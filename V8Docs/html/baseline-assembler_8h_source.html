119  // Push the given values, in the given order. If the stack needs alignment
120  // (looking at you Arm64), the stack is padded from the front (i.e. before the
121  // first value is pushed).
122  //
123  // This supports pushing a RegisterList as the last value -- the list is
124  // iterated and each interpreter Register is pushed.
125  //
126  // The total number of values pushed is returned. Note that this might be
127  // different from sizeof(T...), specifically if there was a RegisterList.
128  template <typename... T>
129  inline int Push(T... vals);
130 
131  // Like Push(vals...), but pushes in reverse order, to support our reversed
132  // order argument JS calling convention. Doesn't return the number of
133  // arguments pushed though.
134  //
135  // Note that padding is still inserted before the first pushed value (i.e. the
136  // last value).
137  template <typename... T>
138  inline void PushReverse(T... vals);
139 
140  // Pop values off the stack into the given registers.
141  //
142  // Note that this inserts into registers in the given order, i.e. in reverse
143  // order if the registers were pushed. This means that to spill registers,
144  // push and pop have to be in reverse order, e.g.
145  //
146  // Push(r1, r2, ..., rN);
147  // ClobberRegisters();
148  // Pop(rN, ..., r2, r1);
149  //
150  // On stack-alignment architectures, any padding is popped off after the last
151  // register. This the behaviour of Push, which means that the above code still
152  // works even if the number of registers doesn't match stack alignment.
153  template <typename... T>
154  inline void Pop(T... registers);
