16 class V8_EXPORT_PRIVATE SandboxHardwareSupport {
17  public:
18  // Allocates a pkey that will be used to optionally block sandbox access. This
19  // function should be called once before any threads are created so that new
20  // threads inherit access to the new pkey.
21  // This function returns true on success, false otherwise. It can generally
22  // fail for one of two reasons:
23  // 1. the current system doesn't support memory protection keys, or
24  // 2. there are no allocatable memory protection keys left.
25  static bool InitializeBeforeThreadCreation();
26 
27  // Try to set up hardware permissions to the sandbox address space. If
28  // successful, future calls to MaybeBlockAccess will block the current thread
29  // from accessing the memory.
30  // This will only fail if InitializeBeforeThreadCreation was unable to
31  // allocate a memory protection key.
32  static bool TryEnable(Address addr, size_t size);
33 
34  // Returns true if hardware sandboxing is enabled.
35  static bool IsEnabled();
36 
37  // Removes the pkey from read only pages. We (currently) still allow read
38  // access to read-only pages inside the sandbox even with an active
39  // DisallowSandboxAccess scope.
40  static void NotifyReadOnlyPageCreated(
41  Address addr, size_t size, PageAllocator::Permission current_permissions);
42 
43  private:
44  friend class DisallowSandboxAccess;
45  friend class AllowSandboxAccess;
46  static int pkey_;
47 };

50 // Scope object to document and enforce that code does not access in-sandbox
51 // data. This provides a certain level of guarantees that code cannot be
52 // influenced by (possibly) attacker-controlled data inside the sandbox.
53 // In DEBUG builds with sandbox hardware support enabled, this property is
54 // enforced at runtime by removing read and write access to the sandbox address
55 // space. The only exception are read-only pages, which will still be readable.
56 //
57 // These scopes can be arbitrarily nested and can be allocated both on the
58 // stack and on the heap. Sandbox access will be disallowed as long as at least
59 // one scope remains active.
60 class V8_NODISCARD V8_ALLOW_UNUSED DisallowSandboxAccess {
61  public:
62 #if defined(DEBUG) && defined(V8_ENABLE_SANDBOX_HARDWARE_SUPPORT)
63  DisallowSandboxAccess();
64  ~DisallowSandboxAccess();
65 
66  // Copying and assigning these scope objects is not allowed as it would not
67  // work correctly.
68  DisallowSandboxAccess(const DisallowSandboxAccess&) = delete;
69  DisallowSandboxAccess& operator=(const DisallowSandboxAccess&) = delete;
70 
71  private:
72  int pkey_;
73 #endif // DEBUG && V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
74 };

76 // Scope object to grant a temporary exception from a DisallowSandboxAccess.
77 // This scope object will re-enable access to in-sandbox memory during its
78 // lifetime even if one or more DisallowSandboxAccess scopes are currently
79 // active. However, in contrast to DisallowSandboxAccess these scopes cannot be
80 // nested and should only be used sparingly and for short durations. It is also
81 // currently not possible to have have another DisallowSandboxAccess inside an
82 // AllowSandboxAccess scope, although that could be implemented if needed.
83 class V8_NODISCARD V8_ALLOW_UNUSED AllowSandboxAccess {
84  public:
85 #if defined(DEBUG) && defined(V8_ENABLE_SANDBOX_HARDWARE_SUPPORT)

87  ~AllowSandboxAccess();
88 
89  // Copying and assigning these scope objects is not allowed as it would not
90  // work correctly.

92  AllowSandboxAccess& operator=(const AllowSandboxAccess&) = delete;
93 
94  private:
95  int pkey_;
96 #endif // DEBUG && V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
97 };
