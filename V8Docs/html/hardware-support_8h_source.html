17 class V8_EXPORT_PRIVATE SandboxHardwareSupport {
18  public:
19  // Try to active hardware sandboxing support. This will attempt to allocate
20  // the memory protection keys needed for sandbox hardware support.
21  //
22  // This function should be called once before any threads are created so that
23  // new threads inherit access to the new pkeys.
24  //
25  // This function returns true on success, false otherwise. It can generally
26  // fail for one of two reasons:
27  // 1. the current system doesn't support memory protection keys, or
28  // 2. there are no allocatable memory protection keys left.
29  static bool TryActivateBeforeThreadCreation();
30 
31  // Returns true if sandbox hardware support is active.
32  //
33  // This will return true iff TryActivateBeforeThreadCreation() was called
34  // before and succeeded in allocating the memory protection keys.
35  static bool IsActive();
36 
37  // Register the memory for the sandbox.
38  //
39  // This will set up the memory protection keys for the sandbox address space.
40  static void RegisterSandboxMemory(Address addr, size_t size);
41 
42  // Register memory outside of the sandbox.
43  //
44  // When in sandboxed execution mode, memory outside of the sandbox cannot be
45  // written to. For now, all such memory needs to be registered through this
46  // method, which will assign the proper memory protection key to it. Once we
47  // always use the default pkey as out_of_sandbox_pkey_, this method will no
48  // longer be required.
49  static void RegisterOutOfSandboxMemory(Address addr, size_t size);
50 
51  // Make additional out-of-sandbox memory accessible to sandboxed code.
52  //
53  // Any use of this function should be considered to be a temporary exception
54  // and be accompanied with a comment describing how and when to get rid of it
55  // (ideally linking to a bug). Eventually, for hardware sandbox support to
56  // become robust, there should be no remaining uses of this function, and
57  // this therefore also serves as documentation where code should be
58  // refactored to not require out-of-sandbox writes from sandboxed code.
59  static void RegisterUnsafeSandboxExtensionMemory(Address addr, size_t size);
60 
61  // Register read-only memory inside the sandbox.
62  // Currently read-only in-sandbox memory is still accessible even with an
63  // active DisallowSandboxAccess scope, and this function configures the
64  // memory protection keys accordingly.
65  static void RegisterReadOnlyMemoryInsideSandbox(
66  Address addr, size_t size, PageAllocator::Permission current_permissions);
67 
68  // Enter and exit sandboxed execution mode for the current thread.
69  //
70  // When in sandboxed execution mode, out-of-sandbox memory is no longer
71  // writable by the current thread (with the exception of "extension" memory
72  // registered via RegisterUnsafeSandboxExtensionMemory). This is achieved by
73  // removing write access for the out-of-sandbox key.
74  static void EnterSandboxedExecutionModeForCurrentThread();
75  static void ExitSandboxedExecutionModeForCurrentThread();
76 
77  // Returns the sandboxing mode of the current thread.
78  //
79  // If sandbox hardware support is not active, this will always return
80  // CodeSandboxingMode::kUnsandboxed.
81  static CodeSandboxingMode CurrentSandboxingMode();
82 
83  // Returns whether the current sandboxing mode matches the given mode.
84  //
85  // If sandbox hardware support is not active, this will always return true.
86  // As such, this method can be used in for example DCHECKs without having to
87  // first check if sandbox hardware support is active.
88  static bool CurrentSandboxingModeIs(CodeSandboxingMode expected_mode);
89 
90  // For use in generated code.
91  static Address sandboxed_mode_pkey_mask_address() {
92  return reinterpret_cast<Address>(&sandboxed_mode_pkey_mask_);
93  }
94 
95  private:
96  friend class DisallowSandboxAccess;
97  friend class AllowSandboxAccess;
98 
99  static bool TryActivate();
100 
101  // This PKEY is used for all (writable) memory inside the sandbox. It can be
102  // used for two different purposes:
103  //
104  // 1. To allow write access to in-sandbox memory while removing write access
105  // to out-of-sandbox memory when executing sandboxed code.
106  //
107  // 2. To remove read access from in-sandbox memory in privileged code. This
108  // can be used to get a certain level of guarantees that code cannot be
109  // influenced by attacker-controlled data. See MaybeBlockAccess().
110  //
111  static int sandbox_pkey_;
112 
113  // This PKEY is used for out-of-sandbox memory, which must not be writable
114  // when executing sandboxed code.
115  //
116  // Ideally, this would simply be the default pkey (key 0) to guarantee that
117  // all out-of-sandbox memory is inaccessible. However, this is somewhat
118  // complicated to get to work, so for now this is a "regular" pkey pwith
119  // which out-of-sandbox memory must manually be tagged.
120  static int out_of_sandbox_pkey_;
121 
122  // This PKEY is used for out-of-sandbox memory that must still be writable
123  // even to sandboxed code.
124  //
125  // Ideally, at some point in the future this key will no longer be needed
126  // because sandboxed code should not need to write any out-of-sandbox memory.
127  static int extension_pkey_;
128 
129  // The mask to apply to the pkru register to enter sandboxed execution mode.
130  //
131  // This is used for the implementation of EnterSandbox and ExitSandbox in
132  // generated code. When hardware sandbox support is not active, this mask
133  // will simply be zero in which case the EnterSandbox and ExitSandbox
134  // routines (in generated code) become nops.
135  static uint32_t sandboxed_mode_pkey_mask_;
136 };
137 #endif // V8_ENABLE_SANDBOX_HARDWARE_SUPPORT

139 // Enter sandboxed execution mode.
140 //
141 // If hardware sandbox support is active, this will disallow write access to
142 // non-sandbox memory for the current thread until ExitSandbox() is called.
143 inline void EnterSandbox() {
144 #ifdef V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
145  SandboxHardwareSupport::EnterSandboxedExecutionModeForCurrentThread();
146 #endif
147 }
148 
149 // Exit sandboxed execution mode.
150 //
151 // This will re-enable write access to out-of-sandbox memory.
152 inline void ExitSandbox() {
153 #ifdef V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
154  SandboxHardwareSupport::ExitSandboxedExecutionModeForCurrentThread();
155 #endif
156 }

178 // Scope object to document and enforce that code does not access in-sandbox
179 // data. This provides a certain level of guarantees that code cannot be
180 // influenced by (possibly) attacker-controlled data inside the sandbox.
181 // In DEBUG builds with sandbox hardware support enabled, this property is
182 // enforced at runtime by removing read and write access to the sandbox address
183 // space. The only exception are read-only pages, which will still be readable.
184 //
185 // These scopes can be arbitrarily nested and can be allocated both on the
186 // stack and on the heap. Sandbox access will be disallowed as long as at least
187 // one scope remains active.
188 class V8_NODISCARD V8_ALLOW_UNUSED DisallowSandboxAccess {
189  public:
190 #if defined(DEBUG) && defined(V8_ENABLE_SANDBOX_HARDWARE_SUPPORT)
191  DisallowSandboxAccess();
192  ~DisallowSandboxAccess();
193 
194  // Copying and assigning these scope objects is not allowed as it would not
195  // work correctly.
196  DisallowSandboxAccess(const DisallowSandboxAccess&) = delete;
197  DisallowSandboxAccess& operator=(const DisallowSandboxAccess&) = delete;
198 
199  private:
200  int pkey_;
201 #endif // DEBUG && V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
202 };

204 // Scope object to grant a temporary exception from a DisallowSandboxAccess.
205 // This scope object will re-enable access to in-sandbox memory during its
206 // lifetime even if one or more DisallowSandboxAccess scopes are currently
207 // active. However, in contrast to DisallowSandboxAccess these scopes cannot be
208 // nested and should only be used sparingly and for short durations. It is also
209 // currently not possible to have have another DisallowSandboxAccess inside an
210 // AllowSandboxAccess scope, although that could be implemented if needed.
211 class V8_NODISCARD V8_ALLOW_UNUSED AllowSandboxAccess {
212  public:
213 #if defined(DEBUG) && defined(V8_ENABLE_SANDBOX_HARDWARE_SUPPORT)
214  AllowSandboxAccess();
215  ~AllowSandboxAccess();
216 
217  // Copying and assigning these scope objects is not allowed as it would not
218  // work correctly.
219  AllowSandboxAccess(const AllowSandboxAccess&) = delete;
220  AllowSandboxAccess& operator=(const AllowSandboxAccess&) = delete;
221 
222  private:
223  int pkey_;
224 #endif // DEBUG && V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
225 };
