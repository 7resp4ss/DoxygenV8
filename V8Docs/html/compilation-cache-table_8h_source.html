83 // A lookup result from the compilation cache for scripts. There are three
84 // possible states:
85 //
86 // 1. Cache miss: script and toplevel_sfi are both null.
87 // 2. Cache hit: script and toplevel_sfi are both non-null. toplevel_sfi is
88 // guaranteed to be compiled, and to stay compiled while this lookup result
89 // instance is alive.
90 // 3. Partial cache hit: script is non-null, but toplevel_sfi is null. The
91 // script may contain an uncompiled toplevel SharedFunctionInfo.
92 class CompilationCacheScriptLookupResult {
93  public:
94  MaybeHandle<Script> script() const { return script_; }
95  MaybeHandle<SharedFunctionInfo> toplevel_sfi() const { return toplevel_sfi_; }
96  IsCompiledScope is_compiled_scope() const { return is_compiled_scope_; }

126 
127  // Eval code only gets cached after a second probe for the
128  // code object. To do so, on first "put" only a hash identifying the
129  // source is entered into the cache, mapping it to a lifetime count of
130  // the hash. On each call to Age all such lifetimes get reduced, and
131  // removed once they reach zero. If a second put is called while such
132  // a hash is live in the cache, the hash gets replaced by an actual
133  // cache entry. Age also removes stale live entries from the cache.
134  // Such entries are identified by SharedFunctionInfos pointing to
135  // either the recompilation stub, or to "old" code. This avoids memory
136  // leaks due to premature caching of eval strings that are
137  // never needed later.
138  static InfoCellPair LookupEval(DirectHandle<CompilationCacheTable> table,
139  DirectHandle<String> src,
140  DirectHandle<SharedFunctionInfo> shared,
141  DirectHandle<NativeContext> native_context,
142  LanguageMode language_mode, int position);
143  // Adds a new native-context-specific feedback back to an existing entry.
144  // Bails out if no existing entry was found.
145  static void UpdateEval(DirectHandle<CompilationCacheTable> table,
146  DirectHandle<String> src,
