65 // The ParallelMoveResolver is used to resolve multiple moves between registers
66 // and stack slots that are intended to happen, semantically, in parallel. It
67 // finds chains of moves that would clobber each other, and emits them in a non
68 // clobbering order; it also detects cycles of moves and breaks them by moving
69 // to a temporary.
70 //
71 // For example, given the moves:
72 //
73 // r1 -> r2
74 // r2 -> r3
75 // r3 -> r4
76 // r4 -> r1
77 // r4 -> r5
78 //
79 // These can be represented as a move graph
80 //
81 // r2 → r3
82 // ↑ ↓
83 // r1 ← r4 → r5
84 //
85 // and safely emitted (breaking the cycle with a temporary) as
86 //
87 // r1 -> tmp
88 // r4 -> r1
89 // r4 -> r5
90 // r3 -> r4
91 // r2 -> r3
92 // tmp -> r2
93 //
94 // It additionally keeps track of materialising moves, which don't have a stack
95 // slot but rather materialise a value from, e.g., a constant. These can safely
96 // be emitted at the end, once all the parallel moves are done.
97 template <typename RegisterT, bool DecompressIfNeeded>
98 class ParallelMoveResolver {
99  static constexpr auto kAllocatableRegistersT =
100  RegisterTHelper<RegisterT>::kAllocatableRegisters;
101  static_assert(!DecompressIfNeeded || std::is_same_v<Register, RegisterT>);
102  static_assert(!DecompressIfNeeded || COMPRESS_POINTERS_BOOL);
103 
104  public:
105  explicit ParallelMoveResolver(MaglevAssembler* masm)
106  : masm_(masm), scratch_(RegisterT::no_reg()) {}

108  void RecordMove(ValueNode* source_node, compiler::InstructionOperand source,
109  compiler::AllocatedOperand target,
110  bool target_needs_to_be_decompressed) {
111  if (target.IsAnyRegister()) {
112  RecordMoveToRegister(source_node, source, ToRegisterT<RegisterT>(target),
113  target_needs_to_be_decompressed);
114  } else {
115  RecordMoveToStackSlot(source_node, source,
116  masm_->GetFramePointerOffsetForStackSlot(target),

118  }

128  void EmitMoves(RegisterT scratch) {
129  DCHECK(!scratch_.is_valid());
130  scratch_ = scratch;
131  for (RegisterT reg : kAllocatableRegistersT) {
132  StartEmitMoveChain(reg);
133  ValueNode* materializing_register_move =
134  materializing_register_moves_[reg.code()];
135  if (materializing_register_move) {
136  materializing_register_move->LoadToRegister(masm_, reg);
137  }
138  }
139  // Emit stack moves until the move set is empty -- each EmitMoveChain will
140  // pop entries off the moves_from_stack_slot map so we can't use a simple
141  // iteration here.
142  while (!moves_from_stack_slot_.empty()) {
143  StartEmitMoveChain(moves_from_stack_slot_.begin()->first);
144  }
145  for (auto [stack_slot, node] : materializing_stack_slot_moves_) {
146  node->LoadToRegister(masm_, scratch_);
147  __ Move(StackSlot{stack_slot}, scratch_);
148  }
149  }

238  void RecordMoveToRegister(ValueNode* node,
239  compiler::InstructionOperand source,
240  RegisterT target_reg,
241  bool target_needs_to_be_decompressed) {
242  // There shouldn't have been another move to this register already.
243  CheckNoExistingMoveToRegister(target_reg);
244 
245  NeedsDecompression needs_decompression = kDoesNotNeedDecompression;
246  if constexpr (DecompressIfNeeded) {
247  if (target_needs_to_be_decompressed &&
248  !node->decompresses_tagged_result()) {
249  needs_decompression = kNeedsDecompression;
250  }
251  } else {
252  DCHECK_IMPLIES(target_needs_to_be_decompressed,
253  node->decompresses_tagged_result());
254  }
255 
256  GapMoveTargets* targets;
257  if (source.IsAnyRegister()) {
258  RegisterT source_reg = ToRegisterT<RegisterT>(source);
259  if (target_reg == source_reg) {
260  // We should never have a register aliasing case that needs
261  // decompression, since this path is only used by exception phis and
262  // they have no reg->reg moves.
263  DCHECK_EQ(needs_decompression, kDoesNotNeedDecompression);
264  return;
265  }
266  targets = &moves_from_register_[source_reg.code()];
267  } else if (source.IsAnyStackSlot()) {
268  int32_t source_slot = masm_->GetFramePointerOffsetForStackSlot(
269  compiler::AllocatedOperand::cast(source));
270  targets = &moves_from_stack_slot_[source_slot];
271  } else {
272  DCHECK(source.IsConstant());
273  DCHECK(IsConstantNode(node->opcode()));
274  materializing_register_moves_[target_reg.code()] = node;
275  // No need to update `targets.needs_decompression`, materialization is
276  // always decompressed.
277  return;
278  }
279 
280  targets->registers.set(target_reg);
281  if (needs_decompression == kNeedsDecompression) {
282  targets->needs_decompression = kNeedsDecompression;
283  }
284  }

286  void RecordMoveToStackSlot(ValueNode* node,
287  compiler::InstructionOperand source,
288  int32_t target_slot,
289  bool target_needs_to_be_decompressed) {
290  // There shouldn't have been another move to this stack slot already.
291  CheckNoExistingMoveToStackSlot(target_slot);
292 
293  NeedsDecompression needs_decompression = kDoesNotNeedDecompression;
294  if constexpr (DecompressIfNeeded) {
295  if (target_needs_to_be_decompressed &&
296  !node->decompresses_tagged_result()) {
297  needs_decompression = kNeedsDecompression;
298  }
299  } else {
300  DCHECK_IMPLIES(target_needs_to_be_decompressed,
301  node->decompresses_tagged_result());
302  }
303 
304  GapMoveTargets* targets;
305  if (source.IsAnyRegister()) {
306  RegisterT source_reg = ToRegisterT<RegisterT>(source);
307  targets = &moves_from_register_[source_reg.code()];
308  } else if (source.IsAnyStackSlot()) {
309  int32_t source_slot = masm_->GetFramePointerOffsetForStackSlot(
310  compiler::AllocatedOperand::cast(source));
311  if (source_slot == target_slot &&
312  needs_decompression == kDoesNotNeedDecompression) {
313  return;
314  }
315  targets = &moves_from_stack_slot_[source_slot];
316  } else {
317  DCHECK(source.IsConstant());
318  DCHECK(IsConstantNode(node->opcode()));
319  materializing_stack_slot_moves_.emplace_back(target_slot, node);
320  // No need to update `targets.needs_decompression`, materialization is
321  // always decompressed.
322  return;
323  }
324 
325  targets->stack_slots.push_back(target_slot);
326  if (needs_decompression == kNeedsDecompression) {
327  targets->needs_decompression = kNeedsDecompression;
328  }
329  }

331  // Finds and clears the targets for a given source. In terms of move graph,
332  // this returns and removes all outgoing edges from the source.
333  GapMoveTargets PopTargets(RegisterT source_reg) {
334  return std::exchange(moves_from_register_[source_reg.code()],
335  GapMoveTargets{});

337  GapMoveTargets PopTargets(int32_t source_slot) {
338  auto handle = moves_from_stack_slot_.extract(source_slot);
339  if (handle.empty()) return {};
340  DCHECK(!handle.mapped().is_empty());
341  return std::move(handle.mapped());

344  // Emit a single move chain starting at the given source (either a register or
345  // a stack slot). This is a destructive operation on the move graph, and
346  // removes the emitted edges from the graph. Subsequent calls with the same
347  // source should emit no code.
348  template <typename SourceT>
349  void StartEmitMoveChain(SourceT source) {
350  DCHECK(!scratch_has_cycle_start_);
351  GapMoveTargets targets = PopTargets(source);
352  if (targets.is_empty()) return;
353 
354  // Start recursively emitting the move chain, with this source as the start
355  // of the chain.
356  bool has_cycle = RecursivelyEmitMoveChainTargets(source, targets);
357 
358  // Each connected component in the move graph can only have one cycle
359  // (proof: each target can only have one incoming edge, so cycles in the
360  // graph can only have outgoing edges, so there's no way to connect two
361  // cycles). This means that if there's a cycle, the saved value must be the
362  // chain start.
363  if (has_cycle) {
364  if (!scratch_has_cycle_start_) {
365  Pop(scratch_);
366  scratch_has_cycle_start_ = true;
367  }
368  EmitMovesFromSource(scratch_, std::move(targets));
369  scratch_has_cycle_start_ = false;
370  __ RecordComment("-- * End of cycle");
371  } else {
372  EmitMovesFromSource(source, std::move(targets));
373  __ RecordComment("-- * Chain emitted with no cycles");
374  }
375  }
376 
377  template <typename ChainStartT, typename SourceT>
378  bool ContinueEmitMoveChain(ChainStartT chain_start, SourceT source) {
379  if constexpr (std::is_same_v<ChainStartT, SourceT>) {
380  // If the recursion has returned to the start of the chain, then this must
381  // be a cycle.
382  if (chain_start == source) {
383  __ RecordComment("-- * Cycle");
384  DCHECK(!scratch_has_cycle_start_);
385  if constexpr (std::is_same_v<ChainStartT, int32_t>) {
386  __ Move(scratch_, StackSlot{chain_start});
387  } else {
388  __ Move(scratch_, chain_start);
389  }
390  scratch_has_cycle_start_ = true;
391  return true;
392  }
393  }
394 
395  GapMoveTargets targets = PopTargets(source);
396  if (targets.is_empty()) {
397  __ RecordComment("-- * End of chain");
398  return false;
399  }
400 
401  bool has_cycle = RecursivelyEmitMoveChainTargets(chain_start, targets);
402 
403  EmitMovesFromSource(source, std::move(targets));
404  return has_cycle;
405  }
406 
407  // Calls RecursivelyEmitMoveChain for each target of a source. This is used to
408  // share target visiting code between StartEmitMoveChain and
409  // ContinueEmitMoveChain.
410  template <typename ChainStartT>
411  bool RecursivelyEmitMoveChainTargets(ChainStartT chain_start,
412  GapMoveTargets& targets) {
413  bool has_cycle = false;
414  for (auto target : targets.registers) {
415  has_cycle |= ContinueEmitMoveChain(chain_start, target);
416  }
417  for (int32_t target_slot : targets.stack_slots) {
418  has_cycle |= ContinueEmitMoveChain(chain_start, target_slot);
419  }
420  return has_cycle;
421  }

423  void EmitMovesFromSource(RegisterT source_reg, GapMoveTargets&& targets) {
424  DCHECK(moves_from_register_[source_reg.code()].is_empty());
425  if constexpr (DecompressIfNeeded) {
426  // The DecompressIfNeeded clause is redundant with the if-constexpr above,
427  // but otherwise this code cannot be compiled by compilers not yet
428  // implementing CWG2518.
429  static_assert(DecompressIfNeeded && COMPRESS_POINTERS_BOOL);
430 
431  if (targets.needs_decompression == kNeedsDecompression) {
432  __ DecompressTagged(source_reg, source_reg);
433  }
434  }
435  for (RegisterT target_reg : targets.registers) {
436  DCHECK(moves_from_register_[target_reg.code()].is_empty());
437  __ Move(target_reg, source_reg);
438  }
439  for (int32_t target_slot : targets.stack_slots) {
440  DCHECK_EQ(moves_from_stack_slot_.find(target_slot),
441  moves_from_stack_slot_.end());
442  __ Move(StackSlot{target_slot}, source_reg);
443  }
444  }

446  void EmitMovesFromSource(int32_t source_slot, GapMoveTargets&& targets) {
447  DCHECK_EQ(moves_from_stack_slot_.find(source_slot),
448  moves_from_stack_slot_.end());
449 
450  // Cache the slot value on a register.
451  RegisterT register_with_slot_value = RegisterT::no_reg();
452  if (!targets.registers.is_empty()) {
453  // If one of the targets is a register, we can move our value into it and
454  // optimize the moves from this stack slot to always be via that register.
455  register_with_slot_value = targets.registers.PopFirst();
456  } else {
457  DCHECK(!targets.stack_slots.empty());
458  // Otherwise, cache the slot value on the scratch register, clobbering it
459  // if necessary.
460  if (scratch_has_cycle_start_) {
461  Push(scratch_);
462  scratch_has_cycle_start_ = false;
463  }
464  register_with_slot_value = scratch_;
465  }
466  // Now emit moves from that cached register instead of from the stack slot.
467  DCHECK(register_with_slot_value.is_valid());
468  DCHECK(moves_from_register_[register_with_slot_value.code()].is_empty());
469  __ Move(register_with_slot_value, StackSlot{source_slot});
470  // Decompress after the first move, subsequent moves reuse this register so
471  // they're guaranteed to be decompressed.
472  if constexpr (DecompressIfNeeded) {
473  // The DecompressIfNeeded clause is redundant with the if-constexpr above,
474  // but otherwise this code cannot be compiled by compilers not yet
475  // implementing CWG2518.
476  static_assert(DecompressIfNeeded && COMPRESS_POINTERS_BOOL);
477 
478  if (targets.needs_decompression == kNeedsDecompression) {
479  __ DecompressTagged(register_with_slot_value, register_with_slot_value);
480  targets.needs_decompression = kDoesNotNeedDecompression;
481  }
482  }
483  EmitMovesFromSource(register_with_slot_value, std::move(targets));
484  }

514  std::vector<std::pair<int32_t, ValueNode*>> materializing_stack_slot_moves_;
515 
516  bool scratch_has_cycle_start_ = false;
517 };

538  void EmitTrampolineFor(NodeBase* node) {
539  DCHECK(node->properties().can_throw());
540 
541  ExceptionHandlerInfo* const handler_info = node->exception_handler_info();
542  if (handler_info->ShouldLazyDeopt()) return;
543  DCHECK(handler_info->HasExceptionHandler());
544  BasicBlock* const catch_block = handler_info->catch_block();
545  LazyDeoptInfo* const deopt_info = node->lazy_deopt_info();
546 
547  // The exception handler trampoline resolves moves for exception phis and
548  // then jumps to the actual catch block. There are a few points worth
549  // noting:
550  //
551  // - All source locations are assumed to be stack slots, except the
552  // accumulator which is stored in kReturnRegister0. We don't emit an
553  // explicit move for it, instead it is pushed and popped at the boundaries
554  // of the entire move sequence (necessary due to materialisation).
555  //
556  // - Some values may require materialisation, i.e. heap number construction
557  // through calls to the NewHeapNumber builtin. To avoid potential conflicts
558  // with other moves (which may happen due to stack slot reuse, i.e. a
559  // target location of move A may equal source location of move B), we
560  // materialise and push results to new temporary stack slots before the
561  // main move sequence, and then pop results into their final target
562  // locations afterwards. Note this is only safe because a) materialised
563  // values are tagged and b) the stack walk treats unknown stack slots as
564  // tagged.
565 
566  const InterpretedDeoptFrame& lazy_frame =
567  deopt_info->GetFrameForExceptionHandler(handler_info);
568 
569  // TODO(v8:7700): Handle inlining.
570  ParallelMoveResolver<Register, COMPRESS_POINTERS_BOOL> direct_moves(masm_);
571  MoveVector materialising_moves;
572  bool save_accumulator = false;
573  RecordMoves(lazy_frame.unit(), catch_block, lazy_frame.frame_state(),
574  &direct_moves, &materialising_moves, &save_accumulator);
575  __ BindJumpTarget(&handler_info->trampoline_entry());
576  __ RecordComment("-- Exception handler trampoline START");
577  EmitMaterialisationsAndPushResults(materialising_moves, save_accumulator);
578 
579  __ RecordComment("EmitMoves");
580  MaglevAssembler::TemporaryRegisterScope temps(masm_);
581  Register scratch = temps.AcquireScratch();
582  direct_moves.EmitMoves(scratch);
583  EmitPopMaterialisedResults(materialising_moves, save_accumulator, scratch);
584  __ Jump(catch_block->label());
585  __ RecordComment("-- Exception handler trampoline END");
586  }

590  void RecordMoves(
591  const MaglevCompilationUnit& unit, BasicBlock* catch_block,
592  const CompactInterpreterFrameState* register_frame,
593  ParallelMoveResolver<Register, COMPRESS_POINTERS_BOOL>* direct_moves,
594  MoveVector* materialising_moves, bool* save_accumulator) {
595  if (!catch_block->has_phi()) return;
596  for (Phi* phi : *catch_block->phis()) {
597  DCHECK(phi->is_exception_phi());
598  if (!phi->has_valid_live_range()) continue;
599 
600  const ValueLocation& target = phi->result();
601  if (phi->owner() == interpreter::Register::virtual_accumulator()) {
602  // If the accumulator is live, then it is the exception object located
603  // at kReturnRegister0. We don't emit a move for it since the value is
604  // already in the right spot, but we do have to ensure it isn't
605  // clobbered by calls to the NewHeapNumber builtin during
606  // materialisation.
607  DCHECK_EQ(target.AssignedGeneralRegister(), kReturnRegister0);
608  *save_accumulator = true;
609  continue;
610  }
611 
612  ValueNode* source = register_frame->GetValueOf(phi->owner(), unit);
613  DCHECK_NOT_NULL(source);
614  if (VirtualObject* vobj = source->TryCast<VirtualObject>()) {
615  DCHECK(vobj->allocation()->HasEscaped());
616  source = vobj->allocation();
617  }
618  // All registers must have been spilled due to the call.
619  // TODO(jgruber): Which call? Because any throw requires at least a call
620  // to Runtime::kThrowFoo?
621  DCHECK(!source->allocation().IsRegister());
622 
623  // The DeoptInfoVisitor should unwrap identity nodes in frame states.
624  DCHECK(!source->Is<Identity>());
625 
626  switch (source->properties().value_representation()) {
627  case ValueRepresentation::kTagged:
628  direct_moves->RecordMove(
629  source, source->allocation(),
630  compiler::AllocatedOperand::cast(target.operand()),
631  phi->decompresses_tagged_result() ? kNeedsDecompression
632  : kDoesNotNeedDecompression);
633  break;
634  case ValueRepresentation::kInt32:
635  case ValueRepresentation::kUint32:
636  case ValueRepresentation::kIntPtr:
637  materialising_moves->emplace_back(target, source);
638  break;
639  case ValueRepresentation::kFloat64:
640  case ValueRepresentation::kHoleyFloat64:
641  materialising_moves->emplace_back(target, source);
642  break;
643  UNREACHABLE();
644  }
645  }
646  }

648  void EmitMaterialisationsAndPushResults(const MoveVector& moves,
649  bool save_accumulator) const {
650  if (moves.empty()) return;
651 
652  // It's possible to optimize this further, at the cost of additional
653  // complexity:
654  //
655  // - If the target location is a register, we could theoretically move the
656  // materialised result there immediately, with the additional complication
657  // that following calls to NewHeapNumber may clobber the register.
658  //
659  // - If the target location is a stack slot which is neither a source nor
660  // target slot for any other moves (direct or materialising), we could move
661  // the result there directly instead of pushing and later popping it. This
662  // doesn't seem worth the extra code complexity though, given we are
663  // talking about a presumably infrequent case for exception handlers.
664 
665  __ RecordComment("EmitMaterialisationsAndPushResults");
666 
667  if (save_accumulator) __ Push(kReturnRegister0);
668 
669 #ifdef DEBUG
670  // Allow calls in these materialisations.
671  __ set_allow_call(true);
672 #endif
673  for (const Move& move : moves) {
674  // We consider constants after all other operations, since constants
675  // don't need to call NewHeapNumber.
676  if (IsConstantNode(move.source->opcode())) continue;
677  __ MaterialiseValueNode(kReturnRegister0, move.source);
678  __ Push(kReturnRegister0);
679  }
680 #ifdef DEBUG
681  __ set_allow_call(false);
682 #endif
683  }
684 
685  void EmitPopMaterialisedResults(const MoveVector& moves,
686  bool save_accumulator,
687  Register scratch) const {
688  if (moves.empty()) return;
689  __ RecordComment("EmitPopMaterialisedResults");
690  for (const Move& move : base::Reversed(moves)) {
691  const ValueLocation& target = move.target;
692  Register target_reg = target.operand().IsAnyRegister()
693  ? target.AssignedGeneralRegister()
694  : scratch;
695  if (IsConstantNode(move.source->opcode())) {
696  __ MaterialiseValueNode(target_reg, move.source);
697  } else {
698  __ Pop(target_reg);
699  }
700  if (target_reg == scratch) {
701  __ Move(masm_->ToMemOperand(target.operand()), scratch);
702  }
703  }
704  if (save_accumulator) __ Pop(kReturnRegister0);
705  }

715  void PreProcessGraph(Graph* graph) {
716  // TODO(victorgomes): I wonder if we want to create a struct that shares
717  // these fields between graph and code_gen_state.
718  code_gen_state()->set_untagged_slots(graph->untagged_stack_slots());
719  code_gen_state()->set_tagged_slots(graph->tagged_stack_slots());
720  code_gen_state()->set_max_deopted_stack_size(
721  graph->max_deopted_stack_size());
722  code_gen_state()->set_max_call_stack_args_(graph->max_call_stack_args());
723 
724  if (v8_flags.maglev_break_on_entry) {
725  __ DebugBreak();
726  }
727 
728  if (graph->is_osr()) {
729  __ OSRPrologue(graph);
730  } else {
731  __ Prologue(graph);
732  }
733 
734  // "Deferred" computation has to be done before block removal, because
735  // block removal doesn't propagate deferredness of removed blocks.
736  int deferred_count = ComputeDeferred(graph);
737 
738  // If we deferred the first block, un-defer it. This can happen because we
739  // defer a block if all its successors are deferred (i.e., lead to an
740  // unconditional deopt). E.g., if we only executed exception throwing code
741  // paths, the non-exception code paths might be untaken, and thus contain
742  // unconditional deopts, so we end up deferring all non-exception code
743  // paths, including the first block.
744  if (graph->blocks()[0]->is_deferred()) {
745  graph->blocks()[0]->set_deferred(false);
746  --deferred_count;
747  }
748 
749  // Reorder the blocks so that dererred blocks are at the end.
750  int non_deferred_count = graph->num_blocks() - deferred_count;
751 
752  ZoneVector<BasicBlock*> new_blocks(graph->num_blocks(), zone_);
753 
754  size_t ix_non_deferred = 0;
755  size_t ix_deferred = non_deferred_count;
756  for (auto block_it = graph->begin(); block_it != graph->end(); ++block_it) {
757  BasicBlock* block = *block_it;
758  DCHECK(!block->is_dead());
759  if (block->is_deferred()) {
760  new_blocks[ix_deferred++] = block;
761  } else {
762  new_blocks[ix_non_deferred++] = block;
763  }
764  }
765  CHECK_EQ(ix_deferred, graph->num_blocks());
766  CHECK_EQ(ix_non_deferred, non_deferred_count);
767  graph->set_blocks(new_blocks);
768

770  ZoneVector<BasicBlock*>& blocks = graph->blocks();
771  size_t current_ix = 0;
772  for (size_t i = 0; i < blocks.size(); ++i) {
773  BasicBlock* block = blocks[i];
774  if (code_gen_state()->RealJumpTarget(block) == block) {
775  // This block cannot be replaced.
776  blocks[current_ix++] = block;
777  }
778  }
779  blocks.resize(current_ix);

883  void EmitBlockEndGapMoves(UnconditionalControlNode* node,
884  const ProcessingState& state) {
885  BasicBlock* target = node->target();
886  if (!target->has_state()) {
887  __ RecordComment("-- Target has no state, must be a fallthrough");
888  return;
889  }
890 
891  int predecessor_id = state.block()->predecessor_id();
892 
893  MaglevAssembler::TemporaryRegisterScope temps(masm_);
894  Register scratch = temps.AcquireScratch();

896 
897  // TODO(leszeks): Move these to fields, to allow their data structure
898  // allocations to be reused. Will need some sort of state resetting.
899  ParallelMoveResolver<Register, false> register_moves(masm_);
900  ParallelMoveResolver<DoubleRegister, false> double_register_moves(masm_);

902  // Remember what registers were assigned to by a Phi, to avoid clobbering
903  // them with RegisterMoves.
904  RegList registers_set_by_phis;
905  DoubleRegList double_registers_set_by_phis;
906 
907  __ RecordComment("-- Gap moves:");
908 
909  if (target->has_phi()) {
910  Phi::List* phis = target->phis();
911  for (Phi* phi : *phis) {
912  // Ignore dead phis.
913  // TODO(leszeks): We should remove dead phis entirely and turn this into
914  // a DCHECK.
915  if (!phi->has_valid_live_range()) {
916  if (v8_flags.code_comments) {
917  std::stringstream ss;
918  ss << "-- * "
919  << phi->input(state.block()->predecessor_id()).operand() << " → "
920  << target << " (n" << graph_labeller()->NodeId(phi)
921  << ") [DEAD]";
922  __ RecordComment(ss.str());
923  }
924  continue;
925  }
926  Input& input = phi->input(state.block()->predecessor_id());
927  ValueNode* input_node = input.node();
928  compiler::InstructionOperand source = input.operand();
929  compiler::AllocatedOperand target_operand =
930  compiler::AllocatedOperand::cast(phi->result().operand());
931  if (v8_flags.code_comments) {
932  std::stringstream ss;
933  ss << "-- * " << source << " → " << target << " (n"
934  << graph_labeller()->NodeId(phi) << ")";
935  __ RecordComment(ss.str());
936  }
937  if (phi->use_double_register()) {
938  DCHECK(!phi->decompresses_tagged_result());
939  double_register_moves.RecordMove(input_node, source, target_operand,
940  false);
941  } else {
942  register_moves.RecordMove(input_node, source, target_operand,
943  kDoesNotNeedDecompression);
944  }
945  if (target_operand.IsAnyRegister()) {
946  if (phi->use_double_register()) {
947  double_registers_set_by_phis.set(
948  target_operand.GetDoubleRegister());

952  }

957  [&](Register reg, RegisterState& state) {
958  // Don't clobber registers set by a Phi.
959  if (registers_set_by_phis.has(reg)) return;
960

963  if (LoadMergeState(state, &node, &merge)) {
964  compiler::InstructionOperand source =
965  merge->operand(predecessor_id);
966  if (v8_flags.code_comments) {
967  std::stringstream ss;
968  ss << "-- * " << source << " → " << reg;
969  __ RecordComment(ss.str());
970  }
971  register_moves.RecordMove(node, source, reg,
972  kDoesNotNeedDecompression);
973  }
974  });
975 
976  register_moves.EmitMoves(scratch);
977 
978  __ RecordComment("-- Double gap moves:");
979 
980  target->state()->register_state().ForEachDoubleRegister(
981  [&](DoubleRegister reg, RegisterState& state) {
982  // Don't clobber registers set by a Phi.
983  if (double_registers_set_by_phis.has(reg)) return;
984 
985  ValueNode* node;
986  RegisterMerge* merge;
987  if (LoadMergeState(state, &node, &merge)) {
988  compiler::InstructionOperand source =
989  merge->operand(predecessor_id);
990  if (v8_flags.code_comments) {
991  std::stringstream ss;
992  ss << "-- * " << source << " → " << reg;
993  __ RecordComment(ss.str());
994  }
995  double_register_moves.RecordMove(node, source, reg,
996  kDoesNotNeedDecompression);
997  }
998  });
999

1043  int ComputeDeferred(Graph* graph) {
1044  int deferred_count = 0;
1045  // Propagate deferredness: If a block is deferred, defer all its successors,
1046  // except if a successor has another predecessor which is not deferred.
1047 
1048  // In addition, if all successors of a block are deferred, defer it too.
1049 
1050  // Work queue is a queue of blocks which are deferred, so we'll need to
1051  // check whether to defer their successors and predecessors.
1052  SmallZoneVector<BasicBlock*, 32> work_queue(zone_);
1053  for (auto block_it = graph->begin(); block_it != graph->end(); ++block_it) {
1054  BasicBlock* block = *block_it;
1055  if (block->is_deferred()) {
1056  ++deferred_count;
1057  work_queue.emplace_back(block);
1058  }
1059  }
1060 
1061  // The algorithm below is O(N * e^2) where e is the maximum number of
1062  // predecessors / successors. We check whether we should defer a block at
1063  // most e times. When doing the check, we check each predecessor / successor
1064  // once.
1065  while (!work_queue.empty()) {
1066  BasicBlock* block = work_queue.back();
1067  work_queue.pop_back();
1068  DCHECK(block->is_deferred());
1069 
1070  // Check if we should defer any successor.
1071  block->ForEachSuccessor([&work_queue,
1072  &deferred_count](BasicBlock* successor) {
1073  if (successor->is_deferred()) {
1074  return;
1075  }
1076  bool should_defer = true;
1077  successor->ForEachPredecessor([&should_defer](BasicBlock* predecessor) {
1078  if (!predecessor->is_deferred()) {
1079  should_defer = false;
1080  }
1081  });
1082  if (should_defer) {
1083  ++deferred_count;
1084  work_queue.emplace_back(successor);
1085  successor->set_deferred(true);
1086  }
1087  });
1088 
1089  // Check if we should defer any predecessor.
1090  block->ForEachPredecessor([&work_queue,
1091  &deferred_count](BasicBlock* predecessor) {
1092  if (predecessor->is_deferred()) {
1093  return;
1094  }
1095  bool should_defer = true;
1096  predecessor->ForEachSuccessor([&should_defer](BasicBlock* successor) {
1097  if (!successor->is_deferred()) {
1098  should_defer = false;
1099  }
1100  });
1101  if (should_defer) {
1102  ++deferred_count;
1103  work_queue.emplace_back(predecessor);
1104  predecessor->set_deferred(true);
1105  }
1106  });
1107  }
1108  return deferred_count;
1109  }

1588  void BuildDeoptFrameValues(
1589  const MaglevCompilationUnit& compilation_unit,
1590  const CompactInterpreterFrameState* checkpoint_state,
1591  const ValueNode* closure, const InputLocation*& input_location,
1592  const VirtualObjectList& virtual_objects,
1593  interpreter::Register result_location, int result_size) {
1594  // TODO(leszeks): The input locations array happens to be in the same
1595  // order as closure+parameters+context+locals+accumulator are accessed
1596  // here. We should make this clearer and guard against this invariant
1597  // failing.
1598 
1599  // Closure
1600  BuildDeoptFrameSingleValue(closure, input_location, virtual_objects);
1601 
1602  // Parameters
1603  {
1604  int i = 0;
1605  checkpoint_state->ForEachParameter(
1606  compilation_unit, [&](ValueNode* value, interpreter::Register reg) {
1607  DCHECK_EQ(reg.ToParameterIndex(), i);
1608  if (LazyDeoptInfo::InReturnValues(reg, result_location,
1609  result_size)) {
1610  translation_array_builder_->StoreOptimizedOut();
1611  } else {
1612  BuildDeoptFrameSingleValue(value, input_location,
1613  virtual_objects);
1614  }
1615  i++;
1616  });
1617  }
1618 
1619  // Context
1620  ValueNode* context_value = checkpoint_state->context(compilation_unit);
1621  BuildDeoptFrameSingleValue(context_value, input_location, virtual_objects);
1622 
1623  // Locals
1624  {
1625  int i = 0;
1626  checkpoint_state->ForEachLocal(
1627  compilation_unit, [&](ValueNode* value, interpreter::Register reg) {
1628  DCHECK_LE(i, reg.index());
1629  if (LazyDeoptInfo::InReturnValues(reg, result_location,
1630  result_size))
1631  return;
1632  while (i < reg.index()) {
1633  translation_array_builder_->StoreOptimizedOut();
1634  i++;
1635  }
1636  DCHECK_EQ(i, reg.index());
1637  BuildDeoptFrameSingleValue(value, input_location, virtual_objects);
1638  i++;
1639  });
1640  while (i < compilation_unit.register_count()) {
1641  translation_array_builder_->StoreOptimizedOut();
1642  i++;
1643  }
1644  }
1645 
1646  // Accumulator
1647  {
1648  if (checkpoint_state->liveness()->AccumulatorIsLive() &&
1649  !LazyDeoptInfo::InReturnValues(
1650  interpreter::Register::virtual_accumulator(), result_location,
1651  result_size)) {
1652  ValueNode* value = checkpoint_state->accumulator(compilation_unit);
1653  BuildDeoptFrameSingleValue(value, input_location, virtual_objects);
1654  } else {
1655  translation_array_builder_->StoreOptimizedOut();
1656  }
1657  }
1658  }

1817 bool MaglevCodeGenerator::EmitDeopts() {
1818  const size_t num_deopts = code_gen_state_.eager_deopts().size() +
1819  code_gen_state_.lazy_deopts().size();
1820  if (num_deopts > Deoptimizer::kMaxNumberOfEntries) {
1821  return false;
1822  }
1823 
1824  MaglevFrameTranslationBuilder translation_builder(
1825  local_isolate_, &masm_, &frame_translation_builder_,
1826  &protected_deopt_literals_, &deopt_literals_);
1827 
1828  // Deoptimization exits must be as small as possible, since their count grows
1829  // with function size. These labels are an optimization which extracts the
1830  // (potentially large) instruction sequence for the final jump to the
1831  // deoptimization entry into a single spot per InstructionStream object. All
1832  // deopt exits can then near-call to this label. Note: not used on all
1833  // architectures.
1834  Label eager_deopt_entry;
1835  Label lazy_deopt_entry;
1836  __ MaybeEmitDeoptBuiltinsCall(
1837  code_gen_state_.eager_deopts().size(), &eager_deopt_entry,
1838  code_gen_state_.lazy_deopts().size(), &lazy_deopt_entry);
1839 
1840  deopt_exit_start_offset_ = __ pc_offset();
1841 
1842  int deopt_index = 0;
1843 
1844  __ RecordComment("-- Non-lazy deopts");
1845  for (EagerDeoptInfo* deopt_info : code_gen_state_.eager_deopts()) {
1846  local_isolate_->heap()->Safepoint();
1847  translation_builder.BuildEagerDeopt(deopt_info);
1848 
1849  if (masm_.compilation_info()->collect_source_positions() ||
1850  AlwaysPreserveDeoptReason(deopt_info->reason()) ||
1851  IsDeoptimizationWithoutCodeInvalidation(deopt_info->reason())) {
1852  // Note: Maglev uses the deopt_reason to tell the deoptimizer not to
1853  // discard optimized code on deopt during ML-TF OSR. This is why we
1854  // unconditionally emit the deopt_reason when
1855  // IsDeoptimizationWithoutCodeInvalidation is true.
1856  __ RecordDeoptReason(deopt_info->reason(), 0,
1857  deopt_info->top_frame().GetSourcePosition(),
1858  deopt_index);
1859  }
1860  __ bind(deopt_info->deopt_entry_label());
1861 
1862  __ CallForDeoptimization(Builtin::kDeoptimizationEntry_Eager, deopt_index,
1863  deopt_info->deopt_entry_label(),
1864  DeoptimizeKind::kEager, nullptr,
1865  &eager_deopt_entry);
1866 
1867  deopt_index++;
1868  }
1869 
1870  __ RecordComment("-- Lazy deopts");
1871  int last_updated_safepoint = 0;
1872  for (LazyDeoptInfo* deopt_info : code_gen_state_.lazy_deopts()) {
1873  local_isolate_->heap()->Safepoint();
1874  translation_builder.BuildLazyDeopt(deopt_info);
1875 
1876  if (masm_.compilation_info()->collect_source_positions()) {
1877  __ RecordDeoptReason(DeoptimizeReason::kUnknown, 0,
1878  deopt_info->top_frame().GetSourcePosition(),
1879  deopt_index);
1880  }
1881  __ BindExceptionHandler(deopt_info->deopt_entry_label());
1882 
1883  __ CallForDeoptimization(Builtin::kDeoptimizationEntry_Lazy, deopt_index,
1884  deopt_info->deopt_entry_label(),
1885  DeoptimizeKind::kLazy, nullptr, &lazy_deopt_entry);
1886 
1887  last_updated_safepoint = safepoint_table_builder_.UpdateDeoptimizationInfo(
1888  deopt_info->deopting_call_return_pc(),
1889  deopt_info->deopt_entry_label()->pos(), last_updated_safepoint,
1890  deopt_index);
1891  deopt_index++;
1892  }
1893 
1894  return true;
1895 }
