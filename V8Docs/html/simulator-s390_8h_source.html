61 static T ComputeRounding(T a, int mode) {
62  switch (mode) {
63  case ROUND_TO_NEAREST_AWAY_FROM_0:
64  return std::round(a);
65  case ROUND_TO_NEAREST_TO_EVEN:
66  return std::nearbyint(a);
67  case ROUND_TOWARD_0:
68  return std::trunc(a);
69  case ROUND_TOWARD_POS_INF:
70  return std::ceil(a);
71  case ROUND_TOWARD_NEG_INF:
72  return std::floor(a);
73  default:
74  UNIMPLEMENTED();
75  }
76  return 0;

292  template <typename T>
293  void SetS390ConditionCode(T lhs, T rhs) {
294  condition_reg_ = 0;
295  if (lhs == rhs) {
296  condition_reg_ |= CC_EQ;
297  } else if (lhs < rhs) {
298  condition_reg_ |= CC_LT;
299  } else if (lhs > rhs) {
300  condition_reg_ |= CC_GT;
301  }
302 
303  // We get down here only for floating point
304  // comparisons and the values are unordered
305  // i.e. NaN
306  if (condition_reg_ == 0) condition_reg_ = unordered;
307  }
308 
309  // Used by arithmetic operations that use carry.
310  template <typename T>
311  void SetS390ConditionCodeCarry(T result, bool overflow) {
312  condition_reg_ = 0;
313  bool zero_result = (result == static_cast<T>(0));
314  if (zero_result && !overflow) {
315  condition_reg_ |= 8;
316  } else if (!zero_result && !overflow) {
317  condition_reg_ |= 4;
318  } else if (zero_result && overflow) {
319  condition_reg_ |= 2;
320  } else if (!zero_result && overflow) {
321  condition_reg_ |= 1;
322  }
323  if (condition_reg_ == 0) UNREACHABLE();
324  }

328  template <typename T1, typename T2, typename T3>
329  T1 FPProcessNaNBinop(T1 fp_lhs, T1 fp_rhs,
330  const std::function<T1(T1, T1)>& op_for_non_nan) {
331  T2 lhs = T2::FromBits(base::bit_cast<T3>(fp_lhs));
332  T2 rhs = T2::FromBits(base::bit_cast<T3>(fp_rhs));
333  if (lhs.is_nan() && !lhs.is_quiet_nan())
334  return lhs.to_quiet_nan().get_scalar();
335  if (rhs.is_nan() && !rhs.is_quiet_nan())
336  return rhs.to_quiet_nan().get_scalar();
337  if (lhs.is_nan()) return lhs.to_quiet_nan().get_scalar();
338  if (rhs.is_nan()) return rhs.to_quiet_nan().get_scalar();
339  return op_for_non_nan(fp_lhs, fp_rhs);
340  }

411  // Vector register lane numbers on IBM machines are reversed compared to
412  // x64. For example, doing an I32x4 extract_lane with lane number 0 on x64
413  // will be equal to lane number 3 on IBM machines. Vector registers are only
414  // used for compiling Wasm code at the moment. Wasm is also little endian
415  // enforced. On s390 native, we manually do a reverse byte whenever values are
416  // loaded/stored from memory to a Simd register. On the simulator however, we
417  // do not reverse the bytes and data is just copied as is from one memory
418  // location to another location which represents a register. To keep the Wasm
419  // simulation accurate, we need to make sure accessing a lane is correctly
420  // simulated and as such we reverse the lane number on the getters and setters
421  // below. We need to be careful when getting/setting values on the Low or High
422  // side of a simulated register. In the simulation, "Low" is equal to the MSB
423  // and "High" is equal to the LSB on memory. "force_ibm_lane_numbering" could
424  // be used to disabled automatic lane number reversal and help with accessing
425  // the Low or High side of a simulated register.
426  template <class T>
427  T get_simd_register_by_lane(int reg, int lane,
428  bool force_ibm_lane_numbering = true) {
429  if (force_ibm_lane_numbering) {
430  lane = (kSimd128Size / sizeof(T)) - 1 - lane;
431  }
432  CHECK_LE(lane, kSimd128Size / sizeof(T));
433  CHECK_LT(reg, kNumFPRs);
434  CHECK_GE(lane, 0);
435  CHECK_GE(reg, 0);
436  return (reinterpret_cast<T*>(&fp_registers_[reg]))[lane];
437  }
438 
439  template <class T>
440  void set_simd_register_by_lane(int reg, int lane, const T& value,
441  bool force_ibm_lane_numbering = true) {
442  if (force_ibm_lane_numbering) {
443  lane = (kSimd128Size / sizeof(T)) - 1 - lane;
444  }
445  CHECK_LE(lane, kSimd128Size / sizeof(T));
446  CHECK_LT(reg, kNumFPRs);
447  CHECK_GE(lane, 0);
448  CHECK_GE(reg, 0);
449  (reinterpret_cast<T*>(&fp_registers_[reg]))[lane] = value;
450  }
