113 void NodeBase::change_input(int index, ValueNode* node) {
114  DCHECK_NE(input(index).node(), nullptr);
115  // After the AnyUseMarkingProcessor the use count can be -1.
116  if (input(index).node()->is_used()) {
117  input(index).node()->remove_use();
118  } else {
119  DCHECK_EQ(input(index).node()->use_count(), -1);
120  }
121 
122 #ifdef DEBUG
123  input(index) = Input(nullptr);
124 #endif
125  set_input(index, node);
126 }

128 void ValueNode::remove_use() {
129  // Make sure a saturated use count won't drop below zero.
130  DCHECK_GT(use_count_, 0);
131  use_count_--;
132  if (auto alloc = TryCast<InlinedAllocation>()) {
133  // Unfortunately we cannot know if the removed use was escaping or not. To
134  // be safe we need to assume it wasn't.
135  alloc->RemoveNonEscapingUses(1);
136  }
137 }

139 void NodeBase::OverwriteWithIdentityTo(ValueNode* node) {
140  // OverwriteWith() checks if the node we're overwriting to has the same
141  // input count and the same properties. Here we don't need to do that, since
142  // overwriting with a node with property pure, we only need to check if
143  // there is at least 1 input. Since the first input is always the one
144  // closest to the input_base().
145  DCHECK_GE(input_count(), 1);
146  set_opcode(NodeBase::opcode_of<Identity>);
147  set_properties(OpProperties::Pure());
148  bitfield_ = InputCountField::update(bitfield_, 1);
149  change_input(0, node);
150 }
