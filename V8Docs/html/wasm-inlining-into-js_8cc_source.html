34  public:
35  WasmIntoJSInlinerImpl(Zone* zone, const wasm::WasmModule* module,
36  MachineGraph* mcgraph, const wasm::FunctionBody& body,
37  base::Vector<const uint8_t> bytes,
38  SourcePositionTable* source_position_table,
39  int inlining_id)
40  : wasm::Decoder(bytes.begin(), bytes.end()),
41  module_(module),
42  mcgraph_(mcgraph),
43  body_(body),
44  graph_(mcgraph->graph()),
45  gasm_(mcgraph, zone),
46  source_position_table_(source_position_table),
47  inlining_id_(inlining_id) {
48  // +1 for instance node.
49  size_t params = body.sig->parameter_count() + 1;
50  Node* start =
51  graph_->NewNode(mcgraph->common()->Start(static_cast<int>(params)));
52  graph_->SetStart(start);
53  graph_->SetEnd(graph_->NewNode(mcgraph->common()->End(0)));
54  gasm_.InitializeEffectControl(start, start);
55 
56  // Initialize parameter nodes.
57  // We have to add another +1 as the minimum parameter index is actually
58  // -1, not 0...
59  size_t params_extended = params + 1;
60  parameters_ = zone->AllocateArray<Node*>(params_extended);
61  for (unsigned i = 0; i < params_extended; i++) {
62  parameters_[i] = nullptr;
63  }
64  // Instance node at parameter 0.
65  trusted_data_node_ = Param(wasm::kWasmInstanceDataParameterIndex);
66  }

68  Node* Param(int index, const char* debug_name = nullptr) {
69  DCHECK_NOT_NULL(graph_->start());
70  // Turbofan allows negative parameter indices.
71  DCHECK_GE(index, kMinParameterIndex);
72  int array_index = index - kMinParameterIndex;
73  if (parameters_[array_index] == nullptr) {
74  Node* param = graph_->NewNode(
75  mcgraph_->common()->Parameter(index, debug_name), graph_->start());
76  if (index > wasm::kWasmInstanceDataParameterIndex) {
77  // Add a type guard to keep type information based on the inlinee's
78  // signature.
79  wasm::ValueType type = body_.sig->GetParam(index - 1);
80  Type tf_type = compiler::Type::Wasm(type, module_, graph_->zone());
81  param = gasm_.TypeGuard(tf_type, param);
82  }
83  parameters_[array_index] = param;
84  }
85  return parameters_[array_index];
86  }

88  bool TryInlining() {
89  if (body_.sig->return_count() > 1) {
90  return false; // Multi-return is not supported.
91  }
92  // Parse locals.
93  if (consume_u32v() != 0) {
94  // Functions with locals are not supported.
95  return false;
96  }
97  // Parse body.
98  base::SmallVector<Value, 4> stack;
99  while (is_inlineable_) {
100  WasmOpcode opcode = ReadOpcode();
101  switch (opcode) {
102  case wasm::kExprAnyConvertExtern:
103  DCHECK(!stack.empty());
104  stack.back() = ParseAnyConvertExtern(stack.back());
105  continue;
106  case wasm::kExprExternConvertAny:
107  DCHECK(!stack.empty());
108  stack.back() = ParseExternConvertAny(stack.back());
109  continue;
110  case wasm::kExprRefCast:
111  case wasm::kExprRefCastNull:
112  DCHECK(!stack.empty());
113  stack.back() =
114  ParseRefCast(stack.back(), opcode == wasm::kExprRefCastNull);
115  continue;
116  case wasm::kExprArrayLen:
117  DCHECK(!stack.empty());
118  stack.back() = ParseArrayLen(stack.back());
119  continue;
120  case wasm::kExprArrayGet:
121  case wasm::kExprArrayGetS:
122  case wasm::kExprArrayGetU: {
123  DCHECK_GE(stack.size(), 2);
124  Value index = stack.back();
125  stack.pop_back();
126  Value array = stack.back();
127  stack.back() = ParseArrayGet(array, index, opcode);
128  continue;
129  }
130  case wasm::kExprArraySet: {
131  DCHECK_GE(stack.size(), 3);
132  Value value = stack.back();
133  stack.pop_back();
134  Value index = stack.back();
135  stack.pop_back();
136  Value array = stack.back();
137  stack.pop_back();
138  ParseArraySet(array, index, value);
139  continue;
140  }
141  case wasm::kExprStructGet:
142  case wasm::kExprStructGetS:
143  case wasm::kExprStructGetU:
144  DCHECK(!stack.empty());
145  stack.back() = ParseStructGet(stack.back(), opcode);
146  continue;
147  case wasm::kExprStructSet: {
148  DCHECK_GE(stack.size(), 2);
149  Value value = stack.back();
150  stack.pop_back();
151  Value wasm_struct = stack.back();
152  stack.pop_back();
153  ParseStructSet(wasm_struct, value);
154  continue;
155  }
156  case wasm::kExprLocalGet:
157  stack.push_back(ParseLocalGet());
158  continue;
159  case wasm::kExprDrop:
160  DCHECK(!stack.empty());
161  stack.pop_back();
162  continue;
163  case wasm::kExprEnd: {
164  DCHECK_LT(stack.size(), 2);
165  int return_count = static_cast<int>(stack.size());
166  base::SmallVector<Node*, 8> buf(return_count + 3);
167  buf[0] = mcgraph_->Int32Constant(0);
168  if (return_count) {
169  buf[1] = stack.back().node;
170  }
171  buf[return_count + 1] = gasm_.effect();
172  buf[return_count + 2] = gasm_.control();
173  Node* ret = graph_->NewNode(mcgraph_->common()->Return(return_count),
174  return_count + 3, buf.data());
175 
176  gasm_.MergeControlToEnd(ret);
177  return true;
178  }
179  default:
180  // Instruction not supported for inlining.
181  return false;
182  }
183  }
184  // The decoder found an instruction it couldn't inline successfully.
185  return false;
186  }

189  Value ParseAnyConvertExtern(Value input) {
190  DCHECK(input.type.is_reference_to(wasm::HeapType::kExtern) ||
191  input.type.is_reference_to(wasm::HeapType::kNoExtern));
192  wasm::ValueType result_type = wasm::ValueType::Generic(
193  wasm::GenericKind::kAny, input.type.nullability(), kNotShared);
194  Node* internalized = gasm_.WasmAnyConvertExtern(input.node);
195  return TypeNode(internalized, result_type);
196  }

198  Value ParseExternConvertAny(Value input) {
199  DCHECK(input.type.is_reference());
200  wasm::ValueType result_type = wasm::ValueType::Generic(
201  wasm::GenericKind::kExtern, input.type.nullability(), kNotShared);
202  Node* internalized = gasm_.WasmExternConvertAny(input.node);
203  return TypeNode(internalized, result_type);
204  }

206  Value ParseLocalGet() {
207  uint32_t index = consume_u32v();
208  DCHECK_LT(index, body_.sig->parameter_count());
209  return TypeNode(Param(index + 1), body_.sig->GetParam(index));

212  Value ParseStructGet(Value struct_val, WasmOpcode opcode) {
213  wasm::ModuleTypeIndex struct_index{consume_u32v()};
214  DCHECK(module_->has_struct(struct_index));
215  const wasm::StructType* struct_type = module_->struct_type(struct_index);
216  uint32_t field_index = consume_u32v();
217  DCHECK_GT(struct_type->field_count(), field_index);
218  const bool is_signed = opcode == wasm::kExprStructGetS;
219  const CheckForNull null_check =
220  struct_val.type.is_nullable() ? kWithNullCheck : kWithoutNullCheck;
221  Node* member = gasm_.StructGet(struct_val.node, struct_type, field_index,
222  is_signed, null_check);
223  SetSourcePosition(member);
224  return TypeNode(member, struct_type->field(field_index).Unpacked());
225  }

227  void ParseStructSet(Value wasm_struct, Value value) {
228  wasm::ModuleTypeIndex struct_index{consume_u32v()};
229  DCHECK(module_->has_struct(struct_index));
230  const wasm::StructType* struct_type = module_->struct_type(struct_index);
231  uint32_t field_index = consume_u32v();
232  DCHECK_GT(struct_type->field_count(), field_index);
233  const CheckForNull null_check =
234  wasm_struct.type.is_nullable() ? kWithNullCheck : kWithoutNullCheck;
235  gasm_.StructSet(wasm_struct.node, value.node, struct_type, field_index,
236  null_check);
237  SetSourcePosition(gasm_.effect());
238  }

240  // TODO(14616): Implement for shared types.
241  Value ParseRefCast(Value input, bool null_succeeds) {
242  auto [heap_index, length] = read_i33v<ValidationTag>(pc_);
243  pc_ += length;
244  if (heap_index < 0) {
245  if ((heap_index & 0x7f) != wasm::kArrayRefCode) {
246  // Abstract casts for non array type are not supported.
247  is_inlineable_ = false;
248  return {};
249  }
250  auto done = gasm_.MakeLabel();
251  // Abstract cast to array.
252  if (input.type.is_nullable() && null_succeeds) {
253  gasm_.GotoIf(gasm_.IsNull(input.node, input.type), &done);
254  }
255  gasm_.TrapIf(gasm_.IsSmi(input.node), TrapId::kTrapIllegalCast);
256  gasm_.TrapUnless(gasm_.HasInstanceType(input.node, WASM_ARRAY_TYPE),
257  TrapId::kTrapIllegalCast);
258  SetSourcePosition(gasm_.effect());
259  gasm_.Goto(&done);
260  gasm_.Bind(&done);
261  // Add TypeGuard for graph typing.
262  TFGraph* graph = mcgraph_->graph();
263  wasm::ValueType result_type = wasm::ValueType::Generic(
264  wasm::GenericKind::kArray,
265  null_succeeds ? wasm::kNullable : wasm::kNonNullable, kNotShared);
266  Node* type_guard =
267  graph->NewNode(mcgraph_->common()->TypeGuard(
268  Type::Wasm(result_type, module_, graph->zone())),
269  input.node, gasm_.effect(), gasm_.control());
270  gasm_.InitializeEffectControl(type_guard, gasm_.control());
271  return TypeNode(type_guard, result_type);
272  }
273  wasm::ModuleTypeIndex target_type_index{static_cast<uint32_t>(heap_index)};
274  if (module_->has_signature(target_type_index)) {
275  is_inlineable_ = false;
276  return {};
277  }
278  wasm::ValueType target_type = wasm::ValueType::RefMaybeNull(
279  module_->heap_type(target_type_index),
280  null_succeeds ? wasm::kNullable : wasm::kNonNullable);
281  Node* rtt = mcgraph_->graph()->NewNode(
282  gasm_.simplified()->RttCanon(target_type.ref_index()),
283  trusted_data_node_);
284  // Technically this is incorrect: the {rtt} node doesn't hold a reference
285  // to an object of type {target_type}, but to such an object's map. But
286  // we only need this type annotation so {ReduceWasmTypeCast} can get to
287  // the {ref_index}, we never need the type's {kind()}.
288  TypeNode(rtt, wasm::ValueType::Ref(target_type.heap_type()));
289  Node* cast = gasm_.WasmTypeCast(
290  input.node, rtt,
291  {input.type, target_type,
292  module_->type(target_type_index).is_final ? kExactMatchOnly
293  : kMayBeSubtype});
294  SetSourcePosition(cast);
295  return TypeNode(cast, target_type);
296  }

298  Value ParseArrayLen(Value input) {
299  DCHECK(wasm::IsHeapSubtypeOf(input.type.heap_type(),
300  wasm::kWasmArrayRef.heap_type(), module_));
301  const CheckForNull null_check =
302  input.type.is_nullable() ? kWithNullCheck : kWithoutNullCheck;
303  Node* len = gasm_.ArrayLength(input.node, null_check);
304  SetSourcePosition(len);
305  return TypeNode(len, wasm::kWasmI32);
306  }

308  Value ParseArrayGet(Value array, Value index, WasmOpcode opcode) {
309  wasm::ModuleTypeIndex array_index{consume_u32v()};
310  DCHECK(module_->has_array(array_index));
311  const wasm::ArrayType* array_type = module_->array_type(array_index);
312  const bool is_signed = opcode == WasmOpcode::kExprArrayGetS;
313  const CheckForNull null_check =
314  array.type.is_nullable() ? kWithNullCheck : kWithoutNullCheck;
315  // Perform bounds check.
316  Node* length = gasm_.ArrayLength(array.node, null_check);
317  SetSourcePosition(length);
318  gasm_.TrapUnless(gasm_.Uint32LessThan(index.node, length),
319  TrapId::kTrapArrayOutOfBounds);
320  SetSourcePosition(gasm_.effect());
321  // Perform array.get.
322  Node* element =
323  gasm_.ArrayGet(array.node, index.node, array_type, is_signed);
324  return TypeNode(element, array_type->element_type().Unpacked());
325  }

327  void ParseArraySet(Value array, Value index, Value value) {
328  wasm::ModuleTypeIndex array_index{consume_u32v()};
329  DCHECK(module_->has_array(array_index));
330  const wasm::ArrayType* array_type = module_->array_type(array_index);
331  const CheckForNull null_check =
332  array.type.is_nullable() ? kWithNullCheck : kWithoutNullCheck;
333  // Perform bounds check.
334  Node* length = gasm_.ArrayLength(array.node, null_check);
335  SetSourcePosition(length);
336  gasm_.TrapUnless(gasm_.Uint32LessThan(index.node, length),
337  TrapId::kTrapArrayOutOfBounds);
338  SetSourcePosition(gasm_.effect());
339  // Perform array.set.
340  gasm_.ArraySet(array.node, index.node, value.node, array_type);
341  }

343  WasmOpcode ReadOpcode() {
344  DCHECK_LT(pc_, end_);
345  instruction_start_ = pc();
346  WasmOpcode opcode = static_cast<WasmOpcode>(*pc_);
347  if (!WasmOpcodes::IsPrefixOpcode(opcode)) {
348  ++pc_;
349  return opcode;
350  }
351  auto [opcode_with_prefix, length] =
352  read_prefixed_opcode<ValidationTag>(pc_);
353  pc_ += length;
354  return opcode_with_prefix;
355  }

357  Value TypeNode(Node* node, wasm::ValueType type) {
358  compiler::NodeProperties::SetType(
359  node, compiler::Type::Wasm(type, module_, graph_->zone()));
360  return {node, type};
361  }

363  void SetSourcePosition(Node* node) {
364  if (!source_position_table_->IsEnabled()) return;
365  int offset = static_cast<int>(instruction_start_ - start());
366  source_position_table_->SetSourcePosition(
367  node, SourcePosition(offset, inlining_id_));
368  }
