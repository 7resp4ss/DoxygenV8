46 Variable* VariableMap::Declare(Zone* zone, Scope* scope,
47  const AstRawString* name, VariableMode mode,
48  VariableKind kind,
49  InitializationFlag initialization_flag,
50  MaybeAssignedFlag maybe_assigned_flag,
51  IsStaticFlag is_static_flag, bool* was_added) {
52  DCHECK_EQ(zone, allocator().zone());
53  // AstRawStrings are unambiguous, i.e., the same string is always represented
54  // by the same AstRawString*.
55  // FIXME(marja): fix the type of Lookup.
56  Entry* p = ZoneHashMap::LookupOrInsert(const_cast<AstRawString*>(name),
57  name->Hash());
58  *was_added = p->value == nullptr;
59  if (*was_added) {
60  // The variable has not been declared yet -> insert it.
61  DCHECK_EQ(name, p->key);
62  Variable* variable =
63  zone->New<Variable>(scope, name, mode, kind, initialization_flag,
64  maybe_assigned_flag, is_static_flag);
65  p->value = variable;
66  }
67  return reinterpret_cast<Variable*>(p->value);
68 }

75 void VariableMap::RemoveDynamic() {
76  Entry* current = ZoneHashMap::Start();
77  while (current != nullptr) {
78  Variable* var = reinterpret_cast<Variable*>(current->value);
79  if (var->is_dynamic()) {
80  ZoneHashMap::Remove(current);
81  if (current->exists()) continue;
82  }
83  current = ZoneHashMap::Next(current);
84  }
85 }

87 void VariableMap::Add(Variable* var) {
88  const AstRawString* name = var->raw_name();
89  Entry* p = ZoneHashMap::LookupOrInsert(const_cast<AstRawString*>(name),
90  name->Hash());
91  DCHECK_NULL(p->value);
92  DCHECK_EQ(name, p->key);
93  p->value = var;
94 }

96 Variable* VariableMap::Lookup(const AstRawString* name) {
97  Entry* p = ZoneHashMap::Lookup(const_cast<AstRawString*>(name), name->Hash());
98  if (p != nullptr) {
99  DCHECK(reinterpret_cast<const AstRawString*>(p->key) == name);
100  DCHECK_NOT_NULL(p->value);
101  return reinterpret_cast<Variable*>(p->value);
102  }
103  return nullptr;
104 }

436 template <typename IsolateT>
437 Scope* Scope::DeserializeScopeChain(IsolateT* isolate, Zone* zone,
438  Tagged<ScopeInfo> scope_info,
439  DeclarationScope* script_scope,
440  AstValueFactory* ast_value_factory,
441  DeserializationMode deserialization_mode,
442  ParseInfo* parse_info) {
443  // Reconstruct the outer scope chain from a closure's context chain.
444  Scope* current_scope = nullptr;
445  Scope* innermost_scope = nullptr;
446  Scope* outer_scope = nullptr;
447  while (!scope_info.is_null()) {
448  if (scope_info->scope_type() == WITH_SCOPE) {
449  if (scope_info->IsDebugEvaluateScope()) {
450  outer_scope =
451  zone->New<DeclarationScope>(zone, FUNCTION_SCOPE, ast_value_factory,
452  handle(scope_info, isolate));
453  outer_scope->set_is_dynamic_scope();
454  } else {
455  // For scope analysis, debug-evaluate is equivalent to a with scope.
456  outer_scope = zone->New<Scope>(zone, WITH_SCOPE, ast_value_factory,
457  handle(scope_info, isolate));
458  }
459  } else if (scope_info->is_script_scope()) {
460  // If we reach a script scope, it's the outermost scope. Install the
461  // scope info of this script context onto the existing script scope to
462  // avoid nesting script scopes.
463  if (deserialization_mode == DeserializationMode::kIncludingVariables) {
464  script_scope->SetScriptScopeInfo(handle(scope_info, isolate));
465  }
466  DCHECK(!scope_info->HasOuterScopeInfo());
467  break;
468  } else if (scope_info->scope_type() == FUNCTION_SCOPE) {
469  outer_scope = zone->New<DeclarationScope>(
470  zone, FUNCTION_SCOPE, ast_value_factory, handle(scope_info, isolate));
471 #if V8_ENABLE_WEBASSEMBLY
472  if (scope_info->IsAsmModule()) {
473  outer_scope->AsDeclarationScope()->set_is_asm_module();
474  }
475 #endif // V8_ENABLE_WEBASSEMBLY
476  } else if (scope_info->scope_type() == EVAL_SCOPE) {
477  outer_scope = zone->New<DeclarationScope>(
478  zone, EVAL_SCOPE, ast_value_factory, handle(scope_info, isolate));
479  } else if (scope_info->scope_type() == CLASS_SCOPE) {
480  outer_scope = zone->New<ClassScope>(isolate, zone, ast_value_factory,
481  handle(scope_info, isolate));
482  } else if (scope_info->scope_type() == BLOCK_SCOPE) {
483  if (scope_info->is_declaration_scope()) {
484  outer_scope = zone->New<DeclarationScope>(
485  zone, BLOCK_SCOPE, ast_value_factory, handle(scope_info, isolate));
486  } else {
487  outer_scope = zone->New<Scope>(zone, BLOCK_SCOPE, ast_value_factory,
488  handle(scope_info, isolate));
489  }
490  } else if (scope_info->scope_type() == MODULE_SCOPE) {
491  outer_scope = zone->New<ModuleScope>(handle(scope_info, isolate),
492  ast_value_factory);
493  if (parse_info) {
494  parse_info->set_has_module_in_scope_chain();
495  }
496  } else {
497  DCHECK_EQ(scope_info->scope_type(), CATCH_SCOPE);
498  DCHECK_EQ(scope_info->ContextLocalCount(), 1);
499  DCHECK_EQ(scope_info->ContextLocalMode(0), VariableMode::kVar);
500  DCHECK_EQ(scope_info->ContextLocalInitFlag(0), kCreatedInitialized);
501  DCHECK(scope_info->HasInlinedLocalNames());
502  Tagged<String> name = scope_info->ContextInlinedLocalName(0);
503  MaybeAssignedFlag maybe_assigned =
504  scope_info->ContextLocalMaybeAssignedFlag(0);
505  outer_scope =
506  zone->New<Scope>(zone,
507  ast_value_factory->GetString(
508  name, SharedStringAccessGuardIfNeeded(isolate)),
509  maybe_assigned, handle(scope_info, isolate));
510  }
511 
512  if (deserialization_mode == DeserializationMode::kScopesOnly) {
513  outer_scope->scope_info_ = Handle<ScopeInfo>::null();
514  }
515 
516  if (current_scope != nullptr) {
517  outer_scope->AddInnerScope(current_scope);
518  }
519  outer_scope->set_start_position(scope_info->StartPosition());
520  outer_scope->set_end_position(scope_info->EndPosition());
521 
522  current_scope = outer_scope;
523  if (innermost_scope == nullptr) innermost_scope = current_scope;
524  scope_info = scope_info->HasOuterScopeInfo() ? scope_info->OuterScopeInfo()
525  : Tagged<ScopeInfo>();
526  }
527 
528  if (deserialization_mode == DeserializationMode::kIncludingVariables) {
529  SetScriptScopeInfo(isolate, script_scope);
530  }
531 
532  if (innermost_scope == nullptr) return script_scope;
533  script_scope->AddInnerScope(current_scope);
534  return innermost_scope;
535 }

608  DCHECK(HasSimpleParameters() || is_block_scope() || is_being_lazily_parsed_);
609  DCHECK_EQ(factory == nullptr, is_being_lazily_parsed_);
610 
611  if (sloppy_block_functions_.is_empty()) return;
612 
613  // In case of complex parameters the current scope is the body scope and the
614  // parameters are stored in the outer scope.
615  Scope* parameter_scope = HasSimpleParameters() ? this : outer_scope_;
616  DCHECK(parameter_scope->is_function_scope() || is_eval_scope() ||
617  is_script_scope());
618 
619  DeclarationScope* decl_scope = GetNonEvalDeclarationScope();
620  Scope* outer_scope = decl_scope->outer_scope();
621 
622  // For each variable which is used as a function declaration in a sloppy
623  // block,
624  for (SloppyBlockFunctionStatement* sloppy_block_function :
625  sloppy_block_functions_) {
626  const AstRawString* name = sloppy_block_function->name();
627 
628  // If the variable wouldn't conflict with a lexical declaration
629  // or parameter,
630 
631  // Check if there's a conflict with a parameter.
632  Variable* maybe_parameter = parameter_scope->LookupLocal(name);
633  if (maybe_parameter != nullptr && maybe_parameter->is_parameter()) {
634  continue;
635  }
636 
637  // Check if there's a conflict with a lexical declaration
638  Scope* query_scope = sloppy_block_function->scope()->outer_scope();
639  bool should_hoist = true;
640 
641  // It is not sufficient to just do a Lookup on query_scope: for
642  // example, that does not prevent hoisting of the function in
643  // `{ let e; try {} catch (e) { function e(){} } }`
644  //
645  // Don't use a generic cache scope, as the cache scope would be the outer
646  // scope and we terminate the iteration there anyway.
647  do {
648  Variable* var = query_scope->LookupInScopeOrScopeInfo(name, query_scope);
649  if (var != nullptr && IsLexicalVariableMode(var->mode()) &&
650  !var->is_sloppy_block_function()) {
651  should_hoist = false;
652  break;
653  }
654  query_scope = query_scope->outer_scope();
655  } while (query_scope != outer_scope);
656 
657  if (!should_hoist) continue;
658 
659  if (factory) {
660  DCHECK(!is_being_lazily_parsed_);
661  int pos = sloppy_block_function->position();
662  bool ok = true;
663  bool was_added;
664  auto declaration = factory->NewVariableDeclaration(pos);
665  // Based on the preceding checks, it doesn't matter what we pass as
666  // sloppy_mode_block_scope_function_redefinition.
667  //
668  // This synthesized var for Annex B functions-in-block (FiB) may be
669  // declared multiple times for the same var scope, such as in the case of
670  // shadowed functions-in-block like the following:
671  //
672  // {
673  // function f() {}
674  // { function f() {} }
675  // }
676  //
677  // Redeclarations for vars do not create new bindings, but the
678  // redeclarations' initializers are still run. That is, shadowed FiB will
679  // result in multiple assignments to the same synthesized var.
680  Variable* var = DeclareVariable(
681  declaration, name, pos, VariableMode::kVar, NORMAL_VARIABLE,
682  Variable::DefaultInitializationFlag(VariableMode::kVar), &was_added,
683  nullptr, &ok);
684  DCHECK(ok);
685  VariableProxy* source =
686  factory->NewVariableProxy(sloppy_block_function->var());
687  VariableProxy* target = factory->NewVariableProxy(var);
688  Assignment* assignment = factory->NewAssignment(
689  sloppy_block_function->init(), target, source, pos);
690  assignment->set_lookup_hoisting_mode(LookupHoistingMode::kLegacySloppy);
691  Statement* statement = factory->NewExpressionStatement(assignment, pos);
692  sloppy_block_function->set_statement(statement);
693  } else {
694  DCHECK(is_being_lazily_parsed_);
695  bool was_added;
696  Variable* var = DeclareVariableName(name, VariableMode::kVar, &was_added);
697  if (sloppy_block_function->init() == Token::kAssign) {
698  var->SetMaybeAssigned();
699  }
700  }
701  }

983 Variable* Scope::LookupInScopeInfo(const AstRawString* name, Scope* cache) {
984  DCHECK(!scope_info_.is_null());
985  DCHECK(this->IsOuterScopeOf(cache));
986  DCHECK_NULL(cache->variables_.Lookup(name));
987  DisallowGarbageCollection no_gc;
988 
989  Tagged<String> name_handle = *name->string();
990  Tagged<ScopeInfo> scope_info = *scope_info_;
991  // The Scope is backed up by ScopeInfo. This means it cannot operate in a
992  // heap-independent mode, and all strings must be internalized immediately. So
993  // it's ok to get the Handle<String> here.
994  bool found = false;
995 
996  VariableLocation location;
997  int index;
998  VariableLookupResult lookup_result;
999 
1000  {
1001  location = VariableLocation::CONTEXT;
1002  index = scope_info->ContextSlotIndex(name_handle, &lookup_result);
1003  found = index >= 0;
1004  }
1005 
1006  if (!found && is_module_scope()) {
1007  location = VariableLocation::MODULE;
1008  index = scope_info->ModuleIndex(name_handle, &lookup_result.mode,
1009  &lookup_result.init_flag,
1010  &lookup_result.maybe_assigned_flag);
1011  found = index != 0;
1012  }
1013 
1014  if (!found) {
1015  index = scope_info->FunctionContextSlotIndex(name_handle);
1016  if (index < 0) return nullptr; // Nowhere found.
1017  Variable* var = AsDeclarationScope()->DeclareFunctionVar(name, cache);
1018  DCHECK_EQ(VariableMode::kConst, var->mode());
1019  var->AllocateTo(VariableLocation::CONTEXT, index);
1020  return cache->variables_.Lookup(name);
1021  }
1022 
1023  if (!is_module_scope()) {
1024  DCHECK_NE(index, scope_info->ReceiverContextSlotIndex());
1025  }
1026 
1027  bool was_added;
1028  Variable* var = cache->variables_.Declare(
1029  zone(), this, name, lookup_result.mode, NORMAL_VARIABLE,
1030  lookup_result.init_flag, lookup_result.maybe_assigned_flag,
1031  IsStaticFlag::kNotStatic, &was_added);
1032  DCHECK(was_added);
1033  var->AllocateTo(location, index);
1034  return var;
1035 }

1080 Variable* Scope::DeclareLocal(const AstRawString* name, VariableMode mode,
1081  VariableKind kind, bool* was_added,
1082  InitializationFlag init_flag) {
1083  DCHECK(!already_resolved_);
1084  // Private methods should be declared with ClassScope::DeclarePrivateName()
1085  DCHECK(!IsPrivateMethodOrAccessorVariableMode(mode));
1086  // This function handles VariableMode::kVar, VariableMode::kLet,
1087  // VariableMode::kConst, VariableMode::kUsing, and VariableMode::kAwaitUsing
1088  // modes. VariableMode::kDynamic variables are introduced during variable
1089  // allocation, and VariableMode::kTemporary variables are allocated via
1090  // NewTemporary().
1091  DCHECK(IsDeclaredVariableMode(mode));
1092  DCHECK_IMPLIES(GetDeclarationScope()->is_being_lazily_parsed(),
1093  mode == VariableMode::kVar || mode == VariableMode::kLet ||
1094  mode == VariableMode::kConst ||
1095  mode == VariableMode::kUsing ||
1096  mode == VariableMode::kAwaitUsing);
1097  DCHECK(!GetDeclarationScope()->was_lazily_parsed());
1098  Variable* var =
1099  Declare(zone(), name, mode, kind, init_flag, kNotAssigned, was_added);
1100 
1101  // Pessimistically assume that top-level variables will be assigned and used.
1102  //
1103  // Top-level variables in a script can be accessed by other scripts or even
1104  // become global properties. While this does not apply to top-level variables
1105  // in a module (assuming they are not exported), we must still mark these as
1106  // assigned because they might be accessed by a lazily parsed top-level
1107  // function, which, for efficiency, we preparse without variable tracking.
1108  if (is_script_scope() || is_module_scope()) {
1109  if (mode != VariableMode::kConst) var->SetMaybeAssigned();
1110  var->set_is_used();
1111  }
1112 
1113  return var;
1114 }

1116 Variable* Scope::DeclareVariable(
1117  Declaration* declaration, const AstRawString* name, int pos,
1118  VariableMode mode, VariableKind kind, InitializationFlag init,
1119  bool* was_added, bool* sloppy_mode_block_scope_function_redefinition,
1120  bool* ok) {
1121  // Private methods should be declared with ClassScope::DeclarePrivateName()
1122  DCHECK(!IsPrivateMethodOrAccessorVariableMode(mode));
1123  DCHECK(IsDeclaredVariableMode(mode));
1124  DCHECK(!already_resolved_);
1125  DCHECK(!GetDeclarationScope()->is_being_lazily_parsed());
1126  DCHECK(!GetDeclarationScope()->was_lazily_parsed());
1127 
1128  if (mode == VariableMode::kVar && !is_declaration_scope()) {
1129  return GetDeclarationScope()->DeclareVariable(
1130  declaration, name, pos, mode, kind, init, was_added,
1131  sloppy_mode_block_scope_function_redefinition, ok);
1132  }
1133  DCHECK(!is_catch_scope());
1134  DCHECK(!is_with_scope());
1135  DCHECK(is_declaration_scope() ||
1136  (IsLexicalVariableMode(mode) && is_block_scope()));
1137 
1138  DCHECK_NOT_NULL(name);
1139 
1140  Variable* var = LookupLocal(name);
1141  // Declare the variable in the declaration scope.
1142  *was_added = var == nullptr;
1143  if (V8_LIKELY(*was_added)) {
1144  if (V8_UNLIKELY(is_eval_scope() && is_sloppy(language_mode()) &&
1145  mode == VariableMode::kVar)) {
1146  // In a var binding in a sloppy direct eval, pollute the enclosing scope
1147  // with this new binding by doing the following:
1148  // The proxy is bound to a lookup variable to force a dynamic declaration
1149  // using the DeclareEvalVar or DeclareEvalFunction runtime functions.
1150  DCHECK_EQ(NORMAL_VARIABLE, kind);
1151  var = NonLocal(name, VariableMode::kDynamic);
1152  // Mark the var as used in case anyone outside the eval wants to use it.
1153  var->set_is_used();
1154  } else {
1155  // Declare the name.
1156  var = DeclareLocal(name, mode, kind, was_added, init);
1157  DCHECK(*was_added);
1158  }
1159  } else {
1160  var->SetMaybeAssigned();
1161  if (V8_UNLIKELY(IsLexicalVariableMode(mode) ||
1162  IsLexicalVariableMode(var->mode()))) {
1163  // The name was declared in this scope before; check for conflicting
1164  // re-declarations. We have a conflict if either of the declarations is
1165  // not a var (in script scope, we also have to ignore legacy const for
1166  // compatibility). There is similar code in runtime.cc in the Declare
1167  // functions. The function CheckConflictingVarDeclarations checks for
1168  // var and let bindings from different scopes whereas this is a check
1169  // for conflicting declarations within the same scope. This check also
1170  // covers the special case
1171  //
1172  // function () { let x; { var x; } }
1173  //
1174  // because the var declaration is hoisted to the function scope where
1175  // 'x' is already bound.
1176  //
1177  // In harmony we treat re-declarations as early errors. See ES5 16 for a
1178  // definition of early errors.
1179  //
1180  // Allow duplicate function decls for web compat, see bug 4693.
1181  *ok = var->is_sloppy_block_function() &&
1182  kind == SLOPPY_BLOCK_FUNCTION_VARIABLE;
1183  *sloppy_mode_block_scope_function_redefinition = *ok;
1184  }
1185  }
1186  DCHECK_NOT_NULL(var);
1187 
1188  // We add a declaration node for every declaration. The compiler
1189  // will only generate code if necessary. In particular, declarations
1190  // for inner local variables that do not represent functions won't
1191  // result in any generated code.
1192  //
1193  // This will lead to multiple declaration nodes for the
1194  // same variable if it is declared several times. This is not a
1195  // semantic issue, but it may be a performance issue since it may
1196  // lead to repeated DeclareEvalVar or DeclareEvalFunction calls.

1289 Declaration* DeclarationScope::CheckConflictingVarDeclarations(
1290  bool* allowed_catch_binding_var_redeclaration) {
1291  if (has_checked_syntax_) return nullptr;
1292  for (Declaration* decl : decls_) {
1293  // Lexical vs lexical conflicts within the same scope have already been
1294  // captured in Parser::Declare. The only conflicts we still need to check
1295  // are lexical vs nested var.
1296  if (decl->IsVariableDeclaration() &&
1297  decl->AsVariableDeclaration()->AsNested() != nullptr) {
1298  Scope* current = decl->AsVariableDeclaration()->AsNested()->scope();
1299  if (decl->var()->mode() != VariableMode::kVar &&
1300  decl->var()->mode() != VariableMode::kDynamic)
1301  continue;
1302  // Iterate through all scopes until the declaration scope.
1303  do {
1304  // There is a conflict if there exists a non-VAR binding.
1305  Variable* other_var = current->LookupLocal(decl->var()->raw_name());
1306  if (current->is_catch_scope()) {
1307  *allowed_catch_binding_var_redeclaration |= other_var != nullptr;
1308  current = current->outer_scope();
1309  continue;
1310  }
1311  if (other_var != nullptr) {
1312  DCHECK(IsLexicalVariableMode(other_var->mode()));
1313  return decl;
1314  }
1315  current = current->outer_scope();
1316  } while (current != this);
1317  }
1318  }
1319 
1320  if (V8_LIKELY(!is_eval_scope())) return nullptr;
1321  if (!is_sloppy(language_mode())) return nullptr;
1322 
1323  // Var declarations in sloppy eval are hoisted to the first non-eval
1324  // declaration scope. Check for conflicts between the eval scope that
1325  // declaration scope.
1326  Scope* end = outer_scope()->GetNonEvalDeclarationScope()->outer_scope();
1327 
1328  for (Declaration* decl : decls_) {
1329  if (IsLexicalVariableMode(decl->var()->mode())) continue;
1330  Scope* current = outer_scope_;
1331  // Iterate through all scopes until and including the declaration scope.
1332  do {
1333  // There is a conflict if there exists a non-VAR binding up to the
1334  // declaration scope in which this sloppy-eval runs.
1335  //
1336  // Use the current scope as the cache. We can't use the regular cache
1337  // since catch scope vars don't result in conflicts, but they will mask
1338  // variables for regular scope resolution. We have to make sure to not put
1339  // masked variables in the cache used for regular lookup.
1340  Variable* other_var =
1341  current->LookupInScopeOrScopeInfo(decl->var()->raw_name(), current);
1342  if (other_var != nullptr && !current->is_catch_scope()) {
1343  // If this is a VAR, then we know that it doesn't conflict with
1344  // anything, so we can't conflict with anything either. The one
1345  // exception is the binding variable in catch scopes, which is handled
1346  // by the if above.
1347  if (!IsLexicalVariableMode(other_var->mode())) break;
1348  return decl;
1349  }
1350  current = current->outer_scope();
1351  } while (current != end);
1352  }
1353  return nullptr;
1354 }

2291 void UpdateNeedsHoleCheck(Variable* var, VariableProxy* proxy, Scope* scope) {
2292  if (var->mode() == VariableMode::kDynamicLocal) {
2293  // Dynamically introduced variables never need a hole check (since they're
2294  // VariableMode::kVar bindings, either from var or function declarations),
2295  // but the variable they shadow might need a hole check, which we want to do
2296  // if we decide that no shadowing variable was dynamically introduced.
2297  DCHECK_EQ(kCreatedInitialized, var->initialization_flag());
2298  return UpdateNeedsHoleCheck(var->local_if_not_shadowed(), proxy, scope);
2299  }
2300 
2301  if (var->initialization_flag() == kCreatedInitialized) return;
2302 
2303  // It's impossible to eliminate module import hole checks here, because it's
2304  // unknown at compilation time whether the binding referred to in the
2305  // exporting module itself requires hole checks.
2306  if (var->location() == VariableLocation::MODULE && !var->IsExport()) {
2307  SetNeedsHoleCheck(var, proxy, Variable::kHasHoleCheckUseInUnknownScope);
2308  return;
2309  }
2310 
2311  // Check if the binding really needs an initialization check. The check
2312  // can be skipped in the following situation: we have a VariableMode::kLet or
2313  // VariableMode::kConst binding, both the Variable and the VariableProxy have
2314  // the same declaration scope (i.e. they are both in global code, in the same
2315  // function or in the same eval code), the VariableProxy is in the source
2316  // physically located after the initializer of the variable, and that the
2317  // initializer cannot be skipped due to a nonlinear scope.
2318  //
2319  // The condition on the closure scopes is a conservative check for
2320  // nested functions that access a binding and are called before the
2321  // binding is initialized:
2322  // function() { f(); let x = 1; function f() { x = 2; } }
2323  //
2324  // The check cannot be skipped on non-linear scopes, namely switch
2325  // scopes, to ensure tests are done in cases like the following:
2326  // switch (1) { case 0: let x = 2; case 1: f(x); }
2327  // The scope of the variable needs to be checked, in case the use is
2328  // in a sub-block which may be linear.
2329  if (var->scope()->GetClosureScope() != scope->GetClosureScope()) {
2330  SetNeedsHoleCheck(var, proxy,
2331  Variable::kHasHoleCheckUseInDifferentClosureScope);
2332  return;
2333  }
2334 
2335  // We should always have valid source positions.
2336  DCHECK_NE(var->initializer_position(), kNoSourcePosition);
2337  DCHECK_NE(proxy->position(), kNoSourcePosition);
2338 
2339  if (var->scope()->is_nonlinear() ||
2340  var->initializer_position() >= proxy->position()) {
2341  SetNeedsHoleCheck(var, proxy, Variable::kHasHoleCheckUseInSameClosureScope);
2342  return;
2343  }
2344 }

2445 void DeclarationScope::AllocateParameterLocals() {
2446  DCHECK(is_function_scope());

2449  if (arguments_ != nullptr) {
2450  DCHECK(!is_arrow_scope());

2607 void Scope::AllocateVariablesRecursively() {
2608  this->ForEach([](Scope* scope) -> Iteration {
2609  DCHECK(!scope->already_resolved_);
2610  if (WasLazilyParsed(scope)) return Iteration::kContinue;
2611  if (scope->sloppy_eval_can_extend_vars_) {
2612  scope->num_heap_slots_ = Context::MIN_CONTEXT_EXTENDED_SLOTS;
2613  }
2614  DCHECK_EQ(scope->ContextHeaderLength(), scope->num_heap_slots_);
2615

2617  // Parameters must be allocated first, if any.
2618  if (scope->is_declaration_scope()) {
2619  scope->AsDeclarationScope()->AllocateReceiver();
2620  if (scope->is_function_scope()) {
2621  scope->AsDeclarationScope()->AllocateParameterLocals();
2622  }
2623  }
2624  scope->AllocateNonParameterLocalsAndDeclaredGlobals();
2625 
2626  // Force allocation of a context for this scope if necessary. For a 'with'
2627  // scope and for a function scope that makes an 'eval' call we need a
2628  // context, even if no local variables were statically allocated in the
2629  // scope. Likewise for modules and function scopes representing asm.js
2630  // modules. Also force a context, if the scope is stricter than the outer
2631  // scope.
2632  bool must_have_context =
2633  scope->is_with_scope() || scope->is_module_scope() ||
2634 #if V8_ENABLE_WEBASSEMBLY
2635  scope->IsAsmModule() ||
2636 #endif // V8_ENABLE_WEBASSEMBLY
2637  scope->ForceContextForLanguageMode() ||
2638  (scope->is_function_scope() &&
2639  scope->AsDeclarationScope()->sloppy_eval_can_extend_vars()) ||
2640  (scope->is_block_scope() && scope->is_declaration_scope() &&
2641  scope->AsDeclarationScope()->sloppy_eval_can_extend_vars());
2642 
2643  // If we didn't allocate any locals in the local context, then we only
2644  // need the minimal number of slots if we must have a context.
2645  if (scope->num_heap_slots_ == scope->ContextHeaderLength() &&
2646  !must_have_context) {
2647  scope->num_heap_slots_ = 0;
2648  }
2649 
2650  // If the number of context slots are over the function context threshold,
2651  // do not allocate context cells.
2652  if (scope->is_function_scope() &&
2653  scope->ContextLocalCount() > v8_flags.function_context_cells_max_size) {
2654  scope->has_context_cells_ = false;
2655  }
2656 
2657  // Allocation done.
2658  DCHECK(scope->num_heap_slots_ == 0 ||
2659  scope->num_heap_slots_ >= scope->ContextHeaderLength());

2661  });

2730 void DeclarationScope::RecalcPrivateNameContextChain() {
2731  // The outermost scope in a class heritage expression is marked to skip the
2732  // class scope during private name resolution. It is possible, however, that
2733  // either the class scope won't require a Context and ScopeInfo, or the
2734  // outermost scope in the heritage position won't. Simply copying the bit from
2735  // full parse into the ScopeInfo will break lazy compilation. In the former
2736  // case the scope that is marked to skip its outer scope will incorrectly skip
2737  // a different class scope than the one we intended to skip. In the latter
2738  // case variables resolved through an inner scope will incorrectly check the
2739  // class scope since we lost the skip bit from the outermost heritage scope.
2740  //
2741  // This method fixes both cases by, in outermost to innermost order, copying
2742  // the value of the skip bit from outer scopes that don't require a Context.
2743  DCHECK(needs_private_name_context_chain_recalc_);
2744  this->ForEach([](Scope* scope) {
2745  Scope* outer = scope->outer_scope();
2746  if (!outer) return Iteration::kDescend;
2747  if (!outer->NeedsContext()) {
2748  scope->private_name_lookup_skips_outer_class_ =
2749  outer->private_name_lookup_skips_outer_class();
2750  }
2751  if (!scope->is_function_scope() ||
2752  scope->AsDeclarationScope()->ShouldEagerCompile()) {
2753  return Iteration::kDescend;
2754  }
2755  return Iteration::kContinue;
2756  });
