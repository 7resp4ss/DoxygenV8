18 // ----------------------------------------------------------------------------
19 // MemoryProtectionKey
20 //
21 // This class has static methods for the different platform specific
22 // functions related to memory protection key support.
23 //
24 // TODO(416209124): Once this has stabilized further, consider moving it into a
25 // MemoryProtectionKeyProvider or similar class expsed via the Platform and
26 // provided by the Embedder.
27 class V8_BASE_EXPORT MemoryProtectionKey {
28  public:
29  // Sentinel value if there is no PKU support or allocation of a key failed.
30  // This is also the return value on an error of pkey_alloc() and has the
31  // benefit that calling pkey_mprotect() with -1 behaves the same as regular
32  // mprotect().
33  static constexpr int kNoMemoryProtectionKey = -1;
34 
35  // The default ProtectionKey can be used to remove pkey assignments.
36  static constexpr int kDefaultProtectionKey = 0;
37 
38  // Permissions for memory protection keys on top of the page's permissions.
39  // NOTE: Since there is no executable bit, the executable permission cannot be
40  // withdrawn by memory protection keys.
41  enum Permission {
42  kNoRestrictions = 0,
43  kDisableAccess = 1,
44  kDisableWrite = 2,
45  };
46 
47 // If sys/mman.h has PKEY support (on newer Linux distributions), ensure that
48 // our definitions of the permissions is consistent with the ones in glibc.
49 #if defined(PKEY_DISABLE_ACCESS)
50  static_assert(kDisableAccess == PKEY_DISABLE_ACCESS);
51  static_assert(kDisableWrite == PKEY_DISABLE_WRITE);
52 #endif
53 
54  // Determine if the operating systems exposes the memory protection key APIs.
55  //
56  // This is a cheap test to see if the necessary library routines are
57  // available. It does not test whether the CPU and/or the kernel support
58  // PKEYs. For that, use the more expensive TestKeyAllocation() routine.
59  static bool HasMemoryProtectionKeyAPIs();
60 
61  // Test whether memory protection keys can be successfully allocated on this
62  // system at runtime, implying that both the CPU and the kernel support PKEYs.
63  // This is a somewhat expensive test as it generally involves two syscalls
64  // (e.g. pkey_alloc and pkey_free on Linux).
65  // Note: as this involves allocating a PKEY, and since there's a limited
66  // number of keys, it will fail if all keys have already been allocated.
67  // Similarly, if this succeeds, it does not guarantee that a PKEY can be
68  // allocated in the future.
69  static bool TestKeyAllocation();
70 
71  // Allocates a new key. Returns kNoMemoryProtectionKey on error.
72  static int AllocateKey();
73 
74  // Register a memory protection key that was allocated through other means.
75  //
76  // This is currently needed for SetDefaultPermissionsForAllKeysInSignalHandler
77  // to work correctly. In the future, we should obtain a
78  // MemoryProtectionKeyProvider from the embedder so that there is a single
79  // entity in the process responsible for allocating keys. Then this will no
80  // longer be needed. See also https://crbug.com/416209124.
81  static void RegisterExternallyAllocatedKey(int key);
82 
83  // Frees the given key which must have been obtained through AllocateKey.
84  static void FreeKey(int key);
85 
86  // Associates a memory protection {key} with the given {region}.
87  // If {key} is {kNoMemoryProtectionKey} this behaves like "plain"
88  // {SetPermissions()} and associates the default key to the region. That is,
89  // explicitly calling with {kNoMemoryProtectionKey} can be used to
90  // disassociate any protection key from a region. This also means "plain"
91  // {SetPermissions()} disassociates the key from a region, making the key's
92  // access restrictions irrelevant/inactive for that region. Returns true if
93  // changing permissions and key was successful. (Returns a bool to be
94  // consistent with {SetPermissions()}). The {page_permissions} are the
95  // permissions of the page, not the key. For changing the permissions of the
96  // key, use {SetPermissionsForKey()} instead.
97  static bool SetPermissionsAndKey(
98  base::AddressRegion region,
99  v8::PageAllocator::Permission page_permissions, int key);
100 
101  // Set the key's permissions. {key} must be valid, i.e. not
102  // {kNoMemoryProtectionKey}.
103  static void SetPermissionsForKey(int key, Permission permissions);
104 
105  // Get the permissions of the protection key {key} for the current thread.
106  static Permission GetKeyPermission(int key);
107 
108  // Set the default permissions for all active keys in a signal handler.
109  //
110  // Signal handlers on some platforms, for example older Linux kernels, run
111  // without access to any non-default pkey. As such, they will crash when
112  // trying to access any memory using another key. To work around that, this
113  // method should be called at the start of a signal handler to restore the
114  // default permissions for all active keys.
115  //
116  // At the moment, the default permission for a key is read-only access
117  // (kDisableWrite). If necessary in the future, we could make that
118  // configurable by passing in the default permissions into AllocateKey and
119  // remembering them alongside the key.
120  static void SetDefaultPermissionsForAllKeysInSignalHandler();
121 };
