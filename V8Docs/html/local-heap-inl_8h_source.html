57 template <typename Callback>
58 V8_INLINE void LocalHeap::ParkAndExecuteCallback(Callback callback) {
59  // This method is given as a callback to the stack trampoline, when the stack
60  // marker has just been set.
61 #if defined(V8_ENABLE_DIRECT_HANDLE) && defined(ENABLE_SLOW_DCHECKS)
62  // Reset the number of direct handles that are below the stack marker.
63  // It will be restored before the method returns.
64  DirectHandleBase::ResetNumberOfHandlesScope scope;
65 #endif // V8_ENABLE_DIRECT_HANDLE && ENABLE_SLOW_DCHECKS
66  ParkedScope parked(this);
67  // Provide the parked scope as a witness, if the callback expects it.
68  if constexpr (std::is_invocable_v<Callback, const ParkedScope&>) {
69  callback(parked);
70  } else {
71  callback();
72  }
73 }
