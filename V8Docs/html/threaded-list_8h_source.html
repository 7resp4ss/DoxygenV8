23 // Represents a linked list that threads through the nodes in the linked list.
24 // Entries in the list are pointers to nodes. By default nodes need to have a
25 // T** next() method that returns the location where the next value is stored.
26 // The kSupportsUnsafeInsertion flag defines whether the list supports insertion
27 // of new elements into the list by just rewiring the next pointers without
28 // updating the list object itself. Such an insertion might invalidate the
29 // pointer to list tail and thus requires additional steps to recover the
30 // pointer to the tail.
31 // The default can be overwritten by providing a ThreadedTraits class.
32 template <typename T, typename BaseClass,
33  typename TLTraits = ThreadedListTraits<T>,
34  bool kSupportsUnsafeInsertion = false>
35 class ThreadedListBase final : public BaseClass {
36  public:
37  ThreadedListBase() : head_(nullptr), tail_(&head_) {}
38  ThreadedListBase(const ThreadedListBase&) = delete;
39  ThreadedListBase& operator=(const ThreadedListBase&) = delete;

41  void Add(T* v) {
42  EnsureValidTail();
43  DCHECK_NULL(*tail_);
44  DCHECK_NULL(*TLTraits::next(v));
45  *tail_ = v;
46  tail_ = TLTraits::next(v);
47  // Check that only one element was added (and that hasn't created a cycle).
48  DCHECK_NULL(*tail_);
49  }
50 
51  void AddFront(T* v) {
52  DCHECK_NULL(*TLTraits::next(v));
53  DCHECK_NOT_NULL(v);
54  T** const next = TLTraits::next(v);
55 
56  *next = head_;
57  if (head_ == nullptr) tail_ = next;
58  head_ = v;
59  }
60 
61  // This temporarily breaks the tail_ invariant, and it should only be called
62  // if we support unsafe insertions.
63  static void AddAfter(T* after_this, T* v) {
64  DCHECK(kSupportsUnsafeInsertion);
65  DCHECK_NULL(*TLTraits::next(v));
66  *TLTraits::next(v) = *TLTraits::next(after_this);
67  *TLTraits::next(after_this) = v;
68  }
69 
70  void DropHead() {
71  DCHECK_NOT_NULL(head_);
72 
73  T* old_head = head_;
74  head_ = *TLTraits::next(head_);
75  if (head_ == nullptr) tail_ = &head_;
76  *TLTraits::next(old_head) = nullptr;
77  }

79  bool Contains(T* v) {
80  for (Iterator it = begin(); it != end(); ++it) {
81  if (*it == v) return true;
82  }
83  return false;
84  }

86  void Append(ThreadedListBase&& list) {
87  if (list.is_empty()) return;
88 
89  EnsureValidTail();
90  *tail_ = list.head_;
91  tail_ = list.tail_;
92  list.Clear();
93  }
94 
95  void Prepend(ThreadedListBase&& list) {
96  if (list.head_ == nullptr) return;
97 
98  EnsureValidTail();
99  T* new_head = list.head_;
100  *list.tail_ = head_;
101  if (head_ == nullptr) {
102  tail_ = list.tail_;
103  }
104  head_ = new_head;
105  list.Clear();
106  }

108  // This is only valid if {v} is in the current list.
109  void TruncateAt(ThreadedListBase* rem, T* v) {
110  CHECK_NOT_NULL(rem);
111  CHECK_NOT_NULL(v);
112  CHECK(rem->is_empty());
113  Iterator it = begin();
114  T* last = nullptr;
115  for (; it != end(); ++it) {
116  if (*it == v) {
117  break;
118  }
119  last = *it;
120  }
121  CHECK_EQ(v, *it);
122 
123  // Remaining list.
124  rem->head_ = v;
125  rem->tail_ = tail_;
126 
127  if (last == nullptr) {
128  // The head must point to v, so we return the empty list.
129  CHECK_EQ(head_, v);
130  Clear();
131  } else {
132  tail_ = TLTraits::next(last);
133  *tail_ = nullptr;
134  }
135  }
136 
137  void Clear() {
138  head_ = nullptr;
139  tail_ = &head_;
140  }

159  bool Remove(T* v) {
160  T* current = first();
161  if (current == v) {
162  DropHead();
163  return true;
164  }
165 
166  EnsureValidTail();
167  while (current != nullptr) {
168  T* next = *TLTraits::next(current);
169  if (next == v) {
170  *TLTraits::next(current) = *TLTraits::next(next);
171  *TLTraits::next(next) = nullptr;
172 
173  if (TLTraits::next(next) == tail_) {
174  tail_ = TLTraits::next(current);
175  }
176  return true;
177  }
178  current = next;
179  }
180  return false;
181  }

276  // Moves the tail of the from_list, starting at the from_location, to the end
277  // of this list.
278  void MoveTail(ThreadedListBase* from_list, Iterator from_location) {
279  if (from_list->end() != from_location) {
280  DCHECK_NULL(*tail_);
281  *tail_ = *from_location;
282  tail_ = from_list->tail_;
283  from_list->Rewind(from_location);
284  }
285  }

287  // Removes the element at `it`, and returns a new iterator pointing to the
288  // element following the removed element (if `it` was pointing to the last
289  // element, then `end()` is returned). The head and the tail are updated. `it`
290  // should not be `end()`. Iterators that are currently on the same element as
291  // `it` are invalidated. Other iterators are not affected. If the last element
292  // is removed, existing `end()` iterators will be invalidated.
293  Iterator RemoveAt(Iterator it) {
294  if (*it.entry_ == head_) {
295  DropHead();
296  return begin();
297  } else if (tail_ == TLTraits::next(*it.entry_)) {
298  tail_ = it.entry_;
299  *it.entry_ = nullptr;
300  return end();
301  } else {
302  T* old_entry = *it.entry_;
303  *it.entry_ = *TLTraits::next(*it.entry_);
304  *TLTraits::next(old_entry) = nullptr;
305  return Iterator(it.entry_);
306  }
307  }

326  bool Verify() const {
327  T* last = this->first();
328  if (last == nullptr) {
329  CHECK_EQ(&head_, tail_);
330  } else {
331  while (*TLTraits::next(last) != nullptr) {
332  last = *TLTraits::next(last);
333  }
334  CHECK_EQ(TLTraits::next(last), tail_);
335  }
336  return true;
337  }

339  inline void EnsureValidTail() const {
340  if (!kSupportsUnsafeInsertion) {
341  DCHECK_EQ(*tail_, nullptr);
342  return;
343  }
344  // If kSupportsUnsafeInsertion, then we support adding a new element by
345  // using the pointer to a certain element. E.g., imagine list A -> B -> C,
346  // we can add D after B, by just moving the pointer of B to D and D to
347  // whatever B used to point to. We do not need to know the beginning of the
348  // list (ie. to have a pointer to the ThreadList class). This however might
349  // break the tail_ invariant. We ensure this here, by manually looking for
350  // the tail of the list.
351  if (*tail_ == nullptr) return;
352  T* last = *tail_;
353  if (last != nullptr) {
354  while (*TLTraits::next(last) != nullptr) {
355  last = *TLTraits::next(last);
356  }
357  tail_ = TLTraits::next(last);
358  }
359  }
360 
361  private:
362  T* head_;
363  mutable T** tail_; // We need to ensure a valid `tail_` even when using a
364  // const Iterator.
365 };
