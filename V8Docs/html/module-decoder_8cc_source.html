77 ModuleResult DecodeWasmModule(
78  WasmEnabledFeatures enabled_features,
79  base::Vector<const uint8_t> wire_bytes, bool validate_functions,
80  ModuleOrigin origin, Counters* counters,
81  std::shared_ptr<metrics::Recorder> metrics_recorder,
82  v8::metrics::Recorder::ContextId context_id, DecodingMethod decoding_method,
83  WasmDetectedFeatures* detected_features) {
84  if (counters) {
85  auto size_counter =
86  SELECT_WASM_COUNTER(counters, origin, wasm, module_size_bytes);
87  static_assert(kV8MaxWasmModuleSize < kMaxInt);
88  size_counter->AddSample(static_cast<int>(wire_bytes.size()));
89  }
90 
91  v8::metrics::WasmModuleDecoded metrics_event;
92  base::ElapsedTimer timer;
93  timer.Start();
94  ModuleResult result =
95  DecodeWasmModule(enabled_features, wire_bytes, validate_functions, origin,
96  detected_features);
97  if (counters && result.ok()) {
98  auto counter =
99  SELECT_WASM_COUNTER(counters, origin, wasm_functions_per, module);
100  counter->AddSample(
101  static_cast<int>(result.value()->num_declared_functions));
102  }
103 
104  // Record event metrics.
105  metrics_event.wall_clock_duration_in_us = timer.Elapsed().InMicroseconds();
106  timer.Stop();
107  metrics_event.success = result.ok();
108  metrics_event.async = decoding_method == DecodingMethod::kAsync ||
109  decoding_method == DecodingMethod::kAsyncStream;
110  metrics_event.streamed = decoding_method == DecodingMethod::kSyncStream ||
111  decoding_method == DecodingMethod::kAsyncStream;
112  if (result.ok()) {
113  metrics_event.function_count = result.value()->num_declared_functions;
114  }
115  metrics_event.module_size_in_bytes = wire_bytes.size();
116  metrics_recorder->DelayMainThreadEvent(metrics_event, context_id);
117 
118  return result;
119 }

227 AsmJsOffsetsResult DecodeAsmJsOffsets(
228  base::Vector<const uint8_t> encoded_offsets) {
229  std::vector<AsmJsOffsetFunctionEntries> functions;
230 
231  Decoder decoder(encoded_offsets);
232  uint32_t functions_count = decoder.consume_u32v("functions count");
233  // Consistency check.
234  DCHECK_GE(encoded_offsets.size(), functions_count);
235  functions.reserve(functions_count);
236 
237  for (uint32_t i = 0; i < functions_count; ++i) {
238  uint32_t size = decoder.consume_u32v("table size");
239  if (size == 0) {
240  functions.emplace_back();
241  continue;
242  }
243  DCHECK(decoder.checkAvailable(size));
244  const uint8_t* table_end = decoder.pc() + size;
245  uint32_t locals_size = decoder.consume_u32v("locals size");
246  int function_start_position = decoder.consume_u32v("function start pos");
247  int function_end_position = function_start_position;
248  int last_byte_offset = locals_size;
249  int last_asm_position = function_start_position;
250  std::vector<AsmJsOffsetEntry> func_asm_offsets;
251  func_asm_offsets.reserve(size / 4); // conservative estimation
252  // Add an entry for the stack check, associated with position 0.
253  func_asm_offsets.push_back(
254  {0, function_start_position, function_start_position});
255  while (decoder.pc() < table_end) {
256  DCHECK(decoder.ok());
257  last_byte_offset += decoder.consume_u32v("byte offset delta");
258  int call_position =
259  last_asm_position + decoder.consume_i32v("call position delta");
260  int to_number_position =
261  call_position + decoder.consume_i32v("to_number position delta");
262  last_asm_position = to_number_position;
263  if (decoder.pc() == table_end) {
264  // The last entry is the function end marker.
265  DCHECK_EQ(call_position, to_number_position);
266  function_end_position = call_position;
267  } else {
268  func_asm_offsets.push_back(
269  {last_byte_offset, call_position, to_number_position});
270  }
271  }
272  DCHECK_EQ(decoder.pc(), table_end);
273  functions.emplace_back(AsmJsOffsetFunctionEntries{
274  function_start_position, function_end_position,
275  std::move(func_asm_offsets)});
276  }
277  DCHECK(decoder.ok());
278  DCHECK(!decoder.more());
279 
280  return decoder.toResult(AsmJsOffsets{std::move(functions)});
281 }

283 std::vector<CustomSectionOffset> DecodeCustomSections(
284  base::Vector<const uint8_t> bytes) {
285  Decoder decoder(bytes);
286  decoder.consume_bytes(4, "wasm magic");
287  decoder.consume_bytes(4, "wasm version");
288 
289  std::vector<CustomSectionOffset> result;
290 
291  while (decoder.more()) {
292  uint8_t section_code = decoder.consume_u8("section code");
293  uint32_t section_length = decoder.consume_u32v("section length");
294  uint32_t section_start = decoder.pc_offset();
295  if (section_code != 0) {
296  // Skip known sections.
297  decoder.consume_bytes(section_length, "section bytes");
298  continue;
299  }
300  uint32_t name_length = decoder.consume_u32v("name length");
301  uint32_t name_offset = decoder.pc_offset();
302  decoder.consume_bytes(name_length, "section name");
303  uint32_t payload_offset = decoder.pc_offset();
304  if (section_length < (payload_offset - section_start)) {
305  decoder.error("invalid section length");
306  break;
307  }
308  uint32_t payload_length = section_length - (payload_offset - section_start);
309  decoder.consume_bytes(payload_length);
310  if (decoder.failed()) break;
311  result.push_back({{section_start, section_length},
312  {name_offset, name_length},
313  {payload_offset, payload_length}});
314  }
315 
316  return result;
317 }

321 bool FindNameSection(Decoder* decoder) {
322  static constexpr int kModuleHeaderSize = 8;
323  decoder->consume_bytes(kModuleHeaderSize, "module header");
324 
325  WasmSectionIterator section_iter(decoder, ITracer::NoTrace);
326 
327  while (decoder->ok() && section_iter.more() &&
328  section_iter.section_code() != kNameSectionCode) {
329  section_iter.advance(true);
330  }
331  if (!section_iter.more()) return false;
332 
333  // Reset the decoder to not read beyond the name section end.
334  decoder->Reset(section_iter.payload(), decoder->pc_offset());
335  return true;
336 }

338 enum class EmptyNames : bool { kAllow, kSkip };

340 void DecodeNameMapInternal(NameMap& target, Decoder& decoder,
341  EmptyNames empty_names = EmptyNames::kSkip) {
342  uint32_t count = decoder.consume_u32v("names count");
343  for (uint32_t i = 0; i < count; i++) {
344  uint32_t index = decoder.consume_u32v("index");
345  WireBytesRef name =
346  consume_string(&decoder, unibrow::Utf8Variant::kLossyUtf8, "name");
347  if (!decoder.ok()) break;
348  if (index > NameMap::kMaxKey) continue;
349  if (empty_names == EmptyNames::kSkip && name.is_empty()) continue;
350  if (!validate_utf8(&decoder, name)) continue;
351  target.Put(index, name);
352  }
353  target.FinishInitialization();
354 }

366 void DecodeIndirectNameMap(IndirectNameMap& target, Decoder& decoder,

387 void DecodeFunctionNames(base::Vector<const uint8_t> wire_bytes,
388  NameMap& names) {
389  Decoder decoder(wire_bytes);
390  if (!FindNameSection(&decoder)) return;
391  while (decoder.ok() && decoder.more()) {
392  uint8_t name_type = decoder.consume_u8("name type");
393  if (name_type & 0x80) break; // no varuint7
394 
395  uint32_t name_payload_len = decoder.consume_u32v("name payload length");
396  if (!decoder.checkAvailable(name_payload_len)) break;
397 
398  if (name_type != NameSectionKindCode::kFunctionCode) {
399  decoder.consume_bytes(name_payload_len, "name subsection payload");
400  continue;
401  }
402  // We need to allow empty function names for spec-conformant stack traces.
403  DecodeNameMapInternal(names, decoder, EmptyNames::kAllow);
404  // The spec allows only one occurrence of each subsection. We could be
405  // more permissive and allow repeated subsections; in that case we'd
406  // have to delay calling {target.FinishInitialization()} on the function
407  // names map until we've seen them all.
408  // For now, we stop decoding after finding the first function names
409  // subsection.
410  return;
411  }
412 }

414 // This follows the decoding logic of DecodeNameMap/DecodeIndirectNameMap, but
415 // processes the results differently, according to the needs of canonical
416 // type names:
417 // - all type indices are canonicalized.
418 // - multiple modules' name sections are merged into the output data structures
419 // {typenames} and {fieldnames}; existing entries are not overwritten.
420 // - all name payloads are copied out of the wire bytes.
421 void DecodeCanonicalTypeNames(
422  base::Vector<const uint8_t> wire_bytes, const WasmModule* module,
423  std::vector<base::OwnedVector<char>>& typenames,
424  std::map<uint32_t, std::vector<base::OwnedVector<char>>>& fieldnames,
425  size_t* total_allocated_size) {
426  bool types_done = false;
427  bool fields_done = false;
428  Decoder decoder(wire_bytes);
429  if (!FindNameSection(&decoder)) return;
430  const char* base = reinterpret_cast<const char*>(wire_bytes.begin());
431  while (decoder.ok() && decoder.more()) {
432  uint8_t name_type = decoder.consume_u8("name type");
433  if (name_type & 0x80) break; // no varuint7
434  uint32_t name_payload_len = decoder.consume_u32v("name payload length");
435  if (!decoder.checkAvailable(name_payload_len)) break;
436  if (name_type == NameSectionKindCode::kTypeCode) {
437  if (types_done) {
438  decoder.consume_bytes(name_payload_len, "name subsection payload");
439  continue; // The spec allows only one occurrence.
440  }
441  types_done = true;
442  uint32_t count = decoder.consume_u32v("names count");
443  for (uint32_t i = 0; i < count; i++) {
444  ModuleTypeIndex module_index{decoder.consume_u32v("index")};
445  WireBytesRef name =
446  consume_string(&decoder, unibrow::Utf8Variant::kLossyUtf8, "name");
447  if (!decoder.ok()) break;
448  if (!module->has_type(module_index)) continue;
449  if (name.is_empty()) continue;
450  CanonicalTypeIndex canonical_index =
451  module->canonical_type_id(module_index);
452  uint32_t index = canonical_index.index;
453  // Callers should have pre-sized the target vector appropriately.
454  DCHECK_LT(index, typenames.size());
455  // TODO(jkummerow): Consider implementing a more refined strategy
456  // for conflict resolution than "first definition wins", if we
457  // encounter that situation in practice. Note that we'll also have
458  // to deal with multiple field names then (in the loop below).
459  if (!typenames[index].empty()) continue; // We already have a name.
460  if (!validate_utf8(&decoder, name)) continue;
461  uint32_t length = name.length();
462  typenames[index] =
463  base::OwnedVector<char>::NewByCopying(base + name.offset(), length);
464  *total_allocated_size += length;
465  }
466  } else if (name_type == NameSectionKindCode::kFieldCode) {
467  if (fields_done) {
468  decoder.consume_bytes(name_payload_len, "name subsection payload");
469  continue; // The spec allows only one occurrence.
470  }
471  fields_done = true;
472  uint32_t types_count = decoder.consume_u32v("types count");
473  for (uint32_t i = 0; i < types_count; ++i) {
474  ModuleTypeIndex module_index{decoder.consume_u32v("type index")};
475  // TODO(jkummerow): Should we try to skip only the invalid entry?
476  if (!module->has_struct(module_index)) return;
477  CanonicalTypeIndex canonical_index =
478  module->canonical_type_id(module_index);
479  uint32_t struct_index = canonical_index.index;
480  if (struct_index >= typenames.size()) return;
481  // Due to {module->has_struct()}, this is guaranteed to succeed.
482  const CanonicalStructType* struct_type =
483  GetTypeCanonicalizer()->LookupStruct(canonical_index);
484  auto const& entry = fieldnames.try_emplace(
485  struct_index, size_t{struct_type->field_count()});
486  std::vector<base::OwnedVector<char>>& field_names = entry.first->second;
487  uint32_t fields_count = decoder.consume_u32v("fields count");
488  for (uint32_t j = 0; j < fields_count; j++) {
489  uint32_t field_index = decoder.consume_u32v("field index");
490  WireBytesRef name = consume_string(
491  &decoder, unibrow::Utf8Variant::kLossyUtf8, "field name");
492  if (!decoder.ok()) break;
493  if (field_index >= field_names.size()) continue;
494  if (!field_names[field_index].empty()) continue;
495  if (!validate_utf8(&decoder, name)) continue;
496  uint32_t length = name.length();
497  field_names[field_index] = base::OwnedVector<char>::NewByCopying(
498  base + name.offset(), length);
499  *total_allocated_size += length;
500  }
501  if (!decoder.ok()) break;
502  }
503  } else {
504  // We don't care about this subsection here.
505  decoder.consume_bytes(name_payload_len, "name subsection payload");
506  }
507  }

511 // A task that validates multiple functions in parallel, storing the earliest
512 // validation error in {this} decoder.
513 class ValidateFunctionsTask : public JobTask {
514  public:
515  explicit ValidateFunctionsTask(
516  base::Vector<const uint8_t> wire_bytes, const WasmModule* module,
517  WasmEnabledFeatures enabled_features, std::function<bool(int)> filter,
518  WasmError* error_out,
519  std::atomic<WasmDetectedFeatures>* detected_features)
520  : wire_bytes_(wire_bytes),
521  module_(module),
522  enabled_features_(enabled_features),
523  filter_(std::move(filter)),
524  next_function_(module->num_imported_functions),
525  after_last_function_(next_function_ + module->num_declared_functions),
526  error_out_(error_out),
527  detected_features_(detected_features) {
528  DCHECK(!error_out->has_error());
529  }
530 
531  void Run(JobDelegate* delegate) override {
532  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.wasm.detailed"),
533  "wasm.ValidateFunctionsTask");
534 
535  WasmDetectedFeatures detected_features;
536  Zone zone(GetWasmEngine()->allocator(), ZONE_NAME);
537  do {
538  // Get the index of the next function to validate.
539  // {fetch_add} might overrun {after_last_function_} by a bit. Since the
540  // number of functions is limited to a value much smaller than the
541  // integer range, this is near impossible to happen.
542  static_assert(kV8MaxWasmTotalFunctions < kMaxInt / 2);
543  int func_index;
544  do {
545  func_index = next_function_.fetch_add(1, std::memory_order_relaxed);
546  if (V8_UNLIKELY(func_index >= after_last_function_)) {
547  UpdateDetectedFeatures(detected_features);
548  return;
549  }
550  DCHECK_LE(0, func_index);
551  } while ((filter_ && !filter_(func_index)) ||
552  module_->function_was_validated(func_index));
553 
554  zone.Reset();
555  if (!ValidateFunction(func_index, &zone, &detected_features)) {
556  // No need to validate any more functions.
557  next_function_.store(after_last_function_, std::memory_order_relaxed);
558  return;
559  }
560  } while (!delegate->ShouldYield());
561  UpdateDetectedFeatures(detected_features);
562  }
563 
564  size_t GetMaxConcurrency(size_t /* worker_count */) const override {
565  int next_func = next_function_.load(std::memory_order_relaxed);
566  return std::max(0, after_last_function_ - next_func);
567  }
568 
569  private:
570  bool ValidateFunction(int func_index, Zone* zone,
571  WasmDetectedFeatures* detected_features) {
572  const WasmFunction& function = module_->functions[func_index];
573  DCHECK_LT(0, function.code.offset());
574  bool is_shared = module_->type(function.sig_index).is_shared;
575  FunctionBody body{function.sig, function.code.offset(),
576  wire_bytes_.begin() + function.code.offset(),
577  wire_bytes_.begin() + function.code.end_offset(),
578  is_shared};
579  DecodeResult validation_result = ValidateFunctionBody(
580  zone, enabled_features_, module_, detected_features, body);
581  if (V8_UNLIKELY(validation_result.failed())) {
582  SetError(func_index, std::move(validation_result).error());
583  return false;
584  }
585  module_->set_function_validated(func_index);
586  return true;
587  }

589  // Set the error from the argument if it's earlier than the error we already
590  // have (or if we have none yet). Thread-safe.
591  void SetError(int func_index, WasmError error) {
592  base::MutexGuard mutex_guard{&set_error_mutex_};
593  if (error_out_->has_error() && error_out_->offset() <= error.offset()) {
594  return;

680 DecodedNameSection::DecodedNameSection(base::Vector<const uint8_t> wire_bytes,
681  WireBytesRef name_section) {
682  if (name_section.is_empty()) return; // No name section.
683  Decoder decoder(wire_bytes.begin() + name_section.offset(),
684  wire_bytes.begin() + name_section.end_offset(),
685  name_section.offset());
686  while (decoder.ok() && decoder.more()) {
687  uint8_t name_type = decoder.consume_u8("name type");
688  if (name_type & 0x80) break; // no varuint7
689 
690  uint32_t name_payload_len = decoder.consume_u32v("name payload length");
691  if (!decoder.checkAvailable(name_payload_len)) break;
692 
693  switch (name_type) {
694  case kModuleCode:
695  case kFunctionCode:
696  // Already handled elsewhere.
697  decoder.consume_bytes(name_payload_len);
698  break;
699  case kLocalCode:
700  static_assert(kV8MaxWasmTotalFunctions <= IndirectNameMap::kMaxKey);
701  static_assert(kV8MaxWasmFunctionLocals <= NameMap::kMaxKey);
702  DecodeIndirectNameMap(local_names_, decoder, name_payload_len);
703  break;
704  case kLabelCode:
705  static_assert(kV8MaxWasmTotalFunctions <= IndirectNameMap::kMaxKey);
706  static_assert(kV8MaxWasmFunctionSize <= NameMap::kMaxKey);
707  DecodeIndirectNameMap(label_names_, decoder, name_payload_len);
708  break;
709  case kTypeCode:
710  static_assert(kV8MaxWasmTypes <= NameMap::kMaxKey);
711  DecodeNameMap(type_names_, decoder, name_payload_len);
712  break;
713  case kTableCode:
714  static_assert(kV8MaxWasmTables <= NameMap::kMaxKey);
715  DecodeNameMap(table_names_, decoder, name_payload_len);
716  break;
717  case kMemoryCode:
718  static_assert(kV8MaxWasmMemories <= NameMap::kMaxKey);
719  DecodeNameMap(memory_names_, decoder, name_payload_len);
720  break;
721  case kGlobalCode:
722  static_assert(kV8MaxWasmGlobals <= NameMap::kMaxKey);
723  DecodeNameMap(global_names_, decoder, name_payload_len);
724  break;
725  case kElementSegmentCode:
726  static_assert(kV8MaxWasmTableInitEntries <= NameMap::kMaxKey);
727  DecodeNameMap(element_segment_names_, decoder, name_payload_len);
728  break;
729  case kDataSegmentCode:
730  static_assert(kV8MaxWasmDataSegments <= NameMap::kMaxKey);
731  DecodeNameMap(data_segment_names_, decoder, name_payload_len);
732  break;
733  case kFieldCode:
734  static_assert(kV8MaxWasmTypes <= IndirectNameMap::kMaxKey);
735  static_assert(kV8MaxWasmStructFields <= NameMap::kMaxKey);
736  DecodeIndirectNameMap(field_names_, decoder, name_payload_len);
737  break;
738  case kTagCode:
739  static_assert(kV8MaxWasmTags <= NameMap::kMaxKey);
740  DecodeNameMap(tag_names_, decoder, name_payload_len);
741  break;
742  }
743  }
