346  template <typename C = SkipResourceCheck>
347  class AvailableLocales {
348  public:
349  AvailableLocales() {
350  UErrorCode status = U_ZERO_ERROR;
351  UEnumeration* uenum =
352  uloc_openAvailableByType(ULOC_AVAILABLE_WITH_LEGACY_ALIASES, &status);
353  DCHECK(U_SUCCESS(status));
354 
355  std::vector<std::string> all_locales;
356  const char* loc;
357  while ((loc = uenum_next(uenum, nullptr, &status)) != nullptr) {
358  DCHECK(U_SUCCESS(status));
359  std::string locstr(loc);
360  std::replace(locstr.begin(), locstr.end(), '_', '-');
361  // Handle special case
362  if (locstr == "en-US-POSIX") locstr = "en-US-u-va-posix";
363  all_locales.push_back(locstr);
364  }
365  uenum_close(uenum);
366 
367  set_ = Intl::BuildLocaleSet(all_locales, C::path(), C::key());
368  }
369  const std::set<std::string>& Get() const { return set_; }
370 
371  private:
372  std::set<std::string> set_;
373  };
