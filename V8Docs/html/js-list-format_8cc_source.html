60 MaybeDirectHandle<JSListFormat> JSListFormat::New(
61  Isolate* isolate, DirectHandle<Map> map, DirectHandle<Object> locales,
62  DirectHandle<Object> input_options) {
63  // 3. Let requestedLocales be ? CanonicalizeLocaleList(locales).
64  Maybe<std::vector<std::string>> maybe_requested_locales =
65  Intl::CanonicalizeLocaleList(isolate, locales);
66  MAYBE_RETURN(maybe_requested_locales, DirectHandle<JSListFormat>());
67  std::vector<std::string> requested_locales =
68  maybe_requested_locales.FromJust();
69 
70  DirectHandle<JSReceiver> options;
71  const char* service = "Intl.ListFormat";
72  // 4. Let options be GetOptionsObject(_options_).
73  ASSIGN_RETURN_ON_EXCEPTION(isolate, options,
74  GetOptionsObject(isolate, input_options, service));
75 
76  // Note: No need to create a record. It's not observable.
77  // 6. Let opt be a new Record.
78 
79  // 7. Let matcher be ? GetOption(options, "localeMatcher", "string", «
80  // "lookup", "best fit" », "best fit").
81  Maybe<Intl::MatcherOption> maybe_locale_matcher =
82  Intl::GetLocaleMatcher(isolate, options, service);
83  MAYBE_RETURN(maybe_locale_matcher, MaybeDirectHandle<JSListFormat>());
84 
85  // 8. Set opt.[[localeMatcher]] to matcher.
86  Intl::MatcherOption matcher = maybe_locale_matcher.FromJust();
87 
88  // 10. Let r be ResolveLocale(%ListFormat%.[[AvailableLocales]],
89  // requestedLocales, opt, undefined, localeData).
90  Intl::ResolvedLocale r;
91  if (!Intl::ResolveLocale(isolate, JSListFormat::GetAvailableLocales(),
92  requested_locales, matcher, {})
93  .To(&r)) {
94  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
95  }
96 
97  DirectHandle<String> locale_str =
98  isolate->factory()->NewStringFromAsciiChecked(r.locale.c_str());
99 
100  // 12. Let t be GetOption(options, "type", "string", «"conjunction",
101  // "disjunction", "unit"», "conjunction").
102  Maybe<Type> maybe_type = GetStringOption<Type>(
103  isolate, options, "type", service,
104  std::to_array<const std::string_view>(
105  {"conjunction", "disjunction", "unit"}),
106  std::array{Type::CONJUNCTION, Type::DISJUNCTION, Type::UNIT},
107  Type::CONJUNCTION);
108  MAYBE_RETURN(maybe_type, MaybeDirectHandle<JSListFormat>());
109  Type type_enum = maybe_type.FromJust();
110 
111  // 14. Let s be ? GetOption(options, "style", "string",
112  // «"long", "short", "narrow"», "long").
113  Maybe<Style> maybe_style = GetStringOption<Style>(
114  isolate, options, "style", service,
115  std::to_array<const std::string_view>({"long", "short", "narrow"}),
116  std::array{Style::LONG, Style::SHORT, Style::NARROW}, Style::LONG);
117  MAYBE_RETURN(maybe_style, MaybeDirectHandle<JSListFormat>());
118  Style style_enum = maybe_style.FromJust();
119 
120  icu::Locale icu_locale = r.icu_locale;
121  UErrorCode status = U_ZERO_ERROR;
122  std::shared_ptr<icu::ListFormatter> formatter{
123  icu::ListFormatter::createInstance(icu_locale, GetIcuType(type_enum),
124  GetIcuWidth(style_enum), status)};
125  if (U_FAILURE(status) || formatter == nullptr) {
126  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
127  }
128 
129  DirectHandle<Managed<icu::ListFormatter>> managed_formatter =
130  Managed<icu::ListFormatter>::From(isolate, 0, std::move(formatter));
131 
132  // Now all properties are ready, so we can allocate the result object.
133  DirectHandle<JSListFormat> list_format =
134  Cast<JSListFormat>(isolate->factory()->NewFastOrSlowJSObjectFromMap(map));
135  DisallowGarbageCollection no_gc;
136  list_format->set_flags(0);
137  list_format->set_icu_formatter(*managed_formatter);
138 
139  // 11. Set listFormat.[[Locale]] to r.[[Locale]].
140  list_format->set_locale(*locale_str);
141 
142  // 13. Set listFormat.[[Type]] to t.
143  list_format->set_type(type_enum);
144 
145  // 15. Set listFormat.[[Style]] to s.
146  list_format->set_style(style_enum);
147 
148  return list_format;
149 }

202 // Extract String from FixedArray into array of UnicodeString
203 Maybe<std::vector<icu::UnicodeString>> ToUnicodeStringArray(
204  Isolate* isolate, DirectHandle<FixedArray> array) {
205  int length = array->length();
206  std::vector<icu::UnicodeString> result;
207  for (int i = 0; i < length; i++) {
208  Handle<Object> item(array->get(i), isolate);
209  DCHECK(IsString(*item));
210  Handle<String> item_str = Cast<String>(item);
211  if (!item_str->IsFlat()) item_str = String::Flatten(isolate, item_str);
212  result.push_back(Intl::ToICUUnicodeString(isolate, item_str));
213  }
214  return Just(result);
215 }

217 template <typename T>
218 MaybeDirectHandle<T> FormatListCommon(
219  Isolate* isolate, DirectHandle<JSListFormat> format,
220  DirectHandle<FixedArray> list,
221  const std::function<MaybeDirectHandle<T>(
222  Isolate*, const icu::FormattedValue&)>& formatToResult) {
223  DCHECK(!IsUndefined(*list));
224  Maybe<std::vector<icu::UnicodeString>> maybe_array =
225  ToUnicodeStringArray(isolate, list);
226  MAYBE_RETURN(maybe_array, DirectHandle<T>());
227  std::vector<icu::UnicodeString> array = maybe_array.FromJust();
228 
229  icu::ListFormatter* formatter = format->icu_formatter()->raw();
230  DCHECK_NOT_NULL(formatter);
231 
232  UErrorCode status = U_ZERO_ERROR;
233  icu::FormattedList formatted = formatter->formatStringsToValue(
234  array.data(), static_cast<int32_t>(array.size()), status);
235  if (U_FAILURE(status)) {
236  THROW_NEW_ERROR(isolate, NewTypeError(MessageTemplate::kIcuError));
237  }
238  return formatToResult(isolate, formatted);
239 }

252 // A helper function to convert the FormattedList to a
253 // MaybeHandle<JSArray> for the implementation of formatToParts.
254 MaybeDirectHandle<JSArray> FormattedListToJSArray(
255  Isolate* isolate, const icu::FormattedValue& formatted) {
256  DirectHandle<JSArray> array = isolate->factory()->NewJSArray(0);
257  icu::ConstrainedFieldPosition cfpos;
258  cfpos.constrainCategory(UFIELD_CATEGORY_LIST);
259  int index = 0;
260  UErrorCode status = U_ZERO_ERROR;
261  icu::UnicodeString string = formatted.toString(status);
262  DirectHandle<String> substring;
263  while (formatted.nextPosition(cfpos, status) && U_SUCCESS(status)) {
264  ASSIGN_RETURN_ON_EXCEPTION(
265  isolate, substring,
266  Intl::ToString(isolate, string, cfpos.getStart(), cfpos.getLimit()));
267  Intl::AddElement(isolate, array, index++,
268  IcuFieldIdToType(isolate, cfpos.getField()), substring);
269  }
270  if (U_FAILURE(status)) {
271  THROW_NEW_ERROR(isolate, NewTypeError(MessageTemplate::kIcuError));
272  }
273  JSObject::ValidateElements(isolate, *array);
274  return array;
275 }
