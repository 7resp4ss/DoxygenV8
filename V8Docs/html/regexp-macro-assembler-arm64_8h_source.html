267  RegisterState GetRegisterState(int register_index) {
268  DCHECK_LE(0, register_index);
269  if (register_index >= kNumCachedRegisters) {
270  return STACKED;
271  } else {
272  if ((register_index % 2) == 0) {
273  return CACHED_LSW;
274  } else {
275  return CACHED_MSW;
276  }
277  }
278  }
