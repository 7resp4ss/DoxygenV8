369 std::string GetGMTTzID(const std::string& input) {
370  std::string ret = "Etc/GMT";
371  switch (input.length()) {
372  case 8:
373  if (input[7] == '0') return ret + '0';
374  break;
375  case 9:
376  if ((input[7] == '+' || input[7] == '-') &&
377  base::IsInRange(input[8], '0', '9')) {
378  return ret + input[7] + input[8];
379  }
380  break;
381  case 10:
382  if ((input[7] == '+' || input[7] == '-') && (input[8] == '1') &&
383  base::IsInRange(input[9], '0', '4')) {
384  return ret + input[7] + input[8] + input[9];
385  }
386  break;
387  }
388  return "";
389 }

398 // for i under Turkish locale while std::toupper may.

408 // Returns titlecased location, bueNos_airES -> Buenos_Aires
409 // or ho_cHi_minH -> Ho_Chi_Minh. It is locale-agnostic and only
410 // deals with ASCII only characters.
411 // 'of', 'au' and 'es' are special-cased and lowercased.
412 // ICU's timezone parsing is case sensitive, but ECMAScript is case insensitive
413 std::string ToTitleCaseTimezoneLocation(const std::string& input) {
414  std::string title_cased;
415  int word_length = 0;
416  for (char ch : input) {
417  // Convert first char to upper case, the rest to lower case
418  if (IsAsciiAlpha(ch)) {
419  title_cased += word_length == 0 ? LocaleIndependentAsciiToUpper(ch)
420  : LocaleIndependentAsciiToLower(ch);
421  word_length++;
422  } else if (ch == '_' || ch == '-' || ch == '/') {
423  // Special case Au/Es/Of to be lower case.
424  if (word_length == 2) {
425  size_t pos = title_cased.length() - 2;
426  std::string substr = title_cased.substr(pos, 2);
427  if (substr == "Of" || substr == "Es" || substr == "Au") {
428  title_cased[pos] = LocaleIndependentAsciiToLower(title_cased[pos]);
429  }
430  }
431  title_cased += ch;
432  word_length = 0;
433  } else {
434  // Invalid input
435  return std::string();
436  }
437  }
438 
439  return title_cased;
440 }

483 // Return the time zone id which match ICU's expectation of title casing
484 // return empty string when error.
485 std::string JSDateTimeFormat::CanonicalizeTimeZoneID(const std::string& input) {
486  std::string upper = input;
487  transform(upper.begin(), upper.end(), upper.begin(),
488  LocaleIndependentAsciiToUpper);
489  if (upper.length() == 3) {
490  if (upper == "GMT") return "UTC";
491  // For id such as "CET", return upper case.
492  return upper;
493  } else if (upper.length() == 7 && '0' <= upper[3] && upper[3] <= '9') {
494  // For id such as "CST6CDT", return upper case.
495  return upper;
496  } else if (upper.length() > 3) {
497  if (memcmp(upper.c_str(), "ETC", 3) == 0) {
498  if (upper == "ETC/UTC" || upper == "ETC/GMT" || upper == "ETC/UCT") {
499  return "UTC";
500  }
501  if (strncmp(upper.c_str(), "ETC/GMT", 7) == 0) {
502  return GetGMTTzID(input);
503  }
504  } else if (memcmp(upper.c_str(), "GMT", 3) == 0) {
505  if (upper == "GMT0" || upper == "GMT+0" || upper == "GMT-0") {
506  return "UTC";
507  }
508  } else if (memcmp(upper.c_str(), "US/", 3) == 0) {
509  std::string title = ToTitleCaseTimezoneLocation(input);
510  if (title.length() >= 2) {
511  // Change "Us/" to "US/"
512  title[1] = 'S';
513  }
514  return title;
515  } else if (strncmp(upper.c_str(), "SYSTEMV/", 8) == 0) {
516  upper.replace(0, 8, "SystemV/");
517  return upper;
518  }
519  }
520  // We expect only _, '-' and / beside ASCII letters.
521 
522  static base::LazyInstance<SpecialTimeZoneMap>::type special_time_zone_map =
523  LAZY_INSTANCE_INITIALIZER;
524 
525  std::string special_case = special_time_zone_map.Pointer()->Find(upper);
526  if (!special_case.empty()) {
527  return special_case;
528  }
529  return ToTitleCaseTimezoneLocation(input);
530 }

561 MaybeHandle<String> JSDateTimeFormat::TimeZoneIdToString(
562  Isolate* isolate, const icu::UnicodeString& id) {
563  // In CLDR (http://unicode.org/cldr/trac/ticket/9943), Etc/UTC is made
564  // a separate timezone ID from Etc/GMT even though they're still the same
565  // timezone. We have Etc/UTC because 'UTC', 'Etc/Universal',
566  // 'Etc/Zulu' and others are turned to 'Etc/UTC' by ICU. Etc/GMT comes
567  // from Etc/GMT0, Etc/GMT+0, Etc/GMT-0, Etc/Greenwich.
568  // ecma402#sec-canonicalizetimezonename step 3
569  if (id == UNICODE_STRING_SIMPLE("Etc/UTC") ||
570  id == UNICODE_STRING_SIMPLE("Etc/GMT")) {
571  return isolate->factory()->UTC_string();
572  }
573  // If the id is in the format of GMT[+-]hh:mm, change it to
574  // [+-]hh:mm.
575  if (id.startsWith(u"GMT", 3)) {
576  return Intl::ToString(isolate, id.tempSubString(3));
577  }
578  return Intl::ToString(isolate, id);
579 }

1092 // This helper function handles Supported fields and Default fields in Table 16
1093 // ( #table-temporal-patterns ). It remove all the fields not stated in keep
1094 // from input, and add the fields in add_default if a skeleton in the same
1095 // category is in the input, with considering the equivalent.
1096 // For example, if input is "yyyyMMhm", keep is {y,M,d} and add_default is
1097 // {y,M,d}, the output will be "yyyyMMd". For example, if input is
1098 // "yyyyMMhmOOOO", keep is {h,m,s,z,O,v} and add_default is {h,m,s}, then the
1099 // output will be "hmOOOOs". The meaning of the skeleton letters is stated in
1100 // UTS35
1101 // https://www.unicode.org/reports/tr35/tr35-dates.html#table-date-field-symbol-table
1102 icu::UnicodeString KeepSupportedAddDefault(
1103  const icu::UnicodeString& input, const std::set<char16_t>& keep,
1104  const std::set<char16_t>& add_default) {
1105  const std::map<char16_t, char16_t> equivalent({{'L', 'M'},
1106  {'h', 'j'},
1107  {'H', 'j'},
1108  {'k', 'j'},
1109  {'K', 'j'},
1110  {'O', 'z'},
1111  {'v', 'z'}});
1112  std::set<char16_t> to_be_added(add_default);
1113  icu::UnicodeString result;
1114  for (int32_t i = 0; i < input.length(); i++) {
1115  char16_t ch = input.charAt(i);
1116  if (keep.find(ch) != keep.end()) {
1117  to_be_added.erase(ch);
1118  auto also = equivalent.find(ch);
1119  if (also != equivalent.end()) {
1120  to_be_added.erase(also->second);
1121  }
1122  result.append(ch);

1603  JSDateTimeFormat::HourCycle hc) {
1604  char16_t replacement;
1605  switch (hc) {

1627  case '\'':
1628  replace = !replace;
1629  result.append(ch);
1630  break;

1636  [[fallthrough]];

1640  result.append(' ');
1641  }

1643  break;
1644  default:
1645  result.append(ch);
1646  break;
1647  }

1773  const icu::UnicodeString pattern) {

1783  break;

1785  if (!in_quote) return JSDateTimeFormat::HourCycle::kH12;
1786  break;
1787  case 'H':
1788  if (!in_quote) return JSDateTimeFormat::HourCycle::kH23;
1789  break;
1790  case 'k':
1791  if (!in_quote) return JSDateTimeFormat::HourCycle::kH24;
1792  break;

1822  case JSDateTimeFormat::HourCycle::kH12:

1824  break;
1825  case JSDateTimeFormat::HourCycle::kH23:
1826  to = 'H';
1827  break;
1828  case JSDateTimeFormat::HourCycle::kH24:

1841  [[fallthrough]];

1845  case 'h':

1847  case 'H':
1848  [[fallthrough]];
1849  case 'K':

1851  case 'k':

1935  generator, hc);
