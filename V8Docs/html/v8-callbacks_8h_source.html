37 /**
38  * A JIT code event is issued each time code is added, moved or removed.
39  *
40  * \note removal events are not currently issued.
41  */
42 struct JitCodeEvent {
43  enum EventType {
44  CODE_ADDED,
45  CODE_MOVED,
46  CODE_REMOVED,
47  CODE_ADD_LINE_POS_INFO,
48  CODE_START_LINE_INFO_RECORDING,
49  CODE_END_LINE_INFO_RECORDING
50  };
51  // Definition of the code position type. The "POSITION" type means the place
52  // in the source code which are of interest when making stack traces to
53  // pin-point the source location of a stack frame as close as possible.
54  // The "STATEMENT_POSITION" means the place at the beginning of each
55  // statement, and is used to indicate possible break locations.
56  enum PositionType { POSITION, STATEMENT_POSITION };
57 
58  // There are three different kinds of CodeType, one for JIT code generated
59  // by the optimizing compiler, one for byte code generated for the
60  // interpreter, and one for code generated from Wasm. For JIT_CODE and
61  // WASM_CODE, |code_start| points to the beginning of jitted assembly code,
62  // while for BYTE_CODE events, |code_start| points to the first bytecode of
63  // the interpreted function.
64  enum CodeType { BYTE_CODE, JIT_CODE, WASM_CODE };
65 
66  // Type of event.
67  EventType type;
68  CodeType code_type;
69  // Start of the instructions.
70  void* code_start;
71  // Size of the instructions.
72  size_t code_len;
73  // Script info for CODE_ADDED event.
74  Local<UnboundScript> script;
75  // User-defined data for *_LINE_INFO_* event. It's used to hold the source
76  // code line information which is returned from the
77  // CODE_START_LINE_INFO_RECORDING event. And it's passed to subsequent
78  // CODE_ADD_LINE_POS_INFO and CODE_END_LINE_INFO_RECORDING events.
79  void* user_data;

146 /**
147  * Applications can register callback functions which will be called before and
148  * after certain garbage collection operations. Allocations are not allowed in
149  * the callback functions, you therefore cannot manipulate objects (set or
150  * delete properties for example) since it is possible such operations will

156  kGCTypeMinorMarkSweep = 1 << 1,
157  kGCTypeMarkSweepCompact = 1 << 2,
158  kGCTypeIncrementalMarking = 1 << 3,
159  kGCTypeProcessWeakCallbacks = 1 << 4,
160  kGCTypeAll = kGCTypeScavenge | kGCTypeMinorMarkSweep |
161  kGCTypeMarkSweepCompact | kGCTypeIncrementalMarking |

165 /**
166  * GCCallbackFlags is used to notify additional information about the GC
167  * callback.
168  * - kGCCallbackFlagConstructRetainedObjectInfos: The GC callback is for
169  * constructing retained object infos.
170  * - kGCCallbackFlagForced: The GC callback is for a forced GC for testing.
171  * - kGCCallbackFlagSynchronousPhantomCallbackProcessing: The GC callback
172  * is called synchronously without getting posted to an idle task.
173  * - kGCCallbackFlagCollectAllAvailableGarbage: The GC callback is called
174  * in a phase where V8 is trying to collect all available garbage
175  * (e.g., handling a low memory notification).
176  * - kGCCallbackScheduleIdleGarbageCollection: The GC callback is called to
177  * trigger an idle garbage collection.
178  */
179 enum GCCallbackFlags {
180  kNoGCCallbackFlags = 0,
181  kGCCallbackFlagConstructRetainedObjectInfos = 1 << 1,
182  kGCCallbackFlagForced = 1 << 2,
183  kGCCallbackFlagSynchronousPhantomCallbackProcessing = 1 << 3,
184  kGCCallbackFlagCollectAllAvailableGarbage = 1 << 4,
185  kGCCallbackFlagCollectAllExternalMemory = 1 << 5,
186  kGCCallbackScheduleIdleGarbageCollection = 1 << 6,
187 };

342 /**
343  * HostImportModuleDynamicallyCallback is called when we
344  * require the embedder to load a module. This is used as part of the dynamic
345  * import syntax.
346  *
347  * The referrer contains metadata about the script/module that calls
348  * import.
349  *
350  * The specifier is the name of the module that should be imported.
351  *
352  * The import_attributes are import attributes for this request in the form:
353  * [key1, value1, key2, value2, ...] where the keys and values are of type
354  * v8::String. Note, unlike the FixedArray passed to ResolveModuleCallback and
355  * returned from ModuleRequest::GetImportAttributes(), this array does not
356  * contain the source Locations of the attributes.
357  *
358  * The embedder must compile, instantiate, evaluate the Module, and
359  * obtain its namespace object.
360  *
361  * The Promise returned from this function is forwarded to userland
362  * JavaScript. The embedder must resolve this promise with the module
363  * namespace object. In case of an exception, the embedder must reject
364  * this promise with the exception. If the promise creation itself
365  * fails (e.g. due to stack overflow), the embedder must propagate
366  * that exception by returning an empty MaybeLocal.
367  */
368 using HostImportModuleDynamicallyCallback = MaybeLocal<Promise> (*)(
369  Local<Context> context, Local<Data> host_defined_options,
370  Local<Value> resource_name, Local<String> specifier,
371  Local<FixedArray> import_attributes);
372 
373 /**
374  * HostImportModuleWithPhaseDynamicallyCallback is called when we
375  * require the embedder to load a module with a specific phase. This is used
376  * as part of the dynamic import syntax.
377  *
378  * The referrer contains metadata about the script/module that calls
379  * import.
380  *
381  * The specifier is the name of the module that should be imported.
382  *
383  * The phase is the phase of the import requested.
384  *
385  * The import_attributes are import attributes for this request in the form:
386  * [key1, value1, key2, value2, ...] where the keys and values are of type
387  * v8::String. Note, unlike the FixedArray passed to ResolveModuleCallback and
388  * returned from ModuleRequest::GetImportAttributes(), this array does not
389  * contain the source Locations of the attributes.
390  *
391  * The Promise returned from this function is forwarded to userland
392  * JavaScript. The embedder must resolve this promise according to the phase
393  * requested:
394  * - For ModuleImportPhase::kSource, the promise must be resolved with a
395  * compiled ModuleSource object, or rejected with a SyntaxError if the
396  * module does not support source representation.
397  * - For ModuleImportPhase::kEvaluation, the promise must be resolved with a
398  * ModuleNamespace object of a module that has been compiled, instantiated,
399  * and evaluated.
400  *
401  * In case of an exception, the embedder must reject this promise with the
402  * exception. If the promise creation itself fails (e.g. due to stack
403  * overflow), the embedder must propagate that exception by returning an empty
404  * MaybeLocal.
405  *
406  * This callback is still experimental and is only invoked for source phase
407  * imports.
408  */
409 using HostImportModuleWithPhaseDynamicallyCallback = MaybeLocal<Promise> (*)(
410  Local<Context> context, Local<Data> host_defined_options,
411  Local<Value> resource_name, Local<String> specifier,
412  ModuleImportPhase phase, Local<FixedArray> import_attributes);
