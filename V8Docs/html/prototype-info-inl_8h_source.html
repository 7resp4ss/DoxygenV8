80 Tagged<MaybeObject> PrototypeInfo::GetDerivedMap(DirectHandle<Map> from) {
81  if (IsUndefined(derived_maps())) {
82  return Tagged<MaybeObject>();
83  }
84  auto derived = Cast<WeakArrayList>(derived_maps());
85  // Index 0 is the map for object create
86  for (int i = 1; i < derived->length(); ++i) {
87  Tagged<MaybeObject> el = derived->Get(i);
88  Tagged<Map> map_obj;
89  if (el.GetHeapObjectIfWeak(&map_obj)) {
90  Tagged<Map> to = Cast<Map>(map_obj);
91  if (to->GetConstructor() == from->GetConstructor() &&
92  to->instance_type() == from->instance_type()) {
93  return el;
94  }
95  }
96  }
97  return Tagged<MaybeObject>();
98 }

101 void PrototypeInfo::AddDerivedMap(DirectHandle<PrototypeInfo> info,
102  DirectHandle<Map> to, Isolate* isolate) {
103  if (IsUndefined(info->derived_maps())) {
104  // Index 0 is the map for object create
105  Tagged<WeakArrayList> derived = *isolate->factory()->NewWeakArrayList(2);
106  // GetConstructMap assumes a weak pointer.
107  derived->Set(0, ClearedValue(isolate));
108  derived->Set(1, MakeWeak(*to));
109  derived->set_length(2);
110  info->set_derived_maps(derived, kReleaseStore);
111  return;
112  }
113  auto derived = handle(Cast<WeakArrayList>(info->derived_maps()), isolate);
114  // Index 0 is the map for object create
115  int i = 1;
116  for (; i < derived->length(); ++i) {
117  Tagged<MaybeObject> el = derived->Get(i);
118  if (el.IsCleared()) {
119  derived->Set(i, MakeWeak(*to));
120  return;
121  }
122  }
123 
124  auto bigger = WeakArrayList::EnsureSpace(isolate, derived, i + 1);
125  bigger->Set(i, MakeWeak(*to));
126  bigger->set_length(i + 1);
127  if (*bigger != *derived) {
128  info->set_derived_maps(*bigger, kReleaseStore);
129  }
130 }
