28 WasmCompilationResult WasmCompilationUnit::ExecuteCompilation(
29  CompilationEnv* env, const WireBytesStorage* wire_bytes_storage,
30  Counters* counters, WasmDetectedFeatures* detected) {
31  DCHECK_GE(func_index_, static_cast<int>(env->module->num_imported_functions));
32  const WasmFunction* func = &env->module->functions[func_index_];
33  base::Vector<const uint8_t> code = wire_bytes_storage->GetCode(func->code);
34  bool is_shared = env->module->type(func->sig_index).is_shared;
35  wasm::FunctionBody func_body{func->sig, func->code.offset(), code.begin(),
36  code.end(), is_shared};
37 
38  std::optional<TimedHistogramScope> wasm_compile_function_time_scope;
39  std::optional<TimedHistogramScope> wasm_compile_huge_function_time_scope;
40  if (counters && base::TimeTicks::IsHighResolution()) {
41  if (func_body.end - func_body.start >= 100 * KB) {
42  auto huge_size_histogram = SELECT_WASM_COUNTER(
43  counters, env->module->origin, wasm, huge_function_size_bytes);
44  huge_size_histogram->AddSample(
45  static_cast<int>(func_body.end - func_body.start));
46  wasm_compile_huge_function_time_scope.emplace(
47  counters->wasm_compile_huge_function_time());
48  }
49  auto timed_histogram = SELECT_WASM_COUNTER(counters, env->module->origin,
50  wasm_compile, function_time);
51  wasm_compile_function_time_scope.emplace(timed_histogram);
52  }
53 
54  // Before executing compilation, make sure that the function was validated.
55  // Both Liftoff and TurboFan compilation do not perform validation, so can
56  // only run on valid functions.
57  if (V8_UNLIKELY(!env->module->function_was_validated(func_index_))) {
58  // This code path can only be reached in
59  // - eager compilation mode,
60  // - with lazy validation,
61  // - with PGO (which compiles some functions eagerly), or
62  // - with compilation hints (which also compiles some functions eagerly).
63  DCHECK(!v8_flags.wasm_lazy_compilation || v8_flags.wasm_lazy_validation ||
64  v8_flags.experimental_wasm_pgo_from_file ||
65  v8_flags.experimental_wasm_compilation_hints);
66  Zone validation_zone{GetWasmEngine()->allocator(), ZONE_NAME};
67  if (ValidateFunctionBody(&validation_zone, env->enabled_features,
68  env->module, detected, func_body)
69  .failed()) {
70  return {};
71  }
72  env->module->set_function_validated(func_index_);
73  }
74 
75  if (v8_flags.trace_wasm_compiler) {
76  PrintF("Compiling wasm function %d with %s\n", func_index_,
77  ExecutionTierToString(tier_));
78  }
79 
80  WasmCompilationResult result;
81  int declared_index = declared_function_index(env->module, func_index_);
82 
83  switch (tier_) {
84  case ExecutionTier::kNone:
85 #if V8_ENABLE_DRUMBRAKE
86  case ExecutionTier::kInterpreter:
87 #endif // V8_ENABLE_DRUMBRAKE
88  UNREACHABLE();
89 
90  case ExecutionTier::kLiftoff: {
91  // The --wasm-tier-mask-for-testing flag can force functions to be
92  // compiled with TurboFan, and the --wasm-debug-mask-for-testing can force
93  // them to be compiled for debugging, see documentation.
94  bool try_liftoff = true;
95  if (V8_UNLIKELY(v8_flags.wasm_tier_mask_for_testing != 0)) {
96  bool must_use_liftoff =
97  v8_flags.liftoff_only ||
98  for_debugging_ != ForDebugging::kNotForDebugging;
99  bool tiering_requested =
100  declared_index < 32 &&
101  (v8_flags.wasm_tier_mask_for_testing & (1 << declared_index));
102  if (!must_use_liftoff && tiering_requested) try_liftoff = false;
103  }
104 
105  if (V8_LIKELY(try_liftoff)) {
106  auto options = LiftoffOptions{}
107  .set_func_index(func_index_)
108  .set_for_debugging(for_debugging_)
109  .set_counters(counters)
110  .set_detected_features(detected);
111  // We do not use the debug side table, we only (optionally) pass it to
112  // cover different code paths in Liftoff for testing.
113  std::unique_ptr<DebugSideTable> unused_debug_sidetable;
114  if (V8_UNLIKELY(declared_index < 32 &&
115  (v8_flags.wasm_debug_mask_for_testing &
116  (1 << declared_index)) != 0)) {
117  options.set_debug_sidetable(&unused_debug_sidetable);
118  if (!for_debugging_) options.set_for_debugging(kForDebugging);
119  }
120  if (v8_flags.wasm_code_coverage &&
121  options.for_debugging == kNotForDebugging) {
122  options.set_for_debugging(kForDebugging);
123  }
124  result = ExecuteLiftoffCompilation(env, func_body, options);
125  if (result.succeeded()) break;
126  }
127 
128  // If --liftoff-only, do not fall back to turbofan, even if compilation
129  // failed.
130  if (v8_flags.liftoff_only) break;
131 
132  // If Liftoff failed, fall back to TurboFan.
133  // TODO(wasm): We could actually stop or remove the tiering unit for this
134  // function to avoid compiling it twice with TurboFan.
135  [[fallthrough]];
136  }
137  case ExecutionTier::kTurbofan: {
138  compiler::WasmCompilationData data(func_body);
139  data.func_index = func_index_;
140  data.wire_bytes_storage = wire_bytes_storage;
141  result = compiler::turboshaft::ExecuteTurboshaftWasmCompilation(
142  env, data, detected, counters);
143  // In exceptional cases it can happen that compilation requests for
144  // debugging end up being executed by Turbofan, e.g. if Liftoff bails out
145  // because of unsupported features or the --wasm-tier-mask-for-testing is
146  // set. In that case we set the for_debugging field for the TurboFan
147  // result to match the requested for_debugging_.
148  result.for_debugging = for_debugging_;
149  break;
150  }
151  }
152 
153  DCHECK(result.succeeded());
154  if (counters) {
155  counters->wasm_generated_code_size()->Increment(
156  result.code_desc.instr_size);
157  counters->wasm_reloc_size()->Increment(result.code_desc.reloc_size);
158  counters->wasm_deopt_data_size()->Increment(
159  static_cast<int>(result.deopt_data.size()));
160  }
161 
162  result.func_index = func_index_;
163 
164  return result;
165 }

168 void WasmCompilationUnit::CompileWasmFunction(Counters* counters,
169  NativeModule* native_module,
170  WasmDetectedFeatures* detected,
171  const WasmFunction* function,
172  ExecutionTier tier) {
173  ModuleWireBytes wire_bytes(native_module->wire_bytes());
174  bool is_shared = native_module->module()->type(function->sig_index).is_shared;
175  FunctionBody function_body{function->sig, function->code.offset(),
176  wire_bytes.start() + function->code.offset(),
177  wire_bytes.start() + function->code.end_offset(),
178  is_shared};
179 
180  DCHECK_LE(native_module->num_imported_functions(), function->func_index);
181  DCHECK_LT(function->func_index, native_module->num_functions());
182  WasmCompilationUnit unit(function->func_index, tier, kNotForDebugging);
183  CompilationEnv env = CompilationEnv::ForModule(native_module);
184  base::FlushDenormalsScope disable_denormals(
185  tier == ExecutionTier::kTurbofan &&
186  native_module->compile_imports().contains(
187  CompileTimeImport::kDisableDenormalFloats));
188  WasmCompilationResult result = unit.ExecuteCompilation(
189  &env, native_module->compilation_state()->GetWireBytesStorage().get(),
190  counters, detected);
191  if (result.succeeded()) {
192  WasmCodeRefScope code_ref_scope;
193  native_module->PublishCode(native_module->AddCompiledCode(result));
194  } else {
195  native_module->compilation_state()->SetError();
196  }
197 }

232 DirectHandle<Code> JSToWasmWrapperCompilationUnit::Finalize() {
233 #if V8_ENABLE_DRUMBRAKE
234  if (v8_flags.wasm_jitless) {
235  return isolate_->builtins()->code_handle(
236  Builtin::kGenericJSToWasmInterpreterWrapper);
237  }
238 #endif // V8_ENABLE_DRUMBRAKE
239 
240  CompilationJob::Status status = job_->FinalizeJob(isolate_);
241  CHECK_EQ(status, CompilationJob::SUCCEEDED);
242  OptimizedCompilationInfo* info =
243  static_cast<compiler::turboshaft::TurboshaftCompilationJob*>(job_.get())
244  ->compilation_info();
245  DirectHandle<Code> code = info->code();
246  if (isolate_->IsLoggingCodeCreation()) {
247  DirectHandle<String> name = isolate_->factory()->NewStringFromAsciiChecked(
248  info->GetDebugName().get());
249  PROFILE(isolate_, CodeCreateEvent(LogEventListener::CodeTag::kStub,
250  Cast<AbstractCode>(code), name));
251  }
252  // Install the compiled wrapper in the cache now.
253  WasmExportWrapperCache::Put(isolate_, sig_index_, receiver_is_first_param_,
254  code);
255  Counters* counters = isolate_->counters();
256  counters->wasm_generated_code_size()->Increment(code->body_size());
257  counters->wasm_reloc_size()->Increment(code->relocation_size());
258  counters->wasm_compiled_export_wrapper()->Increment(1);
259  return code;
260 }
