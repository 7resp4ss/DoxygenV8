36 template <class Next>
37 class WasmInJSInliningReducer : public Next {
38  public:
39  TURBOSHAFT_REDUCER_BOILERPLATE(WasmInJSInlining)
40 
41  V<Any> REDUCE(Call)(V<CallTarget> callee,
42  OptionalV<turboshaft::FrameState> frame_state,
43  base::Vector<const OpIndex> arguments,
44  const TSCallDescriptor* descriptor, OpEffects effects) {
45  if (!descriptor->js_wasm_call_parameters) {
46  // Regular call, nothing to do with Wasm or inlining. Proceed untouched...
47  return Next::ReduceCall(callee, frame_state, arguments, descriptor,
48  effects);
49  }
50 
51  // We shouldn't have attached `JSWasmCallParameters` at this call, unless
52  // we have TS Wasm-in-JS inlining enabled.
53  CHECK(v8_flags.turboshaft_wasm_in_js_inlining);
54 
55  const wasm::WasmModule* module =
56  descriptor->js_wasm_call_parameters->module();
57  wasm::NativeModule* native_module =
58  descriptor->js_wasm_call_parameters->native_module();
59  uint32_t func_idx = descriptor->js_wasm_call_parameters->function_index();
60 
61  V<Any> result =
62  TryInlineWasmCall(module, native_module, func_idx, arguments);
63  if (result.valid()) {
64  return result;
65  } else {
66  // The JS-to-Wasm wrapper was already inlined by the earlier TurboFan
67  // phase, specifically `JSInliner::ReduceJSWasmCall`. However, it did
68  // not toggle the thread-in-Wasm flag, since we don't want to set it
69  // in the inline case above.
70  // For the non-inline case, we need to toggle the flag now.
71  // TODO(dlehmann,353475584): Reuse the code from
72  // `WasmGraphBuilderBase::BuildModifyThreadInWasmFlag`, but that
73  // requires a different assembler stack...
74  OpIndex isolate_root = __ LoadRootRegister();
75  V<WordPtr> thread_in_wasm_flag_address =
76  __ Load(isolate_root, LoadOp::Kind::RawAligned().Immutable(),
77  MemoryRepresentation::UintPtr(),
78  Isolate::thread_in_wasm_flag_address_offset());
79  __ Store(thread_in_wasm_flag_address, __ Word32Constant(1),
80  LoadOp::Kind::RawAligned(), MemoryRepresentation::Int32(),
81  compiler::kNoWriteBarrier);
82 
83  result =
84  Next::ReduceCall(callee, frame_state, arguments, descriptor, effects);
85 
86  __ Store(thread_in_wasm_flag_address, __ Word32Constant(0),
87  LoadOp::Kind::RawAligned(), MemoryRepresentation::Int32(),
88  compiler::kNoWriteBarrier);
89 
90  return result;
91  }
92  }
93 
94  private:
95  V<Any> TryInlineWasmCall(const wasm::WasmModule* module,
96  wasm::NativeModule* native_module, uint32_t func_idx,
97  base::Vector<const OpIndex> arguments);
98 };

156  void StartFunction(FullDecoder* decoder) {
157  locals_.resize(decoder->num_locals());
158 
159  // Initialize the function parameters, which are part of the local space.
160  CHECK_LE(arguments_.size(), locals_.size());
161  std::copy(arguments_.begin(), arguments_.end(), locals_.begin());
162 
163  // Initialize the non-parameter locals.
164  uint32_t index = static_cast<uint32_t>(decoder->sig_->parameter_count());
165  CHECK_EQ(index, arguments_.size());
166  while (index < decoder->num_locals()) {
167  ValueType type = decoder->local_type(index);
168  OpIndex op;
169  if (!type.is_defaultable()) {
170  DCHECK(type.is_reference());
171  op = __ RootConstant(RootIndex::kOptimizedOut);
172  } else {
173  op = DefaultValue(type);
174  }
175  while (index < decoder->num_locals() &&
176  decoder->local_type(index) == type) {
177  locals_[index++] = op;
178  }
179  }

1259  const wasm::WasmModule* module, wasm::NativeModule* native_module,
1260  uint32_t func_idx, base::Vector<const OpIndex> arguments) {

1262 
1263  TRACE("Considering wasm function ["
1264  << func_idx << "] "
1265  << JSInliner::WasmFunctionNameForTrace(native_module, func_idx)
1266  << " of module " << module << " for inlining");
1267 
1268  if (wasm::is_asmjs_module(module)) {
1269  TRACE("- not inlining: asm.js-in-JS inlining is not supported");
1270  return OpIndex::Invalid();
1271  }
1272 
1273  if (func_idx < module->num_imported_functions) {
1274  TRACE("- not inlining: call to an imported function");
1275  return OpIndex::Invalid();
1276  }
1277  DCHECK_LT(func_idx - module->num_imported_functions,
1278  module->num_declared_functions);
1279 
1280  // TODO(42204563): Support shared-everything proposal (at some point, or
1281  // possibly never).
1282  bool is_shared = module->type(func.sig_index).is_shared;
1283  if (is_shared) {
1284  TRACE("- not inlining: shared everything is not supported");
1285  return OpIndex::Invalid();
1286  }
1287 
1288  base::Vector<const uint8_t> module_bytes = native_module->wire_bytes();
1289  const uint8_t* start = module_bytes.begin() + func.code.offset();
1290  const uint8_t* end = module_bytes.begin() + func.code.end_offset();
1291 
1292  wasm::FunctionBody func_body{func.sig, func.code.offset(), start, end,
1293  is_shared};
1294 
1295  auto env = wasm::CompilationEnv::ForModule(native_module);
1296  wasm::WasmDetectedFeatures detected{};
1297 
1298  // JS-to-Wasm wrapper inlining doesn't support multi-value at the moment,
1299  // so we should never reach here with more than 1 return value.
1300  DCHECK_LE(func.sig->return_count(), 1);
1301  base::Vector<const OpIndex> arguments_without_instance =
1302  arguments.SubVectorFrom(1);
1303  V<WasmTrustedInstanceData> trusted_instance_data =
1304  arguments[wasm::kWasmInstanceDataParameterIndex];
1305 
1306  Block* inlinee_body_and_rest = __ NewBlock();
1307  __ Goto(inlinee_body_and_rest);
1308 
1309  // First pass: Decode Wasm body to see if we could inline or would bail out.
1310  // Emit into an unreachable block. We are not interested in the operations at
1311  // this point, only in the decoder status afterwards.
1312  Block* unreachable = __ NewBlock();
1313  __ Bind(unreachable);
1314 
1315  using Interface = WasmInJsInliningInterface<Assembler<ReducerList>>;
1316  using Decoder =
1317  wasm::WasmFullDecoder<typename Interface::ValidationTag, Interface>;
1318  Decoder can_inline_decoder(Asm().phase_zone(), env.module,
1319  env.enabled_features, &detected, func_body, Asm(),
1320  arguments_without_instance, trusted_instance_data);
1321  DCHECK(env.module->function_was_validated(func_idx));
1322  can_inline_decoder.Decode();
1323 
1324  // The function was already validated, so decoding can only fail if we bailed
1325  // out due to an unsupported instruction.
1326  if (!can_inline_decoder.ok()) {
1327  TRACE("- not inlining: " << can_inline_decoder.error().message());
1328  __ Bind(inlinee_body_and_rest);
1329  return OpIndex::Invalid();
1330  }
1331 
1332  // Second pass: Actually emit the inlinee instructions now.
1333  __ Bind(inlinee_body_and_rest);
1334  Decoder emitting_decoder(Asm().phase_zone(), env.module, env.enabled_features,
1335  &detected, func_body, Asm(),
1336  arguments_without_instance, trusted_instance_data);
1337  emitting_decoder.Decode();
1338  DCHECK(emitting_decoder.ok());
1339  DCHECK(emitting_decoder.interface().Result().valid());
1340  TRACE("- inlining");
1341  return emitting_decoder.interface().Result();
1342 }
