23 void Node::OutOfLineInputs::ExtractFrom(Use* old_use_ptr,
24  ZoneNodePtr* old_input_ptr, int count) {
25  DCHECK_GE(count, 0);
26  // Extract the inputs from the old use and input pointers and copy them
27  // to this out-of-line-storage.
28  Use* new_use_ptr = reinterpret_cast<Use*>(this) - 1;
29  ZoneNodePtr* new_input_ptr = inputs();
30  CHECK_IMPLIES(count > 0, Use::InputIndexField::is_valid(count - 1));
31  for (int current = 0; current < count; current++) {
32  new_use_ptr->bit_field_ =
33  Use::InputIndexField::encode(current) | Use::InlineField::encode(false);
34  DCHECK_EQ(old_input_ptr, old_use_ptr->input_ptr());
35  DCHECK_EQ(new_input_ptr, new_use_ptr->input_ptr());
36  Node* old_to = *old_input_ptr;
37  if (old_to) {
38  *old_input_ptr = nullptr;
39  old_to->RemoveUse(old_use_ptr);
40  *new_input_ptr = old_to;
41  old_to->AppendUse(new_use_ptr);
42  } else {
43  *new_input_ptr = nullptr;
44  }
45  old_input_ptr++;
46  new_input_ptr++;
47  old_use_ptr--;
48  new_use_ptr--;
49  }
50  this->count_ = count;
51 }

57 template <typename NodePtrT>
58 Node* Node::NewImpl(Zone* zone, NodeId id, const Operator* op, int input_count,
59  NodePtrT const* inputs, bool has_extensible_inputs) {
60  DCHECK_GE(input_count, 0);
61 
62  ZoneNodePtr* input_ptr;
63  Use* use_ptr;
64  Node* node;
65  bool is_inline;
66 
67  // Verify that none of the inputs are {nullptr}.
68  for (int i = 0; i < input_count; i++) {
69  if (inputs[i] == nullptr) {
70  FATAL("Node::New() Error: #%d:%s[%d] is nullptr", static_cast<int>(id),
71  op->mnemonic(), i);
72  }
73  }
74 
75  if (input_count > kMaxInlineCapacity) {
76  // Allocate out-of-line inputs.
77  int capacity =
78  has_extensible_inputs ? input_count + kMaxInlineCapacity : input_count;
79  OutOfLineInputs* outline = OutOfLineInputs::New(zone, capacity);
80 
81  // Allocate node, with space for OutOfLineInputs pointer.
82  void* node_buffer = zone->Allocate<NodeWithOutOfLineInputs>(
83  sizeof(Node) + sizeof(ZoneOutOfLineInputsPtr));
84  node = new (node_buffer) Node(id, op, kOutlineMarker, 0);
85  node->set_outline_inputs(outline);
86 
87  outline->node_ = node;
88  outline->count_ = input_count;
89 
90  input_ptr = outline->inputs();
91  use_ptr = reinterpret_cast<Use*>(outline);
92  is_inline = false;
93  } else {
94  // Allocate node with inline inputs. Capacity must be at least 1 so that
95  // an OutOfLineInputs pointer can be stored when inputs are added later.
96  int capacity = std::max(1, input_count);
97  if (has_extensible_inputs) {
98  const int max = kMaxInlineCapacity;
99  capacity = std::min(input_count + 3, max);
100  }
101 
102  size_t size = sizeof(Node) + capacity * (sizeof(ZoneNodePtr) + sizeof(Use));
103  intptr_t raw_buffer =
104  reinterpret_cast<intptr_t>(zone->Allocate<NodeWithInLineInputs>(size));
105  void* node_buffer =
106  reinterpret_cast<void*>(raw_buffer + capacity * sizeof(Use));
107 
108  node = new (node_buffer) Node(id, op, input_count, capacity);
109  input_ptr = node->inline_inputs();
110  use_ptr = reinterpret_cast<Use*>(node);
111  is_inline = true;
112  }
113 
114  // Initialize the input pointers and the uses.
115  CHECK_IMPLIES(input_count > 0,
116  Use::InputIndexField::is_valid(input_count - 1));
117  for (int current = 0; current < input_count; ++current) {
118  Node* to = *inputs++;
119  input_ptr[current] = to;
120  Use* use = use_ptr - 1 - current;
121  use->bit_field_ = Use::InputIndexField::encode(current) |
122  Use::InlineField::encode(is_inline);
123  to->AppendUse(use);
124  }
125  node->Verify();
126  return node;
127 }

152 void Node::AppendInput(Zone* zone, Node* new_to) {
153  DCHECK_NOT_NULL(zone);
154  DCHECK_NOT_NULL(new_to);
155 
156  int const inline_count = InlineCountField::decode(bit_field_);
157  int const inline_capacity = InlineCapacityField::decode(bit_field_);
158  if (inline_count < inline_capacity) {
159  // Append inline input.
160  bit_field_ = InlineCountField::update(bit_field_, inline_count + 1);
161  *GetInputPtr(inline_count) = new_to;
162  Use* use = GetUsePtr(inline_count);
163  static_assert(InlineCapacityField::kMax <= Use::InputIndexField::kMax);
164  use->bit_field_ = Use::InputIndexField::encode(inline_count) |
165  Use::InlineField::encode(true);
166  new_to->AppendUse(use);
167  } else {
168  // Append out-of-line input.
169  int const input_count = InputCount();
170  OutOfLineInputs* outline = nullptr;
171  if (inline_count != kOutlineMarker) {
172  // switch to out of line inputs.
173  outline = OutOfLineInputs::New(zone, input_count * 2 + 3);
174  outline->node_ = this;
175  outline->ExtractFrom(GetUsePtr(0), GetInputPtr(0), input_count);
176  bit_field_ = InlineCountField::update(bit_field_, kOutlineMarker);
177  set_outline_inputs(outline);
178  } else {
179  // use current out of line inputs.
180  outline = outline_inputs();
181  if (input_count >= outline->capacity_) {
182  // out of space in out-of-line inputs.
183  outline = OutOfLineInputs::New(zone, input_count * 2 + 3);
184  outline->node_ = this;
185  outline->ExtractFrom(GetUsePtr(0), GetInputPtr(0), input_count);
186  set_outline_inputs(outline);
187  }
188  }
189  outline->count_++;
190  *GetInputPtr(input_count) = new_to;
191  Use* use = GetUsePtr(input_count);
192  CHECK(Use::InputIndexField::is_valid(input_count));
193  use->bit_field_ = Use::InputIndexField::encode(input_count) |
194  Use::InlineField::encode(false);
195  new_to->AppendUse(use);
196  }
197  Verify();
198 }

202  DCHECK_NOT_NULL(zone);
203  DCHECK_LE(0, index);
204  DCHECK_LT(index, InputCount());
205  AppendInput(zone, InputAt(InputCount() - 1));

243  ZoneNodePtr* input_ptr = GetInputPtr(start);

272 void Node::EnsureInputCount(Zone* zone, int new_input_count) {
273  int current_count = InputCount();
274  DCHECK_NE(current_count, 0);
275  if (current_count > new_input_count) {
276  TrimInputCount(new_input_count);
277  } else if (current_count < new_input_count) {
278  Node* dummy = InputAt(current_count - 1);
279  do {
280  AppendInput(zone, dummy);
281  current_count++;
282  } while (current_count < new_input_count);
283  }

304 void Node::ReplaceUses(Node* that) {
305  DCHECK(this->first_use_ == nullptr || this->first_use_->prev == nullptr);
306  DCHECK(that->first_use_ == nullptr || that->first_use_->prev == nullptr);
307 
308  // Update the pointers to {this} to point to {that}.
309  Use* last_use = nullptr;
310  for (Use* use = this->first_use_; use; use = use->next) {
311  *use->input_ptr() = that;
312  last_use = use;
313  }
314  if (last_use) {
315  // Concat the use list of {this} and {that}.
316  last_use->next = that->first_use_;
317  if (that->first_use_) that->first_use_->prev = last_use;
318  that->first_use_ = this->first_use_;
319  }
320  first_use_ = nullptr;
321 }
