45 class V8_EXPORT_PRIVATE MemoryChunk final {
46  public:
47  // All possible flags that can be set on a page. While the value of flags
48  // doesn't matter in principle, keep flags used in the write barrier together
49  // in order to have dense page flag checks in the write barrier.
50  enum Flag : uintptr_t {
51  NO_FLAGS = 0u,
52 
53  // This page belongs to a shared heap.
54  IN_WRITABLE_SHARED_SPACE = 1u << 0,
55 
56  // These two flags are used in the write barrier to catch "interesting"
57  // references.
58  POINTERS_TO_HERE_ARE_INTERESTING = 1u << 1,
59  POINTERS_FROM_HERE_ARE_INTERESTING = 1u << 2,
60 
61  // A page in the from-space or a young large page that was not scavenged
62  // yet.
63  FROM_PAGE = 1u << 3,
64  // A page in the to-space or a young large page that was scavenged.
65  TO_PAGE = 1u << 4,
66 
67  // |INCREMENTAL_MARKING|: Indicates whether incremental marking is currently
68  // enabled.
69  INCREMENTAL_MARKING = 1u << 5,
70 
71  // The memory chunk belongs to the read-only heap and does not participate
72  // in garbage collection. This is used instead of owner for identity
73  // checking since read-only chunks have no owner once they are detached.
74  READ_ONLY_HEAP = 1u << 6,
75 
76  // Used in young generation checks. When sticky mark-bits are enabled and
77  // major GC in progress, treat all objects as old.
78  IS_MAJOR_GC_IN_PROGRESS = 1u << 7,
79 
80  // Used to mark chunks belonging to spaces that do not suppor young gen
81  // allocations. Such chunks can never contain any young objects.
82  CONTAINS_ONLY_OLD = 1u << 8,
83 
84  // Page was allocated during major incremental marking. May only contain old
85  // objects.
86  BLACK_ALLOCATED = 1u << 9,
87 
88  // ----------------------------------------------------------------
89  // Values below here are not critical for the heap write barrier.
90 
91  LARGE_PAGE = 1u << 10,
92  EVACUATION_CANDIDATE = 1u << 11,
93  NEVER_EVACUATE = 1u << 12,
94 
95  // |PAGE_NEW_OLD_PROMOTION|: A page tagged with this flag has been promoted
96  // from new to old space during evacuation.
97  PAGE_NEW_OLD_PROMOTION = 1u << 13,
98 
99  // This flag is intended to be used for testing. Works only when both
100  // v8_flags.stress_compaction and
101  // v8_flags.manual_evacuation_candidates_selection are set. It forces the
102  // page to become an evacuation candidate at next candidates selection
103  // cycle.
104  FORCE_EVACUATION_CANDIDATE_FOR_TESTING = 1u << 14,
105 
106  // This flag is intended to be used for testing.
107  NEVER_ALLOCATE_ON_PAGE = 1u << 15,
108 
109  // The memory chunk is already logically freed, however the actual freeing
110  // still has to be performed.
111  PRE_FREED = 1u << 16,
112 
113  // |COMPACTION_WAS_ABORTED|: Indicates that the compaction in this page
114  // has been aborted and needs special handling by the sweeper.
115  COMPACTION_WAS_ABORTED = 1u << 17,
116 
117  NEW_SPACE_BELOW_AGE_MARK = 1u << 18,
118 
119  // The memory chunk freeing bookkeeping has been performed but the chunk has
120  // not yet been freed.
121  UNREGISTERED = 1u << 19,
122 
123  // The memory chunk is pinned in memory and can't be moved. This is likely
124  // because there exists a potential pointer to somewhere in the chunk which
125  // can't be updated.
126  PINNED = 1u << 20,
127 
128  // A Page with code objects.
129  IS_EXECUTABLE = 1u << 21,
130 
131  // The memory chunk belongs to the trusted space. When the sandbox is
132  // enabled, the trusted space is located outside of the sandbox and so its
133  // content cannot be corrupted by an attacker.
134  IS_TRUSTED = 1u << 22,
135 
136  // A quarantined page that contains objects reachable from stack during a
137  // scavenge. Quarantined pages are not used for further allocations in new
138  // space (to make it easier to keep track of the intermediate generation).
139  // This flag should only ever be set during a scavenge cycle.
140  IS_QUARANTINED = 1u << 23,
141 
142  // A new space page that will be promoted to old space by the end of the GC.
143  // This flag should only ever be set during a scavenge cycle.
144  WILL_BE_PROMOTED = 1u << 24,
145  };

436 // Define special hash function for chunk pointers, to be used with std data
437 // structures, e.g.
438 // std::unordered_set<MemoryChunk*, base::hash<MemoryChunk*>
439 // This hash function discards the trailing zero bits (chunk alignment).
440 // Notice that, when pointer compression is enabled, it also discards the
441 // cage base.
442 template <>
443 struct hash<const i::MemoryChunk*> {
444  V8_INLINE size_t operator()(const i::MemoryChunk* chunk) const {
445  return static_cast<v8::internal::Tagged_t>(
446  reinterpret_cast<uintptr_t>(chunk)) >>
447  kPageSizeBits;
448  }
449 };
450 
451 template <>
452 struct hash<i::MemoryChunk*> {
453  V8_INLINE size_t operator()(i::MemoryChunk* chunk) const {
454  return hash<const i::MemoryChunk*>()(chunk);
455  }
456 };
