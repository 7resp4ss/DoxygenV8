71 bool GcSafeCode::CanDeoptAt(Isolate* isolate, Address pc) const {
72  if (!UnsafeCastToCode()->uses_deoptimization_data()) return false;
73  Tagged<DeoptimizationData> deopt_data = UncheckedCast<DeoptimizationData>(
74  UnsafeCastToCode()->unchecked_deoptimization_data());
75  Address code_start_address = instruction_start();
76  for (int i = 0; i < deopt_data->DeoptCount(); i++) {
77  if (deopt_data->Pc(i).value() == -1) continue;
78  Address address = code_start_address + deopt_data->Pc(i).value();
79  if (address == pc && deopt_data->GetBytecodeOffsetOrBuiltinContinuationId(
80  i) != BytecodeOffset::None()) {
81  return true;
82  }
83  }
84  return false;
85 }

355 uintptr_t Code::GetBaselinePCForBytecodeOffset(
356  int bytecode_offset, BytecodeToPCPosition position,
357  Tagged<BytecodeArray> bytecodes) {
358  DisallowGarbageCollection no_gc;
359  CHECK_EQ(kind(), CodeKind::BASELINE);
360  // The following check ties together the bytecode being executed in
361  // Generate_BaselineOrInterpreterEntry with the bytecode that was used to
362  // compile this baseline code. Together, this ensures that we don't OSR into a
363  // wrong code object.
364  auto maybe_bytecodes = bytecode_or_interpreter_data();
365  if (IsBytecodeArray(maybe_bytecodes)) {
366  SBXCHECK_EQ(maybe_bytecodes, bytecodes);
367  } else {
368  CHECK(IsInterpreterData(maybe_bytecodes));
369  SBXCHECK_EQ(Cast<InterpreterData>(maybe_bytecodes)->bytecode_array(),
370  bytecodes);
371  }
372  baseline::BytecodeOffsetIterator offset_iterator(
373  Cast<TrustedByteArray>(bytecode_offset_table()), bytecodes);
374  offset_iterator.AdvanceToBytecodeOffset(bytecode_offset);
375  uintptr_t pc = 0;
376  if (position == kPcAtStartOfBytecode) {
377  pc = offset_iterator.current_pc_start_offset();
378  } else {
379  DCHECK_EQ(position, kPcAtEndOfBytecode);
380  pc = offset_iterator.current_pc_end_offset();
381  }
382  return pc;
383 }

536 inline void Code::SetMarkedForDeoptimization(Isolate* isolate,
537  LazyDeoptimizeReason reason) {
538  set_marked_for_deoptimization(true);
539  // Eager deopts are already logged by the deoptimizer.
540  if (reason != LazyDeoptimizeReason::kEagerDeopt &&
541  V8_UNLIKELY(v8_flags.trace_deopt || v8_flags.log_deopt)) {
542  TraceMarkForDeoptimization(isolate, reason);
543  }
544 #ifdef V8_ENABLE_LEAPTIERING
545  JSDispatchHandle handle = js_dispatch_handle();
546  if (handle != kNullJSDispatchHandle) {
547  JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
548  Tagged<Code> cur = jdt->GetCode(handle);
549  if (SafeEquals(cur)) {
550  if (v8_flags.reopt_after_lazy_deopts &&
551  isolate->concurrent_recompilation_enabled()) {
552  jdt->SetCodeNoWriteBarrier(
553  handle, *BUILTIN_CODE(isolate, InterpreterEntryTrampoline));
554  // Somewhat arbitrary list of lazy deopt reasons which we expect to be
555  // stable enough to warrant either immediate re-optimization, or
556  // re-optimization after one invocation (to detect potential follow-up
557  // IC changes).
558  // TODO(olivf): We should also work on reducing the number of
559  // dependencies we create in the compilers to require less of these
560  // quick re-compilations.
561  switch (reason) {
562  case LazyDeoptimizeReason::kAllocationSiteTenuringChange:
