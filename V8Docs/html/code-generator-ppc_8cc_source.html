31 // Adds PPC-specific methods to convert InstructionOperands.
32 class PPCOperandConverter final : public InstructionOperandConverter {
33  public:
34  PPCOperandConverter(CodeGenerator* gen, Instruction* instr)
35  : InstructionOperandConverter(gen, instr) {}
36 
37  size_t OutputCount() { return instr_->OutputCount(); }
38 
39  RCBit OutputRCBit() const {
40  switch (instr_->flags_mode()) {
41  case kFlags_branch:
42  case kFlags_conditional_branch:
43  case kFlags_deoptimize:
44  case kFlags_set:
45  case kFlags_conditional_set:
46  case kFlags_trap:
47  case kFlags_select:
48  return SetRC;
49  case kFlags_none:
50  return LeaveRC;
51  }
52  UNREACHABLE();
53  }
54 
55  bool CompareLogical() const {
56  switch (instr_->flags_condition()) {
57  case kUnsignedLessThan:
58  case kUnsignedGreaterThanOrEqual:
59  case kUnsignedLessThanOrEqual:
60  case kUnsignedGreaterThan:
61  return true;
62  default:
63  return false;
64  }
65  UNREACHABLE();
66  }
67 
68  Operand InputImmediate(size_t index) {
69  Constant constant = ToConstant(instr_->InputAt(index));
70  switch (constant.type()) {
71  case Constant::kInt32:
72  return Operand(constant.ToInt32());
73  case Constant::kFloat32:
74  return Operand::EmbeddedNumber(constant.ToFloat32());
75  case Constant::kFloat64:
76  return Operand::EmbeddedNumber(constant.ToFloat64().value());
77  case Constant::kInt64:
78  return Operand(constant.ToInt64());
79  case Constant::kExternalReference:
80  return Operand(constant.ToExternalReference());
81  case Constant::kCompressedHeapObject: {
82  RootIndex root_index;
83  if (gen_->isolate()->roots_table().IsRootHandle(constant.ToHeapObject(),
84  &root_index)) {
85  CHECK(COMPRESS_POINTERS_BOOL);
86  CHECK(V8_STATIC_ROOTS_BOOL || !gen_->isolate()->bootstrapper());
87  Tagged_t ptr =
88  MacroAssemblerBase::ReadOnlyRootPtr(root_index, gen_->isolate());
89  return Operand(ptr);
90  }
91  return Operand(constant.ToHeapObject());
92  }
93  case Constant::kHeapObject:
94  case Constant::kRpoNumber:
95  break;
96  }
97  UNREACHABLE();
98  }
99 
100  MemOperand MemoryOperand(AddressingMode* mode, size_t* first_index) {
101  const size_t index = *first_index;
102  AddressingMode addr_mode = AddressingModeField::decode(instr_->opcode());
103  if (mode) *mode = addr_mode;
104  switch (addr_mode) {
105  case kMode_None:
106  break;
107  case kMode_MRI:
108  *first_index += 2;
109  return MemOperand(InputRegister(index + 0), InputInt64(index + 1));
110  case kMode_MRR:
111  *first_index += 2;
112  return MemOperand(InputRegister(index + 0), InputRegister(index + 1));
113  case kMode_Root:
114  *first_index += 1;
115  return MemOperand(kRootRegister, InputRegister(index));
116  }
117  UNREACHABLE();
118  }
119 
120  MemOperand MemoryOperand(AddressingMode* mode = NULL,
121  size_t first_index = 0) {
122  return MemoryOperand(mode, &first_index);
123  }
124 
125  MemOperand ToMemOperand(InstructionOperand* op) const {
126  DCHECK_NOT_NULL(op);
127  DCHECK(op->IsStackSlot() || op->IsFPStackSlot());
128  return SlotToMemOperand(AllocatedOperand::cast(op)->index());
129  }
130 
131  MemOperand SlotToMemOperand(int slot) const {
132  FrameOffset offset = frame_access_state()->GetFrameOffset(slot);
133  return MemOperand(offset.from_stack_pointer() ? sp : fp, offset.offset());
134  }

191  void Generate() final {
192  ConstantPoolUnavailableScope constant_pool_unavailable(masm());
193  // When storing an indirect pointer, the value will always be a
194  // full/decompressed pointer.
195  if (COMPRESS_POINTERS_BOOL &&
196  mode_ != RecordWriteMode::kValueIsIndirectPointer) {
197  __ DecompressTagged(value_, value_);
198  }
199  __ CheckPageFlag(value_, scratch0_,
200  MemoryChunk::kPointersToHereAreInterestingMask, eq,
201  exit());
202  if (offset_ == no_reg) {
203  __ addi(scratch1_, object_, Operand(offset_immediate_));
204  } else {
205  DCHECK_EQ(0, offset_immediate_);
206  __ add(scratch1_, object_, offset_);
207  }
208  SaveFPRegsMode const save_fp_mode = frame()->DidAllocateDoubleRegisters()
209  ? SaveFPRegsMode::kSave
210  : SaveFPRegsMode::kIgnore;
211  if (must_save_lr_) {
212  // We need to save and restore lr if the frame was elided.
213  __ mflr(scratch0_);
214  __ Push(scratch0_);
215  unwinding_info_writer_->MarkLinkRegisterOnTopOfStack(__ pc_offset());
216  }
217  if (mode_ == RecordWriteMode::kValueIsEphemeronKey) {
218  __ CallEphemeronKeyBarrier(object_, scratch1_, save_fp_mode);
219  } else if (mode_ == RecordWriteMode::kValueIsIndirectPointer) {
220  DCHECK(IsValidIndirectPointerTag(indirect_pointer_tag_));
221  __ CallIndirectPointerBarrier(object_, scratch1_, save_fp_mode,
222  indirect_pointer_tag_);
223 #if V8_ENABLE_WEBASSEMBLY
224  } else if (stub_mode_ == StubCallMode::kCallWasmRuntimeStub) {
225  __ CallRecordWriteStubSaveRegisters(object_, scratch1_, save_fp_mode,
226  StubCallMode::kCallWasmRuntimeStub);
227 #endif // V8_ENABLE_WEBASSEMBLY
228  } else {
229  __ CallRecordWriteStubSaveRegisters(object_, scratch1_, save_fp_mode);
230  }
231  if (must_save_lr_) {
232  // We need to save and restore lr if the frame was elided.
233  __ Pop(scratch0_);
234  __ mtlr(scratch0_);
235  unwinding_info_writer_->MarkPopLinkRegisterFromTopOfStack(__ pc_offset());
236  }
237  }

256 Condition FlagsConditionToCondition(FlagsCondition condition, ArchOpcode op) {
257  switch (condition) {
258  case kEqual:
259  return eq;
260  case kNotEqual:
261  return ne;
262  case kSignedLessThan:
263  case kUnsignedLessThan:
264  return lt;
265  case kSignedGreaterThanOrEqual:
266  case kUnsignedGreaterThanOrEqual:
267  return ge;
268  case kSignedLessThanOrEqual:
269  case kUnsignedLessThanOrEqual:
270  return le;
271  case kSignedGreaterThan:
272  case kUnsignedGreaterThan:
273  return gt;
274  case kOverflow:
275  // Overflow checked for add/sub only.
276  switch (op) {
277  case kPPC_Add32:
278  case kPPC_Add64:
279  case kPPC_Sub:
280  case kPPC_AddWithOverflow32:
281  case kPPC_SubWithOverflow32:
282  return lt;
283  default:
284  break;
285  }
286  break;
287  case kNotOverflow:
288  switch (op) {
289  case kPPC_Add32:
290  case kPPC_Add64:
291  case kPPC_Sub:
292  case kPPC_AddWithOverflow32:
293  case kPPC_SubWithOverflow32:
294  return ge;
295  default:
296  break;
297  }
298  break;
299  default:
300  break;
301  }
302  UNREACHABLE();
303 }

727  StandardFrameConstants::kFixedSlotCountAboveFp;

739  masm->AddS64(sp, sp, Operand(-stack_slot_delta * kSystemPointerSize), r0);
740  state->IncreaseSPDelta(stack_slot_delta);

804 // Assembles an instruction after register allocation, producing machine code.
805 CodeGenerator::CodeGenResult CodeGenerator::AssembleArchInstruction(
806  Instruction* instr) {
807  PPCOperandConverter i(this, instr);
808  ArchOpcode opcode = ArchOpcodeField::decode(instr->opcode());
809 
810  switch (opcode) {
811  case kArchCallCodeObject: {
812  v8::internal::Assembler::BlockTrampolinePoolScope block_trampoline_pool(
813  masm());
814  if (HasRegisterInput(instr, 0)) {
815  Register reg = i.InputRegister(0);
816  DCHECK_IMPLIES(
817  instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister),
818  reg == kJavaScriptCallCodeStartRegister);
819  __ CallCodeObject(reg);
820  } else {
821  __ Call(i.InputCode(0), RelocInfo::CODE_TARGET);
822  }
823  RecordCallPosition(instr);
824  DCHECK_EQ(LeaveRC, i.OutputRCBit());
825  frame_access_state()->ClearSPDelta();
826  break;
827  }
828  case kArchCallBuiltinPointer: {
829  DCHECK(!instr->InputAt(0)->IsImmediate());
830  Register builtin_index = i.InputRegister(0);
831  Register target =
832  instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister)
833  ? kJavaScriptCallCodeStartRegister
834  : builtin_index;
835  __ CallBuiltinByIndex(builtin_index, target);
836  RecordCallPosition(instr);
837  frame_access_state()->ClearSPDelta();
838  break;
839  }
840 #if V8_ENABLE_WEBASSEMBLY
841  case kArchCallWasmFunction:
842  case kArchCallWasmFunctionIndirect: {
843  // We must not share code targets for calls to builtins for wasm code, as
844  // they might need to be patched individually.
845  if (instr->InputAt(0)->IsImmediate()) {
846  DCHECK_EQ(opcode, kArchCallWasmFunction);
847  Constant constant = i.ToConstant(instr->InputAt(0));
848  Address wasm_code = static_cast<Address>(constant.ToInt64());
849  __ Call(wasm_code, constant.rmode());
850  } else if (opcode == kArchCallWasmFunctionIndirect) {
851  __ CallWasmCodePointer(i.InputRegister(0));
852  } else {
853  __ Call(i.InputRegister(0));
854  }
855  RecordCallPosition(instr);
856  DCHECK_EQ(LeaveRC, i.OutputRCBit());
857  frame_access_state()->ClearSPDelta();
858  break;
859  }
860  case kArchTailCallWasm:
861  case kArchTailCallWasmIndirect: {
862  // We must not share code targets for calls to builtins for wasm code, as
863  // they might need to be patched individually.
864  if (instr->InputAt(0)->IsImmediate()) {
865  DCHECK_EQ(opcode, kArchTailCallWasm);
866  Constant constant = i.ToConstant(instr->InputAt(0));
867  Address wasm_code = static_cast<Address>(constant.ToInt64());
868  __ Jump(wasm_code, constant.rmode());
869  } else if (opcode == kArchTailCallWasmIndirect) {
870  __ CallWasmCodePointer(i.InputRegister(0), CallJumpMode::kTailCall);
871  } else {
872  __ Jump(i.InputRegister(0));
873  }
874  DCHECK_EQ(LeaveRC, i.OutputRCBit());
875  frame_access_state()->ClearSPDelta();
876  frame_access_state()->SetFrameAccessToDefault();
877  break;
878  }
879 #endif // V8_ENABLE_WEBASSEMBLY
880  case kArchTailCallCodeObject: {
881  if (HasRegisterInput(instr, 0)) {
882  Register reg = i.InputRegister(0);
883  DCHECK_IMPLIES(
884  instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister),
885  reg == kJavaScriptCallCodeStartRegister);
886  __ JumpCodeObject(reg);
887  } else {
888  // We cannot use the constant pool to load the target since
889  // we've already restored the caller's frame.
890  ConstantPoolUnavailableScope constant_pool_unavailable(masm());
891  __ Jump(i.InputCode(0), RelocInfo::CODE_TARGET);
892  }
893  DCHECK_EQ(LeaveRC, i.OutputRCBit());
894  frame_access_state()->ClearSPDelta();
895  frame_access_state()->SetFrameAccessToDefault();
896  break;
897  }
898  case kArchTailCallAddress: {
899  CHECK(!instr->InputAt(0)->IsImmediate());
900  Register reg = i.InputRegister(0);
901  DCHECK_IMPLIES(
902  instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister),
903  reg == kJavaScriptCallCodeStartRegister);
904  __ Jump(reg);
905  frame_access_state()->ClearSPDelta();
906  frame_access_state()->SetFrameAccessToDefault();
907  break;
908  }
909  case kArchCallJSFunction: {
910  v8::internal::Assembler::BlockTrampolinePoolScope block_trampoline_pool(
911  masm());
912  Register func = i.InputRegister(0);
913  if (v8_flags.debug_code) {
914  // Check the function's context matches the context argument.
915  __ LoadTaggedField(
916  kScratchReg, FieldMemOperand(func, JSFunction::kContextOffset), r0);
917  __ CmpS64(cp, kScratchReg);
918  __ Assert(eq, AbortReason::kWrongFunctionContext);
919  }
920  uint32_t num_arguments =
921  i.InputUint32(instr->JSCallArgumentCountInputIndex());
922  __ CallJSFunction(func, num_arguments, kScratchReg);
923  RecordCallPosition(instr);
924  DCHECK_EQ(LeaveRC, i.OutputRCBit());
925  frame_access_state()->ClearSPDelta();
926  break;
927  }
928  case kArchPrepareCallCFunction: {
929  int const num_parameters = MiscField::decode(instr->opcode());
930  __ PrepareCallCFunction(num_parameters, kScratchReg);
931  // Frame alignment requires using FP-relative frame addressing.
932  frame_access_state()->SetFrameAccessToFP();
933  break;
934  }
935  case kArchSaveCallerRegisters: {
936  fp_mode_ =
937  static_cast<SaveFPRegsMode>(MiscField::decode(instr->opcode()));
938  DCHECK(fp_mode_ == SaveFPRegsMode::kIgnore ||
939  fp_mode_ == SaveFPRegsMode::kSave);
940  // kReturnRegister0 should have been saved before entering the stub.
941  int bytes = __ PushCallerSaved(fp_mode_, ip, r0, kReturnRegister0);
942  DCHECK(IsAligned(bytes, kSystemPointerSize));
943  DCHECK_EQ(0, frame_access_state()->sp_delta());
944  frame_access_state()->IncreaseSPDelta(bytes / kSystemPointerSize);
945  DCHECK(!caller_registers_saved_);
946  caller_registers_saved_ = true;
947  break;
948  }
949  case kArchRestoreCallerRegisters: {
950  DCHECK(fp_mode_ ==
951  static_cast<SaveFPRegsMode>(MiscField::decode(instr->opcode())));
952  DCHECK(fp_mode_ == SaveFPRegsMode::kIgnore ||
953  fp_mode_ == SaveFPRegsMode::kSave);
954  // Don't overwrite the returned value.
955  int bytes = __ PopCallerSaved(fp_mode_, ip, r0, kReturnRegister0);
956  frame_access_state()->IncreaseSPDelta(-(bytes / kSystemPointerSize));
957  DCHECK_EQ(0, frame_access_state()->sp_delta());
958  DCHECK(caller_registers_saved_);
959  caller_registers_saved_ = false;
960  break;
961  }
962  case kArchPrepareTailCall:
963  AssemblePrepareTailCall();
964  break;
965  case kArchComment:
966  __ RecordComment(reinterpret_cast<const char*>(i.InputInt64(0)),
967  SourceLocation());
968  break;
969  case kArchCallCFunction: {
970  uint32_t param_counts = i.InputUint32(instr->InputCount() - 1);
971  int const num_gp_parameters = ParamField::decode(param_counts);
972  int const fp_param_field = FPParamField::decode(param_counts);
973  int num_fp_parameters = fp_param_field;
974  bool has_function_descriptor = false;
975  SetIsolateDataSlots set_isolate_data_slots = SetIsolateDataSlots::kYes;
976 #if ABI_USES_FUNCTION_DESCRIPTORS
977  // AIX/PPC64BE Linux uses a function descriptor
978  int kNumFPParametersMask = kHasFunctionDescriptorBitMask - 1;
979  num_fp_parameters = kNumFPParametersMask & fp_param_field;
980  has_function_descriptor =
981  (fp_param_field & kHasFunctionDescriptorBitMask) != 0;
982 #endif
983 #if V8_ENABLE_WEBASSEMBLY
984  Label start_call;
985  int start_pc_offset = 0;
986  bool isWasmCapiFunction =
987  linkage()->GetIncomingDescriptor()->IsWasmCapiFunction();
988  if (isWasmCapiFunction) {
989  __ mflr(r0);
990  __ LoadPC(kScratchReg);
991  __ bind(&start_call);
992  start_pc_offset = __ pc_offset();
993  // We are going to patch this instruction after emitting
994  // CallCFunction, using a zero offset here as placeholder for now.
995  // patch_pc_address assumes `addi` is used here to
996  // add the offset to pc.
997  __ addi(kScratchReg, kScratchReg, Operand::Zero());
998  __ StoreU64(kScratchReg,
999  MemOperand(fp, WasmExitFrameConstants::kCallingPCOffset));
1000  __ mtlr(r0);
1001  set_isolate_data_slots = SetIsolateDataSlots::kNo;
1002  }
1003 #endif // V8_ENABLE_WEBASSEMBLY
1004  int pc_offset;
1005  if (instr->InputAt(0)->IsImmediate()) {
1006  ExternalReference ref = i.InputExternalReference(0);
1007  pc_offset =
1008  __ CallCFunction(ref, num_gp_parameters, num_fp_parameters,
1009  set_isolate_data_slots, has_function_descriptor);
1010  } else {
1011  Register func = i.InputRegister(0);
1012  pc_offset =
1013  __ CallCFunction(func, num_gp_parameters, num_fp_parameters,
1014  set_isolate_data_slots, has_function_descriptor);
1015  }
1016 #if V8_ENABLE_WEBASSEMBLY
1017  if (isWasmCapiFunction) {
1018  int offset_since_start_call = pc_offset - start_pc_offset;
1019  // Here we are going to patch the `addi` instruction above to use the
1020  // correct offset.

1022  // emitted instruction. Add one more to the offset to point to after the
1023  // Call.

1025  __ patch_pc_address(kScratchReg, start_pc_offset,
1026  offset_since_start_call);
1027  }

1029  RecordSafepoint(instr->reference_map(), pc_offset);
1030

1032  handlers_.push_back({nullptr, pc_offset});
1033  }
1034  if (instr->HasCallDescriptorFlag(CallDescriptor::kNeedsFrameState)) {

1098  break;

1114  case kArchSetStackPointer: {

1119 #endif // V8_ENABLE_WEBASSEMBLY
