59  // A collection of segments in an external entity table.
60  //
61  // For the purpose of memory management, a table is partitioned into segments
62  // of a fixed size (e.g. 64kb). A Space is a collection of segments that all
63  // share the same freelist. As such, entry allocation and freeing (e.g.
64  // through garbage collection) all happen on the level of spaces.
65  //
66  // Spaces allow implementing features such as:
67  // * Young generation GC support (a separate space is used for all entries
68  // belonging to the young generation)
69  // * Having double-width entries in a table (a dedicated space is used that
70  // contains only double-width entries)
71  // * Sharing one table between multiple isolates that perform GC independently
72  // (each Isolate owns one space)
73  struct Space {
74  public:
75  Space() = default;
76  Space(const Space&) = delete;
77  Space& operator=(const Space&) = delete;
78  ~Space();
79 
80  // Determines the number of entries currently on the freelist.
81  // As entries can be allocated from other threads, the freelist size may
82  // have changed by the time this method returns. As such, the returned
83  // value should only be treated as an approximation.
84  uint32_t freelist_length() const;
85 
86  // Returns the current number of segments currently associated with this
87  // space.
88  // The caller must lock the mutex.
89  uint32_t num_segments();
90 
91  // Returns whether this space is currently empty.
92  // The caller must lock the mutex.
93  bool is_empty() { return num_segments() == 0; }
94 
95  // Returns the current capacity of this space.
96  // The capacity of a space is the total number of entries it can contain.
97  // The caller must lock the mutex.
98  uint32_t capacity() { return num_segments() * kEntriesPerSegment; }
99 
100  // Returns true if this space contains the entry with the given index.
101  bool Contains(uint32_t index);
102 
103  // Whether this space is attached to a table's internal read-only segment.
104  bool is_internal_read_only_space() const {
105  return is_internal_read_only_space_;
106  }
107 
108 #ifdef DEBUG
109  // Check whether this space belongs to the given external entity table.
110  bool BelongsTo(const void* table) const { return owning_table_ == table; }
111 #endif // DEBUG
112 
113  // Similar to `num_segments()` but also locks the mutex.
114  uint32_t NumSegmentsForTesting() {
115  base::MutexGuard guard(&mutex_);
116  return num_segments();
117  }
118 
119  protected:
120  friend class ExternalEntityTable<Entry, size>;
121 
122 #ifdef DEBUG
123  // In debug builds we keep track of which table a space belongs to to be
124  // able to insert additional DCHECKs that verify that spaces are always used
125  // with the correct table.
126  std::atomic<void*> owning_table_ = nullptr;
127 #endif
128 
129  // The freelist used by this space.
130  // This contains both the index of the first entry in the freelist and the
131  // total length of the freelist as both values need to be updated together
132  // in a single atomic operation to stay consistent in the case of concurrent
133  // entry allocations.
134  std::atomic<FreelistHead> freelist_head_ = FreelistHead();
135 
136  // The collection of segments belonging to this space.
137  std::set<Segment> segments_;
138 
139  // Whether this is the internal RO space, which has special semantics:
140  // - read-only page permissions after initialization,
141  // - the space is not swept since slots are live by definition,
142  // - contains exactly one segment, located at offset 0, and
143  // - the segment's lifecycle is managed by `owning_table_`.
144  bool is_internal_read_only_space_ = false;
145 
146  // Mutex guarding access to the segments_ set.
147  base::Mutex mutex_;
148  };

165  // Allocates a new entry in the given space and return its index.

173  // Attempts to allocate an entry in the given space below the specified index.
174  //
175  // If there are no free entries at a lower index, this method will fail and
176  // return zero. This method will therefore never allocate a new segment.
177  // This method is atomic and can be called from background threads.
178  uint32_t AllocateEntryBelow(Space* space, uint32_t threshold_index);

180  // Try to allocate the first entry of the freelist.
181  //
182  // This method is mostly a wrapper around an atomic compare-and-swap which
183  // replaces the current freelist head with the next entry in the freelist,
184  // thereby allocating the entry at the start of the freelist.
185  bool TryAllocateEntryFromFreelist(Space* space, FreelistHead freelist);
186 
187  // Trey to allocate a new segment and add it to the given space.
188  //
189  // This should only be called when the freelist of the space is currently
190  // empty. It will then refill the freelist with all entries in the newly
191  // allocated segment. Fails if there is no space left.
192  std::optional<FreelistHead> TryExtend(Space* space);
193 
194  // Sweeps the given space.
195  //
196  // This will free all unmarked entries to the freelist and unmark all live
197  // entries. The table is swept top-to-bottom so that the freelist ends up
198  // sorted. During sweeping, new entries must not be allocated.
199  //
200  // This is a generic implementation of table sweeping and requires that the
201  // Entry type implements the following additional methods:
202  // - bool IsMarked()
203  // - void Unmark()
204  //
205  // Returns the number of live entries after sweeping.
206  uint32_t GenericSweep(Space* space);
207 
208  // Variant of the above that invokes a callback for every live entry.
209  template <typename Callback>
210  uint32_t GenericSweep(Space* space, Callback marked);
211 
212  // Iterate over all entries in the given space.
213  //
214  // The callback function will be invoked for every entry and be passed the
