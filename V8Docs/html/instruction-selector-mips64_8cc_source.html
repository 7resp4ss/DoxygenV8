28  InstructionOperand UseOperand(OpIndex node, InstructionCode opcode) {
29  if (CanBeImmediate(node, opcode)) {
30  return UseImmediate(node);
31  }
32  return UseRegister(node);
33  }

35  // Use the zero register if the node has the immediate value zero, otherwise
36  // assign a register.
37  InstructionOperand UseRegisterOrImmediateZero(OpIndex node) {
38  if (const ConstantOp* constant =
39  selector()->Get(node).TryCast<ConstantOp>()) {
40  if ((constant->IsIntegral() && constant->integral() == 0) ||
41  (constant->kind == ConstantOp::Kind::kFloat32 &&
42  constant->float32().get_bits() == 0) ||
43  (constant->kind == ConstantOp::Kind::kFloat64 &&
44  constant->float64().get_bits() == 0)) {
45  return UseImmediate(node);
46  }
47  }
48  return UseRegister(node);
49  }

56  std::optional<int64_t> GetOptionalIntegerConstant(OpIndex operation) {

63  bool CanBeImmediate(OpIndex node, InstructionCode mode) {
64  const ConstantOp* constant = selector()->Get(node).TryCast<ConstantOp>();
65  if (!constant) return false;
66 
67  int64_t value;
68  return selector()->MatchSignedIntegralConstant(node, &value) &&
69  CanBeImmediate(value, mode);
70  }

72  bool CanBeImmediate(int64_t value, InstructionCode opcode) {
73  switch (ArchOpcodeField::decode(opcode)) {
74  case kMips64Shl:
75  case kMips64Sar:
76  case kMips64Shr:
77  return is_uint5(value);
78  case kMips64Dshl:
79  case kMips64Dsar:
80  case kMips64Dshr:
81  return is_uint6(value);
82  case kMips64Add:
83  case kMips64And32:
84  case kMips64And:
85  case kMips64Dadd:
86  case kMips64Or32:
87  case kMips64Or:
88  case kMips64Tst:
89  case kMips64Xor:
90  return is_uint16(value);
91  case kMips64Lb:
92  case kMips64Lbu:
93  case kMips64Sb:
94  case kMips64Lh:
95  case kMips64Lhu:
96  case kMips64Sh:
97  case kMips64Lw:
98  case kMips64Sw:
99  case kMips64Ld:
100  case kMips64Sd:
101  case kMips64Lwc1:
102  case kMips64Swc1:
103  case kMips64Ldc1:
104  case kMips64Sdc1:
105  return is_int32(value);
106  default:
107  return is_int16(value);
108  }
109  }

115  }

169 struct ExtendingLoadMatcher {
170  ExtendingLoadMatcher(OpIndex node, InstructionSelector* selector)
171  : matches_(false), selector_(selector), immediate_(0) {
172  Initialize(node);
173  }
174 
175  bool Matches() const { return matches_; }
176 
177  OpIndex base() const {
178  DCHECK(Matches());
179  return base_;
180  }
181  int64_t immediate() const {
182  DCHECK(Matches());
183  return immediate_;
184  }
185  ArchOpcode opcode() const {
186  DCHECK(Matches());
187  return opcode_;
188  }
189 
190  private:
191  bool matches_;
192  InstructionSelector* selector_;
193  OpIndex base_{};
194  int64_t immediate_;
195  ArchOpcode opcode_;
196 
197  void Initialize(turboshaft::OpIndex node) {
198  const ShiftOp& shift = selector_->Get(node).template Cast<ShiftOp>();
199  DCHECK(shift.kind == ShiftOp::Kind::kShiftRightArithmetic ||
200  shift.kind == ShiftOp::Kind::kShiftRightArithmeticShiftOutZeros);
201  // When loading a 64-bit value and shifting by 32, we should
202  // just load and sign-extend the interesting 4 bytes instead.
203  // This happens, for example, when we're loading and untagging SMIs.
204  const Operation& lhs = selector_->Get(shift.left());
205  int64_t constant_rhs;
206 
207  if (lhs.Is<LoadOp>() &&
208  selector_->MatchIntegralWord64Constant(shift.right(), &constant_rhs) &&
209  constant_rhs == 32 && selector_->CanCover(node, shift.left())) {
210  Mips64OperandGenerator g(selector_);
211 
212  const LoadOp& load = lhs.Cast<LoadOp>();
213  base_ = load.base();
214  opcode_ = kMips64Lw;
215  if (load.index().has_value()) {
216  int64_t index_constant;
217  if (selector_->MatchIntegralWord64Constant(load.index().value(),
218  &index_constant)) {
219  DCHECK_EQ(load.element_size_log2, 0);
220  immediate_ = index_constant + 4;
221  matches_ = g.CanBeImmediate(immediate_, kMips64Lw);
222  }
223  } else {
224  immediate_ = load.offset + 4;
225  matches_ = g.CanBeImmediate(immediate_, kMips64Lw);
226  }
227  }
228  }
229 };

231 bool TryEmitExtendingLoad(InstructionSelector* selector, OpIndex node,
232  OpIndex output_node) {
233  ExtendingLoadMatcher m(node, selector);
234  Mips64OperandGenerator g(selector);
235  if (m.Matches()) {
236  InstructionOperand inputs[2];
237  inputs[0] = g.UseRegister(m.base());
238  InstructionCode opcode =
239  m.opcode() | AddressingModeField::encode(kMode_MRI);
240  DCHECK(is_int32(m.immediate()));
241  inputs[1] = g.TempImmediate(static_cast<int32_t>(m.immediate()));
242  InstructionOperand outputs[] = {g.DefineAsRegister(output_node)};
243  selector->Emit(opcode, arraysize(outputs), outputs, arraysize(inputs),
244  inputs);
245  return true;
246  }
247  return false;
248 }

263 static void VisitBinop(InstructionSelector* selector, OpIndex node,
264  InstructionCode opcode, bool has_reverse_opcode,
265  InstructionCode reverse_opcode,
266  FlagsContinuation* cont) {
267  Mips64OperandGenerator g(selector);
268  InstructionOperand inputs[2];
269  size_t input_count = 0;
270  InstructionOperand outputs[1];
271  size_t output_count = 0;
272 
273  const Operation& binop = selector->Get(node);
274  OpIndex left_node = binop.input(0);
275  OpIndex right_node = binop.input(1);
276 
277  if (TryMatchImmediate(selector, &opcode, right_node, &input_count,
278  &inputs[1])) {
279  inputs[0] = g.UseRegister(left_node);
280  input_count++;
281  } else if (has_reverse_opcode &&
282  TryMatchImmediate(selector, &reverse_opcode, left_node,
283  &input_count, &inputs[1])) {
284  inputs[0] = g.UseRegister(right_node);
285  opcode = reverse_opcode;
286  input_count++;
287  } else {
288  inputs[input_count++] = g.UseRegister(left_node);
289  inputs[input_count++] = g.UseOperand(right_node, opcode);
290  }
291 
292  outputs[output_count++] = g.DefineAsRegister(node);
293 
294  DCHECK_NE(0u, input_count);
295  DCHECK_EQ(1u, output_count);
296  DCHECK_GE(arraysize(inputs), input_count);
297  DCHECK_GE(arraysize(outputs), output_count);
298 
299  selector->EmitWithContinuation(opcode, output_count, outputs, input_count,
300  inputs, cont);
301 }

336 void EmitLoad(InstructionSelector* selector, OpIndex node,
337  InstructionCode opcode, OpIndex output = OpIndex{}) {
338  Mips64OperandGenerator g(selector);
339  const Operation& op = selector->Get(node);
340  const LoadOp& load = op.Cast<LoadOp>();
341 
342  // The LoadStoreSimplificationReducer transforms all loads into
343  // *(base + index).
344  OpIndex base = load.base();
345  OpIndex index = load.index().value();
346  CHECK_EQ(load.offset, 0);
347  DCHECK_EQ(load.element_size_log2, 0);
348 
349  InstructionOperand inputs[3];
350  size_t input_count = 0;
351  InstructionOperand output_op;
352 
353  // If output is valid, use that as the output register. This is used when we
354  // merge a conversion into the load.
355  output_op = g.DefineAsRegister(output.valid() ? output : node);
356 
357  const Operation& base_op = selector->Get(base);
358  int64_t index_constant;
359  const bool is_index_constant =
360  selector->MatchSignedIntegralConstant(index, &index_constant);
361  if (base_op.Is<Opmask::kExternalConstant>() && is_index_constant) {
362  const ConstantOp& constant_base = base_op.Cast<ConstantOp>();
363  if (selector->CanAddressRelativeToRootsRegister(
364  constant_base.external_reference())) {
365  ptrdiff_t const delta =
366  index_constant +
367  MacroAssemblerBase::RootRegisterOffsetForExternalReference(
368  selector->isolate(), constant_base.external_reference());
369  input_count = 1;
370  // Check that the delta is a 32-bit integer due to the limitations of
371  // immediate operands.
372  if (is_int32(delta)) {
373  inputs[0] = g.UseImmediate(static_cast<int32_t>(delta));
374  opcode |= AddressingModeField::encode(kMode_Root);
375  selector->Emit(opcode, 1, &output_op, input_count, inputs);
376  return;
377  }
378  }
379  }
380 
381  if (base_op.Is<LoadRootRegisterOp>() && is_index_constant) {
382  input_count = 1;
383  inputs[0] = g.UseImmediate64(index_constant);
384  opcode |= AddressingModeField::encode(kMode_Root);
385  selector->Emit(opcode, 1, &output_op, input_count, inputs);
386  return;
387  }
388 
389  if (g.CanBeImmediate(index, opcode)) {
390  selector->Emit(opcode | AddressingModeField::encode(kMode_MRI),
391  g.DefineAsRegister(output.valid() ? output : node),
392  g.UseRegister(base), g.UseImmediate(index));
393  } else {
394  InstructionOperand addr_reg = g.TempRegister();
395  selector->Emit(kMips64Dadd | AddressingModeField::encode(kMode_None),
396  addr_reg, g.UseRegister(index), g.UseRegister(base));
397  // Emit desired load opcode, using temp addr_reg.
398  selector->Emit(opcode | AddressingModeField::encode(kMode_MRI),
399  g.DefineAsRegister(output.valid() ? output : node), addr_reg,
400  g.TempImmediate(0));
401  }
402 }

412 void InstructionSelector::VisitLoad(OpIndex node) {
413  auto load = this->load_view(node);
414  LoadRepresentation load_rep = load.loaded_rep();
415 
416  InstructionCode opcode = kArchNop;
417  switch (load_rep.representation()) {
418  case MachineRepresentation::kFloat32:
419  opcode = kMips64Lwc1;
420  break;
421  case MachineRepresentation::kFloat64:
422  opcode = kMips64Ldc1;
423  break;
424  case MachineRepresentation::kBit: // Fall through.
425  case MachineRepresentation::kWord8:
426  opcode = load_rep.IsUnsigned() ? kMips64Lbu : kMips64Lb;
427  break;
428  case MachineRepresentation::kWord16:
429  opcode = load_rep.IsUnsigned() ? kMips64Lhu : kMips64Lh;
430  break;
431  case MachineRepresentation::kWord32:
432  opcode = kMips64Lw;
433  break;
434  case MachineRepresentation::kTaggedSigned: // Fall through.
435  case MachineRepresentation::kTaggedPointer: // Fall through.
436  case MachineRepresentation::kTagged: // Fall through.
437  case MachineRepresentation::kWord64:
438  opcode = kMips64Ld;
439  break;
440  case MachineRepresentation::kSimd128:
441  opcode = kMips64MsaLd;
442  break;
443  case MachineRepresentation::kFloat16:
444  UNIMPLEMENTED();
445  case MachineRepresentation::kSimd256: // Fall through.
446  case MachineRepresentation::kCompressedPointer: // Fall through.
447  case MachineRepresentation::kProtectedPointer: // Fall through.
448  case MachineRepresentation::kSandboxedPointer: // Fall through.
449  case MachineRepresentation::kCompressed: // Fall through.
450  case MachineRepresentation::kMapWord: // Fall through.
451  case MachineRepresentation::kIndirectPointer: // Fall through.
452  case MachineRepresentation::kFloat16RawBits: // Fall through.
453  case MachineRepresentation::kNone:
454  UNREACHABLE();
455  }
456 
457  EmitLoad(this, node, opcode);
458 }

467 void InstructionSelector::VisitStore(OpIndex node) {
468  Mips64OperandGenerator g(this);
469  StoreView store_view = this->store_view(node);
470  DCHECK_EQ(store_view.displacement(), 0);
471  OpIndex base = store_view.base();
472  OpIndex index = store_view.index().value();
473  OpIndex value = store_view.value();
474 
475  WriteBarrierKind write_barrier_kind =
476  store_view.stored_rep().write_barrier_kind();
477  MachineRepresentation rep = store_view.stored_rep().representation();
478 
479  if (v8_flags.enable_unconditional_write_barriers && CanBeTaggedPointer(rep)) {
480  write_barrier_kind = kFullWriteBarrier;
481  }
482 
483  // TODO(mips): I guess this could be done in a better way.
484  if (write_barrier_kind != kNoWriteBarrier &&
485  !v8_flags.disable_write_barriers) {
486  DCHECK(CanBeTaggedPointer(rep));
487  InstructionOperand inputs[3];
488  size_t input_count = 0;
489  inputs[input_count++] = g.UseUniqueRegister(base);
490  inputs[input_count++] = g.UseUniqueRegister(index);
491  inputs[input_count++] = g.UseUniqueRegister(value);
492  RecordWriteMode record_write_mode =
493  WriteBarrierKindToRecordWriteMode(write_barrier_kind);
494  InstructionOperand temps[] = {g.TempRegister(), g.TempRegister()};
495  size_t const temp_count = arraysize(temps);
496  InstructionCode code = kArchStoreWithWriteBarrier;
497  code |= MiscField::encode(static_cast<int>(record_write_mode));
498  Emit(code, 0, nullptr, input_count, inputs, temp_count, temps);
499  } else {
500  ArchOpcode opcode;
501  switch (rep) {
502  case MachineRepresentation::kFloat32:
503  opcode = kMips64Swc1;
504  break;
505  case MachineRepresentation::kFloat64:
506  opcode = kMips64Sdc1;
507  break;
508  case MachineRepresentation::kBit: // Fall through.
509  case MachineRepresentation::kWord8:
510  opcode = kMips64Sb;
511  break;
512  case MachineRepresentation::kWord16:
513  opcode = kMips64Sh;
514  break;
515  case MachineRepresentation::kWord32:
516  opcode = kMips64Sw;
517  break;
518  case MachineRepresentation::kTaggedSigned: // Fall through.
519  case MachineRepresentation::kTaggedPointer: // Fall through.
520  case MachineRepresentation::kTagged: // Fall through.
521  case MachineRepresentation::kWord64:
522  opcode = kMips64Sd;
523  break;
524  case MachineRepresentation::kSimd128:
525  opcode = kMips64MsaSt;
526  break;
527  case MachineRepresentation::kFloat16:
528  UNIMPLEMENTED();
529  case MachineRepresentation::kSimd256: // Fall through.
530  case MachineRepresentation::kCompressedPointer: // Fall through.
531  case MachineRepresentation::kCompressed: // Fall through.
532  case MachineRepresentation::kProtectedPointer: // Fall through.
533  case MachineRepresentation::kSandboxedPointer: // Fall through.
534  case MachineRepresentation::kMapWord: // Fall through.
535  case MachineRepresentation::kIndirectPointer: // Fall through.
536  case MachineRepresentation::kFloat16RawBits: // Fall through.
537  case MachineRepresentation::kNone:
538  UNREACHABLE();
539  }
540 
541  if (Is<LoadRootRegisterOp>(base)) {
542  // This will only work if {index} is a constant.
543  Emit(opcode | AddressingModeField::encode(kMode_Root), g.NoOutput(),
544  g.UseImmediate(index), g.UseRegisterOrImmediateZero(value));
545  return;
546  }
547 
548  if (g.CanBeImmediate(index, opcode)) {
549  Emit(opcode | AddressingModeField::encode(kMode_MRI), g.NoOutput(),
550  g.UseRegister(base), g.UseImmediate(index),
551  g.UseRegisterOrImmediateZero(value));
552  } else {
553  InstructionOperand addr_reg = g.TempRegister();
554  Emit(kMips64Dadd | AddressingModeField::encode(kMode_None), addr_reg,
555  g.UseRegister(index), g.UseRegister(base));
556  // Emit desired store opcode, using temp addr_reg.
557  Emit(opcode | AddressingModeField::encode(kMode_MRI), g.NoOutput(),
558  addr_reg, g.TempImmediate(0), g.UseRegisterOrImmediateZero(value));
559  }
560  }
561 }

611 void InstructionSelector::VisitWord64Shl(OpIndex node) {
612  const ShiftOp& shift_op = this->Get(node).template Cast<ShiftOp>();
613  const Operation& lhs = this->Get(shift_op.left());
614  const Operation& rhs = this->Get(shift_op.right());

1586  case kNotEqual:

1591  return VisitCompare(selector, opcode, g.UseRegister(right),
1592  g.UseRegister(left), cont);

1917 // Shared routine for word comparisons against zero.
1918 void InstructionSelector::VisitWordCompareZero(OpIndex user, OpIndex value,
1919  FlagsContinuation* cont) {
1920  {
1921  Mips64OperandGenerator g(this);
1922  // Try to combine with comparisons against 0 by simply inverting the branch.
1923  while (const ComparisonOp* equal =
1924  this->TryCast<Opmask::kWord32Equal>(value)) {
1925  if (!CanCover(user, value)) break;
1926  if (!MatchIntegralZero(equal->right())) break;
1927 
1928  user = value;
1929  value = equal->left();
1930  cont->Negate();
1931  }
1932  const Operation& value_op = Get(value);
1933  if (CanCover(user, value)) {
1934  if (const ComparisonOp* comparison = value_op.TryCast<ComparisonOp>()) {
1935  switch (comparison->rep.value()) {
1936  case RegisterRepresentation::Word32():
1937  cont->OverwriteAndNegateIfEqual(
1938  GetComparisonFlagCondition(*comparison));
1939  return VisitWord32Compare(this, value, cont);
1940 
1941  case RegisterRepresentation::Word64():
1942  cont->OverwriteAndNegateIfEqual(
1943  GetComparisonFlagCondition(*comparison));
1944  return VisitWord64Compare(this, value, cont);
1945 
1946  case RegisterRepresentation::Float32():
1947  switch (comparison->kind) {
1948  case ComparisonOp::Kind::kEqual:
1949  cont->OverwriteAndNegateIfEqual(kEqual);
1950  return VisitFloat32Compare(this, value, cont);
1951  case ComparisonOp::Kind::kSignedLessThan:
1952  cont->OverwriteAndNegateIfEqual(kFloatLessThan);
1953  return VisitFloat32Compare(this, value, cont);
1954  case ComparisonOp::Kind::kSignedLessThanOrEqual:
1955  cont->OverwriteAndNegateIfEqual(kFloatLessThanOrEqual);
1956  return VisitFloat32Compare(this, value, cont);
1957  default:
1958  UNREACHABLE();
1959  }
1960 
1961  case RegisterRepresentation::Float64():
1962  switch (comparison->kind) {
1963  case ComparisonOp::Kind::kEqual:
1964  cont->OverwriteAndNegateIfEqual(kEqual);
1965  return VisitFloat64Compare(this, value, cont);
1966  case ComparisonOp::Kind::kSignedLessThan:
1967  cont->OverwriteAndNegateIfEqual(kFloatLessThan);
1968  return VisitFloat64Compare(this, value, cont);
1969  case ComparisonOp::Kind::kSignedLessThanOrEqual:
1970  cont->OverwriteAndNegateIfEqual(kFloatLessThanOrEqual);
1971  return VisitFloat64Compare(this, value, cont);
1972  default:
1973  UNREACHABLE();
1974  }
1975 
1976  default:
1977  break;
1978  }
1979  } else if (const ProjectionOp* projection =
1980  value_op.TryCast<ProjectionOp>()) {
1981  // Check if this is the overflow output projection of an
1982  // <Operation>WithOverflow node.
1983  if (projection->index == 1u) {
1984  // We cannot combine the <Operation>WithOverflow with this branch
1985  // unless the 0th projection (the use of the actual value of the
1986  // <Operation> is either nullptr, which means there's no use of the
1987  // actual value, or was already defined, which means it is scheduled
1988  // *AFTER* this branch).
1989  OpIndex node = projection->input();
1990  OptionalOpIndex result = FindProjection(node, 0);
1991  if (!result.valid() || IsDefined(result.value())) {
1992  if (const OverflowCheckedBinopOp* binop =
1993  TryCast<OverflowCheckedBinopOp>(node)) {
1994  const bool is64 = binop->rep == WordRepresentation::Word64();
1995  switch (binop->kind) {
1996  case OverflowCheckedBinopOp::Kind::kSignedAdd:
1997  cont->OverwriteAndNegateIfEqual(kOverflow);
1998  return VisitBinop(this, node,
1999  is64 ? kMips64DaddOvf : kMips64Dadd, cont);
2000  case OverflowCheckedBinopOp::Kind::kSignedSub:
2001  cont->OverwriteAndNegateIfEqual(kOverflow);
2002  return VisitBinop(this, node,
2003  is64 ? kMips64DsubOvf : kMips64Dsub, cont);
2004  case OverflowCheckedBinopOp::Kind::kSignedMul:
2005  cont->OverwriteAndNegateIfEqual(kOverflow);
2006  return VisitBinop(
2007  this, node, is64 ? kMips64DMulOvf : kMips64MulOvf, cont);
2008  }
2009  }
2010  }
2011  }
2012  } else if (value_op.Is<Opmask::kWord32BitwiseAnd>() ||
2013  value_op.Is<Opmask::kWord64BitwiseAnd>()) {
2014  VisitWordCompare(this, value, kMips64Tst, cont, true);
2015  return;
2016  } else if (value_op.Is<StackPointerGreaterThanOp>()) {
2017  cont->OverwriteAndNegateIfEqual(kStackPointerGreaterThanCondition);
2018  return VisitStackPointerGreaterThan(value, cont);
2019  }
2020  }
2021  // Continuation could not be combined with a compare, emit compare against
2022  // 0.
2023  EmitWordCompareZero(this, value, cont);
2024  }
2025 }
