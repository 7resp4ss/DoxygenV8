54 void WritableRelocInfo::apply(intptr_t delta) {
55  // absolute code pointer inside code object moves with the code object.
56  if (IsInternalReference(rmode_)) {
57  // Jump table entry
58  Address target = Memory<Address>(pc_);
59  jit_allocation_.WriteValue(pc_, target + delta);
60  } else {
61  // mov sequence
62  DCHECK(IsInternalReferenceEncoded(rmode_));
63  Address target = Assembler::target_address_at(pc_, constant_pool_);
64  Assembler::set_target_address_at(pc_, constant_pool_, target + delta,
65  &jit_allocation_, SKIP_ICACHE_FLUSH);
66  }
67 }

90 Address RelocInfo::target_address_address() {
91  DCHECK(HasTargetAddressAddress());
92 
93  if (V8_EMBEDDED_CONSTANT_POOL_BOOL &&
94  Assembler::IsConstantPoolLoadStart(pc_)) {
95  // We return the PC for embedded constant pool since this function is used
96  // by the serializer and expects the address to reside within the code
97  // object.
98  return pc_;
99  }
100 
101  // Read the address of the word containing the target_address in an
102  // instruction stream.
103  // The only architecture-independent user of this function is the serializer.
104  // The serializer uses it to find out how many raw bytes of instruction to
105  // output before the next target.
106  // For an instruction like LIS/ORI where the target bits are mixed into the
107  // instruction bits, the size of the target will be zero, indicating that the
108  // serializer should not step forward in memory after a target is resolved
109  // and written.
110  return pc_;
111 }

202  Assembler::set_target_address_at(pc_, constant_pool_, target.ptr(),

257 // Fetch the 32bit value from the FIXED_SEQUENCE lis/ori
258 Address Assembler::target_address_at(Address pc, Address constant_pool) {
259  if (V8_EMBEDDED_CONSTANT_POOL_BOOL && constant_pool) {
260  ConstantPoolEntry::Access access;
261  if (IsConstantPoolLoadStart(pc, &access))
262  return Memory<Address>(target_constant_pool_address_at(
263  pc, constant_pool, access, ConstantPoolEntry::INTPTR));
264  }
265 
266  Instr instr1 = instr_at(pc);
267  Instr instr2 = instr_at(pc + kInstrSize);
268  // Interpret 2 instructions generated by lis/ori
269  if (IsLis(instr1) && IsOri(instr2)) {
270  Instr instr4 = instr_at(pc + (3 * kInstrSize));
271  Instr instr5 = instr_at(pc + (4 * kInstrSize));
272  // Assemble the 64 bit value.
273  uint64_t hi = (static_cast<uint32_t>((instr1 & kImm16Mask) << 16) |
274  static_cast<uint32_t>(instr2 & kImm16Mask));
275  uint64_t lo = (static_cast<uint32_t>((instr4 & kImm16Mask) << 16) |
276  static_cast<uint32_t>(instr5 & kImm16Mask));
277  return static_cast<Address>((hi << 32) | lo);
278  }
279 
280  UNREACHABLE();
281 }

283 const uint32_t kLoadIntptrOpcode = LD;
284 
285 // Constant pool load sequence detection:
286 // 1) REGULAR access:
287 // load <dst>, kConstantPoolRegister + <offset>
288 //
289 // 2) OVERFLOWED access:
290 // addis <scratch>, kConstantPoolRegister, <offset_high>
291 // load <dst>, <scratch> + <offset_low>
292 bool Assembler::IsConstantPoolLoadStart(Address pc,
293  ConstantPoolEntry::Access* access) {
294  Instr instr = instr_at(pc);
295  uint32_t opcode = instr & kOpcodeMask;
296  if (GetRA(instr) != kConstantPoolRegister) return false;
297  bool overflowed = (opcode == ADDIS);
298 #ifdef DEBUG
299  if (overflowed) {
300  opcode = instr_at(pc + kInstrSize) & kOpcodeMask;
301  }
302  DCHECK(opcode == kLoadIntptrOpcode || opcode == LFD);
303 #endif
304  if (access) {
305  *access = (overflowed ? ConstantPoolEntry::OVERFLOWED
306  : ConstantPoolEntry::REGULAR);
307  }
308  return true;
309 }
310 
311 bool Assembler::IsConstantPoolLoadEnd(Address pc,
312  ConstantPoolEntry::Access* access) {
313  Instr instr = instr_at(pc);
314  uint32_t opcode = instr & kOpcodeMask;
315  bool overflowed = false;
316  if (!(opcode == kLoadIntptrOpcode || opcode == LFD)) return false;
317  if (GetRA(instr) != kConstantPoolRegister) {
318  instr = instr_at(pc - kInstrSize);
319  opcode = instr & kOpcodeMask;
320  if ((opcode != ADDIS) || GetRA(instr) != kConstantPoolRegister) {
321  return false;
322  }
323  overflowed = true;
324  }
325  if (access) {
326  *access = (overflowed ? ConstantPoolEntry::OVERFLOWED
327  : ConstantPoolEntry::REGULAR);
328  }
329  return true;
330 }

332 int Assembler::GetConstantPoolOffset(Address pc,
333  ConstantPoolEntry::Access access,
334  ConstantPoolEntry::Type type) {
335  bool overflowed = (access == ConstantPoolEntry::OVERFLOWED);
336 #ifdef DEBUG
337  ConstantPoolEntry::Access access_check =
338  static_cast<ConstantPoolEntry::Access>(-1);
339  DCHECK(IsConstantPoolLoadStart(pc, &access_check));
340  DCHECK(access_check == access);
341 #endif
342  int offset;
343  if (overflowed) {
344  offset = (instr_at(pc) & kImm16Mask) << 16;
345  offset += SIGN_EXT_IMM16(instr_at(pc + kInstrSize) & kImm16Mask);
346  DCHECK(!is_int16(offset));
347  } else {
348  offset = SIGN_EXT_IMM16((instr_at(pc) & kImm16Mask));
349  }
350  return offset;
351 }

353 void Assembler::PatchConstantPoolAccessInstruction(
354  int pc_offset, int offset, ConstantPoolEntry::Access access,
355  ConstantPoolEntry::Type type) {
356  Address pc = reinterpret_cast<Address>(buffer_start_) + pc_offset;
357  bool overflowed = (access == ConstantPoolEntry::OVERFLOWED);
358  CHECK(overflowed != is_int16(offset));
359 #ifdef DEBUG
360  ConstantPoolEntry::Access access_check =
361  static_cast<ConstantPoolEntry::Access>(-1);
362  DCHECK(IsConstantPoolLoadStart(pc, &access_check));
363  DCHECK(access_check == access);
364 #endif
365  if (overflowed) {
366  int hi_word = static_cast<int>(offset >> 16);
367  int lo_word = static_cast<int>(offset & 0xffff);
368  if (lo_word & 0x8000) hi_word++;
369 
370  Instr instr1 = instr_at(pc);
371  Instr instr2 = instr_at(pc + kInstrSize);
372  instr1 &= ~kImm16Mask;
373  instr1 |= (hi_word & kImm16Mask);
374  instr2 &= ~kImm16Mask;
375  instr2 |= (lo_word & kImm16Mask);
376  instr_at_put(pc, instr1);
377  instr_at_put(pc + kInstrSize, instr2);
378  } else {
379  Instr instr = instr_at(pc);
380  instr &= ~kImm16Mask;
381  instr |= (offset & kImm16Mask);
382  instr_at_put(pc, instr);
383  }
384 }

411 // This code assumes the FIXED_SEQUENCE of lis/ori
412 void Assembler::set_target_address_at(Address pc, Address constant_pool,
413  Address target,
414  WritableJitAllocation* jit_allocation,
415  ICacheFlushMode icache_flush_mode) {
416  if (V8_EMBEDDED_CONSTANT_POOL_BOOL && constant_pool) {
417  ConstantPoolEntry::Access access;
418  if (IsConstantPoolLoadStart(pc, &access)) {
419  if (jit_allocation) {
420  jit_allocation->WriteUnalignedValue<Address>(
421  target_constant_pool_address_at(pc, constant_pool, access,
422  ConstantPoolEntry::INTPTR),
423  target);
424  } else {
425  Memory<Address>(target_constant_pool_address_at(
426  pc, constant_pool, access, ConstantPoolEntry::INTPTR)) = target;
427  }
428  return;
429  }
430  }
431 
432  Instr instr1 = instr_at(pc);
433  Instr instr2 = instr_at(pc + kInstrSize);
434  // Interpret 2 instructions generated by lis/ori
435  if (IsLis(instr1) && IsOri(instr2)) {
436  Instr instr4 = instr_at(pc + (3 * kInstrSize));
437  Instr instr5 = instr_at(pc + (4 * kInstrSize));
438  // Needs to be fixed up when mov changes to handle 64-bit values.
439  uint32_t* p = reinterpret_cast<uint32_t*>(pc);
440  uintptr_t itarget = static_cast<uintptr_t>(target);
441 
442  instr5 &= ~kImm16Mask;
443  instr5 |= itarget & kImm16Mask;
444  itarget = itarget >> 16;
445 
446  instr4 &= ~kImm16Mask;
447  instr4 |= itarget & kImm16Mask;
448  itarget = itarget >> 16;
449 
450  instr2 &= ~kImm16Mask;
451  instr2 |= itarget & kImm16Mask;
452  itarget = itarget >> 16;
453 
454  instr1 &= ~kImm16Mask;
455  instr1 |= itarget & kImm16Mask;
456  itarget = itarget >> 16;
457 
458  if (jit_allocation) {
459  jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(&p[0]),
460  instr1);
461  jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(&p[1]),
462  instr2);
463  jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(&p[3]),
464  instr4);
465  jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(&p[4]),
466  instr5);
467  } else {
468  *p = instr1;
469  *(p + 1) = instr2;
470  *(p + 3) = instr4;
471  *(p + 4) = instr5;
472  }
473  if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
474  FlushInstructionCache(p, 5 * kInstrSize);
475  }
476  return;
477  }
478  UNREACHABLE();
479 }
