41 void Deoptimizer::PatchToJump(Address pc, Address new_pc) {
42  if (Assembler::IsJmpRel(pc)) {
43  // The place holder has been patched already.
44  // TODO(ahaas): Check that the offset in the jump in `pc` matches the offset
45  // of a jump to `new_pc`.
46  return;
47  }
48 
49  RwxMemoryWriteScope rwx_write_scope("Patch jump to deopt trampoline");
50  intptr_t displacement =
51  new_pc - (pc + MacroAssembler::kIntraSegmentJmpInstrSize);
52  CHECK(is_int32(displacement));
53  // We'll overwrite only one instruction of 5-bytes. Give enough
54  // space not to try to grow the buffer.
55  constexpr int kSize = 32;
56  Assembler masm(
57  AssemblerOptions{},
58  ExternalAssemblerBuffer(reinterpret_cast<uint8_t*>(pc), kSize));
59  int offset = static_cast<int>(new_pc - pc);
60  masm.jmp_rel(offset);
61  FlushInstructionCache(pc, kSize);
62 }
