17 // ----------------------------------------------------------------------------
18 // BitField is a help template for encoding and decode bitfield with
19 // unsigned content.
20 // Instantiate them via 'using', which is cheaper than deriving a new class:
21 // using MyBitField = base::BitField<MyEnum, 4, 2>;
22 // The BitField class is final to enforce this style over derivation.
23 
24 template <class T, int shift, int size, class U = uint32_t>
25 class BitField final {
26  public:
27  static_assert(std::is_unsigned_v<U>);
28  static_assert(shift < 8 * sizeof(U)); // Otherwise shifts by {shift} are UB.
29  static_assert(size < 8 * sizeof(U)); // Otherwise shifts by {size} are UB.
30  static_assert(shift + size <= 8 * sizeof(U));
31  static_assert(size > 0);
32 
33  using FieldType = T;
34  using BaseType = U;
35 
36  // A type U mask of bit field. To use all bits of a type U of x bits
37  // in a bitfield without compiler warnings we have to compute 2^x
38  // without using a shift count of x in the computation.
39  static constexpr int kShift = shift;
40  static constexpr int kSize = size;
41  static constexpr U kMask = ((U{1} << kShift) << kSize) - (U{1} << kShift);
42  static constexpr int kLastUsedBit = kShift + kSize - 1;
43  static constexpr U kNumValues = U{1} << kSize;
44  static constexpr U kMax = kNumValues - 1;
45 
46  template <class T2, int size2>
47  using Next = BitField<T2, kShift + kSize, size2, U>;
48 
49  // Tells whether the provided value fits into the bit field.
50  static constexpr bool is_valid(T value) {
51  return (static_cast<U>(value) & ~kMax) == 0;
52  }
53 
54  // Returns a type U with the bit field value encoded.
55  static constexpr U encode(T value) {
56  DCHECK(is_valid(value));
57  return static_cast<U>(value) << kShift;
58  }
59 
60  // Returns a type U with the bit field value updated.
61  V8_NODISCARD static constexpr U update(U previous, T value) {
62  return (previous & ~kMask) | encode(value);
63  }
64 
65  // Extracts the bit field from the value.
66  static constexpr T decode(U value) {
67  return static_cast<T>((value & kMask) >> kShift);
68  }
69 };

71 // ----------------------------------------------------------------------------
72 // BitFieldUnion can be used to combine two linear BitFields.
73 // So far only the static mask is computed. Encoding and decoding tbd.
74 // Can be used for example as a quick combined check:
75 // `if (BitFieldUnion<BFA, BFB>::kMask & bitfield) ...`
76 
77 template <typename A, typename B>
78 class BitFieldUnion final {
79  public:
80  static_assert(std::is_same_v<typename A::BaseType, typename B::BaseType>);
81  static_assert((A::kMask & B::kMask) == 0);
82  static constexpr int kShift = std::min(A::kShift, B::kShift);
83  static constexpr int kMask = A::kMask | B::kMask;
84  static constexpr int kSize =
85  A::kSize + B::kSize + (std::max(A::kShift, B::kShift) - kShift);
86 };

133 // ----------------------------------------------------------------------------
134 // BitSetComputer is a help template for encoding and decoding information for
135 // a variable number of items in an array.
136 //
137 // To encode boolean data in a smi array you would use:
138 // using BoolComputer = BitSetComputer<bool, 1, kSmiValueSize, uint32_t>;
139 //
140 template <class T, int kBitsPerItem, int kBitsPerWord, class U>
141 class BitSetComputer {
142  public:
143  static const int kItemsPerWord = kBitsPerWord / kBitsPerItem;
144  static const int kMask = (1 << kBitsPerItem) - 1;
145 
146  // The number of array elements required to embed T information for each item.
147  static int word_count(int items) {
148  if (items == 0) return 0;
149  return (items - 1) / kItemsPerWord + 1;
150  }
151 
152  // The array index to look at for item.
153  static int index(int base_index, int item) {
154  return base_index + item / kItemsPerWord;
155  }
156 
157  // Extract T data for a given item from data.
158  static T decode(U data, int item) {
159  return static_cast<T>((data >> shift(item)) & kMask);
160  }
161 
162  // Return the encoding for a store of value for item in previous.
163  static U encode(U previous, int item, T value) {
164  int shift_value = shift(item);
165  int set_bits = (static_cast<int>(value) << shift_value);
166  return (previous & ~(kMask << shift_value)) | set_bits;
167  }
168 
169  static int shift(int item) { return (item % kItemsPerWord) * kBitsPerItem; }
170 };
