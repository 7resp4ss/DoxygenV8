225 void HandleScope::CloseScope(Isolate* isolate, Address* prev_next,
226  Address* prev_limit) {
227 #ifdef DEBUG
228  int before = v8_flags.check_handle_count ? NumberOfHandles(isolate) : 0;
229 #endif
230  DCHECK_NOT_NULL(isolate);
231  HandleScopeData* current = isolate->handle_scope_data();
232 
233  std::swap(current->next, prev_next);
234  current->level--;
235  Address* limit = prev_next;
236  if (V8_UNLIKELY(current->limit != prev_limit)) {
237  current->limit = prev_limit;
238  limit = prev_limit;
239  DeleteExtensions(isolate);
240  }
241 #ifdef ENABLE_LOCAL_HANDLE_ZAPPING
242  ZapRange(current->next, limit);
243 #endif
244  MSAN_ALLOCATED_UNINITIALIZED_MEMORY(
245  current->next,
246  static_cast<size_t>(reinterpret_cast<Address>(limit) -
247  reinterpret_cast<Address>(current->next)));
248 #ifdef DEBUG
249  int after = v8_flags.check_handle_count ? NumberOfHandles(isolate) : 0;
250  DCHECK_LT(after - before, kCheckHandleThreshold);
251  DCHECK_LT(before, kCheckHandleThreshold);
252 #endif
253 }

255 template <typename T, template <typename> typename HandleType>
256  requires(std::is_convertible_v<HandleType<T>, DirectHandle<T>>)
257 HandleType<T> HandleScope::CloseAndEscape(HandleType<T> handle_value) {
258  HandleScopeData* current = isolate_->handle_scope_data();
259  Tagged<T> value = *handle_value;
260 #ifdef V8_ENABLE_CHECKS
261  CHECK_EQ(scope_level_, isolate_->handle_scope_data()->level);
262 #endif
263  // Throw away all handles in the current scope.
264  CloseScope(isolate_, prev_next_, prev_limit_);
265  // Allocate one handle in the parent scope.
266  DCHECK(current->level > current->sealed_level);
267  HandleType<T> result(value, isolate_);
268  // Reinitialize the current scope (so that it's ready
269  // to be used or closed again).
270  prev_next_ = current->next;
271  prev_limit_ = current->limit;
272  current->level++;
273  return result;
274 }

276 Address* HandleScope::CreateHandle(Isolate* isolate, Address value) {
277  DCHECK(AllowHandleAllocation::IsAllowed());
278  DCHECK_EQ(isolate, Isolate::TryGetCurrent());
279 #ifdef DEBUG
280  if (!AllowHandleUsageOnAllThreads::IsAllowed()) {
281  DCHECK(isolate->main_thread_local_heap()->IsRunning());
282  DCHECK_WITH_MSG(
283  isolate->thread_id() == ThreadId::Current(),
284  "main-thread handle can only be created on the main thread.");
285  }
286  // We should only allocate handles for objects that can be referenced from the
287  // isolate's heap.
288 #ifdef ENABLE_SLOW_DCHECKS
289  if (!HAS_SMI_TAG(value)) {
290  DCHECK(HAS_STRONG_HEAP_OBJECT_TAG(value));
291  Tagged<HeapObject> obj = UncheckedCast<HeapObject>(Tagged<Object>{value});
292  SLOW_DCHECK(isolate->heap()->CanReferenceHeapObject(obj));
293  }
294 #endif // ENABLE_SLOW_DCHECKS
295 #endif // DEBUG
296  HandleScopeData* data = isolate->handle_scope_data();
297  Address* result = data->next;
298  if (V8_UNLIKELY(result == data->limit)) {
299  result = Extend(isolate);
300  }
301  // Update the current next field, set the value in the created handle,
302  // and return the result.
303  DCHECK_LT(reinterpret_cast<Address>(result),
304  reinterpret_cast<Address>(data->limit));
305  data->next = reinterpret_cast<Address*>(reinterpret_cast<Address>(result) +
306  sizeof(Address));
307  *result = value;
308  return result;
309 }

312 inline SealHandleScope::SealHandleScope(Isolate* isolate) : isolate_(isolate) {
313  // Make sure the current thread is allowed to create handles to begin with.
314  DCHECK(AllowHandleAllocation::IsAllowed());
315  HandleScopeData* current = isolate_->handle_scope_data();
316  // Shrink the current handle scope to make it impossible to do
317  // handle allocations without an explicit handle scope.

321  current->sealed_level = current->level;
322 }
