22 void JSDispatchEntry::MakeJSDispatchEntry(Address object, Address entrypoint,
23  uint16_t parameter_count,
24  bool mark_as_alive) {
25  DCHECK_EQ(object & kHeapObjectTag, 0);
26  DCHECK_EQ((((object - kObjectPointerOffset) << kObjectPointerShift) >>
27  kObjectPointerShift) +
28  kObjectPointerOffset,
29  object);
30  DCHECK_EQ((object - kObjectPointerOffset) + kObjectPointerOffset, object);
31  DCHECK_LT((object - kObjectPointerOffset),
32  1ULL << ((sizeof(encoded_word_) * 8) - kObjectPointerShift));
33 
34  Address payload = ((object - kObjectPointerOffset) << kObjectPointerShift) |
35  (parameter_count & kParameterCountMask);
36  DCHECK(!(payload & kMarkingBit));
37  if (mark_as_alive) payload |= kMarkingBit;
38 #ifdef V8_TARGET_ARCH_32_BIT
39  parameter_count_.store(parameter_count, std::memory_order_relaxed);
40  next_free_entry_.store(0, std::memory_order_relaxed);
41 #endif
42  encoded_word_.store(payload, std::memory_order_relaxed);
43  entrypoint_.store(entrypoint, std::memory_order_relaxed);
44  DCHECK(!IsFreelistEntry());
45 }

52 Address JSDispatchEntry::GetCodePointer() const {
53  CHECK(!IsFreelistEntry());
54  // The pointer tag bit (LSB) of the object pointer is used as marking bit,
55  // and so may be 0 or 1 here. As the return value is a tagged pointer, the
56  // bit must be 1 when returned, so we need to set it here.
57  Address payload = encoded_word_.load(std::memory_order_relaxed);
58  return ((payload >> kObjectPointerShift) + kObjectPointerOffset) |
59  kHeapObjectTag;
60 }

66 uint16_t JSDispatchEntry::GetParameterCount() const {
67  // Loading a pointer out of a freed entry will always result in an invalid
68  // pointer (e.g. upper bits set or nullptr). However, here we're just loading
69  // an integer (the parameter count), so we probably want to make sure that
70  // we're not getting that from a freed entry.
71  CHECK(!IsFreelistEntry());
72 #ifdef V8_TARGET_ARCH_32_BIT
73  return parameter_count_.load(std::memory_order_relaxed);
74 #else
75  static_assert(kParameterCountMask != 0);
76  Address payload = encoded_word_.load(std::memory_order_relaxed);
77  return payload & kParameterCountMask;
78 #endif
79 }

163 std::optional<JSDispatchHandle> JSDispatchTable::TryAllocateAndInitializeEntry(
164  Space* space, uint16_t parameter_count, Tagged<Code> new_code) {
165  DCHECK(space->BelongsTo(this));
166  SBXCHECK(IsCompatibleCode(new_code, parameter_count));
167 
168  uint32_t index;

177  parameter_count, space->allocate_black());

181 void JSDispatchEntry::SetCodeAndEntrypointPointer(Address new_object,
182  Address new_entrypoint) {
183  Address old_payload = encoded_word_.load(std::memory_order_relaxed);
184  Address marking_bit = old_payload & kMarkingBit;
185  Address parameter_count = old_payload & kParameterCountMask;
186  // We want to preserve the marking bit of the entry. Since that happens to
187  // be the tag bit of the pointer, we need to explicitly clear it here.
188  Address object =
189  ((new_object - kObjectPointerOffset) << kObjectPointerShift) &
190  ~kMarkingBit;
191  Address new_payload = object | marking_bit | parameter_count;
192  encoded_word_.store(new_payload, std::memory_order_relaxed);
193  entrypoint_.store(new_entrypoint, std::memory_order_relaxed);
194  DCHECK(!IsFreelistEntry());
195 }

201 void JSDispatchEntry::MakeFreelistEntry(uint32_t next_entry_index) {
202 #ifdef V8_TARGET_ARCH_64_BIT
203  Address payload = kFreeEntryTag | next_entry_index;
204  entrypoint_.store(payload, std::memory_order_relaxed);
205 #else
206  // Store index + 1 such that we can use 0 for non-free entries.
207  next_free_entry_.store(next_entry_index + 1, std::memory_order_relaxed);
208  entrypoint_.store(kNullAddress, std::memory_order_relaxed);
209 #endif
210  encoded_word_.store(kNullAddress, std::memory_order_relaxed);
211  DCHECK(IsFreelistEntry());
212 }

232 void JSDispatchEntry::Mark() {
233  Address old_value = encoded_word_.load(std::memory_order_relaxed);
234  Address new_value = old_value | kMarkingBit;
235  // We don't need this cas to succeed. If marking races with
236  // `SetCodeAndEntrypointPointer`, then we are bound to re-set the mark bit in
237  // the write barrier.
238  static_assert(JSDispatchTable::kWriteBarrierSetsEntryMarkBit);
239  encoded_word_.compare_exchange_strong(old_value, new_value,
240  std::memory_order_relaxed);
241 }

324 // static
325 bool JSDispatchTable::IsCompatibleCode(Tagged<Code> code,
326  uint16_t parameter_count) {
327  if (code->entrypoint_tag() != kJSEntrypointTag) {
328  // Target code doesn't use JS linkage. This cannot be valid.
329  return false;
330  }
331  if (code->parameter_count() == parameter_count) {
332  DCHECK_IMPLIES(code->is_builtin(),
333  parameter_count ==
334  Builtins::GetFormalParameterCount(code->builtin_id()));
335  // Dispatch entry and code have the same signature. This is correct.
336  return true;
337  }
338 
339  // Signature mismatch. This is mostly not safe, except for certain varargs
340  // builtins which are able to correctly handle such a mismatch. Examples
341  // include builtins like the InterpreterEntryTrampoline or the JSToWasm and
342  // JSToJS wrappers which determine their actual parameter count at runtime
343  // (see CodeStubAssembler::SetSupportsDynamicParameterCount()), or internal
344  // builtins that end up tailcalling into other code such as CompileLazy.
345  //
346  // Currently, we also allow this for testing code (from our test suites).
347  // TODO(saelo): maybe we should also forbid this just to be sure.
348  if (code->kind() == CodeKind::FOR_TESTING) {
349  return true;
350  }
351  DCHECK(code->is_builtin());
352  DCHECK_EQ(code->parameter_count(), kDontAdaptArgumentsSentinel);
353  switch (code->builtin_id()) {
354  case Builtin::kIllegal:
355  case Builtin::kCompileLazy:
356  case Builtin::kInterpreterEntryTrampoline:
357  case Builtin::kInstantiateAsmJs:
358  case Builtin::kDebugBreakTrampoline:
359 #ifdef V8_ENABLE_WEBASSEMBLY
360  case Builtin::kJSToWasmWrapper:
361  case Builtin::kJSToJSWrapper:
362  case Builtin::kJSToJSWrapperInvalidSig:
363  case Builtin::kWasmPromising:
364 #if V8_ENABLE_DRUMBRAKE
365  case Builtin::kGenericJSToWasmInterpreterWrapper:
366 #endif
367  case Builtin::kWasmStressSwitch:
368 #endif
369  return true;
370  default:
371  return false;
372  }
373 }
