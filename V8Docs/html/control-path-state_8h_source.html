21 enum NodeUniqueness { kUniqueInstance, kMultipleInstances };

23 // Class for tracking information about path state. It is represented as a
24 // linked list of {NodeState} blocks, each of which corresponds to a block of
25 // code bewteen an IfTrue/IfFalse and a Merge. Each block is in turn represented
26 // as a linked list of {NodeState}s.
27 // If {node_uniqueness} is {kMultipleInstances}, different states can be
28 // assigned to the same node. The most recent state always takes precedence.
29 // States still belong to a block and will be removed if the block gets merged.
30 template <typename NodeState, NodeUniqueness node_uniqueness>
31 class ControlPathState {
32  public:
33  static_assert(std::is_member_function_pointer_v<decltype(&NodeState::IsSet)>,
34  "{NodeState} needs an {IsSet} method");
35  static_assert(
36  std::is_member_object_pointer_v<decltype(&NodeState::node)>,
37  "{NodeState} needs to hold a pointer to the {Node*} owner of the state");
38 
39  explicit ControlPathState(Zone* zone) : states_(zone) {}
40 
41  // Returns the {NodeState} assigned to node, or the default value
42  // {NodeState()} if it is not assigned.
43  NodeState LookupState(Node* node) const;
44 
45  // Adds a state in the current code block, or a new block if the block list is
46  // empty.
47  void AddState(Zone* zone, Node* node, NodeState state, ControlPathState hint);
48 
49  // Adds a state in a new block.
50  void AddStateInNewBlock(Zone* zone, Node* node, NodeState state);
51 
52  // Reset this {ControlPathState} to its longest prefix that is common with
53  // {other}.
54  void ResetToCommonAncestor(ControlPathState other);
55 
56  bool IsEmpty() { return blocks_.Size() == 0; }
57 
58  bool operator==(const ControlPathState& other) const {
59  return blocks_ == other.blocks_;
60  }
61  bool operator!=(const ControlPathState& other) const {
62  return blocks_ != other.blocks_;
63  }
64 
65  private:
66  using NodeWithPathDepth = std::pair<Node*, size_t>;
67 
68  size_t depth(size_t depth_if_multiple_instances) {
69  return node_uniqueness == kMultipleInstances ? depth_if_multiple_instances
70  : 0;
71  }
72 
73 #if DEBUG
74  bool BlocksAndStatesInvariant();
75 #endif
76 
77  FunctionalList<FunctionalList<NodeState>> blocks_;
78  // This is an auxilliary data structure that provides fast lookups in the
79  // set of states. It should hold at any point that the contents of {blocks_}
80  // and {states_} is the same, which is implemented in
81  // {BlocksAndStatesInvariant}.
82  PersistentMap<NodeWithPathDepth, NodeState> states_;
83 };

123 template <typename NodeState, NodeUniqueness node_uniqueness>
124 NodeState ControlPathState<NodeState, node_uniqueness>::LookupState(
125  Node* node) const {
126  if (node_uniqueness == kUniqueInstance) return states_.Get({node, 0});
127  for (size_t depth = blocks_.Size(); depth > 0; depth--) {
128  NodeState state = states_.Get({node, depth});
129  if (state.IsSet()) return state;
130  }
131  return {};
132 }

134 template <typename NodeState, NodeUniqueness node_uniqueness>
135 void ControlPathState<NodeState, node_uniqueness>::AddState(
136  Zone* zone, Node* node, NodeState state,
137  ControlPathState<NodeState, node_uniqueness> hint) {
138  NodeState previous_state = LookupState(node);
139  if (node_uniqueness == kUniqueInstance ? previous_state.IsSet()
140  : previous_state == state) {
141  return;
142  }
143 
144  FunctionalList<NodeState> prev_front = blocks_.Front();
145  if (hint.blocks_.Size() > 0) {
146  prev_front.PushFront(state, zone, hint.blocks_.Front());
147  } else {
148  prev_front.PushFront(state, zone);
149  }
150  blocks_.DropFront();
151  blocks_.PushFront(prev_front, zone);
152  states_.Set({node, depth(blocks_.Size())}, state);
153  SLOW_DCHECK(BlocksAndStatesInvariant());
154 }
155 
156 template <typename NodeState, NodeUniqueness node_uniqueness>
157 void ControlPathState<NodeState, node_uniqueness>::AddStateInNewBlock(
158  Zone* zone, Node* node, NodeState state) {
159  FunctionalList<NodeState> new_block;
160  NodeState previous_state = LookupState(node);
161  if (node_uniqueness == kUniqueInstance ? !previous_state.IsSet()
162  : previous_state != state) {
163  new_block.PushFront(state, zone);
164  states_.Set({node, depth(blocks_.Size() + 1)}, state);
165  }
166  blocks_.PushFront(new_block, zone);
167  SLOW_DCHECK(BlocksAndStatesInvariant());
168 }

170 template <typename NodeState, NodeUniqueness node_uniqueness>
171 void ControlPathState<NodeState, node_uniqueness>::ResetToCommonAncestor(
172  ControlPathState<NodeState, node_uniqueness> other) {
173  while (other.blocks_.Size() > blocks_.Size()) other.blocks_.DropFront();
174  while (blocks_.Size() > other.blocks_.Size()) {
175  for (NodeState state : blocks_.Front()) {
176  states_.Set({state.node, depth(blocks_.Size())}, {});
177  }
178  blocks_.DropFront();
179  }
180  while (blocks_ != other.blocks_) {
181  for (NodeState state : blocks_.Front()) {
182  states_.Set({state.node, depth(blocks_.Size())}, {});
183  }
184  blocks_.DropFront();
185  other.blocks_.DropFront();
186  }
187  SLOW_DCHECK(BlocksAndStatesInvariant());
188 }

190 #if DEBUG
191 template <typename NodeState, NodeUniqueness node_uniqueness>
192 bool ControlPathState<NodeState, node_uniqueness>::BlocksAndStatesInvariant() {
193  PersistentMap<NodeWithPathDepth, NodeState> states_copy(states_);
194  size_t current_depth = blocks_.Size();
195  for (auto block : blocks_) {
196  std::unordered_set<Node*> seen_this_block;
197  for (NodeState state : block) {
198  // Every element of blocks_ has to be in states_.
199  if (seen_this_block.count(state.node) == 0) {
200  if (states_copy.Get({state.node, depth(current_depth)}) != state) {
201  return false;
202  }
203  states_copy.Set({state.node, depth(current_depth)}, {});
204  seen_this_block.emplace(state.node);
205  }
206  }
207  current_depth--;
208  }
209  // Every element of {states_} has to be in {blocks_}. We removed all
210  // elements of blocks_ from states_copy, so if it is not empty, the
211  // invariant fails.
212  return states_copy.begin() == states_copy.end();
213 }

216 template <typename NodeState, NodeUniqueness node_uniqueness>
217 Reduction AdvancedReducerWithControlPathState<
218  NodeState, node_uniqueness>::TakeStatesFromFirstControl(Node* node) {
219  // We just propagate the information from the control input (ideally,
220  // we would only revisit control uses if there is change).
221  Node* input = NodeProperties::GetControlInput(node, 0);
222  if (!reduced_.Get(input)) return NoChange();
223  return UpdateStates(node, node_states_.Get(input));
224 }

226 template <typename NodeState, NodeUniqueness node_uniqueness>
227 Reduction
228 AdvancedReducerWithControlPathState<NodeState, node_uniqueness>::UpdateStates(
229  Node* state_owner, ControlPathState<NodeState, node_uniqueness> new_state) {
230  // Only signal that the node has {Changed} if its state has changed.
231  bool reduced_changed = reduced_.Set(state_owner, true);
232  bool node_states_changed = node_states_.Set(state_owner, new_state);
233  if (reduced_changed || node_states_changed) {
234  return Changed(state_owner);
235  }
236  return NoChange();
237 }

239 template <typename NodeState, NodeUniqueness node_uniqueness>
240 Reduction
241 AdvancedReducerWithControlPathState<NodeState, node_uniqueness>::UpdateStates(
242  Node* state_owner, ControlPathState<NodeState, node_uniqueness> prev_states,
243  Node* additional_node, NodeState additional_state, bool in_new_block) {
244  if (in_new_block || prev_states.IsEmpty()) {
245  prev_states.AddStateInNewBlock(zone_, additional_node, additional_state);
246  } else {
247  ControlPathState<NodeState, node_uniqueness> original =
248  node_states_.Get(state_owner);
249  prev_states.AddState(zone_, additional_node, additional_state, original);
250  }
251  return UpdateStates(state_owner, prev_states);
252 }
