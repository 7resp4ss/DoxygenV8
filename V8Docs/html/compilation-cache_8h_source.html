18 // The compilation cache consists of several sub-caches: one each for evals and
19 // scripts, which use this class as a base class, and a separate generational
20 // sub-cache for RegExps. Since the same source code string has different
21 // compiled code for scripts and evals, we use separate sub-caches for different
22 // compilation modes, to avoid retrieving the wrong result.

67 // Sub-cache for eval scripts. Two caches for eval are used. One for eval calls
68 // in native contexts and one for eval calls in other contexts. The cache
69 // considers the following pieces of information when checking for matching
70 // entries:
71 // 1. The source string.
72 // 2. The shared function info of the calling function.
73 // 3. Whether the source should be compiled as strict code or as sloppy code.
74 // Note: Currently there are clients of CompileEval that always compile
75 // sloppy code even if the calling function is a strict mode function.
76 // More specifically these are the CompileString, DebugEvaluate and
77 // DebugEvaluateGlobal runtime functions.
78 // 4. The start position of the calling scope.
79 class CompilationCacheEval : public CompilationCacheEvalOrScript {
80  public:
81  explicit CompilationCacheEval(Isolate* isolate)
82  : CompilationCacheEvalOrScript(isolate) {}
83

85  DirectHandle<SharedFunctionInfo> outer_info,
86  DirectHandle<NativeContext> native_context,
87  LanguageMode language_mode, int position);
88 
89  void UpdateEval(DirectHandle<String> source,
90  DirectHandle<SharedFunctionInfo> outer_info,
91  DirectHandle<JSFunction> js_function,
92  LanguageMode language_mode, int position);
93 
94  void Put(DirectHandle<String> source,
95  DirectHandle<SharedFunctionInfo> outer_info,
96  DirectHandle<JSFunction> js_function, int position);

141 // The compilation cache keeps shared function infos for compiled
142 // scripts and evals. The shared function infos are looked up using
143 // the source string as the key. For regular expressions the
144 // compilation data is cached.
145 class V8_EXPORT_PRIVATE CompilationCache {
146  public:
147  CompilationCache(const CompilationCache&) = delete;
148  CompilationCache& operator=(const CompilationCache&) = delete;
149 
150  // Finds the Script and root SharedFunctionInfo for a script source string.
151  // Returns empty handles if the cache doesn't contain a script for the given
152  // source string with the right origin.
153  CompilationCacheScript::LookupResult LookupScript(
154  Handle<String> source, const ScriptDetails& script_details,
155  LanguageMode language_mode);
156 
157  // Finds the shared function info for a source string for eval in a
158  // given context. Returns an empty handle if the cache doesn't
159  // contain a script for the given source string.
160  InfoCellPair LookupEval(DirectHandle<String> source,
161  DirectHandle<SharedFunctionInfo> outer_info,
162  DirectHandle<Context> context,
163  LanguageMode language_mode, int position);
164 
165  void UpdateEval(DirectHandle<String> source,
166  DirectHandle<SharedFunctionInfo> outer_info,
167  DirectHandle<JSFunction> js_function,
168  LanguageMode language_mode, int position);
169 
170  // Returns the regexp data associated with the given regexp if it
171  // is in cache, otherwise an empty handle.
172  MaybeDirectHandle<RegExpData> LookupRegExp(DirectHandle<String> source,
173  JSRegExp::Flags flags);
174 
175  // Associate the (source, kind) pair to the shared function
176  // info. This may overwrite an existing mapping.
177  void PutScript(Handle<String> source, LanguageMode language_mode,
178  DirectHandle<SharedFunctionInfo> function_info);
179 
180  // Associate the (source, context->closure()->shared(), kind) triple
181  // with the shared function info. This may overwrite an existing mapping.
182  void PutEval(DirectHandle<String> source,
183  DirectHandle<SharedFunctionInfo> outer_info,
184  DirectHandle<JSFunction> js_function, int position);
185 
186  // Associate the (source, flags) pair to the given regexp data.
187  // This may overwrite an existing mapping.
188  void PutRegExp(DirectHandle<String> source, JSRegExp::Flags flags,
189  DirectHandle<RegExpData> data);
190 
191  // Clear the cache - also used to initialize the cache at startup.
192  void Clear();
193 
194  // Remove given shared function info from all caches.
195  void Remove(DirectHandle<SharedFunctionInfo> function_info);
196 
197  // GC support.
198  void Iterate(RootVisitor* v);
199 
200  // Notify the cache that a mark-sweep garbage collection is about to
201  // take place. This is used to retire entries from the cache to
202  // avoid keeping them alive too long without using them.
203  void MarkCompactPrologue();
204 
205  // Enable/disable compilation cache. Used by debugger to disable compilation
206  // cache during debugging so that eval and new scripts are always compiled.
207  // TODO(bmeurer, chromium:992277): The RegExp cache cannot be enabled and/or
208  // disabled, since it doesn't affect debugging. However ideally the other
209  // caches should also be always on, even in the presence of the debugger,
210  // but at this point there are too many unclear invariants, and so I decided
211  // to just fix the pressing performance problem for RegExp individually first.
212  void EnableScriptAndEval();
213  void DisableScriptAndEval();
214 
215  private:
216  explicit CompilationCache(Isolate* isolate);
217  ~CompilationCache() = default;
218 
219  base::HashMap* EagerOptimizingSet();
220 
221  bool IsEnabledScriptAndEval() const {
222  return v8_flags.compilation_cache && enabled_script_and_eval_;
223  }
224  bool IsEnabledScript(LanguageMode language_mode) {
225  // Tests can change v8_flags.use_strict at runtime. The compilation cache
226  // only contains scripts which were compiled with the default language mode.
227  return IsEnabledScriptAndEval() && language_mode == LanguageMode::kSloppy;
228  }
