150 class CompileTimeImports {
151  public:
152  CompileTimeImports() = default;
153 
154  CompileTimeImports(const CompileTimeImports& other) V8_NOEXCEPT = default;
155  CompileTimeImports& operator=(const CompileTimeImports& other)
156  V8_NOEXCEPT = default;
157  CompileTimeImports(CompileTimeImports&& other) V8_NOEXCEPT {
158  *this = std::move(other);
159  }
160  CompileTimeImports& operator=(CompileTimeImports&& other) V8_NOEXCEPT {
161  bits_ = other.bits_;
162  constants_module_ = std::move(other.constants_module_);
163  return *this;
164  }
165  static CompileTimeImports FromSerialized(
166  CompileTimeImportFlags::StorageType flags,
167  base::Vector<const char> constants_module) {
168  CompileTimeImports result;
169  result.bits_ = CompileTimeImportFlags::FromIntegral(flags);
170  result.constants_module_.assign(constants_module.begin(),
171  constants_module.end());
172  return result;
173  }
174 
175  bool empty() const { return bits_.empty(); }
176  bool has_string_constants(base::Vector<const uint8_t> name) const {
177  return bits_.contains(CompileTimeImport::kStringConstants) &&
178  constants_module_.size() == name.size() &&
179  std::equal(name.begin(), name.end(), constants_module_.begin());
180  }
181  bool contains(CompileTimeImport imp) const { return bits_.contains(imp); }
182 
183  int compare(const CompileTimeImports& other) const {
184  if (bits_.ToIntegral() < other.bits_.ToIntegral()) return -1;
185  if (bits_.ToIntegral() > other.bits_.ToIntegral()) return 1;
186  return constants_module_.compare(other.constants_module_);
187  }
188 
189  void Add(CompileTimeImport imp) { bits_.Add(imp); }
190 
191  std::string& constants_module() { return constants_module_; }
192  const std::string& constants_module() const { return constants_module_; }
193 
194  CompileTimeImportFlags flags() const { return bits_; }
195 
196  private:
197  CompileTimeImportFlags bits_;
198  std::string constants_module_;
199 };
