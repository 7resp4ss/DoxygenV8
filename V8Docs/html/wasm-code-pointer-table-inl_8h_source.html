103 WasmCodePointerTable::FreelistHead WasmCodePointerTable::ReadFreelistHead() {
104  while (true) {
105  FreelistHead freelist = freelist_head_.load(std::memory_order_acquire);
106  if (IsRetryMarker(freelist)) {
107  // The retry marker will only be stored for a short amount of time. We can
108  // check for it in a busy loop.
109  continue;
110  }
111  return freelist;
112  }
113 }

115 WasmCodePointer WasmCodePointerTable::AllocateUninitializedEntry() {
116  DCHECK(is_initialized());
117 
118  while (true) {
119  // Fast path, try to take an entry from the freelist.
120  uint32_t allocated_entry;
121  if (TryAllocateFromFreelist(&allocated_entry)) {
122  return WasmCodePointer{allocated_entry};
123  }
124 
125  // This is essentially DCLP (see
126  // https://preshing.com/20130930/double-checked-locking-is-fixed-in-cpp11/)
127  // and so requires an acquire load as well as a release store in
128  // AllocateTableSegment() to prevent reordering of memory accesses, which
129  // could for example cause one thread to read a freelist entry before it
130  // has been properly initialized.
131 
132  // The freelist is empty. We take a lock to avoid another thread from
133  // allocating a new segment in the meantime. However, the freelist can
134  // still grow if another thread frees an entry, so we'll merge the
135  // freelists atomically in the end.
136  base::MutexGuard guard(&segment_allocation_mutex_);
137 
138  // Reload freelist head in case another thread already grew the table.
139  if (!freelist_head_.load(std::memory_order_relaxed).is_empty()) {
140  // Something changed, retry.
141  continue;
142  }
143 
144  // Freelist is (still) empty so extend this space by another segment.
145  auto [segment, freelist] = AllocateAndInitializeSegment();
146 
147  // Take out the first entry before we link it to the freelist_head.
148  allocated_entry = AllocateEntryFromFreelistNonAtomic(&freelist);
149 
150  // Merge the new freelist entries into our freelist.
151  LinkFreelist(freelist, segment.last_entry());
152 
153  return WasmCodePointer{allocated_entry};
154  }
155 }

157 bool WasmCodePointerTable::TryAllocateFromFreelist(uint32_t* index) {
158  while (true) {
159  FreelistHead current_freelist_head = ReadFreelistHead();
160  if (current_freelist_head.is_empty()) {
161  return false;
162  }
163 
164  // Temporarily replace the freelist head with a marker to gain exclusive
165  // access to it. This avoids a race condition where another thread could
166  // unmap the memory while we're trying to read from it.
167  if (!freelist_head_.compare_exchange_strong(current_freelist_head,
168  kRetryMarker)) {
169  continue;
170  }
171 
172  uint32_t next_freelist_entry =
173  at(current_freelist_head.next()).GetNextFreelistEntryIndex();
174  FreelistHead new_freelist_head(next_freelist_entry,
175  current_freelist_head.length() - 1);
176 
177  // We are allowed to overwrite the freelist_head_ since we stored the
178  // kRetryMarker in there.
179  freelist_head_.store(new_freelist_head, std::memory_order_relaxed);
180 
181  *index = current_freelist_head.next();
182 
183  return true;
184  }
185 }

187 uint32_t WasmCodePointerTable::AllocateEntryFromFreelistNonAtomic(
188  FreelistHead* freelist_head) {
189  DCHECK(!freelist_head->is_empty());
190  uint32_t index = freelist_head->next();
191  uint32_t next_next = at(freelist_head->next()).GetNextFreelistEntryIndex();
192  *freelist_head = FreelistHead(next_next, freelist_head->length() - 1);
193  return index;
194 }

203 WasmCodePointerTable::FreelistHead WasmCodePointerTable::LinkFreelist(
204  FreelistHead freelist_to_link, uint32_t last_element) {
205  DCHECK(!freelist_to_link.is_empty());
206 
207  FreelistHead current_head, new_head;
208  do {
209  current_head = ReadFreelistHead();
210  new_head = FreelistHead(freelist_to_link.next(),
211  freelist_to_link.length() + current_head.length());
212 
213  WriteScope write_scope("write free list entry");
214  at(last_element).MakeFreelistEntry(current_head.next());
215  // This must be a release store since we previously wrote the freelist
216  // entries in AllocateTableSegment() and we need to prevent the writes from
217  // being reordered past this store. See AllocateEntry() for more details.
218  } while (!freelist_head_.compare_exchange_strong(current_head, new_head,
219  std::memory_order_release));
220 
221  return new_head;
222 }
