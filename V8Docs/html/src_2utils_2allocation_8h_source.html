42 template <typename T>
43 T* NewArray(size_t size) {
44  T* result = new (std::nothrow) T[size];
45  if (V8_UNLIKELY(result == nullptr)) {
46  OnCriticalMemoryPressure();
47  result = new (std::nothrow) T[size];
48  if (result == nullptr) V8::FatalProcessOutOfMemory(nullptr, "NewArray");
49  }
50  return result;
51 }

305 // Represents a VirtualMemory reservation along with a BoundedPageAllocator that
306 // can be used to allocate within the reservation.
307 //
308 // Virtual memory cages are used for the pointer compression cage, the code
309 // ranges (on platforms that require code ranges), and trusted ranges (when the
310 // sandbox is enabled). They are configurable via ReservationParams.
311 //
312 // +-----------+------------ ~~~ --+- ~~~ -+
313 // | ... | ... | ... |
314 // +-----------+------------ ~~~ --+- ~~~ -+
315 // ^ ^
316 // cage base allocatable base
317 //
318 // <------------------->
319 // allocatable size
320 // <------------------------------->
321 // cage size
322 // <--------------------------------------->
323 // reservation size
324 //
325 // - The reservation is made using ReservationParams::page_allocator.
326 // - cage base is the start of the virtual memory reservation and the base
327 // address of the cage.
328 // - allocatable base is the cage base rounded up to the nearest
329 // ReservationParams::page_size, and is the start of the allocatable area for
330 // the BoundedPageAllocator.
331 // - cage size is the size of the area from cage base to the end of the
332 // allocatable area.
333 //
334 // - The reservation size is configured by ReservationParams::reservation_size
335 // but it might be actually bigger if we end up over-reserving the virtual
336 // address space.
337 //
338 // Additionally,
339 // - The alignment of the cage base is configured by
340 // ReservationParams::base_alignment.
341 // - The page size of the BoundedPageAllocator is configured by
342 // ReservationParams::page_size.
343 // - A hint for the value of start can be passed by
344 // ReservationParams::requested_start_hint and it must be aligned to
345 // ReservationParams::base_alignment.
346 //
347 // The configuration is subject to the following alignment requirements.
348 // Below, AllocatePageSize is short for
349 // ReservationParams::page_allocator->AllocatePageSize().
350 //
351 // - The reservation size must be AllocatePageSize-aligned.
352 // - If the base alignment is not kAnyBaseAlignment then the base alignment
353 // must be AllocatePageSize-aligned.
354 // - The base alignment may be kAnyBaseAlignment to denote any alignment is
355 // acceptable. In this case the base bias size does not need to be aligned.
356 class VirtualMemoryCage {
357  public:
358  VirtualMemoryCage();
359  virtual ~VirtualMemoryCage();
360 
361  VirtualMemoryCage(const VirtualMemoryCage&) = delete;
362  VirtualMemoryCage& operator=(VirtualMemoryCage&) = delete;
363 
364  VirtualMemoryCage(VirtualMemoryCage&& other) V8_NOEXCEPT;
365  VirtualMemoryCage& operator=(VirtualMemoryCage&& other) V8_NOEXCEPT;
366 
367  Address base() const { return base_; }
368  size_t size() const { return size_; }
369 
370  base::AddressRegion region() const {
371  return base::AddressRegion{base_, size_};
372  }
373 
374  base::BoundedPageAllocator* page_allocator() const {
375  return page_allocator_.get();
376  }
377 
378  VirtualMemory* reservation() { return &reservation_; }
379  const VirtualMemory* reservation() const { return &reservation_; }
380 
381  bool IsReserved() const {
382  DCHECK_EQ(base_ != kNullAddress, reservation_.IsReserved());
383  DCHECK_EQ(base_ != kNullAddress, size_ != 0);
384  return reservation_.IsReserved();
385  }

390  // See diagram above.
