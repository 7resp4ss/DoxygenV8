18 // Wrapper class to hold a field index, usually but not necessarily generated
19 // from a property index. When available, the wrapper class captures additional
20 // information to allow the field index to be translated back into the property
21 // index it was originally generated from.
22 class FieldIndex final {
23  public:

42  bool is_inobject() const { return IsInObjectBits::decode(bit_field_); }
43 
44  bool is_double() const { return EncodingBits::decode(bit_field_) == kDouble; }
45 
46  int offset() const { return OffsetBits::decode(bit_field_); }

50  // Zero-indexed from beginning of the object.
51  int index() const {
52  DCHECK(IsAligned(offset(), kTaggedSize));
53  return offset() / kTaggedSize;
54  }
55 
56  int outobject_array_index() const {
57  DCHECK(!is_inobject());
58  return index() - first_inobject_property_offset() / kTaggedSize;
59  }
60 
61  // Zero-based from the first inobject property. Overflows to out-of-object
62  // properties.
63  int property_index() const {
64  int result = index() - first_inobject_property_offset() / kTaggedSize;
65  if (!is_inobject()) {
66  result += InObjectPropertyBits::decode(bit_field_);
67  }
68  return result;
69  }
70 
71  int GetFieldAccessStubKey() const {
72  return bit_field_ &
73  (IsInObjectBits::kMask | EncodingBits::kMask | OffsetBits::kMask);
74  }
75 
76  bool operator==(FieldIndex const& other) const {
77  return bit_field_ == other.bit_field_;
78  }
79  bool operator!=(FieldIndex const& other) const { return !(*this == other); }
80 
81  private:
82  FieldIndex(bool is_inobject, int offset, Encoding encoding,
83  int inobject_properties, int first_inobject_property_offset) {
84  DCHECK(IsAligned(first_inobject_property_offset, kTaggedSize));
85  bit_field_ = IsInObjectBits::encode(is_inobject) |
86  EncodingBits::encode(encoding) |
87  FirstInobjectPropertyOffsetBits::encode(
88  first_inobject_property_offset) |
89  OffsetBits::encode(offset) |
90  InObjectPropertyBits::encode(inobject_properties);
91  }
92 
93  static Encoding FieldEncoding(Representation representation) {
94  switch (representation.kind()) {
95  case Representation::kNone:
96  case Representation::kSmi:
97  case Representation::kHeapObject:
98  case Representation::kTagged:
99  return kTagged;
100  case Representation::kDouble:
101  return kDouble;
102  default:
103  break;
104  }
105  PrintF("%s\n", representation.Mnemonic());
106  UNREACHABLE();
107  return kTagged;
108  }
109 
110  int first_inobject_property_offset() const {
111  return FirstInobjectPropertyOffsetBits::decode(bit_field_);
112  }
113 
114  static const int kOffsetBitsSize =
115  (kDescriptorIndexBitCount + 1 + kTaggedSizeLog2);
116 
117  // Index from beginning of object.
118  using OffsetBits = base::BitField64<int, 0, kOffsetBitsSize>;
119  using IsInObjectBits = OffsetBits::Next<bool, 1>;
120  using EncodingBits = IsInObjectBits::Next<Encoding, 2>;
121  // Number of inobject properties.
122  using InObjectPropertyBits =
123  EncodingBits::Next<int, kDescriptorIndexBitCount>;
124  // Offset of first inobject property from beginning of object.
125  using FirstInobjectPropertyOffsetBits =
126  InObjectPropertyBits::Next<int, kFirstInobjectPropertyOffsetBitCount>;
127  static_assert(FirstInobjectPropertyOffsetBits::kLastUsedBit < 64);
128 
129  uint64_t bit_field_;
130 };
