65 /**
66  * An instance of a Property Descriptor, see Ecma-262 6.2.4.
67  *
68  * Properties in a descriptor are present or absent. If you do not set
69  * `enumerable`, `configurable`, and `writable`, they are absent. If `value`,
70  * `get`, or `set` are absent, but you must specify them in the constructor, use
71  * empty handles.
72  *
73  * Accessors `get` and `set` must be callable or undefined if they are present.
74  *
75  * \note Only query properties if they are present, i.e., call `x()` only if
76  * `has_x()` returns true.
77  *
78  * \code
79  * // var desc = {writable: false}
80  * v8::PropertyDescriptor d(Local<Value>()), false);
81  * d.value(); // error, value not set
82  * if (d.has_writable()) {
83  * d.writable(); // false
84  * }
85  *
86  * // var desc = {value: undefined}
87  * v8::PropertyDescriptor d(v8::Undefined(isolate));
88  *
89  * // var desc = {get: undefined}
90  * v8::PropertyDescriptor d(v8::Undefined(isolate), Local<Value>()));
91  * \endcode
92  */

94  public:
95  // GenericDescriptor
96  PropertyDescriptor();
97 
98  // DataDescriptor
99  explicit PropertyDescriptor(Local<Value> value);
100 
101  // DataDescriptor with writable property
102  PropertyDescriptor(Local<Value> value, bool writable);
103 
104  // AccessorDescriptor
105  PropertyDescriptor(Local<Value> get, Local<Value> set);
106 
107  ~PropertyDescriptor();
108 
109  Local<Value> value() const;
110  bool has_value() const;
111 
112  Local<Value> get() const;
113  bool has_get() const;
114  Local<Value> set() const;
115  bool has_set() const;
116 
117  void set_enumerable(bool enumerable);
118  bool enumerable() const;
119  bool has_enumerable() const;
120

123  bool has_configurable() const;

915 Local<Data> Object::GetInternalField(int index) {
916 #ifndef V8_ENABLE_CHECKS
917  using A = internal::Address;
918  using I = internal::Internals;
919  A obj = internal::ValueHelper::ValueAsAddress(this);
920  // Fast path: If the object is a plain JSObject, which is the common case, we
921  // know where to find the internal fields and can return the value directly.
922  int instance_type = I::GetInstanceType(obj);
923  if (I::CanHaveInternalField(instance_type)) {
924  int offset = I::kJSAPIObjectWithEmbedderSlotsHeaderSize +
925  (I::kEmbedderDataSlotSize * index);
926  A value = I::ReadRawField<A>(obj, offset);
927 #ifdef V8_COMPRESS_POINTERS
928  // We read the full pointer value and then decompress it in order to avoid
929  // dealing with potential endiannes issues.
930  value = I::DecompressTaggedField(obj, static_cast<uint32_t>(value));
931 #endif
932 
933  auto* isolate = I::GetCurrentIsolate();
934  return Local<Data>::New(isolate, value);
935  }
936 #endif
937  return SlowGetInternalField(index);
938 }
