37 bool IsReplacementCharacter(const uint8_t* octets, int length) {
38  // The replacement character is at codepoint U+FFFD in the Unicode Specials
39  // table. Its UTF-8 encoding is 0xEF 0xBF 0xBD.
40  if (length != 3 || octets[0] != 0xEF || octets[1] != 0xBF ||
41  octets[2] != 0xBD) {
42  return false;
43  }
44  return true;
45 }

47 bool DecodeOctets(const uint8_t* octets, int length,
48  std::vector<base::uc16>* buffer) {
49  size_t cursor = 0;
50  base::uc32 value = unibrow::Utf8::ValueOf(octets, length, &cursor);
51  if (value == unibrow::Utf8::kBadChar &&
52  !IsReplacementCharacter(octets, length)) {
53  return false;
54  }
55 
56  if (value <=
57  static_cast<base::uc32>(unibrow::Utf16::kMaxNonSurrogateCharCode)) {
58  buffer->push_back(value);
59  } else {
60  buffer->push_back(unibrow::Utf16::LeadSurrogate(value));
61  buffer->push_back(unibrow::Utf16::TrailSurrogate(value));
62  }
63  return true;
64 }

66 int TwoDigitHex(base::uc16 character1, base::uc16 character2) {
67  if (character1 > 'f') return -1;
68  int high = base::HexValue(character1);
69  if (high == -1) return -1;
70  if (character2 > 'f') return -1;
71  int low = base::HexValue(character2);
72  if (low == -1) return -1;
73  return (high << 4) + low;
74 }

77 void AddToBuffer(base::uc16 decoded, String::FlatContent* uri_content,
78  int index, bool is_uri, std::vector<T>* buffer) {
79  if (is_uri && IsReservedPredicate(decoded)) {
80  buffer->push_back('%');

82  base::uc16 second = uri_content->Get(index + 2);
83  DCHECK_GT(std::numeric_limits<T>::max(), first);
84  DCHECK_GT(std::numeric_limits<T>::max(), second);
85 
86  buffer->push_back(first);
87  buffer->push_back(second);
88  } else {
89  buffer->push_back(decoded);
90  }
91 }

93 bool IntoTwoByte(int index, bool is_uri, int uri_length,
94  String::FlatContent* uri_content,
95  std::vector<base::uc16>* buffer) {
96  for (int k = index; k < uri_length; k++) {
97  base::uc16 code = uri_content->Get(k);
98  if (code == '%') {
99  int two_digits;
100  if (k + 2 >= uri_length ||
101  (two_digits = TwoDigitHex(uri_content->Get(k + 1),
102  uri_content->Get(k + 2))) < 0) {
103  return false;
104  }
105  k += 2;
106  base::uc16 decoded = static_cast<base::uc16>(two_digits);
107  if (decoded > unibrow::Utf8::kMaxOneByteChar) {
108  uint8_t octets[unibrow::Utf8::kMaxEncodedSize];
109  octets[0] = decoded;
110 
111  int number_of_continuation_bytes = 0;
112  while ((decoded << ++number_of_continuation_bytes) & 0x80) {
113  if (number_of_continuation_bytes > 3 || k + 3 >= uri_length) {
114  return false;
115  }
116  if (uri_content->Get(++k) != '%' ||
117  (two_digits = TwoDigitHex(uri_content->Get(k + 1),
118  uri_content->Get(k + 2))) < 0) {
119  return false;
120  }
121  k += 2;
122  base::uc16 continuation_byte = static_cast<base::uc16>(two_digits);
123  octets[number_of_continuation_bytes] = continuation_byte;
124  }
125 
126  if (!DecodeOctets(octets, number_of_continuation_bytes, buffer)) {
127  return false;
128  }
129  } else {
130  AddToBuffer(decoded, uri_content, k - 2, is_uri, buffer);
131  }
132  } else {
133  buffer->push_back(code);
134  }
135  }
136  return true;
137 }

139 bool IntoOneAndTwoByte(DirectHandle<String> uri, bool is_uri,
140  std::vector<uint8_t>* one_byte_buffer,
141  std::vector<base::uc16>* two_byte_buffer) {
142  DisallowGarbageCollection no_gc;
143  String::FlatContent uri_content = uri->GetFlatContent(no_gc);
144 
145  int uri_length = uri->length();
146  for (int k = 0; k < uri_length; k++) {
147  base::uc16 code = uri_content.Get(k);
148  if (code == '%') {
149  int two_digits;
150  if (k + 2 >= uri_length ||
151  (two_digits = TwoDigitHex(uri_content.Get(k + 1),
152  uri_content.Get(k + 2))) < 0) {
153  return false;
154  }
155 
156  base::uc16 decoded = static_cast<base::uc16>(two_digits);
157  if (decoded > unibrow::Utf8::kMaxOneByteChar) {
158  return IntoTwoByte(k, is_uri, uri_length, &uri_content,
159  two_byte_buffer);
160  }
161 
162  AddToBuffer(decoded, &uri_content, k, is_uri, one_byte_buffer);
163  k += 2;
164  } else {
165  if (code > unibrow::Utf8::kMaxOneByteChar) {
166  return IntoTwoByte(k, is_uri, uri_length, &uri_content,
167  two_byte_buffer);
168  }
169  one_byte_buffer->push_back(code);
170  }
171  }
172  return true;
173 }

177 MaybeDirectHandle<String> Uri::Decode(Isolate* isolate,
178  DirectHandle<String> uri, bool is_uri) {
179  uri = String::Flatten(isolate, uri);
180  std::vector<uint8_t> one_byte_buffer;
181  std::vector<base::uc16> two_byte_buffer;
182 
183  if (!IntoOneAndTwoByte(uri, is_uri, &one_byte_buffer, &two_byte_buffer)) {
184  THROW_NEW_ERROR(isolate, NewURIError());
185  }
186 
187  if (two_byte_buffer.empty()) {
188  return isolate->factory()->NewStringFromOneByte(base::Vector<const uint8_t>(
189  one_byte_buffer.data(), static_cast<int>(one_byte_buffer.size())));
190  }
191 
192  DirectHandle<SeqTwoByteString> result;
193  int result_length =
194  static_cast<int>(one_byte_buffer.size() + two_byte_buffer.size());
195  ASSIGN_RETURN_ON_EXCEPTION(
196  isolate, result, isolate->factory()->NewRawTwoByteString(result_length));
197 
198  DisallowGarbageCollection no_gc;
199  base::uc16* chars = result->GetChars(no_gc);
200  if (!one_byte_buffer.empty()) {
201  CopyChars(chars, one_byte_buffer.data(), one_byte_buffer.size());
202  chars += one_byte_buffer.size();
203  }
204  if (!two_byte_buffer.empty()) {
205  CopyChars(chars, two_byte_buffer.data(), two_byte_buffer.size());
206  }
207 
208  return result;
209 }

280 MaybeDirectHandle<String> Uri::Encode(Isolate* isolate,
281  DirectHandle<String> uri, bool is_uri) {
282  uri = String::Flatten(isolate, uri);
283  int uri_length = uri->length();
284  std::vector<uint8_t> buffer;
285  buffer.reserve(uri_length);
286 
287  bool throw_error = false;
288  {
289  DisallowGarbageCollection no_gc;
290  String::FlatContent uri_content = uri->GetFlatContent(no_gc);
291 
292  for (int k = 0; k < uri_length; k++) {
293  base::uc16 cc1 = uri_content.Get(k);
294  if (unibrow::Utf16::IsLeadSurrogate(cc1)) {
295  k++;
296  if (k < uri_length) {
297  base::uc16 cc2 = uri->Get(k);
298  if (unibrow::Utf16::IsTrailSurrogate(cc2)) {
299  EncodePair(cc1, cc2, &buffer);
300  continue;
301  }
302  }
303  } else if (!unibrow::Utf16::IsTrailSurrogate(cc1)) {
304  if (IsUnescapePredicateInUriComponent(cc1) ||
305  (is_uri && IsUriSeparator(cc1))) {
306  buffer.push_back(cc1);
307  } else {
308  EncodeSingle(cc1, &buffer);
309  }
310  continue;
311  }
312 
313  // String::FlatContent DCHECKs its contents did not change during its
314  // lifetime. Throwing the error inside the loop may cause GC and move the
315  // string contents.
316  throw_error = true;
317  break;
318  }
319  }
320 
321  if (throw_error) THROW_NEW_ERROR(isolate, NewURIError());
322  if (buffer.size() > v8::String::kMaxLength) {
323  THROW_NEW_ERROR(isolate, NewInvalidStringLengthError());
324  }
325  return isolate->factory()->NewStringFromOneByte(base::VectorOf(buffer));
326 }

330 template <typename Char>
331 int UnescapeChar(base::Vector<const Char> vector, int i, int length,
332  int* step) {
333  uint16_t character = vector[i];
334  int32_t hi = 0;
335  int32_t lo = 0;
336  if (character == '%' && i <= length - 6 && vector[i + 1] == 'u' &&
337  (hi = TwoDigitHex(vector[i + 2], vector[i + 3])) > -1 &&
338  (lo = TwoDigitHex(vector[i + 4], vector[i + 5])) > -1) {
339  *step = 6;
340  return (hi << 8) + lo;
341  } else if (character == '%' && i <= length - 3 &&
342  (lo = TwoDigitHex(vector[i + 1], vector[i + 2])) > -1) {
343  *step = 3;
344  return lo;
345  } else {
346  *step = 1;
347  return character;
348  }
349 }

351 template <typename Char>
352 MaybeHandle<String> UnescapeSlow(Isolate* isolate, DirectHandle<String> string,
353  int start_index) {
354  bool one_byte = true;
355  uint32_t length = string->length();
356 
357  int unescaped_length = 0;
358  {
359  DisallowGarbageCollection no_gc;
360  base::Vector<const Char> vector = string->GetCharVector<Char>(no_gc);
361  for (uint32_t i = start_index; i < length; unescaped_length++) {
362  int step;
363  if (UnescapeChar(vector, i, length, &step) >
364  String::kMaxOneByteCharCode) {
365  one_byte = false;
366  }
367  i += step;
368  }
369  }
370 
371  DCHECK_LT(start_index, length);
372  Handle<String> first_part =
373  isolate->factory()->NewProperSubString(string, 0, start_index);
374 
375  int dest_position = 0;
376  Handle<String> second_part;
377  DCHECK_LE(unescaped_length, String::kMaxLength);
378  if (one_byte) {
379  Handle<SeqOneByteString> dest = isolate->factory()
380  ->NewRawOneByteString(unescaped_length)
381  .ToHandleChecked();
382  DisallowGarbageCollection no_gc;
383  base::Vector<const Char> vector = string->GetCharVector<Char>(no_gc);
384  for (uint32_t i = start_index; i < length; dest_position++) {
385  int step;
386  dest->SeqOneByteStringSet(dest_position,
387  UnescapeChar(vector, i, length, &step));
388  i += step;
389  }
390  second_part = dest;
391  } else {
392  Handle<SeqTwoByteString> dest = isolate->factory()
393  ->NewRawTwoByteString(unescaped_length)
394  .ToHandleChecked();
395  DisallowGarbageCollection no_gc;
396  base::Vector<const Char> vector = string->GetCharVector<Char>(no_gc);
397  for (uint32_t i = start_index; i < length; dest_position++) {
398  int step;
399  dest->SeqTwoByteStringSet(dest_position,
400  UnescapeChar(vector, i, length, &step));
401  i += step;
402  }
403  second_part = dest;
404  }
405  return isolate->factory()->NewConsString(first_part, second_part);
406 }

440 template <typename Char>
441 static MaybeHandle<String> EscapePrivate(Isolate* isolate,
442  Handle<String> string) {
443  DCHECK(string->IsFlat());
444  uint32_t escaped_length = 0;
445  uint32_t length = string->length();
446 
447  {
448  DisallowGarbageCollection no_gc;
449  base::Vector<const Char> vector = string->GetCharVector<Char>(no_gc);
450  for (uint32_t i = 0; i < length; i++) {
451  uint16_t c = vector[i];
452  if (c >= 256) {
453  escaped_length += 6;
454  } else if (IsNotEscaped(c)) {
455  escaped_length++;
456  } else {
457  escaped_length += 3;
458  }
459 
460  // We don't allow strings that are longer than a maximal length.
461  DCHECK_LT(String::kMaxLength, 0x7FFFFFFF - 6); // Cannot overflow.
462  if (escaped_length > String::kMaxLength) break; // Provoke exception.
463  }
464  }
465 
466  // No length change implies no change. Return original string if no change.
467  if (escaped_length == length) return string;
468 
469  Handle<SeqOneByteString> dest;
470  ASSIGN_RETURN_ON_EXCEPTION(
471  isolate, dest, isolate->factory()->NewRawOneByteString(escaped_length));
472  int dest_position = 0;
473 
474  {
475  DisallowGarbageCollection no_gc;
476  base::Vector<const Char> vector = string->GetCharVector<Char>(no_gc);
477  for (uint32_t i = 0; i < length; i++) {
478  uint16_t c = vector[i];
479  if (c >= 256) {
480  dest->SeqOneByteStringSet(dest_position, '%');
481  dest->SeqOneByteStringSet(dest_position + 1, 'u');
482  dest->SeqOneByteStringSet(dest_position + 2,
483  base::HexCharOfValue(c >> 12));
484  dest->SeqOneByteStringSet(dest_position + 3,
485  base::HexCharOfValue((c >> 8) & 0xF));
486  dest->SeqOneByteStringSet(dest_position + 4,
487  base::HexCharOfValue((c >> 4) & 0xF));
488  dest->SeqOneByteStringSet(dest_position + 5,
489  base::HexCharOfValue(c & 0xF));
490  dest_position += 6;
491  } else if (IsNotEscaped(c)) {
492  dest->SeqOneByteStringSet(dest_position, c);
493  dest_position++;
494  } else {
495  dest->SeqOneByteStringSet(dest_position, '%');
496  dest->SeqOneByteStringSet(dest_position + 1,
497  base::HexCharOfValue(c >> 4));
498  dest->SeqOneByteStringSet(dest_position + 2,
499  base::HexCharOfValue(c & 0xF));
500  dest_position += 3;
501  }
502  }
503  }
504 
505  return dest;
506 }
