318  template <class T, int size, int lo, int hi>
319  inline T getfield(T value) {
320  DCHECK(lo < hi);
321  DCHECK_GT(size, 0);
322  int mask = hi - lo;
323  int shift = size * 8 - hi;
324  uint32_t mask_value = (mask == 32) ? 0xffffffff : (1 << mask) - 1;
325  return (value & mask_value) << shift;
326  }
