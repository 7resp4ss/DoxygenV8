33 bool HasHeapNumberInputOrOutput(Node* node) {
34  int input_count = node->InputCount();
35  for (int j = 2; j < input_count; ++j) {
36  Node* input = node->InputAt(j);
37  if (input->opcode() == IrOpcode::kNumberConstant) {
38  double value = OpParameter<double>(input->op());
39  if (!IsSmiDouble(value)) {
40  return true;
41  }
42  } else if (input->opcode() == IrOpcode::kChangeFloat64HoleToTagged) {
43  return true;
44  }
45  }
46 
47  for (Edge const edge : node->use_edges()) {
48  if (!NodeProperties::IsValueEdge(edge)) continue;
49  switch (edge.from()->opcode()) {
50  case IrOpcode::kSpeculativeNumberAdd:
51  case IrOpcode::kSpeculativeNumberSubtract:
52  case IrOpcode::kSpeculativeNumberMultiply:
53  case IrOpcode::kSpeculativeNumberPow:
54  case IrOpcode::kSpeculativeNumberDivide:
55  case IrOpcode::kSpeculativeNumberModulus:
56  case IrOpcode::kSpeculativeNumberBitwiseAnd:
57  case IrOpcode::kSpeculativeNumberBitwiseOr:
58  case IrOpcode::kSpeculativeNumberBitwiseXor:
59  case IrOpcode::kSpeculativeNumberShiftLeft:
60  case IrOpcode::kSpeculativeNumberShiftRight:
61  case IrOpcode::kSpeculativeNumberShiftRightLogical:
62  case IrOpcode::kSpeculativeAdditiveSafeIntegerAdd:
63  case IrOpcode::kSpeculativeAdditiveSafeIntegerSubtract:
64  return true;
65 
66  default:
67  break;
68  }
69  }
70 
71  return false;
72 }

74 bool CanConsiderForInlining(JSHeapBroker* broker,
75  FeedbackCellRef feedback_cell) {
76  OptionalFeedbackVectorRef feedback_vector =
77  feedback_cell.feedback_vector(broker);
78  if (!feedback_vector.has_value()) {
79  TRACE("Cannot consider " << feedback_cell
80  << " for inlining (no feedback vector)");
81  return false;
82  }
83  SharedFunctionInfoRef shared = feedback_vector->shared_function_info(broker);
84 
85  if (!shared.HasBytecodeArray()) {
86  TRACE("Cannot consider " << shared << " for inlining (no bytecode)");
87  return false;
88  }
89  // Ensure we have a persistent handle to the bytecode in order to avoid
90  // flushing it during the remaining compilation.
91  shared.GetBytecodeArray(broker);
92 
93  // Read feedback vector again in case it got flushed before we were able to
94  // prevent flushing above.
95  OptionalFeedbackVectorRef feedback_vector_again =
96  feedback_cell.feedback_vector(broker);
97  if (!feedback_vector_again.has_value()) {
98  TRACE("Cannot consider " << shared << " for inlining (no feedback vector)");
99  return false;
100  }
101  if (!feedback_vector_again->equals(*feedback_vector)) {
102  // The new feedback vector likely contains lots of uninitialized slots, so
103  // it doesn't make much sense to inline this function now.
104  TRACE("Not considering " << shared
105  << " for inlining (feedback vector changed)");
106  return false;
107  }
108 
109  SharedFunctionInfo::Inlineability inlineability =
110  shared.GetInlineability(CodeKind::TURBOFAN_JS, broker);
111  if (inlineability != SharedFunctionInfo::kIsInlineable) {
112  TRACE("Cannot consider "
113  << shared << " for inlining (reason: " << inlineability << ")");
114  return false;
115  }
116 
117  TRACE("Considering " << shared << " for inlining with " << *feedback_vector);
118  return true;
119 }

133 JSInliningHeuristic::Candidate JSInliningHeuristic::CollectFunctions(
134  Node* node, int functions_size) {
135  DCHECK_NE(0, functions_size);
136  Node* callee = node->InputAt(0);
137  Candidate out;
138  out.node = node;
139 
140  HeapObjectMatcher m(callee);
141  if (m.HasResolvedValue() && m.Ref(broker()).IsJSFunction()) {
142  JSFunctionRef function = m.Ref(broker()).AsJSFunction();
143  out.functions[0] = function;
144  if (CanConsiderForInlining(broker(), function)) {
145  out.bytecode[0] = function.shared(broker()).GetBytecodeArray(broker());
146  out.num_functions = 1;
147 
148  out.has_heapnumber_params = HasHeapNumberInputOrOutput(node);
149 
150  return out;
151  }
152  }
153  if (m.IsPhi()) {
154  int const value_input_count = m.node()->op()->ValueInputCount();
155  if (value_input_count > functions_size) {
156  out.num_functions = 0;
157  return out;
158  }
159  for (int n = 0; n < value_input_count; ++n) {
160  HeapObjectMatcher m2(callee->InputAt(n));
161  if (!m2.HasResolvedValue() || !m2.Ref(broker()).IsJSFunction()) {
162  out.num_functions = 0;
163  return out;
164  }
165 
166  out.functions[n] = m2.Ref(broker()).AsJSFunction();
167  JSFunctionRef function = out.functions[n].value();
168  if (CanConsiderForInlining(broker(), function)) {
169  out.bytecode[n] = function.shared(broker()).GetBytecodeArray(broker());
170  }
171  }
172  out.num_functions = value_input_count;
173  return out;
174  }
175  if (m.IsCheckClosure()) {
176  DCHECK(!out.functions[0].has_value());
177  FeedbackCellRef feedback_cell = MakeRef(broker(), FeedbackCellOf(m.op()));
178  if (CanConsiderForInlining(broker(), feedback_cell)) {
179  out.shared_info = feedback_cell.shared_function_info(broker()).value();
180  out.bytecode[0] = out.shared_info->GetBytecodeArray(broker());
181  }
182  out.num_functions = 1;
183  return out;
184  }
185  if (m.IsJSCreateClosure()) {
186  DCHECK(!out.functions[0].has_value());
187  JSCreateClosureNode n(callee);
188  FeedbackCellRef feedback_cell = n.GetFeedbackCellRefChecked(broker());
189  if (CanConsiderForInlining(broker(), feedback_cell)) {
190  out.shared_info = feedback_cell.shared_function_info(broker()).value();
191  out.bytecode[0] = out.shared_info->GetBytecodeArray(broker());
192  CHECK(out.shared_info->equals(n.Parameters().shared_info()));
193  }
194  out.num_functions = 1;
195  return out;
196  }
197  out.num_functions = 0;
198  return out;
199 }

201 Reduction JSInliningHeuristic::Reduce(Node* node) {
202 #if V8_ENABLE_WEBASSEMBLY
203  if (mode() == kWasmWrappersOnly || mode() == kWasmFullInlining) {
204  if (node->opcode() == IrOpcode::kJSWasmCall) {
205  return inliner_.ReduceJSWasmCall(node);
206  }
207  return NoChange();
208  }
209 #endif // V8_ENABLE_WEBASSEMBLY
210 
211  DCHECK_EQ(mode(), kJSOnly);
212  if (!IrOpcode::IsInlineeOpcode(node->opcode())) return NoChange();
213 
214  // Check if we already saw that {node} before, and if so, just skip it.
215  if (seen_.find(node->id()) != seen_.end()) return NoChange();
216 
217  // Check if the {node} is an appropriate candidate for inlining.
218  Candidate candidate = CollectFunctions(node, kMaxCallPolymorphism);
219  if (candidate.num_functions == 0) {
220  return NoChange();
221  }
222 
223  bool can_inline_candidate = false, candidate_is_small = true;
224  candidate.total_size = 0;
225  FrameState frame_state{NodeProperties::GetFrameStateInput(node)};
226  FrameStateInfo const& frame_info = frame_state.frame_state_info();
227  Handle<SharedFunctionInfo> frame_shared_info;
228  for (int i = 0; i < candidate.num_functions; ++i) {
229  if (!candidate.bytecode[i].has_value()) {
230  candidate.can_inline_function[i] = false;
231  continue;
232  }
233 
234  SharedFunctionInfoRef shared =
235  candidate.functions[i].has_value()
236  ? candidate.functions[i].value().shared(broker())
237  : candidate.shared_info.value();
238  candidate.can_inline_function[i] = candidate.bytecode[i].has_value();
239  // Because of concurrent optimization, optimization of the inlining
240  // candidate could have been disabled meanwhile.
241  // JSInliner will check this again and not actually inline the function in
242  // this case.
243  CHECK_IMPLIES(
244  candidate.can_inline_function[i],
245  shared.IsInlineable(CodeKind::TURBOFAN_JS, broker()) ||
246  shared.GetInlineability(CodeKind::TURBOFAN_JS, broker()) ==
247  SharedFunctionInfo::kHasOptimizationDisabled);
248  // Do not allow direct recursion i.e. f() -> f(). We still allow indirect
249  // recursion like f() -> g() -> f(). The indirect recursion is helpful in
250  // cases where f() is a small dispatch function that calls the appropriate
251  // function. In the case of direct recursion, we only have some static
252  // information for the first level of inlining and it may not be that useful
253  // to just inline one level in recursive calls. In some cases like tail
254  // recursion we may benefit from recursive inlining, if we have additional
255  // analysis that converts them to iterative implementations. Though it is
256  // not obvious if such an analysis is needed.
257  if (frame_info.shared_info().ToHandle(&frame_shared_info) &&
258  frame_shared_info.equals(shared.object())) {
259  TRACE("Not considering call site #" << node->id() << ":"
260  << node->op()->mnemonic()
261  << ", because of recursive inlining");
262  candidate.can_inline_function[i] = false;
263  }
264  if (candidate.can_inline_function[i]) {
265  can_inline_candidate = true;
266  BytecodeArrayRef bytecode = candidate.bytecode[i].value();
267  candidate.total_size += bytecode.length();
268  candidate.own_size += bytecode.length();
269  unsigned inlined_bytecode_size = 0;
270  if (OptionalJSFunctionRef function = candidate.functions[i]) {
271  if (OptionalCodeRef code = function->code(broker())) {
272  inlined_bytecode_size = code->GetInlinedBytecodeSize();
273  candidate.total_size += inlined_bytecode_size;
274  }
275  }
276 
277  bool this_function_small = false;
278  if (IsSmall(bytecode.length() + inlined_bytecode_size)) {
279  this_function_small = true;
280  } else if (candidate.has_heapnumber_params &&
281  IsSmallWithHeapNumberParam(bytecode.length())) {
282  this_function_small = true;
283  }
284 
285  candidate_is_small = candidate_is_small && this_function_small;
286  }
287  }
288  if (!can_inline_candidate) return NoChange();
289 
290  // Gather feedback on how often this call site has been hit before.
291  if (node->opcode() == IrOpcode::kJSCall) {
292  CallParameters const p = CallParametersOf(node->op());
293  candidate.frequency = p.frequency();
294  } else {
295  ConstructParameters const p = ConstructParametersOf(node->op());
296  candidate.frequency = p.frequency();
297  }
298 
299  // Don't consider a {candidate} whose frequency is below the
300  // threshold, i.e. a call site that is only hit once every N
301  // invocations of the caller.
302  if (candidate.frequency.IsKnown() &&
303  candidate.frequency.value() < v8_flags.min_inlining_frequency) {
304  return NoChange();
305  }
306 
307  // Found a candidate. Insert it into the set of seen nodes s.t. we don't
308  // revisit in the future. Note this insertion happens here and not earlier in
309  // order to make inlining decisions order-independent. A node may not be a
310  // candidate when first seen, but later reductions may turn it into a valid
311  // candidate. In that case, the node should be revisited by
312  // JSInliningHeuristic.
313  seen_.insert(node->id());
314 
315  // Forcibly inline small functions here. In the case of polymorphic inlining
316  // candidate_is_small is set only when all functions are small.
317  if (candidate_is_small &&
318  total_ignored_bytecode_size_ < max_inlined_bytecode_size_small_total_) {
319  TRACE("Inlining small function(s) at call site #"
320  << node->id() << ":" << node->op()->mnemonic());
321  return InlineCandidate(candidate, true);
322  }
323 
324  if (total_inlined_bytecode_size_ >= max_inlined_bytecode_size_absolute_) {
325  return NoChange();
326  }
327 
328  // In the general case we remember the candidate for later.
329  candidates_.insert(candidate);
330  return NoChange();
331 }

333 void JSInliningHeuristic::Finalize() {
334  if (candidates_.empty()) return; // Nothing to do without candidates.
335  if (v8_flags.trace_turbo_inlining) PrintCandidates();
336 
337  // We inline at most one candidate in every iteration of the fixpoint.
338  // This is to ensure that we don't consume the full inlining budget
339  // on things that aren't called very often.
340  // TODO(bmeurer): Use std::priority_queue instead of std::set here.
341  while (!candidates_.empty()) {
342  auto i = candidates_.begin();
343  Candidate candidate = *i;
344  candidates_.erase(i);
345 
346  // Ignore this candidate if it's no longer valid.
347  if (!IrOpcode::IsInlineeOpcode(candidate.node->opcode())) continue;
348  if (candidate.node->IsDead()) continue;
349 
350  // We re-check HasHeapNumberInputOrOutput because the inputs/outputs of this
351  // function could have previously been function calls that now have been
352  // inlined and thusly revealed HeapNumbers.
353  if (HasHeapNumberInputOrOutput(candidate.node) &&
354  IsSmallWithHeapNumberParam(candidate.own_size) &&
355  total_ignored_bytecode_size_ < max_inlined_bytecode_size_small_total_) {
356  Reduction const reduction = InlineCandidate(candidate, true);
357  if (reduction.Changed()) return;
358  }
359 
360  // Make sure we have some extra budget left, so that any small functions
361  // exposed by this function would be given a chance to inline.
362  double size_of_candidate =
363  candidate.total_size * v8_flags.reserve_inline_budget_scale_factor;
364  int total_size =
365  total_inlined_bytecode_size_ + static_cast<int>(size_of_candidate);
366 
367  if (total_size > max_inlined_bytecode_size_cumulative_) {
368  info_->set_could_not_inline_all_candidates();
369  // Try if any smaller functions are available to inline.
370  continue;
371  }
372 
373  Reduction const reduction = InlineCandidate(candidate, false);
374  if (reduction.Changed()) return;
375  }
376 }

385 bool CollectStateValuesOwnedUses(Node* node, Node* state_values,
386  NodeAndIndex* uses_buffer, size_t* use_count,
387  size_t max_uses) {
388  // Only accumulate states that are not shared with other users.
389  if (state_values->UseCount() > 1) return true;
390  for (int i = 0; i < state_values->InputCount(); i++) {
391  Node* input = state_values->InputAt(i);
392  if (input->opcode() == IrOpcode::kStateValues) {
393  if (!CollectStateValuesOwnedUses(node, input, uses_buffer, use_count,
394  max_uses)) {
395  return false;
396  }
397  } else if (input == node) {
398  if (*use_count >= max_uses) return false;
399  uses_buffer[*use_count] = {state_values, i};
400  (*use_count)++;
401  }
402  }
403  return true;
404 }

408 Node* JSInliningHeuristic::DuplicateStateValuesAndRename(Node* state_values,
409  Node* from, Node* to,
410  StateCloneMode mode) {
411  // Only rename in states that are not shared with other users. This needs to
412  // be in sync with the condition in {CollectStateValuesOwnedUses}.
413  if (state_values->UseCount() > 1) return state_values;
414  Node* copy = mode == kChangeInPlace ? state_values : nullptr;
415  for (int i = 0; i < state_values->InputCount(); i++) {
416  Node* input = state_values->InputAt(i);
417  Node* processed;
418  if (input->opcode() == IrOpcode::kStateValues) {
419  processed = DuplicateStateValuesAndRename(input, from, to, mode);
420  } else if (input == from) {
421  processed = to;
422  } else {
423  processed = input;
424  }
425  if (processed != input) {
426  if (!copy) {
427  copy = graph()->CloneNode(state_values);
428  }
429  copy->ReplaceInput(i, processed);
430  }
431  }
432  return copy ? copy : state_values;
433 }

480 bool JSInliningHeuristic::TryReuseDispatch(Node* node, Node* callee,
481  Node** if_successes, Node** calls,
482  Node** inputs, int input_count,
483  int* num_calls) {
484  // We will try to reuse the control flow branch created for computing
485  // the {callee} target of the call. We only reuse the branch if there
486  // is no side-effect between the call and the branch, and if the callee is
487  // only used as the target (and possibly also in the related frame states).
488 
489  // We are trying to match the following pattern:
490  //
491  // C1 C2
492  // . .
493  // | |
494  // Merge(merge) <-----------------+
495  // ^ ^ |
496  // V1 V2 | | E1 E2 |
497  // . . | +----+ . . |
498  // | | | | | | |
499  // Phi(callee) EffectPhi(effect_phi) |
500  // ^ ^ |
501  // | | |
502  // +----+ | |
503  // | | | |
504  // | StateValues | |
505  // | ^ | |
506  // +----+ | | |
507  // | | | | |
508  // | FrameState | |
509  // | ^ | |
510  // | | | +---+
511  // | | | | |
512  // +----+ Checkpoint(checkpoint) |
513  // | | ^ |
514  // | StateValues | +-------------+
515  // | | | |
516  // +-----+ | | |
517  // | | | | |
518  // | FrameState | |
519  // | ^ | |
520  // +-----------+ | | |
521  // Call(node)
522  // |
523  // |
524  // .
525  //
526  // The {callee} here is a phi that merges the possible call targets, {node}
527  // is the actual call that we will try to duplicate and connect to the
528  // control that comes into {merge}. There can be a {checkpoint} between
529  // the call and the calle phi.
530  //
531  // The idea is to get rid of the merge, effect phi and phi, then duplicate
532  // the call (with all the frame states and such), and connect the duplicated
533  // calls and states directly to the inputs of the ex-phi, ex-effect-phi and
534  // ex-merge. The tricky part is to make sure that there is no interference
535  // from the outside. In particular, there should not be any unaccounted uses
536  // of the phi, effect-phi and merge because we will remove them from
537  // the graph.
538  //
539  // V1 E1 C1 V2 E2 C2
540  // . . . . . .
541  // | | | | | |
542  // +----+ | | +----+ |
543  // | | | | | | |
544  // | StateValues | | | StateValues |
545  // | ^ | | | ^ |
546  // +----+ | | | +----+ | |
547  // | | | | | | | | |
548  // | FrameState | | | FrameState |
549  // | ^ | | | ^ |
550  // | | | | | | |
551  // | | | | | | |
552  // +----+ Checkpoint | +----+ Checkpoint |
553  // | | ^ | | | ^ |
554  // | StateValues | | | StateValues | |
555  // | | | | | | | |
556  // +-----+ | | | +-----+ | | |
557  // | | | | | | | | | |
558  // | FrameState | | | FrameState | |
559  // | ^ | | | ^ | |
560  // +-------------+| | | +-------------+| | |
561  // Call----+ Call----+
562  // | |
563  // +-------+ +------------+
564  // | |
565  // Merge
566  // EffectPhi
567  // Phi
568  // |
569  // ...
570 
571  // Bailout if the call is not polymorphic anymore (other reducers might
572  // have replaced the callee phi with a constant).
573  if (callee->opcode() != IrOpcode::kPhi) return false;
574 
575  // If there is a control node between the callee computation
576  // and the call, bail out.
577  Node* merge = NodeProperties::GetControlInput(callee);
578  if (NodeProperties::GetControlInput(node) != merge) return false;
579 
580  // If there is a non-checkpoint effect node between the callee computation
581  // and the call, bail out. We will drop any checkpoint between the call and
582  // the callee phi because the callee computation should have its own
583  // checkpoint that the call can fall back to.
584  Node* checkpoint = nullptr;
585  Node* effect = NodeProperties::GetEffectInput(node);
586  if (effect->opcode() == IrOpcode::kCheckpoint) {
587  checkpoint = effect;
588  if (NodeProperties::GetControlInput(checkpoint) != merge) return false;
589  effect = NodeProperties::GetEffectInput(effect);
590  }
591  if (effect->opcode() != IrOpcode::kEffectPhi) return false;
592  if (NodeProperties::GetControlInput(effect) != merge) return false;
593  Node* effect_phi = effect;
594 
595  // The effect phi, the callee, the call and the checkpoint must be the only
596  // users of the merge.
597  for (Node* merge_use : merge->uses()) {
598  if (merge_use != effect_phi && merge_use != callee && merge_use != node &&
599  merge_use != checkpoint) {
600  return false;
601  }
602  }
603 
604  // The effect phi must be only used by the checkpoint or the call.
605  for (Node* effect_phi_use : effect_phi->uses()) {
606  if (effect_phi_use != node && effect_phi_use != checkpoint) return false;
607  }
608 
609  // We must replace the callee phi with the appropriate constant in
610  // the entire subgraph reachable by inputs from the call (terminating
611  // at phis and merges). Since we do not want to walk (and later duplicate)
612  // the subgraph here, we limit the possible uses to this set:
613  //
614  // 1. In the call (as a target).
615  // 2. The checkpoint between the call and the callee computation merge.
616  // 3. The lazy deoptimization frame state.
617  //
618  // This corresponds to the most common pattern, where the function is
619  // called with only local variables or constants as arguments.
620  //
621  // To check the uses, we first collect all the occurrences of callee in 1, 2
622  // and 3, and then we check that all uses of callee are in the collected
623  // occurrences. If there is an unaccounted use, we do not try to rewire
624  // the control flow.
625  //
626  // Note: With CFG, this would be much easier and more robust - we would just
627  // duplicate all the nodes between the merge and the call, replacing all
628  // occurrences of the {callee} phi with the appropriate constant.
629 
630  // First compute the set of uses that are only reachable from 2 and 3.
631  const size_t kMaxUses = 8;
632  NodeAndIndex replaceable_uses[kMaxUses];
633  size_t replaceable_uses_count = 0;
634 
635  // Collect the uses to check case 2.
636  Node* checkpoint_state = nullptr;
637  if (checkpoint) {
638  checkpoint_state = checkpoint->InputAt(0);
639  if (!CollectFrameStateUniqueUses(callee, FrameState{checkpoint_state},
640  replaceable_uses, &replaceable_uses_count,
641  kMaxUses)) {
642  return false;
643  }
644  }
645 
646  // Collect the uses to check case 3.
647  FrameState frame_state{NodeProperties::GetFrameStateInput(node)};
648  if (!CollectFrameStateUniqueUses(callee, frame_state, replaceable_uses,
649  &replaceable_uses_count, kMaxUses)) {
650  return false;
651  }
652 
653  // Bail out if there is a use of {callee} that is not reachable from 1, 2
654  // and 3.
655  for (Edge edge : callee->use_edges()) {
656  // Case 1 (use by the call as a target).
657  if (edge.from() == node && edge.index() == 0) continue;
658  // Case 2 and 3 - used in checkpoint and/or lazy deopt frame states.
659  bool found = false;
660  for (size_t i = 0; i < replaceable_uses_count; i++) {
661  if (replaceable_uses[i].node == edge.from() &&
662  replaceable_uses[i].index == edge.index()) {
663  found = true;
664  break;
665  }
666  }
667  if (!found) return false;
668  }
669 
670  *num_calls = callee->op()->ValueInputCount();
671 
672  // Clone the call and the framestate, including the uniquely reachable
673  // state values, making sure that we replace the phi with the constant.
674  for (int i = 0; i < *num_calls; ++i) {
675  // Clone the calls for each branch.
676  // We need to specialize the calls to the correct target, effect, and
677  // control. We also need to duplicate the checkpoint and the lazy
678  // frame state, and change all the uses of the callee to the constant
679  // callee.
680  Node* target = callee->InputAt(i);
681  Node* effect_phi_effect = effect_phi->InputAt(i);
682  Node* control = merge->InputAt(i);
683 
684  if (checkpoint) {
685  // Duplicate the checkpoint.
686  FrameState new_checkpoint_state = DuplicateFrameStateAndRename(
687  FrameState{checkpoint_state}, callee, target,
688  (i == *num_calls - 1) ? kChangeInPlace : kCloneState);
689  effect_phi_effect = graph()->NewNode(
690  checkpoint->op(), new_checkpoint_state, effect_phi_effect, control);
691  }
692 
693  // Duplicate the call.
694  FrameState new_lazy_frame_state = DuplicateFrameStateAndRename(
695  frame_state, callee, target,
696  (i == *num_calls - 1) ? kChangeInPlace : kCloneState);
697  inputs[0] = target;
698  inputs[input_count - 3] = new_lazy_frame_state;
699  inputs[input_count - 2] = effect_phi_effect;
700  inputs[input_count - 1] = control;
701  calls[i] = if_successes[i] =
702  graph()->NewNode(node->op(), input_count, inputs);
703  }
704 
705  // Mark the control inputs dead, so that we can kill the merge.
706  node->ReplaceInput(input_count - 1, jsgraph()->Dead());
707  callee->ReplaceInput(*num_calls, jsgraph()->Dead());
708  effect_phi->ReplaceInput(*num_calls, jsgraph()->Dead());
709  if (checkpoint) {
710  checkpoint->ReplaceInput(2, jsgraph()->Dead());
711  }
712 
713  merge->Kill();
714  return true;
715 }

717 void JSInliningHeuristic::CreateOrReuseDispatch(
718  Node* node, Node* callee, Candidate const& candidate, Node** if_successes,
719  Node** calls, Node** inputs, int input_count, int* num_calls) {
720  SourcePositionTable::Scope position(
721  source_positions_, source_positions_->GetSourcePosition(node));
722  if (TryReuseDispatch(node, callee, if_successes, calls, inputs, input_count,
723  num_calls)) {
724  return;
725  }
726 
727  static_assert(JSCallOrConstructNode::kHaveIdenticalLayouts);
728 
729  Node* fallthrough_control = NodeProperties::GetControlInput(node);
730  *num_calls = candidate.num_functions;
731 
732  // Create the appropriate control flow to dispatch to the cloned calls.
733  for (int i = 0; i < *num_calls; ++i) {
734  // TODO(2206): Make comparison be based on underlying SharedFunctionInfo
735  // instead of the target JSFunction reference directly.
736  Node* target =
737  jsgraph()->ConstantNoHole(candidate.functions[i].value(), broker());
738  if (i != (*num_calls - 1)) {
739  Node* check =
740  graph()->NewNode(simplified()->ReferenceEqual(), callee, target);
741  Node* branch =
742  graph()->NewNode(common()->Branch(), check, fallthrough_control);
743  fallthrough_control = graph()->NewNode(common()->IfFalse(), branch);
744  if_successes[i] = graph()->NewNode(common()->IfTrue(), branch);
745  } else {
746  if_successes[i] = fallthrough_control;
747  }
748 
749  // Clone the calls for each branch.
750  // The first input to the call is the actual target (which we specialize
751  // to the known {target}); the last input is the control dependency.
752  // We also specialize the new.target of JSConstruct {node}s if it refers
753  // to the same node as the {node}'s target input, so that we can later
754  // properly inline the JSCreate operations.
755  if (node->opcode() == IrOpcode::kJSConstruct) {
756  // TODO(jgruber, v8:10675): This branch seems unreachable.
757  JSConstructNode n(node);
758  if (inputs[n.TargetIndex()] == inputs[n.NewTargetIndex()]) {
759  inputs[n.NewTargetIndex()] = target;
760  }
761  }
762  inputs[JSCallOrConstructNode::TargetIndex()] = target;
763  inputs[input_count - 1] = if_successes[i];
764  calls[i] = if_successes[i] =
765  graph()->NewNode(node->op(), input_count, inputs);
766  }
767 }

769 Reduction JSInliningHeuristic::InlineCandidate(Candidate const& candidate,
770  bool small_function) {
771  int num_calls = candidate.num_functions;
772  Node* const node = candidate.node;
773 
774 #if V8_ENABLE_WEBASSEMBLY
775  DCHECK_NE(node->opcode(), IrOpcode::kJSWasmCall);
776 #endif // V8_ENABLE_WEBASSEMBLY
777  if (num_calls == 1) {
778  Reduction const reduction = inliner_.ReduceJSCall(node);
779  if (reduction.Changed()) {
780  if (small_function) {
781  total_ignored_bytecode_size_ += candidate.bytecode[0].value().length();
782  } else {
783  total_inlined_bytecode_size_ += candidate.bytecode[0].value().length();
784  }
785  }
786  return reduction;
787  }
788 
789  // Expand the JSCall/JSConstruct node to a subgraph first if
790  // we have multiple known target functions.
791  DCHECK_LT(1, num_calls);
792  Node* calls[kMaxCallPolymorphism + 1];
793  Node* if_successes[kMaxCallPolymorphism];
794  Node* callee = NodeProperties::GetValueInput(node, 0);
795 
796  // Setup the inputs for the cloned call nodes.
797  int const input_count = node->InputCount();
798  Node** inputs = graph()->zone()->AllocateArray<Node*>(input_count);
799  for (int i = 0; i < input_count; ++i) {
800  inputs[i] = node->InputAt(i);
801  }
802 
803  // Create the appropriate control flow to dispatch to the cloned calls.
804  CreateOrReuseDispatch(node, callee, candidate, if_successes, calls, inputs,
805  input_count, &num_calls);
806 
807  // Check if we have an exception projection for the call {node}.
808  Node* if_exception = nullptr;
809  if (NodeProperties::IsExceptionalCall(node, &if_exception)) {
810  Node* if_exceptions[kMaxCallPolymorphism + 1];
811  for (int i = 0; i < num_calls; ++i) {
812  if_successes[i] = graph()->NewNode(common()->IfSuccess(), calls[i]);
813  if_exceptions[i] =
814  graph()->NewNode(common()->IfException(), calls[i], calls[i]);
815  }
816 
817  // Morph the {if_exception} projection into a join.
818  Node* exception_control =
819  graph()->NewNode(common()->Merge(num_calls), num_calls, if_exceptions);
820  if_exceptions[num_calls] = exception_control;
821  Node* exception_effect = graph()->NewNode(common()->EffectPhi(num_calls),
822  num_calls + 1, if_exceptions);
823  Node* exception_value = graph()->NewNode(
824  common()->Phi(MachineRepresentation::kTagged, num_calls), num_calls + 1,
825  if_exceptions);
826  ReplaceWithValue(if_exception, exception_value, exception_effect,
827  exception_control);
828  }
829 
830  // Morph the original call site into a join of the dispatched call sites.
831  Node* control =
832  graph()->NewNode(common()->Merge(num_calls), num_calls, if_successes);
833  calls[num_calls] = control;
834  Node* effect =
835  graph()->NewNode(common()->EffectPhi(num_calls), num_calls + 1, calls);
836  Node* value =
837  graph()->NewNode(common()->Phi(MachineRepresentation::kTagged, num_calls),
838  num_calls + 1, calls);
839  ReplaceWithValue(node, value, effect, control);
840 
841  // Inline the individual, cloned call sites.
842  for (int i = 0; i < num_calls && total_inlined_bytecode_size_ <
843  max_inlined_bytecode_size_absolute_;
844  ++i) {
845  if (candidate.can_inline_function[i] &&
846  (small_function || total_inlined_bytecode_size_ <
847  max_inlined_bytecode_size_cumulative_)) {
848  Node* call = calls[i];
849  Reduction const reduction = inliner_.ReduceJSCall(call);
850  if (reduction.Changed()) {
851  if (small_function) {
852  total_ignored_bytecode_size_ +=
853  candidate.bytecode[i].value().length();
854  } else {
855  total_inlined_bytecode_size_ += candidate.bytecode[i]->length();
856  }
857  // Killing the call node is not strictly necessary, but it is safer to
858  // make sure we do not resurrect the node.
859  call->Kill();
860  }
861  }
862  }
863 
864  return Replace(value);
865 }

867 bool JSInliningHeuristic::CandidateCompare::operator()(
868  const Candidate& left, const Candidate& right) const {
869  constexpr bool kInlineLeftFirst = true, kInlineRightFirst = false;
870 
871  if (right.frequency.IsUnknown()) {
872  if (left.frequency.IsUnknown()) {
873  // If left and right are both unknown then the ordering is indeterminate,
874  // which breaks strict weak ordering requirements, so we fall back to the
875  // node id as a tie breaker.
876  if (left.total_size < right.total_size) {
877  return kInlineLeftFirst;
878  } else if (left.total_size > right.total_size) {
879  return kInlineRightFirst;
880  } else {
881  return left.node->id() > right.node->id();
882  }
883  } else {
884  return kInlineLeftFirst;
885  }
886  } else if (left.frequency.IsUnknown()) {
887  return kInlineRightFirst;
888  }
889 
890  float left_score = left.frequency.value() / left.total_size;
891  float right_score = right.frequency.value() / right.total_size;
892 
893  if (left_score > right_score) {
894  return kInlineLeftFirst;
895  } else if (left_score < right_score) {
896  return kInlineRightFirst;
897  } else {
898  return left.node->id() > right.node->id();
899  }
