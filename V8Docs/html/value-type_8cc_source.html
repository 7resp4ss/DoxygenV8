88 void PrintGenericHeaptypeName(std::ostringstream& buf, ValueTypeBase type) {
89  DCHECK(type.is_generic());
90  if (type.is_nullable()) {
91  GenericKind kind = type.generic_kind();
92  if (kind == GenericKind::kNone) {
93  // The code below would produce "noneref", and we need to keep it
94  // that way for "(ref none)", so we need this special case.
95  buf << (type.is_shared() ? "(shared nullref)" : "nullref");
96  return;
97  } else if (kind == GenericKind::kNoExn) {
98  buf << (type.is_shared() ? "(shared nullexnref)" : "nullexnref");
99  return;
100  } else if (kind == GenericKind::kNoExtern) {
101  buf << (type.is_shared() ? "(shared nullexternref)" : "nullexternref");
102  return;
103  } else if (kind == GenericKind::kNoFunc) {
104  buf << (type.is_shared() ? "(shared nullfuncref)" : "nullfuncref");
105  return;
106  } else if (kind == GenericKind::kNoCont) {
107  buf << (type.is_shared() ? "(shared nullcontref)" : "nullcontref");
108  return;
109  }
110  }
111  buf << type.generic_heaptype_name();

116 std::string ValueTypeBase::name() const {
117  if (is_numeric()) {
118  switch (numeric_kind()) {
119 #define NUMERIC_CASE(kind, log2, code, mtype, shortName, name) \
120  case NumericKind::k##kind: \
121  return name;
122  FOREACH_NUMERIC_VALUE_TYPE(NUMERIC_CASE)
123 #undef NUMERIC_CASE
124  }
125  UNREACHABLE();
126  }
127  std::ostringstream buf;
128  if (has_index()) {
129  buf << "(ref ";
130  if (is_nullable()) buf << "null ";
131  if (is_exact()) buf << "exact ";
132  buf << raw_index().index << ")";
133  } else {
134  DCHECK(is_generic());
135  if (is_nullable()) {
136  GenericKind kind = generic_kind();
137  if (kind == GenericKind::kNone) {
138  // The code below would produce "noneref", and we need to keep it
139  // that way for "(ref none)", so we need this special case.
140  return is_shared() ? "(shared nullref)" : "nullref";
141  } else if (kind == GenericKind::kNoExn) {
142  return is_shared() ? "(shared nullexnref)" : "nullexnref";
143  } else if (kind == GenericKind::kNoExtern) {
144  return is_shared() ? "(shared nullexternref)" : "nullexternref";
145  } else if (kind == GenericKind::kNoFunc) {
146  return is_shared() ? "(shared nullfuncref)" : "nullfuncref";
147  } else if (kind == GenericKind::kNoCont) {
148  return is_shared() ? "(shared nullcontref)" : "nullcontref";
149  }
150  }
151  bool shorthand =
152  (is_nullable() || is_sentinel() || is_string_view()) && !is_exact();
153  bool append_ref = is_nullable() && !is_sentinel() && !is_exact();
154  if (!shorthand) buf << "(ref ";
155  if (!shorthand && is_nullable()) buf << "null ";
156  PrintGenericHeaptypeName(buf, *this);
157  if (append_ref) buf << "ref";
158  if (!shorthand) buf << ")";
159  }
160  return buf.str();
161 }

172 bool EquivalentNumericSig(const CanonicalSig* a, const FunctionSig* b) {
173  if (a->parameter_count() != b->parameter_count()) return false;
174  if (a->return_count() != b->return_count()) return false;
175  base::Vector<const CanonicalValueType> a_types = a->all();
176  base::Vector<const ValueType> b_types = b->all();
177  for (size_t i = 0; i < a_types.size(); i++) {
178  if (!a_types[i].is_numeric()) return false;
179  if (a_types[i].kind() != b_types[i].kind()) return false;
180  }
181  return true;
182 }

200 const wasm::FunctionSig* ReplaceTypeInSig(Zone* zone,
201  const wasm::FunctionSig* sig,
202  wasm::ValueType from,
203  wasm::ValueType to,
204  size_t num_replacements) {
205  size_t param_occurences =
206  std::count(sig->parameters().begin(), sig->parameters().end(), from);
207  size_t return_occurences =
208  std::count(sig->returns().begin(), sig->returns().end(), from);
209  if (param_occurences == 0 && return_occurences == 0) return sig;
210 
211  wasm::FunctionSig::Builder builder(
212  zone, sig->return_count() + return_occurences * (num_replacements - 1),
213  sig->parameter_count() + param_occurences * (num_replacements - 1));
214 
215  for (wasm::ValueType ret : sig->returns()) {
216  if (ret == from) {
217  for (size_t i = 0; i < num_replacements; i++) builder.AddReturn(to);
218  } else {
219  builder.AddReturn(ret);
220  }
221  }
222 
223  for (wasm::ValueType param : sig->parameters()) {
224  if (param == from) {
225  for (size_t i = 0; i < num_replacements; i++) builder.AddParam(to);
226  } else {
227  builder.AddParam(param);
228  }
229  }
230 
231  return builder.Get();
232 }
