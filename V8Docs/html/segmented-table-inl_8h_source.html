61 template <typename Entry, size_t size>
62 void SegmentedTable<Entry, size>::Initialize() {
63  DCHECK(!is_initialized());
64  DCHECK_EQ(vas_, nullptr);
65 
66  VirtualAddressSpace* root_space = GetPlatformVirtualAddressSpace();
67 
68 #ifdef V8_TARGET_ARCH_64_BIT
69  static_assert(kUseContiguousMemory);
70  DCHECK(IsAligned(kReservationSize, root_space->allocation_granularity()));
71 
72  if (root_space->CanAllocateSubspaces()) {
73  auto subspace = root_space->AllocateSubspace(VirtualAddressSpace::kNoHint,
74  kReservationSize, kAlignment,
75  PagePermissions::kReadWrite);
76  vas_ = subspace.release();
77  if (kUseSegmentPool) {
78  segment_pool_grow_mutex_ = new base::Mutex();
79  }
80  } else {
81  // This may be required on old Windows versions that don't support
82  // VirtualAlloc2, which is required for subspaces. In that case, just use a
83  // fully-backed emulated subspace.
84  Address reservation_base = root_space->AllocatePages(
85  VirtualAddressSpace::kNoHint, kReservationSize, kAlignment,
86  PagePermissions::kNoAccess);
87  if (reservation_base) {
88  vas_ = new base::EmulatedVirtualAddressSubspace(
89  root_space, reservation_base, kReservationSize, kReservationSize);
90  }
91  // EmulatedVirtualAddressSubspace does not suppert AllocatePagesArray.
92  DCHECK(!kUseSegmentPool);
93  }
94  if (!vas_) {
95  V8::FatalProcessOutOfMemory(
96  nullptr, "SegmentedTable::InitializeTable (subspace allocation)");
97  }
98 #else // V8_TARGET_ARCH_64_BIT
99  static_assert(!kUseContiguousMemory);
100  vas_ = root_space;
101 #endif
102 
103  base_ = reinterpret_cast<Entry*>(vas_->base());
104 
105  if constexpr (kUseContiguousMemory && kIsWriteProtected) {
106  CHECK(ThreadIsolation::WriteProtectMemory(
107  base(), size, PageAllocator::Permission::kNoAccess));
108  }
109 }

126 template <typename Entry, size_t size>
127 typename SegmentedTable<Entry, size>::FreelistHead
128 SegmentedTable<Entry, size>::InitializeFreeList(Segment segment,
129  uint32_t start_offset) {
130  DCHECK_LT(start_offset, kEntriesPerSegment);
131  uint32_t num_entries = kEntriesPerSegment - start_offset;
132 
133  uint32_t first = segment.first_entry() + start_offset;
134  uint32_t last = segment.last_entry();
135  {
136  WriteIterator it = iter_at(first);
137  while (it.index() != last) {
138  it->MakeFreelistEntry(it.index() + 1);
139  ++it;
140  }
141  it->MakeFreelistEntry(0);
142  }
143 
144  return FreelistHead(first, num_entries);
145 }

147 template <typename Entry, size_t size>
148 std::optional<typename SegmentedTable<Entry, size>::Segment>
149 SegmentedTable<Entry, size>::TryGetSegmentFromPool() {
150  DCHECK(kUseSegmentPool);
151  for (int i = 0; i < static_cast<int>(kSegmentPoolSize); ++i) {
152  uint32_t segment = segment_pool_[i].load(std::memory_order_relaxed);
153  if (segment != 0) {
154  if (segment_pool_[i].compare_exchange_weak(segment, 0,
155  std::memory_order_acq_rel)) {
156  return Segment::At(segment);
157  } else {
158  // Retry if CAS failed. This is needed to ensure the pool really is
159  // empty when this method returns {}.
160  --i;
161  }
162  }
163  }
164  return {};
165 }

167 template <typename Entry, size_t size>
168 std::optional<typename SegmentedTable<Entry, size>::Segment>
169 SegmentedTable<Entry, size>::TryAllocateSegment() {
170  if constexpr (!kUseSegmentPool) {
171  Address start =
172  vas_->AllocatePages(VirtualAddressSpace::kNoHint, kSegmentSize,
173  kSegmentSize, PagePermissions::kReadWrite);
174  if (!start) {
175  return {};
176  }
177  uint32_t offset = static_cast<uint32_t>(start - vas_->base());
178  return Segment::At(offset);
179  }
180 
181  if (auto segment = TryGetSegmentFromPool()) {
182  return segment;
183  }
184 
185  base::MutexGuard guard(*segment_pool_grow_mutex_);
186  // Try again to get a segment from the pool with the mutex held. Either
187  // another thread filled the pool before us and we should get a segment here,
188  // or the pool is guaranteed to be empty and we can proceed to fill it.
189  if (auto segment = TryGetSegmentFromPool()) {
190  return segment;
191  }
192 
193  return FillSegmentsPool(true);
194 }

196 template <typename Entry, size_t size>
197 std::optional<typename SegmentedTable<Entry, size>::Segment>
198 SegmentedTable<Entry, size>::FillSegmentsPool(bool return_a_segment) {
199  std::optional<Segment> res;
200  for (size_t i = 0; i < kSegmentPoolSize; ++i) {
201  DCHECK_EQ(segment_pool_[i].load(std::memory_order_acquire), 0);
202  Address start =
203  vas_->AllocatePages(VirtualAddressSpace::kNoHint, kSegmentSize,
204  kAlignment, PagePermissions::kReadWrite);
205  if (!start) continue;
206  uint32_t offset = static_cast<uint32_t>(start - vas_->base());
207  if (return_a_segment && i == 0) {
208  res.emplace(Segment::At(offset));
209  continue;
210  }
211  DCHECK_NE(offset, 0);
212  segment_pool_[i].store(offset, std::memory_order_release);
213  }
214  return res;
215 }

228 template <typename Entry, size_t size>
229 std::optional<std::pair<typename SegmentedTable<Entry, size>::Segment,
230  typename SegmentedTable<Entry, size>::FreelistHead>>
231 SegmentedTable<Entry, size>::TryAllocateAndInitializeSegment() {
232  auto segment = TryAllocateSegment();
233  if (!segment) return {};
234  DCHECK_IMPLIES(!kUseContiguousMemory,
235  (*segment).number() > kNumReadOnlySegments);
236  FreelistHead freelist = InitializeFreeList(*segment);
237  return {{*segment, freelist}};

240 template <typename Entry, size_t size>
241 void SegmentedTable<Entry, size>::FreeTableSegment(Segment segment) {
242  if (kUseSegmentPool) {
243  // Offset 0 is used as a marker for free entries. Thus we cannot store it in
244  // the pool.
245  if (segment.offset() != 0) {
246  base::MutexGuard guard(*segment_pool_grow_mutex_);
247  for (size_t i = 0; i < kSegmentPoolSize; ++i) {
248  uint32_t cur = segment_pool_[i].load(std::memory_order_relaxed);
249  if (cur == 0) {
250  if (segment_pool_[i].compare_exchange_weak(
251  cur, segment.offset(), std::memory_order_acq_rel)) {
252  return;
253  }
254  }
255  }
256  }
257  }
258  Address segment_start = vas_->base() + segment.offset();
259  vas_->FreePages(segment_start, kSegmentSize);
260 }
