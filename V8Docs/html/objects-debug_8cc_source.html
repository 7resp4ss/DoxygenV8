111 // Heap Verification Overview
112 // --------------------------
113 // - Each InstanceType has a separate XXXVerify method which checks an object's
114 // integrity in isolation.
115 // - --verify-heap will iterate over all gc spaces and call ObjectVerify() on
116 // every encountered tagged pointer.
117 // - Verification should be pushed down to the specific instance type if its
118 // integrity is independent of an outer object.
119 // - In cases where the InstanceType is too generic (e.g. FixedArray) the
120 // XXXVerify of the outer method has to do recursive verification.
121 // - If the corresponding objects have inheritance the parent's Verify method
122 // is called as well.
123 // - For any field containing pointers VerifyPointer(...) should be called.
124 //
125 // Caveats
126 // -------
127 // - Assume that any of the verify methods is incomplete!
128 // - Some integrity checks are only partially done due to objects being in
129 // partially initialized states when a gc happens, for instance when outer
130 // objects are allocated before inner ones.
131 //

413 void BytecodeArray::BytecodeArrayVerify(Isolate* isolate) {
414  ExposedTrustedObjectVerify(isolate);
415 
416  {
417  CHECK(IsSmi(TaggedField<Object>::load(*this, kLengthOffset)));
418  CHECK_LE(0, length());
419  CHECK_LE(length(), kMaxLength);
420  }
421  {
422  auto o = constant_pool();
423  Object::VerifyPointer(isolate, o);
424  CHECK(IsTrustedFixedArray(o));
425  }
426  {
427  auto o = handler_table();
428  Object::VerifyPointer(isolate, o);
429  CHECK(IsTrustedByteArray(o));
430  }
431  {
432  auto o = wrapper();
433  Object::VerifyPointer(isolate, o);
434  CHECK(IsBytecodeWrapper(o));
435  // Our wrapper must point back to us.
436  CHECK_EQ(o->bytecode(isolate), *this);
437  }
438  {
439  // Use the raw accessor here as source positions may not be available.
440  auto o = raw_source_position_table(kAcquireLoad);
441  Object::VerifyPointer(isolate, o);
442  CHECK(o == Smi::zero() || IsTrustedByteArray(o));
443  }
444 
445  // TODO(oth): Walk bytecodes and immediate values to validate sanity.
446  // - All bytecodes are known and well formed.
447  // - Jumps must go to new instructions starts.
448  // - No Illegal bytecodes.
449  // - No consecutive sequences of prefix Wide / ExtraWide.
450  // - String constants for loads should be internalized strings.

507 void JSObject::JSObjectVerify(Isolate* isolate) {
508  TorqueGeneratedClassVerifiers::JSObjectVerify(*this, isolate);
509  VerifyHeapPointer(isolate, elements());
510 
511  CHECK_IMPLIES(HasSloppyArgumentsElements(), IsJSArgumentsObject(*this));
512  if (HasFastProperties()) {
513  int actual_unused_property_fields = map()->GetInObjectProperties() +
514  property_array()->length() -
515  map()->NextFreePropertyIndex();
516  if (map()->UnusedPropertyFields() != actual_unused_property_fields) {
517  // There are two reasons why this can happen:
518  // - in the middle of StoreTransitionStub when the new extended backing
519  // store is already set into the object and the allocation of the
520  // HeapNumber triggers GC while the map isn't updated yet.
521  // - deletion of the last property can leave additional backing store
522  // capacity behind.
523  CHECK_GT(actual_unused_property_fields, map()->UnusedPropertyFields());
524  int delta = actual_unused_property_fields - map()->UnusedPropertyFields();
525  CHECK_EQ(0, delta % JSObject::kFieldsAdded);
526  }
527  Tagged<DescriptorArray> descriptors = map()->instance_descriptors(isolate);
528  bool is_transitionable_fast_elements_kind =
529  IsTransitionableFastElementsKind(map()->elements_kind());
530 
531  for (InternalIndex i : map()->IterateOwnDescriptors()) {
532  PropertyDetails details = descriptors->GetDetails(i);
533  if (details.location() == PropertyLocation::kField) {
534  CHECK_EQ(PropertyKind::kData, details.kind());
535  Representation r = details.representation();
536  FieldIndex index = FieldIndex::ForDetails(map(), details);
537  if (COMPRESS_POINTERS_BOOL && index.is_inobject()) {
538  VerifyObjectField(isolate, index.offset());
539  }
540  Tagged<Object> value = RawFastPropertyAt(index);
541  CHECK_IMPLIES(r.IsDouble(), IsHeapNumber(value));
542  if (IsUninitialized(value, isolate)) continue;
543  CHECK_IMPLIES(r.IsSmi(), IsSmi(value));
544  CHECK_IMPLIES(r.IsHeapObject(), IsHeapObject(value));
545  Tagged<FieldType> field_type = descriptors->GetFieldType(i);
546  bool type_is_none = IsNone(field_type);
547  bool type_is_any = IsAny(field_type);
548  if (r.IsNone()) {
549  CHECK(type_is_none);
550  } else if (r.IsHeapObject()) {
551  CHECK(!type_is_none);
552  if (!type_is_any) {
553  CHECK_IMPLIES(FieldType::NowStable(field_type),
554  map()->is_deprecated() ||
555  FieldType::NowContains(field_type, value));
556  }
557  } else {
558  CHECK(type_is_any);
559  }
560  CHECK_IMPLIES(is_transitionable_fast_elements_kind,

1171  // elements.

1173  }
