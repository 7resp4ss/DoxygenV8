92  // Since this pass changes the representation of Phis, it makes some untagging
93  // operations outdated: if we've decided that a Phi should have Int32
94  // representation, then we don't need to do a kCheckedSmiUntag before using
95  // it. UpdateNodeInputs(n) removes such untagging from {n}'s input (and insert
96  // new conversions if needed, from Int32 to Float64 for instance).
97  template <class NodeT>
98  ProcessResult UpdateNodeInputs(NodeT* n, const ProcessingState* state) {
99  NodeBase* node = static_cast<NodeBase*>(n);
100 
101  ProcessResult result = ProcessResult::kContinue;
102  if (IsUntagging(n->opcode())) {
103  if (node->input(0).node()->Is<Phi>() &&
104  node->input(0).node()->value_representation() !=
105  ValueRepresentation::kTagged) {
106  DCHECK_EQ(node->input_count(), 1);
107  // This untagging conversion is outdated, since its input has been
108  // untagged. Depending on the conversion, it might need to be replaced
109  // by another untagged->untagged conversion, or it might need to be
110  // removed alltogether (or rather, replaced by an identity node).
111  UpdateUntaggingOfPhi(node->input(0).node()->Cast<Phi>(),
112  n->template Cast<ValueNode>());
113  }
114  } else {
115  result = UpdateNonUntaggingNodeInputs(n, state);
116  }
117 
118  return result;
119  }
120 
121  template <class NodeT>
122  ProcessResult UpdateNonUntaggingNodeInputs(NodeT* n,
123  const ProcessingState* state) {
124  NodeBase* node = static_cast<NodeBase*>(n);
125 
126  // It would be bad to re-tag the input of an untagging node, so this
127  // function should never be called on untagging nodes.
128  DCHECK(!IsUntagging(n->opcode()));
129 
130  for (int i = 0; i < n->input_count(); i++) {
131  ValueNode* input = node->input(i).node();
132  if (input->Is<Identity>()) {
133  // Bypassing the identity
134  node->change_input(i, input->input(0).node());
135  } else if (Phi* phi = input->TryCast<Phi>()) {
136  // If the input is a Phi and it was used without any untagging, then
137  // we need to retag it (with some additional checks/changes for some
138  // nodes, cf the overload of UpdateNodePhiInput).
139  ProcessResult result = UpdateNodePhiInput(n, phi, i, state);
140  if (V8_UNLIKELY(result == ProcessResult::kRemove)) {
141  return ProcessResult::kRemove;
142  }
143  }
144  }
145 
146  return ProcessResult::kContinue;
147  }
