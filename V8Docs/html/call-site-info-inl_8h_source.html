40 Tagged<HeapObject> CallSiteInfo::code_object(IsolateForSandbox isolate) const {
41  DCHECK(!IsTrustedPointerFieldEmpty(kCodeObjectOffset));
42  // The field can contain either a Code or a BytecodeArray, so we need to use
43  // the kUnknownIndirectPointerTag. Since we can then no longer rely on the
44  // type-checking mechanism of trusted pointers we need to perform manual type
45  // checks afterwards.
46  Tagged<HeapObject> code_object =
47  ReadTrustedPointerField<kUnknownIndirectPointerTag>(kCodeObjectOffset,
48  isolate);
49  CHECK(IsCode(code_object) || IsBytecodeArray(code_object));
50  return code_object;
51 }

53 void CallSiteInfo::set_code_object(Tagged<HeapObject> code,
54  WriteBarrierMode mode) {
55  DCHECK(IsCode(code) || IsBytecodeArray(code) || IsUndefined(code));
56  if (IsCode(code) || IsBytecodeArray(code)) {
57  WriteTrustedPointerField<kUnknownIndirectPointerTag>(
58  kCodeObjectOffset, Cast<ExposedTrustedObject>(code));
59  CONDITIONAL_TRUSTED_POINTER_WRITE_BARRIER(
60  *this, kCodeObjectOffset, kUnknownIndirectPointerTag, code, mode);
61  } else {
62  DCHECK(IsUndefined(code));
63  ClearTrustedPointerField(kCodeObjectOffset);
64  }
65 }
