38 enum class OptimizationReason : uint8_t {
39 #define OPTIMIZATION_REASON_CONSTANTS(Constant, message) k##Constant,
40  OPTIMIZATION_REASON_LIST(OPTIMIZATION_REASON_CONSTANTS)
41 #undef OPTIMIZATION_REASON_CONSTANTS
42 };

44 char const* OptimizationReasonToString(OptimizationReason reason) {
45  static char const* reasons[] = {
46 #define OPTIMIZATION_REASON_TEXTS(Constant, message) message,
47  OPTIMIZATION_REASON_LIST(OPTIMIZATION_REASON_TEXTS)
48 #undef OPTIMIZATION_REASON_TEXTS
49  };
50  size_t const index = static_cast<size_t>(reason);
51  DCHECK_LT(index, arraysize(reasons));
52  return reasons[index];
53 }

55 #undef OPTIMIZATION_REASON_LIST

57 std::ostream& operator<<(std::ostream& os, OptimizationReason reason) {
58  return os << OptimizationReasonToString(reason);

61 class OptimizationDecision {
62  public:
63  static constexpr OptimizationDecision Maglev() {
64  // TODO(v8:7700): Consider using another reason here.
65  return {OptimizationReason::kHotAndStable, CodeKind::MAGLEV,
66  ConcurrencyMode::kConcurrent};
67  }
68  static constexpr OptimizationDecision TurbofanHotAndStable() {
69  return {OptimizationReason::kHotAndStable, CodeKind::TURBOFAN_JS,
70  ConcurrencyMode::kConcurrent};
71  }
72  static constexpr OptimizationDecision DoNotOptimize() {
73  return {OptimizationReason::kDoNotOptimize,
74  // These values don't matter but we have to pass something.
75  CodeKind::TURBOFAN_JS, ConcurrencyMode::kConcurrent};
76  }
77 
78  constexpr bool should_optimize() const {
79  return optimization_reason != OptimizationReason::kDoNotOptimize;
80  }
81 
82  OptimizationReason optimization_reason;
83  CodeKind code_kind;
84  ConcurrencyMode concurrency_mode;
85 
86  private:
87  OptimizationDecision() = default;
88  constexpr OptimizationDecision(OptimizationReason optimization_reason,
89  CodeKind code_kind,
90  ConcurrencyMode concurrency_mode)
91  : optimization_reason(optimization_reason),
92  code_kind(code_kind),
93  concurrency_mode(concurrency_mode) {}
94 };

156 // Returns true when |function| should be enqueued for sparkplug compilation for
157 // the first time.
158 bool FirstTimeTierUpToSparkplug(Isolate* isolate, Tagged<JSFunction> function) {
159  return !function->has_feedback_vector() ||
160  // We request sparkplug even in the presence of a fbv, if we are
161  // running ignition and haven't enqueued the function for sparkplug
162  // batch compilation yet. This ensures we tier-up to sparkplug when the
163  // feedback vector is allocated eagerly (e.g. for logging function
164  // events; see JSFunction::InitializeFeedbackCell()).
165  (function->ActiveTierIsIgnition(isolate) &&
166  CanCompileWithBaseline(isolate, function->shared()) &&
167  function->shared()->cached_tiering_decision() ==
168  CachedTieringDecision::kPending);
169 }

180 int InterruptBudgetFor(Isolate* isolate, std::optional<CodeKind> code_kind,
181  Tagged<JSFunction> function,
182  CachedTieringDecision cached_tiering_decision,
183  int bytecode_length) {
184  // Avoid interrupts while we're already tiering.
185  if (function->tiering_in_progress()) return INT_MAX / 2;
186 
187  const std::optional<CodeKind> existing_request =
188  function->GetRequestedOptimizationIfAny(isolate);
189  if (existing_request == CodeKind::TURBOFAN_JS ||
190  (code_kind.has_value() && code_kind.value() == CodeKind::TURBOFAN_JS)) {
191  return v8_flags.invocation_count_for_osr * bytecode_length;
192  }
193  if (maglev::IsMaglevOsrEnabled() && existing_request == CodeKind::MAGLEV) {
194  return v8_flags.invocation_count_for_maglev_osr * bytecode_length;
195  }
196 
197  if (TiersUpToMaglev(code_kind) &&
198  !function->IsTieringRequestedOrInProgress()) {
199  if (v8_flags.profile_guided_optimization) {
200  switch (cached_tiering_decision) {
201  case CachedTieringDecision::kDelayMaglev:
202  return (std::max(v8_flags.invocation_count_for_maglev,
203  v8_flags.minimum_invocations_after_ic_update) +
204  v8_flags.invocation_count_for_maglev_with_delay) *
205  bytecode_length;
206  case CachedTieringDecision::kEarlyMaglev:
207  case CachedTieringDecision::kEarlyTurbofan:
208  return v8_flags.invocation_count_for_early_optimization *
209  bytecode_length;
210  case CachedTieringDecision::kPending:
211  case CachedTieringDecision::kEarlySparkplug:
212  case CachedTieringDecision::kNormal:
213  return v8_flags.invocation_count_for_maglev * bytecode_length;
214  }
215  // The enum value is coming from inside the sandbox and while the switch
216  // is exhaustive, it's not guaranteed that value is one of the declared
217  // values.
218  SBXCHECK(false);
219  }
220  return v8_flags.invocation_count_for_maglev * bytecode_length;
221  }
222  return v8_flags.invocation_count_for_turbofan * bytecode_length;
223 }

227 // static
228 int TieringManager::InterruptBudgetFor(
229  Isolate* isolate, Tagged<JSFunction> function,
230  std::optional<CodeKind> override_active_tier) {
231  DCHECK(function->shared()->is_compiled());
232  const int bytecode_length =
233  function->shared()->GetBytecodeArray(isolate)->length();
234 
235  if (FirstTimeTierUpToSparkplug(isolate, function)) {
236  return bytecode_length * v8_flags.invocation_count_for_feedback_allocation;
237  }
238 
239  DCHECK(function->has_feedback_vector());
240  if (bytecode_length > v8_flags.max_optimized_bytecode_size) {
241  // Decrease times of interrupt budget underflow, the reason of not setting
242  // to INT_MAX is the interrupt budget may overflow when doing add
243  // operation for forward jump.
244  return INT_MAX / 2;
245  }
246  return ::i::InterruptBudgetFor(
247  isolate,
248  override_active_tier ? override_active_tier
249  : function->GetActiveTier(isolate),
250  function, function->shared()->cached_tiering_decision(), bytecode_length);
251 }

255 void TrySetOsrUrgency(Isolate* isolate, Tagged<JSFunction> function,
256  int osr_urgency) {
257  Tagged<SharedFunctionInfo> shared = function->shared();
258  if (V8_UNLIKELY(!v8_flags.use_osr)) return;
259  if (V8_UNLIKELY(shared->optimization_disabled(
260  (!v8_flags.maglev ||
261  (v8_flags.turbofan && function->ActiveTierIsMaglev(isolate)))
262  ? CodeKind::TURBOFAN_JS
263  : CodeKind::MAGLEV))) {
264  return;
265  }
266 
267  // We've passed all checks - bump the OSR urgency.
268 
269  Tagged<FeedbackVector> fv = function->feedback_vector();
270  if (V8_UNLIKELY(v8_flags.trace_osr)) {
271  CodeTracer::Scope scope(isolate->GetCodeTracer());
272  PrintF(scope.file(),
273  "[OSR - setting osr urgency. function: %s, old urgency: %d, new "
274  "urgency: %d]\n",
275  function->DebugNameCStr().get(), fv->osr_urgency(), osr_urgency);
276  }
277 
278  DCHECK_GE(osr_urgency, fv->osr_urgency()); // Never lower urgency here.
279  fv->set_osr_urgency(osr_urgency);
280 }

282 void TryIncrementOsrUrgency(Isolate* isolate, Tagged<JSFunction> function) {
283  int old_urgency = function->feedback_vector()->osr_urgency();
284  int new_urgency = std::min(old_urgency + 1, FeedbackVector::kMaxOsrUrgency);
285  TrySetOsrUrgency(isolate, function, new_urgency);
286 }

300 void TieringManager::MaybeOptimizeFrame(Tagged<JSFunction> function,
301  CodeKind current_code_kind) {
302  const bool tiering_in_progress = function->tiering_in_progress();
303  const bool osr_in_progress =
304  function->feedback_vector()->osr_tiering_in_progress();
305  // Attenzione! Update this constant in case the condition below changes.
306  static_assert(kTieringStateInProgressBlocksTierup);
307  if (V8_UNLIKELY(tiering_in_progress) || V8_UNLIKELY(osr_in_progress)) {
308  if (v8_flags.concurrent_recompilation_front_running &&
309  ((tiering_in_progress && function->ActiveTierIsMaglev(isolate_)) ||
310  (osr_in_progress &&
311  function->feedback_vector()->maybe_has_optimized_osr_code()))) {
312  // TODO(olivf): In the case of Maglev we tried a queue with two
313  // priorities, but it seems not actually beneficial. More
314  // investigation is needed.
315  isolate_->IncreaseConcurrentOptimizationPriority(CodeKind::TURBOFAN_JS,
316  function->shared());
317  }
318  // Note: This effectively disables further tiering actions (e.g. OSR, or
319  // tiering up into Maglev) for the function while it is being compiled.
320  TraceInOptimizationQueue(function, current_code_kind);
321  return;
322  }
323 
324  if (V8_UNLIKELY(v8_flags.testing_d8_test_runner) &&
325  ManualOptimizationTable::IsMarkedForManualOptimization(isolate_,
326  function)) {
327  TraceHeuristicOptimizationDisallowed(function);
328  return;
329  }
330 
331  if (V8_UNLIKELY(function->shared()->optimization_disabled(
332  current_code_kind == CodeKind::MAGLEV ? CodeKind::TURBOFAN_JS
333  : CodeKind::MAGLEV))) {
334  return;
335  }
336 
337  if (V8_UNLIKELY(v8_flags.always_osr)) {
338  TryRequestOsrAtNextOpportunity(isolate_, function);
339  // Continue below and do a normal optimized compile as well.
340  }
341 
342  const bool maglev_osr = maglev::IsMaglevOsrEnabled();
343  const CodeKinds available_kinds = function->GetAvailableCodeKinds(isolate_);
344  const bool waiting_for_tierup =
345  (current_code_kind < CodeKind::TURBOFAN_JS &&
346  (available_kinds & CodeKindFlag::TURBOFAN_JS)) ||
347  (maglev_osr && current_code_kind < CodeKind::MAGLEV &&
348  (available_kinds & CodeKindFlag::MAGLEV));
349  // Baseline OSR uses a separate mechanism and must not be considered here,
350  // therefore we limit to kOptimizedJSFunctionCodeKindsMask.
351  if (function->IsOptimizationRequested(isolate_) || waiting_for_tierup) {
352  if (V8_UNLIKELY(maglev_osr && current_code_kind == CodeKind::MAGLEV &&
353  (!v8_flags.osr_from_maglev ||
354  isolate_->EfficiencyModeEnabledForTiering() ||
355  isolate_->BatterySaverModeEnabled()))) {
356  return;
357  }
358 
359  // OSR kicks in only once we've previously decided to tier up, but we are
360  // still in a lower-tier frame (this implies a long-running loop).
361  TryIncrementOsrUrgency(isolate_, function);
362 
363  // Return unconditionally and don't run through the optimization decision
364  // again; we've already decided to tier up previously.
365  return;
366  }
367 
368  const std::optional<CodeKind> existing_request =
369  function->GetRequestedOptimizationIfAny(isolate_);
370  DCHECK(existing_request != CodeKind::TURBOFAN_JS);
371  DCHECK(!function->HasAvailableCodeKind(isolate_, CodeKind::TURBOFAN_JS));
372  OptimizationDecision d =
373  ShouldOptimize(function->feedback_vector(), current_code_kind);
374  // We might be stuck in a baseline frame that wants to tier up to Maglev, but
375  // is in a loop, and can't OSR, because Maglev doesn't have OSR. Allow it to
376  // skip over Maglev by re-checking ShouldOptimize as if we were in Maglev.
377  if (V8_UNLIKELY(!isolate_->EfficiencyModeEnabledForTiering() && !maglev_osr &&
378  d.should_optimize() && d.code_kind == CodeKind::MAGLEV)) {
379  bool is_marked_for_maglev_optimization =
380  existing_request == CodeKind::MAGLEV ||
381  (available_kinds & CodeKindFlag::MAGLEV);
382  if (is_marked_for_maglev_optimization) {
383  d = ShouldOptimize(function->feedback_vector(), CodeKind::MAGLEV);
384  }
385  }
386 
387  if (V8_UNLIKELY(isolate_->EfficiencyModeEnabledForTiering() &&
388  d.code_kind != CodeKind::TURBOFAN_JS)) {
389  d.concurrency_mode = ConcurrencyMode::kSynchronous;
390  }
391 
392  if (d.should_optimize()) Optimize(function, d);
393 }

395 OptimizationDecision TieringManager::ShouldOptimize(
396  Tagged<FeedbackVector> feedback_vector, CodeKind current_code_kind) {
397  Tagged<SharedFunctionInfo> shared = feedback_vector->shared_function_info();
398  if (current_code_kind == CodeKind::TURBOFAN_JS) {
399  return OptimizationDecision::DoNotOptimize();
400  }
401 
402  if (TiersUpToMaglev(current_code_kind) &&
403  shared->PassesFilter(v8_flags.maglev_filter) &&
404  !shared->maglev_compilation_failed()) {
405  if (v8_flags.profile_guided_optimization &&
406  shared->cached_tiering_decision() ==
407  CachedTieringDecision::kEarlyTurbofan) {
408  return OptimizationDecision::TurbofanHotAndStable();
409  }
410  return OptimizationDecision::Maglev();
411  }
412 
413  if (V8_UNLIKELY(!v8_flags.turbofan ||
414  !shared->PassesFilter(v8_flags.turbo_filter) ||
415  (v8_flags.efficiency_mode_disable_turbofan &&
416  isolate_->EfficiencyModeEnabledForTiering()) ||
417  isolate_->BatterySaverModeEnabled())) {
418  return OptimizationDecision::DoNotOptimize();
419  }
420 
421  if (isolate_->EfficiencyModeEnabledForTiering() &&
422  v8_flags.efficiency_mode_delay_turbofan &&
423  feedback_vector->invocation_count() <
424  v8_flags.efficiency_mode_delay_turbofan) {
425  return OptimizationDecision::DoNotOptimize();
426  }
427 
428  Tagged<BytecodeArray> bytecode = shared->GetBytecodeArray(isolate_);
429  if (bytecode->length() > v8_flags.max_optimized_bytecode_size) {
430  return OptimizationDecision::DoNotOptimize();
431  }
432 
433  return OptimizationDecision::TurbofanHotAndStable();
434 }

438 bool ShouldResetInterruptBudgetByICChange(
439  CachedTieringDecision cached_tiering_decision) {
440  switch (cached_tiering_decision) {
441  case CachedTieringDecision::kEarlyMaglev:
442  case CachedTieringDecision::kEarlyTurbofan:
443  return false;
444  case CachedTieringDecision::kPending:
445  case CachedTieringDecision::kEarlySparkplug:
446  case CachedTieringDecision::kDelayMaglev:
447  case CachedTieringDecision::kNormal:
448  return true;
449  }
450  // The enum value is coming from inside the sandbox and while the switch is
451  // exhaustive, it's not guaranteed that value is one of the declared values.
452  SBXCHECK(false);
453 }

457 void TieringManager::NotifyICChanged(Tagged<FeedbackVector> vector) {
458  CodeKind code_kind = vector->shared_function_info()->HasBaselineCode()
459  ? CodeKind::BASELINE
460  : CodeKind::INTERPRETED_FUNCTION;
461 
462 #ifndef V8_ENABLE_LEAPTIERING
463  if (vector->has_optimized_code()) {
464  code_kind = vector->optimized_code(isolate_)->kind();
465  }
466 #endif // !V8_ENABLE_LEAPTIERING
467 
468  if (code_kind == CodeKind::INTERPRETED_FUNCTION &&
469  CanCompileWithBaseline(isolate_, vector->shared_function_info()) &&
470  vector->shared_function_info()->cached_tiering_decision() ==
471  CachedTieringDecision::kPending) {
472  // Don't delay tier-up if we haven't tiered up to baseline yet, but will --
473  // baseline code is feedback independent.
474  return;
475  }
476 
477  OptimizationDecision decision = ShouldOptimize(vector, code_kind);
478  if (decision.should_optimize()) {
479  Tagged<SharedFunctionInfo> shared = vector->shared_function_info();
480  int bytecode_length = shared->GetBytecodeArray(isolate_)->length();
481  Tagged<FeedbackCell> cell = vector->parent_feedback_cell();
482  int invocations = v8_flags.minimum_invocations_after_ic_update;
483  int bytecodes = std::min(bytecode_length, (kMaxInt >> 1) / invocations);
484  int new_budget = invocations * bytecodes;
485  int current_budget = cell->interrupt_budget();
486  if (v8_flags.profile_guided_optimization &&
487  shared->cached_tiering_decision() <=
488  CachedTieringDecision::kEarlySparkplug) {
489  DCHECK_LT(v8_flags.invocation_count_for_early_optimization,
490  FeedbackVector::kInvocationCountBeforeStableDeoptSentinel);
491  if (vector->invocation_count_before_stable() <
492  v8_flags.invocation_count_for_early_optimization) {
493  // Record how many invocation count were consumed before the last IC
494  // change.
495  int new_invocation_count_before_stable;
496  if (vector->interrupt_budget_reset_by_ic_change()) {
497  // Initial interrupt budget is
498  // v8_flags.minimum_invocations_after_ic_update * bytecodes
499  int new_consumed_budget = new_budget - current_budget;
500  new_invocation_count_before_stable =
501  vector->invocation_count_before_stable(kRelaxedLoad) +
502  std::ceil(static_cast<float>(new_consumed_budget) / bytecodes);
503  } else {
504  // Initial interrupt budget is
505  // v8_flags.invocation_count_for_{maglev|turbofan} * bytecodes
506  int total_consumed_budget =
507  (maglev::IsMaglevEnabled()
508  ? v8_flags.invocation_count_for_maglev
509  : v8_flags.invocation_count_for_turbofan) *
510  bytecodes -
511  current_budget;
512  new_invocation_count_before_stable =
513  std::ceil(static_cast<float>(total_consumed_budget) / bytecodes);
514  }
515  if (new_invocation_count_before_stable >=
516  v8_flags.invocation_count_for_early_optimization) {
517  vector->set_invocation_count_before_stable(
518  v8_flags.invocation_count_for_early_optimization, kRelaxedStore);
519  shared->set_cached_tiering_decision(CachedTieringDecision::kNormal);
520  } else {
521  vector->set_invocation_count_before_stable(
522  new_invocation_count_before_stable, kRelaxedStore);
523  }
524  } else {
525  shared->set_cached_tiering_decision(CachedTieringDecision::kNormal);
526  }
527  }
528  if (!v8_flags.profile_guided_optimization ||
529  ShouldResetInterruptBudgetByICChange(
530  shared->cached_tiering_decision())) {
531  if (new_budget > current_budget) {
532  if (v8_flags.trace_opt_verbose) {
533  PrintF("[delaying optimization of %s, IC changed]\n",
534  shared->DebugNameCStr().get());
535  }
536  vector->set_interrupt_budget_reset_by_ic_change(true);
537  cell->set_interrupt_budget(new_budget);
538  }
539  }
540  }

548 void TieringManager::OnInterruptTick(DirectHandle<JSFunction> function,
549  CodeKind code_kind) {
550  IsCompiledScope is_compiled_scope(
551  function->shared()->is_compiled_scope(isolate_));
552 
553  // Remember whether the function had a vector at this point. This is
554  // relevant later since the configuration 'Ignition without a vector' can be
555  // considered a tier on its own. We begin tiering up to tiers higher than
556  // Sparkplug only when reaching this point *with* a feedback vector.
557  const bool had_feedback_vector = function->has_feedback_vector();
558  const bool first_time_tiered_up_to_sparkplug =
559  FirstTimeTierUpToSparkplug(isolate_, *function);
560  // We don't want to trigger GC in the middle of OSR, so do not build a
561  // baseline code for such case.
562  const bool maybe_had_optimized_osr_code =
563  had_feedback_vector &&
564  function->feedback_vector()->maybe_has_optimized_osr_code();
565  const bool compile_sparkplug =
566  CanCompileWithBaseline(isolate_, function->shared()) &&
567  function->ActiveTierIsIgnition(isolate_) && !maybe_had_optimized_osr_code;
568 
569  // Ensure that the feedback vector has been allocated.
570  if (!had_feedback_vector) {
571  if (compile_sparkplug && function->shared()->cached_tiering_decision() ==
572  CachedTieringDecision::kPending) {
573  // Mark the function as compiled with sparkplug before the feedback
574  // vector is created to initialize the interrupt budget for the next
575  // tier.
576  function->shared()->set_cached_tiering_decision(
577  CachedTieringDecision::kEarlySparkplug);
578  }
579  JSFunction::CreateAndAttachFeedbackVector(isolate_, function,
580  &is_compiled_scope);
581  DCHECK(is_compiled_scope.is_compiled());
582  // Also initialize the invocation count here. This is only really needed
583  // for OSR. When we OSR functions with lazy feedback allocation we want to
584  // have a non zero invocation count so we can inline functions.
585  function->feedback_vector()->set_invocation_count(1, kRelaxedStore);
586  }
587 
588  DCHECK(function->has_feedback_vector());
589  DCHECK(function->shared()->is_compiled());
590  DCHECK(function->shared()->HasBytecodeArray());
591 
592  // TODO(jgruber): Consider integrating this into a linear tiering system
593  // controlled by TieringState in which the order is always
594  // Ignition-Sparkplug-Turbofan, and only a single tierup is requested at
595  // once.
596  // It's unclear whether this is possible and/or makes sense - for example,
597  // batching compilation can introduce arbitrary latency between the SP
598  // compile request and fulfillment, which doesn't work with strictly linear
599  // tiering.
600  if (compile_sparkplug) {
601 #ifdef V8_ENABLE_SPARKPLUG
602  if (v8_flags.baseline_batch_compilation) {
603  isolate_->baseline_batch_compiler()->EnqueueFunction(function);
604  } else {
605  IsCompiledScope inner_is_compiled_scope(
606  function->shared()->is_compiled_scope(isolate_));
607  Compiler::CompileBaseline(isolate_, function, Compiler::CLEAR_EXCEPTION,
608  &inner_is_compiled_scope);
609  }
610 #else
611  UNREACHABLE();
612 #endif // V8_ENABLE_SPARKPLUG
613  }
614 
615  // We only tier up beyond sparkplug if we already had a feedback vector.
616  if (first_time_tiered_up_to_sparkplug) {
617  // If we didn't have a feedback vector, the interrupt budget has already
618  // been set by JSFunction::CreateAndAttachFeedbackVector, so no need to
619  // set it again.
620  if (had_feedback_vector) {
621  if (function->shared()->cached_tiering_decision() ==
622  CachedTieringDecision::kPending) {
623  function->shared()->set_cached_tiering_decision(
624  CachedTieringDecision::kEarlySparkplug);
625  }
626  function->SetInterruptBudget(isolate_, BudgetModification::kRaise);
627  }
628  return;
629  }
630 
631  // Don't tier up if Turbofan is disabled.
632  // TODO(jgruber): Update this for a multi-tier world.
633  if (V8_UNLIKELY(!isolate_->use_optimizer())) {
634  function->SetInterruptBudget(isolate_, BudgetModification::kRaise);
635  return;
636  }
637 
638  // --- We've decided to proceed for now. ---
639 
640  DisallowGarbageCollection no_gc;
641  OnInterruptTickScope scope;
642  Tagged<JSFunction> function_obj = *function;
643 
644  MaybeOptimizeFrame(function_obj, code_kind);
645 
646  // Make sure to set the interrupt budget after maybe starting an optimization,
647  // so that the interrupt budget size takes into account tiering state.
648  DCHECK(had_feedback_vector);
649  function->SetInterruptBudget(isolate_, BudgetModification::kRaise);
650 }
651

653 } // namespace v8
