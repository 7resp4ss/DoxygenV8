305 void BaselineCompiler::GenerateCode() {
306  {
307  RCS_BASELINE_SCOPE(PreVisit);
308  // Mark exception handlers as valid indirect jump targets. This is required
309  // when CFI is enabled, to allow indirect jumps into baseline code.
310  HandlerTable table(*bytecode_);
311  for (int i = 0; i < table.NumberOfRangeEntries(); ++i) {
312  MarkIndirectJumpTarget(table.GetRangeHandler(i));
313  }
314  for (; !iterator_.done(); iterator_.Advance()) {
315  PreVisitSingleBytecode();
316  }
317  iterator_.Reset();
318  }
319 
320  // No code generated yet.
321  DCHECK_EQ(__ pc_offset(), 0);
322  __ CodeEntry();
323 
324 #ifdef V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
325  __ AssertInSandboxedExecutionMode();
326 #endif // V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
327 
328  {
329  RCS_BASELINE_SCOPE(Visit);
330  Prologue();
331  AddPosition();
332  for (; !iterator_.done(); iterator_.Advance()) {
333  VisitSingleBytecode();
334  AddPosition();
335  }
336  }
337 }

339 MaybeHandle<Code> BaselineCompiler::Build() {
340  RCS_BASELINE_SCOPE(Build);
341  CodeDesc desc;
342  __ GetCode(local_isolate_, &desc);
343 
344  // Allocate the bytecode offset table.
345  Handle<TrustedByteArray> bytecode_offset_table =
346  bytecode_offset_table_builder_.ToBytecodeOffsetTable(local_isolate_);
347 
348  Factory::CodeBuilder code_builder(local_isolate_, desc, CodeKind::BASELINE);
349  code_builder.set_bytecode_offset_table(bytecode_offset_table);
350  if (shared_function_info_->HasInterpreterData(local_isolate_)) {
351  code_builder.set_interpreter_data(
352  handle(shared_function_info_->interpreter_data(local_isolate_),
353  local_isolate_));
354  } else {
355  code_builder.set_interpreter_data(bytecode_);
356  }
357  code_builder.set_parameter_count(bytecode_->parameter_count());
358  return code_builder.TryBuild();

483 void BaselineCompiler::VisitSingleBytecode() {
484 #ifdef DEBUG
485  effect_state_.clear();
486 #endif
487  int offset = iterator().current_offset();
488  if (IsJumpTarget(offset)) __ Bind(&labels_[offset]);
489  // This is required when CFI is enabled.
490  if (IsIndirectJumpTarget(offset)) {
491  __ JumpTarget();
492  }
493 
494  ASM_CODE_COMMENT_STRING(&masm_, [&]() {
495  std::ostringstream str;
496  iterator().PrintTo(str);
497  return str.str();
498  });
499 
500  VerifyFrame();
501 
502 #ifdef V8_TRACE_UNOPTIMIZED
503  TraceBytecode(Runtime::kTraceUnoptimizedBytecodeEntry);
504 #endif
505 
506  {
507  interpreter::Bytecode bytecode = iterator().current_bytecode();
508 
509 #ifdef DEBUG
510  std::optional<EnsureAccumulatorPreservedScope> accumulator_preserved_scope;
511  // We should make sure to preserve the accumulator whenever the bytecode
512  // isn't registered as writing to it. We can't do this for jumps or switches
513  // though, since the control flow would not match the control flow of this
514  // scope.
515  if (v8_flags.slow_debug_code &&
516  !interpreter::Bytecodes::WritesOrClobbersAccumulator(bytecode) &&
517  !interpreter::Bytecodes::IsJump(bytecode) &&
518  !interpreter::Bytecodes::IsSwitch(bytecode)) {
519  accumulator_preserved_scope.emplace(&basm_);
520  }
521 #endif // DEBUG
522 
523  switch (bytecode) {
524 #define BYTECODE_CASE(name, ...) \
525  case interpreter::Bytecode::k##name: \
526  Visit##name(); \
527  break;
528  BYTECODE_LIST(BYTECODE_CASE, BYTECODE_CASE)
529 #undef BYTECODE_CASE
530  }
531  }
532 
533 #ifdef V8_TRACE_UNOPTIMIZED
534  TraceBytecode(Runtime::kTraceUnoptimizedBytecodeExit);
535 #endif
536 }

540  ASM_CODE_COMMENT(&masm_);
