54 void CompilationCacheRegExp::Age() {
55  static_assert(kGenerations > 1);
56 
57  // Age the generations implicitly killing off the oldest.
58  for (int i = kGenerations - 1; i > 0; i--) {
59  tables_[i] = tables_[i - 1];
60  }
61 
62  // Set the first generation as unborn.
63  tables_[0] = ReadOnlyRoots(isolate()).undefined_value();
64 }

66 void CompilationCacheScript::Age() {
67  DisallowGarbageCollection no_gc;
68  if (IsUndefined(table_, isolate())) return;
69  Tagged<CompilationCacheTable> table = Cast<CompilationCacheTable>(table_);
70 
71  for (InternalIndex entry : table->IterateEntries()) {
72  Tagged<Object> key;
73  if (!table->ToKey(isolate(), entry, &key)) continue;
74  DCHECK(IsWeakFixedArray(key));
75 
76  Tagged<Object> value = table->PrimaryValueAt(entry);
77  if (!IsUndefined(value, isolate())) {
78  Tagged<SharedFunctionInfo> info = Cast<SharedFunctionInfo>(value);
79  // Clear entries after Bytecode was flushed from SFI.
80  if (!info->HasBytecodeArray()) {
81  table->SetPrimaryValueAt(entry,
82  ReadOnlyRoots(isolate()).undefined_value(),
83  SKIP_WRITE_BARRIER);
84  }
85  }
86  }
87 }

89 void CompilationCacheEval::Age() {
90  DisallowGarbageCollection no_gc;
91  if (IsUndefined(table_, isolate())) return;
92  Tagged<CompilationCacheTable> table = Cast<CompilationCacheTable>(table_);
93 
94  for (InternalIndex entry : table->IterateEntries()) {
95  Tagged<Object> key;
96  if (!table->ToKey(isolate(), entry, &key)) continue;
97 
98  DCHECK(IsFixedArray(key));
99  // The ageing mechanism for eval caches.
100  Tagged<SharedFunctionInfo> info =
101  Cast<SharedFunctionInfo>(table->PrimaryValueAt(entry));
102  // Clear entries after Bytecode was flushed from SFI.
103  if (!info->HasBytecodeArray()) {
104  table->RemoveEntry(entry);
105  }
106  }
107 }

135 CompilationCacheScript::LookupResult CompilationCacheScript::Lookup(
136  Handle<String> source, const ScriptDetails& script_details) {
137  DirectHandle<CompilationCacheTable> table = GetTable();
138  LookupResult result = CompilationCacheTable::LookupScript(
139  table, source, script_details, isolate());
140 
141  DirectHandle<Script> script;
142  if (result.script().ToHandle(&script)) {
143  DirectHandle<SharedFunctionInfo> sfi;
144  if (result.toplevel_sfi().ToHandle(&sfi)) {
145  isolate()->counters()->compilation_cache_hits()->Increment();
146  LOG(isolate(), CompilationCacheEvent("hit", "script", *sfi));
147  } else {
148  isolate()->counters()->compilation_cache_partial_hits()->Increment();
149  }
150  } else {
151  isolate()->counters()->compilation_cache_misses()->Increment();
152  }
153  return result;

197 MaybeDirectHandle<RegExpData> CompilationCacheRegExp::Lookup(
198  DirectHandle<String> source, JSRegExp::Flags flags) {
199  HandleScope scope(isolate());
200  // Make sure not to leak the table into the surrounding handle
201  // scope. Otherwise, we risk keeping old tables around even after
202  // having cleared the cache.
203  DirectHandle<Object> result = isolate()->factory()->undefined_value();
204  int generation;
205  for (generation = 0; generation < kGenerations; generation++) {
206  DirectHandle<CompilationCacheTable> table = GetTable(generation);
207  result = table->LookupRegExp(source, flags);
208  if (IsRegExpDataWrapper(*result)) break;
209  }
210  if (IsRegExpDataWrapper(*result)) {
211  Handle<RegExpData> data(Cast<RegExpDataWrapper>(result)->data(isolate()),
212  isolate());
213  if (generation != 0) {
214  Put(source, flags, data);
215  }
216  isolate()->counters()->compilation_cache_hits()->Increment();
217  return scope.CloseAndEscape(data);
218  } else {
219  isolate()->counters()->compilation_cache_misses()->Increment();
220  return MaybeDirectHandle<RegExpData>();
221  }
222 }

262 InfoCellPair CompilationCache::LookupEval(
263  DirectHandle<String> source, DirectHandle<SharedFunctionInfo> outer_info,
264  DirectHandle<Context> context, LanguageMode language_mode, int position) {
265  InfoCellPair result;
266  if (!IsEnabledScriptAndEval()) return result;
267 
268  const char* cache_type;
269 
270  DirectHandle<NativeContext> maybe_native_context;
271  if (TryCast<NativeContext>(context, &maybe_native_context)) {
272  result = eval_global_.Lookup(source, outer_info, maybe_native_context,
273  language_mode, position);
274  cache_type = "eval-global";
275 
276  } else {
277  DCHECK_NE(position, kNoSourcePosition);
278  DirectHandle<NativeContext> native_context(context->native_context(),
279  isolate());
280  result = eval_contextual_.Lookup(source, outer_info, native_context,
281  language_mode, position);
282  cache_type = "eval-contextual";
283  }
284 
285  if (result.has_shared()) {
286  LOG(isolate(), CompilationCacheEvent("hit", cache_type, result.shared()));
287  }
288 
289  return result;
290 }
