25 void LogExecution(Isolate* isolate, DirectHandle<JSFunction> function) {
26  DCHECK(v8_flags.log_function_events);
27  if (!function->has_feedback_vector()) return;
28 #ifdef V8_ENABLE_LEAPTIERING
29  DCHECK(function->IsLoggingRequested(isolate));
30  IsolateGroup::current()->js_dispatch_table()->ResetTieringRequest(
31  function->dispatch_handle());
32 #else
33  if (!function->feedback_vector()->log_next_execution()) return;
34 #endif
35  DirectHandle<SharedFunctionInfo> sfi(function->shared(), isolate);
36  DirectHandle<String> name = SharedFunctionInfo::DebugName(isolate, sfi);
37  DisallowGarbageCollection no_gc;
38  Tagged<SharedFunctionInfo> raw_sfi = *sfi;
39  std::string event_name = "first-execution";
40  CodeKind kind = function->abstract_code(isolate)->kind(isolate);
41  // Not adding "-interpreter" for tooling backwards compatibility.
42  if (kind != CodeKind::INTERPRETED_FUNCTION) {
43  event_name += "-";
44  event_name += CodeKindToString(kind);
45  }
46  LOG(isolate, FunctionEvent(
47  event_name.c_str(), Cast<Script>(raw_sfi->script())->id(), 0,
48  raw_sfi->StartPosition(), raw_sfi->EndPosition(), *name));
49 #ifndef V8_ENABLE_LEAPTIERING
50  function->feedback_vector()->set_log_next_execution(false);
51 #endif // !V8_ENABLE_LEAPTIERING
52 }

55 RUNTIME_FUNCTION(Runtime_CompileLazy) {
56  HandleScope scope(isolate);
57  DCHECK_EQ(1, args.length());
58  DirectHandle<JSFunction> function = args.at<JSFunction>(0);
59  StackLimitCheck check(isolate);
60  if (V8_UNLIKELY(
61  check.JsHasOverflowed(kStackSpaceRequiredForCompilation * KB))) {
62  return isolate->StackOverflow();
63  }
64 
65  DirectHandle<SharedFunctionInfo> sfi(function->shared(), isolate);
66 
67  DCHECK(!function->is_compiled(isolate));
68 #ifdef DEBUG
69  if (v8_flags.trace_lazy && sfi->is_compiled()) {
70  PrintF("[unoptimized: %s]\n", function->DebugNameCStr().get());
71  }
72 #endif
73  IsCompiledScope is_compiled_scope;
74  if (!Compiler::Compile(isolate, function, Compiler::KEEP_EXCEPTION,
75  &is_compiled_scope)) {
76  return ReadOnlyRoots(isolate).exception();
77  }
78 #ifndef V8_ENABLE_LEAPTIERING
79  if (V8_UNLIKELY(v8_flags.log_function_events)) {
80  LogExecution(isolate, function);
81  }
82 #endif // !V8_ENABLE_LEAPTIERING
83  DCHECK(function->is_compiled(isolate));
84  return function->code(isolate);
85 }

150 void CompileOptimized(DirectHandle<JSFunction> function, ConcurrencyMode mode,
151  CodeKind target_kind, Isolate* isolate) {
152  // Ensure that the tiering request is reset even if compilation fails.
153  function->ResetTieringRequests();
154 
155  // As a pre- and post-condition of CompileOptimized, the function *must* be
156  // compiled, i.e. the installed InstructionStream object must not be
157  // CompileLazy.
158  IsCompiledScope is_compiled_scope(function->shared(), isolate);
159  if (V8_UNLIKELY(!is_compiled_scope.is_compiled())) {
160  // This happens if the code is flushed while we still have an optimization
161  // request pending (or if manually an optimization is requested on an
162  // uncompiled function).
163  // Instead of calling into Compiler::Compile and having to do exception
164  // handling here, we reset and return and thus tail-call into CompileLazy.
165  function->ResetIfCodeFlushed(isolate);
166  return;
167  }
168 
169  if (mode == ConcurrencyMode::kConcurrent) {
170  // No need to start another compile job.
171  // Also, various fuzzing flags like --always-turbofan might already compile
172  // this function in the above Compiler::Compile function.
173  if (function->tiering_in_progress() ||
174  function->GetActiveTier(isolate) >= target_kind) {
175  static_assert(kTieringStateInProgressBlocksTierup);
176  function->SetInterruptBudget(isolate, BudgetModification::kRaise);
177  return;
178  }
179  }
180 
181  // Concurrent optimization runs on another thread, thus no additional gap.
182  const int gap =
183  IsConcurrent(mode) ? 0 : kStackSpaceRequiredForCompilation * KB;
184  StackLimitCheck check(isolate);
185  if (check.JsHasOverflowed(gap)) return;
186 
187  Compiler::CompileOptimized(isolate, function, mode, target_kind);
188 
189  DCHECK(function->is_compiled(isolate));
190 }

235  HandleScope scope(isolate);
236  DCHECK_EQ(2, args.length());
237  DirectHandle<JSFunction> function = args.at<JSFunction>(0);
238  bool reoptimize = (*args.at<Smi>(1)).value();
239 
240  IsCompiledScope is_compiled_scope(function->shared(), isolate);
241  if (!is_compiled_scope.is_compiled()) {
242  StackLimitCheck check(isolate);
243  if (V8_UNLIKELY(
244  check.JsHasOverflowed(kStackSpaceRequiredForCompilation * KB))) {
245  return isolate->StackOverflow();
246  }
247  if (!Compiler::Compile(isolate, function, Compiler::KEEP_EXCEPTION,
248  &is_compiled_scope)) {
249  return ReadOnlyRoots(isolate).exception();

251  // In case this code was flushed we should not re-optimize it too quickly.
252  reoptimize = false;

254 
255  function->ResetTieringRequests();
256  if (reoptimize) {
257  // Set the budget such that we have one invocation which allows us to detect
258  // if any ICs need updating before re-optimization.
259  function->raw_feedback_cell()->set_interrupt_budget(1);
260  } else {
261  function->SetInterruptBudget(isolate, BudgetModification::kRaise,
262  CodeKind::INTERPRETED_FUNCTION);
263  }
264  return ReadOnlyRoots(isolate).undefined_value();

269 RUNTIME_FUNCTION(Runtime_CompileOptimized) {
270  HandleScope scope(isolate);
271  DCHECK_EQ(1, args.length());
272  Handle<JSFunction> function = args.at<JSFunction>(0);
273 
274  CodeKind target_kind;
275  ConcurrencyMode mode;
276  DCHECK(function->has_feedback_vector());
277  switch (function->tiering_state()) {
278  case TieringState::kRequestMaglev_Synchronous:
279  target_kind = CodeKind::MAGLEV;
280  mode = ConcurrencyMode::kSynchronous;
281  break;
282  case TieringState::kRequestMaglev_Concurrent:
283  target_kind = CodeKind::MAGLEV;
284  mode = ConcurrencyMode::kConcurrent;
285  break;
286  case TieringState::kRequestTurbofan_Synchronous:
287  target_kind = CodeKind::TURBOFAN_JS;
288  mode = ConcurrencyMode::kSynchronous;
289  break;
290  case TieringState::kRequestTurbofan_Concurrent:
291  target_kind = CodeKind::TURBOFAN_JS;
292  mode = ConcurrencyMode::kConcurrent;
293  break;
294  case TieringState::kNone:
295  case TieringState::kInProgress:
296  UNREACHABLE();
297  }
298 
299  // As a pre- and post-condition of CompileOptimized, the function *must* be
300  // compiled, i.e. the installed InstructionStream object must not be
301  // CompileLazy.
302  IsCompiledScope is_compiled_scope(function->shared(), isolate);
303  DCHECK(is_compiled_scope.is_compiled());
304 
305  StackLimitCheck check(isolate);
306  // Concurrent optimization runs on another thread, thus no additional gap.
307  const int gap =
308  IsConcurrent(mode) ? 0 : kStackSpaceRequiredForCompilation * KB;
309  if (check.JsHasOverflowed(gap)) return isolate->StackOverflow();
310 
311  Compiler::CompileOptimized(isolate, function, mode, target_kind);
312 
313  DCHECK(function->is_compiled(isolate));
314  if (V8_UNLIKELY(v8_flags.log_function_events)) {
315  LogExecution(isolate, function);
316  }
317  return function->code(isolate);
318 }

350 RUNTIME_FUNCTION(Runtime_InstantiateAsmJs) {
351  HandleScope scope(isolate);
352  DCHECK_EQ(args.length(), 4);
353  DirectHandle<JSFunction> function = args.at<JSFunction>(0);
354 
355  DirectHandle<JSReceiver> stdlib;
356  if (IsJSReceiver(args[1])) {
357  stdlib = args.at<JSReceiver>(1);
358  }
359  DirectHandle<JSReceiver> foreign;
360  if (IsJSReceiver(args[2])) {
361  foreign = args.at<JSReceiver>(2);
362  }
363  DirectHandle<JSArrayBuffer> memory;
364  if (IsJSArrayBuffer(args[3])) {
365  memory = args.at<JSArrayBuffer>(3);
366  }
367  DirectHandle<SharedFunctionInfo> shared(function->shared(), isolate);
368 #if V8_ENABLE_WEBASSEMBLY
369  if (shared->HasAsmWasmData()) {
370  DirectHandle<AsmWasmData> data(shared->asm_wasm_data(), isolate);
371  MaybeDirectHandle<Object> result = AsmJs::InstantiateAsmWasm(
372  isolate, shared, data, stdlib, foreign, memory);
373  if (!result.is_null()) {
374  isolate->counters()->asmjs_instantiate_result()->AddSample(
375  kAsmJsInstantiateSuccess);
376  return *result.ToHandleChecked();
377  }
378  if (isolate->has_exception()) {
379  // If instantiation fails, we do not propagate the exception but instead
380  // fall back to JS execution. The only exception (to that rule) is the
381  // termination exception.
382  DCHECK(isolate->is_execution_terminating());
383  return ReadOnlyRoots{isolate}.exception();
384  }
385  isolate->counters()->asmjs_instantiate_result()->AddSample(
386  kAsmJsInstantiateFail);
387 
388  // Remove wasm data, mark as broken for asm->wasm, replace AsmWasmData on
389  // the SFI with UncompiledData and set entrypoint to CompileLazy builtin,
390  // and return a smi 0 to indicate failure.
391  SharedFunctionInfo::DiscardCompiled(isolate, shared);
392  }
393  shared->set_is_asm_wasm_broken(true);
394 #endif
395  DCHECK_EQ(function->code(isolate), *BUILTIN_CODE(isolate, InstantiateAsmJs));
396  function->UpdateCode(isolate, *BUILTIN_CODE(isolate, CompileLazy));
397  DCHECK(!isolate->has_exception());
398  return Smi::zero();
399 }

403 bool TryGetOptimizedOsrCode(Isolate* isolate, Tagged<FeedbackVector> vector,
404  const interpreter::BytecodeArrayIterator& it,
405  Tagged<Code>* code_out) {
406  std::optional<Tagged<Code>> maybe_code =
407  vector->GetOptimizedOsrCode(isolate, {}, it.GetSlotOperand(2));
408  if (maybe_code.has_value()) {
409  *code_out = maybe_code.value();
410  return true;
411  }
412  return false;
413 }
414 
415 // Deoptimize all osr'd loops which contain a particular deopt exit.
416 // We only deoptimize OSR code if the exit is inside the actual loop. The OSR
417 // code might still be affected (if the exit is after the loop, or in the case
418 // of turbofan if the exit is before the loop, but reachable through an outer
419 // loop). Regardless in both of these two cases, the OSR code is still usable to
420 // quickly get out of the loop itself.
421 bool DeoptAllOsrLoopsContainingDeoptExit(Isolate* isolate,
422  Tagged<JSFunction> function,
423  BytecodeOffset deopt_exit_offset) {
424  DisallowGarbageCollection no_gc;
425  DCHECK(!deopt_exit_offset.IsNone());
426 
427  if (!v8_flags.use_ic ||
428  !function->feedback_vector()->maybe_has_optimized_osr_code()) {
429  return false;
430  }
431  Handle<BytecodeArray> bytecode_array(
432  function->shared()->GetBytecodeArray(isolate), isolate);
433  DCHECK(interpreter::BytecodeArrayIterator::IsValidOffset(
434  bytecode_array, deopt_exit_offset.ToInt()));
435 
436  Tagged<FeedbackVector> vector = function->feedback_vector();
437  Tagged<Code> code;
438 
439  bool any_marked = false;
440  bool has_maglev_code = false;
441  bool has_turbofan_code = false;
442 
443  interpreter::BytecodeArrayIterator it(bytecode_array);
444  for (; !it.done(); it.Advance()) {
445  // We're only interested in loop ranges.
446  if (it.current_bytecode() != interpreter::Bytecode::kJumpLoop) continue;
447  bool has_osr_code = TryGetOptimizedOsrCode(isolate, vector, it, &code);
448  if (has_osr_code) {
449  if (base::IsInRange(deopt_exit_offset.ToInt(), it.GetJumpTargetOffset(),
450  it.current_offset())) {
451  code->SetMarkedForDeoptimization(isolate,
452  LazyDeoptimizeReason::kEagerDeopt);
453  any_marked = true;
454  } else {
455  has_turbofan_code |= code->is_turbofanned();
456  has_maglev_code |= code->is_maglevved();
457  }
458  }
459  }
460  Tagged<FeedbackVector> fbv = function->feedback_vector();
461  if (!has_maglev_code && fbv->maybe_has_maglev_osr_code()) {
462  fbv->set_maybe_has_optimized_osr_code(false, CodeKind::MAGLEV);
463  }
464  if (!has_turbofan_code && fbv->maybe_has_turbofan_osr_code()) {
465  fbv->set_maybe_has_optimized_osr_code(false, CodeKind::TURBOFAN_JS);
466  }
467  return any_marked;
468 }

472 RUNTIME_FUNCTION(Runtime_NotifyDeoptimized) {
473  HandleScope scope(isolate);
474  DCHECK_EQ(0, args.length());
475  Deoptimizer* deoptimizer = Deoptimizer::Grab(isolate);
476  DCHECK(CodeKindCanDeoptimize(deoptimizer->compiled_code()->kind()));
477  DCHECK(AllowGarbageCollection::IsAllowed());
478  DCHECK(isolate->context().is_null());
479 
480  TimerEventScope<TimerEventDeoptimizeCode> timer(isolate);
481  TRACE_EVENT0("v8", "V8.DeoptimizeCode");
482  DirectHandle<JSFunction> function = deoptimizer->function();
483  // For OSR the optimized code isn't installed on the function, so get the
484  // code object from deoptimizer.
485  DirectHandle<Code> optimized_code = deoptimizer->compiled_code();
486  const DeoptimizeKind deopt_kind = deoptimizer->deopt_kind();
487  const DeoptimizeReason deopt_reason =
488  deoptimizer->GetDeoptInfo().deopt_reason;
489 
490  // TODO(turbofan): We currently need the native context to materialize
491  // the arguments object, but only to get to its map.
492  isolate->set_context(deoptimizer->function()->native_context());
493 
494  // When this is called from WasmGC code, clear the "thread in wasm" flag,
495  // which is important in case any GC needs to happen.
496  // TODO(40192807): Find a better fix, likely by replacing the global flag.
497  SaveAndClearThreadInWasmFlag clear_wasm_flag(isolate);
498 
499  // Make sure to materialize objects before causing any allocation.
500  deoptimizer->MaterializeHeapObjects();
501  deoptimizer->ProcessDeoptReason(deopt_reason);
502  const BytecodeOffset deopt_exit_offset =
503  deoptimizer->bytecode_offset_in_outermost_frame();
504  delete deoptimizer;
505 
506  // Ensure the context register is updated for materialized objects.
507  JavaScriptStackFrameIterator top_it(isolate);
508  JavaScriptFrame* top_frame = top_it.frame();
509  isolate->set_context(Cast<Context>(top_frame->context()));
510 
511  // Lazy deopts don't invalidate the underlying optimized code since the code
512  // object itself is still valid (as far as we know); the called function
513  // caused the deopt, not the function we're currently looking at.
514  if (deopt_kind == DeoptimizeKind::kLazy) {
515  return ReadOnlyRoots(isolate).undefined_value();
516  }
517 
518  // Some eager deopts also don't invalidate InstructionStream (e.g. when
519  // preparing for OSR from Maglev to Turbofan).
520  if (IsDeoptimizationWithoutCodeInvalidation(deopt_reason)) {
521  if (deopt_reason == DeoptimizeReason::kPrepareForOnStackReplacement &&
522  function->ActiveTierIsMaglev(isolate)) {
523  isolate->tiering_manager()->MarkForTurboFanOptimization(*function);
524  }
525  return ReadOnlyRoots(isolate).undefined_value();
526  }
527 
528  DCHECK_NE(deopt_reason, DeoptimizeReason::kOSREarlyExit);
529  DCHECK_NE(deopt_reason, DeoptimizeReason::kPrepareForOnStackReplacement);
530 
531  // Non-OSR'd code is deoptimized unconditionally. If the deoptimization occurs
532  // inside the outermost loop containing a loop that can trigger OSR
533  // compilation, we remove the OSR code, it will avoid hit the out of date OSR
534  // code and soon later deoptimization.
535  //
536  // For OSR'd code, we keep the optimized code around if deoptimization occurs
537  // outside the outermost loop containing the loop that triggered OSR
538  // compilation. The reasoning is that OSR is intended to speed up the
539  // long-running loop; so if the deoptimization occurs outside this loop it is
540  // still worth jumping to the OSR'd code on the next run. The reduced cost of
541  // the loop should pay for the deoptimization costs.
542 
543  const BytecodeOffset osr_offset = optimized_code->osr_offset();
544  bool any_marked = false;
545 
546  if (osr_offset.IsNone() || Deoptimizer::DeoptExitIsInsideOsrLoop(
547  isolate, *function, deopt_exit_offset,
548  osr_offset, optimized_code->kind())) {
549  function->ResetTieringRequests();
550  if (!optimized_code->marked_for_deoptimization()) {
551  optimized_code->SetMarkedForDeoptimization(
552  isolate, LazyDeoptimizeReason::kEagerDeopt);
553  any_marked = true;
554  }
555  }
556 
557  if (osr_offset.IsNone()) {
558  if (DeoptAllOsrLoopsContainingDeoptExit(isolate, *function,
559  deopt_exit_offset)) {
560  any_marked = true;
561  }
562  }
563 
564  if (any_marked) {
565  Deoptimizer::DeoptimizeMarkedCode(isolate);
566  }
567 
568  return ReadOnlyRoots(isolate).undefined_value();
569 }

618 Tagged<Object> CompileOptimizedOSR(Isolate* isolate,
619  DirectHandle<JSFunction> function,
620  CodeKind min_opt_level,
621  BytecodeOffset osr_offset) {
622  ConcurrencyMode mode =
623  V8_LIKELY(isolate->concurrent_recompilation_enabled() &&
624  v8_flags.concurrent_osr)
625  ? ConcurrencyMode::kConcurrent
626  : ConcurrencyMode::kSynchronous;
627 
628  if (V8_UNLIKELY(isolate->EfficiencyModeEnabledForTiering() &&
629  min_opt_level == CodeKind::MAGLEV)) {
630  mode = ConcurrencyMode::kSynchronous;
631  }
632 
633  DirectHandle<Code> result;
634  if (!Compiler::CompileOptimizedOSR(
635  isolate, function, osr_offset, mode,
636  (maglev::IsMaglevOsrEnabled() && min_opt_level == CodeKind::MAGLEV)
637  ? CodeKind::MAGLEV
638  : CodeKind::TURBOFAN_JS)
639  .ToHandle(&result) ||
640  result->marked_for_deoptimization()) {
641  // An empty result can mean one of two things:
642  // 1) we've started a concurrent compilation job - everything is fine.
643  // 2) synchronous compilation failed for some reason.
644 
645 #ifndef V8_ENABLE_LEAPTIERING
646  if (!function->HasAttachedOptimizedCode(isolate)) {
647  function->UpdateCode(isolate, function->shared()->GetCode(isolate));
648  }
649 #endif // V8_ENABLE_LEAPTIERING
650 
651  return Smi::zero();
652  }
653 
654  DCHECK(!result.is_null());
655  DCHECK(result->is_turbofanned() || result->is_maglevved());
656  DCHECK(CodeKindIsOptimizedJSFunction(result->kind()));
657 
658 #ifdef DEBUG
659  Tagged<DeoptimizationData> data =
660  Cast<DeoptimizationData>(result->deoptimization_data());
661  DCHECK_EQ(BytecodeOffset(data->OsrBytecodeOffset().value()), osr_offset);
662  DCHECK_GE(data->OsrPcOffset().value(), 0);
663 #endif // DEBUG
664 
665  // First execution logging happens in LogOrTraceOptimizedOSREntry
666  return *result;
667 }

685 Tagged<Object> CompileOptimizedOSRFromMaglev(Isolate* isolate,
686  DirectHandle<JSFunction> function,
687  BytecodeOffset osr_offset) {
688  // This path is only relevant for tests (all production configurations enable
689  // concurrent OSR). It's quite subtle, if interested read on:
690  if (V8_UNLIKELY(!isolate->concurrent_recompilation_enabled() ||
691  !v8_flags.concurrent_osr)) {
692  // - Synchronous Turbofan compilation may trigger lazy deoptimization (e.g.
693  // through compilation dependency finalization actions).
694  // - Maglev (currently) disallows marking an opcode as both can_lazy_deopt
695  // and can_eager_deopt.
696  // - Maglev's JumpLoop opcode (the logical caller of this runtime function)
697  // is marked as can_eager_deopt since OSR'ing to Turbofan involves
698  // deoptimizing to Ignition under the hood.
699  // - Thus this runtime function *must not* trigger a lazy deopt, and
700  // therefore cannot trigger synchronous Turbofan compilation (see above).
701  //
702  // We solve this synchronous OSR case by bailing out early to Ignition, and
703  // letting it handle OSR. How do we trigger the early bailout? Returning
704  // any non-null InstructionStream from this function triggers the deopt in
705  // JumpLoop.
706  if (v8_flags.trace_osr) {
707  CodeTracer::Scope scope(isolate->GetCodeTracer());
708  PrintF(scope.file(),
709  "[OSR - Tiering from Maglev to Turbofan failed because "
710  "concurrent_osr is disabled. function: %s, osr offset: %d]\n",
711  function->DebugNameCStr().get(), osr_offset.ToInt());
712  }
713  return Smi::zero();
714  }
715 
716  if (V8_UNLIKELY(isolate->EfficiencyModeEnabledForTiering() ||
717  isolate->BatterySaverModeEnabled())) {
718  function->feedback_vector()->reset_osr_urgency();
719  function->SetInterruptBudget(isolate, BudgetModification::kRaise);
720  return Smi::zero();
721  }
722 
723  return CompileOptimizedOSR(isolate, function, CodeKind::TURBOFAN_JS,
724  osr_offset);
725 }

790 static Tagged<Object> CompileGlobalEval(
791  Isolate* isolate, Handle<i::Object> source_object,
792  DirectHandle<SharedFunctionInfo> outer_info, LanguageMode language_mode,
793  int eval_scope_info_index, int eval_position) {
794  DirectHandle<NativeContext> native_context = isolate->native_context();
795 
796  // Check if native context allows code generation from
797  // strings. Throw an exception if it doesn't.
798  MaybeDirectHandle<String> source;
799  bool unknown_object;
800  std::tie(source, unknown_object) = Compiler::ValidateDynamicCompilationSource(
801  isolate, native_context, source_object);
802  // If the argument is an unhandled string time, bounce to GlobalEval.
803  if (unknown_object) {
804  return native_context->global_eval_fun();
805  }
806  if (source.is_null()) {
807  Handle<Object> error_message =
808  native_context->ErrorMessageForCodeGenerationFromStrings();
809  DirectHandle<Object> error;
810  MaybeDirectHandle<Object> maybe_error = isolate->factory()->NewEvalError(
811  MessageTemplate::kCodeGenFromStrings, error_message);
812  if (maybe_error.ToHandle(&error)) isolate->Throw(*error);
813  return ReadOnlyRoots(isolate).exception();
814  }
815 
816  // Deal with a normal eval call with a string argument. Compile it
817  // and return the compiled function bound in the local context.
818  static const ParseRestriction restriction = NO_PARSE_RESTRICTION;
819  DirectHandle<JSFunction> compiled;
820  DirectHandle<Context> context(isolate->context(), isolate);
821  if (!Is<NativeContext>(*context) && v8_flags.reuse_scope_infos) {
822  Tagged<WeakFixedArray> array = Cast<Script>(outer_info->script())->infos();
823  Tagged<ScopeInfo> stored_info;
824  CHECK(array->get(eval_scope_info_index)
825  .GetHeapObjectIfWeak(isolate, &stored_info));
826  CHECK_EQ(stored_info, context->scope_info());
827  }
828  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
829  isolate, compiled,
830  Compiler::GetFunctionFromEval(
831  isolate, source.ToHandleChecked(), outer_info, context, language_mode,
832  restriction, kNoSourcePosition, eval_position),
833  ReadOnlyRoots(isolate).exception());
834  return *compiled;
835 }
