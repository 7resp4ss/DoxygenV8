70 void DateCache::YearMonthDayFromDays(int days, int* year, int* month,
71  int* day) {
72  if (ymd_valid_) {
73  // Check conservatively if the given 'days' has
74  // the same year and month as the cached 'days'.
75  int new_day = ymd_day_ + (days - ymd_days_);
76  if (new_day >= 1 && new_day <= 28) {
77  ymd_day_ = new_day;
78  ymd_days_ = days;
79  *year = ymd_year_;
80  *month = ymd_month_;
81  *day = new_day;
82  return;
83  }
84  }
85  int save_days = days;
86 
87  days += kDaysOffset;
88  *year = 400 * (days / kDaysIn400Years) - kYearsOffset;
89  days %= kDaysIn400Years;
90 
91  DCHECK_EQ(save_days, DaysFromYearMonth(*year, 0) + days);
92 
93  days--;
94  int yd1 = days / kDaysIn100Years;
95  days %= kDaysIn100Years;
96  *year += 100 * yd1;
97 
98  days++;
99  int yd2 = days / kDaysIn4Years;
100  days %= kDaysIn4Years;
101  *year += 4 * yd2;
102 
103  days--;
104  int yd3 = days / 365;
105  days %= 365;
106  *year += yd3;
107 
108  bool is_leap = (!yd1 || yd2) && !yd3;
109 
110  DCHECK_GE(days, -1);
111  DCHECK(is_leap || (days >= 0));
112  DCHECK((days < 365) || (is_leap && (days < 366)));
113  DCHECK(is_leap == ((*year % 4 == 0) && (*year % 100 || (*year % 400 == 0))));
114  DCHECK(is_leap || ((DaysFromYearMonth(*year, 0) + days) == save_days));
115  DCHECK(!is_leap || ((DaysFromYearMonth(*year, 0) + days + 1) == save_days));
116 
117  days += is_leap;
118 
119  // Check if the date is after February.
120  if (days >= 31 + 28 + (is_leap ? 1 : 0)) {
121  days -= 31 + 28 + (is_leap ? 1 : 0);
122  // Find the date starting from March.
123  for (int i = 2; i < 12; i++) {
124  if (days < kDaysInMonths[i]) {
125  *month = i;
126  *day = days + 1;
127  break;
128  }
129  days -= kDaysInMonths[i];
130  }
131  } else {
132  // Check January and February.
133  if (days < 31) {
134  *month = 0;
135  *day = days + 1;
136  } else {
137  *month = 1;
138  *day = days - 31 + 1;
139  }
140  }
141  DCHECK(DaysFromYearMonth(*year, *month) + *day - 1 == save_days);
142  ymd_valid_ = true;
143  ymd_year_ = *year;
144  ymd_month_ = *month;
145  ymd_day_ = *day;
146  ymd_days_ = save_days;
147 }

149 int DateCache::DaysFromYearMonth(int year, int month) {
150  static const int day_from_month[] = {0, 31, 59, 90, 120, 151,
151  181, 212, 243, 273, 304, 334};
152  static const int day_from_month_leap[] = {0, 31, 60, 91, 121, 152,
153  182, 213, 244, 274, 305, 335};
154 
155  year += month / 12;
156  month %= 12;
157  if (month < 0) {
158  year--;
159  month += 12;
160  }
161 
162  DCHECK_GE(month, 0);
163  DCHECK_LT(month, 12);
164 
165  // year_delta is an arbitrary number such that:
166  // a) year_delta = -1 (mod 400)
167  // b) year + year_delta > 0 for years in the range defined by
168  // ECMA 262 - 15.9.1.1, i.e. upto 100,000,000 days on either side of
169  // Jan 1 1970. This is required so that we don't run into integer
170  // division of negative numbers.
171  // c) there shouldn't be an overflow for 32-bit integers in the following
172  // operations.
173  static const int year_delta = 399999;
174  static const int base_day =
175  365 * (1970 + year_delta) + (1970 + year_delta) / 4 -
176  (1970 + year_delta) / 100 + (1970 + year_delta) / 400;
177 
178  int year1 = year + year_delta;
179  int day_from_year =
180  365 * year1 + year1 / 4 - year1 / 100 + year1 / 400 - base_day;
181 
182  if ((year % 4 != 0) || (year % 100 == 0 && year % 400 != 0)) {
183  return day_from_year + day_from_month[month];
184  }
185  return day_from_year + day_from_month_leap[month];
186 }

188 void DateCache::BreakDownTime(int64_t time_ms, int* year, int* month, int* day,
189  int* weekday, int* hour, int* min, int* sec,
190  int* ms) {
191  int const days = DaysFromTime(time_ms);
192  int const time_in_day_ms = TimeInDay(time_ms, days);
193  YearMonthDayFromDays(days, year, month, day);
194  *weekday = Weekday(days);
195  *hour = time_in_day_ms / (60 * 60 * 1000);
196  *min = (time_in_day_ms / (60 * 1000)) % 60;
197  *sec = (time_in_day_ms / 1000) % 60;
198  *ms = time_in_day_ms % 1000;
199 }

201 // Implements LocalTimeZonedjustment(t, isUTC)
202 // ECMA 262 - ES#sec-local-time-zone-adjustment
203 int DateCache::GetLocalOffsetFromOS(int64_t time_ms, bool is_utc) {
204  double offset;
205 #ifdef V8_INTL_SUPPORT
206  if (v8_flags.icu_timezone_data) {
207  offset = tz_cache_->LocalTimeOffset(static_cast<double>(time_ms), is_utc);
208  } else {
209 #endif
210  // When ICU timezone data is not used, we need to compute the timezone
211  // offset for a given local time.
212  //
213  // The following shows that using DST for (t - LocalTZA - hour) produces
214  // correct conversion where LocalTZA is the timezone offset in winter (no
215  // DST) and the timezone offset is assumed to have no historical change.
216  // Note that it does not work for the past and the future if LocalTZA (no
217  // DST) is different from the current LocalTZA (no DST). For instance,
218  // this will break for Europe/Moscow in 2012 ~ 2013 because LocalTZA was
219  // 4h instead of the current 3h (as of 2018).
220  //
221  // Consider transition to DST at local time L1.
222  // Let L0 = L1 - hour, L2 = L1 + hour,
223  // U1 = UTC time that corresponds to L1,
224  // U0 = U1 - hour.
225  // Transitioning to DST moves local clock one hour forward L1 => L2, so
226  // U0 = UTC time that corresponds to L0 = L0 - LocalTZA,
227  // U1 = UTC time that corresponds to L1 = L1 - LocalTZA,
228  // U1 = UTC time that corresponds to L2 = L2 - LocalTZA - hour.
229  // Note that DST(U0 - hour) = 0, DST(U0) = 0, DST(U1) = 1.
230  // U0 = L0 - LocalTZA - DST(L0 - LocalTZA - hour),
231  // U1 = L1 - LocalTZA - DST(L1 - LocalTZA - hour),
232  // U1 = L2 - LocalTZA - DST(L2 - LocalTZA - hour).
233  //
234  // Consider transition from DST at local time L1.
235  // Let L0 = L1 - hour,
236  // U1 = UTC time that corresponds to L1,
237  // U0 = U1 - hour, U2 = U1 + hour.
238  // Transitioning from DST moves local clock one hour back L1 => L0, so
239  // U0 = UTC time that corresponds to L0 (before transition)
240  // = L0 - LocalTZA - hour.
241  // U1 = UTC time that corresponds to L0 (after transition)
242  // = L0 - LocalTZA = L1 - LocalTZA - hour
243  // U2 = UTC time that corresponds to L1 = L1 - LocalTZA.
244  // Note that DST(U0) = 1, DST(U1) = 0, DST(U2) = 0.
245  // U0 = L0 - LocalTZA - DST(L0 - LocalTZA - hour) = L0 - LocalTZA - DST(U0).
246  // U2 = L1 - LocalTZA - DST(L1 - LocalTZA - hour) = L1 - LocalTZA - DST(U1).
247  // It is impossible to get U1 from local time.
248  if (local_offset_ms_ == kInvalidLocalOffsetInMs) {
249  // This gets the constant LocalTZA (arguments are ignored).
250  local_offset_ms_ =
251  tz_cache_->LocalTimeOffset(static_cast<double>(time_ms), is_utc);
252  }
253  offset = local_offset_ms_;
254  if (!is_utc) {
255  const int kMsPerHour = 3600 * 1000;
256  time_ms -= (offset + kMsPerHour);
257  }
258  offset += DaylightSavingsOffsetInMs(time_ms);
259 #ifdef V8_INTL_SUPPORT
260  }
261 #endif
262  DCHECK_LT(offset, kInvalidLocalOffsetInMs);
263  return static_cast<int>(offset);
264 }

266 void DateCache::ExtendTheAfterSegment(int64_t time_ms, int offset_ms) {
267  if (!InvalidSegment(after_) && after_->offset_ms == offset_ms &&
268  after_->start_ms - kDefaultTimeZoneOffsetDeltaInMs <= time_ms &&
269  time_ms <= after_->end_ms) {
270  // Extend the after_ segment.
271  after_->start_ms = time_ms;
272  } else {
273  // The after_ segment is either invalid or starts too late.
274  if (!InvalidSegment(after_)) {
275  // If the after_ segment is valid, replace it with a new segment.
276  after_ = LeastRecentlyUsedCacheItem(before_);
277  }
278  after_->start_ms = time_ms;
279  after_->end_ms = time_ms;
280  after_->offset_ms = offset_ms;
281  after_->last_used = ++cache_usage_counter_;
282  }
283 }

285 int DateCache::LocalOffsetInMs(int64_t time_ms, bool is_utc) {
286  if (!is_utc) {
287  return GetLocalOffsetFromOS(time_ms, is_utc);
288  }
289 #ifdef ENABLE_SLOW_DCHECKS
290  int known_correct_result = 0;
291  if (v8_flags.enable_slow_asserts) {
292  // When slow DCHECKs are enabled, we always retrieve the known good result
293  // (slow) and check that the result produced by the cache matches it.
294  known_correct_result = GetLocalOffsetFromOS(time_ms, is_utc);
295  }
296 #endif // ENABLE_SLOW_DCHECKS
297  // Invalidate cache if the usage counter is close to overflow.
298  // Note that cache_usage_counter is incremented less than ten times
299  // in this function.
300  if (cache_usage_counter_ >= kMaxInt - 10) {
301  cache_usage_counter_ = 0;
302  for (int i = 0; i < kCacheSize; ++i) {
303  ClearSegment(&cache_[i]);
304  }
305  }
306 
307  // Optimistic fast check.
308  if (before_->start_ms <= time_ms && time_ms <= before_->end_ms) {
309  // Cache hit.
310  before_->last_used = ++cache_usage_counter_;
311  SLOW_DCHECK(before_->offset_ms == known_correct_result);
312  return before_->offset_ms;
313  }
314 
315  ProbeCache(time_ms);
316 
317  DCHECK(InvalidSegment(before_) || before_->start_ms <= time_ms);
318  DCHECK(InvalidSegment(after_) || time_ms < after_->start_ms);
319 
320  if (InvalidSegment(before_)) {
321  // Cache miss.
322  before_->start_ms = time_ms;
323  before_->end_ms = time_ms;
324  before_->offset_ms = GetLocalOffsetFromOS(time_ms, is_utc);
325  before_->last_used = ++cache_usage_counter_;
326  SLOW_DCHECK(before_->offset_ms == known_correct_result);
327  return before_->offset_ms;
328  }
329 
330  if (time_ms <= before_->end_ms) {
331  // Cache hit.
332  before_->last_used = ++cache_usage_counter_;
333  SLOW_DCHECK(before_->offset_ms == known_correct_result);
334  return before_->offset_ms;
335  }
336 
337  if (time_ms - kDefaultTimeZoneOffsetDeltaInMs > before_->end_ms) {
338  // If the before_ segment ends too early, then just
339  // query for the offset of the time_ms
340  int offset_ms = GetLocalOffsetFromOS(time_ms, is_utc);
341  ExtendTheAfterSegment(time_ms, offset_ms);
342  // This swap helps the optimistic fast check in subsequent invocations.
343  CacheItem* temp = before_;
344  before_ = after_;
345  after_ = temp;
346  SLOW_DCHECK(offset_ms == known_correct_result);
347  return offset_ms;
348  }
349 
350  // Now the time_ms is between
351  // before_->end_ms and before_->end_ms + default time zone offset delta.
352  // Update the usage counter of before_ since it is going to be used.
353  before_->last_used = ++cache_usage_counter_;
354 
355  // Check if after_ segment is invalid or starts too late.
356  int64_t new_after_start_ms =
357  before_->end_ms + kDefaultTimeZoneOffsetDeltaInMs;
358  if (InvalidSegment(after_) || new_after_start_ms <= after_->start_ms) {
359  int new_offset_ms = GetLocalOffsetFromOS(new_after_start_ms, is_utc);
360  ExtendTheAfterSegment(new_after_start_ms, new_offset_ms);
361  } else {
362  DCHECK(!InvalidSegment(after_));
363  // Update the usage counter of after_ since it is going to be used.
364  after_->last_used = ++cache_usage_counter_;
365  }
366 
367  // Now the time_ms is between before_->end_ms and after_->start_ms.
368  // Only one daylight savings offset change can occur in this interval.
369 
370  if (before_->offset_ms == after_->offset_ms) {
371  // Merge two segments if they have the same offset.
372  before_->end_ms = after_->end_ms;
373  ClearSegment(after_);
374  SLOW_DCHECK(before_->offset_ms == known_correct_result);
375  return before_->offset_ms;
376  }
377 
378  // Binary search for time zone offset change point,
379  // but give up if we don't find it in five iterations.
380  for (int i = 4; i >= 0; --i) {
381  int64_t delta = after_->start_ms - before_->end_ms;
382  int64_t middle_sec = (i == 0) ? time_ms : before_->end_ms + delta / 2;
383  int offset_ms = GetLocalOffsetFromOS(middle_sec, is_utc);
384  if (before_->offset_ms == offset_ms) {
385  before_->end_ms = middle_sec;
386  if (time_ms <= before_->end_ms) {
387  SLOW_DCHECK(offset_ms == known_correct_result);
388  return offset_ms;
389  }
390  // If we didn't return, we can't be in the last iteration.
391  DCHECK_GT(i, 0);
392  } else {
393  DCHECK(after_->offset_ms == offset_ms);
394  after_->start_ms = middle_sec;
395  if (time_ms >= after_->start_ms) {
396  // This swap helps the optimistic fast check in subsequent invocations.
397  CacheItem* temp = before_;
398  before_ = after_;
399  after_ = temp;
400  SLOW_DCHECK(offset_ms == known_correct_result);
401  return offset_ms;
402  }
403  // If we didn't return, we can't be in the last iteration.
404  DCHECK_GT(i, 0);
405  }
406  }
407  // During the last iteration, we set middle_sec = time_ms and return via one
408  // of the two return statements above. Thus, we never end up here.
409  UNREACHABLE();
410 }

412 void DateCache::ProbeCache(int64_t time_ms) {
413  CacheItem* before = nullptr;
414  CacheItem* after = nullptr;
415  DCHECK(before_ != after_);
416 
417  for (int i = 0; i < kCacheSize; ++i) {
418  if (InvalidSegment(&cache_[i])) {
419  continue;
420  }
421  if (cache_[i].start_ms <= time_ms) {
422  if (before == nullptr || before->start_ms < cache_[i].start_ms) {
423  before = &cache_[i];
424  }
425  } else if (time_ms < cache_[i].end_ms) {
426  if (after == nullptr || after->end_ms > cache_[i].end_ms) {
427  after = &cache_[i];
428  }
429  }
430  }
431 
432  // If before or after segments were not found,
433  // then set them to any invalid segment.
434  if (before == nullptr) {
435  before =
436  InvalidSegment(before_) ? before_ : LeastRecentlyUsedCacheItem(after);
437  }
438  if (after == nullptr) {
439  after = InvalidSegment(after_) && before != after_
440  ? after_
441  : LeastRecentlyUsedCacheItem(before);
442  }
443 
444  DCHECK_NOT_NULL(before);
445  DCHECK_NOT_NULL(after);
446  DCHECK(before != after);
447  DCHECK(InvalidSegment(before) || before->start_ms <= time_ms);
448  DCHECK(InvalidSegment(after) || time_ms < after->start_ms);
449  DCHECK(InvalidSegment(before) || InvalidSegment(after) ||
450  before->end_ms < after->start_ms);
451 
452  before_ = before;
453  after_ = after;
454 }

456 DateCache::CacheItem* DateCache::LeastRecentlyUsedCacheItem(CacheItem* skip) {
457  CacheItem* result = nullptr;
458  for (int i = 0; i < kCacheSize; ++i) {
459  if (&cache_[i] == skip) continue;
460  if (result == nullptr || result->last_used > cache_[i].last_used) {
461  result = &cache_[i];
462  }
463  }
464  ClearSegment(result);
465  return result;
466 }

491 double MakeDay(double year, double month, double date) {
492  if ((kMinYear <= year && year <= kMaxYear) &&
493  (kMinMonth <= month && month <= kMaxMonth) && std::isfinite(date)) {
494  int y = FastD2I(year);
495  int m = FastD2I(month);
496  y += m / 12;
497  m %= 12;
498  if (m < 0) {
499  m += 12;
500  y -= 1;
501  }
502  DCHECK_LE(0, m);
503  DCHECK_LT(m, 12);
504 
505  // kYearDelta is an arbitrary number such that:
506  // a) kYearDelta = -1 (mod 400)
507  // b) year + kYearDelta > 0 for years in the range defined by
508  // ECMA 262 - 15.9.1.1, i.e. upto 100,000,000 days on either side of
509  // Jan 1 1970. This is required so that we don't run into integer
510  // division of negative numbers.
511  // c) there shouldn't be an overflow for 32-bit integers in the following
512  // operations.
513  static const int kYearDelta = 399999;
514  static const int kBaseDay =
515  365 * (1970 + kYearDelta) + (1970 + kYearDelta) / 4 -
516  (1970 + kYearDelta) / 100 + (1970 + kYearDelta) / 400;
517  int day_from_year = 365 * (y + kYearDelta) + (y + kYearDelta) / 4 -
518  (y + kYearDelta) / 100 + (y + kYearDelta) / 400 -
519  kBaseDay;
520  if ((y % 4 != 0) || (y % 100 == 0 && y % 400 != 0)) {
521  static const int kDayFromMonth[] = {0, 31, 59, 90, 120, 151,
522  181, 212, 243, 273, 304, 334};
523  day_from_year += kDayFromMonth[m];
524  } else {
525  static const int kDayFromMonth[] = {0, 31, 60, 91, 121, 152,
526  182, 213, 244, 274, 305, 335};
527  day_from_year += kDayFromMonth[m];
528  }
529  return static_cast<double>(day_from_year - 1) + DoubleToInteger(date);
530  }
531  return std::numeric_limits<double>::quiet_NaN();
532 }

565 DateBuffer ToDateString(double time_val, DateCache* date_cache,
566  ToDateStringMode mode) {
567  if (std::isnan(time_val)) {
568  return FormatDate("Invalid Date");
569  }
570  int64_t time_ms = static_cast<int64_t>(time_val);
571  int64_t local_time_ms = (mode == ToDateStringMode::kUTCDateAndTime ||
572  mode == ToDateStringMode::kISODateAndTime)
573  ? time_ms
574  : date_cache->ToLocal(time_ms);
575  int year, month, day, weekday, hour, min, sec, ms;
576  date_cache->BreakDownTime(local_time_ms, &year, &month, &day, &weekday, &hour,
577  &min, &sec, &ms);
578  int timezone_offset = -date_cache->TimezoneOffset(time_ms);
579  int timezone_hour = std::abs(timezone_offset) / 60;
580  int timezone_min = std::abs(timezone_offset) % 60;
581  const char* local_timezone = date_cache->LocalTimezone(time_ms);
582  switch (mode) {
583  case ToDateStringMode::kLocalDate:

629  }

633  double const time =

635  out[DateParser::SECOND], out[DateParser::MILLISECOND]);

640  date = isolate->date_cache()->ToUTC(static_cast<int64_t>(date));
641  } else {

643  }
