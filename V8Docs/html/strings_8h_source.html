32 // Returns the value (0 .. 15) of a hexadecimal character c.
33 // If c is not a legal hexadecimal character, returns a value < 0.
34 inline int HexValue(uc32 c) {
35  c -= '0';
36  if (static_cast<unsigned>(c) <= 9) return c;
37  c = (c | 0x20) - ('a' - '0'); // detect 0x11..0x16 and 0x31..0x36.
38  if (static_cast<unsigned>(c) <= 5) return c + 10;
39  return -1;
40 }
