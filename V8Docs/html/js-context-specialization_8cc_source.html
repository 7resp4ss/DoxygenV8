29  switch (node->opcode()) {
30  case IrOpcode::kParameter:
31  return ReduceParameter(node);
32  case IrOpcode::kJSLoadContextNoCell:

35  return ReduceJSLoadContext(node);

38  case IrOpcode::kJSStoreContext:
39  return ReduceJSStoreContext(node);
40  case IrOpcode::kJSGetImportMeta:
41  return ReduceJSGetImportMeta(node);
42  default:
43  break;
44  }
45  return NoChange();
46 }

146 // Given a context {node} and the {distance} from that context to the target
147 // context (which we want to read from or store to), try to return a
148 // specialization context. If successful, update {distance} to whatever
149 // distance remains from the specialization context.
150 OptionalContextRef GetSpecializationContext(JSHeapBroker* broker, Node* node,
151  size_t* distance,
152  Maybe<OuterContext> maybe_outer) {
153  switch (node->opcode()) {
154  case IrOpcode::kHeapConstant: {
155  // TODO(jgruber,chromium:1209798): Using kAssumeMemoryFence works around
156  // the fact that the graph stores handles (and not refs). The assumption
157  // is that any handle inserted into the graph is safe to read; but we
158  // don't preserve the reason why it is safe to read. Thus we must
159  // over-approximate here and assume the existence of a memory fence. In
160  // the future, we should consider having the graph store ObjectRefs or
161  // ObjectData pointer instead, which would make new ref construction here
162  // unnecessary.
163  HeapObjectRef object =
164  MakeRefAssumeMemoryFence(broker, HeapConstantOf(node->op()));
165  if (object.IsContext()) return object.AsContext();
166  break;
167  }
168  case IrOpcode::kParameter: {
169  OuterContext outer;
170  if (maybe_outer.To(&outer) && IsContextParameter(node) &&
171  *distance >= outer.distance) {
172  *distance -= outer.distance;
173  return MakeRef(broker, outer.context);
174  }
175  break;
176  }
177  default:
178  break;
179  }
180  return OptionalContextRef();
181 }

185 Reduction JSContextSpecialization::ReduceJSLoadContextNoCell(Node* node) {
186  DCHECK_EQ(IrOpcode::kJSLoadContextNoCell, node->opcode());
187 
188  const ContextAccess& access = ContextAccessOf(node->op());
189  size_t depth = access.depth();
190 
191  // First walk up the context chain in the graph as far as possible.
192  Node* context = NodeProperties::GetOuterContext(node, &depth);
193 
194  OptionalContextRef maybe_concrete =
195  GetSpecializationContext(broker(), context, &depth, outer());
196  if (!maybe_concrete.has_value()) {
197  // We do not have a concrete context object, so we can only partially reduce
198  // the load by folding-in the outer context node.
199  return SimplifyJSLoadContextNoCell(node, context, depth);
200  }
201 
202  // Now walk up the concrete context chain for the remaining depth.
203  ContextRef concrete = maybe_concrete.value();
204  concrete = concrete.previous(broker(), &depth);
205  if (depth > 0) {
206  TRACE_BROKER_MISSING(broker(), "previous value for context " << concrete);
207  return SimplifyJSLoadContextNoCell(
208  node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
209  }
210 
211  if (!access.immutable() &&
212  !broker()->dependencies()->DependOnContextCell(
213  concrete, access.index(), ContextCell::kConst, broker())) {
214  // We found the requested context object but since the context slot is
215  // mutable we can only partially reduce the load.
216  return SimplifyJSLoadContextNoCell(
217  node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
218  }
219 
220  // This will hold the final value, if we can figure it out.
221  OptionalObjectRef maybe_value;
222  maybe_value = concrete.get(broker(), static_cast<int>(access.index()));
223 
224  if (!maybe_value.has_value()) {
225  TRACE_BROKER_MISSING(broker(), "slot value " << access.index()
226  << " for context "
227  << concrete);
228  return SimplifyJSLoadContextNoCell(
229  node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
230  }
231 
232  // Even though the context slot is immutable, the context might have escaped
233  // before the function to which it belongs has initialized the slot.
234  // We must be conservative and check if the value in the slot is currently
235  // the hole or undefined. Only if it is neither of these, can we be sure
236  // that it won't change anymore.
237  if (maybe_value->IsUndefined() || maybe_value->IsTheHole()) {
238  return SimplifyJSLoadContextNoCell(
239  node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
240  }
241 
242  // Success. The context load can be replaced with the constant.
243  Node* constant = jsgraph_->ConstantNoHole(*maybe_value, broker());
244  ReplaceWithValue(node, constant);
245  return Replace(constant);
246 }

248 Reduction JSContextSpecialization::ReduceJSLoadContext(Node* node) {
249  DCHECK_EQ(IrOpcode::kJSLoadContext, node->opcode());
250 
251  const ContextAccess& access = ContextAccessOf(node->op());
252  DCHECK(!access.immutable());
253  size_t depth = access.depth();
254 
255  // First walk up the context chain in the graph as far as possible.
256  Node* effect = NodeProperties::GetEffectInput(node);
257  Node* control = NodeProperties::GetControlInput(node);
258  Node* context = NodeProperties::GetOuterContext(node, &depth);
259 
260  OptionalContextRef maybe_concrete =
261  GetSpecializationContext(broker(), context, &depth, outer());
262  if (!maybe_concrete.has_value()) {
263  // We do not have a concrete context object, so we can only partially reduce
264  // the load by folding-in the outer context node.
265  return SimplifyJSLoadContext(node, context, depth);
266  }
267 
268  // Now walk up the concrete context chain for the remaining depth.
269  ContextRef concrete = maybe_concrete.value();
270  concrete = concrete.previous(broker(), &depth);
271  if (depth > 0) {
272  TRACE_BROKER_MISSING(broker(), "previous value for context " << concrete);
273  return SimplifyJSLoadContext(
274  node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
275  }
276 
277  auto maybe_value = concrete.get(broker(), static_cast<int>(access.index()));
278  if (!maybe_value || maybe_value->IsTheHole() ||
279  maybe_value->IsUndefinedContextCell()) {
280  return SimplifyJSLoadContext(
281  node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
282  }
283  if (!maybe_value->IsContextCell()) {
284  // Do a normal context load.
285  Node* load = effect = jsgraph_->graph()->NewNode(
286  jsgraph_->simplified()->LoadField(
287  AccessBuilder::ForContextSlot(access.index())),
288  jsgraph_->ConstantNoHole(concrete, broker()), effect, control);
289  ReplaceWithValue(node, load, effect, control);
290  return Changed(node);
291  }
292  DCHECK(concrete.object()->HasContextCells());
293 
294  compiler::ContextCellRef slot_ref = maybe_value->AsContextCell();
295  // TODO(victorgomes): Make DependOnContextCell return the state instead.
296  ContextCell::State state = slot_ref.state();
297  switch (state) {
298  case ContextCell::kConst: {
299  auto constant = slot_ref.tagged_value(broker());
300  if (!constant.has_value()) {
301  return SimplifyJSLoadContext(
302  node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
303  }
304  broker()->dependencies()->DependOnContextCell(slot_ref, state);
305  ReplaceWithValue(node, jsgraph_->ConstantNoHole(*constant, broker()),
306  effect, control);
307  return Changed(node);
308  }
309  case ContextCell::kSmi: {
310  broker()->dependencies()->DependOnContextCell(slot_ref, state);
311  Node* load = effect = jsgraph_->graph()->NewNode(
312  jsgraph_->simplified()->LoadField(
313  AccessBuilder::ForContextCellTaggedValue()),
314  jsgraph_->ConstantNoHole(slot_ref, broker()), effect, control);
315  ReplaceWithValue(node, load, effect, control);
316  return Changed(node);
317  }
318  case ContextCell::kInt32: {
319  broker()->dependencies()->DependOnContextCell(slot_ref, state);
320  Node* int32_load = effect = jsgraph_->graph()->NewNode(
321  jsgraph_->simplified()->LoadField(
322  AccessBuilder::ForContextCellInt32Value()),
323  jsgraph_->ConstantNoHole(slot_ref, broker()), effect, control);
324  ReplaceWithValue(node, int32_load, effect, control);
325  return Changed(node);
326  }
327  case ContextCell::kFloat64: {
328  broker()->dependencies()->DependOnContextCell(slot_ref, state);
329  Node* double_load = effect = jsgraph_->graph()->NewNode(
330  jsgraph_->simplified()->LoadField(
331  AccessBuilder::ForContextCellFloat64Value()),
332  jsgraph_->ConstantNoHole(slot_ref, broker()), effect, control);
333  ReplaceWithValue(node, double_load, effect, control);
334  return Changed(node);
335  }
336  case ContextCell::kDetached: {
337  Node* load = effect = jsgraph_->graph()->NewNode(
338  jsgraph_->simplified()->LoadField(
339  AccessBuilder::ForContextSlot(access.index())),
340  jsgraph_->ConstantNoHole(concrete, broker()), effect, control);
341  ReplaceWithValue(node, load, effect, control);
342  return Changed(node);
343  }
344  }
345  UNREACHABLE();
346 }

379 Reduction JSContextSpecialization::ReduceJSStoreContext(Node* node) {
380  DCHECK(v8_flags.script_context_cells || v8_flags.function_context_cells);
381  DCHECK_EQ(IrOpcode::kJSStoreContext, node->opcode());
382 
383  const ContextAccess& access = ContextAccessOf(node->op());
384  size_t depth = access.depth();
385 
386  // First walk up the context chain in the graph until we reduce the depth to 0
387  // or hit a node that does not have a CreateXYZContext operator.
388  Node* context = NodeProperties::GetOuterContext(node, &depth);
389  Node* value = NodeProperties::GetValueInput(node, 0);
390  Effect effect{NodeProperties::GetEffectInput(node)};
391  Control control{NodeProperties::GetControlInput(node)};
392 
393  OptionalContextRef maybe_concrete =
394  GetSpecializationContext(broker(), context, &depth, outer());
395  if (!maybe_concrete.has_value()) {
396  // We do not have a concrete context object, so we can only partially reduce
397  // the load by folding-in the outer context node.
398  return SimplifyJSStoreContext(node, context, depth);
399  }
400 
401  // Now walk up the concrete context chain for the remaining depth.
402  ContextRef concrete = maybe_concrete.value();
403  concrete = concrete.previous(broker(), &depth);
404  if (depth > 0) {
405  TRACE_BROKER_MISSING(broker(), "previous value for context " << concrete);
406  return SimplifyJSStoreContext(
407  node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
408  }
409 
410  auto maybe_value = concrete.get(broker(), static_cast<int>(access.index()));
411  if (!maybe_value || maybe_value->IsTheHole() ||
412  maybe_value->IsUndefinedContextCell()) {
413  return SimplifyJSStoreContext(
414  node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
415  }
416  if (!maybe_value->IsContextCell()) {
417  // Do a normal context store.
418  Node* store = jsgraph()->graph()->NewNode(
419  jsgraph()->simplified()->StoreField(
420  AccessBuilder::ForContextSlot(access.index())),
421  jsgraph()->ConstantNoHole(concrete, broker()), value, effect, control);
422  ReplaceWithValue(node, store, store, control);
423  return Changed(node);
424  }
425  DCHECK(concrete.object()->HasContextCells());
426 
427  compiler::ContextCellRef slot_ref = maybe_value->AsContextCell();
428  PropertyAccessBuilder access_builder(jsgraph(), broker());
429 
430  ContextCell::State state = slot_ref.state();
431  switch (state) {
432  case ContextCell::kConst: {
433  auto constant = slot_ref.tagged_value(broker());
434  if (!constant.has_value() ||
435  (constant->IsString() && !constant->IsInternalizedString())) {
436  return SimplifyJSStoreContext(
437  node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
438  }
439  broker()->dependencies()->DependOnContextCell(slot_ref, state);
440  access_builder.BuildCheckValue(value, &effect, control, *constant);
441  ReplaceWithValue(node, effect, effect, control);
442  return Changed(node);
443  }
444  case ContextCell::kSmi: {
445  broker()->dependencies()->DependOnContextCell(slot_ref, state);
446  Node* smi_value = access_builder.BuildCheckSmi(value, &effect, control);
447  Node* smi_store = jsgraph()->graph()->NewNode(
448  jsgraph()->simplified()->StoreField(
449  AccessBuilder::ForContextCellTaggedValue()),
450  jsgraph()->ConstantNoHole(slot_ref, broker()), smi_value, effect,
451  control);
452  ReplaceWithValue(node, smi_store, smi_store, control);
453  return Changed(node);
454  }
455  case ContextCell::kInt32: {
456  broker()->dependencies()->DependOnContextCell(slot_ref, state);
457  Node* input_number =
458  access_builder.BuildCheckNumberFitsInt32(value, &effect, control);
459  Node* double_store = jsgraph()->graph()->NewNode(
460  jsgraph()->simplified()->StoreField(
461  AccessBuilder::ForContextCellInt32Value()),
462  jsgraph_->ConstantNoHole(slot_ref, broker()), input_number, effect,
463  control);
464  ReplaceWithValue(node, double_store, double_store, control);
465  return Changed(node);
466  }
467  case ContextCell::kFloat64: {
468  broker()->dependencies()->DependOnContextCell(slot_ref, state);
469  Node* input_number =
470  access_builder.BuildCheckNumber(value, &effect, control);
471  Node* double_store = jsgraph()->graph()->NewNode(
472  jsgraph()->simplified()->StoreField(
473  AccessBuilder::ForContextCellFloat64Value()),
474  jsgraph_->ConstantNoHole(slot_ref, broker()), input_number, effect,
475  control);
476  ReplaceWithValue(node, double_store, double_store, control);
477  return Changed(node);
478  }
479  case ContextCell::kDetached: {
480  Node* store = jsgraph()->graph()->NewNode(
481  jsgraph()->simplified()->StoreField(
482  AccessBuilder::ForContextSlot(access.index())),
483  jsgraph()->ConstantNoHole(concrete, broker()), value, effect,
484  control);
485  ReplaceWithValue(node, store, store, control);
486  return Changed(node);
487  }
488  }
489  UNREACHABLE();

492 OptionalContextRef GetModuleContext(JSHeapBroker* broker, Node* node,
493  Maybe<OuterContext> maybe_context) {
494  size_t depth = std::numeric_limits<size_t>::max();
495  Node* context = NodeProperties::GetOuterContext(node, &depth);
496 
497  auto find_context = [broker](ContextRef c) {
498  while (c.map(broker).instance_type() != MODULE_CONTEXT_TYPE) {
499  size_t depth = 1;
500  c = c.previous(broker, &depth);
501  CHECK_EQ(depth, 0);
502  }
503  return c;
504  };
505 
506  switch (context->opcode()) {
507  case IrOpcode::kHeapConstant: {
508  // TODO(jgruber,chromium:1209798): Using kAssumeMemoryFence works around
509  // the fact that the graph stores handles (and not refs). The assumption
510  // is that any handle inserted into the graph is safe to read; but we
511  // don't preserve the reason why it is safe to read. Thus we must
512  // over-approximate here and assume the existence of a memory fence. In
513  // the future, we should consider having the graph store ObjectRefs or
514  // ObjectData pointer instead, which would make new ref construction here
515  // unnecessary.
516  HeapObjectRef object =
517  MakeRefAssumeMemoryFence(broker, HeapConstantOf(context->op()));
518  if (object.IsContext()) {
519  return find_context(object.AsContext());
520  }
521  break;
522  }
523  case IrOpcode::kParameter: {
524  OuterContext outer;
525  if (maybe_context.To(&outer) && IsContextParameter(context)) {
526  return find_context(MakeRef(broker, outer.context));
527  }
528  break;
529  }
530  default:
531  break;
532  }
533 
534  return OptionalContextRef();
535 }

537 Reduction JSContextSpecialization::ReduceJSGetImportMeta(Node* node) {
538  OptionalContextRef maybe_context = GetModuleContext(broker(), node, outer());
539  if (!maybe_context.has_value()) return NoChange();
