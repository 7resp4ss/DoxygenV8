99 void ExternalizeStringExtension::Externalize(
100  const v8::FunctionCallbackInfo<v8::Value>& info) {
101  DCHECK(ValidateCallbackInfo(info));
102  if (info.Length() < 1 || !info[0]->IsString()) {
103  info.GetIsolate()->ThrowError(
104  "First parameter to externalizeString() must be a string.");
105  return;
106  }
107  bool result = false;
108  DirectHandle<String> string =
109  Utils::OpenDirectHandle(*info[0].As<v8::String>());
110  const bool externalize_as_one_byte = string->IsOneByteRepresentation();
111  if (!string->SupportsExternalization(
112  externalize_as_one_byte ? v8::String::Encoding::ONE_BYTE_ENCODING
113  : v8::String::Encoding::TWO_BYTE_ENCODING)) {
114  // If the string is shared, testing with the combination of
115  // --shared-string-table and --isolate in d8 may result in races to
116  // externalize the same string. If GC is stressed in addition, this test
117  // might fail as the string was already externalized (marked for
118  // externalization by another thread and externalized during GC).
119  if (!StringShape(*string).IsExternal()) {
120  info.GetIsolate()->ThrowError("string does not support externalization.");
121  }
122  return;
123  }
124  uint32_t length = string->length();
125  if (externalize_as_one_byte) {
126  uint8_t* data = new uint8_t[length];
127  String::WriteToFlat(*string, data, 0, length);
128  SimpleOneByteStringResource* resource =
129  new SimpleOneByteStringResource(reinterpret_cast<char*>(data), length);
130  result = Utils::ToLocal(string)->MakeExternal(info.GetIsolate(), resource);
131  if (!result) delete resource;
132  } else {
133  base::uc16* data = new base::uc16[length];
134  String::WriteToFlat(*string, data, 0, length);
135  SimpleTwoByteStringResource* resource =
136  new SimpleTwoByteStringResource(data, length);
137  result = Utils::ToLocal(string)->MakeExternal(info.GetIsolate(), resource);
138  if (!result) delete resource;
139  }
140  // If the string is shared, testing with the combination of
141  // --shared-string-table and --isolate in d8 may result in races to
142  // externalize the same string. Those races manifest as externalization
143  // sometimes failing if another thread won and already forwarded the string to
144  // the external resource. Don't consider those races as failures.
145  if (!result && !HasExternalForwardingIndex(string)) {
146  info.GetIsolate()->ThrowError("externalizeString() failed.");
147  return;
148  }
149 }

153 MaybeDirectHandle<String> CopyConsStringToOld(Isolate* isolate,
154  DirectHandle<ConsString> string,
155  v8::String::Encoding encoding) {
156  DirectHandle<String> first = handle(string->first(), isolate);
157  DirectHandle<String> second = handle(string->second(), isolate);
158  if (encoding == v8::String::Encoding::TWO_BYTE_ENCODING &&
159  first->IsOneByteRepresentation() && second->IsOneByteRepresentation()) {
160  isolate->Throw(*isolate->factory()->NewStringFromAsciiChecked(
161  "Cannot create externalizable two-byte string from one-byte "
162  "ConsString. Create at least one part of the ConsString with "
163  "createExternalizableTwoByteString()"));
164  return kNullMaybeHandle;
165  }
166  return isolate->factory()->NewConsString(first, second, AllocationType::kOld);
167 }

169 MaybeDirectHandle<String> CreateExternalizableString(
170  v8::Isolate* isolate, DirectHandle<String> string,
171  v8::String::Encoding encoding) {
172  Isolate* i_isolate = reinterpret_cast<Isolate*>(isolate);
173  DCHECK_IMPLIES(encoding == v8::String::Encoding::ONE_BYTE_ENCODING,
174  string->IsOneByteRepresentation());
175  if (string->SupportsExternalization(encoding)) {
176  return string;
177  }
178  // Return the string if it is already externalized.
179  if (StringShape(*string).IsExternal()) {
180  return string;
181  }
182 
183  // Read-only strings are never externalizable. Don't try to copy them as
184  // some parts of the code might rely on some strings being in RO space (i.e.
185  // empty string).
186  if (HeapLayout::InReadOnlySpace(*string)) {
187  isolate->ThrowError("Read-only strings cannot be externalized.");
188  return kNullMaybeHandle;
189  }
190 #ifdef V8_COMPRESS_POINTERS
191  // Small strings may not be in-place externalizable.
192  if (string->Size() < static_cast<int>(sizeof(UncachedExternalString))) {
193  isolate->ThrowError("String is too short to be externalized.");
194  return kNullMaybeHandle;
195  }
196 #endif
197 
198  // Special handling for ConsStrings, as the ConsString -> ExternalString
199  // migration is special for GC (Tagged pointers to Untagged pointers).
200  // Skip if the ConsString is flat, as we won't be guaranteed a string in old
201  // space in that case. Note that this is also true for non-canonicalized
202  // ConsStrings that TurboFan might create (the first part is empty), so we
203  // explicitly check for that case as well.
204  if (IsConsString(*string, i_isolate) && !string->IsFlat() &&
205  Cast<ConsString>(string)->first()->length() != 0) {
206  DirectHandle<String> result;
207  if (CopyConsStringToOld(i_isolate, Cast<ConsString>(string), encoding)
208  .ToHandle(&result)) {
209  DCHECK(result->SupportsExternalization(encoding));
210  return result;
211  } else {
212  return kNullMaybeHandle;
213  }
214  }
215  // All other strings can be implicitly flattened.
216  if (encoding == v8::String::ONE_BYTE_ENCODING) {
217  MaybeDirectHandle<SeqOneByteString> maybe_result =
218  i_isolate->factory()->NewRawOneByteString(string->length(),
219  AllocationType::kOld);
220  DirectHandle<SeqOneByteString> result;
221  if (maybe_result.ToHandle(&result)) {
222  DisallowGarbageCollection no_gc;
223  String::WriteToFlat(*string, result->GetChars(no_gc), 0,
224  string->length());
225  DCHECK(result->SupportsExternalization(encoding));
226  return result;
227  }
228  } else {
229  MaybeDirectHandle<SeqTwoByteString> maybe_result =
230  i_isolate->factory()->NewRawTwoByteString(string->length(),
231  AllocationType::kOld);
232  DirectHandle<SeqTwoByteString> result;
233  if (maybe_result.ToHandle(&result)) {
234  DisallowGarbageCollection no_gc;
235  String::WriteToFlat(*string, result->GetChars(no_gc), 0,
236  string->length());
237  DCHECK(result->SupportsExternalization(encoding));
238  return result;
239  }
240  }
241  isolate->ThrowError("Unable to create string");
242  return kNullMaybeHandle;
243 }

247 void ExternalizeStringExtension::CreateExternalizableString(
248  const v8::FunctionCallbackInfo<v8::Value>& info) {
249  DCHECK(ValidateCallbackInfo(info));
250  if (info.Length() < 1 || !info[0]->IsString()) {
