73 template <AllocationType type>
74 V8_WARN_UNUSED_RESULT V8_INLINE AllocationResult
75 HeapAllocator::AllocateRaw(int size_in_bytes, AllocationOrigin origin,
76  AllocationAlignment alignment, AllocationHint hint) {
77  DCHECK(!heap_->IsInGC());
78  DCHECK(AllowHandleAllocation::IsAllowed());
79  DCHECK(AllowHeapAllocation::IsAllowed());
80  CHECK(AllowHeapAllocationInRelease::IsAllowed());
81  DCHECK(local_heap_->IsRunning());
82  // We need to have entered the isolate before allocating.
83  DCHECK_EQ(heap_->isolate(), Isolate::TryGetCurrent());
84 #if V8_ENABLE_WEBASSEMBLY
85  if (!v8_flags.wasm_jitless) {
86  trap_handler::AssertThreadNotInWasm();
87  }
88 #endif
89 #if DEBUG
90  local_heap_->VerifyCurrent();
91 #endif
92 
93  if (v8_flags.single_generation.value() && type == AllocationType::kYoung) {
94  return AllocateRaw(size_in_bytes, AllocationType::kOld, origin, alignment);
95  }
96 
97 #ifdef V8_ENABLE_ALLOCATION_TIMEOUT
98  if (V8_UNLIKELY(allocation_timeout_.has_value()) &&
99  ReachedAllocationTimeout()) {
100  return AllocationResult::Failure();
101  }
102 #endif // V8_ENABLE_ALLOCATION_TIMEOUT
103 
104 #ifdef DEBUG
105  IncrementObjectCounters();
106 #endif // DEBUG
107 
108  if (heap_->CanSafepoint()) {
109  local_heap_->Safepoint();
110  }
111 
112  const size_t large_object_threshold = heap_->MaxRegularHeapObjectSize(type);
113  const bool large_object =
114  static_cast<size_t>(size_in_bytes) > large_object_threshold;
115 
116  Tagged<HeapObject> object;
117  AllocationResult allocation;
118 
119  if (V8_UNLIKELY(large_object)) {
120  allocation =
121  AllocateRawLargeInternal(size_in_bytes, type, origin, alignment, hint);
122  } else {
123  switch (type) {
124  case AllocationType::kYoung:
125  allocation = new_space_allocator_->AllocateRaw(size_in_bytes, alignment,
126  origin, hint);
127  break;
128  case AllocationType::kMap:
129  case AllocationType::kOld:
130  allocation = old_space_allocator_->AllocateRaw(size_in_bytes, alignment,
131  origin, hint);
132  DCHECK_IMPLIES(v8_flags.sticky_mark_bits && !allocation.IsFailure(),
133  heap_->marking_state()->IsMarked(allocation.ToObject()));
134  break;
135  case AllocationType::kCode: {
136  DCHECK_EQ(alignment, AllocationAlignment::kTaggedAligned);
137  DCHECK(AllowCodeAllocation::IsAllowed());
138  allocation = code_space_allocator_->AllocateRaw(
139  size_in_bytes, AllocationAlignment::kTaggedAligned, origin, hint);
140  break;
141  }
142  case AllocationType::kReadOnly:
143  DCHECK(read_only_space()->writable());
144  DCHECK_EQ(AllocationOrigin::kRuntime, origin);
145  allocation = read_only_space()->AllocateRaw(size_in_bytes, alignment);
146  break;
147  case AllocationType::kSharedMap:
148  case AllocationType::kSharedOld:
149  allocation = shared_space_allocator_->AllocateRaw(
150  size_in_bytes, alignment, origin, hint);
151  break;
152  case AllocationType::kTrusted:
153  allocation = trusted_space_allocator_->AllocateRaw(
154  size_in_bytes, alignment, origin, hint);
155  break;
156  case AllocationType::kSharedTrusted:
157  allocation = shared_trusted_space_allocator_->AllocateRaw(
158  size_in_bytes, alignment, origin, hint);
159  break;
160  }
161  }
162 
163  if (allocation.To(&object)) {
164  if (heap::ShouldZapGarbage() && AllocationType::kCode == type) {
165  heap::ZapCodeBlock(object.address(), size_in_bytes);
166  }
167 
168  if (local_heap_->is_main_thread()) {
169  for (auto& tracker : heap_->allocation_trackers_) {
170  tracker->AllocationEvent(object.address(), size_in_bytes);
171  }
172  }
173  }
174 
175  return allocation;
176 }

178 AllocationResult HeapAllocator::AllocateRaw(int size_in_bytes,
179  AllocationType type,
180  AllocationOrigin origin,
181  AllocationAlignment alignment,
182  AllocationHint hint) {
183  switch (type) {
184  case AllocationType::kYoung:
185  return AllocateRaw<AllocationType::kYoung>(size_in_bytes, origin,
186  alignment, hint);
187  case AllocationType::kOld:
188  return AllocateRaw<AllocationType::kOld>(size_in_bytes, origin, alignment,
189  hint);
190  case AllocationType::kCode:
191  return AllocateRaw<AllocationType::kCode>(size_in_bytes, origin,
192  alignment, hint);
193  case AllocationType::kMap:
194  return AllocateRaw<AllocationType::kMap>(size_in_bytes, origin, alignment,
195  hint);
196  case AllocationType::kReadOnly:
197  return AllocateRaw<AllocationType::kReadOnly>(size_in_bytes, origin,
198  alignment, hint);
199  case AllocationType::kSharedMap:
200  return AllocateRaw<AllocationType::kSharedMap>(size_in_bytes, origin,
201  alignment, hint);
202  case AllocationType::kSharedOld:
203  return AllocateRaw<AllocationType::kSharedOld>(size_in_bytes, origin,
204  alignment, hint);
205  case AllocationType::kTrusted:
206  return AllocateRaw<AllocationType::kTrusted>(size_in_bytes, origin,
207  alignment, hint);
208  case AllocationType::kSharedTrusted:
209  return AllocateRaw<AllocationType::kSharedTrusted>(size_in_bytes, origin,
210  alignment, hint);
211  }
212  UNREACHABLE();
213 }

215 template <HeapAllocator::AllocationRetryMode mode>
216 V8_WARN_UNUSED_RESULT V8_INLINE Tagged<HeapObject>
217 HeapAllocator::AllocateRawWith(int size, AllocationType allocation,
218  AllocationOrigin origin,
219  AllocationAlignment alignment,
220  AllocationHint hint) {
221  AllocationResult result;
222  Tagged<HeapObject> object;
223  size = ALIGN_TO_ALLOCATION_ALIGNMENT(size);
224  if (allocation == AllocationType::kYoung) {
225  result = AllocateRaw<AllocationType::kYoung>(size, origin, alignment, hint);
226  if (result.To(&object)) {
227  return object;
228  }
229  } else if (allocation == AllocationType::kOld) {
230  result = AllocateRaw<AllocationType::kOld>(size, origin, alignment, hint);
231  if (result.To(&object)) {
232  return object;
233  }
234  }
235  switch (mode) {
236  case kLightRetry:
237  result = AllocateRawWithLightRetrySlowPath(size, allocation, origin,
238  alignment, hint);
239  break;
240  case kRetryOrFail:
241  result = AllocateRawWithRetryOrFailSlowPath(size, allocation, origin,
242  alignment, hint);
243  break;
244  }
245  if (result.To(&object)) {
246  return object;
247  }
248  return HeapObject();
249 }

251 template <typename AllocateFunction>
252 V8_WARN_UNUSED_RESULT std::invoke_result_t<AllocateFunction>
253 HeapAllocator::AllocateRawWithLightRetrySlowPath(AllocateFunction&& Allocate,
254  AllocationType allocation) {
255  DCHECK_NE(AllocationType::kYoung, allocation);
256 
257  if (auto result = Allocate()) [[likely]] {
258  return result;
259  }
260 
261  if (auto result = CollectGarbageAndRetryAllocation(Allocate, allocation)) {
262  return result;
263  }
264 
265  heap_for_allocation(allocation)->CheckHeapLimitReached();
266 
267  return {};
268 }

270 template <typename AllocateFunction>
271 std::invoke_result_t<AllocateFunction>
272 HeapAllocator::AllocateRawWithRetryOrFailSlowPath(AllocateFunction&& Allocate,
273  AllocationType allocation) {
274  if (auto result = Allocate()) [[likely]] {
275  return result;
276  }
277 
278  if (auto result = CollectGarbageAndRetryAllocation(Allocate, allocation)) {
279  return result;
280  }
281 
282  // In the case of young allocations, the GCs above were minor GCs. Try "light"
283  // full GCs before performing the last-resort GCs.
284  if (allocation == AllocationType::kYoung) {
285  if (auto result =
286  CollectGarbageAndRetryAllocation(Allocate, AllocationType::kOld)) {
287  return result;
288  }
289  }
290 
291  // Perform last resort GC. This call will clear more caches and perform more
292  // GCs. It will also enforce the heap limit if still violated.
293  CollectAllAvailableGarbage(allocation);
294 
295  if (auto result = RetryAllocation(Allocate)) {
296  return result;
297  }
298 
299  V8::FatalProcessOutOfMemory(heap_->isolate(), "CALL_AND_RETRY_LAST",
300  V8::kHeapOOM);
301 }

303 template <typename AllocateFunction>
304 std::invoke_result_t<AllocateFunction>
305 HeapAllocator::CollectGarbageAndRetryAllocation(AllocateFunction&& Allocate,
306  AllocationType allocation) {
307  const auto perform_heap_limit_check = v8_flags.late_heap_limit_check
308  ? PerformHeapLimitCheck::kNo
309  : PerformHeapLimitCheck::kYes;
310 
311  for (int i = 0; i < 2; i++) {
312  // Skip the heap limit check in the GC if enabled. The heap limit needs to
313  // be enforced by the caller.
314  CollectGarbage(allocation, perform_heap_limit_check);
315 
316  // As long as we are at or above the heap limit, we definitely need another
317  // GC.
318  if (heap_for_allocation(allocation)->ReachedHeapLimit()) {
319  continue;
320  }
321 
322  if (auto result = RetryAllocation(Allocate)) {
323  return result;
324  }
325  }
326 
327  return {};
328 }

330 template <typename AllocateFunction>
331 std::invoke_result_t<AllocateFunction> HeapAllocator::RetryAllocation(
332  AllocateFunction&& Allocate) {
333  // Initially flags on the LocalHeap are always disabled. They are only
334  // active while this method is running.
335  DCHECK(!local_heap_->IsRetryOfFailedAllocation());
336  local_heap_->SetRetryOfFailedAllocation(true);
337  auto result = Allocate();
338  local_heap_->SetRetryOfFailedAllocation(false);
339  return result;
340 }
