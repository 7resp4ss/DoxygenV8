96  //

104  // Make sure we don't overflow the smi.
105  static_assert(FieldIndexBits::kLastUsedBit < kSmiValueSize);

107  //

109  //
110  using WasmFieldTypeBits = IsWasmStructBits::Next<WasmValueType, 4>;

112  // Make sure we don't overflow the smi.
113  static_assert(WasmFieldOffsetBits::kLastUsedBit < kSmiValueSize);

117  //
118  using AllowOutOfBoundsBits = LookupOnLookupStartObjectBits::Next<bool, 1>;
119 
120  //
121  // Encoding when KindBits contains kElement.
122  //
123  using IsWasmArrayBits = AllowOutOfBoundsBits::Next<bool, 1>;
