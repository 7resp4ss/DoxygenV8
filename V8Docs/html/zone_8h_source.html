28 // The Zone supports very fast allocation of small chunks of
29 // memory. The chunks cannot be deallocated individually, but instead
30 // the Zone supports deallocating all chunks in one fast
31 // operation. The Zone is used to hold temporary data structures like
32 // the abstract syntax tree, which is deallocated after compilation.
33 //
34 // Note: There is no need to initialize the Zone; the first time an
35 // allocation is attempted, a segment of memory will be requested
36 // through the allocator.
37 //
38 // Note: The implementation is inherently not thread safe. Do not use
39 // from multi-threaded code.

48  // Allocate 'size' bytes of uninitialized memory in the Zone; expands the Zone
49  // by allocating new segments of memory on demand using AccountingAllocator
50  // (see AccountingAllocator::AllocateSegment()).
51  //
52  // When V8_ENABLE_PRECISE_ZONE_STATS is defined, the allocated bytes are
53  // associated with the provided TypeTag type.
54  template <typename TypeTag>
55  void* Allocate(size_t size) {
56 #ifdef V8_USE_ADDRESS_SANITIZER
57  return AsanNew(size);
58 #else
59  size = RoundUp(size, kAlignmentInBytes);
60 #ifdef V8_ENABLE_PRECISE_ZONE_STATS
61  if (V8_UNLIKELY(TracingFlags::is_zone_stats_enabled())) {
62  type_stats_.AddAllocated<TypeTag>(size);
63  }
64  allocation_size_for_tracing_ += size;
65 #endif
66  if (V8_UNLIKELY(size > limit_ - position_)) {
67  Expand(size);
68  }
69 
70  DCHECK_LE(position_, limit_);
71  DCHECK_LE(size, limit_ - position_);
72  DCHECK_EQ(0, position_ % kAlignmentInBytes);
73  void* result = reinterpret_cast<void*>(position_);
74  position_ += size;
75  return result;
76 #endif // V8_USE_ADDRESS_SANITIZER
77  }
78 
79  // Return 'size' bytes of memory back to Zone. These bytes can be reused
80  // for following allocations.
81  //
82  // When V8_ENABLE_PRECISE_ZONE_STATS is defined, the deallocated bytes are
83  // associated with the provided TypeTag type.
84  template <typename TypeTag = void>
85  void Delete(void* pointer, size_t size) {
86  DCHECK_NOT_NULL(pointer);
87  DCHECK_NE(size, 0);
88  size = RoundUp(size, kAlignmentInBytes);
89 #ifdef V8_ENABLE_PRECISE_ZONE_STATS
90  if (V8_UNLIKELY(TracingFlags::is_zone_stats_enabled())) {
91  type_stats_.AddDeallocated<TypeTag>(size);
92  }
93  freed_size_for_tracing_ += size;
94 #endif
95 
96 #ifdef DEBUG
97  static const unsigned char kZapDeadByte = 0xcd;
98  memset(pointer, kZapDeadByte, size);
99 #endif
100  }

102  // Allocates memory for T instance and constructs object by calling respective
103  // Args... constructor.
104  //
105  // When V8_ENABLE_PRECISE_ZONE_STATS is defined, the allocated bytes are
106  // associated with the T type.
107  template <typename T, typename... Args>
108  T* New(Args&&... args) {
109  static_assert(alignof(T) <= kAlignmentInBytes);
110  void* memory = Allocate<T>(sizeof(T));
111  return new (memory) T(std::forward<Args>(args)...);
112  }
113 
114  // Allocates uninitialized memory for 'length' number of T instances.
115  //
116  // When V8_ENABLE_PRECISE_ZONE_STATS is defined, the allocated bytes are
117  // associated with the provided TypeTag type. It might be useful to tag
118  // buffer allocations with meaningful names to make buffer allocation sites
119  // distinguishable between each other.
120  template <typename T, typename TypeTag = T[]>
121  T* AllocateArray(size_t length) {
122  static_assert(alignof(T) <= kAlignmentInBytes);
123  // Defense-in-depth: ensure that the multiplication does not overflow as
124  // that would result in a too-small allocation and (likely) memory
125  // corruption afterwards.
126  CHECK_LT(length, std::numeric_limits<size_t>::max() / sizeof(T));
127  return static_cast<T*>(Allocate<TypeTag>(length * sizeof(T)));
128  }
129 
130  // Allocates a Vector with 'length' uninitialized entries.
131  template <typename T, typename TypeTag = T[]>
132  base::Vector<T> AllocateVector(size_t length) {
133  T* new_array = AllocateArray<T, TypeTag>(length);
134  return {new_array, length};
135  }
136 
137  // Allocates a Vector with 'length' elements and value-constructs them.
138  template <typename T, typename TypeTag = T[]>
139  base::Vector<T> NewVector(size_t length) {
140  T* new_array = AllocateArray<T, TypeTag>(length);
141  std::uninitialized_value_construct_n(new_array, length);
142  return {new_array, length};
143  }
144 
145  // Allocates a Vector with 'length' elements and initializes them with
146  // 'value'.
147  template <typename T, typename TypeTag = T[]>
148  base::Vector<T> NewVector(size_t length, T value) {
149  T* new_array = AllocateArray<T, TypeTag>(length);
150  std::uninitialized_fill_n(new_array, length, value);
151  return {new_array, length};
152  }
153 
154  template <typename T, typename TypeTag = std::remove_const_t<T>[]>
155  base::Vector<std::remove_const_t<T>> CloneVector(base::Vector<T> v) {
156  auto* new_array = AllocateArray<std::remove_const_t<T>, TypeTag>(v.size());
157  std::uninitialized_copy(v.begin(), v.end(), new_array);
158  return {new_array, v.size()};
159  }

161  // Return array of 'length' elements back to Zone. These bytes can be reused
162  // for following allocations.

164  // When V8_ENABLE_PRECISE_ZONE_STATS is defined, the deallocated bytes are

166  template <typename T, typename TypeTag = T[]>
167  void DeleteArray(T* pointer, size_t length) {
168  Delete<TypeTag>(pointer, length * sizeof(T));

319 // ZoneObject is an abstraction that helps define classes of objects
320 // allocated in the Zone. Use it as a base class; see ast.h.
321 class ZoneObject {
322  public:
323  // The accidential old-style pattern
324  // new (zone) SomeObject(...)
325  // now produces compilation error. The proper way of allocating objects in
326  // Zones looks like this:
327  // zone->New<SomeObject>(...)
328  void* operator new(size_t, Zone*) = delete; // See explanation above.
329  // Allow non-allocating placement new.
330  void* operator new(size_t size, void* ptr) { // See explanation above.
331  return ptr;
332  }
333 
334  // Ideally, the delete operator should be private instead of
335  // public, but unfortunately the compiler sometimes synthesizes
336  // (unused) destructors for classes derived from ZoneObject, which
337  // require the operator to be visible. MSVC requires the delete
338  // operator to be public.
339 
340  // ZoneObjects should never be deleted individually; use
341  // Zone::DeleteAll() to delete all zone objects in one go.
342  // Note, that descructors will not be called.
343  void operator delete(void*, size_t) { UNREACHABLE(); }
344  void operator delete(void* pointer, Zone* zone) = delete;
345 };
