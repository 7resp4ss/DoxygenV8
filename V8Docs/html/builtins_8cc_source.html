114 const char* Builtins::Lookup(Address pc) {
115  // Off-heap pc's can be looked up through binary search.
116  Builtin builtin = OffHeapInstructionStream::TryLookupCode(isolate_, pc);
117  if (Builtins::IsBuiltinId(builtin)) return name(builtin);
118 
119  // May be called during initialization (disassembler).
120  if (!initialized_) return nullptr;
121  for (Builtin builtin_ix = Builtins::kFirst; builtin_ix <= Builtins::kLast;
122  ++builtin_ix) {
123  if (code(builtin_ix)->contains(isolate_, pc)) {
124  return name(builtin_ix);
125  }
126  }
127  return nullptr;
128 }

166 bool Builtins::CheckFormalParameterCount(
167  Builtin builtin, int function_length,

171  return true;

174  if (!HasJSLinkage(builtin)) {

185  return parameter_count == formal_parameter_count_with_receiver;

189 CallInterfaceDescriptor Builtins::CallInterfaceDescriptorFor(Builtin builtin) {
190  CallDescriptors::Key key;
191  switch (builtin) {
192 // This macro is deliberately crafted so as to emit very little code,
193 // in order to keep binary size of this function under control.
194 #define CASE_OTHER(Name, ...) \
195  case Builtin::k##Name: { \
196  key = Builtin_##Name##_InterfaceDescriptor::key(); \
197  break; \
198  }
199  BUILTIN_LIST(IGNORE_BUILTIN, IGNORE_BUILTIN, IGNORE_BUILTIN, CASE_OTHER,
200  CASE_OTHER, CASE_OTHER, CASE_OTHER, IGNORE_BUILTIN, CASE_OTHER)
201 #undef CASE_OTHER
202  default:
203  Builtins::Kind kind = Builtins::KindOf(builtin);
204  DCHECK_NE(BCH, kind);
205  if (kind == TSJ || kind == TFJ || kind == CPP) {
206  return JSTrampolineDescriptor{};
207  }
208  UNREACHABLE();
209  }
210  return CallInterfaceDescriptor{key};
211 }

366  Builtin* builtin) const {
367  Address* handle_location = maybe_code.location();
368  Address* builtins_table = isolate_->builtin_table();
369  if (handle_location < builtins_table) return false;
370  Address* builtins_table_end = &builtins_table[Builtins::kBuiltinCount];
371  if (handle_location >= builtins_table_end) return false;
372  *builtin = FromInt(static_cast<int>(handle_location - builtins_table));
373  return true;

385  EmbeddedData embedded_data = EmbeddedData::FromBlob(isolate);

400  // T0 tables.

403  isolate_data->builtin_tier0_entry_table()[ii] =

405  isolate_data->builtin_tier0_table()[ii] = isolate_data->builtin_table()[ii];
406  }

409 // static
410 void Builtins::EmitCodeCreateEvents(Isolate* isolate) {
411  if (!isolate->IsLoggingCodeCreation()) return;
412 
413  Address* builtins = isolate->builtin_table();
414  int i = 0;
415  HandleScope scope(isolate);
416  for (; i < ToInt(Builtin::kFirstBytecodeHandler); i++) {
417  auto builtin_code = DirectHandle<Code>::FromSlot(&builtins[i]);
418  DirectHandle<AbstractCode> code = Cast<AbstractCode>(builtin_code);
419  PROFILE(isolate, CodeCreateEvent(LogEventListener::CodeTag::kBuiltin, code,
420  Builtins::name(FromInt(i))));
421  }
422 
423  static_assert(kLastBytecodeHandlerPlusOne == kBuiltinCount);
424  for (; i < kBuiltinCount; i++) {
425  auto builtin_code = DirectHandle<Code>::FromSlot(&builtins[i]);
426  DirectHandle<AbstractCode> code = Cast<AbstractCode>(builtin_code);
427  interpreter::Bytecode bytecode =
428  builtin_metadata[i].data.bytecode_and_scale.bytecode;
429  interpreter::OperandScale scale =
430  builtin_metadata[i].data.bytecode_and_scale.scale;
431  PROFILE(isolate,
432  CodeCreateEvent(
433  LogEventListener::CodeTag::kBytecodeHandler, code,
434  interpreter::Bytecodes::ToString(bytecode, scale).c_str()));
435  }
436 }

438 // static
439 DirectHandle<Code> Builtins::CreateInterpreterEntryTrampolineForProfiling(
440  Isolate* isolate) {
441  DCHECK_NOT_NULL(isolate->embedded_blob_code());
442  DCHECK_NE(0, isolate->embedded_blob_code_size());
443 
444  Tagged<Code> code = isolate->builtins()->code(
445  Builtin::kInterpreterEntryTrampolineForProfiling);
446 
447  CodeDesc desc;
448  desc.buffer = reinterpret_cast<uint8_t*>(code->instruction_start());
449 
450  int instruction_size = code->instruction_size();
451  desc.buffer_size = instruction_size;
452  desc.instr_size = instruction_size;
453 
454  // Ensure the code doesn't require creation of metadata, otherwise respective
455  // fields of CodeDesc should be initialized.
456  DCHECK_EQ(code->safepoint_table_size(), 0);
457  DCHECK_EQ(code->handler_table_size(), 0);
458  DCHECK_EQ(code->constant_pool_size(), 0);
459  // TODO(v8:11036): The following DCHECK currently fails if the mksnapshot is
460  // run with enabled code comments, i.e. --interpreted_frames_native_stack is
461  // incompatible with --code-comments at mksnapshot-time. If ever needed,
462  // implement support.
463  DCHECK_EQ(code->code_comments_size(), 0);
464  DCHECK_EQ(code->unwinding_info_size(), 0);
465 
466  desc.safepoint_table_offset = instruction_size;
467  desc.handler_table_offset = instruction_size;
468  desc.constant_pool_offset = instruction_size;
469  desc.code_comments_offset = instruction_size;
470  desc.jump_table_info_offset = instruction_size;
471 
472  CodeDesc::Verify(&desc);
473 
474  return Factory::CodeBuilder(isolate, desc, CodeKind::BUILTIN)
475  // Mimic the InterpreterEntryTrampoline.
476  .set_builtin(Builtin::kInterpreterEntryTrampoline)
477  .Build();
478 }

561  case TSC:
562  case TFC:
563  case ASM:
