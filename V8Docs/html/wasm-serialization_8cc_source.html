47  template <typename T>
48  void Write(const T& value) {
49  DCHECK_GE(current_size(), sizeof(T));
50  WriteUnalignedValue(reinterpret_cast<Address>(current_location()), value);
51  pos_ += sizeof(T);
52  if (v8_flags.trace_wasm_serialization) {
53  StdoutStream{} << "wrote: " << static_cast<size_t>(value)
54  << " sized: " << sizeof(T) << std::endl;
55  }
56  }

58  template <typename T>
59  void WriteVector(const base::Vector<T> v) {
60  base::Vector<const uint8_t> bytes = base::Vector<const uint8_t>::cast(v);
61  DCHECK_GE(current_size(), bytes.size());
62  if (!bytes.empty()) {
63  memcpy(current_location(), bytes.begin(), bytes.size());
64  pos_ += bytes.size();
65  }
66  if (v8_flags.trace_wasm_serialization) {
67  StdoutStream{} << "wrote vector of " << v.size()
68  << " elements (total size " << bytes.size() << " bytes)"
69  << std::endl;
70  }
71  }
72 
73  void Skip(size_t size) { pos_ += size; }
74 
75  private:
76  uint8_t* const start_;
77  uint8_t* const end_;
78  uint8_t* pos_;
79 };

137 // On Intel, call sites are encoded as a displacement. For linking and for
138 // serialization/deserialization, we want to store/retrieve a tag (the function
139 // index). On Intel, that means accessing the raw displacement.
140 // On ARM64, call sites are encoded as either a literal load or a direct branch.
141 // Other platforms simply require accessing the target address.
142 void SetWasmCalleeTag(WritableRelocInfo* rinfo, uint32_t tag) {
143 #if V8_TARGET_ARCH_X64 || V8_TARGET_ARCH_IA32
144  DCHECK(rinfo->HasTargetAddressAddress());
145  DCHECK(!RelocInfo::IsCompressedEmbeddedObject(rinfo->rmode()));
146  WriteUnalignedValue(rinfo->target_address_address(), tag);
147 #elif V8_TARGET_ARCH_ARM64
148  Instruction* instr = reinterpret_cast<Instruction*>(rinfo->pc());
149  if (instr->IsLdrLiteralX()) {
150  WriteUnalignedValue(rinfo->constant_pool_entry_address(),
151  static_cast<Address>(tag));
152  } else {
153  DCHECK(instr->IsBranchAndLink() || instr->IsUnconditionalBranch());
154  instr->SetBranchImmTarget<UncondBranchType>(
155  reinterpret_cast<Instruction*>(rinfo->pc() + tag * kInstrSize));
156  }
157 #elif V8_TARGET_ARCH_RISCV64 || V8_TARGET_ARCH_RISCV32
158  Instruction* instr = reinterpret_cast<Instruction*>(rinfo->pc());
159  if (instr->IsAUIPC()) {
160  Instr auipc = instr->InstructionBits();
161  Instr jalr = reinterpret_cast<Instruction*>(rinfo->pc() + 1 * kInstrSize)
162  ->InstructionBits();
163  DCHECK(is_int32(tag + 0x800));
164  Assembler::PatchBranchlongOffset(rinfo->pc(), auipc, jalr, (int32_t)tag,
165  nullptr);
166  } else {
167  Assembler::set_target_address_at(rinfo->pc(), rinfo->constant_pool(),
168  static_cast<Address>(tag), nullptr,
169  SKIP_ICACHE_FLUSH);
170  }
171 #else
172  Address addr = static_cast<Address>(tag);
173  if (rinfo->rmode() == RelocInfo::EXTERNAL_REFERENCE) {
174  rinfo->set_target_external_reference(addr, SKIP_ICACHE_FLUSH);
175  } else if (rinfo->rmode() == RelocInfo::WASM_STUB_CALL) {
176  rinfo->set_wasm_stub_call_address(addr);
177  } else {
178  rinfo->set_target_address(addr, SKIP_ICACHE_FLUSH);
179  }
180 #endif
181 }

183 uint32_t GetWasmCalleeTag(RelocInfo* rinfo) {
184 #if V8_TARGET_ARCH_X64 || V8_TARGET_ARCH_IA32
185  DCHECK(!RelocInfo::IsCompressedEmbeddedObject(rinfo->rmode()));
186  return ReadUnalignedValue<uint32_t>(rinfo->target_address_address());
187 #elif V8_TARGET_ARCH_ARM64
188  Instruction* instr = reinterpret_cast<Instruction*>(rinfo->pc());
189  if (instr->IsLdrLiteralX()) {
190  return ReadUnalignedValue<uint32_t>(rinfo->constant_pool_entry_address());
191  } else {
192  DCHECK(instr->IsBranchAndLink() || instr->IsUnconditionalBranch());
193  return static_cast<uint32_t>(instr->ImmPCOffset() / kInstrSize);
194  }
195 #elif V8_TARGET_ARCH_RISCV64 || V8_TARGET_ARCH_RISCV32
196  Instruction* instr = reinterpret_cast<Instruction*>(rinfo->pc());
197  if (instr->IsAUIPC()) {
198  Instr auipc = instr->InstructionBits();
199  Instr jalr = reinterpret_cast<Instruction*>(rinfo->pc() + 1 * kInstrSize)
200  ->InstructionBits();
201  return Assembler::BrachlongOffset(auipc, jalr);
202  } else {
203  return static_cast<uint32_t>(rinfo->target_address());
204  }
205 #else
206  Address addr;
207  if (rinfo->rmode() == RelocInfo::EXTERNAL_REFERENCE) {
208  addr = rinfo->target_external_reference();
209  } else if (rinfo->rmode() == RelocInfo::WASM_STUB_CALL) {
210  addr = rinfo->wasm_stub_call_address();
211  } else {
212  addr = rinfo->target_address();
213  }
214  return static_cast<uint32_t>(addr);
215 #endif
216 }

237 // A List of all isolate-independent external references. This is used to create
238 // a tag from the Address of an external reference and vice versa.
239 class ExternalReferenceList {
240  public:
241  ExternalReferenceList(const ExternalReferenceList&) = delete;
242  ExternalReferenceList& operator=(const ExternalReferenceList&) = delete;
243 
244  uint32_t tag_from_address(Address ext_ref_address) const {
245  auto tag_addr_less_than = [this](uint32_t tag, Address searched_addr) {
246  return external_reference_by_tag_[tag] < searched_addr;
247  };
248  auto it = std::lower_bound(std::begin(tags_ordered_by_address_),
249  std::end(tags_ordered_by_address_),
250  ext_ref_address, tag_addr_less_than);
251  DCHECK_NE(std::end(tags_ordered_by_address_), it);
252  uint32_t tag = *it;
253  DCHECK_EQ(address_from_tag(tag), ext_ref_address);
254  return tag;
255  }
256 
257  Address address_from_tag(uint32_t tag) const {
258  DCHECK_GT(kNumExternalReferences, tag);
259  return external_reference_by_tag_[tag];
260  }
261 
262  static const ExternalReferenceList& Get() {
263  static ExternalReferenceList list; // Lazily initialized.
264  return list;
265  }
266 
267  private:
268  // Private constructor. There will only be a single instance of this object.
269  ExternalReferenceList() {
270  for (uint32_t i = 0; i < kNumExternalReferences; ++i) {
271  tags_ordered_by_address_[i] = i;
272  }
273  auto addr_by_tag_less_than = [this](uint32_t a, uint32_t b) {
274  return external_reference_by_tag_[a] < external_reference_by_tag_[b];
275  };
276  std::sort(std::begin(tags_ordered_by_address_),
277  std::end(tags_ordered_by_address_), addr_by_tag_less_than);
278  }

358 size_t NativeModuleSerializer::Measure() const {
359  // From {WriteHeader}:
360  size_t size = sizeof(WasmDetectedFeatures::StorageType) +
361  sizeof(size_t) + // total code size
362  sizeof(bool) + // all functions validated
363  sizeof(typename CompileTimeImportFlags::StorageType) +
364  sizeof(uint32_t) + // length of constants_module.
365  native_module_->compile_imports().constants_module().size() +
366  import_statuses_.size() * sizeof(WellKnownImport);
367 
368  // From {WriteCode}, called repeatedly.
369  for (WasmCode* code : code_table_) {
370  size += MeasureCode(code);
371  }
372 
373  // Tiering budget, wrote in {Write} directly.
374  size += native_module_->module()->num_declared_functions * sizeof(uint32_t);
375 
376  return size;

379 void NativeModuleSerializer::WriteHeader(Writer* writer,
380  size_t total_code_size) {
381  // TODO(eholk): We need to properly preserve the flag whether the trap
382  // handler was used or not when serializing.
383 
384  // Serialize the set of detected features; this contains
385  // - all features detected during module decoding,
386  // - all features detected during function body decoding (if lazy validation
387  // is disabled), and
388  // - some features detected during compilation; some might still be missing
389  // because installing code and publishing detected features is not atomic.
390  writer->Write(
391  native_module_->compilation_state()->detected_features().ToIntegral());
392 
393  writer->Write(total_code_size);
394 
395  // We do not ship lazy validation, so in most cases all functions will be
396  // validated. Thus only write out a single bit instead of serializing the
397  // information per function.
398  const bool fully_validated = !v8_flags.wasm_lazy_validation;
399  writer->Write(fully_validated);
400 #ifdef DEBUG
401  if (fully_validated) {
402  const WasmModule* module = native_module_->module();
403  for (auto& function : module->declared_functions()) {
404  DCHECK(module->function_was_validated(function.func_index));
405  }
406  }
407 #endif
408 
409  const CompileTimeImports& compile_imports = native_module_->compile_imports();
410  const std::string& constants_module = compile_imports.constants_module();
411  writer->Write(compile_imports.flags().ToIntegral());
412  writer->Write(static_cast<uint32_t>(constants_module.size()));
413  writer->WriteVector(base::VectorOf(constants_module));
414  writer->WriteVector(base::VectorOf(import_statuses_));
415 }

417 void NativeModuleSerializer::WriteCode(
418  const WasmCode* code, Writer* writer,
419  const NativeModule::CallIndirectTargetMap& function_index_map) {
420  if (code == nullptr) {
421  writer->Write(kLazyFunction);
422  return;
423  }
424 
425  DCHECK_EQ(WasmCode::kWasmFunction, code->kind());
426  // Only serialize TurboFan code, as Liftoff code can contain breakpoints or
427  // non-relocatable constants.
428  if (code->tier() != ExecutionTier::kTurbofan) {
429  // We check if the function has been executed already. If so, we serialize
430  // it as {kEagerFunction} so that upon deserialization the function will
431  // get eagerly compiled with Liftoff (if enabled). If the function has not
432  // been executed yet, we serialize it as {kLazyFunction}, and the function
433  // will not get compiled upon deserialization.
434  NativeModule* native_module = code->native_module();
435  uint32_t budget = native_module
436  ->tiering_budget_array()[declared_function_index(
437  native_module->module(), code->index())]
438  .load(std::memory_order_relaxed);
439  writer->Write(budget == static_cast<uint32_t>(v8_flags.wasm_tiering_budget)
440  ? kLazyFunction
441  : kEagerFunction);
442  return;
443  }
444 
445  ++num_turbofan_functions_;
446  writer->Write(kTurboFanFunction);
447  // Write the size of the entire code section, followed by the code header.
448  writer->Write(code->constant_pool_offset());
449  writer->Write(code->safepoint_table_offset());
450  writer->Write(code->handler_table_offset());
451  writer->Write(code->code_comments_offset());
452  writer->Write(code->jump_table_info_offset());
453  writer->Write(code->unpadded_binary_size());
454  writer->Write(code->stack_slots());
455  writer->Write(code->ool_spills());
456  writer->Write(code->raw_tagged_parameter_slots_for_serialization());
457  writer->Write(code->instructions().length());
458  writer->Write(code->reloc_info().length());
459  writer->Write(code->source_positions().length());
460  writer->Write(code->inlining_positions().length());
461  writer->Write(code->deopt_data().length());
462  writer->Write(code->protected_instructions_data().length());
463  writer->Write(code->kind());
464  writer->Write(code->tier());
465 
466  // Get a pointer to the destination buffer, to hold relocated code.
467  uint8_t* serialized_code_start = writer->current_buffer().begin();
468  uint8_t* code_start = serialized_code_start;
469  size_t code_size = code->instructions().size();
470  writer->Skip(code_size);
471  // Write the reloc info, source positions, inlining positions and protected
472  // code.
473  writer->WriteVector(code->reloc_info());
474  writer->WriteVector(code->source_positions());
475  writer->WriteVector(code->inlining_positions());
476  writer->WriteVector(code->deopt_data());
477  writer->WriteVector(code->protected_instructions_data());
478 #if V8_TARGET_ARCH_MIPS64 || V8_TARGET_ARCH_ARM || V8_TARGET_ARCH_PPC64 || \
479  V8_TARGET_ARCH_S390X || V8_TARGET_ARCH_RISCV32 || V8_TARGET_ARCH_RISCV64
480  // On platforms that don't support misaligned word stores, copy to an aligned
481  // buffer if necessary so we can relocate the serialized code.
482  std::unique_ptr<uint8_t[]> aligned_buffer;
483  if (!IsAligned(reinterpret_cast<Address>(serialized_code_start),
484  kSystemPointerSize)) {
485  // 'uint8_t' does not guarantee an alignment but seems to work well enough
486  // in practice.
487  aligned_buffer.reset(new uint8_t[code_size]);
488  code_start = aligned_buffer.get();
489  }
490 #endif
491  memcpy(code_start, code->instructions().begin(), code_size);
492  // Relocate the code.
493  constexpr int kMask =
494  RelocInfo::ModeMask(RelocInfo::WASM_CALL) |
495  RelocInfo::ModeMask(RelocInfo::WASM_STUB_CALL) |
496  RelocInfo::ModeMask(RelocInfo::WASM_CANONICAL_SIG_ID) |
497  RelocInfo::ModeMask(RelocInfo::WASM_CODE_POINTER_TABLE_ENTRY) |
498  RelocInfo::ModeMask(RelocInfo::EXTERNAL_REFERENCE) |
499  RelocInfo::ModeMask(RelocInfo::INTERNAL_REFERENCE) |
500  RelocInfo::ModeMask(RelocInfo::INTERNAL_REFERENCE_ENCODED);
501  RelocIterator orig_iter(code->instructions(), code->reloc_info(),
502  code->constant_pool(), kMask);
503 
504  WritableJitAllocation jit_allocation =
505  WritableJitAllocation::ForNonExecutableMemory(
506  reinterpret_cast<Address>(code_start), code->instructions().size(),
507  ThreadIsolation::JitAllocationType::kWasmCode);
508  for (WritableRelocIterator iter(
509  jit_allocation, {code_start, code->instructions().size()},
510  code->reloc_info(),
511  reinterpret_cast<Address>(code_start) + code->constant_pool_offset(),
512  kMask);
513  !iter.done(); iter.next(), orig_iter.next()) {
514  RelocInfo::Mode mode = orig_iter.rinfo()->rmode();
515  switch (mode) {
516  case RelocInfo::WASM_CALL: {
517  Address orig_target = orig_iter.rinfo()->wasm_call_address();
518  uint32_t tag =
519  native_module_->GetFunctionIndexFromJumpTableSlot(orig_target);
520  SetWasmCalleeTag(iter.rinfo(), tag);
521  } break;
522  case RelocInfo::WASM_STUB_CALL: {
523  Address target = orig_iter.rinfo()->wasm_stub_call_address();
524  uint32_t tag = static_cast<uint32_t>(
525  native_module_->GetBuiltinInJumptableSlot(target));
526  SetWasmCalleeTag(iter.rinfo(), tag);
527  } break;
528  case RelocInfo::WASM_CANONICAL_SIG_ID: {
529  uint32_t canonical_sig_id = orig_iter.rinfo()->wasm_canonical_sig_id();
530  uint32_t module_local_sig_id =
531  CanonicalSigIdToModuleLocalTypeId(canonical_sig_id);
532  iter.rinfo()->set_wasm_canonical_sig_id(module_local_sig_id);
533  } break;
534  case RelocInfo::WASM_CODE_POINTER_TABLE_ENTRY: {
535  WasmCodePointer target =
536  orig_iter.rinfo()->wasm_code_pointer_table_entry();
537  uint32_t function_index = function_index_map.at(target);
538  iter.rinfo()->set_wasm_code_pointer_table_entry(
539  WasmCodePointer{function_index}, SKIP_ICACHE_FLUSH);
540  } break;
541  case RelocInfo::EXTERNAL_REFERENCE: {
542  Address orig_target = orig_iter.rinfo()->target_external_reference();
543  uint32_t ext_ref_tag =
544  ExternalReferenceList::Get().tag_from_address(orig_target);
545  SetWasmCalleeTag(iter.rinfo(), ext_ref_tag);
546  } break;
547  case RelocInfo::INTERNAL_REFERENCE:
548  case RelocInfo::INTERNAL_REFERENCE_ENCODED: {
549  Address orig_target = orig_iter.rinfo()->target_internal_reference();
550  Address offset = orig_target - code->instruction_start();
551  Assembler::deserialization_set_target_internal_reference_at(
552  iter.rinfo()->pc(), offset, jit_allocation, mode);
553  } break;
554  default:
555  UNREACHABLE();
556  }
557  }
558  // If we copied to an aligned buffer, copy code into serialized buffer.
559  if (code_start != serialized_code_start) {
560  memcpy(serialized_code_start, code_start, code_size);
561  }
562  total_written_code_ += code_size;
563 }

738 class DeserializeCodeTask : public JobTask {
739  public:
740  DeserializeCodeTask(NativeModuleDeserializer* deserializer,
741  DeserializationQueue* reloc_queue)
742  : deserializer_(deserializer), reloc_queue_(reloc_queue) {}
743 
744  void Run(JobDelegate* delegate) override {
745  bool finished = false;
746  while (!finished) {
747  // Repeatedly publish everything that was copied already.
748  finished = TryPublishing(delegate);
749 
750  auto batch = reloc_queue_->Pop();
751  if (batch.empty()) break;
752  for (const auto& unit : batch) {
753  deserializer_->CopyAndRelocate(unit);
754  }
755  publish_queue_.Add(std::move(batch));
756  delegate->NotifyConcurrencyIncrease();
757  }
758  }
759 
760  size_t GetMaxConcurrency(size_t /* worker_count */) const override {
761  // Number of copy&reloc batches, plus 1 if there is also something to
762  // publish.
763  bool publish = publishing_.load(std::memory_order_relaxed) == false &&
764  publish_queue_.NumBatches() > 0;
765  return reloc_queue_->NumBatches() + (publish ? 1 : 0);
766  }
767 
768  private:
769  bool TryPublishing(JobDelegate* delegate) {
770  // Publishing is sequential, so only start publishing if no one else is.
771  if (publishing_.exchange(true, std::memory_order_relaxed)) return false;
772 
773  WasmCodeRefScope code_scope;
774  while (true) {
775  bool yield = false;
776  while (!yield) {
777  auto to_publish = publish_queue_.PopAll();
778  if (to_publish.empty()) break;
779  deserializer_->Publish(std::move(to_publish));
780  yield = delegate->ShouldYield();
781  }
782  publishing_.store(false, std::memory_order_relaxed);
783  if (yield) return true;
784  // After finishing publishing, check again if new work arrived in the mean
785  // time. If so, continue publishing.
786  if (publish_queue_.NumBatches() == 0) break;
787  if (publishing_.exchange(true, std::memory_order_relaxed)) break;
788  // We successfully reset {publishing_} from {false} to {true}.
789  }
790  return false;
791  }
792 
793  NativeModuleDeserializer* const deserializer_;
794  DeserializationQueue* const reloc_queue_;
795  DeserializationQueue publish_queue_;
796  std::atomic<bool> publishing_{false};
797 };

802 bool NativeModuleDeserializer::Read(Reader* reader) {
803  DCHECK(!read_called_);
804 #ifdef DEBUG
805  read_called_ = true;
806 #endif
807 
808  ReadHeader(reader);
809  if (compile_imports_.compare(native_module_->compile_imports()) != 0) {
810  return false;
811  }
812 
813  uint32_t total_fns = native_module_->num_functions();
814  uint32_t first_wasm_fn = native_module_->num_imported_functions();
815 
816  if (all_functions_validated_) {
817  native_module_->module()->set_all_functions_validated();
818  }
819 
820  WasmCodeRefScope wasm_code_ref_scope;
821 
822  DeserializationQueue reloc_queue;
823 
824  // Create a new job without any workers; those are spawned on
825  // {NotifyConcurrencyIncrease}.
826  std::unique_ptr<JobHandle> job_handle = V8::GetCurrentPlatform()->CreateJob(
827  TaskPriority::kUserVisible,
828  std::make_unique<DeserializeCodeTask>(this, &reloc_queue));
829 
830  // Choose a batch size such that we do not create too small batches (>=100k
831  // code bytes), but also not too many (<=100 batches).
832  constexpr size_t kMinBatchSizeInBytes = 100000;
833  size_t batch_limit =
834  std::max(kMinBatchSizeInBytes, remaining_code_size_ / 100);
835 
836  std::vector<DeserializationUnit> batch;
837  size_t batch_size = 0;
838  for (uint32_t i = first_wasm_fn; i < total_fns; ++i) {
839  DeserializationUnit unit = ReadCode(i, reader);
840  if (!unit.code) continue;
841  batch_size += unit.code->instructions().size();
842  batch.emplace_back(std::move(unit));
843  if (batch_size >= batch_limit) {
844  reloc_queue.Add(std::move(batch));
845  DCHECK(batch.empty());
846  batch_size = 0;
847  job_handle->NotifyConcurrencyIncrease();
848  }
849  }
850 
851  // We should have read the expected amount of code now, and should have fully
852  // utilized the allocated code space.
853  DCHECK_EQ(0, remaining_code_size_);
854  DCHECK_EQ(0, current_code_space_.size());
855 
856  if (!batch.empty()) {
857  reloc_queue.Add(std::move(batch));
858  job_handle->NotifyConcurrencyIncrease();
859  }
860 
861  // Wait for all tasks to finish, while participating in their work.
862  job_handle->Join();
863 
864  ReadTieringBudget(reader);
865  return reader->current_size() == 0;
866 }

897 DeserializationUnit NativeModuleDeserializer::ReadCode(int fn_index,
898  Reader* reader) {
899  uint8_t code_kind = reader->Read<uint8_t>();
900  if (code_kind == kLazyFunction) {
901  lazy_functions_.push_back(fn_index);
902  return {};
903  }
904  if (code_kind == kEagerFunction) {
905  eager_functions_.push_back(fn_index);
906  return {};
907  }
908 
909  int constant_pool_offset = reader->Read<int>();
910  int safepoint_table_offset = reader->Read<int>();
911  int handler_table_offset = reader->Read<int>();
912  int code_comment_offset = reader->Read<int>();
913  int jump_table_info_offset = reader->Read<int>();
914  int unpadded_binary_size = reader->Read<int>();
915  int stack_slot_count = reader->Read<int>();
916  int ool_spill_count = reader->Read<int>();
917  uint32_t tagged_parameter_slots = reader->Read<uint32_t>();
918  int code_size = reader->Read<int>();
919  int reloc_size = reader->Read<int>();
920  int source_position_size = reader->Read<int>();
921  int inlining_position_size = reader->Read<int>();
922  int deopt_data_size = reader->Read<int>();
923  // TODO(mliedtke): protected_instructions_data is the first part of the
924  // meta_data_ array. Ideally the sizes would be in the same order...
925  int protected_instructions_size = reader->Read<int>();
926  WasmCode::Kind kind = reader->Read<WasmCode::Kind>();
927  ExecutionTier tier = reader->Read<ExecutionTier>();
928 
929  DCHECK(IsAligned(code_size, kCodeAlignment));
930  DCHECK_GE(remaining_code_size_, code_size);
931  if (current_code_space_.size() < static_cast<size_t>(code_size)) {
932  // Allocate the next code space. Don't allocate more than 90% of
933  // {kMaxCodeSpaceSize}, to leave some space for jump tables.
934  size_t max_reservation = RoundUp<kCodeAlignment>(
935  v8_flags.wasm_max_code_space_size_mb * MB * 9 / 10);
936  size_t code_space_size = std::min(max_reservation, remaining_code_size_);
937  std::tie(current_code_space_, current_jump_tables_) =
938  native_module_->AllocateForDeserializedCode(code_space_size);
939  DCHECK_EQ(current_code_space_.size(), code_space_size);
940  CHECK(current_jump_tables_.is_valid());
941  }
942 
943  DeserializationUnit unit;
944  unit.src_code_buffer = reader->ReadVector<uint8_t>(code_size);
945  auto reloc_info = reader->ReadVector<uint8_t>(reloc_size);
946  auto source_pos = reader->ReadVector<uint8_t>(source_position_size);
947  auto inlining_pos = reader->ReadVector<uint8_t>(inlining_position_size);
948  auto deopt_data = reader->ReadVector<uint8_t>(deopt_data_size);
949  auto protected_instructions =
950  reader->ReadVector<uint8_t>(protected_instructions_size);
951 
952  base::Vector<uint8_t> instructions =
953  current_code_space_.SubVector(0, code_size);
954  current_code_space_ += code_size;
955  remaining_code_size_ -= code_size;
956 
957  unit.code = native_module_->AddDeserializedCode(
958  fn_index, instructions, stack_slot_count, ool_spill_count,
959  tagged_parameter_slots, safepoint_table_offset, handler_table_offset,
960  constant_pool_offset, code_comment_offset, jump_table_info_offset,
961  unpadded_binary_size, protected_instructions, reloc_info, source_pos,
962  inlining_pos, deopt_data, kind, tier);
963  unit.jump_tables = current_jump_tables_;
964  return unit;
965 }

967 void NativeModuleDeserializer::CopyAndRelocate(
968  const DeserializationUnit& unit) {
969  WritableJitAllocation jit_allocation = ThreadIsolation::RegisterJitAllocation(
970  reinterpret_cast<Address>(unit.code->instructions().begin()),
971  unit.code->instructions().size(),
972  ThreadIsolation::JitAllocationType::kWasmCode, false);
973 
974  jit_allocation.CopyCode(0, unit.src_code_buffer.begin(),
975  unit.src_code_buffer.size());
976 
977  // Relocate the code.
978  int kMask = RelocInfo::ModeMask(RelocInfo::WASM_CALL) |
979  RelocInfo::ModeMask(RelocInfo::WASM_STUB_CALL) |
980  RelocInfo::ModeMask(RelocInfo::WASM_CANONICAL_SIG_ID) |
981  RelocInfo::ModeMask(RelocInfo::WASM_CODE_POINTER_TABLE_ENTRY) |
982  RelocInfo::ModeMask(RelocInfo::EXTERNAL_REFERENCE) |
983  RelocInfo::ModeMask(RelocInfo::INTERNAL_REFERENCE) |
984  RelocInfo::ModeMask(RelocInfo::INTERNAL_REFERENCE_ENCODED);
985  for (WritableRelocIterator iter(jit_allocation, unit.code->instructions(),
986  unit.code->reloc_info(),
987  unit.code->constant_pool(), kMask);
988  !iter.done(); iter.next()) {
989  RelocInfo::Mode mode = iter.rinfo()->rmode();
990  switch (mode) {
991  case RelocInfo::WASM_CALL: {
992  uint32_t tag = GetWasmCalleeTag(iter.rinfo());
993  Address target =
994  native_module_->GetNearCallTargetForFunction(tag, unit.jump_tables);
995  iter.rinfo()->set_wasm_call_address(target);
996  break;
997  }
998  case RelocInfo::WASM_STUB_CALL: {
999  uint32_t tag = GetWasmCalleeTag(iter.rinfo());
1000  Address target = native_module_->GetJumpTableEntryForBuiltin(
1001  static_cast<Builtin>(tag), unit.jump_tables);
1002  iter.rinfo()->set_wasm_stub_call_address(target);
1003  break;
1004  }
1005  case RelocInfo::WASM_CANONICAL_SIG_ID: {
1006  // This is intentional: in serialized code, we patched embedded
1007  // canonical signature IDs with their module-specific equivalents,
1008  // so although the accessor is called "wasm_canonical_sig_id()", what
1009  // we get back is actually a module-specific signature ID, which we
1010  // now need to translate back to a canonical ID.
1011  ModuleTypeIndex module_local_sig_id{
1012  iter.rinfo()->wasm_canonical_sig_id()};
1013  CanonicalTypeIndex canonical_sig_id =
1014  native_module_->module()->canonical_sig_id(module_local_sig_id);
1015  iter.rinfo()->set_wasm_canonical_sig_id(canonical_sig_id.index);
1016  } break;
1017  case RelocInfo::WASM_CODE_POINTER_TABLE_ENTRY: {
1018  Address function_index =
1019  iter.rinfo()->wasm_code_pointer_table_entry().value();
1020  WasmCodePointer target = native_module_->GetCodePointerHandle(
1021  base::checked_cast<uint32_t>(function_index));
1022  iter.rinfo()->set_wasm_code_pointer_table_entry(target,
1023  SKIP_ICACHE_FLUSH);
1024  } break;
1025  case RelocInfo::EXTERNAL_REFERENCE: {
1026  uint32_t tag = GetWasmCalleeTag(iter.rinfo());
1027  Address address = ExternalReferenceList::Get().address_from_tag(tag);
1028  iter.rinfo()->set_target_external_reference(address, SKIP_ICACHE_FLUSH);
1029  break;
1030  }
1031  case RelocInfo::INTERNAL_REFERENCE:
1032  case RelocInfo::INTERNAL_REFERENCE_ENCODED: {
1033  Address offset = iter.rinfo()->target_internal_reference();
1034  Address target = unit.code->instruction_start() + offset;
1035  Assembler::deserialization_set_target_internal_reference_at(
1036  iter.rinfo()->pc(), target, jit_allocation, mode);
1037  break;
1038  }
1039  default:
1040  UNREACHABLE();
1041  }
1042  }
1043 
1044  // Finally, flush the icache for that code.
1045  FlushInstructionCache(unit.code->instructions().begin(),
1046  unit.code->instructions().size());
1047 }

1091 MaybeDirectHandle<WasmModuleObject> DeserializeNativeModule(
1092  Isolate* isolate, base::Vector<const uint8_t> data,
1093  base::Vector<const uint8_t> wire_bytes_vec,
1094  const CompileTimeImports& compile_imports,
1095  base::Vector<const char> source_url) {

1097  WasmEnabledFeatures::FromIsolate(isolate);
1098  if (!IsWasmCodegenAllowed(isolate, isolate->native_context())) return {};
1099  if (!IsSupportedVersion(data, enabled_features)) return {};
1100 
1101  // Make the copy of the wire bytes early, so we use the same memory for
1102  // decoding, lookup in the native module cache, and insertion into the cache.
1103  base::OwnedVector<const uint8_t> owned_wire_bytes =
1104  base::OwnedCopyOf(wire_bytes_vec);
1105 
1106  WasmDetectedFeatures detected_features;
1107  ModuleResult decode_result = DecodeWasmModule(
1108  enabled_features, owned_wire_bytes.as_vector(), false,
1109  i::wasm::kWasmOrigin, isolate->counters(), isolate->metrics_recorder(),
1110  isolate->GetOrRegisterRecorderContextId(isolate->native_context()),
1111  DecodingMethod::kDeserialize, &detected_features);
