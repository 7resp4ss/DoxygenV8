306 template <typename IsolateT>
307 SharedFunctionInfo::Inlineability SharedFunctionInfo::GetInlineability(
308  CodeKind code_kind, IsolateT* isolate) const {
309  if (!IsScript(script())) return kHasNoScript;
310 
311  if (isolate->is_precise_binary_code_coverage() &&
312  !has_reported_binary_coverage()) {
313  // We may miss invocations if this function is inlined.
314  return kNeedsBinaryCoverage;
315  }
316 
317  // Built-in functions are handled by the JSCallReducer.
318  if (HasBuiltinId()) return kIsBuiltin;
319 
320  if (!IsUserJavaScript()) return kIsNotUserCode;
321 
322  // If there is no bytecode array, it is either not compiled or it is compiled
323  // with WebAssembly for the asm.js pipeline. In either case we don't want to
324  // inline.
325  if (!HasBytecodeArray()) return kHasNoBytecode;
326 
327  if (GetBytecodeArray(isolate)->length() >
328  v8_flags.max_inlined_bytecode_size) {
329  return kExceedsBytecodeLimit;
330  }
331 
332  {
333  MutexGuardIfOffThread<IsolateT> mutex_guard(
334  isolate->shared_function_info_access(), isolate);
335  if (HasBreakInfo(isolate->GetMainThreadIsolateUnsafe())) {
336  return kMayContainBreakPoints;
337  }
338  }
339 
340  if (optimization_disabled(code_kind)) return kHasOptimizationDisabled;
341 
342  return kIsInlineable;
343 }

556 void SharedFunctionInfo::SetScopeInfo(Tagged<ScopeInfo> scope_info,
557  WriteBarrierMode mode) {
558  // Move the existing name onto the ScopeInfo.
559  Tagged<NameOrScopeInfoT> name_or_scope_info =
560  this->name_or_scope_info(kAcquireLoad);
561  Tagged<UnionOf<Smi, String>> name;
562  if (IsScopeInfo(name_or_scope_info)) {
563  name = Cast<ScopeInfo>(name_or_scope_info)->FunctionName();
564  } else {
565  name = Cast<UnionOf<Smi, String>>(name_or_scope_info);
566  }
567  DCHECK(IsString(name) || name == kNoSharedNameSentinel);
568  // ScopeInfo can get promoted to read-only space. Now that we reuse them after
569  // flushing bytecode, we'll actually reinstall read-only scopeinfos on
570  // SharedFunctionInfos if they required a context. The read-only scopeinfos
571  // should already be fully initialized though, and hence will already have the
572  // right FunctionName (and InferredName if relevant).
573  if (scope_info->FunctionName() != name) {
574  scope_info->SetFunctionName(name);
575  }
576  if (HasInferredName() && inferred_name()->length() != 0 &&
577  scope_info->InferredFunctionName() != inferred_name()) {
578  scope_info->SetInferredFunctionName(inferred_name());
579  }

656 IsCompiledScope::IsCompiledScope(const Tagged<SharedFunctionInfo> shared,
657  Isolate* isolate) {
658  Tagged<Object> data_obj = shared->GetTrustedData(isolate);
659  if (Tagged<HeapObject> data; TryCast<HeapObject>(data_obj, &data)) {
660  if (Tagged<Code> code; TryCast<Code>(data, &code)) {
661  DCHECK_EQ(code->kind(), CodeKind::BASELINE);
662  data = code->bytecode_or_interpreter_data();
663  }
664  // Unlike GetBytecodeArray, we don't bother checking for DebugInfo here. If
665  // there is DebugInfo, then it will hold both the debug and original
666  // BytecodeArray strongly, so it doesn't matter which of those we hold.
667  if (Tagged<BytecodeArray> bytecode;
668  TryCast<BytecodeArray>(data, &bytecode)) {
669  retain_code_ = handle(bytecode, isolate);
670  is_compiled_ = true;
671  } else if (Tagged<InterpreterData> interpreter_data;
672  TryCast<InterpreterData>(data, &interpreter_data)) {
673  retain_code_ = handle(interpreter_data->bytecode_array(), isolate);
674  is_compiled_ = true;
675  } else if (Is<UncompiledData>(data)) {
676  retain_code_ = {};
677  is_compiled_ = false;
678  } else {
679  retain_code_ = {};
680  is_compiled_ = shared->is_compiled();
681  }
682  } else {
683  retain_code_ = {};
684  is_compiled_ = shared->is_compiled();
685  }
686 
687  DCHECK_IMPLIES(!retain_code_.is_null(), is_compiled());
688  DCHECK_EQ(shared->is_compiled(), is_compiled());
689 }

1083  GetReadOnlyRoots().uncompiled_data_without_preparse_data_map(),
