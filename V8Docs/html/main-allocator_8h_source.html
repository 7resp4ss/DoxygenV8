119 class LinearAreaOriginalData {
120  public:
121  // Loads top and limit fields using the synchronization protocol without
122  // holding the mutex lock. Note that in the returned (top, limit) pair, top
123  // and limit might not actually belong together for brief moments when
124  // changing LABs. It could happen that the LAB is invalid (limit<top) or that
125  // the LAB is longer than needed (when old limit <= new limit). However, this
126  // is fine for concurrent marking which would just push a few more objects to
127  // the "on_hold" queue then necessary.
128  //
129  // If top and limit need to be consistent with respect to each other use
130  // GetTopAndLimitLocked() instead.
131  std::pair<Address, Address> GetTopAndLimit() const {
132  // The order of the two loads is important. See SetTopAndLimit().
133  auto top = original_top_.load(std::memory_order_acquire);
134  auto limit = original_limit_.load(std::memory_order_relaxed);
135  return std::make_pair(top, limit);
136  }
137 
138  // Same as GetTopAndLimit() but also locks the mutex.
139  std::pair<Address, Address> GetTopAndLimitLocked() const;
140 
141  void SetTopAndLimit(Address top, Address limit);
142 
143  private:
144  // The top and the limit at the time of setting the linear allocation area.
145  // These values can be accessed by background tasks. Protected by mutex_.
146  std::atomic<Address> original_top_ = 0;
147  std::atomic<Address> original_limit_ = 0;
148 
149  // Protects original_top_ and original_limit_.
150  mutable base::Mutex mutex_;
151 };
