598  if (instance_type_ == JS_DATA_VIEW_TYPE) return 0;
599  if (candidates_.empty()) return std::nullopt;
600  int shift = ElementsKindToShiftSize(*candidates_.begin());
601  if (!base::all_of(candidates_, [shift](auto e) {
602  return ElementsKindToShiftSize(e) == shift;
603  })) {
604  return std::nullopt;
605  }
606  return shift;
607  }
608 
609  std::optional<int> TryComputeStaticElementSize() {
610  DCHECK(instance_type_ != JS_RAB_GSAB_DATA_VIEW_TYPE);
611  if (instance_type_ == JS_DATA_VIEW_TYPE) return 1;
612  if (candidates_.empty()) return std::nullopt;
613  int size = ElementsKindToByteSize(*candidates_.begin());
614  if (!base::all_of(candidates_, [size](auto e) {
615  return ElementsKindToByteSize(e) == size;
616  })) {
617  return std::nullopt;
618  }
619  return size;
620  }

622  TNode<UintPtrT> BuildLength(TNode<JSArrayBufferView> view,
623  TNode<Context> context) {
624  auto& a = *assembler_;
625 
626  // Case 1: Normal (backed by AB/SAB) or non-length tracking backed by GSAB
627  // (can't go oob once constructed)
628  auto GsabFixedOrNormal = [&]() {
629  TNode<UintPtrT> byte_length = MachineLoadField<UintPtrT>(
630  AccessBuilder::ForJSArrayBufferViewByteLength(), view,
631  UseInfo::Word());
632  if (auto size_opt = TryComputeStaticElementSize()) {
633  return a.UintPtrDiv(byte_length, a.UintPtrConstant(*size_opt));
634  } else {
635  TNode<Map> typed_array_map = a.LoadField<Map>(
636  AccessBuilder::ForMap(WriteBarrierKind::kNoWriteBarrier), view);
637  TNode<Uint32T> elements_kind = a.LoadElementsKind(typed_array_map);
638  TNode<Uint32T> element_size =
639  a.LookupByteSizeForElementsKind(elements_kind);
640  return a.UintPtrDiv(byte_length, a.ChangeUint32ToUintPtr(element_size));
641  }
642  };
643 
644  // If we statically know we cannot have rab/gsab backed, we can simply
645  // load from the view.
646  if (!maybe_rab_gsab()) {
647  return GsabFixedOrNormal();
648  }
649 
650  // Otherwise, we need to generate the checks for the view's bitfield.
651  TNode<Word32T> bitfield = a.EnterMachineGraph<Word32T>(
652  a.LoadField<Word32T>(AccessBuilder::ForJSArrayBufferViewBitField(),
653  view),
654  UseInfo::TruncatingWord32());
655  TNode<Word32T> length_tracking_bit = a.Word32And(
656  bitfield, a.Uint32Constant(JSArrayBufferView::kIsLengthTracking));
657  TNode<Word32T> backed_by_rab_bit = a.Word32And(
658  bitfield, a.Uint32Constant(JSArrayBufferView::kIsBackedByRab));
659 
660  // Load the underlying buffer.
661  TNode<HeapObject> buffer = a.LoadField<HeapObject>(
662  AccessBuilder::ForJSArrayBufferViewBuffer(), view);
663 
664  // Compute the element size.
665  TNode<Uint32T> element_size;
666  if (auto size_opt = TryComputeStaticElementSize()) {
667  element_size = a.Uint32Constant(*size_opt);
668  } else {
669  DCHECK_EQ(instance_type_, JS_TYPED_ARRAY_TYPE);
670  TNode<Map> typed_array_map = a.LoadField<Map>(
671  AccessBuilder::ForMap(WriteBarrierKind::kNoWriteBarrier), view);
672  TNode<Uint32T> elements_kind = a.LoadElementsKind(typed_array_map);
673  element_size = a.LookupByteSizeForElementsKind(elements_kind);
674  }
675 
676  // 2) Fixed length backed by RAB (can go oob once constructed)
677  auto RabFixed = [&]() {
678  TNode<UintPtrT> unchecked_byte_length = MachineLoadField<UintPtrT>(
679  AccessBuilder::ForJSArrayBufferViewByteLength(), view,
680  UseInfo::Word());
681  TNode<UintPtrT> underlying_byte_length = MachineLoadField<UintPtrT>(
682  AccessBuilder::ForJSArrayBufferByteLength(), buffer, UseInfo::Word());
683  TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
684  AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
685  UseInfo::Word());
686 
687  TNode<UintPtrT> byte_length =
688  a
689  .MachineSelectIf<UintPtrT>(a.UintPtrLessThanOrEqual(
690  a.UintPtrAdd(byte_offset, unchecked_byte_length),
691  underlying_byte_length))
692  .Then([&]() { return unchecked_byte_length; })
693  .Else([&]() { return a.UintPtrConstant(0); })
694  .Value();
695  return a.UintPtrDiv(byte_length, a.ChangeUint32ToUintPtr(element_size));
696  };
697 
698  // 3) Length-tracking backed by RAB (JSArrayBuffer stores the length)
699  auto RabTracking = [&]() {
700  TNode<UintPtrT> byte_length = MachineLoadField<UintPtrT>(
701  AccessBuilder::ForJSArrayBufferByteLength(), buffer, UseInfo::Word());
702  TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
703  AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
704  UseInfo::Word());
705 
706  return a
707  .MachineSelectIf<UintPtrT>(
708  a.UintPtrLessThanOrEqual(byte_offset, byte_length))
709  .Then([&]() {
710  // length = floor((byte_length - byte_offset) / element_size)
711  return a.UintPtrDiv(a.UintPtrSub(byte_length, byte_offset),
712  a.ChangeUint32ToUintPtr(element_size));
713  })
714  .Else([&]() { return a.UintPtrConstant(0); })
715  .ExpectTrue()
716  .Value();
717  };
718 
719  // 4) Length-tracking backed by GSAB (BackingStore stores the length)
720  auto GsabTracking = [&]() {
721  TNode<Number> temp = TNode<Number>::UncheckedCast(a.TypeGuard(
722  TypeCache::Get()->kJSArrayBufferViewByteLengthType,
723  a.JSCallRuntime1(Runtime::kGrowableSharedArrayBufferByteLength,
724  buffer, context, std::nullopt, Operator::kNoWrite)));
725  TNode<UintPtrT> byte_length =
726  a.EnterMachineGraph<UintPtrT>(temp, UseInfo::Word());
727  TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
728  AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
729  UseInfo::Word());
730 
731  return a
732  .MachineSelectIf<UintPtrT>(
733  a.UintPtrLessThanOrEqual(byte_offset, byte_length))
734  .Then([&]() {
735  // length = floor((byte_length - byte_offset) / element_size)
736  return a.UintPtrDiv(a.UintPtrSub(byte_length, byte_offset),
737  a.ChangeUint32ToUintPtr(element_size));
738  })
739  .Else([&]() { return a.UintPtrConstant(0); })
740  .ExpectTrue()
741  .Value();
742  };
743 
744  return a.MachineSelectIf<UintPtrT>(length_tracking_bit)
745  .Then([&]() {
746  return a.MachineSelectIf<UintPtrT>(backed_by_rab_bit)
747  .Then(RabTracking)
748  .Else(GsabTracking)
749  .Value();
750  })
751  .Else([&]() {
752  return a.MachineSelectIf<UintPtrT>(backed_by_rab_bit)
753  .Then(RabFixed)
754  .Else(GsabFixedOrNormal)
755  .Value();
756  })
757  .Value();
758  }

824  a.LookupByteShiftForElementsKind(elements_kind);

835  UseInfo::Word());

838  .MachineSelectIf<UintPtrT>(

840  .Then([&]() {
841  return RoundDownToElementSize(
842  a.UintPtrSub(byte_length, byte_offset));
843  })
844  .Else([&]() { return a.UintPtrConstant(0); })

855  TNode<UintPtrT> byte_length =
856  a.EnterMachineGraph<UintPtrT>(temp, UseInfo::Word());

858  AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
859  UseInfo::Word());

889  TNode<Word32T> BuildDetachedOrOutOfBoundsCheck(
890  TNode<JSArrayBufferView> view) {
891  auto& a = *assembler_;
892 
893  // Load the underlying buffer and its bitfield.
894  TNode<HeapObject> buffer = a.LoadField<HeapObject>(
895  AccessBuilder::ForJSArrayBufferViewBuffer(), view);
896  // Mask the detached bit.
897  TNode<Word32T> detached_bit = a.ArrayBufferDetachedBit(buffer);
898 
899  // If we statically know we cannot have rab/gsab backed, we are done here.
900  if (!maybe_rab_gsab()) {
901  return detached_bit;
902  }
903 
904  // Otherwise, we need to generate the checks for the view's bitfield.
905  TNode<Word32T> bitfield = a.EnterMachineGraph<Word32T>(
906  a.LoadField<Word32T>(AccessBuilder::ForJSArrayBufferViewBitField(),
907  view),
908  UseInfo::TruncatingWord32());
909  TNode<Word32T> length_tracking_bit = a.Word32And(
910  bitfield, a.Uint32Constant(JSArrayBufferView::kIsLengthTracking));
911  TNode<Word32T> backed_by_rab_bit = a.Word32And(
912  bitfield, a.Uint32Constant(JSArrayBufferView::kIsBackedByRab));
913 
914  auto RabLengthTracking = [&]() {
915  TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
916  AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
917  UseInfo::Word());
918 
919  TNode<UintPtrT> underlying_byte_length = MachineLoadField<UintPtrT>(
920  AccessBuilder::ForJSArrayBufferByteLength(), buffer, UseInfo::Word());
921 
922  return a.Word32Or(detached_bit,
923  a.UintPtrLessThan(underlying_byte_length, byte_offset));
924  };
925 
926  auto RabFixed = [&]() {
927  TNode<UintPtrT> unchecked_byte_length = MachineLoadField<UintPtrT>(
928  AccessBuilder::ForJSArrayBufferViewByteLength(), view,
929  UseInfo::Word());
930  TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
931  AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
932  UseInfo::Word());
933 
934  TNode<UintPtrT> underlying_byte_length = MachineLoadField<UintPtrT>(
935  AccessBuilder::ForJSArrayBufferByteLength(), buffer, UseInfo::Word());
936 
937  return a.Word32Or(
938  detached_bit,
939  a.UintPtrLessThan(underlying_byte_length,
940  a.UintPtrAdd(byte_offset, unchecked_byte_length)));
941  };
942 
943  // Dispatch depending on rab/gsab and length tracking.
944  return a.MachineSelectIf<Word32T>(backed_by_rab_bit)
945  .Then([&]() {
946  return a.MachineSelectIf<Word32T>(length_tracking_bit)
947  .Then(RabLengthTracking)
948  .Else(RabFixed)
949  .Value();
950  })
951  .Else([&]() { return detached_bit; })
952  .Value();
953  }

1310 Node* GraphAssembler::AddNode(Node* node) {
1311  if (!inline_reducers_.empty() && !inline_reductions_blocked_) {
1312  // Reducers may add new nodes to the graph using this graph assembler,
1313  // however they should never introduce nodes that need further reduction,
1314  // so block reduction
1315  BlockInlineReduction scope(this);
1316  Reduction reduction;
1317  for (auto reducer : inline_reducers_) {
1318  reduction = reducer->Reduce(node, nullptr);
1319  if (reduction.Changed()) break;
1320  }
1321  if (reduction.Changed()) {
1322  Node* replacement = reduction.replacement();
1323  if (replacement != node) {
1324  // Replace all uses of node and kill the node to make sure we don't
1325  // leave dangling dead uses.
1326  NodeProperties::ReplaceUses(node, replacement, effect(), control());
1327  node->Kill();
1328  return replacement;
1329  }
1330  }
1331  }
1332 
1333  if (node->opcode() == IrOpcode::kTerminate) {
1334  return node;
1335  }
1336 
1337  UpdateEffectControlWith(node);
1338  return node;
1339 }
