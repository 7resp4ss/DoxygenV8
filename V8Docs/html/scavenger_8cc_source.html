91  void VisitExternalPointer(Tagged<HeapObject> host,
92  ExternalPointerSlot slot) override {
93 #ifdef V8_COMPRESS_POINTERS
94  DCHECK(!slot.tag_range().IsEmpty());
95  DCHECK(!IsSharedExternalPointerType(slot.tag_range()));
96  // TODO(chromium:337580006): Remove when pointer compression always uses
97  // EPT.
98  if (!slot.HasExternalPointerHandle()) return;
99  ExternalPointerHandle handle = slot.Relaxed_LoadHandle();
100  Heap* heap = scavenger_->heap();
101  ExternalPointerTable& table = heap->isolate()->external_pointer_table();
102 
103  // For survivor objects, the scavenger marks their EPT entries when they are
104  // copied and then sweeps the young EPT space at the end of collection,
105  // reclaiming unmarked EPT entries.
106  //
107  // However when promoting, we just evacuate the entry from new to old space.
108  // Usually the entry will be unmarked, unless the slot was initialized since
109  // the last GC (external pointer tags have the mark bit set), in which case
110  // it may be marked already. In any case, transfer the color from new to
111  // old EPT space.
112  table.Evacuate(heap->young_external_pointer_space(),
113  heap->old_external_pointer_space(), handle, slot.address(),
114  ExternalPointerTable::EvacuateMarkMode::kTransferMark);
115 #endif // V8_COMPRESS_POINTERS
116  }

132  template <typename TSlot>
133  V8_INLINE void VisitPointersImpl(Tagged<HeapObject> host, TSlot start,
134  TSlot end) {
135  using THeapObjectSlot = typename TSlot::THeapObjectSlot;
136  // Treat weak references as strong.
137  // TODO(marja): Proper weakness handling in the young generation.
138  for (TSlot slot = start; slot < end; ++slot) {
139  typename TSlot::TObject object = *slot;
140  Tagged<HeapObject> heap_object;
141  if (object.GetHeapObject(&heap_object)) {
142  HandleSlot(host, THeapObjectSlot(slot), heap_object);

147  template <typename THeapObjectSlot>
148  V8_INLINE void HandleSlot(Tagged<HeapObject> host, THeapObjectSlot slot,
149  Tagged<HeapObject> target) {
150  static_assert(
151  std::is_same_v<THeapObjectSlot, FullHeapObjectSlot> ||
152  std::is_same_v<THeapObjectSlot, HeapObjectSlot>,
153  "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
154  scavenger_->SynchronizePageAccess(target);
155 
156  if (Heap::InFromPage(target)) {
157  SlotCallbackResult result = scavenger_->ScavengeObject(slot, target);
158  bool success = (*slot).GetHeapObject(&target);
159  USE(success);
160  DCHECK(success);
161 
162  if (result == KEEP_SLOT) {
163  SLOW_DCHECK(IsHeapObject(target));
164  MemoryChunk* chunk = MemoryChunk::FromHeapObject(host);
165  MutablePageMetadata* page =
166  MutablePageMetadata::cast(chunk->Metadata());
167 
168  // Sweeper is stopped during scavenge, so we can directly
169  // insert into its remembered set here.
170  RememberedSet<OLD_TO_NEW>::Insert<AccessMode::ATOMIC>(
171  page, chunk->Offset(slot.address()));
172  }
173  DCHECK(!MarkCompactCollector::IsOnEvacuationCandidate(target));
174  }
175 
176  if (HeapLayout::InWritableSharedSpace(target)) {
177  MemoryChunk* chunk = MemoryChunk::FromHeapObject(host);
178  MutablePageMetadata* page = MutablePageMetadata::cast(chunk->Metadata());
179  RememberedSet<OLD_TO_SHARED>::Insert<AccessMode::ATOMIC>(
180  page, chunk->Offset(slot.address()));
181  }
182  }
183 
184  Scavenger* const scavenger_;

228 void ScavengerCollector::JobTask::Run(JobDelegate* delegate) {
229  DCHECK_LT(delegate->GetTaskId(), scavengers_->size());
230  // Set the current isolate such that trusted pointer tables etc are
231  // available and the cage base is set correctly for multi-cage mode.
232  SetCurrentIsolateScope isolate_scope(collector_->heap_->isolate());
233 
234  collector_->estimate_concurrency_.fetch_add(1, std::memory_order_relaxed);
235 
236  Scavenger* scavenger = (*scavengers_)[delegate->GetTaskId()].get();
237  if (delegate->IsJoiningThread()) {
238  TRACE_GC_WITH_FLOW(collector_->heap_->tracer(),
239  GCTracer::Scope::SCAVENGER_SCAVENGE_PARALLEL, trace_id_,
240  TRACE_EVENT_FLAG_FLOW_IN);
241  ProcessItems(delegate, scavenger);
242  } else {
243  TRACE_GC_EPOCH_WITH_FLOW(
244  collector_->heap_->tracer(),
245  GCTracer::Scope::SCAVENGER_BACKGROUND_SCAVENGE_PARALLEL,
246  ThreadKind::kBackground, trace_id_, TRACE_EVENT_FLAG_FLOW_IN);
247  ProcessItems(delegate, scavenger);
248  }

251 size_t ScavengerCollector::JobTask::GetMaxConcurrency(
252  size_t worker_count) const {
253  // We need to account for local segments held by worker_count in addition to
254  // GlobalPoolSize() of copied_list_, pinned_list_ and promoted_list_.
255  size_t wanted_num_workers =
256  std::max<size_t>(remaining_memory_chunks_.load(std::memory_order_relaxed),
257  worker_count + copied_list_.Size() +
258  pinned_list_.Size() + promoted_list_.Size());
259  if (!collector_->heap_->ShouldUseBackgroundThreads() ||
260  collector_->heap_->ShouldOptimizeForBattery()) {
261  return std::min<size_t>(wanted_num_workers, 1);
262  }
263  return std::min<size_t>(scavengers_->size(), wanted_num_workers);
264 }

266 void ScavengerCollector::JobTask::ProcessItems(JobDelegate* delegate,
267  Scavenger* scavenger) {
268  double scavenging_time = 0.0;
269  {
270  TimedScope scope(&scavenging_time);
271 
272  scavenger->VisitPinnedObjects();
273  ConcurrentScavengePages(scavenger);
274  scavenger->Process(delegate);
275  }
276  if (V8_UNLIKELY(v8_flags.trace_parallel_scavenge)) {
277  PrintIsolate(collector_->heap_->isolate(),
278  "scavenge[%p]: time=%.2f copied=%zu promoted=%zu\n",
279  static_cast<void*>(this), scavenging_time,
280  scavenger->bytes_copied(), scavenger->bytes_promoted());
281  }

284 void ScavengerCollector::JobTask::ConcurrentScavengePages(
285  Scavenger* scavenger) {
286  while (remaining_memory_chunks_.load(std::memory_order_relaxed) > 0) {
287  std::optional<size_t> index = generator_.GetNext();
288  if (!index) {
289  return;
290  }
291  for (size_t i = *index; i < old_to_new_chunks_.size(); ++i) {
292  auto& work_item = old_to_new_chunks_[i];
293  if (!work_item.first.TryAcquire()) {
294  break;
295  }
296  scavenger->ScavengePage(work_item.second);
297  if (remaining_memory_chunks_.fetch_sub(1, std::memory_order_relaxed) <=
298  1) {
299  return;
300  }
301  }
302  }
303 }

312 class GlobalHandlesWeakRootsUpdatingVisitor final : public RootVisitor {
313  public:

325  private:
326  void UpdatePointer(FullObjectSlot p) {
327  Tagged<Object> object = *p;
328  DCHECK(!HasWeakHeapObjectTag(object));
329  // The object may be in the old generation as global handles over
330  // approximates the list of young nodes. This checks also bails out for
331  // Smis.
332  if (!HeapLayout::InYoungGeneration(object)) {
333  return;
334  }
335 
336  Tagged<HeapObject> heap_object = Cast<HeapObject>(object);
337  // TODO(chromium:1336158): Turn the following CHECKs into DCHECKs after
338  // flushing out potential issues.
339  CHECK(Heap::InFromPage(heap_object));
340  MapWord first_word = heap_object->map_word(kRelaxedLoad);
341  CHECK(first_word.IsForwardingAddress());
342  Tagged<HeapObject> dest = first_word.ToForwardingAddress(heap_object);
343  if (heap_object == dest) {
344  DCHECK(Heap::IsLargeObject(heap_object) ||
345  MemoryChunk::FromHeapObject(heap_object)->IsQuarantined());
346  return;
347  }
348  UpdateHeapObjectReferenceSlot(FullHeapObjectSlot(p), dest);
349  // The destination object should be in the "to" space. However, it could
350  // also be a large string if the original object was a shortcut candidate.
351  DCHECK_IMPLIES(HeapLayout::InYoungGeneration(dest),
352  Heap::InToPage(dest) ||
353  (Heap::IsLargeObject(dest) && Heap::InFromPage(dest) &&
354  dest->map_word(kRelaxedLoad).IsForwardingAddress()));
355  }

376 // A conservative stack scanning visitor implementation that:
377 // 1) Filters out non-young objects, and
378 // 2) Use the marking bitmap as a temporary object start bitmap.
379 class YoungGenerationConservativeStackVisitor
380  : public ConservativeStackVisitorBase<

386  DCHECK_NE(isolate->heap()->ConservativeStackScanningModeForMinorGC(),
387  Heap::StackScanMode::kNone);
388  DCHECK(!v8_flags.minor_ms);
389  DCHECK(!v8_flags.sticky_mark_bits);
390  DCHECK(std::all_of(
391  isolate_->heap()->semi_space_new_space()->to_space().begin(),
392  isolate_->heap()->semi_space_new_space()->to_space().end(),
393  [](const PageMetadata* page) {
394  return page->marking_bitmap()->IsClean();
395  }));
396  DCHECK(std::all_of(
397  isolate_->heap()->semi_space_new_space()->from_space().begin(),
398  isolate_->heap()->semi_space_new_space()->from_space().end(),
399  [](const PageMetadata* page) {
400  return page->marking_bitmap()->IsClean();
401  }));
402  }
403 
404  ~YoungGenerationConservativeStackVisitor() {
405  DCHECK(std::all_of(
406  isolate_->heap()->semi_space_new_space()->to_space().begin(),
407  isolate_->heap()->semi_space_new_space()->to_space().end(),
408  [](const PageMetadata* page) {
409  return page->marking_bitmap()->IsClean();
410  }));
411  for (PageMetadata* page :
412  isolate_->heap()->semi_space_new_space()->from_space()) {
413  page->marking_bitmap()->Clear<AccessMode::NON_ATOMIC>();
414  }
415  }
416 
417  private:
418  static constexpr bool kOnlyVisitMainV8Cage [[maybe_unused]] = true;

460 template <typename ConcreteVisitor>
461 class ObjectPinningVisitorBase : public RootVisitor {
462  public:
463  ObjectPinningVisitorBase(const Heap* heap, Scavenger& scavenger,
464  ScavengerCollector::PinnedObjects& pinned_objects)
465  : RootVisitor(),
466  heap_(heap),
467  scavenger_(scavenger),
468  pinned_objects_(pinned_objects) {}
469 
470  void VisitRootPointer(Root root, const char* description,
471  FullObjectSlot p) final {
472  DCHECK(root == Root::kStackRoots || root == Root::kHandleScope);
473  static_cast<ConcreteVisitor*>(this)->HandlePointer(p);
474  }
475 
476  void VisitRootPointers(Root root, const char* description,
477  FullObjectSlot start, FullObjectSlot end) final {
478  DCHECK(root == Root::kStackRoots || root == Root::kHandleScope);
479  for (FullObjectSlot p = start; p < end; ++p) {
480  static_cast<ConcreteVisitor*>(this)->HandlePointer(p);
481  }
482  }
483 
484  protected:
485  void HandleHeapObject(Tagged<HeapObject> object) {
486  DCHECK(!HasWeakHeapObjectTag(object));
487  DCHECK(!MapWord::IsPacked(object.ptr()));
488  DCHECK(!HeapLayout::IsSelfForwarded(object));
489  if (IsAllocationMemento(object)) {
490  // Don't pin allocation mementos since they should not survive a GC.
491  return;
492  }
493  if (scavenger_.PromoteIfLargeObject(object)) {
494  // Large objects are not moved and thus don't require pinning. Instead,
495  // we scavenge large pages eagerly to keep them from being reclaimed (if
496  // the page is only reachable from stack).
497  return;
498  }
499  DCHECK(!MemoryChunk::FromHeapObject(object)->IsLargePage());
500  DCHECK(HeapLayout::InYoungGeneration(object));
501  DCHECK(Heap::InFromPage(object));
502  Address object_address = object.address();
503  MapWord map_word = object->map_word(kRelaxedLoad);
504  DCHECK(!map_word.IsForwardingAddress());
505  DCHECK(std::all_of(
506  pinned_objects_.begin(), pinned_objects_.end(),
507  [object_address](ScavengerCollector::PinnedObjectEntry& entry) {
508  return entry.address != object_address;
509  }));
510  int object_size = object->SizeFromMap(map_word.ToMap());
511  DCHECK_LT(0, object_size);
512  pinned_objects_.push_back(
513  {object_address, map_word, static_cast<size_t>(object_size)});
514  MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
515  if (!chunk->IsQuarantined()) {
516  chunk->SetFlagNonExecutable(MemoryChunk::IS_QUARANTINED);
517  if (v8_flags.scavenger_promote_quarantined_pages &&
518  heap_->semi_space_new_space()->ShouldPageBePromoted(chunk)) {
519  chunk->SetFlagNonExecutable(MemoryChunk::WILL_BE_PROMOTED);
520  }
521  }
522  scavenger_.PinAndPushObject(chunk, object, map_word);
523  }
524 
525  private:
526  const Heap* const heap_;
527  Scavenger& scavenger_;
528  ScavengerCollector::PinnedObjects& pinned_objects_;
529 };

616 void RestorePinnedObjects(
617  SemiSpaceNewSpace& new_space,
618  const ScavengerCollector::PinnedObjects& pinned_objects) {
619  // Restore the maps of quarantined objects. We use the iteration over
620  // quarantined objects to split them based on pages. This will be used below
621  // for sweeping the quarantined pages (since there are no markbits).
622  DCHECK_EQ(0, new_space.QuarantinedPageCount());
623  size_t quarantined_objects_size = 0;
624  for (const auto& [object_address, map_word, object_size] : pinned_objects) {
625  DCHECK(!map_word.IsForwardingAddress());
626  Tagged<HeapObject> object = HeapObject::FromAddress(object_address);
627  DCHECK(HeapLayout::IsSelfForwarded(object));
628  object->set_map_word(map_word.ToMap(), kRelaxedStore);
629  DCHECK(!HeapLayout::IsSelfForwarded(object));
630  MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
631  DCHECK(chunk->IsQuarantined());
632  if (!chunk->IsFlagSet(MemoryChunk::WILL_BE_PROMOTED)) {
633  quarantined_objects_size += object_size;
634  }
635  }
636  new_space.SetQuarantinedSize(quarantined_objects_size);
637 }

639 void QuarantinePinnedPages(SemiSpaceNewSpace& new_space) {
640  PageMetadata* next_page = new_space.from_space().first_page();
641  while (next_page) {
642  PageMetadata* current_page = next_page;
643  next_page = current_page->next_page();
644  MemoryChunk* chunk = current_page->Chunk();
645  DCHECK(chunk->IsFromPage());
646  if (!chunk->IsQuarantined()) {
647  continue;
648  }
649  if (chunk->IsFlagSet(MemoryChunk::WILL_BE_PROMOTED)) {
650  // free list categories will be relinked by the quarantined page sweeper
651  // after sweeping is done.
652  new_space.PromotePageToOldSpace(current_page,
653  FreeMode::kDoNotLinkCategory);
654  DCHECK(!chunk->InYoungGeneration());
655  } else {
656  new_space.MoveQuarantinedPage(chunk);
657  DCHECK(!chunk->IsFromPage());
658  DCHECK(chunk->IsToPage());
659  }
660  DCHECK(current_page->marking_bitmap()->IsClean());
661  DCHECK(!chunk->IsFromPage());
662  DCHECK(!chunk->IsQuarantined());
663  DCHECK(!chunk->IsFlagSet(MemoryChunk::WILL_BE_PROMOTED));
664  }
665 }

679 void ScavengerCollector::QuarantinedPageSweeper::JobTask::Run(
680  JobDelegate* delegate) {
681 #ifdef V8_COMPRESS_POINTERS_IN_MULTIPLE_CAGES
682  SetCurrentIsolateScope current_isolate_scope(heap_->isolate());
683 #endif // V8_COMPRESS_POINTERS_IN_MULTIPLE_CAGES
684  DCHECK_IMPLIES(
685  delegate->IsJoiningThread(),
686  heap_->IsMainThread() ||
687  (!heap_->isolate()->is_shared_space_isolate() &&
688  heap_->isolate()->shared_space_isolate()->heap()->IsMainThread()));
689  const bool is_main_thread =
690  delegate->IsJoiningThread() && heap_->IsMainThread();
691  TRACE_GC_EPOCH_WITH_FLOW(
692  heap_->tracer(),
693  GCTracer::Scope::SCAVENGER_BACKGROUND_QUARANTINED_PAGE_SWEEPING,
694  is_main_thread ? ThreadKind::kMain : ThreadKind::kBackground, trace_id(),
695  TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
696  DCHECK(!is_done_.load(std::memory_order_relaxed));
697  DCHECK(!pinned_objects_.empty());
698  if (pinned_object_per_page_.empty()) {
699  // Populate the per page map.
700  for (const PinnedObjectEntry& entry : pinned_objects_) {
701  DCHECK(!HeapLayout::IsSelfForwarded(
702  HeapObject::FromAddress(entry.address), heap_->isolate()));
703  MemoryChunk* chunk = MemoryChunk::FromAddress(entry.address);
704  DCHECK(!chunk->IsQuarantined());
705  ObjectsAndSizes& objects_for_page = pinned_object_per_page_[chunk];
706  DCHECK(!std::any_of(objects_for_page.begin(), objects_for_page.end(),
707  [entry](auto& object_and_size) {
708  return object_and_size.first == entry.address;
709  }));
710  objects_for_page.emplace_back(entry.address, entry.size);
711  }
712  // Initialize the iterator.
713  next_page_iterator_ = pinned_object_per_page_.begin();
714  DCHECK_NE(next_page_iterator_, pinned_object_per_page_.end());
715  }
716  // Sweep all quarantined pages.
717  while (next_page_iterator_ != pinned_object_per_page_.end()) {
718  if (delegate->ShouldYield()) {
719  TRACE_GC_NOTE("Quarantined page sweeping preempted");
720  return;
721  }
722  MemoryChunk* chunk = next_page_iterator_->first;
723  PageMetadata* page = static_cast<PageMetadata*>(chunk->Metadata());
724  DCHECK(!chunk->IsFromPage());
725  if (chunk->IsToPage()) {
726  SweepPage(CreateFillerFreeSpaceHandler, chunk, page,
727  next_page_iterator_->second);
728  } else {
729  DCHECK_EQ(chunk->Metadata()->owner()->identity(), OLD_SPACE);
730  base::MutexGuard guard(page->mutex());
731  // If for some reason the page is swept twice, this DCHECK will fail.
732  DCHECK_EQ(page->area_size(), page->allocated_bytes());
733  size_t filler_size_on_page =
734  SweepPage(AddToFreeListFreeSpaceHandler, chunk, page,
735  next_page_iterator_->second);
736  DCHECK_EQ(page->owner()->identity(), OLD_SPACE);
737  OldSpace* old_space = static_cast<OldSpace*>(page->owner());
738  old_space->RelinkQuarantinedPageFreeList(page, filler_size_on_page);
739  }
740  next_page_iterator_++;
741  }
742  is_done_.store(true, std::memory_order_relaxed);
743  pinned_object_per_page_.clear();
744  pinned_objects_.clear();
745 }

770 size_t ScavengerCollector::QuarantinedPageSweeper::JobTask::SweepPage(
771  FreeSpaceHandler free_space_handler, MemoryChunk* chunk, PageMetadata* page,
772  ObjectsAndSizes& pinned_objects_on_page) {
773  DCHECK_EQ(page, chunk->Metadata());
774  DCHECK(!pinned_objects_on_page.empty());
775  Address start = page->area_start();
776  std::sort(pinned_objects_on_page.begin(), pinned_objects_on_page.end());
777  size_t filler_size_on_page = 0;
778  for (const auto& [object_adress, object_size] : pinned_objects_on_page) {
779  DCHECK_LE(start, object_adress);
780  if (start != object_adress) {
781  size_t filler_size = object_adress - start;
782  free_space_handler(heap_, start, filler_size, should_zap_);
783  filler_size_on_page += filler_size;
784  }
785  start = object_adress + object_size;
786  }
787  Address end = page->area_end();
788  if (start != end) {
789  size_t filler_size = end - start;
790  free_space_handler(heap_, start, filler_size, should_zap_);
791  filler_size_on_page += filler_size;
792  }
793  return filler_size_on_page;
794 }

812 void ScavengerCollector::CollectGarbage() {
813  ScopedFullHeapCrashKey collect_full_heap_dump_if_crash(isolate_);
814 
815  DCHECK(surviving_new_large_objects_.empty());
816  DCHECK(!quarantined_page_sweeper_.IsSweeping());
817 
818  SemiSpaceNewSpace* new_space = SemiSpaceNewSpace::From(heap_->new_space());
819  new_space->GarbageCollectionPrologue();
820  new_space->SwapSemiSpaces();
821 
822  // We also flip the young generation large object space. All large objects
823  // will be in the from space.
824  heap_->new_lo_space()->Flip();
825  heap_->new_lo_space()->ResetPendingObject();
826 
827  DCHECK(!heap_->allocator()->new_space_allocator()->IsLabValid());
828 
829  Scavenger::EmptyChunksList empty_chunks;
830  Scavenger::CopiedList copied_list;
831  Scavenger::PinnedList pinned_list;
832  Scavenger::PromotedList promoted_list;
833  EphemeronRememberedSet::TableList ephemeron_table_list;
834 
835  PinnedObjects pinned_objects;
836 
837  const int num_scavenge_tasks = NumberOfScavengeTasks();
838  std::vector<std::unique_ptr<Scavenger>> scavengers;
839 
840  const bool is_logging = isolate_->log_object_relocation();
841  for (int i = 0; i < num_scavenge_tasks; ++i) {
842  scavengers.emplace_back(
843  new Scavenger(this, heap_, is_logging, &empty_chunks, &copied_list,
844  &pinned_list, &promoted_list, &ephemeron_table_list));
845  }
846  Scavenger& main_thread_scavenger = *scavengers[kMainThreadId].get();
847 
848  {
849  // Identify weak unmodified handles. Requires an unmodified graph.
850  TRACE_GC(heap_->tracer(),
851  GCTracer::Scope::SCAVENGER_SCAVENGE_WEAK_GLOBAL_HANDLES_IDENTIFY);
852  isolate_->traced_handles()->ComputeWeaknessForYoungObjects();
853  }
854 
855  std::vector<std::pair<ParallelWorkItem, MutablePageMetadata*>>
856  old_to_new_chunks;
857  {
858  // Copy roots.
859  TRACE_GC(heap_->tracer(), GCTracer::Scope::SCAVENGER_SCAVENGE_ROOTS);
860 
861  // We must collect old-to-new pages before starting Scavenge because
862  // pages could be removed from the old generation for allocation which
863  // hides them from the iteration.
864  OldGenerationMemoryChunkIterator::ForAll(
865  heap_, [&old_to_new_chunks](MutablePageMetadata* chunk) {
866  if (chunk->slot_set<OLD_TO_NEW>() ||
867  chunk->typed_slot_set<OLD_TO_NEW>() ||
868  chunk->slot_set<OLD_TO_NEW_BACKGROUND>()) {
869  old_to_new_chunks.emplace_back(ParallelWorkItem{}, chunk);
870  }
871  });
872 
873  const Heap::StackScanMode stack_scan_mode =
874  heap_->ConservativeStackScanningModeForMinorGC();
875  DCHECK_IMPLIES(stack_scan_mode == Heap::StackScanMode::kSelective,
876  heap_->IsGCWithStack());
877  if ((stack_scan_mode != Heap::StackScanMode::kNone) &&
878  heap_->IsGCWithStack()) {
879  // Pinning objects must be the first step and must happen before
880  // scavenging any objects. Specifically we must all pin all objects
881  // before visiting other pinned objects. If we scavenge some object X
882  // and move it before all stack-reachable objects are pinned, and we
883  // later find that we need to pin X, it will be too late to undo the
884  // moving.
885  TRACE_GC(heap_->tracer(),
886  GCTracer::Scope::SCAVENGER_SCAVENGE_PIN_OBJECTS);
887  ConservativeObjectPinningVisitor conservative_pinning_visitor(
888  heap_, main_thread_scavenger, pinned_objects);
889  // Scavenger reuses the page's marking bitmap as a temporary object
890  // start bitmap. Stack scanning will incrementally build the map as it
891  // searches through pages.
892  YoungGenerationConservativeStackVisitor stack_visitor(
893  isolate_, &conservative_pinning_visitor);
894  // Marker was already set by Heap::CollectGarbage.
895  heap_->IterateConservativeStackRoots(&stack_visitor, stack_scan_mode);
896  if (V8_UNLIKELY(v8_flags.stress_scavenger_conservative_object_pinning)) {
897  TreatConservativelyVisitor handles_visitor(&stack_visitor, heap_);
898  heap_->IterateRootsForPrecisePinning(&handles_visitor);
899  }
900  }
901  const bool is_using_precise_pinning =
902  heap_->ShouldUsePrecisePinningForMinorGC();
903  if (is_using_precise_pinning) {
904  PreciseObjectPinningVisitor precise_pinning_visitor(
905  heap_, main_thread_scavenger, pinned_objects);
906  ClearStaleLeftTrimmedPointerVisitor left_trim_visitor(
907  heap_, &precise_pinning_visitor);
908  heap_->IterateRootsForPrecisePinning(&left_trim_visitor);
909  }
910 
911  // Scavenger treats all weak roots except for global handles as strong.
912  // That is why we don't set skip_weak = true here and instead visit
913  // global handles separately.
914  base::EnumSet<SkipRoot> options(
915  {SkipRoot::kExternalStringTable, SkipRoot::kGlobalHandles,
916  SkipRoot::kTracedHandles, SkipRoot::kOldGeneration,
917  SkipRoot::kConservativeStack, SkipRoot::kReadOnlyBuiltins});
918  if (is_using_precise_pinning) {
919  options.Add({SkipRoot::kMainThreadHandles, SkipRoot::kStack});
920  }
921  RootScavengeVisitor root_scavenge_visitor(main_thread_scavenger);
922 
923  heap_->IterateRoots(&root_scavenge_visitor, options);
924  isolate_->global_handles()->IterateYoungStrongAndDependentRoots(
925  &root_scavenge_visitor);
926  isolate_->traced_handles()->IterateYoungRoots(&root_scavenge_visitor);
927  }
928  {
929  // Parallel phase scavenging all copied and promoted objects.
930  TRACE_GC_ARG1(heap_->tracer(),
931  GCTracer::Scope::SCAVENGER_SCAVENGE_PARALLEL_PHASE,
932  "UseBackgroundThreads", heap_->ShouldUseBackgroundThreads());
933 
934  auto job = std::make_unique<JobTask>(
935  this, &scavengers, std::move(old_to_new_chunks), copied_list,
936  pinned_list, promoted_list);
937  TRACE_GC_NOTE_WITH_FLOW("Parallel scavenge started", job->trace_id(),
938  TRACE_EVENT_FLAG_FLOW_OUT);
939  V8::GetCurrentPlatform()
940  ->CreateJob(v8::TaskPriority::kUserBlocking, std::move(job))
941  ->Join();
942  DCHECK(copied_list.IsEmpty());
943  DCHECK(pinned_list.IsEmpty());
944  DCHECK(promoted_list.IsEmpty());
945  }
946 
947  {
948  // Scavenge weak global handles.
949  TRACE_GC(heap_->tracer(),
950  GCTracer::Scope::SCAVENGER_SCAVENGE_WEAK_GLOBAL_HANDLES_PROCESS);
951  GlobalHandlesWeakRootsUpdatingVisitor visitor;
952  isolate_->global_handles()->ProcessWeakYoungObjects(
953  &visitor, &IsUnscavengedHeapObjectSlot);
954  isolate_->traced_handles()->ProcessWeakYoungObjects(
955  &visitor, &IsUnscavengedHeapObjectSlot);
956  }
957 
958  {
959  // Finalize parallel scavenging.
960  TRACE_GC(heap_->tracer(), GCTracer::Scope::SCAVENGER_SCAVENGE_FINALIZE);
961 
962  DCHECK(surviving_new_large_objects_.empty());
963 
964  for (auto& scavenger : scavengers) {
965  scavenger->Finalize();
966  }
967  scavengers.clear();
968 
969 #ifdef V8_COMPRESS_POINTERS
970  // Sweep the external pointer table.
971  DCHECK(heap_->concurrent_marking()->IsStopped());
972  DCHECK(!heap_->incremental_marking()->IsMajorMarking());
973  heap_->isolate()->external_pointer_table().Sweep(
974  heap_->young_external_pointer_space(), heap_->isolate()->counters());
975 #endif // V8_COMPRESS_POINTERS
976 
977  HandleSurvivingNewLargeObjects();
978 
979  heap_->tracer()->SampleConcurrencyEsimate(
980  FetchAndResetConcurrencyEstimate());
981  }
982 
983  {
984  // Update references into new space
985  TRACE_GC(heap_->tracer(), GCTracer::Scope::SCAVENGER_SCAVENGE_UPDATE_REFS);
986  heap_->UpdateYoungReferencesInExternalStringTable(
987  &Heap::UpdateYoungReferenceInExternalStringTableEntry);
988 
989  if (V8_UNLIKELY(v8_flags.always_use_string_forwarding_table)) {
990  isolate_->string_forwarding_table()->UpdateAfterYoungEvacuation();
991  }
992  }
993 
994  ProcessWeakReferences(&ephemeron_table_list);
995 
996  {
997  TRACE_GC(heap_->tracer(),
998  GCTracer::Scope::SCAVENGER_SCAVENGE_RESTORE_AND_QUARANTINE_PINNED);
999  RestorePinnedObjects(*new_space, pinned_objects);
1000  QuarantinePinnedPages(*new_space);
1001  }
1002 
1003  // Need to free new space LAB that was allocated during scavenge.
1004  heap_->allocator()->new_space_allocator()->FreeLinearAllocationArea();
1005 
1006  // Since we promote all surviving large objects immediately, all remaining
1007  // large objects must be dead.
1008  // TODO(hpayer): Don't free all as soon as we have an intermediate generation.
1009  heap_->new_lo_space()->FreeDeadObjects(
1010  [](Tagged<HeapObject>) { return true; });
1011 
1012  new_space->GarbageCollectionEpilogue();
1013 
1014  // Start sweeping quarantined pages.
1015  if (!pinned_objects.empty()) {
1016  quarantined_page_sweeper_.StartSweeping(std::move(pinned_objects));
1017  } else {
1018  // Sweeping is not started since there are no pages to sweep. Mark sweeping
1019  // as completed so that the current GC cycle can be stopped since there is
1020  // no sweeper to mark it has completed later.
1021  heap_->tracer()->NotifyYoungSweepingCompleted();
1022  }
1023 
1024  {
1025  TRACE_GC(heap_->tracer(), GCTracer::Scope::SCAVENGER_FREE_REMEMBERED_SET);
1026  Scavenger::EmptyChunksList::Local empty_chunks_local(empty_chunks);
1027  MutablePageMetadata* chunk;
1028  while (empty_chunks_local.Pop(&chunk)) {
1029  RememberedSet<OLD_TO_NEW>::CheckPossiblyEmptyBuckets(chunk);
1030  RememberedSet<OLD_TO_NEW_BACKGROUND>::CheckPossiblyEmptyBuckets(chunk);
1031  }
1032 
1033 #ifdef DEBUG
1034  OldGenerationMemoryChunkIterator::ForAll(
1035  heap_, [](MutablePageMetadata* chunk) {
1036  if (chunk->slot_set<OLD_TO_NEW>() ||
1037  chunk->typed_slot_set<OLD_TO_NEW>() ||
1038  chunk->slot_set<OLD_TO_NEW_BACKGROUND>()) {
1039  DCHECK(chunk->possibly_empty_buckets()->IsEmpty());
1040  }
1041  });
1042 #endif
1043  }
1044 
1045  SweepArrayBufferExtensions();
1046 
1047  isolate_->global_handles()->UpdateListOfYoungNodes();
1048  isolate_->traced_handles()->UpdateListOfYoungNodes();
1049 
1050  // Update how much has survived scavenge.
1051  heap_->IncrementYoungSurvivorsCounter(heap_->SurvivedYoungObjectSize());
1052 
1053  {
1054  TRACE_GC(heap_->tracer(), GCTracer::Scope::SCAVENGER_RESIZE_NEW_SPACE);
1055  heap_->ResizeNewSpace();
1056  }
1057 }
