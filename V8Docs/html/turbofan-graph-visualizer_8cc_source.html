248 void JsonPrintAllSourceWithPositionsWasm(
249  std::ostream& os, const wasm::WasmModule* module,
250  const wasm::WireBytesStorage* wire_bytes,
251  base::Vector<WasmInliningPosition> positions) {
252  // Filter out duplicate sources. (A single wasm function might be inlined more
253  // than once.)
254  std::vector<int /*function id*/> sources;
255  std::unordered_map<int /*function id*/, size_t /*source index*/> source_map;
256  for (WasmInliningPosition pos : positions) {
257  auto [_, inserted] =
258  source_map.emplace(pos.inlinee_func_index, sources.size());
259  if (inserted) {
260  // The function wasn't inlined yet. Add a new entry to the sources.
261  // The hashmap stores the index to the entry in the source map.
262  sources.push_back(pos.inlinee_func_index);
263  }
264  // Don't do anything if it was already inserted.
265  }
266 
267  // Print inlining sources.
268  os << "\"sources\": {";
269  for (size_t i = 0; i < sources.size(); ++i) {
270  if (i != 0) os << ", ";
271  int function_id = sources[i];
272  const wasm::WasmFunction& fct = module->functions[function_id];
273  os << '"' << i << "\": {\"sourceId\": " << i << ", \"functionName\": \""
274  << fct.func_index << "\", \"sourceName\": \"\", \"sourceText\": \"";
275  base::Vector<const uint8_t> module_bytes{nullptr, 0};
276  std::optional<wasm::ModuleWireBytes> maybe_wire_bytes =
277  wire_bytes->GetModuleBytes();
278  if (maybe_wire_bytes) module_bytes = maybe_wire_bytes->module_bytes();
279  std::ostringstream wasm_str;
280  wasm::DisassembleFunction(module, function_id,
281  wire_bytes->GetCode(fct.code), module_bytes,
282  fct.code.offset(), wasm_str);
283  os << JSONEscaped(wasm_str) << "\"}";
284  }
285  os << "},\n";
286  // Print inlining mappings.
287  // This maps the inlining position to the deduplicated source in the sources
288  // object generated above.
289  os << "\"inlinings\": {";
290  for (size_t i = 0; i < positions.size(); ++i) {
291  if (i != 0) os << ", ";
292  DCHECK(source_map.contains(positions[i].inlinee_func_index));
293  size_t source_id = source_map.find(positions[i].inlinee_func_index)->second;
294  SourcePosition inlining_pos = positions[i].caller_pos;
295  os << '"' << i << "\": {\"inliningId\": " << i
296  << ", \"sourceId\": " << source_id
297  << ", \"inliningPosition\": " << AsJSON(inlining_pos) << "}";
298  }

302 std::unique_ptr<char[]> GetVisualizerLogFileName(OptimizedCompilationInfo* info,
303  const char* optional_base_dir,
304  const char* phase,
305  const char* suffix) {
306  base::EmbeddedVector<char, 256> filename(0);
307  std::unique_ptr<char[]> debug_name = info->GetDebugName();
308  const char* file_prefix = v8_flags.trace_turbo_file_prefix.value();
309  int optimization_id = info->IsOptimizing() ? info->optimization_id() : 0;
310  if (strlen(debug_name.get()) > 0) {
311  if (strcmp(debug_name.get(), "WasmJSFastApiCall") == 0) {
312  // Don't clobber one wrapper's output with another's.
313  static int fast_call_wrappers_count = 0;
314  optimization_id = ++fast_call_wrappers_count;
315  }
316  SNPrintF(filename, "%s-%s-%i", file_prefix, debug_name.get(),
317  optimization_id);
318  } else if (info->has_shared_info()) {
319  SNPrintF(filename, "%s-%p-%i", file_prefix,
320  reinterpret_cast<void*>(info->shared_info()->address()),
321  optimization_id);
322  } else {
323  SNPrintF(filename, "%s-none-%i", file_prefix, optimization_id);
324  }
325  base::EmbeddedVector<char, 256> source_file(0);
326  bool source_available = false;
327  if (v8_flags.trace_file_names && info->has_shared_info() &&
328  IsScript(info->shared_info()->script())) {
329  Tagged<Object> source_name =
330  Cast<Script>(info->shared_info()->script())->name();
331  if (IsString(source_name)) {
332  Tagged<String> str = Cast<String>(source_name);
333  if (str->length() > 0) {
334  SNPrintF(source_file, "%s", str->ToCString().get());
335  std::replace(source_file.begin(),
336  source_file.begin() + source_file.length(), '/', '_');
337  source_available = true;
338  }
339  }
340  }
341  std::replace(filename.begin(), filename.begin() + filename.length(), ' ',
342  '_');
343  std::replace(filename.begin(), filename.begin() + filename.length(), ':',
344  '-');
345 #if V8_OS_WIN
346  std::replace(filename.begin(), filename.begin() + filename.length(), '<',
347  '{');
348  std::replace(filename.begin(), filename.begin() + filename.length(), '>',
349  '}');
350 #endif // V8_OS_WIN
351 
352  base::EmbeddedVector<char, 256> base_dir;
353  if (optional_base_dir != nullptr) {
354  SNPrintF(base_dir, "%s%c", optional_base_dir,
355  base::OS::DirectorySeparator());
356  } else {
357  base_dir[0] = '\0';
358  }
359 
360  base::EmbeddedVector<char, 256> full_filename;
361  if (phase == nullptr && !source_available) {
362  SNPrintF(full_filename, "%s%s.%s", base_dir.begin(), filename.begin(),
363  suffix);
364  } else if (phase != nullptr && !source_available) {
365  SNPrintF(full_filename, "%s%s-%s.%s", base_dir.begin(), filename.begin(),
366  phase, suffix);
367  } else if (phase == nullptr && source_available) {
368  SNPrintF(full_filename, "%s%s_%s.%s", base_dir.begin(), filename.begin(),
369  source_file.begin(), suffix);
370  } else {
371  SNPrintF(full_filename, "%s%s_%s-%s.%s", base_dir.begin(), filename.begin(),
372  source_file.begin(), phase, suffix);
373  }
374 
375  char* buffer = new char[full_filename.length() + 1];
376  memcpy(buffer, full_filename.begin(), full_filename.length());
377  buffer[full_filename.length()] = '\0';
378  return std::unique_ptr<char[]>(buffer);
379 }

421 void JSONGraphWriter::PrintNode(Node* node, bool is_live) {
422  if (first_node_) {
423  first_node_ = false;
424  } else {
425  os_ << ",\n";
426  }
427  std::ostringstream label, title, properties;
428  node->op()->PrintTo(label, Operator::PrintVerbosity::kSilent);
429  node->op()->PrintTo(title, Operator::PrintVerbosity::kVerbose);
430  node->op()->PrintPropsTo(properties);
431  os_ << "{\"id\":" << SafeId(node) << ",\"label\":\"" << JSONEscaped(label)
432  << "\"" << ",\"title\":\"" << JSONEscaped(title) << "\""
433  << ",\"live\": " << (is_live ? "true" : "false") << ",\"properties\":\""
434  << JSONEscaped(properties) << "\"";
435  IrOpcode::Value opcode = node->opcode();
436  if (IrOpcode::IsPhiOpcode(opcode)) {
437  os_ << ",\"rankInputs\":[0," << NodeProperties::FirstControlIndex(node)
438  << "]";
439  os_ << ",\"rankWithInput\":[" << NodeProperties::FirstControlIndex(node)
440  << "]";
441  } else if (opcode == IrOpcode::kIfTrue || opcode == IrOpcode::kIfFalse ||
442  opcode == IrOpcode::kLoop) {
443  os_ << ",\"rankInputs\":[" << NodeProperties::FirstControlIndex(node)
444  << "]";
445  }
446  if (opcode == IrOpcode::kBranch) {
447  os_ << ",\"rankInputs\":[0]";
448  }
449  if (positions_ != nullptr) {
450  SourcePosition position = positions_->GetSourcePosition(node);
451  if (position.IsKnown()) {
452  os_ << ", \"sourcePosition\" : " << AsJSON(position);
453  }
454  }
455  if (origins_) {
456  NodeOrigin origin = origins_->GetNodeOrigin(node);
457  if (origin.IsKnown()) {
458  os_ << ", \"origin\" : " << AsJSON(origin);
459  }
460  }
461  os_ << ",\"opcode\":\"" << IrOpcode::Mnemonic(node->opcode()) << "\"";
462  os_ << ",\"control\":"
463  << (NodeProperties::IsControl(node) ? "true" : "false");
464  os_ << ",\"opinfo\":\"" << node->op()->ValueInputCount() << " v "
465  << node->op()->EffectInputCount() << " eff "
466  << node->op()->ControlInputCount() << " ctrl in, "
467  << node->op()->ValueOutputCount() << " v "
468  << node->op()->EffectOutputCount() << " eff "
469  << node->op()->ControlOutputCount() << " ctrl out\"";
470  if (auto type_opt = GetType(node)) {
471  std::ostringstream type_out;
472  type_opt->PrintTo(type_out);
473  os_ << ",\"type\":\"" << JSONEscaped(type_out) << "\"";
474  }
475  os_ << "}";
476 }

917 const int kUnvisited = 0;
918 const int kOnStack = 1;
919 const int kVisited = 2;
920 
921 std::ostream& operator<<(std::ostream& os, const AsRPO& ar) {
922  AccountingAllocator allocator;
923  Zone local_zone(&allocator, ZONE_NAME);
924 
925  // Do a post-order depth-first search on the RPO graph. For every node,
926  // print:
927  //
928  // - the node id
929  // - the operator mnemonic
930  // - in square brackets its parameter (if present)
931  // - in parentheses the list of argument ids and their mnemonics
932  // - the node type (if it is typed)
933 
934  // Post-order guarantees that all inputs of a node will be printed before
935  // the node itself, if there are no cycles. Any cycles are broken
936  // arbitrarily.
937 
938  ZoneVector<uint8_t> state(ar.graph.NodeCount(), kUnvisited, &local_zone);
939  ZoneStack<Node*> stack(&local_zone);
940 
941  stack.push(ar.graph.end());
942  state[ar.graph.end()->id()] = kOnStack;
943  while (!stack.empty()) {
944  Node* n = stack.top();
945  bool pop = true;
946  for (Node* const i : n->inputs()) {
947  if (state[i->id()] == kUnvisited) {
948  state[i->id()] = kOnStack;
949  stack.push(i);
950  pop = false;
951  break;
952  }
953  }
954  if (pop) {
955  state[n->id()] = kVisited;
956  stack.pop();
957  os << "#" << n->id() << ":" << *n->op() << "(";
958  // Print the inputs.
959  int j = 0;
960  for (Node* const i : n->inputs()) {
961  if (j++ > 0) os << ", ";
962  os << "#" << SafeId(i) << ":" << SafeMnemonic(i);
963  }
964  os << ")";
965  // Print the node type, if any.
966  if (NodeProperties::IsTyped(n)) {
967  os << " [Type: " << NodeProperties::GetType(n) << "]";
968  }
969  os << std::endl;
970  }
971  }
972  return os;
973 }
