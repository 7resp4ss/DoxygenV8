21 bool BinaryOperationHintToNumberOperationHint(
22  BinaryOperationHint binop_hint, NumberOperationHint* number_hint) {
23  switch (binop_hint) {
24  case BinaryOperationHint::kSignedSmall:
25  *number_hint = NumberOperationHint::kSignedSmall;
26  return true;
27  case BinaryOperationHint::kSignedSmallInputs:
28  *number_hint = NumberOperationHint::kSignedSmallInputs;
29  return true;
30  case BinaryOperationHint::kAdditiveSafeInteger:
31  *number_hint = NumberOperationHint::kAdditiveSafeInteger;
32  return true;
33  case BinaryOperationHint::kNumber:
34  *number_hint = NumberOperationHint::kNumber;
35  return true;
36  case BinaryOperationHint::kNumberOrOddball:
37  *number_hint = NumberOperationHint::kNumberOrOddball;
38  return true;
39  case BinaryOperationHint::kAny:
40  case BinaryOperationHint::kNone:
41  case BinaryOperationHint::kString:
42  case BinaryOperationHint::kStringOrStringWrapper:
43  case BinaryOperationHint::kBigInt:
44  case BinaryOperationHint::kBigInt64:
45  break;
46  }
47  return false;
48 }

50 bool BinaryOperationHintToBigIntOperationHint(
51  BinaryOperationHint binop_hint, BigIntOperationHint* bigint_hint) {
52  switch (binop_hint) {
53  case BinaryOperationHint::kSignedSmall:
54  case BinaryOperationHint::kSignedSmallInputs:
55  case BinaryOperationHint::kAdditiveSafeInteger:
56  case BinaryOperationHint::kNumber:
57  case BinaryOperationHint::kNumberOrOddball:
58  case BinaryOperationHint::kAny:
59  case BinaryOperationHint::kNone:
60  case BinaryOperationHint::kString:
61  case BinaryOperationHint::kStringOrStringWrapper:
62  return false;
63  case BinaryOperationHint::kBigInt64:
64  *bigint_hint = BigIntOperationHint::kBigInt64;
65  return true;
66  case BinaryOperationHint::kBigInt:
67  *bigint_hint = BigIntOperationHint::kBigInt;
68  return true;
69  }
70  UNREACHABLE();
71 }

151  const Operator* SpeculativeNumberOp(NumberOperationHint hint) {
152  switch (op_->opcode()) {
153  case IrOpcode::kJSAdd:
154  if (hint == NumberOperationHint::kSignedSmall) {
155  return simplified()->SpeculativeSmallIntegerAdd(hint);
156  } else if (hint == NumberOperationHint::kAdditiveSafeInteger) {
157  return simplified()->SpeculativeAdditiveSafeIntegerAdd(hint);
158  } else {
159  return simplified()->SpeculativeNumberAdd(hint);
160  }
161  case IrOpcode::kJSSubtract:
162  if (hint == NumberOperationHint::kSignedSmall) {
163  return simplified()->SpeculativeSmallIntegerSubtract(hint);
164  } else if (hint == NumberOperationHint::kAdditiveSafeInteger) {
165  return simplified()->SpeculativeAdditiveSafeIntegerSubtract(hint);
166  } else {
167  return simplified()->SpeculativeNumberSubtract(hint);
168  }
169  case IrOpcode::kJSMultiply:
170  return simplified()->SpeculativeNumberMultiply(hint);
171  case IrOpcode::kJSExponentiate:
172  return simplified()->SpeculativeNumberPow(hint);
173  case IrOpcode::kJSDivide:
174  return simplified()->SpeculativeNumberDivide(hint);
175  case IrOpcode::kJSModulus:
176  return simplified()->SpeculativeNumberModulus(hint);
177  case IrOpcode::kJSBitwiseAnd:
178  return simplified()->SpeculativeNumberBitwiseAnd(hint);
179  case IrOpcode::kJSBitwiseOr:
180  return simplified()->SpeculativeNumberBitwiseOr(hint);
181  case IrOpcode::kJSBitwiseXor:
182  return simplified()->SpeculativeNumberBitwiseXor(hint);
183  case IrOpcode::kJSShiftLeft:
184  return simplified()->SpeculativeNumberShiftLeft(hint);
185  case IrOpcode::kJSShiftRight:
186  return simplified()->SpeculativeNumberShiftRight(hint);
187  case IrOpcode::kJSShiftRightLogical:
188  return simplified()->SpeculativeNumberShiftRightLogical(hint);
189  default:
190  break;
191  }
192  UNREACHABLE();
193  }

223  const Operator* SpeculativeNumberCompareOp(NumberOperationHint hint) {
224  switch (op_->opcode()) {
225  case IrOpcode::kJSEqual:
226  return simplified()->SpeculativeNumberEqual(hint);
227  case IrOpcode::kJSLessThan:
228  return simplified()->SpeculativeNumberLessThan(hint);
229  case IrOpcode::kJSGreaterThan:
230  std::swap(left_, right_); // a > b => b < a
231  return simplified()->SpeculativeNumberLessThan(hint);
232  case IrOpcode::kJSLessThanOrEqual:
233  return simplified()->SpeculativeNumberLessThanOrEqual(hint);
234  case IrOpcode::kJSGreaterThanOrEqual:
235  std::swap(left_, right_); // a >= b => b <= a
236  return simplified()->SpeculativeNumberLessThanOrEqual(hint);
237  default:
238  break;
239  }
240  UNREACHABLE();
241  }

244  switch (op_->opcode()) {
245  case IrOpcode::kJSEqual:
246  return simplified()->SpeculativeBigIntEqual(hint);
247  case IrOpcode::kJSLessThan:
248  return simplified()->SpeculativeBigIntLessThan(hint);
249  case IrOpcode::kJSGreaterThan:

251  return simplified()->SpeculativeBigIntLessThan(hint);
252  case IrOpcode::kJSLessThanOrEqual:

263  Node* BuildSpeculativeOperation(const Operator* op) {
264  DCHECK_EQ(2, op->ValueInputCount());
265  DCHECK_EQ(1, op->EffectInputCount());
266  DCHECK_EQ(1, op->ControlInputCount());
267  DCHECK_EQ(false, OperatorProperties::HasFrameStateInput(op));
268  DCHECK_EQ(false, OperatorProperties::HasContextInput(op));
269  DCHECK_EQ(1, op->EffectOutputCount());
270  DCHECK_EQ(0, op->ControlOutputCount());
271  return graph()->NewNode(op, left_, right_, effect_, control_);
272  }

274  Node* TryBuildNumberBinop() {
275  NumberOperationHint hint;
276  if (GetBinaryNumberOperationHint(&hint)) {
277  const Operator* op = SpeculativeNumberOp(hint);
278  Node* node = BuildSpeculativeOperation(op);
279  return node;
280  }
281  return nullptr;
282  }

284  Node* TryBuildBigIntBinop() {
285  BigIntOperationHint hint;
286  if (GetBinaryBigIntOperationHint(&hint)) {
287  const Operator* op = SpeculativeBigIntOp(hint);
288  Node* node = BuildSpeculativeOperation(op);
289  return node;
290  }
291  return nullptr;
292  }

294  Node* TryBuildNumberCompare() {
295  NumberOperationHint hint;
296  if (GetCompareNumberOperationHint(&hint)) {
297  // Equality doesn't not perform ToNumber conversions on Oddballs.
298  if (hint == NumberOperationHint::kNumberOrOddball &&
299  op_->opcode() == IrOpcode::kJSEqual) {
300  return nullptr;
301  }
302  const Operator* op = SpeculativeNumberCompareOp(hint);
303  Node* node = BuildSpeculativeOperation(op);
304  return node;
305  }
306  return nullptr;
307  }

309  Node* TryBuildBigIntCompare() {
310  BigIntOperationHint hint;
311  if (GetCompareBigIntOperationHint(&hint)) {
312  const Operator* op = SpeculativeBigIntCompareOp(hint);
313  Node* node = BuildSpeculativeOperation(op);
314  return node;
315  }
316  return nullptr;
317  }

366 JSTypeHintLowering::LoweringResult JSTypeHintLowering::ReduceUnaryOperation(
367  const Operator* op, Node* operand, Node* effect, Node* control,
368  FeedbackSlot slot) const {
369  if (Node* node = BuildDeoptIfFeedbackIsInsufficient(
370  slot, effect, control,
371  DeoptimizeReason::kInsufficientTypeFeedbackForUnaryOperation)) {
372  return LoweringResult::Exit(node);
373  }
374 
375  // Note: Unary and binary operations collect the same kind of feedback.
376  FeedbackSource feedback(feedback_vector(), slot);
377 
378  Node* node;
379  Node* check = nullptr;
380  switch (op->opcode()) {
381  case IrOpcode::kJSBitwiseNot: {
382  // Lower to a speculative xor with -1 if we have some kind of Number
383  // feedback.
384  JSSpeculativeBinopBuilder b(
385  this, jsgraph()->javascript()->BitwiseXor(feedback), operand,
386  jsgraph()->SmiConstant(-1), effect, control, slot);
387  node = b.TryBuildNumberBinop();
388  break;
389  }
390  case IrOpcode::kJSDecrement: {
391  // Lower to a speculative subtraction of 1 if we have some kind of Number
392  // feedback.
393  JSSpeculativeBinopBuilder b(
394  this, jsgraph()->javascript()->Subtract(feedback), operand,
395  jsgraph()->SmiConstant(1), effect, control, slot);
396  node = b.TryBuildNumberBinop();
397  break;
398  }
399  case IrOpcode::kJSIncrement: {
400  // Lower to a speculative addition of 1 if we have some kind of Number
401  // feedback.
402  JSSpeculativeBinopBuilder b(this, jsgraph()->javascript()->Add(feedback),
403  operand, jsgraph()->SmiConstant(1), effect,
404  control, slot);
405  node = b.TryBuildNumberBinop();
406  break;
407  }
408  case IrOpcode::kJSNegate: {
409  // Lower to a speculative multiplication with -1 if we have some kind of
410  // Number feedback.
411  JSSpeculativeBinopBuilder b(
412  this, jsgraph()->javascript()->Multiply(feedback), operand,
413  jsgraph()->SmiConstant(-1), effect, control, slot);
414  node = b.TryBuildNumberBinop();
415  if (!node) {
416  if (jsgraph()->machine()->Is64()) {
417  if (GetBinaryOperationHint(slot) == BinaryOperationHint::kBigInt) {
418  op = jsgraph()->simplified()->SpeculativeBigIntNegate(
419  BigIntOperationHint::kBigInt);
420  node = jsgraph()->graph()->NewNode(op, operand, effect, control);
421  }
422  }
423  }
424  break;
425  }
426  case IrOpcode::kTypeOf: {
427  TypeOfFeedback::Result hint = broker()->GetFeedbackForTypeOf(feedback);
428  switch (hint) {
429  case TypeOfFeedback::kNumber:
430  check = jsgraph()->graph()->NewNode(
431  jsgraph()->simplified()->CheckNumber(FeedbackSource()), operand,
432  effect, control);
433  node = jsgraph()->ConstantNoHole(broker()->number_string(), broker());
434  break;
435  case TypeOfFeedback::kString:
436  check = jsgraph()->graph()->NewNode(
437  jsgraph()->simplified()->CheckString(FeedbackSource()), operand,
438  effect, control);
439  node = jsgraph()->ConstantNoHole(broker()->string_string(), broker());
440  break;
441  case TypeOfFeedback::kFunction: {
442  Node* condition = jsgraph()->graph()->NewNode(
443  jsgraph()->simplified()->ObjectIsDetectableCallable(), operand);
444  check = jsgraph()->graph()->NewNode(
445  jsgraph()->simplified()->CheckIf(
446  DeoptimizeReason::kNotDetectableReceiver, FeedbackSource()),
447  condition, effect, control);
448  node =
449  jsgraph()->ConstantNoHole(broker()->function_string(), broker());
450  break;
451  }
452  default:
453  node = nullptr;
454  break;
455  }
456  break;
457  }
458  default:
459  UNREACHABLE();
460  }
461 
462  if (node != nullptr) {
463  return LoweringResult::SideEffectFree(node, check ? check : node, control);
464  } else {
465  return LoweringResult::NoChange();
466  }
467 }

469 JSTypeHintLowering::LoweringResult JSTypeHintLowering::ReduceBinaryOperation(
470  const Operator* op, Node* left, Node* right, Node* effect, Node* control,
471  FeedbackSlot slot) const {
472  switch (op->opcode()) {
473  case IrOpcode::kJSStrictEqual: {
474  if (Node* node = BuildDeoptIfFeedbackIsInsufficient(
475  slot, effect, control,
476  DeoptimizeReason::kInsufficientTypeFeedbackForCompareOperation)) {
477  return LoweringResult::Exit(node);
478  }
479  // TODO(turbofan): Should we generally support early lowering of
480  // JSStrictEqual operators here?
481  break;
482  }
483  case IrOpcode::kJSEqual:
484  case IrOpcode::kJSLessThan:
485  case IrOpcode::kJSGreaterThan:
486  case IrOpcode::kJSLessThanOrEqual:
487  case IrOpcode::kJSGreaterThanOrEqual: {
488  if (Node* node = BuildDeoptIfFeedbackIsInsufficient(
489  slot, effect, control,
490  DeoptimizeReason::kInsufficientTypeFeedbackForCompareOperation)) {
491  return LoweringResult::Exit(node);
492  }
493  JSSpeculativeBinopBuilder b(this, op, left, right, effect, control, slot);
494  if (Node* node = b.TryBuildNumberCompare()) {
495  return LoweringResult::SideEffectFree(node, node, control);
496  }
497  if (Node* node = b.TryBuildBigIntCompare()) {
498  return LoweringResult::SideEffectFree(node, node, control);
499  }
500  break;
501  }
502  case IrOpcode::kJSInstanceOf: {
503  if (Node* node = BuildDeoptIfFeedbackIsInsufficient(
504  slot, effect, control,
505  DeoptimizeReason::kInsufficientTypeFeedbackForCompareOperation)) {
506  return LoweringResult::Exit(node);
507  }
508  // TODO(turbofan): Should we generally support early lowering of
509  // JSInstanceOf operators here?
510  break;
511  }
512  case IrOpcode::kJSBitwiseOr:
513  case IrOpcode::kJSBitwiseXor:
514  case IrOpcode::kJSBitwiseAnd:
515  case IrOpcode::kJSShiftLeft:
516  case IrOpcode::kJSShiftRight:
517  case IrOpcode::kJSShiftRightLogical:
518  case IrOpcode::kJSAdd:
519  case IrOpcode::kJSSubtract:
520  case IrOpcode::kJSMultiply:
521  case IrOpcode::kJSDivide:
522  case IrOpcode::kJSModulus:
523  case IrOpcode::kJSExponentiate: {
524  if (Node* node = BuildDeoptIfFeedbackIsInsufficient(
525  slot, effect, control,
526  DeoptimizeReason::kInsufficientTypeFeedbackForBinaryOperation)) {
527  return LoweringResult::Exit(node);
528  }
529  JSSpeculativeBinopBuilder b(this, op, left, right, effect, control, slot);
530  if (Node* node = b.TryBuildNumberBinop()) {
531  return LoweringResult::SideEffectFree(node, node, control);
532  }
533  if (op->opcode() != IrOpcode::kJSShiftRightLogical &&
534  op->opcode() != IrOpcode::kJSExponentiate) {
535  if (Node* node = b.TryBuildBigIntBinop()) {
536  return LoweringResult::SideEffectFree(node, node, control);
537  }
538  }
539  break;
540  }
541  default:
542  UNREACHABLE();
543  }
544  return LoweringResult::NoChange();
545 }
