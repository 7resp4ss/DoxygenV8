19 template <typename T, typename CompressionScheme>
20 Address TaggedMember<T, CompressionScheme>::tagged_to_full(
21  Tagged_t tagged_value) {
22 #ifdef V8_COMPRESS_POINTERS
23  if constexpr (std::is_same_v<Smi, T>) {
24  DCHECK(HAS_SMI_TAG(tagged_value));
25  return CompressionScheme::DecompressTaggedSigned(tagged_value);
26  } else {
27  return CompressionScheme::DecompressTagged(tagged_value);
28  }
29 #else
30  return tagged_value;
31 #endif
32 }

108 template <typename T, typename CompressionScheme>
109 Tagged<T> TaggedMember<T, CompressionScheme>::SeqCst_CompareAndSwap(
110  HeapObjectLayout* host, Tagged<T> expected_value, Tagged<T> value,
111  WriteBarrierMode mode) {
112  Tagged<T> old_value(tagged_to_full(AsAtomicTagged::SeqCst_CompareAndSwap(
113  this->ptr_location(), full_to_tagged(expected_value.ptr()),
114  full_to_tagged(value.ptr()))));
115  if (old_value == expected_value) {
116  WriteBarrier(host, value, mode);
117  }
118  return old_value;

152 template <typename T, typename CompressionScheme>
153 void TaggedMember<T, CompressionScheme>::WriteBarrier(HeapObjectLayout* host,
154  Tagged<T> value,
155  WriteBarrierMode mode) {
156 #ifndef V8_DISABLE_WRITE_BARRIERS
157  if constexpr (!std::is_same_v<Smi, T>) {
158 #if V8_ENABLE_UNCONDITIONAL_WRITE_BARRIERS
159  mode = UPDATE_WRITE_BARRIER;
160 #endif
161  DCHECK(HeapLayout::IsOwnedByAnyHeap(Tagged(host)));
162  WriteBarrier::ForValue(host, this, value, mode);
163  }
164 #endif
165 }
