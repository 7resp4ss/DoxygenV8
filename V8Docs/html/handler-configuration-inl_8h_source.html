113 Handle<Smi> LoadHandler::LoadElement(Isolate* isolate,
114  ElementsKind elements_kind,

118  IsHoleyElementsKind(elements_kind));
119  int config = KindBits::encode(Kind::kElement) |
120  AllowOutOfBoundsBits::encode(LoadModeHandlesOOB(load_mode)) |
121  ElementsKindBits::encode(elements_kind) |
122  AllowHandlingHole::encode(LoadModeHandlesHoles(load_mode)) |
123  IsJsArrayBits::encode(is_js_array);
