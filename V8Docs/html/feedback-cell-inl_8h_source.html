30 void FeedbackCell::clear_padding() {
31  if (FeedbackCell::kAlignedSize == FeedbackCell::kUnalignedSize) return;
32  DCHECK_GE(FeedbackCell::kAlignedSize, FeedbackCell::kUnalignedSize);
33  memset(reinterpret_cast<uint8_t*>(address() + FeedbackCell::kUnalignedSize),
34  0, FeedbackCell::kAlignedSize - FeedbackCell::kUnalignedSize);
35 }
