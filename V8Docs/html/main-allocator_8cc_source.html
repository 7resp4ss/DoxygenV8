140 // Perform an allocation step when the step is reached. size_in_bytes is the
141 // actual size needed for the object (required for InvokeAllocationObservers).
142 // aligned_size_in_bytes is the size of the object including the filler right
143 // before it to reach the right alignment (required to DCHECK the start of the
144 // object). allocation_size is the size of the actual allocation which needs to
145 // be used for the accounting. It can be different from aligned_size_in_bytes in
146 // PagedSpace::AllocateRawAligned, where we have to overallocate in order to be
147 // able to align the allocation afterwards.
148 void MainAllocator::InvokeAllocationObservers(Address soon_object,
149  size_t size_in_bytes,
150  size_t aligned_size_in_bytes,
151  size_t allocation_size) {
152  DCHECK_LE(size_in_bytes, aligned_size_in_bytes);
153  DCHECK_LE(aligned_size_in_bytes, allocation_size);
154  DCHECK(size_in_bytes == aligned_size_in_bytes ||
155  aligned_size_in_bytes == allocation_size);
156 
157  if (!SupportsAllocationObserver() ||
158  !isolate_heap()->IsAllocationObserverActive()) {
159  return;
160  }
161 
162  if (allocation_size >= allocation_counter().NextBytes()) {
163  // Only the first object in a LAB should reach the next step.
164  DCHECK_EQ(soon_object, allocation_info().start() + aligned_size_in_bytes -
165  size_in_bytes);
166 
167  // Right now the LAB only contains that one object.
168  DCHECK_EQ(allocation_info().top() + allocation_size - aligned_size_in_bytes,
169  allocation_info().limit());
170 
171  // Ensure that there is a valid object
172  space_heap()->CreateFillerObjectAt(soon_object,
173  static_cast<int>(size_in_bytes));
174 
175 #if DEBUG
176  // Ensure that allocation_info_ isn't modified during one of the
177  // AllocationObserver::Step methods.
178  LinearAllocationArea saved_allocation_info = allocation_info();
179 #endif
180 
181  // Run AllocationObserver::Step through the AllocationCounter.
182  allocation_counter().InvokeAllocationObservers(soon_object, size_in_bytes,
183  allocation_size);
184 
185  // Ensure that start/top/limit didn't change.
186  DCHECK_EQ(saved_allocation_info.start(), allocation_info().start());
187  DCHECK_EQ(saved_allocation_info.top(), allocation_info().top());
188  DCHECK_EQ(saved_allocation_info.limit(), allocation_info().limit());
189  }
190 
191  DCHECK_LT(allocation_info().limit() - allocation_info().start(),
192  allocation_counter().NextBytes());
193 }

226  int size_in_bytes, AllocationAlignment alignment, AllocationOrigin origin) {

240  aligned_size_in_bytes, max_aligned_size);

347 Address MainAllocator::ComputeLimit(Address start, Address end,
348  size_t min_size) const {
349  DCHECK_GE(end - start, min_size);
350 
351  // Use the full LAB when allocation observers aren't enabled.
352  if (!SupportsAllocationObserver()) return end;
353 
354  // LABs with allocation observers are only used outside GC and on the main
355  // thread.
356  DCHECK(!isolate_heap()->IsInGC());
357  DCHECK(is_main_thread());
358 
359  if (!isolate_heap()->IsInlineAllocationEnabled()) {
360  // LABs are disabled, so we fit the requested area exactly.
361  return start + min_size;
362  }
363 
364  // When LABs are enabled, pick the largest possible LAB size by default.
365  size_t step_size = end - start;
366 
367  if (isolate_heap()->IsAllocationObserverActive()) {
368  // Ensure there are no unaccounted allocations.
369  DCHECK_EQ(allocation_info().start(), allocation_info().top());
370 
371  size_t step = allocation_counter().NextBytes();
372  DCHECK_NE(step, 0);
373  // Generated code may allocate inline from the linear allocation area. To
374  // make sure we can observe these allocations, we use a lower limit.
375  size_t rounded_step = static_cast<size_t>(
376  RoundDown(static_cast<int>(step - 1), ObjectAlignment()));
377  step_size = std::min(step_size, rounded_step);
378  }
379 
380  if (v8_flags.stress_marking) {
381  step_size = std::min(step_size, static_cast<size_t>(64));
382  }
383 
384  DCHECK_LE(start + step_size, end);
385  return start + std::max(step_size, min_size);
386 }

449 bool SemiSpaceNewSpaceAllocatorPolicy::EnsureAllocation(
450  int size_in_bytes, AllocationAlignment alignment, AllocationOrigin origin) {
451  std::optional<base::MutexGuard> guard;
452  if (allocator_->in_gc()) guard.emplace(space_->mutex());
453 
454  FreeLinearAllocationAreaUnsynchronized();
455 
456  std::optional<std::pair<Address, Address>> allocation_result =
457  space_->Allocate(size_in_bytes, alignment);
458  if (!allocation_result) {
459  if (!space_->ReachedTargetCapacity()) {
460  // If allocation failed even though we have not even grown the space to
461  // its target capacity yet, we can bail out early.
462  return false;
463  }
464 
465  if (!space_->heap()->ShouldExpandYoungGenerationOnSlowAllocation(
466  PageMetadata::kPageSize)) {
467  return false;
468  }
469  allocation_result =
470  space_->AllocateOnNewPageBeyondCapacity(size_in_bytes, alignment);
471  if (!allocation_result) return false;
472  }
473 
474  Address start = allocation_result->first;
475  Address end = allocation_result->second;
476 
477  int filler_size = Heap::GetFillToAlign(start, alignment);
478  int aligned_size_in_bytes = size_in_bytes + filler_size;
479  DCHECK_LE(start + aligned_size_in_bytes, end);
480 
481  Address limit;
482 
483  if (allocator_->in_gc()) {
484  // During GC we allow multiple LABs in new space and since Allocate() above
485  // returns the whole remaining page by default, we limit the size of the LAB
486  // here.
487  size_t used = std::max(aligned_size_in_bytes, kLabSizeInGC);
488  limit = std::min(end, start + used);
489  } else {
490  limit = allocator_->ComputeLimit(start, end, aligned_size_in_bytes);
491  }
492  CHECK_LE(limit, end);
493 
494  if (limit != end) {
495  space_->Free(limit, end);
496  }
497 
498  allocator_->ResetLab(start, limit, limit);
499 
500  space_->to_space().AddRangeToActiveSystemPages(allocator_->top(),
501  allocator_->limit());
502  return true;
503 }

538 bool PagedNewSpaceAllocatorPolicy::EnsureAllocation(
539  int size_in_bytes, AllocationAlignment alignment, AllocationOrigin origin) {
540  if (space_->paged_space()->last_lab_page_) {
541  space_->paged_space()->last_lab_page_->DecreaseAllocatedLabSize(
542  allocator_->limit() - allocator_->top());
543  allocator_->ExtendLAB(allocator_->top());
544  // No need to write a filler to the remaining lab because it will either be
545  // reallocated if the lab can be extended or freed otherwise.
546  }
547 
548  if (!paged_space_allocator_policy_->EnsureAllocation(size_in_bytes, alignment,
549  origin)) {
550  if (!TryAllocatePage(size_in_bytes, origin)) {
551  if (!WaitForSweepingForAllocation(size_in_bytes, origin)) {
552  return false;
553  }
554  }
555  }
556 
557  space_->paged_space()->last_lab_page_ =
558  PageMetadata::FromAllocationAreaAddress(allocator_->top());
559  DCHECK_NOT_NULL(space_->paged_space()->last_lab_page_);
560  space_->paged_space()->last_lab_page_->IncreaseAllocatedLabSize(
561  allocator_->limit() - allocator_->top());
562 
563  if (space_heap()->incremental_marking()->IsMinorMarking()) {
564  space_heap()->concurrent_marking()->RescheduleJobIfNeeded(
565  GarbageCollector::MINOR_MARK_SWEEPER);
566  }
567 
568  return true;
569 }

571 bool PagedNewSpaceAllocatorPolicy::WaitForSweepingForAllocation(
572  int size_in_bytes, AllocationOrigin origin) {
573  // This method should be called only when there are no more pages for main
574  // thread to sweep.
575  DCHECK(space_heap()->sweeper()->IsSweepingDoneForSpace(NEW_SPACE));
576  if (!v8_flags.concurrent_sweeping || !space_heap()->sweeping_in_progress())
577  return false;
578  Sweeper* sweeper = space_heap()->sweeper();
579  if (!sweeper->AreMinorSweeperTasksRunning() &&
580  !sweeper->ShouldRefillFreelistForSpace(NEW_SPACE)) {
581 #if DEBUG
582  for (PageMetadata* p : *space_) {
583  DCHECK(p->SweepingDone());
584  p->ForAllFreeListCategories(
585  [space = space_->paged_space()](FreeListCategory* category) {
586  DCHECK_IMPLIES(!category->is_empty(),
587  category->is_linked(space->free_list()));
588  });
589  }
590 #endif // DEBUG
591  // All pages are already swept and relinked to the free list
592  return false;
593  }
594  // When getting here we know that any unswept new space page is currently
595  // being handled by a concurrent sweeping thread. Rather than try to cancel
596  // tasks and restart them, we wait "per page". This should be faster.
597  for (PageMetadata* p : *space_) {
598  if (!p->SweepingDone()) sweeper->WaitForPageToBeSwept(p);
599  }
600  space_->paged_space()->RefillFreeList();
601  DCHECK(!sweeper->ShouldRefillFreelistForSpace(NEW_SPACE));
602  return paged_space_allocator_policy_->TryAllocationFromFreeList(
603  static_cast<size_t>(size_in_bytes), origin);
604 }

663 bool PagedSpaceAllocatorPolicy::RefillLab(int size_in_bytes,
664  AllocationOrigin origin) {
665  // Allocation in this space has failed.
666  DCHECK_GE(size_in_bytes, 0);
667 
668  if (TryExtendLAB(size_in_bytes)) return true;
669 
670  if (TryAllocationFromFreeList(size_in_bytes, origin)) return true;
671 
672  // Don't steal pages from the shared space of the main isolate if running as a
673  // client. The issue is that the concurrent marker may be running on the main
674  // isolate and may reach the page and read its flags, which will then end up
675  // in a race, when the page of the compaction space will be merged back to the
676  // main space. For the same reason, don't take swept pages from the main
677  // shared space.
678  const bool running_from_client_isolate_and_allocating_in_shared_space =
679  (allocator_->identity() == SHARED_SPACE) &&
680  !isolate_heap()->isolate()->is_shared_space_isolate();
681  if (running_from_client_isolate_and_allocating_in_shared_space) {
682  // Avoid OOM crash in the GC in order to invoke NearHeapLimitCallback after
683  // GC and give it a chance to increase the heap limit.
684  if (!isolate_heap()->force_oom() &&
685  TryExpandAndAllocate(size_in_bytes, origin)) {
686  return true;
687  }
688  return false;
689  }
690 
691  // Sweeping is still in progress. The sweeper doesn't work with black
692  // allocated pages, so it's fine for the compaction space to refill the
693  // freelist from just swept pages.
694  if (space_heap()->sweeping_in_progress()) {
695  // First try to refill the free-list, concurrent sweeper threads
696  // may have freed some objects in the meantime.
697  if (space_heap()->sweeper()->ShouldRefillFreelistForSpace(
698  allocator_->identity())) {
699  space_->RefillFreeList();
700 
701  // Retry the free list allocation.
702  if (TryAllocationFromFreeList(static_cast<size_t>(size_in_bytes), origin))
703  return true;
704  }
705 
706  static constexpr int kMaxPagesToSweep = 1;
707  if (ContributeToSweeping(kMaxPagesToSweep)) {
708  if (TryAllocationFromFreeList(size_in_bytes, origin)) {
709  return true;
710  }
711  }
712  }
713 
714  // If there is not enough memory in the compaction space left, try to steal
715  // a page from the corresponding "regular" page space.
716  // Don't do this though when black allocated pages are enabled and incremental
717  // marking is in progress, because otherwise evacuating into a black allocated
718  // page will cause the marker to miss the object.
719  const bool incremental_marking_with_black_allocated_pages_is_running =
720  v8_flags.black_allocated_pages &&
721  space_heap()->incremental_marking()->IsMajorMarking();
722  if (!incremental_marking_with_black_allocated_pages_is_running &&
723  space_->is_compaction_space()) {
724  DCHECK_NE(NEW_SPACE, allocator_->identity());
725  PagedSpaceBase* main_space =
726  space_heap()->paged_space(allocator_->identity());
727  PageMetadata* page = main_space->RemovePageSafe(size_in_bytes);
728  if (page != nullptr) {
729  // Make sure we don't evacuate into a black allocated page.
730  DCHECK_IMPLIES(v8_flags.black_allocated_pages,
731  !page->Chunk()->IsFlagSet(MemoryChunk::BLACK_ALLOCATED));
732  space_->AddPage(page);
733  if (TryAllocationFromFreeList(static_cast<size_t>(size_in_bytes), origin))
734  return true;
735  }
736  }
737 
738  if (allocator_->identity() != NEW_SPACE &&
739  space_heap()->ShouldExpandOldGenerationOnSlowAllocation(
740  allocator_->local_heap(), origin) &&
741  space_heap()->CanExpandOldGeneration(space_->AreaSize())) {
742  if (TryExpandAndAllocate(static_cast<size_t>(size_in_bytes), origin)) {
743  return true;
744  }
745  }
746 
747  // Try sweeping all pages.
748  if (ContributeToSweeping()) {
749  if (TryAllocationFromFreeList(size_in_bytes, origin)) {
750  return true;
751  }
752  }
753 
754  if (allocator_->identity() != NEW_SPACE && allocator_->in_gc() &&
755  !space_heap()->force_oom()) {
756  // Avoid OOM crash in the GC in order to invoke NearHeapLimitCallback after
757  // GC and give it a chance to increase the heap limit.
758  if (TryExpandAndAllocate(size_in_bytes, origin)) {
759  return true;
760  }
761  }
762  return false;
763 }

765 bool PagedSpaceAllocatorPolicy::TryExpandAndAllocate(size_t size_in_bytes,
766  AllocationOrigin origin) {
767  // Run in a loop because concurrent threads might allocate from the new free
768  // list entries before this thread gets a chance.
769  while (space_->TryExpand(allocator_->local_heap(), origin)) {
770  if (TryAllocationFromFreeList(static_cast<size_t>(size_in_bytes), origin)) {
771  return true;
772  }
773  }
774  return false;

777 bool PagedSpaceAllocatorPolicy::ContributeToSweeping(uint32_t max_pages) {
778  if (!space_heap()->sweeping_in_progress_for_space(allocator_->identity()))
779  return false;
780  if (space_heap()->sweeper()->IsSweepingDoneForSpace(allocator_->identity()))
781  return false;
782 
783  const bool is_main_thread =
784  allocator_->is_main_thread() ||
785  (allocator_->in_gc() && isolate_heap()->IsMainThread());
786  const auto sweeping_scope_kind =
787  is_main_thread ? ThreadKind::kMain : ThreadKind::kBackground;
788  const auto sweeping_scope_id = space_heap()->sweeper()->GetTracingScope(
789  allocator_->identity(), is_main_thread);
790 
791  TRACE_GC_EPOCH_WITH_FLOW(
792  isolate_heap()->tracer(), sweeping_scope_id, sweeping_scope_kind,
793  isolate_heap()->sweeper()->GetTraceIdForFlowEvent(sweeping_scope_id),

796  // final atomic pause.

799  : Sweeper::SweepingMode::kLazyOrConcurrent;

801  if (!space_heap()->sweeper()->ParallelSweepSpace(allocator_->identity(),
802  sweeping_mode, max_pages)) {
803  return false;
804  }
805  space_->RefillFreeList();

822 bool PagedSpaceAllocatorPolicy::TryAllocationFromFreeList(
823  size_t size_in_bytes, AllocationOrigin origin) {
824  PagedSpace::ConcurrentAllocationMutex guard(space_);
825  DCHECK(IsAligned(size_in_bytes, kTaggedSize));
826  DCHECK_LE(allocator_->top(), allocator_->limit());
827 #ifdef DEBUG
828  if (allocator_->top() != allocator_->limit()) {
829  DCHECK_EQ(PageMetadata::FromAddress(allocator_->top()),
830  PageMetadata::FromAddress(allocator_->limit() - 1));
831  }
832 #endif
833  // Don't free list allocate if there is linear space available.
834  DCHECK_LT(static_cast<size_t>(allocator_->limit() - allocator_->top()),
835  size_in_bytes);
836 
837  size_t new_node_size = 0;
838  Tagged<FreeSpace> new_node =
839  space_->free_list_->Allocate(size_in_bytes, &new_node_size, origin);
840  if (new_node.is_null()) return false;
841  DCHECK_GE(new_node_size, size_in_bytes);
842 
843  // The old-space-step might have finished sweeping and restarted marking.
844  // Verify that it did not turn the page of the new node into an evacuation
845  // candidate.
846  DCHECK(!MarkCompactCollector::IsOnEvacuationCandidate(new_node));
847 
848  // Mark the old linear allocation area with a free space map so it can be
849  // skipped when scanning the heap. This also puts it back in the free list
850  // if it is big enough.
851  FreeLinearAllocationAreaUnsynchronized();
852 
853  // Memory in the linear allocation area is counted as allocated. We may free
854  // a little of this again immediately - see below.
855  PageMetadata* page = PageMetadata::FromHeapObject(new_node);
856  space_->IncreaseAllocatedBytes(new_node_size, page);
857 
858  DCHECK_EQ(allocator_->allocation_info().start(),
859  allocator_->allocation_info().top());
860  Address start = new_node.address();
861  Address end = new_node.address() + new_node_size;
862  Address limit = allocator_->ComputeLimit(start, end, size_in_bytes);
863  DCHECK_LE(limit, end);
864  DCHECK_LE(size_in_bytes, limit - start);
865  if (limit != end) {
866  if (!allocator_->supports_extending_lab()) {
867  space_->Free(limit, end - limit);
868  end = limit;
869  } else {
870  DCHECK(allocator_->is_main_thread());
871  space_heap()->CreateFillerObjectAt(limit, static_cast<int>(end - limit));
872  }
873  }
874  SetLinearAllocationArea(start, limit, end);
875  space_->AddRangeToActiveSystemPages(page, start, limit);
876 
877  return true;
878 }

880 bool PagedSpaceAllocatorPolicy::TryExtendLAB(int size_in_bytes) {
881  if (!allocator_->supports_extending_lab()) return false;
882  Address current_top = allocator_->top();
883  if (current_top == kNullAddress) return false;
884  Address current_limit = allocator_->limit();
885  Address max_limit = allocator_->extended_limit();
886  if (current_top + size_in_bytes > max_limit) {
887  return false;
888  }
889  allocator_->AdvanceAllocationObservers();
890  Address new_limit =
891  allocator_->ComputeLimit(current_top, max_limit, size_in_bytes);
892  allocator_->ExtendLAB(new_limit);
893  DCHECK(allocator_->is_main_thread());
894  space_heap()->CreateFillerObjectAt(new_limit,
895  static_cast<int>(max_limit - new_limit));
896  PageMetadata* page = PageMetadata::FromAddress(current_top);
897  // No need to create a black allocation area since new space doesn't use
898  // black allocation.
899  DCHECK_EQ(NEW_SPACE, allocator_->identity());
900  space_->AddRangeToActiveSystemPages(page, current_limit, new_limit);
901  return true;
902 }
