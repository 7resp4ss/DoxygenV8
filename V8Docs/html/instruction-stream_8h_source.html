24 // InstructionStream contains the instruction stream for V8-generated code
25 // objects.
26 //
27 // When V8_EXTERNAL_CODE_SPACE is enabled, InstructionStream objects are
28 // allocated in a separate pointer compression cage instead of the cage where
29 // all the other objects are allocated.
30 //
31 // An InstructionStream is a trusted object as it lives outside of the sandbox
32 // and contains trusted content (machine code). However, it is special in that
33 // it doesn't live in the trusted space but instead in the code space.
34 class InstructionStream : public TrustedObject {
35  public:
36  // All InstructionStream objects have the following layout:
37  //
38  // +--------------------------+
39  // | header |
40  // +--------------------------+ <-- body_start()
41  // | instructions | == instruction_start()
42  // | ... |
43  // | padded to meta alignment | see kMetadataAlignment
44  // +--------------------------+ <-- instruction_end()
45  // | metadata | == metadata_start() (MS)
46  // | ... |
47  // | | <-- MS + handler_table_offset()
48  // | | <-- MS + constant_pool_offset()
49  // | | <-- MS + code_comments_offset()
50  // | | <-- MS + unwinding_info_offset()
51  // | padded to obj alignment |
52  // +--------------------------+ <-- metadata_end() == body_end()
53  // | padded to kCodeAlignmentMinusCodeHeader
54  // +--------------------------+
55  //
56  // In other words, the variable-size 'body' consists of 'instructions' and
57  // 'metadata'.
58 
59  // Constants for use in static asserts, stating whether the body is adjacent,
60  // i.e. instructions and metadata areas are adjacent.
61  static constexpr bool kOnHeapBodyIsContiguous = true;
62  static constexpr bool kOffHeapBodyIsContiguous = false;
63  static constexpr bool kBodyIsContiguous =
64  kOnHeapBodyIsContiguous && kOffHeapBodyIsContiguous;

66  inline Address instruction_start() const;

125  // Layout description.
126 #define ISTREAM_FIELDS(V) \
127  V(kCodeOffset, kProtectedPointerSize) \
128  V(kRelocationInfoOffset, kProtectedPointerSize) \
129  /* Data or code not directly visited by GC directly starts here. */ \
130  V(kDataStart, 0) \

132  V(kConstantPoolOffsetOffset, V8_EMBEDDED_CONSTANT_POOL_BOOL ? kIntSize : 0) \

138  static_assert(kCodeAlignment >= kHeaderSize);
139  // We do two things to ensure kCodeAlignment of the entry address:
140  // 1) Add kCodeAlignmentMinusCodeHeader padding once in the beginning of every
141  // MemoryChunk.
142  // 2) Round up all IStream allocations to a multiple of kCodeAlignment, see
143  // TrailingPaddingSizeFor.
144  // Together, the IStream object itself will always start at offset
145  // kCodeAlignmentMinusCodeHeader, which aligns the entry to kCodeAlignment.
146  static constexpr int kCodeAlignmentMinusCodeHeader =
147  kCodeAlignment - kHeaderSize;
