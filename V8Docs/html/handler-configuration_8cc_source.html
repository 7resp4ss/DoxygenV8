26 // TODO(ishell): Remove templatezation once we move common bits from
27 // Load/StoreHandler to the base class.
28 template <typename ICHandler, bool fill_handler = true>
29 int InitPrototypeChecksImpl(Isolate* isolate, DirectHandle<ICHandler> handler,
30  Tagged<Smi>* smi_handler,
31  DirectHandle<Map> lookup_start_object_map,
32  MaybeObjectDirectHandle data1,
33  MaybeObjectDirectHandle maybe_data2) {
34  int data_size = 1;
35  // Holder-is-receiver case itself does not add entries unless there is an
36  // optional data2 value provided.
37 
38  DCHECK_IMPLIES(IsJSGlobalObjectMap(*lookup_start_object_map),
39  lookup_start_object_map->is_prototype_map());
40 
41  if (IsPrimitiveMap(*lookup_start_object_map) ||
42  lookup_start_object_map->is_access_check_needed()) {
43  DCHECK(!IsJSGlobalObjectMap(*lookup_start_object_map));
44  // The validity cell check for primitive and global proxy receivers does
45  // not guarantee that certain native context ever had access to other
46  // native context. However, a handler created for one native context could
47  // be used in other native context through the megamorphic stub cache.
48  // So we record the original native context to which this handler
49  // corresponds.
50  if (fill_handler) {
51  DirectHandle<Context> native_context = isolate->native_context();
52  handler->set_data2(MakeWeak(*native_context));
53  } else {
54  // Enable access checks on the lookup start object.
55  *smi_handler = SetBitFieldValue<
56  typename ICHandler::DoAccessCheckOnLookupStartObjectBits>(
57  isolate, *smi_handler, true);
58  }
59  data_size++;
60  } else if (lookup_start_object_map->is_dictionary_map() &&
61  !IsJSGlobalObjectMap(*lookup_start_object_map)) {
62  if (!fill_handler) {
63  // Enable lookup on lookup start object.
64  *smi_handler =
65  SetBitFieldValue<typename ICHandler::LookupOnLookupStartObjectBits>(
66  isolate, *smi_handler, true);
67  }
68  }
69  if (fill_handler) {
70  handler->set_data1(*data1);
71  }
72  if (!maybe_data2.is_null()) {
73  if (fill_handler) {
74  // This value will go either to data2 or data3 slot depending on whether
75  // data2 slot is already occupied by native context.
76  if (data_size == 1) {
77  handler->set_data2(*maybe_data2);
78  } else {
79  DCHECK_EQ(2, data_size);
80  handler->set_data3(*maybe_data2);
81  }
82  }
83  data_size++;
84  }
85  return data_size;
86 }
87 
88 // Returns 0 if the validity cell check is enough to ensure that the
89 // prototype chain from |lookup_start_object_map| till |holder| did not change.
90 // If the |holder| is an empty handle then the full prototype chain is
91 // checked.
92 template <typename ICHandler>
93 int GetHandlerDataSize(
94  Isolate* isolate, Tagged<Smi>* smi_handler,
95  DirectHandle<Map> lookup_start_object_map, MaybeObjectDirectHandle data1,
96  MaybeObjectDirectHandle maybe_data2 = MaybeObjectDirectHandle()) {
97  DCHECK_NOT_NULL(smi_handler);
98  return InitPrototypeChecksImpl<ICHandler, false>(
99  isolate, DirectHandle<ICHandler>(), smi_handler, lookup_start_object_map,
100  data1, maybe_data2);
101 }

105  Isolate* isolate, DirectHandle<ICHandler> handler,
106  DirectHandle<Map> lookup_start_object_map, MaybeObjectDirectHandle data1,

143 Handle<Object> LoadHandler::LoadFullChain(
144  Isolate* isolate, DirectHandle<Map> lookup_start_object_map,
145  const MaybeObjectDirectHandle& holder, Handle<Smi> smi_handler_handle) {
146  Tagged<Smi> smi_handler = *smi_handler_handle;
147  MaybeObjectDirectHandle data1 = holder;
148  int data_size = GetHandlerDataSize<LoadHandler>(
149  isolate, &smi_handler, lookup_start_object_map, data1);
150 
151  DirectHandle<UnionOf<Smi, Cell>> validity_cell =
152  Map::GetOrCreatePrototypeChainValidityCell(lookup_start_object_map,
153  isolate);
154  if (IsSmi(*validity_cell)) {
155  DCHECK_EQ(1, data_size);
156  // Lookup on lookup start object isn't supported in case of a simple smi
157  // handler.
158  if (!LookupOnLookupStartObjectBits::decode(smi_handler.value())) {
159  return smi_handler_handle;
160  }
161  }
162 
163  Handle<LoadHandler> handler = isolate->factory()->NewLoadHandler(data_size);
164 
165  handler->set_smi_handler(smi_handler);
166  handler->set_validity_cell(*validity_cell);
167  InitPrototypeChecks(isolate, direct_handle(handler), lookup_start_object_map,
168  data1);
169  return handler;

172 Handle<Object> LoadHandler::LoadNonExistent(
173  Isolate* isolate, DirectHandle<Map> lookup_start_object_map) {
174  Tagged<Smi> smi_handler = Smi::FromInt(KindBits::encode(Kind::kNonExistent));
175  MaybeObjectDirectHandle data1 =
176  MaybeObjectDirectHandle(isolate->factory()->null_value());
177  int data_size = GetHandlerDataSize<LoadHandler>(
178  isolate, &smi_handler, lookup_start_object_map, data1);
179 
180  DirectHandle<PrototypeInfo> prototype_info;
181  DirectHandle<UnionOf<Smi, Cell>> validity_cell =
182  Map::GetOrCreatePrototypeChainValidityCell(lookup_start_object_map,
183  isolate, &prototype_info);
184 
185  // Try to fetch cached handler if it was already created.
186  int handler_index;
187 
188  if (IsSmi(*validity_cell)) {
189  DCHECK_EQ(1, data_size);
190  // Lookup on lookup start object isn't supported in case of a simple smi
191  // handler.
192  if (!LookupOnLookupStartObjectBits::decode(smi_handler.value())) {
193  return handle(smi_handler, isolate);
194  }
195  // Handlers with Smi validity cells are not cacheable.
196  handler_index = -1;
197 
198  } else {
199  // |smi_handler| already has do-access-check-on-lookup or lookup-on-receiver
200  // bits set if necessary, so just use them in order to figure out the index
201  // of the cached non-existent handler.
202  int config = smi_handler.value();
203  const uint32_t non_default_mask =
204  DoAccessCheckOnLookupStartObjectBits::kMask |
205  LookupOnLookupStartObjectBits::kMask;
206  if (config & non_default_mask) {
207  if (config & LookupOnLookupStartObjectBits::kMask) {
208  handler_index =
209  PrototypeInfo::kLoadNonExistentHandlerWithLookupOnReceiver;
210  } else {
211  DCHECK(config & DoAccessCheckOnLookupStartObjectBits::kMask);
212  // This is a less often used version, we don't cache it.
213  handler_index = -1;
214  }
215  } else {
216  handler_index = PrototypeInfo::kLoadNonExistentHandlerDefault;
217  }
218 
219  if (handler_index >= 0) {
220  auto maybe_handler = prototype_info->cached_handler(handler_index);
221  if (!IsSmi(maybe_handler)) {
222  Tagged<LoadHandler> handler = Cast<LoadHandler>(maybe_handler);
223  // Handler's validity cell must be the up-to-date one.
224  DCHECK_EQ(handler->validity_cell(), *validity_cell);
225  return handle(handler, isolate);
226  }
227  // Proceed with allocating a handler and caching it in prototype info.
228  }
229  }
230 
231  Handle<LoadHandler> handler = isolate->factory()->NewLoadHandler(data_size);
232 
233  handler->set_smi_handler(smi_handler);
234  handler->set_validity_cell(*validity_cell);
235  InitPrototypeChecks(isolate, direct_handle(handler), lookup_start_object_map,
236  data1);
237 
238  if (handler_index >= 0) {
239  prototype_info->set_cached_handler(handler_index, *handler);
240  }
241  return handler;
242 }

245 KeyedAccessLoadMode LoadHandler::GetKeyedAccessLoadMode(
246  Tagged<MaybeObject> handler) {
247  DisallowGarbageCollection no_gc;
248  if (IsSmi(handler)) {
249  int const raw_handler = handler.ToSmi().value();
250  Kind const kind = KindBits::decode(raw_handler);
251  if (kind == Kind::kElement || kind == Kind::kIndexedString) {
252  bool handle_oob = AllowOutOfBoundsBits::decode(raw_handler);

254  return CreateKeyedAccessLoadMode(handle_oob, handle_holes);
255  }
256  }
257  return KeyedAccessLoadMode::kInBounds;

303  bool is_dictionary_map = transition_map->is_dictionary_map();

305  if (!is_dictionary_map) {

333 // static
334 MaybeObjectHandle StoreHandler::StoreTransition(Isolate* isolate,
335  Handle<Map> transition_map) {
336  bool is_dictionary_map = transition_map->is_dictionary_map();
337  // We should not create transition handlers for fast prototype maps
338  // because they are not reusable anyway.
339  DCHECK_IMPLIES(transition_map->is_prototype_map(), is_dictionary_map);
340 #ifdef DEBUG
341  if (!is_dictionary_map) {
342  InternalIndex descriptor = transition_map->LastAdded();
343  DirectHandle<DescriptorArray> descriptors(
344  transition_map->instance_descriptors(isolate), isolate);
345  // Private fields must be added via StoreOwnTransition handler.
346  DCHECK(!descriptors->GetKey(descriptor)->IsPrivateName());
347  PropertyDetails details = descriptors->GetDetails(descriptor);
348  if (descriptors->GetKey(descriptor)->IsPrivate()) {
349  DCHECK_EQ(DONT_ENUM, details.attributes());
350  } else {
351  DCHECK_EQ(NONE, details.attributes());
352  }
353  Representation representation = details.representation();
354  DCHECK(!representation.IsNone());
355  }
356 #endif
357  // Declarative handlers don't support access checks.
358  DCHECK(!transition_map->is_access_check_needed());
359 
360  // Get validity cell value if it is necessary for the handler.
361  DirectHandle<UnionOf<Smi, Cell>> validity_cell;
362  if (is_dictionary_map || !transition_map->IsPrototypeValidityCellValid()) {
363  validity_cell =
364  Map::GetOrCreatePrototypeChainValidityCell(transition_map, isolate);
365  }
366 
367  if (is_dictionary_map) {
368  DCHECK(!IsJSGlobalObjectMap(*transition_map));
369  Handle<StoreHandler> handler = isolate->factory()->NewStoreHandler(0);
370  // Store normal with enabled lookup on receiver.
371  int config = KindBits::encode(Kind::kNormal) |
372  LookupOnLookupStartObjectBits::encode(true);
373  handler->set_smi_handler(Smi::FromInt(config));
374  handler->set_validity_cell(*validity_cell);
375  return MaybeObjectHandle(handler);
376 
377  } else {
378  // Ensure the transition map contains a valid prototype validity cell.
379  if (!validity_cell.is_null()) {
380  transition_map->set_prototype_validity_cell(*validity_cell,
381  kRelaxedStore);
382  }
383  return MaybeObjectHandle::Weak(transition_map);
384  }
385 }

616  PrintSmiLoadHandler(raw_handler, os);
