39  static void resetContext(const v8::WeakCallbackInfo<WeakCallbackData>& data) {
40  // InspectedContext is alive here because weak handler is still alive.
41  data.GetParameter()->m_context->m_weakCallbackData = nullptr;
42  data.GetParameter()->m_context->m_context.Reset();
43  v8::debug::GetCurrentPlatform()
44  ->GetForegroundTaskRunner(data.GetIsolate())
45  ->PostTask(
46  std::make_unique<ContextCollectedCallbacks>(data.GetParameter()));
47  }
48 
49  private:
50  InspectedContext* m_context;
51  V8InspectorImpl* m_inspector;
52  int m_groupId;
53  int m_contextId;
54 
55  friend class InspectedContext::ContextCollectedCallbacks;
56 };

62 InspectedContext::InspectedContext(V8InspectorImpl* inspector,
63  const V8ContextInfo& info, int contextId)
64  : m_inspector(inspector),
65  m_context(inspector->isolate(), info.context),
66  m_contextId(contextId),
67  m_contextGroupId(info.contextGroupId),
68  m_origin(toString16(info.origin)),
69  m_humanReadableName(toString16(info.humanReadableName)),
70  m_auxData(toString16(info.auxData)),
71  m_uniqueId(internal::V8DebuggerId::generate(inspector)) {
72  v8::debug::SetContextId(info.context, contextId);
73  m_weakCallbackData =
74  new WeakCallbackData(this, m_inspector, m_contextGroupId, m_contextId);
75  m_context.SetWeak(m_weakCallbackData,
76  &InspectedContext::WeakCallbackData::resetContext,
77  v8::WeakCallbackType::kParameter);
78 
79  v8::Context::Scope contextScope(info.context);
80  v8::HandleScope handleScope(inspector->isolate());
81  v8::Local<v8::Object> global = info.context->Global();
82  v8::Local<v8::Value> console;
83  if (!global->Get(info.context, toV8String(inspector->isolate(), "console"))
84  .ToLocal(&console) ||
85  !console->IsObject()) {
86  return;
87  }
88 
89  m_inspector->console()->installAsyncStackTaggingAPI(info.context,
90  console.As<v8::Object>());
91 
92  if (info.hasMemoryOnConsole) {
93  m_inspector->console()->installMemoryGetter(info.context,
94  console.As<v8::Object>());
95  }
96 }

145 bool InspectedContext::addInternalObject(v8::Local<v8::Object> object,
146  V8InternalValueType type) {
147  if (m_internalObjects.IsEmpty()) {
148  m_internalObjects.Reset(isolate(),
149  v8::debug::EphemeronTable::New(isolate()));
150  }
151  v8::Local<v8::debug::EphemeronTable> new_map =
152  m_internalObjects.Get(isolate())->Set(
153  isolate(), object,
154  v8::Integer::New(isolate(), static_cast<int>(type)));
155  m_internalObjects.Reset(isolate(), new_map);
156  return true;
157 }

159 V8InternalValueType InspectedContext::getInternalType(
160  v8::Local<v8::Object> object) {
161  if (m_internalObjects.IsEmpty()) return V8InternalValueType::kNone;
162  v8::Local<v8::Value> typeValue;
163  if (!m_internalObjects.Get(isolate())
164  ->Get(isolate(), object)
165  .ToLocal(&typeValue) ||
166  !typeValue->IsUint32()) {
167  return V8InternalValueType::kNone;
168  }
169  return static_cast<V8InternalValueType>(typeValue.As<v8::Int32>()->Value());
170 }
