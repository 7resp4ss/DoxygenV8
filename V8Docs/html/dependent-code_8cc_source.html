64 void DependentCode::InstallDependency(Isolate* isolate, Handle<Code> code,
65  Handle<HeapObject> object,
66  DependencyGroups groups) {
67  if (V8_UNLIKELY(v8_flags.trace_compilation_dependencies)) {
68  StdoutStream{} << "Installing dependency of [" << code << "] on [" << object
69  << "] in groups [";
70  PrintDependencyGroups(groups);
71  StdoutStream{} << "]\n";
72  }
73  Handle<DependentCode> old_deps(DependentCode::GetDependentCode(*object),
74  isolate);
75  DirectHandle<DependentCode> new_deps =
76  InsertWeakCode(isolate, old_deps, groups, code);
77 
78  // Update the list head if necessary.
79  if (!new_deps.is_identical_to(old_deps)) {
80  DependentCode::SetDependentCode(object, new_deps);
81  }
82 }

84 DirectHandle<DependentCode> DependentCode::InsertWeakCode(
85  Isolate* isolate, Handle<DependentCode> entries, DependencyGroups groups,
86  DirectHandle<Code> code) {
87  if (entries->length() == entries->capacity()) {
88  // We'd have to grow - try to compact first.
89  entries->IterateAndCompact(
90  isolate, [](Tagged<Code>, DependencyGroups) { return false; });
91  }
92 
93  // As the Code object lives outside of the sandbox in trusted space, we need
94  // to use its in-sandbox wrapper object here.
95  MaybeObjectDirectHandle code_slot(MakeWeak(code->wrapper()), isolate);
96  entries = Cast<DependentCode>(WeakArrayList::AddToEnd(
97  isolate, entries, code_slot, Smi::FromInt(groups)));
98  return entries;
99 }

101 template <typename Function>
102 void DependentCode::IterateAndCompact(IsolateForSandbox isolate,
103  const Function& fn) {
104  DisallowGarbageCollection no_gc;
105 
106  int len = length();
107  if (len == 0) return;
108 
109  // We compact during traversal, thus use a somewhat custom loop construct:
110  //
111  // - Loop back-to-front s.t. trailing cleared entries can simply drop off
112  // the back of the list.
113  // - Any cleared slots are filled from the back of the list.
114  int i = len - kSlotsPerEntry;
115  while (i >= 0) {
116  Tagged<MaybeObject> obj = Get(i + kCodeSlotOffset);
117  if (obj.IsCleared()) {
118  len = FillEntryFromBack(i, len);
119  i -= kSlotsPerEntry;
120  continue;
121  }
122 
123  if (fn(Cast<CodeWrapper>(obj.GetHeapObjectAssumeWeak())->code(isolate),
124  static_cast<DependencyGroups>(
125  Get(i + kGroupsSlotOffset).ToSmi().value()))) {
126  len = FillEntryFromBack(i, len);
127  }
128 
129  i -= kSlotsPerEntry;
130  }
131 
132  set_length(len);
133 }

135 bool DependentCode::MarkCodeForDeoptimization(
136  Isolate* isolate, DependentCode::DependencyGroups deopt_groups) {
137  DisallowGarbageCollection no_gc;
138 
139  bool marked_something = false;
140  IterateAndCompact(isolate, [&](Tagged<Code> code, DependencyGroups groups) {
141  if ((groups & deopt_groups) == 0) return false;
142 
143  if (!code->marked_for_deoptimization()) {
144  // Pick a single group out of the applicable deopt groups, to use as the
145  // deopt reason. Only one group is reported to avoid string concatenation.
146  DependencyGroup first_group = static_cast<DependencyGroup>(
147  1 << base::bits::CountTrailingZeros32(groups & deopt_groups));
148  code->SetMarkedForDeoptimization(
149  isolate,
150  DependentCode::DependencyGroupToLazyDeoptReason(first_group));
151  marked_something = true;
152  }
153 
154  return true;
155  });
156 
157  return marked_something;
158 }

160 int DependentCode::FillEntryFromBack(int index, int length) {
161  DCHECK_EQ(index % 2, 0);
162  DCHECK_EQ(length % 2, 0);
163  for (int i = length - kSlotsPerEntry; i > index; i -= kSlotsPerEntry) {
164  Tagged<MaybeObject> obj = Get(i + kCodeSlotOffset);
165  if (obj.IsCleared()) continue;
166 
167  Set(index + kCodeSlotOffset, obj);
168  Set(index + kGroupsSlotOffset, Get(i + kGroupsSlotOffset),
169  SKIP_WRITE_BARRIER);
170  return i;
171  }
172  return index; // No non-cleared entry found.
173 }
