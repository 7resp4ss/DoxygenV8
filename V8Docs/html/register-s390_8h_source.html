56 #if V8_OS_ZOS
57 // z/OS XPLINK 64-bit frame shape (without the 2k stack bias):
58 // [0] Backchain
59 // [1] Environment
60 // [2] Entry Point
61 // [3] Return Address (XPLINK)
62 // [4] GPR8
63 // [5] GPR9
64 // ...
65 // [10] GPR14 / RA Slot
66 // [11] GPR15 / SP Slot
67 // [12] Reserved
68 // [13] Reserved
69 // [14] Debug Area
70 // [15] Reserved
71 // [16] Register Arg1
72 // [17] Register Arg2
73 // [18] Register Arg3
74 // [19] Register Arg4
75 // [20] Register Arg5
76 
77 // Since z/OS port of V8 follows the register assignment from Linux in the
78 // JavaScript context, JS code will set up r2-r6 as parameter registers,
79 // with 6th+ parameters passed on the stack, when calling C functions.
80 // XPLINK allocates stack slots for all parameters regardless of whether
81 // they are passed in registers. To ensure stack slots are available to
82 // store register parameters back to the stack for XPLINK calls, we include
83 // slots for the 5 "register" arguments (r2-r6 as noted above) as part of
84 // the required stack frame slots. Additional params being passed on the
85 // stack will continue to grow from slot 22 and beyond.
86 //
87 // The 2k stack bias for XPLINK will be adjusted from SP into r4 (system
88 // stack pointer) by the CallCFunctionHelper and CEntryStub right before
89 // the actual native call.
90 const int kNumRequiredStackFrameSlots = 21;
91 const int kStackFrameSPSlot = 11;
92 const int kStackFrameRASlot = 10;
93 const int kStackFrameExtraParamSlot = 21;
94 const int kXPLINKStackFrameExtraParamSlot = 19;
95 const int kStackPointerBias = 2048;
96 #else
97 // [0] Back Chain
98 // [1] Reserved for compiler use
99 // [2] GPR 2
100 // [3] GPR 3
101 // ...
102 // [15] GPR 15
103 // [16] FPR 0
104 // [17] FPR 2
105 // [18] FPR 4
106 // [19] FPR 6
107 const int kNumRequiredStackFrameSlots = 20;
108 const int kStackFrameRASlot = 14;
109 const int kStackFrameSPSlot = 15;
110 const int kStackFrameExtraParamSlot = 20;
111 #endif
