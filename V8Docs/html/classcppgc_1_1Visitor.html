Detailed Description

Visitor passed to trace methods.

All managed pointers must have called the Visitor's trace method on them.

class Foo final : public GarbageCollected<Foo> {
public:
void Trace(Visitor* visitor) const {
visitor->Trace(foo_);
visitor->Trace(weak_foo_);
}
private:
Member<Foo> foo_;
WeakMember<Foo> weak_foo_;
};
