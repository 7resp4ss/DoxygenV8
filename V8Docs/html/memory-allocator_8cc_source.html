93 Address MemoryAllocator::AllocateAlignedMemory(
94  size_t chunk_size, size_t area_size, size_t alignment,
95  AllocationSpace space, Executability executable, void* address_hint,
96  VirtualMemory* controller, PageSize page_size, AllocationHint hint) {
97  DCHECK_EQ(space == CODE_SPACE || space == CODE_LO_SPACE,
98  executable == EXECUTABLE);
99  v8::PageAllocator* page_allocator = this->page_allocator(space);
100  DCHECK_LT(area_size, chunk_size);
101 
102  PageAllocator::Permission permissions =
103  executable == EXECUTABLE
104  ? MutablePageMetadata::GetCodeModificationPermission()
105  : PageAllocator::kReadWrite;
106 
107  v8::PageAllocator::AllocationHint page_allocation_hint =
108  v8::PageAllocator::AllocationHint().WithAddress(address_hint);
109  if (hint.MayGrow()) {
110  page_allocation_hint = page_allocation_hint.WithMayGrow();
111  }
112  VirtualMemory reservation(page_allocator, chunk_size, page_allocation_hint,
113  alignment, permissions);
114  if (!reservation.IsReserved()) return HandleAllocationFailure(executable);
115 
116  // We cannot use the last chunk in the address space because we would
117  // overflow when comparing top and limit if this chunk is used for a
118  // linear allocation area.
119  if ((reservation.address() + static_cast<Address>(chunk_size)) == 0u) {
120  CHECK(!reserved_chunk_at_virtual_memory_limit_);
121  reserved_chunk_at_virtual_memory_limit_ = std::move(reservation);
122  CHECK(reserved_chunk_at_virtual_memory_limit_);
123 
124  // Retry reserve virtual memory.
125  reservation = VirtualMemory(page_allocator, chunk_size,
126  page_allocation_hint, alignment, permissions);
127  if (!reservation.IsReserved()) return HandleAllocationFailure(executable);
128  }
129 
130  Address base = reservation.address();
131 
132  if (executable == EXECUTABLE) {
133  ThreadIsolation::RegisterJitPage(base, chunk_size);
134  }
135 
136  UpdateAllocatedSpaceLimits(base, base + chunk_size, executable);
137 
138  *controller = std::move(reservation);
139  return base;
140 }

153 size_t MemoryAllocator::ComputeChunkSize(size_t area_size,
154  AllocationSpace space) {
155  //
156  // +----------------------------+<- base aligned at MemoryChunk::kAlignment
157  // | Header |
158  // +----------------------------+<- area_start_ (base + area_start_)
159  // | Area |
160  // +----------------------------+<- area_end_ (area_start + area_size)
161  // | Committed but not used |
162  // +----------------------------+<- base + chunk_size
163  //
164 
165  return ::RoundUp(
166  MemoryChunkLayout::ObjectStartOffsetInMemoryChunk(space) + area_size,
167  GetCommitPageSize());
168 }

170 std::optional<MemoryAllocator::MemoryChunkAllocationResult>
171 MemoryAllocator::AllocateUninitializedChunkAt(BaseSpace* space,
172  size_t area_size,
173  Executability executable,
174  Address hint, PageSize page_size,
175  AllocationHint allocation_hint) {
176 #ifndef V8_COMPRESS_POINTERS
177  // When pointer compression is enabled, spaces are expected to be at a
178  // predictable address (see mkgrokdump) so we don't supply a hint and rely on
179  // the deterministic behaviour of the BoundedPageAllocator.
180  if (hint == kNullAddress) {
181  hint = reinterpret_cast<Address>(
182  AlignedAddress(isolate_->heap()->GetRandomMmapAddr(),
183  MemoryChunk::GetAlignmentForAllocation()));
184  }
185 #endif
186 
187  VirtualMemory reservation;
188  size_t chunk_size = ComputeChunkSize(area_size, space->identity());
189  DCHECK_EQ(chunk_size % GetCommitPageSize(), 0);
190 
191  Address base = AllocateAlignedMemory(
192  chunk_size, area_size, MemoryChunk::GetAlignmentForAllocation(),
193  space->identity(), executable, reinterpret_cast<void*>(hint),
194  &reservation, page_size, allocation_hint);
195  if (base == kNullAddress) return {};
196 
197  size_ += reservation.size();
198 
199  // Update executable memory size.
200  if (executable == EXECUTABLE) {
201  size_executable_ += reservation.size();
202  }
203 
204  if (heap::ShouldZapGarbage()) {
205  if (executable == EXECUTABLE) {
206  CodePageMemoryModificationScopeForDebugging memory_write_scope(
207  isolate_->heap(), &reservation,
208  base::AddressRegion(base, chunk_size));
209  heap::ZapBlock(base, chunk_size, kZapValue);
210  } else {
211  DCHECK_EQ(executable, NOT_EXECUTABLE);
212  // Zap both page header and object area at once. No guard page in-between.
213  heap::ZapBlock(base, chunk_size, kZapValue);
214  }
215  }
216 
217  LOG(isolate_,
218  NewEvent("MemoryChunk", reinterpret_cast<void*>(base), chunk_size));
219 
220  Address area_start = base + MemoryChunkLayout::ObjectStartOffsetInMemoryChunk(
221  space->identity());
222  Address area_end = area_start + area_size;
223 
224  return MemoryChunkAllocationResult{
225  reinterpret_cast<void*>(base), nullptr, chunk_size, area_start, area_end,
226  std::move(reservation),
227  };
228 }

230 void MemoryAllocator::PartialFreeMemory(MemoryChunkMetadata* chunk,
231  Address start_free,
232  size_t bytes_to_free,
233  Address new_area_end) {
234  VirtualMemory* reservation = chunk->reserved_memory();
235  DCHECK(reservation->IsReserved());
236  chunk->set_size(chunk->size() - bytes_to_free);
237  chunk->set_area_end(new_area_end);
238  if (chunk->Chunk()->IsFlagSet(MemoryChunk::IS_EXECUTABLE)) {
239  // Add guard page at the end.
240  size_t page_size = GetCommitPageSize();
241  DCHECK_EQ(0, chunk->area_end() % static_cast<Address>(page_size));
242  DCHECK_EQ(chunk->ChunkAddress() + chunk->size(), chunk->area_end());
243 
244  if ((V8_HEAP_USE_PTHREAD_JIT_WRITE_PROTECT ||
245  V8_HEAP_USE_BECORE_JIT_WRITE_PROTECT) &&
246  !isolate_->jitless()) {
247  DCHECK(isolate_->RequiresCodeRange());
248  DiscardSealedMemoryScope discard_scope("Partially free memory.");
249  reservation->DiscardSystemPages(chunk->area_end(), page_size);
250  } else {
251  CHECK(reservation->SetPermissions(chunk->area_end(), page_size,
252  PageAllocator::kNoAccess));
253  }
254  }
255  // On e.g. Windows, a reservation may be larger than a page and releasing
256  // partially starting at |start_free| will also release the potentially
257  // unused part behind the current page.
258  const size_t released_bytes = reservation->Release(start_free);
259  DCHECK_GE(size_, released_bytes);
260  size_ -= released_bytes;
261 }

304 void MemoryAllocator::FreeReadOnlyPage(ReadOnlyPageMetadata* chunk) {
305  DCHECK(!chunk->Chunk()->IsFlagSet(MemoryChunk::PRE_FREED));
306  LOG(isolate_, DeleteEvent("MemoryChunk", chunk));
307 
308  UnregisterSharedMemoryChunk(chunk);
309 
310  v8::PageAllocator* allocator = page_allocator(RO_SPACE);
311  VirtualMemory* reservation = chunk->reserved_memory();
312  if (reservation->IsReserved()) {
313  reservation->Free();
314  } else {
315  // Only read-only pages can have a non-initialized reservation object. This
316  // happens when the pages are remapped to multiple locations and where the
317  // reservation would therefore be invalid.
318  FreeMemoryRegion(allocator, chunk->ChunkAddress(),
319  RoundUp(chunk->size(), allocator->AllocatePageSize()));
320  }
321 
322  delete chunk;
323 }

344 void MemoryAllocator::Free(MemoryAllocator::FreeMode mode,
345  MutablePageMetadata* chunk_metadata) {
346  MemoryChunk* chunk = chunk_metadata->Chunk();
347  RecordMemoryChunkDestroyed(chunk);
348  PreFreeMemory(chunk_metadata);
349  chunk_metadata->ReleaseAllAllocatedMemory();
350 
351  switch (mode) {
352  case FreeMode::kImmediately:
353  PerformFreeMemory(chunk_metadata);
354  break;
355  case FreeMode::kPool:
356  // Ensure that we only ever put pages with their markbits cleared into the
357  // pool. This is necessary because `PreFreeMemory` doesn't clear the
358  // marking bitmap and the marking bitmap is reused when this page is taken
359  // out of the pool again.
360  DCHECK(chunk_metadata->IsLivenessClear());
361  DCHECK_EQ(chunk_metadata->size(),
362  static_cast<size_t>(MutablePageMetadata::kPageSize));
363  DCHECK_EQ(chunk->executable(), NOT_EXECUTABLE);
364  // The chunks added to this queue will be cached until memory reducing GC.
365  pool()->Add(isolate_, chunk_metadata);
366  break;
367  }
368 }

370 void MemoryAllocator::FreeLargePagesPooled(
371  std::vector<LargePageMetadata*> large_pages) {
372  for (LargePageMetadata* page : large_pages) {
373  MemoryChunk* chunk = page->Chunk();
374  RecordMemoryChunkDestroyed(chunk);
375  PreFreeMemory(page);
376  page->ReleaseAllAllocatedMemory();
377  }
378 
379  // Try to add all large pages to the page pool. Large pages successfully added
380  // to the pool will be removed from the vector.
381  pool()->AddLarge(isolate_, large_pages);
382 
383  // Pages remaining in the vector couldn't be added to the large page pool.
384  // Free those immediately now.
385  for (LargePageMetadata* page : large_pages) {
386  MemoryAllocator::DeleteMemoryChunk(page);
387  }
388 }

390 PageMetadata* MemoryAllocator::AllocatePage(
391  MemoryAllocator::AllocationMode alloc_mode, Space* space,
392  Executability executable) {
393  const size_t size =
394  MemoryChunkLayout::AllocatableMemoryInMemoryChunk(space->identity());
395  std::optional<MemoryChunkAllocationResult> chunk_info;
396  if (alloc_mode == AllocationMode::kUsePool) {
397  DCHECK_EQ(executable, NOT_EXECUTABLE);
398  chunk_info = AllocateUninitializedPageFromPool(space);
399  }
400 
401  if (!chunk_info) {
402  chunk_info = AllocateUninitializedChunk(
403  space, size, executable, PageSize::kRegular, AllocationHint());
404  }
405 
406  if (!chunk_info) return nullptr;
407 
408  PageMetadata* metadata;
409  if (chunk_info->optional_metadata) {
410  metadata = new (chunk_info->optional_metadata) PageMetadata(
411  isolate_->heap(), space, chunk_info->size, chunk_info->area_start,
412  chunk_info->area_end, std::move(chunk_info->reservation));
413  } else {
414  metadata = new PageMetadata(isolate_->heap(), space, chunk_info->size,
415  chunk_info->area_start, chunk_info->area_end,
416  std::move(chunk_info->reservation));
417  }
418  MemoryChunk* chunk;
419  MemoryChunk::MainThreadFlags flags = metadata->InitialFlags(executable);
420  if (v8_flags.black_allocated_pages && space->identity() != NEW_SPACE &&
421  space->identity() != NEW_LO_SPACE &&
422  isolate_->heap()->incremental_marking()->black_allocation()) {
423  // Disable the write barrier for objects pointing to this page. We don't
424  // need to trigger the barrier for pointers to old black-allocated pages,
425  // since those are never considered for evacuation. However, we have to
426  // keep the old->shared remembered set across multiple GCs, so those
427  // pointers still need to be recorded.
428  if (!IsAnySharedSpace(space->identity())) {
429  flags &= ~MemoryChunk::POINTERS_TO_HERE_ARE_INTERESTING;
430  }
431  // And mark the page as black allocated.
432  flags |= MemoryChunk::BLACK_ALLOCATED;
433  }
434  if (executable) {
435  RwxMemoryWriteScope scope("Initialize a new MemoryChunk.");
436  chunk = new (chunk_info->chunk) MemoryChunk(flags, metadata);
437  } else {
438  chunk = new (chunk_info->chunk) MemoryChunk(flags, metadata);
439  }
440 
441 #ifdef DEBUG
442  if (chunk->executable()) RegisterExecutableMemoryChunk(metadata);
443 #endif // DEBUG
444 
445  DCHECK(metadata->IsLivenessClear());
446  space->InitializePage(metadata);
447  RecordMemoryChunkCreated(chunk);
448  return metadata;
449 }

534 bool MemoryAllocator::ResizeLargePage(LargePageMetadata* page,
535  size_t old_object_size,
536  size_t new_object_size) {
537  const size_t old_reservation_size = page->reservation_.size();
538  const size_t old_page_end = page->reservation_.end();
539  const Address new_area_end = page->area_start() + new_object_size;
540  const Address new_page_end = RoundUp(new_area_end, GetCommitPageSize());
541  const size_t new_page_size = new_page_end - page->ChunkAddress();
542 
543  if (old_page_end == new_page_end) {
544  // We were able to grow the object without growing its owning page.
545  page->set_area_end(new_area_end);
546  return true;
547  }
548 
549  // Currently we only support growing.
550  DCHECK_LT(old_page_end, new_page_end);
551 
552  if (!page->reservation_.Resize(page->ChunkAddress(), new_page_size,
553  PageAllocator::kReadWrite)) {
554  return false;
555  }
556 
557  page->set_area_end(new_area_end);
558  page->set_size(new_page_size);
559 
560 #if DEBUG
561  AllocationSpace space = page->owner_identity();
562  DCHECK(space == NEW_LO_SPACE || space == LO_SPACE);
563 #endif // DEBUG
564 
565  UpdateAllocatedSpaceLimits(page->ChunkAddress(), new_page_end,
566  Executability::NOT_EXECUTABLE);
567 
568  size_ -= old_reservation_size;
569  size_ += page->reservation_.size();
570 
571  return true;
572 }

574 std::optional<MemoryAllocator::MemoryChunkAllocationResult>
575 MemoryAllocator::AllocateUninitializedPageFromPool(Space* space) {
576  MemoryChunkMetadata* chunk_metadata = pool()->Remove(isolate_);
577  if (chunk_metadata == nullptr) return {};
578  const int size = MutablePageMetadata::kPageSize;
579  const Address start = chunk_metadata->ChunkAddress();
580  const Address area_start =
581  start +
582  MemoryChunkLayout::ObjectStartOffsetInMemoryChunk(space->identity());
583  const Address area_end = start + size;
584  // Pooled pages are always regular data pages.
585  DCHECK_NE(CODE_SPACE, space->identity());
586  DCHECK_NE(TRUSTED_SPACE, space->identity());
587  VirtualMemory reservation(data_page_allocator(), start, size);
588  if (heap::ShouldZapGarbage()) {
589  heap::ZapBlock(start, size, kZapValue);
590  }
591  size_ += size;
592  UpdateAllocatedSpaceLimits(start, start + size,
593  Executability::NOT_EXECUTABLE);
594  return MemoryChunkAllocationResult{
595  chunk_metadata->Chunk(), chunk_metadata, size, area_start, area_end,
596  std::move(reservation),
597  };
598 }

600 std::optional<MemoryAllocator::MemoryChunkAllocationResult>
601 MemoryAllocator::TryAllocateUninitializedLargePageFromPool(Space* space,
602  size_t object_size) {
603  const size_t object_start_offset =
604  MemoryChunkLayout::ObjectStartOffsetInMemoryChunk(space->identity());
605  // Select a pooled large page which can store |object_size| bytes. In case the
606  // page is larger than necessary, the next full GC will trim down its size.
607  MemoryChunkMetadata* chunk_metadata =
608  pool()->RemoveLarge(isolate_, object_start_offset + object_size);
609  if (chunk_metadata == nullptr) return {};
610  const Address start = chunk_metadata->ChunkAddress();
611  const size_t size = chunk_metadata->size();
612  const Address area_start = start + object_start_offset;
613  const Address area_end = start + size;
614  VirtualMemory reservation(data_page_allocator(), start, size);
615  if (heap::ShouldZapGarbage()) {
616  heap::ZapBlock(start, size, kZapValue);
617  }
618  size_ += size;
619  UpdateAllocatedSpaceLimits(start, start + size,
620  Executability::NOT_EXECUTABLE);
621  return MemoryChunkAllocationResult{

660 const MemoryChunk* MemoryAllocator::LookupChunkContainingAddressInSafepoint(
661  Address addr) const {
662  // All threads should be either parked or in a safepoint whenever this method
663  // is called, thus pages cannot be allocated or freed at the same time and a
664  // mutex is not required here.
665  // As the address may not correspond to a valid heap object, the chunk we
666  // obtain below is not necessarily a valid chunk.
667  MemoryChunk* chunk = MemoryChunk::FromAddress(addr);
668  // Check if it corresponds to a known normal or large page.
669  if (auto normal_page_it = normal_pages_.find(chunk);
670  normal_page_it != normal_pages_.end()) {
671  // The chunk is a normal page.
672  // auto* normal_page = PageMetadata::cast(chunk);
673  DCHECK_LE((*normal_page_it)->address(), addr);
674  if (chunk->Metadata()->Contains(addr)) return chunk;
675  } else if (auto large_page_it = large_pages_.upper_bound(chunk);
676  large_page_it != large_pages_.begin()) {
677  // The chunk could be inside a large page.
678  DCHECK_IMPLIES(large_page_it != large_pages_.end(),
679  addr < (*large_page_it)->address());
680  auto* large_page_chunk = *std::next(large_page_it, -1);
681  DCHECK_NOT_NULL(large_page_chunk);
682  DCHECK_LE(large_page_chunk->address(), addr);
683  if (large_page_chunk->Metadata()->Contains(addr)) return large_page_chunk;
684  }
685  // Not found in any page.
686  return nullptr;
687 }
