29  void RegisterNode(const NodeBase* node, const MaglevCompilationUnit* unit,
30  BytecodeOffset bytecode_offset, SourcePosition position) {
31  if (nodes_
32  .emplace(node, NodeInfo{next_node_label_,
33  {unit, bytecode_offset, position}})
34  .second) {
35  next_node_label_++;
36  }
37  }

50  void PrintNodeLabel(std::ostream& os, const NodeBase* node) {
51  if (node != nullptr && node->Is<VirtualObject>()) {
52  // VirtualObjects are unregisted nodes, since they are not attached to
53  // the graph, but its inlined allocation is.
54  const VirtualObject* vo = node->Cast<VirtualObject>();
55  os << "VO{" << vo->id() << "}:";
56  node = vo->allocation();
57  }
58  auto node_id_it = nodes_.find(node);
59 
60  if (node_id_it == nodes_.end()) {
61  os << "<unregistered node " << node << ">";
62  return;
63  }
64 
65  if (node->has_id()) {
66  os << "v" << node->id() << "/";
67  }
68  os << "n" << node_id_it->second.label;
69  }
70 
71  void PrintInput(std::ostream& os, const Input& input) {
72  PrintNodeLabel(os, input.node());
73  os << ":" << input.operand();
74  }
75 
76  private:
77  std::map<const NodeBase*, NodeInfo> nodes_;
78  int next_node_label_ = 1;
79 };
