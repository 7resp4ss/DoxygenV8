77 template <typename Impl>
78 template <AllocationType allocation>
79 DirectHandle<Number> FactoryBase<Impl>::NewNumberFromSize(size_t value) {
80  // We can't use Smi::IsValid() here because that operates on a signed
81  // intptr_t, and casting from size_t could create a bogus sign bit.
82  if (value <= static_cast<size_t>(Smi::kMaxValue)) {
83  return direct_handle(Smi::FromIntptr(static_cast<intptr_t>(value)),
84  isolate());
85  }
86  return NewHeapNumber<allocation>(static_cast<double>(value));
87 }

146 template <typename Impl>
147 Tagged<Struct> FactoryBase<Impl>::NewStructInternal(ReadOnlyRoots roots,
148  Tagged<Map> map, int size,
149  AllocationType allocation) {
150  DCHECK_EQ(size, map->instance_size());
151  Tagged<HeapObject> result = AllocateRawWithImmortalMap(size, allocation, map);
152  Tagged<Struct> str = Cast<Struct>(result);
153  Tagged<Undefined> undefined = roots.undefined_value();
154  int length = (size >> kTaggedSizeLog2) - 1;
155  MemsetTagged(str->RawField(Struct::kHeaderSize), undefined, length);
156  return str;
157 }
