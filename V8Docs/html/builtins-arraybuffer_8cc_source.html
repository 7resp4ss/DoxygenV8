42 // Tries to allocate a BackingStore given the input configuration. Either
43 // returns the BackingStore or a message template that should be thrown as
44 // RangeError.
45 std::pair<std::unique_ptr<BackingStore>, std::optional<MessageTemplate>>
46 TryAllocateBackingStore(Isolate* isolate, SharedFlag shared,
47  ResizableFlag resizable, DirectHandle<Object> length,
48  DirectHandle<Object> max_length,
49  InitializedFlag initialized) {
50  DisallowJavascriptExecution no_js(isolate);
51 
52  size_t byte_length;
53  size_t max_byte_length = 0;
54  std::unique_ptr<BackingStore> backing_store;
55 
56  size_t max_allocatable =
57  isolate->array_buffer_allocator()->MaxAllocationSize();
58  DCHECK(max_allocatable <= JSArrayBuffer::kMaxByteLength);
59  static_assert(JSArrayBuffer::kMaxByteLength == JSTypedArray::kMaxByteLength);
60  if (!TryNumberToSize(*length, &byte_length) ||
61  byte_length > max_allocatable) {
62  return {nullptr, MessageTemplate::kInvalidArrayBufferLength};
63  }
64 
65  switch (resizable) {
66  case ResizableFlag::kNotResizable:
67  backing_store =
68  BackingStore::Allocate(isolate, byte_length, shared, initialized);
69  break;
70  case ResizableFlag::kResizable:
71  if (!TryNumberToSize(*max_length, &max_byte_length) ||
72  max_byte_length > max_allocatable) {
73  return {nullptr, MessageTemplate::kInvalidArrayBufferMaxLength};
74  }
75  if (byte_length > max_byte_length) {
76  return {nullptr, MessageTemplate::kInvalidArrayBufferMaxLength};
77  }
78 
79  size_t page_size, initial_pages, max_pages;
80  const auto maybe_range_error =
81  JSArrayBuffer::GetResizableBackingStorePageConfigurationImpl(
82  isolate, byte_length, max_byte_length, &page_size, &initial_pages,
83  &max_pages);
84  if (maybe_range_error.has_value()) {
85  return {nullptr, maybe_range_error.value()};
86  }
87  backing_store = BackingStore::TryAllocateAndPartiallyCommitMemory(
88  isolate, byte_length, max_byte_length, page_size, initial_pages,
89  max_pages, WasmMemoryFlag::kNotWasm, shared);
90  break;
91  }
92 
93  // Range errors bailed out earlier; only the failing allocation needs to be
94  // caught here.
95  if (!backing_store) {
96  return {nullptr, MessageTemplate::kArrayBufferAllocationFailed};
97  }
98  return {std::move(backing_store), std::nullopt};
99 }

101 Tagged<Object> ConstructBuffer(Isolate* isolate,
102  DirectHandle<JSFunction> target,
103  DirectHandle<JSReceiver> new_target,
104  DirectHandle<Object> length,
105  DirectHandle<Object> max_length,
106  InitializedFlag initialized) {
107  // We first try to convert the sizes and collect any possible range errors. If
108  // no errors are observable we create the BackingStore before the
109  // JSArrayBuffer to avoid a complex dance during setup. We then always create
110  // the AB before throwing a possible error as the creation is observable.
111  const SharedFlag shared =
112  *target != target->native_context()->array_buffer_fun()
113  ? SharedFlag::kShared
114  : SharedFlag::kNotShared;
115  const ResizableFlag resizable = max_length.is_null()
116  ? ResizableFlag::kNotResizable
117  : ResizableFlag::kResizable;
118  // BackingStore allocation may GC which is not observable itself.
119  auto [backing_store, range_error] = TryAllocateBackingStore(
120  isolate, shared, resizable, length, max_length, initialized);
121  DirectHandle<JSObject> result;
122  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
123  isolate, result,
124  JSObject::New(target, new_target, {}, NewJSObjectType::kAPIWrapper));
125  auto array_buffer = Cast<JSArrayBuffer>(result);
126  const bool backing_store_creation_failed = !backing_store;
127  array_buffer->Setup(shared, resizable, std::move(backing_store), isolate);
128  if (backing_store_creation_failed) {
129  CHECK(range_error.has_value());
130  THROW_NEW_ERROR_RETURN_FAILURE(isolate, NewRangeError(range_error.value()));
131  }
132  return *array_buffer;
133 }

138 BUILTIN(ArrayBufferConstructor) {
139  HandleScope scope(isolate);
140  DirectHandle<JSFunction> target = args.target();
141  DCHECK(*target == target->native_context()->array_buffer_fun() ||
142  *target == target->native_context()->shared_array_buffer_fun());
143  if (IsUndefined(*args.new_target(), isolate)) { // [[Call]]
144  THROW_NEW_ERROR_RETURN_FAILURE(
145  isolate,
146  NewTypeError(MessageTemplate::kConstructorNotFunction,
147  direct_handle(target->shared()->Name(), isolate)));
148  }
149  // [[Construct]]
150  DirectHandle<JSReceiver> new_target = Cast<JSReceiver>(args.new_target());
151  DirectHandle<Object> length = args.atOrUndefined(isolate, 1);
152 
153  DirectHandle<Object> number_length;
154  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, number_length,
155  Object::ToInteger(isolate, length));
156  if (Object::NumberValue(*number_length) < 0.0) {
157  THROW_NEW_ERROR_RETURN_FAILURE(
158  isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferLength));
159  }
160 
161  DirectHandle<Object> number_max_length;
162  DirectHandle<Object> max_length;
163  DirectHandle<Object> options = args.atOrUndefined(isolate, 2);
164  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
165  isolate, max_length,
166  JSObject::ReadFromOptionsBag(
167  options, isolate->factory()->max_byte_length_string(), isolate));
168 
169  if (!IsUndefined(*max_length, isolate)) {
170  if (*target == target->native_context()->array_buffer_fun()) {
171  isolate->CountUsage(
172  v8::Isolate::UseCounterFeature::kResizableArrayBuffer);
173  } else {
174  isolate->CountUsage(
175  v8::Isolate::UseCounterFeature::kGrowableSharedArrayBuffer);
176  }
177  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, number_max_length,
178  Object::ToInteger(isolate, max_length));
179  if (Object::NumberValue(*number_length) >
180  Object::NumberValue(*number_max_length)) {
181  THROW_NEW_ERROR_RETURN_FAILURE(
182  isolate,
183  NewRangeError(MessageTemplate::kInvalidArrayBufferMaxLength));
184  }
185  }
186  return ConstructBuffer(isolate, target, new_target, number_length,
187  number_max_length, InitializedFlag::kZeroInitialized);
188 }

202 static Tagged<Object> SliceHelper(BuiltinArguments args, Isolate* isolate,
203  const char* kMethodName, bool is_shared) {
204  HandleScope scope(isolate);
205  DirectHandle<Object> start = args.at(1);
206  DirectHandle<Object> end = args.atOrUndefined(isolate, 2);
207 
208  // * If Type(O) is not Object, throw a TypeError exception.
209  // * If O does not have an [[ArrayBufferData]] internal slot, throw a
210  // TypeError exception.
211  CHECK_RECEIVER(JSArrayBuffer, array_buffer, kMethodName);
212  // * [AB] If IsSharedArrayBuffer(O) is true, throw a TypeError exception.
213  // * [SAB] If IsSharedArrayBuffer(O) is false, throw a TypeError exception.
214  CHECK_SHARED(is_shared, array_buffer, kMethodName);
215 
216  // * [AB] If IsDetachedBuffer(buffer) is true, throw a TypeError exception.
217  if (!is_shared && array_buffer->was_detached()) {
218  THROW_NEW_ERROR_RETURN_FAILURE(
219  isolate, NewTypeError(MessageTemplate::kDetachedOperation,
220  isolate->factory()->NewStringFromAsciiChecked(
221  kMethodName)));
222  }
223 
224  // * [AB] Let len be O.[[ArrayBufferByteLength]].
225  // * [SAB] Let len be O.[[ArrayBufferByteLength]].
226  double const len = array_buffer->GetByteLength();
227 
228  // * Let relativeStart be ? ToInteger(start).
229  double relative_start;
230  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
231  isolate, relative_start, Object::IntegerValue(isolate, start));
232 
233  // * If relativeStart < 0, let first be max((len + relativeStart), 0); else
234  // let first be min(relativeStart, len).
235  double const first = (relative_start < 0)
236  ? std::max(len + relative_start, 0.0)
237  : std::min(relative_start, len);
238 
239  // * If end is undefined, let relativeEnd be len; else let relativeEnd be ?
240  // ToInteger(end).
241  double relative_end;
242  if (IsUndefined(*end, isolate)) {
243  relative_end = len;
244  } else {
245  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
246  isolate, relative_end, Object::IntegerValue(isolate, end));
247  }
248 
249  // * If relativeEnd < 0, let final be max((len + relativeEnd), 0); else let
250  // final be min(relativeEnd, len).
251  double const final_ = (relative_end < 0) ? std::max(len + relative_end, 0.0)
252  : std::min(relative_end, len);
253 
254  // * Let newLen be max(final-first, 0).
255  double const new_len = std::max(final_ - first, 0.0);
256  DirectHandle<Object> new_len_obj = isolate->factory()->NewNumber(new_len);
257 
258  // * [AB] Let ctor be ? SpeciesConstructor(O, %ArrayBuffer%).
259  // * [SAB] Let ctor be ? SpeciesConstructor(O, %SharedArrayBuffer%).
260  DirectHandle<JSFunction> constructor_fun =
261  is_shared ? isolate->shared_array_buffer_fun()
262  : isolate->array_buffer_fun();
263  DirectHandle<Object> ctor;
264  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
265  isolate, ctor,
266  Object::SpeciesConstructor(isolate, Cast<JSReceiver>(args.receiver()),
267  constructor_fun));
268 
269  // * Let new be ? Construct(ctor, newLen).
270  DirectHandle<JSReceiver> new_;
271  {
272  constexpr int argc = 1;
273  std::array<DirectHandle<Object>, argc> ctor_args = {new_len_obj};
274 
275  DirectHandle<Object> new_obj;
276  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
277  isolate, new_obj,
278  Execution::New(isolate, ctor, base::VectorOf(ctor_args)));
279 
280  new_ = Cast<JSReceiver>(new_obj);
281  }
282 
283  // * If new does not have an [[ArrayBufferData]] internal slot, throw a
284  // TypeError exception.
285  if (!IsJSArrayBuffer(*new_)) {
286  THROW_NEW_ERROR_RETURN_FAILURE(
287  isolate,
288  NewTypeError(MessageTemplate::kIncompatibleMethodReceiver,
289  isolate->factory()->NewStringFromAsciiChecked(kMethodName),
290  new_));
291  }
292 
293  // * [AB] If IsSharedArrayBuffer(new) is true, throw a TypeError exception.
294  // * [SAB] If IsSharedArrayBuffer(new) is false, throw a TypeError exception.
295  DirectHandle<JSArrayBuffer> new_array_buffer = Cast<JSArrayBuffer>(new_);
296  CHECK_SHARED(is_shared, new_array_buffer, kMethodName);
297 
298  // The created ArrayBuffer might or might not be resizable, since the species
299  // constructor might return a non-resizable or a resizable buffer.
300 
301  // * [AB] If IsDetachedBuffer(new) is true, throw a TypeError exception.
302  if (!is_shared && new_array_buffer->was_detached()) {
303  THROW_NEW_ERROR_RETURN_FAILURE(
304  isolate, NewTypeError(MessageTemplate::kDetachedOperation,
305  isolate->factory()->NewStringFromAsciiChecked(
306  kMethodName)));
307  }
308 
309  // * [AB] If SameValue(new, O) is true, throw a TypeError exception.
310  if (!is_shared && Object::SameValue(*new_, *args.receiver())) {
311  THROW_NEW_ERROR_RETURN_FAILURE(
312  isolate, NewTypeError(MessageTemplate::kArrayBufferSpeciesThis));
313  }
314 
315  // * [SAB] If new.[[ArrayBufferData]] and O.[[ArrayBufferData]] are the same
316  // Shared Data Block values, throw a TypeError exception.
317  if (is_shared &&
318  new_array_buffer->backing_store() == array_buffer->backing_store()) {
319  THROW_NEW_ERROR_RETURN_FAILURE(
320  isolate, NewTypeError(MessageTemplate::kSharedArrayBufferSpeciesThis));
321  }
322 
323  // * If new.[[ArrayBufferByteLength]] < newLen, throw a TypeError exception.
324  size_t new_array_buffer_byte_length = new_array_buffer->GetByteLength();
325  if (new_array_buffer_byte_length < new_len) {
326  THROW_NEW_ERROR_RETURN_FAILURE(
327  isolate,
328  NewTypeError(is_shared ? MessageTemplate::kSharedArrayBufferTooShort
329  : MessageTemplate::kArrayBufferTooShort));
330  }
331 
332  // * [AB] NOTE: Side-effects of the above steps may have detached O.
333  // * [AB] If IsDetachedBuffer(O) is true, throw a TypeError exception.
334  if (!is_shared && array_buffer->was_detached()) {
335  THROW_NEW_ERROR_RETURN_FAILURE(
336  isolate, NewTypeError(MessageTemplate::kDetachedOperation,
337  isolate->factory()->NewStringFromAsciiChecked(
338  kMethodName)));
339  }
340 
341  // * Let fromBuf be O.[[ArrayBufferData]].
342  // * Let toBuf be new.[[ArrayBufferData]].
343  // * Perform CopyDataBlockBytes(toBuf, 0, fromBuf, first, newLen).
344  size_t first_size = first;
345  size_t new_len_size = new_len;
346  DCHECK(new_array_buffer_byte_length >= new_len_size);
347 
348  if (new_len_size != 0) {
349  size_t from_byte_length = array_buffer->GetByteLength();
350  if (V8_UNLIKELY(!is_shared && array_buffer->is_resizable_by_js())) {
351  // The above steps might have resized the underlying buffer. In that case,
352  // only copy the still-accessible portion of the underlying data.
353  if (first_size > from_byte_length) {
354  return *new_; // Nothing to copy.
355  }
356  if (new_len_size > from_byte_length - first_size) {
357  new_len_size = from_byte_length - first_size;
358  }
359  }
360  DCHECK(first_size <= from_byte_length);
361  DCHECK(from_byte_length - first_size >= new_len_size);
362  uint8_t* from_data =
363  reinterpret_cast<uint8_t*>(array_buffer->backing_store()) + first_size;
364  uint8_t* to_data =
365  reinterpret_cast<uint8_t*>(new_array_buffer->backing_store());
366  if (is_shared) {
367  base::Relaxed_Memcpy(reinterpret_cast<base::Atomic8*>(to_data),
368  reinterpret_cast<base::Atomic8*>(from_data),
369  new_len_size);
370  } else {
371  CopyBytes(to_data, from_data, new_len_size);
372  }
373  }
374 
375  return *new_;
376 }

391 static Tagged<Object> ResizeHelper(BuiltinArguments args, Isolate* isolate,
392  const char* kMethodName, bool is_shared) {
393  HandleScope scope(isolate);
394 
395  // 1 Let O be the this value.
396  // 2. Perform ? RequireInternalSlot(O, [[ArrayBufferMaxByteLength]]).
397  CHECK_RECEIVER(JSArrayBuffer, array_buffer, kMethodName);
398  CHECK_RESIZABLE(true, array_buffer, kMethodName);
399 
400  // [RAB] 3. If IsSharedArrayBuffer(O) is true, throw a *TypeError* exception
401  // [GSAB] 3. If IsSharedArrayBuffer(O) is false, throw a *TypeError* exception
402  CHECK_SHARED(is_shared, array_buffer, kMethodName);
403 
404  // Let newByteLength to ? ToIntegerOrInfinity(newLength).
405  DirectHandle<Object> new_length = args.at(1);
406  DirectHandle<Object> number_new_byte_length;
407  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, number_new_byte_length,
408  Object::ToInteger(isolate, new_length));
409 
410  // [RAB] If IsDetachedBuffer(O) is true, throw a TypeError exception.
411  if (!is_shared && array_buffer->was_detached()) {
412  THROW_NEW_ERROR_RETURN_FAILURE(
413  isolate, NewTypeError(MessageTemplate::kDetachedOperation,
414  isolate->factory()->NewStringFromAsciiChecked(
415  kMethodName)));
416  }
417 
418  // [RAB] If newByteLength < 0 or newByteLength >
419  // O.[[ArrayBufferMaxByteLength]], throw a RangeError exception.
420 
421  // [GSAB] If newByteLength < currentByteLength or newByteLength >
422  // O.[[ArrayBufferMaxByteLength]], throw a RangeError exception.
423  size_t new_byte_length;
424  if (!TryNumberToSize(*number_new_byte_length, &new_byte_length)) {
425  THROW_NEW_ERROR_RETURN_FAILURE(
426  isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferResizeLength,
427  isolate->factory()->NewStringFromAsciiChecked(
428  kMethodName)));
429  }
430 
431  if (is_shared && new_byte_length < array_buffer->GetByteLength()) {
432  // GrowableSharedArrayBuffer is only allowed to grow.
433  THROW_NEW_ERROR_RETURN_FAILURE(
434  isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferResizeLength,
435  isolate->factory()->NewStringFromAsciiChecked(
436  kMethodName)));
437  }
438 
439  if (new_byte_length > array_buffer->max_byte_length()) {
440  THROW_NEW_ERROR_RETURN_FAILURE(
441  isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferResizeLength,
442  isolate->factory()->NewStringFromAsciiChecked(
443  kMethodName)));
444  }
445 
446  // [RAB] Let hostHandled be ? HostResizeArrayBuffer(O, newByteLength).
447  // [GSAB] Let hostHandled be ? HostGrowArrayBuffer(O, newByteLength).
448  // If hostHandled is handled, return undefined.
449 
450 #ifdef V8_ENABLE_WEBASSEMBLY
451  auto backing_store = array_buffer->GetBackingStore();
452  if (backing_store->is_wasm_memory()) {
453  size_t old_byte_length =
454  backing_store->byte_length(std::memory_order_seq_cst);
455  // WebAssembly memories cannot shrink, and must be a multiple of the page
456  // size.
457  if (new_byte_length < old_byte_length ||
458  (new_byte_length % wasm::kWasmPageSize) != 0) {
459  THROW_NEW_ERROR_RETURN_FAILURE(
460  isolate,
461  NewRangeError(
462  MessageTemplate::kInvalidArrayBufferResizeLength,
463  isolate->factory()->NewStringFromAsciiChecked(kMethodName)));
464  }
465  Handle<Object> memory =
466  Object::GetProperty(
467  isolate, array_buffer,
468  isolate->factory()->array_buffer_wasm_memory_symbol())
469  .ToHandleChecked();
470  CHECK(IsWasmMemoryObject(*memory));
471  // WasmMemoryObject::Grow handles updating byte_length, as it's used by both
472  // ArrayBuffer.prototype.resize and WebAssembly.Memory.prototype.grow.
473  uint32_t delta_pages =
474  static_cast<uint32_t>(new_byte_length - old_byte_length) /
475  wasm::kWasmPageSize;
476  if (WasmMemoryObject::Grow(isolate, Cast<WasmMemoryObject>(memory),
477  delta_pages) == -1) {
478  THROW_NEW_ERROR_RETURN_FAILURE(
479  isolate, NewRangeError(MessageTemplate::kOutOfMemory,
480  isolate->factory()->NewStringFromAsciiChecked(
481  kMethodName)));
482  }
483  return ReadOnlyRoots(isolate).undefined_value();
484  }
485 #endif // V8_ENABLE_WEBASSEMBLY
486 
487  if (!is_shared) {
488  // [RAB] Let oldBlock be O.[[ArrayBufferData]].
489  // [RAB] Let newBlock be ? CreateByteDataBlock(newByteLength).
490  // [RAB] Let copyLength be min(newByteLength, O.[[ArrayBufferByteLength]]).
491  // [RAB] Perform CopyDataBlockBytes(newBlock, 0, oldBlock, 0, copyLength).
492  // [RAB] NOTE: Neither creation of the new Data Block nor copying from the
493  // old Data Block are observable. Implementations reserve the right to
494  // implement this method as in-place growth or shrinkage.
495  if (array_buffer->GetBackingStore()->ResizeInPlace(isolate,
496  new_byte_length) !=
497  BackingStore::ResizeOrGrowResult::kSuccess) {
498  THROW_NEW_ERROR_RETURN_FAILURE(
499  isolate, NewRangeError(MessageTemplate::kOutOfMemory,
500  isolate->factory()->NewStringFromAsciiChecked(
501  kMethodName)));
502  }
503 
504  // TypedsArrays in optimized code may go out of bounds. Trigger deopts
505  // through the ArrayBufferDetaching protector.
506  if (new_byte_length < array_buffer->byte_length()) {
507  if (Protectors::IsArrayBufferDetachingIntact(isolate)) {
508  Protectors::InvalidateArrayBufferDetaching(isolate);
509  }
510  }
511 
512  isolate->heap()->ResizeArrayBufferExtension(
513  array_buffer->extension(),
514  static_cast<int64_t>(new_byte_length) - array_buffer->byte_length());
515 
516  // [RAB] Set O.[[ArrayBufferByteLength]] to newLength.
517  array_buffer->set_byte_length(new_byte_length);
518  } else {
519  // [GSAB] (Detailed description of the algorithm omitted.)
520  auto result =
521  array_buffer->GetBackingStore()->GrowInPlace(isolate, new_byte_length);
522  if (result == BackingStore::ResizeOrGrowResult::kFailure) {
523  THROW_NEW_ERROR_RETURN_FAILURE(
524  isolate, NewRangeError(MessageTemplate::kOutOfMemory,
525  isolate->factory()->NewStringFromAsciiChecked(
526  kMethodName)));
527  }
528  if (result == BackingStore::ResizeOrGrowResult::kRace) {
529  THROW_NEW_ERROR_RETURN_FAILURE(
530  isolate,
531  NewRangeError(
532  MessageTemplate::kInvalidArrayBufferResizeLength,
533  isolate->factory()->NewStringFromAsciiChecked(kMethodName)));
534  }
535  // Invariant: byte_length for a GSAB is 0 (it needs to be read from the
536  // BackingStore). Don't use the byte_length getter, which DCHECKs that it's
537  // not used on growable SharedArrayBuffers
538  CHECK_EQ(0, array_buffer->byte_length_unchecked());
539  }
540  return ReadOnlyRoots(isolate).undefined_value();
541 }

543 // ES #sec-get-sharedarraybuffer.prototype.bytelength
544 // get SharedArrayBuffer.prototype.byteLength
545 BUILTIN(SharedArrayBufferPrototypeGetByteLength) {
546  const char* const kMethodName = "get SharedArrayBuffer.prototype.byteLength";
547  HandleScope scope(isolate);
548  // 1. Let O be the this value.
549  // 2. Perform ? RequireInternalSlot(O, [[ArrayBufferData]]).
550  CHECK_RECEIVER(JSArrayBuffer, array_buffer, kMethodName);
551  // 3. If IsSharedArrayBuffer(O) is false, throw a TypeError exception.
552  CHECK_SHARED(true, array_buffer, kMethodName);
553 
554  DCHECK_IMPLIES(!array_buffer->GetBackingStore()->is_wasm_memory(),
555  array_buffer->max_byte_length() ==
556  array_buffer->GetBackingStore()->max_byte_length());
557

576 Tagged<Object> ArrayBufferTransfer(Isolate* isolate,
577  DirectHandle<JSArrayBuffer> array_buffer,
578  DirectHandle<Object> new_length,
579  PreserveResizability preserve_resizability,
580  const char* method_name) {
581  size_t max_allocatable =
582  isolate->array_buffer_allocator()->MaxAllocationSize();
583  DCHECK(max_allocatable <= JSArrayBuffer::kMaxByteLength);
584  // 2. If IsSharedArrayBuffer(arrayBuffer) is true, throw a TypeError
585  // exception.
586  CHECK_SHARED(false, array_buffer, method_name);
587 
588  size_t new_byte_length;
589  if (IsUndefined(*new_length, isolate)) {
590  // 3. If newLength is undefined, then
591  // a. Let newByteLength be arrayBuffer.[[ArrayBufferByteLength]].
592  new_byte_length = array_buffer->GetByteLength();
593  } else {
594  // 4. Else,
595  // a. Let newByteLength be ? ToIndex(newLength).
596  DirectHandle<Object> number_new_byte_length;
597  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, number_new_byte_length,
598  Object::ToInteger(isolate, new_length));
599  if (Object::NumberValue(*number_new_byte_length) < 0.0) {
600  THROW_NEW_ERROR_RETURN_FAILURE(
601  isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferLength));
602  }
603  if (!TryNumberToSize(*number_new_byte_length, &new_byte_length) ||
604  new_byte_length > max_allocatable) {
605  THROW_NEW_ERROR_RETURN_FAILURE(
606  isolate,
607  NewRangeError(
608  MessageTemplate::kInvalidArrayBufferResizeLength,
609  isolate->factory()->NewStringFromAsciiChecked(method_name)));
610  }
611  }
612 
613  // 5. If IsDetachedBuffer(arrayBuffer) is true, throw a TypeError exception.
614  if (array_buffer->was_detached()) {
615  THROW_NEW_ERROR_RETURN_FAILURE(
616  isolate, NewTypeError(MessageTemplate::kDetachedOperation,
617  isolate->factory()->NewStringFromAsciiChecked(
618  method_name)));
619  }
620 
621  ResizableFlag resizable;
622  size_t new_max_byte_length;
623  if (preserve_resizability == kPreserveResizability &&
624  array_buffer->is_resizable_by_js()) {
625  // 6. If preserveResizability is preserve-resizability and
626  // IsResizableArrayBuffer(arrayBuffer) is true, then
627  // a. Let newMaxByteLength be arrayBuffer.[[ArrayBufferMaxByteLength]].
628  new_max_byte_length = array_buffer->max_byte_length();
629  resizable = ResizableFlag::kResizable;
630  } else {
631  // 7. Else,
632  // a. Let newMaxByteLength be empty.
633  new_max_byte_length = new_byte_length;
634  resizable = ResizableFlag::kNotResizable;
635  }
636 
637  // 8. If arrayBuffer.[[ArrayBufferDetachKey]] is not undefined, throw a
638  // TypeError exception.
639 
640  if (!array_buffer->is_detachable()) {
641  THROW_NEW_ERROR_RETURN_FAILURE(
642  isolate,
643  NewTypeError(MessageTemplate::kDataCloneErrorNonDetachableArrayBuffer));
644  }
645 
646  // After this point the steps are not observable and are performed out of
647  // spec order.
648 
649  // Case 1: We don't need a BackingStore.
650  if (new_byte_length == 0) {
651  // 15. Perform ! DetachArrayBuffer(arrayBuffer).
652  JSArrayBuffer::Detach(array_buffer).Check();
653 
654  // 9. Let newBuffer be ? AllocateArrayBuffer(%ArrayBuffer%, newByteLength,
655  // newMaxByteLength).
656  //
657  // Nothing to do for steps 10-14.
658  //
659  // 16. Return newBuffer.
660  return *isolate->factory()
661  ->NewJSArrayBufferAndBackingStore(
662  0, new_max_byte_length, InitializedFlag::kUninitialized,
663  resizable)
664  .ToHandleChecked();
665  }
666 
667  // Case 2: We can reuse the same BackingStore.
668  auto from_backing_store = array_buffer->GetBackingStore();
669  if (from_backing_store && !from_backing_store->is_resizable_by_js() &&
670  resizable == ResizableFlag::kNotResizable &&
671  new_byte_length == array_buffer->GetByteLength()) {
672  // TODO(syg): Consider realloc when the default ArrayBuffer allocator's
673  // Reallocate does better than copy.
674  //
675  // See https://crbug.com/330575496#comment27
676 
677  // 15. Perform ! DetachArrayBuffer(arrayBuffer).
678  JSArrayBuffer::Detach(array_buffer).Check();
679 
680  // 9. Let newBuffer be ? AllocateArrayBuffer(%ArrayBuffer%, newByteLength,
681  // newMaxByteLength).
682  // 16. Return newBuffer.
683  return *isolate->factory()->NewJSArrayBuffer(std::move(from_backing_store));
684  }
685 
686  // Case 3: We can't reuse the same BackingStore. Copy the buffer.
687 
688  if (new_byte_length > new_max_byte_length) {
689  THROW_NEW_ERROR_RETURN_FAILURE(
690  isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferLength));
691  }
692 
693  // 9. Let newBuffer be ? AllocateArrayBuffer(%ArrayBuffer%, newByteLength,
694  // newMaxByteLength).
695  DirectHandle<JSArrayBuffer> new_buffer;
696  MaybeDirectHandle<JSArrayBuffer> result =
697  isolate->factory()->NewJSArrayBufferAndBackingStore(
698  new_byte_length, new_max_byte_length, InitializedFlag::kUninitialized,
699  resizable);
700  if (!result.ToHandle(&new_buffer)) {
701  THROW_NEW_ERROR_RETURN_FAILURE(
702  isolate, NewRangeError(MessageTemplate::kArrayBufferAllocationFailed));
703  }
704 
705  // 10. Let copyLength be min(newByteLength,
706  // arrayBuffer.[[ArrayBufferByteLength]]).
707  //
708  // (Size comparison is done manually below instead of using min.)
709 
710  // 11. Let fromBlock be arrayBuffer.[[ArrayBufferData]].
711  uint8_t* from_data =
712  reinterpret_cast<uint8_t*>(array_buffer->backing_store());
713 
714  // 12. Let toBlock be newBuffer.[[ArrayBufferData]].
715  uint8_t* to_data = reinterpret_cast<uint8_t*>(new_buffer->backing_store());
716 
717  // 13. Perform CopyDataBlockBytes(toBlock, 0, fromBlock, 0, copyLength).
718  // 14. NOTE: Neither creation of the new Data Block nor copying from the old
719  // Data Block are observable. Implementations reserve the right to
720  // implement this method as a zero-copy move or a realloc.
721  size_t from_byte_length = array_buffer->GetByteLength();
722  if (new_byte_length <= from_byte_length) {
723  CopyBytes(to_data, from_data, new_byte_length);
724  } else {
725  CopyBytes(to_data, from_data, from_byte_length);
726  memset(to_data + from_byte_length, 0, new_byte_length - from_byte_length);
727  }
728 
729  // 15. Perform ! DetachArrayBuffer(arrayBuffer).
730  JSArrayBuffer::Detach(array_buffer).Check();
731 
732  // 16. Return newBuffer.
733  return *new_buffer;
734 }
