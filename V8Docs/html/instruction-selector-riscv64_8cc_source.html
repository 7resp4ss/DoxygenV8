20 bool RiscvOperandGenerator::CanBeImmediate(int64_t value,
21  InstructionCode opcode) {
22  switch (ArchOpcodeField::decode(opcode)) {
23  case kRiscvShl32:
24  case kRiscvSar32:
25  case kRiscvShr32:
26  return is_uint5(value);
27  case kRiscvShl64:
28  case kRiscvSar64:
29  case kRiscvShr64:
30  return is_uint6(value);
31  case kRiscvAdd32:
32  case kRiscvAnd32:
33  case kRiscvAnd:
34  case kRiscvAdd64:
35  case kRiscvOr32:
36  case kRiscvOr:
37  case kRiscvTst64:
38  case kRiscvTst32:
39  case kRiscvXor:
40  return is_int12(value);
41  case kRiscvLb:
42  case kRiscvLbu:
43  case kRiscvSb:
44  case kRiscvLh:
45  case kRiscvLhu:
46  case kRiscvSh:
47  case kRiscvLw:
48  case kRiscvSw:
49  case kRiscvLd:
50  case kRiscvSd:
51  case kRiscvLoadFloat:
52  case kRiscvStoreFloat:
53  case kRiscvLoadDouble:
54  case kRiscvStoreDouble:
55  return is_int32(value);
56  default:
57  return is_int12(value);
58  }

61 struct ExtendingLoadMatcher {
62  ExtendingLoadMatcher(OpIndex node, InstructionSelector* selector)
63  : matches_(false), selector_(selector), immediate_(0) {
64  Initialize(node);
65  }
66 
67  bool Matches() const { return matches_; }
68 
69  OpIndex base() const {
70  DCHECK(Matches());
71  return base_;
72  }
73  int64_t immediate() const {
74  DCHECK(Matches());
75  return immediate_;
76  }
77  ArchOpcode opcode() const {
78  DCHECK(Matches());
79  return opcode_;
80  }
81 
82  private:
83  bool matches_;
84  InstructionSelector* selector_;
85  OpIndex base_{};
86  int64_t immediate_;
87  ArchOpcode opcode_;
88 
89  void Initialize(turboshaft::OpIndex node) {
90  const ShiftOp& shift = selector_->Get(node).template Cast<ShiftOp>();
91  DCHECK(shift.kind == ShiftOp::Kind::kShiftRightArithmetic ||
92  shift.kind == ShiftOp::Kind::kShiftRightArithmeticShiftOutZeros);
93  // When loading a 64-bit value and shifting by 32, we should
94  // just load and sign-extend the interesting 4 bytes instead.
95  // This happens, for example, when we're loading and untagging SMIs.
96  const Operation& lhs = selector_->Get(shift.left());
97  int64_t constant_rhs;
98 
99  if (lhs.Is<LoadOp>() &&
100  selector_->MatchIntegralWord64Constant(shift.right(), &constant_rhs) &&
101  constant_rhs == 32 && selector_->CanCover(node, shift.left())) {
102  RiscvOperandGenerator g(selector_);
103  const LoadOp& load = lhs.Cast<LoadOp>();
104  base_ = load.base();
105  opcode_ = kRiscvLw;
106  if (load.index().has_value()) {
107  int64_t index_constant;
108  if (selector_->MatchIntegralWord64Constant(load.index().value(),
109  &index_constant)) {
110  DCHECK_EQ(load.element_size_log2, 0);
111  immediate_ = index_constant + 4;
112  matches_ = g.CanBeImmediate(immediate_, kRiscvLw);
113  }
114  } else {
115  immediate_ = load.offset + 4;
116  matches_ = g.CanBeImmediate(immediate_, kRiscvLw);
117  }
118  }
119  }
120 };

122 bool TryEmitExtendingLoad(InstructionSelector* selector, OpIndex node,
123  OpIndex output_node) {
124  ExtendingLoadMatcher m(node, selector);
125  RiscvOperandGenerator g(selector);
126  if (m.Matches()) {
127  InstructionOperand inputs[2];
128  inputs[0] = g.UseRegister(m.base());
129  InstructionCode opcode =
130  m.opcode() | AddressingModeField::encode(kMode_MRI);
131  DCHECK(is_int32(m.immediate()));
132  inputs[1] = g.TempImmediate(static_cast<int32_t>(m.immediate()));
133  InstructionOperand outputs[] = {g.DefineAsRegister(output_node)};
134  selector->Emit(opcode, arraysize(outputs), outputs, arraysize(inputs),
135  inputs);
136  return true;
137  }
138  return false;
139 }

141 void EmitLoad(InstructionSelector* selector, OpIndex node,
142  InstructionCode opcode, OpIndex output = OpIndex()) {
143  RiscvOperandGenerator g(selector);
144 
145  const Operation& op = selector->Get(node);
146  const LoadOp& load = op.Cast<LoadOp>();
147 
148  // The LoadStoreSimplificationReducer transforms all loads into
149  // *(base + index).
150  OpIndex base = load.base();
151  OpIndex index = load.index().value();
152  DCHECK_EQ(load.offset, 0);
153  DCHECK_EQ(load.element_size_log2, 0);
154 
155  InstructionOperand inputs[3];
156  size_t input_count = 0;
157  InstructionOperand output_op;
158 
159  // If output is valid, use that as the output register. This is used when we
160  // merge a conversion into the load.
161  output_op = g.DefineAsRegister(output.valid() ? output : node);
162  const Operation& base_op = selector->Get(base);
163  int64_t index_constant;
164  const bool is_index_constant =
165  selector->MatchSignedIntegralConstant(index, &index_constant);
166  if (base_op.Is<Opmask::kExternalConstant>() && is_index_constant) {
167  const ConstantOp& constant_base = base_op.Cast<ConstantOp>();
168  if (selector->CanAddressRelativeToRootsRegister(
169  constant_base.external_reference())) {
170  ptrdiff_t const delta =
171  index_constant +
172  MacroAssemblerBase::RootRegisterOffsetForExternalReference(
173  selector->isolate(), constant_base.external_reference());
174  input_count = 1;
175  // Check that the delta is a 32-bit integer due to the limitations of
176  // immediate operands.
177  if (is_int32(delta)) {
178  inputs[0] = g.UseImmediate(static_cast<int32_t>(delta));
179  opcode |= AddressingModeField::encode(kMode_Root);
180  selector->Emit(opcode, 1, &output_op, input_count, inputs);
181  return;
182  }
183  }
184  }
185 
186  if (base_op.Is<LoadRootRegisterOp>() && is_index_constant) {
187  DCHECK(is_index_constant);
188  input_count = 1;
189  inputs[0] = g.UseImmediate64(index_constant);
190  opcode |= AddressingModeField::encode(kMode_Root);
191  selector->Emit(opcode, 1, &output_op, input_count, inputs);
192  return;
193  }
194 
195  if (g.CanBeImmediate(index, opcode)) {
196  selector->Emit(opcode | AddressingModeField::encode(kMode_MRI),
197  g.DefineAsRegister(output.valid() ? output : node),
198  g.UseRegister(base), g.UseImmediate(index));
199  } else {
200  InstructionOperand addr_reg = g.TempRegister();
201  selector->Emit(kRiscvAdd64 | AddressingModeField::encode(kMode_None),
202  addr_reg, g.UseRegister(index), g.UseRegister(base));
203  // Emit desired load opcode, using temp addr_reg.
204  selector->Emit(opcode | AddressingModeField::encode(kMode_MRI),
205  g.DefineAsRegister(output.valid() ? output : node), addr_reg,
206  g.TempImmediate(0));
207  }
208 }

210 void EmitS128Load(InstructionSelector* selector, OpIndex node,
211  InstructionCode opcode, VSew sew, Vlmul lmul) {
212  RiscvOperandGenerator g(selector);
213  const Operation& op = selector->Get(node);
214  DCHECK_EQ(op.input_count, 2);
215  OpIndex base = op.input(0);
216  OpIndex index = op.input(1);
217  if (g.CanBeImmediate(index, opcode)) {
218  selector->Emit(opcode | AddressingModeField::encode(kMode_MRI),
219  g.DefineAsRegister(node), g.UseRegister(base),
220  g.UseImmediate(index), g.UseImmediate(sew),
221  g.UseImmediate(lmul));
222  } else {
223  InstructionOperand addr_reg = g.TempRegister();
224  selector->Emit(kRiscvAdd64 | AddressingModeField::encode(kMode_None),
225  addr_reg, g.UseRegister(index), g.UseRegister(base));
226  // Emit desired load opcode, using temp addr_reg.
227  selector->Emit(opcode | AddressingModeField::encode(kMode_MRI),
228  g.DefineAsRegister(node), addr_reg, g.TempImmediate(0),
229  g.UseImmediate(sew), g.UseImmediate(lmul));
230  }

233 void InstructionSelector::VisitStoreLane(OpIndex node) {
234  const Simd128LaneMemoryOp& store = Get(node).Cast<Simd128LaneMemoryOp>();
235  InstructionCode opcode = kRiscvS128StoreLane;
236  opcode |= LaneSizeField::encode(store.lane_size() * kBitsPerByte);
237  if (store.kind.with_trap_handler) {
238  opcode |= AccessModeField::encode(kMemoryAccessProtectedMemOutOfBounds);
239  }
240 
241  RiscvOperandGenerator g(this);
242  const Operation& op = this->Get(node);
243  DCHECK_EQ(op.input_count, 3);
244  OpIndex base = op.input(0);
245  OpIndex index = op.input(1);
246  InstructionOperand addr_reg = g.TempRegister();
247  Emit(kRiscvAdd64, addr_reg, g.UseRegister(base), g.UseRegister(index));
248  InstructionOperand inputs[4] = {
249  g.UseRegister(op.input(2)),
250  g.UseImmediate(store.lane),
251  addr_reg,
252  g.TempImmediate(0),
253  };
254  opcode |= AddressingModeField::encode(kMode_MRI);
255  Emit(opcode, 0, nullptr, 4, inputs);
256 }

279 ArchOpcode GetLoadOpcode(MemoryRepresentation loaded_rep,
280  RegisterRepresentation result_rep) {
281  // NOTE: The meaning of `loaded_rep` = `MemoryRepresentation::AnyTagged()` is
282  // we are loading a compressed tagged field, while `result_rep` =
283  // `RegisterRepresentation::Tagged()` refers to an uncompressed tagged value.
284 
285  switch (loaded_rep) {
286  case MemoryRepresentation::Int8():
287  DCHECK_EQ(result_rep, RegisterRepresentation::Word32());
288  return kRiscvLb;
289  case MemoryRepresentation::Uint8():
290  DCHECK_EQ(result_rep, RegisterRepresentation::Word32());
291  return kRiscvLbu;
292  case MemoryRepresentation::Int16():
293  DCHECK_EQ(result_rep, RegisterRepresentation::Word32());
294  return kRiscvLh;
295  case MemoryRepresentation::Uint16():
296  DCHECK_EQ(result_rep, RegisterRepresentation::Word32());
297  return kRiscvLhu;
298  case MemoryRepresentation::Int32():
299  DCHECK_EQ(result_rep, RegisterRepresentation::Word32());
300  return kRiscvLw;
301  case MemoryRepresentation::Uint32():
302  DCHECK_EQ(result_rep, RegisterRepresentation::Word32());
303  return kRiscvLwu;
304  case MemoryRepresentation::Int64():
305  case MemoryRepresentation::Uint64():
306  DCHECK_EQ(result_rep, RegisterRepresentation::Word64());
307  return kRiscvLd;
308  case MemoryRepresentation::Float16():
309  UNIMPLEMENTED();
310  case MemoryRepresentation::Float32():
311  DCHECK_EQ(result_rep, RegisterRepresentation::Float32());
312  return kRiscvLoadFloat;
313  case MemoryRepresentation::Float64():
314  DCHECK_EQ(result_rep, RegisterRepresentation::Float64());
315  return kRiscvLoadDouble;
316 #ifdef V8_COMPRESS_POINTERS
317  case MemoryRepresentation::AnyTagged():
318  case MemoryRepresentation::TaggedPointer():
319  if (result_rep == RegisterRepresentation::Compressed()) {
320  return kRiscvLwu;
321  }
322  DCHECK_EQ(result_rep, RegisterRepresentation::Tagged());
323  return kRiscvLoadDecompressTagged;
324  case MemoryRepresentation::TaggedSigned():
325  if (result_rep == RegisterRepresentation::Compressed()) {
326  return kRiscvLwu;
327  }
328  DCHECK_EQ(result_rep, RegisterRepresentation::Tagged());
329  return kRiscvLoadDecompressTaggedSigned;
330 #else
331  case MemoryRepresentation::AnyTagged():
332  case MemoryRepresentation::TaggedPointer():
333  case MemoryRepresentation::TaggedSigned():
334  DCHECK_EQ(result_rep, RegisterRepresentation::Tagged());
335  return kRiscvLd;
336 #endif
337  case MemoryRepresentation::AnyUncompressedTagged():
338  case MemoryRepresentation::UncompressedTaggedPointer():
339  case MemoryRepresentation::UncompressedTaggedSigned():
340  DCHECK_EQ(result_rep, RegisterRepresentation::Tagged());
341  return kRiscvLd;
342  case MemoryRepresentation::ProtectedPointer():
343  CHECK(V8_ENABLE_SANDBOX_BOOL);
344  return kRiscvLoadDecompressProtected;
345  case MemoryRepresentation::IndirectPointer():
346  UNREACHABLE();
347  case MemoryRepresentation::SandboxedPointer():
348  return kRiscvLoadDecodeSandboxedPointer;
349  case MemoryRepresentation::Simd128():
350  return kRiscvRvvLd;
351  case MemoryRepresentation::Simd256():
352  UNREACHABLE();
353  }
354 }

399 void InstructionSelector::VisitLoad(OpIndex node) {
400  auto load = load_view(node);
401  InstructionCode opcode = kArchNop;
402  opcode = GetLoadOpcode(load.ts_loaded_rep(), load.ts_result_rep());
403  bool traps_on_null;
404  if (load.is_protected(&traps_on_null)) {
405  if (traps_on_null) {
406  opcode |= AccessModeField::encode(kMemoryAccessProtectedNullDereference);
407  } else {
408  opcode |= AccessModeField::encode(kMemoryAccessProtectedMemOutOfBounds);
409  }
410  }
411  EmitLoad(this, node, opcode);
412 }

418 void InstructionSelector::VisitStore(OpIndex node) {
419  RiscvOperandGenerator g(this);
420  StoreView store_view = this->store_view(node);
421  DCHECK_EQ(store_view.displacement(), 0);
422  OpIndex base = store_view.base();
423  OpIndex index = store_view.index().value();
424  OpIndex value = store_view.value();
425 
426  WriteBarrierKind write_barrier_kind =
427  store_view.stored_rep().write_barrier_kind();
428  const MachineRepresentation rep = store_view.stored_rep().representation();
429 
430  // TODO(riscv): I guess this could be done in a better way.
431  if (write_barrier_kind != kNoWriteBarrier &&
432  V8_LIKELY(!v8_flags.disable_write_barriers)) {
433  DCHECK(CanBeTaggedOrCompressedOrIndirectPointer(rep));
434  InstructionOperand inputs[4];
435  size_t input_count = 0;
436  inputs[input_count++] = g.UseUniqueRegister(base);
437  inputs[input_count++] = g.UseUniqueRegister(index);
438  inputs[input_count++] = g.UseUniqueRegister(value);
439  RecordWriteMode record_write_mode =
440  WriteBarrierKindToRecordWriteMode(write_barrier_kind);
441  InstructionOperand temps[] = {g.TempRegister(), g.TempRegister()};
442  size_t const temp_count = arraysize(temps);
443  InstructionCode code;
444  if (rep == MachineRepresentation::kIndirectPointer) {
445  DCHECK_EQ(write_barrier_kind, kIndirectPointerWriteBarrier);
446  // In this case we need to add the IndirectPointerTag as additional input.
447  code = kArchStoreIndirectWithWriteBarrier;
448  IndirectPointerTag tag = store_view.indirect_pointer_tag();
449  inputs[input_count++] = g.UseImmediate64(static_cast<int64_t>(tag));
450  } else {
451  code = kArchStoreWithWriteBarrier;
452  }
453  code |= RecordWriteModeField::encode(record_write_mode);
454  if (store_view.is_store_trap_on_null()) {
455  code |= AccessModeField::encode(kMemoryAccessProtectedNullDereference);
456  }
457  Emit(code, 0, nullptr, input_count, inputs, temp_count, temps);
458  return;
459  }
460 
461  InstructionCode code;
462  code = GetStoreOpcode(store_view.ts_stored_rep());
463 
464  if (Is<LoadRootRegisterOp>(base)) {
465  Emit(code | AddressingModeField::encode(kMode_Root), g.NoOutput(),
466  g.UseRegisterOrImmediateZero(value), g.UseImmediate(index));
467  return;
468  }
469 
470  if (store_view.is_store_trap_on_null()) {
471  code |= AccessModeField::encode(kMemoryAccessProtectedNullDereference);
472  } else if (store_view.access_kind() ==
473  MemoryAccessKind::kProtectedByTrapHandler) {
474  code |= AccessModeField::encode(kMemoryAccessProtectedMemOutOfBounds);
475  }
476 
477  if (g.CanBeImmediate(index, code)) {
478  Emit(code | AddressingModeField::encode(kMode_MRI), g.NoOutput(),
479  g.UseRegisterOrImmediateZero(value), g.UseRegister(base),
480  g.UseImmediate(index));
481  } else {
482  InstructionOperand addr_reg = g.TempRegister();
483  Emit(kRiscvAdd64 | AddressingModeField::encode(kMode_None), addr_reg,
484  g.UseRegister(index), g.UseRegister(base));
485  // Emit desired store opcode, using temp addr_reg.
486  Emit(code | AddressingModeField::encode(kMode_MRI), g.NoOutput(),
487  g.UseRegisterOrImmediateZero(value), addr_reg, g.TempImmediate(0));
488  }
489 }

519 void InstructionSelector::VisitWord64Shl(OpIndex node) {
520  const ShiftOp& shift_op = this->Get(node).template Cast<ShiftOp>();
521  const Operation& lhs = this->Get(shift_op.left());
522  const Operation& rhs = this->Get(shift_op.right());
523  if ((lhs.Is<Opmask::kChangeInt32ToInt64>() ||
524  lhs.Is<Opmask::kChangeUint32ToUint64>()) &&
525  rhs.Is<Opmask::kWord32Constant>()) {
526  int64_t shift_by = rhs.Cast<ConstantOp>().signed_integral();
527  if (base::IsInRange(shift_by, 32, 63) && CanCover(node, shift_op.left())) {
528  RiscvOperandGenerator g(this);
529  // There's no need to sign/zero-extend to 64-bit if we shift out the
530  // upper 32 bits anyway.
531  Emit(kRiscvShl64, g.DefineSameAsFirst(node),
532  g.UseRegister(lhs.Cast<ChangeOp>().input()),
533  g.UseImmediate64(shift_by));
534  return;
535  }
536  }
537  VisitRRO(this, kRiscvShl64, node);
538 }

544 void InstructionSelector::VisitWord64Sar(OpIndex node) {
545  if (TryEmitExtendingLoad(this, node, node)) return;
546  // Select Sbfx(x, imm, 32-imm) for Word64Sar(ChangeInt32ToInt64(x), imm)
547  // where possible
548  const ShiftOp& shiftop = Get(node).Cast<ShiftOp>();
549  const Operation& lhs = Get(shiftop.left());
550 
551  int64_t constant_rhs;
552  if (lhs.Is<Opmask::kChangeInt32ToInt64>() &&
553  MatchIntegralWord64Constant(shiftop.right(), &constant_rhs) &&
554  is_uint5(constant_rhs) && CanCover(node, shiftop.left())) {
555  // Don't select Sbfx here if Asr(Ldrsw(x), imm) can be selected for
556  // Word64Sar(ChangeInt32ToInt64(Load(x)), imm)
557  OpIndex input = lhs.Cast<ChangeOp>().input();
558  if (!Get(input).Is<LoadOp>() || !CanCover(shiftop.left(), input)) {
559  RiscvOperandGenerator g(this);
560  int right = static_cast<int>(constant_rhs);
561  Emit(kRiscvSar32, g.DefineAsRegister(node), g.UseRegister(input),
562  g.UseImmediate(right));
563  return;
564  }
565  }
566  VisitRRO(this, kRiscvSar64, node);
567 }

796 void InstructionSelector::VisitChangeFloat64ToInt32(OpIndex node) {
797  RiscvOperandGenerator g(this);
798  auto value = this->Get(node).input(0);
799  if (CanCover(node, value)) {
800  const Operation& op = this->Get(value);
801  if (const FloatUnaryOp* load = op.TryCast<FloatUnaryOp>()) {
802  DCHECK(load->rep == FloatRepresentation::Float64());
803  switch (load->kind) {
804  case FloatUnaryOp::Kind::kRoundDown:
805  Emit(kRiscvFloorWD, g.DefineAsRegister(node),
806  g.UseRegister(op.input(0)));
807  return;
808  case FloatUnaryOp::Kind::kRoundUp:
809  Emit(kRiscvCeilWD, g.DefineAsRegister(node),
810  g.UseRegister(op.input(0)));
811  return;
812  case FloatUnaryOp::Kind::kRoundToZero:
813  Emit(kRiscvTruncWD, g.DefineAsRegister(node),
814  g.UseRegister(op.input(0)));
815  return;
816  case FloatUnaryOp::Kind::kRoundTiesEven:
817  Emit(kRiscvRoundWD, g.DefineAsRegister(node),
818  g.UseRegister(op.input(0)));
819  return;
820  default:
821  break;
822  }
823  }
824  if (op.Is<ChangeOp>()) {
825  const ChangeOp& change = op.Cast<ChangeOp>();
826  using Rep = turboshaft::RegisterRepresentation;
827  if (change.from == Rep::Float32() && change.to == Rep::Float64()) {
828  auto next = this->Get(value).input(0);
829  if (CanCover(value, next)) {
830  const Operation& next_op = this->Get(next);
831  if (const FloatUnaryOp* round = next_op.TryCast<FloatUnaryOp>()) {
832  DCHECK(round->rep == FloatRepresentation::Float32());
833  switch (round->kind) {
834  case FloatUnaryOp::Kind::kRoundDown:
835  Emit(kRiscvFloorWS, g.DefineAsRegister(node),
836  g.UseRegister(next_op.input(0)));
837  return;
838  case FloatUnaryOp::Kind::kRoundUp:
839  Emit(kRiscvCeilWS, g.DefineAsRegister(node),
840  g.UseRegister(next_op.input(0)));
841  return;
842  case FloatUnaryOp::Kind::kRoundToZero:
843  Emit(kRiscvTruncWS, g.DefineAsRegister(node),
844  g.UseRegister(next_op.input(0)));
845  return;
846  case FloatUnaryOp::Kind::kRoundTiesEven:
847  Emit(kRiscvRoundWS, g.DefineAsRegister(node),
848  g.UseRegister(next_op.input(0)));
849  return;

851  break;

853  }

856  Emit(kRiscvTruncWS, g.DefineAsRegister(node),

1500  AtomicWidth width) {
1501  using OpIndex = OpIndex;
1502  RiscvOperandGenerator g(selector);
1503  auto load = selector->load_view(node);
1504  OpIndex base = load.base();
1505  OpIndex index = load.index();
1506 
1507  // The memory order is ignored as both acquire and sequentially consistent
1508  // loads can emit LDAR.
1509  // https://www.cl.cam.ac.uk/~pes20/cpp/cpp0xmappings.html
1510  LoadRepresentation load_rep = load.loaded_rep();
1511  InstructionCode code;
1512  switch (load_rep.representation()) {
1513  case MachineRepresentation::kWord8:
1514  DCHECK_IMPLIES(load_rep.IsSigned(), width == AtomicWidth::kWord32);
1515  code = load_rep.IsSigned() ? kAtomicLoadInt8 : kAtomicLoadUint8;
1516  break;
1517  case MachineRepresentation::kWord16:
1518  DCHECK_IMPLIES(load_rep.IsSigned(), width == AtomicWidth::kWord32);
1519  code = load_rep.IsSigned() ? kAtomicLoadInt16 : kAtomicLoadUint16;
1520  break;
1521  case MachineRepresentation::kWord32:
1522  code = kAtomicLoadWord32;
1523  break;
1524  case MachineRepresentation::kWord64:
1525  code = kRiscvWord64AtomicLoadUint64;
1526  break;
1527 #ifdef V8_COMPRESS_POINTERS
1528  case MachineRepresentation::kTaggedSigned:
1529  code = kRiscvAtomicLoadDecompressTaggedSigned;
1530  break;
1531  case MachineRepresentation::kTaggedPointer:
1532  case MachineRepresentation::kTagged:
1533  code = kRiscvAtomicLoadDecompressTagged;
1534  break;
1535 #else
1536  case MachineRepresentation::kTaggedSigned: // Fall through.
1537  case MachineRepresentation::kTaggedPointer: // Fall through.
1538  case MachineRepresentation::kTagged:
1539  if (kTaggedSize == 8) {
1540  code = kRiscvWord64AtomicLoadUint64;
1541  } else {
1542  code = kAtomicLoadWord32;
1543  }
1544  break;
1545 #endif
1546  case MachineRepresentation::kCompressedPointer: // Fall through.
1547  case MachineRepresentation::kCompressed:
1548  DCHECK(COMPRESS_POINTERS_BOOL);
1549  code = kAtomicLoadWord32;

1559  // probably be kMemoryAccessProtectedNullDereference.

1561  code |= AccessModeField::encode(kMemoryAccessProtectedMemOutOfBounds);
1562  }

1565  selector->Emit(code | AddressingModeField::encode(kMode_MRI) |
1566  AtomicWidthField::encode(width),

1568  g.UseImmediate(index));

1572  addr_reg, g.UseRegister(base), g.UseRegister(index));

1762 void InstructionSelector::VisitWordCompareZero(OpIndex user, OpIndex value,
1763  FlagsContinuation* cont) {
1764  // Try to combine with comparisons against 0 by simply inverting the branch.
1765  while (const ComparisonOp* equal =
1766  this->TryCast<Opmask::kWord32Equal>(value)) {
1767  if (!CanCover(user, value)) break;
1768  if (!MatchIntegralZero(equal->right())) break;
1769 
1770  user = value;
1771  value = equal->left();
1772  cont->Negate();
1773  }
1774 
1775  const Operation& value_op = Get(value);
1776  if (CanCover(user, value)) {
1777  if (const ComparisonOp* comparison = value_op.TryCast<ComparisonOp>()) {
1778  switch (comparison->rep.MapTaggedToWord().value()) {
1779  case RegisterRepresentation::Word32():
1780  cont->OverwriteAndNegateIfEqual(
1781  GetComparisonFlagCondition(*comparison));
1782  return VisitWord32Compare(this, value, cont);
1783 
1784  case RegisterRepresentation::Word64():
1785  cont->OverwriteAndNegateIfEqual(
1786  GetComparisonFlagCondition(*comparison));
1787  return VisitWord64Compare(this, value, cont);
1788 
1789  case RegisterRepresentation::Float32():
1790  switch (comparison->kind) {
1791  case ComparisonOp::Kind::kEqual:
1792  cont->OverwriteAndNegateIfEqual(kEqual);
1793  return VisitFloat32Compare(this, value, cont);
1794  case ComparisonOp::Kind::kSignedLessThan:
1795  cont->OverwriteAndNegateIfEqual(kFloatLessThan);
1796  return VisitFloat32Compare(this, value, cont);
1797  case ComparisonOp::Kind::kSignedLessThanOrEqual:
1798  cont->OverwriteAndNegateIfEqual(kFloatLessThanOrEqual);
1799  return VisitFloat32Compare(this, value, cont);
1800  default:
1801  UNREACHABLE();
1802  }
1803  case RegisterRepresentation::Float64():
1804  switch (comparison->kind) {
1805  case ComparisonOp::Kind::kEqual:
1806  cont->OverwriteAndNegateIfEqual(kEqual);
1807  return VisitFloat64Compare(this, value, cont);
1808  case ComparisonOp::Kind::kSignedLessThan:
1809  cont->OverwriteAndNegateIfEqual(kFloatLessThan);
1810  return VisitFloat64Compare(this, value, cont);
1811  case ComparisonOp::Kind::kSignedLessThanOrEqual:
1812  cont->OverwriteAndNegateIfEqual(kFloatLessThanOrEqual);
1813  return VisitFloat64Compare(this, value, cont);
1814  default:
1815  UNREACHABLE();
1816  }
1817  default:
1818  break;
1819  }
1820  } else if (const ProjectionOp* projection =
1821  value_op.TryCast<ProjectionOp>()) {
1822  // Check if this is the overflow output projection of an
1823  // <Operation>WithOverflow node.
1824  if (projection->index == 1u) {
1825  // We cannot combine the <Operation>WithOverflow with this branch
1826  // unless the 0th projection (the use of the actual value of the
1827  // <Operation> is either nullptr, which means there's no use of the
1828  // actual value, or was already defined, which means it is scheduled
1829  // *AFTER* this branch).
1830  OpIndex node = projection->input();
1831  if (const OverflowCheckedBinopOp* binop =
1832  TryCast<OverflowCheckedBinopOp>(node);
1833  binop && CanDoBranchIfOverflowFusion(node)) {
1834  const bool is64 = binop->rep == WordRepresentation::Word64();
1835  switch (binop->kind) {
1836  case OverflowCheckedBinopOp::Kind::kSignedAdd:
1837  cont->OverwriteAndNegateIfEqual(kOverflow);
1838  if (is64) {
1839  return VisitBinop<Int32BinopMatcher>(this, node,
1840  kRiscvAddOvfWord, cont);
1841  } else { // If enable COMPRESS_POINTERS, smi will zero extend to
1842  // 64 bit, kRiscvAdd64 can't process smi overflow.
1843  return VisitBinop<Int32BinopMatcher>(
1844  this, node,
1845  COMPRESS_POINTERS_BOOL ? kRiscvAddOvf32 : kRiscvAdd64,
1846  cont);
1847  }
1848  case OverflowCheckedBinopOp::Kind::kSignedSub:
1849  cont->OverwriteAndNegateIfEqual(kOverflow);
1850  if (is64) {
1851  return VisitBinop<Int32BinopMatcher>(this, node,
1852  kRiscvSubOvfWord, cont);
1853  } else { // If enable COMPRESS_POINTERS, smi will zero extend to
1854  // 64 bit, kRiscvSub64 can't process smi overflow.
1855  return VisitBinop<Int32BinopMatcher>(
1856  this, node,
1857  COMPRESS_POINTERS_BOOL ? kRiscvSubOvf32 : kRiscvSub64,
1858  cont);
1859  }
1860  case OverflowCheckedBinopOp::Kind::kSignedMul:
1861  cont->OverwriteAndNegateIfEqual(kOverflow);
1862  return VisitBinop<Int32BinopMatcher>(
1863  this, node, is64 ? kRiscvMulOvf64 : kRiscvMulOvf32, cont);
1864  }
1865  }
1866  }
1867  }
1868  }
1869 
1870  // Continuation could not be combined with a compare, emit compare against
1871  // 0.
1872  const ComparisonOp* comparison = this->Get(user).TryCast<ComparisonOp>();
1873 #ifdef V8_COMPRESS_POINTERS
1874  if ((comparison &&
1875  comparison->rep.value() == RegisterRepresentation::Word64()) ||
1876  IsWord32Binop(value_op)) {
1877  return EmitWordCompareZero(this, value, cont);
1878  } else {
1879  return EmitWord32CompareZero(this, value, cont);
1880  }
1881 #else
1882  if (comparison &&
1883  comparison->rep.value() == RegisterRepresentation::Word32()) {
1884  return EmitWord32CompareZero(this, value, cont);
1885  } else {
1886  return EmitWordCompareZero(this, value, cont);
1887  }
1888 #endif
1889 }
