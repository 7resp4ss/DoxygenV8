468  bool IsUnalignedSupported(base::EnumSet<MachineRepresentation> unsupported,
469  MachineRepresentation rep) const {
470  // All accesses of bytes in memory are aligned.
471  DCHECK_NE(MachineRepresentation::kWord8, rep);
472  switch (unalignedSupport_) {
473  case kFullSupport:
474  return true;
475  case kNoSupport:
476  return false;
477  case kSomeSupport:
478  return !unsupported.contains(rep);
479  }
480  UNREACHABLE();
481  }
