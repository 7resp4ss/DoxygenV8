306 // Wrapper around heap refs which works roughly like a std::optional, but
307 // doesn't use extra storage for a boolean, but instead uses a null data pointer
308 // as a sentinel no value.
309 template <typename TRef>
310 class OptionalRef {
311  public:
312  // {ArrowOperatorHelper} is returned by {OptionalRef::operator->}. It should
313  // never be stored anywhere or used in any other code; no one should ever have
314  // to spell out {ArrowOperatorHelper} in code. Its only purpose is to be
315  // dereferenced immediately by "operator-> chaining". Returning the address of
316  // the field is valid because this objects lifetime only ends at the end of
317  // the full statement.
318  class ArrowOperatorHelper {
319  public:
320  TRef* operator->() { return &object_; }
321 
322  private:
323  friend class OptionalRef<TRef>;
324  explicit ArrowOperatorHelper(TRef object) : object_(object) {}
325 
326  TRef object_;
327  };
328 
329  OptionalRef() = default;
330  // NOLINTNEXTLINE
331  OptionalRef(std::nullopt_t) : OptionalRef() {}
332 
333  // Allow implicit upcasting from OptionalRefs with compatible refs.
334  template <typename SRef>
335  // NOLINTNEXTLINE
336  V8_INLINE OptionalRef(OptionalRef<SRef> ref)
337  requires std::is_convertible_v<SRef*, TRef*>
338  : data_(ref.data_) {}
339 
340  // Allow implicit upcasting from compatible refs.
341  template <typename SRef>
342  // NOLINTNEXTLINE
343  V8_INLINE OptionalRef(SRef ref)
344  requires std::is_convertible_v<SRef*, TRef*>
345  : data_(ref.data_) {}
346 
347  constexpr bool has_value() const { return data_ != nullptr; }
348  constexpr explicit operator bool() const { return has_value(); }
349 
350  TRef value() const {
351  DCHECK(has_value());
352  return TRef(data_, false);
353  }
354  TRef operator*() const { return value(); }
355  ArrowOperatorHelper operator->() const {
356  return ArrowOperatorHelper(value());
357  }
358 
359  bool equals(OptionalRef other) const { return data_ == other.data_; }
360 
361  size_t hash_value() const {
362  return has_value() ? value().hash_value() : base::hash_value(0);
363  }
364 
365  private:
366  explicit OptionalRef(ObjectData* data) : data_(data) {
367  CHECK_NOT_NULL(data_);
368  }
369  ObjectData* data_ = nullptr;
370 
371  template <typename SRef>
372  friend class OptionalRef;
373 };
