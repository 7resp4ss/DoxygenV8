173 /**
174  * This class serves as a base class for recording event-based metrics in V8.
175  * There a two kinds of metrics, those which are expected to be thread-safe and
176  * whose implementation is required to fulfill this requirement and those whose
177  * implementation does not have that requirement and only needs to be
178  * executable on the main thread. If such an event is triggered from a
179  * background thread, it will be delayed and executed by the foreground task
180  * runner.
181  *
182  * The embedder is expected to call v8::Isolate::SetMetricsRecorder()
183  * providing its implementation and have the virtual methods overwritten
184  * for the events it cares about.
185  */
186 class V8_EXPORT Recorder {
187  public:
188  // A unique identifier for a context in this Isolate.
189  // It is guaranteed to not be reused throughout the lifetime of the Isolate.
190  class ContextId {
191  public:
192  ContextId() : id_(kEmptyId) {}
193 
194  bool IsEmpty() const { return id_ == kEmptyId; }
195  static const ContextId Empty() { return ContextId{kEmptyId}; }
196 
197  bool operator==(const ContextId& other) const { return id_ == other.id_; }
198  bool operator!=(const ContextId& other) const { return id_ != other.id_; }
199 
200  private:
201  friend class ::v8::Context;
202  friend class ::v8::internal::Isolate;
203 
204  explicit ContextId(uintptr_t id) : id_(id) {}
205 
206  static constexpr uintptr_t kEmptyId = 0;
207  uintptr_t id_;
208  };
209 
210  virtual ~Recorder() = default;
211 
212  // Main thread events. Those are only triggered on the main thread, and hence
213  // can access the context.
214 #define ADD_MAIN_THREAD_EVENT(E) \
215  virtual void AddMainThreadEvent(const E&, ContextId) {}
216  ADD_MAIN_THREAD_EVENT(GarbageCollectionFullCycle)
217  ADD_MAIN_THREAD_EVENT(GarbageCollectionFullMainThreadIncrementalMark)
218  ADD_MAIN_THREAD_EVENT(GarbageCollectionFullMainThreadBatchedIncrementalMark)
219  ADD_MAIN_THREAD_EVENT(GarbageCollectionFullMainThreadIncrementalSweep)
220  ADD_MAIN_THREAD_EVENT(GarbageCollectionFullMainThreadBatchedIncrementalSweep)
221  ADD_MAIN_THREAD_EVENT(GarbageCollectionYoungCycle)
222  ADD_MAIN_THREAD_EVENT(WasmModuleDecoded)
223  ADD_MAIN_THREAD_EVENT(WasmModuleCompiled)
224  ADD_MAIN_THREAD_EVENT(WasmModuleInstantiated)
225 #undef ADD_MAIN_THREAD_EVENT
226 
227  // Thread-safe events are not allowed to access the context and therefore do
228  // not carry a context ID with them. These IDs can be generated using
229  // Recorder::GetContextId() and the ID will be valid throughout the lifetime
230  // of the isolate. It is not guaranteed that the ID will still resolve to
231  // a valid context using Recorder::GetContext() at the time the metric is
232  // recorded. In this case, an empty handle will be returned.
233 #define ADD_THREAD_SAFE_EVENT(E) \
234  virtual void AddThreadSafeEvent(const E&) {}
235  ADD_THREAD_SAFE_EVENT(WasmModulesPerIsolate)
236 #undef ADD_THREAD_SAFE_EVENT
237 
238  virtual void NotifyIsolateDisposal() {}
239 
240  // Return the context with the given id or an empty handle if the context
241  // was already garbage collected.
242  static MaybeLocal<Context> GetContext(Isolate* isolate, ContextId id);
243  // Return the unique id corresponding to the given context.
244  static ContextId GetContextId(Local<Context> context);
245 };

247 /**
248  * Experimental API intended for the LongTasks UKM (crbug.com/1173527).
249  * The Reset() method should be called at the start of a potential
250  * long task. The Get() method returns durations of V8 work that
251  * happened during the task.
252  *
253  * This API is experimental and may be removed/changed in the future.
254  */
255 struct V8_EXPORT LongTaskStats {
256  /**
257  * Resets durations of V8 work for the new task.
258  */
259  V8_INLINE static void Reset(Isolate* isolate) {
260  v8::internal::Internals::IncrementLongTasksStatsCounter(isolate);
261  }
262 
263  /**
264  * Returns durations of V8 work that happened since the last Reset().
265  */
266  static LongTaskStats Get(Isolate* isolate);
267

270  int64_t gc_young_wall_clock_duration_us = 0;
