97 template <typename DerivedDescriptor>
98 void StaticCallInterfaceDescriptor<DerivedDescriptor>::Initialize(
99  CallInterfaceDescriptorData* data) {
100  // Static local copy of the Registers array, for platform-specific
101  // initialization
102  static constexpr auto registers = DerivedDescriptor::registers();
103  static constexpr auto double_registers =
104  DerivedDescriptor::double_registers();
105  static constexpr auto return_registers =
106  DerivedDescriptor::return_registers();
107  static constexpr auto return_double_registers =
108  DerivedDescriptor::return_double_registers();
109 
110  // The passed pointer should be a modifiable pointer to our own data.
111  DCHECK_EQ(data, this->data());
112  DCHECK(!data->IsInitialized());
113 
114  if (DerivedDescriptor::kRestrictAllocatableRegisters) {
115  data->RestrictAllocatableRegisters(registers.data(), registers.size());
116  } else {
117  DCHECK(!DerivedDescriptor::kCalleeSaveRegisters);
118  }
119 
120  // Make sure the defined arrays are big enough. The arrays can be filled up
121  // with `no_reg` and `no_dreg` to pass this DCHECK.
122  DCHECK_GE(registers.size(), GetRegisterParameterCount());
123  DCHECK_GE(double_registers.size(), GetRegisterParameterCount());
124  DCHECK_GE(return_registers.size(), DerivedDescriptor::kReturnCount);
125  DCHECK_GE(return_double_registers.size(), DerivedDescriptor::kReturnCount);
126  data->InitializeRegisters(
127  DerivedDescriptor::flags(), DerivedDescriptor::kEntrypointTag,
128  DerivedDescriptor::kSandboxingMode, DerivedDescriptor::kReturnCount,
129  DerivedDescriptor::GetParameterCount(),
130  DerivedDescriptor::kStackArgumentOrder,
131  DerivedDescriptor::GetRegisterParameterCount(), registers.data(),
132  double_registers.data(), return_registers.data(),
133  return_double_registers.data());
134 
135  // InitializeTypes is customizable by the DerivedDescriptor subclass.
136  DerivedDescriptor::InitializeTypes(data);
137 
138  DCHECK(data->IsInitialized());
139  DCHECK(this->CheckFloatingPointParameters(data));
140 #if DEBUG
141  DerivedDescriptor::Verify(data);
142 #endif
143 }

175 template <>

178 // Helper for finding the index of the first invalid register in a register
179 // array.
180 template <size_t N, size_t Index>
181 struct FirstInvalidRegisterHelper {
182  static constexpr int Call(std::array<Register, N> regs) {
183  if (!std::get<Index>(regs).is_valid()) {
184  // All registers after the first invalid one have to also be invalid (this
185  // DCHECK will be checked recursively).
186  DCHECK_EQ((FirstInvalidRegisterHelper<N, Index + 1>::Call(regs)),
187  Index + 1);
188  return Index;
189  }
190  return FirstInvalidRegisterHelper<N, Index + 1>::Call(regs);
191  }
192 };
193 template <size_t N>
194 struct FirstInvalidRegisterHelper<N, N> {
195  static constexpr int Call(std::array<Register, N> regs) { return N; }
196 };
197 template <size_t N, size_t Index = 0>
198 constexpr size_t FirstInvalidRegister(std::array<Register, N> regs) {
199  return FirstInvalidRegisterHelper<N, 0>::Call(regs);
200 }
201 constexpr size_t FirstInvalidRegister(EmptyRegisterArray regs) { return 0; }
202 
203 } // namespace detail

205 // static
206 template <typename DerivedDescriptor>
207 constexpr int
208 StaticCallInterfaceDescriptor<DerivedDescriptor>::GetRegisterParameterCount() {
209  static_assert(
210  detail::IsRegisterArray<decltype(DerivedDescriptor::registers())>::value,
211  "DerivedDescriptor subclass should define a registers() function "
212  "returning a std::array<Register>");
213 
214  // The register parameter count is the minimum of:
215  // 1. The number of named parameters in the descriptor, and
216  // 2. The number of valid registers the descriptor provides with its
217  // registers() function, e.g. for {rax, rbx, no_reg} this number is 2.
218  // 3. The maximum number of register parameters allowed (
219  // kMaxBuiltinRegisterParams for most builtins,
220  // kMaxTFSBuiltinRegisterParams for TFS builtins, customizable by the
221  // subclass otherwise).
222  return std::min<int>({DerivedDescriptor::GetParameterCount(),
223  static_cast<int>(detail::FirstInvalidRegister(
224  DerivedDescriptor::registers())),
225  DerivedDescriptor::kMaxRegisterParams});
226 }

253 template <typename DerivedDescriptor>
254 constexpr MachineType
255 StaticCallInterfaceDescriptor<DerivedDescriptor>::GetParameterType(int i) {
256  if constexpr (!DerivedDescriptor::kCustomMachineTypes) {
257  // If there are no custom machine types, all results and parameters are
258  // tagged.
259  return MachineType::AnyTagged();
260  } else {
261  // All varags are tagged.
262  if (DerivedDescriptor::AllowVarArgs() &&
263  i >= DerivedDescriptor::GetParameterCount()) {
264  return MachineType::AnyTagged();
265  }
266  DCHECK_LT(i, DerivedDescriptor::GetParameterCount());
267  return DerivedDescriptor::kMachineTypes
268  [DerivedDescriptor::GetReturnCount() + i];
269  }
270 }

306 // static
307 constexpr RegList WriteBarrierDescriptor::ComputeSavedRegisters(
308  Register object, Register slot_address) {
309  DCHECK(!AreAliased(object, slot_address));
310  RegList saved_registers;
311 #if V8_TARGET_ARCH_X64
312  // Only push clobbered registers.
313  if (object != ObjectRegister()) saved_registers.set(ObjectRegister());
314  if (slot_address != no_reg && slot_address != SlotAddressRegister()) {
315  saved_registers.set(SlotAddressRegister());
316  }
317 #elif V8_TARGET_ARCH_ARM64 || V8_TARGET_ARCH_ARM || V8_TARGET_ARCH_LOONG64 || \
318  V8_TARGET_ARCH_MIPS64
319  if (object != ObjectRegister()) saved_registers.set(ObjectRegister());
320  // The slot address is always clobbered.
321  saved_registers.set(SlotAddressRegister());
322 #else
323  // TODO(cbruni): Enable callee-saved registers for other platforms.
324  // This is a temporary workaround to prepare code for callee-saved registers.
325  constexpr auto allocated_registers = registers();
326  for (size_t i = 0; i < allocated_registers.size(); ++i) {
327  saved_registers.set(allocated_registers[i]);
328  }
329 #endif
330  return saved_registers;
331 }
