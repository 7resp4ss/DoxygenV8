246  Tagged<Object> maybe_ctor = constructor();

349 // static
350 MaybeHandle<Object> TemplateInfo::ProbeInstantiationsCache(
351  Isolate* isolate, DirectHandle<NativeContext> native_context,
352  DirectHandle<TemplateInfo> info, CachingMode caching_mode) {
353  DCHECK(info->is_cacheable());
354 
355  uint32_t serial_number = info->serial_number();
356  if (serial_number == kUninitializedSerialNumber) {
357  return {};
358  }
359 
360  if (serial_number < kFastTemplateInstantiationsCacheSize) {
361  Tagged<FixedArray> fast_cache =
362  native_context->fast_template_instantiations_cache();
363  Tagged<Object> object = fast_cache->get(serial_number);
364  if (IsTheHole(object, isolate)) {
365  return {};
366  }
367  return handle(object, isolate);
368  }
369  Tagged<EphemeronHashTable> cache =
370  native_context->slow_template_instantiations_cache();
371  ReadOnlyRoots roots(isolate);
372  // Instead of detouring via Object::GetHash() load the hash directly.
373  uint32_t hash = info->GetHash();
374  InternalIndex entry = cache->FindEntry(isolate, roots, info, hash);
375  if (entry.is_found()) {
376  return handle(cache->ValueAt(entry), isolate);
377  }
378  return {};
379 }

381 // static
382 void TemplateInfo::CacheTemplateInstantiation(
383  Isolate* isolate, DirectHandle<NativeContext> native_context,
384  DirectHandle<TemplateInfo> info, CachingMode caching_mode,
385  DirectHandle<Object> object) {
386  DCHECK(info->is_cacheable());
387 
388  uint32_t serial_number = info->EnsureHasSerialNumber(isolate);
389 
390  if (serial_number < kFastTemplateInstantiationsCacheSize) {
391  Handle<FixedArray> fast_cache =
392  handle(native_context->fast_template_instantiations_cache(), isolate);
393  fast_cache->set(serial_number, *object);
394  return;
395  }
396  Handle<EphemeronHashTable> cache =
397  handle(native_context->slow_template_instantiations_cache(), isolate);
398  if (caching_mode == CachingMode::kUnlimited ||
399  (cache->NumberOfElements() < kMaxTemplateInstantiationsCacheSize)) {
400  ReadOnlyRoots roots(isolate);
401  // Instead of detouring via Object::GetHash() load the hash directly.
402  uint32_t hash = info->GetHash();
403  auto new_cache =
404  EphemeronHashTable::Put(isolate, cache, info, object, hash);
405  if (*new_cache != *cache) {
406  native_context->set_slow_template_instantiations_cache(*new_cache);
407  }
408  }
409 }

411 // static
412 void TemplateInfo::UncacheTemplateInstantiation(
413  Isolate* isolate, DirectHandle<NativeContext> native_context,
414  DirectHandle<TemplateInfo> info, CachingMode caching_mode) {
415  int serial_number = info->serial_number();
416  if (serial_number == kUninitializedSerialNumber) return;
417 
418  if (serial_number < kFastTemplateInstantiationsCacheSize) {
419  Tagged<FixedArray> fast_cache =
420  native_context->fast_template_instantiations_cache();
421  DCHECK(!IsUndefined(fast_cache->get(serial_number), isolate));
422  fast_cache->set(serial_number, ReadOnlyRoots{isolate}.the_hole_value(),
423  SKIP_WRITE_BARRIER);
424  return;
425  }
426  Handle<EphemeronHashTable> cache =
427  handle(native_context->slow_template_instantiations_cache(), isolate);
428  // Instead of detouring via Object::GetHash() load the hash directly.
429  uint32_t hash = info->GetHash();
430  bool was_present = false;
431  auto new_cache =
432  EphemeronHashTable::Remove(isolate, cache, info, &was_present, hash);
433  DCHECK(was_present);
434  if (!new_cache.is_identical_to(cache)) {
435  native_context->set_slow_template_instantiations_cache(*new_cache);
436  }
437 }
