29 class OptimizingCompileTaskExecutor::CompileTask : public v8::JobTask {
30  public:
31  explicit CompileTask(OptimizingCompileTaskExecutor* task_executor)
32  : task_executor_(task_executor) {}
33 
34  void Run(JobDelegate* delegate) override {
35  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"), "V8.TurbofanTask");
36  DCHECK_LT(delegate->GetTaskId(), task_executor_->task_states_.size());
37  OptimizingCompileTaskState& task_state =
38  task_executor_->task_states_[delegate->GetTaskId()];
39  bool should_yield = delegate->ShouldYield();
40 
41  while (!should_yield) {
42  // NextInput() sets the isolate for task_state to job->isolate() while
43  // holding the lock.
44  TurbofanCompilationJob* job = task_executor_->NextInput(task_state);
45  if (!job) break;
46 
47  Isolate* const isolate = job->isolate();
48 
49  {
50  base::FlushDenormalsScope flush_denormals_scope(
51  isolate->flush_denormals());
52 
53  // Note that LocalIsolate's lifetime is shorter than the isolate value
54  // in task_state which is only cleared after this LocalIsolate instance
55  // was destroyed.
56  LocalIsolate local_isolate(isolate, ThreadKind::kBackground);
57  DCHECK(local_isolate.heap()->IsParked());
58 
59  do {
60  task_executor_->RunCompilationJob(task_state, isolate, local_isolate,
61  job);
62 
63  should_yield = delegate->ShouldYield();
64  if (should_yield) break;
65 
66  // Reuse the LocalIsolate if the next worklist item has the same
67  // isolate.
68  job = task_executor_->NextInputIfIsolateMatches(task_state);
69  } while (job);
70  }
71 
72  // Reset the isolate in the task state to nullptr. Only do this after the
73  // LocalIsolate was destroyed. This invariant is used by
74  // WaitUntilTasksStoppedForIsolate() to ensure all tasks are stopped for
75  // an isolate.
76  task_executor_->ClearTaskState(task_state);
77  }
78 
79  // Here we are allowed to read the isolate without holding a lock because
80  // only this thread here will ever change this field and the main thread
81  // will only ever read it.
82  DCHECK_NULL(task_state.isolate);
83  }
84 
85  size_t GetMaxConcurrency(size_t worker_count) const override {
86  size_t num_tasks = task_executor_->input_queue_.Length() + worker_count;
87  return std::min(num_tasks, task_executor_->task_states_.size());
88  }
89 
90  private:
91  OptimizingCompileTaskExecutor* task_executor_;
92 };

104  if (is_initialized_) return;
105  is_initialized_ = true;
106 
107  if (v8_flags.concurrent_recompilation ||
108  v8_flags.concurrent_builtin_generation) {
109  int max_tasks;
110 
111  if (v8_flags.concurrent_turbofan_max_threads == 0) {
112  max_tasks = V8::GetCurrentPlatform()->NumberOfWorkerThreads();
113  } else {

147 void OptimizingCompileTaskExecutor::RunCompilationJob(
148  OptimizingCompileTaskState& task_state, Isolate* isolate,
149  LocalIsolate& local_isolate, TurbofanCompilationJob* job) {
150  TRACE_EVENT_WITH_FLOW0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
151  "V8.OptimizeBackground", job->trace_id(),
152  TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
153  TimerEventScope<TimerEventRecompileConcurrent> timer(isolate);
154 
155  if (recompilation_delay_ != 0) {
156  base::OS::Sleep(base::TimeDelta::FromMilliseconds(recompilation_delay_));
157  }
158 
159  RCS_SCOPE(&local_isolate, RuntimeCallCounterId::kOptimizeBackgroundTurbofan);
160 
161  // The function may have already been optimized by OSR. Simply continue.
162  CompilationJob::Status status =
163  job->ExecuteJob(local_isolate.runtime_call_stats(), &local_isolate);
164  USE(status); // Prevent an unused-variable error.
165 
166  // Remove the job first from task_state before adding it to the output queue.
167  // As soon as the job is in the output queue it could be deleted any moment.
168  ResetJob(task_state);
169 
170  isolate->optimizing_compile_dispatcher()->QueueFinishedJob(job);
171 }

227 void OptimizingCompileTaskExecutor::WaitUntilCompilationJobsDoneForIsolate(
228  Isolate* isolate) {
229  // Once we have ensured that no task is working on the given isolate, we also
230  // know that there are no more LocalHeaps for this isolate from CompileTask.
231  // This is because CompileTask::Run() only updates the isolate once the
232  // LocalIsolate/LocalHeap for it was destroyed.
233  base::MutexGuard guard(&input_queue_.mutex_);
234 
235  while (input_queue_.HasJobForIsolate(isolate) ||
236  IsTaskRunningForIsolate(isolate)) {
237  input_queue_.task_finished_.Wait(&input_queue_.mutex_);
238  }
239 }

321 void OptimizingCompileDispatcher::InstallOptimizedFunctions() {
322  HandleScope handle_scope(isolate_);
323 
324  for (;;) {
325  std::unique_ptr<TurbofanCompilationJob> job = output_queue_.Dequeue();
326  if (!job) break;
327 
328  OptimizedCompilationInfo* info = job->compilation_info();
329  DirectHandle<JSFunction> function(*info->closure(), isolate_);
330 
331  // If another racing task has already finished compiling and installing the
332  // requested code kind on the function, throw out the current job.
333  if (!info->is_osr() &&
334  function->HasAvailableCodeKind(isolate_, info->code_kind())) {
335  if (v8_flags.trace_concurrent_recompilation) {
336  PrintF(" ** Aborting compilation for ");
337  ShortPrint(*function);
338  PrintF(" as it has already been optimized.\n");
339  }
340  Compiler::DisposeTurbofanCompilationJob(isolate_, job.get());
341  continue;
342  }
343  // Discard code compiled for a discarded native context without
344  // finalization.
345  if (function->native_context()->global_object()->IsDetached(isolate_)) {
346  Compiler::DisposeTurbofanCompilationJob(isolate_, job.get());
347  continue;
348  }
349 
350  Compiler::FinalizeTurbofanCompilationJob(job.get(), isolate_);
351  }
352 }

377 void OptimizingCompileInputQueue::Prioritize(
378  Isolate* isolate, Tagged<SharedFunctionInfo> function) {
379  // Ensure that we only run this method on the main thread. This makes sure
380  // that we never dereference handles during a safepoint.
381  DCHECK_EQ(isolate->thread_id(), ThreadId::Current());
382  base::MutexGuard access(&mutex_);
383  auto it =
384  std::find_if(queue_.begin(), queue_.end(),
385  [isolate, function](TurbofanCompilationJob* job) {
386  // Early bailout to avoid dereferencing handles from other
387  // isolates. The other isolate could be in a safepoint/GC
388  // and dereferencing the handle is therefore invalid.
389  if (job->isolate() != isolate) return false;
390  return *job->compilation_info()->shared_info() == function;
391  });
392 
393  if (it != queue_.end()) {
394  auto first_for_isolate = std::find_if(
395  queue_.begin(), queue_.end(), [isolate](TurbofanCompilationJob* job) {
396  return job->isolate() == isolate;
397  });
398  DCHECK_NE(first_for_isolate, queue_.end());
399  std::iter_swap(it, first_for_isolate);
400  }
401 }

476 int OptimizingCompileOutputQueue::InstallGeneratedBuiltins(
477  Isolate* isolate, int installed_count) {
478  // Builtin generation needs to be deterministic, meaning heap allocations must
479  // happen in a deterministic order. To ensure determinism with concurrent
480  // compilation, only finalize contiguous builtins in ascending order of their
481  // finalization order, which is set at job creation time.
482 
483  CHECK(isolate->IsGeneratingEmbeddedBuiltins());
484 
485  base::MutexGuard guard(&mutex_);
486 
487  std::sort(queue_.begin(), queue_.end(),
488  [](const TurbofanCompilationJob* job1,
489  const TurbofanCompilationJob* job2) {
490  return job1->FinalizeOrder() < job2->FinalizeOrder();
491  });
492 
493  while (!queue_.empty()) {
494  int current = queue_.front()->FinalizeOrder();
495  CHECK_EQ(installed_count, current);
496  std::unique_ptr<TurbofanCompilationJob> job(queue_.front());
497  queue_.pop_front();
498  CHECK_EQ(CompilationJob::SUCCEEDED, job->FinalizeJob(isolate));
499  installed_count = current + 1;
500  }
501  return installed_count;
502 }
