20 // The GraphProcessor takes a NodeProcessor, and applies it to each Node in the
21 // Graph by calling NodeProcessor::Process on each Node.
22 //
23 // The GraphProcessor also keeps track of the current ProcessingState, and
24 // passes this to the Process method.
25 //
26 // It expects a NodeProcessor class with:
27 //
28 // // A function that processes the graph before the nodes are walked.
29 // void PreProcessGraph(Graph* graph);
30 //
31 // // A function that processes the graph after the nodes are walked.
32 // void PostProcessGraph(Graph* graph);
33 //
34 // // A function that processes each basic block before its nodes are walked.
35 // BlockProcessResult PreProcessBasicBlock(BasicBlock* block);
36 //
37 // // Process methods for each Node type. The GraphProcessor switches over
38 // // the Node's opcode, casts it to the appropriate FooNode, and dispatches
39 // // to NodeProcessor::Process. It's then up to the NodeProcessor to provide
40 // // either distinct Process methods per Node type, or using templates or
41 // // overloading as appropriate to group node processing.
42 // void Process(FooNode* node, const ProcessingState& state) {}
43 //
44 template <typename NodeProcessor, bool visit_identity_nodes = false>
45 class GraphProcessor;
46 
47 enum class BlockProcessResult {
48  kContinue, // Process exited normally.
49  kSkip, // Skip processing this block (no MultiProcessor support).
50 };
51 
52 enum class ProcessResult {
53  kContinue, // Process exited normally, and the following processors will be
54  // called on the node.
55  kRemove, // Remove the current node from the graph (and do not call the
56  // following processors).
57  kHoist, // Hoist the current instruction to the parent basic block
58  // and reset the current instruction to the beginning of the
59  // block. Parent block must be dominating.
60  kAbort, // Stop processing now, do not process subsequent nodes/blocks.
61  // Should not be used when processing Constants.
62  kSkipBlock, // Stop processing this block and skip the remaining nodes (no
63  // MultiProcessor support).
64 };

89 template <typename NodeProcessor, bool visit_identity_nodes>
90 class GraphProcessor {
91  public:
92  template <typename... Args>
93  explicit GraphProcessor(Args&&... args)
94  : node_processor_(std::forward<Args>(args)...) {}
95 
96  void ProcessGraph(Graph* graph) {
97  graph_ = graph;
98 
99  node_processor_.PreProcessGraph(graph);
100 
101  auto process_constants = [&](auto& map) {
102  for (auto it = map.begin(); it != map.end();) {
103  ProcessResult result =
104  node_processor_.Process(it->second, GetCurrentState());
105  switch (result) {
106  [[likely]] case ProcessResult::kContinue:
107  ++it;
108  break;
109  case ProcessResult::kRemove:
110  it = map.erase(it);
111  break;
112  case ProcessResult::kHoist:
113  case ProcessResult::kAbort:
114  case ProcessResult::kSkipBlock:
115  UNREACHABLE();
116  }
117  }
118  };
119  process_constants(graph->constants());
120  process_constants(graph->root());
121  process_constants(graph->smi());
122  process_constants(graph->tagged_index());
123  process_constants(graph->int32());
124  process_constants(graph->uint32());
125  process_constants(graph->intptr());
126  process_constants(graph->float64());
127  process_constants(graph->trusted_constants());
128 
129  for (block_it_ = graph->begin(); block_it_ != graph->end(); ++block_it_) {
130  BasicBlock* block = *block_it_;
131  if (V8_UNLIKELY(block->is_dead())) continue;
132 
133  BlockProcessResult preprocess_result =
134  node_processor_.PreProcessBasicBlock(block);
135  switch (preprocess_result) {
136  [[likely]] case BlockProcessResult::kContinue:
137  break;
138  case BlockProcessResult::kSkip:
139  continue;
140  }
141 
142  if (block->has_phi()) {
143  auto& phis = *block->phis();
144  for (auto it = phis.begin(); it != phis.end();) {
145  Phi* phi = *it;
146  ProcessResult result =
147  node_processor_.Process(phi, GetCurrentState());
148  switch (result) {
149  [[likely]] case ProcessResult::kContinue:
150  ++it;
151  break;
152  case ProcessResult::kRemove:
153  it = phis.RemoveAt(it);
154  break;
155  case ProcessResult::kAbort:
156  return;
157  case ProcessResult::kSkipBlock:
158  goto skip_block;
159  case ProcessResult::kHoist:
160  UNREACHABLE();
161  }
162  }
163  }
164 
165  node_processor_.PostPhiProcessing();
166 
167  for (node_it_ = block->nodes().begin(); node_it_ != block->nodes().end();
168  ++node_it_) {
169  Node* node = *node_it_;
170  if (node == nullptr) continue;
171  ProcessResult result = ProcessNodeBase(node, GetCurrentState());
172  switch (result) {
173  [[likely]] case ProcessResult::kContinue:
174  break;
175  case ProcessResult::kRemove:
176  *node_it_ = nullptr;
177  break;
178  case ProcessResult::kHoist: {
179  DCHECK(block->predecessor_count() == 1 ||
180  (block->predecessor_count() == 2 && block->is_loop()));
181  BasicBlock* target = block->predecessor_at(0);
182  DCHECK(target->successors().size() == 1);
183  Node* cur = *node_it_;
184  cur->set_owner(target);
185  *node_it_ = nullptr;
186  target->nodes().push_back(cur);
187  node_it_ = block->nodes().begin();
188  break;
189  }
190  case ProcessResult::kAbort:
191  return;
192  case ProcessResult::kSkipBlock:
193  goto skip_block;
194  }
195  }
196 
197  {
198  ProcessResult control_result =
199  ProcessNodeBase(block->control_node(), GetCurrentState());
200  switch (control_result) {
201  [[likely]] case ProcessResult::kContinue:
202  case ProcessResult::kSkipBlock:
203  break;
204  case ProcessResult::kAbort:
205  return;
206  case ProcessResult::kRemove:
207  case ProcessResult::kHoist:
208  UNREACHABLE();
209  }
210  }
211  skip_block:
212  node_processor_.PostProcessBasicBlock(block);
213  continue;
214  }
215 
216  node_processor_.PostProcessGraph(graph);
217  }
218 
219  NodeProcessor& node_processor() { return node_processor_; }
220  const NodeProcessor& node_processor() const { return node_processor_; }
221 
222  private:
223  ProcessingState GetCurrentState() {
224  return ProcessingState(block_it_, &node_it_);
225  }
226 
227  ProcessResult ProcessNodeBase(NodeBase* node, const ProcessingState& state) {
228  switch (node->opcode()) {
229 #define CASE(OPCODE) \
230  case Opcode::k##OPCODE: \
231  if constexpr (!visit_identity_nodes && \
232  Opcode::k##OPCODE == Opcode::kIdentity) { \
233  return ProcessResult::kContinue; \
234  } \
235  PreProcess(node->Cast<OPCODE>(), state); \
236  return node_processor_.Process(node->Cast<OPCODE>(), state);
237 
238  NODE_BASE_LIST(CASE)
239 #undef CASE
240  }
241  }
242 
243  void PreProcess(NodeBase* node, const ProcessingState& state) {}
244 
245  NodeProcessor node_processor_;
246  Graph* graph_;
247  BlockConstIterator block_it_;
248  NodeIterator node_it_;
249 };

273 class NodeMultiProcessor<Processor, Processors...>
274  : NodeMultiProcessor<Processors...> {
275  using Base = NodeMultiProcessor<Processors...>;

277  public:
278  template <typename... Args>
279  explicit NodeMultiProcessor(Processor&& processor, Args&&... processors)
280  : Base(std::forward<Args>(processors)...),

286  template <typename Node>
287  ProcessResult Process(Node* node, const ProcessingState& state) {
288  auto res = processor_.Process(node, state);
289  switch (res) {
290  [[likely]] case ProcessResult::kContinue:
291  return Base::Process(node, state);
292  case ProcessResult::kAbort:
293  case ProcessResult::kRemove:
294  return res;
295  case ProcessResult::kHoist:
296  case ProcessResult::kSkipBlock:
297  // TODO(olivf): How to combine these with multiple processors depends on
298  // the needs of the actual processors. Implement once needed.
299  UNREACHABLE();
300  }
301  }
302  void PreProcessGraph(Graph* graph) {
303  processor_.PreProcessGraph(graph);
304  Base::PreProcessGraph(graph);
305  }
306  void PostProcessGraph(Graph* graph) {
307  // Post process in reverse order because that kind of makes sense.
308  Base::PostProcessGraph(graph);
309  processor_.PostProcessGraph(graph);
310  }
311  void PostProcessBasicBlock(BasicBlock* block) {
312  Base::PostProcessBasicBlock(block);
313  processor_.PostProcessBasicBlock(block);
314  }
315  BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
316  BlockProcessResult res = processor_.PreProcessBasicBlock(block);
317  switch (res) {
318  [[likely]] case BlockProcessResult::kContinue:
319  return Base::PreProcessBasicBlock(block);
320  case BlockProcessResult::kSkip:
321  return res;
322  }
323  }
324  void PostPhiProcessing() {
325  processor_.PostPhiProcessing();
326  Base::PostPhiProcessing();
327  }
328 
329  private:
330  Processor processor_;
331 };
