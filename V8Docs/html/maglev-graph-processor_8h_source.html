20 // The GraphProcessor takes a NodeProcessor, and applies it to each Node in the
21 // Graph by calling NodeProcessor::Process on each Node.
22 //
23 // The GraphProcessor also keeps track of the current ProcessingState, and
24 // passes this to the Process method.
25 //
26 // It expects a NodeProcessor class with:
27 //
28 // // A function that processes the graph before the nodes are walked.
29 // void PreProcessGraph(Graph* graph);
30 //
31 // // A function that processes the graph after the nodes are walked.
32 // void PostProcessGraph(Graph* graph);
33 //
34 // // A function that processes each basic block before its nodes are walked.
35 // BlockProcessResult PreProcessBasicBlock(BasicBlock* block);
36 //
37 // // Process methods for each Node type. The GraphProcessor switches over
38 // // the Node's opcode, casts it to the appropriate FooNode, and dispatches
39 // // to NodeProcessor::Process. It's then up to the NodeProcessor to provide
40 // // either distinct Process methods per Node type, or using templates or
41 // // overloading as appropriate to group node processing.
42 // void Process(FooNode* node, const ProcessingState& state) {}
43 //
44 template <typename NodeProcessor, bool visit_identity_nodes = false>
45 class GraphProcessor;
46 
47 enum class BlockProcessResult {
48  kContinue, // Process exited normally.
49  kSkip, // Skip processing this block (no MultiProcessor support).
50 };
51 
52 enum class ProcessResult {
53  kContinue, // Process exited normally, and the following processors will be
54  // called on the node.
55  kRemove, // Remove the current node from the graph (and do not call the
56  // following processors).
57  kHoist, // Hoist the current instruction to the parent basic block
58  // and reset the current instruction to the beginning of the
59  // block. Parent block must be dominating.
60  kAbort, // Stop processing now, do not process subsequent nodes/blocks.
61  // Should not be used when processing Constants.
62  kSkipBlock, // Stop processing this block and skip the remaining nodes (no
63  // MultiProcessor support).
64 };

89 template <typename NodeProcessor, bool visit_identity_nodes>
90 class GraphProcessor {
91  public:
92  template <typename... Args>
93  explicit GraphProcessor(Args&&... args)
94  : node_processor_(std::forward<Args>(args)...) {}
95 
96  void ProcessGraph(Graph* graph) {
97  graph_ = graph;
98 
99  node_processor_.PreProcessGraph(graph);
100 
101  auto process_constants = [&](auto& map) {
102  for (auto it = map.begin(); it != map.end();) {
103  ProcessResult result =
104  node_processor_.Process(it->second, GetCurrentState());
105  switch (result) {
106  [[likely]] case ProcessResult::kContinue:
107  ++it;
108  break;
109  case ProcessResult::kRemove:
110  it = map.erase(it);
111  break;
112  case ProcessResult::kHoist:
113  case ProcessResult::kAbort:
114  case ProcessResult::kSkipBlock:
115  UNREACHABLE();
116  }
117  }
118  };
119  process_constants(graph->constants());
120  process_constants(graph->root());
121  process_constants(graph->smi());
122  process_constants(graph->tagged_index());
123  process_constants(graph->int32());
124  process_constants(graph->uint32());
125  process_constants(graph->intptr());
126  process_constants(graph->float64());
127  process_constants(graph->external_references());
128  process_constants(graph->trusted_constants());
129 
130  for (block_it_ = graph->begin(); block_it_ != graph->end(); ++block_it_) {
131  BasicBlock* block = *block_it_;
132  if (V8_UNLIKELY(block->is_dead())) continue;
133 
134  BlockProcessResult preprocess_result =
135  node_processor_.PreProcessBasicBlock(block);
136  switch (preprocess_result) {
137  [[likely]] case BlockProcessResult::kContinue:
138  break;
139  case BlockProcessResult::kSkip:
140  continue;
141  }
142 
143  if (block->has_phi()) {
144  auto& phis = *block->phis();
145  for (auto it = phis.begin(); it != phis.end();) {
146  Phi* phi = *it;
147  ProcessResult result =
148  node_processor_.Process(phi, GetCurrentState());
149  switch (result) {
150  [[likely]] case ProcessResult::kContinue:
151  ++it;
152  break;
153  case ProcessResult::kRemove:
154  it = phis.RemoveAt(it);
155  break;
156  case ProcessResult::kAbort:
157  return;
158  case ProcessResult::kSkipBlock:
159  goto skip_block;
160  case ProcessResult::kHoist:
161  UNREACHABLE();
162  }
163  }
164  }
165 
166  node_processor_.PostPhiProcessing();
167 
168  for (node_it_ = block->nodes().begin(); node_it_ != block->nodes().end();
169  ++node_it_) {
170  Node* node = *node_it_;
171  if (node == nullptr) continue;
172  ProcessResult result = ProcessNodeBase(node, GetCurrentState());
173  switch (result) {
174  [[likely]] case ProcessResult::kContinue:
175  break;
176  case ProcessResult::kRemove:
177  *node_it_ = nullptr;
178  break;
179  case ProcessResult::kHoist: {
180  DCHECK(block->predecessor_count() == 1 ||
181  (block->predecessor_count() == 2 && block->is_loop()));
182  BasicBlock* target = block->predecessor_at(0);
183  DCHECK(target->successors().size() == 1);
184  Node* cur = *node_it_;
185  cur->set_owner(target);
186  *node_it_ = nullptr;
187  target->nodes().push_back(cur);
188  node_it_ = block->nodes().begin();
189  break;
190  }
191  case ProcessResult::kAbort:
192  return;
193  case ProcessResult::kSkipBlock:
194  goto skip_block;
195  }
196  }
197 
198  {
199  ProcessResult control_result =
200  ProcessNodeBase(block->control_node(), GetCurrentState());
201  switch (control_result) {
202  [[likely]] case ProcessResult::kContinue:
203  case ProcessResult::kSkipBlock:
204  break;
205  case ProcessResult::kAbort:
206  return;
207  case ProcessResult::kRemove:
208  case ProcessResult::kHoist:
209  UNREACHABLE();
210  }
211  }
212  skip_block:
213  node_processor_.PostProcessBasicBlock(block);
214  continue;
215  }
216 
217  node_processor_.PostProcessGraph(graph);
218  }
219 
220  NodeProcessor& node_processor() { return node_processor_; }
221  const NodeProcessor& node_processor() const { return node_processor_; }
222 
223  private:
224  ProcessingState GetCurrentState() {
225  return ProcessingState(block_it_, &node_it_);
226  }
227 
228  ProcessResult ProcessNodeBase(NodeBase* node, const ProcessingState& state) {
229  switch (node->opcode()) {
230 #define CASE(OPCODE) \
231  case Opcode::k##OPCODE: \
232  if constexpr (!visit_identity_nodes && \
233  Opcode::k##OPCODE == Opcode::kIdentity) { \
234  return ProcessResult::kContinue; \
235  } \
236  PreProcess(node->Cast<OPCODE>(), state); \
237  return node_processor_.Process(node->Cast<OPCODE>(), state);
238 
239  NODE_BASE_LIST(CASE)
240 #undef CASE
241  }
242  }
243 
244  void PreProcess(NodeBase* node, const ProcessingState& state) {}
245 
246  NodeProcessor node_processor_;
247  Graph* graph_;
248  BlockConstIterator block_it_;
249  NodeIterator node_it_;
250 };

273 template <typename Processor, typename... Processors>
274 class NodeMultiProcessor<Processor, Processors...>
275  : NodeMultiProcessor<Processors...> {
276  using Base = NodeMultiProcessor<Processors...>;
277 
278  public:
279  template <typename... Args>
280  explicit NodeMultiProcessor(Processor&& processor, Args&&... processors)
281  : Base(std::forward<Args>(processors)...),
282  processor_(std::forward<Processor>(processor)) {}
283  template <typename... Args>
284  explicit NodeMultiProcessor(Args&&... processors)
285  : Base(std::forward<Args>(processors)...) {}
286 
287  template <typename Node>
288  ProcessResult Process(Node* node, const ProcessingState& state) {
289  auto res = processor_.Process(node, state);
290  switch (res) {
291  [[likely]] case ProcessResult::kContinue:
292  return Base::Process(node, state);
293  case ProcessResult::kAbort:
294  case ProcessResult::kRemove:
295  return res;
296  case ProcessResult::kHoist:
297  case ProcessResult::kSkipBlock:
298  // TODO(olivf): How to combine these with multiple processors depends on
299  // the needs of the actual processors. Implement once needed.
300  UNREACHABLE();
301  }
302  }
303  void PreProcessGraph(Graph* graph) {
304  processor_.PreProcessGraph(graph);
305  Base::PreProcessGraph(graph);
306  }
307  void PostProcessGraph(Graph* graph) {
308  // Post process in reverse order because that kind of makes sense.
309  Base::PostProcessGraph(graph);
310  processor_.PostProcessGraph(graph);
311  }
312  void PostProcessBasicBlock(BasicBlock* block) {
313  Base::PostProcessBasicBlock(block);
314  processor_.PostProcessBasicBlock(block);
315  }
316  BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
317  BlockProcessResult res = processor_.PreProcessBasicBlock(block);
318  switch (res) {
319  [[likely]] case BlockProcessResult::kContinue:
320  return Base::PreProcessBasicBlock(block);
321  case BlockProcessResult::kSkip:
322  return res;
323  }
324  }
325  void PostPhiProcessing() {
326  processor_.PostPhiProcessing();
327  Base::PostPhiProcessing();
328  }
329 
330  private:
331  Processor processor_;
332 };
