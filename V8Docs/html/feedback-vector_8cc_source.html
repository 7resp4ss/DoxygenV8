82 template <typename IsolateT>
83 Handle<FeedbackMetadata> FeedbackMetadata::New(IsolateT* isolate,
84  const FeedbackVectorSpec* spec) {
85  auto* factory = isolate->factory();
86 
87  const int slot_count = spec->slot_count();
88  const int create_closure_slot_count = spec->create_closure_slot_count();
89  if (slot_count == 0 && create_closure_slot_count == 0) {
90  return factory->empty_feedback_metadata();
91  }
92 #ifdef DEBUG
93  for (int i = 0; i < slot_count;) {
94  FeedbackSlotKind kind = spec->GetKind(FeedbackSlot(i));
95  int entry_size = FeedbackMetadata::GetSlotSize(kind);
96  for (int j = 1; j < entry_size; j++) {
97  kind = spec->GetKind(FeedbackSlot(i + j));
98  DCHECK_EQ(FeedbackSlotKind::kInvalid, kind);
99  }
100  i += entry_size;
101  }
102 #endif
103 
104  Handle<FeedbackMetadata> metadata =
105  factory->NewFeedbackMetadata(slot_count, create_closure_slot_count);
106 
107  // Initialize the slots. The raw data section has already been pre-zeroed in
108  // NewFeedbackMetadata.
109  for (int i = 0; i < slot_count; i++) {
110  FeedbackSlot slot(i);
111  FeedbackSlotKind kind = spec->GetKind(slot);
112  metadata->SetKind(slot, kind);
113  }
114 
115  for (int i = 0; i < create_closure_slot_count; i++) {
116  uint16_t parameter_count = spec->GetCreateClosureParameterCount(i);
117  metadata->SetCreateClosureParameterCount(i, parameter_count);
118  }
119 
120  return metadata;
121 }

128 bool FeedbackMetadata::SpecDiffersFrom(
129  const FeedbackVectorSpec* other_spec) const {
130  if (other_spec->slot_count() != slot_count()) {
131  return true;
132  }
133 
134  int slots = slot_count();
135  for (int i = 0; i < slots;) {
136  FeedbackSlot slot(i);
137  FeedbackSlotKind kind = GetKind(slot);
138  int entry_size = FeedbackMetadata::GetSlotSize(kind);
139 
140  if (kind != other_spec->GetKind(slot)) {
141  return true;
142  }
143  i += entry_size;
144  }
145  return false;
146 }

217 DirectHandle<ClosureFeedbackCellArray> ClosureFeedbackCellArray::New(
218  Isolate* isolate, DirectHandle<SharedFunctionInfo> shared,
219  AllocationType allocation) {
220  int length = shared->feedback_metadata()->create_closure_slot_count();
221  if (length == 0) {
222  return isolate->factory()->empty_closure_feedback_cell_array();
223  }
224 
225  // Pre-allocate the cells s.t. we can initialize `result` without further
226  // allocation.
227  DirectHandleVector<FeedbackCell> cells(isolate);
228  cells.reserve(length);
229  for (int i = 0; i < length; i++) {
230  DirectHandle<FeedbackCell> cell = isolate->factory()->NewNoClosuresCell();
231 #ifdef V8_ENABLE_LEAPTIERING
232  uint16_t parameter_count =
233  shared->feedback_metadata()->GetCreateClosureParameterCount(i);
234  auto initial_code = BUILTIN_CODE(isolate, CompileLazy);
235  FeedbackCell::AllocateAndInstallJSDispatchHandle(
236  cell, FeedbackCell::kDispatchHandleOffset, isolate, parameter_count,
237  initial_code);
238 #endif
239  cells.push_back(cell);
240  }
241 
242  std::optional<DisallowGarbageCollection> no_gc;
243  auto result = Allocate(isolate, length, &no_gc, allocation);
244  for (int i = 0; i < length; i++) {
245  result->set(i, *cells[i]);
246  }
247 
248  return result;
249 }

251 // static
252 Handle<FeedbackVector> FeedbackVector::New(
253  Isolate* isolate, DirectHandle<SharedFunctionInfo> shared,
254  DirectHandle<ClosureFeedbackCellArray> closure_feedback_cell_array,
255  DirectHandle<FeedbackCell> parent_feedback_cell,
256  IsCompiledScope* is_compiled_scope) {
257  DCHECK(is_compiled_scope->is_compiled());
258  Factory* factory = isolate->factory();
259 
260  DirectHandle<FeedbackMetadata> feedback_metadata(shared->feedback_metadata(),
261  isolate);
262  const int slot_count = feedback_metadata->slot_count();
263 
264  Handle<FeedbackVector> vector = factory->NewFeedbackVector(
265  shared, closure_feedback_cell_array, parent_feedback_cell);
266 
267  DCHECK_EQ(vector->length(), slot_count);
268 
269  DCHECK_EQ(vector->shared_function_info(), *shared);
270  DCHECK_EQ(vector->invocation_count(), 0);
271 #ifndef V8_ENABLE_LEAPTIERING
272  DCHECK_EQ(vector->tiering_state(), TieringState::kNone);
273  DCHECK(!vector->maybe_has_maglev_code());
274  DCHECK(!vector->maybe_has_turbofan_code());
275  DCHECK(vector->maybe_optimized_code().IsCleared());
276 #endif // !V8_ENABLE_LEAPTIERING
277 
278  // Ensure we can skip the write barrier
279  DirectHandle<Symbol> uninitialized_sentinel = UninitializedSentinel(isolate);
280  DCHECK_EQ(ReadOnlyRoots(isolate).uninitialized_symbol(),
281  *uninitialized_sentinel);
282  for (int i = 0; i < slot_count;) {
283  FeedbackSlot slot(i);
284  FeedbackSlotKind kind = feedback_metadata->GetKind(slot);
285  int entry_size = FeedbackMetadata::GetSlotSize(kind);
286 
287  Tagged<MaybeObject> extra_value = *uninitialized_sentinel;
288  switch (kind) {
289  case FeedbackSlotKind::kLoadGlobalInsideTypeof:
290  case FeedbackSlotKind::kLoadGlobalNotInsideTypeof:
291  case FeedbackSlotKind::kStoreGlobalSloppy:
292  case FeedbackSlotKind::kStoreGlobalStrict:
293  case FeedbackSlotKind::kJumpLoop:
294  vector->Set(slot, ClearedValue(isolate), SKIP_WRITE_BARRIER);
295  break;
296  case FeedbackSlotKind::kForIn:
297  case FeedbackSlotKind::kCompareOp:
298  case FeedbackSlotKind::kBinaryOp:
299  case FeedbackSlotKind::kTypeOf:
300  vector->Set(slot, Smi::zero(), SKIP_WRITE_BARRIER);
301  break;
302  case FeedbackSlotKind::kLiteral:
303  vector->Set(slot, Smi::zero(), SKIP_WRITE_BARRIER);
304  break;
305  case FeedbackSlotKind::kCall:
306  vector->Set(slot, *uninitialized_sentinel, SKIP_WRITE_BARRIER);
307  extra_value = Smi::zero();
308  break;
309  case FeedbackSlotKind::kCloneObject:
310  case FeedbackSlotKind::kLoadProperty:
311  case FeedbackSlotKind::kLoadKeyed:
312  case FeedbackSlotKind::kHasKeyed:
313  case FeedbackSlotKind::kSetNamedSloppy:
314  case FeedbackSlotKind::kSetNamedStrict:
315  case FeedbackSlotKind::kDefineNamedOwn:
316  case FeedbackSlotKind::kDefineKeyedOwn:
317  case FeedbackSlotKind::kSetKeyedSloppy:
318  case FeedbackSlotKind::kSetKeyedStrict:
319  case FeedbackSlotKind::kStoreInArrayLiteral:
320  case FeedbackSlotKind::kDefineKeyedOwnPropertyInLiteral:
321  case FeedbackSlotKind::kInstanceOf:
322  vector->Set(slot, *uninitialized_sentinel, SKIP_WRITE_BARRIER);
323  break;
324  case FeedbackSlotKind::kStringAddAndInternalize:
325  vector->Set(slot, Smi::zero(), SKIP_WRITE_BARRIER);
326  break;
327 
328  case FeedbackSlotKind::kInvalid:
329  UNREACHABLE();
330  }
331  for (int j = 1; j < entry_size; j++) {
332  vector->Set(slot.WithOffset(j), extra_value, SKIP_WRITE_BARRIER);
333  }
334  i += entry_size;
335  }
336 
337  if (!isolate->is_best_effort_code_coverage()) {
338  AddToVectorsForProfilingTools(isolate, vector);
339  }
340  parent_feedback_cell->set_value(*vector, kReleaseStore);
341  return vector;
342 }

400 void FeedbackVector::SetOptimizedCode(IsolateForSandbox isolate,
401  Tagged<Code> code) {
402  DCHECK(CodeKindIsOptimizedJSFunction(code->kind()));
403  int32_t state = flags();
404  // Skip setting optimized code if it would cause us to tier down.
405  if (!has_optimized_code()) {
406  state = MaybeHasTurbofanCodeBit::update(state, false);
407  } else if (!CodeKindCanTierUp(optimized_code(isolate)->kind()) ||
408  optimized_code(isolate)->kind() > code->kind()) {
409  if (!v8_flags.stress_concurrent_inlining_attach_code &&
410  !optimized_code(isolate)->marked_for_deoptimization()) {
411  return;
412  }
413  // If we fall through, we may be tiering down. This is fine since we only do
414  // that when the current code is marked for deoptimization, or because we're
415  // stress testing.
416  state = MaybeHasTurbofanCodeBit::update(state, false);
417  }
418  // TODO(mythria): We could see a CompileOptimized state here either from
419  // tests that use %OptimizeFunctionOnNextCall, --always-turbofan or because we
420  // re-mark the function for non-concurrent optimization after an OSR. We
421  // should avoid these cases and also check that marker isn't
422  // TieringState::kRequestTurbofan*.
423  set_maybe_optimized_code(MakeWeak(code->wrapper()));
424  // TODO(leszeks): Reconsider whether this could clear the tiering state vs.
425  // the callers doing so.
426  state = TieringStateBits::update(state, TieringState::kNone);
427  if (code->is_maglevved()) {
428  DCHECK(!MaybeHasTurbofanCodeBit::decode(state));
429  state = MaybeHasMaglevCodeBit::update(state, true);
430  } else {
431  DCHECK(code->is_turbofanned());
432  state = MaybeHasTurbofanCodeBit::update(state, true);
433  state = MaybeHasMaglevCodeBit::update(state, false);
434  }
435  set_flags(state);
436 }

505 bool FeedbackVector::ClearSlots(Isolate* isolate, ClearBehavior behavior) {
506  if (!shared_function_info()->HasFeedbackMetadata()) return false;
507  Tagged<MaybeObject> uninitialized_sentinel =
508  FeedbackVector::RawUninitializedSentinel(isolate);
509 
510  bool feedback_updated = false;
511  FeedbackMetadataIterator iter(metadata());
512  while (iter.HasNext()) {
513  FeedbackSlot slot = iter.Next();
514 
515  Tagged<MaybeObject> obj = Get(slot);
516  if (obj != uninitialized_sentinel) {
517  FeedbackNexus nexus(isolate, *this, slot);
518  feedback_updated |= nexus.Clear(behavior);
519  }
520  }
521  return feedback_updated;
522 }

764 InlineCacheState FeedbackNexus::ic_state() const {
765  Tagged<MaybeObject> feedback, extra;
766  std::tie(feedback, extra) = GetFeedbackPair();
767 
768  switch (kind()) {
769  case FeedbackSlotKind::kLiteral:
770  if (IsSmi(feedback)) return InlineCacheState::UNINITIALIZED;
771  return InlineCacheState::MONOMORPHIC;
772 
773  case FeedbackSlotKind::kStoreGlobalSloppy:
774  case FeedbackSlotKind::kStoreGlobalStrict:
775  case FeedbackSlotKind::kLoadGlobalNotInsideTypeof:
776  case FeedbackSlotKind::kLoadGlobalInsideTypeof:
777  case FeedbackSlotKind::kJumpLoop: {
778  if (IsSmi(feedback)) return InlineCacheState::MONOMORPHIC;
779 
780  DCHECK(feedback.IsWeakOrCleared());
781  if (!feedback.IsCleared() || extra != UninitializedSentinel()) {
782  return InlineCacheState::MONOMORPHIC;
783  }
784  return InlineCacheState::UNINITIALIZED;
785  }
786 
787  case FeedbackSlotKind::kSetNamedSloppy:
788  case FeedbackSlotKind::kSetNamedStrict:
789  case FeedbackSlotKind::kSetKeyedSloppy:
790  case FeedbackSlotKind::kSetKeyedStrict:
791  case FeedbackSlotKind::kStoreInArrayLiteral:
792  case FeedbackSlotKind::kDefineNamedOwn:
793  case FeedbackSlotKind::kDefineKeyedOwn:
794  case FeedbackSlotKind::kLoadProperty:
795  case FeedbackSlotKind::kLoadKeyed:
796  case FeedbackSlotKind::kHasKeyed: {
797  if (feedback == UninitializedSentinel()) {
798  return InlineCacheState::UNINITIALIZED;
799  }
800  if (feedback == MegamorphicSentinel()) {
801  return InlineCacheState::MEGAMORPHIC;
802  }
803  if (feedback == MegaDOMSentinel()) {
804  DCHECK(IsLoadICKind(kind()));
805  return InlineCacheState::MEGADOM;
806  }
807  if (feedback.IsWeakOrCleared()) {
808  // Don't check if the map is cleared.
809  return InlineCacheState::MONOMORPHIC;
810  }
811  Tagged<HeapObject> heap_object;
812  if (feedback.GetHeapObjectIfStrong(&heap_object)) {
813  if (IsWeakFixedArray(heap_object)) {
814  // Determine state purely by our structure, don't check if the maps
815  // are cleared.
816  return InlineCacheState::POLYMORPHIC;
817  }
818  if (IsName(heap_object)) {
819  DCHECK(IsKeyedLoadICKind(kind()) || IsKeyedStoreICKind(kind()) ||
820  IsKeyedHasICKind(kind()) || IsDefineKeyedOwnICKind(kind()));
821  Tagged<Object> extra_object = extra.GetHeapObjectAssumeStrong();
822  Tagged<WeakFixedArray> extra_array =
823  Cast<WeakFixedArray>(extra_object);
824  return extra_array->length() > 2 ? InlineCacheState::POLYMORPHIC
825  : InlineCacheState::MONOMORPHIC;
826  }
827  }
828  // TODO(1393773): Remove once the issue is solved.
829  Address vector_ptr = vector().ptr();
830  config_.isolate()->PushParamsAndDie(
831  reinterpret_cast<void*>(feedback.ptr()),
832  reinterpret_cast<void*>(extra.ptr()),
833  reinterpret_cast<void*>(vector_ptr),
834  reinterpret_cast<void*>(static_cast<intptr_t>(slot_.ToInt())),
835  reinterpret_cast<void*>(static_cast<intptr_t>(kind())),
836  // Include part of the feedback vector containing the slot.
837  reinterpret_cast<void*>(
838  vector_ptr + FeedbackVector::OffsetOfElementAt(slot_.ToInt())));
839  UNREACHABLE();
840  }
841  case FeedbackSlotKind::kCall: {
842  Tagged<HeapObject> heap_object;
843  if (feedback == MegamorphicSentinel()) {
844  return InlineCacheState::GENERIC;
845  } else if (feedback.IsWeakOrCleared()) {
846  if (feedback.GetHeapObjectIfWeak(&heap_object)) {
847  if (IsFeedbackCell(heap_object)) {
848  return InlineCacheState::POLYMORPHIC;
849  }
850  CHECK(IsJSFunction(heap_object) || IsJSBoundFunction(heap_object));
851  }
852  return InlineCacheState::MONOMORPHIC;
853  } else if (feedback.GetHeapObjectIfStrong(&heap_object) &&
854  IsAllocationSite(heap_object)) {
855  return InlineCacheState::MONOMORPHIC;
856  }
857 
858  CHECK_EQ(feedback, UninitializedSentinel());
859  return InlineCacheState::UNINITIALIZED;
860  }
861  case FeedbackSlotKind::kBinaryOp:
862  case FeedbackSlotKind::kStringAddAndInternalize: {
863  BinaryOperationHint hint = GetBinaryOperationFeedback();
864  if (hint == BinaryOperationHint::kNone) {
865  return InlineCacheState::UNINITIALIZED;
866  } else if (hint == BinaryOperationHint::kAny) {
867  return InlineCacheState::GENERIC;
868  }
869 
870  return InlineCacheState::MONOMORPHIC;
871  }
872  case FeedbackSlotKind::kCompareOp: {
873  CompareOperationHint hint = GetCompareOperationFeedback();
874  if (hint == CompareOperationHint::kNone) {
875  return InlineCacheState::UNINITIALIZED;
876  } else if (hint == CompareOperationHint::kAny) {
877  return InlineCacheState::GENERIC;
878  }
879 
880  return InlineCacheState::MONOMORPHIC;
881  }
882  case FeedbackSlotKind::kForIn: {
883  ForInHint hint = GetForInFeedback();
884  if (hint == ForInHint::kNone) {
885  return InlineCacheState::UNINITIALIZED;
886  } else if (hint == ForInHint::kAny) {
887  return InlineCacheState::GENERIC;
888  }
889  return InlineCacheState::MONOMORPHIC;
890  }
891  case FeedbackSlotKind::kTypeOf: {
892  if (feedback == Smi::zero()) {
893  return InlineCacheState::UNINITIALIZED;
894  } else if (feedback == Smi::FromInt(TypeOfFeedback::kAny)) {
895  return InlineCacheState::MEGAMORPHIC;
896  } else if (base::bits::CountPopulation(
897  static_cast<uint32_t>(feedback.ToSmi().value())) == 1) {
898  return InlineCacheState::MONOMORPHIC;
899  } else {
900  return InlineCacheState::POLYMORPHIC;
901  }
902  }
903  case FeedbackSlotKind::kInstanceOf: {
904  if (feedback == UninitializedSentinel()) {
905  return InlineCacheState::UNINITIALIZED;
906  } else if (feedback == MegamorphicSentinel()) {
907  return InlineCacheState::MEGAMORPHIC;
908  }
909  return InlineCacheState::MONOMORPHIC;
910  }
911  case FeedbackSlotKind::kDefineKeyedOwnPropertyInLiteral: {
912  if (feedback == UninitializedSentinel()) {
913  return InlineCacheState::UNINITIALIZED;
914  } else if (feedback.IsWeakOrCleared()) {
915  // Don't check if the map is cleared.
916  return InlineCacheState::MONOMORPHIC;
917  }
918 
919  return InlineCacheState::MEGAMORPHIC;
920  }
921 
922  case FeedbackSlotKind::kCloneObject: {
923  if (feedback == UninitializedSentinel()) {
924  return InlineCacheState::UNINITIALIZED;
925  }
926  if (feedback == MegamorphicSentinel()) {
927  return InlineCacheState::MEGAMORPHIC;
928  }
929  if (feedback.IsWeakOrCleared()) {
930  return InlineCacheState::MONOMORPHIC;
931  }
932 
933  DCHECK(IsWeakFixedArray(feedback.GetHeapObjectAssumeStrong()));
934  return InlineCacheState::POLYMORPHIC;
935  }
936 
937  case FeedbackSlotKind::kInvalid:
938  UNREACHABLE();
939  }
940  return InlineCacheState::UNINITIALIZED;
941 }

991 void FeedbackNexus::ConfigureCloneObject(
992  DirectHandle<Map> source_map, const MaybeObjectHandle& handler_handle) {
993  // TODO(olivf): Introduce a CloneHandler to deal with all the logic of this
994  // state machine which is now spread between Runtime_CloneObjectIC_Miss and
995  // this method.
996  auto GetHandler = [=]() {
997  if (IsSmi(*handler_handle)) {
998  return *handler_handle;
999  }
1000  return MakeWeak(*handler_handle);
1001  };
1002  DCHECK(config()->can_write());
1003  Isolate* isolate = config()->isolate();
1004  DirectHandle<HeapObject> feedback;
1005  {
1006  Tagged<MaybeObject> maybe_feedback = GetFeedback();
1007  if (maybe_feedback.IsStrongOrWeak()) {
1008  feedback = direct_handle(maybe_feedback.GetHeapObject(), isolate);
1009  } else {
1010  DCHECK(maybe_feedback.IsCleared());
1011  }
1012  }
1013  switch (ic_state()) {
1014  case InlineCacheState::UNINITIALIZED:
1015  // Cache the first map seen which meets the fast case requirements.
1016  SetFeedback(MakeWeak(*source_map), UPDATE_WRITE_BARRIER, GetHandler());
1017  break;
1018  case InlineCacheState::MONOMORPHIC:
1019  if (feedback.is_null() || feedback.is_identical_to(source_map) ||
1020  Cast<Map>(*feedback)->is_deprecated()) {
1021  SetFeedback(MakeWeak(*source_map), UPDATE_WRITE_BARRIER, GetHandler());
1022  } else {
1023  // Transition to POLYMORPHIC.
1024  DirectHandle<WeakFixedArray> array =
1025  CreateArrayOfSize(2 * kCloneObjectPolymorphicEntrySize);
1026  DisallowGarbageCollection no_gc;
1027  Tagged<WeakFixedArray> raw_array = *array;
1028  raw_array->set(0, MakeWeak(*feedback));
1029  raw_array->set(1, GetFeedbackExtra());
1030  raw_array->set(2, MakeWeak(*source_map));
1031  raw_array->set(3, GetHandler());
1032  SetFeedback(raw_array, UPDATE_WRITE_BARRIER, ClearedValue(isolate));
1033  }
1034  break;
1035  case InlineCacheState::POLYMORPHIC: {
1036  const int kMaxElements = v8_flags.max_valid_polymorphic_map_count *
1037  kCloneObjectPolymorphicEntrySize;
1038  DirectHandle<WeakFixedArray> array = Cast<WeakFixedArray>(feedback);
1039  int i = 0;
1040  for (; i < array->length(); i += kCloneObjectPolymorphicEntrySize) {
1041  Tagged<MaybeObject> feedback_map = array->get(i);
1042  if (feedback_map.IsCleared()) break;
1043  DirectHandle<Map> cached_map(Cast<Map>(feedback_map.GetHeapObject()),
1044  isolate);
1045  if (cached_map.is_identical_to(source_map) ||
1046  cached_map->is_deprecated())
1047  break;
1048  }
1049 
1050  if (i >= array->length()) {
1051  if (i == kMaxElements) {
1052  // Transition to MEGAMORPHIC.
1053  Tagged<MaybeObject> sentinel = MegamorphicSentinel();
1054  SetFeedback(sentinel, SKIP_WRITE_BARRIER, ClearedValue(isolate));
1055  break;
1056  }
1057 
1058  // Grow polymorphic feedback array.
1059  DirectHandle<WeakFixedArray> new_array = CreateArrayOfSize(
1060  array->length() + kCloneObjectPolymorphicEntrySize);
1061  for (int j = 0; j < array->length(); ++j) {
1062  new_array->set(j, array->get(j));
1063  }
1064  SetFeedback(*new_array);
1065  array = new_array;
1066  }
1067 
1068  array->set(i, MakeWeak(*source_map));
1069  array->set(i + 1, GetHandler());
1070  break;
1071  }
1072 
1073  default:
1074  UNREACHABLE();
1075  }
1076 }
