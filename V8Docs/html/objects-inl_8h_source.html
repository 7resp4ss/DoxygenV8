354 template <typename CompareAndSwapImpl>
355 Tagged<Object> HeapObject::SeqCst_CompareAndSwapField(
356  Tagged<Object> expected, Tagged<Object> value,
357  CompareAndSwapImpl compare_and_swap_impl) {
358  Tagged<Object> actual_expected = expected;
359  do {
360  Tagged<Object> old_value = compare_and_swap_impl(actual_expected, value);
361  if (old_value == actual_expected || !IsNumber(old_value) ||
362  !IsNumber(actual_expected)) {
363  return old_value;
364  }
365  if (!Object::SameNumberValue(
366  Object::NumberValue(Cast<Number>(old_value)),
367  Object::NumberValue(Cast<Number>(actual_expected)))) {
368  return old_value;
369  }
370  // The pointer comparison failed, but the numbers are equal. This can
371  // happen even if both numbers are HeapNumbers with the same value.
372  // Try again in the next iteration.
373  actual_expected = old_value;
374  } while (true);
375 }

705 bool Object::SameNumberValue(double value1, double value2) {
706  // Compare values bitwise, to cover -0 being different from 0 -- we'd need to
707  // look at sign bits anyway if we'd done a double comparison, so we may as

709  uint64_t value1_bits = base::bit_cast<uint64_t>(value1);
710  uint64_t value2_bits = base::bit_cast<uint64_t>(value2);
711  if (value1_bits == value2_bits) {
712  return true;
713  }
714  // SameNumberValue(NaN, NaN) is true even for NaNs with different bit
715  // representations.

1878 Tagged<Object> Object::GetSimpleHash(Tagged<Object> object) {
1879  DisallowGarbageCollection no_gc;
1880  if (IsSmi(object)) {
1881  uint32_t hash = ComputeUnseededHash(Smi::ToInt(object));
1882  return Smi::FromInt(hash & Smi::kMaxValue);
1883  }
1884  auto instance_type = Cast<HeapObject>(object)->map()->instance_type();
1885  if (InstanceTypeChecker::IsHeapNumber(instance_type)) {
1886  double num = Cast<HeapNumber>(object)->value();
1887  if (std::isnan(num)) return Smi::FromInt(Smi::kMaxValue);
1888  // Use ComputeUnseededHash for all values in Signed32 range, including -0,
1889  // which is considered equal to 0 because collections use SameValueZero.
1890  uint32_t hash;
1891  // Check range before conversion to avoid undefined behavior.
1892  if (num >= kMinInt && num <= kMaxInt && FastI2D(FastD2I(num)) == num) {
1893  hash = ComputeUnseededHash(FastD2I(num));
1894  } else {
1895  hash = ComputeLongHash(base::double_to_uint64(num));
1896  }
1897  return Smi::FromInt(hash & Smi::kMaxValue);
1898  } else if (InstanceTypeChecker::IsName(instance_type)) {
1899  uint32_t hash = Cast<Name>(object)->EnsureHash();
1900  return Smi::FromInt(hash);
1901  } else if (InstanceTypeChecker::IsOddball(instance_type)) {
1902  uint32_t hash = Cast<Oddball>(object)->to_string()->EnsureHash();
1903  return Smi::FromInt(hash);
1904  } else if (InstanceTypeChecker::IsBigInt(instance_type)) {
1905  uint32_t hash = Cast<BigInt>(object)->Hash();
1906  return Smi::FromInt(hash & Smi::kMaxValue);
1907  } else if (InstanceTypeChecker::IsSharedFunctionInfo(instance_type)) {
1908  uint32_t hash = Cast<SharedFunctionInfo>(object)->Hash();
1909  return Smi::FromInt(hash & Smi::kMaxValue);
1910  } else if (InstanceTypeChecker::IsScopeInfo(instance_type)) {
1911  uint32_t hash = Cast<ScopeInfo>(object)->Hash();
1912  return Smi::FromInt(hash & Smi::kMaxValue);
1913  } else if (InstanceTypeChecker::IsScript(instance_type)) {
1914  int id = Cast<Script>(object)->id();
1915  return Smi::FromInt(ComputeUnseededHash(id) & Smi::kMaxValue);
1916  } else if (InstanceTypeChecker::IsTemplateInfo(instance_type)) {
1917  uint32_t hash = Cast<TemplateInfo>(object)->GetHash();

1919  return Smi::FromInt(hash);

1939 bool IsShared(Tagged<Object> obj) {
1940  // This logic should be kept in sync with fast paths in
1941  // CodeStubAssembler::SharedValueBarrier.
1942 
1943  // Smis are trivially shared.
1944  if (IsSmi(obj)) return true;
1945 
1946  Tagged<HeapObject> object = Cast<HeapObject>(obj);
1947 
1948  // RO objects are shared when the RO space is shared.
1949  if (HeapLayout::InReadOnlySpace(object)) {
1950  return true;
1951  }
1952 
1953  // Check if this object is already shared.
1954  InstanceType instance_type = object->map()->instance_type();
1955  if (InstanceTypeChecker::IsAlwaysSharedSpaceJSObject(instance_type)) {
1956  DCHECK(HeapLayout::InAnySharedSpace(object));
1957  return true;
1958  }
1959  switch (instance_type) {
1960  case SHARED_SEQ_TWO_BYTE_STRING_TYPE:
1961  case SHARED_SEQ_ONE_BYTE_STRING_TYPE:
1962  case SHARED_EXTERNAL_TWO_BYTE_STRING_TYPE:
1963  case SHARED_EXTERNAL_ONE_BYTE_STRING_TYPE:
1964  case SHARED_UNCACHED_EXTERNAL_TWO_BYTE_STRING_TYPE:
1965  case SHARED_UNCACHED_EXTERNAL_ONE_BYTE_STRING_TYPE:
1966  DCHECK(HeapLayout::InAnySharedSpace(object));
1967  return true;
1968 #if V8_ENABLE_WEBASSEMBLY
1969  case WASM_STRUCT_TYPE:
1970  case WASM_ARRAY_TYPE:
1971  return HeapLayout::InAnySharedSpace(object);
1972 #endif
1973  case INTERNALIZED_TWO_BYTE_STRING_TYPE:
1974  case INTERNALIZED_ONE_BYTE_STRING_TYPE:
1975  case EXTERNAL_INTERNALIZED_TWO_BYTE_STRING_TYPE:
1976  case EXTERNAL_INTERNALIZED_ONE_BYTE_STRING_TYPE:
1977  case UNCACHED_EXTERNAL_INTERNALIZED_TWO_BYTE_STRING_TYPE:
1978  case UNCACHED_EXTERNAL_INTERNALIZED_ONE_BYTE_STRING_TYPE:
1979  if (v8_flags.shared_string_table) {
1980  DCHECK(HeapLayout::InAnySharedSpace(object));
1981  return true;
1982  }
1983  return false;
1984  case HEAP_NUMBER_TYPE:
1985  return HeapLayout::InWritableSharedSpace(object);
