15  DirectHandle<Object> options,
16  const char* method_name) {

18  if (IsUndefined(*options, isolate)) {
19  // a. Return ! ObjectCreate(null).

22  // 2. If Type(options) is Object, then

45 Maybe<bool> GetStringOption(Isolate* isolate, DirectHandle<JSReceiver> options,
46  const char* property,
47  const std::span<const std::string_view> values,
48  const char* method_name,
49  DirectHandle<String>* result) {
50  DirectHandle<String> property_str =
51  isolate->factory()->NewStringFromAsciiChecked(property);
52 
53  // 1. Let value be ? Get(options, property).
54  DirectHandle<Object> value;
55  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
56  isolate, value,
57  Object::GetPropertyOrElement(isolate, options, property_str),
58  Nothing<bool>());
59 
60  if (IsUndefined(*value, isolate)) {
61  return Just(false);
62  }
63 
64  // 2. c. Let value be ? ToString(value).
65  DirectHandle<String> value_str;
66  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
67  isolate, value_str, Object::ToString(isolate, value), Nothing<bool>());
68 
69  // 2. d. if values is not undefined, then
70  if (!values.empty()) {
71  // 2. d. i. If values does not contain an element equal to value,
72  // throw a RangeError exception.
73  for (const auto& val : values) {
74  if (value_str->IsEqualTo(val, isolate)) {
75  // 2. e. return value
76  *result = value_str;
77  return Just(true);
78  }
79  }
80 
81  DirectHandle<String> method_str =
82  isolate->factory()->NewStringFromAsciiChecked(method_name);
83  THROW_NEW_ERROR_RETURN_VALUE(
84  isolate,
85  NewRangeError(MessageTemplate::kValueOutOfRange, value, method_str,
86  property_str),
87  Nothing<bool>());
88  }
89 
90  // 2. e. return value
91  *result = value_str;
92  return Just(true);
93 }

120 // ecma402/#sec-defaultnumberoption
121 Maybe<int> DefaultNumberOption(Isolate* isolate, DirectHandle<Object> value,
122  int min, int max, int fallback,
123  DirectHandle<String> property) {
124  // 2. Else, return fallback.
125  if (IsUndefined(*value)) return Just(fallback);
126 
127  // 1. If value is not undefined, then
128  // a. Let value be ? ToNumber(value).
129  DirectHandle<Number> value_num;
130  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
131  isolate, value_num, Object::ToNumber(isolate, value), Nothing<int>());
132  DCHECK(IsNumber(*value_num));
133 
134  // b. If value is NaN or less than minimum or greater than maximum, throw a
135  // RangeError exception.
136  if (IsNaN(*value_num) || Object::NumberValue(*value_num) < min ||
137  Object::NumberValue(*value_num) > max) {
138  THROW_NEW_ERROR_RETURN_VALUE(
139  isolate,
140  NewRangeError(MessageTemplate::kPropertyValueOutOfRange, property),
141  Nothing<int>());
142  }
143 
144  // The max and min arguments are integers and the above check makes
145  // sure that we are within the integer range making this double to
146  // int conversion safe.
147  //
148  // c. Return floor(value).
149  return Just(FastD2I(floor(Object::NumberValue(*value_num))));
150 }

167 Maybe<double> GetNumberOptionAsDouble(Isolate* isolate,
168  DirectHandle<JSReceiver> options,
169  DirectHandle<String> property,
170  double default_value) {
171  // 1. Let value be ? Get(options, property).
172  DirectHandle<Object> value;
173  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
174  isolate, value, JSReceiver::GetProperty(isolate, options, property),
175  Nothing<double>());
176  // 2. If value is undefined, then
177  if (IsUndefined(*value)) {
178  // b. Return default.
179  return Just(default_value);
180  }
181  // 4. Else if type is "number", then
182  // a. Set value to ? ToNumber(value).
183  DirectHandle<Number> value_num;
184  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
185  isolate, value_num, Object::ToNumber(isolate, value), Nothing<double>());
186  // b. If value is NaN, throw a RangeError exception.
187  if (IsNaN(*value_num)) {
188  THROW_NEW_ERROR_RETURN_VALUE(
189  isolate,
190  NewRangeError(MessageTemplate::kPropertyValueOutOfRange, property),
191  Nothing<double>());
192  }
193 
194  // 7. Return value.
195  return Just(Object::NumberValue(*value_num));
196 }
