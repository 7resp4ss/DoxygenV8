23 const char* Truncation::description() const {
24  switch (kind()) {
25  case TruncationKind::kNone:
26  return "no-value-use";
27  case TruncationKind::kBool:
28  return "truncate-to-bool";
29  case TruncationKind::kWord32:
30  return "truncate-to-word32";
31  case TruncationKind::kWord64:
32  return "truncate-to-word64";
33  case TruncationKind::kOddballAndBigIntToNumber:
34  switch (identify_zeros()) {
35  case kIdentifyZeros:
36  return "truncate-oddball&bigint-to-number (identify zeros)";
37  case kDistinguishZeros:
38  return "truncate-oddball&bigint-to-number (distinguish zeros)";
39  }
40  case TruncationKind::kBooleanAndNullAndBigIntToNumber:
41  switch (identify_zeros()) {
42  case kIdentifyZeros:
43  return "truncate-boolean&null&bigint-to-number (identify zeros)";
44  case kDistinguishZeros:
45  return "truncate-boolean&null&bigint-to-number (distinguish zeros)";
46  }
47  case TruncationKind::kAny:
48  switch (identify_zeros()) {
49  case kIdentifyZeros:
50  return "no-truncation (but identify zeros)";
51  case kDistinguishZeros:
52  return "no-truncation (but distinguish zeros)";
53  }
54  }
55  UNREACHABLE();
56 }

80 // static
81 Truncation::TruncationKind Truncation::Generalize(TruncationKind rep1,
82  TruncationKind rep2) {
83  if (LessGeneral(rep1, rep2)) return rep2;
84  if (LessGeneral(rep2, rep1)) return rep1;
85  // Handle the generalization of float64-representable values.
86  if (LessGeneral(rep1, TruncationKind::kOddballAndBigIntToNumber) &&
87  LessGeneral(rep2, TruncationKind::kOddballAndBigIntToNumber)) {
88  return TruncationKind::kOddballAndBigIntToNumber;
89  }
90  // Handle the generalization of any-representable values.
91  if (LessGeneral(rep1, TruncationKind::kAny) &&
92  LessGeneral(rep2, TruncationKind::kAny)) {
93  return TruncationKind::kAny;
94  }
95  // All other combinations are illegal.
96  FATAL("Tried to combine incompatible truncations");
97 }

109 // static
110 bool Truncation::LessGeneral(TruncationKind rep1, TruncationKind rep2) {
111  switch (rep1) {
112  case TruncationKind::kNone:
113  return true;
114  case TruncationKind::kBool:
115  return rep2 == TruncationKind::kBool || rep2 == TruncationKind::kAny;
116  case TruncationKind::kWord32:
117  return rep2 == TruncationKind::kWord32 ||
118  rep2 == TruncationKind::kWord64 ||
119  rep2 == TruncationKind::kOddballAndBigIntToNumber ||
120  rep2 == TruncationKind::kBooleanAndNullAndBigIntToNumber ||
121  rep2 == TruncationKind::kAny;
122  case TruncationKind::kWord64:
123  return rep2 == TruncationKind::kWord64 ||
124  rep2 == TruncationKind::kOddballAndBigIntToNumber ||
125  rep2 == TruncationKind::kBooleanAndNullAndBigIntToNumber ||
126  rep2 == TruncationKind::kAny;
127  case TruncationKind::kOddballAndBigIntToNumber:
128  return rep2 == TruncationKind::kOddballAndBigIntToNumber ||
129  rep2 == TruncationKind::kBooleanAndNullAndBigIntToNumber ||
130  rep2 == TruncationKind::kAny;
131  case TruncationKind::kBooleanAndNullAndBigIntToNumber:
132  return rep2 == TruncationKind::kBooleanAndNullAndBigIntToNumber ||
133  rep2 == TruncationKind::kAny;
134  case TruncationKind::kAny:
135  return rep2 == TruncationKind::kAny;
136  }
137  UNREACHABLE();
138 }

178 // Changes representation from {output_rep} to {use_rep}. The {truncation}
179 // parameter is only used for checking - if the changer cannot figure
180 // out signedness for the word32->float64 conversion, then we check that the
181 // uses truncate to word32 (so they do not care about signedness).
182 Node* RepresentationChanger::GetRepresentationFor(
183  Node* node, MachineRepresentation output_rep, Type output_type,
184  Node* use_node, UseInfo use_info) {
185  // We are currently not inserting conversions in machine graphs.
186  // We might add that, though.
187  DCHECK_IMPLIES(!output_type.IsNone(), !output_type.Is(Type::Machine()));
188  if (output_rep == MachineRepresentation::kNone && !output_type.IsNone()) {
189  // The output representation should be set if the type is inhabited (i.e.,
190  // if the value is possible).
191  return TypeError(node, output_rep, output_type, use_info.representation());
192  }
193 
194  if (output_type.Is(Type::BigInt()) &&
195  output_rep == MachineRepresentation::kWord64 &&
196  !TypeCheckIsBigInt(use_info.type_check())) {
197  // Rematerialize any truncated BigInt if user is not expecting a BigInt.
198  if (output_type.Is(Type::UnsignedBigInt64())) {
199  node = InsertConversion(node, simplified()->ChangeUint64ToBigInt(),
200  use_node);
201  } else {
202  node =
203  InsertConversion(node, simplified()->ChangeInt64ToBigInt(), use_node);
204  }
205  output_rep = MachineRepresentation::kTaggedPointer;
206  } else if (output_rep == MachineRepresentation::kFloat16RawBits) {
207  // Float16Array elements are loaded as raw bits in a word16 then converted
208  // to float64, since architectures have spotty support for fp16.
209  DCHECK(output_type.Is(Type::Number()));
210  if (machine()->ChangeFloat16RawBitsToFloat64().IsSupported()) {
211  node = jsgraph()->graph()->NewNode(
212  machine()->ChangeFloat16RawBitsToFloat64().op(), node);
213  } else {
214  node = InsertChangeFloat16RawBitsToFloat64Fallback(node);
215  }
216  output_rep = MachineRepresentation::kFloat64;
217  }
218 
219  // Handle the no-op shortcuts when no checking is necessary.
220  if (use_info.type_check() == TypeCheckKind::kNone ||
221  // TODO(nicohartmann@, chromium:1077804): Ignoring {use_info.type_check()}
222  // in case the representation already matches is not correct. For now,
223  // this behavior is disabled only for TypeCheckKind::kBigInt, but should
224  // be fixed for all other type checks.
225  (output_rep != MachineRepresentation::kWord32 &&
226  (use_info.type_check() != TypeCheckKind::kAdditiveSafeInteger &&
227  !TypeCheckIsBigInt(use_info.type_check())))) {
228  if (use_info.representation() == output_rep) {
229  // Representations are the same. That's a no-op.
230  return node;
231  }
232  if (IsWord(use_info.representation()) && IsWord(output_rep)) {
233  // Both are words less than or equal to 32-bits.
234  // Since loads of integers from memory implicitly sign or zero extend the
235  // value to the full machine word size and stores implicitly truncate,
236  // no representation change is necessary.
237  return node;
238  }
239  }
240 
241  switch (use_info.representation()) {
242  case MachineRepresentation::kTaggedSigned:
243  DCHECK(use_info.type_check() == TypeCheckKind::kNone ||
244  use_info.type_check() == TypeCheckKind::kSignedSmall);
245  return GetTaggedSignedRepresentationFor(node, output_rep, output_type,
246  use_node, use_info);
247  case MachineRepresentation::kTaggedPointer:
248  DCHECK(use_info.type_check() == TypeCheckKind::kNone ||
249  use_info.type_check() == TypeCheckKind::kHeapObject ||
250  use_info.type_check() == TypeCheckKind::kBigInt);
251  return GetTaggedPointerRepresentationFor(node, output_rep, output_type,
252  use_node, use_info);
253  case MachineRepresentation::kTagged:
254  DCHECK_EQ(TypeCheckKind::kNone, use_info.type_check());
255  return GetTaggedRepresentationFor(node, output_rep, output_type,
256  use_info.truncation());
257  case MachineRepresentation::kFloat16RawBits:
258  DCHECK_EQ(TypeCheckKind::kNone, use_info.type_check());
259  return GetFloat16RawBitsRepresentationFor(node, output_rep, output_type,
260  use_node, use_info);
261  case MachineRepresentation::kFloat32:
262  DCHECK_EQ(TypeCheckKind::kNone, use_info.type_check());
263  return GetFloat32RepresentationFor(node, output_rep, output_type,
264  use_info.truncation());
265  case MachineRepresentation::kFloat64:
266  DCHECK(use_info.type_check() == TypeCheckKind::kNone ||
267  use_info.type_check() == TypeCheckKind::kNumber ||
268  use_info.type_check() == TypeCheckKind::kNumberOrBoolean ||
269  use_info.type_check() == TypeCheckKind::kNumberOrOddball);
270  return GetFloat64RepresentationFor(node, output_rep, output_type,
271  use_node, use_info);
272  case MachineRepresentation::kBit:
273  DCHECK_EQ(TypeCheckKind::kNone, use_info.type_check());
274  return GetBitRepresentationFor(node, output_rep, output_type);
275  case MachineRepresentation::kWord8:
276  case MachineRepresentation::kWord16:
277  case MachineRepresentation::kWord32:
278  return GetWord32RepresentationFor(node, output_rep, output_type, use_node,
279  use_info);
280  case MachineRepresentation::kWord64:
281  DCHECK(use_info.type_check() == TypeCheckKind::kNone ||
282  use_info.type_check() == TypeCheckKind::kSigned64 ||
283  use_info.type_check() == TypeCheckKind::kAdditiveSafeInteger ||
284  TypeCheckIsBigInt(use_info.type_check()) ||
285  use_info.type_check() == TypeCheckKind::kArrayIndex);
286  return GetWord64RepresentationFor(node, output_rep, output_type, use_node,
287  use_info);
288  case MachineRepresentation::kSimd128:
289  case MachineRepresentation::kSimd256:
290  case MachineRepresentation::kNone:
291  return node;
292  case MachineRepresentation::kFloat16:
293  case MachineRepresentation::kCompressed:
294  case MachineRepresentation::kCompressedPointer:
295  case MachineRepresentation::kSandboxedPointer:
296  case MachineRepresentation::kProtectedPointer:
297  case MachineRepresentation::kIndirectPointer:
298  case MachineRepresentation::kMapWord:
299  UNREACHABLE();
300  }
301  UNREACHABLE();
302 }

304 Node* RepresentationChanger::GetTaggedSignedRepresentationFor(
305  Node* node, MachineRepresentation output_rep, Type output_type,
306  Node* use_node, UseInfo use_info) {
307  // Eagerly fold representation changes for constants.
308  switch (node->opcode()) {
309  case IrOpcode::kNumberConstant:
310  if (output_type.Is(Type::SignedSmall())) {
311  return node;
312  }
313  break;
314  default:
315  break;
316  }
317  // Select the correct X -> Tagged operator.
318  const Operator* op;
319  if (output_type.Is(Type::None())) {
320  // This is an impossible value; it should not be used at runtime.
321  return jsgraph()->graph()->NewNode(
322  jsgraph()->common()->DeadValue(MachineRepresentation::kTaggedSigned),
323  node);
324  } else if (IsWord(output_rep)) {
325  if (output_type.Is(Type::Signed31())) {
326  op = simplified()->ChangeInt31ToTaggedSigned();
327  } else if (output_type.Is(Type::Signed32())) {
328  if (SmiValuesAre32Bits()) {
329  op = simplified()->ChangeInt32ToTagged();
330  } else if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
331  op = simplified()->CheckedInt32ToTaggedSigned(use_info.feedback());
332  } else {
333  return TypeError(node, output_rep, output_type,
334  MachineRepresentation::kTaggedSigned);
335  }
336  } else if (output_type.Is(Type::Unsigned32()) &&
337  use_info.type_check() == TypeCheckKind::kSignedSmall) {
338  op = simplified()->CheckedUint32ToTaggedSigned(use_info.feedback());
339  } else {
340  return TypeError(node, output_rep, output_type,
341  MachineRepresentation::kTaggedSigned);
342  }
343  } else if (output_rep == MachineRepresentation::kWord64) {
344  if (output_type.Is(Type::Signed31())) {
345  // int64 -> int32 -> tagged signed
346  node = InsertTruncateInt64ToInt32(node);
347  op = simplified()->ChangeInt31ToTaggedSigned();
348  } else if (output_type.Is(Type::Signed32()) && SmiValuesAre32Bits()) {
349  // int64 -> int32 -> tagged signed
350  node = InsertTruncateInt64ToInt32(node);
351  op = simplified()->ChangeInt32ToTagged();
352  } else if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
353  if (output_type.Is(cache_->kPositiveSafeInteger)) {
354  op = simplified()->CheckedUint64ToTaggedSigned(use_info.feedback());
355  } else if (output_type.Is(cache_->kSafeInteger)) {
356  op = simplified()->CheckedInt64ToTaggedSigned(use_info.feedback());
357  } else {
358  return TypeError(node, output_rep, output_type,
359  MachineRepresentation::kTaggedSigned);
360  }
361  } else {
362  return TypeError(node, output_rep, output_type,
363  MachineRepresentation::kTaggedSigned);
364  }
365  } else if (output_rep == MachineRepresentation::kFloat64) {
366  if (output_type.Is(Type::Signed31())) {
367  // float64 -> int32 -> tagged signed
368  node = InsertChangeFloat64ToInt32(node);
369  op = simplified()->ChangeInt31ToTaggedSigned();
370  } else if (output_type.Is(Type::Signed32())) {
371  // float64 -> int32 -> tagged signed
372  node = InsertChangeFloat64ToInt32(node);
373  if (SmiValuesAre32Bits()) {
374  op = simplified()->ChangeInt32ToTagged();
375  } else if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
376  op = simplified()->CheckedInt32ToTaggedSigned(use_info.feedback());
377  } else {
378  return TypeError(node, output_rep, output_type,
379  MachineRepresentation::kTaggedSigned);
380  }
381  } else if (output_type.Is(Type::Unsigned32()) &&
382  use_info.type_check() == TypeCheckKind::kSignedSmall) {
383  // float64 -> uint32 -> tagged signed
384  node = InsertChangeFloat64ToUint32(node);
385  op = simplified()->CheckedUint32ToTaggedSigned(use_info.feedback());
386  } else if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
387  node = InsertCheckedFloat64ToInt32(
388  node,
389  output_type.Maybe(Type::MinusZero())
390  ? CheckForMinusZeroMode::kCheckForMinusZero
391  : CheckForMinusZeroMode::kDontCheckForMinusZero,
392  use_info.feedback(), use_node);
393  if (SmiValuesAre32Bits()) {
394  op = simplified()->ChangeInt32ToTagged();
395  } else {
396  op = simplified()->CheckedInt32ToTaggedSigned(use_info.feedback());
397  }
398  } else {
399  return TypeError(node, output_rep, output_type,
400  MachineRepresentation::kTaggedSigned);
401  }
402  } else if (output_rep == MachineRepresentation::kFloat32) {
403  if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
404  node = InsertChangeFloat32ToFloat64(node);
405  node = InsertCheckedFloat64ToInt32(
406  node,
407  output_type.Maybe(Type::MinusZero())
408  ? CheckForMinusZeroMode::kCheckForMinusZero
409  : CheckForMinusZeroMode::kDontCheckForMinusZero,
410  use_info.feedback(), use_node);
411  if (SmiValuesAre32Bits()) {
412  op = simplified()->ChangeInt32ToTagged();
413  } else {
414  op = simplified()->CheckedInt32ToTaggedSigned(use_info.feedback());
415  }
416  } else {
417  return TypeError(node, output_rep, output_type,
418  MachineRepresentation::kTaggedSigned);
419  }
420  } else if (CanBeTaggedPointer(output_rep)) {
421  if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
422  op = simplified()->CheckedTaggedToTaggedSigned(use_info.feedback());
423  } else if (output_type.Is(Type::SignedSmall())) {
424  op = simplified()->ChangeTaggedToTaggedSigned();
425  } else {
426  return TypeError(node, output_rep, output_type,
427  MachineRepresentation::kTaggedSigned);
428  }
429  } else if (output_rep == MachineRepresentation::kBit) {
430  if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
431  // TODO(turbofan): Consider adding a Bailout operator that just deopts.
432  // Also use that for MachineRepresentation::kPointer case above.
433  node = InsertChangeBitToTagged(node);
434  op = simplified()->CheckedTaggedToTaggedSigned(use_info.feedback());
435  } else {
436  return TypeError(node, output_rep, output_type,
437  MachineRepresentation::kTaggedSigned);
438  }
439  } else {
440  return TypeError(node, output_rep, output_type,
441  MachineRepresentation::kTaggedSigned);
442  }
443  return InsertConversion(node, op, use_node);
444 }

446 Node* RepresentationChanger::GetTaggedPointerRepresentationFor(
447  Node* node, MachineRepresentation output_rep, Type output_type,
448  Node* use_node, UseInfo use_info) {
449  // Eagerly fold representation changes for constants.
450  switch (node->opcode()) {
451  case IrOpcode::kHeapConstant:
452  if (TypeCheckIsBigInt(use_info.type_check())) break;
453  return node; // No change necessary.
454  case IrOpcode::kInt32Constant:
455  case IrOpcode::kFloat64Constant:
456  case IrOpcode::kFloat32Constant:
457  UNREACHABLE();
458  default:
459  break;
460  }
461  // Select the correct X -> TaggedPointer operator.
462  Operator const* op;
463  if (output_type.Is(Type::None())) {
464  // This is an impossible value; it should not be used at runtime.
465  return jsgraph()->graph()->NewNode(
466  jsgraph()->common()->DeadValue(MachineRepresentation::kTaggedPointer),
467  node);
468  }
469 
470  if (TypeCheckIsBigInt(use_info.type_check()) &&
471  !output_type.Is(Type::BigInt())) {
472  // BigInt checks can only be performed on tagged representations. Note that
473  // a corresponding check is inserted down below.
474  if (!CanBeTaggedPointer(output_rep)) {
475  Node* unreachable =
476  InsertUnconditionalDeopt(use_node, DeoptimizeReason::kNotABigInt);
477  return jsgraph()->graph()->NewNode(
478  jsgraph()->common()->DeadValue(MachineRepresentation::kTaggedPointer),
479  unreachable);
480  }
481  }
482 
483  if (output_rep == MachineRepresentation::kBit) {
484  if (output_type.Is(Type::Boolean())) {
485  op = simplified()->ChangeBitToTagged();
486  } else {
487  return TypeError(node, output_rep, output_type,
488  MachineRepresentation::kTagged);
489  }
490  } else if (IsWord(output_rep)) {
491  if (output_type.Is(Type::Unsigned32())) {
492  // uint32 -> float64 -> tagged
493  node = InsertChangeUint32ToFloat64(node);
494  } else if (output_type.Is(Type::Signed32())) {
495  // int32 -> float64 -> tagged
496  node = InsertChangeInt32ToFloat64(node);
497  } else {
498  return TypeError(node, output_rep, output_type,
499  MachineRepresentation::kTaggedPointer);
500  }
501  op = simplified()->ChangeFloat64ToTaggedPointer();
502  } else if (output_rep == MachineRepresentation::kWord64) {
503  if (output_type.Is(cache_->kSafeInteger)) {
504  // int64 -> float64 -> tagged pointer
505  op = machine()->ChangeInt64ToFloat64();
506  node = jsgraph()->graph()->NewNode(op, node);
507  op = simplified()->ChangeFloat64ToTaggedPointer();
508  } else if (output_type.Is(Type::SignedBigInt64()) &&
509  use_info.type_check() == TypeCheckKind::kBigInt) {
510  op = simplified()->ChangeInt64ToBigInt();
511  } else if (output_type.Is(Type::UnsignedBigInt64()) &&
512  use_info.type_check() == TypeCheckKind::kBigInt) {
513  op = simplified()->ChangeUint64ToBigInt();
514  } else {
515  return TypeError(node, output_rep, output_type,
516  MachineRepresentation::kTaggedPointer);
517  }
518  } else if (output_rep == MachineRepresentation::kFloat32) {
519  if (output_type.Is(Type::Number())) {
520  // float32 -> float64 -> tagged
521  node = InsertChangeFloat32ToFloat64(node);
522  op = simplified()->ChangeFloat64ToTaggedPointer();
523  } else {
524  return TypeError(node, output_rep, output_type,
525  MachineRepresentation::kTaggedPointer);
526  }
527  } else if (output_rep == MachineRepresentation::kFloat64) {
528  if (output_type.Is(Type::Number())) {
529  // float64 -> tagged
530  op = simplified()->ChangeFloat64ToTaggedPointer();
531  } else {
532  return TypeError(node, output_rep, output_type,
533  MachineRepresentation::kTaggedPointer);
534  }
535  } else if (IsAnyTagged(output_rep)) {
536  if (use_info.type_check() == TypeCheckKind::kBigInt) {
537  if (output_type.Is(Type::BigInt())) {
538  DCHECK_NE(output_rep, MachineRepresentation::kTaggedSigned);
539  return node;
540  }
541  op = simplified()->CheckBigInt(use_info.feedback());
542  } else if (use_info.type_check() == TypeCheckKind::kBigInt64) {
543  if (output_type.Is(Type::SignedBigInt64())) {
544  DCHECK_NE(output_rep, MachineRepresentation::kTaggedSigned);
545  return node;
546  }
547  if (!output_type.Is(Type::BigInt())) {
548  node = InsertConversion(
549  node, simplified()->CheckBigInt(use_info.feedback()), use_node);
550  }
551  op = simplified()->CheckedBigIntToBigInt64(use_info.feedback());
552  } else if (output_rep == MachineRepresentation::kTaggedPointer ||
553  !output_type.Maybe(Type::SignedSmall())) {
554  DCHECK_NE(output_rep, MachineRepresentation::kTaggedSigned);
555  return node;
556  } else {
557  op = simplified()->CheckedTaggedToTaggedPointer(use_info.feedback());
558  }
559  } else {
560  return TypeError(node, output_rep, output_type,
561  MachineRepresentation::kTaggedPointer);
562  }
563  return InsertConversion(node, op, use_node);
564 }

566 Node* RepresentationChanger::GetTaggedRepresentationFor(
567  Node* node, MachineRepresentation output_rep, Type output_type,
568  Truncation truncation) {
569  // Eagerly fold representation changes for constants.
570  switch (node->opcode()) {
571  case IrOpcode::kNumberConstant:
572  case IrOpcode::kHeapConstant:
573  return node; // No change necessary.
574  case IrOpcode::kInt32Constant:
575  case IrOpcode::kFloat64Constant:
576  case IrOpcode::kFloat32Constant:
577  UNREACHABLE();
578  default:
579  break;
580  }
581  if (output_rep == MachineRepresentation::kTaggedSigned ||
582  output_rep == MachineRepresentation::kTaggedPointer ||
583  output_rep == MachineRepresentation::kMapWord) {
584  // this is a no-op.
585  return node;
586  }
587  // Select the correct X -> Tagged operator.
588  const Operator* op;
589  if (output_type.Is(Type::None())) {
590  // This is an impossible value; it should not be used at runtime.
591  return jsgraph()->graph()->NewNode(
592  jsgraph()->common()->DeadValue(MachineRepresentation::kTagged), node);
593  } else if (output_rep == MachineRepresentation::kBit) {
594  if (output_type.Is(Type::Boolean())) {
595  op = simplified()->ChangeBitToTagged();
596  } else {
597  return TypeError(node, output_rep, output_type,
598  MachineRepresentation::kTagged);
599  }
600  } else if (IsWord(output_rep)) {
601  if (output_type.Is(Type::Signed31())) {
602  op = simplified()->ChangeInt31ToTaggedSigned();
603  } else if (output_type.Is(Type::Signed32()) ||
604  (output_type.Is(Type::Signed32OrMinusZero()) &&
605  truncation.IdentifiesZeroAndMinusZero())) {
606  op = simplified()->ChangeInt32ToTagged();
607  } else if (output_type.Is(Type::Unsigned32()) ||
608  (output_type.Is(Type::Unsigned32OrMinusZero()) &&
609  truncation.IdentifiesZeroAndMinusZero()) ||
610  truncation.IsUsedAsWord32()) {
611  // Either the output is uint32 or the uses only care about the
612  // low 32 bits (so we can pick uint32 safely).
613  op = simplified()->ChangeUint32ToTagged();
614  } else {
615  return TypeError(node, output_rep, output_type,
616  MachineRepresentation::kTagged);
617  }
618  } else if (output_rep == MachineRepresentation::kWord64) {
619  if (output_type.Is(Type::Signed31())) {
620  // int64 -> int32 -> tagged signed
621  node = InsertTruncateInt64ToInt32(node);
622  op = simplified()->ChangeInt31ToTaggedSigned();
623  } else if (output_type.Is(Type::Signed32())) {
624  // int64 -> int32 -> tagged
625  node = InsertTruncateInt64ToInt32(node);
626  op = simplified()->ChangeInt32ToTagged();
627  } else if (output_type.Is(Type::Unsigned32())) {
628  // int64 -> uint32 -> tagged
629  node = InsertTruncateInt64ToInt32(node);
630  op = simplified()->ChangeUint32ToTagged();
631  } else if (output_type.Is(cache_->kPositiveSafeInteger)) {
632  // uint64 -> tagged
633  op = simplified()->ChangeUint64ToTagged();
634  } else if (output_type.Is(cache_->kSafeInteger)) {
635  // int64 -> tagged
636  op = simplified()->ChangeInt64ToTagged();
637  } else if (output_type.Is(Type::SignedBigInt64())) {
638  // int64 -> BigInt
639  op = simplified()->ChangeInt64ToBigInt();
640  } else if (output_type.Is(Type::UnsignedBigInt64())) {
641  // uint64 -> BigInt
642  op = simplified()->ChangeUint64ToBigInt();
643  } else {
644  return TypeError(node, output_rep, output_type,
645  MachineRepresentation::kTagged);
646  }
647  } else if (output_rep ==
648  MachineRepresentation::kFloat32) { // float32 -> float64 -> tagged
649  node = InsertChangeFloat32ToFloat64(node);
650  op = simplified()->ChangeFloat64ToTagged(
651  output_type.Maybe(Type::MinusZero())
652  ? CheckForMinusZeroMode::kCheckForMinusZero
653  : CheckForMinusZeroMode::kDontCheckForMinusZero);
654  } else if (output_rep == MachineRepresentation::kFloat64) {
655  if (output_type.Is(Type::Signed31())) { // float64 -> int32 -> tagged
656  node = InsertChangeFloat64ToInt32(node);
657  op = simplified()->ChangeInt31ToTaggedSigned();
658  } else if (output_type.Is(
659  Type::Signed32())) { // float64 -> int32 -> tagged
660  node = InsertChangeFloat64ToInt32(node);
661  op = simplified()->ChangeInt32ToTagged();
662  } else if (output_type.Is(
663  Type::Unsigned32())) { // float64 -> uint32 -> tagged
664  node = InsertChangeFloat64ToUint32(node);
665  op = simplified()->ChangeUint32ToTagged();
666  } else if (output_type.Is(Type::Number()) ||
667  (output_type.Is(Type::NumberOrOddball()) &&
668  truncation.TruncatesOddballAndBigIntToNumber())) {
669  op = simplified()->ChangeFloat64ToTagged(
670  output_type.Maybe(Type::MinusZero())
671  ? CheckForMinusZeroMode::kCheckForMinusZero
672  : CheckForMinusZeroMode::kDontCheckForMinusZero);
673  } else if (output_type.Is(Type::NumberOrUndefined())) {
674  op = simplified()->ChangeFloat64OrUndefinedToTagged(
675  output_type.Maybe(Type::MinusZero())
676  ? CheckForMinusZeroMode::kCheckForMinusZero
677  : CheckForMinusZeroMode::kDontCheckForMinusZero);
678  } else {
679  return TypeError(node, output_rep, output_type,
680  MachineRepresentation::kTagged);
681  }
682  } else {
683  return TypeError(node, output_rep, output_type,
684  MachineRepresentation::kTagged);
685  }
686  return jsgraph()->graph()->NewNode(op, node);
687 }

791 Node* RepresentationChanger::GetFloat64RepresentationFor(
792  Node* node, MachineRepresentation output_rep, Type output_type,
793  Node* use_node, UseInfo use_info) {
794  NumberMatcher m(node);
795  if (m.HasResolvedValue()) {
796  // BigInts are not used as number constants.
797  DCHECK(!TypeCheckIsBigInt(use_info.type_check()));
798  switch (use_info.type_check()) {
799  case TypeCheckKind::kNone:
800  case TypeCheckKind::kNumber:
801  case TypeCheckKind::kNumberOrBoolean:
802  case TypeCheckKind::kNumberOrOddball:
803  return jsgraph()->Float64Constant(m.ResolvedValue());
804  case TypeCheckKind::kBigInt:
805  case TypeCheckKind::kBigInt64:
806  case TypeCheckKind::kHeapObject:
807  case TypeCheckKind::kSigned32:
808  case TypeCheckKind::kSigned64:
809  case TypeCheckKind::kSignedSmall:
810  case TypeCheckKind::kArrayIndex:
811  case TypeCheckKind::kAdditiveSafeInteger:
812  break;
813  }
814  }
815 
816  // Select the correct X -> Float64 operator.
817  const Operator* op = nullptr;
818  if (output_type.Is(Type::None())) {
819  // This is an impossible value; it should not be used at runtime.
820  return jsgraph()->graph()->NewNode(
821  jsgraph()->common()->DeadValue(MachineRepresentation::kFloat64), node);
822  } else if (IsWord(output_rep)) {
823  if (output_type.Is(Type::Signed32()) ||
824  (output_type.Is(Type::Signed32OrMinusZero()) &&
825  use_info.truncation().IdentifiesZeroAndMinusZero())) {
826  op = machine()->ChangeInt32ToFloat64();
827  } else if (output_type.Is(Type::Unsigned32()) ||
828  (output_type.Is(Type::Unsigned32OrMinusZero()) &&
829  use_info.truncation().IdentifiesZeroAndMinusZero()) ||
830  use_info.truncation().IsUsedAsWord32()) {
831  // Either the output is uint32 or the uses only care about the
832  // low 32 bits (so we can pick uint32 safely).
833  op = machine()->ChangeUint32ToFloat64();
834  }
835  } else if (output_rep == MachineRepresentation::kBit) {
836  CHECK(output_type.Is(Type::Boolean()));
837  if (use_info.truncation().TruncatesOddballAndBigIntToNumber() ||
838  use_info.type_check() == TypeCheckKind::kNumberOrBoolean ||
839  use_info.type_check() == TypeCheckKind::kNumberOrOddball) {
840  op = machine()->ChangeUint32ToFloat64();
841  } else {
842  CHECK_NE(use_info.type_check(), TypeCheckKind::kNone);
843  Node* unreachable =
844  InsertUnconditionalDeopt(use_node, DeoptimizeReason::kNotAHeapNumber);
845  return jsgraph()->graph()->NewNode(
846  jsgraph()->common()->DeadValue(MachineRepresentation::kFloat64),
847  unreachable);
848  }
849  } else if (IsAnyTagged(output_rep)) {
850  if (output_type.Is(Type::Undefined())) {
851  if (use_info.type_check() == TypeCheckKind::kNumberOrOddball ||
852  (use_info.type_check() == TypeCheckKind::kNone &&
853  use_info.truncation().TruncatesOddballAndBigIntToNumber())) {
854  return jsgraph()->Float64Constant(
855  std::numeric_limits<double>::quiet_NaN());
856  } else if (use_info.truncation()
857  .TruncatesBooleanAndNullAndBigIntToNumber()) {
858 #ifdef V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
859  return jsgraph()->Float64Constant(UndefinedNan());
860 #else
861  UNREACHABLE();
862 #endif // V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
863  } else {
864  DCHECK(use_info.type_check() == TypeCheckKind::kNone ||
865  use_info.type_check() == TypeCheckKind::kNumber ||
866  use_info.type_check() == TypeCheckKind::kNumberOrBoolean);
867  Node* unreachable = InsertUnconditionalDeopt(
868  use_node, use_info.type_check() == TypeCheckKind::kNumber
869  ? DeoptimizeReason::kNotANumber
870  : DeoptimizeReason::kNotANumberOrBoolean);
871  return jsgraph()->graph()->NewNode(
872  jsgraph()->common()->DeadValue(MachineRepresentation::kFloat64),
873  unreachable);
874  }
875  } else if (output_rep == MachineRepresentation::kTaggedSigned) {
876  node = InsertChangeTaggedSignedToInt32(node);
877  op = machine()->ChangeInt32ToFloat64();
878  } else if (output_type.Is(Type::Number())) {
879  op = simplified()->ChangeTaggedToFloat64();
880  } else if ((output_type.Is(Type::NumberOrOddball()) &&
881  use_info.truncation().TruncatesOddballAndBigIntToNumber()) ||
882  output_type.Is(Type::NumberOrHole())) {
883  // JavaScript 'null' is an Oddball that results in +0 when truncated to
884  // Number. In a context like -0 == null, which must evaluate to false,
885  // this truncation must not happen. For this reason we restrict this
886  // case to when either the user explicitly requested a float (and thus
887  // wants +0 if null is the input) or we know from the types that the
888  // input can only be Number | Hole. The latter is necessary to handle
889  // the operator CheckFloat64Hole. We did not put in the type (Number |
890  // Oddball \ Null) to discover more bugs related to this conversion via
891  // crashes.
892  op = simplified()->TruncateTaggedToFloat64();
893  } else if (output_type.Is(Type::NumberOrUndefined()) &&
894  use_info.truncation()
895  .TruncatesBooleanAndNullAndBigIntToNumber()) {
896  op = simplified()->TruncateTaggedToFloat64PreserveUndefined();
897  } else if (use_info.type_check() == TypeCheckKind::kNumber ||
898  (use_info.type_check() == TypeCheckKind::kNumberOrOddball &&
899  !output_type.Maybe(Type::BooleanOrNullOrNumber()))) {
900  op = simplified()->CheckedTaggedToFloat64(CheckTaggedInputMode::kNumber,
901  use_info.feedback());
902  } else if (use_info.type_check() == TypeCheckKind::kNumberOrBoolean) {
903  op = simplified()->CheckedTaggedToFloat64(
904  CheckTaggedInputMode::kNumberOrBoolean, use_info.feedback());
905  } else if (use_info.type_check() == TypeCheckKind::kNumberOrOddball) {
906  op = simplified()->CheckedTaggedToFloat64(
907  CheckTaggedInputMode::kNumberOrOddball, use_info.feedback());
908  }
909  } else if (output_rep == MachineRepresentation::kFloat32) {
910  op = machine()->ChangeFloat32ToFloat64();
911  } else if (output_rep == MachineRepresentation::kWord64) {
912  if (output_type.Is(cache_->kSafeInteger)) {
913  op = machine()->ChangeInt64ToFloat64();
914  }
915  }
916  if (op == nullptr) {
917  return TypeError(node, output_rep, output_type,
918  MachineRepresentation::kFloat64);
919  }
920  return InsertConversion(node, op, use_node);
921 }

1093  CheckTaggedInputMode::kNumberOrOddball, use_info.feedback());
1094  } else {
1095  return TypeError(node, output_rep, output_type,
1096  MachineRepresentation::kWord32);
1097  }
1098  } else {
1099  return TypeError(node, output_rep, output_type,
1100  MachineRepresentation::kWord32);
1101  }
1102  } else if (output_rep == MachineRepresentation::kWord32) {
1103  // Only the checked case should get here, the non-checked case is
1104  // handled in GetRepresentationFor.
1105  if (use_info.type_check() == TypeCheckKind::kSignedSmall ||

1113  (identify_zeros &&
1114  output_type.Is(Type::Unsigned32OrMinusZero()))) {
1115  op = simplified()->CheckedUint32ToInt32(use_info.feedback());

1117  return TypeError(node, output_rep, output_type,
1118  MachineRepresentation::kWord32);
1119  }
1120  } else if (use_info.type_check() == TypeCheckKind::kAdditiveSafeInteger ||
1121  use_info.type_check() == TypeCheckKind::kNumber ||
1122  use_info.type_check() == TypeCheckKind::kNumberOrOddball) {
1123  return node;
1124  }
1125  } else if (output_rep == MachineRepresentation::kWord8 ||
1126  output_rep == MachineRepresentation::kWord16) {
1127  DCHECK_EQ(MachineRepresentation::kWord32, use_info.representation());
1128  DCHECK(use_info.type_check() == TypeCheckKind::kSignedSmall ||
1129  use_info.type_check() == TypeCheckKind::kSigned32 ||
1130  use_info.type_check() == TypeCheckKind::kAdditiveSafeInteger);
1131  return node;
1132  } else if (output_rep == MachineRepresentation::kWord64) {
1133  if (output_type.Is(Type::Signed32()) ||
1134  (output_type.Is(Type::Unsigned32()) &&
1135  use_info.type_check() == TypeCheckKind::kNone) ||
1136  (output_type.Is(cache_->kSafeInteger) &&
1137  use_info.truncation().IsUsedAsWord32())) {
1138  op = machine()->TruncateInt64ToInt32();
1139  } else if (use_info.type_check() == TypeCheckKind::kSignedSmall ||
1140  use_info.type_check() == TypeCheckKind::kSigned32 ||
1141  use_info.type_check() == TypeCheckKind::kArrayIndex) {
1142  if (output_type.Is(cache_->kPositiveSafeInteger)) {
1143  op = simplified()->CheckedUint64ToInt32(use_info.feedback());
1144  } else if (output_type.Is(cache_->kSafeInteger)) {
1145  op = simplified()->CheckedInt64ToInt32(use_info.feedback());
1146  } else {
1147  return TypeError(node, output_rep, output_type,
1148  MachineRepresentation::kWord32);
1149  }
1150  } else {
1151  return TypeError(node, output_rep, output_type,
1152  MachineRepresentation::kWord32);
1153  }
1154  }
1155 
1156  if (op == nullptr) {
1157  return TypeError(node, output_rep, output_type,
1158  MachineRepresentation::kWord32);
1159  }
1160  return InsertConversion(node, op, use_node);

1177 Node* RepresentationChanger::GetBitRepresentationFor(
1178  Node* node, MachineRepresentation output_rep, Type output_type) {
1179  // Eagerly fold representation changes for constants.
1180  switch (node->opcode()) {
1181  case IrOpcode::kHeapConstant: {
1182  HeapObjectMatcher m(node);
1183  if (m.Is(factory()->false_value())) {
1184  return InsertTypeOverrideForVerifier(
1185  Type::Constant(broker_, broker_->false_value(), jsgraph()->zone()),
1186  jsgraph()->Int32Constant(0));
1187  } else if (m.Is(factory()->true_value())) {
1188  return InsertTypeOverrideForVerifier(
1189  Type::Constant(broker_, broker_->true_value(), jsgraph()->zone()),
1190  jsgraph()->Int32Constant(1));
1191  }
1192  break;
1193  }
1194  default:
1195  break;
1196  }
1197  // Select the correct X -> Bit operator.
1198  const Operator* op;
1199  if (output_type.Is(Type::None())) {
1200  // This is an impossible value; it should not be used at runtime.
1201  return jsgraph()->graph()->NewNode(
1202  jsgraph()->common()->DeadValue(MachineRepresentation::kBit), node);
1203  } else if (output_rep == MachineRepresentation::kTagged ||
1204  output_rep == MachineRepresentation::kTaggedPointer) {
1205  if (output_type.Is(Type::BooleanOrNullOrUndefined())) {
1206  // true is the only trueish Oddball.
1207  op = simplified()->ChangeTaggedToBit();
1208  } else {
1209  if (output_rep == MachineRepresentation::kTagged &&
1210  output_type.Maybe(Type::SignedSmall())) {
1211  op = simplified()->TruncateTaggedToBit();
1212  } else {
1213  // The {output_type} either doesn't include the Smi range,
1214  // or the {output_rep} is known to be TaggedPointer.
1215  op = simplified()->TruncateTaggedPointerToBit();
1216  }
1217  }
1218  } else if (output_rep == MachineRepresentation::kTaggedSigned) {
1219  if (COMPRESS_POINTERS_BOOL) {
1220  node = jsgraph()->graph()->NewNode(machine()->Word32Equal(), node,
1221  jsgraph()->Int32Constant(0));
1222  } else {
1223  node = jsgraph()->graph()->NewNode(machine()->WordEqual(), node,
1224  jsgraph()->IntPtrConstant(0));
1225  }
1226  return jsgraph()->graph()->NewNode(machine()->Word32Equal(), node,
1227  jsgraph()->Int32Constant(0));
1228  } else if (IsWord(output_rep)) {
1229  node = jsgraph()->graph()->NewNode(machine()->Word32Equal(), node,
1230  jsgraph()->Int32Constant(0));
1231  return jsgraph()->graph()->NewNode(machine()->Word32Equal(), node,
1232  jsgraph()->Int32Constant(0));
1233  } else if (output_rep == MachineRepresentation::kWord64) {
1234  node = jsgraph()->graph()->NewNode(machine()->Word64Equal(), node,
1235  jsgraph()->Int64Constant(0));
1236  return jsgraph()->graph()->NewNode(machine()->Word32Equal(), node,
1237  jsgraph()->Int32Constant(0));
1238  } else if (output_rep == MachineRepresentation::kFloat32) {
1239  node = jsgraph()->graph()->NewNode(machine()->Float32Abs(), node);
1240  return jsgraph()->graph()->NewNode(machine()->Float32LessThan(),
1241  jsgraph()->Float32Constant(0.0), node);
1242  } else if (output_rep == MachineRepresentation::kFloat64) {
1243  node = jsgraph()->graph()->NewNode(machine()->Float64Abs(), node);
1244  return jsgraph()->graph()->NewNode(machine()->Float64LessThan(),
1245  jsgraph()->Float64Constant(0.0), node);
1246  } else {
1247  return TypeError(node, output_rep, output_type,
1248  MachineRepresentation::kBit);
1249  }
1250  return jsgraph()->graph()->NewNode(op, node);
1251 }
