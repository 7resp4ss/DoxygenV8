131 // Abstracts details of building TurboFan graph nodes for wasm to separate
132 // the wasm decoder from the internal details of TurboFan.

134  public:
135  // ParameterMode specifies how the instance is passed.
136  enum ParameterMode {
137  // Normal wasm functions pass the instance as an implicit first parameter.
138  kInstanceParameterMode,
139  // For Wasm-to-JS and C-API wrappers, a {WasmImportData} object is
140  // passed as first parameter.
141  kWasmImportDataMode,
142  // For JS-to-Wasm wrappers (which are JS functions), we load the Wasm
143  // instance from the JS function data. The generated code objects live on
144  // the JS heap, so those compilation pass an isolate.
145  kJSFunctionAbiMode,
146  // The JS-to-JS wrapper does not have an associated instance.
147  // The C-entry stub uses a custom ABI (see {CWasmEntryParameters}).
148  kNoSpecialParameterMode
149  };
