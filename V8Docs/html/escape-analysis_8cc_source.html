336 void EffectGraphReducer::ReduceFrom(Node* node) {
337  // Perform DFS and eagerly trigger revisitation as soon as possible.
338  // A stack element {node, i} indicates that input i of node should be visited
339  // next.
340  DCHECK(stack_.empty());
341  stack_.push({node, 0});
342  while (!stack_.empty()) {
343  tick_counter_->TickAndMaybeEnterSafepoint();
344  Node* current = stack_.top().node;
345  int& input_index = stack_.top().input_index;
346  if (input_index < current->InputCount()) {
347  Node* input = current->InputAt(input_index);
348  input_index++;
349  switch (state_.Get(input)) {
350  case State::kVisited:
351  // The input is already reduced.
352  break;
353  case State::kOnStack:
354  // The input is on the DFS stack right now, so it will be revisited
355  // later anyway.
356  break;
357  case State::kUnvisited:
358  case State::kRevisit: {
359  state_.Set(input, State::kOnStack);
360  stack_.push({input, 0});
361  break;
362  }
363  }
364  } else {
365  stack_.pop();
366  Reduction reduction;
367  reduce_(current, &reduction);
368  for (Edge edge : current->use_edges()) {
369  // Mark uses for revisitation.
370  Node* use = edge.from();
371  if (NodeProperties::IsEffectEdge(edge)) {
372  if (reduction.effect_changed()) Revisit(use);
373  } else {
374  if (reduction.value_changed()) Revisit(use);
375  }
376  }
377  state_.Set(current, State::kVisited);
378  // Process the revisitation buffer immediately. This improves performance
379  // of escape analysis. Using a stack for {revisit_} reverses the order in
380  // which the revisitation happens. This also seems to improve performance.
381  while (!revisit_.empty()) {
382  Node* revisit = revisit_.top();
383  if (state_.Get(revisit) == State::kRevisit) {
384  state_.Set(revisit, State::kOnStack);
385  stack_.push({revisit, 0});
386  }
387  revisit_.pop();
388  }
389  }
390  }

439 VariableTracker::State VariableTracker::MergeInputs(Node* effect_phi) {
440  // A variable that is mapped to [nullptr] was not assigned a value on every
441  // execution path to the current effect phi. Relying on the invariant that
442  // every variable is initialized (at least with a sentinel like the Dead
443  // node), this means that the variable initialization does not dominate the
444  // current point. So for loop effect phis, we can keep nullptr for a variable
445  // as long as the first input of the loop has nullptr for this variable. For
446  // non-loop effect phis, we can even keep it nullptr as long as any input has
447  // nullptr.
448  DCHECK_EQ(IrOpcode::kEffectPhi, effect_phi->opcode());
449  int arity = effect_phi->op()->EffectInputCount();
450  Node* control = NodeProperties::GetControlInput(effect_phi, 0);
451  TRACE("control: %s#%d\n", control->op()->mnemonic(), control->id());
452  bool is_loop = control->opcode() == IrOpcode::kLoop;
453  buffer_.reserve(arity + 1);
454 
455  State first_input = table_.Get(NodeProperties::GetEffectInput(effect_phi, 0));
456  State result = first_input;
457  for (std::pair<Variable, Node*> var_value : first_input) {
458  tick_counter_->TickAndMaybeEnterSafepoint();
459  if (Node* value = var_value.second) {
460  Variable var = var_value.first;
461  TRACE("var %i:\n", var.id_);
462  buffer_.clear();
463  buffer_.push_back(value);
464  bool identical_inputs = true;
465  int num_defined_inputs = 1;
466  TRACE(" input 0: %s#%d\n", value->op()->mnemonic(), value->id());
467  for (int i = 1; i < arity; ++i) {
468  Node* next_value =
469  table_.Get(NodeProperties::GetEffectInput(effect_phi, i)).Get(var);
470  if (next_value != value) identical_inputs = false;
471  if (next_value != nullptr) {
472  num_defined_inputs++;
473  TRACE(" input %i: %s#%d\n", i, next_value->op()->mnemonic(),
474  next_value->id());
475  } else {
476  TRACE(" input %i: nullptr\n", i);
477  }
478  buffer_.push_back(next_value);
479  }
480 
481  Node* old_value = table_.Get(effect_phi).Get(var);
482  if (old_value) {
483  TRACE(" old: %s#%d\n", old_value->op()->mnemonic(), old_value->id());
484  } else {
485  TRACE(" old: nullptr\n");
486  }
487  // Reuse a previously created phi node if possible.
488  if (old_value && old_value->opcode() == IrOpcode::kPhi &&
489  NodeProperties::GetControlInput(old_value, 0) == control) {
490  // Since a phi node can never dominate its control node,
491  // [old_value] cannot originate from the inputs. Thus [old_value]
492  // must have been created by a previous reduction of this [effect_phi].
493  for (int i = 0; i < arity; ++i) {
494  Node* old_input = NodeProperties::GetValueInput(old_value, i);
495  Node* new_input = buffer_[i] ? buffer_[i] : graph_->Dead();
496  if (old_input != new_input) {
497  NodeProperties::ReplaceValueInput(old_value, new_input, i);
498  reducer_->Revisit(old_value);
499  }
500  }
501  result.Set(var, old_value);
502  } else {
503  if (num_defined_inputs == 1 && is_loop) {
504  // For loop effect phis, the variable initialization dominates iff it
505  // dominates the first input.
506  DCHECK_EQ(2, arity);
507  DCHECK_EQ(value, buffer_[0]);
508  result.Set(var, value);
509  } else if (num_defined_inputs < arity) {
510  // If the variable is undefined on some input of this non-loop effect
511  // phi, then its initialization does not dominate this point.
512  result.Set(var, nullptr);
513  } else {
514  DCHECK_EQ(num_defined_inputs, arity);
515  // We only create a phi if the values are different.
516  if (identical_inputs) {
517  result.Set(var, value);
518  } else {
519  TRACE("Creating new phi\n");
520  buffer_.push_back(control);
521  Node* phi = graph_->graph()->NewNode(
522  graph_->common()->Phi(MachineRepresentation::kTagged, arity),
523  arity + 1, &buffer_.front());
524  // TODO(turbofan): Computing precise types here is tricky, because
525  // of the necessary revisitations. If we really need this, we should
526  // probably do it afterwards.
527  NodeProperties::SetType(phi, Type::Any());
528  reducer_->AddRoot(phi);
529  result.Set(var, phi);
530  }
531  }
532  }
533 #ifdef DEBUG
534  if (Node* result_node = result.Get(var)) {
535  TRACE(" result: %s#%d\n", result_node->op()->mnemonic(),
536  result_node->id());
537  } else {
538  TRACE(" result: nullptr\n");
539  }
540 #endif
541  }
542  }
543  return result;
544 }

633 void ReduceNode(const Operator* op, EscapeAnalysisTracker::Scope* current,
634  JSGraph* jsgraph) {
635  switch (op->opcode()) {
636  case IrOpcode::kAllocate: {
637  NumberMatcher size(current->ValueInput(0));
638  if (!size.HasResolvedValue()) break;
639  int size_int = static_cast<int>(size.ResolvedValue());
640  if (size_int != size.ResolvedValue()) break;
641  if (const VirtualObject* vobject = current->InitVirtualObject(size_int)) {
642  // Initialize with dead nodes as a sentinel for uninitialized memory.
643  for (Variable field : *vobject) {
644  current->Set(field, jsgraph->Dead());
645  }
646  }
647  break;
648  }
649  case IrOpcode::kFinishRegion:
650  current->SetVirtualObject(current->ValueInput(0));
651  break;
652  case IrOpcode::kStoreField: {
653  Node* object = current->ValueInput(0);
654  Node* value = current->ValueInput(1);
655  const VirtualObject* vobject = current->GetVirtualObject(object);
656  Variable var;
657  if (value->opcode() == IrOpcode::kTrustedHeapConstant) {
658  // TODO(dmercadier): enable escaping objects containing
659  // TrustedHeapConstants. This is currently disabled because it leads to
660  // bugs when Trusted HeapConstant and regular HeapConstant flow into the
661  // same Phi, which can then be marked as Compressed, messing up the
662  // tagging of the Trusted HeapConstant.
663  current->SetEscaped(object);
664  current->SetEscaped(value);
665  break;
666  }
667  // BoundedSize fields cannot currently be materialized by the deoptimizer,
668  // so we must not dematerialze them.
669  if (vobject && !vobject->HasEscaped() &&
670  vobject->FieldAt(OffsetOfFieldAccess(op)).To(&var) &&
671  !FieldAccessOf(op).is_bounded_size_access) {
672  current->Set(var, value);
673  current->MarkForDeletion();
674  } else {
675  current->SetEscaped(object);
676  current->SetEscaped(value);
677  }
678  break;
679  }
680  case IrOpcode::kStoreElement: {
681  Node* object = current->ValueInput(0);
682  Node* index = current->ValueInput(1);
683  Node* value = current->ValueInput(2);
684  const VirtualObject* vobject = current->GetVirtualObject(object);
685  int offset;
686  Variable var;
687  if (vobject && !vobject->HasEscaped() &&
688  OffsetOfElementsAccess(op, index).To(&offset) &&
689  vobject->FieldAt(offset).To(&var)) {
690  current->Set(var, value);
691  current->MarkForDeletion();
692  } else {
693  current->SetEscaped(value);
694  current->SetEscaped(object);
695  }
696  break;
697  }
698  case IrOpcode::kLoadField: {
699  Node* object = current->ValueInput(0);
700  const VirtualObject* vobject = current->GetVirtualObject(object);
701  Variable var;
702  Node* value;
703  if (vobject && !vobject->HasEscaped() &&
704  vobject->FieldAt(OffsetOfFieldAccess(op)).To(&var) &&
705  current->Get(var).To(&value)) {
706  current->SetReplacement(value);
707  } else {
708  current->SetEscaped(object);
709  }
710  break;
711  }
712  case IrOpcode::kLoadElement: {
713  Node* object = current->ValueInput(0);
714  Node* index = current->ValueInput(1);
715  const VirtualObject* vobject = current->GetVirtualObject(object);
716  int offset;
717  Variable var;
718  Node* value;
719  if (vobject && !vobject->HasEscaped() &&
720  OffsetOfElementsAccess(op, index).To(&offset) &&
721  vobject->FieldAt(offset).To(&var) && current->Get(var).To(&value)) {
722  current->SetReplacement(value);
723  break;
724  } else if (vobject && !vobject->HasEscaped()) {
725  // Compute the known length (aka the number of elements) of {object}
726  // based on the virtual object information.
727  ElementAccess const& access = ElementAccessOf(op);
728  int const length =
729  (vobject->size() - access.header_size) >>
730  ElementSizeLog2Of(access.machine_type.representation());
731  Variable var0, var1;
732  Node* value0;
733  Node* value1;
734  if (length == 1 &&
735  vobject->FieldAt(OffsetOfElementAt(access, 0)).To(&var) &&
736  current->Get(var).To(&value) &&
737  (value == nullptr ||
738  NodeProperties::GetType(value).Is(access.type))) {
739  // The {object} has no elements, and we know that the LoadElement
740  // {index} must be within bounds, thus it must always yield this
741  // one element of {object}.
742  current->SetReplacement(value);
743  break;
744  } else if (length == 2 &&
745  vobject->FieldAt(OffsetOfElementAt(access, 0)).To(&var0) &&
746  current->Get(var0).To(&value0) &&
747  (value0 == nullptr ||
748  NodeProperties::GetType(value0).Is(access.type)) &&
749  vobject->FieldAt(OffsetOfElementAt(access, 1)).To(&var1) &&
750  current->Get(var1).To(&value1) &&
751  (value1 == nullptr ||
752  NodeProperties::GetType(value1).Is(access.type))) {
753  if (value0 && value1) {
754  // The {object} has exactly two elements, so the LoadElement
755  // must return one of them (i.e. either the element at index
756  // 0 or the one at index 1). So we can turn the LoadElement
757  // into a Select operation instead (still allowing the {object}
758  // to be scalar replaced). We must however mark the elements
759  // of the {object} itself as escaping.
760  Node* check =
761  jsgraph->graph()->NewNode(jsgraph->simplified()->NumberEqual(),
762  index, jsgraph->ZeroConstant());
763  NodeProperties::SetType(check, Type::Boolean());
764  Node* select = jsgraph->graph()->NewNode(
765  jsgraph->common()->Select(access.machine_type.representation()),
766  check, value0, value1);
767  NodeProperties::SetType(select, access.type);
768  current->SetReplacement(select);
769  current->SetEscaped(value0);
770  current->SetEscaped(value1);
771  break;
772  } else {
773  // If the variables have no values, we have
774  // not reached the fixed-point yet.
775  break;
776  }
777  }
778  }
779  current->SetEscaped(object);
780  break;
781  }
782  case IrOpcode::kTypeGuard: {
783  current->SetVirtualObject(current->ValueInput(0));
784  break;
785  }
786  case IrOpcode::kReferenceEqual: {
787  Node* left = current->ValueInput(0);
788  Node* right = current->ValueInput(1);
789  const VirtualObject* left_object = current->GetVirtualObject(left);
790  const VirtualObject* right_object = current->GetVirtualObject(right);
791  Node* replacement = nullptr;
792  if (left_object && !left_object->HasEscaped()) {
793  if (right_object && !right_object->HasEscaped() &&
794  left_object->id() == right_object->id()) {
795  replacement = jsgraph->TrueConstant();
796  } else {
797  replacement = jsgraph->FalseConstant();
798  }
799  } else if (right_object && !right_object->HasEscaped()) {
800  replacement = jsgraph->FalseConstant();
801  }
802  // TODO(turbofan) This is a workaround for uninhabited types. If we
803  // replaced a value of uninhabited type with a constant, we would
804  // widen the type of the node. This could produce inconsistent
805  // types (which might confuse representation selection). We get
806  // around this by refusing to constant-fold and escape-analyze
807  // if the type is not inhabited.
808  if (replacement && !NodeProperties::GetType(left).IsNone() &&
809  !NodeProperties::GetType(right).IsNone()) {
810  current->SetReplacement(replacement);
811  break;
812  }
813  current->SetEscaped(left);
814  current->SetEscaped(right);
815  break;
816  }
817  case IrOpcode::kCheckMaps: {
818  CheckMapsParameters params = CheckMapsParametersOf(op);
819  Node* checked = current->ValueInput(0);
820  if (CheckMapsHelper(current, checked, params.maps())) {
821  break;
822  }
823  current->SetEscaped(checked);
824  break;
825  }
826  case IrOpcode::kTransitionElementsKindOrCheckMap: {
827  ElementsTransitionWithMultipleSources params =
828  ElementsTransitionWithMultipleSourcesOf(op);
829  Node* checked = current->ValueInput(0);
830  if (CheckMapsHelper(current, checked, ZoneRefSet<Map>(params.target()))) {
831  break;
832  }
833  current->SetEscaped(checked);
834  break;
835  }
836  case IrOpcode::kCompareMaps: {
837  Node* object = current->ValueInput(0);
838  const VirtualObject* vobject = current->GetVirtualObject(object);
839  Variable map_field;
840  Node* object_map;
841  if (vobject && !vobject->HasEscaped() &&
842  vobject->FieldAt(HeapObject::kMapOffset).To(&map_field) &&
843  current->Get(map_field).To(&object_map)) {
844  if (object_map) {
845  current->SetReplacement(LowerCompareMapsWithoutLoad(
846  object_map, CompareMapsParametersOf(op), jsgraph));
847  break;
848  } else {
849  // If the variable has no value, we have not reached the fixed-point
850  // yet.
851  break;
852  }
853  }
854  current->SetEscaped(object);
855  break;
856  }
857  case IrOpcode::kCheckHeapObject: {
858  Node* checked = current->ValueInput(0);
859  switch (checked->opcode()) {
860  case IrOpcode::kAllocate:
861  case IrOpcode::kFinishRegion:
862  case IrOpcode::kHeapConstant:
863  current->SetReplacement(checked);
864  break;
865  default:
866  current->SetEscaped(checked);
867  break;
868  }
869  break;
870  }
871  case IrOpcode::kMapGuard: {
872  Node* object = current->ValueInput(0);
873  const VirtualObject* vobject = current->GetVirtualObject(object);
874  if (vobject && !vobject->HasEscaped()) {
875  current->MarkForDeletion();
876  }
877  break;
878  }
879  case IrOpcode::kStateValues:
880  // We visit StateValue nodes through their correpsonding FrameState node,
881  // so we need to make sure we revisit the FrameState.
882  current->SetValueChanged();
883  break;
884  case IrOpcode::kFrameState: {
885  // We mark the receiver as escaping due to the non-standard `.getThis`
886  // API.
887  FrameState frame_state{current->CurrentNode()};
888  FrameStateType type = frame_state.frame_state_info().type();
889  // This needs to be kept in sync with the frame types supported in
890  // `OptimizedJSFrame::Summarize`.
891  if (type != FrameStateType::kUnoptimizedFunction &&
892  type != FrameStateType::kJavaScriptBuiltinContinuation &&
893  type != FrameStateType::kJavaScriptBuiltinContinuationWithCatch) {
894  break;
895  }
896  if (!current->FrameStateMightLazyDeopt(current->CurrentNode())) {
897  // Only lazy deopt frame states are used to generate stack traces.
898  break;
899  }
900  StateValuesAccess::iterator it =
901  StateValuesAccess(frame_state.parameters()).begin();
902  if (!it.done()) {
903  if (Node* receiver = it.node()) {
904  current->SetEscaped(receiver);
905  }
906  current->SetEscaped(frame_state.function());
907  }
908  break;
909  }
910  default: {
911  // For unknown nodes, treat all value inputs as escaping.
912  int value_input_count = op->ValueInputCount();
913  for (int i = 0; i < value_input_count; ++i) {
914  Node* input = current->ValueInput(i);
915  current->SetEscaped(input);
916  }
917  if (OperatorProperties::HasContextInput(op)) {
918  current->SetEscaped(current->ContextInput());
919  }
920  break;
921  }
922  }
923 }
