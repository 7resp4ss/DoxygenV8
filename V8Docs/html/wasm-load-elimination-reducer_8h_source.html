143  void Invalidate(int offset, wasm::ModuleTypeIndex type_index) {
144  // This is like LateLoadElimination's {InvalidateAtOffset}, but based
145  // on Wasm types instead of tracked JS maps.
146  auto offset_keys = offset_keys_.find(offset);
147  if (offset_keys == offset_keys_.end()) return;
148  for (auto it = offset_keys->second.begin();
149  it != offset_keys->second.end();) {
150  Key key = *it;
151  DCHECK_EQ(offset, key.data().mem.offset);
152  OpIndex base = key.data().mem.base;
153 
154  // If the base is guaranteed non-aliasing, we don't need to clear any
155  // other entries. Any existing entry for this base will be overwritten
156  // by {Insert(set)}.
157  if (non_aliasing_objects_.Get(base)) {
158  ++it;
159  continue;
160  }
161 
162  if (TypesUnrelated(type_index, key.data().mem.type_index)) {
163  ++it;
164  continue;
165  }
166 
167  it = offset_keys->second.RemoveAt(it);
168  Set(key, OpIndex::Invalid());
169  }
170  }

181  // Invalidates all Keys that are not known as non-aliasing.
182  enum class EntriesWithOffsets { kInvalidate, kKeep };
183  template <EntriesWithOffsets offsets = EntriesWithOffsets::kInvalidate>
184  void InvalidateMaybeAliasing() {
185  // We find current active keys through {base_keys_} so that we can bail out
186  // for whole buckets non-aliasing buckets (if we had gone through
187  // {offset_keys_} instead, then for each key we would've had to check
188  // whether it was non-aliasing or not).
189  for (auto& base_keys : base_keys_) {
190  OpIndex base = base_keys.first;
191  if (non_aliasing_objects_.Get(base)) continue;
192  if constexpr (offsets == EntriesWithOffsets::kInvalidate) {
193  for (auto it = base_keys.second.with_offsets.begin();
194  it != base_keys.second.with_offsets.end();) {
195  Key key = *it;
196  if (key.data().mem.mutability == false) {
197  ++it;
198  continue;
199  }
200  // It's important to remove with RemoveAt before Setting the key to
201  // invalid, otherwise OnKeyChange will remove {key} from {base_keys},
202  // which will invalidate {it}.
203  it = base_keys.second.with_offsets.RemoveAt(it);
204  Set(key, OpIndex::Invalid());
205  }
206  }
207  }

284  void Insert(OpIndex base, int32_t offset, wasm::ModuleTypeIndex type_index,
285  uint8_t size, bool mutability, OpIndex value) {
286  DCHECK_EQ(base, ResolveBase(base));
287 
288  WasmMemoryAddress mem{base, offset, type_index, size, mutability};
289  auto existing_key = all_keys_.find(mem);
290  if (existing_key != all_keys_.end()) {
291  if (mutability) {
292  Set(existing_key->second, value);
293  } else {
294  SetNoNotify(existing_key->second, value);
295  }
296  return;
297  }
298 
299  // Creating a new key.
300  Key key = NewKey({mem});
301  all_keys_.insert({mem, key});
302  if (mutability) {
303  Set(key, value);
304  } else {
305  // Call `SetNoNotify` to avoid calls to `OnNewKey` and `OnValueChanged`.
306  SetNoNotify(key, value);

311  OpIndex ResolveBase(OpIndex base) {
312  while (true) {
313  if (replacements_[base] != OpIndex::Invalid()) {
314  base = replacements_[base];
315  continue;
316  }
317  Operation& op = graph_.Get(base);
318  if (AssertNotNullOp* check = op.TryCast<AssertNotNullOp>()) {
319  base = check->object();
320  continue;
321  }
322  if (WasmTypeCastOp* cast = op.TryCast<WasmTypeCastOp>()) {
323  base = cast->object();
324  continue;
325  }
326  break; // Terminate if nothing happened.
327  }
328  return base;

331  void AddKeyInBaseOffsetMaps(Key key) {
332  // Inserting in {base_keys_}.
333  OpIndex base = key.data().mem.base;
334  auto base_keys = base_keys_.find(base);
335  if (base_keys != base_keys_.end()) {
336  base_keys->second.with_offsets.PushFront(key);
337  } else {
338  BaseData data;
339  data.with_offsets.PushFront(key);
340  base_keys_.insert({base, std::move(data)});
341  }
342 
343  // Inserting in {offset_keys_}.
344  int offset = key.data().mem.offset;
345  auto offset_keys = offset_keys_.find(offset);
346  if (offset_keys != offset_keys_.end()) {
347  offset_keys->second.PushFront(key);
348  } else {
349  v8::base::DoublyThreadedList<Key, OffsetListTraits> list;
350  list.PushFront(key);
351  offset_keys_.insert({offset, std::move(list)});

384 class WasmLoadEliminationAnalyzer {
385  public:
386  using AliasTable = SparseOpIndexSnapshotTable<bool>;
387  using AliasKey = AliasTable::Key;
388  using AliasSnapshot = AliasTable::Snapshot;

391  using MemorySnapshot = wle::WasmMemoryContentTable::Snapshot;

393  WasmLoadEliminationAnalyzer(PipelineData* data, Graph& graph,
394  Zone* phase_zone)
395  : graph_(graph),
396  phase_zone_(phase_zone),
397  replacements_(graph.op_id_count(), phase_zone, &graph),
398  non_aliasing_objects_(phase_zone),
399  memory_(data, phase_zone, non_aliasing_objects_, replacements_, graph_),
400  block_to_snapshot_mapping_(graph.block_count(), phase_zone),
401  predecessor_alias_snapshots_(phase_zone),
402  predecessor_memory_snapshots_(phase_zone) {}
403 
404  void Run() {
405  LoopFinder loop_finder(phase_zone_, &graph_, LoopFinder::Config{});
406  AnalyzerIterator iterator(phase_zone_, graph_, loop_finder);
407 
408  bool compute_start_snapshot = true;
409  while (iterator.HasNext()) {
410  const Block* block = iterator.Next();
411 
412  ProcessBlock(*block, compute_start_snapshot);
413  compute_start_snapshot = true;
414 
415  // Consider re-processing for loops.
416  if (const GotoOp* last = block->LastOperation(graph_).TryCast<GotoOp>()) {
417  if (last->destination->IsLoop() &&
418  last->destination->LastPredecessor() == block) {
419  const Block* loop_header = last->destination;
420  // {block} is the backedge of a loop. We recompute the loop header's
421  // initial snapshots, and if they differ from its original snapshot,
422  // then we revisit the loop.
423  if (BeginBlock<true>(loop_header)) {
424  // We set the snapshot of the loop's 1st predecessor to the newly
425  // computed snapshot. It's not quite correct, but this predecessor
426  // is guaranteed to end with a Goto, and we are now visiting the
427  // loop, which means that we don't really care about this
428  // predecessor anymore.
429  // The reason for saving this snapshot is to prevent infinite
430  // looping, since the next time we reach this point, the backedge
431  // snapshot could still invalidate things from the forward edge
432  // snapshot. By restricting the forward edge snapshot, we prevent
433  // this.
434  const Block* loop_1st_pred =
435  loop_header->LastPredecessor()->NeighboringPredecessor();
436  FinishBlock(loop_1st_pred);
437  // And we start a new fresh snapshot from this predecessor.
438  auto pred_snapshots =
439  block_to_snapshot_mapping_[loop_1st_pred->index()];
440  non_aliasing_objects_.StartNewSnapshot(
441  pred_snapshots->alias_snapshot);
442  memory_.StartNewSnapshot(pred_snapshots->memory_snapshot);
443 
444  iterator.MarkLoopForRevisit();
445  compute_start_snapshot = false;
446  } else {
447  SealAndDiscard();
448  }
449  }
450  }
451  }
452  }

582 void WasmLoadEliminationAnalyzer::ProcessBlock(const Block& block,
583  bool compute_start_snapshot) {
584  if (compute_start_snapshot) {
585  BeginBlock(&block);
586  }
587  if (block.IsLoop() && BackedgeHasSnapshot(block)) {
588  // Update the associated snapshot for the forward edge with the merged
589  // snapshot information from the forward- and backward edge.
590  // This will make sure that when evaluating whether a loop needs to be
591  // revisited, the inner loop compares the merged state with the backedge
592  // preventing us from exponential revisits for loops where the backedge
593  // invalidates loads which are eliminatable on the forward edge.
594  StoreLoopSnapshotInForwardPredecessor(block);

599  if (ShouldSkipOptimizationStep()) continue;
600  if (ShouldSkipOperation(op)) continue;

602  case Opcode::kStructGet:
603  ProcessStructGet(op_idx, op.Cast<StructGetOp>());
604  break;
605  case Opcode::kStructSet:
606  ProcessStructSet(op_idx, op.Cast<StructSetOp>());
607  break;
608  case Opcode::kStructAtomicRMW:
609  ProcessAtomicRMW(op_idx, op.Cast<StructAtomicRMWOp>());
610  break;
611  case Opcode::kArrayLength:
612  ProcessArrayLength(op_idx, op.Cast<ArrayLengthOp>());
613  break;
614  case Opcode::kWasmAllocateArray:
615  ProcessWasmAllocateArray(op_idx, op.Cast<WasmAllocateArrayOp>());
616  break;
617  case Opcode::kStringAsWtf16:
618  ProcessStringAsWtf16(op_idx, op.Cast<StringAsWtf16Op>());
619  break;
620  case Opcode::kStringPrepareForGetCodeUnit:
621  ProcessStringPrepareForGetCodeUnit(
622  op_idx, op.Cast<StringPrepareForGetCodeUnitOp>());
623  break;
624  case Opcode::kAnyConvertExtern:
625  ProcessAnyConvertExtern(op_idx, op.Cast<AnyConvertExternOp>());
626  break;
627  case Opcode::kAssertNotNull:
628  // TODO(14108): We'll probably want to handle WasmTypeCast as
629  // a "load-like" instruction too, to eliminate repeated casts.
630  ProcessAssertNotNull(op_idx, op.Cast<AssertNotNullOp>());
631  break;
632  case Opcode::kArraySet:
633  break;
634  case Opcode::kAllocate:
635  // Create new non-alias.
636  ProcessAllocate(op_idx, op.Cast<AllocateOp>());
637  break;
638  case Opcode::kCall:
639  // Invalidate state (+ maybe invalidate aliases).
640  ProcessCall(op_idx, op.Cast<CallOp>());
641  break;
642  case Opcode::kPhi:
643  // Invalidate aliases.
644  ProcessPhi(op_idx, op.Cast<PhiOp>());
645  break;
646  case Opcode::kLoad:
647  // Atomic loads have the "can_write" bit set, because they make
648  // writes on other threads visible. At any rate, we have to
649  // explicitly skip them here.
650  case Opcode::kStore:
651  // We rely on having no raw "Store" operations operating on Wasm
652  // objects at this point in the pipeline.
653  // TODO(jkummerow): Is there any way to DCHECK that?
654  case Opcode::kAssumeMap:
655  case Opcode::kCatchBlockBegin:
656  case Opcode::kRetain:
657  case Opcode::kDidntThrow:
658  case Opcode::kCheckException:
659  case Opcode::kAtomicRMW:
660  case Opcode::kAtomicWord32Pair:
661  case Opcode::kMemoryBarrier:
662  case Opcode::kJSStackCheck:
663  case Opcode::kWasmStackCheck:
664  case Opcode::kSimd128LaneMemory:
665  case Opcode::kGlobalSet:
666  case Opcode::kParameter:
667  case Opcode::kSetStackPointer:
668  // We explicitly break for those operations that have can_write effects
669  // but don't actually write, or cannot interfere with load elimination.
670  break;
671 
672  case Opcode::kWordBinop:
673  // A WordBinop should never invalidate aliases (since the only time when
674  // it should take a non-aliasing object as input is for Smi checks).
675  DcheckWordBinop(op_idx, op.Cast<WordBinopOp>());
676  break;
677 
678  case Opcode::kFrameState:
679  case Opcode::kDeoptimizeIf:
680  case Opcode::kComparison:
681  case Opcode::kTrapIf:
682  // We explicitly break for these opcodes so that we don't call
683  // InvalidateAllNonAliasingInputs on their inputs, since they don't
684  // really create aliases. (and also, they don't write so it's
685  // fine to break)
686  DCHECK(!op.Effects().can_write());
687  break;
688 
689  case Opcode::kDeoptimize:
690  case Opcode::kReturn:
691  // We explicitly break for these opcodes so that we don't call
692  // InvalidateAllNonAliasingInputs on their inputs, since they are block
693  // terminators without successors, meaning that it's not useful for the
694  // rest of the analysis to invalidate anything here.
695  DCHECK(op.IsBlockTerminator() && SuccessorBlocks(op).empty());
696  break;
697 
698  default:
699  // Operations that `can_write` should invalidate the state. All such
700  // operations should be already handled above, which means that we don't
701  // need a `if (can_write) { Invalidate(); }` here.
702  CHECK(!op.Effects().can_write());
703 
704  // Even if the operation doesn't write, it could create an alias to its
705  // input by returning it. This happens for instance in Phis and in
706  // Change (although ChangeOp is already handled earlier by calling
707  // ProcessChange). We are conservative here by calling
708  // InvalidateAllNonAliasingInputs for all operations even though only
709  // few can actually create aliases to fresh allocations, the reason
710  // being that missing such a case would be a security issue, and it
711  // should be rare for fresh allocations to be used outside of
712  // Call/Store/Load/Change anyways.
713  InvalidateAllNonAliasingInputs(op);
714 
715  break;
716  }
717  }
718 
719  FinishBlock(&block);
720 }

722 // Returns true if replacing a load with a RegisterRepresentation
723 // {expected_reg_rep} and size {in_memory_size} with an
724 // operation with RegisterRepresentation {actual} is valid. For instance,
725 // replacing an operation that returns a Float64 by one that returns a Word64 is
726 // not valid. Similarly, replacing a Tagged with an untagged value is probably
727 // not valid because of the GC.
728

730  RegisterRepresentation expected_reg_repr,
731  uint8_t in_memory_size) {
732  if (in_memory_size !=
733  MemoryRepresentation::FromRegisterRepresentation(actual, true)
734  .SizeInBytes()) {
735  // The replacement was truncated when being stored or should be truncated
736  // (or sign-extended) during the load. Since we don't have enough
737  // truncations operators in Turboshaft (eg, we don't have Int32 to Int8
738  // truncation), we just prevent load elimination in this case.
739 
740  // TODO(jkummerow): support eliminating repeated loads of the same i8/i16
741  // field.
742  return false;
743  }
744 
745  return expected_reg_repr == actual;
746 }

1010 template <bool for_loop_revisit>
1011 bool WasmLoadEliminationAnalyzer::BeginBlock(const Block* block) {
1012  DCHECK_IMPLIES(
1013  for_loop_revisit,
1014  block->IsLoop() &&
1015  block_to_snapshot_mapping_[block->LastPredecessor()->index()]
1016  .has_value());
1017 
1018  // Collect the snapshots of all predecessors.
1019  {
1020  predecessor_alias_snapshots_.clear();
1021  predecessor_memory_snapshots_.clear();
1022  for (const Block* p : block->PredecessorsIterable()) {
1023  auto pred_snapshots = block_to_snapshot_mapping_[p->index()];
1024  // When we visit the loop for the first time, the loop header hasn't
1025  // been visited yet, so we ignore it.
1026  DCHECK_IMPLIES(!pred_snapshots.has_value(),
1027  block->IsLoop() && block->LastPredecessor() == p);
1028  if (!pred_snapshots.has_value()) {
1029  DCHECK(!for_loop_revisit);
1030  continue;
1031  }
1032  // Note that the backedge snapshot of an inner loop in kFirstVisit will
1033  // also be taken into account if we are in the kSecondVisit of an outer
1034  // loop. The data in the backedge snapshot could be out-dated, but if it
1035  // is, then it's fine: if the backedge of the outer-loop was more
1036  // restrictive than its forward incoming edge, then the forward incoming
1037  // edge of the inner loop should reflect this restriction.
1038  predecessor_alias_snapshots_.push_back(pred_snapshots->alias_snapshot);
1039  predecessor_memory_snapshots_.push_back(pred_snapshots->memory_snapshot);
1040  }
1041  }
1042 
1043  // Note that predecessors are in reverse order, which means that the backedge
1044  // is at offset 0.
1045  constexpr int kBackedgeOffset = 0;
1046  constexpr int kForwardEdgeOffset = 1;
1047 
1048  bool loop_needs_revisit = false;
1049  // Start a new snapshot for this block by merging information from
1050  // predecessors.
1051  auto merge_aliases = [&](AliasKey key,
1052  base::Vector<const bool> predecessors) -> bool {
1053  if (for_loop_revisit && predecessors[kForwardEdgeOffset] &&
1054  !predecessors[kBackedgeOffset]) {
1055  // The backedge doesn't think that {key} is no-alias, but the loop
1056  // header previously thought it was --> need to revisit.
1057  loop_needs_revisit = true;
1058  }
1059  return base::all_of(predecessors);
1060  };
1061  non_aliasing_objects_.StartNewSnapshot(
1062  base::VectorOf(predecessor_alias_snapshots_), merge_aliases);
1063 
1064  // Merging for {memory_} means setting values to Invalid unless all
1065  // predecessors have the same value.
1066  // TODO(dmercadier): we could insert of Phis during the pass to merge existing
1067  // information. This is a bit hard, because we are currently in an analyzer
1068  // rather than a reducer. Still, we could "prepare" the insertion now and then
1069  // really insert them during the Reduce phase of the CopyingPhase.
1070  auto merge_memory = [&](MemoryKey key,
1071  base::Vector<const OpIndex> predecessors) -> OpIndex {
1072  if (for_loop_revisit && predecessors[kForwardEdgeOffset].valid() &&
1073  predecessors[kBackedgeOffset] != predecessors[kForwardEdgeOffset]) {
1074  // {key} had a value in the loop header, but the backedge and the forward
1075  // edge don't agree on its value, which means that the loop invalidated
1076  // some memory data, and thus needs to be revisited.
1077  loop_needs_revisit = true;
1078  }
1079  return base::all_equal(predecessors) ? predecessors[0] : OpIndex::Invalid();
1080  };
1081  memory_.StartNewSnapshot(base::VectorOf(predecessor_memory_snapshots_),
1082  merge_memory);
1083 
1084  if (block->IsLoop()) return loop_needs_revisit;
1085  return false;
1086 }
