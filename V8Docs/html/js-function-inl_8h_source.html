80 void JSFunction::UpdateOptimizedCode(Isolate* isolate, Tagged<Code> code,
81  WriteBarrierMode mode) {
82  DisallowGarbageCollection no_gc;
83  DCHECK(code->is_optimized_code());
84 #ifdef V8_ENABLE_LEAPTIERING
85  if (code->is_context_specialized()) {
86  // We can only set context-specialized code for single-closure cells.
87  if (raw_feedback_cell()->map() !=
88  ReadOnlyRoots(isolate).one_closure_cell_map()) {
89  return;
90  }
91  }
92  // Required for being able to deoptimize this code.
93  code->set_js_dispatch_handle(dispatch_handle());
94 #endif // V8_ENABLE_LEAPTIERING
95  UpdateCodeImpl(isolate, code, mode, false);
96 }

98 void JSFunction::UpdateCodeImpl(Isolate* isolate, Tagged<Code> value,
99  WriteBarrierMode mode,
100  bool keep_tiering_request) {
101  DisallowGarbageCollection no_gc;
102 
103 #ifdef V8_ENABLE_LEAPTIERING
104  JSDispatchHandle handle = dispatch_handle();
105  if (handle == kNullJSDispatchHandle) {
106  handle = raw_feedback_cell()->dispatch_handle();
107  DCHECK_NE(handle, kNullJSDispatchHandle);
108  set_dispatch_handle(handle, mode);
109  }
110  if (keep_tiering_request) {
111  UpdateDispatchEntryKeepTieringRequest(value, mode);
112  } else {
113  UpdateDispatchEntry(value, mode);
114  }
115 
116  if (V8_UNLIKELY(v8_flags.log_function_events)) {
117  IsolateGroup::current()->js_dispatch_table()->SetTieringRequest(
118  dispatch_handle(), TieringBuiltin::kFunctionLogNextExecution, isolate);
119  }
120 #else
121  WriteCodePointerField(kCodeOffset, value);
122  CONDITIONAL_CODE_POINTER_WRITE_BARRIER(*this, kCodeOffset, value, mode);
123 
124  if (V8_UNLIKELY(v8_flags.log_function_events && has_feedback_vector())) {
125  feedback_vector()->set_log_next_execution(true);
126  }
127 #endif // V8_ENABLE_LEAPTIERING
128 }

337 std::optional<CodeKind> JSFunction::GetRequestedOptimizationIfAny(
338  Isolate* isolate, ConcurrencyMode mode) const {
339 #ifdef V8_ENABLE_LEAPTIERING
340  JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
341  Address entrypoint = jdt->GetEntrypoint(dispatch_handle());
342  const EmbeddedData& embedded_data = EmbeddedData::FromBlob(isolate);
343  auto builtin = ([&]() -> std::optional<TieringBuiltin> {
344 #define CASE(name, ...) \
345  if (entrypoint == embedded_data.InstructionStartOf(Builtin::k##name)) { \
346  DCHECK(jdt->IsTieringRequested(dispatch_handle(), TieringBuiltin::k##name, \
347  isolate)); \
348  return TieringBuiltin::k##name; \
349  }
350  BUILTIN_LIST_BASE_TIERING(CASE)
351 #undef CASE
352  DCHECK(!jdt->IsTieringRequested(dispatch_handle()));
353  return {};
354  })();
355  if (V8_LIKELY(!builtin)) return {};
356  switch (*builtin) {
357  case TieringBuiltin::kOptimizeMaglevEager:
358  if (mode == ConcurrencyMode::kSynchronous) return CodeKind::MAGLEV;
359  break;
360  case TieringBuiltin::kStartMaglevOptimizeJob:
361  if (mode == ConcurrencyMode::kConcurrent) return CodeKind::MAGLEV;
362  break;
363  case TieringBuiltin::kOptimizeTurbofanEager:
364  if (mode == ConcurrencyMode::kSynchronous) return CodeKind::TURBOFAN_JS;
365  break;
366  case TieringBuiltin::kStartTurbofanOptimizeJob:
367  if (mode == ConcurrencyMode::kConcurrent) return CodeKind::TURBOFAN_JS;
368  break;
369  case TieringBuiltin::kMarkLazyDeoptimized:
370  case TieringBuiltin::kMarkReoptimizeLazyDeoptimized:
371  case TieringBuiltin::kFunctionLogNextExecution:
372  break;
373  }
374 #else
375  switch (mode) {
376  case ConcurrencyMode::kConcurrent:
377  if (IsRequestTurbofan_Concurrent(tiering_state())) {
378  return CodeKind::TURBOFAN_JS;
379  }
380  if (IsRequestMaglev_Concurrent(tiering_state())) {
381  return CodeKind::MAGLEV;
382  }
383  break;
384  case ConcurrencyMode::kSynchronous:
385  if (IsRequestTurbofan_Synchronous(tiering_state())) {
386  return CodeKind::TURBOFAN_JS;
387  }
388  if (IsRequestMaglev_Synchronous(tiering_state())) {
389  return CodeKind::MAGLEV;
390  }
391  break;
392  }
393 #endif // !V8_ENABLE_LEAPTIERING
394  return {};
395 }

549 bool JSFunction::NeedsResetDueToFlushedBytecode(Isolate* isolate) {
550  // Do a raw read for shared and code fields here since this function may be
551  // called on a concurrent thread. JSFunction itself should be fully
552  // initialized here but the SharedFunctionInfo, Code objects may not be
553  // initialized. We read using acquire loads to defend against that.
554  // TODO(v8) the branches for !IsSharedFunctionInfo() and !IsCode() are
555  // probably dead code by now. Investigate removing them or replacing them
556  // with CHECKs.
557  Tagged<Object> maybe_shared =
558  ACQUIRE_READ_FIELD(*this, kSharedFunctionInfoOffset);
559  if (!IsSharedFunctionInfo(maybe_shared)) return false;
560 
561  Tagged<Object> maybe_code = raw_code(isolate, kAcquireLoad);
562  if (!IsCode(maybe_code)) return false;
563  Tagged<Code> code = Cast<Code>(maybe_code);
564 
565  Tagged<SharedFunctionInfo> shared = Cast<SharedFunctionInfo>(maybe_shared);
566  return !shared->is_compiled() &&
567  (code->builtin_id() != Builtin::kCompileLazy ||
568  // With leaptiering we can have CompileLazy as the code object but
569  // still an optimization trampoline installed.
570  (V8_ENABLE_LEAPTIERING_BOOL && IsOptimizationRequested(isolate)));
571 }
