5 #ifndef V8_SANDBOX_CHECK_H_
6 #define V8_SANDBOX_CHECK_H_

8 #include "src/sandbox/hardware-support.h"

10 // When the sandbox is enabled, a SBXCHECK behaves exactly like a CHECK, but
11 // indicates that the check is required for the sandbox, i.e. prevents a
12 // sandbox bypass. When the sandbox is off, it becomes a DCHECK.
13 //
14 // As an example, consider a scenario where an in-sandbox object stores an
15 // index into an out-of-sandbox array (or a similar data structure). While
16 // under normal circumstances it can be guaranteed that the index will always
17 // be in bounds, with the sandbox attacker model, we have to assume that the
18 // in-sandbox object can be corrupted by an attacker and so the access can go
19 // out-of-bounds. In that case, a SBXCHECK can be used to both prevent memory
20 // corruption outside of the sandbox and document that there is a
21 // security-critical invariant that may be violated when an attacker can
22 // corrupt memory inside the sandbox, but otherwise holds true.
23 #ifdef V8_ENABLE_SANDBOX
24 
25 // It's unsafe to access sandbox memory during a SBXCHECK since such an access
26 // will be inherently racy as we need to assume an attacker can modify the value
27 // inside the sandbox right before and after the check. If you run into this,
28 // you might want to read the value outside of the SBXCHECK first to ensure that
29 // the SBXCHECK and the code that relies on it use the same value. And if in
30 // doubt, feel free to add someone from the security team as a reviewer. If
31 // sandbox hardware support is enabled, we'll block these accesses temporarily
32 // in debug builds.
33 #define SBXCHECK(condition) \
34  do { \
35  v8::internal::DisallowSandboxAccess no_sandbox_access; \
36  CHECK(condition); \
37  } while (false)
38 
39 #define SBXCHECK_WRAPPED(CONDITION, lhs, rhs) \
40  do { \
41  v8::internal::DisallowSandboxAccess no_sandbox_access; \
42  CHECK_##CONDITION(lhs, rhs); \
43  } while (false)
44 
45 #define SBXCHECK_EQ(lhs, rhs) SBXCHECK_WRAPPED(EQ, lhs, rhs)
46 #define SBXCHECK_NE(lhs, rhs) SBXCHECK_WRAPPED(NE, lhs, rhs)
47 #define SBXCHECK_GT(lhs, rhs) SBXCHECK_WRAPPED(GT, lhs, rhs)
48 #define SBXCHECK_GE(lhs, rhs) SBXCHECK_WRAPPED(GE, lhs, rhs)
49 #define SBXCHECK_LT(lhs, rhs) SBXCHECK_WRAPPED(LT, lhs, rhs)
50 #define SBXCHECK_LE(lhs, rhs) SBXCHECK_WRAPPED(LE, lhs, rhs)
51 #define SBXCHECK_BOUNDS(index, limit) SBXCHECK_WRAPPED(BOUNDS, index, limit)
52 #define SBXCHECK_IMPLIES(when, then) SBXCHECK_WRAPPED(IMPLIES, when, then)
53 #else // V8_ENABLE_SANDBOX
54 #define SBXCHECK(condition) DCHECK(condition)
55 #define SBXCHECK_EQ(lhs, rhs) DCHECK_EQ(lhs, rhs)
56 #define SBXCHECK_NE(lhs, rhs) DCHECK_NE(lhs, rhs)
57 #define SBXCHECK_GT(lhs, rhs) DCHECK_GT(lhs, rhs)
58 #define SBXCHECK_GE(lhs, rhs) DCHECK_GE(lhs, rhs)

60 #define SBXCHECK_LE(lhs, rhs) DCHECK_LE(lhs, rhs)
61 #define SBXCHECK_BOUNDS(index, limit) DCHECK_BOUNDS(index, limit)
62 #define SBXCHECK_IMPLIES(when, then) DCHECK_IMPLIES(when, then)
63 #endif // V8_ENABLE_SANDBOX
