43  private:

194  public:

199  protected:

201  some_lab_is_set_ |=

205 
206  private:
207  bool some_lab_is_set_ = false;
208  };

212  DCHECK(remembered_set_.IsEmpty());

224 void HeapBase::Terminate() {
225  CHECK(!IsMarking());
226  CHECK(!IsGCForbidden());
227  // Cannot use IsGCAllowed() as `Terminate()` will be invoked after detaching
228  // which implies GC is prohibited at this point.
229  CHECK(!sweeper().IsSweepingOnMutatorThread());
230 
231  sweeper().FinishIfRunning();
232 
233 #if defined(CPPGC_YOUNG_GENERATION)
234  if (generational_gc_supported()) {
235  DCHECK(is_young_generation_enabled());
236  HeapHandle::is_young_generation_enabled_ = false;
237  YoungGenerationEnabler::Disable();
238  }
239 #endif // defined(CPPGC_YOUNG_GENERATION)
240 
241  constexpr size_t kMaxTerminationGCs = 20;
242  size_t gc_count = 0;
243  bool more_termination_gcs_needed = false;
244  do {
245  // Clear root sets.
246  strong_persistent_region_.ClearAllUsedNodes();
247  weak_persistent_region_.ClearAllUsedNodes();
248  {
249  PersistentRegionLock guard;
250  strong_cross_thread_persistent_region_.ClearAllUsedNodes();
251  weak_cross_thread_persistent_region_.ClearAllUsedNodes();
252  }
253 
254 #if defined(CPPGC_YOUNG_GENERATION)
255  if (generational_gc_supported()) {
256  // Unmark the heap so that the sweeper destructs all objects.
257  // TODO(chromium:1029379): Merge two heap iterations (unmarking +
258  // sweeping) into forced finalization.
259  SequentialUnmarker unmarker(raw_heap());
260  }
261 #endif // defined(CPPGC_YOUNG_GENERATION)
262 
263  in_atomic_pause_ = true;
264  stats_collector()->NotifyMarkingStarted(CollectionType::kMajor,
265  GCConfig::MarkingType::kAtomic,
266  GCConfig::IsForcedGC::kForced);
267  object_allocator().ResetLinearAllocationBuffers();
268  stats_collector()->NotifyMarkingCompleted(0);
269  ExecutePreFinalizers();
270  // TODO(chromium:1029379): Prefinalizers may black-allocate objects (under a
271  // compile-time option). Run sweeping with forced finalization here.
272  sweeper().Start({SweepingConfig::SweepingType::kAtomic,
273  SweepingConfig::CompactableSpaceHandling::kSweep});
274  in_atomic_pause_ = false;
275  sweeper().FinishIfRunning();
276  more_termination_gcs_needed =
277  strong_persistent_region_.NodesInUse() ||
278  weak_persistent_region_.NodesInUse() || [this]() {
279  PersistentRegionLock guard;
280  return strong_cross_thread_persistent_region_.NodesInUse() ||
281  weak_cross_thread_persistent_region_.NodesInUse();
282  }();
283  gc_count++;
284  } while (more_termination_gcs_needed && (gc_count < kMaxTerminationGCs));
285 
286  CHECK_EQ(0u, strong_persistent_region_.NodesInUse());
287  CHECK_EQ(0u, weak_persistent_region_.NodesInUse());
288  {
289  PersistentRegionLock guard;
290  CHECK_EQ(0u, strong_cross_thread_persistent_region_.NodesInUse());
291  CHECK_EQ(0u, weak_cross_thread_persistent_region_.NodesInUse());
292  }
293  CHECK_LE(gc_count, kMaxTerminationGCs);
294 
295  object_allocator().ResetLinearAllocationBuffers();
296  disallow_gc_scope_++;
297 }

299 HeapStatistics HeapBase::CollectStatistics(
300  HeapStatistics::DetailLevel detail_level) {
301  if (detail_level == HeapStatistics::DetailLevel::kBrief) {
302  const size_t pooled_memory = page_backend_->page_pool().PooledMemory();
303  const size_t committed_memory =
304  stats_collector_->allocated_memory_size() + pooled_memory;
305  const size_t resident_memory =
306  stats_collector_->resident_memory_size() + pooled_memory;
307 
308  return {committed_memory,
309  resident_memory,
310  stats_collector_->allocated_object_size(),
311  pooled_memory,
312  HeapStatistics::DetailLevel::kBrief,
313  {},
314  {}};
315  }
316 
317  sweeper_.FinishIfRunning();
318  object_allocator_.ResetLinearAllocationBuffers();
319  return HeapStatisticsCollector().CollectDetailedStatistics(this);
320 }
