76 InternalIndex DescriptorArray::Search(Tagged<Name> name, int valid_descriptors,
77  bool concurrent_search) {
78  DCHECK(IsUniqueName(name));
79  SLOW_DCHECK_IMPLIES(!concurrent_search, IsSortedNoDuplicates());
80 
81  if (valid_descriptors == 0) {
82  return InternalIndex::NotFound();
83  }
84 
85  // Do linear search for small arrays, and for searches in the background
86  // thread.
87  const int kMaxElementsForLinearSearch = 8;
88  if (valid_descriptors <= kMaxElementsForLinearSearch || concurrent_search) {
89  return LinearSearch(name, valid_descriptors);
90  }
91 
92  return BinarySearch(name, valid_descriptors);
93 }

95 InternalIndex DescriptorArray::BinarySearch(Tagged<Name> name,
96  int valid_descriptors) {
97  // We have to binary search all descriptors, not just valid ones, since the
98  // binary search ordering is across all descriptors.
99  int end = number_of_descriptors();
100  uint32_t hash = name->hash();
101 
102  // Find the first descriptor whose key's hash is greater-than-or-equal-to the
103  // search hash.
104  int number = *std::ranges::lower_bound(std::views::iota(0, end), hash,
105  std::less<>(), [&](int i) {
106  Tagged<Name> entry = GetSortedKey(i);
107  return entry->hash();
108  });
109 
110  // There may have been hash collisions, so search for the name from the first
111  // index until the first non-matching hash.
112  for (; number < end; ++number) {
113  InternalIndex index(GetSortedKeyIndex(number));
114  Tagged<Name> entry = GetKey(index);
115  if (entry == name) {
116  // If we found the entry, but it's outside the owned descriptors of the
117  // caller, return not found.
118  if (index.as_int() >= valid_descriptors) {
119  return InternalIndex::NotFound();
120  }
121  return index;
122  }
123  if (entry->hash() != hash) {
124  return InternalIndex::NotFound();
125  }
126  }
127 
128  return InternalIndex::NotFound();
129 }

168 InternalIndex DescriptorArray::SearchWithCache(Isolate* isolate,
169  Tagged<Name> name,
170  Tagged<Map> map) {
171  DCHECK(IsUniqueName(name));
172  int number_of_own_descriptors = map->NumberOfOwnDescriptors();
173  if (number_of_own_descriptors == 0) return InternalIndex::NotFound();
174 
175  DescriptorLookupCache* cache = isolate->descriptor_lookup_cache();
176  int number = cache->Lookup(map, name);
177 
178  if (number == DescriptorLookupCache::kAbsent) {
179  InternalIndex result = Search(name, number_of_own_descriptors);
180  number = result.is_found() ? result.as_int() : DescriptorArray::kNotFound;
181  cache->Update(map, name, number);
182  }
183  if (number == DescriptorArray::kNotFound) return InternalIndex::NotFound();
184  return InternalIndex(number);
185 }

344 void DescriptorArray::Append(Descriptor* desc) {
345  DisallowGarbageCollection no_gc;
346  int descriptor_number = number_of_descriptors();
347  DCHECK_LE(descriptor_number + 1, number_of_all_descriptors());
348  set_number_of_descriptors(descriptor_number + 1);
349  Set(InternalIndex(descriptor_number), desc);
350 
351  uint32_t desc_hash = desc->GetKey()->hash();
352  // Hash value can't be zero, see String::ComputeAndSetHash()
353  uint32_t collision_hash = 0;
354 
355  int insertion;
356 
357  for (insertion = descriptor_number; insertion > 0; --insertion) {
358  Tagged<Name> key = GetSortedKey(insertion - 1);
359  collision_hash = key->hash();
360  if (collision_hash <= desc_hash) break;
361  SetSortedKey(insertion, GetSortedKeyIndex(insertion - 1));
362  }
363 
364  SetSortedKey(insertion, descriptor_number);
365 
366  // Resetting the fast iterable state is bottlenecked in SetKey().
367  DCHECK_NE(fast_iterable(), FastIterableState::kJsonFast);
368 
369  if (V8_LIKELY(collision_hash != desc_hash)) return;
370 
371  CheckNameCollisionDuringInsertion(desc, desc_hash, insertion);
372 }

380 // static
381 bool DescriptorArrayMarkingState::TryUpdateIndicesToMark(
382  unsigned gc_epoch, Tagged<DescriptorArray> array,
383  DescriptorIndex index_to_mark) {
384  const auto current_epoch = gc_epoch & Epoch::kMask;
385  while (true) {
386  const RawGCStateType raw_gc_state = array->raw_gc_state(kRelaxedLoad);
387  const auto epoch_from_state = Epoch::decode(raw_gc_state);
388  RawGCStateType new_raw_gc_state = 0;
389  if (current_epoch != epoch_from_state) {
390  // If the epochs do not match, then either the raw_gc_state is zero
391  // (freshly allocated descriptor array) or the epoch from value lags
392  // by 1.
393  DCHECK_IMPLIES(raw_gc_state != 0,
394  Epoch::decode(epoch_from_state + 1) == current_epoch);
395  new_raw_gc_state = NewState(current_epoch, 0, index_to_mark);
396  } else {
397  const DescriptorIndex already_marked = Marked::decode(raw_gc_state);
398  const DescriptorIndex delta = Delta::decode(raw_gc_state);
399  if ((already_marked + delta) >= index_to_mark) {
400  return false;
401  }
402  new_raw_gc_state = NewState(current_epoch, already_marked,
403  index_to_mark - already_marked);
404  }
405  if (SwapState(array, raw_gc_state, new_raw_gc_state)) {
406  return true;
407  }
408  }
409 }

412 std::pair<DescriptorArrayMarkingState::DescriptorIndex,
413  DescriptorArrayMarkingState::DescriptorIndex>
414 DescriptorArrayMarkingState::AcquireDescriptorRangeToMark(
415  unsigned gc_epoch, Tagged<DescriptorArray> array) {
416  const auto current_epoch = gc_epoch & Epoch::kMask;
417  while (true) {
418  const RawGCStateType raw_gc_state = array->raw_gc_state(kRelaxedLoad);
419  const DescriptorIndex marked = Marked::decode(raw_gc_state);
420  const DescriptorIndex delta = Delta::decode(raw_gc_state);
421  // We may encounter an array here that was merely pushed to the marker. In
422  // such a case, we process all descriptors (if we succeed). The cases to
423  // check are:
424  // 1. Epoch mismatch: Happens when descriptors survive a GC cycle.
425  // 2. Epoch matches but marked/delta is 0: Can happen when descriptors are
426  // newly allocated in the current cycle.
427  if (current_epoch != Epoch::decode(raw_gc_state) || (marked + delta) == 0) {
428  // In case number of descriptors is 0 and we reach the array through roots
429  // marking, mark also slack to get a proper transition from 0 marked to X
430  // marked. Otherwise, we would need to treat the state [0,0[ for marked
431  // and delta as valid state which leads to double-accounting through the
432  // marking barrier (when nof>1 in the barrier).
433  const int16_t number_of_descriptors =
434  array->number_of_descriptors() ? array->number_of_descriptors()
435  : array->number_of_all_descriptors();
436  DCHECK_GT(number_of_descriptors, 0);
437  if (SwapState(array, raw_gc_state,
438  NewState(current_epoch, number_of_descriptors, 0))) {
439  return {0, number_of_descriptors};
440  }
441  continue;
442  }
443 
444  // The delta is 0, so everything has been processed. Return the marked
445  // indices.
446  if (delta == 0) {
447  return {marked, marked};
448  }
449 
450  if (SwapState(array, raw_gc_state,
451  NewState(current_epoch, marked + delta, 0))) {
452  return {marked, marked + delta};
453  }
454  }
455 }
