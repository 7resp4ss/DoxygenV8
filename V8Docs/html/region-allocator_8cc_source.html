53 RegionAllocator::AllRegionsSet::iterator RegionAllocator::FindRegion(
54  Address address) {
55  if (!whole_region_.contains(address)) return all_regions_.end();
56 
57  Region key(address, 0, RegionState::kFree);
58  AllRegionsSet::iterator iter = all_regions_.upper_bound(&key);
59  // Regions in |all_regions_| are compared by end() values and key's end()
60  // points exactly to the address we are querying, so the upper_bound will
61  // find the region whose |end()| is greater than the requested address.
62  DCHECK_NE(iter, all_regions_.end());
63  DCHECK((*iter)->contains(address));
64  return iter;
65 }

90 void RegionAllocator::FreeListRemoveRegion(Region* region) {
91  DCHECK(region->is_free());
92  auto iter = free_regions_.find(region);
93  DCHECK_NE(iter, free_regions_.end());
94  DCHECK_EQ(region, *iter);
95  DCHECK_LE(region->size(), free_size_);
96  free_size_ -= region->size();
97  free_regions_.erase(iter);

100 RegionAllocator::Region* RegionAllocator::Split(Region* region,
101  size_t new_size) {
102  DCHECK(IsAligned(new_size, page_size_));
103  DCHECK_NE(new_size, 0);
104  DCHECK_GT(region->size(), new_size);
105 
106  if (on_split_) on_split_(region->begin(), new_size);
107 
108  // Create new region and put it to the lists after the |region|.
109  DCHECK(!region->is_excluded());
110  RegionState state = region->state();
111  Region* new_region =
112  new Region(region->begin() + new_size, region->size() - new_size, state);
113  if (state == RegionState::kFree) {
114  // Remove region from the free list before updating it's size.
115  FreeListRemoveRegion(region);
116  }
117  region->set_size(new_size);
118 
119  all_regions_.insert(new_region);
120 
121  if (state == RegionState::kFree) {
122  FreeListAddRegion(region);
123  FreeListAddRegion(new_region);
124  }
125  return new_region;
126 }

128 void RegionAllocator::Merge(AllRegionsSet::iterator prev_iter,
129  AllRegionsSet::iterator next_iter) {
130  Region* prev = *prev_iter;
131  Region* next = *next_iter;
132  DCHECK_EQ(prev->end(), next->begin());
133 
134  if (on_merge_) on_merge_(prev->begin(), prev->size() + next->size());
135 
136  prev->set_size(prev->size() + next->size());
137 
138  all_regions_.erase(next_iter); // prev_iter stays valid.
139 
140  // The |next| region must already not be in the free list.
141  DCHECK_EQ(free_regions_.find(next), free_regions_.end());
142  delete next;
143 }

145 RegionAllocator::Address RegionAllocator::AllocateRegion(
146  size_t size, AllocationStrategy allocation_strategy) {
147  DCHECK_NE(size, 0);
148  DCHECK(IsAligned(size, page_size_));
149 
150  Region* region = nullptr;
151 
152  if (allocation_strategy == AllocationStrategy::kFirstFit) {
153  region = FreeListFindRegion(size);
154  } else {
155  DCHECK_EQ(allocation_strategy, AllocationStrategy::kLargestFit);
156  region = FreeListFindLargestRegion(size);
157  }
158 
159  if (region == nullptr) return kAllocationFailure;
160 
161  if (region->size() != size) {
162  Split(region, size);
163  }
164  DCHECK(IsAligned(region->begin(), page_size_));
165  DCHECK_EQ(region->size(), size);
166 
167  // Mark region as used.
168  FreeListRemoveRegion(region);
169  region->set_state(RegionState::kAllocated);
170  return region->begin();
171 }

173 RegionAllocator::Address RegionAllocator::AllocateRegion(
174  RandomNumberGenerator* rng, size_t size) {
175  if (free_size() >= max_load_for_randomization_) {
176  // There is enough free space for trying to randomize the address.
177  size_t random = 0;
178 
179  for (int i = 0; i < kMaxRandomizationAttempts; i++) {
180  rng->NextBytes(&random, sizeof(random));
181  size_t random_offset = page_size_ * (random % region_size_in_pages_);
182  Address address = begin() + random_offset;
183  if (AllocateRegionAt(address, size, RegionState::kAllocated)) {
184  return address;
185  }
186  }
187  // Fall back to free list allocation.
188  }
189  return AllocateRegion(size);
190 }

192 bool RegionAllocator::AllocateRegionAt(Address requested_address, size_t size,
193  RegionState region_state) {
194  DCHECK(IsAligned(requested_address, page_size_));
195  DCHECK_NE(size, 0);
196  DCHECK(IsAligned(size, page_size_));
197  DCHECK_NE(region_state, RegionState::kFree);
198 
199  Address requested_end = requested_address + size;
200 
201  // Fail if the region would outgrow the total reservation or the addition
202  // overflows.
203  if (requested_end > end() || requested_end < requested_address) {
204  return false;
205  }
206 
207  DCHECK_LE(requested_end, end());
208  Region* region;
209  {
210  AllRegionsSet::iterator region_iter = FindRegion(requested_address);
211  if (region_iter == all_regions_.end()) {
212  return false;
213  }
214  region = *region_iter;
215  }
216  if (!region->is_free() || region->end() < requested_end) {
217  return false;
218  }
219  // Found free region that includes the requested one.
220  if (region->begin() != requested_address) {
221  // Split the region at the |requested_address| boundary.
222  size_t new_size = requested_address - region->begin();
223  DCHECK(IsAligned(new_size, page_size_));
224  region = Split(region, new_size);
225  }
226  if (region->end() != requested_end) {
227  // Split the region at the |requested_end| boundary.
228  Split(region, size);
229  }
230  DCHECK_EQ(region->begin(), requested_address);
231  DCHECK_EQ(region->size(), size);
232 
233  // Mark region as used.
234  FreeListRemoveRegion(region);
235  region->set_state(region_state);
236  return true;
237 }

239 RegionAllocator::Address RegionAllocator::AllocateAlignedRegion(
240  size_t size, size_t alignment) {
241  DCHECK(IsAligned(size, page_size_));
242  DCHECK(IsAligned(alignment, page_size_));
243  DCHECK_GE(alignment, page_size_);
244 
245  const size_t padded_size = size + alignment - page_size_;
246  Region* region = FreeListFindRegion(padded_size);
247  if (region == nullptr) {
248  // In case we are out of space we might still fit an allocation without
249  // padding and the result might still satisfy our alignment requirements.
250  region = FreeListFindRegion(size);
251  }
252  if (region == nullptr) return kAllocationFailure;
253 
254  if (!IsAligned(region->begin(), alignment)) {
255  size_t start = RoundUp(region->begin(), alignment);
256  if (start + size > region->end()) {
257  return kAllocationFailure;
258  }
259  region = Split(region, start - region->begin());
260  DCHECK_EQ(region->begin(), start);
261  DCHECK(IsAligned(region->begin(), alignment));
262  }
263 
264  if (region->size() != size) {
265  Split(region, size);
266  }
267  DCHECK(IsAligned(region->begin(), alignment));
268  DCHECK_EQ(region->size(), size);
269 
270  // Mark region as used.
271  FreeListRemoveRegion(region);
272  region->set_state(RegionState::kAllocated);
273  return region->begin();
274 }

277  size_t size,
278  size_t alignment) {
279  DCHECK(IsAligned(alignment, page_size()));
280  DCHECK(IsAligned(hint, alignment));
281 
282  if (hint && contains(hint, size)) {
283  if (AllocateRegionAt(hint, size)) {
284  return hint;

286  }
287 
288  Address address;
289  if (alignment <= page_size()) {

292  } else {
293  address = AllocateAlignedRegion(size, alignment);
294  }
295 
296  return address;

299 size_t RegionAllocator::TrimRegion(Address address, size_t new_size) {
300  DCHECK(IsAligned(new_size, page_size_));
301 
302  AllRegionsSet::iterator region_iter = FindRegion(address);
303  if (region_iter == all_regions_.end()) {
304  return 0;
305  }
306  Region* region = *region_iter;
307  if (region->begin() != address || !region->is_allocated()) {
308  return 0;
309  }
310 
311  // The region must not be in the free list.
312  DCHECK_EQ(free_regions_.find(*region_iter), free_regions_.end());
313 
314  if (new_size > 0) {
315  region = Split(region, new_size);
316  ++region_iter;
317  }
318  size_t size = region->size();
319  region->set_state(RegionState::kFree);
320 
321  // Merge current region with the surrounding ones if they are free.
322  if (region->end() != whole_region_.end()) {
323  // There must be a range after the current one.
324  AllRegionsSet::iterator next_iter = std::next(region_iter);
325  DCHECK_NE(next_iter, all_regions_.end());
326  if ((*next_iter)->is_free()) {
327  // |next| region object will be deleted during merge, remove it from
328  // the free list.
329  FreeListRemoveRegion(*next_iter);
330  Merge(region_iter, next_iter);
331  }
332  }
333  if (new_size == 0 && region->begin() != whole_region_.begin()) {
334  // There must be a range before the current one.
335  AllRegionsSet::iterator prev_iter = std::prev(region_iter);
336  DCHECK_NE(prev_iter, all_regions_.end());
337  if ((*prev_iter)->is_free()) {
338  // |prev| region's size will change, we'll have to re-insert it into
339  // the proper place of the free list.
340  FreeListRemoveRegion(*prev_iter);
341  Merge(prev_iter, region_iter);
342  // |prev| region becomes the current region.
343  region_iter = prev_iter;
344  region = *region_iter;
345  }
346  }
347  FreeListAddRegion(region);
348  return size;
349 }

351 bool RegionAllocator::TryGrowRegion(Address address, size_t new_size) {
352  DCHECK(IsAligned(new_size, page_size_));
353 
354  AllRegionsSet::iterator region_iter = FindRegion(address);
355  if (region_iter == all_regions_.end()) {
356  return false;
357  }
358  Region* region = *region_iter;
359  if (region->begin() != address || !region->is_allocated()) {
360  return false;
361  }
362 
363  // The region must not be in the free list.
364  DCHECK_EQ(free_regions_.find(*region_iter), free_regions_.end());
365  DCHECK_LT(region->size(), new_size);
366 
367  if (!AllocateRegionAt(region->end(), new_size - region->size())) {
368  return false;
369  }
370 
371  AllRegionsSet::iterator new_region_iter = std::next(region_iter);
372  DCHECK_NE(new_region_iter, all_regions_.end());
373 
374  Merge(region_iter, new_region_iter);
375  return true;
376 }
