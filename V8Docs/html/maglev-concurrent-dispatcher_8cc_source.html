128 CompilationJob::Status MaglevCompilationJob::ExecuteJobImpl(
129  RuntimeCallStats* stats, LocalIsolate* local_isolate) {
130  BeginPhaseKind("V8.MaglevExecuteJob");
131  LocalIsolateScope scope{info(), local_isolate};
132  if (!maglev::MaglevCompiler::Compile(local_isolate, info())) {
133  EndPhaseKind();
134  bailout_reason_ = BailoutReason::kMaglevGraphBuildingFailed;
135  return CompilationJob::FAILED;
136  }
137  EndPhaseKind();
138  // TODO(v8:7700): Actual return codes.
139  return CompilationJob::SUCCEEDED;

142 CompilationJob::Status MaglevCompilationJob::FinalizeJobImpl(Isolate* isolate) {
143  BeginPhaseKind("V8.MaglevFinalizeJob");
144  Handle<Code> code;
145  auto [maybe_code, bailout_reason] =
146  maglev::MaglevCompiler::GenerateCode(isolate, info());
147  if (!maybe_code.ToHandle(&code)) {
148  EndPhaseKind();
149  bailout_reason_ = bailout_reason;
150  return CompilationJob::FAILED;
151  }
152  // Functions with many inline candidates are sensitive to correct call
153  // frequency feedback and should therefore not be tiered up early.
154  if (v8_flags.profile_guided_optimization &&
155  info()->could_not_inline_all_candidates() &&
156  info()->toplevel_function()->shared()->cached_tiering_decision() !=
157  CachedTieringDecision::kDelayMaglev) {
158  info()->toplevel_function()->shared()->set_cached_tiering_decision(
159  CachedTieringDecision::kNormal);
160  }
161  info()->set_code(code);
162  GlobalHandleVector<Map> maps = CollectRetainedMaps(isolate, code);
163  RegisterWeakObjectsInOptimizedCode(
164  isolate, info()->broker()->target_native_context().object(), code,
165  std::move(maps));
166  EndPhaseKind();
167  return CompilationJob::SUCCEEDED;
168 }

178 void MaglevCompilationJob::DisposeOnMainThread(Isolate* isolate) {
179  // Drop canonical handles on the main thread, to avoid (in the case of
180  // background job destruction) needing to unpark the local isolate on the
181  // background thread for unregistering the identity map's strong roots.

204 void MaglevCompilationJob::RecordCompilationStats(Isolate* isolate) const {
205  // Don't record samples from machines without high-resolution timers,
206  // as that can cause serious reporting issues. See the thread at
207  // http://g/chrome-metrics-team/NwwJEyL8odU/discussion for more details.
208  if (base::TimeTicks::IsHighResolution()) {
209  Counters* const counters = isolate->counters();
210  counters->maglev_optimize_prepare()->AddSample(
211  static_cast<int>(time_taken_to_prepare_.InMicroseconds()));
212  counters->maglev_optimize_execute()->AddSample(
213  static_cast<int>(time_taken_to_execute_.InMicroseconds()));
214  counters->maglev_optimize_finalize()->AddSample(
215  static_cast<int>(time_taken_to_finalize_.InMicroseconds()));
216  counters->maglev_optimize_total_time()->AddSample(
217  static_cast<int>(ElapsedTime().InMicroseconds()));
218  }
219  if (v8_flags.trace_opt_stats) {
220  static double compilation_time = 0.0;
221  static int compiled_functions = 0;
222  static int code_size = 0;
223 
224  compilation_time += (time_taken_to_prepare_.InMillisecondsF() +
225  time_taken_to_execute_.InMillisecondsF() +
226  time_taken_to_finalize_.InMillisecondsF());
227  compiled_functions++;
228  code_size += function()->shared()->SourceSize();
229  PrintF(
230  "[maglev] Compiled: %d functions with %d byte source size in %fms.\n",
231  compiled_functions, code_size, compilation_time);
232  }
233 }

235 uint64_t MaglevCompilationJob::trace_id() const {
236  // Xor together the this pointer, the info pointer, and the top level
237  // function's Handle address, to try to make the id more unique on platforms
238  // where just the `this` pointer is likely to be reused.
239  return reinterpret_cast<uint64_t>(this) ^
240  reinterpret_cast<uint64_t>(info_.get()) ^
241  info_->toplevel_function().address() ^
242  info_->toplevel_function()->shared()->function_literal_id(
243  kRelaxedLoad);
244 }

258 // The JobTask is posted to V8::GetCurrentPlatform(). It's responsible for
259 // processing the incoming queue on a worker thread.
260 class MaglevConcurrentDispatcher::JobTask final : public v8::JobTask {
261  public:
262  explicit JobTask(MaglevConcurrentDispatcher* dispatcher)
263  : dispatcher_(dispatcher) {}
264 
265  void Run(JobDelegate* delegate) override {
266  if (incoming_queue()->IsEmpty() && destruction_queue()->IsEmpty()) {
267  return;
268  }
269  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"), "V8.MaglevTask");
270  base::FlushDenormalsScope flush_denormals_scope(
271  isolate()->flush_denormals());
272  LocalIsolate local_isolate(isolate(), ThreadKind::kBackground);
273  DCHECK(local_isolate.heap()->IsParked());
274 
275  std::unique_ptr<MaglevCompilationJob> job_to_destruct;
276  while (!delegate->ShouldYield()) {
277  std::unique_ptr<MaglevCompilationJob> job;
278  if (incoming_queue()->Dequeue(&job)) {
279  DCHECK_NOT_NULL(job);
280  TRACE_EVENT_WITH_FLOW0(
281  TRACE_DISABLED_BY_DEFAULT("v8.compile"), "V8.MaglevBackground",
282  job->trace_id(),
283  TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
284  RCS_SCOPE(&local_isolate,
285  RuntimeCallCounterId::kOptimizeBackgroundMaglev);
286  CompilationJob::Status status =
287  job->ExecuteJob(local_isolate.runtime_call_stats(), &local_isolate);
288  if (status == CompilationJob::SUCCEEDED) {
289  outgoing_queue()->Enqueue(std::move(job));
290  isolate()->stack_guard()->RequestInstallMaglevCode();
291  }
292  } else if (destruction_queue()->Dequeue(&job)) {
293  // Maglev jobs aren't cheap to destruct, so destroy them here in the
294  // background thread rather than on the main thread.
295  DCHECK_NOT_NULL(job);
296  TRACE_EVENT_WITH_FLOW0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
297  "V8.MaglevDestructBackground", job->trace_id(),
298  TRACE_EVENT_FLAG_FLOW_IN);
299  UnparkedScope unparked_scope(&local_isolate);
300  job.reset();
301  } else {
302  break;
303  }
304  }
305  }
306 
307  size_t GetMaxConcurrency(size_t worker_count) const override {
308  size_t num_tasks =
309  incoming_queue()->size() + destruction_queue()->size() + worker_count;
310  size_t max_threads = v8_flags.concurrent_maglev_max_threads;
311  if (max_threads > 0) {
312  return std::min(max_threads, num_tasks);
313  }
314  return num_tasks;
315  }
316 
317  private:
318  Isolate* isolate() const { return dispatcher_->isolate_; }
319  QueueT* incoming_queue() const { return &dispatcher_->incoming_queue_; }
320  QueueT* outgoing_queue() const { return &dispatcher_->outgoing_queue_; }
321  QueueT* destruction_queue() const { return &dispatcher_->destruction_queue_; }
322 
323  MaglevConcurrentDispatcher* const dispatcher_;
324 };

326 MaglevConcurrentDispatcher::MaglevConcurrentDispatcher(Isolate* isolate)
327  : isolate_(isolate) {
328  bool enable = v8_flags.concurrent_recompilation && maglev::IsMaglevEnabled();
329  if (enable) {
330  bool is_tracing =
331  v8_flags.print_maglev_code || v8_flags.trace_maglev_graph_building ||
332  v8_flags.trace_maglev_inlining || v8_flags.print_maglev_deopt_verbose ||
333  v8_flags.print_maglev_graph || v8_flags.print_maglev_graphs ||
334  v8_flags.trace_maglev_phi_untagging || v8_flags.trace_maglev_regalloc;
335 
336  if (is_tracing) {
337  PrintF("Concurrent maglev has been disabled for tracing.\n");
338  enable = false;
339  }
340  }
341  if (enable) {
342  TaskPriority priority = v8_flags.concurrent_maglev_high_priority_threads
343  ? TaskPriority::kUserBlocking
344  : TaskPriority::kUserVisible;
345  job_handle_ = V8::GetCurrentPlatform()->PostJob(
346  priority, std::make_unique<JobTask>(this));
347  DCHECK(is_enabled());
348  } else {
349  DCHECK(!is_enabled());
350  }
351 }

368 void MaglevConcurrentDispatcher::FinalizeFinishedJobs() {
369  HandleScope handle_scope(isolate_);
370  while (!outgoing_queue_.IsEmpty()) {
371  std::unique_ptr<MaglevCompilationJob> job;
372  outgoing_queue_.Dequeue(&job);
373  TRACE_EVENT_WITH_FLOW0(
374  TRACE_DISABLED_BY_DEFAULT("v8.compile"), "V8.MaglevConcurrentFinalize",
375  job->trace_id(), TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
376  RCS_SCOPE(isolate_,
377  RuntimeCallCounterId::kOptimizeConcurrentFinalizeMaglev);
378  Compiler::FinalizeMaglevCompilationJob(job.get(), isolate_);
379  job->DisposeOnMainThread(isolate_);
380  if (v8_flags.maglev_destroy_on_background) {
381  // Maglev jobs aren't cheap to destruct, so re-enqueue them for
382  // destruction on a background thread.
383  destruction_queue_.Enqueue(std::move(job));
384  job_handle_->NotifyConcurrencyIncrease();
385  } else {
386  TRACE_EVENT_WITH_FLOW0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
387  "V8.MaglevDestruct", job->trace_id(),
388  TRACE_EVENT_FLAG_FLOW_IN);
389  job.reset();
390  }
391  }
392 }

394 void MaglevConcurrentDispatcher::AwaitCompileJobs() {
395  // Use Join to wait until there are no more queued or running jobs.
396  {
397  AllowGarbageCollection allow_before_parking;
398  isolate_->main_thread_local_isolate()->ExecuteMainThreadWhileParked(
399  [this]() { job_handle_->Join(); });
400  }
401  // Join kills the job handle, so drop it and post a new one.
402  TaskPriority priority = v8_flags.concurrent_maglev_high_priority_threads
403  ? TaskPriority::kUserBlocking
404  : TaskPriority::kUserVisible;
405  job_handle_ = V8::GetCurrentPlatform()->PostJob(
406  priority, std::make_unique<JobTask>(this));
407  DCHECK(incoming_queue_.IsEmpty());
408 }

410 void MaglevConcurrentDispatcher::Flush(BlockingBehavior behavior) {
411  while (!incoming_queue_.IsEmpty()) {
412  std::unique_ptr<MaglevCompilationJob> job;
413  if (incoming_queue_.Dequeue(&job)) {
414  Compiler::DisposeMaglevCompilationJob(job.get(), isolate_);
415  }
416  }
417  while (!destruction_queue_.IsEmpty()) {
418  std::unique_ptr<MaglevCompilationJob> job;
419  destruction_queue_.Dequeue(&job);
420  }
421  if (behavior == BlockingBehavior::kBlock && job_handle_->IsValid()) {
422  AwaitCompileJobs();
423  }
424  while (!outgoing_queue_.IsEmpty()) {
425  std::unique_ptr<MaglevCompilationJob> job;
426  if (outgoing_queue_.Dequeue(&job)) {
427  Compiler::DisposeMaglevCompilationJob(job.get(), isolate_);
428  }
429  }
