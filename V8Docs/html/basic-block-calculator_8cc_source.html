26 void BasicBlockCalculator::ComputeBasicBlocks() {
27  DCHECK(basic_block_ranges_.empty());
28 
29  // Basic block rules.
30  // ------------------
31  // 1. The first basic block starts at function offset 0.
32  // 2. The 'else' instructions and the 'end' instructions end the current
33  // block. They are the last instruction in the basic block and the next
34  // instruction will start a new block. Likewise, all branch, call, tail-call,
35  // 'unreachable', 'return', 'if' and throw instructions end the current block.
36  // 3. The 'loop' instructions end the current block and start a new block.
37  // 4. The first instruction after the end of a block starts a new block.
38  // There can be unreachable instructions in between, which are not part of
39  // any block.
40  // 5. The last basic block ends at the offset of the final function 'end'
41  // instruction.
42 
43  BodyLocalDecls locals;
44  BytecodeIterator iterator(start_, end_, &locals, zone_);
45  DCHECK_LT(0, locals.encoded_size);
46 
47  // Start function.
48  current_basic_block_ = 0;
49  current_basic_block_start_pos_ = 0;
50  bool at_block_end = false;
51 
52  for (; iterator.has_next(); iterator.next()) {
53  WasmOpcode opcode = iterator.current();
54 
55  if (at_block_end) {
56  StartNewBlock(&iterator);
57  at_block_end = false;
58  }
59 
60  switch (opcode) {
61  case kExprLoop:
62  DCHECK_GE(current_basic_block_, 0);
63  if (iterator.position() > current_basic_block_start_pos_) {
64  basic_block_ranges_.emplace_back(current_basic_block_start_pos_,
65  iterator.position() - 1);
66  StartNewBlock(&iterator);
67  }
68  break;
69 
70  case kExprEnd:
71  case kExprIf:
72  case kExprElse:
73  case kExprUnreachable:
74  case kExprBr:
75  case kExprBrTable:
76  case kExprReturn:
77  case kExprDelegate:
78  case kExprRethrow:
79  case kExprThrow:
80  case kExprThrowRef:
81  case kExprCatch:
82  case kExprCatchAll:
83  case kExprBrIf:
84  case kExprBrOnNull:
85  case kExprBrOnNonNull:
86  case kExprBrOnCast:
87  case kExprBrOnCastFail:
88  case kExprBrOnCastDesc:
89  case kExprBrOnCastDescFail:
90  case kExprReturnCall:
91  case kExprReturnCallIndirect:
92  case kExprReturnCallRef:
93  case kExprCallFunction:
94  case kExprCallIndirect:
95  case kExprCallRef:
96  EndCurrentBlock(&iterator);
97  at_block_end = true;
98  break;
99 
100  default:
101  break;
102  }
103  }
104 
105  if (!at_block_end) {
106  basic_block_ranges_.emplace_back(current_basic_block_start_pos_,
107  iterator.position() - 1);
108  }
109 }
