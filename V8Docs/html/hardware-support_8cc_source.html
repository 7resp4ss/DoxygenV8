93 // static
94 CodeSandboxingMode SandboxHardwareSupport::CurrentSandboxingMode() {
95  if (!IsActive()) return CodeSandboxingMode::kUnsandboxed;
96 
97  auto key_permissions =
98  base::MemoryProtectionKey::GetKeyPermission(out_of_sandbox_pkey_);
99  if (key_permissions == base::MemoryProtectionKey::Permission::kDisableWrite) {
100  return CodeSandboxingMode::kSandboxed;
101  } else {
102  DCHECK_EQ(key_permissions,
103  base::MemoryProtectionKey::Permission::kNoRestrictions);
104  return CodeSandboxingMode::kUnsandboxed;
105  }
106 }

115 // static
116 bool SandboxHardwareSupport::TryActivate() {
117  DCHECK(!IsActive());
118 
119  if (!base::MemoryProtectionKey::HasMemoryProtectionKeyAPIs()) {
120  return false;
121  }
122 
123  sandbox_pkey_ = base::MemoryProtectionKey::AllocateKey();
124  if (sandbox_pkey_ == base::MemoryProtectionKey::kNoMemoryProtectionKey) {
125  return false;
126  }
127 
128  // Ideally, this would be the default protection key.
129  // See the comment at the declaration of these keys for more information
130  // about why that currently isn't the case.
131  out_of_sandbox_pkey_ = base::MemoryProtectionKey::AllocateKey();
132  if (out_of_sandbox_pkey_ ==
133  base::MemoryProtectionKey::kNoMemoryProtectionKey) {
134  base::MemoryProtectionKey::FreeKey(sandbox_pkey_);
135  sandbox_pkey_ = base::MemoryProtectionKey::kNoMemoryProtectionKey;
136  return false;
137  }
138 
139  extension_pkey_ = base::MemoryProtectionKey::AllocateKey();
140  if (extension_pkey_ == base::MemoryProtectionKey::kNoMemoryProtectionKey) {
141  base::MemoryProtectionKey::FreeKey(sandbox_pkey_);
142  base::MemoryProtectionKey::FreeKey(out_of_sandbox_pkey_);
143  sandbox_pkey_ = base::MemoryProtectionKey::kNoMemoryProtectionKey;
144  out_of_sandbox_pkey_ = base::MemoryProtectionKey::kNoMemoryProtectionKey;
145  return false;
146  }
147 
148  // Compute the PKEY mask for entering sandboxed execution mode. For that, we
149  // simply need to remove write access for the out-of-sandbox pkey.
150  sandboxed_mode_pkey_mask_ =
151  base::MemoryProtectionKey::ComputeRegisterMaskForPermissionSwitch(
152  out_of_sandbox_pkey_,
153  base::MemoryProtectionKey::Permission::kDisableWrite);
154  // We use zero to indicate that sandbox hardware support is inactive.
155  CHECK_NE(sandboxed_mode_pkey_mask_, 0);
156 
157  CHECK(IsActive());
158  return true;
159 }

162 // DisallowSandboxAccess scopes can be arbitrarily nested and even attached to
163 // heap-allocated objects (so their lifetime isn't necessarily tied to a stack
164 // frame). For that to work correctly, we need to track the activation count in
165 // a per-thread global variable.
166 thread_local unsigned disallow_sandbox_access_activation_counter_ = 0;
167 // AllowSandboxAccess scopes on the other hand cannot be nested. There must be
168 // at most a single one active at any point in time. These are supposed to only
169 // be used for short sequences of code that's otherwise running with an active
170 // DisallowSandboxAccess scope.
171 thread_local bool has_active_allow_sandbox_access_scope_ = false;
172 
173 DisallowSandboxAccess::DisallowSandboxAccess() {
174  pkey_ = SandboxHardwareSupport::sandbox_pkey_;
175  if (pkey_ == base::MemoryProtectionKey::kNoMemoryProtectionKey) {
176  return;
177  }
178 
179  // Using a DisallowSandboxAccess inside an AllowSandboxAccess isn't currently
180  // allowed, but we could add support for that in the future if needed.
181  DCHECK_WITH_MSG(!has_active_allow_sandbox_access_scope_,
182  "DisallowSandboxAccess cannot currently be nested inside an "
183  "AllowSandboxAccess");
184 
185  if (disallow_sandbox_access_activation_counter_ == 0) {
186  DCHECK_EQ(base::MemoryProtectionKey::GetKeyPermission(pkey_),
187  base::MemoryProtectionKey::Permission::kNoRestrictions);
188  base::MemoryProtectionKey::SetPermissionsForKey(
189  pkey_, base::MemoryProtectionKey::Permission::kDisableAccess);
190  }
191  disallow_sandbox_access_activation_counter_ += 1;
192 }
193 
194 DisallowSandboxAccess::~DisallowSandboxAccess() {
195  if (pkey_ == base::MemoryProtectionKey::kNoMemoryProtectionKey) {
196  return;
197  }
198 
199  DCHECK_NE(disallow_sandbox_access_activation_counter_, 0);
200  disallow_sandbox_access_activation_counter_ -= 1;
201  if (disallow_sandbox_access_activation_counter_ == 0) {
202  DCHECK_EQ(base::MemoryProtectionKey::GetKeyPermission(pkey_),
203  base::MemoryProtectionKey::Permission::kDisableAccess);
204  base::MemoryProtectionKey::SetPermissionsForKey(
205  pkey_, base::MemoryProtectionKey::Permission::kNoRestrictions);
206  }
207 }

209 AllowSandboxAccess::AllowSandboxAccess() {
210  if (disallow_sandbox_access_activation_counter_ == 0) {
211  // This means that either scope enforcement is disabled due to a lack of
212  // PKEY support on the system or that there is no active
213  // DisallowSandboxAccess. In both cases, we don't need to do anything here
214  // and this scope object is just a no-op.
215  pkey_ = base::MemoryProtectionKey::kNoMemoryProtectionKey;
216  return;
217  }
218 
219  DCHECK_WITH_MSG(!has_active_allow_sandbox_access_scope_,
220  "AllowSandboxAccess scopes cannot be nested");
221  has_active_allow_sandbox_access_scope_ = true;
222 
223  pkey_ = SandboxHardwareSupport::sandbox_pkey_;
224  // We must have an active DisallowSandboxAccess so PKEYs must be supported.
225  DCHECK_NE(pkey_, base::MemoryProtectionKey::kNoMemoryProtectionKey);
226 
227  DCHECK_EQ(base::MemoryProtectionKey::GetKeyPermission(pkey_),
228  base::MemoryProtectionKey::Permission::kDisableAccess);
229  base::MemoryProtectionKey::SetPermissionsForKey(
230  pkey_, base::MemoryProtectionKey::Permission::kNoRestrictions);
231 }
232 
233 AllowSandboxAccess::~AllowSandboxAccess() {
234  if (pkey_ == base::MemoryProtectionKey::kNoMemoryProtectionKey) {
235  // There was no DisallowSandboxAccess scope active when this
236  // AllowSandboxAccess scope was created, and we don't expect one to have
237  // been created in the meantime.
238  DCHECK_EQ(disallow_sandbox_access_activation_counter_, 0);
239  return;
240  }
241 
242  // There was an active DisallowSandboxAccess scope when this
243  // AllowSandboxAccess scope was created, and we expect it to still be there.
244  DCHECK_GT(disallow_sandbox_access_activation_counter_, 0);
245 
246  DCHECK(has_active_allow_sandbox_access_scope_);
247  has_active_allow_sandbox_access_scope_ = false;
248 
249  DCHECK_EQ(base::MemoryProtectionKey::GetKeyPermission(pkey_),
250  base::MemoryProtectionKey::Permission::kNoRestrictions);
251  base::MemoryProtectionKey::SetPermissionsForKey(
252  pkey_, base::MemoryProtectionKey::Permission::kDisableAccess);
253 }
