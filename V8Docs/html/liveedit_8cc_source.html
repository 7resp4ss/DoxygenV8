59 // Finds common prefix and suffix in input. This parts shouldn't take space in
60 // linear programming table. Enable subranging in input and output.
61 void NarrowDownInput(SubrangableInput* input, SubrangableOutput* output) {
62  const int len1 = input->GetLength1();
63  const int len2 = input->GetLength2();
64 
65  int common_prefix_len;
66  int common_suffix_len;
67 
68  {
69  common_prefix_len = 0;
70  int prefix_limit = std::min(len1, len2);
71  while (common_prefix_len < prefix_limit &&
72  input->Equals(common_prefix_len, common_prefix_len)) {
73  common_prefix_len++;
74  }
75 
76  common_suffix_len = 0;
77  int suffix_limit =
78  std::min(len1 - common_prefix_len, len2 - common_prefix_len);
79 
80  while (common_suffix_len < suffix_limit &&
81  input->Equals(len1 - common_suffix_len - 1,
82  len2 - common_suffix_len - 1)) {
83  common_suffix_len++;
84  }
85  }
86 
87  if (common_prefix_len > 0 || common_suffix_len > 0) {
88  int new_len1 = len1 - common_suffix_len - common_prefix_len;
89  int new_len2 = len2 - common_suffix_len - common_prefix_len;
90 
91  input->SetSubrange1(common_prefix_len, new_len1);
92  input->SetSubrange2(common_prefix_len, new_len2);
93 
94  output->SetSubrange1(common_prefix_len, new_len1);
95  output->SetSubrange2(common_prefix_len, new_len2);
96  }
97 }

287 struct SourcePositionEvent {
288  enum Type { LITERAL_STARTS, LITERAL_ENDS, DIFF_STARTS, DIFF_ENDS };
289 
290  int position;
291  Type type;
292 
293  union {
294  FunctionLiteral* literal;
295  int pos_diff;
296  };
297 
298  SourcePositionEvent(FunctionLiteral* literal, bool is_start)
299  : position(is_start ? literal->start_position()
300  : literal->end_position()),
301  type(is_start ? LITERAL_STARTS : LITERAL_ENDS),
302  literal(literal) {}
303  SourcePositionEvent(const SourceChangeRange& change, bool is_start)
304  : position(is_start ? change.start_position : change.end_position),
305  type(is_start ? DIFF_STARTS : DIFF_ENDS),
306  pos_diff((change.new_end_position - change.new_start_position) -
307  (change.end_position - change.start_position)) {}
308 
309  static bool LessThan(const SourcePositionEvent& a,
310  const SourcePositionEvent& b) {
311  if (a.position != b.position) return a.position < b.position;
312  if (a.type != b.type) return a.type < b.type;
313  if (a.type == LITERAL_STARTS && b.type == LITERAL_STARTS) {
314  // If the literals start in the same position, we want the one with the
315  // furthest (i.e. largest) end position to be first.
316  if (a.literal->end_position() != b.literal->end_position()) {
317  return a.literal->end_position() > b.literal->end_position();
318  }
319  // If they also end in the same position, we want the first in order of
320  // literal ids to be first.
321  return a.literal->function_literal_id() <
322  b.literal->function_literal_id();
323  } else if (a.type == LITERAL_ENDS && b.type == LITERAL_ENDS) {
324  // If the literals end in the same position, we want the one with the
325  // nearest (i.e. largest) start position to be first.
326  if (a.literal->start_position() != b.literal->start_position()) {
327  return a.literal->start_position() > b.literal->start_position();
328  }
329  // If they also end in the same position, we want the last in order of
330  // literal ids to be first.
331  return a.literal->function_literal_id() >
332  b.literal->function_literal_id();
333  } else {
334  return a.pos_diff < b.pos_diff;
335  }
336  }
337 };

339 struct FunctionLiteralChange {
340  // If any of start/end position is kNoSourcePosition, this literal is

342  int new_start_position;

354 using FunctionLiteralChanges =
355  std::unordered_map<FunctionLiteral*, FunctionLiteralChange>;
356 void CalculateFunctionLiteralChanges(
357  const std::vector<FunctionLiteral*>& literals,
358  const std::vector<SourceChangeRange>& diffs,
359  FunctionLiteralChanges* result) {
360  std::vector<SourcePositionEvent> events;
361  events.reserve(literals.size() * 2 + diffs.size() * 2);
362  for (FunctionLiteral* literal : literals) {
363  events.emplace_back(literal, true);
364  events.emplace_back(literal, false);
365  }
366  for (const SourceChangeRange& diff : diffs) {
367  events.emplace_back(diff, true);
368  events.emplace_back(diff, false);
369  }
370  std::sort(events.begin(), events.end(), SourcePositionEvent::LessThan);
371  bool inside_diff = false;
372  int delta = 0;
373  std::stack<std::pair<FunctionLiteral*, FunctionLiteralChange>> literal_stack;
374  for (const SourcePositionEvent& event : events) {
375  switch (event.type) {
376  case SourcePositionEvent::DIFF_ENDS:
377  DCHECK(inside_diff);
378  inside_diff = false;
379  delta += event.pos_diff;
380  break;
381  case SourcePositionEvent::LITERAL_ENDS: {
382  DCHECK_EQ(literal_stack.top().first, event.literal);
383  FunctionLiteralChange& change = literal_stack.top().second;
384  change.new_end_position = inside_diff
385  ? kNoSourcePosition
386  : event.literal->end_position() + delta;
387  result->insert(literal_stack.top());
388  literal_stack.pop();
389  break;
390  }
391  case SourcePositionEvent::LITERAL_STARTS:
392  literal_stack.push(std::make_pair(
393  event.literal,
394  FunctionLiteralChange(
395  inside_diff ? kNoSourcePosition
396  : event.literal->start_position() + delta,
397  literal_stack.empty() ? nullptr : literal_stack.top().first)));
398  break;
399  case SourcePositionEvent::DIFF_STARTS:
400  DCHECK(!inside_diff);
401  inside_diff = true;
402  if (!literal_stack.empty()) {
403  // Note that outer literal has not necessarily changed, unless the
404  // diff goes past the end of this literal. In this case, we'll mark
405  // this function as damaged and parent as changed later in
406  // MapLiterals.
407  literal_stack.top().second.has_changes = true;
408  }
409  break;
410  }
411  }
412 }

414 // Function which has not changed itself, but if any variable in its
415 // outer context has been added/removed, we must consider this function
416 // as damaged and not update references to it.
417 // This is because old compiled function has hardcoded references to
418 // it's outer context.
419 bool HasChangedScope(FunctionLiteral* a, FunctionLiteral* b) {
420  Scope* scope_a = a->scope()->outer_scope();
421  Scope* scope_b = b->scope()->outer_scope();
422  while (scope_a && scope_b) {
423  std::unordered_map<int, Handle<String>> vars;
424  for (Variable* var : *scope_a->locals()) {
425  if (!var->IsContextSlot()) continue;
426  vars[var->index()] = var->name();
427  }
428  for (Variable* var : *scope_b->locals()) {
429  if (!var->IsContextSlot()) continue;
430  auto it = vars.find(var->index());
431  if (it == vars.end()) return true;
432  if (*it->second != *var->name()) return true;
433  }
434  scope_a = scope_a->outer_scope();
435  scope_b = scope_b->outer_scope();
436  }
437  return scope_a != scope_b;
438 }

442 using LiteralMap = std::unordered_map<FunctionLiteral*, FunctionLiteral*>;
443 void MapLiterals(const FunctionLiteralChanges& changes,
444  const std::vector<FunctionLiteral*>& new_literals,
445  LiteralMap* unchanged, LiteralMap* changed) {
446  // Track the top-level script function separately as it can overlap fully with
447  // another function, e.g. the script "()=>42".
448  const std::pair<int, int> kTopLevelMarker = std::make_pair(-1, -1);
449  std::map<std::pair<int, int>, FunctionLiteral*> position_to_new_literal;
450  for (FunctionLiteral* literal : new_literals) {
451  DCHECK(literal->start_position() != kNoSourcePosition);
452  DCHECK(literal->end_position() != kNoSourcePosition);
453  std::pair<int, int> key =
454  literal->function_literal_id() == kFunctionLiteralIdTopLevel
455  ? kTopLevelMarker
456  : std::make_pair(literal->start_position(),
457  literal->end_position());
458  // Make sure there are no duplicate keys.
459  DCHECK_EQ(position_to_new_literal.find(key), position_to_new_literal.end());
460  position_to_new_literal[key] = literal;
461  }
462  LiteralMap mappings;
463  std::unordered_map<FunctionLiteral*, ChangeState> change_state;
464  for (const auto& change_pair : changes) {
465  FunctionLiteral* literal = change_pair.first;
466  const FunctionLiteralChange& change = change_pair.second;
467  std::pair<int, int> key =
468  literal->function_literal_id() == kFunctionLiteralIdTopLevel
469  ? kTopLevelMarker
470  : std::make_pair(change.new_start_position,
471  change.new_end_position);
472  auto it = position_to_new_literal.find(key);
473  if (it == position_to_new_literal.end() ||
474  HasChangedScope(literal, it->second)) {
475  change_state[literal] = ChangeState::DAMAGED;
476  if (!change.outer_literal) continue;
477  if (change_state[change.outer_literal] != ChangeState::DAMAGED) {
478  change_state[change.outer_literal] = ChangeState::CHANGED;
479  }
480  } else {
481  mappings[literal] = it->second;
482  if (change_state.find(literal) == change_state.end()) {
483  change_state[literal] =
484  change.has_changes ? ChangeState::CHANGED : ChangeState::UNCHANGED;
485  }
486  }
487  }
488  for (const auto& mapping : mappings) {
489  if (change_state[mapping.first] == ChangeState::UNCHANGED) {
490  (*unchanged)[mapping.first] = mapping.second;
491  } else if (change_state[mapping.first] == ChangeState::CHANGED) {
492  (*changed)[mapping.first] = mapping.second;
493  }
494  }
495 }

531 bool ParseScript(Isolate* isolate, Handle<Script> script, ParseInfo* parse_info,
532  MaybeDirectHandle<ScopeInfo> outer_scope_info,
533  bool compile_as_well, std::vector<FunctionLiteral*>* literals,
534  std::map<uint32_t, std::vector<uint32_t>>* eval_calls,

543  } else {
544  success =
545  parsing::ParseProgram(parse_info, script, outer_scope_info, isolate,
546  parsing::ReportStatisticsMode::kYes);
547  if (!success) {
548  // Throw the parser error.
549  parse_info->pending_error_handler()->PrepareErrors(

551  parse_info->pending_error_handler()->ReportErrors(isolate, script);

553  }

557  i::DirectHandle<i::JSMessageObject> msg =

560  result->line_number = msg->GetLineNumber();
561  result->column_number = msg->GetColumnNumber();
562  result->status = debug::LiveEditResult::COMPILE_ERROR;
563  return false;
564  }
565  CollectFunctionLiterals(isolate, parse_info->literal())
566  .Run(literals, eval_calls);
567  return true;

851 void LiveEdit::PatchScript(Isolate* isolate, Handle<Script> script,
852  Handle<String> new_source, bool preview,
853  bool allow_top_frame_live_editing,
854  debug::LiveEditResult* result) {
855  std::vector<SourceChangeRange> diffs;
856  LiveEdit::CompareStrings(isolate,
857  handle(Cast<String>(script->source()), isolate),
858  new_source, &diffs);
859  if (diffs.empty()) {
860  result->status = debug::LiveEditResult::OK;
861  return;
862  }
863 
864  ReusableUnoptimizedCompileState reusable_state(isolate);
865 
866  UnoptimizedCompileState compile_state;
867  UnoptimizedCompileFlags flags =
868  UnoptimizedCompileFlags::ForScriptCompile(isolate, *script);
869  flags.set_is_eager(true);
870  flags.set_is_reparse(true);
871  ParseInfo parse_info(isolate, flags, &compile_state, &reusable_state);
872  MaybeDirectHandle<ScopeInfo> outer_scope_info =
873  DetermineOuterScopeInfo(isolate, script);
874  std::vector<FunctionLiteral*> literals;
875  std::map<uint32_t, std::vector<uint32_t>> eval_calls;
876  if (!ParseScript(isolate, script, &parse_info, outer_scope_info, false,
877  &literals, &eval_calls, result))
878  return;
879 
880  Handle<Script> new_script =
881  isolate->factory()->CloneScript(script, new_source);
882  UnoptimizedCompileState new_compile_state;
883  UnoptimizedCompileFlags new_flags =
884  UnoptimizedCompileFlags::ForScriptCompile(isolate, *new_script);
885  new_flags.set_is_eager(true);
886  ParseInfo new_parse_info(isolate, new_flags, &new_compile_state,
887  &reusable_state);
888  std::vector<FunctionLiteral*> new_literals;
889  std::map<uint32_t, std::vector<uint32_t>> new_eval_calls;
890  if (!ParseScript(isolate, new_script, &new_parse_info, outer_scope_info, true,
891  &new_literals, &new_eval_calls, result)) {
892  return;
893  }
894 
895  FunctionLiteralChanges literal_changes;
896  CalculateFunctionLiteralChanges(literals, diffs, &literal_changes);
897 
898  LiteralMap changed;
899  LiteralMap unchanged;
900  MapLiterals(literal_changes, new_literals, &unchanged, &changed);
901 
902  FunctionDataMap function_data_map;
903  for (const auto& mapping : changed) {
904  function_data_map.AddInterestingLiteral(script->id(), mapping.first);
905  function_data_map.AddInterestingLiteral(new_script->id(), mapping.second);
906  }
907  for (const auto& mapping : unchanged) {
908  function_data_map.AddInterestingLiteral(script->id(), mapping.first);
909  }
910  function_data_map.Fill(isolate);
911 
912  if (!CanPatchScript(changed, script, new_script, function_data_map,
913  allow_top_frame_live_editing, result)) {
914  return;
915  }
916 
917  if (preview) {
918  result->status = debug::LiveEditResult::OK;
919  return;
920  }
921 
922  // Patching a script means that the bytecode on the stack may no longer
923  // correspond to the bytecode of the JSFunction for that frame. As a result
924  // it is no longer safe to flush bytecode since we might flush the new
925  // bytecode for a JSFunction that is on the stack with an old bytecode, which
926  // breaks the invariant that any JSFunction active on the stack is compiled.
927  isolate->set_disable_bytecode_flushing(true);
928 
929  std::map<int, int> start_position_to_unchanged_id;
930  for (const auto& mapping : unchanged) {
931  FunctionData* data = nullptr;
932  if (!function_data_map.Lookup(script, mapping.first, &data)) continue;
933  Handle<SharedFunctionInfo> sfi;
934  if (!data->shared.ToHandle(&sfi)) continue;
935  DCHECK_EQ(sfi->script(), *script);
936 
937  isolate->compilation_cache()->Remove(sfi);
938  isolate->debug()->DeoptimizeFunction(sfi);
939  if (std::optional<Tagged<DebugInfo>> di = sfi->TryGetDebugInfo(isolate)) {
940  DirectHandle<DebugInfo> debug_info(di.value(), isolate);
941  isolate->debug()->RemoveBreakInfoAndMaybeFree(debug_info);
942  }
943  SharedFunctionInfo::EnsureSourcePositionsAvailable(isolate, sfi);
944  UpdatePositions(isolate, sfi, mapping.second, diffs);
945 
946  sfi->set_script(*new_script, kReleaseStore);
947  std::vector<uint32_t> source_infos =
948  eval_calls[sfi->function_literal_id(kRelaxedLoad)];
949  sfi->set_function_literal_id(mapping.second->function_literal_id(),
950  kRelaxedStore);
951  std::vector<uint32_t> target_infos =
952  new_eval_calls[sfi->function_literal_id(kRelaxedLoad)];
953  new_script->infos()->set(mapping.second->function_literal_id(),
954  MakeWeak(*sfi));
955  if (sfi->HasBytecodeArray()) {
956  for (size_t i = 0; i < source_infos.size(); i++) {
957  Tagged<ScopeInfo> scope_info = Cast<ScopeInfo>(
958  script->infos()->get(source_infos[i]).GetHeapObjectAssumeWeak());
959  new_script->infos()->set(target_infos[i], MakeWeak(scope_info));
960  }
961  }
962  DCHECK_EQ(sfi->function_literal_id(kRelaxedLoad),
963  mapping.second->function_literal_id());
964 
965  // Save the new start_position -> id mapping, so that we can recover it when
966  // iterating over changed functions' constant pools.
967  start_position_to_unchanged_id[mapping.second->start_position()] =
968  mapping.second->function_literal_id();
969 
970  if (sfi->HasUncompiledDataWithPreparseData(isolate)) {
971  sfi->ClearPreparseData(isolate);
972  }
973 
974  for (auto& js_function : data->js_functions) {
975  js_function->set_raw_feedback_cell(
976  *isolate->factory()->many_closures_cell());
977  if (!js_function->is_compiled(isolate)) continue;
978  IsCompiledScope is_compiled_scope(
979  js_function->shared()->is_compiled_scope(isolate));
980  JSFunction::EnsureFeedbackVector(isolate, js_function,
981  &is_compiled_scope);
982  }
983 
984  if (!sfi->HasBytecodeArray()) continue;
985  Tagged<TrustedFixedArray> constants =
986  sfi->GetBytecodeArray(isolate)->constant_pool();
987  for (int i = 0; i < constants->length(); ++i) {
988  if (!IsSharedFunctionInfo(constants->get(i))) continue;
989  data = nullptr;
990  if (!function_data_map.Lookup(Cast<SharedFunctionInfo>(constants->get(i)),
991  &data)) {
992  continue;
993  }
994  auto change_it = changed.find(data->literal);
995  if (change_it == changed.end()) continue;
996  if (!function_data_map.Lookup(new_script, change_it->second, &data)) {
997  continue;
998  }
999  Handle<SharedFunctionInfo> new_sfi;
1000  if (!data->shared.ToHandle(&new_sfi)) continue;
1001  constants->set(i, *new_sfi);
1002  }
1003  }
1004  isolate->LocalsBlockListCacheRehash();
1005  for (const auto& mapping : changed) {
1006  FunctionData* data = nullptr;
1007  if (!function_data_map.Lookup(new_script, mapping.second, &data)) continue;
1008  Handle<SharedFunctionInfo> new_sfi;
1009  // In most cases the new FunctionLiteral should also have an SFI, but there
1010  // are some exceptions. E.g the compiler doesn't create SFIs for
1011  // inner functions that are never referenced.
1012  if (!data->shared.ToHandle(&new_sfi)) continue;
1013  DCHECK_EQ(new_sfi->script(), *new_script);
1014 
1015  if (!function_data_map.Lookup(script, mapping.first, &data)) continue;
1016  Handle<SharedFunctionInfo> sfi;
1017  if (!data->shared.ToHandle(&sfi)) continue;
1018 
1019  isolate->debug()->DeoptimizeFunction(sfi);
1020  isolate->compilation_cache()->Remove(sfi);
1021  for (auto& js_function : data->js_functions) {
1022  js_function->set_raw_feedback_cell(
1023  *isolate->factory()->many_closures_cell());
1024 #ifdef V8_ENABLE_LEAPTIERING
1025  auto code = handle(new_sfi->GetCode(isolate), isolate);
1026  JSFunction::AllocateDispatchHandle(
1027  js_function, isolate,
1028  new_sfi->internal_formal_parameter_count_with_receiver(), code);
1029 #endif
1030  js_function->set_shared(*new_sfi);
1031 
1032  if (!js_function->is_compiled(isolate)) continue;
1033  IsCompiledScope is_compiled_scope(
1034  js_function->shared()->is_compiled_scope(isolate));
1035  JSFunction::EnsureFeedbackVector(isolate, js_function,
1036  &is_compiled_scope);
1037  }
1038  }
1039  SharedFunctionInfo::ScriptIterator it(isolate, *new_script);
1040  for (Tagged<SharedFunctionInfo> sfi = it.Next(); !sfi.is_null();
1041  sfi = it.Next()) {
1042  if (!sfi->HasBytecodeArray()) continue;
1043  Tagged<TrustedFixedArray> constants =
1044  sfi->GetBytecodeArray(isolate)->constant_pool();
1045  for (int i = 0; i < constants->length(); ++i) {
1046  if (!IsSharedFunctionInfo(constants->get(i))) continue;
1047  Tagged<SharedFunctionInfo> inner_sfi =
1048  Cast<SharedFunctionInfo>(constants->get(i));
1049  // See if there is a mapping from this function's start position to an
1050  // unchanged function's id.
1051  auto unchanged_it =
1052  start_position_to_unchanged_id.find(inner_sfi->StartPosition());
1053  if (unchanged_it == start_position_to_unchanged_id.end()) continue;
1054

1056  // already been updated in in the unchanged pass.
1057  Tagged<SharedFunctionInfo> old_unchanged_inner_sfi =
1058  Cast<SharedFunctionInfo>(
1059  new_script->infos()->get(unchanged_it->second).GetHeapObject());
1060  if (old_unchanged_inner_sfi == inner_sfi) continue;
1061  DCHECK_NE(old_unchanged_inner_sfi, inner_sfi);
1062  // Now some sanity checks. Make sure that the unchanged SFI has already
1063  // been processed and patched to be on the new script ...
1064  DCHECK_EQ(old_unchanged_inner_sfi->script(), *new_script);
1065  constants->set(i, old_unchanged_inner_sfi);
1066  }
1067  }
1068 #ifdef DEBUG
