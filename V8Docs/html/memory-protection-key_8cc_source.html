28 int GetProtectionFromMemoryPermission(PageAllocator::Permission permission) {
29  // Mappings for PKU are either RWX (for code), no access (for uncommitted
30  // memory), or RO for globals.
31  switch (permission) {
32  case PageAllocator::kNoAccess:
33  return PROT_NONE;
34  case PageAllocator::kRead:
35  return PROT_READ;
36  case PageAllocator::kReadWrite:
37  return PROT_READ | PROT_WRITE;
38  case PageAllocator::kReadWriteExecute:
39  return PROT_READ | PROT_WRITE | PROT_EXEC;
40  default:
41  UNREACHABLE();
42  }
43 }

72 // static
73 int MemoryProtectionKey::AllocateKey() {
74  if (!pkey_alloc) {
75  return kNoMemoryProtectionKey;
76  }
77 
78  int key = pkey_alloc(0, kNoRestrictions);
79  if (key != kNoMemoryProtectionKey) {
80  CHECK_LT(key, kMaxAvailableKeys);
81  DCHECK(!g_active_keys[key]);
82  g_active_keys[key] = true;
83  }
84 
85  return key;
86 }

103 // static
104 bool MemoryProtectionKey::SetPermissionsAndKey(
105  base::AddressRegion region, v8::PageAllocator::Permission page_permissions,
106  int key) {
107  DCHECK_NE(key, kNoMemoryProtectionKey);
108  CHECK_NOT_NULL(pkey_mprotect);
109 
110  void* address = reinterpret_cast<void*>(region.begin());
111  size_t size = region.size();
112 
113  int protection = GetProtectionFromMemoryPermission(page_permissions);
114 
115  return pkey_mprotect(address, size, protection, key) == 0;
116 }

130 MemoryProtectionKey::Permission MemoryProtectionKey::GetKeyPermission(int key) {
131  DCHECK_NE(kNoMemoryProtectionKey, key);
132 
133  // If a valid key was allocated, {pkey_get()} must also be available.
134  DCHECK_NOT_NULL(pkey_get);
135 
136  int permission = pkey_get(key);
137  CHECK(permission == kNoRestrictions || permission == kDisableAccess ||
138  permission == kDisableWrite);
139  return static_cast<Permission>(permission);

143 uint32_t MemoryProtectionKey::ComputeRegisterMaskForPermissionSwitch(
144  int key, Permission permissions) {
145  DCHECK_NE(permissions, kNoRestrictions);
146 
147 #if defined(V8_TARGET_ARCH_X64)
148  constexpr int kBitsPerKey = 2;

158  // NOTE: This code MUST be async-signal safe
159 
160  // As a future optimization, we could compute the register state first (or
