69  template <typename NodeT>
70  ProcessResult Process(NodeT* node, const ProcessingState& state) {
71  if constexpr (NodeT::kProperties.is_call() ||
72  NodeT::kProperties.needs_register_snapshot()) {
73  int node_stack_args = node->MaxCallStackArgs();
74  if constexpr (NodeT::kProperties.needs_register_snapshot()) {
75  // Pessimistically assume that we'll push all registers in deferred
76  // calls.
77  node_stack_args +=
78  kAllocatableGeneralRegisterCount + kAllocatableDoubleRegisterCount;
79  }
80  max_call_stack_args_ = std::max(max_call_stack_args_, node_stack_args);
81  }
82  if constexpr (NodeT::kProperties.can_eager_deopt()) {
83  UpdateMaxDeoptedStackSize(node->eager_deopt_info());
84  }
85  if constexpr (NodeT::kProperties.can_lazy_deopt()) {
86  UpdateMaxDeoptedStackSize(node->lazy_deopt_info());
87  }
88  return ProcessResult::kContinue;
89  }
90 
91  private:
92  void UpdateMaxDeoptedStackSize(DeoptInfo* deopt_info) {
93  const DeoptFrame* deopt_frame = &deopt_info->top_frame();
94  int frame_size = 0;
95  if (deopt_frame->type() == DeoptFrame::FrameType::kInterpretedFrame) {
96  if (&deopt_frame->as_interpreted().unit() == last_seen_unit_) return;
97  last_seen_unit_ = &deopt_frame->as_interpreted().unit();
98  frame_size = deopt_frame->as_interpreted().unit().max_arguments() *
99  kSystemPointerSize;
100  }
101 
102  do {
103  frame_size += ConservativeFrameSize(deopt_frame);
104  deopt_frame = deopt_frame->parent();
105  } while (deopt_frame != nullptr);
106  max_deopted_stack_size_ = std::max(frame_size, max_deopted_stack_size_);
107  }
108  int ConservativeFrameSize(const DeoptFrame* deopt_frame) {
109  switch (deopt_frame->type()) {
110  case DeoptFrame::FrameType::kInterpretedFrame: {
111  auto info = UnoptimizedFrameInfo::Conservative(
112  deopt_frame->as_interpreted().unit().parameter_count(),
113  deopt_frame->as_interpreted().unit().register_count());
114  return info.frame_size_in_bytes();
115  }
116  case DeoptFrame::FrameType::kConstructInvokeStubFrame: {
117  return FastConstructStubFrameInfo::Conservative().frame_size_in_bytes();
118  }
119  case DeoptFrame::FrameType::kInlinedArgumentsFrame: {
120  return std::max(
121  0,
122  static_cast<int>(
123  deopt_frame->as_inlined_arguments().arguments().size() -
124  deopt_frame->as_inlined_arguments().unit().parameter_count()) *
125  kSystemPointerSize);
126  }
127  case DeoptFrame::FrameType::kBuiltinContinuationFrame: {
128  // PC + FP + Closure + Params + Context
129  const RegisterConfiguration* config = RegisterConfiguration::Default();
130  auto info = BuiltinContinuationFrameInfo::Conservative(
131  deopt_frame->as_builtin_continuation().parameters().length(),
132  Builtins::CallInterfaceDescriptorFor(
133  deopt_frame->as_builtin_continuation().builtin_id()),
134  config);
135  return info.frame_size_in_bytes();
136  }
137  }
138  }
139 
140  int max_call_stack_args_ = 0;
141  int max_deopted_stack_size_ = 0;
142  // Optimize UpdateMaxDeoptedStackSize to not re-calculate if it sees the same
143  // compilation unit multiple times in a row.
144  const MaglevCompilationUnit* last_seen_unit_ = nullptr;
145 };

182  template <typename NodeT>
183  void MarkInputUses(NodeT* node, const ProcessingState& state) {
184  LoopUsedNodes* loop_used_nodes = GetCurrentLoopUsedNodes();
185  // Mark input uses in the same order as inputs are assigned in the register
186  // allocator (see StraightForwardRegisterAllocator::AssignInputs).
187  node->ForAllInputsInRegallocAssignmentOrder(
188  [&](NodeBase::InputAllocationPolicy, Input* input) {
189  MarkUse(input->node(), node->id(), input, loop_used_nodes);
190  });
191  if constexpr (NodeT::kProperties.can_eager_deopt()) {
192  MarkCheckpointNodes(node, node->eager_deopt_info(), loop_used_nodes,
193  state);
194  }
195  if constexpr (NodeT::kProperties.can_lazy_deopt()) {
196  MarkCheckpointNodes(node, node->lazy_deopt_info(), loop_used_nodes,
197  state);

206  // Specialize the two unconditional jumps to extend their Phis' inputs' live
207  // ranges.
208 
209  void MarkInputUses(JumpLoop* node, const ProcessingState& state) {
210  int predecessor_id = state.block()->predecessor_id();
211  BasicBlock* target = node->target();
212  uint32_t use = node->id();
213 
214  DCHECK(!loop_used_nodes_.empty());
215  LoopUsedNodes loop_used_nodes = std::move(loop_used_nodes_.back());
216  loop_used_nodes_.pop_back();
217  DCHECK_EQ(loop_used_nodes.header, target);
218 
219  LoopUsedNodes* outer_loop_used_nodes = GetCurrentLoopUsedNodes();
220 
221  if (target->has_phi()) {
222  for (Phi* phi : *target->phis()) {
223  DCHECK(phi->is_used());
224  ValueNode* input = phi->input(predecessor_id).node();
225  MarkUse(input, use, &phi->input(predecessor_id), outer_loop_used_nodes);
226  }
227  }
228 
229  DCHECK_EQ(loop_used_nodes.header, target);
230  if (!loop_used_nodes.used_nodes.empty()) {
231  // Try to avoid unnecessary reloads or spills across the back-edge based
232  // on use positions and calls inside the loop.
233  RegallocInfo::RegallocLoopInfo& loop_info =
234  regalloc_info_->loop_info_
235  .emplace(loop_used_nodes.header->id(), compilation_info_->zone())
236  .first->second;
237  for (auto p : loop_used_nodes.used_nodes) {
238  // If the node is used before the first call and after the last call,
239  // keep it in a register across the back-edge.
240  if (p.second.first_register_use != kInvalidNodeId &&
241  (loop_used_nodes.first_call == kInvalidNodeId ||
242  (p.second.first_register_use <= loop_used_nodes.first_call &&
243  p.second.last_register_use > loop_used_nodes.last_call))) {
244  loop_info.reload_hints_.Add(p.first, compilation_info_->zone());
245  }
246  // If the node is not used, or used after the first call and before the
247  // last call, keep it spilled across the back-edge.
248  if (p.second.first_register_use == kInvalidNodeId ||
249  (loop_used_nodes.first_call != kInvalidNodeId &&
250  p.second.first_register_use > loop_used_nodes.first_call &&
251  p.second.last_register_use <= loop_used_nodes.last_call)) {
252  loop_info.spill_hints_.Add(p.first, compilation_info_->zone());
253  }
254  }
255 
256  // Uses of nodes in this loop may need to propagate to an outer loop, so
257  // that they're lifetime is extended there too.
258  // TODO(leszeks): We only need to extend the lifetime in one outermost
259  // loop, allow nodes to be "moved" between lifetime extensions.
260  base::Vector<Input> used_node_inputs =
261  compilation_info_->zone()->AllocateVector<Input>(
262  loop_used_nodes.used_nodes.size());
263  int i = 0;
264  for (auto& [used_node, info] : loop_used_nodes.used_nodes) {
265  Input* input = new (&used_node_inputs[i++]) Input(used_node);
266  MarkUse(used_node, use, input, outer_loop_used_nodes);
267  }
268  node->set_used_nodes(used_node_inputs);
269  }
270  }
271  void MarkInputUses(Jump* node, const ProcessingState& state) {
272  MarkJumpInputUses(node->id(), node->target(), state);
273  }
274  void MarkInputUses(CheckpointedJump* node, const ProcessingState& state) {
275  MarkJumpInputUses(node->id(), node->target(), state);

277  void MarkJumpInputUses(uint32_t use, BasicBlock* target,
278  const ProcessingState& state) {
279  int i = state.block()->predecessor_id();
280  if (!target->has_phi()) return;
281  LoopUsedNodes* loop_used_nodes = GetCurrentLoopUsedNodes();
282  Phi::List& phis = *target->phis();
283  for (auto it = phis.begin(); it != phis.end();) {
284  Phi* phi = *it;
285  if (!phi->is_used()) {
286  // Skip unused phis -- we're processing phis out of order with the dead
287  // node sweeping processor, so we will still observe unused phis here.
288  // We can eagerly remove them while we're at it so that the dead node
289  // sweeping processor doesn't have to revisit them.
290  it = phis.RemoveAt(it);
291  } else {
292  ValueNode* input = phi->input(i).node();
293  MarkUse(input, use, &phi->input(i), loop_used_nodes);
294  ++it;
295  }
296  }
297  }

318  void MarkUse(ValueNode* node, uint32_t use_id, InputLocation* input,
319  LoopUsedNodes* loop_used_nodes) {
320  DCHECK(!node->Is<Identity>());
321 
322  node->record_next_use(use_id, input);
323 
324  // If we are in a loop, loop_used_nodes is non-null. In this case, check if
325  // the incoming node is from outside the loop, and make sure to extend its
326  // lifetime to the loop end if yes.
327  if (loop_used_nodes) {
328  // If the node's id is smaller than the smallest id inside the loop, then
329  // it must have been created before the loop. This means that it's alive
330  // on loop entry, and therefore has to be alive across the loop back edge
331  // too.
332  if (node->id() < loop_used_nodes->header->first_id()) {
333  auto [it, info] = loop_used_nodes->used_nodes.emplace(
334  node, NodeUse{kInvalidNodeId, kInvalidNodeId});
335  if (input->operand().IsUnallocated()) {
336  const auto& operand =
337  compiler::UnallocatedOperand::cast(input->operand());
338  if (operand.HasRegisterPolicy() || operand.HasFixedRegisterPolicy() ||
339  operand.HasFixedFPRegisterPolicy()) {
340  if (it->second.first_register_use == kInvalidNodeId) {
341  it->second.first_register_use = use_id;
342  }
343  it->second.last_register_use = use_id;
344  }
345  }
346  }
347  }

350  template <typename DeoptInfoT>
351  void MarkCheckpointNodes(NodeBase* node, DeoptInfoT* deopt_info,
352  LoopUsedNodes* loop_used_nodes,
353  const ProcessingState& state) {
354  int use_id = node->id();
355  if (!deopt_info->has_input_locations()) {
356  size_t count = 0;
357  deopt_info->ForEachInput([&](ValueNode*) { count++; });
358  deopt_info->InitializeInputLocations(compilation_info_->zone(), count);
359  }
360  InputLocation* input = deopt_info->input_locations();
361  deopt_info->ForEachInput([&](ValueNode* node) {
362  MarkUse(node, use_id, input, loop_used_nodes);
363  input++;
364  });
365  }
