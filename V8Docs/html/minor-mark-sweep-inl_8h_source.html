38 template <typename TSlot>
39 void YoungGenerationRootMarkingVisitor::VisitPointersImpl(Root root,
40  TSlot start,
41  TSlot end) {
42  if (root == Root::kStackRoots) {
43  for (TSlot slot = start; slot < end; ++slot) {
44  main_marking_visitor_->VisitObjectViaSlot<
45  YoungGenerationMainMarkingVisitor::ObjectVisitationMode::
46  kPushToWorklist,
47  YoungGenerationMainMarkingVisitor::SlotTreatmentMode::kReadOnly>(
48  slot);

53  YoungGenerationMainMarkingVisitor::ObjectVisitationMode::
54  kPushToWorklist,

61 template <typename Visitor>
62 bool YoungGenerationRememberedSetsMarkingWorklist::ProcessNextItem(
63  Visitor* visitor, std::optional<size_t>& index) {
64  if (remaining_remembered_sets_marking_items_.load(
65  std::memory_order_relaxed) == 0) {
66  return false;
67  }
68  while (true) {
69  if (index && (index < remembered_sets_marking_items_.size())) {
70  auto& work_item = remembered_sets_marking_items_[*index];
71  if (work_item.TryAcquire()) {
72  remaining_remembered_sets_marking_items_.fetch_sub(
73  1, std::memory_order_relaxed);
74  work_item.Process(visitor);
75  (*index)++;
76  return true;
77  }
78  }
79  index = remembered_sets_marking_index_generator_.GetNext();
80  if (!index) return false;
81  }
82 }

94 template <typename Visitor>
95 void YoungGenerationRememberedSetsMarkingWorklist::MarkingItem::
96  MarkUntypedPointers(Visitor* visitor) {
97  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.gc"),
98  "MarkingItem::MarkUntypedPointers");
99  auto callback = [this, visitor](MaybeObjectSlot slot) {
100  return CheckAndMarkObject(visitor, slot);
101  };
102  if (slot_set_) {
103  const auto slot_count =
104  RememberedSet<OLD_TO_NEW>::template Iterate<AccessMode::NON_ATOMIC>(
105  slot_set_, chunk_, callback, SlotSet::FREE_EMPTY_BUCKETS);
106  if (slot_count == 0) {
107  SlotSet::Delete(slot_set_);
108  slot_set_ = nullptr;
109  }
110  }
111  if (background_slot_set_) {
112  const auto slot_count =
113  RememberedSet<OLD_TO_NEW_BACKGROUND>::template Iterate<
114  AccessMode::NON_ATOMIC>(background_slot_set_, chunk_, callback,
115  SlotSet::FREE_EMPTY_BUCKETS);
116  if (slot_count == 0) {

123 template <typename Visitor>
124 void YoungGenerationRememberedSetsMarkingWorklist::MarkingItem::
125  MarkTypedPointers(Visitor* visitor) {
126  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.gc"),
127  "MarkingItem::MarkTypedPointers");
128  DCHECK_NULL(background_slot_set_);
129  DCHECK_NOT_NULL(typed_slot_set_);
130  const auto slot_count = RememberedSet<OLD_TO_NEW>::IterateTyped(
131  typed_slot_set_,
132  [this, visitor](SlotType slot_type, Address slot_address) {
133  Tagged<HeapObject> object = UpdateTypedSlotHelper::GetTargetObject(
134  heap(), slot_type, slot_address);
135  FullMaybeObjectSlot slot(&object);
136  return CheckAndMarkObject(visitor, slot);
137  });
138  if (slot_count == 0) {
139  delete typed_slot_set_;
140  typed_slot_set_ = nullptr;
141  }

144 template <typename Visitor, typename TSlot>
145 V8_INLINE SlotCallbackResult
146 YoungGenerationRememberedSetsMarkingWorklist::MarkingItem::CheckAndMarkObject(
147  Visitor* visitor, TSlot slot) {
148  static_assert(
149  std::is_same_v<TSlot, FullMaybeObjectSlot> ||
150  std::is_same_v<TSlot, MaybeObjectSlot>,
151  "Only FullMaybeObjectSlot and MaybeObjectSlot are expected here");
152  return visitor->VisitObjectViaSlotInRememberedSet(slot) ? KEEP_SLOT
153  : REMOVE_SLOT;
