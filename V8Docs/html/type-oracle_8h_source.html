370  static std::optional<const Type*> ImplicitlyConvertableFrom(
371  const Type* to, const Type* from) {
372  while (from != nullptr) {
373  for (GenericCallable* from_constexpr :
374  Declarations::LookupGeneric(kFromConstexprMacroName)) {
375  if (std::optional<const Callable*> specialization =
376  from_constexpr->GetSpecialization({to, from})) {
377  if ((*specialization)->signature().GetExplicitTypes() ==
378  TypeVector{from}) {
379  return from;
380  }
381  }
382  }
383  from = from->parent();
384  }
385  return std::nullopt;
386  }
