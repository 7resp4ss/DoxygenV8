89 // Defines constants and accessor classes to assemble, disassemble and
90 // simulate z/Architecture instructions.
91 //
92 // Section references in the code refer to the "z/Architecture Principles
93 // Of Operation" http://publibfi.boulder.ibm.com/epubs/pdf/dz9zr009.pdf
94 //
95 
96 // Constants for specific fields are defined in their respective named enums.
97 // General constants are in an anonymous enum in class Instr.
98 enum Condition : int {
99  kNoCondition = -1,
100  eq = 0x8, // Equal.
101  ne = 0x7, // Not equal.
102  ge = 0xa, // Greater or equal.
103  lt = 0x4, // Less than.
104  gt = 0x2, // Greater than.
105  le = 0xc, // Less then or equal
106  al = 0xf, // Always.
107 
108  CC_NOP = 0x0, // S390 NOP
109  CC_EQ = 0x08, // S390 condition code 0b1000
110  CC_LT = 0x04, // S390 condition code 0b0100
111  CC_LE = CC_EQ | CC_LT, // S390 condition code 0b1100
112  CC_GT = 0x02, // S390 condition code 0b0010
113  CC_GE = CC_EQ | CC_GT, // S390 condition code 0b1010
114  CC_OF = 0x01, // S390 condition code 0b0001
115  CC_NOF = 0x0E, // S390 condition code 0b1110
116  CC_ALWAYS = 0x0F, // S390 always taken branch
117  unordered = CC_OF, // Floating-point unordered
118  ordered = CC_NOF, // floating-point ordered
119  overflow = CC_OF, // Summary overflow
120  nooverflow = CC_NOF,
121 
122  mask0x0 = 0, // no jumps
123  mask0x1 = 1,
124  mask0x2 = 2,
125  mask0x3 = 3,
126  mask0x4 = 4,
127  mask0x5 = 5,
128  mask0x6 = 6,
129  mask0x7 = 7,
130  mask0x8 = 8,
131  mask0x9 = 9,
132  mask0xA = 10,
133  mask0xB = 11,
134  mask0xC = 12,
135  mask0xD = 13,
136  mask0xE = 14,
137  mask0xF = 15,
138 
139  // Unified cross-platform condition names/aliases.
140  // Do not set unsigned constants equal to their signed variants.
141  // We need to be able to differentiate between signed and unsigned enum
142  // constants in order to emit the right instructions (i.e CmpS64 vs CmpU64).
143  kEqual = eq,
144  kNotEqual = ne,
145  kLessThan = lt,
146  kGreaterThan = gt,
147  kLessThanEqual = le,
148  kGreaterThanEqual = ge,
149  kUnsignedLessThan = 16,
150  kUnsignedGreaterThan = 17,
151  kUnsignedLessThanEqual = 18,
152  kUnsignedGreaterThanEqual = 19,
153  kOverflow = overflow,
154  kNoOverflow = nooverflow,
155  kZero = 20,
156  kNotZero = 21,
157 };

159 inline Condition to_condition(Condition cond) {
160  switch (cond) {
161  case kUnsignedLessThan:
162  return lt;
163  case kUnsignedGreaterThan:
164  return gt;
165  case kUnsignedLessThanEqual:
166  return le;
167  case kUnsignedGreaterThanEqual:
168  return ge;
169  case kZero:
170  return eq;
171  case kNotZero:
172  return ne;
173  default:
174  break;
175  }
176  return cond;
177 }
178 
179 inline bool is_signed(Condition cond) {
180  switch (cond) {
181  case kEqual:
182  case kNotEqual:
183  case kLessThan:
184  case kGreaterThan:
185  case kLessThanEqual:
186  case kGreaterThanEqual:
187  case kOverflow:
188  case kNoOverflow:
189  case kZero:
190  case kNotZero:
191  return true;
192 
193  case kUnsignedLessThan:
194  case kUnsignedGreaterThan:
195  case kUnsignedLessThanEqual:
196  case kUnsignedGreaterThanEqual:
197  return false;
198 
199  default:
200  UNREACHABLE();
201  }
202 }

204 inline Condition NegateCondition(Condition cond) {
205  DCHECK(cond != al);
206  switch (cond) {
207  case eq:
208  return ne;
209  case ne:
210  return eq;
211  case ge:
212  return lt;
213  case gt:
214  return le;
215  case le:
216  return gt;
217  case lt:
218  return ge;
219  case lt | gt:
220  return eq;
221  case le | ge:
222  return CC_OF;
223  case CC_OF:
224  return CC_NOF;
225  case kUnsignedLessThan:
226  return kUnsignedGreaterThanEqual;
227  case kUnsignedGreaterThan:
228  return kUnsignedLessThanEqual;
229  case kUnsignedLessThanEqual:
230  return kUnsignedGreaterThan;
231  case kUnsignedGreaterThanEqual:
232  return kUnsignedLessThan;
233  default:
234  DCHECK(false);
235  }
236  return al;
237 }

2039  // Extract the raw instruction bits
2040  template <typename T>
2041  static inline T InstructionBits(const uint8_t* instr) {
2042 #if !V8_TARGET_LITTLE_ENDIAN
2043  if (sizeof(T) <= 4) {
2044  return *reinterpret_cast<const T*>(instr);
2045  } else {
2046  // We cannot read 8-byte instructon address directly, because for a
2047  // six-byte instruction, the extra 2-byte address might not be
2048  // allocated.
2049  uint64_t fourBytes = *reinterpret_cast<const uint32_t*>(instr);
2050  uint16_t twoBytes = *reinterpret_cast<const uint16_t*>(instr + 4);
2051  return (fourBytes << 16 | twoBytes);
2052  }
2053 #else
2054  // Even on little endian hosts (simulation), the instructions
2055  // are stored as big-endian in order to decode the opcode and
2056  // instruction length.
2057  T instr_bits = 0;
2058 
2059  // 6-byte instrs are represented by uint64_t
2060  uint32_t size = (sizeof(T) == 8) ? 6 : sizeof(T);
2061 
2062  for (T i = 0; i < size; i++) {
2063  instr_bits <<= 8;
2064  instr_bits |= *(instr + i);
2065  }
2066  return instr_bits;
2067 #endif
2068  }

2070  // Set the Instruction Bits to value
2071  template <typename T>
2072  static inline void SetInstructionBits(
2073  uint8_t* instr, T value,
2074  WritableJitAllocation* jit_allocation = nullptr) {
2075 #if V8_TARGET_LITTLE_ENDIAN
2076  // The instruction bits are stored in big endian format even on little
2077  // endian hosts, in order to decode instruction length and opcode.
2078  // The following code will reverse the bytes so that the stores later
2079  // (which are in native endianess) will effectively save the instruction
2080  // in big endian.
2081  if (sizeof(T) == 2) {
2082  // Two Byte Instruction
2083  value = ((value & 0x00FF) << 8) | ((value & 0xFF00) >> 8);
2084  } else if (sizeof(T) == 4) {
2085  // Four Byte Instruction
2086  value = ((value & 0x000000FF) << 24) | ((value & 0x0000FF00) << 8) |
2087  ((value & 0x00FF0000) >> 8) | ((value & 0xFF000000) >> 24);
2088  } else if (sizeof(T) == 8) {
2089  // Six Byte Instruction
2090  uint64_t orig_value = static_cast<uint64_t>(value);
2091  value = (static_cast<uint64_t>(orig_value & 0xFF) << 40) |
2092  (static_cast<uint64_t>((orig_value >> 8) & 0xFF) << 32) |
2093  (static_cast<uint64_t>((orig_value >> 16) & 0xFF) << 24) |
2094  (static_cast<uint64_t>((orig_value >> 24) & 0xFF) << 16) |
2095  (static_cast<uint64_t>((orig_value >> 32) & 0xFF) << 8) |
2096  (static_cast<uint64_t>((orig_value >> 40) & 0xFF));
2097  }
2098 #endif
2099  if (sizeof(T) <= 4) {
2100  if (jit_allocation) {
2101  jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(instr),
2102  value);
2103  } else {
2104  *reinterpret_cast<T*>(instr) = value;
2105  }
2106  } else {
2107 #if V8_TARGET_LITTLE_ENDIAN
2108  uint64_t orig_value = static_cast<uint64_t>(value);
2109  if (jit_allocation) {
2110  jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(instr),
2111  static_cast<uint32_t>(value));
2112  jit_allocation->WriteUnalignedValue(
2113  reinterpret_cast<Address>(instr + 4),
2114  static_cast<uint16_t>((orig_value >> 32) & 0xFFFF));
2115  } else {
2116  *reinterpret_cast<uint32_t*>(instr) = static_cast<uint32_t>(value);
2117  *reinterpret_cast<uint16_t*>(instr + 4) =
2118  static_cast<uint16_t>((orig_value >> 32) & 0xFFFF);
2119  }
2120 #else
2121  if (jit_allocation) {
2122  jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(instr),
2123  static_cast<uint32_t>(value >> 16));
2124  jit_allocation->WriteUnalignedValue(
2125  reinterpret_cast<Address>(instr + 4),
2126  static_cast<uint16_t>(value & 0xFFFF));
2127  } else {
2128  *reinterpret_cast<uint32_t*>(instr) =
2129  static_cast<uint32_t>(value >> 16);
2130  *reinterpret_cast<uint16_t*>(instr + 4) =
2131  static_cast<uint16_t>(value & 0xFFFF);
2132  }
2133 #endif
2134  }
2135  }

2143  // Extract the full opcode from the instruction.
2144  static inline Opcode S390OpcodeValue(const uint8_t* instr) {
2145  OpcodeFormatType opcodeType = getOpcodeFormatType(instr);
2146 
2147  // The native instructions are encoded in big-endian format
2148  // even if running on little-endian host. Hence, we need
2149  // to ensure we use uint8_t* based bit-wise logic.
2150  switch (opcodeType) {
2151  case ONE_BYTE_OPCODE:
2152  // One Byte - Bits 0 to 7
2153  return static_cast<Opcode>(*instr);
2154  case TWO_BYTE_OPCODE:
2155  // Two Bytes - Bits 0 to 15
2156  return static_cast<Opcode>((*instr << 8) | (*(instr + 1)));
2157  case TWO_BYTE_DISJOINT_OPCODE:
2158  // Two Bytes - Bits 0 to 7, 40 to 47
2159  return static_cast<Opcode>((*instr << 8) | (*(instr + 5) & 0xFF));
2160  default:
2161  // case THREE_NIBBLE_OPCODE:
2162  // Three Nibbles - Bits 0 to 7, 12 to 15
2163  return static_cast<Opcode>((*instr << 4) | (*(instr + 1) & 0xF));
2164  }
2165 
2166  UNREACHABLE();
2167  }
