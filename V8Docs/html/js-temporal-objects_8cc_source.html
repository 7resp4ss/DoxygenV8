115 template <typename RetVal>
116 RetVal HandleStringEncodings(
117  Isolate* isolate, DirectHandle<String> string,
118  std::function<RetVal(std::string_view)> utf8_fn,
119  std::function<RetVal(std::u16string_view)> utf16_fn) {
120  string = String::Flatten(isolate, string);
121  DisallowGarbageCollection no_gc;
122  auto flat = string->GetFlatContent(no_gc);
123  if (flat.IsOneByte()) {
124  auto content = flat.ToOneByteVector();
125  // reinterpret_cast since std string types accept signed chars
126  std::string_view view(reinterpret_cast<const char*>(content.data()),
127  content.size());
128  return utf8_fn(view);
129 
130  } else {
131  auto content = flat.ToUC16Vector();
132  std::u16string_view view(reinterpret_cast<const char16_t*>(content.data()),
133  content.size());
134  return utf16_fn(view);
135  }

138 // Take a Rust Result and turn it into a Maybe, suitable for use
139 // with error handling macros.
140 //
141 // Note that a lot of the types returned by Rust code prefer
142 // move semantics, try using MAYBE_MOVE_RETURN_ON_EXCEPTION_VALUE
143 template <typename ContainedValue>
144 Maybe<ContainedValue> ExtractRustResult(
145  Isolate* isolate, TemporalResult<ContainedValue>&& rust_result) {
146  if (rust_result.is_err()) {
147  auto err = std::move(rust_result).err().value();
148  switch (err.kind) {
149  case temporal_rs::ErrorKind::Type:
150  THROW_NEW_ERROR_RETURN_VALUE(isolate,
151  NEW_TEMPORAL_INVALID_ARG_TYPE_ERROR(),
152  Nothing<ContainedValue>());
153  break;
154  case temporal_rs::ErrorKind::Range:
155  THROW_NEW_ERROR_RETURN_VALUE(isolate,
156  NEW_TEMPORAL_INVALID_ARG_RANGE_ERROR(),
157  Nothing<ContainedValue>());
158  break;
159  case temporal_rs::ErrorKind::Syntax:
160  case temporal_rs::ErrorKind::Assert:
161  case temporal_rs::ErrorKind::Generic:
162  default:
163  // These cases shouldn't happen; the spec doesn't currently trigger
164  // these errors
165  THROW_NEW_ERROR_RETURN_VALUE(isolate, NEW_TEMPORAL_INTERNAL_ERROR(),
166  Nothing<ContainedValue>());
167  }
168  return Nothing<ContainedValue>();
169  }
170  return Just(std::move(rust_result).ok().value());
171 }

328 bool IsValidIsoDate(double year, double month, double day) {
329  // 1. If month < 1 or month > 12, then
330  if (month < 1 || month > 12) {
331  // a. Return false.
332  return false;
333  }
334 
335  // This check is technically needed later when we check if things are in the
336  // Temporal range, but we do it now to ensure we can safely cast to int32_t
337  // before passing to Rust See https://github.com/boa-dev/temporal/issues/334.
338  if (year < std::numeric_limits<int32_t>::min() ||
339  year > std::numeric_limits<int32_t>::max()) {
340  return false;
341  }
342 
343  // IsValidIsoDate does not care about years that are "out of Temporal range",
344  // that gets handled later.
345  int64_t year_int = static_cast<int32_t>(year);
346  uint8_t month_int = static_cast<uint8_t>(month);
347  // 2. Let daysInMonth be ISODaysInMonth(year, month).
348  // 3. If day < 1 or day > daysInMonth, then
349  if (day < 1 || day > ISODaysInMonth(year_int, month_int)) {
350  // a. Return false.
351  return false;
352  }
353 
354  // 4. Return true.
355  return true;

385 Maybe<temporal_rs::Precision> GetTemporalFractionalSecondDigitsOption(
386  Isolate* isolate, DirectHandle<JSReceiver> normalized_options,
387  const char* method_name) {
388  auto auto_val =
389  temporal_rs::Precision{.is_minute = false, .precision = std::nullopt};
390 
391  Factory* factory = isolate->factory();
392  // 1. Let digitsValue be ?Get(options, "fractionalSecondDigits").
393  DirectHandle<Object> digits_val;
394  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
395  isolate, digits_val,
396  JSReceiver::GetProperty(isolate, normalized_options,
397  factory->fractionalSecondDigits_string()),
398  Nothing<temporal_rs::Precision>());
399 
400  // 2. If digitsValue is undefined, return auto.
401  if (IsUndefined(*digits_val)) {
402  return Just(auto_val);
403  }
404 
405  // 3. If digitsValue is not a Number, then
406  if (!IsNumber(*digits_val)) {
407  DirectHandle<String> string;
408  ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate, string,
409  Object::ToString(isolate, digits_val),
410  Nothing<temporal_rs::Precision>());
411  // a. If ? ToString(digitsValue) is not "auto", throw a RangeError
412  // exception.
413  if (!String::Equals(isolate, string, factory->auto_string())) {
414  THROW_NEW_ERROR_RETURN_VALUE(
415  isolate,
416  NewRangeError(MessageTemplate::kPropertyValueOutOfRange,
417  factory->fractionalSecondDigits_string()),
418  Nothing<temporal_rs::Precision>());
419  }
420  // b. Return auto.
421  return Just(auto_val);
422  }
423  // 4. If digitsValue is NaN, +∞𝔽, or -∞𝔽, throw a RangeError exception.
424  auto digits_num = Cast<Number>(*digits_val);
425  auto digits_float = Object::NumberValue(digits_num);
426  if (std::isnan(digits_float) || std::isinf(digits_float)) {
427  THROW_NEW_ERROR_RETURN_VALUE(
428  isolate,
429  NewRangeError(MessageTemplate::kPropertyValueOutOfRange,
430  factory->fractionalSecondDigits_string()),
431  Nothing<temporal_rs::Precision>());
432  }
433  // 5. Let digitCount be floor(ℝ(digitsValue)).
434  int64_t digit_count = std::floor(Object::NumberValue(digits_num));
435  // 6. If digitCount < 0 or digitCount > 9, throw a RangeError exception.
436  if (digit_count < 0 || digit_count > 9) {
437  THROW_NEW_ERROR_RETURN_VALUE(
438  isolate,
439  NewRangeError(MessageTemplate::kPropertyValueOutOfRange,
440  factory->fractionalSecondDigits_string()),
441  Nothing<temporal_rs::Precision>());
442  }
443 
444  return Just(
445  temporal_rs::Precision{.is_minute = false, .precision = digit_count});
446 }
