65 // This may return an empty MaybeDirectHandle if an exception is thrown or
66 // we abort due to reaching the recursion limit.
67 MaybeDirectHandle<String> StringReplaceOneCharWithString(
68  Isolate* isolate, DirectHandle<String> subject, DirectHandle<String> search,
69  DirectHandle<String> replace, bool* found, int recursion_limit) {
70  StackLimitCheck stackLimitCheck(isolate);
71  if (stackLimitCheck.HasOverflowed() || (recursion_limit == 0)) {
72  return MaybeDirectHandle<String>();
73  }
74  recursion_limit--;
75  if (IsConsString(*subject)) {
76  Tagged<ConsString> cons = Cast<ConsString>(*subject);
77  DirectHandle<String> first(cons->first(), isolate);
78  DirectHandle<String> second(cons->second(), isolate);
79  DirectHandle<String> new_first;
80  if (!StringReplaceOneCharWithString(isolate, first, search, replace, found,
81  recursion_limit).ToHandle(&new_first)) {
82  return MaybeDirectHandle<String>();
83  }
84  if (*found) return isolate->factory()->NewConsString(new_first, second);
85 
86  DirectHandle<String> new_second;
87  if (!StringReplaceOneCharWithString(isolate, second, search, replace, found,
88  recursion_limit)
89  .ToHandle(&new_second)) {
90  return MaybeDirectHandle<String>();
91  }
92  if (*found) return isolate->factory()->NewConsString(first, new_second);
93 
94  return subject;
95  } else {
96  int index = String::IndexOf(isolate, subject, search, 0);
97  if (index == -1) return subject;
98  *found = true;
99  DirectHandle<String> first =
100  isolate->factory()->NewSubString(subject, 0, index);
101  DirectHandle<String> cons1;
102  ASSIGN_RETURN_ON_EXCEPTION(
103  isolate, cons1, isolate->factory()->NewConsString(first, replace));
104  DirectHandle<String> second =
105  isolate->factory()->NewSubString(subject, index + 1, subject->length());
106  return isolate->factory()->NewConsString(cons1, second);
107  }
108 }

110 RUNTIME_FUNCTION(Runtime_StringReplaceOneCharWithString) {
111  HandleScope scope(isolate);
112  DCHECK_EQ(3, args.length());
113  DirectHandle<String> subject = args.at<String>(0);
114  DirectHandle<String> search = args.at<String>(1);
115  DirectHandle<String> replace = args.at<String>(2);
116 
117  // If the cons string tree is too deep, we simply abort the recursion and
118  // retry with a flattened subject string.
119  const int kRecursionLimit = 0x1000;
120  bool found = false;
121  DirectHandle<String> result;
122  if (StringReplaceOneCharWithString(isolate, subject, search, replace, &found,
123  kRecursionLimit).ToHandle(&result)) {
124  return *result;
125  }
126  if (isolate->has_exception()) return ReadOnlyRoots(isolate).exception();
127 
128  subject = String::Flatten(isolate, subject);
129  if (StringReplaceOneCharWithString(isolate, subject, search, replace, &found,
130  kRecursionLimit).ToHandle(&result)) {
131  return *result;
132  }
133  if (isolate->has_exception()) return ReadOnlyRoots(isolate).exception();
134  // In case of empty handle and no exception we have stack overflow.
135  return isolate->StackOverflow();
136 }

171 Tagged<Object> StringAdd_StringConstant_Internalize(
172  Isolate* isolate, DirectHandle<Object> lhs, DirectHandle<Object> rhs,
173  Handle<HeapObject> maybe_feedback_vector, int slot_index,
174  ASVariant as_variant) {
175  const bool lhs_is_string_constant =
176  as_variant == ASVariant::kLhsIsStringConstant;
177  DirectHandle<Object> const_str_operand = lhs_is_string_constant ? lhs : rhs;
178  DirectHandle<Object> other_operand = lhs_is_string_constant ? rhs : lhs;
179  DCHECK(IsInternalizedString(*const_str_operand));
180 
181  DirectHandle<String> other_operand_string;
182  if (IsString(*other_operand)) {
183  other_operand_string = Cast<String>(other_operand);
184  } else {
185  // According to spec we first have to do ToPrimitive and only then
186  // ToString.
187  // https://tc39.es/ecma262/#sec-applystringornumericbinaryoperator
188  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
189  isolate, other_operand, Object::ToPrimitive(isolate, other_operand));
190  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
191  isolate, other_operand_string,
192  Object::ToString(isolate, other_operand));
193  }
194 
195  auto f = isolate->factory();
196  auto other_operand_internalize =
197  IsInternalizedString(*other_operand_string)
198  ? Cast<InternalizedString>(other_operand_string)
199  : f->InternalizeString(other_operand_string);
200 
201  auto lhs_internalize = lhs_is_string_constant
202  ? Cast<String>(const_str_operand)
203  : Cast<String>(other_operand_internalize);
204  auto rhs_internalize = lhs_is_string_constant
205  ? Cast<String>(other_operand_internalize)
206  : Cast<String>(const_str_operand);
207 
208  if (IsUndefined(*maybe_feedback_vector)) {
209  DirectHandle<String> cons;
210  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
211  isolate, cons, f->NewConsString(lhs_internalize, rhs_internalize));
212  return *f->InternalizeString(cons);
213  }
214 
215  auto feedback_vector = Cast<FeedbackVector>(maybe_feedback_vector);
216 
217  FeedbackSlot cache_slot(FeedbackVector::ToSlot(
218  slot_index + kAdd_StringConstant_Internalize_CacheSlotOffset));
219  DCHECK_LT(cache_slot.ToInt(), feedback_vector->length());
220  Handle<Object> cache_obj(Cast<Object>(feedback_vector->Get(cache_slot)),
221  isolate);
222  Handle<SimpleNameDictionary> cache;
223  if (*cache_obj == ReadOnlyRoots{isolate}.uninitialized_symbol()) {
224  cache = SimpleNameDictionary::New(isolate, 1);
225  feedback_vector->SynchronizedSet(cache_slot, *cache);
226  } else {
227  cache = Cast<SimpleNameDictionary>(cache_obj);
228  }
229 
230  InternalIndex entry = cache->FindEntry(isolate, other_operand_internalize);
231  if (entry.is_found()) {
232  auto result = cache->ValueAt(entry);
233  DCHECK(IsInternalizedString(result));
234  return result;
235  }
236 
237  DirectHandle<String> cons;
238  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
239  isolate, cons, f->NewConsString(lhs_internalize, rhs_internalize));
240 
241  auto internalized = f->InternalizeString(cons);
242  auto new_cache = SimpleNameDictionary::Set(
243  isolate, cache, other_operand_internalize, internalized);
244  if (*new_cache != *cache) {
245  feedback_vector->SynchronizedSet(cache_slot, *new_cache);
246  }
247 
248  return *internalized;
249 }

304 RUNTIME_FUNCTION(Runtime_StringCodePointAt) {
305  HandleScope handle_scope(isolate);
306  DCHECK_EQ(2, args.length());
307 
308  DirectHandle<String> subject = args.at<String>(0);
309  uint32_t i = NumberToUint32(args[1]);
310 
311  // Flatten the string. If someone wants to get a char at an index
312  // in a cons string, it is likely that more indices will be
313  // accessed.
314  subject = String::Flatten(isolate, subject);
315 
316  if (i >= static_cast<uint32_t>(subject->length())) {
317  return ReadOnlyRoots(isolate).nan_value();
318  }
319 
320  int first_code_point = subject->Get(i);
321  if ((first_code_point & 0xFC00) != 0xD800) {
322  return Smi::FromInt(first_code_point);
323  }
324 
325  if (i + 1 >= static_cast<uint32_t>(subject->length())) {
326  return Smi::FromInt(first_code_point);
327  }
328 
329  int second_code_point = subject->Get(i + 1);
330  if ((second_code_point & 0xFC00) != 0xDC00) {
331  return Smi::FromInt(first_code_point);
332  }
333 
334  int surrogate_offset = 0x10000 - (0xD800 << 10) - 0xDC00;
335  return Smi::FromInt((first_code_point << 10) +
336  (second_code_point + surrogate_offset));
337 }

339 RUNTIME_FUNCTION(Runtime_StringBuilderConcat) {
340  HandleScope scope(isolate);
341  DCHECK_EQ(3, args.length());
342  DirectHandle<FixedArray> array = args.at<FixedArray>(0);
343 
344  int array_length = args.smi_value_at(1);
345 
346  DirectHandle<String> special = args.at<String>(2);
347 
348  // This assumption is used by the slice encoding in one or two smis.
349  DCHECK_GE(Smi::kMaxValue, String::kMaxLength);
350 
351  int special_length = special->length();
352 
353  int length;
354  bool one_byte = special->IsOneByteRepresentation();
355 
356  {
357  DisallowGarbageCollection no_gc;
358  Tagged<FixedArray> fixed_array = *array;
359 
360  if (array_length == 0) {

362  } else if (array_length == 1) {
363  Tagged<Object> first = fixed_array->get(0);
364  if (IsString(first)) return first;

367  array_length, &one_byte);

369

371  return isolate->Throw(ReadOnlyRoots(isolate).illegal_argument_string());
372  }
373  if (length == 0) {
374  return ReadOnlyRoots(isolate).empty_string();
375  }

379  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
380  isolate, answer, isolate->factory()->NewRawOneByteString(length));
381  DisallowGarbageCollection no_gc;
382  StringBuilderConcatHelper(*special, answer->GetChars(no_gc), *array,
383  array_length);
384  return *answer;
385  } else {
386  DirectHandle<SeqTwoByteString> answer;
387  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
388  isolate, answer, isolate->factory()->NewRawTwoByteString(length));
389  DisallowGarbageCollection no_gc;
390  StringBuilderConcatHelper(*special, answer->GetChars(no_gc), *array,
391  array_length);
392  return *answer;
393  }

396 // Converts a String to JSArray.
397 // For example, "foo" => ["f", "o", "o"].
398 RUNTIME_FUNCTION(Runtime_StringToArray) {
399  HandleScope scope(isolate);
400  DCHECK_EQ(2, args.length());
401  DirectHandle<String> s = args.at<String>(0);
402  uint32_t limit = NumberToUint32(args[1]);
403 
404  s = String::Flatten(isolate, s);
405  const int length =
406  static_cast<int>(std::min(static_cast<uint32_t>(s->length()), limit));
407 
408  DirectHandle<FixedArray> elements = isolate->factory()->NewFixedArray(length);
409  bool elements_are_initialized = false;
410 
411  if (s->IsFlat() && s->IsOneByteRepresentation()) {
412  DisallowGarbageCollection no_gc;
413  String::FlatContent content = s->GetFlatContent(no_gc);
414  // Use pre-initialized single characters to initialize all the elements.
415  // This can be false if the string is sliced from an externalized
416  // two-byte string that has only one-byte chars, in that case we will do
417  // a LookupSingleCharacterStringFromCode for each of the characters.
418  if (content.IsOneByte()) {
419  base::Vector<const uint8_t> chars = content.ToOneByteVector();
420  ReadOnlyRoots roots(isolate);
421  for (int i = 0; i < length; ++i) {
422  Tagged<String> value = roots.single_character_string(chars[i]);
423  DCHECK(ReadOnlyHeap::Contains(Cast<HeapObject>(value)));
424  // The single-character strings are in RO space so it should
425  // be safe to skip the write barriers.
426  elements->set(i, value, SKIP_WRITE_BARRIER);
427  }
428  elements_are_initialized = true;
429  }
430  }
431 
432  if (!elements_are_initialized) {
433  for (int i = 0; i < length; ++i) {
434  DirectHandle<Object> str =
435  isolate->factory()->LookupSingleCharacterStringFromCode(s->Get(i));
436  elements->set(i, *str);
437  }
438  }
439 
440 #ifdef DEBUG
441  for (int i = 0; i < length; ++i) {
442  DCHECK_EQ(Cast<String>(elements->get(i))->length(), 1);
443  }
444 #endif
445 
446  return *isolate->factory()->NewJSArrayWithElements(elements);
447 }

493 RUNTIME_FUNCTION(Runtime_StringEqual) {
494  SaveAndClearThreadInWasmFlag non_wasm_scope(isolate);
495  HandleScope handle_scope(isolate);
496  DCHECK_EQ(2, args.length());
497  // This function can be called from Wasm: optimized Wasm code calls
498  // straight to the "StringEqual" builtin, which tail-calls here. So on
499  // the stack, the CEntryStub's EXIT frame will sit right on top of the
500  // Wasm frame; and Wasm frames don't scan their outgoing parameters (in
501  // order to support tail-calls between Wasm functions), while the EXIT
502  // frame doesn't scan its incoming parameters (because it expects to be
503  // called from JS).
504  // Working around this by calling through a trampoline builtin is slow.
505  // Teaching the stack walker to be smarter has proven to be difficult.
506  // In the future, Conservative Stack Scanning will trivially solve the
507  // problem. In the meantime, we can work around it by explicitly creating
508  // handles here (rather than treating the on-stack arguments as handles).
509  //
510  // TODO(42203211): Don't create new handles here once direct handles and CSS
511  // are enabled by default.
512  DirectHandle<String> x(*args.at<String>(0), isolate);
513  DirectHandle<String> y(*args.at<String>(1), isolate);
514  return isolate->heap()->ToBoolean(String::Equals(isolate, x, y));
515 }

540 RUNTIME_FUNCTION(Runtime_StringEscapeQuotes) {
541  HandleScope handle_scope(isolate);
542  DCHECK_EQ(1, args.length());
543  DirectHandle<String> string = args.at<String>(0);
544 
545  // Equivalent to global replacement `string.replace(/"/g, "&quot")`, but this
546  // does not modify any global state (e.g. the regexp match info).
547 
548  const int string_length = string->length();
549  DirectHandle<String> quotes =
550  isolate->factory()->LookupSingleCharacterStringFromCode('"');
551 
552  int quote_index = String::IndexOf(isolate, string, quotes, 0);
553 
554  // No quotes, nothing to do.
555  if (quote_index == -1) return *string;
556 
557  // Find all quotes.
558  std::vector<int> indices = {quote_index};
559  while (quote_index + 1 < string_length) {
560  quote_index = String::IndexOf(isolate, string, quotes, quote_index + 1);
561  if (quote_index == -1) break;
562  indices.emplace_back(quote_index);
563  }
564 
565  // Build the replacement string.
566  DirectHandle<String> replacement =
567  isolate->factory()->NewStringFromAsciiChecked("&quot;");
568  const int estimated_part_count = static_cast<int>(indices.size()) * 2 + 1;
569  ReplacementStringBuilder builder(isolate->heap(), string,
570  estimated_part_count);
571 
572  int prev_index = -1; // Start at -1 to avoid special-casing the first match.
573  for (int index : indices) {
574  const int slice_start = prev_index + 1;
575  const int slice_end = index;
576  if (slice_end > slice_start) {
577  builder.AddSubjectSlice(slice_start, slice_end);
578  }
579  builder.AddString(replacement);
580  prev_index = index;
581  }
582 
583  if (prev_index < string_length - 1) {
584  builder.AddSubjectSlice(prev_index + 1, string_length);
585  }
586 
587  DirectHandle<String> result;
588  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, result, builder.ToString());
589  return *result;
590 }

601  HandleScope handle_scope(isolate);
