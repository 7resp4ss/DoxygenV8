20 template <typename T>
21 bool GCAwareObjectTypeCheck(Tagged<Object> object, const Heap* heap) {
22  Tagged<HeapObject> heap_object = UncheckedCast<HeapObject>(object);
23  // `heap_object` must be of type `T`. One of 4 cases can apply:
24  // 1) A scavenger is ongoing and `object` is a promoted large object.
25  if ((heap->gc_state() == Heap::SCAVENGE) &&
26  MemoryChunk::FromHeapObject(heap_object)->IsLargePage()) {
27  return true;
28  }
29  // 2) A conservative scavenge is ongoing and `object` was pinned by
30  // conservative stack scanning.
31  MapWord map_word = heap_object->map_word(
32  PtrComprCageBase(heap->isolate()->cage_base()), kRelaxedLoad);
33  if ((heap->gc_state() == Heap::SCAVENGE) &&
34  HeapLayout::InYoungGeneration(heap_object) &&
35  ((heap->ConservativeStackScanningModeForMinorGC() !=
36  Heap::StackScanMode::kNone) ||
37  heap->ShouldUsePrecisePinningForMinorGC()) &&
38  map_word.IsForwardingAddress() &&
39  HeapLayout::IsSelfForwarded(heap_object, map_word)) {
40  return true;
41  }
42  // 3) A GC is ongoing, `object` was evacuated, and the new instance is a
43  // `T`.
44  if (heap->IsInGC() && map_word.IsForwardingAddress() &&
45  Is<T>(map_word.ToForwardingAddress(heap_object))) {
46  return true;
47  }
48  // 4) If none of the above special cases apply, `heap_object` must be a `T`.
49  return Is<T>(object);
50 }
51 #endif // DEBUG
