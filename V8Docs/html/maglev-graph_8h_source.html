83  inline void RemoveUnreachableBlocks() {
84  DCHECK(may_have_unreachable_blocks());
85 
86  std::vector<bool> reachable_blocks(max_block_id(), false);
87  std::vector<BasicBlock*> worklist;
88 
89  DCHECK(!blocks().empty());
90  BasicBlock* initial_bb = blocks().front();
91  worklist.push_back(initial_bb);
92  reachable_blocks[initial_bb->id()] = true;
93  DCHECK(!initial_bb->is_loop());
94 
95  while (!worklist.empty()) {
96  BasicBlock* current = worklist.back();
97  worklist.pop_back();
98 
99  for (auto handler : current->exception_handlers()) {
100  if (!handler->HasExceptionHandler()) continue;
101  if (handler->ShouldLazyDeopt()) continue;
102  BasicBlock* catch_block = handler->catch_block();
103  if (!reachable_blocks[catch_block->id()]) {
104  reachable_blocks[catch_block->id()] = true;
105  worklist.push_back(catch_block);
106  }
107  }
108  current->ForEachSuccessor([&](BasicBlock* succ) {
109  if (!reachable_blocks[succ->id()]) {
110  reachable_blocks[succ->id()] = true;
111  worklist.push_back(succ);
112  }
113  });
114  }
115 
116  // Sweep dead blocks and remove unreachable predecessors.
117  IterateGraphAndSweepDeadBlocks([&](BasicBlock* bb) {
118  if (!reachable_blocks[bb->id()]) return true;
119  // If block doesn't have a merge state, it has only one predecessor, so
120  // it must be the reachable one.
121  if (!bb->has_state()) return false;
122  if (bb->is_loop() &&
123  !reachable_blocks[bb->backedge_predecessor()->id()]) {
124  // If the backedge predecessor is not reachable, we can turn the loop
125  // into a regular block.
126  bb->state()->TurnLoopIntoRegularBlock();
127  }
128  for (int i = bb->predecessor_count() - 1; i >= 0; i--) {
129  if (!reachable_blocks[bb->predecessor_at(i)->id()]) {
130  bb->state()->RemovePredecessorAt(i);
131  }
132  }
133  return false;
134  });
135 
136  set_may_have_unreachable_blocks(false);
137  }

139  template <typename Function>
140  void IterateGraphAndSweepDeadBlocks(Function&& is_dead) {
141  auto current = blocks_.begin();
142  auto last_non_dead = current;
143  while (current != blocks_.end()) {
144  if (is_dead(*current)) {
145  (*current)->mark_dead();
146  } else {
147  if (current != last_non_dead) {
148  // Move current to last non dead position.
149  *last_non_dead = *current;
150  }
151  ++last_non_dead;
152  }
153  ++current;
154  }
155  if (current != last_non_dead) {
156  blocks_.resize(blocks_.size() - (current - last_non_dead));
157  }
158  }

267  compiler::OptionalScopeInfoRef TryGetScopeInfoForContextLoad(
268  ValueNode* context, int offset, compiler::JSHeapBroker* broker) {
269  compiler::OptionalScopeInfoRef cur = TryGetScopeInfo(context, broker);
270  if (offset == Context::OffsetOfElementAt(Context::EXTENSION_INDEX)) {
271  return cur;
272  }
273  CHECK_EQ(offset, Context::OffsetOfElementAt(Context::PREVIOUS_INDEX));
274  if (cur.has_value()) {
275  cur = (*cur).OuterScopeInfo(broker);
276  while (!cur->HasContext() && cur->HasOuterScopeInfo()) {
277  cur = cur->OuterScopeInfo(broker);
278  }
279  if (cur->HasContext()) {
280  return cur;
281  }
282  }
283  return {};
284  }

286  // Resolve the scope info of a context value.
287  // An empty result means we don't statically know the context's scope.
288  compiler::OptionalScopeInfoRef TryGetScopeInfo(
289  ValueNode* context, compiler::JSHeapBroker* broker) {
290  auto it = scope_infos_.find(context);
291  if (it != scope_infos_.end()) {
292  return it->second;
293  }
294  compiler::OptionalScopeInfoRef res;
295  if (auto context_const = context->TryCast<Constant>()) {
296  res = context_const->object().AsContext().scope_info(broker);
297  DCHECK(res->HasContext());
298  } else if (auto load =
299  context->TryCast<LoadTaggedFieldForContextSlotNoCells>()) {
300  compiler::OptionalScopeInfoRef cur = TryGetScopeInfoForContextLoad(
301  load->input(0).node(), load->offset(), broker);
302  if (cur.has_value()) res = cur;
303  } else if (auto load_script =
304  context->TryCast<LoadTaggedFieldForContextSlot>()) {
305  compiler::OptionalScopeInfoRef cur = TryGetScopeInfoForContextLoad(
306  load_script->input(0).node(), load_script->offset(), broker);
307  if (cur.has_value()) res = cur;
308  } else if (context->Is<InitialValue>()) {
309  // We should only fail to keep track of initial contexts originating from
310  // the OSR prequel.
311  // TODO(olivf): Keep track of contexts when analyzing OSR Prequel.
312  DCHECK(is_osr());
313  } else {
314  // Any context created within a function must be registered in
315  // graph()->scope_infos(). Initial contexts must be registered before
316  // BuildBody. We don't track context in generators (yet) and around eval
317  // the bytecode compiler creates contexts by calling
318  // Runtime::kNewFunctionInfo directly.
319  DCHECK(context->Is<Phi>() || context->Is<GeneratorRestoreRegister>() ||
320  context->Is<RegisterInput>() || context->Is<CallRuntime>());
321  }
322  return scope_infos_[context] = res;
323  }
