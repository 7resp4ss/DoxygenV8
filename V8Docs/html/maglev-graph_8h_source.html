299  template <typename NodeT, typename... Args>
300  NodeT* CreateNewConstantNode(Args&&... args) const {
301  static_assert(IsConstantNode(Node::opcode_of<NodeT>));
302  NodeT* node = NodeBase::New<NodeT>(zone(), std::forward<Args>(args)...);
303  static_assert(!NodeT::kProperties.can_eager_deopt());
304  static_assert(!NodeT::kProperties.can_lazy_deopt());
305  static_assert(!NodeT::kProperties.can_throw());
306  static_assert(!NodeT::kProperties.can_write());
307  if (has_graph_labeller()) graph_labeller()->RegisterNode(node);
308  if (v8_flags.trace_maglev_graph_building) {
309  std::cout << " " << node << " "
310  << PrintNodeLabel(graph_labeller(), node) << ": "
311  << PrintNode(graph_labeller(), node) << std::endl;

314  }

327  compiler::OptionalScopeInfoRef TryGetScopeInfoForContextLoad(
328  ValueNode* context, int offset) {
329  compiler::OptionalScopeInfoRef cur = TryGetScopeInfo(context);
330  if (offset == Context::OffsetOfElementAt(Context::EXTENSION_INDEX)) {
331  return cur;
332  }
333  CHECK_EQ(offset, Context::OffsetOfElementAt(Context::PREVIOUS_INDEX));
334  if (cur.has_value()) {
335  cur = (*cur).OuterScopeInfo(broker());
336  while (!cur->HasContext() && cur->HasOuterScopeInfo()) {
337  cur = cur->OuterScopeInfo(broker());
338  }
339  if (cur->HasContext()) {
340  return cur;
341  }
342  }
343  return {};
344  }

346  template <typename Function>
347  void IterateGraphAndSweepDeadBlocks(Function&& is_dead) {
348  auto current = blocks_.begin();
349  auto last_non_dead = current;
350  while (current != blocks_.end()) {
351  if (is_dead(*current)) {
352  (*current)->mark_dead();
353  } else {
354  if (current != last_non_dead) {
355  // Move current to last non dead position.
356  *last_non_dead = *current;
357  }
358  ++last_non_dead;
359  }
360  ++current;
361  }
362  if (current != last_non_dead) {
363  blocks_.resize(blocks_.size() - (current - last_non_dead));
364  }
365  }
366 };

368 compiler::OptionalScopeInfoRef Graph::TryGetScopeInfo(ValueNode* context) {
369  auto it = scope_infos_.find(context);
370  if (it != scope_infos_.end()) {
371  return it->second;
372  }
373  compiler::OptionalScopeInfoRef res;
374  if (auto context_const = context->TryCast<Constant>()) {
375  res = context_const->object().AsContext().scope_info(broker());
376  DCHECK(res->HasContext());
377  } else if (auto load =
378  context->TryCast<LoadTaggedFieldForContextSlotNoCells>()) {
379  compiler::OptionalScopeInfoRef cur =
380  TryGetScopeInfoForContextLoad(load->input(0).node(), load->offset());
381  if (cur.has_value()) res = cur;
382  } else if (auto load_script =
383  context->TryCast<LoadTaggedFieldForContextSlot>()) {
384  compiler::OptionalScopeInfoRef cur = TryGetScopeInfoForContextLoad(
385  load_script->input(0).node(), load_script->offset());
386  if (cur.has_value()) res = cur;
387  } else if (context->Is<InitialValue>()) {
388  // We should only fail to keep track of initial contexts originating from
389  // the OSR prequel.
390  // TODO(olivf): Keep track of contexts when analyzing OSR Prequel.
391  DCHECK(is_osr());
392  } else {
393  // Any context created within a function must be registered in
394  // graph()->scope_infos(). Initial contexts must be registered before
395  // BuildBody. We don't track context in generators (yet) and around eval
396  // the bytecode compiler creates contexts by calling
397  // Runtime::kNewFunctionInfo directly.
398  DCHECK(context->Is<Phi>() || context->Is<GeneratorRestoreRegister>() ||
399  context->Is<RegisterInput>() || context->Is<CallRuntime>());
400  }
401  return scope_infos_[context] = res;
402 }

404 void Graph::RemoveUnreachableBlocks() {
405  DCHECK(may_have_unreachable_blocks());
406 
407  std::vector<bool> reachable_blocks(max_block_id(), false);
408  std::vector<BasicBlock*> worklist;
409 
410  DCHECK(!blocks().empty());
411  BasicBlock* initial_bb = blocks().front();
412  worklist.push_back(initial_bb);
413  reachable_blocks[initial_bb->id()] = true;
414  DCHECK(!initial_bb->is_loop());
415 
416  while (!worklist.empty()) {
417  BasicBlock* current = worklist.back();
418  worklist.pop_back();
419 
420  for (auto handler : current->exception_handlers()) {
421  if (!handler->HasExceptionHandler()) continue;
422  if (handler->ShouldLazyDeopt()) continue;
423  BasicBlock* catch_block = handler->catch_block();
424  if (!reachable_blocks[catch_block->id()]) {
425  reachable_blocks[catch_block->id()] = true;
426  worklist.push_back(catch_block);
427  }
428  }
429  current->ForEachSuccessor([&](BasicBlock* succ) {
430  if (!reachable_blocks[succ->id()]) {
431  reachable_blocks[succ->id()] = true;
432  worklist.push_back(succ);
433  }
434  });
435  }
436 
437  // Sweep dead blocks and remove unreachable predecessors.
438  IterateGraphAndSweepDeadBlocks([&](BasicBlock* bb) {
439  if (!reachable_blocks[bb->id()]) return true;
440  // If block doesn't have a merge state, it has only one predecessor, so
441  // it must be the reachable one.
442  if (!bb->has_state()) return false;
443  if (bb->is_loop() && !reachable_blocks[bb->backedge_predecessor()->id()]) {
444  // If the backedge predecessor is not reachable, we can turn the loop
445  // into a regular block.
446  bb->state()->TurnLoopIntoRegularBlock();
447  }
448  for (int i = bb->predecessor_count() - 1; i >= 0; i--) {
449  if (!reachable_blocks[bb->predecessor_at(i)->id()]) {
450  bb->state()->RemovePredecessorAt(i);
451  }
452  }
453  return false;
454  });
455 
456  set_may_have_unreachable_blocks(false);
457 }

460  if (ref.IsSmi()) {
