162 inline void PushInput(MaglevAssembler* masm, const Input& input) {
163  if (input.operand().IsConstant()) {
164  input.node()->LoadToRegister(masm, kScratchRegister);
165  masm->Push(kScratchRegister);
166  } else {
167  // TODO(leszeks): Consider special casing the value. (Toon: could possibly
168  // be done through Input directly?)
169  const compiler::AllocatedOperand& operand =
170  compiler::AllocatedOperand::cast(input.operand());
171 
172  if (operand.IsRegister()) {
173  masm->Push(operand.GetRegister());
174  } else {
175  DCHECK(operand.IsStackSlot());
176  masm->Push(masm->GetStackSlot(operand));
177  }
178  }
179 }

181 template <typename T, typename... Args>
182 inline void PushIterator(MaglevAssembler* masm, base::iterator_range<T> range,
183  Args... args) {
184  for (auto iter = range.begin(), end = range.end(); iter != end; ++iter) {
185  masm->Push(*iter);
186  }
187  PushAllHelper<Args...>::Push(masm, args...);
188 }

190 template <typename T, typename... Args>
191 inline void PushIteratorReverse(MaglevAssembler* masm,
192  base::iterator_range<T> range, Args... args) {
193  PushAllHelper<Args...>::PushReverse(masm, args...);
194  for (auto iter = range.rbegin(), end = range.rend(); iter != end; ++iter) {
195  masm->Push(*iter);
196  }
197 }

199 template <typename... Args>
200 struct PushAllHelper<Input, Args...> {
201  static void Push(MaglevAssembler* masm, const Input& arg, Args... args) {
202  PushInput(masm, arg);
203  PushAllHelper<Args...>::Push(masm, args...);
204  }
205  static void PushReverse(MaglevAssembler* masm, const Input& arg,
206  Args... args) {
207  PushAllHelper<Args...>::PushReverse(masm, args...);
208  PushInput(masm, arg);
209  }
210 };
211 template <typename Arg, typename... Args>
212 struct PushAllHelper<Arg, Args...> {
213  static void Push(MaglevAssembler* masm, Arg arg, Args... args) {
214  if constexpr (is_iterator_range<Arg>::value) {
215  PushIterator(masm, arg, args...);
216  } else {
217  masm->MacroAssembler::Push(arg);
218  PushAllHelper<Args...>::Push(masm, args...);
219  }
220  }
221  static void PushReverse(MaglevAssembler* masm, Arg arg, Args... args) {
222  if constexpr (is_iterator_range<Arg>::value) {
223  PushIteratorReverse(masm, arg, args...);
224  } else {
225  PushAllHelper<Args...>::PushReverse(masm, args...);
226  masm->Push(arg);
227  }
228  }
229 };

275  Label::Distance distance) {

962 inline void MaglevAssembler::EmitEagerDeoptStress(Label* target) {
963  if (V8_LIKELY(v8_flags.deopt_every_n_times <= 0)) {
964  return;
965  }
966 
967  ExternalReference counter = ExternalReference::stress_deopt_count(isolate());
968  // The following code assumes that `Isolate::stress_deopt_count_` is 8 bytes
969  // wide.
970  static constexpr size_t kSizeofRAX = 8;
971  static_assert(sizeof(decltype(*isolate()->stress_deopt_count_address())) ==
972  kSizeofRAX);
973 
974  Label fallthrough;
975  pushfq();
976  pushq(rax);
977  load_rax(counter);
978  decl(rax);
979  JumpIf(not_zero, &fallthrough, Label::kNear);
980 
981  RecordComment("-- deopt_every_n_times hit, jump to eager deopt");
982  Move(rax, v8_flags.deopt_every_n_times);
983  store_rax(counter);
984  popq(rax);
985  popfq();
986  JumpToDeopt(target);
987 
988  bind(&fallthrough);
989  store_rax(counter);
990  popq(rax);
991  popfq();
992 }

996  // The least common denominator of all eager deopts is that they eventually
997  // (should) bottom out in `JumpIf`. We use the opportunity here to trigger
998  // extra eager deoptimizations with the `--deopt-every-n-times` stress mode.
