145 // This one is very sneaky. String maps are laid out sequentially, and
146 // alternate between two-byte and one-byte. Since they're sequential, each
147 // address is one Map::kSize larger than the previous. This means that the LSB
148 // of the map size alternates being set and unset for alternating string map
149 // addresses, and therefore is on/off for all two-byte/one-byte strings. Which
150 // of the two has the on-bit depends on the current RO heap layout, so just
151 // sniff this by checking an arbitrary one-byte map's value.
152 static constexpr uint32_t kStringMapEncodingMask =
153  1 << base::bits::CountTrailingZerosNonZero(Map::kSize);
154 static constexpr uint32_t kOneByteStringMapBit =
155  StaticReadOnlyRoot::kSeqOneByteStringMap & kStringMapEncodingMask;
156 static constexpr uint32_t kTwoByteStringMapBit =
157  StaticReadOnlyRoot::kSeqTwoByteStringMap & kStringMapEncodingMask;
158 
159 inline constexpr std::optional<TaggedAddressRange>
160 UniqueMapRangeOfInstanceTypeRange(InstanceType first, InstanceType last) {
161  // Doesn't use range based for loop due to LLVM <11 bug re. constexpr
162  // functions.
163  for (size_t i = 0; i < kUniqueMapRangeOfInstanceTypeRangeList.size(); ++i) {
164  if (kUniqueMapRangeOfInstanceTypeRangeList[i].first.first == first &&
165  kUniqueMapRangeOfInstanceTypeRangeList[i].first.second == last) {
166  return {kUniqueMapRangeOfInstanceTypeRangeList[i].second};
167  }
168  }
169  return {};
170 }

172 inline constexpr std::optional<TaggedAddressRange> UniqueMapRangeOfInstanceType(
173  InstanceType type) {
174  return UniqueMapRangeOfInstanceTypeRange(type, type);
175 }

177 inline bool MayHaveMapCheckFastCase(InstanceType type) {
178  if (UniqueMapOfInstanceType(type)) return true;
179  for (auto& el : kUniqueMapRangeOfInstanceTypeRangeList) {
180  if (el.first.first <= type && type <= el.first.second) {
181  return true;
182  }
183  }
184  return false;

192 inline constexpr bool CheckInstanceMapRange(TaggedAddressRange expected,
193  Tagged<Map> map) {
194  Tagged_t ptr = V8HeapCompressionScheme::CompressObject(map.ptr());
195  // Make the upper limit be the end of the map object, rather than the start;
196  // this helps the C++ optimizer realize there are no "holes" between
197  // neighbouring map ranges, such as string map ranges.
198  return base::IsInRange(ptr, expected.first, expected.second + Map::kSize - 1);
199 }

208 // INSTANCE_TYPE_CHECKER1 is to be used if the instance type is already loaded.
209 // INSTANCE_TYPE_CHECKER2 is preferred since it can sometimes avoid loading the

211 // known map or range of maps.

213 #define INSTANCE_TYPE_CHECKER1(type, forinstancetype) \

215  return instance_type == forinstancetype; \
216  }

220 #define INSTANCE_TYPE_CHECKER2(type, forinstancetype_) \
221  V8_INLINE bool Is##type(Tagged<Map> map_object) { \

249 // Checks if value is in range [lower_limit, higher_limit] using a single
250 // branch. Assumes that the input instance type is valid.
251 template <InstanceType lower_limit, InstanceType upper_limit>
252 struct InstanceRangeChecker {
253  static constexpr bool Check(InstanceType value) {
254  return base::IsInRange(value, lower_limit, upper_limit);
255  }
256 };
257 template <InstanceType upper_limit>
258 struct InstanceRangeChecker<FIRST_TYPE, upper_limit> {
259  static constexpr bool Check(InstanceType value) {
260  DCHECK_LE(FIRST_TYPE, value);
261  return value <= upper_limit;
262  }
263 };
264 template <InstanceType lower_limit>
265 struct InstanceRangeChecker<lower_limit, LAST_TYPE> {
266  static constexpr bool Check(InstanceType value) {
267  DCHECK_GE(LAST_TYPE, value);
268  return value >= lower_limit;
269  }
270 };

562 // Returns true for those heap object types that must be tied to some native
563 // context.
564 V8_INLINE constexpr bool IsNativeContextSpecific(InstanceType instance_type) {
565  // All context map are tied to some native context.
566  if (IsContext(instance_type)) return true;
567  // All non-JSReceivers are never tied to any native context.
568  if (!IsJSReceiver(instance_type)) return false;
569 
570  // Most of the JSReceivers are tied to some native context modulo the
571  // following exceptions.
572  if (IsMaybeReadOnlyJSObject(instance_type)) {
573  // These JSObject types are wrappers around a set of primitive values
574  // and exist only for the purpose of passing the data across V8 Api.
575  // Thus they are not tied to any native context.
576  return false;
577  } else if (InstanceTypeChecker::IsAlwaysSharedSpaceJSObject(instance_type)) {
578  // JSObjects allocated in shared space are never tied to a native context.
579  return false;
580 
581 #if V8_ENABLE_WEBASSEMBLY
582  } else if (InstanceTypeChecker::IsWasmObject(instance_type)) {
