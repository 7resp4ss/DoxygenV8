481 void HandleScopeImplementer::DeleteExtensions(internal::Address* prev_limit) {
482  while (!blocks_.empty()) {
483  internal::Address* block_start = blocks_.back();
484  internal::Address* block_limit = block_start + kHandleBlockSize;
485 
486  // SealHandleScope may make the prev_limit to point inside the block.
487  // Cast possibly-unrelated pointers to plain Address before comparing them
488  // to avoid undefined behavior.
489  if (reinterpret_cast<Address>(block_start) <
490  reinterpret_cast<Address>(prev_limit) &&
491  reinterpret_cast<Address>(prev_limit) <=
492  reinterpret_cast<Address>(block_limit)) {
493 #ifdef ENABLE_LOCAL_HANDLE_ZAPPING
494  internal::HandleScope::ZapRange(prev_limit, block_limit);
495 #endif
496  break;
497  }
498 
499  blocks_.pop_back();
500 #ifdef ENABLE_LOCAL_HANDLE_ZAPPING
501  internal::HandleScope::ZapRange(block_start, block_limit);
502 #endif
503  if (spare_ != nullptr) {
504  DeleteArray(spare_);
505  }
506  spare_ = block_start;
507  }
508  DCHECK((blocks_.empty() && prev_limit == nullptr) ||
509  (!blocks_.empty() && prev_limit != nullptr));
510 }
