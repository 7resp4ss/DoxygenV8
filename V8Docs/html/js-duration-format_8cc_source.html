88 Maybe<DurationUnitOptions> GetDurationUnitOptions(
89  Isolate* isolate, Unit unit, const char* unit_string,
90  const char* display_field, DirectHandle<JSReceiver> options,
91  JSDurationFormat::Style base_style,
92  const std::span<const std::string_view> value_strings,
93  const std::span<const JSDurationFormat::FieldStyle> value_enums,
94  JSDurationFormat::FieldStyle digital_base,
95  JSDurationFormat::FieldStyle prev_style) {
96  const char* method_name = "Intl.DurationFormat";
97  JSDurationFormat::FieldStyle style;
98  // 1. Let style be ? GetOption(options, unit, "string", stylesList,
99  // undefined).
100  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
101  isolate, style,
102  GetStringOption<JSDurationFormat::FieldStyle>(
103  isolate, options, unit_string, method_name, value_strings,
104  value_enums, JSDurationFormat::FieldStyle::kUndefined),
105  Nothing<DurationUnitOptions>());
106 
107  // 2. Let displayDefault be "always".
108  JSDurationFormat::Display display_default =
109  JSDurationFormat::Display::kAlways;
110  // 3. If style is undefined, then
111  if (style == JSDurationFormat::FieldStyle::kUndefined) {
112  // a. If baseStyle is "digital", then
113  if (base_style == JSDurationFormat::Style::kDigital) {
114  // i. If unit is not one of "hours", "minutes", or "seconds", then
115  if (unit != Unit::kHours && unit != Unit::kMinutes &&
116  unit != Unit::kSeconds) {
117  // a. Set displayDefault to "auto".
118  display_default = JSDurationFormat::Display::kAuto;
119  }
120  // ii. Set style to digitalBase.
121  style = digital_base;
122  // b. Else
123  } else {
124  // i. if prevStyle is "fractional", "numeric", or "2-digit", then
125  if (prev_style == JSDurationFormat::FieldStyle::kFractional ||
126  prev_style == JSDurationFormat::FieldStyle::kNumeric ||
127  prev_style == JSDurationFormat::FieldStyle::k2Digit) {
128  // 1. If unit is not one of "minutes" or "seconds", then
129  if (unit != Unit::kMinutes && unit != Unit::kSeconds) {
130  // a. Set displayDefault to "auto".
131  display_default = JSDurationFormat::Display::kAuto;
132  }
133  // 2. Set style to "numeric".
134  style = JSDurationFormat::FieldStyle::kNumeric;
135  // iii. Else,
136  } else {
137  // 1. Set displayDefault to "auto".
138  display_default = JSDurationFormat::Display::kAuto;
139  // 2. Set style to baseStyle.
140  switch (base_style) {
141  case JSDurationFormat::Style::kLong:
142  style = JSDurationFormat::FieldStyle::kLong;
143  break;
144  case JSDurationFormat::Style::kShort:
145  style = JSDurationFormat::FieldStyle::kShort;
146  break;
147  case JSDurationFormat::Style::kNarrow:
148  style = JSDurationFormat::FieldStyle::kNarrow;
149  break;
150  default:
151  UNREACHABLE();
152  }
153  }
154  }
155  }
156  // 4. If style is "numeric", then
157  if (style == JSDurationFormat::FieldStyle::kNumeric) {
158  // a. If unit is one of "milliseconds", "microseconds", or "nanoseconds",
159  // then
160  if (unit == Unit::kMilliseconds || unit == Unit::kMicroseconds ||
161  unit == Unit::kNanoseconds) {
162  // i. Set style to "fractional".
163  style = JSDurationFormat::FieldStyle::kFractional;
164  // ii. Set displayDefault to "auto".
165  display_default = JSDurationFormat::Display::kAuto;
166  }
167  }
168  // 5. Let displayField be the string-concatenation of unit and "Display".
169  // 6. Let display be ? GetOption(options, displayField, "string", « "auto",
170  // "always" », displayDefault).
171  JSDurationFormat::Display display;
172  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
173  isolate, display,
174  GetStringOption<JSDurationFormat::Display>(
175  isolate, options, display_field, method_name,
176  std::to_array<const std::string_view>({"auto", "always"}),
177  std::array{JSDurationFormat::Display::kAuto,
178  JSDurationFormat::Display::kAlways},
179  display_default),
180  Nothing<DurationUnitOptions>());
181  // 7. If display is "always" and style is "fractional", then
182  if (display == JSDurationFormat::Display::kAlways &&
183  style == JSDurationFormat::FieldStyle::kFractional) {
184  // a. Throw a RangeError exception.
185  THROW_NEW_ERROR_RETURN_VALUE(
186  isolate,
187  NewRangeError(MessageTemplate::kInvalid,
188  isolate->factory()->object_string(), options),
189  Nothing<DurationUnitOptions>());
190  }
191  // 8. If prevStyle is "fractional", then
192  if (prev_style == JSDurationFormat::FieldStyle::kFractional) {
193  // a. If style is not "fractional", then
194  if (style != JSDurationFormat::FieldStyle::kFractional) {
195  // i. Throw a RangeError exception.
196  THROW_NEW_ERROR_RETURN_VALUE(
197  isolate,
198  NewRangeError(MessageTemplate::kInvalid,
199  isolate->factory()->object_string(), options),
200  Nothing<DurationUnitOptions>());
201  }
202  }
203  // 7. If prevStyle is "numeric" or "2-digit", then
204  if (prev_style == JSDurationFormat::FieldStyle::kNumeric ||
205  prev_style == JSDurationFormat::FieldStyle::k2Digit) {
206  // a. If style is not "fractional", "numeric" or "2-digit", then
207  if (style != JSDurationFormat::FieldStyle::kFractional &&
208  style != JSDurationFormat::FieldStyle::kNumeric &&
209  style != JSDurationFormat::FieldStyle::k2Digit) {
210  // i. Throw a RangeError exception.
211  THROW_NEW_ERROR_RETURN_VALUE(
212  isolate,
213  NewRangeError(MessageTemplate::kInvalid,
214  isolate->factory()->object_string(), options),
215  Nothing<DurationUnitOptions>());
216  }
217  // b. If unit is "minutes" or "seconds", then
218  if (unit == Unit::kMinutes || unit == Unit::kSeconds) {
219  // i. Set style to "2-digit".
220  style = JSDurationFormat::FieldStyle::k2Digit;
221  }
222  }
223  // 8. Return the Record { [[Style]]: style, [[Display]]: display }.
224  return Just(DurationUnitOptions({style, display}));
225 }

227 JSDurationFormat::Separator GetSeparator(const icu::Locale& l) {
228  UErrorCode status = U_ZERO_ERROR;
229  icu::DateFormatSymbols sym(l, status);
230  if (U_FAILURE(status)) return JSDurationFormat::Separator::kColon;
231  icu::UnicodeString sep;
232  sym.getTimeSeparatorString(sep);
233  if (sep.length() != 1) return JSDurationFormat::Separator::kColon;
234  switch (sep.charAt(0)) {
235  case u'.':
236  return JSDurationFormat::Separator::kFullStop;
237  case u'\uFF1A':
238  return JSDurationFormat::Separator::kFullwidthColon;
239  case u'\u066B':
240  return JSDurationFormat::Separator::kArabicDecimalSeparator;
241  // By default, or if we get anything else, just use ':'.
242  default:
243  return JSDurationFormat::Separator::kColon;
244  }
245 }

254  // 3. Let requestedLocales be ? CanonicalizeLocaleList(locales).

257  isolate, requested_locales,
258  Intl::CanonicalizeLocaleList(isolate, locales),
259  DirectHandle<JSDurationFormat>());
260 
261  // 4. Let options be ? GetOptionsObject(options).
262  DirectHandle<JSReceiver> options;
263  ASSIGN_RETURN_ON_EXCEPTION(
264  isolate, options, GetOptionsObject(isolate, input_options, method_name));
265 
266  // 5. Let matcher be ? GetOption(options, "localeMatcher", "string", «
267  // "lookup", "best fit" », "best fit").
268  Intl::MatcherOption matcher;
269  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
270  isolate, matcher, Intl::GetLocaleMatcher(isolate, options, method_name),
271  DirectHandle<JSDurationFormat>());
272 
273  // 6. Let numberingSystem be ? GetOption(options, "numberingSystem", "string",
274  // undefined, undefined).
275  //
276  // 7. If numberingSystem is not undefined, then
277  //
278  // a. If numberingSystem does not match the Unicode Locale Identifier type
279  // nonterminal, throw a RangeError exception.
280  // Note: The matching test and throw in Step 7-a is throw inside
281  // Intl::GetNumberingSystem.
282  std::string numbering_system_str;
283  bool get;
284  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
285  isolate, get,
286  Intl::GetNumberingSystem(isolate, options, method_name,
287  numbering_system_str),
288  DirectHandle<JSDurationFormat>());
289 
290  // 8. Let opt be the Record { [[localeMatcher]]: matcher, [[nu]]:
291  // numberingSystem }.
292  // 9. Let r be ResolveLocale(%DurationFormat%.[[AvailableLocales]],
293  // requestedLocales, opt, %DurationFormat%.[[RelevantExtensionKeys]],
294  // %DurationFormat%.[[LocaleData]]).
295  Intl::ResolvedLocale r;
296  if (!Intl::ResolveLocale(isolate, JSDurationFormat::GetAvailableLocales(),
297  requested_locales, matcher, {"nu"})
298  .To(&r)) {
299  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
300  }
301 
302  // 10. Let locale be r.[[locale]].
303  icu::Locale r_locale = r.icu_locale;
304  UErrorCode status = U_ZERO_ERROR;
305  // 11. Set durationFormat.[[Locale]] to locale.
306  // 12. Set durationFormat.[[NumberingSystem]] to r.[[nu]].
307  if (get) {
308  auto nu_extension_it = r.extensions.find("nu");
309  if (nu_extension_it != r.extensions.end() &&
310  nu_extension_it->second != numbering_system_str) {
311  r_locale.setUnicodeKeywordValue("nu", nullptr, status);
312  DCHECK(U_SUCCESS(status));
313  }
314  }
315  icu::Locale icu_locale = r_locale;
316  if (get && Intl::IsValidNumberingSystem(numbering_system_str)) {
317  r_locale.setUnicodeKeywordValue("nu", numbering_system_str, status);
318  DCHECK(U_SUCCESS(status));
319  }
320  std::string numbering_system = Intl::GetNumberingSystem(r_locale);
321  Separator separator = GetSeparator(r_locale);
322 
323  // 13. Let style be ? GetOption(options, "style", "string", « "long", "short",
324  // "narrow", "digital" », "long").
325  Style style;
326  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
327  isolate, style,
328  GetStringOption<Style>(isolate, options, "style", method_name,
329  std::to_array<const std::string_view>(
330  {"long", "short", "narrow", "digital"}),
331  std::array{Style::kLong, Style::kShort,
332  Style::kNarrow, Style::kDigital},
333  Style::kShort),
334  DirectHandle<JSDurationFormat>());
335 
336  // 14. Set durationFormat.[[Style]] to style.
337  // 15. Set durationFormat.[[DataLocale]] to r.[[dataLocale]].
338  DirectHandle<Managed<icu::Locale>> managed_locale =
339  Managed<icu::Locale>::From(
340  isolate, 0, std::shared_ptr<icu::Locale>{icu_locale.clone()});
341  // 16. Let prevStyle be the empty String.
342  // 17. For each row of Table 1, except the header row, in table order, do
343  // a. Let styleSlot be the Style Slot value of the current row.
344  // b. Let displaySlot be the Display Slot value of the current row.
345  // c. Let unit be the Unit value.
346  // d. Let valueList be the Values value.
347  // e. Let digitalBase be the Digital Default value.
348  // f. Let unitOptions be ? GetDurationUnitOptions(unit, options, style,
349  // valueList, digitalBase, prevStyle).
350  // of durationFormat to unitOptions.[[Style]].
351  // h. Set the value of the
352  // displaySlot slot of durationFormat to unitOptions.[[Display]].
353  // i. If unit is one of "hours", "minutes", "seconds", "milliseconds",
354  // or "microseconds", then
355  // i. Set prevStyle to unitOptions.[[Style]].
356  // g. Set the value of the styleSlot slot
357  DurationUnitOptions years_option;
358  DurationUnitOptions months_option;
359  DurationUnitOptions weeks_option;
360  DurationUnitOptions days_option;
361  DurationUnitOptions hours_option;
362  DurationUnitOptions minutes_option;
