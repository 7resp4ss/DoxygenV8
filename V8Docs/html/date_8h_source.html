51  // Computes floor(time_ms / kMsPerDay).
52  static int DaysFromTime(int64_t time_ms) {
53  if (time_ms < 0) time_ms -= (kMsPerDay - 1);
54  return static_cast<int>(time_ms / kMsPerDay);
55  }
56 
57  // Computes modulo(time_ms, kMsPerDay) given that
58  // days = floor(time_ms / kMsPerDay).
59  static int TimeInDay(int64_t time_ms, int days) {
60  return static_cast<int>(time_ms - days * kMsPerDay);
61  }

63  // Performs the success path of the ECMA 262 TimeClip operation (when the
64  // value is within the range, truncates it to an integer). Returns false if
65  // the value is outside the range, and should be clipped to NaN.
66  // ECMA 262 - ES#sec-timeclip TimeClip (time)
67  static bool TryTimeClip(double* time) {
68  if (-kMaxTimeInMs <= *time && *time <= kMaxTimeInMs) {
69  // Inline the finite part of DoubleToInteger here, since the range check
70  // already covers the non-finite checks.
71  *time = ((*time > 0) ? std::floor(*time) : std::ceil(*time)) + 0.0;
72  return true;
73  }
74  return false;
75  }

77  // Given the number of days since the epoch, computes the weekday.
78  // ECMA 262 - 15.9.1.6.
79  int Weekday(int days) {
80  int result = (days + 4) % 7;
81  return result >= 0 ? result : result + 7;
82  }

84  bool IsLeap(int year) {
85  return year % 4 == 0 && (year % 100 != 0 || year % 400 == 0);

91  const char* LocalTimezone(int64_t time_ms) {
92  if (time_ms < 0 || time_ms > kMaxEpochTimeInMs) {
93  time_ms = EquivalentTime(time_ms);
94  }
95  bool is_dst = DaylightSavingsOffsetInMs(time_ms) != 0;
96  const char** name = is_dst ? &dst_tz_name_ : &tz_name_;
97  if (*name == nullptr) {
98  *name = tz_cache_->LocalTimezone(static_cast<double>(time_ms));
99  }
100  return *name;
101  }

103  // ECMA 262 - 15.9.5.26
104  int TimezoneOffset(int64_t time_ms) {
105  int64_t local_ms = ToLocal(time_ms);
106  return static_cast<int>((time_ms - local_ms) / kMsPerMin);
107  }
108 
109  // ECMA 262 - ES#sec-localtime-t
110  // LocalTime(t) = t + LocalTZA(t, true)
111  int64_t ToLocal(int64_t time_ms) {
112  return time_ms + LocalOffsetInMs(time_ms, true);
113  }
114 
115  // ECMA 262 - ES#sec-utc-t
116  // UTC(t) = t - LocalTZA(t, false)
117  int64_t ToUTC(int64_t time_ms) {
118  return time_ms - LocalOffsetInMs(time_ms, false);
119  }

121  // Computes a time equivalent to the given time according
122  // to ECMA 262 - 15.9.1.9.
123  // The issue here is that some library calls don't work right for dates
124  // that cannot be represented using a non-negative signed 32 bit integer
125  // (measured in whole seconds based on the 1970 epoch).
126  // We solve this by mapping the time to a year with same leap-year-ness
127  // and same starting day for the year. The ECMAscript specification says
128  // we must do this, but for compatibility with other browsers, we use
129  // the actual year if it is in the range 1970..2037
130  int64_t EquivalentTime(int64_t time_ms) {
131  int days = DaysFromTime(time_ms);
132  int time_within_day_ms = static_cast<int>(time_ms - days * kMsPerDay);
133  int year, month, day;
134  YearMonthDayFromDays(days, &year, &month, &day);
135  int new_days = DaysFromYearMonth(EquivalentYear(year), month) + day - 1;
136  return static_cast<int64_t>(new_days) * kMsPerDay + time_within_day_ms;
137  }
138 
139  // Returns an equivalent year in the range [2008-2035] matching
140  // - leap year,
141  // - week day of first day.
142  // ECMA 262 - 15.9.1.9.
143  int EquivalentYear(int year) {
144  int week_day = Weekday(DaysFromYearMonth(year, 0));
145  int recent_year = (IsLeap(year) ? 1956 : 1967) + (week_day * 12) % 28;
146  // Find the year in the range 2008..2037 that is equivalent mod 28.
147  // Add 3*28 to give a positive argument to the modulus operator.
148  return 2008 + (recent_year + 3 * 28 - 2008) % 28;
149  }

188  // Computes the daylight savings offset for the given time.
189  // ECMA 262 - 15.9.1.8
190  int DaylightSavingsOffsetInMs(int64_t time_ms) {
191  int time_sec = (time_ms >= 0 && time_ms <= kMaxEpochTimeInMs)
192  ? static_cast<int>(time_ms / 1000)
193  : static_cast<int>(EquivalentTime(time_ms) / 1000);
194  return GetDaylightSavingsOffsetFromOS(time_sec);
195  }
