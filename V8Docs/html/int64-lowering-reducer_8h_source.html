54  Word32OrWord32Pair REDUCE(WordBinop)(Word32OrWord32Pair left,
55  Word32OrWord32Pair right,
56  WordBinopOp::Kind kind,
57  WordRepresentation rep) {
58  if (rep == WordRepresentation::Word64()) {
59  V<Word32Pair> left_pair = V<Word32Pair>::Cast(left);
60  V<Word32Pair> right_pair = V<Word32Pair>::Cast(right);
61  switch (kind) {
62  case WordBinopOp::Kind::kAdd:
63  return LowerPairBinOp(left_pair, right_pair,
64  Word32PairBinopOp::Kind::kAdd);
65  case WordBinopOp::Kind::kSub:
66  return LowerPairBinOp(left_pair, right_pair,
67  Word32PairBinopOp::Kind::kSub);
68  case WordBinopOp::Kind::kMul:
69  return LowerPairBinOp(left_pair, right_pair,
70  Word32PairBinopOp::Kind::kMul);
71  case WordBinopOp::Kind::kBitwiseAnd:
72  return LowerBitwiseAnd(left_pair, right_pair);
73  case WordBinopOp::Kind::kBitwiseOr:
74  return LowerBitwiseOr(left_pair, right_pair);
75  case WordBinopOp::Kind::kBitwiseXor:
76  return LowerBitwiseXor(left_pair, right_pair);
77  default:
78  FATAL("WordBinopOp kind %d not supported by int64 lowering",
79  static_cast<int>(kind));
80  }
81  }

109  V<Word32> REDUCE(Comparison)(V<Any> left, V<Any> right,
110  ComparisonOp::Kind kind,
111  RegisterRepresentation rep) {
112  if (rep != WordRepresentation::Word64()) {
113  return Next::ReduceComparison(left, right, kind, rep);
114  }
115 
116  auto [left_low, left_high] = Unpack(V<Word32Pair>::Cast(left));
117  auto [right_low, right_high] = Unpack(V<Word32Pair>::Cast(right));
118  V<Word32> high_comparison;
119  V<Word32> low_comparison;
120  switch (kind) {
121  case ComparisonOp::Kind::kEqual:
122  // TODO(wasm): Use explicit comparisons and && here?
123  return __ Word32Equal(
124  __ Word32BitwiseOr(__ Word32BitwiseXor(left_low, right_low),
125  __ Word32BitwiseXor(left_high, right_high)),
126  0);
127  case ComparisonOp::Kind::kSignedLessThan:
128  high_comparison = __ Int32LessThan(left_high, right_high);
129  low_comparison = __ Uint32LessThan(left_low, right_low);
130  break;
131  case ComparisonOp::Kind::kSignedLessThanOrEqual:
132  high_comparison = __ Int32LessThan(left_high, right_high);
133  low_comparison = __ Uint32LessThanOrEqual(left_low, right_low);
134  break;
135  case ComparisonOp::Kind::kUnsignedLessThan:
136  high_comparison = __ Uint32LessThan(left_high, right_high);
137  low_comparison = __ Uint32LessThan(left_low, right_low);
138  break;
139  case ComparisonOp::Kind::kUnsignedLessThanOrEqual:
140  high_comparison = __ Uint32LessThan(left_high, right_high);
141  low_comparison = __ Uint32LessThanOrEqual(left_low, right_low);
142  break;
143  }
144 
145  return __ Word32BitwiseOr(
146  high_comparison,
147  __ Word32BitwiseAnd(__ Word32Equal(left_high, right_high),
148  low_comparison));
149  }

254  OpIndex REDUCE(Change)(OpIndex input, ChangeOp::Kind kind,
255  ChangeOp::Assumption assumption,
256  RegisterRepresentation from,
257  RegisterRepresentation to) {
258  auto word32 = RegisterRepresentation::Word32();
259  auto word64 = RegisterRepresentation::Word64();
260  auto float64 = RegisterRepresentation::Float64();
261  using Kind = ChangeOp::Kind;
262  if (from != word64 && to != word64) {
263  return Next::ReduceChange(input, kind, assumption, from, to);
264  }
265 
266  if (from == word32 && to == word64) {
267  if (kind == Kind::kZeroExtend) {
268  return __ Tuple(V<Word32>::Cast(input), __ Word32Constant(0));
269  }
270  if (kind == Kind::kSignExtend) {
271  return LowerSignExtend(input);
272  }
273  }
274  if (from == float64 && to == word64) {
275  if (kind == Kind::kBitcast) {
276  return __ Tuple(__ Float64ExtractLowWord32(input),
277  __ Float64ExtractHighWord32(input));
278  }
279  }
280  if (from == word64 && to == float64) {
281  if (kind == Kind::kBitcast) {
282  auto input_w32p = V<Word32Pair>::Cast(input);
283  return __ BitcastWord32PairToFloat64(
284  __ template Projection<1>(input_w32p),
285  __ template Projection<0>(input_w32p));
286  }
287  }
288  if (from == word64 && to == word32 && kind == Kind::kTruncate) {
289  auto input_w32p = V<Word32Pair>::Cast(input);
290  return __ template Projection<0>(input_w32p);
291  }
292  std::stringstream str;
293  str << "ChangeOp " << kind << " from " << from << " to " << to
294  << "not supported by int64 lowering";
295  FATAL("%s", str.str().c_str());

298  std::pair<OptionalV<Word32>, int32_t> IncreaseOffset(OptionalV<Word32> index,
299  int32_t offset,
300  int32_t add_offset,
301  bool tagged_base) {
302  // Note that the offset will just wrap around. Still, we need to always
303  // use an offset that is not std::numeric_limits<int32_t>::min() on tagged
304  // loads.
305  // TODO(dmercadier): Replace LoadOp::OffsetIsValid by taking care of this
306  // special case in the LoadStoreSimplificationReducer instead.
307  int32_t new_offset =
308  static_cast<uint32_t>(offset) + static_cast<uint32_t>(add_offset);
309  OptionalV<Word32> new_index = index;
310  if (!LoadOp::OffsetIsValid(new_offset, tagged_base)) {
311  // We cannot encode the new offset so we use the old offset
312  // instead and use the Index to represent the extra offset.
313  new_offset = offset;
314  if (index.has_value()) {
315  new_index = __ Word32Add(new_index.value(), add_offset);
316  } else {
317  new_index = __ Word32Constant(sizeof(int32_t));
318  }
319  }
320  return {new_index, new_offset};
321  }

323  OpIndex REDUCE(Load)(OpIndex base, OptionalOpIndex index, LoadOp::Kind kind,
324  MemoryRepresentation loaded_rep,
325  RegisterRepresentation result_rep, int32_t offset,
326  uint8_t element_scale) {
327  if (kind.is_atomic) {
328  if (loaded_rep == MemoryRepresentation::Int64() ||
329  loaded_rep == MemoryRepresentation::Uint64()) {
330  // TODO(jkummerow): Support non-zero scales in AtomicWord32PairOp, and
331  // remove the corresponding bailout in MachineOptimizationReducer to
332  // allow generating them.
333  if (element_scale != 0 && index.has_value()) {
334  DCHECK_EQ(element_scale, 3);
335  index = __ Word32ShiftLeft(index.value(), element_scale);
336  }
337  // Manually subtract the pointer tag if present.
338  offset -= kind.tagged_base;
339  return __ AtomicWord32PairLoad(base, index, offset);
340  }
341  if (result_rep == RegisterRepresentation::Word64()) {
342  return __ Tuple(
343  __ Load(base, index, kind, loaded_rep,
344  RegisterRepresentation::Word32(), offset, element_scale),
345  __ Word32Constant(0));
346  }
347  }
348  if (loaded_rep == MemoryRepresentation::Int64() ||
349  loaded_rep == MemoryRepresentation::Uint64()) {
350  auto [high_index, high_offset] =
351  IncreaseOffset(index, offset, sizeof(int32_t), kind.tagged_base);
352  return __ Tuple(
353  Next::ReduceLoad(base, index, kind, MemoryRepresentation::Int32(),
354  RegisterRepresentation::Word32(), offset,
355  element_scale),
356  Next::ReduceLoad(
357  base, high_index, kind, MemoryRepresentation::Int32(),
358  RegisterRepresentation::Word32(), high_offset, element_scale));
359  }
360  return Next::ReduceLoad(base, index, kind, loaded_rep, result_rep, offset,
361  element_scale);
362  }

364  V<None> REDUCE(Store)(OpIndex base, OptionalOpIndex index, OpIndex value,
365  StoreOp::Kind kind, MemoryRepresentation stored_rep,
366  WriteBarrierKind write_barrier, int32_t offset,
367  uint8_t element_size_log2,
368  bool maybe_initializing_or_transitioning,
369  IndirectPointerTag maybe_indirect_pointer_tag) {
370  if (stored_rep == MemoryRepresentation::Int64() ||
371  stored_rep == MemoryRepresentation::Uint64()) {
372  auto [low, high] = Unpack(value);
373  if (kind.is_atomic) {
374  // TODO(jkummerow): Support non-zero scales in AtomicWord32PairOp, and
375  // remove the corresponding bailout in MachineOptimizationReducer to
376  // allow generating them.
377  if (element_size_log2 != 0 && index.has_value()) {
378  DCHECK_EQ(element_size_log2, 3);
379  index = __ Word32ShiftLeft(index.value(), element_size_log2);
380  }
381  // Manually subtract the pointer tag if present.
382  offset -= kind.tagged_base;
383  return __ AtomicWord32PairStore(base, index, low, high, offset);
384  }
385  // low store
386  Next::ReduceStore(base, index, low, kind, MemoryRepresentation::Int32(),
387  write_barrier, offset, element_size_log2,
388  maybe_initializing_or_transitioning,
389  maybe_indirect_pointer_tag);
390  // high store
391  auto [high_index, high_offset] =
392  IncreaseOffset(index, offset, sizeof(int32_t), kind.tagged_base);
393  Next::ReduceStore(
394  base, high_index, high, kind, MemoryRepresentation::Int32(),
395  write_barrier, high_offset, element_size_log2,
396  maybe_initializing_or_transitioning, maybe_indirect_pointer_tag);
397  return V<None>::Invalid();

399  return Next::ReduceStore(base, index, value, kind, stored_rep,
400  write_barrier, offset, element_size_log2,
401  maybe_initializing_or_transitioning,
402  maybe_indirect_pointer_tag);

467  void FixLoopPhi(const PhiOp& input_phi, OpIndex output_index,
468  Block* output_graph_loop) {
469  if (input_phi.rep == RegisterRepresentation::Word64()) {
470  const TupleOp& tuple = __ Get(output_index).template Cast<TupleOp>();
471  DCHECK_EQ(tuple.input_count, 2);
472  OpIndex new_inputs[2] = {__ MapToNewGraph(input_phi.input(0)),
473  __ MapToNewGraph(input_phi.input(1))};
474  for (size_t i = 0; i < 2; ++i) {
475  OpIndex phi_index = tuple.input(i);
476  if (!output_graph_loop->Contains(phi_index)) {
477  continue;
478  }
479 #ifdef DEBUG
480  const PendingLoopPhiOp& pending_phi =
481  __ Get(phi_index).template Cast<PendingLoopPhiOp>();
482  DCHECK_EQ(pending_phi.rep, RegisterRepresentation::Word32());
483  DCHECK_EQ(
484  pending_phi.first(),
485  __ Projection(new_inputs[0], i, RegisterRepresentation::Word32()));
486 #endif
487  __ output_graph().template Replace<PhiOp>(
488  phi_index,
489  base::VectorOf({__ Projection(new_inputs[0], i,
490  RegisterRepresentation::Word32()),
491  __ Projection(new_inputs[1], i,
492  RegisterRepresentation::Word32())}),
493  RegisterRepresentation::Word32());
494  }
495  return;
496  }
497  return Next::FixLoopPhi(input_phi, output_index, output_graph_loop);

713  V<Word32Pair> LowerRotateRight(V<Word32Pair> left, V<Word32> right) {
714  // This reducer assumes that all rotates are mapped to rotate right.
715  DCHECK(!SupportedOperations::word64_rol());
716  auto [left_low, left_high] = Unpack(left);
717  V<Word32> shift = right;
718  uint32_t constant_shift = 0;
719 
720  if (matcher_.MatchIntegralWord32Constant(shift, &constant_shift)) {
721  // Precondition: 0 <= shift < 64.
722  uint32_t shift_value = constant_shift & 0x3F;
723  if (shift_value == 0) {
724  // No-op, return original tuple.
725  return left;
726  }
727  if (shift_value == 32) {
728  // Swap low and high of left.
729  return __ Tuple(left_high, left_low);
730  }
731 
732  V<Word32> low_input = left_high;
733  V<Word32> high_input = left_low;
734  if (shift_value < 32) {
735  low_input = left_low;
736  high_input = left_high;
737  }
738 
739  uint32_t masked_shift_value = shift_value & 0x1F;
740  V<Word32> masked_shift = __ Word32Constant(masked_shift_value);
741  V<Word32> inv_shift = __ Word32Constant(32 - masked_shift_value);
742 
743  V<Word32> low_node = __ Word32BitwiseOr(
744  __ Word32ShiftRightLogical(low_input, masked_shift),
745  __ Word32ShiftLeft(high_input, inv_shift));
746  V<Word32> high_node = __ Word32BitwiseOr(
747  __ Word32ShiftRightLogical(high_input, masked_shift),
748  __ Word32ShiftLeft(low_input, inv_shift));
749  return __ Tuple(low_node, high_node);
750  }
751 
752  V<Word32> safe_shift = shift;
753  if (!SupportedOperations::word32_shift_is_safe()) {
754  // safe_shift = shift % 32
755  safe_shift = __ Word32BitwiseAnd(shift, 0x1F);
756  }
757  V<Word32> all_bits_set = __ Word32Constant(-1);
758  V<Word32> inv_mask = __ Word32BitwiseXor(
759  __ Word32ShiftRightLogical(all_bits_set, safe_shift), all_bits_set);
760  V<Word32> bit_mask = __ Word32BitwiseXor(inv_mask, all_bits_set);
761 
762  V<Word32> less_than_32 = __ Int32LessThan(shift, 32);
763  // The low word and the high word can be swapped either at the input or
764  // at the output. We swap the inputs so that shift does not have to be
765  // kept for so long in a register.
766  ScopedVar<Word32> var_low(this, left_high);
767  ScopedVar<Word32> var_high(this, left_low);
768  IF (less_than_32) {
769  var_low = left_low;
770  var_high = left_high;
771  }
772 
773  V<Word32> rotate_low = __ Word32RotateRight(var_low, safe_shift);
774  V<Word32> rotate_high = __ Word32RotateRight(var_high, safe_shift);
775 
776  V<Word32> low_node =
777  __ Word32BitwiseOr(__ Word32BitwiseAnd(rotate_low, bit_mask),
778  __ Word32BitwiseAnd(rotate_high, inv_mask));
779  V<Word32> high_node =
780  __ Word32BitwiseOr(__ Word32BitwiseAnd(rotate_high, bit_mask),
781  __ Word32BitwiseAnd(rotate_low, inv_mask));
782  return __ Tuple(low_node, high_node);
783  }

785  V<Any> LowerCall(V<CallTarget> callee, OptionalV<FrameState> frame_state,
786  base::Vector<const OpIndex> arguments,
787  const TSCallDescriptor* descriptor, OpEffects effects,
788  bool is_tail_call) {
789  // Iterate over the call descriptor to skip lowering if the signature does
790  // not contain an i64.
791  const CallDescriptor* call_descriptor = descriptor->descriptor;
792  size_t param_count = call_descriptor->ParameterCount();
793  size_t i64_params = 0;
794  for (size_t i = 0; i < param_count; ++i) {
795  i64_params += call_descriptor->GetParameterType(i).representation() ==
796  MachineRepresentation::kWord64;
797  }
798  size_t return_count = call_descriptor->ReturnCount();
799  size_t i64_returns = 0;
800  for (size_t i = 0; i < return_count; ++i) {
801  i64_returns += call_descriptor->GetReturnType(i).representation() ==
802  MachineRepresentation::kWord64;
803  }
804  if (i64_params + i64_returns == 0) {
805  // No lowering required.
806  return is_tail_call ? Next::ReduceTailCall(callee, arguments, descriptor)
807  : Next::ReduceCall(callee, frame_state, arguments,
808  descriptor, effects);
809  }
810 
811  // Transform the BigIntToI64 call descriptor into BigIntToI32Pair (this is
812  // the only use case currently, it may be extended in the future).
813  // The correct target is already set during graph building.
814  CallDescriptor* maybe_special_replacement =
815  wasm::GetWasmEngine()->call_descriptors()->GetLoweredCallDescriptor(
816  call_descriptor);
817  if (maybe_special_replacement) call_descriptor = maybe_special_replacement;
818  // Create descriptor with 2 i32s for every i64.
819  const CallDescriptor* lowered_descriptor =
820  GetI32WasmCallDescriptor(__ graph_zone(), call_descriptor);
821 
822  // Map the arguments by unpacking i64 arguments (which have already been
823  // lowered to Tuple(i32, i32).)
824  base::SmallVector<OpIndex, 16> lowered_args;
825  lowered_args.reserve(param_count + i64_params);
826 
827  DCHECK_EQ(param_count, arguments.size());
828  for (size_t i = 0; i < param_count; ++i) {
829  if (call_descriptor->GetParameterType(i).representation() ==
830  MachineRepresentation::kWord64) {
831  auto [low, high] = Unpack(arguments[i]);
832  lowered_args.push_back(low);
833  lowered_args.push_back(high);
834  } else {
835  lowered_args.push_back(arguments[i]);
836  }
837  }
838 
839  auto lowered_ts_descriptor =
840  TSCallDescriptor::Create(lowered_descriptor, descriptor->can_throw,
841  LazyDeoptOnThrow::kNo, __ graph_zone());
842  OpIndex call =
843  is_tail_call
844  ? Next::ReduceTailCall(callee, base::VectorOf(lowered_args),
845  lowered_ts_descriptor)
846  : Next::ReduceCall(callee, frame_state,
847  base::VectorOf(lowered_args),
848  lowered_ts_descriptor, effects);
849  if (is_tail_call) {
850  // Tail calls don't return anything to the calling function.
851  return call;
852  }
853  if (i64_returns == 0 || return_count == 0) {
854  return call;
855  } else if (return_count == 1) {
856  // There isn't any projection in the input graph for calls returning
857  // exactly one value. Return a tuple of projections for the int64.
858  DCHECK_EQ(i64_returns, 1);
859  return call;
860  }
861 
862  // Wrap the call node with a tuple of projections of the lowered call.
863  // Example for a call returning [int64, int32]:
864  // In: Call(...) -> [int64, int32]
865  // Out: call = Call() -> [int32, int32, int32]
866  // Tuple(
867  // Tuple(Projection(call, 0), Projection(call, 1)),
868  // Projection(call, 2))
869  //
870  // This way projections on the original call node will be automatically
871  // "rewired" to the correct projection of the lowered call.
872  auto word32 = RegisterRepresentation::Word32();
873  base::SmallVector<V<Any>, 16> tuple_inputs;
874  tuple_inputs.reserve(return_count);
875  size_t projection_index = 0; // index of the lowered call results.
876 
877  for (size_t i = 0; i < return_count; ++i) {
878  MachineRepresentation machine_rep =
879  call_descriptor->GetReturnType(i).representation();
880  if (machine_rep == MachineRepresentation::kWord64) {
881  tuple_inputs.push_back(
882  __ Tuple(__ Projection(call, projection_index, word32),
883  __ Projection(call, projection_index + 1, word32)));
884  projection_index += 2;
885  } else {
886  tuple_inputs.push_back(__ Projection(
887  call, projection_index++,
888  RegisterRepresentation::FromMachineRepresentation(machine_rep)));
889  }
890  }
891  DCHECK_EQ(projection_index, return_count + i64_returns);
892  return __ Tuple(base::VectorOf(tuple_inputs));
893  }
