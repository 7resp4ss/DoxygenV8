47 // Simd128UseMap computes uses of revectorizable Simd128 operations of the given
48 // turboshaft graph and skips other operations.
49 class Simd128UseMap : public UseMap, public NON_EXPORTED_BASE(ZoneObject) {
50  public:
51  Simd128UseMap(const Graph& graph, Zone* zone)
52  : UseMap(
53  graph, zone,
54  [](const Graph& graph, const Operation& op, Zone* zone) {
55  auto has_revectorizable_simd128_output = [](const Operation& op) {
56  // Revectorizable Simd128 operations only have single output.
57  return op.outputs_rep().size() == 1 &&
58  op.outputs_rep()[0] == RegisterRepresentation::Simd128();
59  };
60 
61  if (has_revectorizable_simd128_output(op)) return false;
62 
63  // Identify input_reps of ReturnOp and FrameStateOp (which doesn't
64  // track inputs_rep, except for the Word32 popcount input of
65  // ReturnOp).
66  if (op.TryCast<ReturnOp>() || op.TryCast<FrameStateOp>()) {
67  for (auto input : op.inputs()) {
68  const Operation& input_op = graph.Get(input);
69  if (has_revectorizable_simd128_output(input_op)) return false;
70  }
71  return true;
72  }
73 
74  ZoneVector<MaybeRegisterRepresentation> storage(zone);
75  for (auto rep : op.inputs_rep(storage)) {
76  if (rep == MaybeRegisterRepresentation::Simd128()) return false;
77  }
78 
79  SLOW_DCHECK(std::none_of(
80  op.inputs().begin(), op.inputs().end(),
81  [&](const OpIndex input) {
82  return has_revectorizable_simd128_output(graph.Get(input));
83  }));
84  return true;
85  }) {}
86 };
