175 bool Script::HasValidSource() {
176  Tagged<Object> src = this->source();
177  if (!IsString(src)) return true;
178  Tagged<String> src_str = Cast<String>(src);
179  if (!StringShape(src_str).IsExternal()) return true;
180  if (src_str->IsOneByteRepresentation()) {
181  return Cast<ExternalOneByteString>(src)->resource() != nullptr;
182  } else if (src_str->IsTwoByteRepresentation()) {
183  return Cast<ExternalTwoByteString>(src)->resource() != nullptr;
184  }
185  return true;
186 }

225 Tagged<Script> Script::GetEvalOrigin() {
226  DisallowGarbageCollection no_gc;
227  Tagged<Script> origin_script = *this;
228  while (origin_script->has_eval_from_shared()) {
229  Tagged<HeapObject> maybe_script =
230  origin_script->eval_from_shared()->script();
231  CHECK(IsScript(maybe_script));
232  origin_script = Cast<Script>(maybe_script);
233  }
234  return origin_script;
235 }
