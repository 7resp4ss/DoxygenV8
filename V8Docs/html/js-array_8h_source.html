20 // The JSArray describes JavaScript Arrays

22 // - fast, backing storage is a FixedArray and length <= elements.length();

85  // Support for Array.prototype.join().
86  // Writes a fixed array of strings and separators to a single destination
87  // string. This helpers assumes the fixed array encodes separators in two
88  // ways:
89  // 1) Explicitly with a smi, whos value represents the number of repeated
90  // separators.
91  // 2) Implicitly between two consecutive strings a single separator.
92  //
93  // In addition repeated strings are represented by a negative smi, indicating
94  // how many times the previously written string has to be repeated.
95  //
96  // Here are some input/output examples given the separator string is ',':
97  //
98  // [1, 'hello', 2, 'world', 1] => ',hello,,world,'
99  // ['hello', 'world'] => 'hello,world'
100  // ['hello', -2, 'world'] => 'hello,hello,hello,world'
101  //
102  // To avoid any allocations, this helper assumes the destination string is the
103  // exact length necessary to write the strings and separators from the fixed
104  // array.
105  // Since this is called via ExternalReferences, it uses raw Address values:
106  // - {raw_fixed_array} is a tagged FixedArray pointer.
107  // - {raw_separator} and {raw_dest} are tagged String pointers.
108  // - Returns a tagged String pointer.
109  static Address ArrayJoinConcatToSequentialString(Isolate* isolate,
110  Address raw_fixed_array,
111  intptr_t length,
112  Address raw_separator,
113  Address raw_dest);
114 
115  // Checks whether the Array has the current realm's Array.prototype as its

117  // approximation, erring on the side of false, in particular with respect
