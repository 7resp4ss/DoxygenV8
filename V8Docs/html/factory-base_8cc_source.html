83 template <typename Impl>
84 Handle<Code> FactoryBase<Impl>::NewCode(const NewCodeOptions& options) {
85  DirectHandle<CodeWrapper> wrapper = NewCodeWrapper();
86  Tagged<Map> map = read_only_roots().code_map();
87  int size = map->instance_size();
88  Tagged<Code> code = Cast<Code>(
89  AllocateRawWithImmortalMap(size, AllocationType::kTrusted, map));
90  DisallowGarbageCollection no_gc;
91  code->init_self_indirect_pointer(isolate());
92  code->initialize_flags(options.kind, options.is_context_specialized,
93  options.is_turbofanned);
94  code->set_builtin_id(options.builtin);
95  code->set_instruction_size(options.instruction_size);
96  code->set_metadata_size(options.metadata_size);
97  code->set_inlined_bytecode_size(options.inlined_bytecode_size);
98  code->set_osr_offset(options.osr_offset);
99  SBXCHECK_IMPLIES(options.deoptimization_data.is_null(),
100  options.osr_offset.IsNone());
101  code->set_handler_table_offset(options.handler_table_offset);
102  code->set_constant_pool_offset(options.constant_pool_offset);
103  code->set_code_comments_offset(options.code_comments_offset);
104  code->set_jump_table_info_offset(options.jump_table_info_offset);
105  code->set_unwinding_info_offset(options.unwinding_info_offset);
106  code->set_parameter_count(options.parameter_count);
107 #ifdef V8_ENABLE_LEAPTIERING
108  code->set_js_dispatch_handle(kNullJSDispatchHandle);
109 #endif // V8_ENABLE_LEAPTIERING
110 
111  // Set bytecode/interpreter data or deoptimization data.
112  if (CodeKindUsesBytecodeOrInterpreterData(options.kind)) {
113  DCHECK(options.deoptimization_data.is_null());
114  Tagged<TrustedObject> data =
115  *options.bytecode_or_interpreter_data.ToHandleChecked();
116  DCHECK(IsBytecodeArray(data) || IsInterpreterData(data));
117  code->set_bytecode_or_interpreter_data(data);
118  } else if (CodeKindUsesDeoptimizationData(options.kind)) {
119  DCHECK(options.bytecode_or_interpreter_data.is_null());
120  code->set_deoptimization_data(
121  *options.deoptimization_data.ToHandleChecked());
122  } else {
123  DCHECK(options.deoptimization_data.is_null());
124  DCHECK(options.bytecode_or_interpreter_data.is_null());
125  code->clear_deoptimization_data_and_interpreter_data();
126  }
127 
128  // Set bytecode offset table or source position table.
129  if (CodeKindUsesBytecodeOffsetTable(options.kind)) {
130  DCHECK(options.source_position_table.is_null());
131  code->set_bytecode_offset_table(
132  *options.bytecode_offset_table.ToHandleChecked());
133  } else if (CodeKindMayLackSourcePositionTable(options.kind)) {
134  DCHECK(options.bytecode_offset_table.is_null());
135  Handle<TrustedByteArray> table;
136  if (options.source_position_table.ToHandle(&table)) {
137  code->set_source_position_table(*table);
138  } else {
139  code->clear_source_position_table_and_bytecode_offset_table();
140  }
141  } else {
142  DCHECK(options.bytecode_offset_table.is_null());
143  code->set_source_position_table(
144  *options.source_position_table.ToHandleChecked());
145  }
146 
147  // Set instruction stream and entrypoint.
148  Handle<InstructionStream> istream;
149  if (options.instruction_stream.ToHandle(&istream)) {
150  DCHECK_EQ(options.instruction_start, kNullAddress);
151  code->SetInstructionStreamAndInstructionStart(isolate(), *istream);
152  } else {
153  DCHECK_NE(options.instruction_start, kNullAddress);
154  code->set_raw_instruction_stream(Smi::zero(), SKIP_WRITE_BARRIER);
155  code->SetInstructionStartForOffHeapBuiltin(isolate(),
156  options.instruction_start);
157  }
158 
159  wrapper->set_code(code);
160  code->set_wrapper(*wrapper);
161 
162  code->clear_padding();
163  return handle(code, isolate());
164 }

329 template <typename Impl>
330 Handle<BytecodeArray> FactoryBase<Impl>::NewBytecodeArray(
331  int length, const uint8_t* raw_bytecodes, int frame_size,
332  uint16_t parameter_count, uint16_t max_arguments,
333  DirectHandle<TrustedFixedArray> constant_pool,
334  DirectHandle<TrustedByteArray> handler_table, AllocationType allocation) {
335  DCHECK(allocation == AllocationType::kTrusted ||
336  allocation == AllocationType::kSharedTrusted);
337  if (length < 0 || length > BytecodeArray::kMaxLength) {
338  FATAL("Fatal JavaScript invalid size error %d", length);
339  UNREACHABLE();
340  }
341  DirectHandle<BytecodeWrapper> wrapper = NewBytecodeWrapper();
342  int size = BytecodeArray::SizeFor(length);

345  DisallowGarbageCollection no_gc;
346  Tagged<BytecodeArray> instance = Cast<BytecodeArray>(result);
347  instance->init_self_indirect_pointer(isolate());
348  instance->set_length(length);
349  instance->set_frame_size(frame_size);
350  instance->set_parameter_count(parameter_count);
351  instance->set_max_arguments(max_arguments);
352  instance->set_incoming_new_target_or_generator_register(
353  interpreter::Register::invalid_value());
354  instance->set_constant_pool(*constant_pool);
355  instance->set_handler_table(*handler_table);
356  instance->clear_source_position_table(kReleaseStore);
357  instance->set_wrapper(*wrapper);
358  CopyBytes(reinterpret_cast<uint8_t*>(instance->GetFirstBytecodeAddress()),

360  instance->clear_padding();
361  wrapper->set_bytecode(instance);
362  return handle(instance, isolate());

871 template <typename Impl>
872 template <template <typename> typename HandleType>
873  requires(std::is_convertible_v<HandleType<String>, DirectHandle<String>>)
874 HandleType<String>::MaybeType FactoryBase<Impl>::NewConsString(
875  HandleType<String> left, HandleType<String> right,
876  AllocationType allocation) {
877  if (IsThinString(*left)) {
878  left = HandleType<String>(Cast<ThinString>(*left)->actual(), isolate());
879  }
880  if (IsThinString(*right)) {
881  right = HandleType<String>(Cast<ThinString>(*right)->actual(), isolate());
882  }
883  uint32_t left_length = left->length();
884  if (left_length == 0) return right;
885  uint32_t right_length = right->length();
886  if (right_length == 0) return left;
887 
888  uint32_t length = left_length + right_length;
889 
890  if (length == 2) {
891  uint16_t c1 = left->Get(0, isolate());
892  uint16_t c2 = right->Get(0, isolate());
893  return MakeOrFindTwoCharacterString(c1, c2);
894  }
895 
896  // Make sure that an out of memory exception is thrown if the length
897  // of the new cons string is too large.
898  if (length > String::kMaxLength || length < 0) {
899  THROW_NEW_ERROR(isolate(), NewInvalidStringLengthError());
900  }
901 
902  bool left_is_one_byte = left->IsOneByteRepresentation();
903  bool right_is_one_byte = right->IsOneByteRepresentation();
904  bool is_one_byte = left_is_one_byte && right_is_one_byte;
905 
906  // If the resulting string is small make a flat string.
907  if (length < ConsString::kMinLength) {
908  // Note that neither of the two inputs can be a slice because:
909  static_assert(ConsString::kMinLength <= SlicedString::kMinLength);
910  DCHECK(left->IsFlat());
911  DCHECK(right->IsFlat());
912 
913  static_assert(ConsString::kMinLength <= String::kMaxLength);
914  if (is_one_byte) {
915  HandleType<SeqOneByteString> result =
916  NewRawOneByteString(length, allocation).ToHandleChecked();
917  DisallowGarbageCollection no_gc;
918  SharedStringAccessGuardIfNeeded access_guard(isolate());
919  uint8_t* dest = result->GetChars(no_gc, access_guard);
920  // Copy left part.
921  {
922  const uint8_t* src =
923  left->template GetDirectStringChars<uint8_t>(no_gc, access_guard);
924  CopyChars(dest, src, left_length);
925  }
926  // Copy right part.
927  {
928  const uint8_t* src =
929  right->template GetDirectStringChars<uint8_t>(no_gc, access_guard);
930  CopyChars(dest + left_length, src, right_length);
931  }
932  return result;
933  }
934 
935  HandleType<SeqTwoByteString> result =
936  NewRawTwoByteString(length, allocation).ToHandleChecked();
937 
938  DisallowGarbageCollection no_gc;
939  SharedStringAccessGuardIfNeeded access_guard(isolate());
940  base::uc16* sink = result->GetChars(no_gc, access_guard);
941  String::WriteToFlat(*left, sink, 0, left->length(), access_guard);
942  String::WriteToFlat(*right, sink + left->length(), 0, right->length(),
943  access_guard);
944  return result;
945  }
946 
947  return NewConsString(left, right, length, is_one_byte, allocation);
948 }
