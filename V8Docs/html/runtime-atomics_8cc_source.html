143 #define ATOMIC_OPS(type, suffix, vctype) \
144  inline type ExchangeSeqCst(type* p, type value) { \
145  return InterlockedExchange##suffix(reinterpret_cast<vctype*>(p), \
146  base::bit_cast<vctype>(value)); \
147  } \
148  inline type CompareExchangeSeqCst(type* p, type oldval, type newval) { \
149  return InterlockedCompareExchange##suffix(reinterpret_cast<vctype*>(p), \
150  base::bit_cast<vctype>(newval), \
151  base::bit_cast<vctype>(oldval)); \
152  } \
153  inline type AddSeqCst(type* p, type value) { \
154  return InterlockedExchangeAdd##suffix(reinterpret_cast<vctype*>(p), \
155  base::bit_cast<vctype>(value)); \
156  } \
157  inline type SubSeqCst(type* p, type value) { \
158  return InterlockedExchangeAdd##suffix(reinterpret_cast<vctype*>(p), \
159  -base::bit_cast<vctype>(value)); \
160  } \
161  inline type AndSeqCst(type* p, type value) { \
162  return InterlockedAnd##suffix(reinterpret_cast<vctype*>(p), \
163  base::bit_cast<vctype>(value)); \
164  } \
165  inline type OrSeqCst(type* p, type value) { \
166  return InterlockedOr##suffix(reinterpret_cast<vctype*>(p), \
167  base::bit_cast<vctype>(value)); \
168  } \
169  inline type XorSeqCst(type* p, type value) { \
170  return InterlockedXor##suffix(reinterpret_cast<vctype*>(p), \
171  base::bit_cast<vctype>(value)); \
172  }

405 // This is https://tc39.github.io/ecma262/#sec-getmodifysetvalueinbuffer
406 // but also includes the ToInteger/ToBigInt conversion that's part of
407 // https://tc39.github.io/ecma262/#sec-atomicreadmodifywrite
408 template <template <typename> class Op>
409 Tagged<Object> GetModifySetValueInBuffer(RuntimeArguments args,
410  Isolate* isolate,
411  const char* method_name) {
412  HandleScope scope(isolate);
413  DCHECK_EQ(3, args.length());
414  Handle<JSTypedArray> sta = args.at<JSTypedArray>(0);
415  size_t index = NumberToSize(args[1]);
416  Handle<Object> value_obj = args.at(2);
417 
418  uint8_t* source =
419  static_cast<uint8_t*>(sta->GetBuffer(isolate)->backing_store()) +
420  sta->byte_offset();
421 
422  if (sta->type() >= kExternalBigInt64Array) {
423  Handle<BigInt> bigint;
424  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, bigint,
425  BigInt::FromObject(isolate, value_obj));
426 
427  THROW_ERROR_RETURN_FAILURE_ON_DETACHED_OR_OUT_OF_BOUNDS(isolate, sta, index,
428  method_name);
429 
430  CHECK_LT(index, sta->GetLength());
431  if (sta->type() == kExternalBigInt64Array) {
432  return Op<int64_t>::Do(isolate, source, index, bigint);
433  }
434  DCHECK(sta->type() == kExternalBigUint64Array);
435  return Op<uint64_t>::Do(isolate, source, index, bigint);
436  }
437 
438  Handle<Object> value;
439  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, value,
440  Object::ToInteger(isolate, value_obj));
441 
442  THROW_ERROR_RETURN_FAILURE_ON_DETACHED_OR_OUT_OF_BOUNDS(isolate, sta, index,
443  method_name);
444 
445  CHECK_LT(index, sta->GetLength());
446 
447  switch (sta->type()) {
448 #define TYPED_ARRAY_CASE(Type, typeName, TYPE, ctype) \
449  case kExternal##Type##Array: \
450  return Op<ctype>::Do(isolate, source, index, value);
451 
452  INTEGER_TYPED_ARRAYS(TYPED_ARRAY_CASE)
453 #undef TYPED_ARRAY_CASE
454 
455  default:
456  break;
457  }
458 
459  UNREACHABLE();
460 }

517 RUNTIME_FUNCTION(Runtime_AtomicsCompareExchange) {
518  HandleScope scope(isolate);
519  DCHECK_EQ(4, args.length());
520  Handle<JSTypedArray> sta = args.at<JSTypedArray>(0);
521  size_t index = NumberToSize(args[1]);
522  Handle<Object> old_value_obj = args.at(2);
523  Handle<Object> new_value_obj = args.at(3);
524 
525  uint8_t* source =
526  static_cast<uint8_t*>(sta->GetBuffer(isolate)->backing_store()) +
527  sta->byte_offset();
528 
529  if (sta->type() >= kExternalBigInt64Array) {
530  Handle<BigInt> old_bigint;
531  Handle<BigInt> new_bigint;
532  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
533  isolate, old_bigint, BigInt::FromObject(isolate, old_value_obj));
534  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
535  isolate, new_bigint, BigInt::FromObject(isolate, new_value_obj));
536 
537  THROW_ERROR_RETURN_FAILURE_ON_DETACHED_OR_OUT_OF_BOUNDS(
538  isolate, sta, index, "Atomics.compareExchange");
539 
540  CHECK_LT(index, sta->GetLength());
541  if (sta->type() == kExternalBigInt64Array) {
542  return DoCompareExchange<int64_t>(isolate, source, index, old_bigint,
543  new_bigint);
544  }
545  DCHECK(sta->type() == kExternalBigUint64Array);
546  return DoCompareExchange<uint64_t>(isolate, source, index, old_bigint,
547  new_bigint);
548  }
549 
550  Handle<Object> old_value;
551  Handle<Object> new_value;
552  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, old_value,
553  Object::ToInteger(isolate, old_value_obj));
554  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, new_value,
555  Object::ToInteger(isolate, new_value_obj));
556 
557  THROW_ERROR_RETURN_FAILURE_ON_DETACHED_OR_OUT_OF_BOUNDS(
558  isolate, sta, index, "Atomics.compareExchange");
559 
560  switch (sta->type()) {
561 #define TYPED_ARRAY_CASE(Type, typeName, TYPE, ctype) \
562  case kExternal##Type##Array: \
563  return DoCompareExchange<ctype>(isolate, source, index, old_value, \
564  new_value);
565 
566  INTEGER_TYPED_ARRAYS(TYPED_ARRAY_CASE)
567 #undef TYPED_ARRAY_CASE
568 
569  default:
570  break;
571  }
572 
573  UNREACHABLE();
574 }

648 template <typename WriteOperation>
649 Tagged<Object> AtomicFieldWrite(Isolate* isolate, DirectHandle<JSObject> object,
650  Handle<Name> field_name,
651  DirectHandle<Object> value,
652  WriteOperation write_operation) {
653  LookupIterator it(isolate, object, PropertyKey(isolate, field_name),
654  LookupIterator::OWN);
655  Maybe<bool> result = Nothing<bool>();
656  if (it.IsFound()) {
657  if (!it.IsReadOnly()) {
658  return write_operation(it);
659  }
660  // Shared structs and arrays are non-extensible and have non-configurable,
661  // writable, enumerable properties. The only exception is SharedArrays'
662  // "length" property, which is non-writable.
663  result = Object::WriteToReadOnlyProperty(&it, value, Just(kThrowOnError));
664  } else {
665  // Shared structs are non-extensible. Instead of duplicating logic, call
666  // Object::AddDataProperty to handle the error case.
667  result = Object::AddDataProperty(&it, value, NONE, Just(kThrowOnError),
668  StoreOrigin::kMaybeKeyed);
669  }
670  // Treat as strict code and always throw an error.
671  DCHECK(result.IsNothing());
672  USE(result);
673  return ReadOnlyRoots(isolate).exception();
674 }
