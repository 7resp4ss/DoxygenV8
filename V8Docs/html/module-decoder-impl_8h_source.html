64 // Reads a length-prefixed string, checking that it is within bounds. Returns
65 // the offset of the string, and the length as an out parameter.
66 inline WireBytesRef consume_string(Decoder* decoder,
67  unibrow::Utf8Variant grammar,
68  const char* name, ITracer* tracer) {
69  if (tracer) {
70  tracer->Description(name);
71  tracer->Description(" ");
72  }
73  uint32_t length = decoder->consume_u32v("length", tracer);
74  if (tracer) {
75  tracer->Description(": ");
76  tracer->Description(length);
77  tracer->NextLine();
78  }
79  uint32_t offset = decoder->pc_offset();
80  const uint8_t* string_start = decoder->pc();
81  // Consume bytes before validation to guarantee that the string is not oob.
82  if (length > 0) {
83  if (tracer) {
84  tracer->Bytes(decoder->pc(), length);
85  tracer->Description(name);
86  tracer->Description(": ");
87  tracer->Description(reinterpret_cast<const char*>(decoder->pc()), length);
88  tracer->NextLine();
89  }
90  decoder->consume_bytes(length, name);
91  if (decoder->ok()) {
92  switch (grammar) {
93  case unibrow::Utf8Variant::kLossyUtf8:
94  break;
95  case unibrow::Utf8Variant::kUtf8:
96  if (!unibrow::Utf8::ValidateEncoding(string_start, length)) {
97  decoder->errorf(string_start, "%s: no valid UTF-8 string", name);
98  }
99  break;
100  case unibrow::Utf8Variant::kWtf8:
101  if (!unibrow::Wtf8::ValidateEncoding(string_start, length)) {
102  decoder->errorf(string_start, "%s: no valid WTF-8 string", name);
103  }
104  break;
105  case unibrow::Utf8Variant::kUtf8NoTrap:
106  UNREACHABLE();
107  }
108  }
109  }
110  return {offset, decoder->failed() ? 0 : length};
111 }

124 inline SectionCode IdentifyUnknownSectionInternal(Decoder* decoder,
125  ITracer* tracer) {
126  WireBytesRef string = consume_utf8_string(decoder, "section name", tracer);
127  if (decoder->failed()) {
128  return kUnknownSectionCode;
129  }
130  const uint8_t* section_name_start =
131  decoder->start() + decoder->GetBufferRelativeOffset(string.offset());
132 
133  TRACE(" +%d section name : \"%.*s\"\n",
134  static_cast<int>(section_name_start - decoder->start()),
135  string.length() < 20 ? string.length() : 20, section_name_start);

138  static constexpr SpecialSectionPair kSpecialSections[]{
139  {base::StaticCharVector(kNameString), kNameSectionCode},
140  {base::StaticCharVector(kSourceMappingURLString),
141  kSourceMappingURLSectionCode},
142  {base::StaticCharVector(kInstTraceString), kInstTraceSectionCode},
143  {base::StaticCharVector(kCompilationHintsString),
144  kCompilationHintsSectionCode},
145  {base::StaticCharVector(kBranchHintsString), kBranchHintsSectionCode},
146  {base::StaticCharVector(kDebugInfoString), kDebugInfoSectionCode},
147  {base::StaticCharVector(kExternalDebugInfoString),
148  kExternalDebugInfoSectionCode},
149  {base::StaticCharVector(kBuildIdString), kBuildIdSectionCode},
150  {base::StaticCharVector(kDescriptorsString), kDescriptorsSectionCode}};
151 
152  auto name_vec = base::Vector<const char>::cast(
153  base::VectorOf(section_name_start, string.length()));

155  if (name_vec == special_section.first) return special_section.second;
156  }

158  return kUnknownSectionCode;

161 // An iterator over the sections in a wasm binary module.
162 // Automatically skips all unknown sections.
163 class WasmSectionIterator {
164  public:
165  explicit WasmSectionIterator(Decoder* decoder, ITracer* tracer)
166  : decoder_(decoder),
167  tracer_(tracer),
168  section_code_(kUnknownSectionCode),
169  section_start_(decoder->pc()),
170  section_end_(decoder->pc()) {
171  next();
172  }
173 
174  bool more() const { return decoder_->ok() && decoder_->more(); }
175 
176  SectionCode section_code() const { return section_code_; }
177 
178  const uint8_t* section_start() const { return section_start_; }
179 
180  uint32_t section_length() const {
181  return static_cast<uint32_t>(section_end_ - section_start_);
182  }
183 
184  base::Vector<const uint8_t> payload() const {
185  return {payload_start_, payload_length()};
186  }
187 
188  const uint8_t* payload_start() const { return payload_start_; }
189 
190  uint32_t payload_length() const {
191  return static_cast<uint32_t>(section_end_ - payload_start_);
192  }
193 
194  const uint8_t* section_end() const { return section_end_; }
195 
196  // Advances to the next section, checking that decoding the current section
197  // stopped at {section_end_}.
198  void advance(bool move_to_section_end = false) {
199  if (move_to_section_end && decoder_->pc() < section_end_) {
200  decoder_->consume_bytes(
201  static_cast<uint32_t>(section_end_ - decoder_->pc()));
202  }
203  if (decoder_->pc() != section_end_) {
204  const char* msg = decoder_->pc() < section_end_ ? "shorter" : "longer";
205  decoder_->errorf(decoder_->pc(),
206  "section was %s than expected size "
207  "(%u bytes expected, %zu decoded)",
208  msg, section_length(),
209  static_cast<size_t>(decoder_->pc() - section_start_));
210  }
211  next();
212  }
213 
214  private:
215  Decoder* decoder_;
216  ITracer* tracer_;
217  SectionCode section_code_;
218  const uint8_t* section_start_;
219  const uint8_t* payload_start_;
220  const uint8_t* section_end_;
221 
222  // Reads the section code/name at the current position and sets up
223  // the embedder fields.
224  void next() {
225  if (!decoder_->more()) {
226  section_code_ = kUnknownSectionCode;
227  return;
228  }
229  section_start_ = decoder_->pc();
230  // Empty line before next section.
231  if (tracer_) tracer_->NextLine();
232  uint8_t section_code = decoder_->consume_u8("section kind", tracer_);
233  if (tracer_) {
234  tracer_->Description(": ");
235  tracer_->Description(SectionName(static_cast<SectionCode>(section_code)));
236  tracer_->NextLine();
237  }
238  // Read and check the section size.
239  uint32_t section_length = decoder_->consume_u32v("section length", tracer_);
240  if (tracer_) {
241  tracer_->Description(section_length);
242  tracer_->NextLine();
243  }
244  payload_start_ = decoder_->pc();
245  section_end_ = payload_start_ + section_length;
246  if (section_length > decoder_->available_bytes()) {
247  decoder_->errorf(
248  section_start_,
249  "section (code %u, \"%s\") extends past end of the module "
250  "(length %u, remaining bytes %u)",
251  section_code, SectionName(static_cast<SectionCode>(section_code)),
252  section_length, decoder_->available_bytes());
253  section_end_ = payload_start_;
254  }
255 
256  if (section_code == kUnknownSectionCode) {
257  // Check for the known "name", "sourceMappingURL", or "compilationHints"
258  // section.
259  // To identify the unknown section we set the end of the decoder bytes to
260  // the end of the custom section, so that we do not read the section name
261  // beyond the end of the section.
262  const uint8_t* module_end = decoder_->end();
263  decoder_->set_end(section_end_);
264  section_code = IdentifyUnknownSectionInternal(decoder_, tracer_);
265  if (decoder_->ok()) decoder_->set_end(module_end);
266  // As a side effect, the above function will forward the decoder to after
267  // the identifier string.
268  payload_start_ = decoder_->pc();
269  } else if (!IsValidSectionCode(section_code)) {
270  decoder_->errorf(decoder_->pc(), "unknown section code #0x%02x",
271  section_code);
272  }
273  section_code_ = decoder_->failed() ? kUnknownSectionCode
274  : static_cast<SectionCode>(section_code);
275 
276  if (section_code_ == kUnknownSectionCode && section_end_ > decoder_->pc()) {
277  // Skip to the end of the unknown section.
278  uint32_t remaining = static_cast<uint32_t>(section_end_ - decoder_->pc());
279  decoder_->consume_bytes(remaining, "section payload", tracer_);
280  }
281  }

284 inline void DumpModule(const base::Vector<const uint8_t> module_bytes,
285  bool ok) {
286  std::string path;
287  if (v8_flags.dump_wasm_module_path) {
288  path = v8_flags.dump_wasm_module_path;
289  if (path.size() && !base::OS::isDirectorySeparator(path[path.size() - 1])) {
290  path += base::OS::DirectorySeparator();
291  }
292  }
293  // File are named `<hash>.{ok,failed}.wasm`.
294  // Limit the hash to 8 characters (32 bits).
295  uint32_t hash = static_cast<uint32_t>(GetWireBytesHash(module_bytes));
296  base::EmbeddedVector<char, 32> buf;
297  SNPrintF(buf, "%08x.%s.wasm", hash, ok ? "ok" : "failed");
298  path += buf.begin();
299  size_t rv = 0;
300  if (FILE* file = base::OS::FOpen(path.c_str(), "wb")) {
301  rv = fwrite(module_bytes.begin(), module_bytes.length(), 1, file);
302  base::Fclose(file);
303  }
304  if (rv != 1) {
305  OFStream os(stderr);
306  os << "Error while dumping wasm file to " << path << std::endl;
307  }

329  void DecodeModuleHeader(base::Vector<const uint8_t> bytes) {
330  if (failed()) return;
331  Reset(bytes);
332 
333  const uint8_t* pos = pc_;
334  uint32_t magic_word = consume_u32("wasm magic", tracer_);
335  if (tracer_) tracer_->NextLine();
336 #define BYTES(x) (x & 0xFF), (x >> 8) & 0xFF, (x >> 16) & 0xFF, (x >> 24) & 0xFF
337  if (magic_word != kWasmMagic) {
338  errorf(pos,
339  "expected magic word %02x %02x %02x %02x, "
340  "found %02x %02x %02x %02x",
341  BYTES(kWasmMagic), BYTES(magic_word));
342  }
343 
344  pos = pc_;
345  {
346  uint32_t magic_version = consume_u32("wasm version", tracer_);
347  if (tracer_) tracer_->NextLine();
348  if (magic_version != kWasmVersion) {
349  errorf(pos,
350  "expected version %02x %02x %02x %02x, "
351  "found %02x %02x %02x %02x",
352  BYTES(kWasmVersion), BYTES(magic_version));
353  }
354  }

358  bool CheckSectionOrder(SectionCode section_code) {
359  // Check the order of ordered sections.
360  if (section_code >= kFirstSectionInModule &&
361  section_code < kFirstUnorderedSection) {
362  if (section_code < next_ordered_section_) {
363  errorf(pc(), "unexpected section <%s>", SectionName(section_code));
364  return false;
365  }
366  next_ordered_section_ = section_code + 1;
367  return true;
368  }
369 
370  // Ignore ordering problems in unknown / custom sections. Even allow them to
371  // appear multiple times. As optional sections we use them on a "best
372  // effort" basis.
373  if (section_code == kUnknownSectionCode) return true;
374  if (section_code > kLastKnownModuleSection) return true;
375 
376  // The rest is standardized unordered sections; they are checked more
377  // thoroughly..
378  DCHECK_LE(kFirstUnorderedSection, section_code);
379  DCHECK_GE(kLastKnownModuleSection, section_code);
380 
381  // Check that unordered sections don't appear multiple times.
382  if (has_seen_unordered_section(section_code)) {
383  errorf(pc(), "Multiple %s sections not allowed",
384  SectionName(section_code));
385  return false;
386  }
387  set_seen_unordered_section(section_code);
388 
389  // Define a helper to ensure that sections <= {before} appear before the
390  // current unordered section, and everything >= {after} appears after it.
391  auto check_order = [this, section_code](SectionCode before,
392  SectionCode after) -> bool {
393  DCHECK_LT(before, after);
394  if (next_ordered_section_ > after) {
395  errorf(pc(), "The %s section must appear before the %s section",
396  SectionName(section_code), SectionName(after));
397  return false;
398  }
399  if (next_ordered_section_ <= before) next_ordered_section_ = before + 1;
400  return true;
401  };
402 
403  // Now check the ordering constraints of specific unordered sections.
404  switch (section_code) {
405  case kDataCountSectionCode:
406  return check_order(kElementSectionCode, kCodeSectionCode);
407  case kTagSectionCode:
408  return check_order(kMemorySectionCode, kGlobalSectionCode);
409  case kStringRefSectionCode:
410  // TODO(12868): If there's a tag section, assert that we're after the
411  // tag section.
412  return check_order(kMemorySectionCode, kGlobalSectionCode);
413  case kInstTraceSectionCode:
414  // Custom section following code.metadata tool convention containing
415  // offsets specifying where trace marks should be emitted.
416  // Be lenient with placement of instruction trace section. All except
417  // first occurrence after function section and before code section are
418  // ignored.
419  return true;
420  default:
421  return true;
422  }
423  }

425  void DecodeSection(SectionCode section_code,
426  base::Vector<const uint8_t> bytes, uint32_t offset) {
427  if (failed()) return;
428  Reset(bytes, offset);
429  TRACE("Section: %s\n", SectionName(section_code));
430  TRACE("Decode Section %p - %p\n", bytes.begin(), bytes.end());
431 
432  if (!CheckSectionOrder(section_code)) return;
433 
434  switch (section_code) {
435  case kUnknownSectionCode:
436  break;
437  case kTypeSectionCode:
438  DecodeTypeSection();
439  break;
440  case kImportSectionCode:
441  DecodeImportSection();
442  break;
443  case kFunctionSectionCode:
444  DecodeFunctionSection();
445  break;
446  case kTableSectionCode:
447  DecodeTableSection();
448  break;
449  case kMemorySectionCode:
450  DecodeMemorySection();
451  break;
452  case kGlobalSectionCode:
453  DecodeGlobalSection();
454  break;
455  case kExportSectionCode:
456  DecodeExportSection();
457  break;
458  case kStartSectionCode:
459  DecodeStartSection();
460  break;
461  case kCodeSectionCode:
462  DecodeCodeSection();
463  break;
464  case kElementSectionCode:
465  DecodeElementSection();
466  break;
467  case kDataSectionCode:
468  DecodeDataSection();
469  break;
470  case kNameSectionCode:
471  DecodeNameSection();
472  break;
473  case kSourceMappingURLSectionCode:
474  DecodeSourceMappingURLSection();
475  break;
476  case kDebugInfoSectionCode:
477  module_->debug_symbols[WasmDebugSymbols::Type::EmbeddedDWARF] = {
478  WasmDebugSymbols::Type::EmbeddedDWARF, {}};
479  consume_bytes(static_cast<uint32_t>(end_ - start_), ".debug_info");
480  break;
481  case kExternalDebugInfoSectionCode:
482  DecodeExternalDebugInfoSection();
483  break;
484  case kBuildIdSectionCode:
485  DecodeBuildIdSection();
486  break;
487  case kInstTraceSectionCode:
488  if (enabled_features_.has_instruction_tracing()) {
489  DecodeInstTraceSection();
490  } else {
491  // Ignore this section when feature is disabled. It is an optional
492  // custom section anyways.
493  consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
494  }
495  break;
496  case kCompilationHintsSectionCode:
497  // TODO(jkummerow): We're missing tracing support for well-known
498  // custom sections. This confuses `wami --full-hexdump` e.g.
499  // for the modules created by
500  // mjsunit/wasm/compilation-hints-streaming-compilation.js.
501  if (enabled_features_.has_compilation_hints()) {
502  DecodeCompilationHintsSection();
503  } else {
504  // Ignore this section when feature was disabled. It is an optional
505  // custom section anyways.
506  consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
507  }
508  break;
509  case kBranchHintsSectionCode:
510  if (enabled_features_.has_branch_hinting()) {
511  DecodeBranchHintsSection();
512  } else {
513  // Ignore this section when feature was disabled. It is an optional
514  // custom section anyways.
515  consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
516  }
517  break;
518  case kDescriptorsSectionCode:
519  DecodeDescriptorsSection();
520  break;
521  case kDataCountSectionCode:
522  DecodeDataCountSection();
523  break;
524  case kTagSectionCode:
525  DecodeTagSection();
526  break;
527  case kStringRefSectionCode:
528  if (enabled_features_.has_stringref()) {
529  DecodeStringRefSection();
530  } else {
531  errorf(pc(),
532  "unexpected section <%s> (enable with "
533  "--experimental-wasm-stringref)",
534  SectionName(section_code));
535  }
536  break;
537  default:
538  errorf(pc(), "unexpected section <%s>", SectionName(section_code));
539  return;
540  }
541 
542  if (pc() != bytes.end()) {
543  const char* msg = pc() < bytes.end() ? "shorter" : "longer";
544  errorf(pc(),
545  "section was %s than expected size "
546  "(%zu bytes expected, %zu decoded)",
547  msg, bytes.size(), static_cast<size_t>(pc() - bytes.begin()));
548  }
549  }

563  TypeDefinition consume_base_type_definition(bool is_descriptor,
564  bool is_shared) {
565  const bool is_final = true;
566  uint8_t kind = consume_u8(" kind", tracer_);
567  if (tracer_) {
568  tracer_->Description(": ");
569  tracer_->Description(TypeKindName(kind));
570  }
571  switch (kind) {
572  case kWasmFunctionTypeCode: {
573  const FunctionSig* sig = consume_sig(&module_->signature_zone);
574  if (sig == nullptr) {
575  CHECK(!ok());
576  return {};
577  }
578 
579  return {sig, kNoSuperType, is_final, is_shared};
580  }
581  case kWasmStructTypeCode: {
582  module_->is_wasm_gc = true;
583  const StructType* type =
584  consume_struct(&module_->signature_zone, is_descriptor, is_shared);
585  if (type == nullptr) {
586  CHECK(!ok());
587  return {};
588  }
589  return {type, kNoSuperType, is_final, is_shared};
590  }
591  case kWasmArrayTypeCode: {
592  module_->is_wasm_gc = true;
593  const ArrayType* type = consume_array(&module_->signature_zone);
594  if (type == nullptr) {
595  CHECK(!ok());
596  return {};
597  }
598  return {type, kNoSuperType, is_final, is_shared};
599  }
600  case kWasmContTypeCode: {
601  if (!enabled_features_.has_wasmfx()) {
602  error(pc() - 1,
603  "core stack switching not enabled (enable with "
604  "--experimental-wasm-wasmfx)");
605  return {};
606  }
607 
608  const uint8_t* pos = pc();
609  HeapType hp = consume_heap_type();
610 
611  if (!hp.is_index()) {
612  error(pos, "cont type must refer to a signature index");
613  return {};
614  }
615 
616  ContType* type = module_->signature_zone.New<ContType>(hp.ref_index());
617  return {type, kNoSuperType, is_final, is_shared};
618  }
619  default:
620  if (tracer_) tracer_->NextLine();
621  errorf(pc() - 1, "unknown type form: %d", kind);
622  return {};
623  }
624  }

626  TypeDefinition consume_described_type(bool is_descriptor, bool is_shared) {
627  uint8_t kind = read_u8<Decoder::FullValidationTag>(pc(), "type kind");
628  if (kind == kWasmDescriptorCode) {
629  if (!enabled_features_.has_custom_descriptors()) {
630  error(pc(),
631  "descriptor types need --experimental-wasm-custom-descriptors");
632  return {};
633  }
634  detected_features_->add_custom_descriptors();
635  consume_bytes(1, " described_by", tracer_);
636  const uint8_t* pos = pc();
637  uint32_t descriptor = consume_u32v("descriptor", tracer_);
638  if (descriptor >= module_->types.size()) {
639  errorf(pos, "descriptor type index %u is out of bounds", descriptor);
640  return {};
641  }
642  if (tracer_) tracer_->NextLine();
643  TypeDefinition type =
644  consume_base_type_definition(is_descriptor, is_shared);
645  if (type.kind != TypeDefinition::kStruct) {
646  error(pos - 1, "'descriptor' may only be used with structs");
647  return {};
648  }
649  type.descriptor = ModuleTypeIndex{descriptor};
650  return type;
651  } else {
652  return consume_base_type_definition(is_descriptor, is_shared);
653  }
654  }
655 
656  TypeDefinition consume_describing_type(size_t current_type_index,
657  bool is_shared) {
658  uint8_t kind = read_u8<Decoder::FullValidationTag>(pc(), "type kind");
659  if (kind == kWasmDescribesCode) {
660  if (!enabled_features_.has_custom_descriptors()) {
661  error(pc(),
662  "descriptor types need --experimental-wasm-custom-descriptors");
663  return {};
664  }
665  detected_features_->add_custom_descriptors();
666  consume_bytes(1, " describes", tracer_);
667  const uint8_t* pos = pc();
668  uint32_t describes = consume_u32v("describes", tracer_);
669  if (describes >= current_type_index) {
670  error(pos, "types can only describe previously-declared types");
671  return {};
672  }
673  if (tracer_) tracer_->NextLine();
674  TypeDefinition type = consume_described_type(true, is_shared);
675  if (type.kind != TypeDefinition::kStruct) {
676  error(pos - 1, "'describes' may only be used with structs");
677  return {};
678  }
679  type.describes = ModuleTypeIndex{describes};
680  return type;
681  } else {
682  return consume_described_type(false, is_shared);
683  }
684  }
685 
686  TypeDefinition consume_shared_type(size_t current_type_index) {
687  uint8_t kind = read_u8<Decoder::FullValidationTag>(pc(), "type kind");
688  if (kind == kSharedFlagCode) {
689  if (!enabled_features_.has_shared()) {
690  errorf(pc() - 1,
691  "unknown type form: %d, enable with --experimental-wasm-shared",
692  kind);
693  return {};
694  }
695  module_->has_shared_part = true;
696  consume_bytes(1, " shared", tracer_);
697  TypeDefinition type = consume_describing_type(current_type_index, true);
698  DCHECK(type.is_shared);
699  if (type.kind == TypeDefinition::kFunction ||
700  type.kind == TypeDefinition::kCont) {
701  // TODO(42204563): Support shared functions/continuations.
702  error(pc() - 1, "shared functions/continuations are not supported yet");
703  return {};
704  }
705  return type;
706  } else {
707  return consume_describing_type(current_type_index, false);
708  }
709  }
710 
711  // {current_type_index} is the index of the type that's being decoded.
712  // Any supertype must have a lower index.
713  TypeDefinition consume_subtype_definition(size_t current_type_index) {
714  uint8_t kind = read_u8<Decoder::FullValidationTag>(pc(), "type kind");
715  if (kind == kWasmSubtypeCode || kind == kWasmSubtypeFinalCode) {
716  module_->is_wasm_gc = true;
717  bool is_final = kind == kWasmSubtypeFinalCode;
718  consume_bytes(1, is_final ? " subtype final, " : " subtype extensible, ",
719  tracer_);
720  constexpr uint32_t kMaximumSupertypes = 1;
721  uint32_t supertype_count =
722  consume_count("supertype count", kMaximumSupertypes);
723  uint32_t supertype = kNoSuperType.index;
724  if (supertype_count == 1) {
725  supertype = consume_u32v("supertype", tracer_);
726  if (supertype >= current_type_index) {
727  errorf("type %u: invalid supertype %u", current_type_index,
728  supertype);
729  return {};
730  }
731  if (tracer_) {
732  tracer_->Description(supertype);
733  tracer_->NextLine();
734  }
735  }
736  TypeDefinition type = consume_shared_type(current_type_index);
737  type.supertype = ModuleTypeIndex{supertype};
738  type.is_final = is_final;
739  return type;
740  } else {
741  return consume_shared_type(current_type_index);
742  }
743  }

745  void DecodeTypeSection() {
746  TypeCanonicalizer* type_canon = GetTypeCanonicalizer();
747  uint32_t types_count = consume_count("types count", kV8MaxWasmTypes);
748 
749  for (uint32_t i = 0; ok() && i < types_count; ++i) {
750  TRACE("DecodeType[%d] module+%d\n", i, static_cast<int>(pc_ - start_));
751  uint8_t kind = read_u8<Decoder::FullValidationTag>(pc(), "type kind");
752  size_t initial_size = module_->types.size();
753  uint32_t group_size = 1;
754  if (kind == kWasmRecursiveTypeGroupCode) {
755  module_->is_wasm_gc = true;
756  uint32_t rec_group_offset = pc_offset();
757  consume_bytes(1, "rec. group definition", tracer_);
758  if (tracer_) tracer_->NextLine();
759  group_size = consume_count("recursive group size", kV8MaxWasmTypes);
760  if (tracer_) tracer_->RecGroupOffset(rec_group_offset, group_size);
761  }
762  if (initial_size + group_size > kV8MaxWasmTypes) {
763  errorf(pc(), "Type definition count exceeds maximum %zu",
764  kV8MaxWasmTypes);
765  return;
766  }
767  // We need to resize types before decoding the type definitions in this
768  // group, so that the correct type size is visible to type definitions.
769  module_->types.resize(initial_size + group_size);
770  for (uint32_t j = 0; j < group_size; j++) {
771  if (tracer_) tracer_->TypeOffset(pc_offset());
772  TypeDefinition type = consume_subtype_definition(initial_size + j);
773  if (failed()) return;
774  module_->types[initial_size + j] = type;
775  }
776  FinalizeRecgroup(group_size, type_canon);
777  if (kind == kWasmRecursiveTypeGroupCode && tracer_) {
778  tracer_->Description("end of rec. group");
779  tracer_->NextLine();
780  }
781  }

784  void FinalizeRecgroup(uint32_t group_size, TypeCanonicalizer* type_canon) {
785  // Now that we have decoded the entire recgroup, check its validity,
786  // initialize additional data, and canonicalize it:
787  // - check supertype validity
788  // - propagate subtyping depths
789  // - validate is_shared bits and set up RefTypeKind fields
790  WasmModule* module = module_.get();
791  uint32_t end_index = static_cast<uint32_t>(module->types.size());
792  uint32_t start_index = end_index - group_size;
793  for (uint32_t i = start_index; ok() && i < end_index; ++i) {
794  TypeDefinition& type_def = module_->types[i];
795  bool is_shared = type_def.is_shared;
796  switch (type_def.kind) {
797  case TypeDefinition::kFunction: {
798  base::Vector<const ValueType> all = type_def.function_sig->all();
799  size_t count = all.size();
800  ValueType* storage = const_cast<ValueType*>(all.begin());
801  for (uint32_t j = 0; j < count; j++) {
802  value_type_reader::Populate(&storage[j], module);
803  ValueType type = storage[j];
804  if (is_shared && !type.is_shared()) {
805  DCHECK(enabled_features_.has_shared());
806  uint32_t retcount =
807  static_cast<uint32_t>(type_def.function_sig->return_count());
808  const char* param_or_return =
809  j < retcount ? "return" : "parameter";
810  uint32_t index = j < retcount ? j : j - retcount;
811  // {pc_} isn't very accurate, it's pointing at the end of the
812  // recgroup. So to ease debugging, we print the type's index.
813  errorf(pc_,
814  "Type %u: shared signature must have shared %s types, "
815  "actual type for %s %d is %s",
816  i, param_or_return, param_or_return, index,
817  type.name().c_str());
818  return;
819  }
820  }
821  break;
822  }
823  case TypeDefinition::kStruct: {
824  size_t count = type_def.struct_type->field_count();
825  ValueType* storage =
826  const_cast<ValueType*>(type_def.struct_type->fields().begin());
827  for (uint32_t j = 0; j < count; j++) {
828  value_type_reader::Populate(&storage[j], module);
829  ValueType type = storage[j];
830  if (is_shared && !type.is_shared()) {
831  errorf(pc_,
832  "Type %u: shared struct must have shared field types, "
833  "actual type for field %d is %s",
834  i, j, type.name().c_str());
835  return;
836  }
837  }
838  if (type_def.descriptor.valid()) {
839  const TypeDefinition& descriptor =
840  module->type(type_def.descriptor);
841  if (descriptor.describes.index != i) {
842  uint32_t d = type_def.descriptor.index;
843  errorf(pc_,
844  "Type %u has descriptor %u but %u doesn't describe %u", i,
845  d, d, i);
846  return;
847  }
848  if (descriptor.is_shared != type_def.is_shared) {
849  errorf(pc_,
850  "Type %u and its descriptor %u must have same sharedness",
851  i, type_def.descriptor.index);
852  return;
853  }
854  }
855  if (type_def.describes.valid()) {
856  if (module->type(type_def.describes).descriptor.index != i) {
857  uint32_t d = type_def.describes.index;
858  errorf(pc_,
859  "Type %u describes %u but %u isn't a descriptor for %u", i,
860  d, d, i);
861  return;
862  }
863  }
864  break;
865  }
866  case TypeDefinition::kArray: {
867  value_type_reader::Populate(
868  const_cast<ArrayType*>(type_def.array_type)
869  ->element_type_writable_ptr(),
870  module);
871  ValueType type = type_def.array_type->element_type();
872  if (is_shared && !type.is_shared()) {
873  errorf(pc_,
874  "Type %u: shared array must have shared element type, "
875  "actual element type is %s",
876  i, type.name().c_str());
877  return;
878  }
879  break;
880  }
881  case TypeDefinition::kCont: {
882  ModuleTypeIndex contfun_typeid =
883  type_def.cont_type->contfun_typeindex();
884  const TypeDefinition contfun_type =
885  module_->types[contfun_typeid.index];
886  if (contfun_type.kind != TypeDefinition::kFunction) {
887  errorf(pc_,
888  "Type %u: cont type must refer to a signature index, "
889  "actual type is %s",
890  i, module_->heap_type(contfun_typeid).name().c_str());
891  return;
892  }
893  if (is_shared && !contfun_type.is_shared) {
894  errorf(pc_,
895  "Type %u: shared cont type must refer to a shared signature,"
896  " actual type is %s",
897  module_->heap_type(contfun_typeid).name().c_str());
898  return;
899  }
900  break;
901  }
902  }
903  }
904  module->isorecursive_canonical_type_ids.resize(end_index);
905  type_canon->AddRecursiveGroup(module, group_size);
906  for (uint32_t i = start_index; ok() && i < end_index; ++i) {
907  TypeDefinition& type_def = module_->types[i];
908  ModuleTypeIndex explicit_super = type_def.supertype;
909  if (!explicit_super.valid()) continue; // No supertype.
910  DCHECK_LT(explicit_super.index, i); // Checked during decoding.
911  uint32_t depth = module->type(explicit_super).subtyping_depth + 1;
912  type_def.subtyping_depth = depth;
913  DCHECK_GE(depth, 0);
914  if (depth > kV8MaxRttSubtypingDepth) {
915  errorf("type %u: subtyping depth is greater than allowed", i);
916  return;
917  }
918  // This check is technically redundant; we include for the improved error
919  // message.
920  if (module->type(explicit_super).is_final) {
921  errorf("type %u extends final type %u", i, explicit_super.index);
922  return;
923  }
924  if (!ValidSubtypeDefinition(ModuleTypeIndex{i}, explicit_super, module)) {
925  errorf("type %u has invalid explicit supertype %u", i,
926  explicit_super.index);
927  return;
928  }
929  }
930  }

932  void DecodeImportSection() {
933  uint32_t import_table_count =
934  consume_count("imports count", kV8MaxWasmImports);
935  module_->import_table.reserve(import_table_count);
936  for (uint32_t i = 0; ok() && i < import_table_count; ++i) {
937  TRACE("DecodeImportTable[%d] module+%d\n", i,
938  static_cast<int>(pc_ - start_));
939  if (tracer_) tracer_->ImportOffset(pc_offset());
940 
941  const uint8_t* pos = pc_;
942  WireBytesRef module_name =
943  consume_utf8_string(this, "module name", tracer_);
944  WireBytesRef field_name =
945  consume_utf8_string(this, "field name", tracer_);
946  ImportExportKindCode kind =
947  static_cast<ImportExportKindCode>(consume_u8("kind", tracer_));
948  if (tracer_) {
949  tracer_->Description(": ");
950  tracer_->Description(ExternalKindName(kind));
951  }
952  module_->import_table.push_back(WasmImport{
953  .module_name = module_name, .field_name = field_name, .kind = kind});
954  WasmImport* import = &module_->import_table.back();
955  switch (kind) {
956  case kExternalFunction: {
957  // ===== Imported function ===========================================
958  import->index = static_cast<uint32_t>(module_->functions.size());
959  module_->num_imported_functions++;
960  module_->functions.push_back(WasmFunction{
961  .func_index = import->index,
962  .imported = true,
963  });
964  WasmFunction* function = &module_->functions.back();
965  function->sig_index =
966  consume_sig_index(module_.get(), &function->sig);
967  break;
968  }
969  case kExternalTable: {
970  // ===== Imported table ==============================================
971  import->index = static_cast<uint32_t>(module_->tables.size());
972  const uint8_t* type_position = pc();
973  ValueType type = consume_value_type(module_.get());
974  if (!type.is_object_reference()) {
975  errorf(type_position, "Invalid table type %s", type.name().c_str());
976  break;
977  }
978  module_->num_imported_tables++;
979  module_->tables.push_back(WasmTable{
980  .type = type,
981  .imported = true,
982  });
983  WasmTable* table = &module_->tables.back();
984  consume_table_flags(table);
985  DCHECK_IMPLIES(table->shared,
986  enabled_features_.has_shared() || !ok());
987  if (table->shared && enabled_features_.has_shared()) {
988  module_->has_shared_part = true;
989  if (!type.is_shared()) {
990  errorf(type_position,
991  "Shared table %i must have shared element type, actual "
992  "type %s",
993  i, type.name().c_str());
994  break;
995  }
996  }
997  // Note that we should not throw an error if the declared maximum size
998  // is oob. We will instead fail when growing at runtime.
999  uint64_t kNoMaximum = kMaxUInt64;
1000  consume_resizable_limits(
1001  "table", "elements", wasm::max_table_size(), &table->initial_size,
1002  table->has_maximum_size, kNoMaximum, &table->maximum_size,
1003  table->is_table64() ? k64BitLimits : k32BitLimits);
1004  break;
1005  }
1006  case kExternalMemory: {
1007  // ===== Imported memory =============================================
1008  static_assert(kV8MaxWasmMemories <= kMaxUInt32);
1009  if (module_->memories.size() >= kV8MaxWasmMemories - 1) {
1010  errorf("At most %u imported memories are supported",
1011  kV8MaxWasmMemories);
1012  break;
1013  }
1014  uint32_t mem_index = static_cast<uint32_t>(module_->memories.size());
1015  import->index = mem_index;
1016  module_->memories.emplace_back();
1017  WasmMemory* external_memory = &module_->memories.back();
1018  external_memory->imported = true;
1019  external_memory->index = mem_index;
1020 
1021  consume_memory_flags(external_memory);
1022  uint32_t max_pages = external_memory->is_memory64()
1023  ? kSpecMaxMemory64Pages
1024  : kSpecMaxMemory32Pages;
1025  consume_resizable_limits(
1026  "memory", "pages", max_pages, &external_memory->initial_pages,
1027  external_memory->has_maximum_pages, max_pages,
1028  &external_memory->maximum_pages,
1029  external_memory->is_memory64() ? k64BitLimits : k32BitLimits);
1030  break;
1031  }
1032  case kExternalGlobal: {
1033  // ===== Imported global =============================================
1034  import->index = static_cast<uint32_t>(module_->globals.size());
1035  ValueType type = consume_value_type(module_.get());
1036  auto [mutability, shared] = consume_global_flags();
1037  if (V8_UNLIKELY(failed())) break;
1038  if (V8_UNLIKELY(shared && !type.is_shared())) {
1039  error("shared imported global must have shared type");
1040  break;
1041  }
1042  module_->globals.push_back(
1043  WasmGlobal{.type = type,
1044  .mutability = mutability,
1045  .index = 0, // set later in CalculateGlobalOffsets
1046  .shared = shared,
1047  .imported = true});
1048  module_->num_imported_globals++;
1049  DCHECK_EQ(module_->globals.size(), module_->num_imported_globals);
1050  if (shared) module_->has_shared_part = true;
1051  if (mutability) module_->num_imported_mutable_globals++;
1052  if (tracer_) tracer_->NextLine();
1053  break;
1054  }
1055  case kExternalTag: {
1056  // ===== Imported tag ================================================
1057  import->index = static_cast<uint32_t>(module_->tags.size());
1058  module_->num_imported_tags++;
1059  const WasmTagSig* tag_sig = nullptr;
1060  consume_exception_attribute(); // Attribute ignored for now.
1061  ModuleTypeIndex sig_index =
1062  consume_tag_sig_index(module_.get(), &tag_sig);
1063  module_->tags.emplace_back(tag_sig, sig_index);
1064  break;
1065  }
1066  default:
1067  errorf(pos, "unknown import kind 0x%02x", kind);
1068  break;
1069  }
1070  }
1071  if (module_->memories.size() > 1) {
1072  detected_features_->add_multi_memory();
1073  if (v8_flags.wasm_jitless) {
1074  error("Multiple memories not supported in Wasm jitless mode");
1075  }
1076  }
1077  UpdateComputedMemoryInformation();
1078  module_->type_feedback.well_known_imports.Initialize(
1079  module_->num_imported_functions);
1080  if (tracer_) tracer_->ImportsDone(module_.get());
1081  }

1083  void DecodeFunctionSection() {
1084  uint32_t functions_count =
1085  consume_count("functions count", v8_flags.max_wasm_functions);
1086  DCHECK_EQ(module_->functions.size(), module_->num_imported_functions);
1087  uint32_t total_function_count =
1088  module_->num_imported_functions + functions_count;

1090  module_->num_declared_functions = functions_count;

1100  0xff);
1101  }
1102 
1103  for (uint32_t func_index = module_->num_imported_functions;
1104  func_index < total_function_count; ++func_index) {
1105  WasmFunction* function = &module_->functions[func_index];

1110  }
1111  }

1165  consume_resizable_limits(
1166  "table", "elements", wasm::max_table_size(), &table->initial_size,
1167  table->has_maximum_size, kNoMaximum, &table->maximum_size,
1168  table->is_table64() ? k64BitLimits : k32BitLimits);
1169 
1170  if (has_initializer) {
1171  table->initial_value =
1172  consume_init_expr(module_.get(), table_type, table->shared);
1173  }
1174  }

1177  void DecodeMemorySection() {
1178  const uint8_t* mem_count_pc = pc();
1179  static_assert(kV8MaxWasmMemories <= kMaxUInt32);
1180  // Use {kV8MaxWasmMemories} here, but only allow for >1 memory if
1181  // multi-memory is enabled (checked below). This allows for better error
1182  // messages.
1183  uint32_t memory_count = consume_count("memory count", kV8MaxWasmMemories);
1184  size_t imported_memories = module_->memories.size();
1185  DCHECK_GE(kV8MaxWasmMemories, imported_memories);
1186  if (memory_count > kV8MaxWasmMemories - imported_memories) {
1187  errorf(mem_count_pc,
1188  "Exceeding maximum number of memories (%u; declared %u, "
1189  "imported %zu)",
1190  kV8MaxWasmMemories, memory_count, imported_memories);
1191  }
1192  module_->memories.resize(imported_memories + memory_count);
1193 
1194  for (uint32_t i = 0; ok() && i < memory_count; i++) {
1195  WasmMemory* memory = module_->memories.data() + imported_memories + i;
1196  memory->index = static_cast<uint32_t>(imported_memories + i);
1197  if (tracer_) tracer_->MemoryOffset(pc_offset());
1198  consume_memory_flags(memory);
1199  uint32_t max_pages =
1200  memory->is_memory64() ? kSpecMaxMemory64Pages : kSpecMaxMemory32Pages;
1201  consume_resizable_limits(
1202  "memory", "pages", max_pages, &memory->initial_pages,
1203  memory->has_maximum_pages, max_pages, &memory->maximum_pages,
1204  memory->is_memory64() ? k64BitLimits : k32BitLimits);
1205  }
1206  if (module_->memories.size() > 1) {

1403  void DecodeCodeSection() {
1404  // Make sure global offset were calculated before they get accessed during
1405  // function compilation.
1406  CalculateGlobalOffsets(module_.get());
1407  uint32_t code_section_start = pc_offset();
1408  uint32_t functions_count = consume_u32v("functions count", tracer_);
1409  if (tracer_) {
1410  tracer_->Description(functions_count);
1411  tracer_->NextLine();
1412  }
1413  CheckFunctionsCount(functions_count, code_section_start);
1414 
1415  auto inst_traces_it = this->inst_traces_.begin();
1416  std::vector<std::pair<uint32_t, uint32_t>> inst_traces;
1417 
1418  for (uint32_t i = 0; ok() && i < functions_count; ++i) {
1419  int function_index = module_->num_imported_functions + i;
1420  if (tracer_) {
1421  tracer_->Description("function #");
1422  tracer_->FunctionName(function_index);
1423  tracer_->NextLine();
1424  }
1425  const uint8_t* pos = pc();
1426  uint32_t size = consume_u32v("body size", tracer_);
1427  if (tracer_) {
1428  tracer_->Description(size);
1429  tracer_->NextLine();
1430  }
1431  if (size > kV8MaxWasmFunctionSize) {
1432  errorf(pos, "size %u > maximum function size %zu", size,
1433  kV8MaxWasmFunctionSize);
1434  return;
1435  }
1436  uint32_t offset = pc_offset();
1437  consume_bytes(size, "function body");
1438  if (failed()) break;
1439  DecodeFunctionBody(function_index, size, offset);
1440 
1441  // Now that the function has been decoded, we can compute module offsets.
1442  for (; inst_traces_it != this->inst_traces_.end() &&
1443  std::get<0>(*inst_traces_it) == i;
1444  ++inst_traces_it) {
1445  uint32_t trace_offset = offset + std::get<1>(*inst_traces_it);
1446  uint32_t mark_id = std::get<2>(*inst_traces_it);
1447  std::pair<uint32_t, uint32_t> trace_mark = {trace_offset, mark_id};
1448  inst_traces.push_back(trace_mark);
1449  }
1450  }
1451  // If we have actually decoded traces and they were all decoded without
1452  // error, then we can move them to the module. If any errors are found, it
1453  // is safe to throw away all traces.
1454  if (V8_UNLIKELY(!inst_traces.empty() &&
1455  inst_traces_it == this->inst_traces_.end())) {
1456  // This adds an invalid entry at the end of the traces. An invalid entry
1457  // is defined as having an module offset of 0 and a markid of 0.
1458  inst_traces.push_back({0, 0});
1459  this->module_->inst_traces = std::move(inst_traces);
1460  }
1461  DCHECK_GE(pc_offset(), code_section_start);
1462  module_->code = {code_section_start, pc_offset() - code_section_start};
1463  }

1642

1646  for (uint32_t i = 0; i < func_count; i++) {

1670  break;
1671  }

1680  inst_traces.push_back(mark_tuple);

1689  if (inner.ok()) {
1690  this->inst_traces_ = std::move(inst_traces);
1691  }
1692  }
1693

1695  consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
1696  }

1698  void DecodeCompilationHintsSection() {
1699  TRACE("DecodeCompilationHints module+%d\n", static_cast<int>(pc_ - start_));
1700 
1701  // TODO(frgossen): Find a way to report compilation hint errors as warnings.
1702  // All except first occurrence after function section and before code
1703  // section are ignored.
1704  const bool before_function_section =
1705  next_ordered_section_ <= kFunctionSectionCode;
1706  const bool after_code_section = next_ordered_section_ > kCodeSectionCode;
1707  if (before_function_section || after_code_section ||
1708  has_seen_unordered_section(kCompilationHintsSectionCode)) {
1709  return;
1710  }
1711  set_seen_unordered_section(kCompilationHintsSectionCode);
1712 
1713  // TODO(frgossen) Propagate errors to outer decoder in experimental phase.
1714  // We should use an inner decoder later and propagate its errors as
1715  // warnings.
1716  Decoder& decoder = *this;
1717  // Decoder decoder(start_, pc_, end_, buffer_offset_);
1718 
1719  // Ensure exactly one compilation hint per function.
1720  uint32_t hint_count = decoder.consume_u32v("compilation hint count");
1721  if (hint_count != module_->num_declared_functions) {
1722  decoder.errorf(decoder.pc(), "Expected %u compilation hints (%u found)",
1723  module_->num_declared_functions, hint_count);
1724  }
1725 
1726  // Decode sequence of compilation hints.
1727  if (decoder.ok()) {
1728  module_->compilation_hints.reserve(hint_count);
1729  }
1730  for (uint32_t i = 0; decoder.ok() && i < hint_count; i++) {
1731  TRACE("DecodeCompilationHints[%d] module+%d\n", i,
1732  static_cast<int>(pc_ - start_));
1733 
1734  // Compilation hints are encoded in one byte each.
1735  // +-------+----------+---------------+----------+
1736  // | 2 bit | 2 bit | 2 bit | 2 bit |
1737  // | ... | Top tier | Baseline tier | Strategy |
1738  // +-------+----------+---------------+----------+
1739  uint8_t hint_byte = decoder.consume_u8("compilation hint");
1740  if (!decoder.ok()) break;
1741 
1742  // Validate the hint_byte.
1743  // For the compilation strategy, all 2-bit values are valid. For the tier,
1744  // only 0x0, 0x1, and 0x2 are allowed.
1745  static_assert(
1746  static_cast<int>(WasmCompilationHintTier::kDefault) == 0 &&
1747  static_cast<int>(WasmCompilationHintTier::kBaseline) == 1 &&
1748  static_cast<int>(WasmCompilationHintTier::kOptimized) == 2,
1749  "The check below assumes that 0x03 is the only invalid 2-bit number "
1750  "for a compilation tier");
1751  if (((hint_byte >> 2) & 0x03) == 0x03 ||
1752  ((hint_byte >> 4) & 0x03) == 0x03) {
1753  decoder.errorf(decoder.pc(),

1802 
1803  uint32_t func_count = inner.consume_u32v("number of functions");

1806  for (uint32_t i = 0; i < func_count; i++) {
1807  uint32_t func_idx = inner.consume_u32v("function index");

1810  break;
1811  }

1816  static_cast<int>(inner.pc() - inner.start()));
1817  // Keep track of the previous branch offset to validate the ordering
1818  int64_t last_br_off = -1;
1819  for (uint32_t j = 0; j < num_hints; ++j) {
1820  uint32_t br_off = inner.consume_u32v("branch instruction offset");
1821  if (int64_t{br_off} <= last_br_off) {
1822  inner.errorf("Invalid branch offset: %d", br_off);
1823  break;
1824  }
1825  last_br_off = br_off;
1826  uint32_t data_size = inner.consume_u32v("data size");
1827  if (data_size != 1) {
1828  inner.errorf("Invalid data size: %#x. Expected 1.", data_size);
1829  break;
1830  }
1831  uint32_t br_dir = inner.consume_u8("branch direction");
1832  TRACE("DecodeBranchHints[%d][%d] module+%d\n", func_idx, br_off,
1833  static_cast<int>(inner.pc() - inner.start()));
1834  BranchHint hint;
1835  switch (br_dir) {
1836  case 0:
1837  hint = BranchHint::kFalse;
1838  break;
1839  case 1:
1840  hint = BranchHint::kTrue;
1841  break;
1842  default:
1843  hint = BranchHint::kNone;
1844  inner.errorf(inner.pc(), "Invalid branch hint %#x", br_dir);
1845  break;
1846  }
1847  if (!inner.ok()) {
1848  break;
1849  }
1850  func_branch_hints.insert(br_off, hint);
1851  }
1852  if (!inner.ok()) {
1853  break;
1854  }

1863  if (inner.ok()) {
1864  module_->branch_hints = std::move(branch_hints);

1866  }

2067  // function should be called after all globals have been defined, which is
2068  // after the import section and the global section, but before the global

2073  if (module->globals.empty() || module->untagged_globals_buffer_size != 0 ||

2346  bool is_shared,

2349 #define TYPE_CHECK(found) \

2351  errorf(pc() + 1, \
2352  "type error in constant expression[0] (expected %s, got %s)", \

2359  // expressions, we replicate their decoding and validation here. The
2360  // manually handled cases correspond to {ConstantExpression}'s kinds.
2361  // We need to make sure to check that the expression ends in {kExprEnd};
2362  // otherwise, it is just the first operand of a composite expression, and we
2363  // fall back to the default case.
2364  if (!more()) {
2365  error("Beyond end of code");
2366  return {};

2376  tracer_->InitializerExpression(pc_, pc_ + length + 2, kWasmI32);
2377  }
2378  consume_bytes(length + 2);
2379  return ConstantExpression::I32Const(value);
2380  }
2381  break;
2382  }
2383  case kExprRefFunc: {

2385  read_u32v<FullValidationTag>(pc() + 1, "ref.func");
2386  if (V8_UNLIKELY(failed())) return {};

2389  errorf(pc() + 1, "function index %u out of bounds", index);
2390  return {};

2393  bool functype_is_shared = module->type(functype).is_shared;

2395  RefTypeKind::kFunction)

2404  tracer_->InitializerExpression(pc_, pc_ + length + 2, type);
2405  }
2406  consume_bytes(length + 2);

2408  }
2409  break;
2410  }
2411  case kExprRefNull: {
2412  auto [type, length] =
2413  value_type_reader::read_heap_type<FullValidationTag>(
2414  this, pc() + 1, enabled_features_, detected_features_);

2416  module, type);

2419  if (V8_LIKELY(lookahead(1 + length, kExprEnd))) {
2420  TYPE_CHECK(ValueType::RefNull(type))
2421  if (V8_UNLIKELY(is_shared && !type.is_shared())) {
2422  error(pc(), "ref.null does not have a shared type");
2423  return {};
2424  }

2426  tracer_->InitializerExpression(pc_, pc_ + length + 2,
2427  ValueType::RefNull(type));
2428  }

2430  return ConstantExpression::RefNull(type);
2431  }

2433  }

2435  break;
2436  }

2438 
2439  auto sig = FixedSizeSignature<ValueType>::Returns(expected);
2440  FunctionBody body(&sig, this->pc_offset(), pc_, end_, is_shared);
2441  WasmDetectedFeatures detected;
2442  ConstantExpression result;
2443  {
2444  // We need a scope for the decoder because its destructor resets some Zone
2445  // elements, which has to be done before we reset the Zone afterwards.
2446  WasmFullDecoder<Decoder::FullValidationTag, ConstantExpressionInterface,
2447  kConstantExpression>
2448  decoder(&init_expr_zone_, module, enabled_features_, &detected, body,
2449  module);
2450

2452

2454 
2455  if (tracer_) {
2456  // In case of error, decoder.end() is set to the position right before
2457  // the byte(s) that caused the error. For debugging purposes, we should
2458  // print these bytes, but we don't know how many of them there are, so
2459  // for now we have to guess. For more accurate behavior, we'd have to
2460  // pass {num_invalid_bytes} to every {decoder->DecodeError()} call.
2461  static constexpr size_t kInvalidBytesGuess = 4;
2462  const uint8_t* end =
2463  decoder.ok() ? decoder.end()
2464  : std::min(decoder.end() + kInvalidBytesGuess, end_);

2636  WasmElemSegment consume_element_segment_header() {
2637  const uint8_t* pos = pc();

2640  // active or not (0 is active).
2641  constexpr uint8_t kNonActiveMask = 1 << 0;
2642  // The mask for the bit in the flag which indicates:
2643  // - for active tables, if the segment has an explicit table index field.
2644  // - for non-active tables, whether the table is declarative (vs. passive).
2645  constexpr uint8_t kHasTableIndexOrIsDeclarativeMask = 1 << 1;
2646  // The mask for the bit in the flag which indicates if the functions of this
2647  // segment are defined as function indices (0) or constant expressions (1).
2648  constexpr uint8_t kExpressionsAsElementsMask = 1 << 2;
2649  // The mask for the bit which denotes whether this segment is shared.
2650  constexpr uint8_t kSharedFlag = 1 << 3;
2651  constexpr uint8_t kFullMask = kNonActiveMask |
2652  kHasTableIndexOrIsDeclarativeMask |
2653  kExpressionsAsElementsMask | kSharedFlag;
2654

2656  if ((flag & kFullMask) != flag) {
2657  errorf(pos, "illegal flag value %u", flag);
2658  return {};
2659  }
2660

2662  if (is_shared && !enabled_features_.has_shared()) {
2663  errorf(pos, "illegal flag value %u", flag);
2664  return {};

2668  error(pos, "shared element segments are not supported yet.");
2669  return {};

2671

2675  (flag & kNonActiveMask) ? (flag & kHasTableIndexOrIsDeclarativeMask)

2677  : WasmElemSegment::kStatusPassive
2678  : WasmElemSegment::kStatusActive;

2680  if (tracer_) {

2683  : status == WasmElemSegment::kStatusPassive
2684  ? "passive,"
2685  : "declarative,");

2689  flag & kExpressionsAsElementsMask
2690  ? WasmElemSegment::kExpressionElements
2691  : WasmElemSegment::kFunctionIndexElements;

2694  is_active && (flag & kHasTableIndexOrIsDeclarativeMask);

2702  // always have the implicit table index 0.

2704  has_table_index ? "" : " implicit", table_index);
2705  return {};
2706  }

2709  is_active ? module_->tables[table_index].type : kWasmBottom;

2715  tracer_->NextLine();

2720  is_shared);

2722  if (failed()) return {};
2723  }

2727  is_active && !(flag & kHasTableIndexOrIsDeclarativeMask);

2729  if (element_type == WasmElemSegment::kExpressionElements) {
