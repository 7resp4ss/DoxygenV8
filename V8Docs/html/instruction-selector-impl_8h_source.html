18 struct CaseInfo {
19  int32_t value; // The case value.
20  int32_t order; // The order for lowering to comparisons (less means earlier).
21  turboshaft::Block*
22  branch; // The basic blocks corresponding to the case value.
23 };

25 inline bool operator<(const CaseInfo& l, const CaseInfo& r) {
26  return l.order < r.order;

29 // Helper struct containing data about a table or lookup switch.
30 class SwitchInfo {
31  public:
32  SwitchInfo(ZoneVector<CaseInfo> const& cases, int32_t min_value,
33  int32_t max_value, turboshaft::Block* default_branch)
34  : cases_(cases),
35  min_value_(min_value),
36  max_value_(max_value),
37  default_branch_(default_branch) {
38  if (cases.size() != 0) {
39  DCHECK_LE(min_value, max_value);
40  // Note that {value_range} can be 0 if {min_value} is -2^31 and
41  // {max_value} is 2^31-1, so don't assume that it's non-zero below.
42  value_range_ = 1u + base::bit_cast<uint32_t>(max_value) -
43  base::bit_cast<uint32_t>(min_value);
44  } else {
45  value_range_ = 0;
46  }
47  }
48 
49  std::vector<CaseInfo> CasesSortedByValue() const {
50  std::vector<CaseInfo> result(cases_.begin(), cases_.end());
51  std::stable_sort(result.begin(), result.end(),
52  [](CaseInfo a, CaseInfo b) { return a.value < b.value; });
53  return result;
54  }
55  const ZoneVector<CaseInfo>& CasesUnsorted() const { return cases_; }
56  int32_t min_value() const { return min_value_; }
57  int32_t max_value() const { return max_value_; }
58  size_t value_range() const { return value_range_; }
59  size_t case_count() const { return cases_.size(); }
60  turboshaft::Block* default_branch() const { return default_branch_; }
61 
62  private:
63  const ZoneVector<CaseInfo>& cases_;
64  int32_t min_value_; // minimum value of {cases_}
65  int32_t max_value_; // maximum value of {cases_}
66  size_t value_range_; // |max_value - min_value| + 1
67  turboshaft::Block* default_branch_;
68 };

352  Constant ToConstant(turboshaft::OpIndex node) {
353  using Kind = turboshaft::ConstantOp::Kind;
354  if (const turboshaft::ConstantOp* constant =
355  selector_->TryCast<turboshaft::ConstantOp>(node)) {
356  switch (constant->kind) {
357  case Kind::kWord32:
358  return Constant(static_cast<int32_t>(constant->word32()));
359  case Kind::kWord64:
360  return Constant(static_cast<int64_t>(constant->word64()));
361  case Kind::kSmi:
362  if constexpr (Is64()) {
363  return Constant(static_cast<int64_t>(constant->smi().ptr()));
364  } else {
365  return Constant(static_cast<int32_t>(constant->smi().ptr()));
366  }
367  case Kind::kHeapObject:
368  case Kind::kCompressedHeapObject:
369  case Kind::kTrustedHeapObject:
370  return Constant(constant->handle(),
371  constant->kind == Kind::kCompressedHeapObject);
372  case Kind::kExternal:
373  return Constant(constant->external_reference());
374  case Kind::kNumber:
375  return Constant(constant->number());
376  case Kind::kFloat32:
377  return Constant(constant->float32());
378  case Kind::kFloat64:
379  return Constant(constant->float64());
380  case Kind::kTaggedIndex: {
381  // Unencoded index value.
382  intptr_t value = static_cast<intptr_t>(constant->tagged_index());
383  DCHECK(TaggedIndex::IsValid(value));
384  // Generate it as 32/64-bit constant in a tagged form.
385  Address tagged_index = TaggedIndex::FromIntptr(value).ptr();
386  if (kSystemPointerSize == kInt32Size) {
387  return Constant(static_cast<int32_t>(tagged_index));
388  } else {
389  return Constant(static_cast<int64_t>(tagged_index));
390  }
391  }
392  case Kind::kRelocatableWasmCall:
393  case Kind::kRelocatableWasmStubCall: {
394  uint64_t value = constant->integral();
395  auto mode = constant->kind == Kind::kRelocatableWasmCall
396  ? RelocInfo::WASM_CALL
397  : RelocInfo::WASM_STUB_CALL;
398  using constant_type = std::conditional_t<Is64(), int64_t, int32_t>;
399  return Constant(RelocatablePtrConstantInfo(
400  base::checked_cast<constant_type>(value), mode));
401  }
402  case Kind::kRelocatableWasmCanonicalSignatureId:
403  return Constant(RelocatablePtrConstantInfo(
404  base::checked_cast<int32_t>(constant->integral()),
405  RelocInfo::WASM_CANONICAL_SIG_ID));
406  case Kind::kRelocatableWasmIndirectCallTarget:
407  uint64_t value = constant->integral();
408  return Constant(RelocatablePtrConstantInfo(
409  base::checked_cast<int32_t>(value),
410  RelocInfo::WASM_CODE_POINTER_TABLE_ENTRY));
411  }
412  }
413  UNREACHABLE();
414  }

462  UnallocatedOperand ToUnallocatedOperand(LinkageLocation location,
463  int virtual_register) {
464  if (location.IsAnyRegister() || location.IsNullRegister()) {
465  // any machine register.
466  return UnallocatedOperand(UnallocatedOperand::MUST_HAVE_REGISTER,
467  virtual_register);
468  }
469  if (location.IsCallerFrameSlot()) {
470  // a location on the caller frame.
471  return UnallocatedOperand(UnallocatedOperand::FIXED_SLOT,
472  location.AsCallerFrameSlot(), virtual_register);
473  }
474  if (location.IsCalleeFrameSlot()) {
475  // a spill location on this (callee) frame.
476  return UnallocatedOperand(UnallocatedOperand::FIXED_SLOT,
477  location.AsCalleeFrameSlot(), virtual_register);
478  }
479  // a fixed register.
480  if (IsFloatingPoint(location.GetType().representation())) {
481  return UnallocatedOperand(UnallocatedOperand::FIXED_FP_REGISTER,
482  location.AsRegister(), virtual_register);
483  }
484  return UnallocatedOperand(UnallocatedOperand::FIXED_REGISTER,
485  location.AsRegister(), virtual_register);
486  }
