110  void advance_hashtable_index() {
111  DisallowGarbageCollection no_gc;
112  ReadOnlyRoots roots = GetReadOnlyRoots();
113  InternalIndex max = range_->max_index();
114  // Increment until iterator points to a valid key or max.
115  while (index_ < max) {
116  Tagged<Object> key = table()->KeyAt(index_);
117  if (table()->IsKey(roots, key)) break;
118  ++index_;
119  }
120  }

123  };

126 
127  InternalIndex max_index() const {
128  int max = inlined()
129  ? scope_info_->ContextLocalCount()
130  : scope_info_->context_local_names_hashtable()->Capacity();
131  return InternalIndex(max);
132  }
