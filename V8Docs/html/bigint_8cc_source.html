44 // The MutableBigInt class is an implementation detail designed to prevent
45 // accidental mutation of a BigInt after its construction. Step-by-step
46 // construction of a BigInt must happen in terms of MutableBigInt, the
47 // final result is then passed through MutableBigInt::MakeImmutable and not
48 // modified further afterwards.
49 // Many of the functions in this class use arguments of type {BigIntBase},

151 template <typename T, typename Isolate>
152 MaybeHandle<T> ThrowBigIntTooBig(Isolate* isolate) {
153  // If the result of a BigInt computation is truncated to 64 bit, Turbofan
154  // can sometimes truncate intermediate results already, which can prevent
155  // those from exceeding the maximum length, effectively preventing a
156  // RangeError from being thrown. As this is a performance optimization, this
157  // behavior is accepted. To prevent the correctness fuzzer from detecting this
158  // difference, we crash the program.
159  if (v8_flags.correctness_fuzzer_suppressions) {
160  FATAL("Aborting on invalid BigInt length");
161  }
162  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kBigIntTooBig));
163 }

166 MaybeHandle<MutableBigInt> MutableBigInt::New(IsolateT* isolate,
167  uint32_t length,
168  AllocationType allocation) {
169  if (length > BigInt::kMaxLength) {
170  return ThrowBigIntTooBig<MutableBigInt>(isolate);
171  }

177 #endif

181 Handle<BigInt> MutableBigInt::NewFromInt(Isolate* isolate, int value) {
182  if (value == 0) return Zero(isolate);
183  Handle<MutableBigInt> result =
184  Cast<MutableBigInt>(isolate->factory()->NewBigInt(1));
185  bool sign = value < 0;
186  result->initialize_bitfield(sign, 1);
187  if (!sign) {
188  result->set_digit(0, value);
189  } else {
190  if (value == kMinInt) {
191  static_assert(kMinInt == -kMaxInt - 1);
192  result->set_digit(0, static_cast<BigInt::digit_t>(kMaxInt) + 1);
193  } else {
194  result->set_digit(0, -value);
195  }
196  }
197  return MakeImmutable(result);
198 }

200 Handle<BigInt> MutableBigInt::NewFromDouble(Isolate* isolate, double value) {
201  DCHECK_EQ(value, std::floor(value));
202  if (value == 0) return Zero(isolate);
203 
204  bool sign = value < 0; // -0 was already handled above.
205  uint64_t double_bits = base::bit_cast<uint64_t>(value);
206  int32_t raw_exponent =
207  static_cast<int32_t>(double_bits >>
208  base::Double::kPhysicalSignificandSize) &
209  0x7FF;
210  DCHECK_NE(raw_exponent, 0x7FF);
211  DCHECK_GE(raw_exponent, 0x3FF);
212  uint32_t exponent = raw_exponent - 0x3FF;
213  uint32_t digits = exponent / kDigitBits + 1;
214  Handle<MutableBigInt> result =
215  Cast<MutableBigInt>(isolate->factory()->NewBigInt(digits));
216  result->initialize_bitfield(sign, digits);
217 
218  // We construct a BigInt from the double {value} by shifting its mantissa
219  // according to its exponent and mapping the bit pattern onto digits.
220  //
221  // <----------- bitlength = exponent + 1 ----------->
222  // <----- 52 ------> <------ trailing zeroes ------>
223  // mantissa: 1yyyyyyyyyyyyyyyyy 0000000000000000000000000000000
224  // digits: 0001xxxx xxxxxxxx xxxxxxxx xxxxxxxx xxxxxxxx xxxxxxxx
225  // <--> <------>
226  // msd_topbit kDigitBits
227  //
228  uint64_t mantissa =
229  (double_bits & base::Double::kSignificandMask) | base::Double::kHiddenBit;
230  const uint32_t kMantissaTopBit =
231  base::Double::kSignificandSize - 1; // 0-indexed.
232  // 0-indexed position of most significant bit in the most significant digit.
233  uint32_t msd_topbit = exponent % kDigitBits;
234  // Number of unused bits in {mantissa}. We'll keep them shifted to the
235  // left (i.e. most significant part) of the underlying uint64_t.
236  uint32_t remaining_mantissa_bits = 0;
237  // Next digit under construction.
238  digit_t digit;
239 
240  // First, build the MSD by shifting the mantissa appropriately.
241  if (msd_topbit < kMantissaTopBit) {
242  remaining_mantissa_bits = kMantissaTopBit - msd_topbit;
243  digit = mantissa >> remaining_mantissa_bits;
244  mantissa = mantissa << (64 - remaining_mantissa_bits);
245  } else {
246  DCHECK_GE(msd_topbit, kMantissaTopBit);
247  digit = mantissa << (msd_topbit - kMantissaTopBit);
248  mantissa = 0;
249  }
250  result->set_digit(digits - 1, digit);
251  // Then fill in the rest of the digits.
252  static_assert(BigInt::kMaxLength < kMaxInt);
253  for (int32_t digit_index = digits - 2; digit_index >= 0; digit_index--) {
254  if (remaining_mantissa_bits > 0) {
255  remaining_mantissa_bits -= kDigitBits;
256  if (sizeof(digit) == 4) {
257  digit = mantissa >> 32;
258  mantissa = mantissa << 32;
259  } else {
260  DCHECK_EQ(sizeof(digit), 8);
261  digit = mantissa;
262  mantissa = 0;
263  }
264  } else {
265  digit = 0;
266  }
267  result->set_digit(digit_index, digit);
268  }
269  return MakeImmutable(result);
270 }

298 void MutableBigInt::Canonicalize(Tagged<MutableBigInt> result) {
299  // Check if we need to right-trim any leading zero-digits.
300  uint32_t old_length = result->length();
301  uint32_t new_length = old_length;
302  while (new_length > 0 && result->digit(new_length - 1) == 0) new_length--;
303  uint32_t to_trim = old_length - new_length;
304  if (to_trim != 0) {
305  Heap* heap = Isolate::Current()->heap();
306  if (!heap->IsLargeObject(result)) {
307  uint32_t old_size =
308  ALIGN_TO_ALLOCATION_ALIGNMENT(BigInt::SizeFor(old_length));
309  uint32_t new_size =
310  ALIGN_TO_ALLOCATION_ALIGNMENT(BigInt::SizeFor(new_length));
311  heap->NotifyObjectSizeChange(result, old_size, new_size,
312  ClearRecordedSlots::kNo);
313  }
314  result->set_length(new_length, kReleaseStore);
315 
316  // Canonicalize -0n.
317  if (new_length == 0) {
318  result->set_sign(false);
319  // TODO(jkummerow): If we cache a canonical 0n, return that here.
320  }
321  }
322  DCHECK_IMPLIES(result->length() > 0,
323  result->digit(result->length() - 1) != 0); // MSD is non-zero.
324  // Callers that don't require trimming must ensure this themselves.
325  DCHECK_IMPLIES(result->length() == 0, result->sign() == false);
326 }

358 MaybeDirectHandle<BigInt> BigInt::Exponentiate(Isolate* isolate,
359  DirectHandle<BigInt> base,
360  DirectHandle<BigInt> exponent) {
361  // 1. If exponent is < 0, throw a RangeError exception.
362  if (exponent->sign()) {
363  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kMustBePositive));
364  }
365  // 2. If base is 0n and exponent is 0n, return 1n.
366  if (exponent->is_zero()) {
367  return MutableBigInt::NewFromInt(isolate, 1);
368  }
369  // 3. Return a BigInt representing the mathematical value of base raised
370  // to the power exponent.
371  if (base->is_zero()) return base;
372  if (base->length() == 1 && base->digit(0) == 1) {
373  // (-1) ** even_number == 1.
374  if (base->sign() && (exponent->digit(0) & 1) == 0) {
375  return UnaryMinus(isolate, base);
376  }
377  // (-1) ** odd_number == -1; 1 ** anything == 1.
378  return base;
379  }
380  // For all bases >= 2, very large exponents would lead to unrepresentable
381  // results.
382  static_assert(kMaxLengthBits < std::numeric_limits<digit_t>::max());
383  if (exponent->length() > 1) {
384  return ThrowBigIntTooBig<BigInt>(isolate);
385  }
386  digit_t exp_value = exponent->digit(0);
387  if (exp_value == 1) return base;
388  if (exp_value >= kMaxLengthBits) {
389  return ThrowBigIntTooBig<BigInt>(isolate);
390  }
391  static_assert(kMaxLengthBits <= kMaxInt);
392  int n = static_cast<int>(exp_value);
393  if (base->length() == 1 && base->digit(0) == 2) {
394  // Fast path for 2^n.
395  int needed_digits = 1 + (n / kDigitBits);
396  Handle<MutableBigInt> result;
397  if (!MutableBigInt::New(isolate, needed_digits).ToHandle(&result))
398  return {};
399  result->InitializeDigits(needed_digits);
400  // All bits are zero. Now set the n-th bit.
401  digit_t msd = static_cast<digit_t>(1) << (n % kDigitBits);
402  result->set_digit(needed_digits - 1, msd);
403  // Result is negative for odd powers of -2n.
404  if (base->sign()) result->set_sign((n & 1) != 0);
405  return MutableBigInt::MakeImmutable(result);
406  }
407  DirectHandle<BigInt> result;
408  DirectHandle<BigInt> running_square = base;
409  // This implicitly sets the result's sign correctly.
410  if (n & 1) result = base;
411  n >>= 1;
412  for (; n != 0; n >>= 1) {
413  MaybeDirectHandle<BigInt> maybe_result =
414  Multiply(isolate, running_square, running_square);
415  if (!maybe_result.ToHandle(&running_square)) return maybe_result;
416  if (n & 1) {
417  if (result.is_null()) {
418  result = running_square;
419  } else {
420  maybe_result = Multiply(isolate, result, running_square);
421  if (!maybe_result.ToHandle(&result)) return maybe_result;
422  }
423  }
424  }
425  return result;
426 }

428 MaybeHandle<BigInt> BigInt::Multiply(Isolate* isolate, DirectHandle<BigInt> x,
429  DirectHandle<BigInt> y) {
430  if (x->is_zero()) return indirect_handle(x, isolate);
431  if (y->is_zero()) return indirect_handle(y, isolate);
432  uint32_t result_length =
433  bigint::MultiplyResultLength(x->digits(), y->digits());
434  Handle<MutableBigInt> result;
435  if (!MutableBigInt::New(isolate, result_length).ToHandle(&result)) {
436  return {};
437  }
438  DisallowGarbageCollection no_gc;
439  bigint::Status status = isolate->bigint_processor()->Multiply(
440  result->rw_digits(), x->digits(), y->digits());
441  if (status == bigint::Status::kInterrupted) {
442  AllowGarbageCollection terminating_anyway;
443  isolate->TerminateExecution();
444  return {};
445  }
446  result->set_sign(x->sign() != y->sign());
447  return MutableBigInt::MakeImmutable(result);

450 MaybeHandle<BigInt> BigInt::Divide(Isolate* isolate, DirectHandle<BigInt> x,
451  DirectHandle<BigInt> y) {
452  // 1. If y is 0n, throw a RangeError exception.
453  if (y->is_zero()) {
454  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kBigIntDivZero));
455  }
456  // 2. Let quotient be the mathematical value of x divided by y.
457  // 3. Return a BigInt representing quotient rounded towards 0 to the next
458  // integral value.
459  if (bigint::Compare(x->digits(), y->digits()) < 0) {
460  return Zero(isolate);
461  }
462  bool result_sign = x->sign() != y->sign();
463  if (y->length() == 1 && y->digit(0) == 1) {
464  return result_sign == x->sign() ? indirect_handle(x, isolate)
465  : UnaryMinus(isolate, x);
466  }
467  Handle<MutableBigInt> quotient;
468  uint32_t result_length = bigint::DivideResultLength(x->digits(), y->digits());
469  if (!MutableBigInt::New(isolate, result_length).ToHandle(&quotient)) {
470  return {};
471  }
472  DisallowGarbageCollection no_gc;
473  bigint::Status status = isolate->bigint_processor()->Divide(
474  quotient->rw_digits(), x->digits(), y->digits());
475  if (status == bigint::Status::kInterrupted) {
476  AllowGarbageCollection terminating_anyway;
477  isolate->TerminateExecution();
478  return {};
479  }
480  quotient->set_sign(result_sign);
481  return MutableBigInt::MakeImmutable(quotient);
482 }

484 MaybeHandle<BigInt> BigInt::Remainder(Isolate* isolate, DirectHandle<BigInt> x,
485  DirectHandle<BigInt> y) {
486  // 1. If y is 0n, throw a RangeError exception.
487  if (y->is_zero()) {
488  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kBigIntDivZero));
489  }
490  // 2. Return the BigInt representing x modulo y.
491  // See https://github.com/tc39/proposal-bigint/issues/84 though.
492  if (bigint::Compare(x->digits(), y->digits()) < 0)
493  return indirect_handle(x, isolate);
494  if (y->length() == 1 && y->digit(0) == 1) return Zero(isolate);
495  Handle<MutableBigInt> remainder;
496  uint32_t result_length = bigint::ModuloResultLength(y->digits());
497  if (!MutableBigInt::New(isolate, result_length).ToHandle(&remainder)) {
498  return {};
499  }
500  DisallowGarbageCollection no_gc;
501  bigint::Status status = isolate->bigint_processor()->Modulo(
502  remainder->rw_digits(), x->digits(), y->digits());
503  if (status == bigint::Status::kInterrupted) {
504  AllowGarbageCollection terminating_anyway;
505  isolate->TerminateExecution();
506  return {};
507  }
508  remainder->set_sign(x->sign());
509  return MutableBigInt::MakeImmutable(remainder);
510 }

512 MaybeHandle<BigInt> BigInt::Add(Isolate* isolate, DirectHandle<BigInt> x,
513  DirectHandle<BigInt> y) {
514  if (x->is_zero()) return indirect_handle(y, isolate);
515  if (y->is_zero()) return indirect_handle(x, isolate);
516  bool xsign = x->sign();
517  bool ysign = y->sign();

519  bigint::AddSignedResultLength(x->length(), y->length(), xsign == ysign);
520  Handle<MutableBigInt> result;

523  return {};

526  bool result_sign = bigint::AddSigned(result->rw_digits(), x->digits(), xsign,
527  y->digits(), ysign);
528  result->set_sign(result_sign);
529  return MutableBigInt::MakeImmutable(result);

704 ComparisonResult BigInt::CompareToDouble(DirectHandle<BigInt> x, double y) {
705  if (std::isnan(y)) return ComparisonResult::kUndefined;
706  if (y == V8_INFINITY) return ComparisonResult::kLessThan;
707  if (y == -V8_INFINITY) return ComparisonResult::kGreaterThan;
708  bool x_sign = x->sign();
709  // Note that this is different from the double's sign bit for -0. That's
710  // intentional because -0 must be treated like 0.
711  bool y_sign = (y < 0);
712  if (x_sign != y_sign) return UnequalSign(x_sign);
713  if (y == 0) {
714  DCHECK(!x_sign);
715  return x->is_zero() ? ComparisonResult::kEqual
716  : ComparisonResult::kGreaterThan;
717  }
718  if (x->is_zero()) {
719  DCHECK(!y_sign);
720  return ComparisonResult::kLessThan;
721  }
722  uint64_t double_bits = base::bit_cast<uint64_t>(y);
723  int32_t raw_exponent =
724  static_cast<int32_t>(double_bits >>
725  base::Double::kPhysicalSignificandSize) &
726  0x7FF;
727  uint64_t mantissa = double_bits & base::Double::kSignificandMask;
728  // Non-finite doubles are handled above.
729  DCHECK_NE(raw_exponent, 0x7FF);
730  int32_t exponent = raw_exponent - 0x3FF;
731  if (exponent < 0) {
732  // The absolute value of the double is less than 1. Only 0n has an
733  // absolute value smaller than that, but we've already covered that case.
734  DCHECK(!x->is_zero());
735  return AbsoluteGreater(x_sign);
736  }
737  uint32_t x_length = x->length();
738  digit_t x_msd = x->digit(x_length - 1);
739  uint32_t msd_leading_zeros = base::bits::CountLeadingZeros(x_msd);
740  uint32_t x_bitlength = x_length * kDigitBits - msd_leading_zeros;
741  uint32_t y_bitlength = exponent + 1;
742  if (x_bitlength < y_bitlength) return AbsoluteLess(x_sign);
743  if (x_bitlength > y_bitlength) return AbsoluteGreater(x_sign);
744 
745  // At this point, we know that signs and bit lengths (i.e. position of
746  // the most significant bit in exponent-free representation) are identical.
747  // {x} is not zero, {y} is finite and not denormal.
748  // Now we virtually convert the double to an integer by shifting its
749  // mantissa according to its exponent, so it will align with the BigInt {x},
750  // and then we compare them bit for bit until we find a difference or the
751  // least significant bit.
752  // <----- 52 ------> <-- virtual trailing zeroes -->
753  // y / mantissa: 1yyyyyyyyyyyyyyyyy 0000000000000000000000000000000
754  // x / digits: 0001xxxx xxxxxxxx xxxxxxxx ...
755  // <--> <------>
756  // msd_topbit kDigitBits
757  //
758  mantissa |= base::Double::kHiddenBit;
759  const uint32_t kMantissaTopBit = 52; // 0-indexed.
760  // 0-indexed position of {x}'s most significant bit within the {msd}.
761  uint32_t msd_topbit = kDigitBits - 1 - msd_leading_zeros;
762  DCHECK_EQ(msd_topbit, (x_bitlength - 1) % kDigitBits);
763  // Shifted chunk of {mantissa} for comparing with {digit}.
764  digit_t compare_mantissa;
765  // Number of unprocessed bits in {mantissa}. We'll keep them shifted to
766  // the left (i.e. most significant part) of the underlying uint64_t.
767  uint32_t remaining_mantissa_bits = 0;
768 
769  // First, compare the most significant digit against the beginning of
770  // the mantissa.
771  if (msd_topbit < kMantissaTopBit) {
772  remaining_mantissa_bits = (kMantissaTopBit - msd_topbit);
773  compare_mantissa = mantissa >> remaining_mantissa_bits;
774  mantissa = mantissa << (64 - remaining_mantissa_bits);
775  } else {
776  DCHECK_GE(msd_topbit, kMantissaTopBit);
777  compare_mantissa = mantissa << (msd_topbit - kMantissaTopBit);
778  mantissa = 0;
779  }
780  if (x_msd > compare_mantissa) return AbsoluteGreater(x_sign);
781  if (x_msd < compare_mantissa) return AbsoluteLess(x_sign);
782 
783  // Then, compare additional digits against any remaining mantissa bits.
784  static_assert(BigInt::kMaxLength < kMaxInt);
785  for (int32_t digit_index = x_length - 2; digit_index >= 0; digit_index--) {
786  if (remaining_mantissa_bits > 0) {
787  remaining_mantissa_bits -= kDigitBits;
788  if (sizeof(mantissa) != sizeof(x_msd)) {
789  compare_mantissa = mantissa >> (64 - kDigitBits);
790  // "& 63" to appease compilers. kDigitBits is 32 here anyway.
791  mantissa = mantissa << (kDigitBits & 63);
792  } else {
793  compare_mantissa = mantissa;
794  mantissa = 0;
795  }
796  } else {
797  compare_mantissa = 0;
798  }
799  digit_t digit = x->digit(digit_index);
800  if (digit > compare_mantissa) return AbsoluteGreater(x_sign);
801  if (digit < compare_mantissa) return AbsoluteLess(x_sign);
802  }
803 
804  // Integer parts are equal; check whether {y} has a fractional part.
805  if (mantissa != 0) {
806  DCHECK_GT(remaining_mantissa_bits, 0);
807  return AbsoluteLess(x_sign);
808  }
809  return ComparisonResult::kEqual;
810 }

831 MaybeHandle<String> BigInt::ToString(Isolate* isolate,
832  DirectHandle<BigInt> bigint, int radix,
833  ShouldThrow should_throw) {
834  if (bigint->is_zero()) {
835  return isolate->factory()->zero_string();
836  }
837  const bool sign = bigint->sign();
838  uint32_t chars_allocated;
839  uint32_t chars_written;
840  Handle<SeqOneByteString> result;
841  if (bigint->length() == 1 && radix == 10) {
842  // Fast path for the most common case, to avoid call/dispatch overhead.
843  // The logic is the same as what the full implementation does below,
844  // just inlined and specialized for the preconditions.
845  // Microbenchmarks rejoice!
846  digit_t digit = bigint->digit(0);
847  uint32_t bit_length = kDigitBits - base::bits::CountLeadingZeros(digit);
848  constexpr uint32_t kShift = 7;
849  // This is Math.log2(10) * (1 << kShift), scaled just far enough to
850  // make the computations below always precise (after rounding).
851  constexpr uint32_t kShiftedBitsPerChar = 425;
852  chars_allocated = (bit_length << kShift) / kShiftedBitsPerChar + 1 + sign;
853  result = isolate->factory()
854  ->NewRawOneByteString(chars_allocated)
855  .ToHandleChecked();
856  DisallowGarbageCollection no_gc;
857  uint8_t* start = result->GetChars(no_gc);
858  uint8_t* out = start + chars_allocated;
859  while (digit != 0) {
860  *(--out) = '0' + (digit % 10);
861  digit /= 10;
862  }
863  if (sign) *(--out) = '-';
864  if (out == start) {
865  chars_written = chars_allocated;
866  } else {
867  DCHECK_LT(start, out);
868  // The result is one character shorter than predicted. This is
869  // unavoidable, e.g. a 4-bit BigInt can be as big as "10" or as small as
870  // "9", so we must allocate 2 characters for it, and will only later find
871  // out whether all characters were used.
872  chars_written = chars_allocated - static_cast<uint32_t>(out - start);
873  std::memmove(start, out, chars_written);
874  memset(start + chars_written, 0, chars_allocated - chars_written);
875  }
876  } else {
877  // Generic path, handles anything.
878  DCHECK(radix >= 2 && radix <= 36);
879  chars_allocated =
880  bigint::ToStringResultLength(bigint->digits(), radix, sign);
881  if (chars_allocated > String::kMaxLength) {
882  if (should_throw == kThrowOnError) {
883  THROW_NEW_ERROR(isolate, NewInvalidStringLengthError());
884  } else {
885  return {};
886  }
887  }
888  result = isolate->factory()
889  ->NewRawOneByteString(chars_allocated)
890  .ToHandleChecked();
891  chars_written = chars_allocated;
892  DisallowGarbageCollection no_gc;
893  char* characters = reinterpret_cast<char*>(result->GetChars(no_gc));
894  bigint::Status status = isolate->bigint_processor()->ToString(
895  characters, &chars_written, bigint->digits(), radix, sign);
896  if (status == bigint::Status::kInterrupted) {
897  AllowGarbageCollection terminating_anyway;
898  isolate->TerminateExecution();
899  return {};
900  }
901  }
902 
903  // Right-trim any over-allocation (which can happen due to conservative
904  // estimates).
905  RightTrimString(isolate, result, chars_allocated, chars_written);
906 #if DEBUG
907  // Verify that all characters have been written.
908  DCHECK(result->length() == chars_written);
909  DisallowGarbageCollection no_gc;
910  uint8_t* chars = result->GetChars(no_gc);
911  for (uint32_t i = 0; i < chars_written; i++) {
912  DCHECK_NE(chars[i], bigint::kStringZapValue);
913  }
914 #endif
915  return result;
916 }

1031 double MutableBigInt::ToDouble(DirectHandle<BigIntBase> x) {
1032  if (x->is_zero()) return 0.0;
1033  uint32_t x_length = x->length();
1034  digit_t x_msd = x->digit(x_length - 1);
1035  uint32_t msd_leading_zeros = base::bits::CountLeadingZeros(x_msd);
1036  uint32_t x_bitlength = x_length * kDigitBits - msd_leading_zeros;
1037  if (x_bitlength > 1024) return x->sign() ? -V8_INFINITY : V8_INFINITY;
1038  uint64_t exponent = x_bitlength - 1;
1039  // We need the most significant bit shifted to the position of a double's
1040  // "hidden bit". We also need to hide that MSB, so we shift it out.
1041  uint64_t current_digit = x_msd;
1042  uint32_t digit_index = x_length - 1;
1043  uint32_t shift = msd_leading_zeros + 1 + (64 - kDigitBits);
1044  DCHECK_LE(1, shift);
1045  DCHECK_LE(shift, 64);
1046  uint64_t mantissa = (shift == 64) ? 0 : current_digit << shift;
1047  mantissa >>= 12;
1048  int32_t mantissa_bits_unset = shift - 12;
1049  // If not all mantissa bits are defined yet, get more digits as needed.
1050  if (mantissa_bits_unset >= static_cast<int32_t>(kDigitBits) &&
1051  digit_index > 0) {
1052  digit_index--;
1053  current_digit = static_cast<uint64_t>(x->digit(digit_index));
1054  mantissa |= (current_digit << (mantissa_bits_unset - kDigitBits));
1055  mantissa_bits_unset -= kDigitBits;
1056  }
1057  if (mantissa_bits_unset > 0 && digit_index > 0) {
1058  DCHECK_LT(mantissa_bits_unset, kDigitBits);
1059  digit_index--;
1060  current_digit = static_cast<uint64_t>(x->digit(digit_index));
1061  mantissa |= (current_digit >> (kDigitBits - mantissa_bits_unset));
1062  mantissa_bits_unset -= kDigitBits;
1063  }
1064  // If there are unconsumed digits left, we may have to round.
1065  Rounding rounding =
1066  DecideRounding(x, mantissa_bits_unset, digit_index, current_digit);
1067  if (rounding == kRoundUp || (rounding == kTie && (mantissa & 1) == 1)) {
1068  mantissa++;
1069  // Incrementing the mantissa can overflow the mantissa bits. In that case
1070  // the new mantissa will be all zero (plus hidden bit).
1071  if ((mantissa >> base::Double::kPhysicalSignificandSize) != 0) {
1072  mantissa = 0;
1073  exponent++;
1074  // Incrementing the exponent can overflow too.
1075  if (exponent > 1023) {
1076  return x->sign() ? -V8_INFINITY : V8_INFINITY;
1077  }
1078  }
1079  }
1080  // Assemble the result.
1081  uint64_t sign_bit = x->sign() ? (static_cast<uint64_t>(1) << 63) : 0;
1082  exponent = (exponent + 0x3FF) << base::Double::kPhysicalSignificandSize;
1083  uint64_t double_bits = sign_bit | exponent | mantissa;
1084  return base::bit_cast<double>(double_bits);
1085 }

1087 // This is its own function to simplify control flow. The meaning of the
1088 // parameters is defined by {ToDouble}'s local variable usage.
1089 MutableBigInt::Rounding MutableBigInt::DecideRounding(
1090  DirectHandle<BigIntBase> x, int mantissa_bits_unset, int digit_index,
1091  uint64_t current_digit) {
1092  if (mantissa_bits_unset > 0) return kRoundDown;
1093  int top_unconsumed_bit;
1094  if (mantissa_bits_unset < 0) {
1095  // There are unconsumed bits in {current_digit}.
1096  top_unconsumed_bit = -mantissa_bits_unset - 1;
1097  } else {
1098  DCHECK_EQ(mantissa_bits_unset, 0);
1099  // {current_digit} fit the mantissa exactly; look at the next digit.
1100  if (digit_index == 0) return kRoundDown;
1101  digit_index--;
1102  current_digit = static_cast<uint64_t>(x->digit(digit_index));
1103  top_unconsumed_bit = kDigitBits - 1;
1104  }
1105  // If the most significant remaining bit is 0, round down.
1106  uint64_t bitmask = static_cast<uint64_t>(1) << top_unconsumed_bit;
1107  if ((current_digit & bitmask) == 0) {
1108  return kRoundDown;
1109  }
1110  // If any other remaining bit is set, round up.
1111  bitmask -= 1;
1112  if ((current_digit & bitmask) != 0) return kRoundUp;
1113  while (digit_index > 0) {
1114  digit_index--;
1115  if (x->digit(digit_index) != 0) return kRoundUp;
1116  }
1117  return kTie;
1118 }

1248  Isolate* isolate, uint32_t bitfield,

1251  DCHECK_EQ(static_cast<uint32_t>(digits_storage.length()), bytelength);
1252  bool sign = SignBits::decode(bitfield);

1257  Cast<MutableBigInt>(isolate->factory()->NewBigInt(length));
1258  result->initialize_bitfield(sign, length);
1259  UnalignedValueMember<digit_t>* digits = result->raw_digits();

1263  reinterpret_cast<void*>(reinterpret_cast<Address>(digits) + bytelength);
1264  memset(padding_start, 0, length * kDigitSize - bytelength);
1265 #elif defined(V8_TARGET_BIG_ENDIAN)

1268  reinterpret_cast<const digit_t*>(digits_storage.begin());

1270  *digit = ByteReverse(*digit_storage);
1271  digit_storage++;
1272  digit++;

1282  digit_byte--;

1284  }
