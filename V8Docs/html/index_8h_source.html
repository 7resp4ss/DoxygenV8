36 // `OpIndex` is an offset from the beginning of the operations buffer.
37 // Compared to `Operation*`, it is more memory efficient (32bit) and stable when
38 // the operations buffer is re-allocated.
39 class OpIndex {
40  protected:
41  // We make this constructor protected so that integers are not easily
42  // convertible to OpIndex. FromOffset should be used instead to create an
43  // OpIndex from an offset.
44  explicit constexpr OpIndex(uint32_t offset) : offset_(offset) {
45  SLOW_DCHECK(CheckInvariants());
46  }
47  friend class OperationBuffer;
48 
49  public:
50  static constexpr OpIndex FromOffset(uint32_t offset,
51  uint32_t generation_bit) {
52 #ifdef DEBUG
53  DCHECK(generation_bit == 0 || generation_bit == 1);
54  DCHECK_EQ(offset & kGenerationMask, 0);
55  offset |= (generation_bit << kGenerationMaskShift);
56 #endif
57  return OpIndex(offset);
58  }
59  constexpr OpIndex() : offset_(std::numeric_limits<uint32_t>::max()) {}
60  template <typename T, typename C>
61  OpIndex(const ConstOrV<T, C>&) { // NOLINT(runtime/explicit)
62  static_assert(base::tmp::lazy_false<T>::value,
63  "Cannot initialize OpIndex from ConstOrV<>. Did you forget "
64  "to resolve() it in the assembler?");
65  }
66 
67  constexpr uint32_t id() const {
68  // Operations are stored at an offset that's a multiple of
69  // `sizeof(OperationStorageSlot)`. In addition, an operation occupies at
70  // least `kSlotsPerId` many `OperationSlot`s. Therefore, we can assign id's
71  // by dividing by `kSlotsPerId`. A compact id space is important, because it
72  // makes side-tables smaller.
73  SLOW_DCHECK(CheckInvariants());
74  return offset_ / sizeof(OperationStorageSlot) / kSlotsPerId;
75  }
76  uint32_t hash() const {
77  // It can be useful to hash OpIndex::Invalid(), so we have this `hash`
78  // function, which returns the id, but without DCHECKing that Invalid is
79  // valid.
80  SLOW_DCHECK_IMPLIES(valid(), CheckInvariants());
81  return offset_ / sizeof(OperationStorageSlot) / kSlotsPerId;
82  }
83  uint32_t offset() const {
84  SLOW_DCHECK(CheckInvariants());
85 #ifdef DEBUG
86  return offset_ & kUnmaskGenerationMask;
87 #else
88  return offset_;
89 #endif
90  }
91 
92  constexpr bool valid() const { return *this != Invalid(); }
93 
94  static constexpr OpIndex Invalid() { return OpIndex(); }
95 
96  // Encode a sea-of-nodes node id in the `OpIndex` type.
97  // Only used for node origins that actually point to sea-of-nodes graph nodes.
98  static OpIndex EncodeTurbofanNodeId(uint32_t id) {
99  OpIndex result = OpIndex(id * sizeof(OperationStorageSlot));
100  result.offset_ += kTurbofanNodeIdFlag;
101  return result;
102  }
103  uint32_t DecodeTurbofanNodeId() const {
104  DCHECK(IsTurbofanNodeId());
105  return offset_ / sizeof(OperationStorageSlot);
106  }
107  bool IsTurbofanNodeId() const {
108  return offset_ % sizeof(OperationStorageSlot) == kTurbofanNodeIdFlag;
109  }
110 
111  constexpr bool operator==(OpIndex other) const {
112  return offset_ == other.offset_;
113  }
114  constexpr bool operator!=(OpIndex other) const {
115  return offset_ != other.offset_;
116  }
117  constexpr bool operator<(OpIndex other) const {
118  return offset_ < other.offset_;
119  }
120  constexpr bool operator>(OpIndex other) const {
121  return offset_ > other.offset_;
122  }
123  constexpr bool operator<=(OpIndex other) const {
124  return offset_ <= other.offset_;
125  }
126  constexpr bool operator>=(OpIndex other) const {
127  return offset_ >= other.offset_;
128  }
129 
130 #ifdef DEBUG
131  int generation_mod2() const {
132  return (offset_ & kGenerationMask) >> kGenerationMaskShift;
133  }
134  void set_generation_mod2(int generation_mod2) {
135  DCHECK_LE(generation_mod2, 1);
136  offset_ |= generation_mod2 << kGenerationMaskShift;
137  }
138 
139  constexpr bool CheckInvariants() const {
140  DCHECK(valid());
141  // The second lowest significant bit of the offset is used to store the
142  // graph generation modulo 2. The lowest and 3rd lowest bits should always
143  // be 0 (as long as sizeof(OperationStorageSlot) is 8).
144  static_assert(sizeof(OperationStorageSlot) == 8);
145  return (offset_ & 0b101) == 0;
146  }
147 #endif
148 
149  protected:
150  static constexpr uint32_t kGenerationMaskShift = 1;
151  static constexpr uint32_t kGenerationMask = 1 << kGenerationMaskShift;
152  static constexpr uint32_t kUnmaskGenerationMask = ~kGenerationMask;
153 
154  // In DEBUG builds, the offset's second lowest bit contains the graph
155  // generation % 2, so one should keep this in mind when looking at the value
156  // of the offset.
157  uint32_t offset_;
158 
159  static constexpr uint32_t kTurbofanNodeIdFlag = 1;
160 
161  template <typename H>
162  friend H AbslHashValue(H h, const OpIndex& idx) {
163  return H::combine(std::move(h), idx.offset_);
164  }
165 };

608 // V<> represents an SSA-value that is parameterized with the type of the value.
609 // Types from the `Object` hierarchy can be provided as well as the abstract
610 // representation classes (`Word32`, ...) defined above.
611 // Prefer using V<> instead of a plain OpIndex where possible.
612 template <typename T>
613 class V : public OpIndex {
614  // V<T> is implicitly constructible from V<U> iff
615  // `v_traits<T>::implicitly_constructible_from<U>::value`. This is typically
616  // the case if T == U or T is a subclass of U. Different types may specify
617  // different conversion rules in the corresponding `v_traits` when necessary.
618  template <typename U>
619  constexpr static bool implicitly_constructible_from =
620  v_traits<T>::template implicitly_constructible_from<U>::value;
621 
622  public:
623  using type = T;
624  static constexpr auto rep = v_traits<type>::rep;
625  constexpr V() : OpIndex() {}
626 
627  // V<T> is implicitly constructible from V<U> iff
628  // `v_traits<T>::implicitly_constructible_from<U>::value`. This is typically
629  // the case if T == U or T is a subclass of U. Different types may specify
630  // different conversion rules in the corresponding `v_traits` when necessary.
631  template <typename U>
632  requires implicitly_constructible_from<U>
633  V(V<U> index) : OpIndex(index) {} // NOLINT(runtime/explicit)
634 
635  static V Invalid() { return V<T>(OpIndex::Invalid()); }
636 
637  template <typename U>
638  static V<T> Cast(V<U> index) {
639  return V<T>(OpIndex{index});
640  }
641  static V<T> Cast(OpIndex index) { return V<T>(index); }
