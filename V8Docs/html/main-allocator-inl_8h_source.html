18 AllocationResult MainAllocator::AllocateRaw(int size_in_bytes,
19  AllocationAlignment alignment,
20  AllocationOrigin origin,
21  AllocationHint hint) {
22  size_in_bytes = ALIGN_TO_ALLOCATION_ALIGNMENT(size_in_bytes);
23 
24  DCHECK_EQ(in_gc(), origin == AllocationOrigin::kGC);
25  DCHECK_EQ(in_gc(), isolate_heap()->IsInGC());
26 
27  // We are not supposed to allocate in fast c calls.
28  DCHECK_IMPLIES(is_main_thread(),
29  v8_flags.allow_allocation_in_fast_api_call ||
30  !isolate_heap()->isolate()->InFastCCall());
31 
32  AllocationResult result;
33 
34  if (alignment != kTaggedAligned) [[unlikely]] {
35  result = AllocateFastAligned(size_in_bytes, nullptr, alignment, origin);
36  } else {
37  result = AllocateFastUnaligned(size_in_bytes, origin);
38  }
39 
40  return result.IsFailure() ? AllocateRawSlow(size_in_bytes, alignment, origin)
41  : result;
42 }

44 AllocationResult MainAllocator::AllocateFastUnaligned(int size_in_bytes,
45  AllocationOrigin origin) {
46  size_in_bytes = ALIGN_TO_ALLOCATION_ALIGNMENT(size_in_bytes);
47  if (!allocation_info().CanIncrementTop(size_in_bytes)) {
48  return AllocationResult::Failure();
49  }
50  Tagged<HeapObject> obj =
51  HeapObject::FromAddress(allocation_info().IncrementTop(size_in_bytes));
52 
53  MSAN_ALLOCATED_UNINITIALIZED_MEMORY(obj.address(), size_in_bytes);
54 
55  DCHECK_IMPLIES(black_allocation_ == BlackAllocation::kAlwaysEnabled,
56  space_heap()->marking_state()->IsMarked(obj));
57 
58  return AllocationResult::FromObject(obj);
59 }

61 AllocationResult MainAllocator::AllocateFastAligned(
62  int size_in_bytes, int* result_aligned_size_in_bytes,
63  AllocationAlignment alignment, AllocationOrigin origin) {
64  Address top = allocation_info().top();
65  int filler_size = Heap::GetFillToAlign(top, alignment);
66  int aligned_size_in_bytes = size_in_bytes + filler_size;
67 
68  if (!allocation_info().CanIncrementTop(aligned_size_in_bytes)) {
69  return AllocationResult::Failure();
70  }
71  Tagged<HeapObject> obj = HeapObject::FromAddress(
72  allocation_info().IncrementTop(aligned_size_in_bytes));
73  if (result_aligned_size_in_bytes)
74  *result_aligned_size_in_bytes = aligned_size_in_bytes;
75 
76  if (filler_size > 0) {
77  obj = space_heap()->PrecedeWithFiller(obj, filler_size);
78  }
79 
80  MSAN_ALLOCATED_UNINITIALIZED_MEMORY(obj.address(), size_in_bytes);
81 
82  DCHECK_IMPLIES(black_allocation_ == BlackAllocation::kAlwaysEnabled,
83  space_heap()->marking_state()->IsMarked(obj));
84 
85  return AllocationResult::FromObject(obj);
86 }
