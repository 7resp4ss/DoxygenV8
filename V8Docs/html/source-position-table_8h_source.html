79 class V8_EXPORT_PRIVATE SourcePositionTableIterator {
80  public:
81  // Filter that applies when advancing the iterator. If the filter isn't
82  // satisfied, we advance the iterator again.
83  enum IterationFilter { kJavaScriptOnly = 0, kExternalOnly = 1, kAll = 2 };
84  // Filter that applies only to the first entry of the source position table.
85  // If it is kSkipFunctionEntry, it will skip the FunctionEntry entry if it
86  // exists.
87  enum FunctionEntryFilter {
88  kSkipFunctionEntry = 0,
89  kDontSkipFunctionEntry = 1
90  };
91 
92  // Used for saving/restoring the iterator.
93  struct IndexAndPositionState {
94  int index_;
95  PositionTableEntry position_;
96  IterationFilter iteration_filter_;
97  FunctionEntryFilter function_entry_filter_;
98  };
99 
100  // We expose three flavours of the iterator, depending on the argument passed
101  // to the constructor:
102 
103  // Handlified iterator allows allocation, but it needs a handle (and thus
104  // a handle scope). This is the preferred version.
105  explicit SourcePositionTableIterator(
106  Handle<TrustedByteArray> byte_array,
107  IterationFilter iteration_filter = kJavaScriptOnly,
108  FunctionEntryFilter function_entry_filter = kSkipFunctionEntry);
109 
110  // Non-handlified iterator does not need a handle scope, but it disallows
111  // allocation during its lifetime. This is useful if there is no handle
112  // scope around.
113  explicit SourcePositionTableIterator(
114  Tagged<TrustedByteArray> byte_array,
115  IterationFilter iteration_filter = kJavaScriptOnly,
116  FunctionEntryFilter function_entry_filter = kSkipFunctionEntry);
117 
118  // Handle-safe iterator based on an a vector located outside the garbage
119  // collected heap, allows allocation during its lifetime.
120  explicit SourcePositionTableIterator(
121  base::Vector<const uint8_t> bytes,
122  IterationFilter iteration_filter = kJavaScriptOnly,
123  FunctionEntryFilter function_entry_filter = kSkipFunctionEntry);
124 
125  void Advance();
126 
127  int code_offset() const {
128  DCHECK(!done());
129  return current_.code_offset;
130  }
131  SourcePosition source_position() const {
132  DCHECK(!done());
133  return SourcePosition::FromRaw(current_.source_position);
134  }
135  bool is_statement() const {
136  DCHECK(!done());
137  return current_.is_statement;
138  }
139  bool is_breakable() const {
140  DCHECK(!done());
141  return current_.is_breakable;
142  }
143  bool done() const { return index_ == kDone; }
144 
145  IndexAndPositionState GetState() const {
146  return {index_, current_, iteration_filter_, function_entry_filter_};
147  }
148 
149  void RestoreState(const IndexAndPositionState& saved_state) {
150  index_ = saved_state.index_;
151  current_ = saved_state.position_;
152  iteration_filter_ = saved_state.iteration_filter_;
153  function_entry_filter_ = saved_state.function_entry_filter_;
154  }
155 
156  private:
157  // Initializes the source position interator with the first valid bytecode.
158  // Also sets the FunctionEntry SourcePosition if it exists.
159  void Initialize();
160 
161  static const int kDone = -1;
162 
163  base::Vector<const uint8_t> raw_table_;
164  Handle<TrustedByteArray> table_;
165  int index_ = 0;
166  PositionTableEntry current_;
167  IterationFilter iteration_filter_;
168  FunctionEntryFilter function_entry_filter_;
169  DISALLOW_GARBAGE_COLLECTION(no_gc)
170 };
