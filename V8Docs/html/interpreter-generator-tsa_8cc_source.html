41 template <typename Next>
42 class BytecodeHandlerReducer : public Next {
43  public:
44  BUILTIN_REDUCER(BytecodeHandler)
45 
46  ~BytecodeHandlerReducer() {
47  // If the following check fails the handler does not use the
48  // accumulator in the way described in the bytecode definitions in
49  // bytecodes.h.
50  DCHECK_EQ(data_.implicit_register_use,
51  Bytecodes::GetImplicitRegisterUse(data_.bytecode));
52  }
53 
54  void InitializeParameters(V<Object> accumulator,
55  V<BytecodeArray> bytecode_array,
56  V<WordPtr> bytecode_offset,
57  V<WordPtr> dispatch_table) {
58  accumulator_ = accumulator;
59  bytecode_array_ = bytecode_array;
60  bytecode_offset_ = bytecode_offset;
61  dispatch_table_ = dispatch_table;
62  }
63 
64  V<Object> GetAccumulator() {
65  DCHECK(Bytecodes::ReadsAccumulator(data_.bytecode));
66  TrackRegisterUse(ImplicitRegisterUse::kReadAccumulator);
67  return accumulator_;
68  }
69 
70  void SetAccumulator(V<Object> value) {
71  DCHECK(Bytecodes::WritesAccumulator(data_.bytecode));
72  TrackRegisterUse(ImplicitRegisterUse::kWriteAccumulator);
73  accumulator_ = value;
74  }
75 
76  V<Context> GetContext() {
77  return V<Context>::Cast(LoadRegister(Register::current_context()));
78  }
79 
80  void Dispatch() {
81  __ CodeComment("========= Dispatch");
82  DCHECK_IMPLIES(Bytecodes::MakesCallAlongCriticalPath(data_.bytecode),
83  data_.made_call);
84  V<WordPtr> target_offset = Advance(CurrentBytecodeSize());
85  V<WordPtr> target_bytecode = LoadBytecode(target_offset);
86  DispatchToBytecodeWithOptionalStarLookahead(target_bytecode);
87  }
88 
89  void DispatchToBytecodeWithOptionalStarLookahead(V<WordPtr> target_bytecode) {
90  if (Bytecodes::IsStarLookahead(data_.bytecode, operand_scale())) {
91  StarDispatchLookahead(target_bytecode);
92  }
93  DispatchToBytecode(target_bytecode, BytecodeOffset());
94  }
95 
96  void DispatchToBytecode(V<WordPtr> target_bytecode,
97  V<WordPtr> new_bytecode_offset) {
98 #ifdef V8_IGNITION_DISPATCH_COUNTING
99  TraceBytecodeDispatch(target_bytecode);
100 #endif
101 
102  static_assert(kSystemPointerSizeLog2 ==
103  MemoryRepresentation::UintPtr().SizeInBytesLog2());
104  V<WordPtr> target_code_entry =
105  __ LoadOffHeap(DispatchTablePointer(), target_bytecode, 0,
106  MemoryRepresentation::UintPtr());
107 
108  DispatchToBytecodeHandlerEntry(target_code_entry, new_bytecode_offset);
109  }
110 
111  void DispatchToBytecodeHandlerEntry(V<WordPtr> handler_entry,
112  V<WordPtr> bytecode_offset) {
113  TailCallBytecodeDispatch(
114  InterpreterDispatchDescriptor{}, handler_entry, accumulator_.Get(),
115  bytecode_offset, BytecodeArrayTaggedPointer(), DispatchTablePointer());
116  }
117 
118  void StarDispatchLookahead(V<WordPtr> target_bytecode) { UNIMPLEMENTED(); }

120  template <typename... Args>
121  void TailCallBytecodeDispatch(const CallInterfaceDescriptor& descriptor,
122  V<WordPtr> target, Args... args) {
123  DCHECK_EQ(descriptor.GetParameterCount(), sizeof...(Args));
124  auto call_descriptor = compiler::Linkage::GetBytecodeDispatchCallDescriptor(
125  graph_zone_, descriptor, descriptor.GetStackParameterCount());
126  auto ts_call_descriptor =
127  TSCallDescriptor::Create(call_descriptor, compiler::CanThrow::kNo,
128  compiler::LazyDeoptOnThrow::kNo, graph_zone_);
129 
130  std::initializer_list<const OpIndex> arguments{args...};
131  __ TailCall(target, base::VectorOf(arguments), ts_call_descriptor);
132  }

134  V<WordPtr> Advance(ConstOrV<WordPtr> delta) {
135  V<WordPtr> next_offset = __ WordPtrAdd(BytecodeOffset(), delta);
136  bytecode_offset_ = next_offset;
137  return next_offset;
138  }
139 
140  V<Object> LoadRegister(Register reg) {
141  const int offset = reg.ToOperand() * kSystemPointerSize;
142  return __ LoadOffHeap(GetInterpretedFramePointer(), offset,
143  MemoryRepresentation::AnyTagged());
144  }
145 
146  V<WordPtr> GetInterpretedFramePointer() {
147  if (!interpreted_frame_pointer_.Get().valid()) {
148  interpreted_frame_pointer_ = __ ParentFramePointer();
149  } else if (Bytecodes::MakesCallAlongCriticalPath(data_.bytecode) &&
150  data_.made_call && data_.reloaded_frame_ptr) {
151  interpreted_frame_pointer_ = __ ParentFramePointer();
152  data_.reloaded_frame_ptr = true;
153  }
154  return interpreted_frame_pointer_;
155  }
156 
157  V<WordPtr> BytecodeOffset() {
158  if (Bytecodes::MakesCallAlongCriticalPath(data_.bytecode) &&
159  data_.made_call && (bytecode_offset_ == bytecode_offset_parameter_)) {
160  bytecode_offset_ = ReloadBytecodeOffset();
161  }
162  return bytecode_offset_;
163  }
164 
165  V<WordPtr> ReloadBytecodeOffset() {
166  V<WordPtr> offset = LoadAndUntagRegister(Register::bytecode_offset());
167  if (operand_scale() == OperandScale::kSingle) {
168  return offset;
169  }
170 
171  // Add one to the offset such that it points to the actual bytecode rather
172  // than the Wide / ExtraWide prefix bytecode.
173  return __ WordPtrAdd(offset, 1);
174  }
175 
176  V<Word32> LoadFromBytecodeArrayAt(MemoryRepresentation loaded_rep,
177  V<WordPtr> bytecode_offset,
178  int additional_offset = 0) {
179  return __ Load(BytecodeArrayTaggedPointer(), bytecode_offset,
180  LoadOp::Kind::TaggedBase(), loaded_rep,
181  additional_offset + kHeapObjectTag);
182  }
183 
184  V<WordPtr> LoadBytecode(V<WordPtr> bytecode_offset) {
185  V<Word32> bytecode = __ Load(BytecodeArrayTaggedPointer(), bytecode_offset,
186  LoadOp::Kind::TaggedBase(),
187  MemoryRepresentation::Uint8(), kHeapObjectTag);
188  return __ ChangeUint32ToUintPtr(bytecode);
189  }
190 
191  V<WordPtr> LoadAndUntagRegister(Register reg) {
192  V<WordPtr> base = GetInterpretedFramePointer();
193  int index = reg.ToOperand() * kSystemPointerSize;
194  if (SmiValuesAre32Bits()) {
195 #if V8_TARGET_LITTLE_ENDIAN
196  index += 4;
197 #endif
198  return __ ChangeInt32ToIntPtr(
199  __ LoadOffHeap(base, index, MemoryRepresentation::Int32()));
200  } else {
201  return __ ChangeInt32ToIntPtr(__ UntagSmi(
202  __ LoadOffHeap(base, index, MemoryRepresentation::TaggedSigned())));
203  }
204  }
205 
206  // TODO(nicohartmann): Consider providing a V<ExternalReference>.
207  V<WordPtr> DispatchTablePointer() {
208  if (Bytecodes::MakesCallAlongCriticalPath(data_.bytecode) &&
209  data_.made_call && (dispatch_table_ == dispatch_table_parameter_)) {
210  dispatch_table_ = __ ExternalConstant(
211  ExternalReference::interpreter_dispatch_table_address(isolate_));
212  }
213  return dispatch_table_;
214  }
215 
216  V<BytecodeArray> BytecodeArrayTaggedPointer() {
217  // Force a re-load of the bytecode array after every call in case the
218  // debugger has been activated.
219  if (!data_.bytecode_array_valid) {
220  bytecode_array_ = LoadRegister(Register::bytecode_array());
221  data_.bytecode_array_valid = true;
222  }
223  return V<BytecodeArray>::Cast(bytecode_array_);
224  }
225 
226  V<Word32> BytecodeOperandIdxInt32(int operand_index) {
227  DCHECK_EQ(OperandType::kIdx,
228  Bytecodes::GetOperandType(data_.bytecode, operand_index));
229  OperandSize operand_size = Bytecodes::GetOperandSize(
230  data_.bytecode, operand_index, operand_scale());
231  return BytecodeUnsignedOperand(operand_index, operand_size);
232  }
233 
234  V<Word32> BytecodeUnsignedOperand(int operand_index,
235  OperandSize operand_size) {
236  return BytecodeOperand(operand_index, operand_size);
237  }
238 
239  V<Word32> BytecodeOperand(int operand_index, OperandSize operand_size) {
240  DCHECK_LT(operand_index, Bytecodes::NumberOfOperands(bytecode()));
241  DCHECK_EQ(operand_size, Bytecodes::GetOperandSize(bytecode(), operand_index,
242  operand_scale()));
243  MemoryRepresentation loaded_rep;
244  switch (operand_size) {
245  case OperandSize::kByte:
246  loaded_rep = MemoryRepresentation::Uint8();
247  break;
248  case OperandSize::kShort:
249  loaded_rep = MemoryRepresentation::Uint16();
250  break;
251  case OperandSize::kQuad:
252  loaded_rep = MemoryRepresentation::Uint32();
253  break;
254  case OperandSize::kNone:
255  UNREACHABLE();
256  }
257  return LoadFromBytecodeArrayAt(loaded_rep, BytecodeOffset(),
258  OperandOffset(operand_index));
259  }
260 
261  int OperandOffset(int operand_index) const {
262  return Bytecodes::GetOperandOffset(bytecode(), operand_index,
263  operand_scale());
264  }
265 
266  private:
267  Bytecode bytecode() const { return data_.bytecode; }
268  OperandScale operand_scale() const { return data_.operand_scale; }
269 
270  int CurrentBytecodeSize() const {
271  return Bytecodes::Size(data_.bytecode, data_.operand_scale);
272  }
273 
274  void TrackRegisterUse(ImplicitRegisterUse use) {
275  data_.implicit_register_use = data_.implicit_register_use | use;
276  }
277 
278  Isolate* isolate_ = __ data() -> isolate();
279  ZoneWithName<compiler::kGraphZoneName>& graph_zone_ =
280  __ data() -> graph_zone();
281  BytecodeHandlerData& data_ = *__ data() -> bytecode_handler_data();
282  // TODO(nicohartmann): Replace with Var<T>s.
283  OpIndex bytecode_offset_parameter_;
284  OpIndex dispatch_table_parameter_;
285  template <typename T>
286  using Var = compiler::turboshaft::Var<T, assembler_t>;
287  Var<Object> accumulator_{this};
288  Var<WordPtr> interpreted_frame_pointer_{this};
289  Var<WordPtr> bytecode_offset_{this};
290  Var<Object> bytecode_array_{this};
291  Var<WordPtr> dispatch_table_{this};
292 };

313  void EmitBytecodeHandlerProlog() {
314  // Bind an entry block.
315  __ Bind(__ NewBlock());
316  // Initialize parameters.
317  V<Object> acc = __ template Parameter<Object>(

319  V<WordPtr> bytecode_offset = __ template Parameter<WordPtr>(
320  InterpreterDispatchDescriptor::kBytecodeOffset);
321  V<BytecodeArray> bytecode_array = __ template Parameter<BytecodeArray>(
322  InterpreterDispatchDescriptor::kBytecodeArray);
323  V<WordPtr> dispatch_table = __ template Parameter<WordPtr>(
324  InterpreterDispatchDescriptor::kDispatchTable);
325  __ InitializeParameters(acc, bytecode_array, bytecode_offset,
326  dispatch_table);
