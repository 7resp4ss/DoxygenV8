121 class LanguageNames : public LocaleDisplayNamesCommon {
122  public:
123  LanguageNames(const icu::Locale& locale, JSDisplayNames::Style style,
124  bool fallback, bool dialect)
125  : LocaleDisplayNamesCommon(locale, style, fallback, dialect) {}
126 
127  ~LanguageNames() override = default;
128 
129  const char* type() const override { return "language"; }
130 
131  Maybe<icu::UnicodeString> of(Isolate* isolate,
132  const char* code) const override {
133  UErrorCode status = U_ZERO_ERROR;
134  // 1.a If code does not match the unicode_language_id production, throw a
135  // RangeError exception.
136  icu::Locale tagLocale = icu::Locale::forLanguageTag(code, status);
137  icu::Locale l(tagLocale.getBaseName());
138  if (U_FAILURE(status) || tagLocale != l ||
139  !JSLocale::StartsWithUnicodeLanguageId(code)) {
140  THROW_NEW_ERROR_RETURN_VALUE(
141  isolate, NewRangeError(MessageTemplate::kInvalidArgument),
142  Nothing<icu::UnicodeString>());
143  }
144 
145  // 1.b If IsStructurallyValidLanguageTag(code) is false, throw a RangeError
146  // exception.
147 
148  // 1.c Set code to CanonicalizeUnicodeLocaleId(code).
149  l.canonicalize(status);
150  std::string checked = l.toLanguageTag<std::string>(status);
151 
152  if (U_FAILURE(status)) {
153  THROW_NEW_ERROR_RETURN_VALUE(
154  isolate, NewRangeError(MessageTemplate::kInvalidArgument),
155  Nothing<icu::UnicodeString>());
156  }
157 
158  icu::UnicodeString result;
159  locale_display_names()->localeDisplayName(checked.c_str(), result);
160 
161  return Just(result);
162  }
163 };

308 UDateTimePatternField StringToUDateTimePatternField(const char* code) {
309  switch (code[0]) {
310  case 'd':
311  if (strcmp(code, "day") == 0) return UDATPG_DAY_FIELD;
312  if (strcmp(code, "dayPeriod") == 0) return UDATPG_DAYPERIOD_FIELD;
313  break;
314  case 'e':
315  if (strcmp(code, "era") == 0) return UDATPG_ERA_FIELD;
316  break;
317  case 'h':
318  if (strcmp(code, "hour") == 0) return UDATPG_HOUR_FIELD;
319  break;
320  case 'm':
321  if (strcmp(code, "minute") == 0) return UDATPG_MINUTE_FIELD;
322  if (strcmp(code, "month") == 0) return UDATPG_MONTH_FIELD;
323  break;
324  case 'q':
325  if (strcmp(code, "quarter") == 0) return UDATPG_QUARTER_FIELD;
326  break;
327  case 's':
328  if (strcmp(code, "second") == 0) return UDATPG_SECOND_FIELD;
329  break;
330  case 't':
331  if (strcmp(code, "timeZoneName") == 0) return UDATPG_ZONE_FIELD;
332  break;
333  case 'w':
334  if (strcmp(code, "weekOfYear") == 0) return UDATPG_WEEK_OF_YEAR_FIELD;
335  if (strcmp(code, "weekday") == 0) return UDATPG_WEEKDAY_FIELD;
336  break;
337  case 'y':
338  if (strcmp(code, "year") == 0) return UDATPG_YEAR_FIELD;
339  break;
340  default:
341  break;
342  }
343  return UDATPG_FIELD_COUNT;
344 }

380 DisplayNamesInternal* CreateInternal(const icu::Locale& locale,
381  JSDisplayNames::Style style, Type type,
382  bool fallback, bool dialect) {
383  switch (type) {
384  case Type::kLanguage:
385  return new LanguageNames(locale, style, fallback, dialect);
386  case Type::kRegion:
387  return new RegionNames(locale, style, fallback, false);
388  case Type::kScript:
389  return new ScriptNames(locale, style, fallback, false);
390  case Type::kCurrency:
391  return new CurrencyNames(locale, style, fallback, false);
392  case Type::kCalendar:

394  case Type::kDateTimeField:
395  return new DateTimeFieldNames(locale, style, fallback);
396  default:
397  UNREACHABLE();

403 // ecma402 #sec-Intl.DisplayNames
404 MaybeDirectHandle<JSDisplayNames> JSDisplayNames::New(
405  Isolate* isolate, DirectHandle<Map> map, DirectHandle<Object> locales,
406  DirectHandle<Object> input_options) {
407  const char* service = "Intl.DisplayNames";
408  Factory* factory = isolate->factory();
409 
410  DirectHandle<JSReceiver> options;
411  // 3. Let requestedLocales be ? CanonicalizeLocaleList(locales).
412  Maybe<std::vector<std::string>> maybe_requested_locales =
413  Intl::CanonicalizeLocaleList(isolate, locales);
414  MAYBE_RETURN(maybe_requested_locales, DirectHandle<JSDisplayNames>());
415  std::vector<std::string> requested_locales =
416  maybe_requested_locales.FromJust();
417 
418  // 4. Let options be ? GetOptionsObject(options).
419  ASSIGN_RETURN_ON_EXCEPTION(isolate, options,
420  GetOptionsObject(isolate, input_options, service));
421 
422  // Note: No need to create a record. It's not observable.
423  // 5. Let opt be a new Record.
424 
425  // 6. Let localeData be %DisplayNames%.[[LocaleData]].
426 
427  // 7. Let matcher be ? GetOption(options, "localeMatcher", "string",
428  // "lookup", "best fit" , "best fit").
429  Maybe<Intl::MatcherOption> maybe_locale_matcher =
430  Intl::GetLocaleMatcher(isolate, options, service);
431  MAYBE_RETURN(maybe_locale_matcher, MaybeDirectHandle<JSDisplayNames>());
432 
433  // 8. Set opt.[[localeMatcher]] to matcher.
434  Intl::MatcherOption matcher = maybe_locale_matcher.FromJust();
435 
436  // ecma402/#sec-Intl.DisplayNames-internal-slots
437  // The value of the [[RelevantExtensionKeys]] internal slot is
438  // .
439  // 9. Let r be ResolveLocale(%DisplayNames%.[[AvailableLocales]],
440  // requestedLocales, opt, %DisplayNames%.[[RelevantExtensionKeys]]).
441  Intl::ResolvedLocale r;
442  if (!Intl::ResolveLocale(isolate, JSDisplayNames::GetAvailableLocales(),
443  requested_locales, matcher, {})
444  .To(&r)) {
445  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
446  }
447 
448  icu::Locale icu_locale = r.icu_locale;
449 
450  // 10. Let s be ? GetOption(options, "style", "string",
451  // "long", "short", "narrow", "long").
452  Maybe<Style> maybe_style = GetStringOption<Style>(
453  isolate, options, "style", service,
454  std::to_array<const std::string_view>({"long", "short", "narrow"}),
455  std::array{Style::kLong, Style::kShort, Style::kNarrow}, Style::kLong);
456  MAYBE_RETURN(maybe_style, MaybeDirectHandle<JSDisplayNames>());
457  Style style_enum = maybe_style.FromJust();
458 
459  // 11. Set displayNames.[[Style]] to style.
460 
461  // 12. Let type be ? GetOption(options, "type", "string", "language",
462  // "region", "script", "currency" , "calendar", "dateTimeField", "unit",
463  // undefined).
464  Maybe<Type> maybe_type = GetStringOption<Type>(
465  isolate, options, "type", service,
466  std::to_array<const std::string_view>({"language", "region", "script",
467  "currency", "calendar",
468  "dateTimeField"}),
469  std::array{Type::kLanguage, Type::kRegion, Type::kScript, Type::kCurrency,
470  Type::kCalendar, Type::kDateTimeField},
471  Type::kUndefined);
472  MAYBE_RETURN(maybe_type, MaybeDirectHandle<JSDisplayNames>());
473  Type type_enum = maybe_type.FromJust();
474 
475  // 13. If type is undefined, throw a TypeError exception.
476  if (type_enum == Type::kUndefined) {
477  THROW_NEW_ERROR(isolate, NewTypeError(MessageTemplate::kInvalidArgument));
478  }
479 
480  // 14. Set displayNames.[[Type]] to type.
481 
482  // 15. Let fallback be ? GetOption(options, "fallback", "string",
483  // "code", "none" , "code").
484  Maybe<Fallback> maybe_fallback = GetStringOption<Fallback>(
485  isolate, options, "fallback", service,
486  std::to_array<const std::string_view>({"code", "none"}),
487  std::array{Fallback::kCode, Fallback::kNone}, Fallback::kCode);
488  MAYBE_RETURN(maybe_fallback, MaybeDirectHandle<JSDisplayNames>());
489  Fallback fallback_enum = maybe_fallback.FromJust();
490 
491  // 16. Set displayNames.[[Fallback]] to fallback.
492 
493  LanguageDisplay language_display_enum = LanguageDisplay::kDialect;
494  // 24. Let languageDisplay be ? GetOption(options, "languageDisplay",
495  // "string", "dialect", "standard" , "dialect").
496  Maybe<LanguageDisplay> maybe_language_display =
497  GetStringOption<LanguageDisplay>(
498  isolate, options, "languageDisplay", service,
499  std::to_array<const std::string_view>({"dialect", "standard"}),
500  std::array{LanguageDisplay::kDialect, LanguageDisplay::kStandard},
501  LanguageDisplay::kDialect);
502  MAYBE_RETURN(maybe_language_display, MaybeDirectHandle<JSDisplayNames>());
503  // 25. If type is "language", then
504  if (type_enum == Type::kLanguage) {
505  // a. Set displayNames.[[LanguageDisplay]] to languageDisplay.
506  language_display_enum = maybe_language_display.FromJust();
507  }
508 
509  // Set displayNames.[[Fallback]] to fallback.
510 
511  // 17. Set displayNames.[[Locale]] to the value of r.[[Locale]].
512 
513  // Let dataLocale be r.[[dataLocale]].
514 
515  // Let dataLocaleData be localeData.[[<dataLocale>]].
516 
517  // Let types be dataLocaleData.[[types]].
518 
519  // Assert: types is a Record (see 1.3.3).
520 
521  // Let typeFields be types.[[<type>]].
522 
523  // Assert: typeFields is a Record (see 1.3.3).
524 
525  // Let styleFields be typeFields.[[<style>]].
526 
527  // Assert: styleFields is a Record (see 1.3.3).
528 
529  // Set displayNames.[[Fields]] to styleFields.
530 
531  std::shared_ptr<DisplayNamesInternal> internal{CreateInternal(
532  icu_locale, style_enum, type_enum, fallback_enum == Fallback::kCode,
533  language_display_enum == LanguageDisplay::kDialect)};
534  if (internal == nullptr) {
535  THROW_NEW_ERROR(isolate, NewTypeError(MessageTemplate::kIcuError));
536  }
537 
538  DirectHandle<Managed<DisplayNamesInternal>> managed_internal =
539  Managed<DisplayNamesInternal>::From(isolate, 0, std::move(internal));
540 
541  DirectHandle<JSDisplayNames> display_names =
542  Cast<JSDisplayNames>(factory->NewFastOrSlowJSObjectFromMap(map));
543  display_names->set_flags(0);
544  display_names->set_style(style_enum);
545  display_names->set_fallback(fallback_enum);
546  display_names->set_language_display(language_display_enum);
547 
548  DisallowGarbageCollection no_gc;
549  display_names->set_internal(*managed_internal);
550 
551  // Return displayNames.
552  return display_names;
553 }

556 DirectHandle<JSObject> JSDisplayNames::ResolvedOptions(
557  Isolate* isolate, DirectHandle<JSDisplayNames> display_names) {
558  Factory* factory = isolate->factory();
559  // 4. Let options be ! ObjectCreate(%ObjectPrototype%).
560  DirectHandle<JSObject> options =
561  factory->NewJSObject(isolate->object_function());
562 
563  DisplayNamesInternal* internal = display_names->internal()->raw();
564 
565  Maybe<std::string> maybe_locale = Intl::ToLanguageTag(internal->locale());
566  DCHECK(maybe_locale.IsJust());
567  DirectHandle<String> locale = isolate->factory()->NewStringFromAsciiChecked(
568  maybe_locale.FromJust().c_str());
569  DirectHandle<String> style = display_names->StyleAsString(isolate);
570  DirectHandle<String> type =
571  factory->NewStringFromAsciiChecked(internal->type());
572  DirectHandle<String> fallback = display_names->FallbackAsString(isolate);
573  DirectHandle<String> language_display =
574  display_names->LanguageDisplayAsString(isolate);
575 
576  Maybe<bool> maybe_create_locale = JSReceiver::CreateDataProperty(
577  isolate, options, factory->locale_string(), locale, Just(kDontThrow));
578  DCHECK(maybe_create_locale.FromJust());
579  USE(maybe_create_locale);
580 
581  Maybe<bool> maybe_create_style = JSReceiver::CreateDataProperty(
582  isolate, options, factory->style_string(), style, Just(kDontThrow));
583  DCHECK(maybe_create_style.FromJust());
584  USE(maybe_create_style);
585 
586  Maybe<bool> maybe_create_type = JSReceiver::CreateDataProperty(
587  isolate, options, factory->type_string(), type, Just(kDontThrow));
588  DCHECK(maybe_create_type.FromJust());
589  USE(maybe_create_type);
590 
591  Maybe<bool> maybe_create_fallback = JSReceiver::CreateDataProperty(
592  isolate, options, factory->fallback_string(), fallback, Just(kDontThrow));
593  DCHECK(maybe_create_fallback.FromJust());
594  USE(maybe_create_fallback);
595 
596  if (std::strcmp("language", internal->type()) == 0) {
597  Maybe<bool> maybe_create_language_display =
598  JSReceiver::CreateDataProperty(isolate, options,
599  factory->languageDisplay_string(),
600  language_display, Just(kDontThrow));
601  DCHECK(maybe_create_language_display.FromJust());
602  USE(maybe_create_language_display);
603  }
604 
605  return options;
606 }

608 // ecma402 #sec-Intl.DisplayNames.prototype.of
609 MaybeDirectHandle<Object> JSDisplayNames::Of(
610  Isolate* isolate, DirectHandle<JSDisplayNames> display_names,
611  Handle<Object> code_obj) {
612  DirectHandle<String> code;
613  ASSIGN_RETURN_ON_EXCEPTION(isolate, code,
614  Object::ToString(isolate, code_obj));
615  DisplayNamesInternal* internal = display_names->internal()->raw();
616  Maybe<icu::UnicodeString> maybe_result =
617  internal->of(isolate, code->ToCString().get());
618  MAYBE_RETURN(maybe_result, DirectHandle<Object>());
619  icu::UnicodeString result = maybe_result.FromJust();
620  if (result.isBogus()) {
621  return isolate->factory()->undefined_value();
622  }
623  return Intl::ToString(isolate, result).ToHandleChecked();
624 }
