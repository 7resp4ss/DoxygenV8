200 // Representation of a thread in the interpreter.
201 class V8_EXPORT_PRIVATE WasmInterpreterThread {
202  public:
203  // State machine for a WasmInterpreterThread:
204  //
205  // STOPPED
206  // |
207  // Run()|
208  // V
209  // RUNNING <-----------------------------------+
210  // | |
211  // | |
212  // +-------------+---------------+---------------+ |
213  // |Stop() |Trap() |Finish() | |
214  // V V V V |
215  // STOPPED <---- TRAPPED FINISHED EH_UNWINDING --+
216  // ^ |
217  // +---------------------------------------------+
218  //
219  // In more detail:
220  // - For each loaded instance, an InterpreterHandler is created that owns a
221  // WasmInterpreter that owns a WasmInterpreterRuntime object.
222  //
223  // - The WasmInterpreterThread is created in STOPPED state.
224  //
225  // - InterpreterHandle::Execute(func_index, ...) executes Wasm code in
226  // the interpreter:
227  // - WasmInterpreter::BeginExecution ->
228  // WasmInterpreterRuntime::BeginExecution ->
229  // WasmInterpreterThread::StartActivation() -> Run() -> RUNNING
230  // state.
231  // - WasmInterpreter::ContinueExecution ->
232  // WasmInterpreterRuntime::ContinueExecution ->
233  // WasmInterpreterRuntime::ExecuteFunction
234  //
235  // WasmInterpreterRuntime::ExecuteFunction(..., func_index, ...) executes a
236  // specific Wasm function.
237  // If 'func_index' indicates an imported function, and the call fails ->
238  // Stop() -> STOPPED state.
239  // If 'func_index' indicates an not-imported function, we start executing a
240  // sequence of instruction handlers. One of these handlers can cause a
241  // Trap() -> TRAPPED state.
242  // From these instructions sequence we can make several kinds of direct or
243  // indirect wasm calls to:
244  // . An external JS function ->
245  // WasmInterpreterRuntime::CallExternalJSFunction() ->
246  // If the call fails -> Stop() -> STOPPED state.
247  // . A Wasm function in the same module instance, recursively calling
248  // WasmInterpreterRuntime::ExecuteFunction().
249  // . A Wasm function in a different module instance. In this case we
250  // recusively call InterpreterHandle::Execute with the
251  // InterpreterHandle of that different instance. If the call fails ->
252  // Stop() -> STOPPED state.
253  //
254  // After WasmInterpreterRuntime::ExecuteFunction() completes, if we ended up
255  // in the TRAPPED state we raise a JS exception -> RaiseException() ->
256  // Stop() -> STOPPED state.
257  //
258  // If an exception can be handled by Wasm code, according to the Wasm
259  // Exception Handling proposal, the thread can go to the EH_UNWINDING state
260  // while looking for a Wasm function in the call stack that has a {catch}
261  // instruction that can handle that exception. If no such catch handler is
262  // found, the thread goes to STOPPED.
263  //
264  // If we are running the WasmInterpreter of instance A and we can call
265  // from a function of a different instance B (via
266  // InterpreterHandle::Execute()) the execution of instance A "suspends"
267  // waiting for the execution in the WasmInterpreter of instance B to
268  // complete. Instance B can call back into instance A, and so on... This
269  // means that in the call stack we might have a sequence of stack frames for
270  // the WasmInterpreter A followed by frames for instance B followed by
271  // more frames of instance A.
272  // To manage this case WasmInterpreterThread maintains a stack of
273  // Activations, which represents the set of StackFrames for a given module
274  // instance. Only when the last active Activation terminates we call
275  // Finish() -> FINISHED state.
276 
277  enum State { STOPPED, RUNNING, FINISHED, TRAPPED, EH_UNWINDING };

708 // {OperatorMode}s are used for the
709 // v8_flags.drumbrake_register_optimization. The prototype of instruction
710 // handlers contains two arguments int64_t r0 and double fp0 that can be used to
711 // pass in an integer or floating-point register the values that is at the top
712 // of the Wasm execution stack.
713 //
714 // For this reasons, whenever possible we define four different versions of each
715 // instruction handler, all identified by the following prefixes:
716 //
717 // - r2r_*: Wasm instruction handlers called when the stack top value is in a
718 // register and that put the result in a register.
719 // - r2s_*: Wasm instruction handlers called when the stack top value is in a
720 // register and that push the result on the stack.
721 // - s2r_*: Wasm instruction handlers called when the stack top value is not in
722 // a register and that put the result in a register.
723 // - s2s_*: Wasm instruction handlers called when the stack top value is not in
724 // a register and that push the result on the stack.
725 //
726 enum OperatorMode { kR2R = 0, kR2S, kS2R, kS2S };
727 static const size_t kOperatorModeCount = 4;

729 // {RegMode} and {RegModeTransform} specifies how an instruction handler can
730 // leverage the --drumbrake-register-optimization.
731 //
732 // {RegModeTransform} defines a pair of {RegMode}s, that specify whether an
733 // instruction handler can take its input or provide its output from the stack
734 // or from registers.
735 //
736 // For example:
737 // {kF32Reg, kI32Reg}, // 0x5b F32Eq
738 // declares that the F32Eq instruction handler can read the stack top value from
739 // a floating point register as a F32 and pass the result to the next handler in
740 // an integer register as an I32, so saving one stack pop and one stack push
741 // operations.
742 enum class RegMode {
743  kNoReg, // The instruction handler only gets inputs from stack slots or
744  // provide the result into a stack slot.
745 
746  kI32Reg, // The instruction handler can be optimized to work with the integer
747  kI64Reg, // register 'r0'.
748 
749  kF32Reg, // The instruction handler can be optimized to work with the
750  kF64Reg, // floating point register 'fp0'.
751 
752  kAnyReg, // The instruction handler can be optimized to work either with the
753  // integer or fp register; the specific register depends on the
754  // type of the type of the value at the top of the stack. This is
755  // used for instructions like 'drop', 'select' and 'local.set.
756 };
