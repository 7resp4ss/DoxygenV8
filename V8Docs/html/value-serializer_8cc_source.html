51 // WARNING: This serialization format MUST remain backward compatible!
52 //
53 // This format is used by APIs to persist values to disk, e.g. IndexedDB.
54 //
55 // Backward compatibility means that when the format changes, deserializing
56 // valid values in the older format must behave identically as before the
57 // change. To maintain compatibility, either a format change does not affect the
58 // deserializing behavior of valid values in the older format, or the
59 // kLatestVersion constant is bumped.
60 //
61 // Adding a new tag is backwards compatible because no valid serialized value in
62 // older formats would contain the new object tag.
63 //
64 // On the other hand, changing the format of a particular tag is backwards
65 // incompatible and the version must be bumped. For example, a JSArrayBufferView
66 // tag prior to version 14 was followed by the sub-tag, the byte offset, and the
67 // byte length. Starting with version 14, a JSArrayBufferView tag is followed by
68 // the sub-tag, the byte offset, the byte length, and flags. Due the addition of
69 // flags, older valid serialized values for JSArrayBufferViews would be
70 // misinterpreted by newer deserializers. This requires the version to be bumped
71 // and the deserializer to handle both the old and new formats depending on the
72 // version.
73 
74 // Version 9: (imported from Blink)
75 // Version 10: one-byte (Latin-1) strings
76 // Version 11: properly separate undefined from the hole in arrays
77 // Version 12: regexp and string objects share normal string encoding
78 // Version 13: host objects have an explicit tag (rather than handling all
79 // unknown tags)
80 // Version 14: flags for JSArrayBufferViews
81 // Version 15: support for shared objects with an explicit tag
82 //
83 // WARNING: Increasing this value is a change which cannot safely be rolled
84 // back without breaking compatibility with data stored on disk. It is
85 // strongly recommended that you do not make such changes near a release
86 // milestone branch point.
87 //
88 // Recent changes are routinely reverted in preparation for branch, and this
89 // has been the cause of at least one bug in the past.
90 static const uint32_t kLatestVersion = 15;
91 static_assert(kLatestVersion == v8::CurrentValueSerializerFormatVersion(),
92  "Exported format version must match latest version.");

105 template <typename T>
106 static size_t BytesNeededForVarint(T value) {
107  static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
108  "Only unsigned integer types can be written as varints.");
109  size_t result = 0;
110  do {
111  result++;
112  value >>= 7;
113  } while (value);
114  return result;
115 }

333 template <typename T>
334 void ValueSerializer::WriteVarint(T value) {
335  // Writes an unsigned integer as a base-128 varint.
336  // The number is written, 7 bits at a time, from the least significant to the
337  // most significant 7 bits. Each byte, except the last, has the MSB set.
338  // See also https://developers.google.com/protocol-buffers/docs/encoding
339  static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
340  "Only unsigned integer types can be written as varints.");
341  uint8_t stack_buffer[sizeof(T) * 8 / 7 + 1];
342  uint8_t* next_byte = &stack_buffer[0];
343  do {
344  *next_byte = (value & 0x7F) | 0x80;
345  next_byte++;
346  value >>= 7;
347  } while (value);
348  *(next_byte - 1) &= 0x7F;
349  WriteRawBytes(stack_buffer, next_byte - stack_buffer);
350 }

352 template <typename T>
353 void ValueSerializer::WriteZigZag(T value) {
354  // Writes a signed integer as a varint using ZigZag encoding (i.e. 0 is
355  // encoded as 0, -1 as 1, 1 as 2, -2 as 3, and so on).
356  // See also https://developers.google.com/protocol-buffers/docs/encoding
357  // Note that this implementation relies on the right shift being arithmetic.
358  static_assert(std::is_integral_v<T> && std::is_signed_v<T>,
359  "Only signed integer types can be written as zigzag.");
360  using UnsignedT = std::make_unsigned_t<T>;
361  WriteVarint((static_cast<UnsignedT>(value) << 1) ^
362  (value >> (8 * sizeof(T) - 1)));
363 }

401 Maybe<uint8_t*> ValueSerializer::ReserveRawBytes(size_t bytes) {
402  size_t old_size = buffer_size_;
403  size_t new_size = old_size + bytes;
404  if (V8_UNLIKELY(new_size > buffer_capacity_)) {
405  bool ok;
406  if (!ExpandBuffer(new_size).To(&ok)) {
407  return Nothing<uint8_t*>();
408  }
409  }
410  buffer_size_ = new_size;
411  return Just(&buffer_[old_size]);
412 }

414 Maybe<bool> ValueSerializer::ExpandBuffer(size_t required_capacity) {
415  DCHECK_GT(required_capacity, buffer_capacity_);
416  size_t requested_capacity =
417  std::max(required_capacity, buffer_capacity_ * 2) + 64;
418  size_t provided_capacity = 0;
419  void* new_buffer = nullptr;
420  if (delegate_) {
421  new_buffer = delegate_->ReallocateBufferMemory(buffer_, requested_capacity,
422  &provided_capacity);
423  } else {
424  new_buffer = base::Realloc(buffer_, requested_capacity);
425  provided_capacity = requested_capacity;
426  }
427  if (new_buffer) {
428  DCHECK(provided_capacity >= requested_capacity);
429  buffer_ = reinterpret_cast<uint8_t*>(new_buffer);
430  buffer_capacity_ = provided_capacity;
431  return Just(true);
432  } else {
433  out_of_memory_ = true;
434  return Nothing<bool>();
435  }
436 }

468 Maybe<bool> ValueSerializer::WriteObject(DirectHandle<Object> object) {
469  // There is no sense in trying to proceed if we've previously run out of
470  // memory. Bail immediately, as this likely implies that some write has
471  // previously failed and so the buffer is corrupt.
472  if (V8_UNLIKELY(out_of_memory_)) return ThrowIfOutOfMemory();
473 
474  if (IsSmi(*object)) {
475  WriteSmi(Cast<Smi>(*object));
476  return ThrowIfOutOfMemory();
477  }
478 
479  DCHECK(IsHeapObject(*object));
480  InstanceType instance_type =
481  Cast<HeapObject>(*object)->map(isolate_)->instance_type();
482  switch (instance_type) {
483  case ODDBALL_TYPE:
484  WriteOddball(Cast<Oddball>(*object));
485  return ThrowIfOutOfMemory();
486  case HEAP_NUMBER_TYPE:
487  WriteHeapNumber(Cast<HeapNumber>(*object));
488  return ThrowIfOutOfMemory();
489  case BIGINT_TYPE:
490  WriteBigInt(Cast<BigInt>(*object));
491  return ThrowIfOutOfMemory();
492  case JS_TYPED_ARRAY_TYPE:
493  case JS_DATA_VIEW_TYPE:
494  case JS_RAB_GSAB_DATA_VIEW_TYPE: {
495  // Despite being JSReceivers, these have their wrapped buffer serialized
496  // first. That makes this logic a little quirky, because it needs to
497  // happen before we assign object IDs.
498  // TODO(jbroman): It may be possible to avoid materializing a typed
499  // array's buffer here.
500  DirectHandle<JSArrayBufferView> view = Cast<JSArrayBufferView>(object);
501  if (!id_map_.Find(view) && !treat_array_buffer_views_as_host_objects_) {
502  DirectHandle<JSArrayBuffer> buffer(
503  InstanceTypeChecker::IsJSTypedArray(instance_type)
504  ? Cast<JSTypedArray>(view)->GetBuffer(isolate_)
505  : direct_handle(Cast<JSArrayBuffer>(view->buffer()), isolate_));
506  if (!WriteJSReceiver(buffer).FromMaybe(false)) return Nothing<bool>();
507  }
508  return WriteJSReceiver(view);
509  }
510  default:
511  if (InstanceTypeChecker::IsString(instance_type)) {
512  WriteString(Cast<String>(object));
513  return ThrowIfOutOfMemory();
514  } else if (InstanceTypeChecker::IsJSReceiver(instance_type)) {
515  return WriteJSReceiver(Cast<JSReceiver>(object));
516  } else {
517  return ThrowDataCloneError(MessageTemplate::kDataCloneError, object);
518  }
519  }
520 }

559 void ValueSerializer::WriteString(DirectHandle<String> string) {
560  string = String::Flatten(isolate_, string);
561  DisallowGarbageCollection no_gc;
562  String::FlatContent flat = string->GetFlatContent(no_gc);
563  DCHECK(flat.IsFlat());
564  if (flat.IsOneByte()) {
565  base::Vector<const uint8_t> chars = flat.ToOneByteVector();
566  WriteTag(SerializationTag::kOneByteString);
567  WriteOneByteString(chars);
568  } else if (flat.IsTwoByte()) {
569  base::Vector<const base::uc16> chars = flat.ToUC16Vector();
570  uint32_t byte_length = chars.length() * sizeof(base::uc16);
571  // The existing reading code expects 16-byte strings to be aligned.
572  if ((buffer_size_ + 1 + BytesNeededForVarint(byte_length)) & 1)
573  WriteTag(SerializationTag::kPadding);
574  WriteTag(SerializationTag::kTwoByteString);
575  WriteTwoByteString(chars);
576  } else {
577  UNREACHABLE();
578  }
579 }

581 Maybe<bool> ValueSerializer::WriteJSReceiver(
582  DirectHandle<JSReceiver> receiver) {
583  // If the object has already been serialized, just write its ID.
584  auto find_result = id_map_.FindOrInsert(receiver);
585  if (find_result.already_exists) {
586  WriteTag(SerializationTag::kObjectReference);
587  WriteVarint(*find_result.entry - 1);
588  return ThrowIfOutOfMemory();
589  }
590 
591  // Otherwise, allocate an ID for it.
592  uint32_t id = next_id_++;
593  *find_result.entry = id + 1;
594 
595  // Eliminate callable and exotic objects, which should not be serialized.
596  InstanceType instance_type = receiver->map()->instance_type();
597  if (IsCallable(*receiver) ||
598  (IsSpecialReceiverInstanceType(instance_type) &&
599  instance_type != JS_SPECIAL_API_OBJECT_TYPE
600 #if V8_ENABLE_WEBASSEMBLY
601  && instance_type != WASM_STRUCT_TYPE && instance_type != WASM_ARRAY_TYPE
602 #endif
603  )) {
604  return ThrowDataCloneError(MessageTemplate::kDataCloneError, receiver);
605  }
606 
607  // If we are at the end of the stack, abort. This function may recurse.
608  STACK_CHECK(isolate_, Nothing<bool>());
609 
610  HandleScope scope(isolate_);
611  switch (instance_type) {
612  case JS_ARRAY_TYPE:
613  return WriteJSArray(Cast<JSArray>(receiver));
614  case JS_ARRAY_ITERATOR_PROTOTYPE_TYPE:
615  case JS_ITERATOR_PROTOTYPE_TYPE:
616  case JS_MAP_ITERATOR_PROTOTYPE_TYPE:
617  case JS_OBJECT_PROTOTYPE_TYPE:
618  case JS_OBJECT_TYPE:
619  case JS_PROMISE_PROTOTYPE_TYPE:
620  case JS_REG_EXP_PROTOTYPE_TYPE:
621  case JS_SET_ITERATOR_PROTOTYPE_TYPE:
622  case JS_SET_PROTOTYPE_TYPE:
623  case JS_STRING_ITERATOR_PROTOTYPE_TYPE:
624  case JS_TYPED_ARRAY_PROTOTYPE_TYPE:
625  case JS_API_OBJECT_TYPE: {
626  DirectHandle<JSObject> js_object = Cast<JSObject>(receiver);
627  Maybe<bool> is_host_object = IsHostObject(js_object);
628  if (is_host_object.IsNothing()) {
629  return is_host_object;
630  }
631  if (is_host_object.FromJust()) {
632  return WriteHostObject(js_object);
633  } else {
634  return WriteJSObject(js_object);
635  }
636  }
637  case JS_SPECIAL_API_OBJECT_TYPE:
638  return WriteHostObject(Cast<JSObject>(receiver));
639  case JS_DATE_TYPE:
640  WriteJSDate(Cast<JSDate>(*receiver));
641  return ThrowIfOutOfMemory();
642  case JS_PRIMITIVE_WRAPPER_TYPE:
643  return WriteJSPrimitiveWrapper(Cast<JSPrimitiveWrapper>(receiver));
644  case JS_REG_EXP_TYPE:
645  WriteJSRegExp(Cast<JSRegExp>(receiver));
646  return ThrowIfOutOfMemory();
647  case JS_MAP_TYPE:
648  return WriteJSMap(Cast<JSMap>(receiver));
649  case JS_SET_TYPE:
650  return WriteJSSet(Cast<JSSet>(receiver));
651  case JS_ARRAY_BUFFER_TYPE:
652  return WriteJSArrayBuffer(Cast<JSArrayBuffer>(receiver));
653  case JS_TYPED_ARRAY_TYPE:
654  case JS_DATA_VIEW_TYPE:
655  case JS_RAB_GSAB_DATA_VIEW_TYPE:
656  return WriteJSArrayBufferView(Cast<JSArrayBufferView>(*receiver));
657  case JS_ERROR_TYPE:
658  return WriteJSError(Cast<JSObject>(receiver));
659  case JS_SHARED_ARRAY_TYPE:
660  return WriteJSSharedArray(Cast<JSSharedArray>(receiver));
661  case JS_SHARED_STRUCT_TYPE:
662  return WriteJSSharedStruct(Cast<JSSharedStruct>(receiver));
663  case JS_ATOMICS_MUTEX_TYPE:
664  case JS_ATOMICS_CONDITION_TYPE:
665  return WriteSharedObject(receiver);
666 #if V8_ENABLE_WEBASSEMBLY
667  case WASM_MODULE_OBJECT_TYPE:
668  return WriteWasmModule(Cast<WasmModuleObject>(receiver));
669  case WASM_MEMORY_OBJECT_TYPE:
670  return WriteWasmMemory(Cast<WasmMemoryObject>(receiver));
671  case WASM_STRUCT_TYPE:
672  case WASM_ARRAY_TYPE:
673  if (HeapLayout::InAnySharedSpace(*receiver)) {
674  return WriteSharedObject(receiver);
675  }
676  break;
677 #endif // V8_ENABLE_WEBASSEMBLY
678  default:
679  break;
680  }
681 
682  return ThrowDataCloneError(MessageTemplate::kDataCloneError, receiver);
683 }

685 Maybe<bool> ValueSerializer::WriteJSObject(DirectHandle<JSObject> object) {
686  DCHECK(!IsCustomElementsReceiverMap(object->map()));
687  const bool can_serialize_fast =
688  object->HasFastProperties(isolate_) && object->elements()->length() == 0;
689  if (!can_serialize_fast) return WriteJSObjectSlow(object);
690 
691  DirectHandle<Map> map(object->map(), isolate_);
692  WriteTag(SerializationTag::kBeginJSObject);
693 
694  // Write out fast properties as long as they are only data properties and the
695  // map doesn't change.
696  uint32_t properties_written = 0;
697  bool map_changed = false;
698  for (InternalIndex i : map->IterateOwnDescriptors()) {
699  DirectHandle<Name> key(map->instance_descriptors(isolate_)->GetKey(i),
700  isolate_);
701  if (!IsString(*key, isolate_)) continue;
702  PropertyDetails details =
703  map->instance_descriptors(isolate_)->GetDetails(i);
704  if (details.IsDontEnum()) continue;
705 
706  DirectHandle<Object> value;
707  if (V8_LIKELY(!map_changed)) map_changed = *map != object->map();
708  if (V8_LIKELY(!map_changed &&
709  details.location() == PropertyLocation::kField)) {
710  DCHECK_EQ(PropertyKind::kData, details.kind());
711  FieldIndex field_index = FieldIndex::ForDetails(*map, details);
712  value = direct_handle(object->RawFastPropertyAt(field_index), isolate_);
713  } else {
714  // This logic should essentially match WriteJSObjectPropertiesSlow.
715  // If the property is no longer found, do not serialize it.
716  // This could happen if a getter deleted the property.
717  LookupIterator it(isolate_, object, key, LookupIterator::OWN);
718  if (!it.IsFound()) continue;
719  if (!Object::GetProperty(&it).ToHandle(&value)) return Nothing<bool>();
720  }
721 
722  if (!WriteObject(key).FromMaybe(false) ||
723  !WriteObject(value).FromMaybe(false)) {
724  return Nothing<bool>();
725  }
726  properties_written++;
727  }
728 
729  WriteTag(SerializationTag::kEndJSObject);
730  WriteVarint<uint32_t>(properties_written);
731  return ThrowIfOutOfMemory();
732 }

749 Maybe<bool> ValueSerializer::WriteJSArray(DirectHandle<JSArray> array) {
750  PtrComprCageBase cage_base(isolate_);
751  uint32_t length = 0;
752  bool valid_length = Object::ToArrayLength(array->length(), &length);
753  DCHECK(valid_length);
754  USE(valid_length);
755 
756  // To keep things simple, for now we decide between dense and sparse
757  // serialization based on elements kind. A more principled heuristic could
758  // count the elements, but would need to take care to note which indices
759  // existed (as only indices which were enumerable own properties at this point
760  // should be serialized).
761  const bool should_serialize_densely =
762  array->HasFastElements(cage_base) && !array->HasHoleyElements(cage_base);
763 
764  if (should_serialize_densely) {
765  DCHECK_LE(length, static_cast<uint32_t>(FixedArray::kMaxLength));
766  WriteTag(SerializationTag::kBeginDenseJSArray);
767  WriteVarint<uint32_t>(length);
768  uint32_t i = 0;
769 
770  // Fast paths. Note that PACKED_ELEMENTS in particular can bail due to the
771  // structure of the elements changing.
772  switch (array->GetElementsKind(cage_base)) {
773  case PACKED_SMI_ELEMENTS: {
774  DisallowGarbageCollection no_gc;
775  Tagged<FixedArray> elements = Cast<FixedArray>(array->elements());
776  for (i = 0; i < length; i++) {
777  WriteSmi(Cast<Smi>(elements->get(i)));
778  }
779  break;
780  }
781  case PACKED_DOUBLE_ELEMENTS: {
782  // Elements are empty_fixed_array, not a FixedDoubleArray, if the array
783  // is empty. No elements to encode in this case anyhow.
784  if (length == 0) break;
785  DisallowGarbageCollection no_gc;
786  Tagged<FixedDoubleArray> elements =
787  Cast<FixedDoubleArray>(array->elements());
788  for (i = 0; i < length; i++) {
789  WriteTag(SerializationTag::kDouble);
790  WriteDouble(elements->get_scalar(i));
791  }
792  break;
793  }
794  case PACKED_ELEMENTS: {
795  DirectHandle<Object> old_length(array->length(cage_base), isolate_);
796  for (; i < length; i++) {
797  if (array->length(cage_base) != *old_length ||
798  array->GetElementsKind(cage_base) != PACKED_ELEMENTS) {
799  // Fall back to slow path.
800  break;
801  }
802  DirectHandle<Object> element(
803  Cast<FixedArray>(array->elements())->get(i), isolate_);
804  if (!WriteObject(element).FromMaybe(false)) return Nothing<bool>();
805  }
806  break;
807  }
808  default:
809  break;
810  }
811 
812  // If there are elements remaining, serialize them slowly.
813  for (; i < length; i++) {
814  // Serializing the array's elements can have arbitrary side effects, so we
815  // cannot rely on still having fast elements, even if it did to begin
816  // with.
817  DirectHandle<Object> element;
818  LookupIterator it(isolate_, array, i, array, LookupIterator::OWN);
819  if (!it.IsFound()) {
820  // This can happen in the case where an array that was originally dense
821  // became sparse during serialization. It's too late to switch to the
822  // sparse format, but we can mark the elements as absent.
823  WriteTag(SerializationTag::kTheHole);
824  continue;
825  }
826  if (!Object::GetProperty(&it).ToHandle(&element) ||
827  !WriteObject(element).FromMaybe(false)) {
828  return Nothing<bool>();
829  }
830  }
831 
832  DirectHandle<FixedArray> keys;
833  if (!KeyAccumulator::GetKeys(isolate_, array, KeyCollectionMode::kOwnOnly,
834  ENUMERABLE_STRINGS,
835  GetKeysConversion::kKeepNumbers, false, true)
836  .ToHandle(&keys)) {
837  return Nothing<bool>();
838  }
839 
840  uint32_t properties_written;
841  if (!WriteJSObjectPropertiesSlow(array, keys).To(&properties_written)) {
842  return Nothing<bool>();
843  }
844  WriteTag(SerializationTag::kEndDenseJSArray);
845  WriteVarint<uint32_t>(properties_written);
846  WriteVarint<uint32_t>(length);
847  } else {
848  WriteTag(SerializationTag::kBeginSparseJSArray);
849  WriteVarint<uint32_t>(length);
850  DirectHandle<FixedArray> keys;
851  uint32_t properties_written = 0;
852  if (!KeyAccumulator::GetKeys(isolate_, array, KeyCollectionMode::kOwnOnly,
853  ENUMERABLE_STRINGS)
854  .ToHandle(&keys) ||
855  !WriteJSObjectPropertiesSlow(array, keys).To(&properties_written)) {
856  return Nothing<bool>();
857  }
858  WriteTag(SerializationTag::kEndSparseJSArray);
859  WriteVarint<uint32_t>(properties_written);
860  WriteVarint<uint32_t>(length);
861  }
862  return ThrowIfOutOfMemory();
863 }

973 Maybe<bool> ValueSerializer::WriteJSArrayBuffer(
974  DirectHandle<JSArrayBuffer> array_buffer) {
975  if (array_buffer->is_shared()) {
976  if (!delegate_) {
977  return ThrowDataCloneError(MessageTemplate::kDataCloneError,
978  array_buffer);
979  }
980 
981  v8::Isolate* v8_isolate = reinterpret_cast<v8::Isolate*>(isolate_);
982  Maybe<uint32_t> index = delegate_->GetSharedArrayBufferId(
983  v8_isolate, Utils::ToLocalShared(array_buffer));
984  RETURN_VALUE_IF_EXCEPTION(isolate_, Nothing<bool>());
985 
986  WriteTag(SerializationTag::kSharedArrayBuffer);
987  WriteVarint(index.FromJust());
988 
989 #if V8_ENABLE_WEBASSEMBLY
990  // SharedArrayBuffers that are actually WebAssembly memories need special
991  // handling because they have metadata that may diverge from their
992  // BackingStore's.
993  //
994  // These are handled by also serializing/deserializing their corresponding
995  // WebAssembly.Memory object.
996  //
997  // See comment for WasmMemoryObject::FixUpResizableArrayBuffer for details.
998  auto backing_store = array_buffer->GetBackingStore();

1000  if (array_buffer->is_resizable_by_js()) {

1004  isolate_->factory()->array_buffer_wasm_memory_symbol())

1009  // read the same ArrayBuffer/WebAssembly.Memory recursively and fail.

1011  WriteVarint(static_cast<uint8_t>(
1012  WasmMemoryArrayBufferTag::kResizableFollowedByWasmMemory));

1014  return Nothing<bool>();
1015  }
1016  } else {
1017  WriteVarint(static_cast<uint8_t>(
1018  WasmMemoryArrayBufferTag::kResizableNotFollowedByWasmMemory));
1019  }
1020  } else {
1021  WriteVarint(
1022  static_cast<uint8_t>(WasmMemoryArrayBufferTag::kFixedLength));
1023  }
1024  }
1025 #endif // V8_ENABLE_WEBASSEMBLY
1026 
1027  return ThrowIfOutOfMemory();
1028  }
1029 
1030  uint32_t* transfer_entry = array_buffer_transfer_map_.Find(array_buffer);
1031  if (transfer_entry) {
1032  WriteTag(SerializationTag::kArrayBufferTransfer);
1033  WriteVarint(*transfer_entry);
1034  return ThrowIfOutOfMemory();
1035  }
1036  if (array_buffer->was_detached()) {
1037  return ThrowDataCloneError(
1038  MessageTemplate::kDataCloneErrorDetachedArrayBuffer);
1039  }

1041  if (byte_length > std::numeric_limits<uint32_t>::max()) {
1042  return ThrowDataCloneError(MessageTemplate::kDataCloneError, array_buffer);
1043  }
1044  if (array_buffer->is_resizable_by_js()) {
1045  size_t max_byte_length = array_buffer->max_byte_length();
1046  if (max_byte_length > std::numeric_limits<uint32_t>::max()) {
1047  return ThrowDataCloneError(MessageTemplate::kDataCloneError,
1048  array_buffer);
1049  }
1050 
1051  WriteTag(SerializationTag::kResizableArrayBuffer);
1052  WriteVarint<uint32_t>(static_cast<uint32_t>(byte_length));
1053  WriteVarint<uint32_t>(static_cast<uint32_t>(max_byte_length));
1054  WriteRawBytes(array_buffer->backing_store(), byte_length);
1055  return ThrowIfOutOfMemory();
1056  }
1057  WriteTag(SerializationTag::kArrayBuffer);
1058  WriteVarint<uint32_t>(static_cast<uint32_t>(byte_length));
1059  WriteRawBytes(array_buffer->backing_store(), byte_length);
1060  return ThrowIfOutOfMemory();
1061 }

1103 Maybe<bool> ValueSerializer::WriteJSError(DirectHandle<JSObject> error) {
1104  DirectHandle<Object> stack;
1105  PropertyDescriptor message_desc;
1106  Maybe<bool> message_found = JSReceiver::GetOwnPropertyDescriptor(
1107  isolate_, error, isolate_->factory()->message_string(), &message_desc);
1108  MAYBE_RETURN(message_found, Nothing<bool>());
1109  PropertyDescriptor cause_desc;
1110  Maybe<bool> cause_found = JSReceiver::GetOwnPropertyDescriptor(
1111  isolate_, error, isolate_->factory()->cause_string(), &cause_desc);
1112 
1113  WriteTag(SerializationTag::kError);
1114 
1115  DirectHandle<Object> name_object;
1116  if (!JSObject::GetProperty(isolate_, error, "name").ToHandle(&name_object)) {
1117  return Nothing<bool>();
1118  }
1119  DirectHandle<String> name;
1120  if (!Object::ToString(isolate_, name_object).ToHandle(&name)) {
1121  return Nothing<bool>();
1122  }
1123 
1124  if (name->IsOneByteEqualTo(base::CStrVector("EvalError"))) {
1125  WriteVarint(static_cast<uint8_t>(ErrorTag::kEvalErrorPrototype));
1126  } else if (name->IsOneByteEqualTo(base::CStrVector("RangeError"))) {
1127  WriteVarint(static_cast<uint8_t>(ErrorTag::kRangeErrorPrototype));
1128  } else if (name->IsOneByteEqualTo(base::CStrVector("ReferenceError"))) {
1129  WriteVarint(static_cast<uint8_t>(ErrorTag::kReferenceErrorPrototype));
1130  } else if (name->IsOneByteEqualTo(base::CStrVector("SyntaxError"))) {
1131  WriteVarint(static_cast<uint8_t>(ErrorTag::kSyntaxErrorPrototype));
1132  } else if (name->IsOneByteEqualTo(base::CStrVector("TypeError"))) {
1133  WriteVarint(static_cast<uint8_t>(ErrorTag::kTypeErrorPrototype));
1134  } else if (name->IsOneByteEqualTo(base::CStrVector("URIError"))) {
1135  WriteVarint(static_cast<uint8_t>(ErrorTag::kUriErrorPrototype));
1136  } else {
1137  // The default prototype in the deserialization side is Error.prototype, so
1138  // we don't have to do anything here.
1139  }
1140 
1141  if (message_found.FromJust() &&
1142  PropertyDescriptor::IsDataDescriptor(&message_desc)) {
1143  DirectHandle<String> message;
1144  if (!Object::ToString(isolate_, message_desc.value()).ToHandle(&message)) {
1145  return Nothing<bool>();
1146  }
1147  WriteVarint(static_cast<uint8_t>(ErrorTag::kMessage));
1148  WriteString(message);
1149  }
1150 
1151  if (!Object::GetProperty(isolate_, error, isolate_->factory()->stack_string())
1152  .ToHandle(&stack)) {
1153  return Nothing<bool>();
1154  }
1155  if (IsString(*stack)) {
1156  WriteVarint(static_cast<uint8_t>(ErrorTag::kStack));
1157  WriteString(Cast<String>(stack));
1158  }

1408 template <typename T>
1409 Maybe<T> ValueDeserializer::ReadVarint() {
1410  // Reads an unsigned integer as a base-128 varint.
1411  // The number is written, 7 bits at a time, from the least significant to the
1412  // most significant 7 bits. Each byte, except the last, has the MSB set.
1413  // If the varint is larger than T, any more significant bits are discarded.
1414  // See also https://developers.google.com/protocol-buffers/docs/encoding
1415  static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
1416  "Only unsigned integer types can be read as varints.");
1417  if (sizeof(T) > 4) return ReadVarintLoop<T>();
1418  auto max_read_position = position_ + sizeof(T) + 1;
1419  if (V8_UNLIKELY(max_read_position >= end_)) return ReadVarintLoop<T>();
1420 #ifdef DEBUG
1421  // DCHECK code to make sure the manually unrolled loop yields the exact
1422  // same end state and result.
1423  auto previous_position = position_;
1424  Maybe<T> maybe_expected_value = ReadVarintLoop<T>();
1425  // ReadVarintLoop can't return Nothing here; all such conditions have been
1426  // checked above.
1427  T expected_value = maybe_expected_value.ToChecked();
1428  auto expected_position = position_;
1429  position_ = previous_position;
1430 #endif // DEBUG
1431 #define EXIT_DCHECK() \
1432  DCHECK_LE(position_, end_); \
1433  DCHECK_EQ(position_, expected_position); \
1434  DCHECK_EQ(value, expected_value)
1435 
1436  T value = 0;
1437 #define ITERATION_SHIFTED(shift) \
1438  if (shift < sizeof(T) * 8) { \
1439  uint8_t byte = *position_; \
1440  position_++; \
1441  if (byte < 0x80) { \
1442  value |= static_cast<T>(byte) << shift; \
1443  EXIT_DCHECK(); \
1444  return Just(value); \
1445  } else { \
1446  value |= static_cast<T>(byte & 0x7F) << shift; \
1447  } \
1448  }
1449  // Manually unroll the loop to achieve the best measured performance.
1450  // This is ~15% faster than ReadVarintLoop.
1451  ITERATION_SHIFTED(0);
1452  ITERATION_SHIFTED(7);
1453  ITERATION_SHIFTED(14);
1454  ITERATION_SHIFTED(21);
1455  ITERATION_SHIFTED(28);
1456 
1457  EXIT_DCHECK();
1458  return Just(value);
1459 #undef ITERATION_SHIFTED
1460 #undef EXIT_DCHECK
1461 }

1463 template <typename T>
1464 Maybe<T> ValueDeserializer::ReadVarintLoop() {
1465  static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
1466  "Only unsigned integer types can be read as varints.");
1467  T value = 0;
1468  unsigned shift = 0;
1469  bool has_another_byte;
1470  do {
1471  if (position_ >= end_) return Nothing<T>();
1472  uint8_t byte = *position_;
1473  has_another_byte = byte & 0x80;
1474  if (V8_LIKELY(shift < sizeof(T) * 8)) {
1475  value |= static_cast<T>(byte & 0x7F) << shift;
1476  shift += 7;
1477  } else {
1478  // For consistency with the fast unrolled loop in ReadVarint we return
1479  // after we have read size(T) + 1 bytes.
1480 #ifdef V8_VALUE_DESERIALIZER_HARD_FAIL
1481  CHECK(!has_another_byte);
1482 #endif // V8_VALUE_DESERIALIZER_HARD_FAIL
1483  return Just(value);
1484  }
1485  position_++;
1486  } while (has_another_byte);
1487  return Just(value);
1488 }

1490 template <typename T>
1491 Maybe<T> ValueDeserializer::ReadZigZag() {
1492  // Writes a signed integer as a varint using ZigZag encoding (i.e. 0 is
1493  // encoded as 0, -1 as 1, 1 as 2, -2 as 3, and so on).
1494  // See also https://developers.google.com/protocol-buffers/docs/encoding
1495  static_assert(std::is_integral_v<T> && std::is_signed_v<T>,
1496  "Only signed integer types can be read as zigzag.");
1497  using UnsignedT = std::make_unsigned_t<T>;
1498  UnsignedT unsigned_value;
1499  if (!ReadVarint<UnsignedT>().To(&unsigned_value)) return Nothing<T>();
1500  return Just(static_cast<T>((unsigned_value >> 1) ^
1501  -static_cast<T>(unsigned_value & 1)));
1502 }

1647  // Read the count and ignore it.
1648  if (ReadVarint<uint32_t>().IsNothing())
1649  return MaybeDirectHandle<Object>();
1650  return ReadObject();
1651  case SerializationTag::kUndefined:
1652  return isolate_->factory()->undefined_value();
1653  case SerializationTag::kNull:
1654  return isolate_->factory()->null_value();
1655  case SerializationTag::kTrue:
1656  return isolate_->factory()->true_value();

1658  return isolate_->factory()->false_value();
1659  case SerializationTag::kInt32: {

1671  if (number.IsNothing()) return MaybeDirectHandle<Object>();
1672  return isolate_->factory()->NewNumber(number.FromJust());
1673  }
1674  case SerializationTag::kBigInt:
1675  return ReadBigInt();
1676  case SerializationTag::kUtf8String:
1677  return ReadUtf8String();
1678  case SerializationTag::kOneByteString:
1679  return ReadOneByteString();
1680  case SerializationTag::kTwoByteString:
1681  return ReadTwoByteString();

1683  uint32_t id;

1685  return GetObjectWithID(id);

1688  return ReadJSObject();
1689  case SerializationTag::kBeginSparseJSArray:
1690  return ReadSparseJSArray();
1691  case SerializationTag::kBeginDenseJSArray:
1692  return ReadDenseJSArray();
1693  case SerializationTag::kDate:
1694  return ReadJSDate();
1695  case SerializationTag::kTrueObject:
1696  case SerializationTag::kFalseObject:
1697  case SerializationTag::kNumberObject:
1698  case SerializationTag::kBigIntObject:
1699  case SerializationTag::kStringObject:
1700  return ReadJSPrimitiveWrapper(tag);
1701  case SerializationTag::kRegExp:
1702  return ReadJSRegExp();
1703  case SerializationTag::kBeginJSMap:
1704  return ReadJSMap();
1705  case SerializationTag::kBeginJSSet:
1706  return ReadJSSet();
1707  case SerializationTag::kArrayBuffer: {
1708  constexpr bool is_shared = false;
1709  constexpr bool is_resizable = false;
1710  return ReadJSArrayBuffer(is_shared, is_resizable);
1711  }
1712  case SerializationTag::kResizableArrayBuffer: {
1713  constexpr bool is_shared = false;
1714  constexpr bool is_resizable = true;
1715  return ReadJSArrayBuffer(is_shared, is_resizable);
1716  }
1717  case SerializationTag::kArrayBufferTransfer: {

1719  }
1720  case SerializationTag::kSharedArrayBuffer: {
1721  constexpr bool is_shared = true;
1722  constexpr bool is_resizable = false;
1723  return ReadJSArrayBuffer(is_shared, is_resizable);
1724  }
1725  case SerializationTag::kError:

1728  case SerializationTag::kWasmModuleTransfer:
