400  // In analogy to Flag::CheckFlagChange() in src/flags/flag.cc, only allow
401  // repeated flags for identical boolean values. We allow exceptions for flags
402  // with enum-like arguments since their conflicts can also be specified
403  // completely.
404  template <class T, bool kAllowIdenticalAssignment = std::is_same_v<T, bool>>
405  class DisallowReassignment {
406  public:
407  DisallowReassignment(const char* name, T value)
408  : name_(name), value_(value) {}
409 
410  operator T() const { return value_; }
411  T get() const { return value_; }
412  DisallowReassignment& operator=(T value) {
413  if (check_d8_flag_contradictions) {
414  if (kAllowIdenticalAssignment) {
415  if (specified_ && value_ != value) {
416  FATAL("Contradictory values for d8 flag --%s", name_);
417  }
418  } else {
419  if (specified_) {
420  FATAL("Repeated specification of d8 flag --%s", name_);
421  }
422  }
423  }
424  value_ = value;
425  specified_ = true;
426  return *this;
427  }
428  void Overwrite(T value) { value_ = value; }
429 
430  private:
431  const char* name_;
432  T value_;
433  bool specified_ = false;
434  };
