15 // The IfElseCascadeToSwitchReducer transforms if-else cascades
16 // that check if a given value is equal to a 32-bit constant from a given set
17 // into a switch with cases corresponding to the constants in the set.
18 //
19 // So for example code like:
20 // [only pure ops 1]
21 // if (x == 3) {
22 // B1;
23 // } else {
24 // [only pure ops 2]
25 // if (x == 5) {
26 // B2;
27 // } else {
28 // B3;
29 // }
30 // }
31 //
32 // will be transformed to:
33 // [only pure ops 1]
34 // [only pure ops 2]
35 // switch (x) {
36 // case 3:
37 // B1;
38 // case 5:
39 // B2;
40 // default:
41 // B3;
42 // }
43 //
44 // Or represented graphically:
45 // [only pure ops 1]
46 // [only pure ops 1] [only pure ops 2]
47 // x == 3 Switch(x)
48 // Branch | | |
49 // | | ----- | ------
50 // ----- ------ case 3 | | | default
51 // | | | | |
52 // T | | F v | |
53 // v v B1 | v
54 // B1 [only pure ops 2] becomes | B3
55 // x == 5 ======> case 5 |
56 // Branch v
57 // | | B2
58 // ----- ------
59 // | |
60 // T | | F
61 // v v
62 // B2 B3
63 //
64

78 template <class Next>
79 class IfElseCascadeToSwitchReducer : public Next {
80  public:
81  TURBOSHAFT_REDUCER_BOILERPLATE(IfElseCascadeToSwitch)
82 
83  OpIndex ReduceInputGraphBranch(OpIndex input_index, const BranchOp& branch) {
84  LABEL_BLOCK(no_change) {
85  return Next::ReduceInputGraphBranch(input_index, branch);
86  }
87  if (ShouldSkipOptimizationStep()) goto no_change;
88 
89  TRACE("[structural] Calling ReduceInputGraphBranch for index: %u\n",
90  static_cast<unsigned int>(input_index.id()));
91 
92  base::SmallVector<SwitchOp::Case, 16> cases;
93  base::SmallVector<const Block*, 16> false_blocks;
94 
95  Block* current_if_true;
96  Block* current_if_false;
97  const BranchOp* current_branch = &branch;
98  BranchHint current_branch_hint;
99  BranchHint next_hint = BranchHint::kNone;
100 
101  OpIndex switch_var = OpIndex::Invalid();
102  uint32_t value;
103  while (true) {
104  // If we encounter a condition that is not equality, we can't turn it
105  // into a switch case.
106  const Operation& cond =
107  Asm().input_graph().Get(current_branch->condition());
108 
109  if (!cond.template Is<ComparisonOp>()) {
110  // 'if(x==0)' may be optimized to 'if(x)', we should take this into
111  // consideration.
112 
113  // The "false" destination will be inlined before the switch is emitted,
114  // so it should only contain pure operations.
115  if (!ContainsOnlyPureOps(current_branch->if_true,
116  Asm().input_graph())) {
117  TRACE("\t [break] End of only-pure-ops cascade reached.\n");
118  break;
119  }
120 
121  OpIndex current_var = current_branch->condition();
122  if (!switch_var.valid()) {
123  switch_var = current_var;
124  } else if (switch_var != current_var) {
125  TRACE("\t [bailout] Not all branches compare the same variable.\n");
126  break;
127  }
128  value = 0;
129  // The true/false of 'if(x)' is reversed from 'if(x==0)'
130  current_if_true = current_branch->if_false;
131  current_if_false = current_branch->if_true;
132  const BranchHint hint = current_branch->hint;
133  current_branch_hint = hint == BranchHint::kNone ? BranchHint::kNone
134  : hint == BranchHint::kTrue ? BranchHint::kFalse
135  : BranchHint::kTrue;
136  } else {
137  const ComparisonOp* equal =
138  cond.template TryCast<Opmask::kWord32Equal>();
139  if (!equal) {
140  TRACE(
141  "\t [bailout] Branch with different condition than Word32 "
142  "Equal.\n");
143  break;
144  }
145  // MachineOptimizationReducer should normalize equality to put constants
146  // right.
147  const Operation& right_op = Asm().input_graph().Get(equal->right());
148  if (!right_op.Is<Opmask::kWord32Constant>()) {
149  TRACE(
150  "\t [bailout] No Word32 constant on the right side of Equal.\n");
151  break;
152  }
153 
154  // The "false" destination will be inlined before the switch is emitted,
155  // so it should only contain pure operations.
156  if (!ContainsOnlyPureOps(current_branch->if_false,
157  Asm().input_graph())) {
158  TRACE("\t [break] End of only-pure-ops cascade reached.\n");
159  break;
160  }
161  const ConstantOp& const_op = right_op.Cast<ConstantOp>();
162  value = const_op.word32();
163 
164  // If we encounter equal to a different value, we can't introduce
165  // a switch.
166  OpIndex current_var = equal->left();
167  if (!switch_var.valid()) {
168  switch_var = current_var;
169  } else if (switch_var != current_var) {
170  TRACE("\t [bailout] Not all branches compare the same variable.\n");
171  break;
172  }
173 
174  current_if_true = current_branch->if_true;
175  current_if_false = current_branch->if_false;
176  current_branch_hint = current_branch->hint;
177  }
178 
179  DCHECK(current_if_true && current_if_false);
180 
181  // We can't just use `current_branch->hint` for every case. Consider:
182  //
183  // if (a) { }
184  // else if (b) { }
185  // else if (likely(c)) { }
186  // else if (d) { }
187  // else { }
188  //
189  // The fact that `c` is Likely doesn't tell anything about the likelyness
190  // of `a` and `b` compared to `c`, which means that `c` shouldn't have the
191  // Likely hint in the switch. However, since `c` is likely here, it means
192  // that `d` and "default" are both unlikely, even in the switch.
193  //
194  // So, for the 1st case, we use `current_branch->hint`.
195  // Then, when we encounter a Likely hint, we mark all of the subsequent
196  // cases are Unlikely, but don't mark the current one as Likely. This is
197  // done with the `next_hint` variable, which is initially kNone, but
198  // because kFalse when we encounter a Likely branch.
199  // We never set `next_hint` as kTrue as it would only apply to subsequent
200  // cases and not to already-emitted cases. The only case that could thus
201  // have a kTrue annotation is the 1st one.
202  DCHECK_NE(next_hint, BranchHint::kTrue);
203  BranchHint hint = next_hint;
204  if (cases.size() == 0) {
205  // The 1st case gets its original hint.
206  hint = current_branch_hint;
207  } else if (current_branch_hint == BranchHint::kFalse) {
208  // For other cases, if the branch has a kFalse hint, we do use it,
209  // regardless of `next_hint`.
210  hint = BranchHint::kNone;
211  }
212  if (current_branch_hint == BranchHint::kTrue) {
213  // This branch is likely true, which means that all subsequent cases are
214  // unlikely.
215  next_hint = BranchHint::kFalse;
216  }
217 
218  // The current_if_true block becomes the corresponding switch case block.
219  cases.emplace_back(value, Asm().MapToNewGraph(current_if_true), hint);
220 
221  // All pure ops from the if_false block should be executed before
222  // the switch, except the last Branch operation (which we drop).
223  false_blocks.push_back(current_if_false);
224 
225  // If we encounter a if_false block that doesn't end with a Branch,
226  // this means we've reached the end of the cascade.
227  const Operation& maybe_branch =
228  current_if_false->LastOperation(Asm().input_graph());
229  if (!maybe_branch.Is<BranchOp>()) {
230  TRACE("\t [break] Reached end of the if-else cascade.\n");
231  break;
232  }
233 
234  // Iterate to the next if_false block in the cascade.
235  current_branch = &maybe_branch.template Cast<BranchOp>();
236  }
237 
238  // Probably better to keep short if-else cascades as they are.
239  if (cases.size() <= 2) {
240  TRACE("\t [bailout] Cascade with less than 2 levels of nesting.\n");
241  goto no_change;
242  }
243  CHECK_EQ(cases.size(), false_blocks.size());
244 
245  // Sorting the cases because it will help figure out if there is a duplicate
246  // case (in which case we bailout, since this is not well handled by the
247  // code generator).
248  // Note that this isn't wasted work: there is a good chance that the
249  // instruction selector will emit a binary search for this switch, which
250  // will require the cases to be sorted.
251  std::stable_sort(
252  cases.begin(), cases.end(),
253  [](SwitchOp::Case a, SwitchOp::Case b) { return a.value < b.value; });
254  auto it = std::adjacent_find(
255  cases.begin(), cases.end(),
256  [](SwitchOp::Case a, SwitchOp::Case b) { return a.value == b.value; });
257  if (it != cases.end()) {
258  TRACE("\t [bailout] Multiple cases with the value %d.\n", (*it).value);
259  goto no_change;
260  }
261 
262  TRACE("[reduce] Successfully emit a Switch with %zu cases.\n",
263  cases.size());
264  return EmitSwitch(switch_var, cases, false_blocks, current_if_false,
265  next_hint);
266  }
267 
268  private:
269  static bool ContainsOnlyPureOps(const Block* block, const Graph& graph) {
270  for (const auto& op : base::IterateWithoutLast(graph.operations(*block))) {
271  // We are moving the block content to before the switch, effectively
272  // moving it before the previously existing branches.
273  if (!op.Effects().hoistable_before_a_branch()) {
274  return false;
275  }
276  }
277  return true;
278  }
279 
280  // Visits and emits {input_block} right now (ie, in the current block)
281  // until the one before the last operation is reached.
282  void InlineAllOperationsWithoutLast(const Block* input_block) {
283  base::iterator_range<Graph::OpIndexIterator> all_ops =
284  Asm().input_graph().OperationIndices(*input_block);
285 
286  for (OpIndex op : base::IterateWithoutLast(all_ops)) {
287  Asm().InlineOp(op, input_block);
288  }
289  }
290 
291  V<None> EmitSwitch(OpIndex switch_var,
292  base::SmallVector<SwitchOp::Case, 16>& cases,
293  base::SmallVector<const Block*, 16>& false_blocks,
294  Block* current_if_false, BranchHint next_hint) {
295  // We're skipping the last false block, as it becomes the default block.
296  for (size_t i = 0; i < false_blocks.size() - 1; ++i) {
297  const Block* block = false_blocks[i];
298  InlineAllOperationsWithoutLast(block);
299  }
300 
301  // The last current_if_true block that ends the cascade becomes the default
302  // case.
303  Block* default_block = current_if_false;
304  Asm().Switch(
305  Asm().MapToNewGraph(switch_var),
306  Asm().output_graph().graph_zone()->CloneVector(base::VectorOf(cases)),
307  Asm().MapToNewGraph(default_block), next_hint);
308  return OpIndex::Invalid();
309  }
310 };
