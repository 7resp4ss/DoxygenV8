45 Response arrayToProtocolValue(v8::Local<v8::Context> context,
46  v8::Local<v8::Array> array, int maxDepth,
47  std::unique_ptr<protocol::ListValue>* result) {
48  std::unique_ptr<protocol::ListValue> inspectorArray =
49  protocol::ListValue::create();
50  uint32_t length = array->Length();
51  for (uint32_t i = 0; i < length; i++) {
52  v8::Local<v8::Value> value;
53  if (!array->Get(context, i).ToLocal(&value))
54  return Response::InternalError();
55  std::unique_ptr<protocol::Value> element;
56  Response response = toProtocolValue(context, value, maxDepth - 1, &element);
57  if (!response.IsSuccess()) return response;
58  inspectorArray->pushValue(std::move(element));
59  }
60  *result = std::move(inspectorArray);
61  return Response::Success();
62 }

64 Response objectToProtocolValue(
65  v8::Local<v8::Context> context, v8::Local<v8::Object> object, int maxDepth,
66  std::unique_ptr<protocol::DictionaryValue>* result) {
67  std::unique_ptr<protocol::DictionaryValue> jsonObject =
68  protocol::DictionaryValue::create();
69  v8::Local<v8::Array> propertyNames;
70  if (!object->GetOwnPropertyNames(context).ToLocal(&propertyNames))
71  return Response::InternalError();
72  uint32_t length = propertyNames->Length();
73  for (uint32_t i = 0; i < length; i++) {
74  v8::Local<v8::Value> name;
75  if (!propertyNames->Get(context, i).ToLocal(&name))
76  return Response::InternalError();
77  if (name->IsString()) {
78  v8::Maybe<bool> hasRealNamedProperty =
79  object->HasRealNamedProperty(context, name.As<v8::String>());
80  // Don't access properties with interceptors.
81  if (hasRealNamedProperty.IsNothing() || !hasRealNamedProperty.FromJust())
82  continue;
83  }
84  v8::Local<v8::String> propertyName;
85  if (!name->ToString(context).ToLocal(&propertyName)) continue;
86  v8::Local<v8::Value> property;
87  if (!object->Get(context, name).ToLocal(&property))
88  return Response::InternalError();
89  if (property->IsUndefined()) continue;
90  std::unique_ptr<protocol::Value> propertyValue;
91  Response response =
92  toProtocolValue(context, property, maxDepth - 1, &propertyValue);
93  if (!response.IsSuccess()) return response;
94  jsonObject->setValue(
95  toProtocolString(v8::Isolate::GetCurrent(), propertyName),
96  std::move(propertyValue));
97  }
98  *result = std::move(jsonObject);
99  return Response::Success();
100 }

102 std::unique_ptr<protocol::FundamentalValue> toProtocolValue(
103  double doubleValue) {
104  if (doubleValue >= std::numeric_limits<int>::min() &&
105  doubleValue <= std::numeric_limits<int>::max() &&
106  v8::base::bit_cast<int64_t>(doubleValue) !=
107  v8::base::bit_cast<int64_t>(-0.0)) {
108  int intValue = static_cast<int>(doubleValue);
109  if (intValue == doubleValue) {
110  return protocol::FundamentalValue::create(intValue);
111  }
112  }
113  return protocol::FundamentalValue::create(doubleValue);
114 }

116 Response toProtocolValue(v8::Local<v8::Context> context,
117  v8::Local<v8::Value> value, int maxDepth,
118  std::unique_ptr<protocol::Value>* result) {
119  if (maxDepth <= 0)
120  return Response::ServerError("Object reference chain is too long");
121 
122  if (value->IsNull() || value->IsUndefined()) {
123  *result = protocol::Value::null();
124  return Response::Success();
125  }
126  if (value->IsBoolean()) {
127  *result =
128  protocol::FundamentalValue::create(value.As<v8::Boolean>()->Value());
129  return Response::Success();
130  }
131  if (value->IsNumber()) {
132  double doubleValue = value.As<v8::Number>()->Value();
133  *result = toProtocolValue(doubleValue);
134  return Response::Success();
135  }
136  if (value->IsString()) {
137  *result = protocol::StringValue::create(
138  toProtocolString(v8::Isolate::GetCurrent(), value.As<v8::String>()));
139  return Response::Success();
140  }
141  if (value->IsArray()) {
142  v8::Local<v8::Array> array = value.As<v8::Array>();
143  std::unique_ptr<protocol::ListValue> list_result;
144  auto response =
145  arrayToProtocolValue(context, array, maxDepth, &list_result);
146  *result = std::move(list_result);
147  return response;
148  }
149  if (value->IsObject()) {
150  v8::Local<v8::Object> object = value.As<v8::Object>();
151  std::unique_ptr<protocol::DictionaryValue> dict_result;
152  auto response =
153  objectToProtocolValue(context, object, maxDepth, &dict_result);
154  *result = std::move(dict_result);
155  return response;
156  }
157 
158  return Response::ServerError("Object couldn't be returned by value");
159 }

192 String16 abbreviateString(const String16& value, AbbreviateMode mode) {
193  const size_t maxLength = 100;
194  if (value.length() <= maxLength) return value;
195  UChar ellipsis = static_cast<UChar>(0x2026);
196  if (mode == kMiddle) {
197  return String16::concat(
198  value.substring(0, maxLength / 2), String16(&ellipsis, 1),
199  value.substring(value.length() - maxLength / 2 + 1));
200  }
201  return String16::concat(value.substring(0, maxLength - 1), ellipsis);
202 }

259 v8::MaybeLocal<v8::Value> getErrorProperty(v8::Local<v8::Context> context,
260  v8::Local<v8::Object> object,
261  v8::Local<v8::String> name) {
262  v8::Isolate* isolate = v8::Isolate::GetCurrent();
263  v8::TryCatch tryCatch(isolate);
264  v8::MicrotasksScope microtasksScope(context,
265  v8::MicrotasksScope::kDoNotRunMicrotasks);
266  v8::Local<v8::Value> descriptor;
267  if (!object->GetOwnPropertyDescriptor(context, name).ToLocal(&descriptor)) {
268  tryCatch.Reset();
269  return object->Get(context, name);
270  }
271  if (!descriptor->IsObject()) return object->Get(context, name);
272 
273  v8::Local<v8::Object> descriptorObject = descriptor.As<v8::Object>();
274  v8::Local<v8::Value> getDescriptor;
275  if (!descriptorObject->HasOwnProperty(context, toV8String(isolate, "get"))
276  .FromJust()) {
277  tryCatch.Reset();
278  return object->Get(context, name);
279  }
280  if (!descriptorObject->Get(context, toV8String(isolate, "get"))
281  .ToLocal(&getDescriptor)) {
282  tryCatch.Reset();
283  return object->Get(context, name);
284  }
285 
286  if (getDescriptor->IsFunction()) {
287  v8::Local<v8::Function> function = getDescriptor.As<v8::Function>();
288  if (deepBoundFunction(function)->ScriptId() !=
289  v8::UnboundScript::kNoScriptId) {
290  return v8::MaybeLocal<v8::Value>();
291  }
292  }
293 
294  return object->Get(context, name);
295 }

297 // Build a description from an exception using the following pattern:
298 // * The first line is "<name || constructor name>: <message property>". We
299 // use the constructor name if the "name" property is "Error". Most custom
300 // Error subclasses don't overwrite the "name" property.
301 // * The rest is the content of the "stack" property but only with the actual
302 // stack trace part.
303 String16 descriptionForError(v8::Local<v8::Context> context,
304  v8::Local<v8::Object> object) {
305  v8::Isolate* isolate = v8::Isolate::GetCurrent();
306  v8::TryCatch tryCatch(isolate);
307 
308  String16 name = toProtocolString(isolate, object->GetConstructorName());
309  {
310  v8::Local<v8::Value> nameValue;
311  if (getErrorProperty(context, object, toV8String(isolate, "name"))
312  .ToLocal(&nameValue) &&
313  nameValue->IsString()) {
314  v8::Local<v8::String> nameString = nameValue.As<v8::String>();
315  if (nameString->Length() > 0 &&
316  !nameString->StringEquals(toV8String(isolate, "Error"))) {
317  name = toProtocolString(isolate, nameString);
318  }
319  }
320  }
321 
322  std::optional<String16> stack;
323  {
324  v8::Local<v8::Value> stackValue;
325  if (getErrorProperty(context, object, toV8String(isolate, "stack"))
326  .ToLocal(&stackValue) &&
327  stackValue->IsString()) {
328  String16 stackString =
329  toProtocolString(isolate, stackValue.As<v8::String>());
330  size_t pos = stackString.find("\n at ");
331  if (pos != String16::kNotFound) {
332  stack = stackString.substring(pos);
333  }
334  }
335  }
336 
337  std::optional<String16> message;
338  {
339  v8::Local<v8::Value> messageValue;
340  if (getErrorProperty(context, object, toV8String(isolate, "message"))
341  .ToLocal(&messageValue) &&
342  messageValue->IsString()) {
343  String16 msg = toProtocolStringWithTypeCheck(isolate, messageValue);
344  if (!msg.isEmpty()) message = msg;
345  }
346  }
347 
348  String16 description = name;
349  if (message.has_value() && message->length() > 0) {
350  description += ": " + *message;
351  }
352 
353  if (stack.has_value() && stack->length() > 0) {
354  description += *stack;
355  }
356  return description;
357 }

412 String16 descriptionForEntry(v8::Local<v8::Context> context,
413  v8::Local<v8::Object> object) {
414  v8::Isolate* isolate = v8::Isolate::GetCurrent();
415  String16 key;
416  v8::Local<v8::Value> tmp;
417  if (object->GetRealNamedProperty(context, toV8String(isolate, "key"))
418  .ToLocal(&tmp)) {
419  auto wrapper = ValueMirror::create(context, tmp);
420  if (wrapper) {
421  std::unique_ptr<ObjectPreview> preview;
422  int limit = 5;
423  wrapper->buildEntryPreview(context, &limit, &limit, &preview);
424  if (preview) {
425  key = preview->getDescription(String16());
426  if (preview->getType() == RemoteObject::TypeEnum::String) {
427  key = String16::concat('\"', key, '\"');
428  }
429  }
430  }
431  }
432 
433  String16 value;
434  if (object->GetRealNamedProperty(context, toV8String(isolate, "value"))
435  .ToLocal(&tmp)) {
436  auto wrapper = ValueMirror::create(context, tmp);
437  if (wrapper) {
438  std::unique_ptr<ObjectPreview> preview;
439  int limit = 5;
440  wrapper->buildEntryPreview(context, &limit, &limit, &preview);
441  if (preview) {
442  value = preview->getDescription(String16());
443  if (preview->getType() == RemoteObject::TypeEnum::String) {
444  value = String16::concat('\"', value, '\"');
445  }
446  }
447  }
448  }
449 
450  return key.length() ? ("{" + key + " => " + value + "}") : value;
451 }

552  Response buildDeepSerializedValue(
553  v8::Local<v8::Context> context, int maxDepth,
554  v8::Local<v8::Object> additionalParameters,
555  V8SerializationDuplicateTracker& duplicateTracker,
556  std::unique_ptr<protocol::DictionaryValue>* result) const override {
557  v8::Local<v8::Value> value = v8Value(v8::Isolate::GetCurrent());
558  if (value->IsUndefined()) {
559  *result = protocol::DictionaryValue::create();
560  (*result)->setString(
561  "type", protocol::Runtime::DeepSerializedValue::TypeEnum::Undefined);
562  return Response::Success();
563  }
564  if (value->IsNull()) {
565  *result = protocol::DictionaryValue::create();
566  (*result)->setString(
567  "type", protocol::Runtime::DeepSerializedValue::TypeEnum::Null);
568  return Response::Success();
569  }
570  if (value->IsString()) {
571  *result = protocol::DictionaryValue::create();
572  (*result)->setString(
573  "type", protocol::Runtime::DeepSerializedValue::TypeEnum::String);
574  (*result)->setString("value", toProtocolString(v8::Isolate::GetCurrent(),
575  value.As<v8::String>()));
576  return Response::Success();
577  }
578  if (value->IsBoolean()) {
579  *result = protocol::DictionaryValue::create();
580  (*result)->setString(
581  "type", protocol::Runtime::DeepSerializedValue::TypeEnum::Boolean);
582  (*result)->setBoolean("value", value.As<v8::Boolean>()->Value());
583  return Response::Success();
584  }
585 
586  // Fallback in case of unexpected type.
587  bool isKnown;
588  *result = duplicateTracker.LinkExistingOrCreate(value, &isKnown);
589  if (isKnown) {
590  return Response::Success();
591  }
592 
593  (*result)->setString(
594  "type", protocol::Runtime::DeepSerializedValue::TypeEnum::Object);
595  return Response::Success();

1326  &overflow, properties.get());

1330  indexLimit, &overflow, &mirrors)) {

1345  .setType(PropertyPreview::TypeEnum::Accessor)
1346  .build();

1349  preview->setValuePreview(std::move(valuePreview));

1386  .setOverflow(overflow)
1387  .setProperties(std::move(properties))
1388  .build();

1390  if (entriesPreview) (*result)->setEntries(std::move(entriesPreview));

1525 bool ValueMirror::getProperties(v8::Local<v8::Context> context,
1526  v8::Local<v8::Object> object,
1527  bool ownProperties, bool accessorPropertiesOnly,
1528  bool nonIndexedPropertiesOnly,
1529  PropertyAccumulator* accumulator) {

1531  v8::TryCatch tryCatch(isolate);
1532  v8::Local<v8::Set> set = v8::Set::New(isolate);
1533 
1534  v8::MicrotasksScope microtasksScope(context,
1535  v8::MicrotasksScope::kDoNotRunMicrotasks);
1536  V8InternalValueType internalType = v8InternalValueTypeFrom(context, object);
1537  if (internalType == V8InternalValueType::kScope) {
1538  v8::Local<v8::Value> value;
1539  if (!object->Get(context, toV8String(isolate, "object")).ToLocal(&value) ||
1540  !value->IsObject()) {
1541  return false;
1542  } else {
1543  object = value.As<v8::Object>();
1544  }
1545  }
1546  if (internalType == V8InternalValueType::kScopeList ||
1547  internalType == V8InternalValueType::kPrivateMethodList) {

1554  nonIndexedPropertiesOnly);
1555  if (!iterator) {
1556  CHECK(tryCatch.HasCaught());
1557  return false;
1558  }

1561  if (!isOwn && ownProperties) break;

1563  v8::Maybe<bool> result = set->Has(context, v8Name);
1564  if (result.IsNothing()) return false;
1565  if (result.FromJust()) {
1566  if (!iterator->Advance().FromMaybe(false)) {
1567  CHECK(tryCatch.HasCaught());
1568  return false;
1569  }
1570  continue;
1571  }
1572  if (!set->Add(context, v8Name).ToLocal(&set)) return false;
1573 
1574  String16 name;
1575  std::unique_ptr<ValueMirror> symbolMirror;
1576  if (v8Name->IsString()) {
1577  name = toProtocolString(isolate, v8Name.As<v8::String>());
1578  } else {
1579  v8::Local<v8::Symbol> symbol = v8Name.As<v8::Symbol>();
1580  name = descriptionForSymbol(context, symbol);
1581  symbolMirror = ValueMirror::create(context, symbol);
1582  }

1585  std::unique_ptr<ValueMirror> valueMirror;
1586  std::unique_ptr<ValueMirror> getterMirror;
1587  std::unique_ptr<ValueMirror> setterMirror;
1588  std::unique_ptr<ValueMirror> exceptionMirror;
1589  bool writable = false;
1590  bool enumerable = false;
1591  bool configurable = false;
1592 
1593  bool isAccessorProperty = false;

1595  if (!iterator->attributes().To(&attributes)) {
1596  exceptionMirror =
1597  ValueMirror::create(context, tryCatchAttributes.Exception());
1598  } else {
1599  if (iterator->is_native_accessor()) {
1600  if (iterator->has_native_getter()) {
1601  getterMirror = createNativeGetter(context, object, v8Name);
1602  }
1603  if (iterator->has_native_setter()) {
1604  setterMirror = createNativeSetter(context, object, v8Name);
1605  }

1607  enumerable = !(attributes & v8::PropertyAttribute::DontEnum);
1608  configurable = !(attributes & v8::PropertyAttribute::DontDelete);
1609  isAccessorProperty = getterMirror || setterMirror;
1610  } else {

1613  if (!iterator->descriptor().To(&descriptor)) {
1614  exceptionMirror =
1615  ValueMirror::create(context, tryCatchDescriptor.Exception());
1616  } else {
1617  writable = descriptor.has_writable ? descriptor.writable : false;
1618  enumerable =
1619  descriptor.has_enumerable ? descriptor.enumerable : false;
1620  configurable =
1621  descriptor.has_configurable ? descriptor.configurable : false;
1622  if (!descriptor.value.IsEmpty()) {
1623  valueMirror = ValueMirror::create(context, descriptor.value);
1624  }
1625  v8::Local<v8::Function> getterFunction;
1626  if (!descriptor.get.IsEmpty()) {
1627  v8::Local<v8::Value> get = descriptor.get;
1628  getterMirror = ValueMirror::create(context, get);
1629  if (get->IsFunction()) getterFunction = get.As<v8::Function>();
1630  }
1631  if (!descriptor.set.IsEmpty()) {
1632  setterMirror = ValueMirror::create(context, descriptor.set);
1633  }
1634  isAccessorProperty = getterMirror || setterMirror;
1635  if (name != "__proto__" && !getterFunction.IsEmpty() &&
1636  deepBoundFunction(getterFunction)->ScriptId() ==
1637  v8::UnboundScript::kNoScriptId &&
1638  !doesAttributeHaveObservableSideEffectOnGet(context, object,

1640  v8::TryCatch tryCatchFunction(isolate);

1643  if (value->IsPromise() &&
1644  value.As<v8::Promise>()->State() == v8::Promise::kRejected) {
1645  value.As<v8::Promise>()->MarkAsHandled();
1646  } else {
1647  valueMirror = ValueMirror::create(context, value);
1648  setterMirror = nullptr;
1649  getterMirror = nullptr;
1650  }
1651  }
1652  }
1653  }
1654  }
1655  }
1656  if (accessorPropertiesOnly && !isAccessorProperty) continue;

1658  writable,
1659  configurable,
1660  enumerable,
1661  isOwn,
1662  iterator->is_array_index(),

1665  std::move(getterMirror),
1666  std::move(setterMirror),
1667  std::move(symbolMirror),
1668  std::move(exceptionMirror)};
1669  if (!accumulator->Add(std::move(mirror))) return true;
1670 
1671  if (!iterator->Advance().FromMaybe(false)) {
1672  CHECK(tryCatchAttributes.HasCaught());
1673  return false;
1674  }
1675  }
1676  return true;
1677 }

1818  v8::Local<v8::Value> value) {
1819  v8::Isolate* isolate = v8::Isolate::GetCurrent();
1820  if (value->IsNull()) {
1821  return std::make_unique<PrimitiveValueMirror>(
1822  isolate, value.As<v8::Primitive>(), RemoteObject::TypeEnum::Object);
1823  }
1824  if (value->IsBoolean()) {
1825  return std::make_unique<PrimitiveValueMirror>(
1826  isolate, value.As<v8::Primitive>(), RemoteObject::TypeEnum::Boolean);
1827  }
1828  if (value->IsNumber()) {
1829  return std::make_unique<NumberMirror>(isolate, value.As<v8::Number>());
1830  }
1831  if (value->IsString()) {
1832  return std::make_unique<PrimitiveValueMirror>(
1833  isolate, value.As<v8::Primitive>(), RemoteObject::TypeEnum::String);
1834  }
1835  if (value->IsBigInt()) {
1836  return std::make_unique<BigIntMirror>(isolate, value.As<v8::BigInt>());
1837  }
1838  if (value->IsSymbol()) {
1839  return std::make_unique<SymbolMirror>(isolate, value.As<v8::Symbol>());
1840  }
1841  if (value->IsUndefined()) {
1842  return std::make_unique<PrimitiveValueMirror>(
1843  isolate, value.As<v8::Primitive>(), RemoteObject::TypeEnum::Undefined);
1844  }
1845  if (!value->IsObject()) {
1846  return nullptr;
1847  }
1848  v8::Local<v8::Object> object = value.As<v8::Object>();
1849  auto clientSubtype = clientFor(context)->valueSubtype(object);
1850  if (clientSubtype) {
1851  String16 subtype = toString16(clientSubtype->string());
1852  return clientMirror(context, object, subtype);
1853  }
1854  if (object->IsRegExp()) {
1855  v8::Local<v8::RegExp> regexp = object.As<v8::RegExp>();
1856  return std::make_unique<ObjectMirror>(
1857  regexp, RemoteObject::SubtypeEnum::Regexp,
1858  descriptionForRegExp(isolate, regexp));
1859  }
1860  if (object->IsProxy()) {
1861  v8::Local<v8::Proxy> proxy = object.As<v8::Proxy>();
1862  return std::make_unique<ObjectMirror>(proxy,
1863  RemoteObject::SubtypeEnum::Proxy,
1864  descriptionForProxy(isolate, proxy));
1865  }
1866  if (object->IsFunction()) {
1867  v8::Local<v8::Function> function = object.As<v8::Function>();
1868  return std::make_unique<FunctionMirror>(function);
1869  }
1870  if (object->IsDate()) {
1871  v8::Local<v8::Date> date = object.As<v8::Date>();
1872  return std::make_unique<ObjectMirror>(date, RemoteObject::SubtypeEnum::Date,
1873  descriptionForDate(context, date));
1874  }
1875  if (object->IsPromise()) {
1876  v8::Local<v8::Promise> promise = object.As<v8::Promise>();
1877  return std::make_unique<ObjectMirror>(
1878  promise, RemoteObject::SubtypeEnum::Promise,
1879  descriptionForObject(isolate, promise));
1880  }
1881  if (object->IsNativeError()) {
1882  return std::make_unique<ObjectMirror>(object,
1883  RemoteObject::SubtypeEnum::Error,
1884  descriptionForError(context, object));
1885  }
1886  if (object->IsMap()) {
1887  v8::Local<v8::Map> map = object.As<v8::Map>();
1888  return std::make_unique<ObjectMirror>(
1889  map, RemoteObject::SubtypeEnum::Map,
1890  descriptionForCollection(isolate, map, map->Size()));
1891  }
1892  if (object->IsSet()) {
1893  v8::Local<v8::Set> set = object.As<v8::Set>();
1894  return std::make_unique<ObjectMirror>(
1895  set, RemoteObject::SubtypeEnum::Set,
1896  descriptionForCollection(isolate, set, set->Size()));
1897  }
1898  if (object->IsWeakMap()) {
1899  return std::make_unique<ObjectMirror>(
1900  object, RemoteObject::SubtypeEnum::Weakmap,
1901  descriptionForObject(isolate, object));
1902  }
1903  if (object->IsWeakSet()) {
1904  return std::make_unique<ObjectMirror>(
1905  object, RemoteObject::SubtypeEnum::Weakset,
1906  descriptionForObject(isolate, object));
1907  }
1908  if (object->IsMapIterator() || object->IsSetIterator()) {
1909  return std::make_unique<ObjectMirror>(
1910  object, RemoteObject::SubtypeEnum::Iterator,
1911  descriptionForObject(isolate, object));
1912  }
1913  if (object->IsGeneratorObject()) {
1914  return std::make_unique<ObjectMirror>(
1915  object, RemoteObject::SubtypeEnum::Generator,
1916  descriptionForObject(isolate, object));
1917  }
1918  if (object->IsTypedArray()) {
1919  v8::Local<v8::TypedArray> array = object.As<v8::TypedArray>();
1920  return std::make_unique<ObjectMirror>(
1921  array, RemoteObject::SubtypeEnum::Typedarray,
1922  descriptionForCollection(isolate, array, array->Length()));
1923  }
1924  if (object->IsArrayBuffer()) {
1925  v8::Local<v8::ArrayBuffer> buffer = object.As<v8::ArrayBuffer>();
1926  return std::make_unique<ObjectMirror>(
1927  buffer, RemoteObject::SubtypeEnum::Arraybuffer,
1928  descriptionForCollection(isolate, buffer, buffer->ByteLength()));
1929  }
1930  if (object->IsSharedArrayBuffer()) {
1931  v8::Local<v8::SharedArrayBuffer> buffer =
1932  object.As<v8::SharedArrayBuffer>();
1933  return std::make_unique<ObjectMirror>(
1934  buffer, RemoteObject::SubtypeEnum::Arraybuffer,
1935  descriptionForCollection(isolate, buffer, buffer->ByteLength()));
1936  }
1937  if (object->IsDataView()) {
1938  v8::Local<v8::DataView> view = object.As<v8::DataView>();
1939  return std::make_unique<ObjectMirror>(
1940  view, RemoteObject::SubtypeEnum::Dataview,
1941  descriptionForCollection(isolate, view, view->ByteLength()));
1942  }
1943  if (object->IsWasmMemoryObject()) {
1944  v8::Local<v8::WasmMemoryObject> memory = object.As<v8::WasmMemoryObject>();
1945  return std::make_unique<ObjectMirror>(
1946  memory, RemoteObject::SubtypeEnum::Webassemblymemory,
1947  descriptionForCollection(
1948  isolate, memory, memory->Buffer()->ByteLength() / kWasmPageSize));
1949  }
1950 #if V8_ENABLE_WEBASSEMBLY
1951  if (v8::debug::WasmValueObject::IsWasmValueObject(object)) {
1952  v8::Local<v8::debug::WasmValueObject> value_object =
1953  object.As<v8::debug::WasmValueObject>();
1954  return std::make_unique<ObjectMirror>(
1955  value_object, RemoteObject::SubtypeEnum::Wasmvalue,
1956  descriptionForWasmValueObject(context, value_object));
1957  }
1958 #endif // V8_ENABLE_WEBASSEMBLY
1959  if (!value->IsObject()) {
1960  return nullptr;
1961  }
1962  V8InternalValueType internalType = v8InternalValueTypeFrom(context, object);
1963  if (internalType == V8InternalValueType::kScopeList) {
1964  v8::Local<v8::Array> array = value.As<v8::Array>();
1965  return std::make_unique<ObjectMirror>(array, "internal#scopeList",
1966  descriptionForScopeList(array));
1967  }
1968  if (internalType == V8InternalValueType::kPrivateMethodList) {
1969  v8::Local<v8::Array> array = object.As<v8::Array>();
1970  return std::make_unique<ObjectMirror>(
1971  array, "internal#privateMethodList",
1972  descriptionForPrivateMethodList(array));
1973  }
1974  if (internalType == V8InternalValueType::kEntry) {
1975  return std::make_unique<ObjectMirror>(object, "internal#entry",
1976  descriptionForEntry(context, object));
1977  }
1978  if (internalType == V8InternalValueType::kScope) {
1979  return std::make_unique<ObjectMirror>(object, "internal#scope",
1980  descriptionForScope(context, object));
1981  }
1982  if (internalType == V8InternalValueType::kPrivateMethod) {
1983  return std::make_unique<ObjectMirror>(
1984  object, "internal#privateMethod",
1985  descriptionForPrivateMethod(context, object));
1986  }
1987  size_t length = 0;
1988  if (isArrayLike(context, object, &length)) {
1989  return std::make_unique<ObjectMirror>(
1990  object, RemoteObject::SubtypeEnum::Array,
1991  descriptionForCollection(isolate, object, length));
1992  }
1993  return std::make_unique<ObjectMirror>(object,
1994  descriptionForObject(isolate, object));
1995 }
