59 template <typename T>
60 T ReverseBytes(T value, int block_bytes_log2) {
61  DCHECK((sizeof(value) == 4) || (sizeof(value) == 8));
62  DCHECK((1ULL << block_bytes_log2) <= sizeof(value));
63  // Split the 64-bit value into an 8-bit array, where b[0] is the least
64  // significant byte, and b[7] is the most significant.
65  uint8_t bytes[8];
66  uint64_t mask = 0xff00000000000000;
67  for (int i = 7; i >= 0; i--) {
68  bytes[i] = (static_cast<uint64_t>(value) & mask) >> (i * 8);
69  mask >>= 8;
70  }
71 
72  // Permutation tables for REV instructions.
73  // permute_table[0] is used by REV16_x, REV16_w
74  // permute_table[1] is used by REV32_x, REV_w
75  // permute_table[2] is used by REV_x
76  DCHECK((0 < block_bytes_log2) && (block_bytes_log2 < 4));
77  static const uint8_t permute_table[3][8] = {{6, 7, 4, 5, 2, 3, 0, 1},
78  {4, 5, 6, 7, 0, 1, 2, 3},
79  {0, 1, 2, 3, 4, 5, 6, 7}};
80  std::make_unsigned_t<T> result = 0;
81  for (int i = 0; i < 8; i++) {
82  result <<= 8;
83  result |= bytes[permute_table[block_bytes_log2 - 1][i]];
84  }
85  return result;
86 }
