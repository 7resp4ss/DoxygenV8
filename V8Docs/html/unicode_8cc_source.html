55 /**
56  * Look up a character in the Unicode table using a mix of binary and
57  * interpolation search. For a uniformly distributed array
58  * interpolation search beats binary search by a wide margin. However,
59  * in this case interpolation search degenerates because of some very
60  * high values in the lower end of the table so this function uses a
61  * combination. The average number of steps to look up the information
62  * about a character is around 10, slightly higher if there is no
63  * information available about the character.
64  */
65 static bool LookupPredicate(const int32_t* table, uint16_t size, uchar chr) {
66  static const int kEntryDist = 1;
67  uint16_t value = chr & (kChunkBits - 1);
68  unsigned int low = 0;
69  unsigned int high = size - 1;
70  while (high != low) {
71  unsigned int mid = low + ((high - low) >> 1);
72  uchar current_value = GetEntry(TableGet<kEntryDist>(table, mid));
73  // If we've found an entry less than or equal to this one, and the
74  // next one is not also less than this one, we've arrived.
75  if ((current_value <= value) &&
76  (mid + 1 == size ||
77  GetEntry(TableGet<kEntryDist>(table, mid + 1)) > value)) {
78  low = mid;
79  break;
80  } else if (current_value < value) {
81  low = mid + 1;
82  } else if (current_value > value) {
83  // If we've just checked the bottom-most value and it's not
84  // the one we're looking for, we're done.
85  if (mid == 0) break;
86  high = mid - 1;
87  }
88  }
89  int32_t field = TableGet<kEntryDist>(table, low);
90  uchar entry = GetEntry(field);
91  bool is_start = IsStart(field);
92  return (entry == value) || (entry < value && is_start);

137  // If we've just checked the bottom-most value and it's not
