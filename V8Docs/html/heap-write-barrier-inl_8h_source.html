24 // static
25 void WriteBarrier::CombinedWriteBarrierInternal(Tagged<HeapObject> host,
26  HeapObjectSlot slot,
27  Tagged<HeapObject> value,
28  WriteBarrierMode mode) {
29  DCHECK_EQ(mode, UPDATE_WRITE_BARRIER);
30 
31  MemoryChunk* host_chunk = MemoryChunk::FromHeapObject(host);
32  MemoryChunk* value_chunk = MemoryChunk::FromHeapObject(value);
33 
34  const bool is_marking = host_chunk->IsMarking();
35 
36  if (v8_flags.sticky_mark_bits) {
37  // TODO(333906585): Support shared barrier.
38  if (!HeapLayout::InYoungGeneration(host_chunk, host) &&
39  HeapLayout::InYoungGeneration(value_chunk, value)) {
40  // Generational or shared heap write barrier (old-to-new or
41  // old-to-shared).
42  CombinedGenerationalAndSharedBarrierSlow(host, slot.address(), value);
43  }
44  } else {
45  const bool pointers_from_here_are_interesting =
46  !host_chunk->IsYoungOrSharedChunk();
47  if (pointers_from_here_are_interesting &&
48  value_chunk->IsYoungOrSharedChunk()) {
49  // Generational or shared heap write barrier (old-to-new or
50  // old-to-shared).
51  CombinedGenerationalAndSharedBarrierSlow(host, slot.address(), value);
52  }
53  }
54 
55  // Marking barrier: mark value & record slots when marking is on.
56  if (V8_UNLIKELY(is_marking)) {
57  MarkingSlow(host, HeapObjectSlot(slot), value);
58  }
59 }

61 // static
62 inline WriteBarrierMode WriteBarrier::GetWriteBarrierModeForObject(
63  Tagged<HeapObject> object, const DisallowGarbageCollection& promise) {
64  if (v8_flags.disable_write_barriers) {
65  return SKIP_WRITE_BARRIER;
66  }
67  DCHECK(PageFlagsAreConsistent(object));
68  MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
69  if (chunk->IsMarking()) {
70  return UPDATE_WRITE_BARRIER;
71  }
72  if (HeapLayout::InYoungGeneration(chunk, object)) {
73  return SKIP_WRITE_BARRIER;
74  }
75  return UPDATE_WRITE_BARRIER;
76 }

85 void WriteBarrier::ForRelocInfo(Tagged<InstructionStream> host,
86  RelocInfo* rinfo, Tagged<HeapObject> value,
87  WriteBarrierMode mode) {
88  if (mode == SKIP_WRITE_BARRIER) {
89  SLOW_DCHECK(!WriteBarrier::IsRequired(host, value));
90  return;
91  }
92 
93  // Used during InstructionStream initialization where we update the write
94  // barriers together separate from the field writes.
95  if (mode == UNSAFE_SKIP_WRITE_BARRIER) {
96  DCHECK(!DisallowGarbageCollection::IsAllowed());
97  return;
98  }
99 
100  DCHECK_EQ(mode, UPDATE_WRITE_BARRIER);
101  GenerationalForRelocInfo(host, rinfo, value);
102  SharedForRelocInfo(host, rinfo, value);
103  MarkingForRelocInfo(host, rinfo, value);
104 }

137 // static
138 void WriteBarrier::ForEphemeronHashTable(Tagged<EphemeronHashTable> host,
139  ObjectSlot slot, Tagged<Object> value,
140  WriteBarrierMode mode) {
141  if (mode == SKIP_WRITE_BARRIER) {
142  SLOW_DCHECK(!WriteBarrier::IsRequired(host, value));
143  return;
144  }
145 
146  DCHECK_EQ(mode, UPDATE_WRITE_BARRIER);
147  if (!value.IsHeapObject()) return;
148 
149  MemoryChunk* host_chunk = MemoryChunk::FromHeapObject(host);
150 
151  Tagged<HeapObject> heap_object_value = Cast<HeapObject>(value);
152  MemoryChunk* value_chunk = MemoryChunk::FromHeapObject(heap_object_value);
153 
154  const bool pointers_from_here_are_interesting =
155  !host_chunk->IsYoungOrSharedChunk();
156  const bool is_marking = host_chunk->IsMarking();
157 
158  if (pointers_from_here_are_interesting &&
159  value_chunk->IsYoungOrSharedChunk()) {
160  CombinedGenerationalAndSharedEphemeronBarrierSlow(host, slot.address(),
161  heap_object_value);
162  }
163 
164  // Marking barrier: mark value & record slots when marking is on.
165  if (is_marking) {
166  MarkingSlow(host, HeapObjectSlot(slot), heap_object_value);
167  }
168 }

352 void WriteBarrier::ForCppHeapPointer(Tagged<CppHeapPointerWrapperObjectT> host,
353  CppHeapPointerSlot slot, void* value) {
354  // Note: this is currently a combined barrier for marking both the
355  // CppHeapPointerTable entry and the referenced object.
356 
357  if (V8_LIKELY(!IsMarking(host))) {
358 #if defined(CPPGC_YOUNG_GENERATION)
359  // There is no young-gen CppHeapPointerTable space so we should not mark
360  // the table entry in this case.
361  if (value) {
362  GenerationalBarrierForCppHeapPointer(host, value);
363  }
364 #endif
365  return;
366  }
367  MarkingBarrier* marking_barrier = CurrentMarkingBarrier(host);
368  if (marking_barrier->is_minor()) {
369  // TODO(v8:13012): We do not currently mark Oilpan objects while MinorMS is
370  // active. Once Oilpan uses a generational GC with incremental marking and
371  // unified heap, this barrier will be needed again.
372  return;
373  }
374 
375  MarkingSlowFromCppHeapWrappable(marking_barrier->heap(), host, slot, value);
376 }
