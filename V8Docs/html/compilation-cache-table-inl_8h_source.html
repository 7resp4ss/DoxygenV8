46 // The key in a script cache is a WeakFixedArray containing a weak pointer to
47 // the Script. The corresponding value can be either the root SharedFunctionInfo
48 // or undefined. The purpose of storing the root SharedFunctionInfo as the value
49 // is to keep it alive, not to save a lookup on the Script. A newly added entry
50 // always contains the root SharedFunctionInfo. After the root
51 // SharedFunctionInfo has aged sufficiently, it is replaced with undefined. In
52 // this way, all strong references to large objects are dropped, but there is
53 // still a way to get the Script if it happens to still be alive.
54 class ScriptCacheKey : public HashTableKey {

59  kEnd,

77  static std::optional<Tagged<String>> SourceFromObject(Tagged<Object> obj) {
78  DisallowGarbageCollection no_gc;
79  DCHECK(IsWeakFixedArray(obj));
80  Tagged<WeakFixedArray> array = Cast<WeakFixedArray>(obj);
81  DCHECK_EQ(array->length(), kEnd);
82 
83  Tagged<MaybeObject> maybe_script = array->get(kWeakScript);
84  if (Tagged<HeapObject> script; maybe_script.GetHeapObjectIfWeak(&script)) {
85  Tagged<PrimitiveHeapObject> source_or_undefined =
86  Cast<Script>(script)->source();
87  // Scripts stored in the script cache should always have a source string.
88  return Cast<String>(source_or_undefined);
89  }
90 
91  DCHECK(maybe_script.IsCleared());
92  return {};
93  }

111 uint32_t CompilationCacheShape::EvalHash(Tagged<String> source,
112  Tagged<SharedFunctionInfo> shared,
113  LanguageMode language_mode,
114  int position) {
115  uint32_t hash = source->EnsureHash();
116  if (shared->HasSourceCode()) {
117  // Instead of using the SharedFunctionInfo pointer in the hash
118  // code computation, we use a combination of the hash of the
119  // script source code and the start position of the calling scope.
120  // We do this to ensure that the cache entries can survive garbage
121  // collection.
122  Tagged<Script> script(Cast<Script>(shared->script()));
123  hash ^= Cast<String>(script->source())->EnsureHash();
124  }
125  static_assert(LanguageModeSize == 2);
126  if (is_strict(language_mode)) hash ^= 0x8000;
127  hash += position;
128  return hash;
129 }

131 uint32_t CompilationCacheShape::HashForObject(ReadOnlyRoots roots,
132  Tagged<Object> object) {
133  // Eval: The key field contains the hash as a Number.
134  if (IsNumber(object))
135  return static_cast<uint32_t>(Object::NumberValue(object));
136 
137  // Code: The key field contains the SFI key.
138  if (IsSharedFunctionInfo(object)) {
139  return Cast<SharedFunctionInfo>(object)->Hash();
140  }
141 
142  // Script.
143  if (IsWeakFixedArray(object)) {
144  return static_cast<uint32_t>(Smi::ToInt(
145  Cast<WeakFixedArray>(object)->get(ScriptCacheKey::kHash).ToSmi()));
146  }
147 
148  // RegExpData: The key field (and the value field) contains the RegExpData
149  // object.
150  if (IsRegExpDataWrapper(object)) {
151  Tagged<RegExpDataWrapper> re_wrapper = Cast<RegExpDataWrapper>(object);
152  Isolate* isolate = GetIsolateFromWritableObject(re_wrapper);
153  Tagged<RegExpData> data = re_wrapper->data(isolate);
154  return RegExpHash(data->source(), Smi::FromInt(data->flags()));
155  }
156 
157  // Eval: See EvalCacheKey::ToHandle for the encoding.
158  Tagged<FixedArray> val = Cast<FixedArray>(object);
159  DCHECK_EQ(val->map(), roots.fixed_cow_array_map());
160  DCHECK_EQ(4, val->length());
161  Tagged<String> source = Cast<String>(val->get(1));
162  int language_unchecked = Smi::ToInt(val->get(2));
163  DCHECK(is_valid_language_mode(language_unchecked));
164  LanguageMode language_mode = static_cast<LanguageMode>(language_unchecked);
165  int position = Smi::ToInt(val->get(3));
166  Tagged<Object> shared = val->get(0);
167  return EvalHash(source, Cast<SharedFunctionInfo>(shared), language_mode,
168  position);
169 }
