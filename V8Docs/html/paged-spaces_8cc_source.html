54 // We have hit the end of the page and should advance to the next block of
55 // objects. This happens at the end of the page.
56 bool PagedSpaceObjectIterator::AdvanceToNextPage() {
57  if (current_page_ == page_range_.end()) return false;
58  const PageMetadata* cur_page = *(current_page_++);

60  cur_ = heap_objects.begin();

95 void PagedSpaceBase::TearDown() {
96  const bool is_marking = heap_->isolate()->isolate_data()->is_marking();
97  while (!memory_chunk_list_.Empty()) {
98  MutablePageMetadata* chunk = memory_chunk_list_.front();
99  memory_chunk_list_.Remove(chunk);
100  const auto mode = (id_ == NEW_SPACE || id_ == OLD_SPACE)
101  ? MemoryAllocator::FreeMode::kPool
102  : MemoryAllocator::FreeMode::kImmediately;
103  if (mode == MemoryAllocator::FreeMode::kPool &&
104  (is_marking || V8_ENABLE_STICKY_MARK_BITS_BOOL)) {
105  chunk->ClearLiveness();
106  }
107  heap()->memory_allocator()->Free(mode, chunk);
108  }
109  accounting_stats_.Clear();
110 }

112 void PagedSpaceBase::MergeCompactionSpace(CompactionSpace* other) {
113  base::MutexGuard guard(mutex());
114 
115  DCHECK_NE(NEW_SPACE, identity());
116  DCHECK_NE(NEW_SPACE, other->identity());
117 
118  // Move over pages.
119  for (auto it = other->begin(); it != other->end();) {
120  PageMetadata* p = *(it++);
121 
122  // Ensure that pages are initialized before objects on it are discovered by
123  // concurrent markers.
124  p->Chunk()->InitializationMemoryFence();
125 
126  // Relinking requires the category to be unlinked.
127  other->RemovePage(p);
128  AddPage(p);
129  DCHECK_IMPLIES(
130  !p->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE),
131  p->AvailableInFreeList() == p->AvailableInFreeListFromAllocatedBytes());
132 
133  // TODO(leszeks): Here we should allocation step, but:
134  // 1. Allocation groups are currently not handled properly by the sampling
135  // allocation profiler, and
136  // 2. Observers might try to take the space lock, which isn't reentrant.
137  // We'll have to come up with a better solution for allocation stepping
138  // before shipping, which will likely be using LocalHeap.
139  }
140  const bool is_from_client_heap =
141  (other->destination_heap() ==
142  CompactionSpace::DestinationHeap::kSharedSpaceHeap);
143  DCHECK_IMPLIES(is_from_client_heap, identity() == SHARED_SPACE);
144  for (auto p : other->GetNewPages()) {
145  heap()->NotifyOldGenerationExpansion(
146  heap()->main_thread_local_heap(), identity(), p,
147  is_from_client_heap
148  ? Heap::OldGenerationExpansionNotificationOrigin::kFromClientHeap
149  : Heap::OldGenerationExpansionNotificationOrigin::kFromSameHeap);
150  }
151 
152  DCHECK_EQ(0u, other->Size());
153  DCHECK_EQ(0u, other->Capacity());
154 }

164 void PagedSpaceBase::IncrementCommittedPhysicalMemory(size_t increment_value) {
165  if (!base::OS::HasLazyCommits() || increment_value == 0) return;
166  size_t old_value = committed_physical_memory_.fetch_add(
167  increment_value, std::memory_order_relaxed);
168  USE(old_value);
169  DCHECK_LT(old_value, old_value + increment_value);
170 }

172 void PagedSpaceBase::DecrementCommittedPhysicalMemory(size_t decrement_value) {
173  if (!base::OS::HasLazyCommits() || decrement_value == 0) return;
174  size_t old_value = committed_physical_memory_.fetch_sub(
175  decrement_value, std::memory_order_relaxed);
176  USE(old_value);
177  DCHECK_GT(old_value, old_value - decrement_value);
178 }

201 void PagedSpaceBase::RefineAllocatedBytesAfterSweeping(PageMetadata* page) {
202  CHECK(page->SweepingDone());
203  // The live_byte on the page was accounted in the space allocated
204  // bytes counter. After sweeping allocated_bytes() contains the
205  // accurate live byte count on the page.
206  size_t old_counter = page->live_bytes();
207  size_t new_counter = page->allocated_bytes();
208  DCHECK_GE(old_counter, new_counter);
209  if (old_counter > new_counter) {
210  size_t counter_diff = old_counter - new_counter;
211  if (identity() == NEW_SPACE) size_at_last_gc_ -= counter_diff;
212  DecreaseAllocatedBytes(counter_diff, page);
213  DCHECK_EQ(new_counter, accounting_stats_.AllocatedOnPage(page));
214  AdjustDifferenceInAllocatedBytes(counter_diff);
215  }
216  if (!v8_flags.sticky_mark_bits) {
217  // With sticky mark-bits the counter is reset on unmarking.
218  page->SetLiveBytes(0);
219  }
220 }

230 void PagedSpaceBase::AddPageImpl(PageMetadata* page) {
231  DCHECK_NOT_NULL(page);
232  CHECK(page->SweepingDone());
233  page->set_owner(this);
234  DCHECK_IMPLIES(identity() == NEW_SPACE,
235  page->Chunk()->IsFlagSet(MemoryChunk::TO_PAGE));
236  DCHECK_IMPLIES(identity() != NEW_SPACE,
237  !page->Chunk()->IsFlagSet(MemoryChunk::TO_PAGE));
238  memory_chunk_list_.PushBack(page);
239  AccountCommitted(page->size());
240  IncreaseCapacity(page->area_size());
241  IncreaseAllocatedBytes(page->allocated_bytes(), page);
242  ForAll<ExternalBackingStoreType>(
243  [this, page](ExternalBackingStoreType type, int index) {
244  IncrementExternalBackingStoreBytes(
245  type, page->ExternalBackingStoreBytes(type));
246  });
247  IncrementCommittedPhysicalMemory(page->CommittedPhysicalMemory());
248 }

255 void PagedSpaceBase::RemovePage(PageMetadata* page) {
256  CHECK(page->SweepingDone());
257  DCHECK_IMPLIES(identity() == NEW_SPACE,
258  page->Chunk()->IsFlagSet(MemoryChunk::TO_PAGE));
259  memory_chunk_list_.Remove(page);
260  UnlinkFreeListCategories(page);
261  // Pages are only removed from new space when they are promoted to old space
262  // during a GC. This happens after sweeping as started and the allocation
263  // counters have been reset.
264  DCHECK_IMPLIES(identity() == NEW_SPACE,
265  heap()->gc_state() != Heap::NOT_IN_GC);
266  if (identity() == NEW_SPACE) {
267  page->ReleaseFreeListCategories();
268  } else {
269  DecreaseAllocatedBytes(page->allocated_bytes(), page);
270  free_list()->decrease_wasted_bytes(page->wasted_memory());
271  }
272  DecreaseCapacity(page->area_size());
273  AccountUncommitted(page->size());
274  ForAll<ExternalBackingStoreType>(
275  [this, page](ExternalBackingStoreType type, int index) {
276  DecrementExternalBackingStoreBytes(
277  type, page->ExternalBackingStoreBytes(type));
278  });
279  DecrementCommittedPhysicalMemory(page->CommittedPhysicalMemory());
280 }

290 bool PagedSpaceBase::TryExpand(LocalHeap* local_heap, AllocationOrigin origin) {
291  DCHECK_EQ(!local_heap, origin == AllocationOrigin::kGC);
292  const size_t accounted_size =
293  MemoryChunkLayout::AllocatableMemoryInMemoryChunk(identity());
294  if (origin != AllocationOrigin::kGC && identity() != NEW_SPACE) {
295  base::MutexGuard expansion_guard(heap_->heap_expansion_mutex());
296  if (!heap()->IsOldGenerationExpansionAllowed(accounted_size,
297  expansion_guard)) {
298  return false;
299  }
300  }
301  const MemoryAllocator::AllocationMode allocation_mode =
302  (identity() == NEW_SPACE || identity() == OLD_SPACE)
303  ? MemoryAllocator::AllocationMode::kUsePool
304  : MemoryAllocator::AllocationMode::kRegular;
305  PageMetadata* page = heap()->memory_allocator()->AllocatePage(
306  allocation_mode, this, executable());
307  if (page == nullptr) return false;
308  DCHECK_EQ(page->area_size(), accounted_size);
309  ConcurrentAllocationMutex guard(this);
310  AddPage(page);
311  if (origin != AllocationOrigin::kGC && identity() != NEW_SPACE) {
312  heap()->NotifyOldGenerationExpansion(local_heap, identity(), page);
313  }
314  Free(page->area_start(), page->area_size());
315  NotifyNewPage(page);
316  return true;
317 }

366 #ifdef VERIFY_HEAP
367 void PagedSpaceBase::Verify(Isolate* isolate,
368  SpaceVerificationVisitor* visitor) const {
369  CHECK_IMPLIES(identity() != NEW_SPACE, size_at_last_gc_ == 0);
370 
371  size_t external_space_bytes[static_cast<int>(
372  ExternalBackingStoreType::kNumValues)] = {0};
373  PtrComprCageBase cage_base(isolate);
374  for (const PageMetadata* page : *this) {
375  size_t external_page_bytes[static_cast<int>(
376  ExternalBackingStoreType::kNumValues)] = {0};
377 
378  CHECK_EQ(page->owner(), this);
379  CHECK_IMPLIES(identity() != NEW_SPACE, page->AllocatedLabSize() == 0);
380  visitor->VerifyPage(page);
381 
382  CHECK(page->SweepingDone());
383  Address end_of_previous_object = page->area_start();
384  Address top = page->area_end();
385 
386  for (Tagged<HeapObject> object : HeapObjectRange(page)) {
387  CHECK(end_of_previous_object <= object.address());
388 
389  // Invoke verification method for each object.
390  visitor->VerifyObject(object);
391 
392  // All the interior pointers should be contained in the heap.
393  int size = object->Size(cage_base);
394  CHECK(object.address() + size <= top);
395  end_of_previous_object = object.address() + size;
396 
397  if (IsExternalString(object, cage_base)) {
398  Tagged<ExternalString> external_string = Cast<ExternalString>(object);
399  size_t payload_size = external_string->ExternalPayloadSize();
400  external_page_bytes[static_cast<int>(
401  ExternalBackingStoreType::kExternalString)] += payload_size;
402  }
403  }
404  ForAll<ExternalBackingStoreType>(
405  [page, external_page_bytes, &external_space_bytes](
406  ExternalBackingStoreType type, int index) {
407  CHECK_EQ(external_page_bytes[index],
408  page->ExternalBackingStoreBytes(type));
409  external_space_bytes[index] += external_page_bytes[index];
410  });
411 
412  visitor->VerifyPageDone(page);
413  }
414  ForAll<ExternalBackingStoreType>(
415  [this, external_space_bytes](ExternalBackingStoreType type, int index) {
416  if (type == ExternalBackingStoreType::kArrayBuffer) {
417  return;
418  }
419  CHECK_EQ(external_space_bytes[index], ExternalBackingStoreBytes(type));
420  });
421 
422  if (!v8_flags.concurrent_array_buffer_sweeping) {
423  if (identity() == OLD_SPACE) {
424  size_t bytes = heap()->array_buffer_sweeper()->old().BytesSlow();
425  CHECK_EQ(bytes, ExternalBackingStoreBytes(
426  ExternalBackingStoreType::kArrayBuffer));
427  } else if (identity() == NEW_SPACE) {
428  CHECK(v8_flags.minor_ms);
429  size_t bytes = heap()->array_buffer_sweeper()->young().BytesSlow();
430  CHECK_EQ(bytes, ExternalBackingStoreBytes(
431  ExternalBackingStoreType::kArrayBuffer));
432  }
433  }
434 
435 #ifdef DEBUG
436  VerifyCountersAfterSweeping(isolate->heap());
437 #endif
438 }

440 void PagedSpaceBase::VerifyLiveBytes() const {
441  MarkingState* marking_state = heap()->marking_state();
442  PtrComprCageBase cage_base(heap()->isolate());
443  for (const PageMetadata* page : *this) {
444  CHECK(page->SweepingDone());
445  int black_size = 0;
446  for (Tagged<HeapObject> object : HeapObjectRange(page)) {
447  // All the interior pointers should be contained in the heap.
448  if (marking_state->IsMarked(object)) {
449  black_size += object->Size(cage_base);
450  }
451  }
452  CHECK_LE(black_size, page->live_bytes());
453  }
454 }

457 #ifdef DEBUG
458 void PagedSpaceBase::VerifyCountersAfterSweeping(Heap* heap) const {
459  size_t total_capacity = 0;
460  size_t total_allocated = 0;
461  PtrComprCageBase cage_base(heap->isolate());
462  for (const PageMetadata* page : *this) {
463  DCHECK(page->SweepingDone());
464  total_capacity += page->area_size();
465  size_t real_allocated = 0;
466  for (Tagged<HeapObject> object : HeapObjectRange(page)) {
467  if (!IsFreeSpaceOrFiller(object)) {
468  real_allocated +=
469  ALIGN_TO_ALLOCATION_ALIGNMENT(object->Size(cage_base));
470  }
471  }
472  total_allocated += page->allocated_bytes();
473  // The real size can be smaller than the accounted size if array trimming,
474  // object slack tracking happened after sweeping.
475  DCHECK_LE(real_allocated, accounting_stats_.AllocatedOnPage(page));
476  DCHECK_EQ(page->allocated_bytes(), accounting_stats_.AllocatedOnPage(page));
477  }
478  DCHECK_EQ(total_capacity, accounting_stats_.Capacity());
479  DCHECK_EQ(total_allocated, accounting_stats_.Size());
480 }

542 void PagedSpaceBase::RefillFreeList() {
543  // Any PagedSpace might invoke RefillFreeList.
544  DCHECK(identity() == OLD_SPACE || identity() == CODE_SPACE ||
545  identity() == SHARED_SPACE || identity() == NEW_SPACE ||
546  identity() == TRUSTED_SPACE || identity() == SHARED_TRUSTED_SPACE);
547  DCHECK_IMPLIES(identity() == NEW_SPACE, heap_->IsMainThread());
548  DCHECK(!is_compaction_space());
549 
550  for (PageMetadata* p : heap()->sweeper()->GetAllSweptPagesSafe(this)) {
551  // We regularly sweep NEVER_ALLOCATE_ON_PAGE pages. We drop the freelist
552  // entries here to make them unavailable for allocations.
553  if (p->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE)) {
554  free_list_->EvictFreeListItems(p);
555  }
556 
557  ConcurrentAllocationMutex guard(this);
558  DCHECK_EQ(this, p->owner());
559  RefineAllocatedBytesAfterSweeping(p);
560  RelinkFreeListCategories(p);
561  }
562 }

571 void CompactionSpace::NotifyNewPage(PageMetadata* page) {
572  // Incremental marking can be running on the main thread isolate, so when
573  // allocating a new page for the client's compaction space we can get a black
574  // allocated page. This is fine, since the page is not observed the main
575  // isolate until it's merged.
576  DCHECK_IMPLIES(identity() != SHARED_SPACE ||
577  destination_heap() != DestinationHeap::kSharedSpaceHeap,
578  !page->Chunk()->IsFlagSet(MemoryChunk::BLACK_ALLOCATED));
579  new_pages_.push_back(page);
580 }

582 void CompactionSpace::RefillFreeList() {
583  DCHECK_NE(NEW_SPACE, identity());
584 
585  Sweeper* sweeper = heap()->sweeper();
586  size_t added = 0;
587  PageMetadata* p = nullptr;
588  while ((added <= kCompactionMemoryWanted) &&
589  (p = sweeper->GetSweptPageSafe(this))) {
590  // We regularly sweep NEVER_ALLOCATE_ON_PAGE pages. We drop the freelist
591  // entries here to make them unavailable for allocations.
592  if (p->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE)) {
593  free_list()->EvictFreeListItems(p);
594  }
595 
596  // Only during compaction pages can actually change ownership. This is
597  // safe because there exists no other competing action on the page links
598  // during compaction.
599  DCHECK_NE(this, p->owner());
600  PagedSpace* owner = static_cast<PagedSpace*>(p->owner());
601  base::MutexGuard guard(owner->mutex());
602  owner->RefineAllocatedBytesAfterSweeping(p);
603  owner->RemovePage(p);
604  added += AddPage(p);
605  added += p->wasted_memory();
606  }
