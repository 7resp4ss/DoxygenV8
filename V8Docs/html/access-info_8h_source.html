32 // This class encapsulates all information required to access a certain element.
33 class ElementAccessInfo final {
34  public:
35  ElementAccessInfo(ZoneVector<MapRef>&& lookup_start_object_maps,
36  ElementsKind elements_kind, Zone* zone);
37  // For Proxy-on-Prototype mode.
38  ElementAccessInfo(MapRef map, ObjectRef accessor, ObjectRef target,
39  bool string_keys, Zone* zone);
40 
41  ElementsKind elements_kind() const { return elements_kind_; }
42  ZoneVector<MapRef> const& lookup_start_object_maps() const {
43  return lookup_start_object_maps_;
44  }
45  ZoneVector<MapRef> const& transition_sources() const {
46  return transition_sources_;
47  }
48 
49  void AddTransitionSource(MapRef map) {
50  CHECK_EQ(lookup_start_object_maps_.size(), 1);
51  transition_sources_.push_back(map);
52  }
53 
54  // Indicates whether this {ElementAccessInfo} is for the standard case
55  // where elements are directly on the receiver, or the special case where
56  // Proxy traps found on the prototype chain are handling the access.
57  bool is_proxy_on_prototype() const {
58  // Either both of {accessor_} and {target_} must be present, or neither.
59  DCHECK_EQ(accessor_.has_value(), target_.has_value());
60  DCHECK_IMPLIES(!accessor_.has_value(), !string_keys_);
61  return accessor_.has_value();
62  }
63 
64  // Only for Proxy-on-Prototype mode: records whether the Proxy trap expects
65  // string keys or integer keys.
66  bool string_keys() const { return string_keys_; }
67  OptionalObjectRef accessor() const { return accessor_; }
68  OptionalObjectRef target() const { return target_; }
69 
70  private:
71  ElementsKind elements_kind_;
72  bool string_keys_{false};
73  ZoneVector<MapRef> lookup_start_object_maps_;
74  ZoneVector<MapRef> transition_sources_;
75  OptionalObjectRef accessor_;
76  OptionalObjectRef target_;
77 };
