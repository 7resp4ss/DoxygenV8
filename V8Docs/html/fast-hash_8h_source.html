16 // fast_hash_combine() / fast_hash_value() produce a bad but very fast to
17 // compute hash, intended for hash-tables and only usable for data that is
18 // sufficiently random already and has high variance in their low bits.
19 
20 V8_INLINE size_t fast_hash_combine() { return 0u; }
21 V8_INLINE size_t fast_hash_combine(size_t acc) { return acc; }
22 V8_INLINE size_t fast_hash_combine(size_t acc, size_t value) {
23  return 17 * acc + value;
24 }

28 template <class T>
29 struct fast_hash {
30  size_t operator()(const T& v) const {
31  if constexpr (std::is_enum_v<T>) {
32  return static_cast<size_t>(v);
33  } else {
34  return base::hash<T>()(v);
35  }
36  }

39 template <typename T>

42  return v.has_value() ? (fast_hash()(*v) << 1) + 1 : 0;

53 template <class... Ts>

66 template <typename T, typename... Ts>
67 V8_INLINE size_t fast_hash_combine(T const& v, Ts const&... vs) {
68  return fast_hash_combine(fast_hash_combine(vs...), fast_hash<T>()(v));
69 }

71 template <typename Iterator>
72 V8_INLINE size_t fast_hash_range(Iterator first, Iterator last) {
73  size_t acc = 0;
74  for (; first != last; ++first) {
75  acc = fast_hash_combine(acc, *first);
76  }
77  return acc;
78 }
