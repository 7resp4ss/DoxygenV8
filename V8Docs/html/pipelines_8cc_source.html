29 bool Pipeline::AllocateRegisters(const RegisterConfiguration* config,
30  CallDescriptor* call_descriptor,
31  bool run_verifier) {
32  // Don't track usage for this zone in compiler stats.
33  std::unique_ptr<Zone> verifier_zone;
34  RegisterAllocatorVerifier* verifier = nullptr;
35  if (run_verifier) {
36  AccountingAllocator* allocator = data()->allocator();
37  DCHECK_NOT_NULL(allocator);
38  verifier_zone.reset(
39  new Zone(allocator, kRegisterAllocatorVerifierZoneName));
40  verifier = verifier_zone->New<RegisterAllocatorVerifier>(
41  verifier_zone.get(), config, data()->sequence(), data()->frame());
42  }
43 
44 #ifdef DEBUG
45  data_->sequence()->ValidateEdgeSplitForm();
46  data_->sequence()->ValidateDeferredBlockEntryPaths();
47  data_->sequence()->ValidateDeferredBlockExitPaths();
48 #endif
49 
50  data_->InitializeRegisterComponent(config, call_descriptor);
51 
52  RUN_MAYBE_ABORT(MeetRegisterConstraintsPhase);
53  RUN_MAYBE_ABORT(ResolvePhisPhase);
54  RUN_MAYBE_ABORT(BuildLiveRangesPhase);
55  RUN_MAYBE_ABORT(BuildLiveRangeBundlesPhase);
56 
57  TraceSequence("before register allocation");
58  if (verifier != nullptr) {
59  CHECK(!data_->register_allocation_data()->ExistsUseWithoutDefinition());
60  CHECK(data_->register_allocation_data()
61  ->RangesDefinedInDeferredStayInDeferred());
62  }
63 
64  if (data_->info()->trace_turbo_json() && !MayHaveUnverifiableGraph()) {
65  TurboCfgFile tcf(data_->isolate());
66  tcf << AsC1VRegisterAllocationData("PreAllocation",
67  data_->register_allocation_data());
68  }
69 
70  RUN_MAYBE_ABORT(AllocateGeneralRegistersPhase<LinearScanAllocator>);
71 
72  if (data_->sequence()->HasFPVirtualRegisters()) {
73  RUN_MAYBE_ABORT(AllocateFPRegistersPhase<LinearScanAllocator>);
74  }
75 
76  if (data_->sequence()->HasSimd128VirtualRegisters() &&
77  (kFPAliasing == AliasingKind::kIndependent)) {
78  RUN_MAYBE_ABORT(AllocateSimd128RegistersPhase<LinearScanAllocator>);
79  }
80 
81  RUN_MAYBE_ABORT(DecideSpillingModePhase);
82  RUN_MAYBE_ABORT(AssignSpillSlotsPhase);
83  RUN_MAYBE_ABORT(CommitAssignmentPhase);
84 
85  // TODO(chromium:725559): remove this check once
86  // we understand the cause of the bug. We keep just the
87  // check at the end of the allocation.
88  if (verifier != nullptr) {
89  verifier->VerifyAssignment("Immediately after CommitAssignmentPhase.");
90  }
91 
92  RUN_MAYBE_ABORT(ConnectRangesPhase);
93 
94  RUN_MAYBE_ABORT(ResolveControlFlowPhase);
95 
96  RUN_MAYBE_ABORT(PopulateReferenceMapsPhase);
97 
98  if (v8_flags.turbo_move_optimization) {
99  RUN_MAYBE_ABORT(OptimizeMovesPhase);
100  }
101 
102  TraceSequence("after register allocation");
103 
104  if (verifier != nullptr) {
105  verifier->VerifyAssignment("End of regalloc pipeline.");
106  verifier->VerifyGapMoves();
107  }
108 
109  if (data_->info()->trace_turbo_json() && !MayHaveUnverifiableGraph()) {
110  TurboCfgFile tcf(data_->isolate());
111  tcf << AsC1VRegisterAllocationData("CodeGen",
112  data_->register_allocation_data());
113  }
114 
115  data()->ClearRegisterComponent();
116  return true;
117 }

119 [[nodiscard]] bool Pipeline::GenerateCode(
120  Linkage* linkage, std::shared_ptr<OsrHelper> osr_helper,
121  JumpOptimizationInfo* jump_optimization_info,
122  const ProfileDataFromFile* profile, int initial_graph_hash) {
123  // Run code generation. If we optimize jumps, we repeat this a second time.
124  data()->InitializeCodegenComponent(osr_helper, jump_optimization_info);
125 
126  // Perform instruction selection and register allocation.
127  if (!PrepareForInstructionSelection(profile)) return false;
128  if (!SelectInstructions(linkage)) return false;
129  if (!AllocateRegisters(linkage->GetIncomingDescriptor())) return false;
130  if (!AssembleCode(linkage)) return false;
131 
132  if (v8_flags.turbo_profiling) {
133  info()->profiler_data()->SetHash(initial_graph_hash);
134  }
135 
136  if (jump_optimization_info && jump_optimization_info->is_optimizable()) {
137  // Reset data for a second run of instruction selection.
138  data()->ClearCodegenComponent();
139  jump_optimization_info->set_optimizing();
140 
141  // Perform instruction selection and register allocation.
142  data()->InitializeCodegenComponent(osr_helper, jump_optimization_info);
143  if (!SelectInstructions(linkage)) {
144  return false;
145  }
146  if (!AllocateRegisters(linkage->GetIncomingDescriptor())) return false;
147  // Generate the final machine code.
148  if (!AssembleCode(linkage)) return false;
149  }
150  return true;
151 }

153 void BuiltinPipeline::OptimizeBuiltin() {
154  Tracing::Scope tracing_scope(data()->info());
155 
156  CHECK(Run<CsaEarlyMachineOptimizationPhase>());
157  CHECK(Run<CsaLoadEliminationPhase>());
158  CHECK(Run<CsaLateEscapeAnalysisPhase>());
159  CHECK(Run<CsaBranchEliminationPhase>());
160 
161  if (data()->isolate()->builtins_effects_analyzer() != nullptr) {
162  // Effect computations has to run before CsaMemoryOptimizationPhase, so that
163  // Allocate aren't lowered to builtin calls.
164  CHECK(Run<CsaEffectsComputationPhase>());
165  }
166 
167  CHECK(Run<CsaMemoryOptimizationPhase>());
168 
169  if (v8_flags.turboshaft_enable_debug_features) {
170  CHECK(Run<DebugFeatureLoweringPhase>());
171  }
172 
173  CHECK(Run<CodeEliminationAndSimplificationPhase>());
174 }
