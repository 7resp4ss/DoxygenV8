26 class SharedToCounterMap
27  : public base::TemplateHashMapImpl<Tagged<SharedFunctionInfo>, uint32_t,
28  base::KeyEqualityMatcher<Tagged<Object>>,
29  base::DefaultAllocationPolicy> {
30  public:
31  using Entry =
32  base::TemplateHashMapEntry<Tagged<SharedFunctionInfo>, uint32_t>;
33  inline void Add(Tagged<SharedFunctionInfo> key, uint32_t count) {
34  Entry* entry = LookupOrInsert(key, Hash(key), []() { return 0; });
35  uint32_t old_count = entry->value;
36  if (UINT32_MAX - count < old_count) {
37  entry->value = UINT32_MAX;
38  } else {
39  entry->value = old_count + count;
40  }
41  }
42 
43  inline uint32_t Get(Tagged<SharedFunctionInfo> key) {
44  Entry* entry = Lookup(key, Hash(key));
45  if (entry == nullptr) return 0;
46  return entry->value;
47  }
48 
49  private:
50  static uint32_t Hash(Tagged<SharedFunctionInfo> key) {
51  return static_cast<uint32_t>(key.ptr());
52  }
53 
54  DISALLOW_GARBAGE_COLLECTION(no_gc)
55 };

64 bool CompareCoverageBlock(const CoverageBlock& a, const CoverageBlock& b) {
65  DCHECK_NE(kNoSourcePosition, a.start);
66  DCHECK_NE(kNoSourcePosition, b.start);
67  if (a.start == b.start) return a.end > b.end;
68  return a.start < b.start;

76 std::vector<CoverageBlock> GetSortedBlockData(
77  Isolate* isolate, Tagged<SharedFunctionInfo> shared) {
78  DCHECK(shared->HasCoverageInfo(isolate));
79 
80  Tagged<CoverageInfo> coverage_info =
81  Cast<CoverageInfo>(shared->GetDebugInfo(isolate)->coverage_info());
82 
83  std::vector<CoverageBlock> result;
84  if (coverage_info->slot_count() == 0) return result;
85 
86  for (int i = 0; i < coverage_info->slot_count(); i++) {
87  const int start_pos = coverage_info->slots_start_source_position(i);
88  const int until_pos = coverage_info->slots_end_source_position(i);
89  const int count = coverage_info->slots_block_count(i);
90 
91  DCHECK_NE(kNoSourcePosition, start_pos);
92  result.emplace_back(start_pos, until_pos, count);
93  }
94 
95  SortBlockData(result);
96 
97  return result;
98 }

100 // A utility class to simplify logic for performing passes over block coverage
101 // ranges. Provides access to the implicit tree structure of ranges (i.e. access
102 // to parent and sibling blocks), and supports efficient in-place editing and
103 // deletion. The underlying backing store is the array of CoverageBlocks stored
104 // on the CoverageFunction.
105 class CoverageBlockIterator final {
106  public:
107  explicit CoverageBlockIterator(CoverageFunction* function)
108  : function_(function) {
109  DCHECK(std::is_sorted(function_->blocks.begin(), function_->blocks.end(),
110  CompareCoverageBlock));
111  }
112 
113  ~CoverageBlockIterator() {
114  Finalize();
115  DCHECK(std::is_sorted(function_->blocks.begin(), function_->blocks.end(),
116  CompareCoverageBlock));
117  }
118 
119  bool HasNext() const {
120  return read_index_ + 1 < static_cast<int>(function_->blocks.size());
121  }
122 
123  bool Next() {
124  if (!HasNext()) {
125  if (!ended_) MaybeWriteCurrent();
126  ended_ = true;
127  return false;
128  }
129 
130  // If a block has been deleted, subsequent iteration moves trailing blocks
131  // to their updated position within the array.
132  MaybeWriteCurrent();
133 
134  if (read_index_ == -1) {
135  // Initialize the nesting stack with the function range.
136  nesting_stack_.emplace_back(function_->start, function_->end,
137  function_->count);
138  } else if (!delete_current_) {
139  nesting_stack_.emplace_back(GetBlock());
140  }
141 
142  delete_current_ = false;
143  read_index_++;
144 
145  DCHECK(IsActive());
146 
147  CoverageBlock& block = GetBlock();
148  while (nesting_stack_.size() > 1 &&
149  nesting_stack_.back().end <= block.start) {
150  nesting_stack_.pop_back();
151  }
152 
153  DCHECK_IMPLIES(block.start >= function_->end,
154  block.end == kNoSourcePosition);
155  DCHECK_NE(block.start, kNoSourcePosition);
156  DCHECK_LE(block.end, GetParent().end);
157 
158  return true;
159  }
160 
161  CoverageBlock& GetBlock() {
162  DCHECK(IsActive());
163  return function_->blocks[read_index_];
164  }
165 
166  CoverageBlock& GetNextBlock() {
167  DCHECK(IsActive());
168  DCHECK(HasNext());
169  return function_->blocks[read_index_ + 1];
170  }
171 
172  CoverageBlock& GetPreviousBlock() {
173  DCHECK(IsActive());
174  DCHECK_GT(read_index_, 0);
175  return function_->blocks[read_index_ - 1];
176  }
177 
178  CoverageBlock& GetParent() {
179  DCHECK(IsActive());
180  return nesting_stack_.back();
181  }
182 
183  bool HasSiblingOrChild() {
184  DCHECK(IsActive());
185  return HasNext() && GetNextBlock().start < GetParent().end;
186  }
187 
188  CoverageBlock& GetSiblingOrChild() {
189  DCHECK(HasSiblingOrChild());
190  DCHECK(IsActive());
191  return GetNextBlock();
192  }
193 
194  // A range is considered to be at top level if its parent range is the
195  // function range.
196  bool IsTopLevel() const { return nesting_stack_.size() == 1; }
197 
198  void DeleteBlock() {
199  DCHECK(!delete_current_);
200  DCHECK(IsActive());
201  delete_current_ = true;
202  }
203 
204  private:
205  void MaybeWriteCurrent() {
206  if (delete_current_) return;
207  if (read_index_ >= 0 && write_index_ != read_index_) {
208  function_->blocks[write_index_] = function_->blocks[read_index_];
209  }
210  write_index_++;
211  }
212 
213  void Finalize() {
214  while (Next()) {
215  // Just iterate to the end.
216  }
217  function_->blocks.resize(write_index_);
218  }
219 
220  bool IsActive() const { return read_index_ >= 0 && !ended_; }
221 
222  CoverageFunction* function_;
223  std::vector<CoverageBlock> nesting_stack_;
224  bool ended_ = false;
225  bool delete_current_ = false;
226  int read_index_ = -1;
227  int write_index_ = -1;
228 };

234 void MergeDuplicateRanges(CoverageFunction* function) {
235  CoverageBlockIterator iter(function);
236 
237  while (iter.Next() && iter.HasNext()) {
238  CoverageBlock& block = iter.GetBlock();
239  CoverageBlock& next_block = iter.GetNextBlock();
240 
241  if (!HaveSameSourceRange(block, next_block)) continue;
242 
243  DCHECK_NE(kNoSourcePosition, block.end); // Non-singleton range.
244  next_block.count = std::max(block.count, next_block.count);
245  iter.DeleteBlock();
246  }
247 }

249 // Rewrite position singletons (produced by unconditional control flow
250 // like return statements, and by continuation counters) into source
251 // ranges that end at the next sibling range or the end of the parent
252 // range, whichever comes first.
253 void RewritePositionSingletonsToRanges(CoverageFunction* function) {
254  CoverageBlockIterator iter(function);
255 
256  while (iter.Next()) {
257  CoverageBlock& block = iter.GetBlock();
258  CoverageBlock& parent = iter.GetParent();
259 
260  if (block.start >= function->end) {
261  DCHECK_EQ(block.end, kNoSourcePosition);
262  iter.DeleteBlock();
263  } else if (block.end == kNoSourcePosition) {
264  // The current block ends at the next sibling block (if it exists) or the
265  // end of the parent block otherwise.
266  if (iter.HasSiblingOrChild()) {
267  block.end = iter.GetSiblingOrChild().start;
268  } else if (iter.IsTopLevel()) {
269  // See https://crbug.com/v8/6661. Functions are special-cased because
270  // we never want the closing brace to be uncovered. This is mainly to
271  // avoid a noisy UI.
272  block.end = parent.end - 1;
273  } else {
274  block.end = parent.end;
275  }
276  }
277  }
278 }

280 void MergeConsecutiveRanges(CoverageFunction* function) {
281  CoverageBlockIterator iter(function);
282 
283  while (iter.Next()) {
284  CoverageBlock& block = iter.GetBlock();
285 
286  if (iter.HasSiblingOrChild()) {
287  CoverageBlock& sibling = iter.GetSiblingOrChild();
288  if (sibling.start == block.end && sibling.count == block.count) {
289  // Best-effort: this pass may miss mergeable siblings in the presence of
290  // child blocks.
291  sibling.start = block.start;
292  iter.DeleteBlock();
293  }
294  }
295  }
296 }

298 void MergeNestedRanges(CoverageFunction* function) {
299  CoverageBlockIterator iter(function);
300 
301  while (iter.Next()) {
302  CoverageBlock& block = iter.GetBlock();
303  CoverageBlock& parent = iter.GetParent();
304 
305  if (parent.count == block.count) {
306  // Transformation may not be valid if sibling blocks exist with a
307  // differing count.
308  iter.DeleteBlock();
309  }
310  }
311 }

313 void RewriteFunctionScopeCounter(CoverageFunction* function) {
314  // Every function must have at least the top-level function counter.
315  DCHECK(!function->blocks.empty());
316 
317  CoverageBlockIterator iter(function);
318  if (iter.Next()) {
319  DCHECK(iter.IsTopLevel());
320 
321  CoverageBlock& block = iter.GetBlock();
322  if (block.start == SourceRange::kFunctionLiteralSourcePosition &&
323  block.end == SourceRange::kFunctionLiteralSourcePosition) {
324  // If a function-scope block exists, overwrite the function count. It has
325  // a more reliable count than what we get from the FeedbackVector (which
326  // is imprecise e.g. for generator functions and optimized code).
327  function->count = block.count;
328 
329  // Then delete it; for compatibility with non-block coverage modes, the
330  // function-scope block is expected in CoverageFunction, not as a
331  // CoverageBlock.
332  iter.DeleteBlock();
333  }
334  }
335 }

337 void FilterAliasedSingletons(CoverageFunction* function) {
338  CoverageBlockIterator iter(function);
339 
340  iter.Next(); // Advance once since we reference the previous block later.
341 
342  while (iter.Next()) {
343  CoverageBlock& previous_block = iter.GetPreviousBlock();
344  CoverageBlock& block = iter.GetBlock();
345 
346  bool is_singleton = block.end == kNoSourcePosition;
347  bool aliases_start = block.start == previous_block.start;
348 
349  if (is_singleton && aliases_start) {
350  // The previous block must have a full range since duplicate singletons
351  // have already been merged.
352  DCHECK_NE(previous_block.end, kNoSourcePosition);
353  // Likewise, the next block must have another start position since
354  // singletons are sorted to the end.
355  DCHECK_IMPLIES(iter.HasNext(), iter.GetNextBlock().start != block.start);
356  iter.DeleteBlock();
357  }
358  }
359 }

361 void FilterUncoveredRanges(CoverageFunction* function) {
362  CoverageBlockIterator iter(function);
363 
364  while (iter.Next()) {
365  CoverageBlock& block = iter.GetBlock();
366  CoverageBlock& parent = iter.GetParent();
367  if (block.count == 0 && parent.count == 0) iter.DeleteBlock();
368  }
369 }

371 void FilterEmptyRanges(CoverageFunction* function) {
372  CoverageBlockIterator iter(function);
373 
374  while (iter.Next()) {
375  CoverageBlock& block = iter.GetBlock();
376  if (block.start == block.end) iter.DeleteBlock();
377  }

420 void CollectBlockCoverageInternal(Isolate* isolate, CoverageFunction* function,
421  Tagged<SharedFunctionInfo> info,
422  debug::CoverageMode mode) {
423  DCHECK(IsBlockMode(mode));
424 
425  // Functions with empty source ranges are not interesting to report. This can
426  // happen e.g. for internally-generated functions like class constructors.
427  if (!function->HasNonEmptySourceRange()) return;
428 
429  function->has_block_coverage = true;
430  function->blocks = GetSortedBlockData(isolate, info);
431 
432  // If in binary mode, only report counts of 0/1.
433  if (mode == debug::CoverageMode::kBlockBinary) ClampToBinary(function);
434 
435  // To stay compatible with non-block coverage modes, the function-scope count
436  // is expected to be in the CoverageFunction, not as part of its blocks.
437  // This finds the function-scope counter, overwrites CoverageFunction::count,
438  // and removes it from the block list.
439  //
440  // Important: Must be called before other transformation passes.
441  RewriteFunctionScopeCounter(function);
442 
443  // Functions without blocks don't need to be processed further.
444  if (!function->HasBlocks()) return;
445 
446  // Remove singleton ranges with the same start position as a full range and
447  // throw away their counts.
448  // Singleton ranges are only intended to split existing full ranges and should
449  // never expand into a full range. Consider 'if (cond) { ... } else { ... }'
450  // as a problematic example; if the then-block produces a continuation
451  // singleton, it would incorrectly expand into the else range.
452  // For more context, see https://crbug.com/v8/8237.
453  FilterAliasedSingletons(function);
454 
455  // Rewrite all singletons (created e.g. by continuations and unconditional
456  // control flow) to ranges.
457  RewritePositionSingletonsToRanges(function);
458 
459  // Merge nested and consecutive ranges with identical counts.
460  // Note that it's necessary to merge duplicate ranges prior to merging nested
461  // changes in order to avoid invalid transformations. See crbug.com/827530.
462  MergeConsecutiveRanges(function);
463 
464  SortBlockData(function->blocks);
465  MergeDuplicateRanges(function);
466  MergeNestedRanges(function);
467 
468  MergeConsecutiveRanges(function);
469 
470  // Filter out ranges with count == 0 unless the immediate parent range has
471  // a count != 0.
472  FilterUncoveredRanges(function);
473 
474  // Filter out ranges of zero length.
475  FilterEmptyRanges(function);
476 }

506 void CollectAndMaybeResetCounts(Isolate* isolate,
507  SharedToCounterMap* counter_map,
508  v8::debug::CoverageMode coverage_mode) {
509  const bool reset_count =
510  coverage_mode != v8::debug::CoverageMode::kBestEffort;
511 
512  switch (isolate->code_coverage_mode()) {
513  case v8::debug::CoverageMode::kBlockBinary:
514  case v8::debug::CoverageMode::kBlockCount:
515  case v8::debug::CoverageMode::kPreciseBinary:
516  case v8::debug::CoverageMode::kPreciseCount: {
517  // Feedback vectors are already listed to prevent losing them to GC.
518  DCHECK(IsArrayList(
519  *isolate->factory()->feedback_vectors_for_profiling_tools()));
520  auto list = Cast<ArrayList>(
521  isolate->factory()->feedback_vectors_for_profiling_tools());
522  for (int i = 0; i < list->length(); i++) {
523  Tagged<FeedbackVector> vector = Cast<FeedbackVector>(list->get(i));
524  Tagged<SharedFunctionInfo> shared = vector->shared_function_info();
525  DCHECK(shared->IsSubjectToDebugging());
526  uint32_t count = static_cast<uint32_t>(vector->invocation_count());
527  if (reset_count) vector->clear_invocation_count(kRelaxedStore);
528  counter_map->Add(shared, count);
529  }
530  break;
531  }
532  case v8::debug::CoverageMode::kBestEffort: {
533  DCHECK(!IsArrayList(
534  *isolate->factory()->feedback_vectors_for_profiling_tools()));
535  DCHECK_EQ(v8::debug::CoverageMode::kBestEffort, coverage_mode);
536  AllowGarbageCollection allow_gc;
537  HeapObjectIterator heap_iterator(isolate->heap());
538  for (Tagged<HeapObject> current_obj = heap_iterator.Next();
539  !current_obj.is_null(); current_obj = heap_iterator.Next()) {
540  if (!IsJSFunction(current_obj)) continue;
541  Tagged<JSFunction> func = Cast<JSFunction>(current_obj);
542  Tagged<SharedFunctionInfo> shared = func->shared();
543  if (!shared->IsSubjectToDebugging()) continue;
544  if (!(func->has_feedback_vector() ||
545  func->has_closure_feedback_cell_array())) {
546  continue;
547  }
548  uint32_t count = 0;
549  if (func->has_feedback_vector()) {
550  count = static_cast<uint32_t>(
551  func->feedback_vector()->invocation_count());
552  } else if (func->shared()->HasBytecodeArray() &&
553  func->raw_feedback_cell()->interrupt_budget() <
554  TieringManager::InterruptBudgetFor(isolate, func, {})) {
555  // We haven't allocated feedback vector, but executed the function
556  // atleast once. We don't have precise invocation count here.
557  count = 1;
558  }
559  counter_map->Add(shared, count);
560  }
561 
562  // Also check functions on the stack to collect the count map. With lazy
563  // feedback allocation we may miss counting functions if the feedback
564  // vector wasn't allocated yet and the function's interrupt budget wasn't
565  // updated (i.e. it didn't execute return / jump).
566  for (JavaScriptStackFrameIterator it(isolate); !it.done(); it.Advance()) {
567  Tagged<SharedFunctionInfo> shared = it.frame()->function()->shared();
568  if (counter_map->Get(shared) != 0) continue;
569  counter_map->Add(shared, 1);
570  }
571  break;
572  }
573  }
574 }

576 // A {SFI, count} tuple is used to sort by source range (stored on
577 // the SFI) and call count (in the counter map).
578 struct SharedFunctionInfoAndCount {
579  SharedFunctionInfoAndCount(Handle<SharedFunctionInfo> info, uint32_t count)
580  : info(info),
581  count(count),
582  start(StartPosition(*info)),
583  end(info->EndPosition()) {}
584 
585  // Sort by:
586  // - start, ascending.
587  // - end, descending.
588  // - info.is_toplevel() first
589  // - count, descending.
590  bool operator<(const SharedFunctionInfoAndCount& that) const {
591  if (this->start != that.start) return this->start < that.start;
592  if (this->end != that.end) return this->end > that.end;
593  if (this->info->is_toplevel() != that.info->is_toplevel()) {
594  return this->info->is_toplevel();
595  }
596  return this->count > that.count;
597  }
598 
599  Handle<SharedFunctionInfo> info;
600  uint32_t count;
601  int start;
602  int end;
603 };

627 std::unique_ptr<Coverage> Coverage::CollectWasmData(Isolate* isolate) {
628  // Unsupported if jitless mode is enabled at build-time since related
629  // optimizations deactivate invocation count updates.
630  CHECK(!V8_JITLESS_BOOL);
631 
632  // Iterate shared function infos of every script and build a mapping
633  // between source ranges and invocation counts.
634  std::unique_ptr<Coverage> result(new Coverage());
635 
636  // We cannot allocate inside the iterator, so we temporarily store function
637  // names as std::strings in this vector.
638  std::vector<std::string> function_names;
639  {
640  Script::Iterator script_it(isolate);
641  for (Tagged<Script> script = script_it.Next(); !script.is_null();
642  script = script_it.Next()) {
643  if (script->type() != Script::Type::kWasm) continue;
644 
645  // Create and add new script data.
646  result->emplace_back(handle(script, isolate));
647  std::vector<CoverageFunction>* functions = &result->back().functions;
648  const wasm::NativeModule* native_module = script->wasm_native_module();
649  const wasm::WasmModule* wasm_module = native_module->module();
650  const wasm::WasmModuleCoverageData* coverage_data =
651  native_module->coverage_data().get();
652  for (int declared_function_index = 0;
653  declared_function_index <
654  static_cast<int>(coverage_data->function_count());
655  declared_function_index++) {
656  const int function_index =
657  declared_function_index + wasm_module->num_imported_functions;
658  wasm::StringBuilder sb;
659  wasm::NamesProvider names_provider(wasm_module,
660  native_module->wire_bytes());

703 std::unique_ptr<Coverage> Coverage::Collect(
704  Isolate* isolate, v8::debug::CoverageMode collection_mode) {
705  // Unsupported if jitless mode is enabled at build-time since related
706  // optimizations deactivate invocation count updates.
707  CHECK(!V8_JITLESS_BOOL);
708 
709  // Collect call counts for all functions.
710  SharedToCounterMap counter_map;
711  CollectAndMaybeResetCounts(isolate, &counter_map, collection_mode);
712 
713  // Iterate shared function infos of every script and build a mapping
714  // between source ranges and invocation counts.
715  std::unique_ptr<Coverage> result(new Coverage());
716 
717  std::vector<Handle<Script>> scripts;
718  Script::Iterator script_it(isolate);
719  for (Tagged<Script> script = script_it.Next(); !script.is_null();
720  script = script_it.Next()) {
721  if (script->IsUserJavaScript()) scripts.push_back(handle(script, isolate));
722  }
723 
724  for (Handle<Script> script : scripts) {
725  // Create and add new script data.
726  result->emplace_back(script);
727  std::vector<CoverageFunction>* functions = &result->back().functions;
728 
729  std::vector<SharedFunctionInfoAndCount> sorted;
730 
731  {
732  // Sort functions by start position, from outer to inner functions.
733  SharedFunctionInfo::ScriptIterator infos(isolate, *script);
734  for (Tagged<SharedFunctionInfo> info = infos.Next(); !info.is_null();
735  info = infos.Next()) {
736  sorted.emplace_back(handle(info, isolate), counter_map.Get(info));
737  }
738  std::sort(sorted.begin(), sorted.end());
739  }
740 
741  // Stack to track nested functions, referring function by index.
742  std::vector<size_t> nesting;
743 
744  // Use sorted list to reconstruct function nesting.
745  for (const SharedFunctionInfoAndCount& v : sorted) {
746  DirectHandle<SharedFunctionInfo> info = v.info;
747  int start = v.start;
748  int end = v.end;
749  uint32_t count = v.count;
750 
751  // Find the correct outer function based on start position.
752  //
753  // This is, in general, not robust when considering two functions with
754  // identical source ranges; then the notion of inner and outer is unclear.
755  // Identical source ranges arise when the source range of top-most entity
756  // (e.g. function) in the script is identical to the whole script, e.g.
757  // <script>function foo() {}<script>. The script has its own shared
758  // function info, which has the same source range as the SFI for `foo`.
759  // Node.js creates an additional wrapper for scripts (again with identical
760  // source range) and those wrappers will have a call count of zero even if
761  // the wrapped script was executed (see v8:9212). We mitigate this issue
762  // by sorting top-level SFIs first among SFIs with the same source range:
763  // This ensures top-level SFIs are processed first. If a top-level SFI has
764  // a non-zero call count, it gets recorded due to `function_is_relevant`
765  // below (e.g. script wrappers), while top-level SFIs with zero call count
766  // do not get reported (this ensures node's extra wrappers do not get
767  // reported). If two SFIs with identical source ranges get reported, we
768  // report them in decreasing order of call count, as in all known cases
769  // this corresponds to the nesting order. In the case of the script tag
770  // example above, we report the zero call count of `foo` last. As it turns
771  // out, embedders started to rely on functions being reported in nesting
772  // order.
773  // TODO(jgruber): Investigate whether it is possible to remove node's
774  // extra top-level wrapper script, or change its source range, or ensure
775  // that it follows the invariant that nesting order is descending count
776  // order for SFIs with identical source ranges.
777  while (!nesting.empty() && functions->at(nesting.back()).end <= start) {
778  nesting.pop_back();
779  }
780 
781  if (count != 0) {
782  switch (collection_mode) {
783  case v8::debug::CoverageMode::kBlockCount:
784  case v8::debug::CoverageMode::kPreciseCount:
785  break;
786  case v8::debug::CoverageMode::kBlockBinary:
787  case v8::debug::CoverageMode::kPreciseBinary:
788  count = info->has_reported_binary_coverage() ? 0 : 1;
789  info->set_has_reported_binary_coverage(true);
790  break;
791  case v8::debug::CoverageMode::kBestEffort:
792  count = 1;
793  break;
794  }
795  }
796 
797  Handle<String> name = SharedFunctionInfo::DebugName(isolate, info);
798  CoverageFunction function(start, end, count, name);
799 
800  if (IsBlockMode(collection_mode) && info->HasCoverageInfo(isolate)) {
801  CollectBlockCoverage(isolate, &function, *info, collection_mode);
802  }
803 
804  // Only include a function range if itself or its parent function is
805  // covered, or if it contains non-trivial block coverage.
806  bool is_covered = (count != 0);
807  bool parent_is_covered =
808  (!nesting.empty() && functions->at(nesting.back()).count != 0);
809  bool has_block_coverage = !function.blocks.empty();
810  bool function_is_relevant =
811  (is_covered || parent_is_covered || has_block_coverage);
812 
813  // It must also have a non-empty source range (otherwise it is not
814  // interesting to report).
815  bool has_nonempty_source_range = function.HasNonEmptySourceRange();
816 
817  if (has_nonempty_source_range && function_is_relevant) {
818  nesting.push_back(functions->size());
819  functions->emplace_back(function);
820  }
821 
822  if (v8_flags.trace_block_coverage) {
823  PrintBlockCoverage(&function, *info, has_nonempty_source_range,
824  function_is_relevant);
825  }
826  }
827 
828  // Remove entries for scripts that have no coverage.
829  if (functions->empty()) result->pop_back();
830  }
831  return result;
832 }

834 void Coverage::SelectMode(Isolate* isolate, debug::CoverageMode mode) {
835  if (mode != isolate->code_coverage_mode()) {
836  // Changing the coverage mode can change the bytecode that would be
837  // generated for a function, which can interfere with lazy source positions,
838  // so just force source position collection whenever there's such a change.
839  isolate->CollectSourcePositionsForAllBytecodeArrays();
840  // Changing the coverage mode changes the generated bytecode and hence it is
841  // not safe to flush bytecode. Set a flag here, so we can disable bytecode
842  // flushing.
843  isolate->set_disable_bytecode_flushing(true);
844  }
845 
846  switch (mode) {
847  case debug::CoverageMode::kBestEffort:
848  // Note that DevTools switches back to best-effort coverage once the
849  // recording is stopped. Since we delete coverage infos at that point, any
850  // following coverage recording (without reloads) will be at function
851  // granularity.
852  isolate->debug()->RemoveAllCoverageInfos();
853  isolate->SetFeedbackVectorsForProfilingTools(
854  ReadOnlyRoots(isolate).undefined_value());
855  break;
856  case debug::CoverageMode::kBlockBinary:
857  case debug::CoverageMode::kBlockCount:
858  case debug::CoverageMode::kPreciseBinary:
859  case debug::CoverageMode::kPreciseCount: {
860  HandleScope scope(isolate);
861 
862  // Remove all optimized function. Optimized and inlined functions do not
863  // increment invocation count.
864  Deoptimizer::DeoptimizeAll(isolate);
865 
866  std::vector<Handle<JSFunction>> funcs_needing_feedback_vector;
867  {
868  HeapObjectIterator heap_iterator(isolate->heap());
869  for (Tagged<HeapObject> o = heap_iterator.Next(); !o.is_null();
870  o = heap_iterator.Next()) {
871  if (IsJSFunction(o)) {
872  Tagged<JSFunction> func = Cast<JSFunction>(o);
873  if (func->has_closure_feedback_cell_array()) {
874  funcs_needing_feedback_vector.push_back(
875  Handle<JSFunction>(func, isolate));
876  }
877  } else if (IsBinaryMode(mode) && IsSharedFunctionInfo(o)) {
878  // If collecting binary coverage, reset
879  // SFI::has_reported_binary_coverage to avoid optimizing / inlining
880  // functions before they have reported coverage.
881  Tagged<SharedFunctionInfo> shared = Cast<SharedFunctionInfo>(o);
882  shared->set_has_reported_binary_coverage(false);
883  } else if (IsFeedbackVector(o)) {
884  // In any case, clear any collected invocation counts.
885  Cast<FeedbackVector>(o)->clear_invocation_count(kRelaxedStore);
886  }
887  }
888  }
889 
890  for (DirectHandle<JSFunction> func : funcs_needing_feedback_vector) {
891  IsCompiledScope is_compiled_scope(
892  func->shared()->is_compiled_scope(isolate));
893  CHECK(is_compiled_scope.is_compiled());
894  JSFunction::EnsureFeedbackVector(isolate, func, &is_compiled_scope);
895  }
896 
897  // Root all feedback vectors to avoid early collection.
898  isolate->MaybeInitializeVectorListFromHeap();
899 
900  break;
901  }
902  }
903  isolate->set_code_coverage_mode(mode);
904 }
