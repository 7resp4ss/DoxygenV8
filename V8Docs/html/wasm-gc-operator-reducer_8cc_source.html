26 Reduction WasmGCOperatorReducer::Reduce(Node* node) {
27  switch (node->opcode()) {
28  case IrOpcode::kStart:
29  return ReduceStart(node);
30  case IrOpcode::kWasmStructGet:
31  case IrOpcode::kWasmStructSet:
32  return ReduceWasmStructOperation(node);
33  case IrOpcode::kWasmArrayLength:
34  return ReduceWasmArrayLength(node);
35  case IrOpcode::kAssertNotNull:
36  return ReduceAssertNotNull(node);
37  case IrOpcode::kIsNull:
38  case IrOpcode::kIsNotNull:
39  return ReduceCheckNull(node);
40  case IrOpcode::kWasmTypeCheck:
41  return ReduceWasmTypeCheck(node);
42  case IrOpcode::kWasmTypeCheckAbstract:
43  return ReduceWasmTypeCheckAbstract(node);
44  case IrOpcode::kWasmTypeCast:
45  return ReduceWasmTypeCast(node);
46  case IrOpcode::kWasmTypeCastAbstract:
47  return ReduceWasmTypeCastAbstract(node);
48  case IrOpcode::kTypeGuard:
49  return ReduceTypeGuard(node);
50  case IrOpcode::kWasmAnyConvertExtern:
51  return ReduceWasmAnyConvertExtern(node);
52  case IrOpcode::kMerge:
53  return ReduceMerge(node);
54  case IrOpcode::kIfTrue:
55  return ReduceIf(node, true);
56  case IrOpcode::kIfFalse:
57  return ReduceIf(node, false);
58  case IrOpcode::kDead:
59  return NoChange();
60  case IrOpcode::kLoop:
61  return TakeStatesFromFirstControl(node);
62  default:
63  if (node->op()->ControlOutputCount() > 0) {
64  DCHECK_EQ(1, node->op()->ControlInputCount());
65  return TakeStatesFromFirstControl(node);
66  } else {
67  return NoChange();
68  }
69  }

104 Reduction WasmGCOperatorReducer::UpdateNodeAndAliasesTypes(
105  Node* state_owner, ControlPathTypes parent_state, Node* node,
106  wasm::TypeInModule type, bool in_new_block) {
107  ControlPathTypes previous_knowledge = GetState(state_owner);
108  if (!previous_knowledge.IsEmpty()) {
109  NodeWithType current_info = previous_knowledge.LookupState(node);
110  if (current_info.IsSet() && current_info.type == type) return NoChange();
111  }
112  Node* current = node;
113  ControlPathTypes current_state = parent_state;
114  while (current != nullptr) {
115  UpdateStates(state_owner, current_state, current, {current, type},
116  in_new_block);
117  current = GetAlias(current);
118  current_state = GetState(state_owner);
119  in_new_block = false;
120  }
121  return Changed(state_owner);
122 }

128 wasm::TypeInModule WasmGCOperatorReducer::ObjectTypeFromContext(
129  Node* object, Node* control, bool allow_non_wasm) {
130  if (object->opcode() == IrOpcode::kDead ||
131  object->opcode() == IrOpcode::kDeadValue) {
132  return {};
133  }
134  if (!IsReduced(control)) return {};
135  if (allow_non_wasm && !NodeProperties::IsTyped(object)) return {};
136  Type raw_type = NodeProperties::GetType(object);
137  if (allow_non_wasm && !raw_type.IsWasm()) return {};
138  wasm::TypeInModule type_from_node = raw_type.AsWasm();
139  ControlPathTypes state = GetState(control);
140  NodeWithType type_from_state = state.LookupState(object);
141  // We manually resolve TypeGuard aliases in the state.
142  while (object->opcode() == IrOpcode::kTypeGuard && !type_from_state.IsSet()) {
143  object = NodeProperties::GetValueInput(object, 0);
144  type_from_state = state.LookupState(object);
145  }
146  if (!type_from_state.IsSet()) return type_from_node;
147  return compiler::Intersection(type_from_node, type_from_state.type);
148 }

150 Reduction WasmGCOperatorReducer::ReduceWasmStructOperation(Node* node) {
151  DCHECK(node->opcode() == IrOpcode::kWasmStructGet ||
152  node->opcode() == IrOpcode::kWasmStructSet);
153  Node* control = NodeProperties::GetControlInput(node);
154  if (!IsReduced(control)) return NoChange();
155  Node* object = NodeProperties::GetValueInput(node, 0);
156 
157  wasm::TypeInModule object_type = ObjectTypeFromContext(object, control);
158  if (object_type.type.is_uninhabited()) return NoChange();
159 
160  if (object_type.type.is_non_nullable()) {
161  // If the object is known to be non-nullable in the context, remove the null
162  // check.
163  auto op_params = OpParameter<WasmFieldInfo>(node->op());
164  const Operator* new_op =
165  node->opcode() == IrOpcode::kWasmStructGet
166  ? simplified()->WasmStructGet(op_params.type, op_params.field_index,
167  op_params.is_signed,
168  kWithoutNullCheck)
169  : simplified()->WasmStructSet(op_params.type, op_params.field_index,
170  kWithoutNullCheck);
171  NodeProperties::ChangeOp(node, new_op);
172  }
173 
174  object_type.type = object_type.type.AsNonNull();
175 
176  return UpdateNodeAndAliasesTypes(node, GetState(control), object, object_type,
177  false);
178 }

202 // If the condition of this node's branch is a type check or a null check,
203 // add the additional information about the type-checked node to the path
204 // state.
205 Reduction WasmGCOperatorReducer::ReduceIf(Node* node, bool condition) {
206  DCHECK(node->opcode() == IrOpcode::kIfTrue ||
207  node->opcode() == IrOpcode::kIfFalse);
208  Node* branch = NodeProperties::GetControlInput(node);
209  if (branch->opcode() == IrOpcode::kDead) return NoChange();
210  DCHECK_EQ(branch->opcode(), IrOpcode::kBranch);
211  if (!IsReduced(branch)) return NoChange();
212  ControlPathTypes parent_state = GetState(branch);
213  Node* condition_node = NodeProperties::GetValueInput(branch, 0);
214  switch (condition_node->opcode()) {
215  case IrOpcode::kWasmTypeCheck:
216  case IrOpcode::kWasmTypeCheckAbstract: {
217  if (!condition) break;
218  Node* object = NodeProperties::GetValueInput(condition_node, 0);
219  wasm::TypeInModule object_type = ObjectTypeFromContext(object, branch);
220  if (object_type.type.is_uninhabited()) return NoChange();
221 
222  wasm::ValueType to_type =
223  OpParameter<WasmTypeCheckConfig>(condition_node->op()).to;
224 
225  wasm::TypeInModule new_type =
226  wasm::Intersection(object_type.type, to_type, module_);
227  return UpdateNodeAndAliasesTypes(node, parent_state, object, new_type,
228  true);
229  }
230  case IrOpcode::kIsNull:
231  case IrOpcode::kIsNotNull: {
232  Node* object = NodeProperties::GetValueInput(condition_node, 0);
233  Node* control = NodeProperties::GetControlInput(condition_node);
234  wasm::TypeInModule object_type = ObjectTypeFromContext(object, control);
235  if (object_type.type.is_uninhabited()) return NoChange();
236  // If the checked value is null, narrow the type to the corresponding
237  // null type, otherwise to a non-null reference.
238  bool is_null =
239  condition == (condition_node->opcode() == IrOpcode::kIsNull);
240  object_type.type = is_null ? wasm::ToNullSentinel(object_type)
241  : object_type.type.AsNonNull();
242  return UpdateNodeAndAliasesTypes(node, parent_state, object, object_type,
243  true);
244  }
245  default:
246  break;
247  }
248  return TakeStatesFromFirstControl(node);
249 }

251 Reduction WasmGCOperatorReducer::ReduceMerge(Node* node) {
252  // Shortcut for the case when we do not know anything about some
253  // input.
254  Node::Inputs inputs = node->inputs();
255  for (Node* input : inputs) {
256  if (!IsReduced(input)) return NoChange();
257  }
258 
259  auto input_it = inputs.begin();
260 
261  DCHECK_GT(inputs.count(), 0);
262 
263  ControlPathTypes types = GetState(*input_it);
264  ++input_it;
265 
266  auto input_end = inputs.end();
267  for (; input_it != input_end; ++input_it) {
268  // Change the current type block list to a longest common prefix of this
269  // state list and the other list. (The common prefix should correspond to
270  // the state of the common dominator.)
271  // TODO(manoskouk): Consider computing unions for some types.
272  types.ResetToCommonAncestor(GetState(*input_it));
273  }
274  return UpdateStates(node, types);
275 }

277 Reduction WasmGCOperatorReducer::ReduceAssertNotNull(Node* node) {
278  DCHECK_EQ(node->opcode(), IrOpcode::kAssertNotNull);
279  Node* object = NodeProperties::GetValueInput(node, 0);
280  Node* control = NodeProperties::GetControlInput(node);
281 
282  wasm::TypeInModule object_type = ObjectTypeFromContext(object, control);
283  if (object_type.type.is_uninhabited()) return NoChange();
284 
285  // Optimize the check away if the argument is known to be non-null.
286  if (object_type.type.is_non_nullable()) {
287  // First, relax control.
288  ReplaceWithValue(node, node, node, control);
289  // Use a TypeGuard node to not lose any type information.
290  NodeProperties::ChangeOp(
291  node, common()->TypeGuard(NodeProperties::GetType(node)));
292  return Changed(node);
293  }
294 
295  object_type.type = object_type.type.AsNonNull();
296  return UpdateNodeAndAliasesTypes(node, GetState(control), node, object_type,
297  false);
298 }

300 Reduction WasmGCOperatorReducer::ReduceCheckNull(Node* node) {
301  DCHECK(node->opcode() == IrOpcode::kIsNull ||
302  node->opcode() == IrOpcode::kIsNotNull);
303  Node* object = NodeProperties::GetValueInput(node, 0);
304  Node* control = NodeProperties::GetControlInput(node);
305 
306  wasm::TypeInModule object_type = ObjectTypeFromContext(object, control);
307  if (object_type.type.is_uninhabited()) return NoChange();
308 
309  // Optimize the check away if the argument is known to be non-null.
310  if (object_type.type.is_non_nullable()) {
311  ReplaceWithValue(node,
312  SetType(gasm_.Int32Constant(
313  node->opcode() == IrOpcode::kIsNull ? 0 : 1),
314  wasm::kWasmI32));
315  node->Kill();
316  return Replace(object); // Irrelevant replacement.
317  }
318 
319  // Optimize the check away if the argument is known to be null.
320  if (object->opcode() == IrOpcode::kNull) {
321  ReplaceWithValue(node,
322  SetType(gasm_.Int32Constant(
323  node->opcode() == IrOpcode::kIsNull ? 1 : 0),
324  wasm::kWasmI32));
325  node->Kill();
326  return Replace(object); // Irrelevant replacement.
327  }
328 
329  return NoChange();
330 }

332 Reduction WasmGCOperatorReducer::ReduceWasmAnyConvertExtern(Node* node) {
333  DCHECK_EQ(node->opcode(), IrOpcode::kWasmAnyConvertExtern);
334  // Remove redundant any.convert_extern(extern.convert_any(...)) pattern.
335  Node* input = NodeProperties::GetValueInput(node, 0);
336  while (input->opcode() == IrOpcode::kTypeGuard) {
337  input = NodeProperties::GetValueInput(input, 0);
338  }
339  if (input->opcode() == IrOpcode::kDead ||
340  input->opcode() == IrOpcode::kDeadValue) {
341  return NoChange();
342  }
343  if (input->opcode() == IrOpcode::kWasmExternConvertAny) {
344  // "Skip" the extern.convert_any which doesn't have an effect on the value.
345  input = NodeProperties::GetValueInput(input, 0);
346  ReplaceWithValue(node, input);
347  node->Kill();
348  return Replace(input);
349  }
350  return TakeStatesFromFirstControl(node);
351 }

353 Reduction WasmGCOperatorReducer::ReduceTypeGuard(Node* node) {
354  DCHECK_EQ(node->opcode(), IrOpcode::kTypeGuard);
355  Node* control = NodeProperties::GetControlInput(node);
356  Node* object = NodeProperties::GetValueInput(node, 0);
357 
358  // Since TypeGuards can be generated for JavaScript, and this phase is run
359  // for wasm-into-JS inlining, we cannot assume the object has a wasm type.
360  wasm::TypeInModule object_type =
361  ObjectTypeFromContext(object, control, /* allow_non_wasm = */ true);
362  if (object_type.type.is_uninhabited()) return NoChange();
363  Type guarded_type = TypeGuardTypeOf(node->op());
364  if (!guarded_type.IsWasm()) return NoChange();
365 
366  wasm::TypeInModule new_type =
367  compiler::Intersection(object_type, guarded_type.AsWasm());
368 
369  return UpdateNodeAndAliasesTypes(node, GetState(control), node, new_type,
370  false);

373 Reduction WasmGCOperatorReducer::ReduceWasmTypeCast(Node* node) {
374  DCHECK_EQ(node->opcode(), IrOpcode::kWasmTypeCast);
375  Node* effect = NodeProperties::GetEffectInput(node);
376  Node* control = NodeProperties::GetControlInput(node);
377  Node* object = NodeProperties::GetValueInput(node, 0);
378  Node* rtt = NodeProperties::GetValueInput(node, 1);
379 
380  wasm::TypeInModule object_type = ObjectTypeFromContext(object, control);
381  if (object_type.type.is_uninhabited()) return NoChange();
382  if (InDeadBranch(rtt)) return NoChange();
383  wasm::TypeInModule rtt_type = NodeProperties::GetType(rtt).AsWasm();
384  DCHECK_EQ(object_type.module, rtt_type.module);
385  bool to_nullable =
386  OpParameter<WasmTypeCheckConfig>(node->op()).to.is_nullable();
387 
388  if (wasm::IsHeapSubtypeOf(object_type.type.heap_type(),
389  rtt_type.type.heap_type(), object_type.module)) {
390  if (to_nullable) {
391  // Type cast will always succeed. Turn it into a TypeGuard to not lose any
392  // type information.
393  // First, relax control.
394  ReplaceWithValue(node, node, node, control);
395  // Remove rtt input.
396  node->RemoveInput(1);
397  NodeProperties::ChangeOp(
398  node, common()->TypeGuard(NodeProperties::GetType(node)));
399  return Changed(node);
400  } else {
401  gasm_.InitializeEffectControl(effect, control);
402  Node* assert_not_null = gasm_.AssertNotNull(object, object_type.type,
403  TrapId::kTrapIllegalCast);
404  UpdateSourcePosition(assert_not_null, node);
405  return Replace(SetType(assert_not_null, object_type.type.AsNonNull()));
406  }
407  }
408 
409  if (wasm::HeapTypesUnrelated(object_type.type.heap_type(),
410  rtt_type.type.heap_type(), object_type.module)) {
411  gasm_.InitializeEffectControl(effect, control);
412  // A cast between unrelated types can only succeed if the argument is null.
413  // Otherwise, it always fails.
414  Node* non_trapping_condition = object_type.type.is_nullable() && to_nullable
415  ? gasm_.IsNull(object, object_type.type)
416  : gasm_.Int32Constant(0);
417  gasm_.TrapUnless(SetType(non_trapping_condition, wasm::kWasmI32),
418  TrapId::kTrapIllegalCast);
419  UpdateSourcePosition(gasm_.effect(), node);
420  Node* null_node = SetType(gasm_.Null(object_type.type),
421  wasm::ToNullSentinel(object_type));
422  ReplaceWithValue(node, null_node, gasm_.effect(), gasm_.control());
423  node->Kill();
424  return Replace(null_node);
425  }
426 
427  // Update the from-type in the type cast.
428  WasmTypeCheckConfig current_config =
429  OpParameter<WasmTypeCheckConfig>(node->op());
430  NodeProperties::ChangeOp(node, gasm_.simplified()->WasmTypeCast(
431  {object_type.type, current_config.to,
432  current_config.exactness}));
433 
434  wasm::TypeInModule new_type =
435  wasm::Intersection(object_type.type, rtt_type.type.AsNullable(), module_);
436 
437  return UpdateNodeAndAliasesTypes(node, GetState(control), node, new_type,
438  false);
439 }

441 Reduction WasmGCOperatorReducer::ReduceWasmTypeCastAbstract(Node* node) {
442  DCHECK_EQ(node->opcode(), IrOpcode::kWasmTypeCastAbstract);
443  Node* effect = NodeProperties::GetEffectInput(node);
444  Node* control = NodeProperties::GetControlInput(node);
445  Node* object = NodeProperties::GetValueInput(node, 0);
446  WasmTypeCheckConfig config = OpParameter<WasmTypeCheckConfig>(node->op());
447 
448  wasm::TypeInModule object_type = ObjectTypeFromContext(object, control);
449  if (object_type.type.is_uninhabited()) return NoChange();
450  const bool to_nullable = config.to.is_nullable();
451 
452  if (wasm::IsHeapSubtypeOf(object_type.type.heap_type(), config.to.heap_type(),
453  object_type.module)) {
454  if (to_nullable || object_type.type.is_non_nullable()) {
455  // Type cast will always succeed. Turn it into a TypeGuard to not lose any
456  // type information.
457  // First, relax control.
458  ReplaceWithValue(node, node, node, control);
459  NodeProperties::ChangeOp(
460  node, common()->TypeGuard(NodeProperties::GetType(node)));
461  return Changed(node);
462  } else {
463  gasm_.InitializeEffectControl(effect, control);
464  Node* assert_not_null = gasm_.AssertNotNull(object, object_type.type,
465  TrapId::kTrapIllegalCast);
466  UpdateSourcePosition(assert_not_null, node);
467  return Replace(SetType(assert_not_null, object_type.type.AsNonNull()));
468  }
469  }
470 
471  if (wasm::HeapTypesUnrelated(object_type.type.heap_type(),
472  config.to.heap_type(), object_type.module)) {
473  gasm_.InitializeEffectControl(effect, control);
474  // A cast between unrelated types can only succeed if the argument is null.
475  // Otherwise, it always fails.
476  Node* non_trapping_condition = object_type.type.is_nullable() && to_nullable
477  ? gasm_.IsNull(object, object_type.type)
478  : gasm_.Int32Constant(0);
479  gasm_.TrapUnless(SetType(non_trapping_condition, wasm::kWasmI32),
480  TrapId::kTrapIllegalCast);
481  UpdateSourcePosition(gasm_.effect(), node);
482  Node* null_node = SetType(gasm_.Null(object_type.type),
483  wasm::ToNullSentinel(object_type));
484  ReplaceWithValue(node, null_node, gasm_.effect(), gasm_.control());
485  node->Kill();
486  return Replace(null_node);
487  }
488 
489  // Update the from-type in the type cast.
490  NodeProperties::ChangeOp(
491  node, gasm_.simplified()->WasmTypeCastAbstract(
492  {object_type.type, config.to, config.exactness}));
493 
494  wasm::TypeInModule new_type =
495  wasm::Intersection(object_type.type, config.to, module_);
496 
497  return UpdateNodeAndAliasesTypes(node, GetState(control), node, new_type,
498  false);

501 Reduction WasmGCOperatorReducer::ReduceWasmTypeCheck(Node* node) {
502  DCHECK_EQ(node->opcode(), IrOpcode::kWasmTypeCheck);
503  Node* object = NodeProperties::GetValueInput(node, 0);
504  Node* rtt = NodeProperties::GetValueInput(node, 1);
505  Node* effect = NodeProperties::GetEffectInput(node);
506  Node* control = NodeProperties::GetControlInput(node);
507 
508  wasm::TypeInModule object_type = ObjectTypeFromContext(object, control);
509  if (object_type.type.is_uninhabited()) return NoChange();
510  if (InDeadBranch(rtt)) return NoChange();
511  wasm::TypeInModule rtt_type = NodeProperties::GetType(rtt).AsWasm();
512  DCHECK_EQ(object_type.module, rtt_type.module);
513 
514  if (wasm::IsHeapSubtypeOf(object_type.type.heap_type(),
515  rtt_type.type.heap_type(), object_type.module)) {
516  bool null_succeeds =
517  OpParameter<WasmTypeCheckConfig>(node->op()).to.is_nullable();
518  // Type cast will fail only on null.
519  gasm_.InitializeEffectControl(effect, control);
520  Node* condition = SetType(object_type.type.is_nullable() && !null_succeeds
521  ? gasm_.IsNotNull(object, object_type.type)
522  : gasm_.Int32Constant(1),
523  wasm::kWasmI32);
524  ReplaceWithValue(node, condition);
525  node->Kill();
526  return Replace(condition);
527  }
528 
529  if (wasm::HeapTypesUnrelated(object_type.type.heap_type(),
530  rtt_type.type.heap_type(), object_type.module)) {
531  bool null_succeeds =
532  OpParameter<WasmTypeCheckConfig>(node->op()).to.is_nullable();
533  Node* condition = nullptr;
534  if (null_succeeds && object_type.type.is_nullable()) {
535  // The cast only succeeds in case of null.
536  gasm_.InitializeEffectControl(effect, control);
537  condition =
538  SetType(gasm_.IsNull(object, object_type.type), wasm::kWasmI32);
539  } else {
540  // The cast never succeeds.
541  condition = SetType(gasm_.Int32Constant(0), wasm::kWasmI32);
542  }
543  ReplaceWithValue(node, condition);
544  node->Kill();
545  return Replace(condition);
546  }
547 
548  // TODO(12166): Think about modules below if we have cross-module inlining.
549 
550  // Update the from-type in the type cast.
551  WasmTypeCheckConfig current_config =
552  OpParameter<WasmTypeCheckConfig>(node->op());
553  NodeProperties::ChangeOp(node, gasm_.simplified()->WasmTypeCheck(
554  {object_type.type, current_config.to,
555  current_config.exactness}));
556 
557  return TakeStatesFromFirstControl(node);
558 }

560 Reduction WasmGCOperatorReducer::ReduceWasmTypeCheckAbstract(Node* node) {
561  DCHECK_EQ(node->opcode(), IrOpcode::kWasmTypeCheckAbstract);
562  Node* object = NodeProperties::GetValueInput(node, 0);
563  Node* effect = NodeProperties::GetEffectInput(node);
564  Node* control = NodeProperties::GetControlInput(node);
565  WasmTypeCheckConfig config = OpParameter<WasmTypeCheckConfig>(node->op());
566 
567  wasm::TypeInModule object_type = ObjectTypeFromContext(object, control);
568  if (object_type.type.is_uninhabited()) return NoChange();
569  const bool null_succeeds = config.to.is_nullable();
570 
571  if (wasm::IsHeapSubtypeOf(object_type.type.heap_type(), config.to.heap_type(),
572  object_type.module)) {
573  // Type cast will fail only on null.
574  gasm_.InitializeEffectControl(effect, control);
575  Node* condition = SetType(object_type.type.is_nullable() && !null_succeeds
576  ? gasm_.IsNotNull(object, object_type.type)
577  : gasm_.Int32Constant(1),
578  wasm::kWasmI32);
579  ReplaceWithValue(node, condition);
580  node->Kill();
581  return Replace(condition);
582  }
583 
584  // This can never result from user code, only from internal shortcuts,
585  // e.g. when using externrefs as strings.
586  const bool implicit_internalize =
587  config.from.AsNullable() == wasm::kWasmExternRef &&
588  wasm::IsHeapSubtypeOf(config.to.heap_type(), wasm::kWasmAnyRef,
589  object_type.module);
590  if (!implicit_internalize &&
591  wasm::HeapTypesUnrelated(object_type.type.heap_type(),
592  config.to.heap_type(), object_type.module)) {
593  Node* condition = nullptr;
594  if (null_succeeds && object_type.type.is_nullable()) {
595  // The cast only succeeds in case of null.
596  gasm_.InitializeEffectControl(effect, control);
597  condition =
598  SetType(gasm_.IsNull(object, object_type.type), wasm::kWasmI32);
599  } else {
600  // The cast never succeeds.
601  condition = SetType(gasm_.Int32Constant(0), wasm::kWasmI32);
602  }
603  ReplaceWithValue(node, condition);
604  node->Kill();
605  return Replace(condition);
606  }
607 
608  // Update the from-type in the type cast.
609  NodeProperties::ChangeOp(
610  node, gasm_.simplified()->WasmTypeCheckAbstract(
611  {object_type.type, config.to, config.exactness}));
612 
613  return TakeStatesFromFirstControl(node);
614 }
