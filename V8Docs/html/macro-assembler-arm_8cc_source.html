124 MemOperand MacroAssembler::ExternalReferenceAsOperand(
125  ExternalReference reference, Register scratch) {
126  if (root_array_available()) {
127  if (reference.IsIsolateFieldId()) {
128  return MemOperand(kRootRegister, reference.offset_from_root_register());
129  }
130  if (options().enable_root_relative_access) {
131  intptr_t offset =
132  RootRegisterOffsetForExternalReference(isolate(), reference);
133  if (is_int32(offset)) {
134  return MemOperand(kRootRegister, static_cast<int32_t>(offset));
135  }
136  }
137  if (options().isolate_independent_code) {
138  if (IsAddressableThroughRootRegister(isolate(), reference)) {
139  // Some external references can be efficiently loaded as an offset from
140  // kRootRegister.
141  intptr_t offset =
142  RootRegisterOffsetForExternalReference(isolate(), reference);
143  CHECK(is_int32(offset));
144  return MemOperand(kRootRegister, static_cast<int32_t>(offset));
145  } else {
146  // Otherwise, do a memory load from the external reference table.
147  ldr(scratch,
148  MemOperand(kRootRegister,
149  RootRegisterOffsetForExternalReferenceTableEntry(
150  isolate(), reference)));
151  return MemOperand(scratch, 0);
152  }
153  }
154  }
155  Move(scratch, reference);
156  return MemOperand(scratch, 0);
157 }

159 void MacroAssembler::GetLabelAddress(Register dest, Label* target) {
160  // This should be just a
161  // add(dest, pc, branch_offset(target));
162  // but current implementation of Assembler::bind_to()/target_at_put() add
163  // (InstructionStream::kHeaderSize - kHeapObjectTag) to a position of a label
164  // in a "linked" state and thus making it usable only for mov_label_offset().
165  // TODO(ishell): fix branch_offset() and re-implement
166  // RegExpMacroAssemblerARM::PushBacktrack() without mov_label_offset().
167  mov_label_offset(dest, target);
168  // mov_label_offset computes offset of the |target| relative to the "current
169  // InstructionStream object pointer" which is essentially pc_offset() of the
170  // label added with (InstructionStream::kHeaderSize - kHeapObjectTag).
171  // Compute "current InstructionStream object pointer" and add it to the
172  // offset in |lr| register.
173  int current_instr_code_object_relative_offset =
174  pc_offset() + Instruction::kPcLoadDelta +
175  (InstructionStream::kHeaderSize - kHeapObjectTag);
176  add(dest, pc, dest);
177  sub(dest, dest, Operand(current_instr_code_object_relative_offset));
178 }

222 void MacroAssembler::Call(Address target, RelocInfo::Mode rmode, Condition cond,
223  TargetAddressStorageMode mode,
224  bool check_constant_pool) {
225  // Check if we have to emit the constant pool before we block it.
226  if (check_constant_pool) MaybeCheckConstPool();
227  // Block constant pool for the call instruction sequence.
228  BlockConstPoolScope block_const_pool(this);
229 
230  bool old_predictable_code_size = predictable_code_size();
231  if (mode == NEVER_INLINE_TARGET_ADDRESS) {
232  set_predictable_code_size(true);
233  }
234 
235  // Use ip directly instead of using UseScratchRegisterScope, as we do not
236  // preserve scratch registers across calls.
237 
238  // Call sequence on V7 or later may be :
239  // movw ip, #... @ call address low 16
240  // movt ip, #... @ call address high 16
241  // blx ip
242  // @ return address
243  // Or for pre-V7 or values that may be back-patched
244  // to avoid ICache flushes:
245  // ldr ip, [pc, #...] @ call address
246  // blx ip
247  // @ return address
248 
249  mov(ip, Operand(target, rmode));
250  blx(ip, cond);
251 
252  if (mode == NEVER_INLINE_TARGET_ADDRESS) {
253  set_predictable_code_size(old_predictable_code_size);
254  }
255 }

308 void MacroAssembler::CallBuiltin(Builtin builtin, Condition cond) {
309  ASM_CODE_COMMENT_STRING(this, CommentForOffHeapTrampoline("call", builtin));
310  // Use ip directly instead of using UseScratchRegisterScope, as we do not
311  // preserve scratch registers across calls.
312  switch (options().builtin_call_jump_mode) {
313  case BuiltinCallJumpMode::kAbsolute: {
314  mov(ip, Operand(BuiltinEntry(builtin), RelocInfo::OFF_HEAP_TARGET));
315  Call(ip, cond);
316  break;
317  }
318  case BuiltinCallJumpMode::kPCRelative:
319  UNREACHABLE();
320  case BuiltinCallJumpMode::kIndirect:
321  ldr(ip, EntryFromBuiltinAsOperand(builtin));
322  Call(ip, cond);
323  break;
324  case BuiltinCallJumpMode::kForMksnapshot: {
325  if (options().use_pc_relative_calls_and_jumps_for_mksnapshot) {
326  Handle<Code> code = isolate()->builtins()->code_handle(builtin);
327  int32_t code_target_index = AddCodeTarget(code);
328  bl(code_target_index * kInstrSize, cond,
329  RelocInfo::RELATIVE_CODE_TARGET);
330  } else {
331  ldr(ip, EntryFromBuiltinAsOperand(builtin));
332  Call(ip, cond);
333  }
334  break;
335  }
336  }
337 }

490 void MacroAssembler::StoreReturnAddressAndCall(Register target) {
491  ASM_CODE_COMMENT(this);
492  // This generates the final instruction sequence for calls to C functions
493  // once an exit frame has been constructed.
494  //
495  // Note that this assumes the caller code (i.e. the InstructionStream object
496  // currently being generated) is immovable or that the callee function cannot
497  // trigger GC, since the callee function will return to it.
498 
499  // Compute the return address in lr to return to after the jump below. The pc
500  // is already at '+ 8' from the current instruction; but return is after three
501  // instructions, so add another 4 to pc to get the return address.
502  Assembler::BlockConstPoolScope block_const_pool(this);
503  add(lr, pc, Operand(4));
504  str(lr, MemOperand(sp));
505  Call(target);

560 void MacroAssembler::PushArray(Register array, Register size, Register scratch,
561  PushArrayOrder order) {
562  ASM_CODE_COMMENT(this);
563  UseScratchRegisterScope temps(this);
564  Register counter = scratch;
565  Register tmp = temps.Acquire();
566  DCHECK(!AreAliased(array, size, counter, tmp));
567  Label loop, entry;
568  if (order == PushArrayOrder::kReverse) {
569  mov(counter, Operand(0));
570  b(&entry);
571  bind(&loop);
572  ldr(tmp, MemOperand(array, counter, LSL, kSystemPointerSizeLog2));
573  push(tmp);
574  add(counter, counter, Operand(1));
575  bind(&entry);
576  cmp(counter, size);
577  b(lt, &loop);
578  } else {
579  mov(counter, size);
580  b(&entry);
581  bind(&loop);
582  ldr(tmp, MemOperand(array, counter, LSL, kSystemPointerSizeLog2));
583  push(tmp);
584  bind(&entry);
585  sub(counter, counter, Operand(1), SetCC);
586  b(ge, &loop);
587  }
588 }

906 void MacroAssembler::MoveObjectAndSlot(Register dst_object, Register dst_slot,
907  Register object, Operand offset) {
908  DCHECK_NE(dst_object, dst_slot);
909  DCHECK(offset.IsRegister() || offset.IsImmediate());
910  // If `offset` is a register, it cannot overlap with `object`.
911  DCHECK_IMPLIES(offset.IsRegister(), offset.rm() != object);
912 
913  // If the slot register does not overlap with the object register, we can
914  // overwrite it.
915  if (dst_slot != object) {
916  add(dst_slot, object, offset);
917  Move(dst_object, object);
918  return;
919  }
920 
921  DCHECK_EQ(dst_slot, object);
922 
923  // If the destination object register does not overlap with the offset
924  // register, we can overwrite it.
925  if (!offset.IsRegister() || (offset.rm() != dst_object)) {
926  Move(dst_object, dst_slot);
927  add(dst_slot, dst_slot, offset);
928  return;
929  }
930 
931  DCHECK_EQ(dst_object, offset.rm());
932 
933  // We only have `dst_slot` and `dst_object` left as distinct registers so we
934  // have to swap them. We write this as a add+sub sequence to avoid using a
935  // scratch register.
936  add(dst_slot, dst_slot, dst_object);
937  sub(dst_object, dst_slot, dst_object);

944  SaveFPRegsMode fp_mode, SmiCheck smi_check) {

1156  SwVfpRegister::from_code(src_code));

1685 void MacroAssembler::InvokePrologue(Register expected_parameter_count,
1686  Register actual_parameter_count,
1687  InvokeType type) {
1688  ASM_CODE_COMMENT(this);
1689  Label regular_invoke;
1690  // r0: actual arguments count
1691  // r1: function (passed through to callee)
1692  // r2: expected arguments count
1693  DCHECK_EQ(actual_parameter_count, r0);
1694  DCHECK_EQ(expected_parameter_count, r2);
1695 
1696  // If overapplication or if the actual argument count is equal to the
1697  // formal parameter count, no need to push extra undefined values.
1698  sub(expected_parameter_count, expected_parameter_count,
1699  actual_parameter_count, SetCC);
1700  b(le, &regular_invoke);
1701 
1702  Label stack_overflow;
1703  Register scratch = r4;
1704  StackOverflowCheck(expected_parameter_count, scratch, &stack_overflow);
1705 
1706  // Underapplication. Move the arguments already in the stack, including the
1707  // receiver and the return address.
1708  {
1709  Label copy, check;
1710  Register num = r5, src = r6, dest = r9; // r7 and r8 are context and root.
1711  mov(src, sp);
1712  // Update stack pointer.
1713  lsl(scratch, expected_parameter_count, Operand(kSystemPointerSizeLog2));
1714  AllocateStackSpace(scratch);
1715  mov(dest, sp);
1716  mov(num, actual_parameter_count);
1717  b(&check);
1718  bind(&copy);
1719  ldr(scratch, MemOperand(src, kSystemPointerSize, PostIndex));
1720  str(scratch, MemOperand(dest, kSystemPointerSize, PostIndex));
1721  sub(num, num, Operand(1), SetCC);
1722  bind(&check);
1723  b(gt, &copy);
1724  }
1725 
1726  // Fill remaining expected arguments with undefined values.
1727  LoadRoot(scratch, RootIndex::kUndefinedValue);
1728  {
1729  Label loop;
1730  bind(&loop);
1731  str(scratch, MemOperand(r9, kSystemPointerSize, PostIndex));
1732  sub(expected_parameter_count, expected_parameter_count, Operand(1), SetCC);
1733  b(gt, &loop);
1734  }
1735  b(&regular_invoke);
1736 
1737  bind(&stack_overflow);
1738  {
1739  FrameScope frame(
1740  this, has_frame() ? StackFrame::NO_FRAME_TYPE : StackFrame::INTERNAL);
1741  CallRuntime(Runtime::kThrowStackOverflow);
1742  bkpt(0);
1743  }
1744 
1745  bind(&regular_invoke);
1746 }

2462  }

2756  int num_double_arguments) {
2757  int stack_passed_words = 0;
2758  if (use_eabi_hardfloat()) {
2759  // In the hard floating point calling convention, we can use the first 8
2760  // registers to pass doubles.
2761  if (num_double_arguments > kDoubleRegisterPassedArguments) {
2762  stack_passed_words +=
2763  2 * (num_double_arguments - kDoubleRegisterPassedArguments);
2764  }
2765  } else {
2766  // In the soft floating point calling convention, every double
2767  // argument is passed using two registers.
2768  num_reg_arguments += 2 * num_double_arguments;
2769  }
2770  // Up to four simple arguments are passed in registers r0..r3.
2771  if (num_reg_arguments > kRegisterPassedArguments) {
2772  stack_passed_words += num_reg_arguments - kRegisterPassedArguments;
2773  }
2774  return stack_passed_words;

2833  int num_double_arguments,
2834  SetIsolateDataSlots set_isolate_data_slots,
2835  Label* return_label) {
2836  ASM_CODE_COMMENT(this);
2837  DCHECK_LE(num_reg_arguments + num_double_arguments, kMaxCParameters);
2838  DCHECK(has_frame());
2839  // Make sure that the stack is aligned before calling a C function unless
2840  // running in the simulator. The simulator has its own alignment check which
2841  // provides more information.
2842 #if V8_HOST_ARCH_ARM
2843  if (v8_flags.debug_code) {
2844  int frame_alignment = base::OS::ActivationFrameAlignment();
2845  int frame_alignment_mask = frame_alignment - 1;
2846  if (frame_alignment > kPointerSize) {
2847  ASM_CODE_COMMENT_STRING(this, "Check stack alignment");
2848  DCHECK(base::bits::IsPowerOfTwo(frame_alignment));
2849  Label alignment_as_expected;
2850  tst(sp, Operand(frame_alignment_mask));
2851  b(eq, &alignment_as_expected);
2852  // Don't use Check here, as it will call Runtime_Abort possibly
2853  // re-entering here.
2854  stop();
2855  bind(&alignment_as_expected);
2856  }
2857  }
2858 #endif
2859 
2860  Label get_pc;
2861 
2862  if (set_isolate_data_slots == SetIsolateDataSlots::kYes) {
2863  Register pc_scratch = r5;
2864  Push(pc_scratch);
2865  GetLabelAddress(pc_scratch, &get_pc);
2866 
2867  // Save the frame pointer and PC so that the stack layout remains iterable,
2868  // even without an ExitFrame which normally exists between JS and C frames.
2869  CHECK(root_array_available());
2870  str(pc_scratch,
2871  ExternalReferenceAsOperand(IsolateFieldId::kFastCCallCallerPC));
2872  str(fp, ExternalReferenceAsOperand(IsolateFieldId::kFastCCallCallerFP));
2873 
2874  Pop(pc_scratch);
2875  }
2876 
2877  int call_pc_offset;
2878  {
2879  BlockConstPoolScope block_const_pool_scope(this);
2880  Call(function);
2881  call_pc_offset = pc_offset();
2882  bind(&get_pc);
2883  if (return_label) bind(return_label);
2884 
2885  int before_offset = pc_offset();
2886  int stack_passed_arguments =
2887  CalculateStackPassedWords(num_reg_arguments, num_double_arguments);
2888  if (ActivationFrameAlignment() > kPointerSize) {
2889  ldr(sp, MemOperand(sp, stack_passed_arguments * kPointerSize));
2890  } else {
2891  add(sp, sp, Operand(stack_passed_arguments * kPointerSize));
2892  }
2893  // We assume that the move instruction uses kMaxSizeOfMoveAfterFastCall
2894  // bytes. When we patch in the deopt trampoline, we patch it in after the
2895  // move instruction, so that the stack has been restored correctly.
2896  CHECK_EQ(kMaxSizeOfMoveAfterFastCall, pc_offset() - before_offset);
2897  }
2898 
2899  if (set_isolate_data_slots == SetIsolateDataSlots::kYes) {
2900  // We don't unset the PC; the FP is the source of truth.
2901  Register zero_scratch = r5;
2902  Push(zero_scratch);
2903  mov(zero_scratch, Operand::Zero());
2904 
2905  str(zero_scratch,
2906  ExternalReferenceAsOperand(IsolateFieldId::kFastCCallCallerFP));
2907 
2908  Pop(zero_scratch);
2909  }
2910 
2911  return call_pc_offset;
