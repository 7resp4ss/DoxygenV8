56 // The wrapper cache is shared per-process; but it is initialized on demand, and
57 // this action is triggered by some isolate; so we use this isolate for error
58 // reporting and running GCs if required.
59 void WasmImportWrapperCache::LazyInitialize(Isolate* triggering_isolate) {
60  base::MutexGuard lock(&mutex_);
61  if (code_allocator_.get() != nullptr) return; // Already initialized.
62  // Most wrappers are small (200-300 bytes), most modules don't need many.
63  // 32K is enough for ~100 wrappers.
64  static constexpr size_t kInitialReservationSize = 1 << 15;
65  // See {NewNativeModule} for reasoning.
66  static constexpr int kAllocationRetries = 2;
67  VirtualMemory code_space;
68  for (int retries = 0;; ++retries) {
69  code_space = GetWasmCodeManager()->TryAllocate(kInitialReservationSize);
70  if (code_space.IsReserved()) break;
71  if (retries == kAllocationRetries) {
72  V8::FatalProcessOutOfMemory(
73  triggering_isolate,
74  "Failed to allocate code space for import wrappers");
75  UNREACHABLE();
76  }
77  triggering_isolate->heap()->MemoryPressureNotification(
78  MemoryPressureLevel::kCritical, true);
79  }
80  code_allocator_.reset(
81  new WasmCodeAllocator(triggering_isolate->async_counters()));
82  base::AddressRegion initial_region = code_space.region();
83  code_allocator_->Init(std::move(code_space));
84  code_allocator_->InitializeCodeRange(nullptr, initial_region);
85 }

87 WasmCode* WasmImportWrapperCache::ModificationScope::AddWrapper(
88  WasmCompilationResult result, WasmCode::Kind kind, uint64_t signature_hash,
89  std::shared_ptr<wasm::WasmImportWrapperHandle> wrapper_handle) {
90  cache_->mutex_.AssertHeld();
91  DCHECK_NOT_NULL(wrapper_handle);
92  // Equivalent of NativeModule::AddCode().
93  const CodeDesc& desc = result.code_desc;
94  base::Vector<uint8_t> code_space =
95  cache_->code_allocator_->AllocateForWrapper(desc.instr_size);
96 
97  // Equivalent of NativeModule::AddCodeWithCodeSpace().
98  base::Vector<uint8_t> reloc_info{
99  desc.buffer + desc.buffer_size - desc.reloc_size,
100  static_cast<size_t>(desc.reloc_size)};
101  // Contrary to the NativeModule, we don't track code size here, because we
102  // have no source to attribute it to.
103  const int safepoint_table_offset =
104  desc.safepoint_table_size == 0 ? 0 : desc.safepoint_table_offset;
105  const int handler_table_offset = desc.handler_table_offset;
106  const int constant_pool_offset = desc.constant_pool_offset;
107  const int code_comments_offset = desc.code_comments_offset;
108  const int jump_table_info_offset = desc.jump_table_info_offset;
109  const int instr_size = desc.instr_size;
110  {
111  WritableJitAllocation jit_allocation =
112  ThreadIsolation::RegisterJitAllocation(
113  reinterpret_cast<Address>(code_space.begin()), code_space.size(),
114  ThreadIsolation::JitAllocationType::kWasmCode, true);
115  jit_allocation.CopyCode(0, desc.buffer, desc.instr_size);
116 
117  intptr_t delta = code_space.begin() - desc.buffer;
118  Address code_start = reinterpret_cast<Address>(code_space.begin());
119  Address constant_pool_start = code_start + constant_pool_offset;
120  for (WritableRelocIterator it(jit_allocation, code_space, reloc_info,
121  constant_pool_start, RelocInfo::kApplyMask);
122  !it.done(); it.next()) {
123  // Wrappers should contain no direct calls to Wasm functions.
124  DCHECK(!RelocInfo::IsWasmCall(it.rinfo()->rmode()));
125  // Wrappers should not call builtins via a Wasm jump table.
126  DCHECK(!RelocInfo::IsWasmStubCall(it.rinfo()->rmode()));
127  it.rinfo()->apply(delta);
128  }
129  jit_allocation.UpdateWasmCodePointer(wrapper_handle->code_pointer(),
130  signature_hash);
131  }
132  FlushInstructionCache(code_space.begin(), code_space.size());
133  const int frame_slot_count = result.frame_slot_count;
134  const int ool_spill_count = result.ool_spill_count;
135  constexpr bool frame_has_feedback_slot = false;
136  WasmCode* code = new WasmCode{nullptr /* no NativeModule */,
137  kAnonymousFuncIndex,
138  code_space,
139  frame_slot_count,
140  ool_spill_count,
141  result.tagged_parameter_slots,
142  safepoint_table_offset,
143  handler_table_offset,
144  constant_pool_offset,
145  code_comments_offset,
146  jump_table_info_offset,
147  instr_size,
148  result.protected_instructions_data.as_vector(),
149  reloc_info,
150  result.source_positions.as_vector(),
151  result.inlining_positions.as_vector(),
152  result.deopt_data.as_vector(),
153  kind,
154  ExecutionTier::kNone,
155  wasm::kNotForDebugging,
156  signature_hash,
157  frame_has_feedback_slot};
158 
159  code->Validate();
160 
161  wrapper_handle->set_code(code);
162 
163  // As an optimization, we assume that wrappers are allocated in increasing
164  // memory locations.
165  std::map<Address, WasmCode*>& codes = cache_->codes_;
166  codes.emplace_hint(codes.end(), code->instruction_start(), code);
167  return code;
168 }

177 void WasmImportWrapperCache::Free(std::vector<WasmCode*>& wrappers) {
178  base::MutexGuard lock(&mutex_);
179  if (codes_.empty() || wrappers.empty()) return;
180  // {WasmCodeAllocator::FreeCode()} wants code objects to be sorted.
181  std::sort(wrappers.begin(), wrappers.end(), [](WasmCode* a, WasmCode* b) {
182  return a->instruction_start() < b->instruction_start();
183  });
184  for (WasmCode* wrapper : wrappers) {
185  codes_.erase(wrapper->instruction_start());
186  }
187  for (auto it = entry_map_.begin(); it != entry_map_.end();) {
188  if (it->second.expired()) {
189  it = entry_map_.erase(it);
190  } else {
191  it++;
192  }
193  }
194  code_allocator_->FreeCode(base::VectorOf(wrappers));
195  for (WasmCode* wrapper : wrappers) {
196  // TODO(407003348): Drop this check if it doesn't trigger in the wild.
197  CHECK_EQ(wrapper->ref_count_bitfield_.load(std::memory_order_acquire),
198  WasmCode::kIsDyingMask);
199  delete wrapper;
200  }
201  // Make sure nobody tries to access stale pointers.
202  wrappers.clear();
203 }

220 std::shared_ptr<WasmImportWrapperHandle> WasmImportWrapperCache::Get(
221  Isolate* isolate, ImportCallKind kind, CanonicalTypeIndex type_index,
222  int expected_arity, Suspend suspend, const wasm::CanonicalSig* sig) {
223  CacheKey cache_key(kind, type_index, expected_arity, suspend);
224 
225  {
226  base::MutexGuard lock(&mutex_);
227 
228  // If we have an entry already, just return that.
229  auto it = entry_map_.find(cache_key);
230  if (it != entry_map_.end()) {
231  std::shared_ptr<WasmImportWrapperHandle> handle = it->second.lock();
232  if (handle) {
233  return handle;
234  }
235  }
236 
237  // Check if we should use a builtin.
238  std::optional<Builtin> builtin = BuiltinForWrapper(kind, sig, suspend);
239  if (builtin) {
240  Address call_target = Builtins::EmbeddedEntryOf(*builtin);
241  static_assert(Builtins::kAllBuiltinsAreIsolateIndependent);
242  std::shared_ptr<WasmImportWrapperHandle> wrapper_handle =
243  std::make_shared<WasmImportWrapperHandle>(call_target,
244  sig->signature_hash());
245  entry_map_[cache_key] = wrapper_handle;
246  return wrapper_handle;
247  }
248  }
249 
250  // Fall back to compile a new wrapper.
251  return CompileWrapper(isolate, cache_key, sig, nullptr);
252 }

307 std::shared_ptr<WasmImportWrapperHandle> WasmImportWrapperCache::CompileWrapper(
308  Isolate* isolate, const CacheKey& cache_key, const CanonicalSig* sig,
309  std::shared_ptr<WasmImportWrapperHandle> wrapper_handle) {
310  wasm::WasmCompilationResult result;
311  wasm::WasmCode::Kind code_kind;
312  switch (cache_key.kind) {
313  case ImportCallKind::kLinkError:
314  case ImportCallKind::kWasmToWasm:
315  case ImportCallKind::kWasmToJSFastApi:
316  UNREACHABLE();
317  case ImportCallKind::kJSFunction:
318  case ImportCallKind::kUseCallBuiltin:
319  case ImportCallKind::kRuntimeTypeError:
320  result = compiler::CompileWasmImportCallWrapper(
321  cache_key.kind, sig, cache_key.expected_arity, cache_key.suspend);
322  code_kind = WasmCode::Kind::kWasmToJsWrapper;
323  break;
324  case ImportCallKind::kWasmToCapi:
325  result = compiler::CompileWasmCapiCallWrapper(sig);
326  code_kind = wasm::WasmCode::Kind::kWasmToCapiWrapper;
327  break;
328  }
329 
330  WasmCode* wasm_code;
331  {
332  wasm::WasmImportWrapperCache::ModificationScope cache_scope(this);
333 
334  if (!wrapper_handle) {
335  auto it = entry_map_.find(cache_key);
336  if (it != entry_map_.end()) {
337  wrapper_handle = it->second.lock();
338  }
339  }
340 
341  if (wrapper_handle && wrapper_handle->has_code()) {
342  // There was a race and we have code now.
343  return wrapper_handle;
344  }
345 
346  if (!wrapper_handle) {
347  wrapper_handle = std::make_shared<WasmImportWrapperHandle>(
348  kNullAddress, sig->signature_hash());
349  entry_map_[cache_key] = wrapper_handle;
350  }
351 
352  wasm_code = cache_scope.AddWrapper(std::move(result), code_kind,
353  sig->signature_hash(), wrapper_handle);
354  }
355 
356  // To avoid lock order inversion, code printing must happen after the
357  // end of the {cache_scope}.
358  wasm_code->MaybePrint();
359  if (isolate) {
360  isolate->counters()->wasm_generated_code_size()->Increment(
361  wasm_code->instructions().length());
362  isolate->counters()->wasm_reloc_size()->Increment(
363  wasm_code->reloc_info().length());
364  }
365  if (GetWasmEngine()->LogWrapperCode(wasm_code) && isolate) {
366  // Log the code immediately in the current isolate.
367  GetWasmEngine()->LogOutstandingCodesForIsolate(isolate);
368  }
369 
370  return wrapper_handle;
371 }

407 WasmCode* WasmImportWrapperCache::Lookup(Address pc) const {
408  // This can be called from the disassembler via `code->MaybePrint()` in
409  // `AddWrapper()` above, so we need a recursive mutex.
410  base::MutexGuard lock(&mutex_);
411  auto iter = codes_.upper_bound(pc);
412  if (iter == codes_.begin()) return nullptr;
413  --iter;
414  WasmCode* candidate = iter->second;
415  DCHECK_EQ(candidate->instruction_start(), iter->first);
416  if (!candidate->contains(pc)) return nullptr;
417  WasmCodeRefScope::AddRef(candidate);
418  // Note: this function is used for iterating the stack, where dying
419  // code objects can still have their last few activations, so we
420  // must return {candidate} even if {candidate->is_dying()}.
421  return candidate;
422 }
