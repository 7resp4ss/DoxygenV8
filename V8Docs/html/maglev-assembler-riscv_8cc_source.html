26 template <typename T>
27 void AllocateRaw(MaglevAssembler* masm, Isolate* isolate,
28  RegisterSnapshot register_snapshot, Register object,
29  T size_in_bytes, AllocationType alloc_type,
30  AllocationAlignment alignment) {
31  DCHECK(masm->allow_allocate());
32  // TODO(victorgomes): Call the runtime for large object allocation.
33  // TODO(victorgomes): Support double alignment.
34  DCHECK_EQ(alignment, kTaggedAligned);
35  if (v8_flags.single_generation) {
36  alloc_type = AllocationType::kOld;
37  }
38  ExternalReference top = SpaceAllocationTopAddress(isolate, alloc_type);
39  ExternalReference limit = SpaceAllocationLimitAddress(isolate, alloc_type);
40 
41  ZoneLabelRef done(masm);
42  MaglevAssembler::TemporaryRegisterScope temps(masm);
43  Register scratch = temps.AcquireScratch();
44  // We are a bit short on registers, so we use the same register for {object}
45  // and {new_top}. Once we have defined {new_top}, we don't use {object} until
46  // {new_top} is used for the last time. And there (at the end of this
47  // function), we recover the original {object} from {new_top} by subtracting
48  // {size_in_bytes}.
49  Register new_top = object;
50  // Check if there is enough space.
51  __ LoadWord(object, __ ExternalReferenceAsOperand(top, scratch));
52  __ AddWord(new_top, object, Operand(size_in_bytes));
53  __ LoadWord(scratch, __ ExternalReferenceAsOperand(limit, scratch));
54 
55  // Call runtime if new_top >= limit.
56  __ MacroAssembler::Branch(
57  __ MakeDeferredCode(
58  [](MaglevAssembler* masm, RegisterSnapshot register_snapshot,
59  Register object, AllocationType alloc_type, T size_in_bytes,
60  ZoneLabelRef done) {
61  AllocateSlow(masm, register_snapshot, object,
62  AllocateBuiltin(alloc_type), size_in_bytes, done);
63  },
64  register_snapshot, object, alloc_type, size_in_bytes, done),
65  ge, new_top, Operand(scratch));
66 
67  // Store new top and tag object.
68  __ Move(__ ExternalReferenceAsOperand(top, scratch), new_top);
69  SubSizeAndTagObject(masm, object, size_in_bytes);
70  __ bind(*done);
71 }

89 void MaglevAssembler::OSRPrologue(Graph* graph) {
90  DCHECK(graph->is_osr());
91  CHECK(!graph->has_recursive_calls());
92 
93  uint32_t source_frame_size =
94  graph->min_maglev_stackslots_for_unoptimized_frame_size();
95 
96  if (v8_flags.maglev_assert_stack_size && v8_flags.debug_code) {
97  MaglevAssembler::TemporaryRegisterScope temps(this);
98  Register scratch = temps.AcquireScratch();
99  int32_t expected_osr_stack_size =
100  source_frame_size * kSystemPointerSize +
101  StandardFrameConstants::kFixedFrameSizeFromFp;
102  AddWord(scratch, sp, Operand(expected_osr_stack_size));
103  MacroAssembler::SbxCheck(eq, AbortReason::kOsrUnexpectedStackSize, scratch,
104  Operand(fp));
105  }
106 
107  uint32_t target_frame_size =
108  graph->tagged_stack_slots() + graph->untagged_stack_slots();
109  // CHECK_EQ(target_frame_size % 2, 1);
110  CHECK_LE(source_frame_size, target_frame_size);
111  if (source_frame_size < target_frame_size) {
112  ASM_CODE_COMMENT_STRING(this, "Growing frame for OSR");
113  uint32_t additional_tagged =
114  source_frame_size < graph->tagged_stack_slots()
115  ? graph->tagged_stack_slots() - source_frame_size
116  : 0;
117  for (size_t i = 0; i < additional_tagged; ++i) {
118  Push(zero_reg);
119  }
120  uint32_t size_so_far = source_frame_size + additional_tagged;
121  CHECK_LE(size_so_far, target_frame_size);
122  if (size_so_far < target_frame_size) {
123  Sub64(sp, sp,
124  Operand((target_frame_size - size_so_far) * kSystemPointerSize));
125  }
126  }
127 }

129 void MaglevAssembler::Prologue(Graph* graph) {
130  ASM_CODE_COMMENT(this);
131  MaglevAssembler::TemporaryRegisterScope temps(this);
132  // We add two extra registers to the scope. Ideally we could add all the
133  // allocatable general registers, except Context, JSFunction, NewTarget and
134  // ArgCount. Unfortunately, OptimizeCodeOrTailCallOptimizedCodeSlot and
135  // LoadFeedbackVectorFlagsAndJumpIfNeedsProcessing pick random registers and
136  // we could alias those.
137  // TODO(victorgomes): Fix these builtins to either use the scope or pass the
138  // used registers manually.
139  temps.Include({s7, s8}); // use register not overlapping with flags,
140  // feedback and so on
141  DCHECK(!graph->is_osr());
142 
143  CallTarget();
144  BailoutIfDeoptimized();
145 
146  if (graph->has_recursive_calls()) {
147  BindCallTarget(code_gen_state()->entry_label());
148  }
149 
150  // Tiering support.
151 #ifndef V8_ENABLE_LEAPTIERING
152  if (v8_flags.turbofan) {
153  using D = MaglevOptimizeCodeOrTailCallOptimizedCodeSlotDescriptor;
154  Register flags = D::GetRegisterParameter(D::kFlags);
155  Register feedback_vector = D::GetRegisterParameter(D::kFeedbackVector);
156  DCHECK(!AreAliased(
157  flags, feedback_vector,
158  kJavaScriptCallArgCountRegister, // flags - t4, feedback - a6,
159  // kJavaScriptCallArgCountRegister -
160  // a0
161  kJSFunctionRegister, kContextRegister,
162  kJavaScriptCallNewTargetRegister));
163  DCHECK(!temps.Available().has(flags));
164  DCHECK(!temps.Available().has(feedback_vector));
165  Move(feedback_vector,
166  compilation_info()->toplevel_compilation_unit()->feedback().object());
167  Label needs_processing, done;
168  LoadFeedbackVectorFlagsAndJumpIfNeedsProcessing(
169  flags, feedback_vector, CodeKind::MAGLEV, &needs_processing);
170  Jump(&done);
171  bind(&needs_processing);
172  TailCallBuiltin(Builtin::kMaglevOptimizeCodeOrTailCallOptimizedCodeSlot);
173  bind(&done);
174  }
175 #endif
176 
177  EnterFrame(StackFrame::MAGLEV);
178  // Save arguments in frame.
179  // TODO(leszeks): Consider eliding this frame if we don't make any calls
180  // that could clobber these registers.
181  // Push the context and the JSFunction.
182  Push(kContextRegister);
183  Push(kJSFunctionRegister);
184  // Push the actual argument count and a _possible_ stack slot.
185  Push(kJavaScriptCallArgCountRegister);
186  // Initialize stack slots.
187  if (graph->tagged_stack_slots() > 0) {
188  ASM_CODE_COMMENT_STRING(this, "Initializing stack slots");
189 
190  // Magic value. Experimentally, an unroll size of 8 doesn't seem any
191  // worse than fully unrolled pushes.
192  const int kLoopUnrollSize = 8;
193  int tagged_slots = graph->tagged_stack_slots();
194 
195  if (tagged_slots < 2 * kLoopUnrollSize) {
196  // If the frame is small enough, just unroll the frame fill
197  // completely.
198  for (int i = 0; i < tagged_slots; ++i) {
199  Push(zero_reg);
200  }
201  } else {
202  // Extract the first few slots to round to the unroll size.
203  int first_slots = tagged_slots % kLoopUnrollSize;
204  for (int i = 0; i < first_slots; ++i) {
205  Push(zero_reg);
206  }
207  MaglevAssembler::TemporaryRegisterScope temps(this);
208  Register count = temps.AcquireScratch();
209  Move(count, tagged_slots / kLoopUnrollSize);
210  // We enter the loop unconditionally, so make sure we need to loop at
211  // least once.
212  DCHECK_GT(tagged_slots / kLoopUnrollSize, 0);
213  Label loop;
214  bind(&loop);
215  for (int i = 0; i < kLoopUnrollSize; ++i) {
216  Push(zero_reg);
217  }
218  Sub64(count, count, Operand(1));
219  MacroAssembler::Branch(&loop, gt, count, Operand(zero_reg), Label::kNear);
220  }
221  }
222  if (graph->untagged_stack_slots() > 0) {
223  // Extend sp by the size of the remaining untagged part of the frame,
224  // no need to initialise these.
225  Sub64(sp, sp, Operand(graph->untagged_stack_slots() * kSystemPointerSize));
226  }
227 }

267 void MaglevAssembler::StringFromCharCode(RegisterSnapshot register_snapshot,
268  Label* char_code_fits_one_byte,
269  Register result, Register char_code,
270  Register scratch,
271  CharCodeMaskMode mask_mode) {
272  ZeroExtendWord(char_code, char_code);
273  AssertZeroExtended(char_code);
274  DCHECK_NE(char_code, scratch);
275  ZoneLabelRef done(this);
276  if (mask_mode == CharCodeMaskMode::kMustApplyMask) {
277  And(char_code, char_code, Operand(0xFFFF));
278  }
279  // Allocate two-bytes string if {char_code} doesn't fit one byte.
280  MacroAssembler::Branch( // FIXME: reimplement with JumpToDeferredIf
281  MakeDeferredCode(
282  [](MaglevAssembler* masm, RegisterSnapshot register_snapshot,
283  ZoneLabelRef done, Register result, Register char_code,
284  Register scratch) {
285  MaglevAssembler::TemporaryRegisterScope temps(masm);
286  // Ensure that {result} never aliases {scratch}, otherwise use
287  // a temporary register to restore {result} at the end.
288  const bool need_restore_result = (scratch == result);
289  Register string =
290  need_restore_result ? temps.AcquireScratch() : result;
291  // Ensure that {char_code} never aliases {result}, otherwise use
292  // the given {scratch} register.
293  if (char_code == result) {
294  __ Move(scratch, char_code);
295  char_code = scratch;
296  }
297  DCHECK(char_code != string);
298  DCHECK(scratch != string);
299  DCHECK(!register_snapshot.live_tagged_registers.has(char_code));
300  register_snapshot.live_registers.set(char_code);
301  __ AllocateTwoByteString(register_snapshot, string, 1);
302  __ And(scratch, char_code, Operand(0xFFFF));
303  __ Sh(scratch, FieldMemOperand(
304  string, OFFSET_OF_DATA_START(SeqTwoByteString)));
305  if (need_restore_result) {
306  __ Move(result, string);
307  }
308  __ jmp(*done);
309  },
310  register_snapshot, done, result, char_code, scratch),
311  Ugreater_equal, char_code, Operand(String::kMaxOneByteCharCode));
312 
313  if (char_code_fits_one_byte != nullptr) {
314  bind(char_code_fits_one_byte);
315  }
316  LoadSingleCharacterString(result, char_code, scratch);
317  bind(*done);
318 }
319 // Sets equality flag in pseudo flags reg.
320 void MaglevAssembler::IsObjectType(Register object, Register scratch1,
321  Register scratch2, InstanceType type) {
322  ASM_CODE_COMMENT(this);
323  constexpr Register flags = MaglevAssembler::GetFlagsRegister();
324  Label ConditionMet, Done;
325  if (V8_STATIC_ROOTS_BOOL &&
326  InstanceTypeChecker::UniqueMapOfInstanceType(type)) {
327  LoadCompressedMap(scratch1, object);
328  std::optional<RootIndex> expected =
329  InstanceTypeChecker::UniqueMapOfInstanceType(type);
330  Tagged_t expected_ptr = ReadOnlyRootPtr(*expected);
331  li(scratch2, expected_ptr);
332  Sll32(scratch2, scratch2, Operand(0));
333  MacroAssembler::Branch(&ConditionMet, Condition::kEqual, scratch1,
334  Operand(scratch2), Label::kNear);
335  } else {
336  CompareObjectTypeAndJump(object, scratch1, scratch2, type,
337  Condition::kEqual, &ConditionMet, Label::kNear);
338  }
339  Li(flags, 1); // Condition is not met by default and
340  // flags is set after a scratch is used,
341  // so no harm if they are aliased.
342  Jump(&Done, Label::kNear);
343  bind(&ConditionMet);
344  Mv(flags, zero_reg); // Condition is met
345  bind(&Done);
346 }

348 void MaglevAssembler::StringCharCodeOrCodePointAt(
349  BuiltinStringPrototypeCharCodeOrCodePointAt::Mode mode,
350  RegisterSnapshot& register_snapshot, Register result, Register string,
351  Register index, Register instance_type, [[maybe_unused]] Register scratch2,
352  Label* result_fits_one_byte) {
353  ASM_CODE_COMMENT(this);
354  ZoneLabelRef done(this);
355  Label seq_string;
356  Label cons_string;
357  Label sliced_string;
358 
359  Label* deferred_runtime_call = MakeDeferredCode(
360  [](MaglevAssembler* masm,
361  BuiltinStringPrototypeCharCodeOrCodePointAt::Mode mode,
362  RegisterSnapshot register_snapshot, ZoneLabelRef done, Register result,
363  Register string, Register index) {
364  DCHECK(!register_snapshot.live_registers.has(result));
365  DCHECK(!register_snapshot.live_registers.has(string));
366  DCHECK(!register_snapshot.live_registers.has(index));
367  {
368  SaveRegisterStateForCall save_register_state(masm, register_snapshot);
369  __ SmiTag(index);
370  __ Push(string, index);
371  __ Move(kContextRegister, masm->native_context().object());
372  // This call does not throw nor can deopt.
373  if (mode ==
374  BuiltinStringPrototypeCharCodeOrCodePointAt::kCodePointAt) {
375  __ CallRuntime(Runtime::kStringCodePointAt);
376  } else {
377  DCHECK_EQ(mode,
378  BuiltinStringPrototypeCharCodeOrCodePointAt::kCharCodeAt);
379  __ CallRuntime(Runtime::kStringCharCodeAt);
380  }
381  save_register_state.DefineSafepoint();
382  __ SmiUntag(kReturnRegister0);
383  __ Move(result, kReturnRegister0);
384  }
385  __ jmp(*done);
386  },
387  mode, register_snapshot, done, result, string, index);
388 
389  // We might need to try more than one time for ConsString, SlicedString and
390  // ThinString.
391  Label loop;
392  bind(&loop);
393 
394  if (v8_flags.debug_code) {
395  Register scratch = instance_type;
396 
397  // Check if {string} is a string.
398  AssertObjectTypeInRange(string, FIRST_STRING_TYPE, LAST_STRING_TYPE,
399  AbortReason::kUnexpectedValue);
400 
401  Lw(scratch, FieldMemOperand(string, offsetof(String, length_)));
402  Check(kUnsignedLessThan, AbortReason::kUnexpectedValue, index,
403  Operand(scratch));
404  }
405 
406  // Get instance type.
407  LoadInstanceType(instance_type, string);
408 
409  {
410  MaglevAssembler::TemporaryRegisterScope temps(this);
411  Register representation = temps.AcquireScratch();
412 
413  // TODO(victorgomes): Add fast path for external strings.
414  And(representation, instance_type, Operand(kStringRepresentationMask));
415  MacroAssembler::Branch(&seq_string, kEqual, representation,
416  Operand(kSeqStringTag), Label::kNear);
417  MacroAssembler::Branch(&cons_string, kEqual, representation,
418  Operand(kConsStringTag), Label::kNear);
419  MacroAssembler::Branch(&sliced_string, kEqual, representation,
420  Operand(kSlicedStringTag), Label::kNear);
421  MacroAssembler::Branch(deferred_runtime_call, kNotEqual, representation,
422  Operand(kThinStringTag));
423  // Fallthrough to thin string.
424  }
425 
426  // Is a thin string.
427  {
428  LoadTaggedField(string, string, offsetof(ThinString, actual_));
429  MacroAssembler::Branch(&loop, Label::kNear);
430  }
431 
432  bind(&sliced_string);
433  {
434  MaglevAssembler::TemporaryRegisterScope temps(this);
435  Register offset = temps.AcquireScratch();
436 
437  LoadAndUntagTaggedSignedField(offset, string,
438  offsetof(SlicedString, offset_));
439  LoadTaggedField(string, string, offsetof(SlicedString, parent_));
440  Add32(index, index, Operand(offset));
441  MacroAssembler::Branch(&loop, Label::kNear);
442  }
443 
444  bind(&cons_string);
445  {
446  // Reuse {instance_type} register here, since CompareRoot requires a scratch
447  // register as well.
448  Register second_string = instance_type;
449  LoadTaggedFieldWithoutDecompressing(second_string, string,
450  offsetof(ConsString, second_));
451  CompareRoot(second_string,
452  RootIndex::kempty_string); // Sets 1 to flag if not equal
453  JumpIf(ne, deferred_runtime_call); // Check the flag to not be equal 0
454  LoadTaggedField(string, string, offsetof(ConsString, first_));
455  MacroAssembler::Branch(&loop,
456  Label::kNear); // Try again with first string.
457  }
458 
459  bind(&seq_string);
460  {
461  Label two_byte_string;
462  And(instance_type, instance_type, Operand(kStringEncodingMask));
463  MacroAssembler::Branch(&two_byte_string, equal, instance_type,
464  Operand(kTwoByteStringTag), Label::kNear);
465  // The result of one-byte string will be the same for both modes
466  // (CharCodeAt/CodePointAt), since it cannot be the first half of a
467  // surrogate pair.
468  SeqOneByteStringCharCodeAt(result, string, index);
469  MacroAssembler::Branch(result_fits_one_byte);
470 
471  bind(&two_byte_string);
472  // {instance_type} is unused from this point, so we can use as scratch.
473  Register scratch = instance_type;
474 
475  CalcScaledAddress(result, string, index, 1);
476  Lhu(result, MemOperand(result, OFFSET_OF_DATA_START(SeqTwoByteString) -
477  kHeapObjectTag));
478 
479  if (mode == BuiltinStringPrototypeCharCodeOrCodePointAt::kCodePointAt) {
480  Register first_code_point = scratch;
481  And(first_code_point, result, Operand(0xfc00));
482  MacroAssembler::Branch(*done, kNotEqual, first_code_point,
483  Operand(0xd800), Label::kNear);
484 
485  Register length = scratch;
486  Lw(length, FieldMemOperand(string, offsetof(String, length_)));
487  Add32(index, index, Operand(1));
488  MacroAssembler::Branch(*done, kGreaterThanEqual, index, Operand(length),
489  Label::kNear);
490 
491  Register second_code_point = scratch;
492  CalcScaledAddress(second_code_point, string, index, 1);
493  Lhu(second_code_point,
494  MemOperand(second_code_point,
495  OFFSET_OF_DATA_START(SeqTwoByteString) - kHeapObjectTag));
496 
497  // {index} is not needed at this point.
498  Register scratch2 = index;
499  And(scratch2, second_code_point, Operand(0xfc00));
500  MacroAssembler::Branch(*done, kNotEqual, scratch2, Operand(0xdc00),
501  Label::kNear);
502 
503  int surrogate_offset = 0x10000 - (0xd800 << 10) - 0xdc00;
504  Add32(second_code_point, second_code_point, Operand(surrogate_offset));
505  Sll32(result, result, Operand(10));
506  Add32(result, result, Operand(second_code_point));
507  }
508 
509  // Fallthrough.
510  }
511 
512  bind(*done);
513 
514  if (v8_flags.debug_code) {
515  // We make sure that the user of this macro is not relying in string and
516  // index to not be clobbered.
517  if (result != string) {
518  Li(string, 0xdeadbeef);
519  }
520  if (result != index) {
521  Li(index, 0xdeadbeef);
522  }
523  }
524 }

526 void MaglevAssembler::SeqOneByteStringCharCodeAt(Register result,
527  Register string,
528  Register index) {
529  ASM_CODE_COMMENT(this);
530  if (v8_flags.debug_code) {
531  TemporaryRegisterScope scope(this);
532  Register scratch = scope.AcquireScratch();
533  // Check if {string} is a string.
534  AssertNotSmi(string);
535  LoadMap(scratch, string);
536  Check(CompareInstanceTypeRange(scratch, scratch, FIRST_STRING_TYPE,
537  LAST_STRING_TYPE),
538  AbortReason::kUnexpectedValue, MaglevAssembler::GetFlagsRegister(),
539  Operand(zero_reg));
540  // Check if {string} is a sequential one-byte string.
541  AndInt32(scratch, kStringRepresentationAndEncodingMask);
542  CompareInt32AndAssert(scratch, kSeqOneByteStringTag, kEqual,
543  AbortReason::kUnexpectedValue);
544  LoadInt32(scratch, FieldMemOperand(string, offsetof(String, length_)));
545  // Use kMaglevFlagsRegister to avoid no enough scratch reg.
546  scope.IncludeScratch({kMaglevFlagsRegister});
547  CompareInt32AndAssert(index, scratch, kUnsignedLessThan,
548  AbortReason::kUnexpectedValue);
549  }
550  TemporaryRegisterScope scope(this);
551  Register scratch = scope.AcquireScratch();
552  AddWord(scratch, index,
553  Operand(OFFSET_OF_DATA_START(SeqOneByteString) - kHeapObjectTag));
554  AddWord(scratch, string, Operand(scratch));
555  Lbu(result, MemOperand(scratch, 0));
556 }

562 void MaglevAssembler::TruncateDoubleToInt32(Register dst, DoubleRegister src) {
563  ZoneLabelRef done(this);
564  Label* slow_path = MakeDeferredCode(
565  [](MaglevAssembler* masm, DoubleRegister src, Register dst,
566  ZoneLabelRef done) {
567  __ push(ra);
568  __ AllocateStackSpace(kDoubleSize);
569  __ StoreDouble(src, MemOperand(sp, 0));
570  __ CallBuiltin(Builtin::kDoubleToI);
571  __ LoadWord(dst, MemOperand(sp, 0));
572  __ AddWord(sp, sp, Operand(kDoubleSize));
573  __ pop(ra);
574  __ Jump(*done);
575  },
576  src, dst, done);
577  TryInlineTruncateDoubleToI(dst, src, *done);

583 void MaglevAssembler::TryTruncateDoubleToInt32(Register dst, DoubleRegister src,
584  Label* fail) {
585  MaglevAssembler::TemporaryRegisterScope temps(this);
586  DoubleRegister converted_back = temps.AcquireScratchDouble();
587  Register rcmp = temps.AcquireScratch();
588 
589  // Convert the input float64 value to int32.
590  Trunc_w_d(dst, src);
591  // Convert that int32 value back to float64.
592  Cvt_d_w(converted_back, dst);
593  // Check that the result of the float64->int32->float64 is equal to the input
594  // (i.e. that the conversion didn't truncate).
595  CompareF64(rcmp, EQ, src, converted_back); // rcmp is 0 if not equal
596  MacroAssembler::Branch(
597  fail, eq, rcmp, Operand(zero_reg)); // if we don't know branch distance
598  // then lets use MacroAssembler::Branch, it will make sure we fit
599 
600  // Check if {input} is -0.
601  Label check_done;
602  BranchShort(&check_done, ne, dst, Operand(zero_reg));
603 
604  // In case of 0, we need to check for the IEEE 0 pattern (which is all zeros).
605  MacroAssembler::Move(
606  rcmp, src); // FIXME: should we enable this in MaglevAssembler as well?
607 
608  MacroAssembler::Branch(fail, ne, rcmp, Operand(zero_reg));
609 
610  bind(&check_done);
