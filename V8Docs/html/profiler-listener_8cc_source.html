108 void ProfilerListener::CodeCreateEvent(CodeTag tag,
109  DirectHandle<AbstractCode> abstract_code,
110  DirectHandle<SharedFunctionInfo> shared,
111  DirectHandle<Name> script_name, int line,
112  int column) {
113  CodeEventsContainer evt_rec(CodeEventRecord::Type::kCodeCreation);
114  CodeCreateEventRecord* rec = &evt_rec.CodeCreateEventRecord_;
115  PtrComprCageBase cage_base(isolate_);
116  rec->instruction_start = abstract_code->InstructionStart(cage_base);
117  std::unique_ptr<SourcePositionTable> line_table;
118  std::unordered_map<int, std::vector<CodeEntryAndPosition>> inline_stacks;
119  std::unordered_set<CodeEntry*, CodeEntry::Hasher, CodeEntry::Equals>
120  cached_inline_entries;
121  bool is_shared_cross_origin = false;
122  if (IsScript(shared->script(cage_base), cage_base)) {
123  DirectHandle<Script> script(Cast<Script>(shared->script(cage_base)),
124  isolate_);
125  line_table.reset(new SourcePositionTable());
126 
127  is_shared_cross_origin = script->origin_options().IsSharedCrossOrigin();
128 
129  bool is_baseline = abstract_code->kind(cage_base) == CodeKind::BASELINE;
130  Handle<TrustedByteArray> source_position_table(
131  abstract_code->SourcePositionTable(isolate_, *shared), isolate_);
132  std::unique_ptr<baseline::BytecodeOffsetIterator> baseline_iterator;
133  if (is_baseline) {
134  Handle<BytecodeArray> bytecodes(shared->GetBytecodeArray(isolate_),
135  isolate_);
136  Handle<TrustedByteArray> bytecode_offsets(
137  abstract_code->GetCode()->bytecode_offset_table(), isolate_);
138  baseline_iterator = std::make_unique<baseline::BytecodeOffsetIterator>(
139  bytecode_offsets, bytecodes);
140  }
141  // Add each position to the source position table and store inlining stacks
142  // for inline positions. We store almost the same information in the
143  // profiler as is stored on the code object, except that we transform source
144  // positions to line numbers here, because we only care about attributing
145  // ticks to a given line.
146  for (SourcePositionTableIterator it(source_position_table); !it.done();
147  it.Advance()) {
148  int position = it.source_position().ScriptOffset();
149  int inlining_id = it.source_position().InliningId();
150  int code_offset = it.code_offset();
151  if (is_baseline) {
152  // Use the bytecode offset to calculate pc offset for baseline code.
153  baseline_iterator->AdvanceToBytecodeOffset(code_offset);
154  code_offset =
155  static_cast<int>(baseline_iterator->current_pc_start_offset());
156  }
157 
158  if (inlining_id == SourcePosition::kNotInlined) {
159  Script::PositionInfo info;
160  script->GetPositionInfo(position, &info,
161  Script::OffsetFlag::kWithOffset);
162  line_table->SetPosition(code_offset, {info.line + 1, info.column + 1},
163  inlining_id);
164  } else {
165  DCHECK(!is_baseline);
166  DCHECK(IsCode(*abstract_code, cage_base));
167  std::vector<SourcePositionInfo> stack =
168  it.source_position().InliningStack(isolate_,
169  abstract_code->GetCode());
170  DCHECK(!stack.empty());
171 
172  // When we have an inlining id and we are doing cross-script inlining,
173  // then the script of the inlined frames may be different to the script
174  // of |shared|.
175  LineAndColumn line_and_column{stack.front().line + 1,
176  stack.front().column + 1};
177  line_table->SetPosition(code_offset, line_and_column, inlining_id);
178 
179  std::vector<CodeEntryAndPosition> inline_stack;
180  for (SourcePositionInfo& pos_info : stack) {
181  if (pos_info.position.ScriptOffset() == kNoSourcePosition) continue;
182  if (pos_info.script.is_null()) continue;
183 
184  Script::PositionInfo script_pos_info;
185  pos_info.script->GetPositionInfo(pos_info.position.ScriptOffset(),
186  &script_pos_info,
187  Script::OffsetFlag::kWithOffset);
188  line_and_column = {script_pos_info.line + 1,
189  script_pos_info.column + 1};
190 
191  const char* resource_name =
192  (IsName(pos_info.script->name()))
193  ? GetName(Cast<Name>(pos_info.script->name()))
194  : CodeEntry::kEmptyResourceName;
195 
196  bool inline_is_shared_cross_origin =
197  pos_info.script->origin_options().IsSharedCrossOrigin();
198 
199  // We need the start line number and column number of the function for
200  // kLeafNodeLineNumbers mode. Creating a SourcePositionInfo is a handy
201  // way of getting both easily.
202  SourcePositionInfo start_pos_info(
203  isolate_, SourcePosition(pos_info.shared->StartPosition()),
204  pos_info.shared);
205 
206  CodeEntry* inline_entry = code_entries_.Create(
207  tag, GetFunctionName(*pos_info.shared), resource_name,
208  LineAndColumn{start_pos_info.line + 1, start_pos_info.column + 1},
209  nullptr, inline_is_shared_cross_origin);
210  inline_entry->FillFunctionInfo(*pos_info.shared);
211 
212  // Create a canonical CodeEntry for each inlined frame and then reuse
213  // them for subsequent inline stacks to avoid a lot of duplication.
214  CodeEntry* cached_entry = GetOrInsertCachedEntry(
215  &cached_inline_entries, inline_entry, code_entries_);
216 
217  inline_stack.push_back({cached_entry, line_and_column});
218  }
219  DCHECK(!inline_stack.empty());
220  inline_stacks.emplace(inlining_id, std::move(inline_stack));
221  }
222  }
223  }
224  rec->entry =
225  code_entries_.Create(tag, GetFunctionName(*shared),
226  GetName(InferScriptName(*script_name, *shared)),
227  LineAndColumn{line, column}, std::move(line_table),
228  is_shared_cross_origin);
229  if (!inline_stacks.empty()) {
230  rec->entry->SetInlineStacks(std::move(cached_inline_entries),
231  std::move(inline_stacks));
232  }
233 
234  rec->entry->FillFunctionInfo(*shared);
235  rec->instruction_size = abstract_code->InstructionSize(cage_base);
236  weak_code_registry_.Track(rec->entry, abstract_code);
237  DispatchCodeEvent(evt_rec);
238 }

373 const char* ProfilerListener::GetName(base::Vector<const char> name) {
374  // TODO(all): Change {StringsStorage} to accept non-null-terminated strings.
375  base::OwnedVector<char> null_terminated =
376  base::OwnedVector<char>::New(name.size() + 1);

406 void ProfilerListener::AttachDeoptInlinedFrames(DirectHandle<Code> code,
407  CodeDeoptEventRecord* rec) {
408  int deopt_id = rec->deopt_id;
409  SourcePosition last_position = SourcePosition::Unknown();
410  int mask = RelocInfo::ModeMask(RelocInfo::DEOPT_ID) |
411  RelocInfo::ModeMask(RelocInfo::DEOPT_SCRIPT_OFFSET) |
412  RelocInfo::ModeMask(RelocInfo::DEOPT_INLINING_ID);
413 
414  rec->deopt_frames = nullptr;
415  rec->deopt_frame_count = 0;
416 
417  for (RelocIterator it(*code, mask); !it.done(); it.next()) {
418  RelocInfo* info = it.rinfo();
419  if (info->rmode() == RelocInfo::DEOPT_SCRIPT_OFFSET) {
420  int script_offset = static_cast<int>(info->data());
421  it.next();
422  DCHECK(it.rinfo()->rmode() == RelocInfo::DEOPT_INLINING_ID);
423  int inlining_id = static_cast<int>(it.rinfo()->data());
424  last_position = SourcePosition(script_offset, inlining_id);
425  continue;
426  }
427  if (info->rmode() == RelocInfo::DEOPT_ID) {
428  if (deopt_id != static_cast<int>(info->data())) continue;
429  if (!last_position.IsKnown()) continue;
430 
431  // SourcePosition::InliningStack allocates a handle for the SFI of each
432  // frame. These don't escape this function, but quickly add up. This
433  // scope limits their lifetime.
434  HandleScope scope(isolate_);
435  std::vector<SourcePositionInfo> stack =
436  last_position.InliningStack(isolate_, *code);
437  CpuProfileDeoptFrame* deopt_frames =
438  new CpuProfileDeoptFrame[stack.size()];
439 
440  int deopt_frame_count = 0;
441  for (SourcePositionInfo& pos_info : stack) {
442  if (pos_info.position.ScriptOffset() == kNoSourcePosition) continue;
443  if (pos_info.script.is_null()) continue;
444  int script_id = pos_info.script->id();
445  size_t offset = static_cast<size_t>(pos_info.position.ScriptOffset());
446  deopt_frames[deopt_frame_count++] = {script_id, offset};
447  }
448  rec->deopt_frames = deopt_frames;
449  rec->deopt_frame_count = deopt_frame_count;
450  break;
451  }
452  }
453 }
