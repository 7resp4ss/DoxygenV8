84 AssemblerOptions BuiltinAssemblerOptions(Isolate* isolate, Builtin builtin) {
85  AssemblerOptions options = AssemblerOptions::Default(isolate);
86  CHECK(!options.isolate_independent_code);
87  CHECK(!options.collect_win64_unwind_info);
88 
89 #if V8_ENABLE_WEBASSEMBLY
90  if (wasm::BuiltinLookup::IsWasmBuiltinId(builtin) ||
91  builtin == Builtin::kJSToWasmWrapper ||
92  builtin == Builtin::kJSToWasmHandleReturns ||
93  builtin == Builtin::kWasmToJsWrapperCSA) {
94  options.is_wasm = true;
95  }
96 #endif
97  if (!isolate->IsGeneratingEmbeddedBuiltins()) {
98  return options;
99  }
100 
101  const base::AddressRegion& code_region = isolate->heap()->code_region();
102  bool pc_relative_calls_fit_in_code_range =
103  !code_region.is_empty() &&
104  std::ceil(static_cast<float>(code_region.size() / MB)) <=
105  kMaxPCRelativeCodeRangeInMB;
106 
107  // Mksnapshot ensures that the code range is small enough to guarantee that
108  // PC-relative call/jump instructions can be used for builtin to builtin
109  // calls/tail calls. The embedded builtins blob generator also ensures that.
110  // However, there are serializer tests, where we force isolate creation at
111  // runtime and at this point, Code space isn't restricted to a
112  // size s.t. PC-relative calls may be used. So, we fall back to an indirect
113  // mode.
114  options.use_pc_relative_calls_and_jumps_for_mksnapshot =
115  pc_relative_calls_fit_in_code_range;
116 
117  options.builtin_call_jump_mode = BuiltinCallJumpMode::kForMksnapshot;
118  options.isolate_independent_code = true;
119  options.collect_win64_unwind_info = true;
120 
121  if (builtin == Builtin::kInterpreterEntryTrampolineForProfiling) {
122  // InterpreterEntryTrampolineForProfiling must be generated in a position
123  // independent way because it might be necessary to create a copy of the
124  // builtin in the code space if the v8_flags.interpreted_frames_native_stack
125  // is enabled.
126  options.builtin_call_jump_mode = BuiltinCallJumpMode::kIndirect;
127  }
128 
129  return options;
130 }

139 DirectHandle<Code> BuildPlaceholder(Isolate* isolate, Builtin builtin) {
140  HandleScope scope(isolate);
141  uint8_t buffer[kBufferSize];
142  MacroAssembler masm(isolate, CodeObjectRequired::kYes,
143  ExternalAssemblerBuffer(buffer, kBufferSize));
144  DCHECK(!masm.has_frame());
145  {
146  FrameScope frame_scope(&masm, StackFrame::NO_FRAME_TYPE);
147  // The contents of placeholder don't matter, as long as they don't create
148  // embedded constants or external references.
149  masm.Move(kJavaScriptCallCodeStartRegister, Smi::zero());
150  masm.Call(kJavaScriptCallCodeStartRegister);
151  }
152  CodeDesc desc;
153  masm.GetCode(isolate, &desc);
154  Handle<Code> code = Factory::CodeBuilder(isolate, desc, CodeKind::BUILTIN)
155  .set_self_reference(masm.CodeObject())
156  .set_builtin(builtin)
157  .Build();
158  return scope.CloseAndEscape(code);
159 }

161 V8_NOINLINE Tagged<Code> BuildWithMacroAssembler(
162  Isolate* isolate, Builtin builtin, MacroAssemblerGenerator generator,
163  const char* s_name) {
164  HandleScope scope(isolate);
165  uint8_t buffer[kBufferSize];
166 
167  MacroAssembler masm(isolate, BuiltinAssemblerOptions(isolate, builtin),
168  CodeObjectRequired::kYes,
169  ExternalAssemblerBuffer(buffer, kBufferSize));
170  masm.set_builtin(builtin);
171  DCHECK(!masm.has_frame());
172  masm.CodeEntry();
173  generator(&masm);
174 
175  int handler_table_offset = 0;
176 
177  // JSEntry builtins are a special case and need to generate a handler table.
178  DCHECK_EQ(Builtins::KindOf(Builtin::kJSEntry), Builtins::ASM);
179  DCHECK_EQ(Builtins::KindOf(Builtin::kJSConstructEntry), Builtins::ASM);
180  DCHECK_EQ(Builtins::KindOf(Builtin::kJSRunMicrotasksEntry), Builtins::ASM);
181  if (Builtins::IsJSEntryVariant(builtin)) {
182  handler_table_offset = HandlerTable::EmitReturnTableStart(&masm);
183  HandlerTable::EmitReturnEntry(
184  &masm, 0, isolate->builtins()->js_entry_handler_offset());
185 #if V8_ENABLE_DRUMBRAKE
186  } else if (builtin == Builtin::kWasmInterpreterCWasmEntry) {
187  handler_table_offset = HandlerTable::EmitReturnTableStart(&masm);
188  HandlerTable::EmitReturnEntry(
189  &masm, 0,
190  isolate->builtins()->cwasm_interpreter_entry_handler_offset());
191 #endif // V8_ENABLE_DRUMBRAKE
192  }
193 #if V8_ENABLE_WEBASSEMBLY && \
194  (V8_TARGET_ARCH_X64 || V8_TARGET_ARCH_ARM64 || V8_TARGET_ARCH_IA32 || \
195  V8_TARGET_ARCH_ARM || V8_TARGET_ARCH_RISCV64 || V8_TARGET_ARCH_LOONG64 || \
196  V8_TARGET_ARCH_PPC64 || V8_TARGET_ARCH_S390X || V8_TARGET_ARCH_RISCV32)
197  if (builtin == Builtin::kWasmReturnPromiseOnSuspendAsm) {
198  handler_table_offset = HandlerTable::EmitReturnTableStart(&masm);
199  HandlerTable::EmitReturnEntry(
200  &masm, 0, isolate->builtins()->jspi_prompt_handler_offset());
201  }
202 #endif
203 
204  CodeDesc desc;
205  masm.GetCode(isolate->main_thread_local_isolate(), &desc,
206  MacroAssembler::kNoSafepointTable, handler_table_offset);
207 
208  DirectHandle<Code> code =
209  Factory::CodeBuilder(isolate, desc, CodeKind::BUILTIN)
210  .set_self_reference(masm.CodeObject())
211  .set_builtin(builtin)
212  .Build();
213 #if defined(V8_OS_WIN64)
214  isolate->SetBuiltinUnwindData(builtin, masm.GetUnwindInfo());
215 #endif // V8_OS_WIN64
216  return *code;
217 }

219 Tagged<Code> BuildAdaptor(Isolate* isolate, Builtin builtin,
220  Address builtin_address, const char* name) {
221  HandleScope scope(isolate);
222  uint8_t buffer[kBufferSize];
223  MacroAssembler masm(isolate, BuiltinAssemblerOptions(isolate, builtin),
224  CodeObjectRequired::kYes,
225  ExternalAssemblerBuffer(buffer, kBufferSize));
226  masm.set_builtin(builtin);
227  DCHECK(!masm.has_frame());
228  int formal_parameter_count = Builtins::GetFormalParameterCount(builtin);
229  Builtins::Generate_Adaptor(&masm, formal_parameter_count, builtin_address);
230  CodeDesc desc;
231  masm.GetCode(isolate, &desc);
232  DirectHandle<Code> code =
233  Factory::CodeBuilder(isolate, desc, CodeKind::BUILTIN)
234  .set_self_reference(masm.CodeObject())
235  .set_builtin(builtin)
236  .set_parameter_count(formal_parameter_count)
237  .Build();
238  return *code;
239 }

348 void SetupIsolateDelegate::PopulateWithPlaceholders(Isolate* isolate) {
349  // Fill the builtins list with placeholders. References to these placeholder
350  // builtins are eventually replaced by the actual builtins. This is to
351  // support circular references between builtins.
352  Builtins* builtins = isolate->builtins();
353  HandleScope scope(isolate);
354  for (Builtin builtin = Builtins::kFirst; builtin <= Builtins::kLast;
355  ++builtin) {
356  DirectHandle<Code> placeholder = BuildPlaceholder(isolate, builtin);
357  AddBuiltin(builtins, builtin, *placeholder);
358  }
359 }

361 // static
362 void SetupIsolateDelegate::ReplacePlaceholders(Isolate* isolate) {
363  // Replace references from all builtin code objects to placeholders.
364  Builtins* builtins = isolate->builtins();
365  DisallowGarbageCollection no_gc;
366  static const int kRelocMask =
367  RelocInfo::ModeMask(RelocInfo::CODE_TARGET) |
368  RelocInfo::ModeMask(RelocInfo::FULL_EMBEDDED_OBJECT) |
369  RelocInfo::ModeMask(RelocInfo::COMPRESSED_EMBEDDED_OBJECT) |
370  RelocInfo::ModeMask(RelocInfo::RELATIVE_CODE_TARGET);
371  PtrComprCageBase cage_base(isolate);
372  for (Builtin builtin = Builtins::kFirst; builtin <= Builtins::kLast;
373  ++builtin) {
374  Tagged<Code> code = builtins->code(builtin);
375  Tagged<InstructionStream> istream = code->instruction_stream();
376  WritableJitAllocation jit_allocation = ThreadIsolation::LookupJitAllocation(
377  istream.address(), istream->Size(),
378  ThreadIsolation::JitAllocationType::kInstructionStream, true);
379  bool flush_icache = false;
380  for (WritableRelocIterator it(jit_allocation, istream,
381  code->constant_pool(), kRelocMask);
382  !it.done(); it.next()) {
383  WritableRelocInfo* rinfo = it.rinfo();
384  if (RelocInfo::IsCodeTargetMode(rinfo->rmode())) {
385  Tagged<Code> target_code =
386  Code::FromTargetAddress(rinfo->target_address());
387  DCHECK_IMPLIES(
388  RelocInfo::IsRelativeCodeTarget(rinfo->rmode()),
389  Builtins::IsIsolateIndependent(target_code->builtin_id()));
390  if (!target_code->is_builtin()) continue;
391  Tagged<Code> new_target = builtins->code(target_code->builtin_id());
392  rinfo->set_target_address(istream, new_target->instruction_start(),
393  UPDATE_WRITE_BARRIER, SKIP_ICACHE_FLUSH);
394  } else {
395  DCHECK(RelocInfo::IsEmbeddedObjectMode(rinfo->rmode()));
396  Tagged<Object> object = rinfo->target_object(cage_base);
397  if (!IsCode(object, cage_base)) continue;
398  Tagged<Code> target = Cast<Code>(object);
399  if (!target->is_builtin()) continue;
400  Tagged<Code> new_target = builtins->code(target->builtin_id());
401  rinfo->set_target_object(istream, new_target, UPDATE_WRITE_BARRIER,
402  SKIP_ICACHE_FLUSH);
403  }
404  flush_icache = true;
405  }
406  if (flush_icache) {
407  FlushInstructionCache(code->instruction_start(),
408  code->instruction_size());
409  }
410  }
411 }

413 // static
414 void SetupIsolateDelegate::SetupBuiltinsInternal(
415  Isolate* isolate, bool compute_builtins_effects) {
416  Builtins* builtins = isolate->builtins();
417  DCHECK(!builtins->initialized_);
418 
419  BuiltinsEffectsAnalyzer* builtins_effects_analyzer = nullptr;
420  if (compute_builtins_effects) {
421  builtins_effects_analyzer = BuiltinsEffectsAnalyzer::Setup(isolate);
422  }
423 
424  if (v8_flags.dump_builtins_hashes_to_file) {
425  // Create an empty file.
426  std::ofstream(v8_flags.dump_builtins_hashes_to_file, std::ios_base::trunc);
427  }
428 
429  PopulateWithPlaceholders(isolate);
430 
431  // Create a scope for the handles in the builtins.
432  HandleScope scope(isolate);
433 
434  // Generated builtins are temporarily stored in this array to avoid data races
435  // on the isolate's builtin table.
436  std::array<Handle<Code>, Builtins::kBuiltinCount> generated_builtins;
437  auto install_generated_builtin = [&generated_builtins, isolate](
438  Builtin builtin, Handle<Code> code) {
439  DCHECK_EQ(ThreadId::Current(), isolate->thread_id());
440  USE(isolate);
441  generated_builtins[Builtins::ToInt(builtin)] = code;
442  };
443 
444  int builtins_built_without_job_count = 0;
445  int job_creation_order = 0;
446  BuiltinCompilationScheduler scheduler;
447  Tagged<Code> code;
448 
449 #define BUILD_CPP_WITHOUT_JOB(Name, Argc) \
450  code = BuildAdaptor(isolate, Builtin::k##Name, \
451  FUNCTION_ADDR(Builtin_##Name), #Name); \
452  generated_builtins[Builtins::ToInt(Builtin::k##Name)] = \
453  handle(code, isolate); \
454  builtins_built_without_job_count++;
455 
456 #define BUILD_TSJ_WITHOUT_JOB(Name, Argc, ...) \
457  code = BuildWithTurboshaftAssemblerJS( \
458  isolate, Builtin::k##Name, &Builtins::Generate_##Name, Argc, #Name); \
459  AddBuiltin(builtins, Builtin::k##Name, code); \
460  builtins_built_without_job_count++;
461 
462 #define BUILD_TFJ_WITH_JOB(Name, Argc, ...) \
463  CompileJSLinkageCodeStubBuiltin(isolate, Builtin::k##Name, \
464  &Builtins::Generate_##Name, \
465  install_generated_builtin, Argc, #Name, \
466  job_creation_order++, scheduler);
467 
468 #define BUILD_TSC_WITHOUT_JOB(Name, InterfaceDescriptor) \
469  /* Return size is from the provided CallInterfaceDescriptor. */ \
470  code = BuildWithTurboshaftAssemblerCS( \
471  isolate, Builtin::k##Name, &Builtins::Generate_##Name, \
472  CallDescriptors::InterfaceDescriptor, #Name); \
473  generated_builtins[Builtins::ToInt(Builtin::k##Name)] = \
474  handle(code, isolate); \
475  builtins_built_without_job_count++;
476 
477 #define BUILD_TFC_WITH_JOB(Name, InterfaceDescriptor) \
478  /* Return size is from the provided CallInterfaceDescriptor. */ \
479  CompileCSLinkageCodeStubBuiltin( \
480  isolate, Builtin::k##Name, &Builtins::Generate_##Name, \
481  install_generated_builtin, CallDescriptors::InterfaceDescriptor, #Name, \
482  job_creation_order++, scheduler);
483 
484 #define BUILD_TFS_WITH_JOB(Name, ...) \
485  /* Return size for generic TF builtins (stub linkage) is always 1. */ \
486  CompileCSLinkageCodeStubBuiltin( \
487  isolate, Builtin::k##Name, &Builtins::Generate_##Name, \
488  install_generated_builtin, CallDescriptors::Name, #Name, \
489  job_creation_order++, scheduler);
490 
491 #define BUILD_TFH_WITH_JOB(Name, InterfaceDescriptor) \
492  /* Return size for IC builtins/handlers is always 1. */ \
493  CompileCSLinkageCodeStubBuiltin( \
494  isolate, Builtin::k##Name, &Builtins::Generate_##Name, \
495  install_generated_builtin, CallDescriptors::InterfaceDescriptor, #Name, \
496  job_creation_order++, scheduler);
497 
498 #define BUILD_BCH_WITH_JOB(Name, OperandScale, Bytecode) \
499  CompileBytecodeHandler(isolate, Builtin::k##Name, OperandScale, Bytecode, \
500  install_generated_builtin, job_creation_order++, \
501  scheduler);
502 
503 #define BUILD_ASM_WITHOUT_JOB(Name, InterfaceDescriptor) \
504  code = BuildWithMacroAssembler(isolate, Builtin::k##Name, \
505  Builtins::Generate_##Name, #Name); \
506  generated_builtins[Builtins::ToInt(Builtin::k##Name)] = \
507  handle(code, isolate); \
508  builtins_built_without_job_count++;
509 
510 #define NOP(...)
511 
512  // Build all builtins without jobs first. When concurrent builtin generation
513  // is enabled, allocations needs to be deterministic. Having main-thread
514  // generated builtins in the middle of the list breaks determinism.
515  BUILTIN_LIST(BUILD_CPP_WITHOUT_JOB, BUILD_TSJ_WITHOUT_JOB, NOP,
516  BUILD_TSC_WITHOUT_JOB, NOP, NOP, NOP, NOP,
517  BUILD_ASM_WITHOUT_JOB);
518  BUILTIN_LIST(NOP, NOP, BUILD_TFJ_WITH_JOB, NOP, BUILD_TFC_WITH_JOB,
519  BUILD_TFS_WITH_JOB, BUILD_TFH_WITH_JOB, BUILD_BCH_WITH_JOB, NOP)
520 
521 #undef BUILD_CPP_WITHOUT_JOB
522 #undef BUILD_TSJ_WITHOUT_JOB
523 #undef BUILD_TFJ_WITH_JOB
524 #undef BUILD_TSC_WITHOUT_JOB
525 #undef BUILD_TFC_WITH_JOB
526 #undef BUILD_TFS_WITH_JOB
527 #undef BUILD_TFH_WITH_JOB
528 #undef BUILD_BCH_WITH_JOB
529 #undef BUILD_ASM_WITHOUT_JOB
530 #undef NOP
531 
532  scheduler.AwaitAndFinalizeCurrentBatch(isolate);
533  CHECK_EQ(Builtins::kBuiltinCount, builtins_built_without_job_count +
534  scheduler.builtins_installed_count());
535 
536  if (compute_builtins_effects) {
537  DCHECK_NOT_NULL(builtins_effects_analyzer);
538  builtins_effects_analyzer->Finalize();
539  }
