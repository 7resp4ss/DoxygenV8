9 UseMap::UseMap(const Graph& graph, Zone* zone, FunctionType filter)
10  : table_(graph.op_id_count(), zone, &graph),
11  uses_(zone),
12  saturated_uses_(zone) {
13  ZoneVector<std::pair<OpIndex, OpIndex>> delayed_phi_uses(zone);
14 
15  // We preallocate for 2 uses per operation.
16  uses_.reserve(graph.op_id_count() * 2);
17 
18  // We skip {offset:0} to use {offset == 0} as uninitialized.
19  uint32_t offset = 1;
20  for (uint32_t index = 0; index < graph.block_count(); ++index) {
21  BlockIndex block_index(index);
22  const Block& block = graph.Get(block_index);
23 
24  auto block_ops = graph.OperationIndices(block);
25  for (OpIndex op_index : block_ops) {
26  const Operation& op = graph.Get(op_index);
27  // When we see a definition, we allocate space in the {uses_}.
28  DCHECK_EQ(table_[op_index].offset, 0);
29  DCHECK_EQ(table_[op_index].count, 0);
30 
31  if (op.saturated_use_count.IsSaturated()) {
32  table_[op_index].offset =
33  -static_cast<int32_t>(saturated_uses_.size()) - 1;
34  saturated_uses_.emplace_back(zone);
35  saturated_uses_.back().reserve(std::numeric_limits<uint8_t>::max());
36  } else {
37  table_[op_index].offset = offset;
38  offset += op.saturated_use_count.Get();
39  uses_.resize(offset);
40  }
41 
42  if (filter(graph, op, zone)) continue;
43 
44  if (block.IsLoop()) {
45  if (op.Is<PhiOp>()) {
46  DCHECK_EQ(op.input_count, 2);
47  DCHECK_EQ(PhiOp::kLoopPhiBackEdgeIndex, 1);
48  AddUse(&graph, op.input(0), op_index);
49  // Delay back edge of loop Phis.
50  delayed_phi_uses.emplace_back(op.input(1), op_index);
51  continue;
52  }
53  }
54 
55  // Add uses.
56  for (OpIndex input_index : op.inputs()) {
57  AddUse(&graph, input_index, op_index);
58  }
59  }
60  }
61 
62  for (auto [input_index, op_index] : delayed_phi_uses) {
63  AddUse(&graph, input_index, op_index);
64  }
65 }

67 base::Vector<const OpIndex> UseMap::uses(OpIndex index) const {
68  DCHECK(index.valid());
69  int32_t offset = table_[index].offset;
70  uint32_t count = table_[index].count;
71  DCHECK_NE(offset, 0);
72  if (V8_LIKELY(offset > 0)) {
73  return base::Vector<const OpIndex>(uses_.data() + offset, count);
74  } else {
75  DCHECK_EQ(count, saturated_uses_[-offset - 1].size());
76  return base::Vector<const OpIndex>(saturated_uses_[-offset - 1].data(),
77  count);
78  }
79 }

81 void UseMap::AddUse(const Graph* graph, OpIndex node, OpIndex use) {
82  int32_t input_offset = table_[node].offset;
83  uint32_t& input_count = table_[node].count;
84  DCHECK_NE(input_offset, 0);
85  if (V8_LIKELY(input_offset > 0)) {
86  DCHECK_LT(input_count, graph->Get(node).saturated_use_count.Get());
87  DCHECK(!uses_[input_offset + input_count].valid());
88  uses_[input_offset + input_count] = use;
89  } else {
90  ZoneVector<OpIndex>& uses = saturated_uses_[-input_offset - 1];
91  DCHECK_EQ(uses.size(), input_count);
92  uses.emplace_back(use);
93  }
94  ++input_count;
95 }
