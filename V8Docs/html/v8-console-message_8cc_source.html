72 class V8ValueStringBuilder {
73  public:
74  static String16 toString(v8::Local<v8::Value> value,
75  v8::Local<v8::Context> context) {
76  V8ValueStringBuilder builder(context);
77  if (!builder.append(value)) return String16();
78  return builder.toString();
79  }
80 
81  private:
82  enum {
83  IgnoreNull = 1 << 0,
84  IgnoreUndefined = 1 << 1,
85  };
86 
87  explicit V8ValueStringBuilder(v8::Local<v8::Context> context)
88  : m_arrayLimit(maxArrayItemsLimit),
89  m_isolate(v8::Isolate::GetCurrent()),
90  m_visitedArrays(v8::Isolate::GetCurrent()),
91  m_tryCatch(v8::Isolate::GetCurrent()),
92  m_context(context) {}
93 
94  bool append(v8::Local<v8::Value> value, unsigned ignoreOptions = 0) {
95  if (value.IsEmpty()) return true;
96  if ((ignoreOptions & IgnoreNull) && value->IsNull()) return true;
97  if ((ignoreOptions & IgnoreUndefined) && value->IsUndefined()) return true;
98  if (value->IsBigIntObject()) {
99  value = value.As<v8::BigIntObject>()->ValueOf();
100  } else if (value->IsBooleanObject()) {
101  value =
102  v8::Boolean::New(m_isolate, value.As<v8::BooleanObject>()->ValueOf());
103  } else if (value->IsNumberObject()) {
104  value =
105  v8::Number::New(m_isolate, value.As<v8::NumberObject>()->ValueOf());
106  } else if (value->IsStringObject()) {
107  value = value.As<v8::StringObject>()->ValueOf();
108  } else if (value->IsSymbolObject()) {
109  value = value.As<v8::SymbolObject>()->ValueOf();
110  }
111  if (value->IsString()) return append(value.As<v8::String>());
112  if (value->IsBigInt()) return append(value.As<v8::BigInt>());
113  if (value->IsSymbol()) return append(value.As<v8::Symbol>());
114  if (value->IsArray()) return append(value.As<v8::Array>());
115  if (value->IsProxy()) {
116  m_builder.append("[object Proxy]");
117  return true;
118  }
119  if (value->IsObject() && !value->IsDate() && !value->IsFunction() &&
120  !value->IsNativeError() && !value->IsRegExp()) {
121  v8::Local<v8::Object> object = value.As<v8::Object>();
122  v8::Local<v8::String> stringValue;
123  if (object->ObjectProtoToString(m_context).ToLocal(&stringValue))
124  return append(stringValue);
125  }
126  v8::Local<v8::String> stringValue;
127  if (!value->ToString(m_context).ToLocal(&stringValue)) return false;
128  return append(stringValue);
129  }
130 
131  bool append(v8::Local<v8::Array> array) {
132  for (const auto& it : m_visitedArrays) {
133  if (it == array) return true;
134  }
135  uint32_t length = array->Length();
136  if (length > m_arrayLimit) return false;
137  if (m_visitedArrays.size() > maxStackDepthLimit) return false;
138 
139  bool result = true;
140  m_arrayLimit -= length;
141  m_visitedArrays.push_back(array);
142  for (uint32_t i = 0; i < length; ++i) {
143  if (i) m_builder.append(',');
144  v8::Local<v8::Value> value;
145  if (!array->Get(m_context, i).ToLocal(&value)) continue;
146  if (!append(value, IgnoreNull | IgnoreUndefined)) {
147  result = false;
148  break;
149  }
150  }
151  m_visitedArrays.pop_back();
152  return result;
153  }
154 
155  bool append(v8::Local<v8::Symbol> symbol) {
156  m_builder.append("Symbol(");
157  bool result = append(symbol->Description(m_isolate), IgnoreUndefined);
158  m_builder.append(')');
159  return result;
160  }
161 
162  bool append(v8::Local<v8::BigInt> bigint) {
163  v8::Local<v8::String> bigint_string;
164  if (!bigint->ToString(m_context).ToLocal(&bigint_string)) return false;
165  bool result = append(bigint_string);
166  if (m_tryCatch.HasCaught()) return false;
167  m_builder.append('n');
168  return result;
169  }
170 
171  bool append(v8::Local<v8::String> string) {
172  if (m_tryCatch.HasCaught()) return false;
173  if (!string.IsEmpty()) {
174  m_builder.append(toProtocolString(m_isolate, string));
175  }
176  return true;
177  }
178 
179  String16 toString() {
180  if (m_tryCatch.HasCaught()) return String16();
181  return m_builder.toString();
182  }
183 
184  uint32_t m_arrayLimit;
185  v8::Isolate* m_isolate;
186  String16Builder m_builder;
187  v8::LocalVector<v8::Array> m_visitedArrays;
188  v8::TryCatch m_tryCatch;
189  v8::Local<v8::Context> m_context;
190 };

254 std::unique_ptr<protocol::Array<protocol::Runtime::RemoteObject>>
255 V8ConsoleMessage::wrapArguments(V8InspectorSessionImpl* session,
256  bool generatePreview) const {
257  V8InspectorImpl* inspector = session->inspector();
258  int contextGroupId = session->contextGroupId();
259  int contextId = m_contextId;
260  if (m_arguments.empty() || !contextId) return nullptr;
261  InspectedContext* inspectedContext =
262  inspector->getContext(contextGroupId, contextId);
263  if (!inspectedContext) return nullptr;
264 
265  v8::Isolate* isolate = inspectedContext->isolate();
266  v8::HandleScope handles(isolate);
267  v8::Local<v8::Context> context = inspectedContext->context();
268 
269  auto args =
270  std::make_unique<protocol::Array<protocol::Runtime::RemoteObject>>();
271 
272  v8::Local<v8::Value> value = m_arguments[0]->Get(isolate);
273  if (value->IsObject() && m_type == ConsoleAPIType::kTable &&
274  generatePreview) {
275  v8::MaybeLocal<v8::Array> columns;
276  if (m_arguments.size() > 1) {
277  v8::Local<v8::Value> secondArgument = m_arguments[1]->Get(isolate);
278  if (secondArgument->IsArray()) {
279  columns = secondArgument.As<v8::Array>();
280  } else if (secondArgument->IsString()) {
281  v8::TryCatch tryCatch(isolate);
282  v8::Local<v8::Array> array = v8::Array::New(isolate);
283  if (array->Set(context, 0, secondArgument).IsJust()) {
284  columns = array;
285  }
286  }
287  }
288  std::unique_ptr<protocol::Runtime::RemoteObject> wrapped =
289  session->wrapTable(context, value.As<v8::Object>(), columns);
290  inspectedContext = inspector->getContext(contextGroupId, contextId);
291  if (!inspectedContext) return nullptr;
292  if (wrapped) {
293  args->emplace_back(std::move(wrapped));
294  } else {
295  args = nullptr;
296  }
297  } else {
298  for (size_t i = 0; i < m_arguments.size(); ++i) {
299  std::unique_ptr<protocol::Runtime::RemoteObject> wrapped =
300  session->wrapObject(context, m_arguments[i]->Get(isolate), "console",
301  generatePreview);
302  inspectedContext = inspector->getContext(contextGroupId, contextId);
303  if (!inspectedContext) return nullptr;
304  if (!wrapped) {
305  args = nullptr;
306  break;
307  }
308  args->emplace_back(std::move(wrapped));
309  }
310  }
311  return args;
312 }

314 void V8ConsoleMessage::reportToFrontend(protocol::Runtime::Frontend* frontend,
315  V8InspectorSessionImpl* session,
316  bool generatePreview) const {
317  int contextGroupId = session->contextGroupId();
318  V8InspectorImpl* inspector = session->inspector();
319  // Protect against reentrant debugger calls via interrupts.
320  v8::debug::PostponeInterruptsScope no_interrupts(inspector->isolate());
321 
322  if (m_origin == V8MessageOrigin::kException) {
323  if (!inspector->hasConsoleMessageStorage(contextGroupId)) return;
324  v8::HandleScope scope(inspector->isolate());
325  auto maybeScriptOrigin =
326  v8::debug::GetScriptOrigin(inspector->isolate(), m_scriptId);
327  const bool isSharedCrossOrigin =
328  maybeScriptOrigin ? maybeScriptOrigin->Options().IsSharedCrossOrigin()
329  : false;
330  std::unique_ptr<protocol::Runtime::RemoteObject> exception =
331  isSharedCrossOrigin ? wrapException(session, generatePreview) : nullptr;
332  const bool includeException = isSharedCrossOrigin && exception;
333  std::unique_ptr<protocol::Runtime::ExceptionDetails> exceptionDetails =
334  protocol::Runtime::ExceptionDetails::create()
335  .setExceptionId(m_exceptionId)
336  .setText(includeException
337  ? m_message
338  : (m_detailedMessage.length() ? m_detailedMessage
339  : m_message))
340  .setLineNumber(m_lineNumber ? m_lineNumber - 1 : 0)
341  .setColumnNumber(m_columnNumber ? m_columnNumber - 1 : 0)
342  .build();
343  if (m_scriptId)
344  exceptionDetails->setScriptId(String16::fromInteger(m_scriptId));
345  if (!m_url.isEmpty()) exceptionDetails->setUrl(m_url);
346  if (m_stackTrace) {
347  exceptionDetails->setStackTrace(
348  m_stackTrace->buildInspectorObjectImpl(inspector->debugger()));
349  }
350  if (m_contextId) exceptionDetails->setExecutionContextId(m_contextId);
351  if (includeException) {
352  exceptionDetails->setException(std::move(exception));
353  }
354  std::unique_ptr<protocol::DictionaryValue> data =
355  getAssociatedExceptionData(inspector, session);
356  if (data) exceptionDetails->setExceptionMetaData(std::move(data));
357  frontend->exceptionThrown(m_timestamp, std::move(exceptionDetails));
358  return;
359  }
360  if (m_origin == V8MessageOrigin::kRevokedException) {
361  frontend->exceptionRevoked(m_message, m_revokedExceptionId);
362  return;
363  }
364  if (m_origin == V8MessageOrigin::kConsole) {
365  std::unique_ptr<protocol::Array<protocol::Runtime::RemoteObject>>
366  arguments = wrapArguments(session, generatePreview);
367  if (!inspector->hasConsoleMessageStorage(contextGroupId)) return;
368  if (!arguments) {
369  arguments =
370  std::make_unique<protocol::Array<protocol::Runtime::RemoteObject>>();
371  if (!m_message.isEmpty()) {
372  std::unique_ptr<protocol::Runtime::RemoteObject> messageArg =
373  protocol::Runtime::RemoteObject::create()
374  .setType(protocol::Runtime::RemoteObject::TypeEnum::String)
375  .build();
376  messageArg->setValue(protocol::StringValue::create(m_message));
377  arguments->emplace_back(std::move(messageArg));
378  }
379  }
380  std::optional<String16> consoleContext;
381  if (!m_consoleContext.isEmpty()) consoleContext = m_consoleContext;
382  std::unique_ptr<protocol::Runtime::StackTrace> stackTrace;
383  if (m_stackTrace) {
384  switch (m_type) {
385  case ConsoleAPIType::kAssert:
386  case ConsoleAPIType::kError:
387  case ConsoleAPIType::kTrace:
388  case ConsoleAPIType::kWarning:
389  stackTrace =
390  m_stackTrace->buildInspectorObjectImpl(inspector->debugger());
391  break;
392  default:
393  stackTrace =
394  m_stackTrace->buildInspectorObjectImpl(inspector->debugger(), 0);
395  break;
396  }
397  }
398  frontend->consoleAPICalled(
399  consoleAPITypeValue(m_type), std::move(arguments), m_contextId,
400  m_timestamp, std::move(stackTrace), std::move(consoleContext));
401  return;
402  }
403  UNREACHABLE();
404 }

409  if (m_arguments.empty() || !m_contextId) return nullptr;

442 // static
443 std::unique_ptr<V8ConsoleMessage> V8ConsoleMessage::createForConsoleAPI(
444  v8::Local<v8::Context> v8Context, int contextId, int groupId,
445  V8InspectorImpl* inspector, double timestamp, ConsoleAPIType type,
446  v8::MemorySpan<const v8::Local<v8::Value>> arguments,
447  const String16& consoleContext,
448  std::unique_ptr<V8StackTraceImpl> stackTrace) {
449  v8::Isolate* isolate = v8::Isolate::GetCurrent();
450 
451  std::unique_ptr<V8ConsoleMessage> message(
452  new V8ConsoleMessage(V8MessageOrigin::kConsole, timestamp, String16()));
453  if (stackTrace && !stackTrace->isEmpty()) {
454  message->m_url = toString16(stackTrace->topSourceURL());
455  message->m_lineNumber = stackTrace->topLineNumber();
456  message->m_columnNumber = stackTrace->topColumnNumber();
457  }
458  message->m_stackTrace = std::move(stackTrace);
459  message->m_consoleContext = consoleContext;
460  message->m_type = type;
461  message->m_contextId = contextId;
462  for (v8::Local<v8::Value> arg : arguments) {
463  std::unique_ptr<v8::Global<v8::Value>> argument(
464  new v8::Global<v8::Value>(isolate, arg));
465  argument->AnnotateStrongRetainer(kGlobalConsoleMessageHandleLabel);
466  message->m_arguments.push_back(std::move(argument));
467  message->m_v8Size += v8::debug::EstimatedValueSize(isolate, arg);
468  }
469  bool sep = false;
470  for (v8::Local<v8::Value> arg : arguments) {
471  if (sep) {
472  message->m_message += String16(" ");
473  } else {
474  sep = true;
475  }
476  message->m_message += V8ValueStringBuilder::toString(arg, v8Context);
477  }
478 
479  v8::Isolate::MessageErrorLevel clientLevel = v8::Isolate::kMessageInfo;
480  if (type == ConsoleAPIType::kDebug || type == ConsoleAPIType::kCount ||
481  type == ConsoleAPIType::kTimeEnd) {
482  clientLevel = v8::Isolate::kMessageDebug;
483  } else if (type == ConsoleAPIType::kError ||
484  type == ConsoleAPIType::kAssert) {
485  clientLevel = v8::Isolate::kMessageError;
486  } else if (type == ConsoleAPIType::kWarning) {
487  clientLevel = v8::Isolate::kMessageWarning;
488  } else if (type == ConsoleAPIType::kInfo) {
489  clientLevel = v8::Isolate::kMessageInfo;
490  } else if (type == ConsoleAPIType::kLog) {
491  clientLevel = v8::Isolate::kMessageLog;
492  }
493 
494  if (type != ConsoleAPIType::kClear) {
495  inspector->client()->consoleAPIMessage(
496  groupId, clientLevel, toStringView(message->m_message),
497  toStringView(message->m_url), message->m_lineNumber,
498  message->m_columnNumber, message->m_stackTrace.get());
499  }
500 
501  return message;
502 }

579  V8InspectorImpl* inspector = m_inspector;
580  if (message->type() == ConsoleAPIType::kClear) clear();
581 
582  TraceV8ConsoleMessageEvent(message->origin(), message->type());

584  inspector->forEachSession(
585  contextGroupId, [&message](V8InspectorSessionImpl* session) {
586  if (message->origin() == V8MessageOrigin::kConsole)
587  session->consoleAgent()->messageAdded(message.get());
588  session->runtimeAgent()->messageAdded(message.get());
589  });
590  if (!inspector->hasConsoleMessageStorage(contextGroupId)) return;
591 
592  DCHECK(m_messages.size() <= maxConsoleMessageCount);
593  if (m_messages.size() == maxConsoleMessageCount) {
594  m_estimatedSize -= m_messages.front()->estimatedSize();
595  m_messages.pop_front();
596  }

598  !m_messages.empty()) {
599  m_estimatedSize -= m_messages.front()->estimatedSize();
600  m_messages.pop_front();
601  }
602 
603  m_messages.push_back(std::move(message));
604  m_estimatedSize += m_messages.back()->estimatedSize();
605 }
