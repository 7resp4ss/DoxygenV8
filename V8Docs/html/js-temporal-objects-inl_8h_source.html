46 #define TEMPORAL_INLINE_SIGNED_GETTER_SETTER(T, data, field, lower, upper, B) \
47  inline void T::set_##field(int32_t field) { \
48  DCHECK_GE(upper, field); \
49  DCHECK_LE(lower, field); \
50  int hints = data(); \
51  /* Mask out unrelated bits */ \
52  field &= (static_cast<uint32_t>(int32_t{-1})) ^ \
53  (static_cast<uint32_t>(int32_t{-1}) << B##Bits::kSize); \
54  hints = B##Bits::update(hints, field); \
55  set_##data(hints); \
56  } \
57  inline int32_t T::field() const { \
58  int32_t v = B##Bits::decode(data()); \
59  /* Restore bits for negative values based on the MSB in that field */ \
60  v |= ((int32_t{1} << (B##Bits::kSize - 1) & v) \
61  ? (static_cast<uint32_t>(int32_t{-1}) << B##Bits::kSize) \
62  : 0); \
63  DCHECK_GE(upper, v); \
64  DCHECK_LE(lower, v); \
