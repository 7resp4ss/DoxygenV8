72  ? SafepointKind::kGlobal

78 Tagged<Code> DeoptimizableCodeIterator::Next() {
79  while (true) {
80  Tagged<HeapObject> object = object_iterator_->Next();
81  if (object.is_null()) {
82  // No objects left in the current iterator, try to move to the next space
83  // based on the state.
84  switch (state_) {
85  case kIteratingCodeSpace: {
86  object_iterator_ =
87  isolate_->heap()->code_lo_space()->GetObjectIterator(
88  isolate_->heap());
89  state_ = kIteratingCodeLOSpace;
90  continue;
91  }
92  case kIteratingCodeLOSpace:
93  // No other spaces to iterate, so clean up and we're done. Keep the
94  // object iterator so that it keeps returning null on Next(), to avoid
95  // needing to branch on state_ before the while loop, but drop the
96  // safepoint scope since we no longer need to stop the heap from
97  // moving.
98  safepoint_scope_.reset();
99  state_ = kDone;
100  [[fallthrough]];
101  case kDone:
102  return Code();
103  }
104  }
105  Tagged<InstructionStream> istream = Cast<InstructionStream>(object);
106  Tagged<Code> code;
107  if (!istream->TryGetCode(&code, kAcquireLoad)) continue;
108  if (!CodeKindCanDeoptimize(code->kind())) continue;
109  return code;
110  }
111 }

300 
301  TranslatedState translated_values(frame);

303 
304  TranslatedState::iterator frame_it = translated_values.end();
305  int counter = jsframe_index;
306  for (auto it = translated_values.begin(); it != translated_values.end();

320  // We only include kJavaScriptBuiltinContinuation frames above to get the
321  // counting right.
322  CHECK_EQ(frame_it->kind(), TranslatedFrame::kUnoptimizedFunction);

324  DeoptimizedFrameInfo* info =

331 class ActivationsFinder : public ThreadVisitor {
332  public:
333  ActivationsFinder(Tagged<GcSafeCode> topmost_optimized_code,
334  bool safe_to_deopt_topmost_optimized_code) {
335 #ifdef DEBUG
336  topmost_ = topmost_optimized_code;
337  safe_to_deopt_ = safe_to_deopt_topmost_optimized_code;
338 #endif
339  }

341  // Find the frames with activations of codes marked for deoptimization, search
342  // for the trampoline to the deoptimizer call respective to each code, and use
343  // it to replace the current pc on the stack.
344  void VisitThread(Isolate* isolate, ThreadLocalTop* top) override {
345  for (StackFrameIterator it(isolate, top); !it.done(); it.Advance()) {
346  if (it.frame()->is_optimized_js()) {
347  Tagged<GcSafeCode> code = it.frame()->GcSafeLookupCode();
348  if (CodeKindCanDeoptimize(code->kind()) &&
349  code->marked_for_deoptimization()) {
350  // Obtain the trampoline to the deoptimizer call.
351  int trampoline_pc;
352  if (code->is_maglevved()) {
353  MaglevSafepointEntry safepoint = MaglevSafepointTable::FindEntry(
354  isolate, code, it.frame()->pc());
355  trampoline_pc = safepoint.trampoline_pc();
356  } else {
357  SafepointEntry safepoint = SafepointTable::FindEntry(
358  isolate, code, it.frame()->maybe_unauthenticated_pc());
359  trampoline_pc = safepoint.trampoline_pc();
360  }
361  // TODO(saelo): currently we have to use full pointer comparison as
362  // builtin Code is still inside the sandbox while runtime-generated
363  // Code is in trusted space.
364  static_assert(!kAllCodeObjectsLiveInTrustedSpace);
365  DCHECK_IMPLIES(code.SafeEquals(topmost_), safe_to_deopt_);
366  static_assert(SafepointEntry::kNoTrampolinePC == -1);
367  CHECK_GE(trampoline_pc, 0);
368  Address new_pc = code->instruction_start() + trampoline_pc;
369  if (it.frame()->InFastCCall()) {
370  // Fast C calls call directly to C++, so there may not be a return
371  // address on the stack. Instead we patch the code after the call.
372  // For C calls the caller has to pop parameters off the stack. This
373  // has to happen before deoptimization. Therefore we add the offset
374  // here that is needed for popping the arguments.
375  Address pc =
376  *it.frame()->pc_address() + kMaxSizeOfMoveAfterFastCall;
377  Deoptimizer::PatchToJump(pc, new_pc);
378  } else {
379  if (v8_flags.cet_compatible) {
380  Address pc = *it.frame()->pc_address();
381  Deoptimizer::PatchToJump(pc, new_pc);
382  } else {
383  // Replace the current pc on the stack with the trampoline.
384  // TODO(v8:10026): avoid replacing a signed pointer.
385  Address* pc_addr = it.frame()->pc_address();
386  PointerAuthentication::ReplacePC(pc_addr, new_pc,
387  kSystemPointerSize);
388  }
389  }
390  }
391  }
392  }
393  }
394 
395  private:
396 #ifdef DEBUG
397  Tagged<GcSafeCode> topmost_;
398  bool safe_to_deopt_;
399 #endif
400 };

403 // Replace pc on the stack for codes marked for deoptimization.
404 // static
405 void Deoptimizer::DeoptimizeMarkedCode(Isolate* isolate) {
406  DisallowGarbageCollection no_gc;
407 
408  Tagged<GcSafeCode> topmost_optimized_code;
409  bool safe_to_deopt_topmost_optimized_code = false;
410 #ifdef DEBUG
411  // Make sure all activations of optimized code can deopt at their current PC.
412  // The topmost optimized code has special handling because it cannot be
413  // deoptimized due to weak object dependency.
414  for (StackFrameIterator it(isolate, isolate->thread_local_top()); !it.done();
415  it.Advance()) {
416  if (it.frame()->is_optimized_js()) {
417  Tagged<GcSafeCode> code = it.frame()->GcSafeLookupCode();
418  Tagged<JSFunction> function =
419  static_cast<OptimizedJSFrame*>(it.frame())->function();
420  TraceFoundActivation(isolate, function);
421  bool safe_if_deopt_triggered;
422  if (code->is_maglevved()) {
423  MaglevSafepointEntry safepoint =
424  MaglevSafepointTable::FindEntry(isolate, code, it.frame()->pc());
425  safe_if_deopt_triggered = safepoint.has_deoptimization_index();
426  } else {
427  SafepointEntry safepoint = SafepointTable::FindEntry(
428  isolate, code, it.frame()->maybe_unauthenticated_pc());
429  safe_if_deopt_triggered = safepoint.has_deoptimization_index();
430  }
431 
432  // Deopt is checked when we are patching addresses on stack.
433  bool is_builtin_code = code->kind() == CodeKind::BUILTIN;
434  DCHECK(topmost_optimized_code.is_null() || safe_if_deopt_triggered ||
435  is_builtin_code);
436  if (topmost_optimized_code.is_null()) {
437  topmost_optimized_code = code;
438  safe_to_deopt_topmost_optimized_code = safe_if_deopt_triggered;
439  }
440  }
441  }
442 #endif
443 
444  ActivationsFinder visitor(topmost_optimized_code,
445  safe_to_deopt_topmost_optimized_code);
446  // Iterate over the stack of this thread.
447  visitor.VisitThread(isolate, isolate->thread_local_top());
448  // In addition to iterate over the stack of this thread, we also
449  // need to consider all the other threads as they may also use
450  // the code currently beings deoptimized.
451  isolate->thread_manager()->IterateArchivedThreads(&visitor);
452 }

454 void Deoptimizer::DeoptimizeAll(Isolate* isolate) {
455  RCS_SCOPE(isolate, RuntimeCallCounterId::kDeoptimizeCode);
456  TimerEventScope<TimerEventDeoptimizeCode> timer(isolate);

467  }
468  }
469

474 void Deoptimizer::DeoptimizeFunction(Tagged<JSFunction> function,
475  LazyDeoptimizeReason reason,
476  Tagged<Code> code) {
477  Isolate* isolate = Isolate::Current();
478  RCS_SCOPE(isolate, RuntimeCallCounterId::kDeoptimizeCode);
479  TimerEventScope<TimerEventDeoptimizeCode> timer(isolate);
480  TRACE_EVENT0("v8", "V8.DeoptimizeCode");
481  function->ResetIfCodeFlushed(isolate);
482  if (code.is_null()) code = function->code(isolate);
483 
484  if (CodeKindCanDeoptimize(code->kind())) {
485  // Mark the code for deoptimization and unlink any functions that also
486  // refer to that code. The code cannot be shared across native contexts,
487  // so we only need to search one.
488  code->SetMarkedForDeoptimization(isolate, reason);
489 #ifndef V8_ENABLE_LEAPTIERING_BOOL
490  // The code in the function's optimized code feedback vector slot might
491  // be different from the code on the function - evict it if necessary.
492  function->feedback_vector()->EvictOptimizedCodeMarkedForDeoptimization(
493  isolate, function->shared(), "unlinking code marked for deopt");
494 #endif // !V8_ENABLE_LEAPTIERING_BOOL
495 
496  DeoptimizeMarkedCode(isolate);
497  }

602 Deoptimizer::Deoptimizer(Isolate* isolate, Tagged<JSFunction> function,
603  DeoptimizeKind kind, Address from, int fp_to_sp_delta)
604  : isolate_(isolate),
605  function_(function),
606  deopt_exit_index_(kFixedExitSizeMarker),
607  deopt_kind_(kind),
608  from_(from),
609  fp_to_sp_delta_(fp_to_sp_delta),
610  deoptimizing_throw_(false),
611  catch_handler_data_(-1),
612  catch_handler_pc_offset_(-1),
613  restart_frame_index_(-1),
614  input_(nullptr),
615  output_count_(0),
616  output_(nullptr),
617  caller_frame_top_(0),
618  caller_fp_(0),
619  caller_pc_(0),
620  caller_constant_pool_(0),
621  actual_argument_count_(0),
622  stack_fp_(0),
623  trace_scope_(v8_flags.trace_deopt || v8_flags.log_deopt
624  ? new CodeTracer::Scope(isolate->GetCodeTracer())
625  : nullptr) {
626  if (isolate->deoptimizer_lazy_throw()) {
627  CHECK_EQ(kind, DeoptimizeKind::kLazy);
628  isolate->set_deoptimizer_lazy_throw(false);
629  deoptimizing_throw_ = true;
630  }
631 
632  if (isolate->debug()->IsRestartFrameScheduled()) {
633  CHECK(deoptimizing_throw_);
634  restart_frame_index_ = isolate->debug()->restart_inline_frame_index();
635  CHECK_GE(restart_frame_index_, 0);
636  isolate->debug()->clear_restart_frame();
637  }
638 
639  DCHECK_NE(from, kNullAddress);
640 
641 #ifdef DEBUG
642  DCHECK(AllowGarbageCollection::IsAllowed());
643  disallow_garbage_collection_ = new DisallowGarbageCollection();
644 #endif // DEBUG
645 
646 #if V8_ENABLE_WEBASSEMBLY
647  if (v8_flags.wasm_deopt && function.is_null()) {
648  // From now on we should not be accessing any in-sandbox data as all deopt
649  // data is trusted and so stored outside the heap.
650  no_sandbox_access_during_wasm_deopt_.emplace();
651 
652  wasm::WasmCode* code =
653  wasm::GetWasmCodeManager()->LookupCode(isolate, from);
654  compiled_optimized_wasm_code_ = code;
655  DCHECK_NOT_NULL(code);
656  CHECK_EQ(code->kind(), wasm::WasmCode::kWasmFunction);
657  wasm::WasmDeoptView deopt_view(code->deopt_data());
658  const wasm::WasmDeoptData& deopt_data = deopt_view.GetDeoptData();
659  DCHECK_NE(deopt_data.translation_array_size, 0);
660  CHECK_GE(from, deopt_data.deopt_exit_start_offset);
661  Address deopt_exit_offset = from - code->instruction_start();
662  // All eager deopt exits are calls "at the end" of the code to the builtin
663  // generated by Generate_DeoptimizationEntry_Eager. These calls have a fixed
664  // size kEagerDeoptExitsSize and the deopt data contains the offset of the
665  // first such call to the beginning of the code, so we can map any PC of
666  // such call to a unique index for this deopt point.
667  deopt_exit_index_ =
668  static_cast<uint32_t>(deopt_exit_offset -
669  deopt_data.deopt_exit_start_offset -
670  kEagerDeoptExitSize) /
671  kEagerDeoptExitSize;
672 
673  // Note: The parameter stack slots are not really part of the frame.
674  // However, the deoptimizer needs access to the incoming parameter values
675  // and therefore they need to be included in the FrameDescription. Between
676  // the parameters and the actual frame there are 2 pointers (the caller's pc
677  // and saved stack pointer) that therefore also need to be included. Both
678  // pointers as well as the incoming parameter stack slots are going to be
679  // copied into the outgoing FrameDescription which will "push" them back
680  // onto the stack. (This is consistent with how JS handles this.)
681  const wasm::FunctionSig* sig =
682  code->native_module()->module()->functions[code->index()].sig;
683  int parameter_stack_slots, return_stack_slots;
684  GetWasmStackSlotsCounts(sig, &parameter_stack_slots, &return_stack_slots);
685 
686  unsigned input_frame_size = fp_to_sp_delta +
687  parameter_stack_slots * kSystemPointerSize +
688  CommonFrameConstants::kFixedFrameSizeAboveFp;
689  input_ = FrameDescription::Create(input_frame_size, parameter_stack_slots,
690  isolate_);
691  return;
692  }
693 #endif
694 
695  compiled_code_ = isolate_->heap()->FindCodeForInnerPointer(from);
696  DCHECK(!compiled_code_.is_null());
697  DCHECK(IsCode(compiled_code_));
698 
699  DCHECK(IsJSFunction(function));
700  CHECK(CodeKindCanDeoptimize(compiled_code_->kind()));
701  {
702  HandleScope scope(isolate_);
703  PROFILE(isolate_, CodeDeoptEvent(direct_handle(compiled_code_, isolate_),
704  kind, from_, fp_to_sp_delta_));
705  }
706  unsigned size = ComputeInputFrameSize();
707  const int parameter_count = compiled_code_->parameter_count();
708  DCHECK_EQ(
709  parameter_count,
710  function->shared()->internal_formal_parameter_count_with_receiver());
711  input_ = FrameDescription::Create(size, parameter_count, isolate_);
712 
713  DCHECK_EQ(deopt_exit_index_, kFixedExitSizeMarker);
714  // Calculate the deopt exit index from return address.
715  DCHECK_GT(kEagerDeoptExitSize, 0);
716  DCHECK_GT(kLazyDeoptExitSize, 0);
717  Tagged<DeoptimizationData> deopt_data =
718  Cast<DeoptimizationData>(compiled_code_->deoptimization_data());
719  Address deopt_start = compiled_code_->instruction_start() +
720  deopt_data->DeoptExitStart().value();
721  int eager_deopt_count = deopt_data->EagerDeoptCount().value();
722  Address lazy_deopt_start =
723  deopt_start + eager_deopt_count * kEagerDeoptExitSize;
724  // The deoptimization exits are sorted so that lazy deopt exits appear after
725  // eager deopts.
726  static_assert(static_cast<int>(DeoptimizeKind::kLazy) ==
727  static_cast<int>(kLastDeoptimizeKind),
728  "lazy deopts are expected to be emitted last");
729  // from_ is the value of the link register after the call to the
730  // deoptimizer, so for the last lazy deopt, from_ points to the first
731  // non-lazy deopt, so we use <=, similarly for the last non-lazy deopt and
732  // the first deopt with resume entry.
733  if (from_ <= lazy_deopt_start) {
734  DCHECK_EQ(kind, DeoptimizeKind::kEager);
735  int offset = static_cast<int>(from_ - kEagerDeoptExitSize - deopt_start);
736  DCHECK_EQ(0, offset % kEagerDeoptExitSize);
737  deopt_exit_index_ = offset / kEagerDeoptExitSize;
738  } else {
739  DCHECK_EQ(kind, DeoptimizeKind::kLazy);
740  int offset =
741  static_cast<int>(from_ - kLazyDeoptExitSize - lazy_deopt_start);
742  DCHECK_EQ(0, offset % kLazyDeoptExitSize);
743  deopt_exit_index_ = eager_deopt_count + (offset / kLazyDeoptExitSize);
744  }

978 FrameDescription* Deoptimizer::DoComputeWasmLiftoffFrame(
979  TranslatedFrame& frame, wasm::NativeModule* native_module,
980  Tagged<WasmTrustedInstanceData> wasm_trusted_instance, int frame_index,
981  std::stack<intptr_t>& shadow_stack) {
982  // Given inlined frames where function a calls b, b is considered the topmost
983  // because b is on top of the call stack! This is aligned with the names used
984  // by the JS deopt.
985  const bool is_bottommost = frame_index == 0;
986  const bool is_topmost = output_count_ - 1 == frame_index;
987  // Recompile the liftoff (unoptimized) wasm code for the input frame.
988  // TODO(mliedtke): This recompiles every single function even if it never got
989  // optimized and exists as a liftoff variant in the WasmCodeManager as we also
990  // need to compute the deopt information. Can we avoid some of the extra work
991  // here?
992  auto [wasm_code, liftoff_description] = CompileWithLiftoffAndGetDeoptInfo(
993  native_module, frame.wasm_function_index(), frame.bytecode_offset(),
994  is_topmost);
995 
996  DCHECK(liftoff_description);
997 
998  int parameter_stack_slots, return_stack_slots;
999  const wasm::FunctionSig* sig =
1000  native_module->module()->functions[frame.wasm_function_index()].sig;
1001  GetWasmStackSlotsCounts(sig, &parameter_stack_slots, &return_stack_slots);
1002 
1003  // Allocate and populate the FrameDescription describing the output frame.
1004  const uint32_t output_frame_size = liftoff_description->total_frame_size;
1005  const uint32_t total_output_frame_size =
1006  output_frame_size + parameter_stack_slots * kSystemPointerSize +
1007  CommonFrameConstants::kFixedFrameSizeAboveFp;
1008 
1009  if (verbose_tracing_enabled()) {
1010  std::ostringstream outstream;
1011  outstream << " Liftoff stack & register state for function index "
1012  << frame.wasm_function_index() << ", frame size "
1013  << output_frame_size << ", total frame size "
1014  << total_output_frame_size << '\n';
1015  size_t index = 0;
1016  for (const wasm::LiftoffVarState& state : liftoff_description->var_state) {
1017  outstream << " " << index++ << ": " << state << '\n';
1018  }
1019  FILE* file = trace_scope()->file();
1020  PrintF(file, "%s", outstream.str().c_str());
1021  }
1022 
1023  FrameDescription* output_frame = FrameDescription::Create(
1024  total_output_frame_size, parameter_stack_slots, isolate());
1025 
1026  // Copy the parameter stack slots.
1027  static_assert(CommonFrameConstants::kFixedFrameSizeAboveFp ==
1028  2 * kSystemPointerSize);
1029  uint32_t output_offset = total_output_frame_size;
1030  // Zero out the incoming parameter slots. This will make sure that tagged
1031  // values are safely ignored by the gc.
1032  // Note that zero is clearly not the correct value. Still, liftoff copies
1033  // all parameters into "its own" stack slots at the beginning and always
1034  // uses these slots to restore parameters from the stack.
1035  for (int i = 0; i < parameter_stack_slots; ++i) {
1036  output_offset -= kSystemPointerSize;
1037  output_frame->SetFrameSlot(output_offset, 0);
1038  }
1039 
1040  // Calculate top and update previous caller's pc.
1041  Address top = is_bottommost ? caller_frame_top_ - total_output_frame_size
1042  : output_[frame_index - 1]->GetTop() -
1043  total_output_frame_size;
1044  output_frame->SetTop(top);
1045  Address pc = wasm_code->instruction_start() + liftoff_description->pc_offset;
1046  // Sign the PC. Note that for the non-topmost frames the stack pointer at
1047  // which the PC is stored as the "caller pc" / return address depends on the
1048  // amount of parameter stack slots of the callee. To simplify the code, we
1049  // just sign it as if there weren't any parameter stack slots.
1050  // When building up the next frame we can check and "move" the caller PC by
1051  // signing it again with the correct stack pointer.
1052  output_frame->SetPc(PointerAuthentication::SignAndCheckPC(
1053  isolate(), pc, output_frame->GetTop()));
1054 #ifdef V8_ENABLE_CET_SHADOW_STACK
1055  if (v8_flags.cet_compatible) {
1056  if (is_topmost) {
1057  shadow_stack.push(pc);
1058  } else {
1059  shadow_stack.push(wasm_code->instruction_start() +
1060  liftoff_description->adapt_shadow_stack_pc_offset);
1061  }
1062  }
1063 #endif // V8_ENABLE_CET_SHADOW_STACK
1064 
1065  // Sign the previous frame's PC.
1066  if (is_bottommost) {
1067  Address old_context =
1068  caller_frame_top_ - input_->parameter_count() * kSystemPointerSize;
1069  Address new_context =
1070  caller_frame_top_ - parameter_stack_slots * kSystemPointerSize;
1071  caller_pc_ = PointerAuthentication::MoveSignedPC(isolate(), caller_pc_,
1072  new_context, old_context);
1073  } else if (parameter_stack_slots != 0) {
1074  // The previous frame's PC is stored at a different stack slot, so we need
1075  // to re-sign the PC for the new context (stack pointer).
1076  FrameDescription* previous_frame = output_[frame_index - 1];
1077  Address old_context = previous_frame->GetTop();
1078  Address new_context =
1079  old_context - parameter_stack_slots * kSystemPointerSize;
1080  Address signed_pc = PointerAuthentication::MoveSignedPC(
1081  isolate(), previous_frame->GetPc(), new_context, old_context);
1082  previous_frame->SetPc(signed_pc);
1083  }
1084 
1085  // Store the caller PC.
1086  output_offset -= kSystemPointerSize;
1087  output_frame->SetFrameSlot(
1088  output_offset,
1089  is_bottommost ? caller_pc_ : output_[frame_index - 1]->GetPc());
1090  // Store the caller frame pointer.
1091  output_offset -= kSystemPointerSize;
1092  output_frame->SetFrameSlot(
1093  output_offset,
1094  is_bottommost ? caller_fp_ : output_[frame_index - 1]->GetFp());
1095 
1096  CHECK_EQ(output_frame_size, output_offset);
1097  int base_offset = output_frame_size;
1098 
1099  // Set trusted instance data on output frame.
1100  output_frame->SetFrameSlot(
1101  base_offset - WasmLiftoffFrameConstants::kInstanceDataOffset,
1102  wasm_trusted_instance.ptr());
1103  if (liftoff_description->trusted_instance != no_reg) {
1104  output_frame->SetRegister(liftoff_description->trusted_instance.code(),
1105  wasm_trusted_instance.ptr());
1106  }
1107 
1108  DCHECK_GE(translated_state_.frames().size(), 1);
1109  auto liftoff_iter = liftoff_description->var_state.begin();
1110  if constexpr (Is64()) {
1111  // On 32 bit platforms int64s are represented as 2 values on Turbofan.
1112  // Liftoff on the other hand treats them as 1 value (a register pair).
1113  CHECK_EQ(liftoff_description->var_state.size(), frame.GetValueCount());
1114  }
1115 
1116  bool int64_lowering_is_low = true;
1117 
1118  for (const TranslatedValue& value : frame) {
1119  bool skip_increase_liftoff_iter = false;
1120  switch (liftoff_iter->loc()) {
1121  case wasm::LiftoffVarState::kIntConst:
1122  if (!Is64() && liftoff_iter->kind() == wasm::ValueKind::kI64) {
1123  if (int64_lowering_is_low) skip_increase_liftoff_iter = true;
1124  int64_lowering_is_low = !int64_lowering_is_low;
1125  }
1126  break; // Nothing to be done for constants in liftoff frame.
1127  case wasm::LiftoffVarState::kRegister:
1128  if (liftoff_iter->is_gp_reg()) {
1129  intptr_t reg_value = kZapValue;
1130  switch (value.kind()) {
1131  case TranslatedValue::Kind::kInt32:
1132  // Ensure that the upper half is zeroed out.
1133  reg_value = static_cast<uint32_t>(value.int32_value());
1134  break;
1135  case TranslatedValue::Kind::kTagged:
1136  reg_value = value.raw_literal().ptr();
1137  break;
1138  case TranslatedValue::Kind::kInt64:
1139  reg_value = value.int64_value();
1140  break;
1141  default:
1142  UNIMPLEMENTED();
1143  }
1144  output_frame->SetRegister(liftoff_iter->reg().gp().code(), reg_value);
1145  } else if (liftoff_iter->is_fp_reg()) {
1146  switch (value.kind()) {
1147  case TranslatedValue::Kind::kDouble:
1148  output_frame->SetDoubleRegister(liftoff_iter->reg().fp().code(),
1149  value.double_value());
1150  break;
1151  case TranslatedValue::Kind::kFloat:
1152  // Liftoff doesn't have a concept of floating point registers.
1153  // This is an important distinction as e.g. on arm s1 and d1 are
1154  // two completely distinct registers.
1155  static_assert(std::is_same_v<decltype(liftoff_iter->reg().fp()),
1156  DoubleRegister>);
1157  output_frame->SetDoubleRegister(
1158  liftoff_iter->reg().fp().code(),
1159  Float64::FromBits(value.float_value().get_bits()));
1160  break;
1161  case TranslatedValue::Kind::kSimd128:
1162  output_frame->SetSimd128Register(liftoff_iter->reg().fp().code(),
1163  value.simd_value());
1164  break;
1165  default:
1166  UNIMPLEMENTED();
1167  }
1168  } else if (!Is64() && liftoff_iter->is_gp_reg_pair()) {
1169  intptr_t reg_value = kZapValue;
1170  switch (value.kind()) {
1171  case TranslatedValue::Kind::kInt32:
1172  // Ensure that the upper half is zeroed out.
1173  reg_value = static_cast<uint32_t>(value.int32_value());
1174  break;
1175  case TranslatedValue::Kind::kTagged:
1176  reg_value = value.raw_literal().ptr();
1177  break;
1178  default:
1179  UNREACHABLE();
1180  }
1181  int8_t reg = int64_lowering_is_low
1182  ? liftoff_iter->reg().low_gp().code()
1183  : liftoff_iter->reg().high_gp().code();
1184  output_frame->SetRegister(reg, reg_value);
1185  if (int64_lowering_is_low) skip_increase_liftoff_iter = true;
1186  int64_lowering_is_low = !int64_lowering_is_low;
1187  } else if (!Is64() && liftoff_iter->is_fp_reg_pair()) {
1188  CHECK_EQ(value.kind(), TranslatedValue::Kind::kSimd128);
1189  Simd128 simd_value = value.simd_value();
1190  Address val_ptr = reinterpret_cast<Address>(&simd_value);
1191  output_frame->SetDoubleRegister(
1192  liftoff_iter->reg().low_fp().code(),
1193  Float64::FromBits(base::ReadUnalignedValue<uint64_t>(val_ptr)));
1194  output_frame->SetDoubleRegister(
1195  liftoff_iter->reg().high_fp().code(),
1196  Float64::FromBits(base::ReadUnalignedValue<uint64_t>(
1197  val_ptr + sizeof(double))));
1198  } else {
1199  UNREACHABLE();
1200  }
1201  break;
1202  case wasm::LiftoffVarState::kStack:
1203 #ifdef V8_TARGET_BIG_ENDIAN
1204  static constexpr int kLiftoffStackBias = 4;
1205 #else
1206  static constexpr int kLiftoffStackBias = 0;
1207 #endif
1208  switch (liftoff_iter->kind()) {
1209  case wasm::ValueKind::kI32:
1210  CHECK(value.kind() == TranslatedValue::Kind::kInt32 ||
1211  value.kind() == TranslatedValue::Kind::kUint32);
1212  output_frame->SetLiftoffFrameSlot32(
1213  base_offset - liftoff_iter->offset() + kLiftoffStackBias,
1214  value.int32_value_);
1215  break;
1216  case wasm::ValueKind::kF32:
1217  CHECK_EQ(value.kind(), TranslatedValue::Kind::kFloat);
1218  output_frame->SetLiftoffFrameSlot32(
1219  base_offset - liftoff_iter->offset() + kLiftoffStackBias,
1220  value.float_value().get_bits());
1221  break;
1222  case wasm::ValueKind::kI64:
1223  if constexpr (Is64()) {
1224  CHECK(value.kind() == TranslatedValue::Kind::kInt64 ||
1225  value.kind() == TranslatedValue::Kind::kUint64);
1226  output_frame->SetLiftoffFrameSlot64(
1227  base_offset - liftoff_iter->offset(), value.int64_value_);
1228  } else {
1229  CHECK(value.kind() == TranslatedValue::Kind::kInt32 ||
1230  value.kind() == TranslatedValue::Kind::kUint32);
1231  // TODO(bigendian): Either the offsets or the default for
1232  // int64_lowering_is_low might have to be swapped.
1233  if (int64_lowering_is_low) {
1234  skip_increase_liftoff_iter = true;
1235  output_frame->SetLiftoffFrameSlot32(
1236  base_offset - liftoff_iter->offset(), value.int32_value_);
1237  } else {
1238  output_frame->SetLiftoffFrameSlot32(
1239  base_offset - liftoff_iter->offset() + sizeof(int32_t),
1240  value.int32_value_);
1241  }
1242  int64_lowering_is_low = !int64_lowering_is_low;
1243  }
1244  break;
1245  case wasm::ValueKind::kS128: {
1246  int64x2 values = value.simd_value().to_i64x2();
1247  const int offset = base_offset - liftoff_iter->offset();
1248  output_frame->SetLiftoffFrameSlot64(offset, values.val[0]);
1249  output_frame->SetLiftoffFrameSlot64(offset + sizeof(int64_t),
1250  values.val[1]);
1251  break;
1252  }
1253  case wasm::ValueKind::kF64:
1254  CHECK_EQ(value.kind(), TranslatedValue::Kind::kDouble);
1255  output_frame->SetLiftoffFrameSlot64(
1256  base_offset - liftoff_iter->offset(),
1257  value.double_value().get_bits());
1258  break;
1259  case wasm::ValueKind::kRef:
1260  case wasm::ValueKind::kRefNull:
1261  CHECK_EQ(value.kind(), TranslatedValue::Kind::kTagged);
1262  output_frame->SetLiftoffFrameSlotPointer(
1263  base_offset - liftoff_iter->offset(), value.raw_literal_.ptr());
1264  break;
1265  default:
1266  UNIMPLEMENTED();
1267  }
1268  break;
1269  }
1270  DCHECK_IMPLIES(skip_increase_liftoff_iter, !Is64());
1271  if (!skip_increase_liftoff_iter) {
1272  ++liftoff_iter;
1273  }
1274  }
1275 
1276  // Store frame kind.
1277  uint32_t frame_type_offset =
1278  base_offset + WasmLiftoffFrameConstants::kFrameTypeOffset;
1279  output_frame->SetFrameSlot(frame_type_offset,
1280  StackFrame::TypeToMarker(StackFrame::WASM));
1281  // Fill feedback vector stack slot.
1282  // Instead of storing the actual feedback vector, we simply store the declared
1283  // function index of the wasm function. This is done because the feedback
1284  // vector may not exist yet (in case of multiple instantiations of the same
1285  // wasm module) and heap allocations during a deoptimization aren't allowed.
1286  // The Smi in the feedback vector slot will be overwritten with the actual
1287  // feedback vector object in the runtime function WasmLiftoffDeoptFinish.
1288  uint32_t feedback_offset =
1289  base_offset - WasmLiftoffFrameConstants::kFeedbackVectorOffset;
1290  uint32_t fct_feedback_index = wasm::declared_function_index(
1291  native_module->module(), frame.wasm_function_index());
1292  output_frame->SetFrameSlot(feedback_offset,
1293  Smi::FromInt(fct_feedback_index).ptr());
1294 
1295  // Instead of a builtin continuation for wasm the deopt builtin will
1296  // call a c function to destroy the Deoptimizer object and then directly
1297  // return to the liftoff code.
1298  output_frame->SetContinuation(0);
1299 
1300  const intptr_t fp_value = top + output_frame_size;
1301  output_frame->SetFp(fp_value);
1302  Register fp_reg = JavaScriptFrame::fp_register();
1303  output_frame->SetRegister(fp_reg.code(), fp_value);
1304  output_frame->SetRegister(kRootRegister.code(), isolate()->isolate_root());
1305 #ifdef V8_COMPRESS_POINTERS
1306  output_frame->SetRegister(kPtrComprCageBaseRegister.code(),
1307  isolate()->cage_base());
1308 #endif
1309 
1310  return output_frame;
1311 }

1659  CHECK_EQ(shadow_stack_count_, 0);
1660  shadow_stack_ = new intptr_t[count + 1];
1661 
1662  // We should jump to the continuation through AdaptShadowStack to avoid
1663  // security exception.
1664  // Clear the continuation so that DeoptimizationEntry does not push the
1665  // address onto the stack, and push it to the shadow stack instead.
1666  if (output_[count - 1]->GetContinuation()) {
1667  shadow_stack_[shadow_stack_count_++] =
1668  output_[count - 1]->GetContinuation();
1669  output_[count - 1]->SetContinuation(0);
1670  }
1671 
1672  // Add topmost frame's pc to the shadow stack.
1673  shadow_stack_[shadow_stack_count_++] =
1674  output_[count - 1]->GetPc() -
1675  Deoptimizer::kAdaptShadowStackOffsetToSubtract;
1676 
1677  // Add return addresses to the shadow stack, except for the bottommost.
1678  // The bottommost frame's return address already exists in the shadow stack.
1679  for (int i = static_cast<int>(count) - 1; i > 0; i--) {
1680  if (!output_[i]->HasCallerPc()) continue;
1681  shadow_stack_[shadow_stack_count_++] =
1682  output_[i]->GetCallerPc() -
1683  Deoptimizer::kAdaptShadowStackOffsetToSubtract;
1684  }
1685  }
1686 #endif // V8_ENABLE_CET_SHADOW_STACK
1687 
1688  // Don't reset the tiering state for OSR code since we might reuse OSR code
1689  // after deopt, and we still want to tier up to non-OSR code even if OSR code
1690  // deoptimized.
1691  bool osr_early_exit = Deoptimizer::GetDeoptInfo().deopt_reason ==
1692  DeoptimizeReason::kOSREarlyExit;
1693  // TODO(saelo): We have to use full pointer comparisons here while not all
1694  // Code objects have been migrated into trusted space.
1695  static_assert(!kAllCodeObjectsLiveInTrustedSpace);
1696  if (IsJSFunction(function_) &&
1697  (compiled_code_->osr_offset().IsNone()
1698  ? function_->code(isolate()).SafeEquals(compiled_code_)
1699  : (!osr_early_exit &&
1700  DeoptExitIsInsideOsrLoop(
1701  isolate(), function_, bytecode_offset_in_outermost_frame_,
1702  compiled_code_->osr_offset(), compiled_code_->kind())))) {
1703  if (v8_flags.profile_guided_optimization &&
1704  function_->shared()->cached_tiering_decision() !=
1705  CachedTieringDecision::kDelayMaglev) {
1706  if (DeoptimizedMaglevvedCodeEarly(isolate(), function_, compiled_code_)) {
1707  function_->shared()->set_cached_tiering_decision(
1708  CachedTieringDecision::kDelayMaglev);
1709  } else {
1710  function_->shared()->set_cached_tiering_decision(
1711  CachedTieringDecision::kNormal);
1712  }
1713  }
1714  function_->ResetTieringRequests();
1715  // This allows us to quickly re-spawn a new compilation request even if
1716  // there is already one running. In particular it helps to squeeze in a
1717  // maglev compilation when there is a long running turbofan one that was
1718  // started right before the deopt.
1719  function_->SetTieringInProgress(isolate_, false);
1720  function_->SetInterruptBudget(isolate_, BudgetModification::kReset,
1721  CodeKind::INTERPRETED_FUNCTION);
1722  function_->feedback_vector()->set_was_once_deoptimized();
1723  }
1724 
1725  // Print some helpful diagnostic information.
1726  if (verbose_tracing_enabled()) {
1727  TraceDeoptEnd(timer.Elapsed().InMillisecondsF());
1728  }
1729  isolate()->counters()->deopts()->Increment();

1731  // The following invariant is fairly tricky to guarantee, since the size of
1732  // an optimized frame and its deoptimized counterparts usually differs. We
1733  // thus need to consider the case in which deoptimized frames are larger than
1734  // the optimized frame in stack checks in optimized code. We do this by
1735  // applying an offset to stack checks (see kArchStackPointerGreaterThan in the
1736  // code generator).
1737  // Note that we explicitly allow deopts to exceed the limit by a certain
1738  // number of slack bytes.
1739  CHECK_GT(
1740  static_cast<uintptr_t>(caller_frame_top_) - total_output_frame_size,
1741  stack_guard->real_jslimit() - kStackLimitSlackForDeoptimizationInBytes);
1742 }

1745 bool Deoptimizer::DeoptExitIsInsideOsrLoop(Isolate* isolate,
1746  Tagged<JSFunction> function,
1747  BytecodeOffset deopt_exit_offset,
1748  BytecodeOffset osr_offset,
1749  CodeKind code_kind) {
1750  DisallowGarbageCollection no_gc;
1751  HandleScope scope(isolate);
1752  DCHECK(!deopt_exit_offset.IsNone());
1753  DCHECK(!osr_offset.IsNone());
1754 
1755  Handle<BytecodeArray> bytecode_array(
1756  function->shared()->GetBytecodeArray(isolate), isolate);
1757  DCHECK(interpreter::BytecodeArrayIterator::IsValidOffset(
1758  bytecode_array, deopt_exit_offset.ToInt()));
1759 
1760  interpreter::BytecodeArrayIterator it(bytecode_array, osr_offset.ToInt());
1761  CHECK(it.CurrentBytecodeIsValidOSREntry());
1762  const int osr_loop_nesting_level = it.GetImmediateOperand(1);
1763 
1764  for (; !it.done(); it.Advance()) {
1765  const int current_offset = it.current_offset();
1766  // If we've reached the deopt exit, it's contained in the current loop
1767  // (this is covered by IsInRange below, but this check lets us avoid
1768  // useless iteration).
1769  if (current_offset == deopt_exit_offset.ToInt()) return true;
1770  // We're only interested in loop ranges.
1771  if (it.current_bytecode() != interpreter::Bytecode::kJumpLoop) continue;
1772  // Is the deopt exit contained in the current loop?
1773  if (base::IsInRange(deopt_exit_offset.ToInt(), it.GetJumpTargetOffset(),
1774  current_offset)) {
1775  return true;
1776  }
1777  // We've reached nesting level 0, i.e. the current JumpLoop concludes a
1778  // top-level loop.
1779  const int loop_nesting_level = it.GetImmediateOperand(1);
1780  if (loop_nesting_level == 0) return false;
1781  // Maglev never jumps above the OSR loop.
1782  if (code_kind == CodeKind::MAGLEV &&
1783  loop_nesting_level < osr_loop_nesting_level) {
1784  return false;
1785  }
1786  }
1787 
1788  UNREACHABLE();

1832  translated_frame->raw_shared_info()->TryGetDebugInfo(isolate());

1834  // TODO(leszeks): Validate this bytecode.

1851  // simulating what normal handlers do upon completion of the operation.
1852  // For baseline frames, the PC will be a builtin to convert the interpreter

1857  (!is_topmost || (deopt_kind_ == DeoptimizeKind::kLazy)) &&
1858  !goto_catch_handler;
1859  const bool restart_frame = goto_catch_handler && is_restart_frame();

1861  builtins->code(DispatchBuiltinFor(advance_bc, restart_frame));

1864  PrintF(trace_scope()->file(), " translating interpreted frame ");
1865  std::unique_ptr<char[]> name =
1866  translated_frame->raw_shared_info()->DebugNameCStr();
1867  PrintF(trace_scope()->file(), "%s", name.get());

1869  real_bytecode_offset);

1872  goto_catch_handler ? " (throw)" : "");

1874

1876  // this frame's size.

1878  is_bottommost ? caller_frame_top_ - output_frame_size
1879  : output_[frame_index - 1]->GetTop() - output_frame_size;
1880  output_frame->SetTop(top_address);
1881

1883  ReadOnlyRoots roots(isolate());

1885  for (int i = 0; i < ArgumentPaddingSlots(parameters_count); ++i) {
1886  frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
1887  }
1888  }
1889

1891  actual_argument_count_ > parameters_count) {
1892  PrintF(trace_scope_->file(),
1893  " -- %d extra argument(s) already in the stack --\n",
1894  actual_argument_count_ - parameters_count);
1895  }
1896  frame_writer.PushStackJSArguments(value_iterator, parameters_count);
1897

1899  frame_writer.top_offset());

1901  PrintF(trace_scope()->file(), " -------------------------\n");
1902  }

1904  // There are no translation commands for the caller's pc and fp, the
1905  // context, the function and the bytecode offset. Synthesize
1906  // their values and set them up
1907  // explicitly.
1908  //
1909  // The caller's pc for the bottommost output frame is the same as in the
1910  // input frame. For all subsequent output frames, it can be read from the
1911  // previous one. This frame's pc can be computed from the non-optimized
1912  // function code and bytecode offset of the bailout.
1913  if (is_bottommost) {
1914  frame_writer.PushBottommostCallerPc(caller_pc_);
1915  } else {
1916  frame_writer.PushApprovedCallerPc(output_[frame_index - 1]->GetPc());
1917  }
1918 
1919  // The caller's frame pointer for the bottommost output frame is the same
1920  // as in the input frame. For all subsequent output frames, it can be
1921  // read from the previous one. Also compute and set this frame's frame
1922  // pointer.
1923  const intptr_t caller_fp =
1924  is_bottommost ? caller_fp_ : output_[frame_index - 1]->GetFp();
1925  frame_writer.PushCallerFp(caller_fp);
1926 
1927  const intptr_t fp_value = top_address + frame_writer.top_offset();
1928  output_frame->SetFp(fp_value);
1929  if (is_topmost) {
1930  Register fp_reg = UnoptimizedJSFrame::fp_register();
1931  output_frame->SetRegister(fp_reg.code(), fp_value);
1932  }
1933 
1934  if (V8_EMBEDDED_CONSTANT_POOL_BOOL) {
1935  // For the bottommost output frame the constant pool pointer can be gotten
1936  // from the input frame. For subsequent output frames, it can be read from
1937  // the previous frame.
1938  const intptr_t caller_cp =
1939  is_bottommost ? caller_constant_pool_
1940  : output_[frame_index - 1]->GetConstantPool();
1941  frame_writer.PushCallerConstantPool(caller_cp);
1942  }
1943 
1944  // For the bottommost output frame the context can be gotten from the input
1945  // frame. For all subsequent output frames it can be gotten from the function
1946  // so long as we don't inline functions that need local contexts.
1947 
1948  // When deoptimizing into a catch block, we need to take the context
1949  // from a register that was specified in the handler table.
1950  TranslatedFrame::iterator context_pos = value_iterator++;
1951  if (goto_catch_handler) {
1952  // Skip to the translated value of the register specified
1953  // in the handler table.
1954  for (int i = 0; i < catch_handler_data_ + 1; ++i) {
1955  context_pos++;
1956  }
1957  }
1958  // Read the context from the translations.
1959  frame_writer.PushTranslatedValue(context_pos, "context");
1960 
1961  // The function was mentioned explicitly in the BEGIN_FRAME.
1962  frame_writer.PushTranslatedValue(function_iterator, "function");
1963 
1964  // Actual argument count.
1965  int argc;
1966  if (is_bottommost) {
1967  argc = actual_argument_count_;
1968  } else {
1969  TranslatedFrame::Kind previous_frame_kind =
1970  (translated_state_.frames()[frame_index - 1]).kind();
1971  argc = previous_frame_kind == TranslatedFrame::kInlinedExtraArguments
1972  ? output_[frame_index - 1]->parameter_count()
1973  : parameters_count;
1974  }
1975  frame_writer.PushRawValue(argc, "actual argument count\n");
1976

1978  frame_writer.PushRawObject(bytecode_array, "bytecode array\n");
1979

1988

1990  PrintF(trace_scope()->file(), " -------------------------\n");
1991  }
1992

1994  // The return_value_offset is counted from the top. Here, we compute the
1995  // register index (counted from the start).

1997  locals_count - translated_frame->return_value_offset();

2001  // normally to a lazy deopt point.
2002  if (is_topmost && !goto_catch_handler &&
2003  deopt_kind_ == DeoptimizeKind::kLazy && i >= return_value_first_reg &&

2006  if (return_index == 0) {
2007  frame_writer.PushRawValue(input_->GetRegister(kReturnRegister0.code()),
2008  "return value 0\n");

2010  // the accumulator and another one into an ordinary register. Since
2011  // the interpreter should never create such situation, just assert
2012  // this does not happen.
2013  CHECK_LE(return_value_first_reg + return_value_count, locals_count);
2014  } else {
2015  CHECK_EQ(return_index, 1);
2016  frame_writer.PushRawValue(input_->GetRegister(kReturnRegister1.code()),
2017  "return value 1\n");
2018  }
2019  } else {
2020  // This is not return value, just write the value from the translations.
2021  frame_writer.PushTranslatedValue(value_iterator, "stack parameter");
2022  }
2023  }
2024

2026  DCHECK_LE(register_slots_written, frame_info.register_stack_slot_count());

2031  frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");

2033

2035  if (is_topmost) {
2036  for (int i = 0; i < ArgumentPaddingSlots(1); ++i) {
2037  frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");

2040  // {NotifyDeoptimized} builtin pops it off the topmost frame (possibly
2041  // after materialization).
2042  if (goto_catch_handler) {
2043  // If we are lazy deopting to a catch handler, we set the accumulator to
2044  // the exception (which lives in the result register).
2045  intptr_t accumulator_value =
2046  input_->GetRegister(kInterpreterAccumulatorRegister.code());

2055  CHECK_EQ(translated_frame->return_value_count(), 1);
2056  frame_writer.PushRawValue(input_->GetRegister(kReturnRegister0.code()),

2058  } else {
2059  frame_writer.PushTranslatedValue(value_iterator, "accumulator");

2061  }

2063  } else {

2065  // frames, the return value from the callee will become the accumulator.

2067  }
2068  CHECK_EQ(translated_frame->end(), value_iterator);

2072  static_cast<intptr_t>(dispatch_builtin->instruction_start()) +

2078  isolate(), pc, frame_writer.frame()->GetTop());

2080  } else {
2081  output_frame->SetPc(pc);
2082  }
2083

2085  if (V8_EMBEDDED_CONSTANT_POOL_BOOL) {

2089  if (is_topmost) {
2090  Register constant_pool_reg =
2091  UnoptimizedJSFrame::constant_pool_pointer_register();
2092  output_frame->SetRegister(constant_pool_reg.code(), constant_pool_value);
2093  }
2094  }
2095

2097  // and will be materialized by {Runtime_NotifyDeoptimized}. For additional

2100  if (is_topmost) {
2101  intptr_t context_value = static_cast<intptr_t>(Smi::zero().ptr());
2102  Register context_reg = JavaScriptFrame::context_register();
2103  output_frame->SetRegister(context_reg.code(), context_value);
2104  // Set the continuation for the topmost frame.
2105  Tagged<Code> continuation = builtins->code(Builtin::kNotifyDeoptimized);
2106  output_frame->SetContinuation(
2107  static_cast<intptr_t>(continuation->instruction_start()));
2108  }

2111 void Deoptimizer::DoComputeInlinedExtraArguments(
2112  TranslatedFrame* translated_frame, int frame_index) {
2113  // Inlined arguments frame can not be the topmost, nor the bottom most frame.
2114  CHECK(frame_index < output_count_ - 1);
2115  CHECK_GT(frame_index, 0);
2116  CHECK_NULL(output_[frame_index]);
2117 
2118  // During deoptimization we need push the extra arguments of inlined functions
2119  // (arguments with index greater than the formal parameter count).
2120  // For more info, see the design document:
2121  // https://docs.google.com/document/d/150wGaUREaZI6YWqOQFD5l2mWQXaPbbZjcAIJLOFrzMs
2122 
2123  TranslatedFrame::iterator value_iterator = translated_frame->begin();
2124  const int argument_count_without_receiver = translated_frame->height() - 1;
2125  const int formal_parameter_count_without_receiver =
2126  translated_frame->formal_parameter_count() - 1;
2127  SBXCHECK_GE(formal_parameter_count_without_receiver, 0);
2128  const int extra_argument_count =
2129  argument_count_without_receiver - formal_parameter_count_without_receiver;
2130  // The number of pushed arguments is the maximum of the actual argument count
2131  // and the formal parameter count + the receiver.
2132  const int padding =
2133  ArgumentPaddingSlots(std::max(argument_count_without_receiver,
2134  formal_parameter_count_without_receiver) +
2135  1);
2136  const int output_frame_size =
2137  (std::max(0, extra_argument_count) + padding) * kSystemPointerSize;
2138  if (verbose_tracing_enabled()) {
2139  PrintF(trace_scope_->file(),
2140  " translating inlined arguments frame => variable_size=%d\n",
2141  output_frame_size);
2142  }
2143 
2144  // Allocate and store the output frame description.
2145  FrameDescription* output_frame = FrameDescription::Create(
2146  output_frame_size, JSParameterCount(argument_count_without_receiver),
2147  isolate());
2148  // The top address of the frame is computed from the previous frame's top and
2149  // this frame's size.
2150  const intptr_t top_address =
2151  output_[frame_index - 1]->GetTop() - output_frame_size;
2152  output_frame->SetTop(top_address);
2153  // This is not a real frame, we take PC and FP values from the parent frame.
2154  output_frame->SetPc(output_[frame_index - 1]->GetPc());
2155  output_frame->SetFp(output_[frame_index - 1]->GetFp());
2156  output_[frame_index] = output_frame;
2157 
2158  FrameWriter frame_writer(this, output_frame, verbose_trace_scope());
2159 
2160  ReadOnlyRoots roots(isolate());
2161  for (int i = 0; i < padding; ++i) {
2162  frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
2163  }
2164 
2165  if (extra_argument_count > 0) {
2166  // The receiver and arguments with index below the formal parameter
2167  // count are in the fake adaptor frame, because they are used to create the
2168  // arguments object. We should however not push them, since the interpreter
2169  // frame will do that.
2170  value_iterator++; // Skip function.
2171  value_iterator++; // Skip receiver.
2172  for (int i = 0; i < formal_parameter_count_without_receiver; i++)
2173  value_iterator++;
2174  frame_writer.PushStackJSArguments(value_iterator, extra_argument_count);
2175  }

2546 // BuiltinContinuationFrames capture the machine state that is expected as input
2547 // to a builtin, including both input register values and stack parameters. When
2548 // the frame is reactivated (i.e. the frame below it returns), a
2549 // ContinueToBuiltin stub restores the register state from the frame and tail
2550 // calls to the actual target builtin, making it appear that the stub had been
2551 // directly called by the frame above it. The input values to populate the frame
2552 // are taken from the deopt's FrameState.
2553 //
2554 // Frame translation happens in two modes, EAGER and LAZY. In EAGER mode, all of
2555 // the parameters to the Builtin are explicitly specified in the TurboFan
2556 // FrameState node. In LAZY mode, there is always one fewer parameters specified
2557 // in the FrameState than expected by the Builtin. In that case, construction of
2558 // BuiltinContinuationFrame adds the final missing parameter during
2559 // deoptimization, and that parameter is always on the stack and contains the
2560 // value returned from the callee of the call site triggering the LAZY deopt
2561 // (e.g. rax on x64). This requires that continuation Builtins for LAZY deopts
2562 // must have at least one stack parameter.
2563 //
2564 // TO
2565 // | .... |
2566 // +-------------------------+
2567 // | arg padding (arch dept) |<- at most 1*kSystemPointerSize
2568 // +-------------------------+
2569 // | builtin param 0 |<- FrameState input value n becomes
2570 // +-------------------------+
2571 // | ... |
2572 // +-------------------------+
2573 // | builtin param m |<- FrameState input value n+m-1, or in
2574 // +-----needs-alignment-----+ the LAZY case, return LAZY result value
2575 // | ContinueToBuiltin entry |
2576 // +-------------------------+
2577 // | | saved frame (FP) |
2578 // | +=====needs=alignment=====+<- fpreg
2579 // | |constant pool (if ool_cp)|
2580 // v +-------------------------+
2581 // |BUILTIN_CONTINUATION mark|
2582 // +-------------------------+
2583 // | JSFunction (or zero) |<- only if JavaScript builtin
2584 // +-------------------------+
2585 // | frame height above FP |
2586 // +-------------------------+
2587 // | context |<- this non-standard context slot contains
2588 // +-------------------------+ the context, even for non-JS builtins.
2589 // | builtin index |
2590 // +-------------------------+
2591 // | builtin input GPR reg0 |<- populated from deopt FrameState using
2592 // +-------------------------+ the builtin's CallInterfaceDescriptor
2593 // | ... | to map a FrameState's 0..n-1 inputs to
2594 // +-------------------------+ the builtin's n input register params.
2595 // | builtin input GPR regn |
2596 // +-------------------------+
2597 // | reg padding (arch dept) |
2598 // +-----needs--alignment----+
2599 // | res padding (arch dept) |<- only if {is_topmost}; result is pop'd by
2600 // +-------------------------+<- kNotifyDeopt ASM stub and moved to acc
2601 // | result value |<- reg, as ContinueToBuiltin stub expects.
2602 // +-----needs-alignment-----+<- spreg
2603 //
