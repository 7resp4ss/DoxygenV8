135 // Lazy deopt points where the frame state is associated with a call get an
136 // additional parameter for the return result from the call. The return result
137 // is added by the deoptimizer and not explicitly specified in the frame state.
138 // Lazy deopt points which can catch exceptions further get an additional
139 // parameter, namely the exception thrown. The exception is also added by the
140 // deoptimizer.
141 uint8_t DeoptimizerParameterCountFor(ContinuationFrameStateMode mode) {
142  switch (mode) {
143  case ContinuationFrameStateMode::EAGER:
144  return 0;
145  case ContinuationFrameStateMode::LAZY:
146  return 1;
147  case ContinuationFrameStateMode::LAZY_WITH_CATCH:
148  return 2;
149  }
150  UNREACHABLE();
151 }

153 FrameState CreateBuiltinContinuationFrameStateCommon(
154  JSGraph* jsgraph, FrameStateType frame_type, Builtin name, Node* closure,
155  Node* context, Node** parameters, int parameter_count,
156  Node* outer_frame_state,
157  Handle<SharedFunctionInfo> shared = Handle<SharedFunctionInfo>(),
158  const wasm::CanonicalSig* signature = nullptr) {
159  TFGraph* const graph = jsgraph->graph();
160  CommonOperatorBuilder* const common = jsgraph->common();
161 
162  const Operator* op_param =
163  common->StateValues(parameter_count, SparseInputMask::Dense());
164  Node* params_node = graph->NewNode(op_param, parameter_count, parameters);
165 
166  BytecodeOffset bailout_id = Builtins::GetContinuationBytecodeOffset(name);
167 #if V8_ENABLE_WEBASSEMBLY
168  const FrameStateFunctionInfo* state_info =
169  signature ? common->CreateJSToWasmFrameStateFunctionInfo(
170  frame_type, parameter_count, 0, shared, signature)
171  : common->CreateFrameStateFunctionInfo(
172  frame_type, parameter_count, 0, 0, shared, {});
173 #else
174  DCHECK_NULL(signature);
175  const FrameStateFunctionInfo* state_info =
176  common->CreateFrameStateFunctionInfo(frame_type, parameter_count, 0, 0,
177  shared, {});
178 #endif // V8_ENABLE_WEBASSEMBLY
179 
180  const Operator* op = common->FrameState(
181  bailout_id, OutputFrameStateCombine::Ignore(), state_info);
182  return FrameState(graph->NewNode(op, params_node, jsgraph->EmptyStateValues(),
183  jsgraph->EmptyStateValues(), context,
184  closure, outer_frame_state));
185 }

189 FrameState CreateStubBuiltinContinuationFrameState(
190  JSGraph* jsgraph, Builtin name, Node* context, Node* const* parameters,
191  int parameter_count, Node* outer_frame_state,
192  ContinuationFrameStateMode mode, const wasm::CanonicalSig* signature) {
193  Callable callable = Builtins::CallableFor(jsgraph->isolate(), name);
194  CallInterfaceDescriptor descriptor = callable.descriptor();
195 
196  std::vector<Node*> actual_parameters;
197  // Stack parameters first. Depending on {mode}, final parameters are added
198  // by the deoptimizer and aren't explicitly passed in the frame state.
199  int stack_parameter_count =
200  descriptor.GetStackParameterCount() - DeoptimizerParameterCountFor(mode);
201 
202  // Deopt should just cause a crash, no need for parameter count validation.
203  if (name == Builtin::kAbort) {
204  stack_parameter_count = 0;
205  }
206 
207  // Ensure the parameters added by the deoptimizer are passed on the stack.
208  // This check prevents using TFS builtins as continuations while doing the
209  // lazy deopt. Use TFC or TFJ builtin as a lazy deopt continuation which
210  // would pass the result parameter on the stack.
211  DCHECK_GE(stack_parameter_count, 0);
212 
213  // Reserving space in the vector.
214  actual_parameters.reserve(stack_parameter_count +
215  descriptor.GetRegisterParameterCount());
216  for (int i = 0; i < stack_parameter_count; ++i) {
217  actual_parameters.push_back(
218  parameters[descriptor.GetRegisterParameterCount() + i]);
219  }
220  // Register parameters follow, context will be added by instruction selector
221  // during FrameState translation.
222  for (int i = 0; i < descriptor.GetRegisterParameterCount(); ++i) {
223  actual_parameters.push_back(parameters[i]);
224  }
225 
226  FrameStateType frame_state_type = FrameStateType::kBuiltinContinuation;
227 #if V8_ENABLE_WEBASSEMBLY
228  if (name == Builtin::kJSToWasmLazyDeoptContinuation) {
229  CHECK_NOT_NULL(signature);
230  frame_state_type = FrameStateType::kJSToWasmBuiltinContinuation;
231  }
232 #endif // V8_ENABLE_WEBASSEMBLY
233  return CreateBuiltinContinuationFrameStateCommon(
234  jsgraph, frame_state_type, name, jsgraph->UndefinedConstant(), context,
235  actual_parameters.data(), static_cast<int>(actual_parameters.size()),
236  outer_frame_state, Handle<SharedFunctionInfo>(), signature);
237 }

255 FrameState CreateJavaScriptBuiltinContinuationFrameState(
256  JSGraph* jsgraph, SharedFunctionInfoRef shared, Builtin name, Node* target,
257  Node* context, Node* const* stack_parameters, int stack_parameter_count,
258  Node* outer_frame_state, ContinuationFrameStateMode mode) {
259  // Depending on {mode}, final parameters are added by the deoptimizer
260  // and aren't explicitly passed in the frame state.
261  DCHECK_EQ(Builtins::GetStackParameterCount(name),
262  stack_parameter_count + DeoptimizerParameterCountFor(mode));
263 
264  Node* argc = jsgraph->ConstantNoHole(Builtins::GetStackParameterCount(name));
265 
266  // Stack parameters first. They must be first because the receiver is expected
267  // to be the second value in the translation when creating stack crawls
268  // (e.g. Error.stack) of optimized JavaScript frames.
269  std::vector<Node*> actual_parameters;
270  actual_parameters.reserve(stack_parameter_count);
271  for (int i = 0; i < stack_parameter_count; ++i) {
272  actual_parameters.push_back(stack_parameters[i]);
273  }
274 
275  Node* new_target = jsgraph->UndefinedConstant();
276 
277  // Register parameters follow stack parameters. The context will be added by
278  // instruction selector during FrameState translation.
279  DCHECK_EQ(
280  Builtins::CallInterfaceDescriptorFor(name).GetRegisterParameterCount(),
281  V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE_BOOL ? 4 : 3);
282  actual_parameters.push_back(target); // kJavaScriptCallTargetRegister
283  actual_parameters.push_back(new_target); // kJavaScriptCallNewTargetRegister
284  actual_parameters.push_back(argc); // kJavaScriptCallArgCountRegister
285 #ifdef V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE
286  // The dispatch handle isn't used by the continuation builtins.
287  Node* handle = jsgraph->ConstantNoHole(kInvalidDispatchHandle.value());
288  actual_parameters.push_back(handle); // kJavaScriptDispatchHandleRegister
289 #endif
290 
291  return CreateBuiltinContinuationFrameStateCommon(
292  jsgraph,
293  mode == ContinuationFrameStateMode::LAZY_WITH_CATCH
294  ? FrameStateType::kJavaScriptBuiltinContinuationWithCatch
295  : FrameStateType::kJavaScriptBuiltinContinuation,
296  name, target, context, &actual_parameters[0],
297  static_cast<int>(actual_parameters.size()), outer_frame_state,
298  shared.object());
299 }
