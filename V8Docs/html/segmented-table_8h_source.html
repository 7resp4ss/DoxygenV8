15 /**
16  * A thread-safe table with a fixed maximum size split into segments.
17  *
18  * The table provides thread-safe methods to allocate and free of segments and
19  * an inline freelist implementation. Allocation and Freeing of entries is
20  * implemented in subclasses since it depends on if the table is manually
21  * managed or GCed.
22  *
23  * For the purpose of memory management, the table is partitioned into Segments
24  * (for example 64kb memory chunks) that are grouped together in "Spaces". All
25  * segments in a space share a freelist, and so entry allocation and garbage
26  * collection happen on the level of spaces.
27  *
28  * The Entry type defines how the freelist is represented. For that, it must
29  * implement the following methods:
30  * - void MakeFreelistEntry(uint32_t next_entry_index)
31  * - uint32_t GetNextFreelistEntry()
32  */
33 template <typename Entry, size_t size>
34 class V8_EXPORT_PRIVATE SegmentedTable {
35  protected:
36  static constexpr bool kIsWriteProtected = Entry::IsWriteProtected;
37  static constexpr int kEntrySize = sizeof(Entry);
38 
39 #ifdef V8_TARGET_ARCH_64_BIT
40  // On 64 bit, we use a large address space reservation for the table memory.
41  static constexpr bool kUseContiguousMemory = true;
42  static constexpr size_t kReservationSize = size;
43  static constexpr size_t kMaxCapacity = kReservationSize / kEntrySize;
44 #if defined(V8_TARGET_OS_WIN) || defined(V8_HOST_ARCH_PPC64)
45  // On windows the allocation granularity is 64KB and thus we cannot make a
46  // segment smaller than that.
47  // PPC64 can utilize a 64KB page size.
48  static constexpr bool kUseSegmentPool = false;
49  static constexpr size_t kSegmentSize = 64 * KB;
50 #else
51  static constexpr bool kUseSegmentPool = true;
52  static constexpr size_t kSegmentSize = 16 * KB;
53 #endif
54 #else
55  // On 32 bit, segments are individually mapped.
56  static constexpr bool kUseContiguousMemory = false;
57  static constexpr bool kUseSegmentPool = false;
58 #ifdef V8_TARGET_OS_WIN
59  // On windows the allocation granularity is 64KB.
60  static constexpr size_t kSegmentSize = 64 * KB;
61 #else
62  static constexpr size_t kSegmentSize = 16 * KB;
63 #endif
64 #endif // V8_TARGET_ARCH_64_BIT
65 
66  // The sandbox relies on not being able to access any SegmentedTable out of
67  // bounds.
68  static_assert(kUseContiguousMemory || !V8_ENABLE_SANDBOX_BOOL);
69 
70  // For managing the table's backing memory, the table is partitioned into
71  // segments of this size. Segments can then be allocated and freed using the
72  // AllocateAndInitializeSegment() and FreeTableSegment() routines.
73  static constexpr size_t kSegmentPoolSize = 4;
74  static constexpr size_t kEntriesPerSegment = kSegmentSize / kEntrySize;
75  static constexpr size_t kAlignment =
76  kSegmentSize * (kUseSegmentPool ? kSegmentPoolSize : 1);
77  // These need to be identical such that the first normal segment starts at a
78  // normal segment pool offset.
79  static constexpr size_t kNumReadOnlySegments =
80  (kUseSegmentPool ? kSegmentPoolSize : 1);
81 
82  // Struct representing a segment of the table.
83  struct Segment {
84  public:
85  // Initialize a segment given its number.
86  explicit Segment(uint32_t number) : number_(number) {}
87 
88  // Returns the segment starting at the specified offset from the base of the
89  // table.
90  static Segment At(uint32_t offset);
91 
92  // Returns the segment containing the entry at the given index.
93  static Segment Containing(uint32_t entry_index);
94 
95  // The segments of a table are numbered sequentially. This method returns
96  // the number of this segment.
97  uint32_t number() const { return number_; }
98 
99  // Returns the offset of this segment from the table base.
100  uint32_t offset() const { return number_ * kSegmentSize; }
101 
102  // Returns the index of the first entry in this segment.
103  uint32_t first_entry() const { return number_ * kEntriesPerSegment; }
104 
105  // Return the index of the last entry in this segment.
106  uint32_t last_entry() const {
107  return first_entry() + kEntriesPerSegment - 1;
108  }
109 
110  // Segments are ordered by their id/offset.
111  bool operator<(const Segment& other) const {
112  return number_ < other.number_;
113  }
114 
115  private:
116  // A segment is identified by its number, which is its offset from the base
117  // of the table divided by the segment size.
118  const uint32_t number_;
119  };
120 
121  // Struct representing the head of the freelist.
122  //
123  // A segmented table uses simple, singly-linked lists to manage free entries.
124  // Each entry on the freelist contains the 32-bit index of the next entry. The
125  // last entry points to zero.
126  struct FreelistHead {
127  constexpr FreelistHead() : next_(0), length_(0) {}
128  constexpr FreelistHead(uint32_t next, uint32_t length)
129  : next_(next), length_(length) {}
130 
131  // Returns the index of the next entry on the freelist.
132  // If the freelist is empty, this returns zero.
133  uint32_t next() const { return next_; }
134 
135  // Returns the total length of the freelist.
136  uint32_t length() const { return length_; }
137 
138  bool is_empty() const { return length_ == 0; }
139 
140  private:
141  uint32_t next_;
142  uint32_t length_;
143  };
144 
145  // We expect the FreelistHead struct to fit into a single atomic word.
146  // Otherwise, access to it would be slow.
147  static_assert(std::atomic<FreelistHead>::is_always_lock_free);
148 
149  SegmentedTable() = default;
150  SegmentedTable(const SegmentedTable&) = delete;
151  SegmentedTable& operator=(const SegmentedTable&) = delete;

153  // This Iterator also acts as a scope object to temporarily lift any
154  // write-protection (if kIsWriteProtected is true).
155  class WriteIterator {
156  public:
157  explicit WriteIterator(Entry* base, uint32_t index);
158 
159  uint32_t index() const { return index_; }
160  Entry* operator->() {
161  DCHECK(!crossed_segment_);
162  return &base_[index_];
163  }
164  Entry& operator*() {
165  DCHECK(!crossed_segment_);
166  return base_[index_];
167  }
168  WriteIterator& operator++() {
169  index_++;
170 #ifdef DEBUG
171  if (IsAligned(index_, kEntriesPerSegment)) {
172  crossed_segment_ = true;
173  }
174 #endif
175  return *this;
176  }
177  WriteIterator& operator--() {
178  DCHECK_GT(index_, 0);
179 #ifdef DEBUG
180  if (IsAligned(index_, kEntriesPerSegment)) {
181  crossed_segment_ = true;
182  }
183 #endif
184  index_--;
185  return *this;
186  }
187 
188  private:
189  Entry* base_;
190  uint32_t index_;
191  std::conditional_t<kIsWriteProtected, CFIMetadataWriteScope,
192  NopRwxMemoryWriteScope>
193  write_scope_;
194 #ifdef DEBUG
195  bool crossed_segment_ = false;
196 #endif
197  };

227  // Note that you don't need to call this function on segments allocated with

243  // The pointer to the base of the virtual address space backing this table.
244  // All entry accesses happen through this pointer.
245  // It is equivalent to |vas_->base()| and is effectively const after
246  // initialization since the backing memory is never reallocated.
247  Entry* base_ = nullptr;
248 
249  // The virtual address space backing this table.
250  // This is used to manage the underlying OS pages, in particular to allocate
251  // and free the segments that make up the table.
252  VirtualAddressSpace* vas_ = nullptr;
253 
254  // The segment pool contains allocated but not yet initialized pages. It is
255  // only used in combination with VirtualAddressSubSpaces. The pool reduces
256  // lock-contention on startup. Since we split allocations into smaller
257  // segments it takes less time to initialize the freelists.
258  base::Mutex* segment_pool_grow_mutex_ = nullptr;
259  std::array<std::atomic<uint32_t>, kSegmentPoolSize> segment_pool_ = {};
260  // Used during set-up of read only segments.
261  uint32_t read_only_segments_used_ = 0;
262 };
