92 // StatsCounters are dynamically created values which can be tracked in the
93 // StatsTable. They are designed to be lightweight to create and easy to use.
94 //
95 // Internally, a counter represents a value in a row of a StatsTable.
96 // The row has a 32bit value for each process/thread in the table and also
97 // a name (stored in the table metadata). Since the storage location can be
98 // thread-specific, this class cannot be shared across threads.
99 // This class is thread-safe.
100 class StatsCounter {
101  public:
102  const char* name() const { return name_; }
103 
104  void Set(int value) { GetPtr()->store(value, std::memory_order_relaxed); }
105  int Get() { return GetPtr()->load(); }
106 
107  void Increment(int value = 1) {
108  GetPtr()->fetch_add(value, std::memory_order_relaxed);
109  }
110 
111  void Decrement(int value = 1) {
112  GetPtr()->fetch_sub(value, std::memory_order_relaxed);
113  }
114 
115  // Returns true if this counter is enabled (a lookup function was provided and
116  // it returned a non-null pointer).
117  V8_EXPORT_PRIVATE bool Enabled();
118 
119  // Get the internal pointer to the counter. This is used
120  // by the code generator to emit code that manipulates a
121  // given counter without calling the runtime system.
122  std::atomic<int>* GetInternalPointer() { return GetPtr(); }
123 
124  private:
125  friend class Counters;
126  friend class CountersInitializer;
127  friend class StatsCounterResetter;
128 
129  void Initialize(const char* name, Counters* counters) {
130  DCHECK_NULL(counters_);
131  DCHECK_NOT_NULL(counters);
132  // Counter names always start with "c:V8.".
133  DCHECK_EQ(0, memcmp(name, "c:V8.", 5));
134  counters_ = counters;
135  name_ = name;
136  }
137 
138  V8_NOINLINE V8_EXPORT_PRIVATE std::atomic<int>* SetupPtrFromStatsTable();
139 
140  // Reset the cached internal pointer.
141  void Reset() { ptr_.store(nullptr, std::memory_order_relaxed); }
142 
143  // Returns the cached address of this counter location.
144  std::atomic<int>* GetPtr() {
145  auto* ptr = ptr_.load(std::memory_order_acquire);
146  if (V8_LIKELY(ptr)) return ptr;
147  return SetupPtrFromStatsTable();
148  }
149 
150  Counters* counters_ = nullptr;
151  const char* name_ = nullptr;
152  // A pointer to an atomic, set atomically in {GetPtr}.
153  std::atomic<std::atomic<int>*> ptr_{nullptr};
154 };

316 // A histogram timer that can aggregate events within a larger scope.
317 //
318 // Intended use of this timer is to have an outer (aggregating) and an inner
319 // (to be aggregated) scope, where the inner scope measure the time of events,
320 // and all those inner scope measurements will be summed up by the outer scope.
321 // An example use might be to aggregate the time spent in lazy compilation
322 // while running a script.
323 //
324 // Helpers:
325 // - AggregatingHistogramTimerScope, the "outer" scope within which
326 // times will be summed up.
327 // - AggregatedHistogramTimerScope, the "inner" scope which defines the
328 // events to be timed.
329 class AggregatableHistogramTimer : public Histogram {
330  public:
331  // Start/stop the "outer" scope.
332  void Start() { time_ = base::TimeDelta(); }
333  void Stop() {
334  if (time_ != base::TimeDelta()) {
335  // Only add non-zero samples, since zero samples represent situations
336  // where there were no aggregated samples added.
337  AddSample(static_cast<int>(time_.InMicroseconds()));
338  }
339  }
340 
341  // Add a time value ("inner" scope).
342  void Add(base::TimeDelta other) { time_ += other; }
343 
344  private:
345  friend class Counters;
346 
347  AggregatableHistogramTimer() = default;
348  AggregatableHistogramTimer(const AggregatableHistogramTimer&) = delete;
349  AggregatableHistogramTimer& operator=(const AggregatableHistogramTimer&) =
350  delete;
351 
352  base::TimeDelta time_;
353 };

385 // AggretatedMemoryHistogram collects (time, value) sample pairs and turns
386 // them into time-uniform samples for the backing histogram, such that the
387 // backing histogram receives one sample every T ms, where the T is controlled
388 // by the v8_flags.histogram_interval.
389 //
390 // More formally: let F be a real-valued function that maps time to sample
391 // values. We define F as a linear interpolation between adjacent samples. For
392 // each time interval [x; x + T) the backing histogram gets one sample value
393 // that is the average of F(t) in the interval.
394 template <typename Histogram>
395 class AggregatedMemoryHistogram {
396  public:
397  // Note: public for testing purposes only.
398  explicit AggregatedMemoryHistogram(Histogram* backing_histogram)
399  : AggregatedMemoryHistogram() {
400  backing_histogram_ = backing_histogram;
401  }
402 
403  // Invariants that hold before and after AddSample if
404  // is_initialized_ is true:
405  //
406  // 1) For we processed samples that came in before start_ms_ and sent the
407  // corresponding aggregated samples to backing histogram.
408  // 2) (last_ms_, last_value_) is the last received sample.
409  // 3) last_ms_ < start_ms_ + v8_flags.histogram_interval.
410  // 4) aggregate_value_ is the average of the function that is constructed by
411  // linearly interpolating samples received between start_ms_ and last_ms_.
412  void AddSample(double current_ms, double current_value);
413 
414  private:
415  friend class Counters;
416 
417  AggregatedMemoryHistogram()
418  : is_initialized_(false),
419  start_ms_(0.0),
420  last_ms_(0.0),
421  aggregate_value_(0.0),
422  last_value_(0.0),
423  backing_histogram_(nullptr) {}
424  double Aggregate(double current_ms, double current_value);
425 
426  bool is_initialized_;
427  double start_ms_;
428  double last_ms_;
429  double aggregate_value_;
430  double last_value_;
431  Histogram* backing_histogram_;
432 };

434 template <typename Histogram>
435 void AggregatedMemoryHistogram<Histogram>::AddSample(double current_ms,
436  double current_value) {
437  if (!is_initialized_) {
438  aggregate_value_ = current_value;
439  start_ms_ = current_ms;
440  last_value_ = current_value;
441  last_ms_ = current_ms;
442  is_initialized_ = true;
443  } else {
444  const double kEpsilon = 1e-6;
445  const int kMaxSamples = 1000;
446  if (current_ms < last_ms_ + kEpsilon) {
447  // Two samples have the same time, remember the last one.
448  last_value_ = current_value;
449  } else {
450  double sample_interval_ms = v8_flags.histogram_interval;
451  double end_ms = start_ms_ + sample_interval_ms;
452  if (end_ms <= current_ms + kEpsilon) {
453  // Linearly interpolate between the last_ms_ and the current_ms.
454  double slope = (current_value - last_value_) / (current_ms - last_ms_);
455  int i;
456  // Send aggregated samples to the backing histogram from the start_ms
457  // to the current_ms.
458  for (i = 0; i < kMaxSamples && end_ms <= current_ms + kEpsilon; i++) {
459  double end_value = last_value_ + (end_ms - last_ms_) * slope;
460  double sample_value;
461  if (i == 0) {
462  // Take aggregate_value_ into account.
463  sample_value = Aggregate(end_ms, end_value);
464  } else {
465  // There is no aggregate_value_ for i > 0.
466  sample_value = (last_value_ + end_value) / 2;
467  }
468  backing_histogram_->AddSample(static_cast<int>(sample_value + 0.5));
469  last_value_ = end_value;
470  last_ms_ = end_ms;
471  end_ms += sample_interval_ms;
472  }
473  if (i == kMaxSamples) {
474  // We hit the sample limit, ignore the remaining samples.
475  aggregate_value_ = current_value;
476  start_ms_ = current_ms;
477  } else {
478  aggregate_value_ = last_value_;
479  start_ms_ = last_ms_;
480  }
481  }
482  aggregate_value_ = current_ms > start_ms_ + kEpsilon
483  ? Aggregate(current_ms, current_value)
484  : aggregate_value_;
485  last_value_ = current_value;
486  last_ms_ = current_ms;
487  }
488  }
489 }
490 
491 template <typename Histogram>
492 double AggregatedMemoryHistogram<Histogram>::Aggregate(double current_ms,
493  double current_value) {
494  double interval_ms = current_ms - start_ms_;
495  double value = (current_value + last_value_) / 2;
496  // The aggregate_value_ is the average for [start_ms_; last_ms_].
497  // The value is the average for [last_ms_; current_ms].
498  // Return the weighted average of the aggregate_value_ and the value.
499  return aggregate_value_ * ((last_ms_ - start_ms_) / interval_ms) +
500  value * ((current_ms - last_ms_) / interval_ms);
501 }
