17 // We'll use a simple encoding scheme to record the source positions.
18 // Conceptually, each position consists of:
19 // - code_offset: An integer index into the BytecodeArray or code.
20 // - source_position: An integer index into the source string.
21 // - position type: Each position is either a statement or an expression.
22 //
23 // The basic idea for the encoding is to use a variable-length integer coding,
24 // where each byte contains 7 bits of payload data, and 1 'more' bit that
25 // determines whether additional bytes follow. Additionally:
26 // - we record the difference from the previous position,
27 // - we just stuff one bit for the type into the code offset,
28 // - we write least-significant bits first,
29 // - we use zig-zag encoding to encode both positive and negative numbers.
30 
31 namespace {

33 // Each byte is encoded as MoreBit | ValueBits.
34 using MoreBit = base::BitField8<bool, 7, 1>;
35 using ValueBits = base::BitField8<unsigned, 0, 7>;

37 // Helper: Add the offsets from 'other' to 'value'. Also set is_statement.
38 void AddAndSetEntry(PositionTableEntry* value,
39  const PositionTableEntry& other) {
40  value->code_offset += other.code_offset;
41  DCHECK_IMPLIES(value->code_offset != kFunctionEntryBytecodeOffset,
42  value->code_offset >= 0);
43  value->source_position += other.source_position;
44  DCHECK_LE(0, value->source_position);
45  value->is_statement = other.is_statement;
46  value->is_breakable = other.is_breakable;
47 }

49 // Helper: Subtract the offsets from 'other' from 'value'.
50 void SubtractFromEntry(PositionTableEntry* value,
51  const PositionTableEntry& other) {
52  value->code_offset -= other.code_offset;
53  value->source_position -= other.source_position;
54 }

56 template <typename T>
57 inline void EncodeUInt(ZoneVector<uint8_t>* bytes, T value) {
58  bool more;
59  do {
60  more = value > ValueBits::kMax;
61  uint8_t current =
62  MoreBit::encode(more) | ValueBits::encode(value & ValueBits::kMask);
63  bytes->push_back(current);
64  value >>= ValueBits::kSize;
65  } while (more);
66 }

68 // Helper: Encode an integer.
69 template <typename T>
70 inline void EncodeInt(ZoneVector<uint8_t>* bytes, T value) {
71  using unsigned_type = std::make_unsigned_t<T>;
72  // Zig-zag encoding.
73  static constexpr int kShift = sizeof(T) * kBitsPerByte - 1;
74  value = ((static_cast<unsigned_type>(value) << 1) ^ (value >> kShift));
75  DCHECK_GE(value, 0);
76  unsigned_type encoded = static_cast<unsigned_type>(value);
77 
78  EncodeUInt(bytes, encoded);
79 }

81 // Encode a PositionTableEntry.
82 void EncodeEntry(ZoneVector<uint8_t>* bytes, const PositionTableEntry& entry) {
83  // We only accept ascending code offsets.
84  DCHECK_LE(0, entry.code_offset);
85  // All but the first entry must be *strictly* ascending (no two entries for
86  // the same position).
87  // TODO(11496): This DCHECK fails tests.
88  // DCHECK_IMPLIES(!bytes->empty(), entry.code_offset > 0);
89  // The least significant bit is used to encode is_breakable.
90  // The second least significant bit is used to encode is_statement.
91  uint32_t encoded_entry = (entry.code_offset << 2) |
92  (entry.is_breakable << 1) | (entry.is_statement);
93 
94  EncodeUInt(bytes, encoded_entry);
95  EncodeInt(bytes, entry.source_position);
96 }

98 template <typename T>
99 T DecodeUInt(base::Vector<const uint8_t> bytes, int* index) {
100  uint8_t current;
101  int shift = 0;
102  T decoded = 0;
103  bool more;
104  do {
105  current = bytes[(*index)++];
106  decoded |= static_cast<std::make_unsigned_t<T>>(ValueBits::decode(current))
107  << shift;
108  more = MoreBit::decode(current);
109  shift += ValueBits::kSize;
110  } while (more);
111  DCHECK_GE(decoded, 0);
112  return decoded;
113 }

115 // Helper: Decode an integer.
116 template <typename T>
117 T DecodeInt(base::Vector<const uint8_t> bytes, int* index) {
118  using unsigned_type = std::make_unsigned_t<T>;
119 
120  unsigned_type zigzag_value = DecodeUInt<unsigned_type>(bytes, index);
121  T result = static_cast<T>((zigzag_value >> 1) ^ (-(zigzag_value & 1)));
122  return result;
123 }
124 
125 void DecodeEntry(base::Vector<const uint8_t> bytes, int* index,
126  PositionTableEntry* entry) {
127  // The least significant bit is used to encode is_breakable.
128  // The second least significant bit is used to encode is_statement.
129  int tmp = DecodeUInt<int>(bytes, index);
130  entry->code_offset = tmp >> 2;
131  entry->is_breakable = (tmp & 0b10) >> 1;
132  entry->is_statement = tmp & 1;
133  entry->source_position = DecodeInt<int64_t>(bytes, index);
134 }

141 #ifdef ENABLE_SLOW_DCHECKS
142 void CheckTableEquals(const ZoneVector<PositionTableEntry>& raw_entries,
143  SourcePositionTableIterator* encoded) {
144  // Brute force testing: Record all positions and decode
145  // the entire table to verify they are identical.
146  auto raw = raw_entries.begin();
147  for (; !encoded->done(); encoded->Advance(), raw++) {
148  DCHECK(raw != raw_entries.end());
149  DCHECK_EQ(encoded->code_offset(), raw->code_offset);
150  DCHECK_EQ(encoded->source_position().raw(), raw->source_position);
151  DCHECK_EQ(encoded->is_statement(), raw->is_statement);
152  }
153  DCHECK(raw == raw_entries.end());
154 }
155 #endif

180  const PositionTableEntry& entry) {
181  PositionTableEntry tmp(entry);

283 void SourcePositionTableIterator::Advance() {
284  base::Vector<const uint8_t> bytes =
285  table_.is_null() ? raw_table_ : VectorFromByteArray(*table_);
286  DCHECK(!done());
287  DCHECK(index_ >= 0 && index_ <= bytes.length());
288  bool filter_satisfied = false;
289  while (!done() && !filter_satisfied) {
290  if (index_ >= bytes.length()) {
291  index_ = kDone;
292  } else {
293  PositionTableEntry tmp;
294  DecodeEntry(bytes, &index_, &tmp);
295  AddAndSetEntry(&current_, tmp);
296  SourcePosition p = source_position();
297  filter_satisfied =
298  (iteration_filter_ == kAll) ||
299  (iteration_filter_ == kJavaScriptOnly && p.IsJavaScript()) ||
300  (iteration_filter_ == kExternalOnly && p.IsExternal());
301  }
302  }
