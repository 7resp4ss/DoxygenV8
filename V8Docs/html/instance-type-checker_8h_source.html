60 // Maps for primitive objects and a select few JS objects are allocated in r/o
61 // space. All JS_RECEIVER maps must come after primitive object maps, i.e. they
62 // have a compressed address above the last primitive object map root. If we
63 // have a receiver and need to distinguish whether it is either a primitive
64 // object or a JS receiver, it suffices to check if its map is allocated above
65 // the following limit address.
66 constexpr Tagged_t kNonJsReceiverMapLimit =
67  StaticReadOnlyRootsPointerTable[static_cast<size_t>(
68  RootIndex::kFirstJSReceiverMapRoot)] &
69  ~0xFFF;
70 
71 // Maps for strings allocated as the first maps in r/o space, so their lower
72 // bound is zero.
73 constexpr Tagged_t kStringMapLowerBound = 0;
74 // If we have a receiver and need to distinguish whether it is a string or not,
75 // it suffices to check whether it is less-than-equal to the following value.
76 constexpr Tagged_t kStringMapUpperBound =
77  StaticReadOnlyRoot::kThinOneByteStringMap;
78 
79 #define ASSERT_IS_LAST_STRING_MAP(instance_type, size, name, Name) \
80  static_assert(StaticReadOnlyRoot::k##Name##Map <= kStringMapUpperBound);
81 STRING_TYPE_LIST(ASSERT_IS_LAST_STRING_MAP)
82 #undef ASSERT_IS_LAST_STRING_MAP
83 
84 // For performance, the limit is chosen to be encodable as an Arm64
85 // constant. See Assembler::IsImmAddSub in assembler-arm64.cc.
86 static constexpr Tagged_t kMaxFastImmediateConstantArm64 = 0xfff;
87 // If this assert fails, then you have perturbed the allocation pattern in
88 // Heap::CreateReadOnlyHeapObjects. Currently this limit is ensured to exist by
89 // allocating the first JSReceiver map in RO space a sufficiently large distance
90 // away from the last non-JSReceiver map.
91 static_assert(kNonJsReceiverMapLimit != 0 &&
92  is_uint12(kNonJsReceiverMapLimit >> 12) &&
93  ((kNonJsReceiverMapLimit & kMaxFastImmediateConstantArm64) == 0));
