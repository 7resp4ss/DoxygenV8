109 #ifdef V8_EXTERNAL_CODE_SPACE
110 // Compression scheme used for fields containing InstructionStream objects
111 // (namely for the Code::code field).
112 // Unlike the V8HeapCompressionScheme this one allows the cage to cross 4GB
113 // boundary at a price of making decompression slightly more complex.
114 // The former outweighs the latter because it gives us more flexibility in
115 // allocating the code range closer to .text section in the process address
116 // space. At the same time decompression of the external code field happens
117 // relatively rarely during GC.
118 // The base can be any value such that [base, base + 4GB) contains the whole
119 // code range.
120 //
121 // This scheme works as follows:
122 // --|----------{---------|------}--------------|--
123 // 4GB | 4GB | 4GB
124 // +-- code range --+
125 // |
126 // cage base
127 //
128 // * Cage base value is OS page aligned for simplicity (although it's not
129 // strictly necessary).
130 // * Code range size is smaller than or equal to 4GB.
131 // * Compression is just a truncation to 32-bits value.
132 // * Decompression of a pointer:
133 // - if "compressed" cage base is <= than compressed value then one just
134 // needs to OR the upper 32-bits of the case base to get the decompressed
135 // value.
136 // - if compressed value is smaller than "compressed" cage base then ORing
137 // the upper 32-bits of the cage base is not enough because the resulting
138 // value will be off by 4GB, which has to be added to the result.
139 // - note that decompression doesn't modify the lower 32-bits of the value.
140 // * Decompression of Smi values is made a no-op for simplicity given that
141 // on the hot paths of decompressing the Code pointers it's already known
142 // that the value is not a Smi.
143 //
144 class ExternalCodeCompressionScheme {
145  public:
146  V8_INLINE static Address PrepareCageBaseAddress(Address on_heap_addr);
147 
148  // Note that this compression scheme doesn't allow reconstruction of the cage
149  // base value from any arbitrary value, thus the cage base has to be passed
150  // explicitly to the decompression functions.
151  static Address GetPtrComprCageBaseAddress(Address on_heap_addr) = delete;
152 
153  V8_INLINE static Address GetPtrComprCageBaseAddress(
154  PtrComprCageBase cage_base);
155 
156  // Compresses full-pointer representation of a tagged value to on-heap
157  // representation.
158  // Must only be used for compressing object pointers (incl. SMI) since this
159  // function assumes pointers to be inside the pointer compression cage.
160  V8_INLINE static Tagged_t CompressObject(Address tagged);
161  // Compress anything that does not follow the above requirements (e.g. a maybe
162  // object, or a marker bit pattern).
163  V8_INLINE static constexpr Tagged_t CompressAny(Address tagged);
164 
165  // Decompresses any tagged value, preserving both weak- and smi- tags.
166  V8_INLINE static Address DecompressTagged(Tagged_t raw_value);
167 
168  // Process-wide cage base value used for decompression.
169  V8_INLINE static void InitBase(Address base);
170  V8_INLINE static Address base();

197 template <typename V>
198 static inline V ReadMaybeUnalignedValue(Address p) {
199  // Pointer compression causes types larger than kTaggedSize to be unaligned.
200 #ifdef V8_COMPRESS_POINTERS
201  constexpr bool v8_pointer_compression_unaligned = sizeof(V) > kTaggedSize;
202 #else
203  constexpr bool v8_pointer_compression_unaligned = false;
204 #endif
205  // Bug(v8:8875) Double fields may be unaligned.
206  constexpr bool unaligned_double_field =
207  std::is_same_v<V, double> && kDoubleSize > kTaggedSize;
208  if (unaligned_double_field || v8_pointer_compression_unaligned) {
209  return base::ReadUnalignedValue<V>(p);
210  } else {
211  return base::Memory<V>(p);
212  }
213 }

215 template <typename V>
216 static inline void WriteMaybeUnalignedValue(Address p, V value) {
217  // Pointer compression causes types larger than kTaggedSize to be unaligned.
218 #ifdef V8_COMPRESS_POINTERS
219  constexpr bool v8_pointer_compression_unaligned = sizeof(V) > kTaggedSize;
220 #else
221  constexpr bool v8_pointer_compression_unaligned = false;
222 #endif
223  // Bug(v8:8875) Double fields may be unaligned.
224  constexpr bool unaligned_double_field =
225  std::is_same_v<V, double> && kDoubleSize > kTaggedSize;
226  if (unaligned_double_field || v8_pointer_compression_unaligned) {
227  base::WriteUnalignedValue<V>(p, value);
228  } else {
229  base::Memory<V>(p) = value;
230  }
231 }
