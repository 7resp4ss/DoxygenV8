33 void SameThreadEnabledCheckingPolicyBase::CheckPointerImpl(
34  const void* ptr, bool points_to_payload, bool check_off_heap_assignments) {
35  // `ptr` must not reside on stack.
36  DCHECK(!IsOnStack(ptr));
37 #if defined(CPPGC_CAGED_HEAP)
38  // `ptr` must reside in the cage.
39  DCHECK(CagedHeapBase::IsWithinCage(ptr));
40 #endif // defined(CPPGC_CAGED_HEAP)
41  // Check for the most commonly used wrong sentinel value (-1).
42  DCHECK_NE(reinterpret_cast<void*>(-1), ptr);
43  auto* base_page = BasePage::FromPayload(ptr);
44  // Large objects do not support mixins. This also means that `base_page` is
45  // valid for large objects.
46  DCHECK_IMPLIES(base_page->is_large(), points_to_payload);
47 
48  // References cannot change their heap association which means that state is
49  // immutable once it is set.
50  bool is_on_heap = true;
51  if (!heap_) {
52  heap_ = &base_page->heap();
53  if (!heap_->page_backend()->Lookup(reinterpret_cast<Address>(this))) {
54  // If `this` is not contained within the heap of `ptr`, we must deal with
55  // an on-stack or off-heap reference. For both cases there should be no
56  // heap registered.
57  is_on_heap = false;
58  CHECK(!HeapRegistry::TryFromManagedPointer(this));
59  }
60  }
61 
62  // Member references should never mix heaps.
63  DCHECK_EQ(heap_, &base_page->heap());
64 
65  DCHECK(heap_->CurrentThreadIsHeapThread());
66 
67  // Header checks.
68  const HeapObjectHeader* header = nullptr;
69  if (points_to_payload) {
70  // Unmask the alignment bits for the tagged pointer.
71  const void* untagged = reinterpret_cast<const void*>(
72  reinterpret_cast<uintptr_t>(ptr) & ~kAllocationMask);
73  header = &HeapObjectHeader::FromObject(untagged);
74  DCHECK_EQ(
75  header,
76  &base_page->ObjectHeaderFromInnerAddress<AccessMode::kAtomic>(ptr));
77  } else {
78  // Mixin case. Access the ObjectStartBitmap atomically since sweeping can be
79  // in progress.
80  header =
81  &base_page->ObjectHeaderFromInnerAddress<AccessMode::kAtomic>(ptr);
82  DCHECK_LE(header->ObjectStart(), ptr);
83  DCHECK_GT(header->ObjectEnd<AccessMode::kAtomic>(), ptr);
84  }
85  if (header) {
86  DCHECK(!header->IsFree());
87  }
88 
89 #ifdef CPPGC_VERIFY_HEAP
90  if (check_off_heap_assignments || is_on_heap) {
91  if (heap_->prefinalizer_handler()->IsInvokingPreFinalizers()) {
92  // Slot can be in a large object.
93  const auto* slot_page = BasePage::FromInnerAddress(heap_, this);
94  // Off-heap slots (from other heaps or on-stack) are considered live.
95  bool slot_is_live =
96  !slot_page ||
97  slot_page->ObjectHeaderFromInnerAddress(this).IsMarked();
98  // During prefinalizers invocation, check that if the slot is live then
99  // |ptr| refers to a live object.
100  DCHECK_IMPLIES(slot_is_live, header->IsMarked());
101  USE(slot_is_live);
102  }
103  }
104 #else
105  USE(is_on_heap);
106 #endif // CPPGC_VERIFY_HEAP
107 }
