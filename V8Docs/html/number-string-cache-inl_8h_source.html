37 // static
38 Handle<Object> SmiStringCache::Get(Isolate* isolate, InternalIndex entry,
39  Tagged<Smi> number) {
40  DisallowGarbageCollection no_gc;
41  Factory* factory = isolate->factory();
42  Tagged<SmiStringCache> cache = *factory->smi_string_cache();
43  uint32_t entry_index = entry.as_uint32() * kEntrySize;
44 
45  Tagged<Object> key = cache->get(entry_index + kEntryKeyIndex);
46  const bool match = (key == number);
47 
48  if (V8_UNLIKELY(v8_flags.native_code_counters)) {
49  Counters* counters = isolate->counters();
50  counters->number_string_cache_smi_probes()->Increment();
51  if (!match) {
52  counters->number_string_cache_smi_misses()->Increment();
53  }
54  }
55 
56  if (match) {
57  auto string = Cast<String>(cache->get(entry_index + kEntryValueIndex));
58  return handle(string, isolate);
59  }
60  return factory->undefined_value();
61 }

63 // static
64 void SmiStringCache::Set(Isolate* isolate, InternalIndex entry,
65  Tagged<Smi> number, DirectHandle<String> string) {
66  // Smi::zero() must never be put into this cache, one must read the value
67  // from preallocated_number_string_table instead.
68  DCHECK_NE(number, kEmptySentinel);
69 
70  DirectHandle<SmiStringCache> cache = isolate->factory()->smi_string_cache();
71  DCHECK_EQ(entry, cache->GetEntryFor(number));
72  uint32_t entry_index = entry.as_uint32() * kEntrySize;
73 
74  // Switch to full-size mode if the entry is already occupied.
75  if (cache->capacity() == kInitialSize &&
76  cache->get(entry_index + kEntryKeyIndex) != kEmptySentinel &&
77  !isolate->MemorySaverModeEnabled()) {
78  // Allocate full-size cache, recompute the entry and proceed with adding
79  // the entry.
80  uint32_t full_size =
81  base::bits::RoundUpToPowerOfTwo32(v8_flags.smi_string_cache_size);
82  CHECK_LT(kInitialSize, full_size);
83  CHECK_LE(full_size, kMaxCapacity);
84  // TODO(ishell): consider copying entries from previous table.
85  cache = SmiStringCache::New(isolate, full_size);
86  isolate->heap()->SetSmiStringCache(*cache);
87 
88  entry = cache->GetEntryFor(number);
89  entry_index = entry.as_uint32() * kEntrySize;
90  }
91  DisallowGarbageCollection no_gc;
92  cache->set(entry_index + kEntryKeyIndex, number);
93  cache->set(entry_index + kEntryValueIndex, *string);
94 }

158 void DoubleStringCache::Set(Isolate* isolate, InternalIndex entry_index,
159  uint64_t number_bits, DirectHandle<String> string) {
160  DirectHandle<DoubleStringCache> cache =
161  isolate->factory()->double_string_cache();
162  DCHECK_EQ(entry_index, cache->GetEntryFor(number_bits));
163 
164  auto* entry = &cache->entries()[entry_index.as_uint32()];
165 
166  // Switch to full-size mode if the entry is already occupied.
167  if (cache->capacity() == kInitialSize &&
168  entry->value_.load() != kEmptySentinel &&
169  !isolate->MemorySaverModeEnabled()) {
170  // Allocate full-size cache, recompute the entry and proceed with adding
171  // the entry.
172  uint32_t full_size =
173  base::bits::RoundUpToPowerOfTwo32(v8_flags.double_string_cache_size);
174  CHECK_LT(kInitialSize, full_size);
175  CHECK_LE(full_size, kMaxCapacity);
176  cache = DoubleStringCache::New(isolate, full_size);
177  isolate->heap()->SetDoubleStringCache(*cache);
178 
179  entry_index = cache->GetEntryFor(number_bits);
180  entry = &cache->entries()[entry_index.as_uint32()];
181  }
182  DisallowGarbageCollection no_gc;
183  entry->key_.set_value_as_bits(number_bits);
184  entry->value_.store(&**cache, *string);
185 }

189 DirectHandle<DoubleStringCache> DoubleStringCache::New(IsolateT* isolate,
190  int capacity) {
191  DCHECK_GE(capacity, kInitialSize);
192  DCHECK_LE(capacity, kMaxCapacity);
193 
194  Tagged<DoubleStringCache> xs =
195  UncheckedCast<DoubleStringCache>(isolate->factory()->AllocateRawArray(
196  SizeFor(capacity), AllocationType::kOld));
197 
198  ReadOnlyRoots roots{isolate};
199  xs->set_map_after_allocation(isolate, roots.double_string_cache_map(),
200  SKIP_WRITE_BARRIER);
201  xs->capacity_ = capacity;
202 #if TAGGED_SIZE_8_BYTES
203  xs->optional_padding_ = 0;
204 #endif
205  xs->Clear();
206 
207  return handle(xs, isolate);
208 }
