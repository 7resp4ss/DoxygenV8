14 // Defines the list of valid indirect pointer tags.
15 //
16 // When accessing a trusted/indirect pointer, an IndirectPointerTag must be
17 // provided which indicates the expected instance type of the pointed-to
18 // object. When the sandbox is enabled, this tag is used to ensure type-safe
19 // access to objects referenced via trusted pointers: if the provided tag
20 // doesn't match the tag of the object in the trusted pointer table, an
21 // inaccessible pointer will be returned.
22 //
23 // We use the shifted instance type as tag and an AND-based type-checking
24 // mechanism in the TrustedPointerTable, similar to the one used by the
25 // ExternalPointerTable: the entry in the table is ORed with the tag and then
26 // ANDed with the inverse of the tag upon access. This has the benefit that the
27 // type check and the removal of the marking bit can be folded into a single
28 // bitwise operations. However, it is not technically guaranteed that simply
29 // using the instance type as tag works for this scheme as the bits of one
30 // instance type may happen to be a superset of those of another instance type,
31 // thereby causing the type check to incorrectly pass. As such, the chance of
32 // getting "incompabitle" tags increases when adding more tags here so we may
33 // at some point want to consider manually assigning tag values that are
34 // guaranteed to work (similar for how we do it for ExternalPointerTags).
35 
36 constexpr int kIndirectPointerTagShift = 48;
37 constexpr uint64_t kIndirectPointerTagMask = 0x7fff'0000'0000'0000;
38 constexpr uint64_t kTrustedPointerTableMarkBit = 0x8000'0000'0000'0000;
39 // We use a reserved bit for the free entry tag so that the
40 // kUnknownIndirectPointerTag cannot untag free entries. Due to that, not all
41 // tags in the kAllTagsForAndBasedTypeChecking are usable here (which is
42 // ensured by static asserts below, see VALIDATE_INDIRECT_POINTER_TAG).
43 // However, in practice this would probably be fine since the payload is a
44 // table index, and so would likely always crash when treated as a pointer. As
45 // such, if there is ever need for more tags, this can be reconsidered.
46 // Note that we use a bit in the 2nd most significant byte here due to top byte
47 // ignore (TBI), which allows dereferencing pointers even if bits in the most
48 // significant byte are set.
49 constexpr uint64_t kTrustedPointerTableFreeEntryBit = 0x0080'0000'0000'0000;
50 constexpr uint64_t kIndirectPointerTagMaskWithoutFreeEntryBit =
51  0x7f7f'0000'0000'0000;

108 // TODO(saelo): consider renaming this to something like TypeTag or
109 // InstanceTypeTag since that better captures what this represents.
110 enum IndirectPointerTag : uint64_t {
111  // The null tag. Usually used to express the lack of a valid tag, for example
112  // in non-sandbox builds.
113  kIndirectPointerNullTag = 0,
114 
115  // This tag can be used when an indirect pointer field can legitimately refer
116  // to objects of different types.
117  // NOTE: this tag effectively disables the built-in type-checking mechanism.
118  // As such, in virtually all cases the caller needs to perform runtime-type
119  // checks (i.e. IsXyzObject(obj))` afterwards which need to be able to
120  // correctly handle unexpected types. The last point is worth stressing
121  // further. As an example, the following code is NOT correct:
122  //
123  // auto obj = LoadTrustedPointerField<kUnknownIndirectPointerTag>(...);
124  // if (IsFoo(obj)) {
125  // Cast<Foo>(obj)->foo();
126  // } else if (IsBar(obj)) {
127  // Cast<Bar>(obj)->bar();
128  // } else {
129  // // Potential type confusion here!
130  // Cast<Baz>(obj)->baz();
131  // }
132  //
133  // This is because an attacker can swap trusted pointers and thereby cause an
134  // object of a different/unexpected type to be returned. Instead, in this
135  // case a CHECK can for example be used to make the code correct:
136  //
137  // // ...
138  // } else {
139  // // Must be a Baz object
140  // CHECK(IsBaz(obj));
141  // Cast<Baz>(obj)->baz();
142  // }
143  //
144  kUnknownIndirectPointerTag = kIndirectPointerTagMaskWithoutFreeEntryBit,
145 
146  // Tag used internally by the trusted pointer table to mark free entries.
147  // See also the comment above kTrustedPointerTableFreeEntryBit for why this
148  // uses a dedicated bit.
149  kFreeTrustedPointerTableEntryTag = kTrustedPointerTableFreeEntryBit,
150 
151 // "Regular" tags. One per supported instance type.
152 #define INDIRECT_POINTER_TAG_ENUM_DECL(name, tag_id) name = MAKE_TAG(tag_id),
153  INDIRECT_POINTER_TAG_LIST(INDIRECT_POINTER_TAG_ENUM_DECL)
154 #undef INDIRECT_POINTER_TAG_ENUM_DECL
155 };

159 #define VALIDATE_INDIRECT_POINTER_TAG(name, tag_id) \
160  static_assert((name & kIndirectPointerTagMask) == name); \
161  static_assert((name & kIndirectPointerTagMaskWithoutFreeEntryBit) == name);
162 INDIRECT_POINTER_TAG_LIST(VALIDATE_INDIRECT_POINTER_TAG)
163 #undef VALIDATE_INDIRECT_POINTER_TAG
164 static_assert((kFreeTrustedPointerTableEntryTag & kIndirectPointerTagMask) ==
165  kFreeTrustedPointerTableEntryTag);
166 static_assert((kFreeTrustedPointerTableEntryTag &
167  kIndirectPointerTagMaskWithoutFreeEntryBit) == 0);

169 // True if the external pointer must be accessed from the shared isolate's
170 // external pointer table.
171 V8_INLINE static constexpr bool IsSharedTrustedPointerType(
172  IndirectPointerTag tag) {
173  static_assert(IndirectPointerTag::kFirstSharedTrustedTag <=
174  IndirectPointerTag::kLastSharedTrustedTag);
175  return tag >= IndirectPointerTag::kFirstSharedTrustedTag &&
176  tag <= IndirectPointerTag::kLastSharedTrustedTag;
177 }
178 
179 V8_INLINE static constexpr bool IsPerIsolateTrustedPointerType(
180  IndirectPointerTag tag) {
181  static_assert(IndirectPointerTag::kFirstPerIsolateTrustedTag <=
182  IndirectPointerTag::kLastPerIsolateTrustedTag);
183  return tag >= IndirectPointerTag::kFirstPerIsolateTrustedTag &&
184  tag <= IndirectPointerTag::kLastPerIsolateTrustedTag;
185 }

187 V8_INLINE constexpr bool IsValidIndirectPointerTag(IndirectPointerTag tag) {
188  return IsPerIsolateTrustedPointerType(tag) || IsSharedTrustedPointerType(tag);
189 }

191 // Migrating objects into trusted space is typically performed in multiple
192 // steps, where all references to the object from inside the sandbox are first
193 // changed to indirect pointers before actually moving the object out of the
194 // sandbox. As we have CHECKs that trusted pointer table entries point outside
195 // of the sandbox, we need this helper function to disable that CHECK for
196 // objects that are in the process of being migrated into trusted space.
197 V8_INLINE constexpr bool IsTrustedSpaceMigrationInProgressForObjectsWithTag(
198  IndirectPointerTag tag) {
199  return false;
200 }

206 V8_INLINE IndirectPointerTag
207 IndirectPointerTagFromInstanceType(InstanceType instance_type, bool shared) {
208  switch (instance_type) {
209  case CODE_TYPE:
210  return kCodeIndirectPointerTag;
211  case BYTECODE_ARRAY_TYPE:
212  return kBytecodeArrayIndirectPointerTag;
213  case INTERPRETER_DATA_TYPE:
214  return kInterpreterDataIndirectPointerTag;
215  case UNCOMPILED_DATA_WITHOUT_PREPARSE_DATA_TYPE:
216  case UNCOMPILED_DATA_WITH_PREPARSE_DATA_TYPE:
217  case UNCOMPILED_DATA_WITHOUT_PREPARSE_DATA_WITH_JOB_TYPE:
218  case UNCOMPILED_DATA_WITH_PREPARSE_DATA_AND_JOB_TYPE:
219  // TODO(saelo): Consider adding support for inheritance hierarchies in
220  // our tag checking mechanism.
221  return kUncompiledDataIndirectPointerTag;
222  case ATOM_REG_EXP_DATA_TYPE:
223  case IR_REG_EXP_DATA_TYPE:
224  // TODO(saelo): Consider adding support for inheritance hierarchies in
225  // our tag checking mechanism.
226  return kRegExpDataIndirectPointerTag;
227 #if V8_ENABLE_WEBASSEMBLY
228  case WASM_DISPATCH_TABLE_TYPE:
229  return shared ? kSharedWasmDispatchTableIndirectPointerTag
230  : kWasmDispatchTableIndirectPointerTag;
231  case WASM_TRUSTED_INSTANCE_DATA_TYPE:
232  return shared ? kSharedWasmTrustedInstanceDataIndirectPointerTag
233  : kWasmTrustedInstanceDataIndirectPointerTag;
234  case WASM_INTERNAL_FUNCTION_TYPE:
235  return kWasmInternalFunctionIndirectPointerTag;
236  case WASM_FUNCTION_DATA_TYPE:
237  case WASM_EXPORTED_FUNCTION_DATA_TYPE:
238  case WASM_JS_FUNCTION_DATA_TYPE:
239  case WASM_CAPI_FUNCTION_DATA_TYPE:
240  // TODO(saelo): Consider adding support for inheritance hierarchies in
241  // our tag checking mechanism.
242  return kWasmFunctionDataIndirectPointerTag;
243 #endif // V8_ENABLE_WEBASSEMBLY
244  default:
245  UNREACHABLE();
246  }
247 }
