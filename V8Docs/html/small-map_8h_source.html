16 // SmallMap is a container with a std::map-like interface. It starts out backed
17 // by an unsorted array but switches to some other container type if it grows
18 // beyond this fixed size.
19 //
20 // PROS
21 //
22 // - Good memory locality and low overhead for smaller maps.
23 // - Handles large maps without the degenerate performance of an array.
24 //
25 // CONS
26 //
27 // - Larger code size than the alternatives.
28 //
29 // IMPORTANT NOTES
30 //
31 // - Iterators are invalidated across mutations.
32 //
33 // DETAILS
34 //
35 // SmallMap will pick up the comparator from the underlying map type. In
36 // std::map only a "less" operator is defined, which requires us to do two
37 // comparisons per element when doing the brute-force search in the simple
38 // array. std::unordered_map has a key_equal function which will be used.
39 //
40 // We define default overrides for the common map types to avoid this
41 // double-compare, but you should be aware of this if you use your own operator<
42 // for your map and supply your own version of == to the SmallMap. You can use
43 // regular operator== by just doing:
44 //
45 // SmallMap<std::map<MyKey, MyValue>, 4, std::equal_to<MyKey>>
46 //
47 //
48 // USAGE
49 // -----
50 //
51 // NormalMap: The map type to fall back to. This also defines the key and value
52 // types for the SmallMap.
53 // kArraySize: The size of the initial array of results. This will be allocated
54 // with the SmallMap object rather than separately on the heap.
55 // Once the map grows beyond this size, the map type will be used
56 // instead.
57 // EqualKey: A functor which tests two keys for equality. If the wrapped map
58 // type has a "key_equal" member (unordered_map does), then that will
59 // be used by default. If the wrapped map type has a strict weak
60 // ordering "key_compare" (std::map does), that will be used to
61 // implement equality by default.
62 // MapInit: A functor that takes a NormalMap* and uses it to initialize the map.
63 // This functor will be called at most once per SmallMap, when the map
64 // exceeds the threshold of kArraySize and we are about to copy values
65 // from the array to the map. The functor *must* initialize the
66 // NormalMap* argument with placement new, since after it runs we
67 // assume that the NormalMap has been initialized.
68 //
69 // Example:
70 // SmallMap<std::map<string, int>> days;
71 // days["sunday" ] = 0;
72 // days["monday" ] = 1;
73 // days["tuesday" ] = 2;
74 // days["wednesday"] = 3;
75 // days["thursday" ] = 4;
76 // days["friday" ] = 5;
77 // days["saturday" ] = 6;

87 // has_key_equal<M>::value is true iff there exists a type M::key_equal. This is
88 // used to dispatch to one of the select_equal_key<> metafunctions below.
89 template <typename M>
90 struct has_key_equal {
91  typedef char sml; // "small" is sometimes #defined so we use an abbreviation.
92  typedef struct {
93  char dummy[2];
94  } big;
95  // Two functions, one accepts types that have a key_equal member, and one that
96  // accepts anything. They each return a value of a different size, so we can
97  // determine at compile-time which function would have been called.
98  template <typename U>
99  static big test(typename U::key_equal*);
100  template <typename>
101  static sml test(...);
102  // Determines if M::key_equal exists by looking at the size of the return
103  // type of the compiler-chosen test() function.
104  static const bool value = (sizeof(test<M>(0)) == sizeof(big));
105 };
106 template <typename M>
107 const bool has_key_equal<M>::value;
108 
109 // Base template used for map types that do NOT have an M::key_equal member,
110 // e.g., std::map<>. These maps have a strict weak ordering comparator rather
111 // than an equality functor, so equality will be implemented in terms of that
112 // comparator.
113 //
114 // There's a partial specialization of this template below for map types that do
115 // have an M::key_equal member.
116 template <typename M, bool has_key_equal_value>
117 struct select_equal_key {
118  struct equal_key {
119  bool operator()(const typename M::key_type& left,
120  const typename M::key_type& right) {
121  // Implements equality in terms of a strict weak ordering comparator.
122  typename M::key_compare comp;
123  return !comp(left, right) && !comp(right, left);
124  }
125  };
126 };
127 
128 // Partial template specialization handles case where M::key_equal exists, e.g.,
129 // unordered_map<>.
130 template <typename M>
131 struct select_equal_key<M, true> {
132  typedef typename M::key_equal equal_key;
133 };

137 template <typename NormalMap, size_t kArraySize = 4,

143  std::numeric_limits<size_t>::max();

167  if (&src == this) return;
168

321  iterator find(const key_type& key) {
322  key_equal compare;
323 
324  if (UsingFullMap()) {
325  return iterator(map()->find(key));
326  }
327 
328  for (size_t i = 0; i < size_; ++i) {
329  if (compare(array_[i].first, key)) {
330  return iterator(array_ + i);
331  }
332  }
333  return iterator(array_ + size_);

351  // Invalidates iterators.
352  data_type& operator[](const key_type& key) {
353  key_equal compare;
354 
355  if (UsingFullMap()) {
356  return map_[key];
357  }
358 
359  // Search backwards to favor recently-added elements.
360  for (size_t i = size_; i > 0; --i) {
361  const size_t index = i - 1;
362  if (compare(array_[index].first, key)) {
363  return array_[index].second;
364  }
365  }
366 
367  if (V8_UNLIKELY(size_ == kArraySize)) {
368  ConvertToRealMap();
369  return map_[key];
370  }
371 
372  DCHECK(size_ < kArraySize);
373  new (&array_[size_]) value_type(key, data_type());
374  return array_[size_++].second;
375  }

377  // Invalidates iterators.
378  std::pair<iterator, bool> insert(const value_type& x) {
379  key_equal compare;
380 
381  if (UsingFullMap()) {
382  std::pair<typename NormalMap::iterator, bool> ret = map_.insert(x);
383  return std::make_pair(iterator(ret.first), ret.second);
384  }
385 
386  for (size_t i = 0; i < size_; ++i) {
387  if (compare(array_[i].first, x.first)) {
388  return std::make_pair(iterator(array_ + i), false);
389  }
390  }
391 
392  if (V8_UNLIKELY(size_ == kArraySize)) {
393  ConvertToRealMap(); // Invalidates all iterators!
394  std::pair<typename NormalMap::iterator, bool> ret = map_.insert(x);
395  return std::make_pair(iterator(ret.first), ret.second);
396  }
397 
398  DCHECK(size_ < kArraySize);
399  new (&array_[size_]) value_type(x);
400  return std::make_pair(iterator(array_ + size_++), true);
401  }

500  // Invalidates iterators. Returns iterator following the last removed element.
501  iterator erase(const iterator& position) {
502  if (UsingFullMap()) {
503  return iterator(map_.erase(position.map_iter_));
504  }
505 
506  size_t i = static_cast<size_t>(position.array_iter_ - array_);
507  // TODO(crbug.com/817982): When we have a checked iterator, this CHECK might
508  // not be necessary.
509  CHECK_LE(i, size_);
510  array_[i].~value_type();
511  --size_;
512  if (i != size_) {
513  new (&array_[i]) value_type(std::move(array_[size_]));
514  array_[size_].~value_type();
515  return iterator(array_ + i);
516  }
517  return end();
518  }

552  // When `size_ == kUsingFullMapSentinel`, we have switched storage strategies
553  // from `array_[kArraySize] to `NormalMap map_`. See ConvertToRealMap and
554  // UsingFullMap.
555  size_t size_;
556 
557  MapInit functor_;
558 
559  // We want to call constructors and destructors manually, but we don't want
560  // to allocate and deallocate the memory used for them separately. Since
561  // array_ and map_ are mutually exclusive, we'll put them in a union.
562  union {
563  value_type array_[kArraySize];
564  NormalMap map_;
565  };

567  V8_NOINLINE V8_PRESERVE_MOST void ConvertToRealMap() {
568  // Storage for the elements in the temporary array. This is intentionally
569  // declared as a union to avoid having to default-construct |kArraySize|
570  // elements, only to move construct over them in the initial loop.
571  union Storage {
572  Storage() {}
573  ~Storage() {}
574  value_type array[kArraySize];
575  } temp;
576 
577  // Move the current elements into a temporary array.
578  for (size_t i = 0; i < kArraySize; ++i) {
579  new (&temp.array[i]) value_type(std::move(array_[i]));
580  array_[i].~value_type();
581  }
582 
583  // Initialize the map.
584  size_ = kUsingFullMapSentinel;
585  functor_(&map_);
586 
587  // Insert elements into it.
588  for (size_t i = 0; i < kArraySize; ++i) {
589  map_.insert(std::move(temp.array[i]));
590  temp.array[i].~value_type();
591  }
592  }

594  // Helpers for constructors and destructors.
595  void InitFrom(const SmallMap& src) {
596  functor_ = src.functor_;
597  size_ = src.size_;
598  if (src.UsingFullMap()) {
599  functor_(&map_);
600  map_ = src.map_;
601  } else {
602  for (size_t i = 0; i < size_; ++i) {
603  new (&array_[i]) value_type(src.array_[i]);
604  }
605  }
606  }
607 
608  void Destroy() {
609  if (UsingFullMap()) {
610  map_.~NormalMap();
611  } else {
612  for (size_t i = 0; i < size_; ++i) {
613  array_[i].~value_type();
614  }
615  }
616  }
