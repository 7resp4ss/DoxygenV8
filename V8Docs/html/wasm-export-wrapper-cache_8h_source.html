20 // Provides functionality for caching JS-to-Wasm wrappers in an {Isolate}.
21 // Since these wrappers are JS-compatible {Code} objects living in managed
22 // code space, the cache is also Isolate-specific. It uses a WeakFixedArray
23 // as backing store in order to not leak unused wrappers, and a HashTable-
24 // inspired indexing scheme rather than a dense array layout because we expect
25 // only a minority of canonical type indices to have a corresponding wrapper.
26 // For sandbox purposes, the cache itself is untrusted, but calls through the
27 // wrappers are protected by our regular signature verification scheme.
28 class WasmExportWrapperCache : public AllStatic {
29  public:
30  // DCHECK-fails if a caller attempts to overwrite an existing wrapper.
31  static void Put(Isolate* isolate, CanonicalTypeIndex sig_index,
32  bool receiver_is_first_param, DirectHandle<Code> code);
33 
34  // Returns an empty Tagged if no cache entry was found.
35  static Tagged<CodeWrapper> Get(Isolate* isolate, CanonicalTypeIndex sig_index,
36  bool receiver_is_first_param);
37 
38 #ifdef VERIFY_HEAP
39  static void Verify(Heap* heap);
40 #endif // VERIFY_HEAP
41 
42  // Defined in cctest/test-run-wasm-wrappers.cc to not bloat shipping binaries.
43  static int CountWrappersForTesting(Isolate* isolate);
44 
45  private:
46  static constexpr int kUnused = -1;
47 
48  static constexpr int kNumUsedElementsIndex = 0;
49  static constexpr int kReservedSlots = 1;
50 
51  // Key (Smi), value (Weak<CodeWrapper>).
52  static constexpr int kSlotsPerEntry = 2;
53 
54  static DirectHandle<WeakFixedArray> New(Isolate* isolate, int capacity);
55 
56  static uint32_t Capacity(Tagged<WeakFixedArray> cache);
57 
58  static uint32_t Hash(CanonicalTypeIndex sig_index,
59  bool receiver_is_first_param) {
60  // We expect most signatures to get only one wrapper, so putting the
61  // {receiver_is_first_param} bit at the top should give good distribution
62  // of hash values.
63  static_assert(kMaxCanonicalTypes < (1u << 20));
64  return ((receiver_is_first_param ? 1 : 0) << 20) | sig_index.index;
65  }
66 
67  // Maps the n-th entry to an index in the underlying WeakFixedArray.
68  static int ToArraySlot(InternalIndex entry) {
69  return kReservedSlots + entry.as_int() * kSlotsPerEntry;
70  }
71 
72  static InternalIndex FirstProbe(uint32_t hash, uint32_t capacity) {
73  // {New()} ensures that {capacity} is always a power of two.
74  return InternalIndex(hash & (capacity - 1));
75  }
76 
77  static InternalIndex NextProbe(InternalIndex last, uint32_t number,
78  uint32_t capacity) {
79  // {New()} ensures that {capacity} is always a power of two.
80  return InternalIndex((last.as_uint32() + number) & (capacity - 1));
81  }

88  Tagged<MaybeWeak<Object>> value);
89 };
