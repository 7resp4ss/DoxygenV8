32  platform_max_pages, memory->initial_pages)) *
33  kWasmPageSize;
34  memory->max_memory_size = static_cast<uintptr_t>(std::min<uint64_t>(

64  uint32_t count = 0;
65  uint32_t max = 0;
66  DCHECK_EQ(mode_, kInitializing);
67  for (const auto& entry : *map_) {
68  count++;
69  max = std::max(max, entry.first);

111 // static
112 int GetNearestWasmFunction(const WasmModule* module, uint32_t byte_offset) {
113  const std::vector<WasmFunction>& functions = module->functions;
114 
115  // Binary search for a function containing the given position.
116  int left = 0; // inclusive
117  int right = static_cast<int>(functions.size()); // exclusive
118  if (right == 0) return -1;
119  while (right - left > 1) {
120  int mid = left + (right - left) / 2;
121  if (functions[mid].code.offset() <= byte_offset) {
122  left = mid;
123  } else {
124  right = mid;
125  }
126  }
127 
128  return left;
129 }

131 // static
132 int GetContainingWasmFunction(const WasmModule* module, uint32_t byte_offset) {
133  int func_index = GetNearestWasmFunction(module, byte_offset);
134 
135  if (func_index >= 0) {
136  // If the found function does not contain the given position, return -1.
137  const WasmFunction& func = module->functions[func_index];
138  if (byte_offset < func.code.offset() ||
139  byte_offset >= func.code.end_offset()) {
140  return -1;
141  }
142  }
143  return func_index;
144 }

165 int AsmJsOffsetInformation::GetSourcePosition(int declared_func_index,
166  int byte_offset,
167  bool is_at_number_conversion) {
168  EnsureDecodedOffsets();
169 
170  DCHECK_LE(0, declared_func_index);
171  DCHECK_GT(decoded_offsets_->functions.size(), declared_func_index);
172  std::vector<AsmJsOffsetEntry>& function_offsets =
173  decoded_offsets_->functions[declared_func_index].entries;
174 
175  auto byte_offset_less = [](const AsmJsOffsetEntry& a,
176  const AsmJsOffsetEntry& b) {
177  return a.byte_offset < b.byte_offset;
178  };
179  SLOW_DCHECK(std::is_sorted(function_offsets.begin(), function_offsets.end(),
180  byte_offset_less));
181  // If there are no positions recorded, map offset 0 (for function entry) to
182  // position 0.
183  if (function_offsets.empty() && byte_offset == 0) return 0;
184  auto it =
185  std::lower_bound(function_offsets.begin(), function_offsets.end(),
186  AsmJsOffsetEntry{byte_offset, 0, 0}, byte_offset_less);
187  DCHECK_NE(function_offsets.end(), it);
188  DCHECK_EQ(byte_offset, it->byte_offset);
189  return is_at_number_conversion ? it->source_position_number_conversion
190  : it->source_position_call;
191 }

290 DirectHandle<JSObject> GetTypeForFunction(Isolate* isolate,
291  const FunctionSig* sig,
292  bool for_exception) {
293  Factory* factory = isolate->factory();
294 
295  // Extract values for the {ValueType[]} arrays.
296  int param_index = 0;
297  int param_count = static_cast<int>(sig->parameter_count());
298  DirectHandle<FixedArray> param_values = factory->NewFixedArray(param_count);
299  for (ValueType type : sig->parameters()) {
300  DirectHandle<String> type_value = ToValueTypeString(isolate, type);
301  param_values->set(param_index++, *type_value);
302  }
303 
304  // Create the resulting {FunctionType} object.
305  DirectHandle<JSFunction> object_function = isolate->object_function();
306  DirectHandle<JSObject> object = factory->NewJSObject(object_function);
307  DirectHandle<JSArray> params = factory->NewJSArrayWithElements(param_values);
308  DirectHandle<String> params_string =
309  factory->InternalizeUtf8String("parameters");
310  DirectHandle<String> results_string =
311  factory->InternalizeUtf8String("results");
312  JSObject::AddProperty(isolate, object, params_string, params, NONE);
313 
314  // Now add the result types if needed.
315  if (for_exception) {
316  DCHECK_EQ(sig->returns().size(), 0);
317  } else {
318  int result_index = 0;
319  int result_count = static_cast<int>(sig->return_count());
320  DirectHandle<FixedArray> result_values =
321  factory->NewFixedArray(result_count);
322  for (ValueType type : sig->returns()) {
323  DirectHandle<String> type_value = ToValueTypeString(isolate, type);
324  result_values->set(result_index++, *type_value);
325  }
326  DirectHandle<JSArray> results =
327  factory->NewJSArrayWithElements(result_values);
328  JSObject::AddProperty(isolate, object, results_string, results, NONE);
329  }
330 
331  return object;
332 }

425 DirectHandle<JSArray> GetImports(Isolate* isolate,
426  DirectHandle<WasmModuleObject> module_object) {
427  auto enabled_features = i::wasm::WasmEnabledFeatures::FromIsolate(isolate);
428  Factory* factory = isolate->factory();
429 
430  DirectHandle<String> module_string = factory->InternalizeUtf8String("module");
431  DirectHandle<String> name_string = factory->name_string();
432  DirectHandle<String> kind_string = factory->InternalizeUtf8String("kind");
433  DirectHandle<String> type_string = factory->InternalizeUtf8String("type");
434 
435  DirectHandle<String> function_string = factory->function_string();
436  DirectHandle<String> table_string = factory->InternalizeUtf8String("table");
437  DirectHandle<String> memory_string = factory->InternalizeUtf8String("memory");
438  DirectHandle<String> global_string = factory->global_string();
439  DirectHandle<String> tag_string = factory->InternalizeUtf8String("tag");
440 
441  // Create the result array.
442  NativeModule* native_module = module_object->native_module();
443  const WasmModule* module = native_module->module();
444  int num_imports = static_cast<int>(module->import_table.size());
445  DirectHandle<JSArray> array_object =
446  factory->NewJSArray(PACKED_ELEMENTS, 0, 0);
447  DirectHandle<FixedArray> storage = factory->NewFixedArray(num_imports);
448  JSArray::SetContent(isolate, array_object, storage);
449 
450  DirectHandle<JSFunction> object_function = DirectHandle<JSFunction>(
451  isolate->native_context()->object_function(), isolate);
452 
453  // Populate the result array.
454  const WellKnownImportsList& well_known_imports =
455  module->type_feedback.well_known_imports;
456  const std::string& magic_string_constants =
457  native_module->compile_imports().constants_module();
458  const bool has_magic_string_constants =
459  native_module->compile_imports().contains(
460  CompileTimeImport::kStringConstants);
461 
462  int cursor = 0;
463  for (int index = 0; index < num_imports; ++index) {
464  const WasmImport& import = module->import_table[index];
465 
466  DirectHandle<JSObject> entry = factory->NewJSObject(object_function);
467 
468  DirectHandle<String> import_kind;
469  DirectHandle<JSObject> type_value;
470  switch (import.kind) {
471  case kExternalFunction:
472  if (IsCompileTimeImport(well_known_imports.get(import.index))) {
473  continue;
474  }
475  if (enabled_features.has_type_reflection()) {
476  auto& func = module->functions[import.index];
477  type_value = GetTypeForFunction(isolate, func.sig);
478  }
479  import_kind = function_string;
480  break;
481  case kExternalTable:
482  if (enabled_features.has_type_reflection()) {
483  auto& table = module->tables[import.index];
484  std::optional<uint32_t> maximum_size;
485  if (table.has_maximum_size) maximum_size.emplace(table.maximum_size);
486  type_value = GetTypeForTable(isolate, table.type, table.initial_size,
487  maximum_size, table.address_type);
488  }
489  import_kind = table_string;
490  break;
491  case kExternalMemory:
492  if (enabled_features.has_type_reflection()) {
493  auto& memory = module->memories[import.index];
494  std::optional<uint32_t> maximum_size;
495  if (memory.has_maximum_pages) {
496  maximum_size.emplace(memory.maximum_pages);
497  }
498  type_value =
499  GetTypeForMemory(isolate, memory.initial_pages, maximum_size,
500  memory.is_shared, memory.address_type);
501  }
502  import_kind = memory_string;
503  break;
504  case kExternalGlobal:
505  if (has_magic_string_constants &&
506  import.module_name.length() == magic_string_constants.size() &&
507  std::equal(magic_string_constants.begin(),
508  magic_string_constants.end(),
509  module_object->native_module()->wire_bytes().begin() +
510  import.module_name.offset())) {
511  continue;
512  }
513  if (enabled_features.has_type_reflection()) {
514  auto& global = module->globals[import.index];
515  type_value =
516  GetTypeForGlobal(isolate, global.mutability, global.type);
517  }
518  import_kind = global_string;
519  break;
520  case kExternalTag:
521  import_kind = tag_string;
522  break;
523  }
524  DCHECK(!import_kind.is_null());
525 
526  DirectHandle<String> import_module =
527  WasmModuleObject::ExtractUtf8StringFromModuleBytes(
528  isolate, module_object, import.module_name, kInternalize);
529 
530  DirectHandle<String> import_name =
531  WasmModuleObject::ExtractUtf8StringFromModuleBytes(
532  isolate, module_object, import.field_name, kInternalize);
533 
534  JSObject::AddProperty(isolate, entry, module_string, import_module, NONE);
535  JSObject::AddProperty(isolate, entry, name_string, import_name, NONE);
536  JSObject::AddProperty(isolate, entry, kind_string, import_kind, NONE);
537  if (!type_value.is_null()) {
538  JSObject::AddProperty(isolate, entry, type_string, type_value, NONE);
539  }
540 
541  storage->set(cursor++, *entry);
542  }
543 
544  array_object->set_length(Smi::FromInt(cursor));
545  // Make sure that values after the cursor are holes.
546  for (int index = cursor; index < num_imports; ++index) {
547  storage->set_the_hole(isolate, cursor++);
548  }
549  JSObject::ValidateElements(isolate, *array_object);
550  return array_object;
551 }
