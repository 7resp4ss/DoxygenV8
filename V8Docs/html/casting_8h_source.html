16 // CastTraits<T> is a type trait that defines type checking behaviour for
17 // tagged object casting. The expected specialization is:
18 //
19 // template<>
20 // struct CastTraits<SomeObject> {
21 // template<typename From>
22 // static bool AllowFrom(Tagged<From> value) {
23 // return IsSomeObject(value);
24 // }
25 // };
26 //
27 // or, likely, just specializations of AllowFrom for Object and HeapObject,
28 // under the assumption that the HeapObject implementation is the same for all
29 // HeapObjects and the Object implementation has additional overhead in Smi
30 // checks.
31 //
32 // struct CastTraits<Object> {
33 // static bool AllowFrom(Tagged<HeapObject> value) {
34 // return IsSomeObject(value);
35 // }
36 // static bool AllowFrom(Tagged<Object> value) {
37 // return IsSomeObject(value);
38 // }
39 // };
40 //
41 template <typename To>
42 struct CastTraits;

73 // HasCastImplementation is a concept that checks for the existence of
74 // Is<To>(Holder<From>) and UncheckedCast<To>(Holder<From>).
75 template <template <typename> class Holder, typename To, typename From>
76 concept HasCastImplementation = requires(Holder<From> value) {
77  { Is<To>(value) } -> std::same_as<bool>;
78  { UncheckedCast<To>(value) } -> std::same_as<Holder<To>>;
79 };

81 // `TryCast<T>(value, &out)` casts `value` to a tagged object of type `T` and
82 // writes the value to `out`, returning true if the cast succeeded and false if
83 // it failed.
84 template <typename To, typename From, template <typename> class Holder>
85  requires HasCastImplementation<Holder, To, From>
86 inline bool TryCast(Holder<From> value, Holder<To>* out) {
87  if (!Is<To>(value)) return false;
88  *out = UncheckedCast<To>(value);
89  return true;
90 }

104 // `GCSafeCast<T>(value)` casts `object` to a tagged object of type `T` and
105 // should be used when the cast can be called from within a GC. The case all
106 // includes a debug check that `object` is either a tagged object of type `T`,
107 // or one of few special cases possible during GC (see GCAwareObjectTypeCheck):
108 // 1) `object` was already evacuated and the forwarding address refers to a
109 // tagged object of type `T`.
110 // 2) During Scavenger, `object` is a large object.
111 // 3) During a conservative Scavenger, `object` is a pinned object.
112 template <typename T>
113 Tagged<T> GCSafeCast(Tagged<Object> object, const Heap* heap) {
114  DCHECK(GCAwareObjectTypeCheck<T>(object, heap));
115  return UncheckedCast<T>(object);
116 }

154 // `Is<T>(maybe_weak_value)` specialization for possible weak values and strong
155 // target `T`, that additionally first checks whether `maybe_weak_value` is
156 // actually a strong value (or a Smi, which can't be weak).
157 template <typename T, typename U>
158 inline bool Is(Tagged<MaybeWeak<U>> value) {
159  // Cast from maybe weak to strong needs to be strong or smi.
160  if constexpr (!is_maybe_weak_v<T>) {
161  if (!value.IsStrongOrSmi()) return false;
162  return CastTraits<T>::AllowFrom(Tagged<U>(value.ptr()));
163  } else {
164  // Dispatches to CastTraits<MaybeWeak<T>> below.
165  return CastTraits<T>::AllowFrom(value);
166  }
167 }
168 template <typename T, typename... U>
169 constexpr inline bool Is(Tagged<Union<U...>> value) {
170  using UnionU = Union<U...>;
171  if constexpr (is_subtype_v<UnionU, HeapObject>) {
172  return Is<T>(Tagged<HeapObject>(value));
173  } else if constexpr (is_subtype_v<UnionU, MaybeWeak<HeapObject>>) {
174  return Is<T>(Tagged<MaybeWeak<HeapObject>>(value));
175  } else if constexpr (is_subtype_v<UnionU, Object>) {
176  return Is<T>(Tagged<Object>(value));
177  } else {
178  static_assert(is_subtype_v<UnionU, MaybeWeak<Object>>);
179  return Is<T>(Tagged<MaybeWeak<Object>>(value));
180  }
181 }

183 // Specialization for maybe weak cast targets, which first converts the incoming
184 // value to a strong reference and then checks if the cast to the strong T
185 // is allowed. Cleared weak references always return true.
186 template <typename T>
187 struct CastTraits<MaybeWeak<T>> {
188  template <typename U>
189  static bool AllowFrom(Tagged<U> value) {
190  if constexpr (is_maybe_weak_v<U>) {
191  // Cleared values are always ok.
192  if (value.IsCleared()) return true;
193  // TODO(leszeks): Skip Smi check for values that are known to not be Smi.
194  if (value.IsSmi()) {
195  return CastTraits<T>::AllowFrom(Tagged<Smi>(value.ptr()));
196  }
197  return CastTraits<T>::AllowFrom(MakeStrong(value));
198  } else {
199  return CastTraits<T>::AllowFrom(value);
200  }
201  }
202 };
