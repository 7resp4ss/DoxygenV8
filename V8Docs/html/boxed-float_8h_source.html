20 // Safety wrapper for a 32-bit floating-point value to make sure we don't lose
21 // the exact bit pattern during deoptimization when passing this value.
22 class Float32 {
23  public:
24  Float32() = default;
25 
26  // This constructor does not guarantee that bit pattern of the input value
27  // is preserved if the input is a NaN.
28  explicit Float32(float value)
29  : bit_pattern_(base::bit_cast<uint32_t>(value)) {
30  // Check that the provided value is not a NaN, because the bit pattern of a
31  // NaN may be changed by a base::bit_cast, e.g. for signalling NaNs on
32  // ia32.
33  DCHECK(!std::isnan(value));
34  }
35 
36  uint32_t get_bits() const { return bit_pattern_; }
37 
38  float get_scalar() const { return base::bit_cast<float>(bit_pattern_); }
39 
40  bool is_nan() const {
41  // Even though {get_scalar()} might set the quiet NaN bit, it's ok here,
42  // because this does not change the is_nan property.
43  bool nan = std::isnan(get_scalar());
44  DCHECK_EQ(nan, exponent() == 0xff && mantissa() != 0);
45  return nan;
46  }
47 
48  bool is_quiet_nan() const { return is_nan() && (bit_pattern_ & (1 << 22)); }
49 
50  V8_WARN_UNUSED_RESULT Float32 to_quiet_nan() const {
51  DCHECK(is_nan());
52  Float32 quiet_nan{bit_pattern_ | (1 << 22)};
53  DCHECK(quiet_nan.is_quiet_nan());
54  return quiet_nan;
55  }
56 
57  // Return a pointer to the field storing the bit pattern. Used in code
58  // generation tests to store generated values there directly.
59  uint32_t* get_bits_address() { return &bit_pattern_; }
60 
61  static constexpr Float32 FromBits(uint32_t bits) { return Float32(bits); }
62 
63  private:
64  explicit constexpr Float32(uint32_t bit_pattern)
65  : bit_pattern_(bit_pattern) {}
66 
67  uint32_t exponent() const { return (bit_pattern_ >> 23) & 0xff; }
68  uint32_t mantissa() const { return bit_pattern_ & ((1 << 23) - 1); }
69 
70  uint32_t bit_pattern_ = 0;
71 };
72 
73 ASSERT_TRIVIALLY_COPYABLE(Float32);

75 // Safety wrapper for a 64-bit floating-point value to make sure we don't lose
76 // the exact bit pattern during deoptimization when passing this value.
77 // TODO(ahaas): Unify this class with Double in double.h
78 class Float64 {
79  public:
80  Float64() = default;
81 
82  // This constructor does not guarantee that bit pattern of the input value
83  // is preserved if the input is a NaN.
84  explicit Float64(double value)
85  : bit_pattern_(base::bit_cast<uint64_t>(value)) {
86  // Check that the provided value is not a NaN, because the bit pattern of a
87  // NaN may be changed by a base::bit_cast, e.g. for signalling NaNs on
88  // ia32.
89  DCHECK(!std::isnan(value));
90  }
91 
92  explicit Float64(base::Double value) : bit_pattern_(value.AsUint64()) {}
93 
94  uint64_t get_bits() const { return bit_pattern_; }
95  double get_scalar() const { return base::bit_cast<double>(bit_pattern_); }
96  bool is_hole_nan() const { return bit_pattern_ == kHoleNanInt64; }
97 #ifdef V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
98  bool is_undefined_nan() const { return bit_pattern_ == kUndefinedNanInt64; }
99 #endif // V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
100 
101  bool is_nan() const {
102  // Even though {get_scalar()} might set the quiet NaN bit, it's ok here,
103  // because this does not change the is_nan property.
104  bool nan = std::isnan(get_scalar());
105  DCHECK_EQ(nan, exponent() == 0x7ff && mantissa() != 0);
106  return nan;
107  }
108 
109  bool is_quiet_nan() const {
110  return is_nan() && (bit_pattern_ & (uint64_t{1} << 51));
111  }
112 
113  V8_WARN_UNUSED_RESULT Float64 to_quiet_nan() const {
114  DCHECK(is_nan());
115  Float64 quiet_nan{bit_pattern_ | (uint64_t{1} << 51)};
116  DCHECK(quiet_nan.is_quiet_nan());
117  return quiet_nan;
118  }
119 
120  // Return a pointer to the field storing the bit pattern. Used in code
121  // generation tests to store generated values there directly.
122  uint64_t* get_bits_address() { return &bit_pattern_; }
123 
124  static constexpr Float64 FromBits(uint64_t bits) { return Float64(bits); }
125 
126  // Unlike doubles, equality is defined as equally behaving as far as the
127  // optimizers are concerned. I.e., two NaN's are equal as long as they are
128  // both the hole nor not.
129  bool operator==(const Float64& other) const {
130  if (is_nan() && other.is_nan()) {
131  return is_hole_nan() == other.is_hole_nan();
132  }
133  return get_scalar() == other.get_scalar();
134  }
135 
136  friend size_t hash_value(internal::Float64 f64) { return f64.bit_pattern_; }
137 
138  private:
139  explicit constexpr Float64(uint64_t bit_pattern)
140  : bit_pattern_(bit_pattern) {}
141 
142  uint64_t exponent() const { return (bit_pattern_ >> 52) & ((1 << 11) - 1); }
143  uint64_t mantissa() const { return bit_pattern_ & ((uint64_t{1} << 52) - 1); }
144 
145  uint64_t bit_pattern_ = 0;
146 };
