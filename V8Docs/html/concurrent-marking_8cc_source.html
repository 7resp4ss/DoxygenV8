57 // This class caches page live bytes during concurrent marking. This
58 // avoids costly CAS operations on MutablePageMetadata::live_byte_count_ for
59 // each traced object.
60 //
61 // Page live bytes are cached in a fixed-size hash map. In the case of
62 // collisions the existing entry is simply written back to
63 // MutablePageMetadata::live_byte_count_ with a CAS. Afterwards it can be
64 // replaced with the new entry.
65 class MemoryChunkLiveBytesMap {
66  public:
67  MemoryChunkLiveBytesMap() = default;
68 
69  MemoryChunkLiveBytesMap(const MemoryChunkLiveBytesMap&) = delete;
70  MemoryChunkLiveBytesMap& operator=(const MemoryChunkLiveBytesMap&) = delete;
71 
72  void Increment(MutablePageMetadata* page, intptr_t live);
73  void FlushAndClear();
74  void Erase(MutablePageMetadata* page);
75 
76 #if DEBUG
77  void AssertEmpty();
78 #endif // DEBUG
79 
80  private:
81  struct Entry {
82  MutablePageMetadata* page;
83  intptr_t live_bytes;
84  };
85 
86  static constexpr size_t kTableSize = 32;
87 
88  Entry& lookup_entry(MutablePageMetadata* page) {
89  size_t hash = std::hash<MutablePageMetadata*>{}(page);
90  static_assert(base::bits::IsPowerOfTwo(kTableSize));
91  return map_[hash % kTableSize];
92  }
93 
94  std::array<Entry, kTableSize> map_ = {};
95 };
96 
97 void MemoryChunkLiveBytesMap::Increment(MutablePageMetadata* page,
98  intptr_t bytes) {
99  Entry& entry = lookup_entry(page);
100  if (entry.page == page) {
101  entry.live_bytes += bytes;
102  } else if (entry.page == nullptr) {
103  entry.page = page;
104  entry.live_bytes = bytes;
105  } else {
106  // Write back the existing entry.
107  entry.page->IncrementLiveBytesAtomically(entry.live_bytes);
108  // Now just replace it with the new entry.
109  entry.page = page;
110  entry.live_bytes = bytes;
111  }
112 }
113 
114 void MemoryChunkLiveBytesMap::Erase(MutablePageMetadata* page) {
115  Entry& entry = lookup_entry(page);
116  if (entry.page == page) {
117  entry.page = nullptr;
118  entry.live_bytes = 0;
119  }
120 }
121 
122 void MemoryChunkLiveBytesMap::FlushAndClear() {
123  for (auto& entry : map_) {
124  if (entry.page) {
125  entry.page->IncrementLiveBytesAtomically(entry.live_bytes);
126  entry.page = nullptr;
127  entry.live_bytes = 0;
128  }
129  }
130 }

348 void ConcurrentMarking::RunMajor(JobDelegate* delegate,
349  base::EnumSet<CodeFlushMode> code_flush_mode,
350  unsigned mark_compact_epoch,
351  bool should_keep_ages_unchanged) {
352  size_t kBytesUntilInterruptCheck = 64 * KB;
353  int kObjectsUntilInterruptCheck = 1000;
354  uint8_t task_id = delegate->GetTaskId() + 1;
355  TaskState* task_state = task_state_[task_id].get();
356  auto* cpp_heap = CppHeap::From(heap_->cpp_heap());
357  MarkingWorklists::Local local_marking_worklists(
358  marking_worklists_, cpp_heap
359  ? cpp_heap->CreateCppMarkingState()
360  : MarkingWorklists::Local::kNoCppMarkingState);
361  WeakObjects::Local local_weak_objects(weak_objects_);
362  ConcurrentMarkingVisitor visitor(
363  &local_marking_worklists, &local_weak_objects, heap_, mark_compact_epoch,
364  code_flush_mode, should_keep_ages_unchanged,
365  heap_->tracer()->CodeFlushingIncrease(),
366  &task_state->memory_chunk_live_bytes_map,
367  &task_state->memory_chunk_typed_slots_map);
368  NativeContextInferrer native_context_inferrer;
369  NativeContextStats& native_context_stats = task_state->native_context_stats;
370  double time_ms;
371  size_t marked_bytes = 0;
372  Isolate* isolate = heap_->isolate();
373  if (v8_flags.trace_concurrent_marking) {
374  isolate->PrintWithTimestamp("Starting major concurrent marking task %d\n",
375  task_id);
376  }
377  bool another_ephemeron_iteration = false;
378  MainAllocator* const new_space_allocator =
379  heap_->use_new_space() ? heap_->allocator()->new_space_allocator()
380  : nullptr;
381 
382  {
383  TimedScope scope(&time_ms);
384 
385  {
386  Ephemeron ephemeron;
387  while (local_weak_objects.current_ephemerons_local.Pop(&ephemeron)) {
388  if (visitor.ProcessEphemeron(ephemeron.key, ephemeron.value)) {
389  another_ephemeron_iteration = true;
390  }
391  }
392  }
393  PtrComprCageBase cage_base(isolate);
394  bool is_per_context_mode = local_marking_worklists.IsPerContextMode();
395  bool done = false;
396  while (!done) {
397  size_t current_marked_bytes = 0;
398  int objects_processed = 0;
399  while (current_marked_bytes < kBytesUntilInterruptCheck &&
400  objects_processed < kObjectsUntilInterruptCheck) {
401  Tagged<HeapObject> object;
402  if (!local_marking_worklists.Pop(&object)) {
403  done = true;
404  break;
405  }
406  DCHECK(!HeapLayout::InReadOnlySpace(object));
407  DCHECK_EQ(HeapUtils::GetOwnerHeap(object), heap_);
408  objects_processed++;
409 
410  Address new_space_top = kNullAddress;
411  Address new_space_limit = kNullAddress;
412  Address new_large_object = kNullAddress;
413 
414  if (new_space_allocator) {
415  std::tie(new_space_top, new_space_limit) =
416  new_space_allocator->GetOriginalTopAndLimit();
417  }
418 
419  if (heap_->new_lo_space()) {
420  new_large_object = heap_->new_lo_space()->pending_object();
421  }
422 
423  Address addr = object.address();
424 
425  if ((new_space_top <= addr && addr < new_space_limit) ||
426  addr == new_large_object) {
427  local_marking_worklists.PushOnHold(object);
428  } else {
429  Tagged<Map> map = object->map(cage_base, kAcquireLoad);
430  // The marking worklist should never contain filler objects.
431  CHECK(!IsFreeSpaceOrFillerMap(map));
432  if (is_per_context_mode) {
433  Address context;
434  if (native_context_inferrer.Infer(cage_base, map, object,
435  &context)) {
436  local_marking_worklists.SwitchToContext(context);
437  }
438  }
439  const auto visited_size = visitor.Visit(map, object);
440  visitor.IncrementLiveBytesCached(
441  MutablePageMetadata::cast(
442  MemoryChunkMetadata::FromHeapObject(object)),
443  ALIGN_TO_ALLOCATION_ALIGNMENT(visited_size));
444  if (is_per_context_mode) {
445  native_context_stats.IncrementSize(
446  local_marking_worklists.Context(), map, object, visited_size);
447  }
448  current_marked_bytes += visited_size;
449  }
450  }
451  if (objects_processed > 0) another_ephemeron_iteration = true;
452  marked_bytes += current_marked_bytes;
453  base::AsAtomicWord::Relaxed_Store<size_t>(&task_state->marked_bytes,
454  marked_bytes);
455  if (delegate->ShouldYield()) {
456  TRACE_GC_NOTE("ConcurrentMarking::RunMajor Preempted");
457  break;
458  }
459  }
460 
461  CHECK(local_weak_objects.current_ephemerons_local.IsLocalEmpty());
462 
463  local_marking_worklists.Publish();
464  local_weak_objects.Publish();
465  base::AsAtomicWord::Relaxed_Store<size_t>(&task_state->marked_bytes, 0);
466  total_marked_bytes_ += marked_bytes;
467 
468  if (another_ephemeron_iteration) {
469  set_another_ephemeron_iteration(true);
470  }
471  }
472  if (v8_flags.trace_concurrent_marking) {
473  heap_->isolate()->PrintWithTimestamp(
474  "Major task %d concurrently marked %dKB in %.2fms\n", task_id,
475  static_cast<int>(marked_bytes / KB), time_ms);
476  }
477 
478  DCHECK(task_state->local_pretenuring_feedback.empty());
479 }

516 template <YoungGenerationMarkingVisitationMode marking_mode>
517 V8_INLINE size_t ConcurrentMarking::RunMinorImpl(JobDelegate* delegate,
518  TaskState* task_state) {
519  static constexpr size_t kBytesUntilInterruptCheck = 64 * KB;
520  static constexpr int kObjectsUntilInterruptCheck = 1000;
521  size_t marked_bytes = 0;
522  size_t current_marked_bytes = 0;
523  int objects_processed = 0;
524  YoungGenerationMarkingVisitor<marking_mode> visitor(
525  heap_, &task_state->local_pretenuring_feedback);
526  YoungGenerationRememberedSetsMarkingWorklist::Local remembered_sets(
527  heap_->minor_mark_sweep_collector()->remembered_sets_marking_handler());
528  auto& marking_worklists_local = visitor.marking_worklists_local();
529  Isolate* isolate = heap_->isolate();
530  minor_marking_state_->MarkerStarted();
531  MainAllocator* const new_space_allocator =
532  heap_->allocator()->new_space_allocator();
533  NewLargeObjectSpace* const new_lo_space = heap_->new_lo_space();
534 
535  do {
536  if (delegate->IsJoiningThread()) {
537  marking_worklists_local.MergeOnHold();
538  }
539  Tagged<HeapObject> heap_object;
540  TRACE_GC_EPOCH(heap_->tracer(),
541  GCTracer::Scope::MINOR_MS_BACKGROUND_MARKING_CLOSURE,
542  ThreadKind::kBackground);
543  while (marking_worklists_local.Pop(&heap_object)) {
544  if (IsYoungObjectInLab(new_space_allocator, new_lo_space, heap_object)) {
545  visitor.marking_worklists_local().PushOnHold(heap_object);
546  } else {
547  Tagged<Map> map = heap_object->map(isolate);
548  const auto visited_size = visitor.Visit(map, heap_object);
549  if (visited_size) {
550  current_marked_bytes += visited_size;
551  visitor.IncrementLiveBytesCached(
552  MutablePageMetadata::FromHeapObject(heap_object),
553  ALIGN_TO_ALLOCATION_ALIGNMENT(visited_size));
554  }
555  }
556 
557  if (current_marked_bytes >= kBytesUntilInterruptCheck ||
558  ++objects_processed >= kObjectsUntilInterruptCheck) {
559  marked_bytes += current_marked_bytes;
560  if (delegate->ShouldYield()) {
561  TRACE_GC_NOTE("ConcurrentMarking::RunMinor Preempted");
562  minor_marking_state_->MarkerDone();
563  return marked_bytes;
564  }
565  objects_processed = 0;
566  current_marked_bytes = 0;
567  }
568  }
569  } while (remembered_sets.ProcessNextItem(&visitor));
570  if (minor_marking_state_->MarkerDone()) {
571  // This is the last active marker and it ran out of work. Request GC
572  // finalization.
573  heap_->minor_mark_sweep_collector()->RequestGC();
574  }
575  return marked_bytes + current_marked_bytes;
576 }

578 void ConcurrentMarking::RunMinor(JobDelegate* delegate) {
579  DCHECK(heap_->use_new_space());
580  DCHECK_NOT_NULL(heap_->new_lo_space());
581  uint8_t task_id = delegate->GetTaskId() + 1;
582  DCHECK_LT(task_id, task_state_.size());
583  TaskState* task_state = task_state_[task_id].get();
584  double time_ms;
585  size_t marked_bytes = 0;
586  Isolate* isolate = heap_->isolate();
587  if (v8_flags.trace_concurrent_marking) {
588  isolate->PrintWithTimestamp("Starting minor concurrent marking task %d\n",
589  task_id);
590  }
591 
592  {
593  TimedScope scope(&time_ms);
594  if (heap_->minor_mark_sweep_collector()->is_in_atomic_pause()) {
595  // This gets a lower bound for estimated concurrency as we may have marked

647 void ConcurrentMarking::TryScheduleJob(GarbageCollector garbage_collector,
648  TaskPriority priority) {
649  DCHECK(v8_flags.parallel_marking || v8_flags.concurrent_marking ||
650  v8_flags.concurrent_minor_ms_marking);
651  DCHECK(!heap_->IsTearingDown());
652  DCHECK(IsStopped());
653 
654  DCHECK_NE(garbage_collector, GarbageCollector::SCAVENGER);
655  if (garbage_collector == GarbageCollector::MARK_COMPACTOR &&
656  !heap_->mark_compact_collector()->UseBackgroundThreadsInCycle()) {
657  return;
658  }
659  if (garbage_collector == GarbageCollector::MINOR_MARK_SWEEPER &&
660  !heap_->minor_mark_sweep_collector()->UseBackgroundThreadsInCycle()) {
661  return;
662  }
663 
664  if (v8_flags.concurrent_marking_high_priority_threads) {
665  priority = TaskPriority::kUserBlocking;
666  }
667 
668 #if DEBUG
669  if (garbage_collector_.has_value()) {
670  // Concurrent marking resumes. In this case the used collectors need to
671  // match.
672  DCHECK_EQ(*garbage_collector_, garbage_collector);
673  } else {
674  // If a new concurrent marking cycle starts, TaskState should not contain
675  // any data.
676  for (auto& task_state : task_state_) {
677  task_state->memory_chunk_live_bytes_map.AssertEmpty();
678  DCHECK(task_state->memory_chunk_typed_slots_map.empty());
679  DCHECK(task_state->native_context_stats.Empty());
680  DCHECK(task_state->local_pretenuring_feedback.empty());
681  DCHECK_EQ(0, task_state->marked_bytes);
682  }
683  DCHECK_EQ(0, total_marked_bytes_.load());
684  }
685 
686  if (garbage_collector == GarbageCollector::MARK_COMPACTOR) {
687  // The full GC never makes use of local_pretenuring_feedback. It needs to be
688  // empty even if resuming concurrent marking.
689  for (auto& task_state : task_state_) {
690  DCHECK(task_state->local_pretenuring_feedback.empty());
691  }
692  }
693 
694  if (minor_marking_state_) {
695  // Minor marking state can only be alive if the concurrent marker was
696  // previously paused.
697  DCHECK(garbage_collector_.has_value());
698  DCHECK_EQ(*garbage_collector_, GarbageCollector::MINOR_MARK_SWEEPER);
699  DCHECK_EQ(garbage_collector, GarbageCollector::MINOR_MARK_SWEEPER);
700  }
701 #endif // DEBUG
702 
703  garbage_collector_ = garbage_collector;
704  if (garbage_collector == GarbageCollector::MARK_COMPACTOR) {
705  heap_->mark_compact_collector()->local_marking_worklists()->Publish();
706  marking_worklists_ = heap_->mark_compact_collector()->marking_worklists();
707  auto job = std::make_unique<JobTaskMajor>(
708  this, heap_->mark_compact_collector()->epoch(),
709  heap_->mark_compact_collector()->code_flush_mode(),
710  heap_->ShouldCurrentGCKeepAgesUnchanged());
711  current_job_trace_id_.emplace(job->trace_id());
712  TRACE_GC_NOTE_WITH_FLOW("Major concurrent marking started", job->trace_id(),
713  TRACE_EVENT_FLAG_FLOW_OUT);
714  job_handle_ = V8::GetCurrentPlatform()->PostJob(priority, std::move(job));
715  } else {
716  DCHECK(garbage_collector == GarbageCollector::MINOR_MARK_SWEEPER);
717  minor_marking_state_ = std::make_unique<MinorMarkingState>();
718  heap_->minor_mark_sweep_collector()->local_marking_worklists()->Publish();
719  marking_worklists_ =
720  heap_->minor_mark_sweep_collector()->marking_worklists();
721  auto job = std::make_unique<JobTaskMinor>(this);
722  current_job_trace_id_.emplace(job->trace_id());
723  TRACE_GC_NOTE_WITH_FLOW("Minor concurrent marking started", job->trace_id(),
724  TRACE_EVENT_FLAG_FLOW_OUT);
725  job_handle_ = V8::GetCurrentPlatform()->PostJob(priority, std::move(job));
726  }
727  DCHECK(job_handle_->IsValid());
728 }

743 void ConcurrentMarking::RescheduleJobIfNeeded(
744  GarbageCollector garbage_collector, TaskPriority priority) {
745  DCHECK(v8_flags.parallel_marking || v8_flags.concurrent_marking ||
746  v8_flags.concurrent_minor_ms_marking);
747 
748  if (garbage_collector == GarbageCollector::MARK_COMPACTOR &&
749  !heap_->mark_compact_collector()->UseBackgroundThreadsInCycle()) {
750  return;
751  }
752 
753  if (garbage_collector == GarbageCollector::MINOR_MARK_SWEEPER &&
754  !heap_->minor_mark_sweep_collector()->UseBackgroundThreadsInCycle()) {
755  return;
756  }
757 
758  if (heap_->IsTearingDown()) return;
759 
760  if (IsStopped()) {
761  // This DCHECK is for the case that concurrent marking was paused.
762  DCHECK_IMPLIES(garbage_collector_.has_value(),
763  garbage_collector == garbage_collector_);
764  TryScheduleJob(garbage_collector, priority);
765  } else {
766  DCHECK(garbage_collector_.has_value());
767  DCHECK_EQ(garbage_collector, garbage_collector_.value());
768  if (garbage_collector == GarbageCollector::MARK_COMPACTOR) {
769  heap_->mark_compact_collector()->local_marking_worklists()->Publish();
770  } else {
771  heap_->minor_mark_sweep_collector()->local_marking_worklists()->Publish();
772  }
773  if (!IsWorkLeft()) return;
774  if (priority != TaskPriority::kUserVisible)
775  job_handle_->UpdatePriority(priority);
776  DCHECK(current_job_trace_id_.has_value());
777  TRACE_GC_NOTE_WITH_FLOW(
778  garbage_collector_ == GarbageCollector::MARK_COMPACTOR
779  ? "Major concurrent marking rescheduled"
780  : "Minor concurrent marking rescheduled",
781  current_job_trace_id_.value(),
782  TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
783  job_handle_->NotifyConcurrencyIncrease();
784  }
