185  // Convert a stack frame type to a marker that can be stored on the stack.
186  //
187  // The marker is an opaque value, not intended to be interpreted in any way
188  // except being checked by IsTypeMarker or converted by MarkerToType.
189  // It has the same tagging as Smis, so any marker value that does not pass
190  // IsTypeMarker can instead be interpreted as a tagged pointer.
191  //
192  // Note that the marker is not a Smi: Smis on 64-bit architectures are stored
193  // in the top 32 bits of a 64-bit value, which in turn makes them expensive
194  // (in terms of code/instruction size) to push as immediates onto the stack.
195  static constexpr int32_t TypeToMarker(Type type) {
196  DCHECK_GE(type, 0);
197  return (type << kSmiTagSize) | kSmiTag;
198  }
199 
200  // Convert a marker back to a stack frame type.
201  //
202  // Unlike the return value of TypeToMarker, this takes an intptr_t, as that is
203  // the type of the value on the stack.
204  static constexpr Type MarkerToType(intptr_t marker) {
205  DCHECK(IsTypeMarker(marker));
206  return static_cast<Type>(marker >> kSmiTagSize);
207  }
208 
209  // Check if a marker is a stack frame type marker.
210  //
211  // Returns true if the given marker is tagged as a stack frame type marker,
212  // and should be converted back to a stack frame type using MarkerToType.
213  static constexpr bool IsTypeMarker(uintptr_t function_or_marker) {
214  static_assert(kSmiTag == 0);
215  static_assert((std::numeric_limits<uintptr_t>::max() >> kSmiTagSize) >
216  Type::NUMBER_OF_TYPES);
217  return (function_or_marker & kSmiTagMask) == kSmiTag &&
218  function_or_marker < (Type::NUMBER_OF_TYPES << kSmiTagSize);
219  }
