184  // Check if output is known to be Smi when both operands of bitwise operation
185  // are Smi.
186  bool IsBitwiseOutputKnownSmi(Operation bitwise_op) {
187  switch (bitwise_op) {
188  case Operation::kBitwiseAnd:
189  case Operation::kBitwiseOr:
190  case Operation::kBitwiseXor:
191  case Operation::kShiftRight:
192  return true;
193  default:
194  return false;
195  }
