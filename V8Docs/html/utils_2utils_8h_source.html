59 template <typename T>
60 static T ArithmeticShiftRight(T x, int shift) {
61  DCHECK_LE(0, shift);
62  if (x < 0) {
63  // Right shift of signed values is implementation defined. Simulate a
64  // true arithmetic right shift by adding leading sign bits.
65  using UnsignedT = std::make_unsigned_t<T>;
66  UnsignedT mask = ~(static_cast<UnsignedT>(~0) >> shift);
67  return (static_cast<UnsignedT>(x) >> shift) | mask;
68  } else {
69  return x >> shift;
70  }
71 }

73 // Returns the maximum of the two parameters according to JavaScript semantics.
74 template <typename T>
75 T JSMax(T x, T y) {
76  if (std::isnan(x)) return x;
77  if (std::isnan(y)) return y;
78  if (std::signbit(x) < std::signbit(y)) return x;
79  return x > y ? x : y;

91 // Returns the absolute value of its argument.
92 template <typename T>
93 std::make_unsigned_t<T> Abs(T a)
94  requires std::is_signed_v<T>
95 {
96  // This is a branch-free implementation of the absolute value function and is
97  // described in Warren's "Hacker's Delight", chapter 2. It avoids undefined
98  // behavior with the arithmetic negation operation on signed values as well.
99  using unsignedT = std::make_unsigned_t<T>;
100  unsignedT x = static_cast<unsignedT>(a);
101  unsignedT y = static_cast<unsignedT>(a >> (sizeof(T) * 8 - 1));
102  return (x ^ y) - y;
103 }

105 inline double Modulo(double x, double y) {
106 #if defined(V8_OS_WIN)
107  // Workaround MS fmod bugs. ECMA-262 says:
108  // dividend is finite and divisor is an infinity => result equals dividend
109  // dividend is a zero and divisor is nonzero finite => result equals dividend
110  if (!(std::isfinite(x) && (!std::isfinite(y) && !std::isnan(y))) &&
111  !(x == 0 && (y != 0 && std::isfinite(y)))) {
112  double result = fmod(x, y);
113  // Workaround MS bug in VS CRT in some OS versions, https://crbug.com/915045
114  // fmod(-17, +/-1) should equal -0.0 but now returns 0.0.
115  if (x < 0 && result == 0) result = -0.0;
116  x = result;
117  }
118  return x;
119 #elif defined(V8_OS_AIX)
120  // AIX raises an underflow exception for (Number.MIN_VALUE % Number.MAX_VALUE)
121  feclearexcept(FE_ALL_EXCEPT);
122  double result = std::fmod(x, y);
123  int exception = fetestexcept(FE_UNDERFLOW);
124  return (exception ? x : result);
125 #else
126  return std::fmod(x, y);
127 #endif
128 }

130 template <typename T>
131 T SaturateAdd(T a, T b) {
132  if (std::is_signed_v<T>) {
133  if (a > 0 && b > 0) {
134  if (a > std::numeric_limits<T>::max() - b) {
135  return std::numeric_limits<T>::max();
136  }
137  } else if (a < 0 && b < 0) {
138  if (a < std::numeric_limits<T>::min() - b) {
139  return std::numeric_limits<T>::min();
140  }
141  }
142  } else {
143  CHECK(std::is_unsigned_v<T>);
144  if (a > std::numeric_limits<T>::max() - b) {
145  return std::numeric_limits<T>::max();
146  }
147  }
148  return a + b;
149 }

151 template <typename T>
152 T SaturateSub(T a, T b) {
153  if (std::is_signed_v<T>) {
154  if (a >= 0 && b < 0) {
155  if (a > std::numeric_limits<T>::max() + b) {
156  return std::numeric_limits<T>::max();
157  }
158  } else if (a < 0 && b > 0) {
159  if (a < std::numeric_limits<T>::min() + b) {
160  return std::numeric_limits<T>::min();
161  }
162  }
163  } else {
164  CHECK(std::is_unsigned_v<T>);
165  if (a < b) {
166  return static_cast<T>(0);
167  }
168  }
169  return a - b;
170 }

172 template <typename T>
173 T SaturateRoundingQMul(T a, T b) {
174  // Saturating rounding multiplication for Q-format numbers. See
175  // https://en.wikipedia.org/wiki/Q_(number_format) for a description.
176  // Specifically this supports Q7, Q15, and Q31. This follows the
177  // implementation in simulator-logic-arm64.cc (sqrdmulh) to avoid overflow
178  // when a == b == int32 min.
179  static_assert(std::is_integral_v<T>, "only integral types");
180 
181  constexpr int size_in_bits = sizeof(T) * 8;
182  int round_const = 1 << (size_in_bits - 2);
183  int64_t product = a * b;
184  product += round_const;
185  product >>= (size_in_bits - 1);
186  return base::saturated_cast<T>(product);
187 }

189 // Multiply two numbers, returning a result that is twice as wide, no overflow.
190 // Put Wide first so we can use function template argument deduction for Narrow,
191 // and callers can provide only Wide.
192 template <typename Wide, typename Narrow>
193 Wide MultiplyLong(Narrow a, Narrow b) {
194  static_assert(std::is_integral_v<Narrow> && std::is_integral_v<Wide>,
195  "only integral types");
196  static_assert(std::is_signed_v<Narrow> == std::is_signed_v<Wide>,
197  "both must have same signedness");
198  static_assert(sizeof(Narrow) * 2 == sizeof(Wide), "only twice as long");
199 
200  return static_cast<Wide>(a) * static_cast<Wide>(b);

273 // Thomas Wang, Integer Hash Functions.
274 // http://www.concentric.net/~Ttwang/tech/inthash.htm`
275 inline uint32_t ComputeUnseededHash(uint32_t key) {
276  uint32_t hash = key;
277  hash = ~hash + (hash << 15); // hash = (hash << 15) - hash - 1;
278  hash = hash ^ (hash >> 12);
279  hash = hash + (hash << 2);
280  hash = hash ^ (hash >> 4);
281  hash = hash * 2057; // hash = (hash + (hash << 3)) + (hash << 11);
282  hash = hash ^ (hash >> 16);
283  return hash & 0x3fffffff;
284 }
285 
286 inline uint32_t ComputeLongHash(uint64_t key) {
287  uint64_t hash = key;
288  hash = ~hash + (hash << 18); // hash = (hash << 18) - hash - 1;
289  hash = hash ^ (hash >> 31);
290  hash = hash * 21; // hash = (hash + (hash << 2)) + (hash << 4);
291  hash = hash ^ (hash >> 11);
292  hash = hash + (hash << 6);
293  hash = hash ^ (hash >> 22);
294  return static_cast<uint32_t>(hash & 0x3fffffff);
295 }

358 template <typename Char>
359 V8_INLINE bool SimdMemEqual(const Char* lhs, const Char* rhs, size_t count,
360  size_t order) {
361  static_assert(sizeof(Char) == 1);
362  DCHECK_GE(order, 5);
363 
364  static constexpr uint16_t kSIMDMatched16Mask = UINT16_MAX;
365  static constexpr uint32_t kSIMDMatched32Mask = UINT32_MAX;
366 
367  if (order == 5) { // count: [17, 32]
368  // Utilize more simd registers for better pipelining.
369  const __m128i lhs128_start =
370  _mm_lddqu_si128(reinterpret_cast<const __m128i*>(lhs));
371  const __m128i lhs128_end = _mm_lddqu_si128(
372  reinterpret_cast<const __m128i*>(lhs + count - sizeof(__m128i)));
373  const __m128i rhs128_start =
374  _mm_lddqu_si128(reinterpret_cast<const __m128i*>(rhs));
375  const __m128i rhs128_end = _mm_lddqu_si128(
376  reinterpret_cast<const __m128i*>(rhs + count - sizeof(__m128i)));
377  const __m128i res_start = _mm_cmpeq_epi8(lhs128_start, rhs128_start);
378  const __m128i res_end = _mm_cmpeq_epi8(lhs128_end, rhs128_end);
379  const uint32_t res =
380  _mm_movemask_epi8(res_start) << 16 | _mm_movemask_epi8(res_end);
381  return res == kSIMDMatched32Mask;
382  }
383 
384  // count: [33, ...]
385  const __m128i lhs128_unrolled =
386  _mm_lddqu_si128(reinterpret_cast<const __m128i*>(lhs));
387  const __m128i rhs128_unrolled =
388  _mm_lddqu_si128(reinterpret_cast<const __m128i*>(rhs));
389  const __m128i res_unrolled = _mm_cmpeq_epi8(lhs128_unrolled, rhs128_unrolled);
390  const uint16_t res_unrolled_mask = _mm_movemask_epi8(res_unrolled);
391  if (res_unrolled_mask != kSIMDMatched16Mask) return false;
392 
393  for (size_t i = count % sizeof(__m128i); i < count; i += sizeof(__m128i)) {
394  const __m128i lhs128 =
395  _mm_lddqu_si128(reinterpret_cast<const __m128i*>(lhs + i));
396  const __m128i rhs128 =
397  _mm_lddqu_si128(reinterpret_cast<const __m128i*>(rhs + i));
398  const __m128i res = _mm_cmpeq_epi8(lhs128, rhs128);
399  const uint16_t res_mask = _mm_movemask_epi8(res);
400  if (res_mask != kSIMDMatched16Mask) return false;
401  }
402  return true;
403 }

407 // We intentionally use misaligned read/writes for NEON intrinsics, disable
408 // alignment sanitization explicitly.
409 template <typename Char>
410 V8_INLINE V8_CLANG_NO_SANITIZE("alignment") bool SimdMemEqual(const Char* lhs,
411  const Char* rhs,
412  size_t count,
413  size_t order) {
414  static_assert(sizeof(Char) == 1);
415  DCHECK_GE(order, 5);
416 
417  if (order == 5) { // count: [17, 32]
418  // Utilize more simd registers for better pipelining.
419  const auto lhs0 = vld1q_u8(lhs);
420  const auto lhs1 = vld1q_u8(lhs + count - sizeof(uint8x16_t));
421  const auto rhs0 = vld1q_u8(rhs);
422  const auto rhs1 = vld1q_u8(rhs + count - sizeof(uint8x16_t));
423  const auto xored0 = veorq_u8(lhs0, rhs0);
424  const auto xored1 = veorq_u8(lhs1, rhs1);
425  const auto ored = vorrq_u8(xored0, xored1);
426  return !static_cast<bool>(
427  vgetq_lane_u64(vreinterpretq_u64_u8(vpmaxq_u8(ored, ored)), 0));
428  }
429 
430  // count: [33, ...]
431  const auto first_lhs0 = vld1q_u8(lhs);
432  const auto first_rhs0 = vld1q_u8(rhs);
433  const auto first_xored = veorq_u8(first_lhs0, first_rhs0);
434  if (static_cast<bool>(vgetq_lane_u64(
435  vreinterpretq_u64_u8(vpmaxq_u8(first_xored, first_xored)), 0))) {
436  return false;
437  }
438  for (size_t i = count % sizeof(uint8x16_t); i < count;
439  i += sizeof(uint8x16_t)) {
440  const auto lhs0 = vld1q_u8(lhs + i);
441  const auto rhs0 = vld1q_u8(rhs + i);
442  const auto xored = veorq_u8(lhs0, rhs0);
443  if (static_cast<bool>(
444  vgetq_lane_u64(vreinterpretq_u64_u8(vpmaxq_u8(xored, xored)), 0)))
445  return false;
446  }
447  return true;
448 }

463 template <typename Char>
464 V8_CLANG_NO_SANITIZE("alignment")
465 V8_INLINE bool SimdMemEqual(const Char* lhs, const Char* rhs, size_t count) {
466  static_assert(sizeof(Char) == 1);
467  if (count == 0) {
468  return true;
469  }
470  if (count == 1) {
471  return *lhs == *rhs;
472  }
473  const size_t order =
474  sizeof(count) * CHAR_BIT - base::bits::CountLeadingZeros(count - 1);
475  switch (order) {
476  case 1: // count: [2, 2]
477  return *reinterpret_cast<const uint16_t*>(lhs) ==
478  *reinterpret_cast<const uint16_t*>(rhs);
479  case 2: // count: [3, 4]
480  return OverlappingCompare<uint16_t>(lhs, rhs, count);
481  case 3: // count: [5, 8]
482  return OverlappingCompare<uint32_t>(lhs, rhs, count);
483  case 4: // count: [9, 16]
484  return OverlappingCompare<uint64_t>(lhs, rhs, count);
485  default:
486  return SimdMemEqual(lhs, rhs, count, order);
487  }
488 }

490 // Compare 8bit/16bit chars to 8bit/16bit chars.
491 template <typename lchar, typename rchar>
492 inline bool CompareCharsEqualUnsigned(const lchar* lhs, const rchar* rhs,
493  size_t chars) {
494  static_assert(std::is_unsigned_v<lchar>);
495  static_assert(std::is_unsigned_v<rchar>);
496  if constexpr (sizeof(*lhs) == sizeof(*rhs)) {
497 #if defined(__SSE3__) || defined(V8_OPTIMIZE_WITH_NEON)
498  if constexpr (sizeof(*lhs) == 1) {
499  return SimdMemEqual(lhs, rhs, chars);
500  }
501 #endif
502  // memcmp compares byte-by-byte, but for equality it doesn't matter whether
503  // two-byte char comparison is little- or big-endian.
504  return memcmp(lhs, rhs, chars * sizeof(*lhs)) == 0;
505  }
506  for (const lchar* limit = lhs + chars; lhs < limit; ++lhs, ++rhs) {
507  if (*lhs != *rhs) return false;
508  }
509  return true;
510 }

521 // Compare 8bit/16bit chars to 8bit/16bit chars.
522 template <typename lchar, typename rchar>
523 inline int CompareCharsUnsigned(const lchar* lhs, const rchar* rhs,
524  size_t chars) {
525  static_assert(std::is_unsigned_v<lchar>);
526  static_assert(std::is_unsigned_v<rchar>);
527  if (sizeof(*lhs) == sizeof(char) && sizeof(*rhs) == sizeof(char)) {
528  // memcmp compares byte-by-byte, yielding wrong results for two-byte
529  // strings on little-endian systems.
530  return memcmp(lhs, rhs, chars);
531  }
532  for (const lchar* limit = lhs + chars; lhs < limit; ++lhs, ++rhs) {
533  int r = static_cast<int>(*lhs) - static_cast<int>(*rhs);
534  if (r != 0) return r;
535  }
536  return 0;
537 }

784 #if V8_HAS_BUILTIN_BSWAP64
785  return __builtin_bswap64(value);
786 #else

788  return value << (bits_of_v - 8) |

790  ((value << (bits_of_v - 40)) & 0x0000FF0000000000) |
791  ((value << (bits_of_v - 56)) & 0x000000FF00000000) |
792  ((value >> (bits_of_v - 56)) & 0x00000000FF000000) |
793  ((value >> (bits_of_v - 40)) & 0x0000000000FF0000) |
794  ((value >> (bits_of_v - 24)) & 0x000000000000FF00) |
795  ((value >> (bits_of_v - 8)) & 0x00000000000000FF);
796 #endif
