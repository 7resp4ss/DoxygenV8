61 struct PropertyAccessTarget {

66  struct Hash {
67  size_t operator()(const PropertyAccessTarget& pair) const {
68  return base::hash_combine(
69  base::hash_combine(pair.map.object().address(),
70  pair.name.object().address()),
71  static_cast<int>(pair.mode));
72  }
73  };
74  struct Equal {
75  bool operator()(const PropertyAccessTarget& lhs,
76  const PropertyAccessTarget& rhs) const {
77  return lhs.map.equals(rhs.map) && lhs.name.equals(rhs.name) &&
78  lhs.mode == rhs.mode;
79  }
80  };

274  std::optional<RootIndex> FindRootIndex(HeapObjectRef object) {
275  // No root constant is a JSReceiver.
276  if (object.IsJSReceiver()) return {};
277  RootIndex root_index;
278  if (root_index_map_.Lookup(*object.object(), &root_index)) {
279  return root_index;
280  }
281  return {};
282  }
283 
284  // Return the corresponding canonical persistent handle for {object}. Create
285  // one if it does not exist.
286  // If a local isolate is attached, we can create the persistent handle through
287  // it. This commonly happens during the Execute phase.
288  // If we don't, that means we are calling this method from serialization. If
289  // that happens, we should be inside a persistent handle scope. Then, we would
290  // just use the regular handle creation.
291  template <typename T>
292  Handle<T> CanonicalPersistentHandle(Tagged<T> object) {
293  DCHECK_NOT_NULL(canonical_handles_);
294  if (Tagged<HeapObject> heap_object;
295  TryCast<HeapObject>(object, &heap_object)) {
296  RootIndex root_index;
297  // The root index map only contains immortal, immutable objects; it never
298  // contains any instances of type JSObject, since JSObjects must exist
299  // within a NativeContext, and NativeContexts can be created and
300  // destroyed. Thus, we can skip the lookup in the root index map for those
301  // values and save a little time.
302  if constexpr (std::is_convertible_v<T, JSObject> ||
303  std::is_convertible_v<T, Context>) {
304  DCHECK(!root_index_map_.Lookup(heap_object, &root_index));
305  } else if (root_index_map_.Lookup(heap_object, &root_index)) {
306  return Handle<T>(isolate_->root_handle(root_index).location());
307  }
308  }
309 
310  auto find_result = canonical_handles_->FindOrInsert(object);
311  if (find_result.already_exists) return Handle<T>(*find_result.entry);
312 
313  // Allocate new PersistentHandle if one wasn't created before.
314  if (local_isolate()) {
315  *find_result.entry =
316  local_isolate()->heap()->NewPersistentHandle(object).location();
317  } else {
318  DCHECK(PersistentHandlesScope::IsActive(isolate()));
319  *find_result.entry = IndirectHandle<T>(object, isolate()).location();
320  }
321  return Handle<T>(*find_result.entry);
322  }
323 
324  template <typename T>
325  Handle<T> CanonicalPersistentHandle(Handle<T> object) {
326  if (object.is_null()) return object; // Can't deref a null handle.
327  return CanonicalPersistentHandle<T>(*object);
328  }
329 
330  // Checks if a canonical persistent handle for {object} exists.
331  template <typename T>
332  bool IsCanonicalHandle(Handle<T> handle) {
333  DCHECK_NOT_NULL(canonical_handles_);
334  if (Tagged<HeapObject> heap_object;
335  TryCast<HeapObject>(*handle, &heap_object)) {
336  RootIndex root_index;
337  if (root_index_map_.Lookup(heap_object, &root_index)) {
338  return true;
339  }
340  // Builtins use pseudo handles that are canonical and persistent by
341  // design.
342  if (isolate()->IsBuiltinTableHandleLocation(handle.location())) {
343  return true;
344  }
345  }
346  return canonical_handles_->Find(*handle) != nullptr;
347  }

588 template <class T>
589 OptionalRef<typename ref_traits<T>::ref_type> TryMakeRef(JSHeapBroker* broker,
590  ObjectData* data)
591  requires(is_subtype_v<T, Object>)
592 {
593  if (data == nullptr) return {};
594  return {typename ref_traits<T>::ref_type(data)};

597 // Usage:
598 //
599 // OptionalFooRef ref = TryMakeRef(broker, o);
600 // if (!ref.has_value()) return {}; // bailout
601 //
602 // or
603 //
604 // FooRef ref = MakeRef(broker, o);
605 template <class T>
606 OptionalRef<typename ref_traits<T>::ref_type> TryMakeRef(
607  JSHeapBroker* broker, Tagged<T> object, GetOrCreateDataFlags flags = {})
608  requires(is_subtype_v<T, Object>)
609 {
610  ObjectData* data = broker->TryGetOrCreateData(object, flags);
611  if (data == nullptr) {
612  TRACE_BROKER_MISSING(broker, "ObjectData for " << Brief(object));
613  }
614  return TryMakeRef<T>(broker, data);
615 }
