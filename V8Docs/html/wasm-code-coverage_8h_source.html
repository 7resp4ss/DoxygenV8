20 // Represents a range of code byte offsets in the form [start, end], inclusive
21 // at both ends.
22 // Offsets are calculated from the start of the function wire bytes, not from
23 // the start of the module.
24 struct WasmCodeRange {
25  constexpr WasmCodeRange() = default;
26  constexpr WasmCodeRange(int start, int end) : start(start), end(end) {}
27 
28  constexpr bool operator==(const WasmCodeRange& rhs) const = default;
29 
30  int start = kNoCodePosition;
31  int end = kNoCodePosition;
32 
33  private:
34  static constexpr int kNoCodePosition = -1;
35 };

58 class WasmModuleCoverageData {
59  public:
60  explicit WasmModuleCoverageData(uint32_t declared_function_count) {
61  function_data_ =
62  base::OwnedVector<std::unique_ptr<WasmFunctionCoverageData>>::New(
63  declared_function_count);
64  }
65 
66  WasmFunctionCoverageData* InstallCoverageData(
67  int declared_function_index,
68  base::Vector<const WasmCodeRange> code_ranges) {
69  base::MutexGuard guard{&mutex_};
70 
71  if (!function_data_[declared_function_index]) {
72  function_data_[declared_function_index] =
73  std::make_unique<WasmFunctionCoverageData>(code_ranges);
74  } else {
75  DCHECK_EQ(function_data_[declared_function_index]->code_ranges(),
76  code_ranges);
77  }
78  return function_data_[declared_function_index].get();
79  }
80 
81  size_t function_count() const { return function_data_.size(); }
82 
83  WasmFunctionCoverageData* GetFunctionCoverageData(
84  size_t function_index) const {
85  base::MutexGuard guard{&mutex_};
86 
87  DCHECK_LT(function_index, function_data_.size());
88  return function_data_[function_index].get();
89  }
90 
91  private:
92  // There is a single WasmModuleCoverageData per NativeModule, that can be
93  // accessed concurrently from multiple Isolates.
94  mutable base::Mutex mutex_;
95  base::OwnedVector<std::unique_ptr<WasmFunctionCoverageData>> function_data_;
96 };
