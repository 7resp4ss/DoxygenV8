21 bool TypesUnrelated(wasm::ValueType type1, wasm::ValueType type2,
22  const wasm::WasmModule* module) {
23  return !wasm::IsSubtypeOf(type1, type2, module) &&
24  !wasm::IsSubtypeOf(type2, type1, module);
25 }

27 bool TypesUnrelated(Node* lhs, Node* rhs) {
28  wasm::TypeInModule type1 = NodeProperties::GetType(lhs).AsWasm();
29  wasm::TypeInModule type2 = NodeProperties::GetType(rhs).AsWasm();
30  DCHECK_EQ(type1.module, type2.module);
31  return TypesUnrelated(type1.type, type2.type, type1.module);
32 }

44 bool MayAlias(Node* lhs, Node* rhs) {
45  if (lhs == rhs) return true;
46  if (TypesUnrelated(lhs, rhs) || (IsFresh(lhs) && IsFresh(rhs)) ||
47  (IsFresh(lhs) && IsConstant(rhs)) || (IsConstant(lhs) && IsFresh(rhs))) {
48  return false;
49  }
50  return true;
51 }

53 Node* ResolveAliases(Node* node) {
54  while (node->opcode() == IrOpcode::kWasmTypeCast ||
55  node->opcode() == IrOpcode::kWasmTypeCastAbstract ||
56  node->opcode() == IrOpcode::kAssertNotNull ||
57  node->opcode() == IrOpcode::kTypeGuard) {
58  node = NodeProperties::GetValueInput(node, 0);
59  }
60  return node;

71 Reduction WasmLoadElimination::UpdateState(Node* node,
72  AbstractState const* state) {
73  AbstractState const* original = node_states_.Get(node);
74  // Only signal that the {node} has Changed, if the information about {state}
75  // has changed wrt. the {original}.
76  if (state != original) {
77  if (original == nullptr || !state->Equals(original)) {
78  node_states_.Set(node, state);
79  return Changed(node);
80  }
81  }
82  return NoChange();
83 }

85 std::tuple<Node*, Node*> WasmLoadElimination::TruncateAndExtendOrType(
86  Node* value, Node* effect, Node* control, wasm::ValueType field_type,
87  bool is_signed) {
88  if (field_type == wasm::kWasmI8 || field_type == wasm::kWasmI16) {
89  Node* ret = nullptr;
90  if (is_signed) {
91  int shift = 32 - 8 * field_type.value_kind_size();
92  ret = graph()->NewNode(machine()->Word32Sar(),
93  graph()->NewNode(machine()->Word32Shl(), value,
94  jsgraph()->Int32Constant(shift)),
95  jsgraph()->Int32Constant(shift));
96  } else {
97  int mask = (1 << 8 * field_type.value_kind_size()) - 1;
98  ret = graph()->NewNode(machine()->Word32And(), value,
99  jsgraph()->Int32Constant(mask));
100  }
101 
102  NodeProperties::SetType(ret, NodeProperties::GetType(value));
103  return {ret, effect};
104  }
105 
106  // The value might be untyped in case of wasm inlined into JS if the value
107  // comes from a JS node.
108  if (!NodeProperties::IsTyped(value)) {
109  return {value, effect};
110  }
111 
112  Type value_type = NodeProperties::GetType(value);
113  if (!value_type.IsWasm()) {
114  return {value, effect};
115  }
116 
117  wasm::TypeInModule node_type = value_type.AsWasm();
118 
119  if (TypesUnrelated(node_type.type, field_type, node_type.module)) {
120  // Unrelated types can occur as a result of unreachable code.
121  // Example: Storing a value x of type A in a struct, then casting the struct
122  // to a different struct type to then load type B from the same offset
123  // results in trying to replace the load with value x.
124  return {dead(), dead()};
125  }
126  if (!wasm::IsSubtypeOf(node_type.type, field_type, node_type.module)) {
127  Type type = Type::Wasm({field_type, node_type.module}, graph()->zone());
128  Node* ret =
129  graph()->NewNode(common()->TypeGuard(type), value, effect, control);
130  NodeProperties::SetType(ret, type);
131  return {ret, ret};
132  }
133 
134  return {value, effect};
135 }

139 Reduction WasmLoadElimination::Reduce(Node* node) {
140  if (v8_flags.trace_turbo_load_elimination) {
141  // TODO(manoskouk): Add some tracing.
142  }
143  switch (node->opcode()) {
144  case IrOpcode::kWasmStructGet:
145  return ReduceWasmStructGet(node);
146  case IrOpcode::kWasmStructSet:
147  return ReduceWasmStructSet(node);
148  case IrOpcode::kWasmArrayLength:
149  return ReduceWasmArrayLength(node);
150  case IrOpcode::kWasmArrayInitializeLength:
151  return ReduceWasmArrayInitializeLength(node);
152  case IrOpcode::kStringPrepareForGetCodeunit:
153  return ReduceStringPrepareForGetCodeunit(node);
154  case IrOpcode::kStringAsWtf16:
155  return ReduceStringAsWtf16(node);
156  case IrOpcode::kWasmAnyConvertExtern:
157  return ReduceAnyConvertExtern(node);
158  case IrOpcode::kEffectPhi:
159  return ReduceEffectPhi(node);
160  case IrOpcode::kDead:
161  return NoChange();
162  case IrOpcode::kStart:
163  return ReduceStart(node);
164  default:
165  return ReduceOtherNode(node);
166  }
167 }

169 Reduction WasmLoadElimination::ReduceWasmStructGet(Node* node) {
170  DCHECK_EQ(node->opcode(), IrOpcode::kWasmStructGet);
171  Node* input_struct = NodeProperties::GetValueInput(node, 0);
172  Node* object = ResolveAliases(input_struct);
173  Node* effect = NodeProperties::GetEffectInput(node);
174  Node* control = NodeProperties::GetControlInput(node);
175 
176  if (object->opcode() == IrOpcode::kDead) return NoChange();
177  AbstractState const* state = node_states_.Get(effect);
178  if (state == nullptr) return NoChange();
179 
180  const WasmFieldInfo& field_info = OpParameter<WasmFieldInfo>(node->op());
181  bool is_mutable = field_info.type->mutability(field_info.field_index);
182 
183  if (!NodeProperties::IsTyped(input_struct) ||
184  !NodeProperties::GetType(input_struct).IsWasm()) {
185  // The input should always be typed. https://crbug.com/1507106 reported
186  // that we can end up with Type None here instead of a wasm type.
187  // In the worst case this only means that we miss a potential optimization,
188  // still the assumption is that all inputs into StructGet should be typed.
189  return NoChange();
190  }
191  // Skip reduction if the input type is nullref. in this case, the struct get
192  // will always trap.
193  wasm::ValueType struct_type =
194  NodeProperties::GetType(input_struct).AsWasm().type;
195  if (struct_type == wasm::kWasmNullRef) {
196  return NoChange();
197  }
198  // The node is in unreachable code if its input is uninhabitable (bottom or
199  // ref none type). It can also be treated as unreachable if the field index is
200  // in the wrong half state. This can happen if an object gets cast to two
201  // unrelated types subsequently (as the state only tracks the field index)
202  // independent of the underlying type.
203  if (struct_type.is_uninhabited() ||
204  !(is_mutable ? state->immutable_state : state->mutable_state)
205  .LookupField(field_info.field_index, object)
206  .IsEmpty()) {
207  ReplaceWithValue(node, dead(), dead(), dead());
208  MergeControlToEnd(graph(), common(),
209  graph()->NewNode(common()->Throw(), effect, control));
210  node->Kill();
211  return Replace(dead());
212  }
213  // If the input type is not (ref null? none) or bottom and we don't have type
214  // inconsistencies, then the result type must be valid.
215  DCHECK(!NodeProperties::GetType(node).AsWasm().type.is_bottom());
216 
217  HalfState const* half_state =
218  is_mutable ? &state->mutable_state : &state->immutable_state;
219 
220  FieldOrElementValue lookup_result =
221  half_state->LookupField(field_info.field_index, object);
222 
223  if (!lookup_result.IsEmpty() && !lookup_result.value->IsDead()) {
224  std::tuple<Node*, Node*> replacement = TruncateAndExtendOrType(
225  lookup_result.value, effect, control,
226  field_info.type->field(field_info.field_index), field_info.is_signed);
227  if (std::get<0>(replacement) == dead()) {
228  // If the value is dead (unreachable), this whole code path is unreachable
229  // and we can mark this control flow path as dead.
230  ReplaceWithValue(node, dead(), dead(), dead());
231  MergeControlToEnd(graph(), common(),
232  graph()->NewNode(common()->Throw(), effect, control));
233  node->Kill();
234  return Replace(dead());
235  }
236  ReplaceWithValue(node, std::get<0>(replacement), std::get<1>(replacement),
237  control);
238  node->Kill();
239  return Replace(std::get<0>(replacement));
240  }
241 
242  half_state = half_state->AddField(field_info.field_index, object, node);
243 
244  AbstractState const* new_state =
245  is_mutable
246  ? zone()->New<AbstractState>(*half_state, state->immutable_state)
247  : zone()->New<AbstractState>(state->mutable_state, *half_state);
248 
249  return UpdateState(node, new_state);
250 }

252 Reduction WasmLoadElimination::ReduceWasmStructSet(Node* node) {
253  DCHECK_EQ(node->opcode(), IrOpcode::kWasmStructSet);
254  Node* input_struct = NodeProperties::GetValueInput(node, 0);
255  Node* object = ResolveAliases(input_struct);
256  Node* value = NodeProperties::GetValueInput(node, 1);
257  Node* effect = NodeProperties::GetEffectInput(node);
258  Node* control = NodeProperties::GetControlInput(node);
259 
260  if (object->opcode() == IrOpcode::kDead) return NoChange();
261  AbstractState const* state = node_states_.Get(effect);
262  if (state == nullptr) return NoChange();
263 
264  if (!NodeProperties::IsTyped(input_struct) ||
265  !NodeProperties::GetType(input_struct).IsWasm()) {
266  // Also see the same pattern in ReduceWasmStructGet. Note that this is
267  // reached for cases where the StructSet has a value input that is
268  // DeadValue(). Above we check for `object->opcode() == IrOpcode::kDead.
269  // As an alternative that check could be extended to also check for
270  // ... || object->opcode() == IrOpcode::kDeadValue.
271  // It seems that the DeadValue may be caused by
272  // DeadCodeElimination::ReducePureNode. If that finds any input that is a
273  // Dead() node, it will replace that input with a DeadValue().
274  return NoChange();
275  }
276 
277  // Skip reduction if the input type is nullref. in this case, the struct get
278  // will always trap.
279  wasm::ValueType struct_type =
280  NodeProperties::GetType(input_struct).AsWasm().type;
281  if (struct_type == wasm::kWasmNullRef) {
282  return NoChange();
283  }
284 
285  const WasmFieldInfo& field_info = OpParameter<WasmFieldInfo>(node->op());
286  bool is_mutable = field_info.type->mutability(field_info.field_index);
287 
288  // The struct.set is unreachable if its input struct is an uninhabitable type.
289  // It can also be treated as unreachable if the field index is in the wrong
290  // half state. This can happen if an object gets cast to two unrelated types
291  // subsequently (as the state only tracks the field index) independent of the
292  // underlying type.
293  if (struct_type.is_uninhabited() ||
294  !(is_mutable ? state->immutable_state : state->mutable_state)
295  .LookupField(field_info.field_index, object)
296  .IsEmpty()) {
297  ReplaceWithValue(node, dead(), dead(), dead());
298  MergeControlToEnd(graph(), common(),
299  graph()->NewNode(common()->Throw(), effect, control));
300  node->Kill();
301  return Replace(dead());
302  }
303 
304  if (is_mutable) {
305  HalfState const* mutable_state =
306  state->mutable_state.KillField(field_info.field_index, object);
307  mutable_state =
308  mutable_state->AddField(field_info.field_index, object, value);
309  AbstractState const* new_state =
310  zone()->New<AbstractState>(*mutable_state, state->immutable_state);
311  return UpdateState(node, new_state);
312  } else {
313  // We should not initialize the same immutable field twice.
314  DCHECK(state->immutable_state.LookupField(field_info.field_index, object)
315  .IsEmpty());
316  HalfState const* immutable_state =
317  state->immutable_state.AddField(field_info.field_index, object, value);
318  AbstractState const* new_state =
319  zone()->New<AbstractState>(state->mutable_state, *immutable_state);
320  return UpdateState(node, new_state);
321  }
322 }

324 Reduction WasmLoadElimination::ReduceLoadLikeFromImmutable(Node* node,
325  int index) {
326  // The index must be negative as it is not a real load, to not confuse it with
327  // actual loads.
328  DCHECK_LT(index, 0);
329  Node* object = ResolveAliases(NodeProperties::GetValueInput(node, 0));
330  Node* effect = NodeProperties::GetEffectInput(node);
331  Node* control = NodeProperties::GetControlInput(node);
332 
333  if (object->opcode() == IrOpcode::kDead) return NoChange();
334  AbstractState const* state = node_states_.Get(effect);
335  if (state == nullptr) return NoChange();
336 
337  HalfState const* immutable_state = &state->immutable_state;
338 
339  FieldOrElementValue lookup_result =
340  immutable_state->LookupField(index, object);
341 
342  if (!lookup_result.IsEmpty() && !lookup_result.value->IsDead()) {
343  ReplaceWithValue(node, lookup_result.value, effect, control);
344  node->Kill();
345  return Replace(lookup_result.value);
346  }
347 
348  immutable_state = immutable_state->AddField(index, object, node);
349 
350  AbstractState const* new_state =
351  zone()->New<AbstractState>(state->mutable_state, *immutable_state);
352 
353  return UpdateState(node, new_state);
354 }

381 Reduction WasmLoadElimination::ReduceStringPrepareForGetCodeunit(Node* node) {
382  DCHECK_EQ(node->opcode(), IrOpcode::kStringPrepareForGetCodeunit);
383  Node* object = ResolveAliases(NodeProperties::GetValueInput(node, 0));
384  Node* effect = NodeProperties::GetEffectInput(node);

412  zone()->New<AbstractState>(*mutable_state, state->immutable_state);

430 Reduction WasmLoadElimination::ReduceOtherNode(Node* node) {
431  if (node->op()->EffectOutputCount() == 0) return NoChange();
432  DCHECK_EQ(node->op()->EffectInputCount(), 1);
433  Node* const effect = NodeProperties::GetEffectInput(node);
434  AbstractState const* state = node_states_.Get(effect);
435  // If we do not know anything about the predecessor, do not propagate just
436  // yet because we will have to recompute anyway once we compute the
437  // predecessor.
438  if (state == nullptr) return NoChange();
439  // If this {node} has some uncontrolled side effects (i.e. it is a call
440  // without {kNoWrite}), set its state to the immutable half-state of its
441  // input state, otherwise to its input state.
442  // Any cached StringPrepareForGetCodeUnit nodes must be killed at any point
443  // that can cause internalization of strings (i.e. that can turn sequential
444  // strings into thin strings). Currently, that can only happen in JS, so
445  // from Wasm's point of view only in calls.
446  return UpdateState(node, node->opcode() == IrOpcode::kCall &&
447  !node->op()->HasProperty(Operator::kNoWrite)
448  ? zone()->New<AbstractState>(
449  HalfState(zone()), state->immutable_state)
450  : state);
451 }

457 Reduction WasmLoadElimination::ReduceEffectPhi(Node* node) {
458  DCHECK_EQ(node->opcode(), IrOpcode::kEffectPhi);
459  Node* const effect0 = NodeProperties::GetEffectInput(node, 0);
460  Node* const control = NodeProperties::GetControlInput(node);
461  AbstractState const* state0 = node_states_.Get(effect0);
462  if (state0 == nullptr) return NoChange();
463  if (control->opcode() == IrOpcode::kLoop) {
464  // Here we rely on having only reducible loops:
465  // The loop entry edge always dominates the header, so we can just take
466  // the state from the first input, and compute the loop state based on it.
467  AbstractState const* state = ComputeLoopState(node, state0);
468  return UpdateState(node, state);
469  }
470  DCHECK_EQ(IrOpcode::kMerge, control->opcode());
471 
472  // Shortcut for the case when we do not know anything about some input.
473  int const input_count = node->op()->EffectInputCount();
474  for (int i = 1; i < input_count; ++i) {
475  Node* const effect = NodeProperties::GetEffectInput(node, i);
476  if (node_states_.Get(effect) == nullptr) return NoChange();
477  }
478 
479  // Make a copy of the first input's state and intersect it with the state
480  // from other inputs.
481  // TODO(manoskouk): Consider computing phis for at least a subset of the
482  // state.
483  AbstractState* state = zone()->New<AbstractState>(*state0);
484  for (int i = 1; i < input_count; ++i) {
485  Node* const input = NodeProperties::GetEffectInput(node, i);
486  state->IntersectWith(node_states_.Get(input));
487  }
488  return UpdateState(node, state);
489 }

506 WasmLoadElimination::HalfState const* WasmLoadElimination::HalfState::KillField(
507  int field_index, Node* object) const {
508  const InnerMap& same_index_map = fields_.Get(field_index);
509  InnerMap new_map(same_index_map);
510  for (std::pair<Node*, FieldOrElementValue> pair : same_index_map) {

514  }
515  HalfState* result = zone_->New<HalfState>(*this);
516  result->fields_.Set(field_index, new_map);

520 WasmLoadElimination::AbstractState const* WasmLoadElimination::ComputeLoopState(
521  Node* node, AbstractState const* state) const {
522  DCHECK_EQ(node->opcode(), IrOpcode::kEffectPhi);
523  if (state->mutable_state.IsEmpty()) return state;
524  std::queue<Node*> queue;
525  AccountingAllocator allocator;
526  Zone temp_set_zone(&allocator, ZONE_NAME);
527  ZoneUnorderedSet<Node*> visited(&temp_set_zone);
528  visited.insert(node);
529  for (int i = 1; i < node->InputCount() - 1; ++i) {
530  queue.push(node->InputAt(i));
531  }
532  while (!queue.empty()) {
533  Node* const current = queue.front();
534  queue.pop();
535  if (visited.insert(current).second) {
536  if (current->opcode() == IrOpcode::kWasmStructSet) {
537  Node* object = NodeProperties::GetValueInput(current, 0);
538  if (object->opcode() == IrOpcode::kDead ||
539  object->opcode() == IrOpcode::kDeadValue) {
540  // We are in dead code. Bail out with no mutable state.
541  return zone()->New<AbstractState>(HalfState(zone()),
542  state->immutable_state);
543  }
544  WasmFieldInfo field_info = OpParameter<WasmFieldInfo>(current->op());
545  bool is_mutable = field_info.type->mutability(field_info.field_index);
546  if (is_mutable) {
547  const HalfState* new_mutable_state =
548  state->mutable_state.KillField(field_info.field_index, object);
549  state = zone()->New<AbstractState>(*new_mutable_state,
550  state->immutable_state);
551  } else {
552  // TODO(manoskouk): DCHECK
553  }
554  } else if (current->opcode() == IrOpcode::kCall &&
555  !current->op()->HasProperty(Operator::kNoWrite)) {
556  return zone()->New<AbstractState>(HalfState(zone()),
557  state->immutable_state);
558  }
559  for (int i = 0; i < current->op()->EffectInputCount(); ++i) {
560  queue.push(NodeProperties::GetEffectInput(current, i));
561  }
