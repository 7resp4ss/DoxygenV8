23 class ManagedZones final {
24  public:
25  static std::optional<VirtualMemory> GetOrCreateMemoryForSegment(
26  Isolate* isolate, size_t bytes) {
27  DCHECK_EQ(0, bytes % kMinZonePageSize);
28  // Only consult the pool if we have an isolate and the size is exactly the
29  // zone page size. Larger sizes may be required and just bypass the pool.
30  if (isolate && bytes == kMinZonePageSize) {
31  auto maybe_reservation =
32  IsolateGroup::current()->page_pool()->RemoveZoneReservation(isolate);
33  if (maybe_reservation) {
34  return maybe_reservation;
35  }
36  }
37 
38  v8::PageAllocator* platform_page_allocator = GetPlatformPageAllocator();
39  VirtualMemory memory(platform_page_allocator, bytes,
40  v8::PageAllocator::AllocationHint(),
41  kMinZonePageAlignment, v8::PageAllocator::kReadWrite);
42  if (V8_UNLIKELY(!memory.IsReserved())) {
43  return std::nullopt;
44  }
45  CHECK(IsAligned(memory.address(), kMinZonePageAlignment));
46  return memory;
47  }
48 
49  static base::AllocationResult<void*> AllocateSegment(Isolate* isolate,
50  size_t bytes) {
51  static constexpr size_t kMaxSize = size_t{2} * GB;
52  if (bytes >= kMaxSize) {
53  return {nullptr, bytes};
54  }
55  void* memory = nullptr;
56  bytes = RoundUp(bytes, ManagedZones::kMinZonePageSize);
57  std::optional<VirtualMemory> maybe_reservation =
58  ManagedZones::GetOrCreateMemoryForSegment(isolate, bytes);
59  if (V8_LIKELY(maybe_reservation)) {
60  VirtualMemory reservation = std::move(maybe_reservation.value());
61  DCHECK(reservation.IsReserved());
62  DCHECK_EQ(reservation.size(), bytes);
63  memory = reinterpret_cast<void*>(reservation.address());
64  // Don't let the reservation be freed by destructor. From now on the
65  // reservation will be managed via a pair {memory, bytes}.
66  reservation.Reset();
67  }
68  return {memory, bytes};
69  }
70 
71  static void ReturnSegment(Isolate* isolate, void* memory, size_t bytes) {
72  VirtualMemory reservation(GetPlatformPageAllocator(),
73  reinterpret_cast<Address>(memory), bytes);
74  if (reservation.size() == ManagedZones::kMinZonePageSize) {
75  IsolateGroup::current()->page_pool()->AddZoneReservation(
76  isolate, std::move(reservation));
77  }
78  // Reservation will be automatically freed here otherwise.
79  }
80 
81  private:
82  static constexpr size_t kMinZonePageSize = 512 * KB;
83  static constexpr size_t kMinZonePageAlignment = 16 * KB;
84 };

95 Segment* AccountingAllocator::AllocateSegment(size_t requested_bytes) {
96  base::AllocationResult<void*> memory;
97  if (v8_flags.managed_zone_memory && isolate_) {
98  memory = ManagedZones::AllocateSegment(isolate_, requested_bytes);
99  } else {
100  memory = AllocAtLeastWithRetry(requested_bytes);
101  }
102  if (V8_UNLIKELY(memory.ptr == nullptr)) {
103  return nullptr;
104  }
105 
106  size_t current =
107  current_memory_usage_.fetch_add(memory.count, std::memory_order_relaxed) +
108  memory.count;
109  size_t max = max_memory_usage_.load(std::memory_order_relaxed);
110  while (current > max && !max_memory_usage_.compare_exchange_weak(
111  max, current, std::memory_order_relaxed)) {
112  // {max} was updated by {compare_exchange_weak}; retry.
113  }
114  DCHECK_LE(sizeof(Segment), memory.count);
115  return new (memory.ptr) Segment(memory.count);
116 }

118 void AccountingAllocator::ReturnSegment(Segment* segment) {
119  segment->ZapContents();
120  size_t segment_size = segment->total_size();
121  current_memory_usage_.fetch_sub(segment_size, std::memory_order_relaxed);
122  segment->ZapHeader();
123  if (isolate_ && v8_flags.managed_zone_memory) {
124  ManagedZones::ReturnSegment(isolate_, segment, segment_size);
125  } else {
126  free(segment);
127  }
128 }
