49 bool RootsTable::IsRootHandleLocation(Address* handle_location,
50  RootIndex* index) const {
51  FullObjectSlot location(handle_location);
52  FullObjectSlot first_root(&roots_[0]);
53  FullObjectSlot last_root(&roots_[kEntriesCount]);
54  if (location >= last_root) return false;
55  if (location < first_root) return false;
56  *index = static_cast<RootIndex>(location - first_root);
57  return true;
58 }

60 template <typename T>
61 bool RootsTable::IsRootHandle(IndirectHandle<T> handle,
62  RootIndex* index) const {
63  // This can't use handle.location() because it is called from places
64  // where handle dereferencing is disallowed. Comparing the handle's
65  // location against the root handle list is safe though.
66  Address* handle_location = reinterpret_cast<Address*>(handle.address());
67  return IsRootHandleLocation(handle_location, index);
68 }

147 Address ReadOnlyRoots::address_at(RootIndex root_index) const {
148 #if V8_STATIC_ROOTS_BOOL
149  return V8HeapCompressionScheme::DecompressTagged(
150  StaticReadOnlyRootsPointerTable[static_cast<int>(root_index)]);
151 #else
152  size_t index = static_cast<size_t>(root_index);
153  DCHECK_LT(index, kEntriesCount);
154  // Filler objects must be created before the free space map is initialized.
155  // Bootstrapping is able to handle kNullAddress being returned here.
156  DCHECK_IMPLIES(read_only_roots_[index] == kNullAddress,
157  root_index == RootIndex::kFreeSpaceMap);
158  return read_only_roots_[index];
159 #endif
160 }
