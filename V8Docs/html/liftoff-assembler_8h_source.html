306  // Returns whether this was the last use.
307  void dec_used(LiftoffRegister reg) {
308  // Note that we do not DCHECK(!frozen) here due to a special case: When
309  // performign a call_indirect, we first create an OOL trap label (which
310  // freezes the state to make sure that the safe point table remains valid)
311  // and then we drop values (which doesn't invalidate safe point table, so
312  // it is actually fine to do it.)
313  DCHECK(is_used(reg));
314  if (reg.is_pair()) {
315  dec_used(reg.low());
316  dec_used(reg.high());
317  return;
318  }
319  int code = reg.liftoff_code();
320  DCHECK_LT(0, register_use_count[code]);
321  if (--register_use_count[code] == 0) used_registers.clear(reg);
322  }

452  // Returns the register which holds the value of stack slot {index}. If the
453  // value is not stored in a register yet, a register is allocated for it. The
454  // register is then assigned to the stack slot. The value stack height is not
455  // modified. The top of the stack is index 0, i.e. {PopToRegister()} and
456  // {PeekToRegister(0)} should result in the same register.
457  // When the value is finally popped, the use counter of its register has to be
458  // decremented. This can be done by popping the value with {DropValues}.
459  LiftoffRegister PeekToRegister(int index, LiftoffRegList pinned) {
460  DCHECK_LT(index, cache_state_.stack_state.size());
461  VarState& slot = cache_state_.stack_state.end()[-1 - index];
462  if (V8_LIKELY(slot.is_reg())) return slot.reg();
463  LiftoffRegister reg = LoadToRegister(slot, pinned);
464  cache_state_.inc_used(reg);
465  slot.MakeRegister(reg);
466  return reg;
467  }
