18 DirectHandle<JSRegExpResultIndices> JSRegExpResultIndices::BuildIndices(
19  Isolate* isolate, DirectHandle<RegExpMatchInfo> match_info,
20  DirectHandle<Object> maybe_names) {
21  DirectHandle<JSRegExpResultIndices> indices(
22  Cast<JSRegExpResultIndices>(isolate->factory()->NewJSObjectFromMap(
23  isolate->regexp_result_indices_map())));
24 
25  // Initialize indices length to avoid having a partially initialized object
26  // should GC be triggered by creating a NewFixedArray.
27  indices->set_length(Smi::zero());
28 
29  // Build indices array from RegExpMatchInfo.
30  int num_indices = match_info->number_of_capture_registers();
31  int num_results = num_indices >> 1;
32  DirectHandle<FixedArray> indices_array =
33  isolate->factory()->NewFixedArray(num_results);
34  JSArray::SetContent(isolate, indices, indices_array);
35 
36  for (int i = 0; i < num_results; i++) {
37  const int start_offset =
38  match_info->capture(RegExpMatchInfo::capture_start_index(i));
39  const int end_offset =
40  match_info->capture(RegExpMatchInfo::capture_end_index(i));
41 
42  // Any unmatched captures are set to undefined, otherwise we set them to a
43  // subarray of the indices.
44  if (start_offset == -1) {
45  indices_array->set(i, ReadOnlyRoots(isolate).undefined_value());
46  } else {
47  DirectHandle<FixedArray> indices_sub_array(
48  isolate->factory()->NewFixedArray(2));
49  indices_sub_array->set(0, Smi::FromInt(start_offset));
50  indices_sub_array->set(1, Smi::FromInt(end_offset));
51  DirectHandle<JSArray> indices_sub_jsarray =
52  isolate->factory()->NewJSArrayWithElements(indices_sub_array,
53  PACKED_SMI_ELEMENTS, 2);
54  indices_array->set(i, *indices_sub_jsarray);
55  }
56  }
57 
58  // If there are no capture groups, set the groups property to undefined.
59  FieldIndex groups_index = FieldIndex::ForDescriptor(
60  indices->map(), InternalIndex(kGroupsDescriptorIndex));
61  if (IsUndefined(*maybe_names, isolate)) {
62  indices->FastPropertyAtPut(groups_index,
63  ReadOnlyRoots(isolate).undefined_value());
64  return indices;
65  }
66 
67  // Create a groups property which returns a dictionary of named captures to
68  // their corresponding capture indices.
69  auto names = Cast<FixedArray>(maybe_names);
70  int num_names = names->length() >> 1;
71  DirectHandle<HeapObject> group_names;
72  if constexpr (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
73  group_names = isolate->factory()->NewSwissNameDictionary(num_names);
74  } else {
75  group_names = isolate->factory()->NewNameDictionary(num_names);
76  }
77  DirectHandle<PropertyDictionary> group_names_dict =
78  Cast<PropertyDictionary>(group_names);
79  for (int i = 0; i < num_names; i++) {
80  int base_offset = i * 2;
81  int name_offset = base_offset;
82  int index_offset = base_offset + 1;
83  DirectHandle<String> name(Cast<String>(names->get(name_offset)), isolate);
84  Tagged<Smi> smi_index = Cast<Smi>(names->get(index_offset));
85  DirectHandle<Object> capture_indices(indices_array->get(smi_index.value()),
86  isolate);
87  if (!IsUndefined(*capture_indices, isolate)) {
88  capture_indices = Cast<JSArray>(capture_indices);
89  }
90  InternalIndex group_entry = group_names_dict->FindEntry(isolate, name);
91  // Duplicate group entries are possible if the capture groups are in
92  // different alternatives, i.e. only one of them can actually match.
93  // Therefore when we find a duplicate entry, either the current entry is
94  // undefined (didn't match anything) or the indices for the current capture
95  // are undefined. In the latter case we don't do anything, in the former
96  // case we update the entry.
97  if (group_entry.is_found()) {
98  DCHECK(v8_flags.js_regexp_duplicate_named_groups);
99  if (!IsUndefined(*capture_indices, isolate)) {
100  DCHECK(IsUndefined(group_names_dict->ValueAt(group_entry), isolate));
101  group_names_dict->ValueAtPut(group_entry, *capture_indices);
102  }
103  } else {
104  group_names_dict =
105  PropertyDictionary::Add(isolate, group_names_dict, name,
106  capture_indices, PropertyDetails::Empty());
107  }
108  }
109 
110  // Convert group_names to a JSObject and store at the groups property of the
111  // result indices.
112  DirectHandle<FixedArrayBase> elements =
113  isolate->factory()->empty_fixed_array();
114  DirectHandle<Null> null = isolate->factory()->null_value();
115  DirectHandle<JSObject> js_group_names =
116  isolate->factory()->NewSlowJSObjectWithPropertiesAndElements(
117  null, group_names, elements);
118  indices->FastPropertyAtPut(groups_index, *js_group_names);
119  return indices;
120 }

123 std::optional<JSRegExp::Flags> JSRegExp::FlagsFromString(
124  Isolate* isolate, DirectHandle<String> flags) {
125  const int length = flags->length();
126 
127  // A longer flags string cannot be valid.
128  if (length > JSRegExp::kFlagCount) return {};
129 
130  RegExpFlags value;
131  FlatStringReader reader(isolate, String::Flatten(isolate, flags));
132 
133  for (int i = 0; i < length; i++) {
134  std::optional<RegExpFlag> flag = JSRegExp::FlagFromChar(reader.Get(i));
135  if (!flag.has_value()) return {};
136  if (value & flag.value()) return {}; // Duplicate.
137  value |= flag.value();
138  }
139 
140  return JSRegExp::AsJSRegExpFlags(value);
141 }

188 // TODO(jgruber): Consider merging CountAdditionalEscapeChars and
189 // WriteEscapedRegExpSource into a single function to deduplicate dispatch logic
190 // and move related code closer to each other.
191 template <typename Char>
192 uint32_t CountAdditionalEscapeChars(DirectHandle<String> source,
193  bool* needs_escapes_out) {
194  DisallowGarbageCollection no_gc;
195  uint32_t escapes = 0;
196  // The maximum growth-factor is 5 (for \u2028 and \u2029). Make sure that we
197  // won't overflow |escapes| given the current constraints on string length.
198  static_assert(uint64_t{String::kMaxLength} * 5 <
199  std::numeric_limits<decltype(escapes)>::max());
200  bool needs_escapes = false;
201  bool in_character_class = false;
202  base::Vector<const Char> src = source->GetCharVector<Char>(no_gc);
203  for (int i = 0; i < src.length(); i++) {
204  const Char c = src[i];
205  if (c == '\\') {
206  if (i + 1 < src.length() && IsLineTerminator(src[i + 1])) {
207  // This '\' is ignored since the next character itself will be escaped.
208  escapes--;
209  } else {
210  // Escape. Skip next character, which will be copied verbatim;
211  i++;
212  }
213  } else if (c == '/' && !in_character_class) {
214  // Not escaped forward-slash needs escape.
215  needs_escapes = true;
216  escapes++;
217  } else if (c == '[') {
218  in_character_class = true;
219  } else if (c == ']') {
220  in_character_class = false;
221  } else if (c == '\n') {
222  needs_escapes = true;
223  escapes++;
224  } else if (c == '\r') {
225  needs_escapes = true;
226  escapes++;
227  } else if (static_cast<int>(c) == 0x2028) {
228  needs_escapes = true;
229  escapes += std::strlen("\\u2028") - 1;
230  } else if (static_cast<int>(c) == 0x2029) {
231  needs_escapes = true;
232  escapes += std::strlen("\\u2029") - 1;
233  } else {
234  DCHECK(!IsLineTerminator(c));
235  }
236  }
237  DCHECK(!in_character_class);
238  DCHECK_IMPLIES(escapes != 0, needs_escapes);
239  *needs_escapes_out = needs_escapes;
240  return escapes;
241 }

250 template <typename Char, typename StringType>
251 DirectHandle<StringType> WriteEscapedRegExpSource(
252  DirectHandle<String> source, DirectHandle<StringType> result) {
253  DisallowGarbageCollection no_gc;
254  base::Vector<const Char> src = source->GetCharVector<Char>(no_gc);
255  base::Vector<Char> dst(result->GetChars(no_gc), result->length());
256  uint32_t s = 0;
257  uint32_t d = 0;
258  bool in_character_class = false;
259  while (s < src.size()) {
260  const Char c = src[s];
261  if (c == '\\') {
262  if (s + 1 < src.size() && IsLineTerminator(src[s + 1])) {
263  // This '\' is ignored since the next character itself will be escaped.
264  s++;
265  continue;
266  } else {
267  // Escape. Copy this and next character.
268  dst[d++] = src[s++];
269  }
270  if (s == src.size()) break;
271  } else if (c == '/' && !in_character_class) {
272  // Not escaped forward-slash needs escape.
273  dst[d++] = '\\';
274  } else if (c == '[') {
275  in_character_class = true;
276  } else if (c == ']') {
277  in_character_class = false;
278  } else if (c == '\n') {
279  WriteStringToCharVector(dst, &d, "\\n");
280  s++;
281  continue;
282  } else if (c == '\r') {
283  WriteStringToCharVector(dst, &d, "\\r");
284  s++;
285  continue;
286  } else if (static_cast<int>(c) == 0x2028) {
287  WriteStringToCharVector(dst, &d, "\\u2028");
288  s++;
289  continue;
290  } else if (static_cast<int>(c) == 0x2029) {
291  WriteStringToCharVector(dst, &d, "\\u2029");
292  s++;
293  continue;
294  } else {
295  DCHECK(!IsLineTerminator(c));
296  }
297  dst[d++] = src[s++];
298  }
299  DCHECK_EQ(result->length(), d);
300  DCHECK(!in_character_class);
301  return result;
302 }

304 MaybeDirectHandle<String> EscapeRegExpSource(Isolate* isolate,
305  DirectHandle<String> source) {
306  DCHECK(source->IsFlat());
307  if (source->length() == 0) return isolate->factory()->query_colon_string();
308  bool one_byte = String::IsOneByteRepresentationUnderneath(*source);
309  bool needs_escapes = false;
310  uint32_t additional_escape_chars =
311  one_byte ? CountAdditionalEscapeChars<uint8_t>(source, &needs_escapes)
312  : CountAdditionalEscapeChars<base::uc16>(source, &needs_escapes);
313  if (!needs_escapes) return source;
314  DCHECK_LE(static_cast<uint64_t>(source->length()) + additional_escape_chars,
315  std::numeric_limits<uint32_t>::max());
316  uint32_t length = source->length() + additional_escape_chars;
317  if (one_byte) {
318  DirectHandle<SeqOneByteString> result;
319  ASSIGN_RETURN_ON_EXCEPTION(isolate, result,
320  isolate->factory()->NewRawOneByteString(length));
321  return WriteEscapedRegExpSource<uint8_t>(source, result);
322  } else {
323  DirectHandle<SeqTwoByteString> result;
324  ASSIGN_RETURN_ON_EXCEPTION(isolate, result,
325  isolate->factory()->NewRawTwoByteString(length));
326  return WriteEscapedRegExpSource<base::uc16>(source, result);
327  }
328 }

332 // static
333 MaybeDirectHandle<JSRegExp> JSRegExp::Initialize(Isolate* isolate,
334  DirectHandle<JSRegExp> regexp,
335  DirectHandle<String> source,
336  Flags flags,
337  uint32_t backtrack_limit) {
338  Factory* factory = isolate->factory();
339  // If source is the empty string we set it to "(?:)" instead as
340  // suggested by ECMA-262, 5th, section 15.10.4.1.
341  if (source->length() == 0) source = factory->query_colon_string();
342 
343  source = String::Flatten(isolate, source);
344 
345  RETURN_ON_EXCEPTION(isolate, RegExp::Compile(isolate, regexp, source,
346  JSRegExp::AsRegExpFlags(flags),
347  backtrack_limit));
348 
349  DirectHandle<String> escaped_source;
350  ASSIGN_RETURN_ON_EXCEPTION(isolate, escaped_source,
351  EscapeRegExpSource(isolate, source));
352 
353  regexp->set_source(*escaped_source);
354  regexp->set_flags(Smi::FromInt(flags));
355 
356  Tagged<Map> map = regexp->map();
357  Tagged<Object> constructor = map->GetConstructor();
358  if (IsJSFunction(constructor) &&
359  Cast<JSFunction>(constructor)->initial_map() == map) {
360  // If we still have the original map, set in-object properties directly.
361  regexp->InObjectPropertyAtPut(JSRegExp::kLastIndexFieldIndex,
362  Smi::FromInt(kInitialLastIndexValue),
363  SKIP_WRITE_BARRIER);
364  } else {
365  // Map has changed, so use generic, but slower, method.
366  RETURN_ON_EXCEPTION(
367  isolate,
368  Object::SetProperty(
369  isolate, regexp, factory->lastIndex_string(),
370  DirectHandle<Smi>(Smi::FromInt(kInitialLastIndexValue), isolate)));
371  }
372 
373  return regexp;
374 }
