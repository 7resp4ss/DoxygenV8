137  // A transition group is a target and a possibly empty set of sources that can
138  // transition to the target. It is represented as a non-empty vector with the
139  // target at index 0.
140  using TransitionGroup = ZoneVector<MapRef>;
141  ZoneVector<TransitionGroup> const& transition_groups() const;
142 
143  bool HasOnlyStringMaps(JSHeapBroker* broker) const;
144 
145  void AddGroup(TransitionGroup&& group);
146 
147  // Refine {this} by trying to restrict it to the maps in {inferred_maps}. A
148  // transition group's target is kept iff it is in {inferred_maps} or if more
149  // than one of its sources is in {inferred_maps}. Here's an (unrealistic)
150  // example showing all the possible situations:
151  //
152  // inferred_maps = [a0, a2, c1, c2, d1, e0, e1]
153  //
154  // Groups before: Groups after:
155  // [a0, a1, a2] [a0, a2]
156  // [b0]
157  // [c0, c1, c2, c3] [c0, c1, c2]
158  // [d0, d1] [d1]
159  // [e0, e1] [e0, e1]
160  //
161  ElementAccessFeedback const& Refine(
162  JSHeapBroker* broker, ZoneVector<MapRef> const& inferred_maps) const;
163  ElementAccessFeedback const& Refine(
164  JSHeapBroker* broker, ZoneRefSet<Map> const& inferred_maps,
165  bool always_keep_group_target = true) const;
166  NamedAccessFeedback const& Refine(JSHeapBroker* broker, NameRef name) const;
167 
168  private:
169  KeyedAccessMode const keyed_mode_;
170  ZoneVector<TransitionGroup> transition_groups_;
171 };

225 template <class T, ProcessedFeedback::Kind K>
226 class SingleValueFeedback : public ProcessedFeedback {
227  public:
228  explicit SingleValueFeedback(T value, FeedbackSlotKind slot_kind)
229  : ProcessedFeedback(K, slot_kind), value_(value) {
230  DCHECK(
231  (K == kBinaryOperation && slot_kind == FeedbackSlotKind::kBinaryOp) ||
232  (K == kBinaryOperation &&
233  slot_kind == FeedbackSlotKind::kStringAddAndInternalize) ||
234  (K == kTypeOf && slot_kind == FeedbackSlotKind::kTypeOf) ||
235  (K == kCompareOperation && slot_kind == FeedbackSlotKind::kCompareOp) ||
236  (K == kForIn && slot_kind == FeedbackSlotKind::kForIn) ||
237  (K == kInstanceOf && slot_kind == FeedbackSlotKind::kInstanceOf) ||
238  ((K == kLiteral || K == kRegExpLiteral || K == kTemplateObject) &&
239  slot_kind == FeedbackSlotKind::kLiteral));
240  }
241 
242  T value() const { return value_; }
243 
244  private:
245  T const value_;
246 };
