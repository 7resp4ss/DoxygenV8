7 // This file implements the support code for the out of bounds trap handler.
8 // Nothing in here actually runs in the trap handler, but the code here
9 // manipulates data structures used by the trap handler so we still need to be
10 // careful. In order to minimize this risk, here are some rules to follow.
11 //
12 // 1. Avoid introducing new external dependencies. The files in src/trap-handler
13 // should be as self-contained as possible to make it easy to audit the code.
14 //
15 // 2. Any changes must be reviewed by someone from the crash reporting
16 // or security team. See OWNERS for suggested reviewers.
17 //
18 // For more information, see:
19 // https://docs.google.com/document/d/17y4kxuHFrVxAiuCP_FFtFA2HP5sNPsCD10KEx17Hz6M
20 //
21 // For the code that runs in the trap handler itself, see handler-inside.cc.

64 // Verify that the code range does not overlap any that have already been
65 // registered.
66 void VerifyCodeRangeIsDisjoint(const CodeProtectionInfo* code_info) {
67  for (size_t i = 0; i < gNumCodeObjects; ++i) {
68  TH_DCHECK(IsDisjoint(code_info, gCodeObjects[i].code_info));

72 void ValidateCodeObjects() {
73  // Sanity-check the code objects
74  for (unsigned i = 0; i < gNumCodeObjects; ++i) {
75  const auto* data = gCodeObjects[i].code_info;
76 
77  if (data == nullptr) continue;
78 
79  // Do some sanity checks on the protected instruction data
80  for (unsigned j = 0; j < data->num_protected_instructions; ++j) {
81  TH_DCHECK(data->instructions[j].instr_offset >= 0);
82  TH_DCHECK(data->instructions[j].instr_offset < data->size);
83  }
84  }
85 
86  // Check the validity of the free list.
87 #ifdef DEBUG
88  size_t free_count = 0;
89  for (size_t i = gNextCodeObject; i != gNumCodeObjects;
90  i = gCodeObjects[i].next_free) {
91  TH_DCHECK(i < gNumCodeObjects);
92  ++free_count;
93  // This check will fail if we encounter a cycle.
94  TH_DCHECK(free_count <= gNumCodeObjects);
95  }
96 
97  // Check that all free entries are reachable via the free list.
98  size_t free_count2 = 0;
99  for (size_t i = 0; i < gNumCodeObjects; ++i) {
100  if (gCodeObjects[i].code_info == nullptr) {
101  ++free_count2;
102  }
103  }
104  TH_DCHECK(free_count == free_count2);
105 #endif
106 }

109 CodeProtectionInfo* CreateHandlerData(
110  uintptr_t base, size_t size, size_t num_protected_instructions,
111  const ProtectedInstructionData* protected_instructions) {
112  const size_t alloc_size = HandlerDataSize(num_protected_instructions);
113  CodeProtectionInfo* data =
114  reinterpret_cast<CodeProtectionInfo*>(malloc(alloc_size));
115 
116  if (data == nullptr) {
117  return nullptr;
118  }
119 
120  data->base = base;
121  data->size = size;
122  data->num_protected_instructions = num_protected_instructions;
123 
124  if (num_protected_instructions > 0) {
125  memcpy(data->instructions, protected_instructions,
126  num_protected_instructions * sizeof(ProtectedInstructionData));
127  }
128 
129  return data;
130 }

132 int RegisterHandlerData(
133  uintptr_t base, size_t size, size_t num_protected_instructions,
134  const ProtectedInstructionData* protected_instructions) {
135  CodeProtectionInfo* data = CreateHandlerData(
136  base, size, num_protected_instructions, protected_instructions);
137 
138  if (data == nullptr) {
139  abort();
140  }
141 
142  MetadataLock lock;
143 
144  if (kEnableSlowChecks) {
145  VerifyCodeRangeIsDisjoint(data);
146  }
147 
148  size_t i = gNextCodeObject;
149 
150  // Explicitly convert std::numeric_limits<int>::max() to unsigned to avoid
151  // compiler warnings about signed/unsigned comparisons. We aren't worried
152  // about sign extension because we know std::numeric_limits<int>::max() is
153  // positive.
154  const size_t int_max = std::numeric_limits<int>::max();
155 
156  // We didn't find an opening in the available space, so grow.
157  if (i == gNumCodeObjects) {
158  size_t new_size = gNumCodeObjects > 0
159  ? gNumCodeObjects * kCodeObjectGrowthFactor
160  : kInitialCodeObjectSize;
161 
162  // Because we must return an int, there is no point in allocating space for
163  // more objects than can fit in an int.
164  if (new_size > int_max) {
165  new_size = int_max;
166  }
167  if (new_size == gNumCodeObjects) {
168  free(data);
169  return kInvalidIndex;
170  }
171 
172  // Now that we know our new size is valid, we can go ahead and realloc the
173  // array.
174  gCodeObjects = static_cast<CodeProtectionInfoListEntry*>(
175  realloc(gCodeObjects, sizeof(*gCodeObjects) * new_size));
176 
177  if (gCodeObjects == nullptr) {
178  abort();
179  }
180 
181  memset(gCodeObjects + gNumCodeObjects, 0,
182  sizeof(*gCodeObjects) * (new_size - gNumCodeObjects));
183  for (size_t j = gNumCodeObjects; j < new_size; ++j) {
184  gCodeObjects[j].next_free = j + 1;
185  }
186  gNumCodeObjects = new_size;
187  }
188 
189  TH_DCHECK(gCodeObjects[i].code_info == nullptr);
190 
191  // Find out where the next entry should go.
192  gNextCodeObject = gCodeObjects[i].next_free;
193 
194  if (i <= int_max) {
195  gCodeObjects[i].code_info = data;
196 
197  if (kEnableSlowChecks) {
198  ValidateCodeObjects();
199  }
200 
201  return static_cast<int>(i);
202  } else {
203  free(data);
204  return kInvalidIndex;
205  }
206 }

208 void ReleaseHandlerData(int index) {
209  if (index == kInvalidIndex) {
210  return;
211  }
212  TH_DCHECK(index >= 0);
213 
214  // Remove the data from the global list if it's there.
215  CodeProtectionInfo* data = nullptr;
216  {
217  MetadataLock lock;
218 
219  data = gCodeObjects[index].code_info;
220  gCodeObjects[index].code_info = nullptr;
221 
222  gCodeObjects[index].next_free = gNextCodeObject;
223  gNextCodeObject = index;
224 
225  if (kEnableSlowChecks) {
226  ValidateCodeObjects();
227  }
228  }
229  // TODO(eholk): on debug builds, ensure there are no more copies in
230  // the list.
231  TH_DCHECK(data); // make sure we're releasing legitimate handler data.
232  free(data);
233 }

235 bool RegisterV8Sandbox(uintptr_t base, size_t size) {
236  SandboxRecordsLock lock;
237 
238 #ifdef DEBUG
239  for (SandboxRecord* current = gSandboxRecordsHead; current != nullptr;
240  current = current->next) {
241  TH_DCHECK(current->base != base);
242  }
243 #endif
244 
245  SandboxRecord* new_record =
246  reinterpret_cast<SandboxRecord*>(malloc(sizeof(SandboxRecord)));
247  if (new_record == nullptr) {
248  return false;
249  }
250 
251  new_record->base = base;
252  new_record->size = size;
253  new_record->next = gSandboxRecordsHead;
254  gSandboxRecordsHead = new_record;
255  return true;
256 }

258 void UnregisterV8Sandbox(uintptr_t base, size_t size) {
259  SandboxRecordsLock lock;
260 
261  SandboxRecord* current = gSandboxRecordsHead;
262  SandboxRecord* previous = nullptr;
263  while (current != nullptr) {
264  if (current->base == base) {
265  break;
266  }
267  previous = current;
268  current = current->next;
269  }
270 
271  TH_CHECK(current != nullptr);
272  TH_CHECK(current->size == size);
273  if (previous) {
274  previous->next = current->next;
275  } else {
276  gSandboxRecordsHead = current->next;
277  }
278  free(current);
279 }

299 bool EnableTrapHandler(bool use_v8_handler) {
300  // We should only enable the trap handler once, and before any call to
301  // {IsTrapHandlerEnabled}. Enabling the trap handler late can lead to problems
302  // because code or objects might have been generated under the assumption that
303  // trap handlers are disabled.
304  bool can_enable =
305  g_can_enable_trap_handler.exchange(false, std::memory_order_relaxed);
306  // EnableTrapHandler called twice, or after IsTrapHandlerEnabled.
307  TH_CHECK(can_enable);
308  if (!V8_TRAP_HANDLER_SUPPORTED) {
309  return false;
310  }
311  if (use_v8_handler) {
312  g_is_trap_handler_enabled = RegisterDefaultTrapHandler();
313  return g_is_trap_handler_enabled;
314  }
315  g_is_trap_handler_enabled = true;
316  return true;
317 }
