63 template <typename Phase>
64 struct HasProperRunMethod {
65  using parameters = base::tmp::call_parameters_t<decltype(&Phase::Run)>;
66  static_assert(
67  base::tmp::length_v<parameters> >= 2,
68  "Phase::Run needs at least two parameters (PipelineData* and Zone*)");
69  using parameter0 = base::tmp::element_t<parameters, 0>;
70  using parameter1 = base::tmp::element_t<parameters, 1>;
71  static constexpr bool value = std::is_same_v<parameter0, PipelineData*> &&
72  std::is_same_v<parameter1, Zone*>;
73 };
74 
75 template <typename Phase, typename... Args>
76 concept TurboshaftPhase =
77  HasProperRunMethod<Phase>::value &&
78  requires(Phase p) { p.kKind == PhaseKind::kTurboshaft; };
