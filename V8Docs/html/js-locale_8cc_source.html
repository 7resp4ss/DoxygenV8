51 // Inserts tags from options into locale string.
52 Maybe<bool> InsertOptionsIntoLocale(Isolate* isolate,
53  DirectHandle<JSReceiver> options,
54  icu::LocaleBuilder* builder) {
55  DCHECK(isolate);
56 
57  static const auto hour_cycle_values =
58  std::to_array<const std::string_view>({"h11", "h12", "h23", "h24"});
59  static const auto case_first_values =
60  std::to_array<const std::string_view>({"upper", "lower", "false"});
61  const auto empty_values = std::span<std::string_view>();
62  const std::array<OptionData, 7> kOptionToUnicodeTagMap = {
63  {{"calendar", "ca", empty_values, false},
64  {"collation", "co", empty_values, false},
65  {"firstDayOfWeek", "fw", empty_values, false},
66  {"hourCycle", "hc", hour_cycle_values, false},
67  {"caseFirst", "kf", case_first_values, false},
68  {"numeric", "kn", empty_values, true},
69  {"numberingSystem", "nu", empty_values, false}}};
70 
71  // TODO(cira): Pass in values as per the spec to make this to be
72  // spec compliant.
73 
74  for (const auto& option_to_bcp47 : kOptionToUnicodeTagMap) {
75  DirectHandle<String> value_str;
76  bool value_bool = false;
77  Maybe<bool> maybe_found =
78  option_to_bcp47.is_bool_value
79  ? GetBoolOption(isolate, options, option_to_bcp47.name, "locale",
80  &value_bool)
81  : GetStringOption(isolate, options, option_to_bcp47.name,
82  option_to_bcp47.possible_values, "locale",
83  &value_str);
84  MAYBE_RETURN(maybe_found, Nothing<bool>());
85 
86  // TODO(cira): Use fallback value if value is not found to make
87  // this spec compliant.
88  if (!maybe_found.FromJust()) continue;
89 
90  std::string value_stdstr;
91 
92  const char* type = nullptr;
93 
94  if (!value_str.is_null()) {
95  value_stdstr = value_str->ToStdString();
96  type = value_stdstr.data();
97  }
98 
99  if (strcmp(option_to_bcp47.key, "fw") == 0) {
100  const std::array<ValueAndType, 8> kFirstDayValuesAndTypes = {
101  {{"0", "sun"},
102  {"1", "mon"},
103  {"2", "tue"},
104  {"3", "wed"},
105  {"4", "thu"},
106  {"5", "fri"},
107  {"6", "sat"},
108  {"7", "sun"}}};
109  for (const auto& value_to_type : kFirstDayValuesAndTypes) {
110  if (strcmp(type, value_to_type.value) == 0) {
111  type = value_to_type.type;
112  break;
113  }
114  }
115  } else if (option_to_bcp47.is_bool_value) {
116  type = value_bool ? "true" : "false";
117  }
118  DCHECK_NOT_NULL(type);
119 
120  // Overwrite existing, or insert new key-value to the locale string.
121  if (!uloc_toLegacyType(uloc_toLegacyKey(option_to_bcp47.key), type)) {
122  return Just(false);
123  }
124  builder->setUnicodeLocaleKeyword(option_to_bcp47.key, type);
125  }
126  return Just(true);
127 }

136  if (status == U_ILLEGAL_ARGUMENT_ERROR || value.empty()) {
137  return isolate->factory()->undefined_value();

144  }
145  return isolate->factory()->NewStringFromAsciiChecked(value.c_str());
146 }

148 bool IsCheckRange(std::string_view str, size_t min, size_t max,
149  bool(range_check_func)(char)) {
150  if (!base::IsInRange(str.length(), min, max)) return false;
151  for (size_t i = 0; i < str.length(); i++) {
152  if (!range_check_func(str[i])) return false;
153  }
154  return true;
155 }
156 bool IsAlpha(std::string_view str, size_t min, size_t max) {
157  return IsCheckRange(str, min, max, [](char c) -> bool {
158  return base::IsInRange(c, 'a', 'z') || base::IsInRange(c, 'A', 'Z');
159  });
160 }
161 
162 bool IsDigit(std::string_view str, size_t min, size_t max) {
163  return IsCheckRange(str, min, max, [](char c) -> bool {
164  return base::IsInRange(c, '0', '9');
165  });
166 }

168 bool IsAlphanum(std::string_view str, size_t min, size_t max) {
169  return IsCheckRange(str, min, max, [](char c) -> bool {
170  return base::IsInRange(c, 'a', 'z') || base::IsInRange(c, 'A', 'Z') ||
171  base::IsInRange(c, '0', '9');
172  });
173 }

175 bool IsUnicodeLanguageSubtag(std::string_view value) {
176  // unicode_language_subtag = alpha{2,3} | alpha{5,8};
177  return IsAlpha(value, 2, 3) || IsAlpha(value, 5, 8);
178 }
179 
180 bool IsUnicodeScriptSubtag(std::string_view value) {
181  // unicode_script_subtag = alpha{4} ;
182  return IsAlpha(value, 4, 4);
183 }
184 
185 bool IsUnicodeRegionSubtag(std::string_view value) {
186  // unicode_region_subtag = (alpha{2} | digit{3});
187  return IsAlpha(value, 2, 2) || IsDigit(value, 3, 3);
188 }

190 bool IsDigitAlphanum3(std::string_view value) {
191  return value.length() == 4 && base::IsInRange(value[0], '0', '9') &&
192  IsAlphanum(value.substr(1), 3, 3);
193 }
194 
195 bool IsUnicodeVariantSubtag(std::string_view value) {
196  // unicode_variant_subtag = (alphanum{5,8} | digit alphanum{3}) ;
197  return IsAlphanum(value, 5, 8) || IsDigitAlphanum3(value);
198 }

210 // Implemented as iteration instead of recursion to avoid stack overflow for
211 // very long input strings.
212 bool JSLocale::Is38AlphaNumList(std::string_view in) {
213  std::string_view value = in;
214  while (true) {
215  std::size_t found_dash = value.find('-');
216  if (found_dash == std::string::npos) {
217  return IsAlphanum(value, 3, 8);
218  }
219  if (!IsAlphanum(value.substr(0, found_dash), 3, 8)) return false;
220  value = value.substr(found_dash + 1);
221  }
222 }

226 // TODO(ftang) Replace the following check w/ icu::LocaleBuilder
227 // once ICU64 land in March 2019.
228 bool JSLocale::StartsWithUnicodeLanguageId(std::string_view value) {
229  // unicode_language_id =
230  // unicode_language_subtag (sep unicode_script_subtag)?
231  // (sep unicode_region_subtag)? (sep unicode_variant_subtag)* ;
232  if (value.empty()) return false;
233  std::vector<std::string_view> tokens;
234  size_t token_start = 0;
235  size_t token_end;
236  for (token_end = 0; token_end < value.size(); ++token_end) {
237  if (value[token_end] == '-') {
238  tokens.emplace_back(&value[token_start], token_end - token_start);
239  token_start = token_end + 1;
240  }
241  }
242  if (token_start != token_end) {
243  tokens.emplace_back(&value[token_start], token_end - token_start);
244  }
245  DCHECK(!tokens.empty());
246 
247  // length >= 1
248  if (!IsUnicodeLanguageSubtag(tokens[0])) return false;
249 
250  if (tokens.size() == 1) return true;
251 
252  // length >= 2
253  if (IsExtensionSingleton(tokens[1])) return true;
254 
255  size_t index = 1;
256  if (IsUnicodeScriptSubtag(tokens[index])) {
257  index++;
258  if (index == tokens.size()) return true;
259  }
260  if (IsUnicodeRegionSubtag(tokens[index])) {
261  index++;
262  }
263  while (index < tokens.size()) {
264  if (IsExtensionSingleton(tokens[index])) return true;
265  if (!IsUnicodeVariantSubtag(tokens[index])) return false;
266  index++;
267  }
268  return true;
269 }

272 Maybe<bool> ApplyOptionsToTag(Isolate* isolate, DirectHandle<String> tag,
273  DirectHandle<JSReceiver> options,
274  icu::LocaleBuilder* builder) {
275  v8::Isolate* v8_isolate = reinterpret_cast<v8::Isolate*>(isolate);
276  if (tag->length() == 0) {
277  THROW_NEW_ERROR_RETURN_VALUE(
278  isolate, NewRangeError(MessageTemplate::kLocaleNotEmpty),
279  Nothing<bool>());
280  }
281 
282  v8::String::Utf8Value bcp47_tag(v8_isolate, v8::Utils::ToLocal(tag));
283  builder->setLanguageTag(
284  {*bcp47_tag, static_cast<int32_t>(bcp47_tag.length())});
285  DCHECK_LT(0, bcp47_tag.length());
286  DCHECK_NOT_NULL(*bcp47_tag);
287  // 2. If IsStructurallyValidLanguageTag(tag) is false, throw a RangeError
288  // exception.
289  if (!JSLocale::StartsWithUnicodeLanguageId(*bcp47_tag)) {
290  return Just(false);
291  }
292  UErrorCode status = U_ZERO_ERROR;
293  icu::Locale canonicalized = builder->build(status);
294  canonicalized.canonicalize(status);
295  if (U_FAILURE(status)) {
296  return Just(false);
297  }
298  builder->setLocale(canonicalized);
299 
300  // 3. Let language be ? GetOption(options, "language", "string", undefined,
301  // undefined).
302  DirectHandle<String> language_str;
303  Maybe<bool> maybe_language = GetStringOption(
304  isolate, options, "language", std::span<std::string_view>(),
305  "ApplyOptionsToTag", &language_str);
306  MAYBE_RETURN(maybe_language, Nothing<bool>());
307 
308  // 4. If language is not undefined, then
309  if (maybe_language.FromJust()) {
310  std::string language_stdstr = language_str->ToStdString();
311  builder->setLanguage(language_stdstr);
312  builder->build(status);
313  // a. If language does not match the unicode_language_subtag production,
314  // throw a RangeError exception.
315  if (U_FAILURE(status) || language_stdstr.empty() ||
316  IsAlpha(language_stdstr, 4, 4)) {
317  return Just(false);
318  }
319  }
320  // 5. Let script be ? GetOption(options, "script", "string", undefined,
321  // undefined).
322  DirectHandle<String> script_str;
323  Maybe<bool> maybe_script =
324  GetStringOption(isolate, options, "script", std::span<std::string_view>(),
325  "ApplyOptionsToTag", &script_str);
326  MAYBE_RETURN(maybe_script, Nothing<bool>());
327  // 6. If script is not undefined, then
328  if (maybe_script.FromJust()) {
329  std::string script_stdstr = script_str->ToStdString();
330  builder->setScript(script_stdstr);
331  builder->build(status);
332  // a. If script does not match the unicode_script_subtag production, throw
333  // a RangeError exception.
334  if (U_FAILURE(status) || script_stdstr.empty()) {
335  return Just(false);
336  }
337  }
338  // 7. Let region be ? GetOption(options, "region", "string", undefined,
339  // undefined).
340  DirectHandle<String> region_str;
341  Maybe<bool> maybe_region =
342  GetStringOption(isolate, options, "region", std::span<std::string_view>(),
343  "ApplyOptionsToTag", &region_str);
344  MAYBE_RETURN(maybe_region, Nothing<bool>());
345  // 8. If region is not undefined, then
346  if (maybe_region.FromJust()) {
347  std::string region_stdstr = region_str->ToStdString();
348  // a. If region does not match the region production, throw a RangeError
349  // exception.
350  builder->setRegion(region_stdstr);
351  builder->build(status);
352  if (U_FAILURE(status) || region_stdstr.empty()) {
353  return Just(false);
354  }
355  }
356 
357  // 9. Set tag to CanonicalizeLanguageTag(tag).
358  // 10. If language is not undefined,
359  // a. Assert: tag matches the unicode_locale_id production.
360  // b. Set tag to tag with the substring corresponding to the
361  // unicode_language_subtag production replaced by the string language.
362  // 11. If script is not undefined, then
363  // a. If tag does not contain a unicode_script_subtag production, then
364  // i. Set tag to the concatenation of the unicode_language_subtag
365  // production of tag, "-", script, and the rest of tag.
366  // b. Else,
367  // i. Set tag to tag with the substring corresponding to the
368  // unicode_script_subtag production replaced by the string script.
369  // 12. If region is not undefined, then
370  // a. If tag does not contain a unicode_region_subtag production, then
371  // i. Set tag to the concatenation of the unicode_language_subtag
372  // production of tag, the substring corresponding to the "-"
373  // unicode_script_subtag production if present, "-", region, and
374  // the rest of tag.
375  // b. Else,
376  // i. Set tag to tag with the substring corresponding to the
377  // unicode_region_subtag production replaced by the string region.
378  // 13. Return CanonicalizeLanguageTag(tag).
379  return Just(true);
380 }

384 MaybeDirectHandle<JSLocale> JSLocale::New(Isolate* isolate,
385  DirectHandle<Map> map,
386  DirectHandle<String> locale_str,
387  DirectHandle<JSReceiver> options) {
388  icu::LocaleBuilder builder;
389  Maybe<bool> maybe_apply =
390  ApplyOptionsToTag(isolate, locale_str, options, &builder);
391  MAYBE_RETURN(maybe_apply, MaybeDirectHandle<JSLocale>());
392  if (!maybe_apply.FromJust()) {
393  THROW_NEW_ERROR(isolate,
394  NewRangeError(MessageTemplate::kLocaleBadParameters));
395  }
396 
397  Maybe<bool> maybe_insert =
398  InsertOptionsIntoLocale(isolate, options, &builder);
399  MAYBE_RETURN(maybe_insert, MaybeDirectHandle<JSLocale>());
400  UErrorCode status = U_ZERO_ERROR;
401  icu::Locale icu_locale = builder.build(status);
402 
403  icu_locale.canonicalize(status);
404 
405  if (!maybe_insert.FromJust() || U_FAILURE(status)) {
406  THROW_NEW_ERROR(isolate,
407  NewRangeError(MessageTemplate::kLocaleBadParameters));
408  }
409 
410  // 31. Set locale.[[Locale]] to r.[[locale]].
411  DirectHandle<Managed<icu::Locale>> managed_locale =
412  Managed<icu::Locale>::From(
413  isolate, 0, std::shared_ptr<icu::Locale>{icu_locale.clone()});
414 
415  // Now all properties are ready, so we can allocate the result object.
416  DirectHandle<JSLocale> locale =
417  Cast<JSLocale>(isolate->factory()->NewFastOrSlowJSObjectFromMap(map));
418  DisallowGarbageCollection no_gc;
419  locale->set_icu_locale(*managed_locale);
420  return locale;
421 }

448 MaybeDirectHandle<JSLocale> JSLocale::Maximize(Isolate* isolate,
449  DirectHandle<JSLocale> locale) {
450  // ICU has limitation on the length of the locale while addLikelySubtags
451  // is called. Work around the issue by only perform addLikelySubtags
452  // on the base locale and merge the extension if needed.
453  icu::Locale source(*(locale->icu_locale()->raw()));
454  icu::Locale result = icu::Locale::createFromName(source.getBaseName());
455  UErrorCode status = U_ZERO_ERROR;
456  result.addLikelySubtags(status);
457  if (strlen(source.getBaseName()) != strlen(result.getBaseName())) {
458  // Base name is changed
459  if (strlen(source.getBaseName()) != strlen(source.getName())) {
460  // the source has extensions, get the extensions from the source.
461  result = icu::LocaleBuilder()
462  .setLocale(source)
463  .setLanguage(result.getLanguage())
464  .setRegion(result.getCountry())
465  .setScript(result.getScript())
466  .setVariant(result.getVariant())
467  .build(status);
468  }
469  } else {
470  // Base name is not changed
471  result = source;
472  }
473  if (U_FAILURE(status) || result.isBogus()) {
474  // Due to https://unicode-org.atlassian.net/browse/ICU-21639
475  // Valid but super long locale will fail. Just throw here for now.
476  THROW_NEW_ERROR(isolate,
477  NewRangeError(MessageTemplate::kLocaleBadParameters));
478  }
479  return Construct(isolate, result);
480 }

516 template <typename T>
517 MaybeDirectHandle<JSArray> GetKeywordValuesFromLocale(
518  Isolate* isolate, const char* key, const char* unicode_key,
519  const icu::Locale& locale, bool (*removes)(const char*), bool commonly_used,
520  bool sort) {
521  Factory* factory = isolate->factory();
522  UErrorCode status = U_ZERO_ERROR;
523  std::string ext =
524  locale.getUnicodeKeywordValue<std::string>(unicode_key, status);
525  if (!ext.empty()) {
526  DirectHandle<FixedArray> fixed_array = factory->NewFixedArray(1);
527  DirectHandle<String> str = factory->NewStringFromAsciiChecked(ext.c_str());
528  fixed_array->set(0, *str);
529  return factory->NewJSArrayWithElements(fixed_array);
530  }
531  status = U_ZERO_ERROR;
532  std::unique_ptr<icu::StringEnumeration> enumeration(
533  T::getKeywordValuesForLocale(key, locale, commonly_used, status));
534  if (U_FAILURE(status)) {
535  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
536  }
537  return Intl::ToJSArray(isolate, unicode_key, enumeration.get(), removes,
538  sort);
539 }

571 MaybeDirectHandle<JSArray> JSLocale::GetHourCycles(
572  Isolate* isolate, DirectHandle<JSLocale> locale) {
573  // Let preferred be loc.[[HourCycle]].
574  // Let locale be loc.[[Locale]].
575  icu::Locale icu_locale(*(locale->icu_locale()->raw()));
576  Factory* factory = isolate->factory();
577 
578  // Assert: locale matches the unicode_locale_id production.
579 
580  // Let list be a List of 1 or more hour cycle identifiers, which must be
581  // String values indicating either the 12-hour format ("h11", "h12") or the
582  // 24-hour format ("h23", "h24"), sorted in descending preference of those in
583  // common use in the locale for date and time formatting.
584 
585  // Return CreateArrayFromListAndPreferred( list, preferred ).
586  DirectHandle<FixedArray> fixed_array = factory->NewFixedArray(1);
587  UErrorCode status = U_ZERO_ERROR;
588  std::string ext =
589  icu_locale.getUnicodeKeywordValue<std::string>("hc", status);
590  if (!ext.empty()) {
591  DirectHandle<String> str = factory->NewStringFromAsciiChecked(ext.c_str());
592  fixed_array->set(0, *str);
