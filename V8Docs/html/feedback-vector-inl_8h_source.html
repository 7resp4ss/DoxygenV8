230 Tagged<Code> FeedbackVector::optimized_code(IsolateForSandbox isolate) const {
231  Tagged<MaybeWeak<HeapObject>> slot = maybe_optimized_code();
232  DCHECK(slot.IsWeakOrCleared());
233  Tagged<HeapObject> heap_object;
234  Tagged<Code> code;
235  if (slot.GetHeapObject(&heap_object)) {
236  code = Cast<CodeWrapper>(heap_object)->code(isolate);
237  }
238  // It is possible that the maybe_optimized_code slot is cleared but the flags
239  // haven't been updated yet. We update them when we execute the function next
240  // time / when we create new closure.
241  DCHECK_IMPLIES(!code.is_null(),
242  maybe_has_maglev_code() || maybe_has_turbofan_code());
243  DCHECK_IMPLIES(!code.is_null() && code->is_maglevved(),
244  maybe_has_maglev_code());
245  DCHECK_IMPLIES(!code.is_null() && code->is_turbofanned(),
246  maybe_has_turbofan_code());
247  return code;
248 }

275 std::optional<Tagged<Code>> FeedbackVector::GetOptimizedOsrCode(
276  Isolate* isolate, Handle<BytecodeArray> bytecode, FeedbackSlot slot) {
277  Tagged<MaybeObject> maybe_code = Get(isolate, slot);
278  if (maybe_code.IsCleared()) return {};
279 
280  Tagged<Code> code =
281  Cast<CodeWrapper>(maybe_code.GetHeapObject())->code(isolate);
282  if (code->marked_for_deoptimization()) {
283  // Clear the cached Code object if deoptimized.
284  // TODO(jgruber): Add tracing.
285  Set(slot, ClearedValue(isolate));
286  if (!bytecode.is_null()) {
287  RecomputeOptimizedOsrCodeFlags(isolate, bytecode);
288  }
289  return {};
290  }
291 
292  return code;
293 }
294 
295 void FeedbackVector::RecomputeOptimizedOsrCodeFlags(
296  Isolate* isolate, Handle<BytecodeArray> bytecode_array) {
297  bool turbofan = false;
298  bool maglev = false;
299  interpreter::BytecodeArrayIterator it(bytecode_array);
300  for (; !it.done(); it.Advance()) {
301  if (it.current_bytecode() != interpreter::Bytecode::kJumpLoop) continue;
302  if (auto code = GetOptimizedOsrCode(isolate, {}, it.GetSlotOperand(2))) {
303  if ((*code)->marked_for_deoptimization()) continue;
304  turbofan |= (*code)->is_turbofanned();
305  maglev |= (*code)->is_maglevved();
306  }
307  }
308  if (!maglev && maybe_has_maglev_osr_code()) {
309  set_maybe_has_optimized_osr_code(false, CodeKind::MAGLEV);
310  }
311  if (!turbofan && maybe_has_turbofan_osr_code()) {
312  set_maybe_has_optimized_osr_code(false, CodeKind::TURBOFAN_JS);
313  }
314 }

430 CompareOperationHint CompareOperationHintFromFeedback(int type_feedback) {
431  if (Is<CompareOperationFeedback::kNone>(type_feedback)) {
432  return CompareOperationHint::kNone;
433  }
434 
435  if (Is<CompareOperationFeedback::kSignedSmall>(type_feedback)) {
436  return CompareOperationHint::kSignedSmall;
437  } else if (Is<CompareOperationFeedback::kNumber>(type_feedback)) {
438  return CompareOperationHint::kNumber;
439  } else if (Is<CompareOperationFeedback::kNumberOrBoolean>(type_feedback)) {
440  return CompareOperationHint::kNumberOrBoolean;
441  } else if (Is<CompareOperationFeedback::kNumberOrOddball>(type_feedback)) {
442  return CompareOperationHint::kNumberOrOddball;
443  }
444 
445  if (Is<CompareOperationFeedback::kInternalizedString>(type_feedback)) {
446  return CompareOperationHint::kInternalizedString;
447  } else if (Is<CompareOperationFeedback::kString>(type_feedback)) {
448  return CompareOperationHint::kString;
449  } else if (Is<CompareOperationFeedback::kStringOrOddball>(type_feedback) &&
450  !Is<CompareOperationFeedback::kOddball>(type_feedback)) {
451  // Don't return the StringOrOddball feedback for pure oddball comparisons,
452  // that would be too confusing.
453  return CompareOperationHint::kStringOrOddball;
454  }
455 
456  if (Is<CompareOperationFeedback::kReceiver>(type_feedback)) {
457  return CompareOperationHint::kReceiver;
458  } else if (Is<CompareOperationFeedback::kReceiverOrNullOrUndefined>(
459  type_feedback)) {
460  return CompareOperationHint::kReceiverOrNullOrUndefined;
461  }
462 
463  if (Is<CompareOperationFeedback::kBigInt64>(type_feedback)) {
464  return CompareOperationHint::kBigInt64;
465  } else if (Is<CompareOperationFeedback::kBigInt>(type_feedback)) {
466  return CompareOperationHint::kBigInt;
467  }
468 
469  if (Is<CompareOperationFeedback::kSymbol>(type_feedback)) {
470  return CompareOperationHint::kSymbol;
471  }
472 
473  DCHECK(Is<CompareOperationFeedback::kAny>(type_feedback));
474  return CompareOperationHint::kAny;
475 }

577 std::pair<Tagged<MaybeObject>, Tagged<MaybeObject>>
578 FeedbackNexus::GetFeedbackPair() const {
579  if (config()->mode() == NexusConfig::BackgroundThread &&
580  feedback_cache_.has_value()) {
581  return std::make_pair(FromHandle(feedback_cache_->first),
582  FromHandle(feedback_cache_->second));
583  }
584  auto pair = FeedbackMetadata::GetSlotSize(kind()) == 2
585  ? config()->GetFeedbackPair(vector(), slot())
586  : std::make_pair(config()->GetFeedback(vector(), slot()),
587  Tagged<MaybeObject>());
588  if (config()->mode() == NexusConfig::BackgroundThread &&
589  !feedback_cache_.has_value()) {
590  feedback_cache_ =
591  std::make_pair(ToHandle(pair.first), ToHandle(pair.second));
592  }
