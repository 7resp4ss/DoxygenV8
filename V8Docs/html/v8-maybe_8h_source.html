22 /**
23  * A simple Maybe type, representing an object which may or may not have a
24  * value, see https://hackage.haskell.org/package/base/docs/Data-Maybe.html.
25  *
26  * If an API method returns a Maybe<>, the API method can potentially fail
27  * either because an exception is thrown, or because an exception is pending,
28  * e.g. because a previous API call threw an exception that hasn't been caught
29  * yet, or because a TerminateExecution exception was thrown. In that case, a
30  * "Nothing" value is returned.
31  */
32 template <class T>
33 class Maybe : public cppgc::internal::ConditionalStackAllocatedBase<T> {
34  public:
35  constexpr Maybe() = default;
36 
37  V8_INLINE bool IsNothing() const { return !has_value_; }
38  V8_INLINE bool IsJust() const { return has_value_; }
39 
40  /**
41  * An alias for |FromJust|. Will crash if the Maybe<> is nothing.
42  */
43  V8_INLINE T ToChecked() const { return FromJust(); }
44 
45  /**
46  * Short-hand for ToChecked(), which doesn't return a value. To be used, where
47  * the actual value of the Maybe is not needed like Object::Set.
48  */
49  V8_INLINE void Check() const {
50  if (V8_UNLIKELY(!IsJust())) api_internal::FromJustIsNothing();
51  }
52 
53  /**
54  * Converts this Maybe<> to a value of type T. If this Maybe<> is
55  * nothing (empty), |false| is returned and |out| is left untouched.
56  */
57  V8_WARN_UNUSED_RESULT V8_INLINE bool To(T* out) const {
58  if (V8_LIKELY(IsJust())) *out = value_;
59  return IsJust();
60  }
61 
62  /**
63  * Converts this Maybe<> to a value of type T, moving out of it. If this
64  * Maybe<> is nothing (empty), |false| is returned and |out| is left
65  * untouched.
66  */
67  V8_WARN_UNUSED_RESULT V8_INLINE bool MoveTo(T* out) && {
68  if (V8_LIKELY(IsJust())) *out = std::move(value_);
69  return IsJust();
70  }
71 
72  /**
73  * Converts this Maybe<> to a value of type T. If this Maybe<> is
74  * nothing (empty), V8 will crash the process.
75  */
76  V8_INLINE T FromJust() const& {
77  if (V8_UNLIKELY(!IsJust())) api_internal::FromJustIsNothing();
78  return value_;
79  }
80 
81  /**
82  * Converts this Maybe<> to a value of type T. If this Maybe<> is
83  * nothing (empty), V8 will crash the process.
84  */
85  V8_INLINE T FromJust() && {
86  if (V8_UNLIKELY(!IsJust())) api_internal::FromJustIsNothing();
87  return std::move(value_);
88  }
89 
90  /**
91  * Converts this Maybe<> to a value of type T, using a default value if this
92  * Maybe<> is nothing (empty).
93  */
94  V8_INLINE T FromMaybe(const T& default_value) const {
95  return has_value_ ? value_ : default_value;
96  }
97 
98  V8_INLINE bool operator==(const Maybe& other) const {
99  return (IsJust() == other.IsJust()) &&
100  (!IsJust() || FromJust() == other.FromJust());
101  }
102 
103  V8_INLINE bool operator!=(const Maybe& other) const {
104  return !operator==(other);
105  }
106 
107  private:
108  explicit Maybe(const T& t) : has_value_(true), value_(t) {}
109  explicit Maybe(T&& t) : has_value_(true), value_(std::move(t)) {}
110 
111  bool has_value_ = false;
112  T value_;
113 
114  template <class U>
115  friend Maybe<U> Just(const U& u);
116  template <class U, std::enable_if_t<!std::is_lvalue_reference_v<U>>*>
117  friend Maybe<U> Just(U&& u);
118 };
119 
120 template <class T>
121 inline constexpr Maybe<T> Nothing() {
122  return {};
123 }
124 
125 template <class T>
126 inline Maybe<T> Just(const T& t) {
127  return Maybe<T>(t);
128 }
129 
130 // Don't use forwarding references here but instead use two overloads.
131 // Forwarding references only work when type deduction takes place, which is not
132 // the case for callsites such as Just<Type>(t).
133 template <class T, std::enable_if_t<!std::is_lvalue_reference_v<T>>* = nullptr>
134 inline Maybe<T> Just(T&& t) {
135  return Maybe<T>(std::move(t));
136 }

138 // A template specialization of Maybe<T> for the case of T = void.
139 template <>
140 class Maybe<void> {
141  public:
142  constexpr Maybe() = default;
143 
144  V8_INLINE bool IsNothing() const { return !is_valid_; }
145  V8_INLINE bool IsJust() const { return is_valid_; }
146 
147  V8_INLINE bool operator==(const Maybe& other) const {
148  return IsJust() == other.IsJust();
149  }
150 
151  V8_INLINE bool operator!=(const Maybe& other) const {
152  return !operator==(other);
153  }
154 
155  private:
156  struct JustTag {};
157 
158  explicit Maybe(JustTag) : is_valid_(true) {}
159 
160  bool is_valid_ = false;
161 
162  friend Maybe<void> JustVoid();
163 };
