67 // The prefix used for mangled symbols, per the Itanium C++ ABI:
68 // http://www.codesourcery.com/cxx-abi/abi.html#mangling
69 const char kMangledSymbolPrefix[] = "_Z";
70 
71 // Characters that can be used for symbols, generated by Ruby:
72 // (('a'..'z').to_a+('A'..'Z').to_a+('0'..'9').to_a + ['_']).join
73 const char kSymbolCharacters[] =
74  "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789_";
75 
76 #if HAVE_EXECINFO_H
77 // Demangles C++ symbols in the given text. Example:
78 //
79 // "out/Debug/base_unittests(_ZN10StackTraceC1Ev+0x20) [0x817778c]"
80 // =>
81 // "out/Debug/base_unittests(StackTrace::StackTrace()+0x20) [0x817778c]"
82 void DemangleSymbols(std::string* text) {
83  // Note: code in this function is NOT async-signal safe (std::string uses
84  // malloc internally).
85 
86 
87  std::string::size_type search_from = 0;
88  while (search_from < text->size()) {
89  // Look for the start of a mangled symbol, from search_from.
90  std::string::size_type mangled_start =
91  text->find(kMangledSymbolPrefix, search_from);
92  if (mangled_start == std::string::npos) {
93  break; // Mangled symbol not found.
94  }
95 
96  // Look for the end of the mangled symbol.
97  std::string::size_type mangled_end =
98  text->find_first_not_of(kSymbolCharacters, mangled_start);
99  if (mangled_end == std::string::npos) {
100  mangled_end = text->size();
101  }
102  std::string mangled_symbol =
103  text->substr(mangled_start, mangled_end - mangled_start);
104 
105  // Try to demangle the mangled symbol candidate.
106  int status = 0;
107  std::unique_ptr<char, FreeDeleter> demangled_symbol(
108  abi::__cxa_demangle(mangled_symbol.c_str(), nullptr, nullptr, &status));
109  if (status == 0) { // Demangling is successful.
110  // Remove the mangled symbol.
111  text->erase(mangled_start, mangled_end - mangled_start);
112  // Insert the demangled symbol.
113  text->insert(mangled_start, demangled_symbol.get());
114  // Next time, we'll start right after the demangled symbol we inserted.
115  search_from = mangled_start + strlen(demangled_symbol.get());
116  } else {
117  // Failed to demangle. Retry after the "_Z" we just found.
118  search_from = mangled_start + 2;
119  }
120  }
121 }
122 #endif // HAVE_EXECINFO_H

148 void ProcessBacktrace(void* const* trace, size_t size,
149  BacktraceOutputHandler* handler) {
150  // NOTE: This code MUST be async-signal safe (it's used by in-process
151  // stack dumping signal handler). NO malloc or stdio is allowed here.
152  handler->HandleOutput("\n");
153  handler->HandleOutput("==== C stack trace ===============================\n");
154  handler->HandleOutput("\n");
155 
156  bool printed = false;
157 
158  // Below part is async-signal unsafe (uses malloc), so execute it only
159  // when we are not executing the signal handler.
160  if (in_signal_handler == 0) {
161  std::unique_ptr<char*, FreeDeleter> trace_symbols(
162  backtrace_symbols(trace, static_cast<int>(size)));
163  if (trace_symbols) {
164  for (size_t i = 0; i < size; ++i) {
165  std::string trace_symbol = trace_symbols.get()[i];
166  DemangleSymbols(&trace_symbol);
167  handler->HandleOutput(" ");
168  handler->HandleOutput(trace_symbol.c_str());
169  handler->HandleOutput("\n");
170  }
171 
172  printed = true;
173  }
174  } else if (handler->OutputFileDescriptor() != 0) {
175  // In this case, we can use backtrace_symbols_fd to write directly to the
176  // output file descriptor. This isn't quite as nice as we don't control the
177  // formatting and because mangled function names will be used, but still
178  // better than just raw addresses (which are also included in this output).
179  backtrace_symbols_fd(trace, static_cast<int>(size),
180  handler->OutputFileDescriptor());
181  printed = true;
182  }
183 
184  if (!printed) {
185  for (size_t i = 0; i < size; ++i) {
186  handler->HandleOutput(" [");
187  OutputPointer(trace[i], handler);
188  handler->HandleOutput("]\n");
189  }
190  }
191 }
192 #endif // HAVE_EXECINFO_H

201 void StackDumpSignalHandler(int signal, siginfo_t* info, void* void_context) {
202  // NOTE: This code MUST be async-signal safe.
203  // NO malloc or stdio is allowed here.
204 
205  // Record the fact that we are in the signal handler now, so that the rest
206  // of StackTrace can behave in an async-signal-safe manner.
207  in_signal_handler = 1;
208 
209 #if V8_HAS_PKU_SUPPORT
210  MemoryProtectionKey::SetDefaultPermissionsForAllKeysInSignalHandler();
211 #endif
212 
213  PrintToStderr("Received signal ");
214  char buf[1024] = {0};
215  internal::itoa_r(signal, buf, sizeof(buf), 10, 0);
216  PrintToStderr(buf);
217  if (signal == SIGBUS) {
218  if (info->si_code == BUS_ADRALN)
219  PrintToStderr(" BUS_ADRALN ");
220  else if (info->si_code == BUS_ADRERR)
221  PrintToStderr(" BUS_ADRERR ");
222  else if (info->si_code == BUS_OBJERR)
223  PrintToStderr(" BUS_OBJERR ");
224  else
225  PrintToStderr(" <unknown> ");
226  } else if (signal == SIGFPE) {
227  if (info->si_code == FPE_FLTDIV)
228  PrintToStderr(" FPE_FLTDIV ");
229  else if (info->si_code == FPE_FLTINV)
230  PrintToStderr(" FPE_FLTINV ");
231  else if (info->si_code == FPE_FLTOVF)
232  PrintToStderr(" FPE_FLTOVF ");
233  else if (info->si_code == FPE_FLTRES)
234  PrintToStderr(" FPE_FLTRES ");
235  else if (info->si_code == FPE_FLTSUB)
236  PrintToStderr(" FPE_FLTSUB ");
237  else if (info->si_code == FPE_FLTUND)
238  PrintToStderr(" FPE_FLTUND ");
239  else if (info->si_code == FPE_INTDIV)
240  PrintToStderr(" FPE_INTDIV ");
241  else if (info->si_code == FPE_INTOVF)
242  PrintToStderr(" FPE_INTOVF ");
243  else
244  PrintToStderr(" <unknown> ");
245  } else if (signal == SIGILL) {
246  if (info->si_code == ILL_BADSTK)
247  PrintToStderr(" ILL_BADSTK ");
248  else if (info->si_code == ILL_COPROC)
249  PrintToStderr(" ILL_COPROC ");
250  else if (info->si_code == ILL_ILLOPN)
251  PrintToStderr(" ILL_ILLOPN ");
252  else if (info->si_code == ILL_ILLADR)
253  PrintToStderr(" ILL_ILLADR ");
254  else if (info->si_code == ILL_ILLTRP)
255  PrintToStderr(" ILL_ILLTRP ");
256  else if (info->si_code == ILL_PRVOPC)
257  PrintToStderr(" ILL_PRVOPC ");
258  else if (info->si_code == ILL_PRVREG)
259  PrintToStderr(" ILL_PRVREG ");
260  else
261  PrintToStderr(" <unknown> ");
262  } else if (signal == SIGSEGV) {
263  if (info->si_code == SEGV_MAPERR)
264  PrintToStderr(" SEGV_MAPERR ");
265  else if (info->si_code == SEGV_ACCERR)
266  PrintToStderr(" SEGV_ACCERR ");
267  else
268  PrintToStderr(" <unknown> ");
269  }
270  if (signal == SIGBUS || signal == SIGFPE || signal == SIGILL ||
271  signal == SIGSEGV) {
272  internal::itoa_r(reinterpret_cast<intptr_t>(info->si_addr), buf,
273  sizeof(buf), 16, 12);
274  PrintToStderr(buf);
275  }
276  PrintToStderr("\n");
277  if (dump_stack_in_signal_handler) {
278  debug::StackTrace().Print();
279  PrintToStderr("[end of stack trace]\n");
280  }
281 
282  if (::signal(signal, SIG_DFL) == SIG_ERR) _exit(1);
283 }

349 bool EnableInProcessStackDumping() {
350  // When running in an application, our code typically expects SIGPIPE
351  // to be ignored. Therefore, when testing that same code, it should run
352  // with SIGPIPE ignored as well.
353  struct sigaction sigpipe_action;
354  memset(&sigpipe_action, 0, sizeof(sigpipe_action));
355  sigpipe_action.sa_handler = SIG_IGN;
356  sigemptyset(&sigpipe_action.sa_mask);
357  bool success = (sigaction(SIGPIPE, &sigpipe_action, nullptr) == 0);
358 
359  // Avoid hangs during backtrace initialization, see above.
360  WarmUpBacktrace();
361 
362  struct sigaction action;
363  memset(&action, 0, sizeof(action));
364  // Use SA_ONSTACK so that iff an alternate stack has been registered, the
365  // handler will run on that stack instead of the default stack. This can be
366  // useful for example if the stack pointer gets corrupted or in case of stack
367  // overflows, since that might prevent the handler from running properly.
368  action.sa_flags = SA_RESETHAND | SA_SIGINFO | SA_ONSTACK;
369  action.sa_sigaction = &StackDumpSignalHandler;
370  sigemptyset(&action.sa_mask);
371 
372  success &= (sigaction(SIGILL, &action, nullptr) == 0);
373  success &= (sigaction(SIGABRT, &action, nullptr) == 0);
374  success &= (sigaction(SIGFPE, &action, nullptr) == 0);
375  success &= (sigaction(SIGBUS, &action, nullptr) == 0);
376  success &= (sigaction(SIGSEGV, &action, nullptr) == 0);
377  success &= (sigaction(SIGSYS, &action, nullptr) == 0);
378 
379  dump_stack_in_signal_handler = true;
380 
381  return success;
382 }

420 // NOTE: code from sandbox/linux/seccomp-bpf/demo.cc.
421 char* itoa_r(intptr_t i, char* buf, size_t sz, int base, size_t padding) {
422  // Make sure we can write at least one NUL byte.
423  size_t n = 1;
424  if (n > sz) return nullptr;
425 
426  if (base < 2 || base > 16) {
427  buf[0] = '\0';
428  return nullptr;
429  }
430 
431  char* start = buf;
432 
433  uintptr_t j = i;
434 
435  // Handle negative numbers (only for base 10).
436  if (i < 0 && base == 10) {
437  // This does "j = -i" while avoiding integer overflow.
438  j = static_cast<uintptr_t>(-(i + 1)) + 1;
439 
440  // Make sure we can write the '-' character.
441  if (++n > sz) {
442  buf[0] = '\0';
443  return nullptr;
444  }
445  *start++ = '-';
446  }
447 
448  // Loop until we have converted the entire number. Output at least one
449  // character (i.e. '0').
450  char* ptr = start;
451  do {
452  // Make sure there is still enough space left in our output buffer.
453  if (++n > sz) {
454  buf[0] = '\0';
455  return nullptr;
456  }
457 
458  // Output the next digit.
459  *ptr++ = "0123456789abcdef"[j % base];
460  j /= base;
461 
462  if (padding > 0) padding--;
463  } while (j > 0 || padding > 0);
464 
465  // Terminate the output with a NUL character.
466  *ptr = '\0';
467 
468  // Conversion to ASCII actually resulted in the digits being in reverse
469  // order. We can't easily generate them in forward order, as we can't tell
470  // the number of characters needed until we are done converting.
471  // So, now, we reverse the string (except for the possible "-" sign).
472  while (--ptr > start) {
473  char ch = *ptr;
474  *ptr = *start;
475  *start++ = ch;
476  }
477  return buf;
478 }
