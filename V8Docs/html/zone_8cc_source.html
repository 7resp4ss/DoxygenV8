42 void* Zone::AsanNew(size_t size) {
43  CHECK(!sealed_);
44 
45  // Round up the requested size to fit the alignment.
46  size = RoundUp(size, kAlignmentInBytes);
47 
48  // Check if the requested size is available without expanding.
49  const size_t size_with_redzone = size + kASanRedzoneBytes;
50  DCHECK_LE(position_, limit_);
51  if (V8_UNLIKELY(size_with_redzone > limit_ - position_)) {
52  Expand(size_with_redzone);
53  }
54  DCHECK_LE(size_with_redzone, limit_ - position_);
55 
56  Address result = position_;
57  position_ += size_with_redzone;
58 
59  Address redzone_position = result + size;
60  DCHECK_EQ(redzone_position + kASanRedzoneBytes, position_);
61  ASAN_POISON_MEMORY_REGION(reinterpret_cast<void*>(redzone_position),
62  kASanRedzoneBytes);
63 
64  // Check that the result has the proper alignment and return it.
65  DCHECK(IsAligned(result, kAlignmentInBytes));
66  return reinterpret_cast<void*>(result);
67 }

69 void Zone::Reset() {
70  if (!segment_head_) return;
71  Segment* keep = segment_head_;
72  segment_head_ = segment_head_->next();
73  if (segment_head_ != nullptr) {
74  // Reset the position to the end of the new head, and uncommit its
75  // allocation size (which will be re-committed in DeleteAll).
76  position_ = segment_head_->end();
77  allocation_size_ -= segment_head_->end() - segment_head_->start();
78  }
79  keep->set_next(nullptr);
80  DeleteAll();
81  allocator_->TraceZoneCreation(this);
82 
83  // Un-poison the kept segment content so we can zap and reuse it.
84  ASAN_UNPOISON_MEMORY_REGION(reinterpret_cast<void*>(keep->start()),
85  keep->capacity());
86  keep->ZapContents();
87 
88  segment_head_ = keep;
89  position_ = RoundUp(keep->start(), kAlignmentInBytes);
90  limit_ = keep->end();
91  DCHECK_LT(allocation_size(), kAlignmentInBytes);
92  DCHECK_EQ(segment_bytes_allocated_, keep->total_size());
93 }

110 void Zone::DeleteAll() {
111  Segment* current = segment_head_;
112  if (current) {
113  // Commit the allocation_size_ of segment_head_ and disconnect the segments
114  // list from the zone in order to ensure that tracing accounting allocator
115  // will observe value including memory from the head segment.
116  allocation_size_ = allocation_size();
117  segment_head_ = nullptr;
118  }
119  allocator_->TraceZoneDestruction(this);
120 
121  // Traverse the chained list of segments and return them all to the allocator.
122  while (current) {
123  Segment* next = current->next();
124  segment_bytes_allocated_ -= current->total_size();
125  ReleaseSegment(current);
126  current = next;
127  }
128 
129  position_ = limit_ = 0;
130  allocation_size_ = 0;
131 #ifdef V8_ENABLE_PRECISE_ZONE_STATS
132  allocation_size_for_tracing_ = 0;
133 #endif
134 }

143 void Zone::Expand(size_t size) {
144  // Make sure the requested size is already properly aligned and that
145  // there isn't enough room in the Zone to satisfy the request.
146  DCHECK_EQ(size, RoundDown(size, kAlignmentInBytes));
147  DCHECK_LT(limit_ - position_, size);
148 
149  // Compute the new segment size. We use a 'high water mark'
150  // strategy, where we increase the segment size every time we expand
151  // except that we employ a maximum segment size when we delete. This
152  // is to avoid excessive malloc() and free() overhead.
153  //
154  // TODO(409953791): This can be simplified when using the page pool for
155  // managing zone memory as it works with equal-sized segments.
156  Segment* head = segment_head_;
157  const size_t old_size = head ? head->total_size() : 0;
158  static const size_t kSegmentOverhead = sizeof(Segment) + kAlignmentInBytes;
159  const size_t new_size_no_overhead = size + (old_size << 1);
160  size_t new_size = kSegmentOverhead + new_size_no_overhead;
161  const size_t min_new_size = kSegmentOverhead + size;
162  // Guard against integer overflow.
163  if (new_size_no_overhead < size || new_size < kSegmentOverhead) {
164  V8::FatalProcessOutOfMemory(nullptr, "Zone");
165  }
166  if (new_size < kMinimumSegmentSize) {
167  new_size = kMinimumSegmentSize;
168  } else if (new_size >= kMaximumSegmentSize) {
169  // Limit the size of new segments to avoid growing the segment size
170  // exponentially, thus putting pressure on contiguous virtual address space.
171  // All the while making sure to allocate a segment large enough to hold the
172  // requested size.
173  new_size = std::max({min_new_size, kMaximumSegmentSize});
174  }
175  if (new_size > INT_MAX) {
176  V8::FatalProcessOutOfMemory(nullptr, "Zone");
177  }
178  Segment* segment = allocator_->AllocateSegment(new_size);
179  if (segment == nullptr) {
180  V8::FatalProcessOutOfMemory(nullptr, "Zone");
181  }
182 
183  DCHECK_GE(segment->total_size(), new_size);
184  segment_bytes_allocated_ += segment->total_size();
185  segment->set_zone(this);
186  segment->set_next(segment_head_);
187  // Commit the allocation_size_ of segment_head_ if any, in order to ensure
188  // that tracing accounting allocator will observe value including memory
189  // from the previous head segment.
190  allocation_size_ = allocation_size();
191  segment_head_ = segment;
192  allocator_->TraceAllocateSegment(segment);
193 
194  // Recompute 'top' and 'limit' based on the new segment.
195  position_ = RoundUp(segment->start(), kAlignmentInBytes);
196  limit_ = segment->end();
197  DCHECK_LE(position_, limit_);
198  DCHECK_LE(size, limit_ - position_);
199 }

214 void ZoneSnapshot::Restore(Zone* zone) const {
215  // Release segments up to the stored segment_head_.
216  Segment* current = zone->segment_head_;
217  while (current != segment_head_) {
218  // If this check failed, then either you passed a wrong zone, or the zone
219  // was reset to an earlier snapshot already. We cannot move forward again.
220  CHECK_NOT_NULL(current);
221  Segment* next = current->next();
222  zone->ReleaseSegment(current);
223  current = next;
224  }
225 
226  // Un-poison the trailing segment content so we can reuse or zap it later.
227  if (segment_head_ != nullptr) {
228  void* const start = reinterpret_cast<void*>(position_);
229  DCHECK_GE(start, reinterpret_cast<void*>(current->start()));
230  DCHECK_LE(start, reinterpret_cast<void*>(current->end()));
231  const size_t length = current->end() - reinterpret_cast<Address>(start);
232  ASAN_UNPOISON_MEMORY_REGION(start, length);
233  }
234 
235  // Reset the Zone to the stored state.
236  zone->allocation_size_ = allocation_size_;
237  zone->segment_bytes_allocated_ = segment_bytes_allocated_;
238  zone->position_ = position_;
239  zone->limit_ = limit_;
240  zone->segment_head_ = segment_head_;
241 #ifdef V8_ENABLE_PRECISE_ZONE_STATS
242  zone->allocation_size_for_tracing_ = allocation_size_for_tracing_;
243  zone->freed_size_for_tracing_ = freed_size_for_tracing_;
244 #endif
245 }
