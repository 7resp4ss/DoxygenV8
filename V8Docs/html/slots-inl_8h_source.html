205 void ExternalPointerSlot::init(IsolateForSandbox isolate,
206  Tagged<HeapObject> host, Address value,
207  ExternalPointerTag tag) {
208 #ifdef V8_ENABLE_SANDBOX
209  ExternalPointerTable& table = isolate.GetExternalPointerTableFor(tag);
210  ExternalPointerHandle handle = table.AllocateAndInitializeEntry(
211  isolate.GetExternalPointerTableSpaceFor(tag, host.address()), value, tag);
212  // Use a Release_Store to ensure that the store of the pointer into the
213  // table is not reordered after the store of the handle. Otherwise, other
214  // threads may access an uninitialized table entry and crash.
215  Release_StoreHandle(handle);
216 #else
217  store(isolate, value, tag);
218 #endif // V8_ENABLE_SANDBOX
219 }

431 template <IndirectPointerSlot::TagCheckStrictness allow_unpublished>
432 Tagged<Object> IndirectPointerSlot::ResolveHandle(
433  IndirectPointerHandle handle, IsolateForSandbox isolate) const {
434 #ifdef V8_ENABLE_SANDBOX
435  // TODO(saelo) Maybe come up with a different entry encoding scheme that
436  // returns Smi::zero for kNullCodePointerHandle?
437  if (!handle) return Smi::zero();
438 
439  // Resolve the handle. The tag implies the pointer table to use.
440  if (tag_ == kUnknownIndirectPointerTag) {
441  // In this case we have to rely on the handle marking to determine which
442  // pointer table to use.
443  if (handle & kCodePointerHandleMarker) {
444  return ResolveCodePointerHandle(handle);
445  } else {
446  return ResolveTrustedPointerHandle<allow_unpublished>(handle, isolate);
447  }
448  } else if (tag_ == kCodeIndirectPointerTag) {
449  return ResolveCodePointerHandle(handle);
450  } else {
451  return ResolveTrustedPointerHandle<allow_unpublished>(handle, isolate);
452  }
453 #else
454  UNREACHABLE();
455 #endif // V8_ENABLE_SANDBOX
456 }

458 #ifdef V8_ENABLE_SANDBOX
459 template <IndirectPointerSlot::TagCheckStrictness allow_unpublished>
460 Tagged<Object> IndirectPointerSlot::ResolveTrustedPointerHandle(
461  IndirectPointerHandle handle, IsolateForSandbox isolate) const {
462  DCHECK_NE(handle, kNullIndirectPointerHandle);
463  const TrustedPointerTable& table = isolate.GetTrustedPointerTableFor(tag_);
464  if constexpr (allow_unpublished == kAllowUnpublishedEntries) {
465  return Tagged<Object>(table.GetMaybeUnpublished(handle, tag_));
466  }
467  return Tagged<Object>(table.Get(handle, tag_));
468 }

488 // Copies tagged words from |src| to |dst|. The data spans must not overlap.
489 // |src| and |dst| must be kTaggedSize-aligned.
490 inline void CopyTagged(Address dst, const Address src, size_t num_tagged) {
491  static const size_t kBlockCopyLimit = 16;

493  reinterpret_cast<const Tagged_t*>(src), num_tagged);
