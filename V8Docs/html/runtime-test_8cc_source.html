307 bool CanOptimizeFunction(CodeKind target_kind,
308  DirectHandle<JSFunction> function, Isolate* isolate,
309  IsCompiledScope* is_compiled_scope) {
310  // The following conditions were lifted (in part) from the DCHECK inside
311  // JSFunction::MarkForOptimization().
312 
313  // If function isn't compiled, compile it now.
314  if (!is_compiled_scope->is_compiled() &&
315  !Compiler::Compile(isolate, function, Compiler::CLEAR_EXCEPTION,
316  is_compiled_scope)) {
317  return CrashUnlessFuzzingReturnFalse(isolate);
318  }
319 
320  if (target_kind == CodeKind::TURBOFAN_JS && !v8_flags.turbofan) return false;
321  if (target_kind == CodeKind::MAGLEV && !maglev::IsMaglevEnabled()) {
322  return false;
323  }
324 
325  if (function->shared()->optimization_disabled(target_kind)) {
326  return false;
327  }
328 
329  if (IsAsmWasmFunction(isolate, *function)) {
330  return CrashUnlessFuzzingReturnFalse(isolate);
331  }
332 
333  if (v8_flags.testing_d8_test_runner) {
334  if (!CheckMarkedForManualOptimization(isolate, *function)) {
335  return CrashUnlessFuzzingReturnFalse(isolate);
336  }
337  }
338 
339  if (function->is_compiled(isolate) &&
340  !function->HasAvailableCodeKind(isolate,
341  CodeKind::INTERPRETED_FUNCTION)) {
342  return CrashUnlessFuzzingReturnFalse(isolate);
343  }
344 
345  if (function->HasAvailableCodeKind(isolate, target_kind) ||
346  function->HasAvailableHigherTierCodeThan(isolate, target_kind)) {
347  DCHECK(function->HasAttachedOptimizedCode(isolate) ||
348  function->ChecksTieringState(isolate));
349  return false;
350  }
351 
352  return true;
353 }

355 Tagged<Object> OptimizeFunctionOnNextCall(RuntimeArguments& args,
356  Isolate* isolate,
357  CodeKind target_kind) {
358  if (args.length() != 1 && args.length() != 2) {
359  return CrashUnlessFuzzing(isolate);
360  }
361 
362  DirectHandle<Object> function_object = args.at(0);
363  if (!IsJSFunction(*function_object)) return CrashUnlessFuzzing(isolate);
364  DirectHandle<JSFunction> function = Cast<JSFunction>(function_object);
365 
366  IsCompiledScope is_compiled_scope(
367  function->shared()->is_compiled_scope(isolate));
368  if (!CanOptimizeFunction(target_kind, function, isolate,
369  &is_compiled_scope)) {
370  return ReadOnlyRoots(isolate).undefined_value();
371  }
372 
373  ConcurrencyMode concurrency_mode = ConcurrencyMode::kSynchronous;
374  if (args.length() == 2) {
375  DirectHandle<Object> type = args.at(1);
376  if (!IsString(*type)) return CrashUnlessFuzzing(isolate);
377  if (Cast<String>(type)->IsOneByteEqualTo(
378  base::StaticCharVector("concurrent")) &&
379  isolate->concurrent_recompilation_enabled()) {
380  concurrency_mode = ConcurrencyMode::kConcurrent;
381  }
382  }
383 
384  // This function may not have been lazily compiled yet, even though its shared
385  // function has.
386  if (!function->is_compiled(isolate)) {
387  DCHECK(function->shared()->HasBytecodeArray());
388  Tagged<Code> code = *BUILTIN_CODE(isolate, InterpreterEntryTrampoline);
389  if (function->shared()->HasBaselineCode()) {
390  code = function->shared()->baseline_code(kAcquireLoad);
391  }
392  function->UpdateCode(isolate, code);
393  }
394 
395  TraceManualRecompile(*function, target_kind, concurrency_mode);
396  JSFunction::EnsureFeedbackVector(isolate, function, &is_compiled_scope);
397  if (function->GetActiveTier(isolate) != target_kind) {
398  function->RequestOptimization(isolate, target_kind, concurrency_mode);
399  }
400 
401  return ReadOnlyRoots(isolate).undefined_value();
402 }

404 bool EnsureCompiledAndFeedbackVector(Isolate* isolate,
405  DirectHandle<JSFunction> function,
406  IsCompiledScope* is_compiled_scope) {
407  *is_compiled_scope = function->shared()->is_compiled_scope(isolate);
408 
409  // If function isn't compiled, compile it now.
410  if (!is_compiled_scope->is_compiled()) {
411  // Check function allows lazy compilation.
412  DCHECK(function->shared()->allows_lazy_compilation());
413  if (!Compiler::Compile(isolate, function, Compiler::CLEAR_EXCEPTION,
414  is_compiled_scope)) {
415  return false;
416  }
417  }
418 
419  // Ensure function has a feedback vector to hold type feedback for
420  // optimization.
421  if (!function->shared()->HasFeedbackMetadata()) {
422  return false;
423  }
424  JSFunction::EnsureFeedbackVector(isolate, function, is_compiled_scope);
425  return true;
426 }

654 BytecodeOffset OffsetOfNextJumpLoop(Isolate* isolate,
655  Handle<BytecodeArray> bytecode_array,
656  int current_offset) {
657  interpreter::BytecodeArrayIterator it(bytecode_array, current_offset);
658 
659  // First, look for a loop that contains the current bytecode offset.
660  for (; !it.done(); it.Advance()) {
661  if (it.current_bytecode() != interpreter::Bytecode::kJumpLoop) {
662  continue;
663  }
664  if (!base::IsInRange(current_offset, it.GetJumpTargetOffset(),
665  it.current_offset())) {
666  continue;
667  }
668 
669  return BytecodeOffset(it.current_offset());
670  }
671 
672  // Fall back to any loop after the current offset.
673  it.SetOffset(current_offset);
674  for (; !it.done(); it.Advance()) {
675  if (it.current_bytecode() == interpreter::Bytecode::kJumpLoop) {
676  return BytecodeOffset(it.current_offset());
677  }
678  }
679 
680  return BytecodeOffset::None();
681 }

685 RUNTIME_FUNCTION(Runtime_OptimizeOsr) {
686  HandleScope handle_scope(isolate);
687 
688  DirectHandle<JSFunction> function;
689 
690  // The optional parameter determines the frame being targeted.
691  int stack_depth = 0;
692  if (args.length() == 1) {
693  if (!IsSmi(args[0])) return CrashUnlessFuzzing(isolate);
694  stack_depth = args.smi_value_at(0);
695  }
696 
697  // Find the JavaScript function on the top of the stack.
698  JavaScriptStackFrameIterator it(isolate);
699  while (!it.done() && stack_depth--) it.Advance();
700  if (!it.done()) {
701  if (it.frame()->is_turbofan()) {
702  if (v8_flags.trace_osr) {
703  CodeTracer::Scope scope(isolate->GetCodeTracer());
704  PrintF(scope.file(),
705  "[OSR - %%OptimizeOsr failed because the current function could "
706  "not be found.]\n");
707  }
708  // This can happen if %OptimizeOsr is in inlined function.
709  return ReadOnlyRoots(isolate).undefined_value();
710  } else if (it.frame()->is_maglev()) {
711  function = MaglevFrame::cast(it.frame())->GetInnermostFunction();
712  } else {
713  function = direct_handle(it.frame()->function(), isolate);
714  }
715  }
716  if (function.is_null()) return CrashUnlessFuzzing(isolate);
717 
718  if (V8_UNLIKELY((!v8_flags.turbofan && !maglev::IsMaglevEnabled()) ||
719  (!v8_flags.use_osr && !maglev::IsMaglevOsrEnabled()))) {
720  return ReadOnlyRoots(isolate).undefined_value();
721  }
722 
723  if (!function->shared()->allows_lazy_compilation()) {
724  return CrashUnlessFuzzing(isolate);
725  }
726 
727  if (function->shared()->all_optimization_disabled()) {
728  return CrashUnlessFuzzing(isolate);
729  }
730 
731  if (v8_flags.testing_d8_test_runner) {
732  if (!CheckMarkedForManualOptimization(isolate, *function)) {
733  return CrashUnlessFuzzing(isolate);
734  }
735  }
736 
737  if (function->HasAvailableOptimizedCode(isolate) &&
738  (!function->code(isolate)->is_maglevved() || !v8_flags.osr_from_maglev)) {
739  DCHECK(function->HasAttachedOptimizedCode(isolate) ||
740  function->ChecksTieringState(isolate));
741  // If function is already optimized, return.
742  return ReadOnlyRoots(isolate).undefined_value();
743  }
744 
745  if (!it.frame()->is_unoptimized() &&
746  (!it.frame()->is_maglev() || !v8_flags.osr_from_maglev)) {
747  // Nothing to be done.
748  return ReadOnlyRoots(isolate).undefined_value();
749  }
750 
751  IsCompiledScope is_compiled_scope(
752  function->shared()->is_compiled_scope(isolate));
753  JSFunction::EnsureFeedbackVector(isolate, function, &is_compiled_scope);
754  isolate->tiering_manager()->RequestOsrAtNextOpportunity(*function);
755 
756  // If concurrent OSR is enabled, the testing workflow is a bit tricky. We
757  // must guarantee that the next JumpLoop installs the finished OSR'd code
758  // object, but we still want to exercise concurrent code paths. To do so,
759  // we attempt to find the next JumpLoop, start an OSR job for it now, and
760  // immediately force finalization.
761  // If this succeeds and we correctly match up the next JumpLoop, once we
762  // reach the JumpLoop we'll hit the OSR cache and install the generated code.
763  // If not (e.g. because we enter a nested loop first), the next JumpLoop will
764  // see the cached OSR code with a mismatched offset, and trigger
765  // non-concurrent OSR compilation and installation.
766  // To tier up from Maglev to TF we always do this, because the non-concurrent
767  // recompilation in `CompileOptimizedOSRFromMaglev` is broken. See the comment
768  // in `runtime-compiler.cc`.
769  bool concurrent_osr =
770  isolate->concurrent_recompilation_enabled() && v8_flags.concurrent_osr;
771  bool is_maglev = false;
772  if (it.frame()->is_maglev() || concurrent_osr) {
773  BytecodeOffset osr_offset = BytecodeOffset::None();
774  if (it.frame()->is_unoptimized()) {
775  UnoptimizedJSFrame* frame = UnoptimizedJSFrame::cast(it.frame());
776  Handle<BytecodeArray> bytecode_array(frame->GetBytecodeArray(), isolate);
777  const int current_offset = frame->GetBytecodeOffset();
778  osr_offset =
779  OffsetOfNextJumpLoop(isolate, bytecode_array, current_offset);
780  } else {
781  MaglevFrame* frame = MaglevFrame::cast(it.frame());
782  Handle<BytecodeArray> bytecode_array(
783  function->shared()->GetBytecodeArray(isolate), isolate);
784  const BytecodeOffset current_offset = frame->GetBytecodeOffsetForOSR();
785  osr_offset = OffsetOfNextJumpLoop(
786  isolate, bytecode_array,
787  current_offset.IsNone() ? 0 : current_offset.ToInt());
788  is_maglev = true;
789  }
790 
791  if (osr_offset.IsNone()) {
792  // The loop may have been elided by bytecode generation (e.g. for
793  // patterns such as `do { ... } while (false);` or we are in an inlined
794  // constructor stub.
795  return ReadOnlyRoots(isolate).undefined_value();
796  }
797 
798  // Finalize first to ensure all pending tasks are done (since we can't
799  // queue more than one OSR job for each function).
800  if (concurrent_osr) {
801  FinalizeOptimization(isolate);
802  }
803 
804  // Queue the job.
805  auto unused_result = Compiler::CompileOptimizedOSR(
806  isolate, function, osr_offset,
807  concurrent_osr ? ConcurrencyMode::kConcurrent
808  : ConcurrencyMode::kSynchronous,
809  (maglev::IsMaglevOsrEnabled() && !it.frame()->is_maglev())
810  ? CodeKind::MAGLEV
811  : CodeKind::TURBOFAN_JS);
812  USE(unused_result);
813 
814  // Finalize again to finish the queued job. The next call into
815  // Runtime::kCompileOptimizedOSR will pick up the cached InstructionStream
816  // object.
817  if (concurrent_osr) {
818  FinalizeOptimization(isolate);
819  }
820 
821  if (is_maglev) {
822  // Maglev ignores the maybe_has_optimized_osr_code flag, thus we also need
823  // to set a maximum urgency.
824  function->feedback_vector()->set_osr_urgency(
825  FeedbackVector::kMaxOsrUrgency);
826  }
827  }
828 
829  return ReadOnlyRoots(isolate).undefined_value();
830 }

894  HandleScope scope(isolate);
895  DCHECK_EQ(args.length(), 1);
896 
897  int status = 0;
898  if (v8_flags.lite_mode || v8_flags.jitless || !V8_ENABLE_TURBOFAN_BOOL) {
899  // These modes cannot optimize. Unit tests should handle these the same
900  // way.
901  status |= static_cast<int>(OptimizationStatus::kLiteMode);
902  }
903  if (!isolate->use_optimizer()) {
904  status |= static_cast<int>(OptimizationStatus::kNeverOptimize);
905  }
906  if (v8_flags.always_turbofan || v8_flags.prepare_always_turbofan) {
907  status |= static_cast<int>(OptimizationStatus::kAlwaysOptimize);
908  }
909  if (v8_flags.deopt_every_n_times) {
910  status |= static_cast<int>(OptimizationStatus::kMaybeDeopted);
911  }
912  if (v8_flags.optimize_on_next_call_optimizes_to_maglev) {
913  status |= static_cast<int>(
914  OptimizationStatus::kOptimizeOnNextCallOptimizesToMaglev);
915  }
916  if (v8_flags.optimize_maglev_optimizes_to_turbofan) {
917  status |= static_cast<int>(
918  OptimizationStatus::kOptimizeMaglevOptimizesToTurbofan);
919  }
920 
921  Handle<Object> function_object = args.at(0);
922  if (IsUndefined(*function_object)) return Smi::FromInt(status);
923  if (!IsJSFunction(*function_object)) return CrashUnlessFuzzing(isolate);
924 
925  auto function = Cast<JSFunction>(function_object);
926  status |= static_cast<int>(OptimizationStatus::kIsFunction);
927 
928  if (function->has_feedback_vector()) {
929  if (function->tiering_in_progress()) {
930  status |= static_cast<int>(OptimizationStatus::kOptimizingConcurrently);
931  } else if (function->GetRequestedOptimizationIfAny(
932  isolate, ConcurrencyMode::kConcurrent) == CodeKind::MAGLEV) {
933  status |= static_cast<int>(
934  OptimizationStatus::kMarkedForConcurrentMaglevOptimization);
935  } else if (function->GetRequestedOptimizationIfAny(
936  isolate, ConcurrencyMode::kSynchronous) ==
937  CodeKind::MAGLEV) {
938  status |=
939  static_cast<int>(OptimizationStatus::kMarkedForMaglevOptimization);
940  } else if (function->GetRequestedOptimizationIfAny(
941  isolate, ConcurrencyMode::kConcurrent) ==
942  CodeKind::TURBOFAN_JS) {
943  status |= static_cast<int>(
944  OptimizationStatus::kMarkedForConcurrentOptimization);
945  } else if (function->GetRequestedOptimizationIfAny(
946  isolate, ConcurrencyMode::kSynchronous) ==
947  CodeKind::TURBOFAN_JS) {
948  status |= static_cast<int>(OptimizationStatus::kMarkedForOptimization);
949  }
950  }
951 
952  if (function->HasAttachedOptimizedCode(isolate)) {
953  Tagged<Code> code = function->code(isolate);
954  if (code->marked_for_deoptimization()) {
955  status |= static_cast<int>(OptimizationStatus::kMarkedForDeoptimization);
956  } else {
957  status |= static_cast<int>(OptimizationStatus::kOptimized);
958  }
959  if (code->is_maglevved()) {
960  status |= static_cast<int>(OptimizationStatus::kMaglevved);
961  } else if (code->is_turbofanned()) {
962  status |= static_cast<int>(OptimizationStatus::kTurboFanned);
963  }
964  }
965  if (function->HasAttachedCodeKind(isolate, CodeKind::BASELINE)) {
966  status |= static_cast<int>(OptimizationStatus::kBaseline);
967  }
968  if (function->ActiveTierIsIgnition(isolate)) {
969  status |= static_cast<int>(OptimizationStatus::kInterpreted);
970  }
971  if (!function->is_compiled(isolate)) {
972  status |= static_cast<int>(OptimizationStatus::kIsLazy);
973  }
974 
975  // Additionally, detect activations of this frame on the stack, and report the
976  // status of the topmost frame.
977  JavaScriptFrame* frame = nullptr;
978  JavaScriptStackFrameIterator it(isolate);
979  while (!it.done()) {
980  if (it.frame()->function() == *function) {
981  frame = it.frame();
982  break;
983  }
984  it.Advance();
985  }
986  if (frame != nullptr) {
987  status |= static_cast<int>(OptimizationStatus::kIsExecuting);
988  if (frame->is_turbofan()) {
989  status |=
990  static_cast<int>(OptimizationStatus::kTopmostFrameIsTurboFanned);
991  } else if (frame->is_interpreted()) {
992  status |=
993  static_cast<int>(OptimizationStatus::kTopmostFrameIsInterpreted);
994  } else if (frame->is_baseline()) {
995  status |= static_cast<int>(OptimizationStatus::kTopmostFrameIsBaseline);
996  } else if (frame->is_maglev()) {
997  status |= static_cast<int>(OptimizationStatus::kTopmostFrameIsMaglev);
998  }
999  }
1000 
1001  return Smi::FromInt(status);
1002 }
