60 bool MaglevCompiler::Compile(LocalIsolate* local_isolate,
61  MaglevCompilationInfo* compilation_info) {
62  compiler::CurrentHeapBrokerScope current_broker(compilation_info->broker());
63  Graph* graph = Graph::New(compilation_info);
64 
65  bool is_tracing_enabled = false;
66  {
67  UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
68 
69  // Build graph.
70  if (v8_flags.print_maglev_code || v8_flags.code_comments ||
71  v8_flags.print_maglev_graph || v8_flags.print_maglev_graphs ||
72  v8_flags.trace_maglev_graph_building ||
73  v8_flags.trace_maglev_escape_analysis ||
74  v8_flags.trace_maglev_phi_untagging || v8_flags.trace_maglev_regalloc ||
75  v8_flags.trace_maglev_object_tracking) {
76  is_tracing_enabled = compilation_info->toplevel_compilation_unit()
77  ->shared_function_info()
78  .object()
79  ->PassesFilter(v8_flags.maglev_print_filter);
80  compilation_info->set_graph_labeller(new MaglevGraphLabeller());
81  }
82 
83  if (is_tracing_enabled &&
84  (v8_flags.print_maglev_code || v8_flags.print_maglev_graph ||
85  v8_flags.print_maglev_graphs || v8_flags.trace_maglev_graph_building ||
86  v8_flags.trace_maglev_phi_untagging ||
87  v8_flags.trace_maglev_regalloc)) {
88  MaglevCompilationUnit* top_level_unit =
89  compilation_info->toplevel_compilation_unit();
90  std::cout << "Compiling " << Brief(*compilation_info->toplevel_function())
91  << " with Maglev\n";
92  BytecodeArray::Disassemble(top_level_unit->bytecode().object(),
93  std::cout);
94  if (v8_flags.maglev_print_feedback) {
95  Print(*top_level_unit->feedback().object(), std::cout);
96  }
97  }
98 
99  {
100  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
101  "V8.Maglev.GraphBuilding");
102  MaglevGraphBuilder graph_builder(
103  local_isolate, compilation_info->toplevel_compilation_unit(), graph);
104  graph_builder.Build();
105 
106  if (is_tracing_enabled && v8_flags.print_maglev_graphs) {
107  std::cout << "\nAfter graph building" << std::endl;
108  PrintGraph(std::cout, graph);
109  }
110  }
111 
112 #ifdef DEBUG
113  {
114  GraphProcessor<MaglevGraphVerifier, /* visit_identity_nodes */ true>
115  verifier(compilation_info);
116  verifier.ProcessGraph(graph);
117  }
118 #endif
119 
120  if (v8_flags.maglev_non_eager_inlining) {
121  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
122  "V8.Maglev.Inlining");
123 
124  MaglevInliner inliner(graph);
125  inliner.Run(is_tracing_enabled);
126 
127  // TODO(victorgomes): We need to remove all identity nodes before
128  // PhiRepresentationSelector. Since Identity has different semantics
129  // there. Check if we can remove the identity nodes during
130  // PhiRepresentationSelector instead.
131  GraphProcessor<SweepIdentityNodes, /* visit_identity_nodes */ true> sweep;
132  sweep.ProcessGraph(graph);
133  }
134 
135 #ifdef DEBUG
136  {
137  GraphProcessor<MaglevGraphVerifier, /* visit_identity_nodes */ true>
138  verifier(compilation_info);
139  verifier.ProcessGraph(graph);
140  }
141 #endif
142 
143  if (v8_flags.maglev_licm) {
144  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
145  "V8.Maglev.LoopOptimizations");
146 
147  GraphProcessor<LoopOptimizationProcessor> loop_optimizations(
148  compilation_info);
149  loop_optimizations.ProcessGraph(graph);
150 
151  if (is_tracing_enabled && v8_flags.print_maglev_graphs) {
152  std::cout << "\nAfter loop optimizations" << std::endl;
153  PrintGraph(std::cout, graph);
154  }
155  }
156 
157 #ifdef DEBUG
158  {
159  GraphProcessor<MaglevGraphVerifier, /* visit_identity_nodes */ true>
160  verifier(compilation_info);
161  verifier.ProcessGraph(graph);
162  }
163 #endif
164 
165  if (v8_flags.maglev_untagged_phis) {
166  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
167  "V8.Maglev.PhiUntagging");
168 
169  GraphProcessor<MaglevPhiRepresentationSelector> representation_selector(
170  graph);
171  representation_selector.ProcessGraph(graph);
172 
173  if (is_tracing_enabled && v8_flags.print_maglev_graphs) {
174  std::cout << "\nAfter Phi untagging" << std::endl;
175  PrintGraph(std::cout, graph);
176  }
177  }
178  }
179 
180 #ifdef DEBUG
181  {
182  GraphProcessor<MaglevGraphVerifier, /* visit_identity_nodes */ true>
183  verifier(compilation_info);
184  verifier.ProcessGraph(graph);
185  }
186 #endif
187 
188  {
189  // Post-hoc optimisation:
190  // - Remove unreachable blocks
191  // - Dead node marking
192  // - Cleaning up identity nodes
193  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
194  "V8.Maglev.DeadCodeMarking");
195  if (graph->may_have_unreachable_blocks()) {
196  graph->RemoveUnreachableBlocks();
197  }
198  GraphProcessor<AnyUseMarkingProcessor> processor;
199  processor.ProcessGraph(graph);
200  }
201 
202  if (is_tracing_enabled && v8_flags.print_maglev_graphs) {
203  UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
204  std::cout << "After use marking" << std::endl;
205  PrintGraph(std::cout, graph);
206  }
207 
208 #ifdef DEBUG
209  {
210  GraphProcessor<MaglevGraphVerifier, /* visit_identity_nodes */ true>
211  verifier(compilation_info);
212  verifier.ProcessGraph(graph);
213  }
214 #endif
215 
216  {
217  RegallocInfo regalloc_info;
218  {
219  // Preprocessing for register allocation and code gen:
220  // - Remove dead nodes
221  // - Collect input/output location constraints
222  // - Find the maximum number of stack arguments passed to calls
223  // - Collect use information, for SSA liveness and next-use distance.
224  // - Mark
225  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
226  "V8.Maglev.NodeProcessing");
227  UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
228  GraphMultiProcessor<DeadNodeSweepingProcessor,
229  ValueLocationConstraintProcessor,
230  MaxCallDepthProcessor, LiveRangeAndNextUseProcessor,
231  DecompressedUseMarkingProcessor>
232  processor(LiveRangeAndNextUseProcessor{compilation_info, graph,
233  &regalloc_info});
234  processor.ProcessGraph(graph);
235  }
236 
237  if (is_tracing_enabled && v8_flags.print_maglev_graphs) {
238  UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
239  std::cout << "After register allocation pre-processing" << std::endl;
240  PrintGraph(std::cout, graph);
241  }
242 
243  {
244  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
245  "V8.Maglev.RegisterAllocation");
246  StraightForwardRegisterAllocator allocator(compilation_info, graph,
247  &regalloc_info);
248  }
249 
250  if (is_tracing_enabled &&
251  (v8_flags.print_maglev_graph || v8_flags.print_maglev_graphs)) {
252  UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
253  std::cout << "After register allocation" << std::endl;
254  PrintGraph(std::cout, graph);
255  }
256  }
257 
258  {
259  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
260  "V8.Maglev.CodeAssembly");
261  UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
262  std::unique_ptr<MaglevCodeGenerator> code_generator =
263  std::make_unique<MaglevCodeGenerator>(local_isolate, compilation_info,
264  graph);
265  bool success = code_generator->Assemble();
266  if (!success) {
267  return false;
268  }
269 
270  // Stash the compiled code_generator on the compilation info.
271  compilation_info->set_code_generator(std::move(code_generator));
272  }
273 
274  return true;
275 }

278 std::pair<MaybeHandle<Code>, BailoutReason> MaglevCompiler::GenerateCode(
279  Isolate* isolate, MaglevCompilationInfo* compilation_info) {
280  compiler::CurrentHeapBrokerScope current_broker(compilation_info->broker());
281  MaglevCodeGenerator* const code_generator =
282  compilation_info->code_generator();
283  DCHECK_NOT_NULL(code_generator);
284 
285  Handle<Code> code;
286  {
287  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
288  "V8.Maglev.CodeGeneration");
289  if (compilation_info->is_detached() ||
290  !code_generator->Generate(isolate).ToHandle(&code)) {
291  compilation_info->toplevel_compilation_unit()
292  ->shared_function_info()
293  .object()
294  ->set_maglev_compilation_failed(true);
295  return {{}, BailoutReason::kMaglevCodeGenerationFailed};
296  }
297  }
298 
299  {
300  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
301  "V8.Maglev.CommittingDependencies");
302  if (!compilation_info->broker()->dependencies()->Commit(code)) {
303  // Don't `set_maglev_compilation_failed` s.t. we may reattempt
304  // compilation.
305  // TODO(v8:7700): Make this more robust, i.e.: don't recompile endlessly.
306  compilation_info->toplevel_function()->SetInterruptBudget(
307  isolate, BudgetModification::kReduce);
308  return {{}, BailoutReason::kBailedOutDueToDependencyChange};
309  }
310  }
311 
312  if (v8_flags.print_maglev_code) {
313 #ifdef OBJECT_PRINT
314  std::unique_ptr<char[]> debug_name =
315  compilation_info->toplevel_function()->shared()->DebugNameCStr();
316  CodeTracer::StreamScope tracing_scope(isolate->GetCodeTracer());
317  auto& os = tracing_scope.stream();
318  code->CodePrint(os, debug_name.get());
319 #else
320  Print(*code);
321 #endif
322  }
323 
324  return {code, BailoutReason::kNoReason};
325 }
