70 i::Handle<i::String> GetBigIntStringPresentationHandle(
71  i::Isolate* i_isolate, i::DirectHandle<i::BigInt> i_bigint) {
72  // For large BigInts computing the decimal string representation
73  // can take a long time, so we go with hexadecimal in that case.
74  int radix = (i_bigint->Words64Count() > 100 * 1000) ? 16 : 10;
75  i::Handle<i::String> string_value =
76  i::BigInt::ToString(i_isolate, i_bigint, radix, i::kDontThrow)
77  .ToHandleChecked();
78  if (radix == 16) {
79  if (i_bigint->IsNegative()) {
80  string_value =
81  i_isolate->factory()
82  ->NewConsString(
83  i_isolate->factory()->NewStringFromAsciiChecked("-0x"),
84  i_isolate->factory()->NewProperSubString(
85  string_value, 1, string_value->length() - 1))
86  .ToHandleChecked();
87  } else {
88  string_value =
89  i_isolate->factory()
90  ->NewConsString(
91  i_isolate->factory()->NewStringFromAsciiChecked("0x"),
92  string_value)
93  .ToHandleChecked();
94  }
95  }
96  return string_value;
97 }

200 using FlagFilter = std::function<bool(i::IsStaticFlag)>;
201 using VariableModeFilter = std::function<bool(i::VariableMode)>;
202 using ContextLocalIterator = std::function<void(
203  i::VariableMode, i::Handle<i::String>, i::Handle<i::Object>)>;
204 
205 void ForEachContextLocal(i::Isolate* isolate,
206  i::DirectHandle<i::Context> context,
207  const VariableModeFilter& var_mode_filter,
208  const FlagFilter& flag_filter,
209  const ContextLocalIterator& context_local_it) {
210  i::DisallowJavascriptExecutionDebugOnly no_execution(isolate);
211  i::DisallowExceptionsDebugOnly no_exceptions(isolate);
212  i::DirectHandle<i::ScopeInfo> scope_info(context->scope_info(), isolate);
213  for (auto it : i::ScopeInfo::IterateLocalNames(scope_info)) {
214  i::Handle<i::String> name(it->name(), isolate);
215  i::VariableMode mode = scope_info->ContextLocalMode(it->index());
216  if (!var_mode_filter(mode)) {
217  continue;
218  }
219  i::IsStaticFlag flag = scope_info->ContextLocalIsStaticFlag(it->index());
220  if (!flag_filter(flag)) {
221  continue;
222  }
223  int context_index = scope_info->ContextHeaderLength() + it->index();
224  i::Handle<i::Object> slot_value(context->GetNoCell(context_index), isolate);
225  context_local_it(mode, name, slot_value);
226  }
227 }

231 bool GetPrivateMembers(Local<Context> context, Local<Object> object, int filter,
232  LocalVector<Value>* names_out,
233  LocalVector<Value>* values_out) {
234  i::Isolate* isolate = i::Isolate::Current();
235  ApiRuntimeCallStatsScope rcs_scope(
236  isolate, i::RuntimeCallCounterId::kAPI_debug_GetPrivateMembers);
237  EnterV8NoScriptNoExceptionScope api_scope(isolate);
238 
239  bool include_methods =
240  filter & static_cast<int>(PrivateMemberFilter::kPrivateMethods);
241  bool include_fields =
242  filter & static_cast<int>(PrivateMemberFilter::kPrivateFields);
243  bool include_accessors =
244  filter & static_cast<int>(PrivateMemberFilter::kPrivateAccessors);
245  bool include_methods_or_accessors = include_methods || include_accessors;
246 
247  auto var_mode_filter =
248  include_methods
249  ? (include_accessors ? i::IsPrivateMethodOrAccessorVariableMode
250  : i::IsPrivateMethodVariableMode)
251  : i::IsPrivateAccessorVariableMode;
252  auto constexpr instance_filter = [](i::IsStaticFlag flag) {
253  return flag == i::IsStaticFlag::kNotStatic;
254  };
255  auto constexpr static_filter = [](i::IsStaticFlag flag) {
256  return flag == i::IsStaticFlag::kStatic;
257  };
258 
259  i::DirectHandle<i::JSReceiver> receiver = Utils::OpenDirectHandle(*object);
260 
261  i::PropertyFilter key_filter =
262  static_cast<i::PropertyFilter>(i::PropertyFilter::PRIVATE_NAMES_ONLY);
263  i::DirectHandle<i::FixedArray> keys;
264  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
265  isolate, keys,
266  i::KeyAccumulator::GetKeys(isolate, receiver,
267  i::KeyCollectionMode::kOwnOnly, key_filter,
268  i::GetKeysConversion::kConvertToString),
269  false);
270 
271  // Estimate number of private fields and private instance methods/accessors.
272  int private_entries_count = 0;
273  auto count_private_entry =
274  [&](i::VariableMode mode, i::DirectHandle<i::String>,
275  i::DirectHandle<i::Object>) { private_entries_count++; };
276  for (int i = 0; i < keys->length(); ++i) {
277  // Exclude the private brand symbols.
278  i::DirectHandle<i::Symbol> key(i::Cast<i::Symbol>(keys->get(i)), isolate);
279  if (key->is_private_brand()) {
280  if (include_methods_or_accessors) {
281  i::DirectHandle<i::Object> value;
282  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
283  isolate, value, i::Object::GetProperty(isolate, receiver, key),
284  false);
285 
286  i::DirectHandle<i::Context> value_context(i::Cast<i::Context>(*value),
287  isolate);
288  ForEachContextLocal(isolate, value_context, var_mode_filter,
289  instance_filter, count_private_entry);
290  }
291  } else if (include_fields) {
292  private_entries_count++;
293  }
294  }
295 
296  // Estimate number of static private methods/accessors for classes.
297  bool has_static_private_methods_or_accessors = false;
298  if (include_methods_or_accessors) {
299  if (IsJSFunction(*receiver)) {
300  i::DirectHandle<i::JSFunction> func(i::Cast<i::JSFunction>(*receiver),
301  isolate);
302  i::DirectHandle<i::SharedFunctionInfo> shared(func->shared(), isolate);
303  if (shared->is_class_constructor() &&
304  shared->has_static_private_methods_or_accessors()) {
305  has_static_private_methods_or_accessors = true;
306  i::DirectHandle<i::Context> func_context(func->context(), isolate);
307  ForEachContextLocal(isolate, func_context, var_mode_filter,
308  static_filter, count_private_entry);
309  }
310  }
311  }
312 
313  DCHECK(names_out->empty());
314  names_out->reserve(private_entries_count);
315  DCHECK(values_out->empty());
316  values_out->reserve(private_entries_count);
317 
318  auto add_private_entry = [&](i::VariableMode mode,
319  i::DirectHandle<i::String> name,
320  i::DirectHandle<i::Object> value) {
321  DCHECK_IMPLIES(mode == i::VariableMode::kPrivateMethod,
322  IsJSFunction(*value));
323  DCHECK_IMPLIES(mode != i::VariableMode::kPrivateMethod,
324  IsAccessorPair(*value));
325  names_out->push_back(Utils::ToLocal(name));
326  values_out->push_back(Utils::ToLocal(value));
327  };
328  if (has_static_private_methods_or_accessors) {
329  i::DirectHandle<i::Context> receiver_context(
330  i::Cast<i::JSFunction>(*receiver)->context(), isolate);
331  ForEachContextLocal(isolate, receiver_context, var_mode_filter,
332  static_filter, add_private_entry);
333  }
334 
335  for (int i = 0; i < keys->length(); ++i) {
336  i::DirectHandle<i::Object> obj_key(keys->get(i), isolate);
337  i::DirectHandle<i::Symbol> key(i::Cast<i::Symbol>(*obj_key), isolate);
338  CHECK(key->is_private_name());
339  i::DirectHandle<i::Object> value;
340  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
341  isolate, value, i::Object::GetProperty(isolate, receiver, key), false);
342  if (key->is_private_brand()) {
343  if (include_methods_or_accessors) {
344  DCHECK(IsContext(*value));
345  i::DirectHandle<i::Context> value_context(i::Cast<i::Context>(*value),
346  isolate);
347  ForEachContextLocal(isolate, value_context, var_mode_filter,
348  instance_filter, add_private_entry);
349  }
350  } else if (include_fields) { // Private fields
351  i::DirectHandle<i::String> name(
352  i::Cast<i::String>(i::Cast<i::Symbol>(*key)->description()), isolate);
353  names_out->push_back(Utils::ToLocal(name));
354  values_out->push_back(Utils::ToLocal(value));
355  }
356  }
357 
358  DCHECK_EQ(names_out->size(), values_out->size());
359  DCHECK_LE(names_out->size(), private_entries_count);
360  return true;
361 }

620 bool Script::GetPossibleBreakpoints(
621  const Location& start, const Location& end, bool restrict_to_function,
622  std::vector<BreakLocation>* locations) const {
623  CHECK(!start.IsEmpty());
624  i::Handle<i::Script> script = Utils::OpenHandle(this);
625 #if V8_ENABLE_WEBASSEMBLY
626  if (script->type() == i::Script::Type::kWasm) {
627  i::wasm::NativeModule* native_module = script->wasm_native_module();
628  return i::WasmScript::GetPossibleBreakpoints(native_module, start, end,
629  locations);
630  }
631 #endif // V8_ENABLE_WEBASSEMBLY
632 
633  i::Isolate* isolate = i::Isolate::Current();
634 
635  int start_offset, end_offset;
636  if (!GetSourceOffset(start, GetSourceOffsetMode::kClamp).To(&start_offset)) {
637  return false;
638  }
639  if (end.IsEmpty()) {
640  end_offset = std::numeric_limits<int>::max();
641  } else if (!GetSourceOffset(end, GetSourceOffsetMode::kClamp)
642  .To(&end_offset)) {
643  return false;
644  }
645  if (start_offset >= end_offset) return true;
646 
647  std::vector<i::BreakLocation> v8_locations;
648  if (!isolate->debug()->GetPossibleBreakpoints(
649  script, start_offset, end_offset, restrict_to_function,
650  &v8_locations)) {
651  return false;
652  }
653 
654  std::sort(v8_locations.begin(), v8_locations.end(), CompareBreakLocation);
655  for (const auto& v8_location : v8_locations) {
656  Location location = GetSourceLocation(v8_location.position());
657  locations->emplace_back(location.GetLineNumber(),
658  location.GetColumnNumber(), v8_location.type());
659  }
660  return true;
661 }

663 Maybe<int> Script::GetSourceOffset(const Location& location,
664  GetSourceOffsetMode mode) const {
665  i::DirectHandle<i::Script> script = Utils::OpenDirectHandle(this);
666 #if V8_ENABLE_WEBASSEMBLY
667  if (script->type() == i::Script::Type::kWasm) {
668  return location.GetLineNumber() == 0 ? Just(location.GetColumnNumber())
669  : Nothing<int>();
670  }
671 #endif // V8_ENABLE_WEBASSEMBLY
672 
673  int line = location.GetLineNumber();
674  int column = location.GetColumnNumber();
675  if (!script->HasSourceURLComment()) {
676  // Line/column number for inline <script>s with sourceURL annotation
677  // are supposed to be related to the <script> tag, otherwise they
678  // are relative to the parent file. Keep this in sync with the logic
679  // in GetSourceLocation() below.
680  line -= script->line_offset();
681  if (line == 0) column -= script->column_offset();
682  }
683 
684  i::Isolate* isolate = i::Isolate::Current();
685  i::Script::InitLineEnds(isolate, script);
686  auto line_ends =
687  i::Cast<i::FixedArray>(i::direct_handle(script->line_ends(), isolate));
688  if (line < 0) {
689  if (mode == GetSourceOffsetMode::kClamp) {
690  return Just(0);
691  }
692  return {};
693  }
694  if (line >= line_ends->length()) {
695  if (mode == GetSourceOffsetMode::kClamp) {
696  return Just(GetSmiValue(line_ends, line_ends->length() - 1));
697  }
698  return {};
699  }
700  if (column < 0) {
701  if (mode != GetSourceOffsetMode::kClamp) {
702  return {};
703  }
704  column = 0;
705  }
706  int offset = column;
707  if (line > 0) {
708  int prev_line_end_offset = GetSmiValue(line_ends, line - 1);
709  offset += prev_line_end_offset + 1;
710  }
711  int line_end_offset = GetSmiValue(line_ends, line);
712  if (offset > line_end_offset) {
713  // Be permissive with columns that don't exist,
714  // as long as they are clearly within the range
715  // of the script.
716  if (line < line_ends->length() - 1 || mode == GetSourceOffsetMode::kClamp) {
717  return Just(line_end_offset);
718  }
719  return {};
720  }
721  return Just(offset);
722 }
