472 void Map::AccountAddedPropertyField() {
473  // Update used instance size and unused property fields number.
474  static_assert(JSObject::kFieldsAdded == JSObject::kHeaderSize / kTaggedSize);
475 #ifdef DEBUG
476  int new_unused = UnusedPropertyFields() - 1;
477  if (new_unused < 0) new_unused += JSObject::kFieldsAdded;
478 #endif
479  int value = used_or_unused_instance_size_in_words();
480  if (value >= JSObject::kFieldsAdded) {
481  if (value == instance_size_in_words()) {
482  AccountAddedOutOfObjectPropertyField(0);
483  } else {
484  // The property is added in-object, so simply increment the counter.
485  set_used_or_unused_instance_size_in_words(value + 1);
486  }
487  } else {
488  AccountAddedOutOfObjectPropertyField(value);
489  }
490  DCHECK_EQ(new_unused, UnusedPropertyFields());
491 }

493 void Map::AccountAddedOutOfObjectPropertyField(int unused_in_property_array) {
494  unused_in_property_array--;
495  if (unused_in_property_array < 0) {
496  unused_in_property_array += JSObject::kFieldsAdded;
497  }
498  CHECK_LT(static_cast<unsigned>(unused_in_property_array),
499  JSObject::kFieldsAdded);
500  set_used_or_unused_instance_size_in_words(unused_in_property_array);
501  DCHECK_EQ(unused_in_property_array, UnusedPropertyFields());
502 }
