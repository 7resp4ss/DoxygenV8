175 class V8_EXPORT_PRIVATE CallInterfaceDescriptorData {
176  public:
177  enum Flag {
178  kNoFlags = 0u,
179  kNoContext = 1u << 0,
180  // This indicates that the code uses a special frame that does not scan the
181  // stack arguments, e.g. EntryFrame. And this allows the code to use
182  // untagged stack arguments.
183  kNoStackScan = 1u << 1,
184  // In addition to the specified parameters, additional arguments can be
185  // passed on the stack.
186  // This does not indicate if arguments adaption is used or not.
187  kAllowVarArgs = 1u << 2,
188  // Callee save allocatable_registers.
189  kCalleeSaveRegisters = 1u << 3,
190  };
191  using Flags = base::Flags<Flag>;
192 
193  static constexpr int kUninitializedCount = -1;
194 
195  CallInterfaceDescriptorData() = default;
196 
197  CallInterfaceDescriptorData(const CallInterfaceDescriptorData&) = delete;
198  CallInterfaceDescriptorData& operator=(const CallInterfaceDescriptorData&) =
199  delete;

202  // expected to exceed that of this data. In practice, they are expected to
203  // be in a static local.

205  int parameter_count, StackArgumentOrder stack_order,
206  int register_parameter_count,
207  const Register* registers,

209  const Register* return_registers,
210  const DoubleRegister* return_double_registers);
211 
212  // if machine_types is null, then an array of size
213  // (return_count + parameter_count) will be created with
214  // MachineType::AnyTagged() for each member.
215  //
216  // if machine_types is not null, then it should be of the size
217  // (return_count + parameter_count). Those members of the parameter array will

219  // MachineType::AnyTagged().

335 // To support all possible cases, we must limit the number of register args for
336 // TFS builtins on ia32 to 3. Out of the 6 allocatable registers, esi is taken
337 // as the context register and ebx is the root register. One register must
338 // remain available to store the jump/call target. Thus 3 registers remain for
339 // arguments. The reason this applies to TFS builtins specifically is because
340 // this becomes relevant for builtins used as targets of Torque function
341 // pointers (which must have a register available to store the target).
342 // TODO(jgruber): Ideally we should just decrement kMaxBuiltinRegisterParams but
343 // that comes with its own set of complications. It's possible, but requires
344 // refactoring the calling convention of other existing stubs.

479 class StaticCallInterfaceDescriptor : public CallInterfaceDescriptor {
480  public:
481  // ===========================================================================
482  // The following are the descriptor's CRTP configuration points, overwritable
483  // by DerivedDescriptor.
484  static constexpr int kReturnCount =
485  CallInterfaceDescriptorData::kUninitializedCount;
486  static constexpr int kParameterCount =
487  CallInterfaceDescriptorData::kUninitializedCount;
488  static constexpr bool kNoContext = false;
489  static constexpr bool kAllowVarArgs = false;
490  static constexpr bool kNoStackScan = false;
491  static constexpr auto kStackArgumentOrder = StackArgumentOrder::kDefault;
492 
493  // The set of registers available to the parameters, as a
494  // std::array<Register,N>. Can be larger or smaller than kParameterCount; if
495  // larger then any remaining registers are ignored; if smaller, any parameters
496  // after registers().size() will be stack registers.
497  //
498  // Defaults to CallInterfaceDescriptor::DefaultRegisterArray().
499  static constexpr inline auto registers();
500  static constexpr inline auto double_registers();
501  static constexpr inline auto return_registers();
502  static constexpr inline auto return_double_registers();
503 
504  // An additional limit on the number of register parameters allowed. This is
505  // here so that it can be overwritten to kMaxTFSBuiltinRegisterParams for TFS
506  // builtins, see comment on kMaxTFSBuiltinRegisterParams above.
507  static constexpr int kMaxRegisterParams = kMaxBuiltinRegisterParams;
508 
509  // If set to true, the descriptor will restrict the set of allocatable
510  // registers to the set returned by registers(). Then, it is expected that
511  // the first kParameterCount registers() are the parameters of the builtin.
512  static constexpr bool kRestrictAllocatableRegisters = false;
513 
514  // If set to true, builtins will callee save the set returned by registers().
515  static constexpr bool kCalleeSaveRegisters = false;
516 
517  // If set to true, the descriptor will define a kMachineTypes array with the
518  // types of each result value and parameter.
519  static constexpr bool kCustomMachineTypes = false;
520 
521  // End of customization points.
522  // ===========================================================================
523 
524  static constexpr inline Flags flags() {
525  return Flags((DerivedDescriptor::kNoContext
526  ? CallInterfaceDescriptorData::kNoContext
527  : 0) |
528  (DerivedDescriptor::kAllowVarArgs
529  ? CallInterfaceDescriptorData::kAllowVarArgs
530  : 0) |
531  (DerivedDescriptor::kNoStackScan
532  ? CallInterfaceDescriptorData::kNoStackScan
533  : 0) |
534  (DerivedDescriptor::kCalleeSaveRegisters
535  ? CallInterfaceDescriptorData::kCalleeSaveRegisters
536  : 0));
537  }
538  static constexpr inline bool AllowVarArgs() {
539  return DerivedDescriptor::kAllowVarArgs;
540  }
541  static constexpr inline bool HasContextParameter() {

543  }
