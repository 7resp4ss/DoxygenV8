119 // Dependencies can only be fully deduplicated immediately prior to
120 // installation (because PrepareInstall may create the object on which the dep
121 // will be installed). We gather and dedupe deps in this class, and install
122 // them from here.
123 class PendingDependencies final {
124  public:
125  explicit PendingDependencies(Zone* zone)
126  : deps_(8, {}, ZoneAllocationPolicy(zone)) {}
127 
128  void Register(Handle<HeapObject> object,
129  DependentCode::DependencyGroup group) {
130  // InstructionStream, which are per-local Isolate, cannot depend on objects
131  // in the shared or RO heaps. Shared and RO heap dependencies are designed
132  // to never invalidate assumptions. E.g., maps for shared structs do not
133  // have transitions or change the shape of their fields. See
134  // DependentCode::DeoptimizeDependencyGroups for corresponding DCHECK.
135  if (HeapLayout::InWritableSharedSpace(*object) ||
136  HeapLayout::InReadOnlySpace(*object))
137  return;
138  deps_.LookupOrInsert(object, HandleValueHash(object))->value |= group;
139  }
140 
141  void InstallAll(Isolate* isolate, Handle<Code> code) {
142  if (V8_UNLIKELY(v8_flags.predictable)) {
143  InstallAllPredictable(isolate, code);
144  return;
145  }
146 
147  // With deduplication done we no longer rely on the object address for
148  // hashing.
149  AllowGarbageCollection yes_gc;
150  for (auto* entry = deps_.Start(); entry != nullptr;
151  entry = deps_.Next(entry)) {
152  DependentCode::InstallDependency(isolate, code, entry->key, entry->value);
153  }
154  deps_.Invalidate();
155  }
156 
157  void InstallAllPredictable(Isolate* isolate, Handle<Code> code) {
158  CHECK(v8_flags.predictable);
159  // First, guarantee predictable iteration order.
160  using DepsMap = decltype(deps_);
161  std::vector<const DepsMap::Entry*> entries;
162  entries.reserve(deps_.occupancy());
163  for (auto* entry = deps_.Start(); entry != nullptr;
164  entry = deps_.Next(entry)) {
165  entries.push_back(entry);
166  }
167 
168  std::sort(entries.begin(), entries.end(),
169  [](const DepsMap::Entry* lhs, const DepsMap::Entry* rhs) {
170  return lhs->key->ptr() < rhs->key->ptr();
171  });
172 
173  // With deduplication done we no longer rely on the object address for
174  // hashing.
175  AllowGarbageCollection yes_gc;
176  for (const auto* entry : entries) {
177  DependentCode::InstallDependency(isolate, code, entry->key, entry->value);
178  }
179  deps_.Invalidate();
180  }
181 
182  private:
183  uint32_t HandleValueHash(DirectHandle<HeapObject> handle) {
184  return static_cast<uint32_t>(base::hash_value(handle->ptr()));
185  }
186  struct HandleValueEqual {
187  bool operator()(uint32_t hash1, uint32_t hash2,
188  DirectHandle<HeapObject> lhs,
189  Handle<HeapObject> rhs) const {
190  return hash1 == hash2 && lhs.is_identical_to(rhs);
191  }
192  };
193 
194  base::TemplateHashMapImpl<Handle<HeapObject>, DependentCode::DependencyGroups,
195  HandleValueEqual, ZoneAllocationPolicy>
196  deps_;
197 };

317 class ConstantInDictionaryPrototypeChainDependency final
318  : public CompilationDependency {
319  public:
320  explicit ConstantInDictionaryPrototypeChainDependency(
321  const MapRef receiver_map, const NameRef property_name,
322  const ObjectRef constant, PropertyKind kind)
323  : CompilationDependency(kConstantInDictionaryPrototypeChain),
324  receiver_map_(receiver_map),
325  property_name_{property_name},
326  constant_{constant},
327  kind_{kind} {
328  DCHECK(V8_DICT_PROPERTY_CONST_TRACKING_BOOL);
329  }
330 
331  // Checks that |constant_| is still the value of accessing |property_name_|
332  // starting at |receiver_map_|.
333  bool IsValid(JSHeapBroker* broker) const override {
334  return !GetHolderIfValid(broker).is_null();
335  }
336 
337  void Install(JSHeapBroker* broker, PendingDependencies* deps) const override {
338  SLOW_DCHECK(IsValid(broker));
339  Isolate* isolate = broker->isolate();
340  DirectHandle<JSObject> holder = GetHolderIfValid(broker).ToHandleChecked();
341  Handle<Map> map = receiver_map_.object();
342 
343  while (map->prototype() != *holder) {
344  map = handle(map->prototype()->map(), isolate);
345  DCHECK(IsJSObjectMap(*map)); // Due to IsValid holding.
346  deps->Register(map, DependentCode::kPrototypeCheckGroup);
347  }
348 
349  DCHECK(IsJSObjectMap(map->prototype()->map())); // Due to IsValid holding.
350  deps->Register(handle(map->prototype()->map(), isolate),
351  DependentCode::kPrototypeCheckGroup);
352  }
353 
354  private:
355  // If the dependency is still valid, returns holder of the constant. Otherwise
356  // returns null.
357  // TODO(neis) Currently, invoking IsValid and then Install duplicates the call
358  // to GetHolderIfValid. Instead, consider letting IsValid change the state
359  // (and store the holder), or merge IsValid and Install.
360  MaybeHandle<JSObject> GetHolderIfValid(JSHeapBroker* broker) const {
361  DisallowGarbageCollection no_gc;
362  Isolate* isolate = broker->isolate();
363 
364  Tagged<HeapObject> prototype = receiver_map_.object()->prototype();
365 
366  enum class ValidationResult { kFoundCorrect, kFoundIncorrect, kNotFound };
367  auto try_load = [&](auto dictionary) -> ValidationResult {
368  InternalIndex entry =
369  dictionary->FindEntry(isolate, property_name_.object());
370  if (entry.is_not_found()) {
371  return ValidationResult::kNotFound;
372  }
373 
374  PropertyDetails details = dictionary->DetailsAt(entry);
375  if (details.constness() != PropertyConstness::kConst) {
376  return ValidationResult::kFoundIncorrect;
377  }
378 
379  Tagged<Object> dictionary_value = dictionary->ValueAt(entry);
380  Tagged<Object> value;
381  // We must be able to detect the case that the property |property_name_|
382  // of |holder_| was originally a plain function |constant_| (when creating
383  // this dependency) and has since become an accessor whose getter is
384  // |constant_|. Therefore, we cannot just look at the property kind of
385  // |details|, because that reflects the current situation, not the one
386  // when creating this dependency.
387  if (details.kind() != kind_) {
388  return ValidationResult::kFoundIncorrect;
389  }
390  if (kind_ == PropertyKind::kAccessor) {
391  if (!IsAccessorPair(dictionary_value)) {
392  return ValidationResult::kFoundIncorrect;
393  }
394  // Only supporting loading at the moment, so we only ever want the
395  // getter.
396  value = Cast<AccessorPair>(dictionary_value)
397  ->get(AccessorComponent::ACCESSOR_GETTER);
398  } else {
399  value = dictionary_value;
400  }
401  return value == *constant_.object() ? ValidationResult::kFoundCorrect
402  : ValidationResult::kFoundIncorrect;
403  };
404 
405  while (IsJSObject(prototype)) {
406  // We only care about JSObjects because that's the only type of holder
407  // (and types of prototypes on the chain to the holder) that
408  // AccessInfoFactory::ComputePropertyAccessInfo allows.
409  Tagged<JSObject> object = Cast<JSObject>(prototype);
410 
411  // We only support dictionary mode prototypes on the chain for this kind
412  // of dependency.
413  CHECK(!object->HasFastProperties());
414 
415  ValidationResult result =
416  V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL
417  ? try_load(object->property_dictionary_swiss())
418  : try_load(object->property_dictionary());
419 
420  if (result == ValidationResult::kFoundCorrect) {
421  return handle(object, isolate);
422  } else if (result == ValidationResult::kFoundIncorrect) {
423  return MaybeHandle<JSObject>();
424  }
425 
426  // In case of kNotFound, continue walking up the chain.
427  prototype = object->map()->prototype();
428  }
429 
430  return MaybeHandle<JSObject>();
431  }
432 
433  size_t Hash() const override {
434  ObjectRef::Hash h;
435  return base::hash_combine(h(receiver_map_), h(property_name_), h(constant_),
436  static_cast<int>(kind_));
437  }
438 
439  bool Equals(const CompilationDependency* that) const override {
440  const ConstantInDictionaryPrototypeChainDependency* const zat =
441  that->AsConstantInDictionaryPrototypeChain();
442  return receiver_map_.equals(zat->receiver_map_) &&
443  property_name_.equals(zat->property_name_) &&
444  constant_.equals(zat->constant_) && kind_ == zat->kind_;
445  }
446 
447  const MapRef receiver_map_;
448  const NameRef property_name_;
449  const ObjectRef constant_;
450  const PropertyKind kind_;
451 };

1456 bool CompilationDependencies::Commit(Handle<Code> code) {
1457  if (!PrepareInstall()) return false;
1458 
1459  {
1460  PendingDependencies pending_deps(zone_);
1461  DisallowCodeDependencyChange no_dependency_change;
1462  for (const CompilationDependency* dep : dependencies_) {
1463  // Check each dependency's validity again right before installing it,
1464  // because the first iteration above might have invalidated some
1465  // dependencies. For example, PrototypePropertyDependency::PrepareInstall
1466  // can call EnsureHasInitialMap, which can invalidate a
1467  // StableMapDependency on the prototype object's map.
1468  if (!dep->IsValid(broker_)) {
1469  if (v8_flags.trace_compilation_dependencies) {
1470  TraceInvalidCompilationDependency(broker_, dep);
1471  }
1472  dependencies_.clear();
1473  return false;
1474  }
1475  dep->Install(broker_, &pending_deps);
1476  }
1477  pending_deps.InstallAll(broker_->isolate(), code);
1478  }
1479 
1480  // It is even possible that a GC during the above installations invalidated
1481  // one of the dependencies. However, this should only affect
1482  //
1483  // 1. pretenure mode dependencies, or
1484  // 2. function consistency dependencies,
1485  //
1486  // which we assert below. It is safe to return successfully in these cases,
1487  // because
1488  //
1489  // 1. once the code gets executed it will do a stack check that triggers its
1490  // deoptimization.
1491  // 2. since the function state was deemed consistent above, that means the
1492  // compilation saw a self-consistent state of the jsfunction.
1493  if (v8_flags.stress_gc_during_compilation) {
1494  broker_->isolate()->heap()->PreciseCollectAllGarbage(
1495  GCFlag::kForced, GarbageCollectionReason::kTesting, kNoGCCallbackFlags);
1496  }
1497 #ifdef DEBUG
1498  for (auto dep : dependencies_) {
1499  CHECK_IMPLIES(!dep->IsValid(broker_),
1500  dep->IsPretenureMode() || dep->IsConsistentJSFunctionView());
1501  }
1502 #endif
1503 
1504  dependencies_.clear();
1505  return true;
1506 }
