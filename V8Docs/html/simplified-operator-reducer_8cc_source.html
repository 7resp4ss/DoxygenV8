25 Decision DecideObjectIsSmi(Node* const input) {
26  NumberMatcher m(input);
27  if (m.HasResolvedValue()) {
28  return IsSmiDouble(m.ResolvedValue()) ? Decision::kTrue : Decision::kFalse;
29  }
30  if (m.IsAllocate()) return Decision::kFalse;
31  if (m.IsChangeBitToTagged()) return Decision::kFalse;
32  if (m.IsChangeInt31ToTaggedSigned()) return Decision::kTrue;
33  if (m.IsHeapConstant()) return Decision::kFalse;
34  return Decision::kUnknown;
35 }

50 Reduction SimplifiedOperatorReducer::Reduce(Node* node) {
51  switch (node->opcode()) {
52  case IrOpcode::kBooleanNot: {
53  HeapObjectMatcher m(node->InputAt(0));
54  if (m.Is(factory()->true_value())) return ReplaceBoolean(false);
55  if (m.Is(factory()->false_value())) return ReplaceBoolean(true);
56  if (m.IsBooleanNot()) return Replace(m.InputAt(0));
57  break;
58  }
59  case IrOpcode::kChangeBitToTagged: {
60  Int32Matcher m(node->InputAt(0));
61  if (m.Is(0)) return Replace(jsgraph()->FalseConstant());
62  if (m.Is(1)) return Replace(jsgraph()->TrueConstant());
63  if (m.IsChangeTaggedToBit()) return Replace(m.InputAt(0));
64  break;
65  }
66  case IrOpcode::kChangeTaggedToBit: {
67  HeapObjectMatcher m(node->InputAt(0));
68  if (m.HasResolvedValue()) {
69  std::optional<bool> maybe_result =
70  m.Ref(broker()).TryGetBooleanValue(broker());
71  if (maybe_result.has_value()) return ReplaceInt32(*maybe_result);
72  }
73  if (m.IsChangeBitToTagged()) return Replace(m.InputAt(0));
74  break;
75  }
76  case IrOpcode::kChangeFloat64ToTagged: {
77  Float64Matcher m(node->InputAt(0));
78  if (m.HasResolvedValue()) return ReplaceNumber(m.ResolvedValue());
79  if (m.IsChangeTaggedToFloat64()) return Replace(m.node()->InputAt(0));
80  break;
81  }
82  case IrOpcode::kChangeFloat64OrUndefinedToTagged: {
83  // TODO(385155404): Handle some constants here.
84  break;
85  }
86  case IrOpcode::kChangeInt31ToTaggedSigned:
87  case IrOpcode::kChangeInt32ToTagged: {
88  Int32Matcher m(node->InputAt(0));
89  if (m.HasResolvedValue()) return ReplaceNumber(m.ResolvedValue());
90  if (m.IsChangeTaggedSignedToInt32()) {
91  return Replace(m.InputAt(0));
92  }
93  break;
94  }
95  case IrOpcode::kChangeTaggedToFloat64:
96  case IrOpcode::kTruncateTaggedToFloat64: {
97  NumberMatcher m(node->InputAt(0));
98  if (m.HasResolvedValue()) return ReplaceFloat64(m.ResolvedValue());
99  // TODO(385155404): Consider handling ChangeFloat64OrUndefinedToTagged
100  // here.
101  if (m.IsChangeFloat64ToTagged() || m.IsChangeFloat64ToTaggedPointer()) {
102  return Replace(m.node()->InputAt(0));
103  }
104  if (m.IsChangeInt31ToTaggedSigned() || m.IsChangeInt32ToTagged()) {
105  return Change(node, machine()->ChangeInt32ToFloat64(), m.InputAt(0));
106  }
107  if (m.IsChangeUint32ToTagged()) {
108  return Change(node, machine()->ChangeUint32ToFloat64(), m.InputAt(0));
109  }
110  break;
111  }
112  case IrOpcode::kChangeTaggedSignedToInt32:
113  case IrOpcode::kChangeTaggedToInt32: {
114  NumberMatcher m(node->InputAt(0));
115  if (m.HasResolvedValue())
116  return ReplaceInt32(DoubleToInt32(m.ResolvedValue()));
117  // TODO(385155404): Consider handling ChangeFloat64OrUndefinedToTagged
118  // here.
119  if (m.IsChangeFloat64ToTagged() || m.IsChangeFloat64ToTaggedPointer()) {
120  return Change(node, machine()->ChangeFloat64ToInt32(), m.InputAt(0));
121  }
122  if (m.IsChangeInt31ToTaggedSigned() || m.IsChangeInt32ToTagged()) {
123  return Replace(m.InputAt(0));
124  }
125  break;
126  }
127  case IrOpcode::kChangeTaggedToUint32: {
128  NumberMatcher m(node->InputAt(0));
129  if (m.HasResolvedValue())
130  return ReplaceUint32(DoubleToUint32(m.ResolvedValue()));
131  // TODO(385155404): Consider handling ChangeFloat64OrUndefinedToTagged
132  // here.
133  if (m.IsChangeFloat64ToTagged() || m.IsChangeFloat64ToTaggedPointer()) {
134  return Change(node, machine()->ChangeFloat64ToUint32(), m.InputAt(0));
135  }
136  if (m.IsChangeUint32ToTagged()) return Replace(m.InputAt(0));
137  break;
138  }
139  case IrOpcode::kChangeUint32ToTagged: {
140  Uint32Matcher m(node->InputAt(0));
141  if (m.HasResolvedValue())
142  return ReplaceNumber(FastUI2D(m.ResolvedValue()));
143  break;
144  }
145  case IrOpcode::kTruncateTaggedToWord32: {
146  NumberMatcher m(node->InputAt(0));
147  if (m.HasResolvedValue())
148  return ReplaceInt32(DoubleToInt32(m.ResolvedValue()));
149  if (m.IsChangeInt31ToTaggedSigned() || m.IsChangeInt32ToTagged() ||
150  m.IsChangeUint32ToTagged()) {
151  return Replace(m.InputAt(0));
152  }
153  // TODO(385155404): Consider handling ChangeFloat64OrUndefinedToTagged
154  // here.
155  if (m.IsChangeFloat64ToTagged() || m.IsChangeFloat64ToTaggedPointer()) {
156  return Change(node, machine()->TruncateFloat64ToWord32(), m.InputAt(0));
157  }
158  break;
159  }
160  case IrOpcode::kCheckedFloat64ToInt32: {
161  Float64Matcher m(node->InputAt(0));
162  if (m.HasResolvedValue() && IsInt32Double(m.ResolvedValue())) {
163  Node* value =
164  jsgraph()->Int32Constant(static_cast<int32_t>(m.ResolvedValue()));
165  ReplaceWithValue(node, value);
166  return Replace(value);
167  }
168  break;
169  }
170  case IrOpcode::kCheckedTaggedToArrayIndex:
171  case IrOpcode::kCheckedTaggedToInt32:
172  case IrOpcode::kCheckedTaggedSignedToInt32: {
173  NodeMatcher m(node->InputAt(0));
174  if (m.IsConvertTaggedHoleToUndefined()) {
175  node->ReplaceInput(0, m.InputAt(0));
176  return Changed(node);
177  }
178  break;
179  }
180  case IrOpcode::kCheckIf: {
181  HeapObjectMatcher m(node->InputAt(0));
182  if (m.Is(factory()->true_value())) {
183  Node* const effect = NodeProperties::GetEffectInput(node);
184  return Replace(effect);
185  }
186  break;
187  }
188  case IrOpcode::kCheckNumberFitsInt32:
189  case IrOpcode::kCheckNumber: {
190  NodeMatcher m(node->InputAt(0));
191  if (m.IsConvertTaggedHoleToUndefined()) {
192  node->ReplaceInput(0, m.InputAt(0));
193  return Changed(node);
194  }
195  break;
196  }
197  case IrOpcode::kCheckHeapObject: {
198  Node* const input = node->InputAt(0);
199  if (DecideObjectIsSmi(input) == Decision::kFalse) {
200  ReplaceWithValue(node, input);
201  return Replace(input);
202  }
203  NodeMatcher m(input);
204  if (m.IsCheckHeapObject()) {
205  ReplaceWithValue(node, input);
206  return Replace(input);
207  }
208  break;
209  }
210  case IrOpcode::kCheckSmi: {
211  Node* const input = node->InputAt(0);
212  if (DecideObjectIsSmi(input) == Decision::kTrue) {
213  ReplaceWithValue(node, input);
214  return Replace(input);
215  }
216  NodeMatcher m(input);
217  if (m.IsCheckSmi()) {
218  ReplaceWithValue(node, input);
219  return Replace(input);
220  } else if (m.IsConvertTaggedHoleToUndefined()) {
221  node->ReplaceInput(0, m.InputAt(0));
222  return Changed(node);
223  }
224  break;
225  }
226  case IrOpcode::kObjectIsSmi: {
227  Node* const input = node->InputAt(0);
228  switch (DecideObjectIsSmi(input)) {
229  case Decision::kTrue:
230  return ReplaceBoolean(true);
231  case Decision::kFalse:
232  return ReplaceBoolean(false);
233  case Decision::kUnknown:
234  break;
235  }
236  break;
237  }
238  case IrOpcode::kNumberAbs: {
239  NumberMatcher m(node->InputAt(0));
240  if (m.HasResolvedValue())
241  return ReplaceNumber(std::fabs(m.ResolvedValue()));
242  break;
243  }
244  case IrOpcode::kReferenceEqual: {
245  HeapObjectBinopMatcher m(node);
246  if (m.left().node() == m.right().node()) return ReplaceBoolean(true);
247  break;
248  }
249  case IrOpcode::kCheckedInt32Add: {
250  // (x + a) + b => x + (a + b) where a and b are constants and have the
251  // same sign.
252  Int32BinopMatcher m(node);
253  if (m.right().HasResolvedValue()) {
254  Node* checked_int32_add = m.left().node();
255  if (checked_int32_add->opcode() == IrOpcode::kCheckedInt32Add) {
256  Int32BinopMatcher n(checked_int32_add);
257  if (n.right().HasResolvedValue() &&
258  (n.right().ResolvedValue() >= 0) ==
259  (m.right().ResolvedValue() >= 0)) {
260  int32_t val;
261  bool overflow = base::bits::SignedAddOverflow32(
262  n.right().ResolvedValue(), m.right().ResolvedValue(), &val);
263  if (!overflow) {
264  bool has_no_other_uses = true;
265  for (Edge edge : checked_int32_add->use_edges()) {
266  if (!edge.from()->IsDead() && edge.from() != node) {
267  has_no_other_uses = false;
268  break;
269  }
270  }
271  if (has_no_other_uses) {
272  node->ReplaceInput(0, n.left().node());
273  node->ReplaceInput(1, jsgraph()->Int32Constant(val));
274  RelaxEffectsAndControls(checked_int32_add);
275  checked_int32_add->Kill();
276  return Changed(node);
277  }
278  }
279  }
280  }
281  }
282  break;
283  }
284  default:
285  break;
286  }
287  return NoChange();
288 }

290 Reduction SimplifiedOperatorReducer::Change(Node* node, const Operator* op,
291  Node* a) {
292  DCHECK_EQ(node->InputCount(), OperatorProperties::GetTotalInputCount(op));
293  DCHECK_LE(1, node->InputCount());
294  node->ReplaceInput(0, a);
295  NodeProperties::ChangeOp(node, op);
296  return Changed(node);
297 }
