34 std::atomic<int>* StatsCounter::SetupPtrFromStatsTable() {
35  // {Init} must have been called.
36  DCHECK_NOT_NULL(counters_);
37  DCHECK_NOT_NULL(name_);
38  int* location = counters_->FindLocation(name_);
39  std::atomic<int>* ptr =
40  location ? base::AsAtomicPtr(location) : &unused_counter_dump;
41 #ifdef DEBUG
42  std::atomic<int>* old_ptr = ptr_.exchange(ptr, std::memory_order_release);
43  DCHECK_IMPLIES(old_ptr, old_ptr == ptr);
44 #else
45  ptr_.store(ptr, std::memory_order_release);
46 #endif
47  return ptr;
48 }

123 void CountersInitializer::Visit(LegacyMemoryHistogram* histogram,
124  const char* caption) {
125  // Exponential histogram assigns bucket limits to points
126  // p[1], p[2], ... p[n] such that p[i+1] / p[i] = constant.
127  // The constant factor is equal to the n-th root of (high / low),
128  // where the n is the number of buckets, the low is the lower limit,
129  // the high is the upper limit.
130  // For n = 50, low = 1000, high = 500000: the factor = 1.13.
131  histogram->Initialize(caption, 1000, 500000, 50, counters());
132 }
