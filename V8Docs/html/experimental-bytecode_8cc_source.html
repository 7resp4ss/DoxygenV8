117 // The maximum number of digits required to display a non-negative number < n
118 // in base 10.
119 int DigitsRequiredBelow(int n) {
120  DCHECK_GE(n, 0);
121 
122  int result = 1;
123  for (int i = 10; i < n; i *= 10) {
124  result += 1;
125  }
126  return result;
127 }
