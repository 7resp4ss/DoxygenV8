51 bool IsWasmCompileAllowed(v8::Isolate* isolate, v8::Local<v8::Value> value,
52  bool is_async) {
53  base::MutexGuard guard(g_PerIsolateWasmControlsMutex.Pointer());
54  DCHECK_GT(GetPerIsolateWasmControls()->count(isolate), 0);
55  const WasmCompileControls& ctrls = GetPerIsolateWasmControls()->at(isolate);
56  return (is_async && ctrls.AllowAnySizeForAsync) ||
57  (value->IsArrayBuffer() && value.As<v8::ArrayBuffer>()->ByteLength() <=
58  ctrls.MaxWasmBufferSize) ||
59  (value->IsArrayBufferView() &&
60  value.As<v8::ArrayBufferView>()->ByteLength() <=
61  ctrls.MaxWasmBufferSize);
62 }

64 // Use the compile controls for instantiation, too
65 bool IsWasmInstantiateAllowed(v8::Isolate* isolate,
66  v8::Local<v8::Value> module_or_bytes,
67  bool is_async) {
68  base::MutexGuard guard(g_PerIsolateWasmControlsMutex.Pointer());
69  DCHECK_GT(GetPerIsolateWasmControls()->count(isolate), 0);
70  const WasmCompileControls& ctrls = GetPerIsolateWasmControls()->at(isolate);
71  if (is_async && ctrls.AllowAnySizeForAsync) return true;
72  if (!module_or_bytes->IsWasmModuleObject()) {
73  return IsWasmCompileAllowed(isolate, module_or_bytes, is_async);
74  }
75  v8::Local<v8::WasmModuleObject> module =
76  v8::Local<v8::WasmModuleObject>::Cast(module_or_bytes);
77  return static_cast<uint32_t>(
78  module->GetCompiledModule().GetWireBytesRef().size()) <=
79  ctrls.MaxWasmBufferSize;
80 }

158 RUNTIME_FUNCTION(Runtime_CountUnoptimizedWasmToJSWrapper) {
159  SealHandleScope shs(isolate);
160  if (args.length() != 1 || !IsWasmInstanceObject(args[0])) {
161  return CrashUnlessFuzzing(isolate);
162  }
163  Tagged<WasmInstanceObject> instance_object =
164  Cast<WasmInstanceObject>(args[0]);
165  Tagged<WasmTrustedInstanceData> trusted_data =
166  instance_object->trusted_data(isolate);
167  Address wrapper_entry =
168  Builtins::EmbeddedEntryOf(Builtin::kWasmToJsWrapperAsm);
169 
170  int result = 0;
171  Tagged<WasmDispatchTable> dispatch_table =
172  trusted_data->dispatch_table_for_imports();
173  int import_count = dispatch_table->length();
174  wasm::WasmCodePointerTable* cpt = wasm::GetProcessWideWasmCodePointerTable();
175  for (int i = 0; i < import_count; ++i) {
176  if (cpt->EntrypointEqualTo(dispatch_table->target(i), wrapper_entry)) {
177  ++result;
178  }
179  }
180  Tagged<ProtectedFixedArray> dispatch_tables = trusted_data->dispatch_tables();
181  int table_count = dispatch_tables->length();
182  for (int table_index = 0; table_index < table_count; ++table_index) {
183  if (dispatch_tables->get(table_index) == Smi::zero()) continue;
184  Tagged<WasmDispatchTable> table =
185  Cast<WasmDispatchTable>(dispatch_tables->get(table_index));
186  int table_size = table->length();
187  for (int entry_index = 0; entry_index < table_size; ++entry_index) {
188  WasmCodePointer target = table->target(entry_index);
189  if (target != wasm::kInvalidWasmCodePointer &&
190  cpt->EntrypointEqualTo(target, wrapper_entry))
191  ++result;
192  }
193  }
194  return Smi::FromInt(result);
195 }

289 RUNTIME_FUNCTION(Runtime_WasmTraceExit) {
290  HandleScope shs(isolate);
291  // This isn't exposed to fuzzers so doesn't need to handle invalid arguments.
292  DCHECK_EQ(1, args.length());
293  Tagged<Smi> return_addr_smi = Cast<Smi>(args[0]);
294 
295  PrintIndentation(WasmStackSize(isolate));
296  PrintF("}");
297 
298  // Find the caller wasm frame.
299  wasm::WasmCodeRefScope wasm_code_ref_scope;
300  DebuggableStackFrameIterator it(isolate);
301  DCHECK(!it.done());
302  DCHECK(it.is_wasm());
303 #if V8_ENABLE_DRUMBRAKE
304  DCHECK(!it.is_wasm_interpreter_entry());
305 #endif // V8_ENABLE_DRUMBRAKE
306  WasmFrame* frame = WasmFrame::cast(it.frame());
307  int func_index = frame->function_index();
308  const wasm::WasmModule* module = frame->trusted_instance_data()->module();
309  const wasm::FunctionSig* sig = module->functions[func_index].sig;
310 
311  size_t num_returns = sig->return_count();
312  // If we have no returns, we should have passed {Smi::zero()}.
313  DCHECK_IMPLIES(num_returns == 0, IsZero(return_addr_smi));
314  if (num_returns == 1) {
315  wasm::ValueType return_type = sig->GetReturn(0);
316  switch (return_type.kind()) {
317  case wasm::kI32: {
318  int32_t value =
319  base::ReadUnalignedValue<int32_t>(return_addr_smi.ptr());
320  PrintF(" -> %d\n", value);
321  break;
322  }
323  case wasm::kI64: {
324  int64_t value =
325  base::ReadUnalignedValue<int64_t>(return_addr_smi.ptr());
326  PrintF(" -> %" PRId64 "\n", value);
327  break;
328  }
329  case wasm::kF32: {
330  float value = base::ReadUnalignedValue<float>(return_addr_smi.ptr());
331  PrintF(" -> %f\n", value);
332  break;
333  }
334  case wasm::kF64: {
335  double value = base::ReadUnalignedValue<double>(return_addr_smi.ptr());
336  PrintF(" -> %f\n", value);
337  break;
338  }
339  default:
340  PrintF(" -> Unsupported type\n");
341  break;
342  }
343  } else {
344  // TODO(wasm) Handle multiple return values.
345  PrintF("\n");
346  }
347 
348  return ReadOnlyRoots(isolate).undefined_value();
349 }

680  wasm::NativeModule* native_module,

698  module->set_function_validated(func_index);

757 static Tagged<Object> CreateWasmObject(Isolate* isolate,
758  base::Vector<const uint8_t> module_bytes,
759  bool is_struct) {
760  if (module_bytes.size() > v8_flags.wasm_max_module_size) {
761  return CrashUnlessFuzzing(isolate);
762  }
763  // Create and compile the wasm module.
764  wasm::ErrorThrower thrower(isolate, "CreateWasmObject");
765  base::OwnedVector<const uint8_t> bytes = base::OwnedCopyOf(module_bytes);
766  wasm::WasmEngine* engine = wasm::GetWasmEngine();
767  MaybeDirectHandle<WasmModuleObject> maybe_module_object = engine->SyncCompile(
768  isolate, wasm::WasmEnabledFeatures(), wasm::CompileTimeImports(),
769  &thrower, std::move(bytes));
770  CHECK(!thrower.error());
771  DirectHandle<WasmModuleObject> module_object;
772  if (!maybe_module_object.ToHandle(&module_object)) {
773  DCHECK(isolate->has_exception());
774  return ReadOnlyRoots(isolate).exception();
775  }
776  // Instantiate the module.
777  MaybeDirectHandle<WasmInstanceObject> maybe_instance =
778  engine->SyncInstantiate(isolate, &thrower, module_object,
779  Handle<JSReceiver>::null(),
780  MaybeDirectHandle<JSArrayBuffer>());
781  CHECK(!thrower.error());
782  DirectHandle<WasmInstanceObject> instance;
783  if (!maybe_instance.ToHandle(&instance)) {
784  DCHECK(isolate->has_exception());
785  return ReadOnlyRoots(isolate).exception();
786  }
787  wasm::WasmValue value(int64_t{0x7AADF00DBAADF00D});
788  wasm::ModuleTypeIndex type_index{0};
789  Tagged<Map> map = Tagged<Map>::cast(
790  instance->trusted_data(isolate)->managed_object_maps()->get(
791  type_index.index));
792  if (is_struct) {
793  const wasm::StructType* struct_type =
794  instance->module()->struct_type(type_index);
795  DCHECK_EQ(struct_type->field_count(), 1);
796  DCHECK_EQ(struct_type->field(0), wasm::kWasmI64);
797  return *isolate->factory()->NewWasmStruct(struct_type, &value,
798  direct_handle(map, isolate));
799  } else {
800  DCHECK_EQ(instance->module()->array_type(type_index)->element_type(),
801  wasm::kWasmI64);
802  return *isolate->factory()->NewWasmArray(wasm::kWasmI64, 1, value,
803  direct_handle(map, isolate));
804  }
805 }

1085  auto add_input_bytes = [&input_bytes](void* bytes, size_t max_bytes) {
1086  size_t num_bytes = std::min(kMaxInputBytes - input_bytes.size(), max_bytes);
1087  input_bytes.resize(input_bytes.size() + num_bytes);
1088  memcpy(input_bytes.end() - num_bytes, bytes, num_bytes);
1089  };
1090  if (args.length() == 0) {
1091  // If we are called without any arguments, use the RNG from the isolate to
1092  // generate between 1 and kMaxInputBytes random bytes.
1093  int num_bytes =
1094  1 + isolate->random_number_generator()->NextInt(kMaxInputBytes);
1095  input_bytes.resize(num_bytes);
1096  isolate->random_number_generator()->NextBytes(input_bytes.data(),
1097  num_bytes);
1098  } else {
1099  for (int i = 0; i < args.length(); ++i) {
1100  if (IsJSTypedArray(args[i])) {
1101  Tagged<JSTypedArray> typed_array = Cast<JSTypedArray>(args[i]);
1102  add_input_bytes(typed_array->DataPtr(), typed_array->GetByteLength());
1103  } else if (IsJSArrayBuffer(args[i])) {
1104  Tagged<JSArrayBuffer> array_buffer = Cast<JSArrayBuffer>(args[i]);
1105  add_input_bytes(array_buffer->backing_store(),
1106  array_buffer->GetByteLength());
1107  } else if (IsSmi(args[i])) {
1108  int smi_value = Cast<Smi>(args[i]).value();
1109  add_input_bytes(&smi_value, kIntSize);
1110  } else if (IsHeapNumber(args[i])) {
1111  double value = Cast<HeapNumber>(args[i])->value();
1112  add_input_bytes(&value, kDoubleSize);
1113  } else {
1114  // TODO(14637): Extract bytes from more types.
1115  }
1116  }
1117  }
1118 
1119  // Avoid generating SIMD if the CPU does not support it, or it's disabled via
1120  // flags. Otherwise, do not limit the generated expressions and types.
1121  constexpr auto kAllOptions =
1122  wasm::fuzzing::WasmModuleGenerationOptions::All();
1123  constexpr auto kNoSimdOptions = wasm::fuzzing::WasmModuleGenerationOptions{
1124  {wasm::fuzzing::WasmModuleGenerationOption::kGenerateWasmGC}};
1125  static_assert(
1126  (kNoSimdOptions |
1127  wasm::fuzzing::WasmModuleGenerationOptions{
1128  {wasm::fuzzing::WasmModuleGenerationOption::kGenerateSIMD}}) ==
1129  kAllOptions);
1130  auto options =
1131  wasm::CheckHardwareSupportsSimd() ? kAllOptions : kNoSimdOptions;
1132 
1133  base::Vector<const uint8_t> module_bytes =
1134  wasm::fuzzing::GenerateRandomWasmModule(&temporary_zone, options,
1135  base::VectorOf(input_bytes));
1136 
1137  // Fuzzers can set `--wasm-max-module-size` to small values and then call
1138  // %WasmGenerateRandomModule() (see https://crbug.com/382816108).
1139  if (module_bytes.size() > v8_flags.wasm_max_module_size) {
1140  return CrashUnlessFuzzing(isolate);
1141  }
1142 
1143  wasm::ErrorThrower thrower{isolate, "WasmGenerateRandomModule"};
1144  MaybeDirectHandle<WasmModuleObject> maybe_module_object =
1145  wasm::GetWasmEngine()->SyncCompile(isolate,
1146  wasm::WasmEnabledFeatures::FromFlags(),
1147  wasm::CompileTimeImports{}, &thrower,
1148  base::OwnedCopyOf(module_bytes));
1149  if (thrower.error()) {
1150  FATAL(
1151  "wasm::GenerateRandomWasmModule produced a module which did not "
1152  "compile: %s",
1153  thrower.error_msg());
1154  }
1155  return *maybe_module_object.ToHandleChecked();
1156 }
