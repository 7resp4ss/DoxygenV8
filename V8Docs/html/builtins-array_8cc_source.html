69 // This method may transition the elements kind of the JSArray once, to make
70 // sure that all elements provided as arguments in the specified range can be
71 // added without further elements kinds transitions.
72 void MatchArrayElementsKindToArguments(Isolate* isolate,
73  DirectHandle<JSArray> array,
74  BuiltinArguments* args,
75  int first_arg_index, int num_arguments) {
76  int args_length = args->length();
77  if (first_arg_index >= args_length) return;
78 
79  ElementsKind origin_kind = array->GetElementsKind();
80 
81  // We do not need to transition for PACKED/HOLEY_ELEMENTS.
82  if (IsObjectElementsKind(origin_kind)) return;
83 
84  ElementsKind target_kind = origin_kind;
85  {
86  DisallowGarbageCollection no_gc;
87  int last_arg_index = std::min(first_arg_index + num_arguments, args_length);
88  for (int i = first_arg_index; i < last_arg_index; i++) {
89  Tagged<Object> arg = (*args)[i];
90  if (IsHeapObject(arg)) {
91  if (IsHeapNumber(arg)) {
92  target_kind = PACKED_DOUBLE_ELEMENTS;
93  } else {
94  target_kind = PACKED_ELEMENTS;
95  break;
96  }
97  }
98  }
99  }
100  if (target_kind != origin_kind) {
101  // Use a short-lived HandleScope to avoid creating several copies of the
102  // elements handle which would cause issues when left-trimming later-on.
103  HandleScope scope(isolate);
104  JSObject::TransitionElementsKind(isolate, array, target_kind);
105  }
106 }

125 // Checks if the receiver is a JSArray with fast elements which can add new
126 // elements. Returns |false| if not applicable.
127 V8_WARN_UNUSED_RESULT
128 inline bool IsJSArrayWithAddableFastElements(Isolate* isolate,
129  DirectHandle<Object> receiver,
130  ElementsKind* kind,
131  DirectHandle<JSArray>* array) {
132  if (!IsJSArrayWithExtensibleFastElements(isolate, receiver, kind, array))
133  return false;
134 
135  // If there may be elements accessors in the prototype chain, the fast path
136  // cannot be used if there arguments to add to the array.
137  if (!IsJSArrayFastElementMovingAllowed(isolate, **array)) return false;
138

140  // it has no elements. Handle that elsewhere.
141  if (isolate->IsInitialArrayPrototype(**array)) return false;
142 
143  return true;

146 // If |index| is Undefined, returns init_if_undefined.
147 // If |index| is negative, returns length + index.
148 // If |index| is positive, returns index.
149 // Returned value is guaranteed to be in the interval of [0, length].
150 V8_WARN_UNUSED_RESULT Maybe<uint64_t> GetRelativeIndex(
151  Isolate* isolate, uint64_t length, DirectHandle<Object> index,
152  uint64_t init_if_undefined) {
153  int64_t relative_index = init_if_undefined;
154  if (!IsUndefined(*index)) {
155  double provided_relative_index;
156  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate, provided_relative_index,
157  Object::IntegerValue(isolate, index),
158  Nothing<uint64_t>());
159  if (std::abs(provided_relative_index) > length) {
160  return Just(provided_relative_index < 0 ? 0 : length);
161  }
162  relative_index = static_cast<int64_t>(provided_relative_index);
163  }
164 
165  if (relative_index < 0) {
166  return Just(static_cast<uint64_t>(
167  std::max<int64_t>(static_cast<int64_t>(length) + relative_index, 0)));
168  }
169 
170  return Just(std::min(static_cast<uint64_t>(relative_index), length));
171 }

235 // Get a map which replaces the elements kind of an array. Unlike
236 // `JSObject::GetElementsTransitionMap`, this is allowed to go backwards in the
237 // ElementsKinds manifold, and may return a map which doesn't match the current
238 // elements array. The caller must handle the elements array, potentially
239 // reallocating if the FixedArray type doesn't match, and fill a valid value.
240 V8_WARN_UNUSED_RESULT MaybeDirectHandle<Map> GetReplacedElementsKindsMap(
241  Isolate* isolate, DirectHandle<JSArray> array, ElementsKind origin_kind,
242  ElementsKind target_kind) {
243  Tagged<Map> map = array->map();
244 
245  // Fast check for JSArrays without properties.
246  {
247  DisallowGarbageCollection no_gc;
248  Tagged<Context> native_context = map->map()->native_context();
249  if (native_context->GetInitialJSArrayMap(origin_kind) == map) {
250  Tagged<Object> maybe_target_map =
251  native_context->GetNoCell(Context::ArrayMapIndex(target_kind));
252  if (Tagged<Map> target_map; TryCast<Map>(maybe_target_map, &target_map)) {
253  map->NotifyLeafMapLayoutChange(isolate);
254  return direct_handle(target_map, isolate);
255  }
256  }
257  }
258 
259  // Ideally here we would reconfigure the map to an earlier ElementsKind with:
260  // ```
261  // MapUpdater{isolate, direct_handle(map, isolate)}
262  // .ReconfigureElementsKind(target_kind);
263  // ```
264  // However, this currently treats this as an invalid ElementsKind transition,
265  // and normalizes the map.
266  // TODO(leszeks): Handle this case.
267  return {};
268 }

270 V8_WARN_UNUSED_RESULT bool TryFastArrayFill(
271  Isolate* isolate, BuiltinArguments* args, DirectHandle<JSReceiver> receiver,
272  DirectHandle<Object> value, uint64_t start_index, uint64_t end_index) {
273  // If indices are too large, use generic path since they are stored as
274  // properties, not in the element backing store.
275  if (end_index > kMaxUInt32) return false;
276  if (!IsJSObject(*receiver)) return false;
277 
278  DirectHandle<JSArray> array;
279  ElementsKind kind;
280  if (!IsJSArrayWithAddableFastElements(isolate, receiver, &kind, &array)) {
281  return false;
282  }
283 
284  DCHECK_LE(start_index, kMaxUInt32);
285  DCHECK_LE(end_index, kMaxUInt32);
286  DCHECK_LT(start_index, end_index);
287 
288  uint32_t start = static_cast<uint32_t>(start_index);
289  uint32_t end = static_cast<uint32_t>(end_index);
290 
291  // The end index should be truncated to the array length in the argument
292  // resolution part of Array.p.fill, which means it should be within the
293  // FixedArray max length since we know the array has fast elements (and that
294  // both FixedArray and FixedDoubleArray have the same max length).
295  //
296  // However, it _can_ be larger than the _current_ array length, because
297  // of side-effects in Array.p.fill argument resolution (`ToInteger` on `start`
298  // and `end`, see: https://tc39.es/ecma262/#sec-array.prototype.fill). In
299  // particular, those could have transitioned a dictionary-elements array back
300  // to fast elements, by adding enough data element properties to make it dense
301  // (an easy way to do this is another call to Array.p.fill).
302  static_assert(FixedArray::kMaxLength == FixedDoubleArray::kMaxLength);
303  if (end > FixedArray::kMaxLength) return false;
304 
305  // Need to ensure that the fill value can be contained in the array.
306  ElementsKind origin_kind = array->GetElementsKind();
307  ElementsKind target_kind = Object::OptimalElementsKind(*value, isolate);
308 
309  if (target_kind != origin_kind) {
310  // Use a short-lived HandleScope to avoid creating several copies of the
311  // elements handle which would cause issues when left-trimming later-on.
312  HandleScope scope(isolate);
313 
314  bool is_replacing_all_elements =
315  (start == 0 && end == Object::NumberValue(array->length()));
316  bool did_transition_map = false;
317  if (is_replacing_all_elements) {
318  // For the case where we are replacing all elements, we can migrate the
319  // map backwards in the elements kind chain and ignore the current
320  // contents of the elements array.
321  DirectHandle<Map> new_map;
322 
323  if (GetReplacedElementsKindsMap(isolate, array, origin_kind, target_kind)
324  .ToHandle(&new_map)) {
325  DirectHandle<FixedArrayBase> elements(array->elements(), isolate);
326  if (IsDoubleElementsKind(origin_kind) !=
327  IsDoubleElementsKind(target_kind) ||
328  elements->IsCowArray()) {
329  // Reallocate the elements if doubleness doesn't match or the array is
330  // copy-on-write.
331  if (IsDoubleElementsKind(target_kind)) {
332  elements = isolate->factory()->NewFixedDoubleArray(end);
333  } else {
334  elements = isolate->factory()->NewFixedArrayWithZeroes(end);
335  }
336  }
337  JSObject::SetMapAndElements(isolate, array, new_map, elements);
338  if (IsMoreGeneralElementsKindTransition(origin_kind, target_kind)) {
339  // Transition through the allocation site as well if present, but
340  // only if this is a forward transition.
341  JSObject::UpdateAllocationSite(isolate, array, target_kind);
342  }
343  did_transition_map = true;
344  }
345  }
346 
347  if (!did_transition_map) {
348  target_kind = GetMoreGeneralElementsKind(origin_kind, target_kind);
349  JSObject::TransitionElementsKind(isolate, array, target_kind);
350  }
351  }
352 
353  ElementsAccessor* accessor = array->GetElementsAccessor();
354  accessor->Fill(isolate, array, value, start, end).Check();
355 
356  // It's possible the JSArray's 'length' property was assigned to after the
357  // length was loaded due to user code during argument coercion of the start
358  // and end parameters. The spec algorithm does a Set, meaning the length would
359  // grow as needed during the fill.
360  //
361  // ElementAccessor::Fill is able to grow the backing store as needed, but we
362  // need to ensure the JSArray's length is correctly set in case the user
363  // assigned a smaller value.
364  if (Object::NumberValue(array->length()) < end) {
365  CHECK(accessor->SetLength(isolate, array, end).FromJust());
366  }
367 
368  return true;
369 }

395  // else let relativeEnd be ? ToInteger(end).
396  // 6. If relativeEnd < 0, let final be max((len + relativeEnd), 0);
397  // else let final be min(relativeEnd, len).
398  DirectHandle<Object> end = args.atOrUndefined(isolate, 3);
399 
400  uint64_t end_index;
401  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
402  isolate, end_index, GetRelativeIndex(isolate, length, end, length));
403

406  // Ensure indexes are within array bounds
407  DCHECK_LE(0, start_index);
408  DCHECK_LE(start_index, end_index);
409  DCHECK_LE(end_index, length);
410 
411  DirectHandle<Object> value = args.atOrUndefined(isolate, 1);
412 
413  if (TryFastArrayFill(isolate, &args, receiver, value, start_index,
414  end_index)) {
415  return *receiver;
416  }
417  return GenericArrayFill(isolate, receiver, value, start_index, end_index);

421 V8_WARN_UNUSED_RESULT Tagged<Object> GenericArrayPush(Isolate* isolate,
422  BuiltinArguments* args) {
423  // 1. Let O be ? ToObject(this value).
424  DirectHandle<JSReceiver> receiver;
425  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
426  isolate, receiver, Object::ToObject(isolate, args->receiver()));
427 
428  // 2. Let len be ? ToLength(? Get(O, "length")).
429  DirectHandle<Object> raw_length_number;
430  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
431  isolate, raw_length_number,
432  Object::GetLengthFromArrayLike(isolate, receiver));
433 
434  // 3. Let args be a List whose elements are, in left to right order,
435  // the arguments that were passed to this function invocation.
436  // 4. Let arg_count be the number of elements in args.
437  int arg_count = args->length() - 1;
438 
439  // 5. If len + arg_count > 2^53-1, throw a TypeError exception.
440  double length = Object::NumberValue(*raw_length_number);
441  if (arg_count > kMaxSafeInteger - length) {
442  THROW_NEW_ERROR_RETURN_FAILURE(
443  isolate, NewTypeError(MessageTemplate::kPushPastSafeLength,
444  isolate->factory()->NewNumberFromInt(arg_count),
445  raw_length_number));
446  }
447 
448  // 6. Repeat, while args is not empty.
449  for (int i = 0; i < arg_count; ++i) {
450  // a. Remove the first element from args and let E be the value of the
451  // element.
452  DirectHandle<Object> element = args->at(i + 1);
453 
454  // b. Perform ? Set(O, ! ToString(len), E, true).
455  if (length <= JSObject::kMaxElementIndex) {
456  RETURN_FAILURE_ON_EXCEPTION(
457  isolate, Object::SetElement(isolate, receiver, length, element,
458  ShouldThrow::kThrowOnError));
459  } else {
460  PropertyKey key(isolate, length);
461  LookupIterator it(isolate, receiver, key);
462  MAYBE_RETURN(Object::SetProperty(&it, element, StoreOrigin::kMaybeKeyed,
463  Just(ShouldThrow::kThrowOnError)),
464  ReadOnlyRoots(isolate).exception());
465  }
466 
467  // c. Let len be len+1.
468  ++length;
469  }
470 
471  // 7. Perform ? Set(O, "length", len, true).
472  DirectHandle<Object> final_length = isolate->factory()->NewNumber(length);
473  RETURN_FAILURE_ON_EXCEPTION(
474  isolate, Object::SetProperty(isolate, receiver,
475  isolate->factory()->length_string(),
476  final_length, StoreOrigin::kMaybeKeyed,
477  Just(ShouldThrow::kThrowOnError)));
478 
479  // 8. Return len.
480  return *final_length;
481 }

609  isolate->factory()->length_string(),
610  Object::TypeOf(isolate, array), array));
611  }

613  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
614  isolate, set_len_ok, JSArray::SetLength(isolate, array, new_length));
615 
616  return *result;

637  Isolate* isolate, DirectHandle<JSReceiver> receiver, double length) {
638  // 4. Let first be ? Get(O, "0").
639  DirectHandle<Object> first;

668  isolate, receiver, to, from_val,

675  ReadOnlyRoots(isolate).exception());
676  }
677 
678  // f. Increase k by 1.
679  });

684  isolate, receiver, new_length_key, LanguageMode::kStrict),
685  ReadOnlyRoots(isolate).exception());
686 
687  // 8. Perform ? Set(O, "length", len-1, true).
688  RETURN_FAILURE_ON_EXCEPTION(isolate,
689  SetLengthProperty(isolate, receiver, length - 1));
690 
691  // 9. Return first.
692  return *first;

760 /**
761  * A simple visitor visits every element of Array's.
762  * The backend storage can be a fixed array for fast elements case,
763  * or a dictionary for sparse array. Since Dictionary is a subtype
764  * of FixedArray, the class can be used by both fast and slow cases.
765  * The second parameter of the constructor, fast_elements, specifies
766  * whether the storage is a FixedArray or Dictionary.
767  *
768  * An index limit is used to deal with the situation that a result array
769  * length overflows 32-bit non-negative integer.
770  */
771 class ArrayConcatVisitor {
772  public:
773  ArrayConcatVisitor(
774  Isolate* isolate,
775  DirectHandle<UnionOf<JSReceiver, FixedArray, NumberDictionary>> storage,
776  bool fast_elements)
777  : isolate_(isolate),
778  storage_(isolate->global_handles()->Create(*storage)),
779  index_offset_(0u),
780  bit_field_(FastElementsField::encode(fast_elements) |
781  ExceedsLimitField::encode(false) |
782  IsFixedArrayField::encode(IsFixedArray(*storage, isolate)) |
783  HasSimpleElementsField::encode(
784  IsFixedArray(*storage, isolate) ||
785  // Don't take fast path for storages that might have
786  // side effects when storing to them.
787  (!IsCustomElementsReceiverMap(storage->map(isolate)) &&
788  !IsJSTypedArray(*storage, isolate)))) {
789  DCHECK_IMPLIES(this->fast_elements(), is_fixed_array());
790  }
791 
792  ~ArrayConcatVisitor() { clear_storage(); }
793 
794  V8_WARN_UNUSED_RESULT bool visit(uint32_t i, DirectHandle<Object> elm) {
795  uint32_t index = index_offset_ + i;
796 
797  // Note we use >=kMaxArrayLength instead of the more appropriate
798  // >kMaxArrayIndex here due to overflowing arithmetic and
799  // increase_index_offset.
800  if (i >= JSArray::kMaxArrayLength - index_offset_) {
801  set_exceeds_array_limit(true);
802  // Exception hasn't been thrown at this point. Return true to
803  // break out, and caller will throw. !visit would imply that
804  // there is already an exception.
805  return true;
806  }
807 
808  if (!is_fixed_array()) {
809  MAYBE_RETURN(JSReceiver::CreateDataProperty(
810  isolate_, direct_handle(Cast<JSReceiver>(storage_)),
811  PropertyKey(isolate_, index), elm, Just(kThrowOnError)),
812  false);
813  return true;
814  }
815 
816  if (fast_elements()) {
817  if (index < static_cast<uint32_t>(storage_fixed_array()->length())) {
818  storage_fixed_array()->set(index, *elm);
819  return true;
820  }
821  // Our initial estimate of length was foiled, possibly by
822  // getters on the arrays increasing the length of later arrays
823  // during iteration.
824  // This shouldn't happen in anything but pathological cases.
825  SetDictionaryMode();
826  // Fall-through to dictionary mode.
827  }
828  DCHECK(!fast_elements());
829  DirectHandle<NumberDictionary> dict(Cast<NumberDictionary>(*storage_),
830  isolate_);
831  // The object holding this backing store has just been allocated, so
832  // it cannot yet be used as a prototype.
833  DirectHandle<JSObject> not_a_prototype_holder;
834  DirectHandle<NumberDictionary> result = NumberDictionary::Set(
835  isolate_, dict, index, elm, not_a_prototype_holder);
836  if (!result.is_identical_to(dict)) {
837  // Dictionary needed to grow.
838  clear_storage();
839  set_storage(*result);
840  }
841  return true;
842  }
843 
844  uint32_t index_offset() const { return index_offset_; }
845 
846  void increase_index_offset(uint32_t delta) {
847  if (JSArray::kMaxArrayLength - index_offset_ < delta) {
848  index_offset_ = JSArray::kMaxArrayLength;
849  } else {
850  index_offset_ += delta;
851  }
852  // If the initial length estimate was off (see special case in visit()),
853  // but the array blowing the limit didn't contain elements beyond the
854  // provided-for index range, go to dictionary mode now.
855  if (fast_elements() &&
856  index_offset_ >
857  static_cast<uint32_t>(Cast<FixedArrayBase>(*storage_)->length())) {
858  SetDictionaryMode();
859  }
860  }
861 
862  bool exceeds_array_limit() const {
863  return ExceedsLimitField::decode(bit_field_);
864  }
865 
866  DirectHandle<JSArray> ToArray() {
867  DCHECK(is_fixed_array());
868  DirectHandle<JSArray> array = isolate_->factory()->NewJSArray(0);
869  DirectHandle<Number> length =
870  isolate_->factory()->NewNumber(static_cast<double>(index_offset_));
871  DirectHandle<Map> map = JSObject::GetElementsTransitionMap(
872  isolate_, array,
873  fast_elements() ? HOLEY_ELEMENTS : DICTIONARY_ELEMENTS);
874  {
875  DisallowGarbageCollection no_gc;
876  Tagged<JSArray> raw = *array;
877  raw->set_length(*length);
878  raw->set_elements(*storage_fixed_array());
879  raw->set_map(isolate_, *map, kReleaseStore);
880  }
881  return array;
882  }
883 
884  V8_WARN_UNUSED_RESULT MaybeDirectHandle<JSReceiver> ToJSReceiver() {
885  DCHECK(!is_fixed_array());
886  DirectHandle<JSReceiver> result = Cast<JSReceiver>(storage_);
887  DirectHandle<Object> length =
888  isolate_->factory()->NewNumber(static_cast<double>(index_offset_));
889  RETURN_ON_EXCEPTION(
890  isolate_, Object::SetProperty(isolate_, result,
891  isolate_->factory()->length_string(),
892  length, StoreOrigin::kMaybeKeyed,
893  Just(ShouldThrow::kThrowOnError)));
894  return result;
895  }
896  bool has_simple_elements() const {
897  return HasSimpleElementsField::decode(bit_field_);
898  }
899 
900  private:
901  // Convert storage to dictionary mode.
902  void SetDictionaryMode() {
903  DCHECK(fast_elements() && is_fixed_array());
904  DirectHandle<FixedArray> current_storage = storage_fixed_array();
905  DirectHandle<NumberDictionary> slow_storage(
906  NumberDictionary::New(isolate_, current_storage->length()));
907  uint32_t current_length = static_cast<uint32_t>(current_storage->length());
908  FOR_WITH_HANDLE_SCOPE(
909  isolate_, uint32_t, i = 0, i, i < current_length, i++, {
910  DirectHandle<Object> element(current_storage->get(i), isolate_);
911  if (!IsTheHole(*element, isolate_)) {
912  // The object holding this backing store has just been allocated, so
913  // it cannot yet be used as a prototype.
914  DirectHandle<JSObject> not_a_prototype_holder;
915  DirectHandle<NumberDictionary> new_storage = NumberDictionary::Set(
916  isolate_, slow_storage, i, element, not_a_prototype_holder);
917  if (!new_storage.is_identical_to(slow_storage)) {
918  slow_storage = loop_scope.CloseAndEscape(new_storage);
919  }
920  }
921  });
922  clear_storage();
923  set_storage(*slow_storage);
924  set_fast_elements(false);
925  }
926 
927  inline void clear_storage() { GlobalHandles::Destroy(storage_.location()); }
928 
929  inline void set_storage(Tagged<FixedArray> storage) {
930  DCHECK(is_fixed_array());
931  DCHECK(has_simple_elements());
932  storage_ = isolate_->global_handles()->Create(storage);
933  }

935  using FastElementsField = base::BitField<bool, 0, 1>;
936  using ExceedsLimitField = base::BitField<bool, 1, 1>;
937  using IsFixedArrayField = base::BitField<bool, 2, 1>;
938  using HasSimpleElementsField = base::BitField<bool, 3, 1>;
939 
940  bool fast_elements() const { return FastElementsField::decode(bit_field_); }
941  void set_fast_elements(bool fast) {
942  bit_field_ = FastElementsField::update(bit_field_, fast);
943  }
944  void set_exceeds_array_limit(bool exceeds) {
945  bit_field_ = ExceedsLimitField::update(bit_field_, exceeds);
946  }
947  bool is_fixed_array() const { return IsFixedArrayField::decode(bit_field_); }
948  DirectHandle<FixedArray> storage_fixed_array() {
949  DCHECK(is_fixed_array());
950  DCHECK(has_simple_elements());
951  return Cast<FixedArray>(storage_);
952  }
953 
954  Isolate* isolate_;
955  IndirectHandle<UnionOf<JSReceiver, FixedArray, NumberDictionary>>
956  storage_; // Always a global handle.
957  // Index after last seen index. Always less than or equal to
958  // JSArray::kMaxArrayLength.
959  uint32_t index_offset_;
960  uint32_t bit_field_;
961 };

963 uint32_t EstimateElementCount(Isolate* isolate, DirectHandle<JSArray> array) {
964  DisallowGarbageCollection no_gc;
965  uint32_t length = static_cast<uint32_t>(Object::NumberValue(array->length()));
966  int element_count = 0;
967  switch (array->GetElementsKind()) {
968  case PACKED_SMI_ELEMENTS:
969  case HOLEY_SMI_ELEMENTS:
970  case PACKED_ELEMENTS:
971  case PACKED_FROZEN_ELEMENTS:
972  case PACKED_SEALED_ELEMENTS:
973  case PACKED_NONEXTENSIBLE_ELEMENTS:
974  case HOLEY_FROZEN_ELEMENTS:
975  case HOLEY_SEALED_ELEMENTS:
976  case HOLEY_NONEXTENSIBLE_ELEMENTS:
977  case HOLEY_ELEMENTS: {
978  // Fast elements can't have lengths that are not representable by
979  // a 32-bit signed integer.
980  static_assert(static_cast<int32_t>(FixedArray::kMaxLength) >= 0);
981  int fast_length = static_cast<int>(length);
982  Tagged<FixedArray> elements = Cast<FixedArray>(array->elements());
983  for (int i = 0; i < fast_length; i++) {
984  if (!IsTheHole(elements->get(i), isolate)) element_count++;
985  }
986  break;
987  }
988  case PACKED_DOUBLE_ELEMENTS:
989  case HOLEY_DOUBLE_ELEMENTS: {
990  // Fast elements can't have lengths that are not representable by
991  // a 32-bit signed integer.
992  static_assert(static_cast<int32_t>(FixedDoubleArray::kMaxLength) >= 0);
993  int fast_length = static_cast<int>(length);
994  if (IsFixedArray(array->elements())) {
995  DCHECK_EQ(Cast<FixedArray>(array->elements())->length(), 0);
996  break;
997  }
998  Tagged<FixedDoubleArray> elements =
999  Cast<FixedDoubleArray>(array->elements());
1000  for (int i = 0; i < fast_length; i++) {
1001  if (!elements->is_the_hole(i)) element_count++;
1002  }
1003  break;
1004  }
1005  case DICTIONARY_ELEMENTS: {
1006  Tagged<NumberDictionary> dictionary =
1007  Cast<NumberDictionary>(array->elements());
1008  ReadOnlyRoots roots(isolate);
1009  for (InternalIndex i : dictionary->IterateEntries()) {
1010  Tagged<Object> key = dictionary->KeyAt(i);
1011  if (dictionary->IsKey(roots, key)) {
1012  element_count++;
1013  }
1014  }
1015  break;
1016  }
1017 #define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) case TYPE##_ELEMENTS:
1018 
1019  TYPED_ARRAYS(TYPED_ARRAY_CASE)
1020  RAB_GSAB_TYPED_ARRAYS(TYPED_ARRAY_CASE)
1021  // External arrays are always dense.
1022  return length;
1023 
1024 #undef TYPED_ARRAY_CASE
1025  case NO_ELEMENTS:
1026  return 0;
1027  case FAST_SLOPPY_ARGUMENTS_ELEMENTS:
1028  case SLOW_SLOPPY_ARGUMENTS_ELEMENTS:
1029  case FAST_STRING_WRAPPER_ELEMENTS:
1030  case SLOW_STRING_WRAPPER_ELEMENTS:
1031  case WASM_ARRAY_ELEMENTS:
1032  case SHARED_ARRAY_ELEMENTS:
1033  UNREACHABLE();
1034  }
1035  // As an estimate, we assume that the prototype doesn't contain any
1036  // inherited elements.
1037  return element_count;
1038 }

1040 void CollectElementIndices(Isolate* isolate, DirectHandle<JSObject> object,
1041  uint32_t range, std::vector<uint32_t>* indices) {
1042  ElementsKind kind = object->GetElementsKind();
1043  switch (kind) {
1044  case PACKED_SMI_ELEMENTS:
1045  case PACKED_ELEMENTS:
1046  case PACKED_FROZEN_ELEMENTS:
1047  case PACKED_SEALED_ELEMENTS:
1048  case PACKED_NONEXTENSIBLE_ELEMENTS:
1049  case HOLEY_SMI_ELEMENTS:
1050  case HOLEY_FROZEN_ELEMENTS:
1051  case HOLEY_SEALED_ELEMENTS:
1052  case HOLEY_NONEXTENSIBLE_ELEMENTS:
1053  case HOLEY_ELEMENTS: {
1054  DisallowGarbageCollection no_gc;
1055  Tagged<FixedArray> elements = Cast<FixedArray>(object->elements());
1056  uint32_t length = static_cast<uint32_t>(elements->length());
1057  if (range < length) length = range;
1058  for (uint32_t i = 0; i < length; i++) {
1059  if (!IsTheHole(elements->get(i), isolate)) {
1060  indices->push_back(i);
1061  }
1062  }
1063  break;
1064  }
1065  case HOLEY_DOUBLE_ELEMENTS:
1066  case PACKED_DOUBLE_ELEMENTS: {
1067  if (IsFixedArray(object->elements())) {
1068  DCHECK_EQ(object->elements()->length(), 0);
1069  break;
1070  }
1071  DirectHandle<FixedDoubleArray> elements(
1072  Cast<FixedDoubleArray>(object->elements()), isolate);
1073  uint32_t length = static_cast<uint32_t>(elements->length());
1074  if (range < length) length = range;
1075  for (uint32_t i = 0; i < length; i++) {
1076  if (!elements->is_the_hole(i)) {
1077  indices->push_back(i);
1078  }
1079  }
1080  break;
1081  }
1082  case DICTIONARY_ELEMENTS: {
1083  DisallowGarbageCollection no_gc;
1084  Tagged<NumberDictionary> dict =
1085  Cast<NumberDictionary>(object->elements());
1086  uint32_t capacity = dict->Capacity();
1087  ReadOnlyRoots roots(isolate);
1088  FOR_WITH_HANDLE_SCOPE(isolate, uint32_t, j = 0, j, j < capacity, j++, {
1089  Tagged<Object> k = dict->KeyAt(InternalIndex(j));
1090  if (!dict->IsKey(roots, k)) continue;
1091  DCHECK(IsNumber(k));
1092  uint32_t index = static_cast<uint32_t>(Object::NumberValue(k));
1093  if (index < range) {
1094  indices->push_back(index);
1095  }
1096  });
1097  break;
1098  }
1099 #define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) case TYPE##_ELEMENTS:
1100 
1101  TYPED_ARRAYS(TYPED_ARRAY_CASE) RAB_GSAB_TYPED_ARRAYS(TYPED_ARRAY_CASE) {
1102  size_t length = Cast<JSTypedArray>(object)->GetLength();
1103  if (range <= length) {
1104  length = range;
1105  // We will add all indices, so we might as well clear it first
1106  // and avoid duplicates.
1107  indices->clear();
1108  }
1109  // {range} puts a cap on {length}.
1110  DCHECK_LE(length, std::numeric_limits<uint32_t>::max());
1111  for (uint32_t i = 0; i < length; i++) {
1112  indices->push_back(i);
1113  }
1114  if (length == range) return; // All indices accounted for already.
1115  break;
1116  }
1117 
1118 #undef TYPED_ARRAY_CASE
1119  case FAST_SLOPPY_ARGUMENTS_ELEMENTS:
1120  case SLOW_SLOPPY_ARGUMENTS_ELEMENTS: {
1121  DisallowGarbageCollection no_gc;
1122  DisableGCMole no_gc_mole;
1123  Tagged<FixedArrayBase> elements = object->elements();
1124  Tagged<JSObject> raw_object = *object;
1125  ElementsAccessor* accessor = object->GetElementsAccessor();
1126  for (uint32_t i = 0; i < range; i++) {
1127  if (accessor->HasElement(isolate, raw_object, i, elements)) {
1128  indices->push_back(i);
1129  }
1130  }
1131  break;
1132  }
1133  case FAST_STRING_WRAPPER_ELEMENTS:
1134  case SLOW_STRING_WRAPPER_ELEMENTS: {
1135  DCHECK(IsJSPrimitiveWrapper(*object));
1136  auto js_value = Cast<JSPrimitiveWrapper>(object);
1137  DCHECK(IsString(js_value->value()));
1138  DirectHandle<String> string(Cast<String>(js_value->value()), isolate);
1139  uint32_t length = static_cast<uint32_t>(string->length());
1140  uint32_t i = 0;
1141  uint32_t limit = std::min(length, range);
1142  for (; i < limit; i++) {
1143  indices->push_back(i);
1144  }
1145  ElementsAccessor* accessor = object->GetElementsAccessor();
1146  for (; i < range; i++) {
1147  if (accessor->HasElement(isolate, *object, i)) {
1148  indices->push_back(i);
1149  }
1150  }
1151  break;
1152  }
1153  case WASM_ARRAY_ELEMENTS:
1154  // TODO(ishell): implement
1155  UNIMPLEMENTED();
1156  case SHARED_ARRAY_ELEMENTS: {
1157  uint32_t length = Cast<JSSharedArray>(object)->elements()->length();
1158  if (range <= length) {
1159  length = range;
1160  indices->clear();
1161  }
1162  for (uint32_t i = 0; i < length; i++) {
1163  // JSSharedArrays are created non-resizable and do not have holes.
1164  SLOW_DCHECK(object->GetElementsAccessor()->HasElement(
1165  isolate, *object, i, object->elements()));
1166  indices->push_back(i);
1167  }
1168  if (length == range) return;
1169  break;
1170  }
1171  case NO_ELEMENTS:
1172  break;
1173  }
1174 
1175  PrototypeIterator iter(isolate, object);
1176  if (!iter.IsAtEnd()) {
1177  // The prototype will usually have no inherited element indices,
1178  // but we have to check.
1179  // Casting to JSObject is safe because we ran {HasOnlySimpleElements} on
1180  // the receiver before, which checks the prototype chain.
1181  CollectElementIndices(
1182  isolate, PrototypeIterator::GetCurrent<JSObject>(iter), range, indices);
1183  }
1184 }
