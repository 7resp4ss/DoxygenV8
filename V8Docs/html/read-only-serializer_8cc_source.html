20 // Preprocess an object to prepare it for serialization.
21 class ObjectPreProcessor final {
22  public:
23  explicit ObjectPreProcessor(Isolate* isolate)
24  : isolate_(isolate), extref_encoder_(isolate) {}

26 #define PRE_PROCESS_TYPE_LIST(V) \
27  V(AccessorInfo) \
28  V(InterceptorInfo) \
29  V(JSExternalObject) \
30  V(FunctionTemplateInfo) \

33  void PreProcessIfNeeded(Tagged<HeapObject> o) {
34  const InstanceType itype = o->map(isolate_)->instance_type();
35 #define V(TYPE) \
36  if (InstanceTypeChecker::Is##TYPE(itype)) { \
37  return PreProcess##TYPE(Cast<TYPE>(o)); \
38  }
39  PRE_PROCESS_TYPE_LIST(V)
40 #undef V
41  // If we reach here, no preprocessing is needed for this object.
42  }
43 #undef PRE_PROCESS_TYPE_LIST

45  private:
46  void EncodeExternalPointerSlot(ExternalPointerSlot slot) {
47  Address value = slot.load(isolate_);
48  EncodeExternalPointerSlot(slot, value);
49  }
50 
51  void EncodeExternalPointerSlot(ExternalPointerSlot slot, Address value) {
52  // Note it's possible that `value != slot.load(...)`, e.g. for
53  // AccessorInfo::remove_getter_indirection.
54  ExternalReferenceEncoder::Value encoder_value =
55  extref_encoder_.Encode(value);
56  DCHECK_LT(encoder_value.index(),
57  1UL << ro::EncodedExternalReference::kIndexBits);
58  ro::EncodedExternalReference encoded{encoder_value.is_from_api(),
59  encoder_value.index()};
60  // Constructing no_gc here is not the intended use pattern (instead we
61  // should pass it along the entire callchain); but there's little point of
62  // doing that here - all of the code in this file relies on GC being
63  // disabled, and that's guarded at entry points.
64  DisallowGarbageCollection no_gc;
65  slot.ReplaceContentWithIndexForSerialization(no_gc, encoded.ToUint32());
66  }
67  void PreProcessAccessorInfo(Tagged<AccessorInfo> o) {
68  EncodeExternalPointerSlot(
69  o->RawExternalPointerField(AccessorInfo::kMaybeRedirectedGetterOffset,
70  kAccessorInfoGetterTag),
71  o->getter(isolate_)); // Pass the non-redirected value.
72  EncodeExternalPointerSlot(o->RawExternalPointerField(
73  AccessorInfo::kSetterOffset, kAccessorInfoSetterTag));
74  }
75  void PreProcessInterceptorInfo(Tagged<InterceptorInfo> o) {
76  const bool is_named = o->is_named();
77 
78 #define PROCESS_FIELD(Name, name) \
79  EncodeExternalPointerSlot(o->RawExternalPointerField( \
80  InterceptorInfo::k##Name##Offset, \
81  is_named ? kApiNamedProperty##Name##CallbackTag \
82  : kApiIndexedProperty##Name##CallbackTag));
83 
84  INTERCEPTOR_INFO_CALLBACK_LIST(PROCESS_FIELD)
85 #undef PROCESS_FIELD
86  }
87  void PreProcessJSExternalObject(Tagged<JSExternalObject> o) {
88  EncodeExternalPointerSlot(
89  o->RawExternalPointerField(JSExternalObject::kValueOffset,
90  kExternalObjectValueTag),
91  reinterpret_cast<Address>(o->value(isolate_)));
92  }
93  void PreProcessFunctionTemplateInfo(Tagged<FunctionTemplateInfo> o) {
94  EncodeExternalPointerSlot(
95  o->RawExternalPointerField(
96  FunctionTemplateInfo::kMaybeRedirectedCallbackOffset,
97  kFunctionTemplateInfoCallbackTag),
98  o->callback(isolate_)); // Pass the non-redirected value.
99  }
100 #if V8_ENABLE_GEARBOX
101  V8_INLINE void ResetGearboxPlaceholderBuiltin(Tagged<Code> code) {
102  // In order to ensure predictable state of placeholder builtins Code
103  // objects after serialization we replace their fields with the contents
104  // of kIllegal builtin.
105  if (code->is_gearbox_placeholder_builtin()) {
106  Builtin variant_builtin_id = code->builtin_id();
107  Builtin placeholder_builtin_id =
108  Builtins::GetGearboxPlaceholderFromVariant(variant_builtin_id);
109  DCHECK_EQ(

111  code->builtin_id());
112  Tagged<Code> src = isolate_->builtins()->code(Builtin::kIllegal);
113  Code::CopyFieldsWithGearboxForSerialization(code, src, isolate_);
114  // We should use the placeholder id instead of kIllegal.
115  code->set_builtin_id(placeholder_builtin_id);
116  }
117  }
118 #endif
119  void PreProcessCode(Tagged<Code> o) {
120  o->ClearInstructionStartForSerialization(isolate_);
121  CHECK(!o->has_source_position_table_or_bytecode_offset_table());
122  CHECK(!o->has_deoptimization_data_or_interpreter_data());
123 #ifdef V8_ENABLE_LEAPTIERING
124  CHECK_EQ(o->js_dispatch_handle(), kNullJSDispatchHandle);
125 #endif
126 #if V8_ENABLE_GEARBOX
127  ResetGearboxPlaceholderBuiltin(o);
128 #endif
129  }

135 struct ReadOnlySegmentForSerialization {
136  ReadOnlySegmentForSerialization(Isolate* isolate,
137  const ReadOnlyPageMetadata* page,
138  Address segment_start, size_t segment_size,
139  ObjectPreProcessor* pre_processor)
140  : page(page),
141  segment_start(segment_start),
142  segment_size(segment_size),
143  segment_offset(segment_start - page->area_start()),
144  contents(new uint8_t[segment_size]),
145  tagged_slots(segment_size / kTaggedSize) {
146  // .. because tagged_slots records a bit for each slot:
147  DCHECK(IsAligned(segment_size, kTaggedSize));
148  // Ensure incoming pointers to this page are representable.
149  CHECK_LT(isolate->read_only_heap()->read_only_space()->IndexOf(page),
150  1UL << ro::EncodedTagged::kPageIndexBits);
151 
152  MemCopy(contents.get(), reinterpret_cast<void*>(segment_start),
153  segment_size);
154  PreProcessSegment(pre_processor);
155  if (!V8_STATIC_ROOTS_BOOL) EncodeTaggedSlots(isolate);
156  }
157 
158  void PreProcessSegment(ObjectPreProcessor* pre_processor) {
159  // Iterate the RO page and the contents copy in lockstep, preprocessing
160  // objects as we go along.
161  //
162  // See also ObjectSerializer::OutputRawData.
163  DCHECK_GE(segment_start, page->area_start());
164  const Address segment_end = segment_start + segment_size;
165  ReadOnlyPageObjectIterator it(page, segment_start);
166  for (Tagged<HeapObject> o = it.Next(); !o.is_null(); o = it.Next()) {
167  if (o.address() >= segment_end) break;
168  size_t o_offset = o.ptr() - segment_start;
169  Address o_dst = reinterpret_cast<Address>(contents.get()) + o_offset;
170  pre_processor->PreProcessIfNeeded(
171  Cast<HeapObject>(Tagged<Object>(o_dst)));
172  }
173  }
174 
175  void EncodeTaggedSlots(Isolate* isolate);
176 
177  const ReadOnlyPageMetadata* const page;
178  const Address segment_start;
179  const size_t segment_size;
180  const size_t segment_offset;
181  // The (mutated) off-heap copy of the on-heap segment.
182  std::unique_ptr<uint8_t[]> contents;
183  // The relocation table.
184  ro::BitSet tagged_slots;
185 
186  friend class EncodeRelocationsVisitor;

189 ro::EncodedTagged Encode(Isolate* isolate, Tagged<HeapObject> o) {
190  Address o_address = o.address();
191  MemoryChunkMetadata* chunk = MemoryChunkMetadata::FromAddress(o_address);
192 
193  ReadOnlySpace* ro_space = isolate->read_only_heap()->read_only_space();
194  int index = static_cast<int>(ro_space->IndexOf(chunk));
195  uint32_t offset = static_cast<int>(chunk->Offset(o_address));
196  DCHECK(IsAligned(offset, kTaggedSize));
197 
198  return ro::EncodedTagged(index, offset / kTaggedSize);
199 }

274  private:
275  void ProcessSlot(MaybeObjectSlot slot) {
276  Tagged<MaybeObject> o = *slot;
277  if (!o.IsStrongOrWeak()) return; // Smis don't need relocation.
278  DCHECK(o.IsStrong());
279 
280  int slot_offset = SegmentOffsetOf(slot);
281  DCHECK(IsAligned(slot_offset, kTaggedSize));
282 
283  // Encode:
284  ro::EncodedTagged encoded = Encode(isolate_, o.GetHeapObject());
285  memcpy(segment_->contents.get() + slot_offset, &encoded,
286  ro::EncodedTagged::kSize);
287 
288  // Record:
289  segment_->tagged_slots.set(AsSlot(slot_offset));
290  }

292  template <class SlotT>
293  int SegmentOffsetOf(SlotT slot) const {
294  Address addr = slot.address();
295  DCHECK_GE(addr, segment_->segment_start);
296  DCHECK_LT(addr, segment_->segment_start + segment_->segment_size);
297  return static_cast<int>(addr - segment_->segment_start);
298  }

300  static constexpr int AsSlot(int byte_offset) {
301  return byte_offset / kTaggedSize;
302  }

308 void ReadOnlySegmentForSerialization::EncodeTaggedSlots(Isolate* isolate) {
309  DCHECK(!V8_STATIC_ROOTS_BOOL);
310  EncodeRelocationsVisitor v(isolate, this);
311  PtrComprCageBase cage_base(isolate);
312 
313  DCHECK_GE(segment_start, page->area_start());
314  const Address segment_end = segment_start + segment_size;
315  ReadOnlyPageObjectIterator it(page, segment_start,
316  SkipFreeSpaceOrFiller::kNo);
317  for (Tagged<HeapObject> o = it.Next(); !o.is_null(); o = it.Next()) {
318  if (o.address() >= segment_end) break;
319  VisitObject(isolate, o, &v);
320  }

323 class ReadOnlyHeapImageSerializer {
324  public:
325  struct MemoryRegion {
326  Address start;
327  size_t size;
328  };
329 
330  static void Serialize(Isolate* isolate, SnapshotByteSink* sink,
331  const std::vector<MemoryRegion>& unmapped_regions) {
332  ReadOnlyHeapImageSerializer{isolate, sink}.SerializeImpl(unmapped_regions);
333  }
334 
335  private:
336  using Bytecode = ro::Bytecode;
337 
338  ReadOnlyHeapImageSerializer(Isolate* isolate, SnapshotByteSink* sink)
339  : isolate_(isolate), sink_(sink), pre_processor_(isolate) {}
340 
341  void SerializeImpl(const std::vector<MemoryRegion>& unmapped_regions) {
342  DCHECK_EQ(sink_->Position(), 0);
343 
344  ReadOnlySpace* ro_space = isolate_->read_only_heap()->read_only_space();
345 
346  // Allocate all pages first s.t. the deserializer can easily handle forward
347  // references (e.g.: an object on page i points at an object on page i+1).
348  for (const ReadOnlyPageMetadata* page : ro_space->pages()) {
349  EmitAllocatePage(page, unmapped_regions);
350  }
351 
352  // Now write the page contents.
353  for (const ReadOnlyPageMetadata* page : ro_space->pages()) {
354  SerializePage(page, unmapped_regions);
355  }
356 
357  EmitReadOnlyRootsTable();
358  sink_->Put(Bytecode::kFinalizeReadOnlySpace, "space end");
359  }
360 
361  uint32_t IndexOf(const ReadOnlyPageMetadata* page) {
362  ReadOnlySpace* ro_space = isolate_->read_only_heap()->read_only_space();
363  return static_cast<uint32_t>(ro_space->IndexOf(page));
364  }
365 
366  void EmitAllocatePage(const ReadOnlyPageMetadata* page,
367  const std::vector<MemoryRegion>& unmapped_regions) {
368  if (V8_STATIC_ROOTS_BOOL) {
369  sink_->Put(Bytecode::kAllocatePageAt, "fixed page begin");
370  } else {
371  sink_->Put(Bytecode::kAllocatePage, "page begin");
372  }
373  sink_->PutUint30(IndexOf(page), "page index");
374  sink_->PutUint30(
375  static_cast<uint32_t>(page->HighWaterMark() - page->area_start()),
376  "area size in bytes");
377  if (V8_STATIC_ROOTS_BOOL) {
378  auto page_addr = page->ChunkAddress();
379  sink_->PutUint32(V8HeapCompressionScheme::CompressAny(page_addr),
380  "page start offset");
381  }
382  }
383 
384  void SerializePage(const ReadOnlyPageMetadata* page,
385  const std::vector<MemoryRegion>& unmapped_regions) {
386  Address pos = page->area_start();
387 
388  // If this page contains unmapped regions split it into multiple segments.
389  for (auto r = unmapped_regions.begin(); r != unmapped_regions.end(); ++r) {
390  // Regions must be sorted and non-overlapping.
391  if (r + 1 != unmapped_regions.end()) {
392  CHECK(r->start < (r + 1)->start);
393  CHECK(r->start + r->size < (r + 1)->start);
394  }
395  if (base::IsInRange(r->start, pos, page->HighWaterMark())) {
396  size_t segment_size = r->start - pos;
397  ReadOnlySegmentForSerialization segment(isolate_, page, pos,
398  segment_size, &pre_processor_);
399  EmitSegment(&segment);
400  pos += segment_size + r->size;
401  }
402  }
403 
404  // Pages are shrunk, but memory at the end of the area is still
405  // uninitialized and we do not want to include it in the snapshot.
406  size_t segment_size = page->HighWaterMark() - pos;
407  ReadOnlySegmentForSerialization segment(isolate_, page, pos, segment_size,
408  &pre_processor_);
409  EmitSegment(&segment);
410  }
411 
412  void EmitSegment(const ReadOnlySegmentForSerialization* segment) {
413  sink_->Put(Bytecode::kSegment, "segment begin");
414  sink_->PutUint30(IndexOf(segment->page), "page index");
415  sink_->PutUint30(static_cast<uint32_t>(segment->segment_offset),
416  "segment start offset");
417  sink_->PutUint30(static_cast<uint32_t>(segment->segment_size),
418  "segment byte size");
419  sink_->PutRaw(segment->contents.get(),
420  static_cast<int>(segment->segment_size), "page");
421  if (!V8_STATIC_ROOTS_BOOL) {
422  sink_->Put(Bytecode::kRelocateSegment, "relocate segment");
423  sink_->PutRaw(segment->tagged_slots.data(),
424  static_cast<int>(segment->tagged_slots.size_in_bytes()),
425  "tagged_slots");
426  }
427  }
428 
429  void EmitReadOnlyRootsTable() {
430  sink_->Put(Bytecode::kReadOnlyRootsTable, "read only roots table");
431  if (!V8_STATIC_ROOTS_BOOL) {
432  ReadOnlyRoots roots(isolate_);
433  for (size_t i = 0; i < ReadOnlyRoots::kEntriesCount; i++) {
434  RootIndex rudi = static_cast<RootIndex>(i);
435  Tagged<HeapObject> rudolf = Cast<HeapObject>(roots.object_at(rudi));
436  ro::EncodedTagged encoded = Encode(isolate_, rudolf);
437  sink_->PutUint32(encoded.ToUint32(), "read only roots entry");
438  }
439  }
440  }

447 std::vector<ReadOnlyHeapImageSerializer::MemoryRegion> GetUnmappedRegions(
448  Isolate* isolate) {
449 #ifdef V8_STATIC_ROOTS
450  // WasmNull's payload is aligned to the OS page and consists of
451  // WasmNull::kPayloadSize bytes of unmapped memory. To avoid inflating the
452  // snapshot size and accessing uninitialized and/or unmapped memory, the
453  // serializer skips the padding bytes and the payload.
454  ReadOnlyRoots ro_roots(isolate);
455  Tagged<WasmNull> wasm_null = ro_roots.wasm_null();
456  Tagged<HeapObject> wasm_null_padding = ro_roots.wasm_null_padding();
457  CHECK(IsFreeSpace(wasm_null_padding));
458  Address wasm_null_padding_start =
459  wasm_null_padding.address() + sizeof(FreeSpace);
460  std::vector<ReadOnlyHeapImageSerializer::MemoryRegion> unmapped;
461  if (wasm_null.address() > wasm_null_padding_start) {
462  unmapped.push_back({wasm_null_padding_start,
463  wasm_null.address() - wasm_null_padding_start});
464  }
465  unmapped.push_back({wasm_null->payload(), WasmNull::kPayloadSize});
466  return unmapped;
467 #else
468  return {};
469 #endif // V8_STATIC_ROOTS
470 }
