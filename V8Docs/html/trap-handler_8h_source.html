162 inline bool IsTrapHandlerEnabled() {
163  TH_DCHECK(!g_is_trap_handler_enabled || V8_TRAP_HANDLER_SUPPORTED);
164  // Disallow enabling the trap handler after retrieving the current value.
165  // Re-enabling them late can produce issues because code or objects might have
166  // been generated under the assumption that trap handlers are disabled.
167  // Note: We test before setting to avoid contention by an unconditional write.
168  if (g_can_enable_trap_handler.load(std::memory_order_relaxed)) {
169  g_can_enable_trap_handler.store(false, std::memory_order_relaxed);
170  }
171  return g_is_trap_handler_enabled;
172 }

192 inline void SetThreadInWasm() {
193  if (IsTrapHandlerEnabled()) {
194  TH_DCHECK(!IsThreadInWasm());
