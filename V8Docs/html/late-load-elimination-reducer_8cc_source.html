35 void LateLoadEliminationAnalyzer::Run() {
36  TRACE("LateLoadElimination: Starting analysis");
37  LoopFinder loop_finder(phase_zone_, &graph_, LoopFinder::Config{});
38  AnalyzerIterator iterator(phase_zone_, graph_, loop_finder);
39 
40  bool compute_start_snapshot = true;
41  while (iterator.HasNext()) {
42  const Block* block = iterator.Next();
43 
44  ProcessBlock(*block, compute_start_snapshot);
45  compute_start_snapshot = true;
46 
47  // Consider re-processing for loops.
48  if (const GotoOp* last = block->LastOperation(graph_).TryCast<GotoOp>()) {
49  if (last->destination->IsLoop() &&
50  last->destination->LastPredecessor() == block) {
51  TRACE("> Considering reprocessing loop header " << last->destination);
52  const Block* loop_header = last->destination;
53  // {block} is the backedge of a loop. We recompute the loop header's
54  // initial snapshots, and if they differ from its original snapshot,
55  // then we revisit the loop.
56  if (BeginBlock<true>(loop_header)) {
57  TRACE(">> Will need to revisit loop");
58  // We set the snapshot of the loop's 1st predecessor to the newly
59  // computed snapshot. It's not quite correct, but this predecessor
60  // is guaranteed to end with a Goto, and we are now visiting the
61  // loop, which means that we don't really care about this
62  // predecessor anymore.
63  // The reason for saving this snapshot is to prevent infinite
64  // looping, since the next time we reach this point, the backedge
65  // snapshot could still invalidate things from the forward edge
66  // snapshot. By restricting the forward edge snapshot, we prevent
67  // this.
68  const Block* loop_1st_pred =
69  loop_header->LastPredecessor()->NeighboringPredecessor();
70  FinishBlock(loop_1st_pred);
71  // And we start a new fresh snapshot from this predecessor.
72  auto pred_snapshots =
73  block_to_snapshot_mapping_[loop_1st_pred->index()];
74  non_aliasing_objects_.StartNewSnapshot(
75  pred_snapshots->alias_snapshot);
76  object_maps_.StartNewSnapshot(pred_snapshots->maps_snapshot);
77  memory_.StartNewSnapshot(pred_snapshots->memory_snapshot);
78 
79  iterator.MarkLoopForRevisit();
80  compute_start_snapshot = false;
81  } else {
82  TRACE(">> No need to revisit loop");
83  SealAndDiscard();
84  }
85  }
86  }
87  }
88 
89  FixedOpIndexSidetable<SaturatedUint8> total_use_counts(graph_.op_id_count(),
90  phase_zone_, &graph_);
91  // Incorpoare load elimination decisions into int32-truncation data.
92  for (auto it = int32_truncated_loads_.begin();
93  it != int32_truncated_loads_.end();) {
94  OpIndex load_idx = it->first;
95  auto& truncations = it->second;
96  Replacement replacement = GetReplacement(load_idx);
97  // We distinguish a few different cases.
98  if (!replacement.IsLoadElimination()) {
99  // Case 1: This load is not going to be eliminated.
100  total_use_counts[load_idx] += graph_.Get(load_idx).saturated_use_count;
101  // Check if all uses we know so far, are all truncating uses.
102  if (total_use_counts[load_idx].IsSaturated() ||
103  total_use_counts[load_idx].Get() > truncations.size()) {
104  // We do know that we cannot int32-truncate this load, so eliminate
105  // it from the candidates.
106  int32_truncated_loads_.erase(it++);
107  continue;
108  }
109  // Otherwise, keep this candidate.
110  ++it;
111  continue;
112  } else {
113  OpIndex replaced_by_idx = replacement.replacement();
114  const Operation& replaced_by = graph_.Get(replaced_by_idx);
115  if (!replaced_by.Is<LoadOp>()) {
116  // Case 2: This load is replaced by a non-load (e.g. by the value
117  // stored that the load would read). This load cannot be truncated
118  // (because we are not going to have a load anymore), so eliminate it
119  // from the candidates.
120  int32_truncated_loads_.erase(it++);
121  continue;
122  } else {
123  // Case 3: This load is replaced by another load, so the truncating
124  // and the total uses have to be merged into the replacing use.
125  auto it2 = int32_truncated_loads_.find(replaced_by_idx);
126  if (it2 == int32_truncated_loads_.end()) {
127  // Case 3a: The replacing load is not tracked, so we assume it has
128  // non-truncating uses, so we can also ignore this load.
129  int32_truncated_loads_.erase(it++);
130  continue;
131  } else {
132  // Case 3b: The replacing load might have be a candidate for int32
133  // truncation, we merge the information into that load.
134  total_use_counts[replaced_by_idx] +=
135  graph_.Get(load_idx).saturated_use_count;
136  it2->second.insert(truncations.begin(), truncations.end());
137  int32_truncated_loads_.erase(it++);
138  continue;
139  }
140  }
141  }
142  }
143 
144  // We have prepared everything and now extract the necessary replacement
145  // information.
146  for (const auto& [load_idx, int32_truncations] : int32_truncated_loads_) {
147  if (int32_truncations.empty()) continue;
148  if (!total_use_counts[load_idx].IsSaturated() &&
149  total_use_counts[load_idx].Get() == int32_truncations.size()) {
150  // All uses of this load are int32-truncating loads, so we replace them.
151  DCHECK(GetReplacement(load_idx).IsNone() ||
152  GetReplacement(load_idx).IsTaggedLoadToInt32Load());
153  for (const auto [change_idx, bitcast_idx] : int32_truncations) {
154  replacements_[change_idx] =
155  Replacement::Int32TruncationElimination(load_idx);
156  replacements_[bitcast_idx] = Replacement::TaggedBitcastElimination();
157  replacements_[load_idx] = Replacement::TaggedLoadToInt32Load();
158  }
159  }
160  }
161 }

163 void LateLoadEliminationAnalyzer::ProcessBlock(const Block& block,
164  bool compute_start_snapshot) {
165  TRACE("> ProcessBlock(" << block.index() << ")");
166  if (compute_start_snapshot) {
167  BeginBlock(&block);
168  }
169  if (block.IsLoop() && BackedgeHasSnapshot(block)) {
170  // Update the associated snapshot for the forward edge with the merged
171  // snapshot information from the forward- and backward edge.
172  // This will make sure that when evaluating whether a loop needs to be
173  // revisited, the inner loop compares the merged state with the backedge
174  // preventing us from exponential revisits for loops where the backedge
175  // invalidates loads which are eliminatable on the forward edge.
176  StoreLoopSnapshotInForwardPredecessor(block);
177  }
178 
179  for (OpIndex op_idx : graph_.OperationIndices(block)) {
180  Operation& op = graph_.Get(op_idx);
181  if (ShouldSkipOptimizationStep()) continue;
182  if (ShouldSkipOperation(op)) continue;
183  switch (op.opcode) {
184  case Opcode::kLoad:
185  // Eliminate load or update state
186  ProcessLoad(op_idx, op.Cast<LoadOp>());
187  break;
188  case Opcode::kStore:
189  // Update state (+ maybe invalidate aliases)
190  ProcessStore(op_idx, op.Cast<StoreOp>());
191  break;
192  case Opcode::kAllocate:
193  // Create new non-alias
194  ProcessAllocate(op_idx, op.Cast<AllocateOp>());
195  break;
196  case Opcode::kCall:
197  // Invalidate state (+ maybe invalidate aliases)
198  ProcessCall(op_idx, op.Cast<CallOp>());
199  break;
200  case Opcode::kAssumeMap:
201  // Update known maps
202  ProcessAssumeMap(op_idx, op.Cast<AssumeMapOp>());
203  break;
204  case Opcode::kChange:
205  // Check for tagged -> word32 load replacement
206  ProcessChange(op_idx, op.Cast<ChangeOp>());
207  break;
208 
209  case Opcode::kWordBinop:
210  // A WordBinop should never invalidate aliases (since the only time when
211  // it should take a non-aliasing object as input is for Smi checks).
212  DcheckWordBinop(op_idx, op.Cast<WordBinopOp>());
213  break;
214 
215  case Opcode::kFrameState:
216  case Opcode::kDeoptimizeIf:
217  case Opcode::kComparison:
218 #ifdef V8_ENABLE_WEBASSEMBLY
219  case Opcode::kTrapIf:
220 #endif
221  // We explicitly break for these opcodes so that we don't call
222  // InvalidateAllNonAliasingInputs on their inputs, since they don't
223  // really create aliases. (and also, they don't write so it's
224  // fine to break)
225  DCHECK(!op.Effects().can_write());
226  break;
227 
228  case Opcode::kDeoptimize:
229  case Opcode::kReturn:
230  // We explicitly break for these opcodes so that we don't call
231  // InvalidateAllNonAliasingInputs on their inputs, since they are block
232  // terminators without successors, meaning that it's not useful for the
233  // rest of the analysis to invalidate anything here.
234  DCHECK(op.IsBlockTerminator() && SuccessorBlocks(op).empty());
235  break;
236 
237  case Opcode::kAtomicRMW:
238  ProcessAtomicRMW(op_idx, op.Cast<AtomicRMWOp>());
239  break;
240 
241  case Opcode::kCatchBlockBegin:
242  case Opcode::kRetain:
243  case Opcode::kDidntThrow:
244  case Opcode::kCheckException:
245  case Opcode::kAtomicWord32Pair:
246  case Opcode::kMemoryBarrier:
247  case Opcode::kParameter:
248  case Opcode::kDebugBreak:
249  case Opcode::kJSStackCheck:
250 #ifdef V8_ENABLE_WEBASSEMBLY
251  case Opcode::kWasmStackCheck:
252  case Opcode::kSimd128LaneMemory:
253  case Opcode::kGlobalSet:
254  case Opcode::kArraySet:
255  case Opcode::kStructSet:
256  case Opcode::kSetStackPointer:
257 #endif // V8_ENABLE_WEBASSEMBLY
258  // We explicitly break for those operations that have can_write effects
259  // but don't actually write, or cannot interfere with load elimination.
260  break;
261  default:
262  // Operations that `can_write` should invalidate the state. All such
263  // operations should be already handled above, which means that we don't
264  // need a `if (can_write) { Invalidate(); }` here.
265  CHECK(!op.Effects().can_write());
266 
267  // Even if the operation doesn't write, it could create an alias to its
268  // input by returning it. This happens for instance in Phis and in
269  // Change (although ChangeOp is already handled earlier by calling
270  // ProcessChange). We are conservative here by calling
271  // InvalidateAllNonAliasingInputs for all operations even though only
272  // few can actually create aliases to fresh allocations, the reason
273  // being that missing such a case would be a security issue, and it
274  // should be rare for fresh allocations to be used outside of
275  // Call/Store/Load/Change anyways.
276  TRACE("> Process other op (id=" << op_idx << ")");
277  InvalidateAllNonAliasingInputs(op);
278 
279  break;
280  }
281  }
282 
283  FinishBlock(&block);
284 }

288 // Returns true if replacing a Load with a RegisterRepresentation
289 // {expected_reg_rep} and MemoryRepresentation of {expected_loaded_repr} with an
290 // operation with RegisterRepresentation {actual} is valid. For instance,
291 // replacing an operation that returns a Float64 by one that returns a Word64 is
292 // not valid. Similarly, replacing a Tagged with an untagged value is probably
293 // not valid because of the GC.
294 bool RepIsCompatible(RegisterRepresentation actual,
295  RegisterRepresentation expected_reg_repr,
296  MemoryRepresentation expected_loaded_repr) {
297  if (expected_loaded_repr.SizeInBytes() !=
298  MemoryRepresentation::FromRegisterRepresentation(actual, true)
299  .SizeInBytes()) {
300  // The replacement was truncated when being stored or should be truncated
301  // (or sign-extended) during the load. Since we don't have enough
302  // truncations operators in Turboshaft (eg, we don't have Int32 to Int8
303  // truncation), we just prevent load elimination in this case.
304 
305  // TODO(dmercadier): add more truncations operators to Turboshaft, and
306  // insert the correct truncation when there is a mismatch between
307  // {expected_loaded_repr} and {actual}.
308 
309  return false;
310  }
311 
312  return expected_reg_repr == actual;
313 }

317 void LateLoadEliminationAnalyzer::ProcessLoad(OpIndex op_idx,
318  const LoadOp& load) {
319  TRACE("> ProcessLoad(" << op_idx << ")");
320 
321  if (!load.kind.load_eliminable) {
322  // We don't optimize Loads/Stores to addresses that could be accessed
323  // non-canonically.
324  TRACE(">> Not load-eliminable; skipping");
325  return;
326  }
327  if (load.kind.is_atomic) {
328  // Atomic loads cannot be eliminated away, but potential concurrency
329  // invalidates known stored values.
330  TRACE(">> Atomic load, invalidating related memory");
331  memory_.Invalidate(load.base(), load.index(), load.offset);
332  return;
333  }
334 
335  // We need to insert the load into the truncation mapping as a key, because
336  // all loads need to be revisited during processing.
337  int32_truncated_loads_[op_idx];
338 
339  if (OpIndex existing = memory_.Find(load); existing.valid()) {
340  const Operation& replacement = graph_.Get(existing);
341  // We need to make sure that {load} and {replacement} have the same output
342  // representation. In particular, in unreachable code, it's possible that
343  // the two of them have incompatible representations (like one could be
344  // Tagged and the other one Float64).
345  DCHECK_EQ(replacement.outputs_rep().size(), 1);
346  DCHECK_EQ(load.outputs_rep().size(), 1);
347  TRACE(">> Found potential replacement at offset " << existing);
348  if (RepIsCompatible(replacement.outputs_rep()[0], load.outputs_rep()[0],
349  load.loaded_rep)) {
350  TRACE(">>> Confirming replacement");
351  replacements_[op_idx] = Replacement::LoadElimination(existing);
352  return;
353  } else {
354  TRACE(">>> Replacement has wrong representation: "
355  << replacement.outputs_rep()[0] << " instead of "
356  << load.outputs_rep()[0]);
357  }
358  }
359  // Reset the replacement of {op_idx} to Invalid, in case a previous visit of a
360  // loop has set it to something else.
361  replacements_[op_idx] = Replacement::None();
362 
363  // TODO(dmercadier): if we precisely track maps, then we could know from the
364  // map what we are loading in some cases. For instance, if the elements_kind
365  // of the map is *_DOUBLE_ELEMENTS, then a load at offset
366  // JSObject::kElementsOffset always load a FixedDoubleArray, with map
367  // fixed_double_array_map.
368 
369  if (const ConstantOp* base = graph_.Get(load.base()).TryCast<ConstantOp>();
370  base != nullptr && base->kind == ConstantOp::Kind::kExternal) {
371  // External constants can be written by other threads, so we don't
372  // load-eliminate them, in order to always reload them.
373  TRACE(">> Ignoring load from External constant");
374  return;
375  }
376 
377  memory_.Insert(load, op_idx);
378 }

380 void LateLoadEliminationAnalyzer::ProcessStore(OpIndex op_idx,
381  const StoreOp& store) {
382  TRACE("> ProcessStore(" << op_idx << ")");
383  // If we have a raw base and we allow those to be inner pointers, we can
384  // overwrite arbitrary values and need to invalidate anything that is
385  // potentially aliasing.
386  const bool invalidate_maybe_aliasing =
387  !store.kind.tagged_base &&
388  raw_base_assumption_ == RawBaseAssumption::kMaybeInnerPointer;
389 
390  if (invalidate_maybe_aliasing) {
391  TRACE(
392  ">> Raw base or maybe inner pointer ==> Invalidating whole "
393  "maybe-aliasing memory");
394  memory_.InvalidateMaybeAliasing();
395  }
396 
397  if (!store.kind.load_eliminable) {
398  // We don't optimize Loads/Stores to addresses that could be accessed
399  // non-canonically.
400  TRACE(">> Not load-eliminable; skipping");
401  return;
402  }
403 
404  // Updating the known stored values.
405  if (!invalidate_maybe_aliasing) memory_.Invalidate(store);
406  memory_.Insert(store);
407 
408  // Updating aliases if the value stored was known as non-aliasing.
409  OpIndex value = store.value();
410  if (non_aliasing_objects_.HasKeyFor(value)) {
411  TRACE(">> Invalidate non-alias for value=" << value);
412  non_aliasing_objects_.Set(value, false);
413  }
414 
415  // If we just stored a map, invalidate all object_maps_.
416  if (store.offset == HeapObject::kMapOffset && !store.index().valid()) {
417  // TODO(dmercadier): can we only do this for objects that are potentially
418  // aliasing with the `base` (based on their maps and the maps of `base`)?
419  // Also, it might be worth to record a new map if this is actually a map
420  // store.
421  // TODO(dmercadier): do this only if `value` is a Constant with kind
422  // kHeapObject, since all map stores should store a known constant maps.
423  TRACE(">> Wiping all maps\n");
424  for (auto it : object_maps_) {
425  object_maps_.Set(it.second, MapMaskAndOr{});
426  }
427  }
428 }

430 void LateLoadEliminationAnalyzer::ProcessAtomicRMW(OpIndex op_idx,
431  const AtomicRMWOp& store) {
432 #if V8_ENABLE_WEBASSEMBLY
433  TRACE("> ProcessAtomicRMW(" << op_idx << ")");
434  // With shared-everything-treads atomic rmw operations are also used for heap
435  // operations. If the atomic operation is not operating on linear memory, we
436  // need to invalidate it. TODO(mliedtke): Only invalidate the potentially
437  // aliasing information.
438  if (!v8_flags.experimental_wasm_shared ||
439  store.base_rep == RegisterRepresentation::WordPtr()) {
440  TRACE(">> Skipping operation on linear memory");
441  return;
442  }
443  TRACE(">> Invalidating whole maybe-aliasing memory");
444  memory_.InvalidateMaybeAliasing();
445 #endif
446 }
447 
448 // Since we only loosely keep track of what can or can't alias, we assume that
449 // anything that was guaranteed to not alias with anything (because it's in
450 // {non_aliasing_objects_}) can alias with anything when coming back from the
451 // call if it was an argument of the call.
452 void LateLoadEliminationAnalyzer::ProcessCall(OpIndex op_idx,
453  const CallOp& op) {
454  TRACE("> ProcessCall(" << op_idx << ")");
455  const Operation& callee = graph_.Get(op.callee());
456 #ifdef DEBUG
457  if (const ConstantOp* external_constant =
458  callee.template TryCast<Opmask::kExternalConstant>()) {
459  if (external_constant->external_reference() ==
460  ExternalReference::check_object_type()) {
461  return;
462  }
463  }
464 #endif
465 
466  // Some builtins do not create aliases and do not invalidate existing
467  // memory, and some even return fresh objects. For such cases, we don't
468  // invalidate the state, and record the non-alias if any.
469  if (!op.Effects().can_write()) {
470  TRACE(">> Call doesn't write, skipping");
471  return;
472  }
473 
474  auto builtin_id = TryGetBuiltinId(callee.TryCast<ConstantOp>(), broker_);
475 
476  if (builtin_id) {
477  switch (*builtin_id) {
478  // TODO(dmercadier): extend this list.
479  case Builtin::kCopyFastSmiOrObjectElements:
480  // This function just replaces the Elements array of an object.
481  // It doesn't invalidate any alias or any other memory than this
482  // Elements array.
483  TRACE(
484  ">> Call is CopyFastSmiOrObjectElements, invalidating only "
485  "Elements for "
486  << op.arguments()[0]);
487  memory_.Invalidate(op.arguments()[0], OpIndex::Invalid(),
488  JSObject::kElementsOffset);
489  return;
490  default:
491  break;
492  }
493  }
494 
495  // Not a builtin call, or not a builtin that we know doesn't invalidate
496  // memory.
497  InvalidateAllNonAliasingInputs(op);
498 
499  if (builtin_id) {
500  switch (*builtin_id) {
501  case Builtin::kCreateShallowObjectLiteral:
502  // This builtin creates a fresh non-aliasing object.
503  non_aliasing_objects_.Set(op_idx, true);
504  break;
505  default:
506  break;
507  }
508  }
509 
510  // The call could modify arbitrary memory, so we invalidate every
511  // potentially-aliasing object.
512  memory_.InvalidateMaybeAliasing();
513 }
514 
515 // The only time an Allocate should flow into a WordBinop is for Smi checks
516 // (which, by the way, should be removed by MachineOptimizationReducer (since
517 // Allocate never returns a Smi), but there is no guarantee that this happens
518 // before load elimination). So, there is no need to invalidate non-aliases, and
519 // we just DCHECK in this function that indeed, nothing else than a Smi check
520 // happens on non-aliasing objects.
521 void LateLoadEliminationAnalyzer::DcheckWordBinop(OpIndex op_idx,
522  const WordBinopOp& binop) {
523 #ifdef DEBUG
524  auto check = [&](V<Word> left, V<Word> right) {
525  if (auto key = non_aliasing_objects_.TryGetKeyFor(left);
526  key.has_value() && non_aliasing_objects_.Get(*key)) {
527  int64_t cst;
528  DCHECK_EQ(binop.kind, WordBinopOp::Kind::kBitwiseAnd);
529  DCHECK(OperationMatcher(graph_).MatchSignedIntegralConstant(right, &cst));
530  DCHECK_EQ(cst, kSmiTagMask);
531  }
532  };
533  check(binop.left(), binop.right());
534  check(binop.right(), binop.left());
535 #endif
536 }

546 void LateLoadEliminationAnalyzer::InvalidateIfAlias(OpIndex op_idx) {
547  if (auto key = non_aliasing_objects_.TryGetKeyFor(op_idx);
548  key.has_value() && non_aliasing_objects_.Get(*key)) {
549  // An known non-aliasing object was passed as input to the Call; the Call
550  // could create aliases, so we have to consider going forward that this
551  // object could actually have aliases.
552  TRACE(">> InvalidateIfAlias: invalidating for id=" << op_idx);
553  non_aliasing_objects_.Set(*key, false);
554  }
555  if (const FrameStateOp* frame_state =
556  graph_.Get(op_idx).TryCast<FrameStateOp>()) {
557  TRACE(
558  ">> InvalidateIfAlias: recursively invalidating for FrameState inputs "
559  "(FrameState id="
560  << op_idx << ")");
561  // We also mark the arguments of FrameState passed on to calls as
562  // potentially-aliasing, because they could be accessed by the caller with a
563  // function_name.arguments[index].
564  // TODO(dmercadier): this is more conservative that we'd like, since only a
565  // few functions use .arguments. Using a native-context-specific protector
566  // for .arguments might allow to avoid invalidating frame states' content.
567  for (OpIndex input : frame_state->inputs()) {
568  InvalidateIfAlias(input);
569  }
570  }
571 }

587 bool IsInt32TruncatedLoadPattern(const Graph& graph, OpIndex change_idx,
588  const ChangeOp& change, OpIndex* bitcast_idx,
589  OpIndex* load_idx) {
590  DCHECK_EQ(change_idx, graph.Index(change));
591 
592  if (!change.Is<Opmask::kTruncateWord64ToWord32>()) return false;
593  const TaggedBitcastOp* bitcast =
594  graph.Get(change.input())
595  .TryCast<Opmask::kBitcastTaggedToWordPtrForTagAndSmiBits>();
596  if (bitcast == nullptr) return false;
597  // We require that the bitcast has no other uses. This could be slightly
598  // generalized by allowing multiple int32-truncating uses, but that is more
599  // expensive to detect and it is very unlikely that we ever see such a case
600  // (e.g. because of GVN).
601  if (!bitcast->saturated_use_count.IsOne()) return false;
602  const LoadOp* load = graph.Get(bitcast->input()).TryCast<LoadOp>();
603  if (load == nullptr) return false;
604  if (load->loaded_rep.SizeInBytesLog2() !=
605  MemoryRepresentation::Int32().SizeInBytesLog2()) {
606  return false;
607  }
608  if (bitcast_idx) *bitcast_idx = change.input();
609  if (load_idx) *load_idx = bitcast->input();
610  return true;
611 }

613 void LateLoadEliminationAnalyzer::ProcessChange(OpIndex op_idx,
614  const ChangeOp& change) {
615  TRACE("> ProcessChange(" << op_idx << ")");
616  // We look for this special case:
617  // TruncateWord64ToWord32(BitcastTaggedToWordPtrForTagAndSmiBits(Load(x))) =>
618  // Load(x)
619  // where the new Load uses Int32 rather than the tagged representation.
620  OpIndex bitcast_idx, load_idx;
621  if (IsInt32TruncatedLoadPattern(graph_, op_idx, change, &bitcast_idx,
622  &load_idx)) {
623  int32_truncated_loads_[load_idx][op_idx] = bitcast_idx;
624  }
625 
626  InvalidateIfAlias(change.input());
627 }

663 template <bool for_loop_revisit>
664 bool LateLoadEliminationAnalyzer::BeginBlock(const Block* block) {
665  DCHECK_IMPLIES(
666  for_loop_revisit,
667  block->IsLoop() &&
668  block_to_snapshot_mapping_[block->LastPredecessor()->index()]
669  .has_value());
670 
671  // Collect the snapshots of all predecessors.
672  {
673  predecessor_alias_snapshots_.clear();
674  predecessor_maps_snapshots_.clear();
675  predecessor_memory_snapshots_.clear();
676  for (const Block* p : block->PredecessorsIterable()) {
677  auto pred_snapshots = block_to_snapshot_mapping_[p->index()];
678  // When we visit the loop for the first time, the loop header hasn't
679  // been visited yet, so we ignore it.
680  DCHECK_IMPLIES(!pred_snapshots.has_value(),
681  block->IsLoop() && block->LastPredecessor() == p);
682  if (!pred_snapshots.has_value()) {
683  DCHECK(!for_loop_revisit);
684  continue;
685  }
686  // Note that the backedge snapshot of an inner loop in kFirstVisit will
687  // also be taken into account if we are in the kSecondVisit of an outer
688  // loop. The data in the backedge snapshot could be out-dated, but if it
689  // is, then it's fine: if the backedge of the outer-loop was more
690  // restrictive than its forward incoming edge, then the forward incoming
691  // edge of the inner loop should reflect this restriction.
692  predecessor_alias_snapshots_.push_back(pred_snapshots->alias_snapshot);
693  predecessor_memory_snapshots_.push_back(pred_snapshots->memory_snapshot);
694  if (p->NeighboringPredecessor() != nullptr || !block->IsLoop() ||
695  block->LastPredecessor() != p) {
696  // We only add a MapSnapshot predecessor for non-backedge predecessor.
697  // This is because maps coming from inside of the loop may be wrong
698  // until a specific check has been executed.
699  predecessor_maps_snapshots_.push_back(pred_snapshots->maps_snapshot);
700  }
701  }
702  }
703 
704  // Note that predecessors are in reverse order, which means that the backedge
705  // is at offset 0.
706  constexpr int kBackedgeOffset = 0;
707  constexpr int kForwardEdgeOffset = 1;
708 
709  bool loop_needs_revisit = false;
710  // Start a new snapshot for this block by merging information from
711  // predecessors.
712  auto merge_aliases = [&](AliasKey key,
713  base::Vector<const bool> predecessors) -> bool {
714  if (for_loop_revisit && predecessors[kForwardEdgeOffset] &&
715  !predecessors[kBackedgeOffset]) {
716  // The backedge doesn't think that {key} is no-alias, but the loop
717  // header previously thought it was --> need to revisit.
718  loop_needs_revisit = true;
719  }
720  return base::all_of(predecessors);
721  };
722  non_aliasing_objects_.StartNewSnapshot(
723  base::VectorOf(predecessor_alias_snapshots_), merge_aliases);
724 
725  auto merge_maps =
726  [&](MapKey key,
727  base::Vector<const MapMaskAndOr> predecessors) -> MapMaskAndOr {
728  MapMaskAndOr minmax;
729  for (const MapMaskAndOr pred : predecessors) {
730  if (is_empty(pred)) {
731  // One of the predecessors doesn't have maps for this object, so we have
732  // to assume that this object could have any map.
733  return MapMaskAndOr{};
734  }
735  minmax = CombineMinMax(minmax, pred);
736  }
737  return minmax;
738  };
739  object_maps_.StartNewSnapshot(base::VectorOf(predecessor_maps_snapshots_),
740  merge_maps);
741 
742  // Merging for {memory_} means setting values to Invalid unless all
743  // predecessors have the same value.
744  // TODO(dmercadier): we could insert of Phis during the pass to merge existing
745  // information. This is a bit hard, because we are currently in an analyzer
746  // rather than a reducer. Still, we could "prepare" the insertion now and then
747  // really insert them during the Reduce phase of the CopyingPhase.
748  auto merge_memory = [&](MemoryKey key,
749  base::Vector<const OpIndex> predecessors) -> OpIndex {
750  if (for_loop_revisit && predecessors[kForwardEdgeOffset].valid() &&
751  predecessors[kBackedgeOffset] != predecessors[kForwardEdgeOffset]) {
752  // {key} had a value in the loop header, but the backedge and the forward
753  // edge don't agree on its value, which means that the loop invalidated
754  // some memory data, and thus needs to be revisited.
755  loop_needs_revisit = true;
756  }
757  return base::all_equal(predecessors) ? predecessors[0] : OpIndex::Invalid();
758  };
759  memory_.StartNewSnapshot(base::VectorOf(predecessor_memory_snapshots_),
760  merge_memory);
761 
762  if (block->IsLoop()) return loop_needs_revisit;
763  return false;
764 }
