118 template <ExternalPointerTag tag>
119 Handle<Foreign> Factory::NewForeign(Address addr,
120  AllocationType allocation_type) {
121  // Statically ensure that it is safe to allocate foreigns in paged spaces.
122  static_assert(Foreign::kSize <= kMaxRegularHeapObjectSize);
123  Tagged<Map> map = *foreign_map();
124  Tagged<Foreign> foreign = Cast<Foreign>(
125  AllocateRawWithImmortalMap(map->instance_size(), allocation_type, map));
126  DisallowGarbageCollection no_gc;
127  foreign->init_foreign_address<tag>(isolate(), addr);
128  return handle(foreign, isolate());
129 }
