14 void CheckRegisterConfiguration(int count, const Register* registers,
15  const DoubleRegister* double_registers) {
16  // Make sure that the registers are all valid, and don't alias each other.
17  RegList reglist;
18  DoubleRegList double_reglist;
19  for (int i = 0; i < count; ++i) {
20  Register reg = registers[i];
21  DoubleRegister dreg = double_registers[i];
22  DCHECK(reg.is_valid() || dreg.is_valid());
23  DCHECK_NE(reg, kRootRegister);
24 #ifdef V8_COMPRESS_POINTERS
25  DCHECK_NE(reg, kPtrComprCageBaseRegister);
26 #endif
27  if (reg.is_valid()) {
28  DCHECK(!reglist.has(reg));
29  reglist.set(reg);
30  }
31  if (dreg.is_valid()) {
32  DCHECK(!double_reglist.has(dreg));
33  double_reglist.set(dreg);
34  }
35  }
36 }
37 #endif

69 void CallInterfaceDescriptorData::InitializeTypes(
70  const MachineType* machine_types, int machine_types_length) {
71  DCHECK(IsInitializedRegisters());
72  const int types_length = return_count_ + param_count_;
73 
74  // Machine types are either fully initialized or null.
75  if (machine_types == nullptr) {
76  machine_types_ =
77  NewArray<MachineType>(types_length, MachineType::AnyTagged());
78  } else {
79  DCHECK_EQ(machine_types_length, types_length);
80  machine_types_ = NewArray<MachineType>(types_length);
81  for (int i = 0; i < types_length; i++) machine_types_[i] = machine_types[i];
82  }
83 
84  if (!(flags_ & kNoStackScan)) DCHECK(AllStackParametersAreTagged());
85 }

159 void WriteBarrierDescriptor::Verify(CallInterfaceDescriptorData* data) {
160  DCHECK(!AreAliased(ObjectRegister(), SlotAddressRegister(), ValueRegister()));
161  // The default parameters should not clobber vital registers in order to
162  // reduce code size:
163  DCHECK(!AreAliased(ObjectRegister(), kContextRegister,
164  kInterpreterAccumulatorRegister));
165  DCHECK(!AreAliased(SlotAddressRegister(), kContextRegister,
166  kInterpreterAccumulatorRegister));
167  DCHECK(!AreAliased(ValueRegister(), kContextRegister,
168  kInterpreterAccumulatorRegister));
169  DCHECK(!AreAliased(SlotAddressRegister(), kJavaScriptCallNewTargetRegister));
170  // Coincidental: to make calling from various builtins easier.
171  DCHECK_EQ(ObjectRegister(), kJSFunctionRegister);
172  // We need a certain set of registers by default:
173  RegList allocatable_regs = data->allocatable_registers();
174  DCHECK(allocatable_regs.has(kContextRegister));
175  DCHECK(allocatable_regs.has(kReturnRegister0));
176  VerifyArgumentRegisterCount(data, 4);
177 }
