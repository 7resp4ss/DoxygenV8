604  uint32_t length) {
605  DCHECK(IsAligned(reinterpret_cast<Address>(chars), sizeof(base::uc16)));
606  const uint16_t* start = chars;
607  const uint16_t* limit = chars + length;
608 
609  if (static_cast<size_t>(length) >= kUIntptrSize) {
610  // Check unaligned chars.
611  while (!IsAligned(reinterpret_cast<Address>(chars), kUIntptrSize)) {
612  if (*chars > unibrow::Latin1::kMaxChar) {
613  return static_cast<uint32_t>(chars - start);
614  }
615  ++chars;
616  }
617 
618  // Check aligned words.
619  static_assert(unibrow::Latin1::kMaxChar == 0xFF);
620 #ifdef V8_TARGET_LITTLE_ENDIAN
621  const uintptr_t non_one_byte_mask = kUintptrAllBitsSet / 0xFFFF * 0xFF00;
622 #else
623  const uintptr_t non_one_byte_mask = kUintptrAllBitsSet / 0xFFFF * 0x00FF;
624 #endif
625  while (chars + sizeof(uintptr_t) <= limit) {
626  if (*reinterpret_cast<const uintptr_t*>(chars) & non_one_byte_mask) {
627  break;
628  }
629  chars += (sizeof(uintptr_t) / sizeof(base::uc16));
630  }
631  }
632 
633  // Check remaining unaligned chars, or find non-one-byte char in word.
634  while (chars < limit) {
635  if (*chars > unibrow::Latin1::kMaxChar) {
636  return static_cast<uint32_t>(chars - start);
637  }
638  ++chars;
639  }
640 
641  return static_cast<uint32_t>(chars - start);
642  }
