19 enum class MachineRepresentation : uint8_t {
20  kNone,
21  kBit,
22  // Integral representations must be consecutive, in order of increasing order.
23  kWord8,
24  kWord16,
25  kWord32,
26  kWord64,
27  // (uncompressed) MapWord
28  // kMapWord is the representation of a map word, i.e. a map in the header
29  // of a HeapObject.
30  // If V8_MAP_PACKING is disabled, a map word is just the map itself. Hence
31  // kMapWord is equivalent to kTaggedPointer -- in fact it will be
32  // translated to kTaggedPointer during memory lowering.
33  // If V8_MAP_PACKING is enabled, a map word is a Smi-like encoding of a map
34  // and some meta data. Memory lowering of kMapWord loads/stores
35  // produces low-level kTagged loads/stores plus the necessary
36  // decode/encode operations.
37  // In either case, the kMapWord representation is not used after memory
38  // lowering.
39  kMapWord,
40  kTaggedSigned, // (uncompressed) Smi
41  kTaggedPointer, // (uncompressed) HeapObject
42  kTagged, // (uncompressed) Object (Smi or HeapObject)
43  kCompressedPointer, // (compressed) HeapObject
44  kCompressed, // (compressed) Object (Smi or HeapObject)
45  kProtectedPointer, // (uncompressed) TrustedObject
46  kIndirectPointer, // (indirect) HeapObject
47  // A 64-bit pointer encoded in a way (e.g. as offset) that guarantees it will
48  // point into the sandbox.
49  kSandboxedPointer,
50  // kFloat16RawBits is not a real FP representation! It's representation of a
51  // float16 bitcast to a word16, and is used for conversions to and from
52  // Float16Array elements. It is not used after machine lowering.
53  kFloat16RawBits,
54  // FP and SIMD representations must be last, and in order of increasing size.
55  kFloat16,
56  kFloat32,
57  kFloat64,
58  kSimd128,
59  kSimd256,
60  kFirstFPRepresentation = kFloat16,
61  kLastRepresentation = kSimd256
62 };

280  static MachineType TypeForRepresentation(const MachineRepresentation& rep,
281  bool isSigned = true) {
282  switch (rep) {
283  case MachineRepresentation::kNone:
284  return MachineType::None();
285  case MachineRepresentation::kBit:
286  return MachineType::Bool();
287  case MachineRepresentation::kWord8:
288  return isSigned ? MachineType::Int8() : MachineType::Uint8();
289  case MachineRepresentation::kWord16:
290  return isSigned ? MachineType::Int16() : MachineType::Uint16();
291  case MachineRepresentation::kWord32:
292  return isSigned ? MachineType::Int32() : MachineType::Uint32();
293  case MachineRepresentation::kWord64:
294  return isSigned ? MachineType::Int64() : MachineType::Uint64();
295  case MachineRepresentation::kFloat16:
296  return MachineType::Float16();
297  case MachineRepresentation::kFloat32:
298  return MachineType::Float32();
299  case MachineRepresentation::kFloat64:
300  return MachineType::Float64();
301  case MachineRepresentation::kSimd128:
302  return MachineType::Simd128();
303  case MachineRepresentation::kSimd256:
304  return MachineType::Simd256();
305  case MachineRepresentation::kTagged:
306  return MachineType::AnyTagged();
307  case MachineRepresentation::kTaggedSigned:
308  return MachineType::TaggedSigned();
309  case MachineRepresentation::kTaggedPointer:
310  return MachineType::TaggedPointer();
311  case MachineRepresentation::kCompressed:
312  return MachineType::AnyCompressed();
313  case MachineRepresentation::kIndirectPointer:
314  return MachineType::IndirectPointer();
315  case MachineRepresentation::kCompressedPointer:
316  return MachineType::CompressedPointer();
317  case MachineRepresentation::kSandboxedPointer:
318  return MachineType::SandboxedPointer();
319  default:
320  UNREACHABLE();
321  }
322  }

440 // Gets the log2 of the element size in bytes of the machine type.
441 V8_EXPORT_PRIVATE inline constexpr int ElementSizeLog2Of(
442  MachineRepresentation rep) {
443  switch (rep) {
444  case MachineRepresentation::kBit:
445  case MachineRepresentation::kWord8:
446  return 0;
447  case MachineRepresentation::kWord16:
448  case MachineRepresentation::kFloat16:
449  return 1;
450  case MachineRepresentation::kWord32:
451  case MachineRepresentation::kFloat32:
452  case MachineRepresentation::kIndirectPointer:
453  return 2;
454  case MachineRepresentation::kWord64:
455  case MachineRepresentation::kFloat64:
456  return 3;
457  case MachineRepresentation::kSimd128:
458  return 4;
459  case MachineRepresentation::kSimd256:
460  return 5;
461  case MachineRepresentation::kTaggedSigned:
462  case MachineRepresentation::kTaggedPointer:
463  case MachineRepresentation::kTagged:
464  case MachineRepresentation::kMapWord:
465  case MachineRepresentation::kCompressedPointer:
466  case MachineRepresentation::kCompressed:
467  case MachineRepresentation::kProtectedPointer:
468  return kTaggedSizeLog2;
469  case MachineRepresentation::kSandboxedPointer:
470  return kSystemPointerSizeLog2;
471  default:
472  UNREACHABLE();
473  }
474 }
