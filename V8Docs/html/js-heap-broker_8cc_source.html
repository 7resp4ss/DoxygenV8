213 ElementAccessFeedback const& ElementAccessFeedback::Refine(
214  JSHeapBroker* broker, ZoneRefSet<Map> const& inferred_maps,
215  bool always_keep_group_target) const {
216  ElementAccessFeedback& refined_feedback =
217  *broker->zone()->New<ElementAccessFeedback>(broker->zone(), keyed_mode(),
218  slot_kind());
219  if (inferred_maps.size() == 0) return refined_feedback;
220 
221  for (auto const& group : transition_groups()) {
222  DCHECK(!group.empty());
223  TransitionGroup new_group(broker->zone());
224  for (size_t i = 1; i < group.size(); ++i) {
225  MapRef source = group[i];
226  if (inferred_maps.contains(source)) {
227  new_group.push_back(source);
228  }
229  }
230 
231  MapRef target = group.front();
232  bool const keep_target = always_keep_group_target ||
233  inferred_maps.contains(target) ||
234  new_group.size() > 1;
235  if (keep_target) {
236  new_group.push_back(target);
237  // The target must be at the front, the order of sources doesn't matter.
238  std::swap(new_group[0], new_group[new_group.size() - 1]);
239  }
240 
241  if (!new_group.empty()) {
242  DCHECK(new_group.size() == 1 || new_group.front().equals(target));
243  refined_feedback.transition_groups_.push_back(std::move(new_group));
244  }
245  }
246  return refined_feedback;
247 }

318  FeedbackSlotKind kind = nexus.kind();
319  constexpr bool kStringKeysUnsupported = false;
320  if (IsKeyedLoadICKind(kind)) {
321  bool string_keys = nexus.GetKeyType() == IcCheckType::kProperty;
322  return KeyedAccessMode(AccessMode::kLoad, nexus.GetKeyedAccessLoadMode(),
323  string_keys);
324  }
325  if (IsKeyedHasICKind(kind)) {
326  return KeyedAccessMode(AccessMode::kHas, nexus.GetKeyedAccessLoadMode(),
327  kStringKeysUnsupported);

330  return KeyedAccessMode(AccessMode::kDefine, nexus.GetKeyedAccessStoreMode(),

339  IsDefineKeyedOwnPropertyInLiteralKind(kind)) {

342  kStringKeysUnsupported);

469 ProcessedFeedback const& JSHeapBroker::ReadFeedbackForPropertyAccess(
470  FeedbackSource const& source, AccessMode mode,
471  OptionalNameRef static_name) {
472  FeedbackNexus nexus(source.vector, source.slot, feedback_nexus_config());
473  FeedbackSlotKind kind = nexus.kind();
474  if (nexus.IsUninitialized()) return NewInsufficientFeedback(kind);
475 
476  ZoneVector<MapRef> maps(zone());
477  bool has_deprecated_map_without_migration_target = false;
478  nexus.IterateMapsWithUnclearedHandler(
479  [this, &maps, &has_deprecated_map_without_migration_target](
480  DirectHandle<Map> map_handle) {
481  MapRef map = MakeRefAssumeMemoryFence(this, *map_handle);
482  // May change concurrently at any time - must be guarded by a
483  // dependency if non-deprecation is important.
484  if (map.is_deprecated()) {
485  // TODO(ishell): support fast map updating if we enable it.
486  CHECK(!v8_flags.fast_map_update);
487  std::optional<Tagged<Map>> maybe_map = MapUpdater::TryUpdateNoLock(
488  isolate(), *map.object(), ConcurrencyMode::kConcurrent);
489  if (maybe_map.has_value()) {
490  map = MakeRefAssumeMemoryFence(this, maybe_map.value());
491  } else {
492  return; // Couldn't update the deprecated map.
493  }
494  if (!map.is_migration_target()) {
495  // Maps are marked as migration targets only when an object
496  // migrates, so it's possible to have a deprecated map whose updated
497  // counterpart is not a migration target.
498  has_deprecated_map_without_migration_target = true;
499  }
500  }
501  if (map.is_abandoned_prototype_map()) return;
502  maps.push_back(map);
503  });
504 
505  OptionalNameRef name =
506  static_name.has_value() ? static_name : GetNameFeedback(nexus);
507 
508  if (nexus.ic_state() == InlineCacheState::MEGADOM) {
509  DCHECK(maps.empty());
510  MaybeObjectHandle maybe_handler = nexus.ExtractMegaDOMHandler();
511  if (!maybe_handler.is_null()) {
512  DirectHandle<MegaDomHandler> handler =
513  Cast<MegaDomHandler>(maybe_handler.object());
514  if (!handler->accessor(kAcquireLoad).IsCleared()) {
515  FunctionTemplateInfoRef info = MakeRefAssumeMemoryFence(
516  this, Cast<FunctionTemplateInfo>(
517  handler->accessor(kAcquireLoad).GetHeapObject()));
518  return *zone()->New<MegaDOMPropertyAccessFeedback>(info, kind);
519  }
520  }
521  }
522 
523  // If no maps were found for a non-megamorphic access, then our maps died
524  // and we should soft-deopt.
525  if (maps.empty() && nexus.ic_state() != InlineCacheState::MEGAMORPHIC) {
526  return NewInsufficientFeedback(kind);
527  }
528 
529  if (name.has_value()) {
530  // We rely on this invariant in JSGenericLowering.
531  DCHECK_IMPLIES(maps.empty(),
532  nexus.ic_state() == InlineCacheState::MEGAMORPHIC);
533  return *zone()->New<NamedAccessFeedback>(
534  *name, maps, kind, has_deprecated_map_without_migration_target);
535  } else if (nexus.GetKeyType() == IcCheckType::kElement && !maps.empty()) {
536  return ProcessFeedbackMapsForElementAccess(
537  maps, KeyedAccessMode::FromNexus(nexus), kind);
538  } else if (nexus.IsOneMapManyNames() && maps.size() == 1) {
539  return ProcessFeedbackMapsForKeyedPropertyAccess(
540  maps, KeyedAccessMode::FromNexus(nexus), kind);
541  } else {
542  // No actionable feedback.
543  DCHECK(maps.empty());
544  DCHECK_EQ(nexus.ic_state(), InlineCacheState::MEGAMORPHIC);
545  // TODO(neis): Using ElementAccessFeedback here is kind of an abuse.
546  return *zone()->New<ElementAccessFeedback>(
547  zone(), KeyedAccessMode::FromNexus(nexus), kind);
548  }
549 }

551 ProcessedFeedback const& JSHeapBroker::ReadFeedbackForGlobalAccess(
552  JSHeapBroker* broker, FeedbackSource const& source) {
553  FeedbackNexus nexus(source.vector, source.slot, feedback_nexus_config());
554  DCHECK(nexus.kind() == FeedbackSlotKind::kLoadGlobalInsideTypeof ||
555  nexus.kind() == FeedbackSlotKind::kLoadGlobalNotInsideTypeof ||
556  nexus.kind() == FeedbackSlotKind::kStoreGlobalSloppy ||
557  nexus.kind() == FeedbackSlotKind::kStoreGlobalStrict);
558  if (nexus.IsUninitialized()) return NewInsufficientFeedback(nexus.kind());
559  if (nexus.ic_state() != InlineCacheState::MONOMORPHIC ||
560  nexus.GetFeedback().IsCleared()) {
561  return *zone()->New<GlobalAccessFeedback>(nexus.kind());
562  }
563 
564  Handle<Object> feedback_value =
565  CanonicalPersistentHandle(nexus.GetFeedback().GetHeapObjectOrSmi());
566 
567  if (IsSmi(*feedback_value)) {
568  // The wanted name belongs to a script-scope variable and the feedback
569  // tells us where to find its value.
570  int const number = Object::NumberValue(*feedback_value);
571  int const script_context_index =
572  FeedbackNexus::ContextIndexBits::decode(number);
573  int const context_slot_index = FeedbackNexus::SlotIndexBits::decode(number);
574  ContextRef context = MakeRefAssumeMemoryFence(
575  this,
576  target_native_context().script_context_table(broker).object()->get(
577  script_context_index, kAcquireLoad));
578 
579  OptionalObjectRef contents = context.get(broker, context_slot_index);
580  if (contents.has_value()) CHECK(!contents->IsTheHole());
581 
582  return *zone()->New<GlobalAccessFeedback>(
583  context, context_slot_index,
584  FeedbackNexus::ImmutabilityBit::decode(number), nexus.kind());
585  }
586 
587  CHECK(IsPropertyCell(*feedback_value));
588  // The wanted name belongs (or did belong) to a property on the global
589  // object and the feedback is the cell holding its value.
590  return *zone()->New<GlobalAccessFeedback>(

850 ElementAccessFeedback const& JSHeapBroker::ProcessFeedbackMapsForElementAccess(
851  ZoneVector<MapRef>& maps, KeyedAccessMode const& keyed_mode,
852  FeedbackSlotKind slot_kind) {
853  DCHECK(!maps.empty());
854  DCHECK(!keyed_mode.string_keys());
855 
856  // Collect possible transition targets.
857  MapHandles possible_transition_targets(isolate());
858  possible_transition_targets.reserve(maps.size());
859  for (MapRef& map : maps) {
860  if (map.CanInlineElementAccess() &&
861  IsFastElementsKind(map.elements_kind()) &&
862  GetInitialFastElementsKind() != map.elements_kind()) {
863  possible_transition_targets.push_back(map.object());
864  }
865  }
866 
867  using TransitionGroup = ElementAccessFeedback::TransitionGroup;
868  ZoneRefMap<MapRef, TransitionGroup> transition_groups(zone());
869 
870  // Separate the actual receiver maps and the possible transition sources.
871  for (MapRef map : maps) {
872  Tagged<Map> transition_target;
873 
874  // Don't generate elements kind transitions from stable maps.
875  if (!map.is_stable()) {
876  // The lock is needed for UnusedPropertyFields (called deep inside
877  // FindElementsKindTransitionedMap).
878  MapUpdaterGuardIfNeeded mumd_scope(this);
879 
880  transition_target = map.object()->FindElementsKindTransitionedMap(
881  isolate(),
882  MapHandlesSpan(possible_transition_targets.begin(),
883  possible_transition_targets.end()),
884  ConcurrencyMode::kConcurrent);
885  }
886 
887  if (transition_target.is_null()) {
888  TransitionGroup group(1, map, zone());
889  transition_groups.insert({map, group});
890  } else {
891  MapRef target = MakeRefAssumeMemoryFence(this, transition_target);
892  TransitionGroup new_group(1, target, zone());
893  TransitionGroup& actual_group =
894  transition_groups.insert({target, new_group}).first->second;
895  actual_group.push_back(map);
896  }
897  }
898 
899  ElementAccessFeedback* result =
900  zone()->New<ElementAccessFeedback>(zone(), keyed_mode, slot_kind);
901  for (auto& entry : transition_groups) {
902  result->AddGroup(std::move(entry.second));
903  }
904 
905  CHECK(!result->transition_groups().empty());
906  return *result;
907 }

909 void ElementAccessFeedback::AddGroup(TransitionGroup&& group) {
910  CHECK(!group.empty());
911  transition_groups_.push_back(std::move(group));
912 
913 #ifdef ENABLE_SLOW_DCHECKS
914  // Check that each of the group's maps occurs exactly once in the whole
915  // feedback. This implies that "a source is not a target".
916  for (MapRef map : group) {
917  int count = 0;
918  for (TransitionGroup const& some_group : transition_groups()) {
919  count +=
920  std::count_if(some_group.begin(), some_group.end(),
921  [&](MapRef some_map) { return some_map.equals(map); });
922  }
923  CHECK_EQ(count, 1);
924  }
925 #endif
926 }
