33 int Mapping<T, s>::get(uchar c, uchar n, uchar* result) {
34  CacheEntry entry = entries_[c & kMask];
35  if (entry.code_point_ == c) {

49  bool allow_caching = true;
50  int length = T::Convert(c, n, result, &allow_caching);
51  if (allow_caching) {
52  if (length == 1) {
53  entries_[c & kMask] = CacheEntry(c, result[0] - c);

55  } else {

59  } else {

70 // Decodes UTF-8 bytes incrementally, allowing the decoding of bytes as they
71 // stream in. This **must** be followed by a call to ValueOfIncrementalFinish
72 // when the stream is complete, to ensure incomplete sequences are handled.
73 uchar Utf8::ValueOfIncremental(const uint8_t** cursor, State* state,
74  Utf8IncrementalBuffer* buffer) {
75  DCHECK_NOT_NULL(buffer);
76  State old_state = *state;
77  uint8_t next = **cursor;
78  *cursor += 1;
79 
80  if (V8_LIKELY(next <= kMaxOneByteChar && old_state == State::kAccept)) {
81  DCHECK_EQ(0u, *buffer);
82  return static_cast<uchar>(next);
83  }
84 
85  // So we're at the lead byte of a 2/3/4 sequence, or we're at a continuation
86  // char in that sequence.
87  Utf8DfaDecoder::Decode(next, state, buffer);
88 
89  switch (*state) {
90  case State::kAccept: {
91  uchar t = *buffer;
92  *buffer = 0;
93  return t;
94  }
95 
96  case State::kReject:
97  *state = State::kAccept;
98  *buffer = 0;
99 
100  // If we hit a bad byte, we need to determine if we were trying to start
101  // a sequence or continue one. If we were trying to start a sequence,
102  // that means it's just an invalid lead byte and we need to continue to
103  // the next (which we already did above). If we were already in a
104  // sequence, we need to reprocess this same byte after resetting to the
105  // initial state.
106  if (old_state != State::kAccept) {
107  // We were trying to continue a sequence, so let's reprocess this byte
108  // next time.
109  *cursor -= 1;
110  }
111  return kBadChar;
112 
113  default:
114  return kIncomplete;
115  }
116 }

118 unsigned Utf8::EncodeOneByte(char* str, uint8_t c) {
119  static const int kMask = ~(1 << 6);
120  if (c <= kMaxOneByteChar) {
121  str[0] = c;
122  return 1;
123  } else {
124  str[0] = 0xC0 | (c >> 6);
125  str[1] = 0x80 | (c & kMask);
126  return 2;

130 // Encode encodes the UTF-16 code units c and previous into the given str
131 // buffer, and combines surrogate code units into single code points. If
132 // replace_invalid is set to true, orphan surrogate code units will be replaced
133 // with kBadChar.
134 unsigned Utf8::Encode(char* str, uchar c, int previous, bool replace_invalid) {
135  static const int kMask = ~(1 << 6);
136  if (c <= kMaxOneByteChar) {
137  str[0] = c;
138  return 1;
139  } else if (c <= kMaxTwoByteChar) {
140  str[0] = 0xC0 | (c >> 6);
141  str[1] = 0x80 | (c & kMask);
142  return 2;
143  } else if (c <= kMaxThreeByteChar) {
144  DCHECK(!Utf16::IsLeadSurrogate(Utf16::kNoPreviousCharacter));
145  if (Utf16::IsSurrogatePair(previous, c)) {
146  const int kUnmatchedSize = kSizeOfUnmatchedSurrogate;
147  return Encode(str - kUnmatchedSize,
148  Utf16::CombineSurrogatePair(previous, c),
149  Utf16::kNoPreviousCharacter, replace_invalid) -
150  kUnmatchedSize;
151  } else if (replace_invalid &&
152  (Utf16::IsLeadSurrogate(c) || Utf16::IsTrailSurrogate(c))) {
153  c = kBadChar;
154  }
155  str[0] = 0xE0 | (c >> 12);
156  str[1] = 0x80 | ((c >> 6) & kMask);
157  str[2] = 0x80 | (c & kMask);
158  return 3;
159  } else {
160  str[0] = 0xF0 | (c >> 18);
161  str[1] = 0x80 | ((c >> 12) & kMask);
162  str[2] = 0x80 | ((c >> 6) & kMask);
163  str[3] = 0x80 | (c & kMask);
164  return 4;
165  }
166 }

168 uchar Utf8::ValueOf(const uint8_t* bytes, size_t length, size_t* cursor) {
169  if (length == 0) return kBadChar;
170  uint8_t first = bytes[0];
171  // Characters between 0000 and 007F are encoded as a single character
172  if (V8_LIKELY(first <= kMaxOneByteChar)) {
173  *cursor += 1;
174  return first;
175  }
176  return CalculateValue(bytes, length, cursor);
177 }

187 unsigned Utf8::Length(uchar c, int previous) {
188  if (c <= kMaxOneByteChar) {
189  return 1;

191  return 2;

193  DCHECK(!Utf16::IsLeadSurrogate(Utf16::kNoPreviousCharacter));

203 bool Utf8::IsValidCharacter(uchar c) {
204  return c < 0xD800u || (c >= 0xE000u && c < 0xFDD0u) ||
205  (c > 0xFDEFu && c <= 0x10FFFFu && (c & 0xFFFEu) != 0xFFFEu &&
206  c != kBadChar);

209 template <typename Char>
210 Utf8::EncodingResult Utf8::Encode(v8::base::Vector<const Char> string,
211  char* buffer, size_t capacity,
212  bool write_null, bool replace_invalid_utf8) {
213  constexpr bool kSourceIsOneByte = sizeof(Char) == 1;
214 
215  if constexpr (kSourceIsOneByte) {
216  // Only 16-bit characters can contain invalid unicode.
217  replace_invalid_utf8 = false;
218  }
219 
220  size_t write_index = 0;
221  const Char* characters = string.begin();
222  size_t content_capacity = capacity - write_null;
223  CHECK_LE(content_capacity, capacity);
224  uint16_t last = Utf16::kNoPreviousCharacter;
225  size_t read_index = 0;
226  for (; read_index < string.size(); read_index++) {
227  Char character = characters[read_index];
228 
229  size_t required_capacity;
230  if constexpr (kSourceIsOneByte) {
231  required_capacity = Utf8::LengthOneByte(character);
232  } else {
233  required_capacity = Utf8::Length(character, last);
234  }
235  size_t remaining_capacity = content_capacity - write_index;
236  if (remaining_capacity < required_capacity) {
237  // Not enough space left, so stop here.
238  if (Utf16::IsSurrogatePair(last, character)) {
239  DCHECK_GE(write_index, Utf8::kSizeOfUnmatchedSurrogate);
240  // We're in the middle of a surrogate pair. Delete the first part again.
241  write_index -= Utf8::kSizeOfUnmatchedSurrogate;
242  // We've already read at least one character which is a lead surrogate
243  DCHECK_NE(read_index, 0);
244  --read_index;
245  }
246  break;
247  }
248 
249  if constexpr (kSourceIsOneByte) {
250  write_index += Utf8::EncodeOneByte(buffer + write_index, character);
251  } else {
252  // Handle the case where we cut off in the middle of a surrogate pair.
253  if ((read_index + 1 < string.size()) &&
254  Utf16::IsSurrogatePair(character, characters[read_index + 1])) {
255  write_index += Utf8::kSizeOfUnmatchedSurrogate;
256  } else {
257  write_index += Utf8::Encode(buffer + write_index, character, last,
258  replace_invalid_utf8);
259  }
260  }
261 
262  last = character;
263  }
264  DCHECK_LE(write_index, capacity);
265 
266  if (write_null) {
267  DCHECK_LT(write_index, capacity);
268  buffer[write_index++] = '\0';
269  }
270 
271  size_t bytes_written = write_index;
272  size_t characters_processed = read_index;
273  return {bytes_written, characters_processed};
274 }
