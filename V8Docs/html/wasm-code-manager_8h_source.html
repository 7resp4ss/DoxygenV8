268  void IncRef() {
269  [[maybe_unused]] uint32_t old_field =
270  ref_count_bitfield_.fetch_add(1, std::memory_order_acq_rel);
271  DCHECK_LE(1, refcount(old_field));
272  DCHECK_GT(kMaxInt, refcount(old_field));
273  }
274 
275  // Returns true if the refcount was incremented, false if {this->is_dying()}.
276  bool IncRefIfNotDying() {
277  uint32_t old_field = ref_count_bitfield_.load(std::memory_order_acquire);
278  while (true) {
279  if (is_dying(old_field)) return false;
280  if (ref_count_bitfield_.compare_exchange_weak(
281  old_field, old_field + 1, std::memory_order_acq_rel)) {
282  return true;
283  }
284  }
285  }
286 
287  // Decrement the ref count. Returns whether this code becomes dead and needs
288  // to be freed.
289  V8_WARN_UNUSED_RESULT bool DecRef() {
290  uint32_t old_field = ref_count_bitfield_.load(std::memory_order_acquire);
291  while (true) {
292  DCHECK_LE(1, refcount(old_field));
293  if (V8_UNLIKELY(refcount(old_field) == 1)) {
294  if (is_dying(old_field)) {
295  // The code was already on the path to deletion, only temporary
296  // C++ references to it are left. Decrement the refcount, and
297  // return true if it drops to zero.
298  return DecRefOnDeadCode();
299  }
300  // Otherwise, the code enters the path to destruction now.
301  if (ref_count_bitfield_.compare_exchange_weak(
302  old_field, old_field | kIsDyingMask,
303  std::memory_order_acq_rel)) {
304  // No other thread got in the way. Commit to the decision.
305  DecRefOnPotentiallyDeadCode();
306  return false;
307  }
308  // Another thread interfered. Re-evaluate what to do.
309  continue;
310  }
311  DCHECK_LT(1, refcount(old_field));
312  if (ref_count_bitfield_.compare_exchange_weak(
313  old_field, old_field - 1, std::memory_order_acq_rel)) {
314  return false;
315  }
316  }
317  }
318 
319  // Decrement the ref count on code that is known to be in use (i.e. the ref
320  // count cannot drop to zero here).
321  void DecRefOnLiveCode() {
322  [[maybe_unused]] uint32_t old_bitfield_value =
323  ref_count_bitfield_.fetch_sub(1, std::memory_order_acq_rel);
324  DCHECK_LE(2, refcount(old_bitfield_value));
325  }

327  // Decrement the ref count on code that is known to be dead, even though there
328  // might still be C++ references. Returns whether this drops the last
329  // reference and the code needs to be freed.
330  V8_WARN_UNUSED_RESULT bool DecRefOnDeadCode() {
331  uint32_t old_bitfield_value =
332  ref_count_bitfield_.fetch_sub(1, std::memory_order_acq_rel);
333  return refcount(old_bitfield_value) == 1;
334  }

507  // 3) The set of potentially dead code in the {WasmEngine}.
508  // If a decrement of (1) would drop the ref count to 0, that code becomes a

510  // *before* decrementing the counter to ensure the code stays alive as long as
511  // it's being used. Once the ref count drops to zero (i.e. after being removed

513  // machine code is freed.

911  // This function tries to set the fast API call target of function import
912  // `index`. If the call target has been set before with a different value,
913  // then this function returns false, and this import will be marked as not
914  // suitable for wellknown imports, i.e. all existing compiled code of the
915  // module gets flushed, and future calls to this import will not use fast API
916  // calls.
917  bool TrySetFastApiCallTarget(int func_index, Address target) {
918  Address old_val =
919  fast_api_targets_[func_index].load(std::memory_order_relaxed);
920  if (old_val == target) {
921  return true;
922  }
923  if (old_val != kNullAddress) {
924  // If already a different target is stored, then there are conflicting
925  // targets and fast api calls are not possible. In that case the import
926  // will be marked as not suitable for wellknown imports, and the
927  // `fast_api_target` of this import will never be used anymore in the
928  // future.
929  return false;
930  }
931  if (fast_api_targets_[func_index].compare_exchange_strong(
932  old_val, target, std::memory_order_relaxed)) {
933  return true;
934  }
935  // If a concurrent call to `TrySetFastAPICallTarget` set the call target to
936  // the same value as this call, we consider also this call successful.
937  return old_val == target;
938  }
