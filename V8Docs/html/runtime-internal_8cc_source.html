99 Tagged<Object> NewError(Isolate* isolate, RuntimeArguments args,
100  Handle<JSFunction> (Isolate::*constructor_fn)()) {
101  HandleScope scope(isolate);
102  DCHECK_LE(1, args.length());
103  int message_id_smi = args.smi_value_at(0);
104 
105  constexpr int kMaxMessageArgs = 3;
106  DirectHandle<Object> message_args[kMaxMessageArgs];
107  int num_message_args = 0;
108  while (num_message_args < kMaxMessageArgs &&
109  args.length() > num_message_args + 1) {
110  message_args[num_message_args] = args.at(num_message_args + 1);
111  ++num_message_args;
112  }
113 
114  MessageTemplate message_id = MessageTemplateFromInt(message_id_smi);
115 
116  return *isolate->factory()->NewError(
117  (isolate->*constructor_fn)(), message_id,
118  base::VectorOf(message_args, num_message_args));
119 }

128 RUNTIME_FUNCTION(Runtime_ThrowRangeError) {
129  if (v8_flags.correctness_fuzzer_suppressions) {
130  DCHECK_LE(1, args.length());
131  int message_id_smi = args.smi_value_at(0);
132 
133  // If the result of a BigInt computation is truncated to 64 bit, Turbofan
134  // can sometimes truncate intermediate results already, which can prevent
135  // those from exceeding the maximum length, effectively preventing a
136  // RangeError from being thrown. As this is a performance optimization, this
137  // behavior is accepted. To prevent the correctness fuzzer from detecting
138  // this difference, we crash the program.
139  if (MessageTemplateFromInt(message_id_smi) ==
140  MessageTemplate::kBigIntTooBig) {
141  FATAL("Aborting on invalid BigInt length");
142  }
143  }
144 
145  return ThrowError(isolate, args, &Isolate::range_error_function);
146 }

373 Tagged<Object> BytecodeBudgetInterruptWithStackCheck(Isolate* isolate,
374  RuntimeArguments& args,
375  CodeKind code_kind) {
376  HandleScope scope(isolate);
377  DCHECK_EQ(1, args.length());
378  DirectHandle<JSFunction> function = args.at<JSFunction>(0);
379  TRACE_EVENT0("v8.execute", "V8.BytecodeBudgetInterruptWithStackCheck");
380 
381  // Check for stack interrupts here so that we can fold the interrupt check
382  // into bytecode budget interrupts.
383  StackLimitCheck check(isolate);
384  if (check.JsHasOverflowed()) {
385  // We ideally wouldn't actually get StackOverflows here, since we stack
386  // check on bytecode entry, but it's possible that this check fires due to
387  // the runtime function call being what overflows the stack.
388  return isolate->StackOverflow();
389  } else if (check.InterruptRequested()) {
390  Tagged<Object> return_value = isolate->stack_guard()->HandleInterrupts();
391  if (!IsUndefined(return_value, isolate)) {
392  return return_value;
393  }
394  }
395 
396  isolate->tiering_manager()->OnInterruptTick(function, code_kind);
397  return ReadOnlyRoots(isolate).undefined_value();
398 }
