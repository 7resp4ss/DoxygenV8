483 // The comparator is passed two indices |a| and |b|, and it returns < 0 when the
484 // property at index |a| comes before the property at index |b| in the
485 // enumeration order.
486 template <typename Dictionary>
487 struct EnumIndexComparator {
488  explicit EnumIndexComparator(Tagged<Dictionary> dict) : dict(dict) {}
489  bool operator()(Tagged_t a, Tagged_t b) {
490  PropertyDetails details_a(dict->DetailsAt(
491  InternalIndex(Tagged<Smi>(static_cast<Address>(a)).value())));
492  PropertyDetails details_b(dict->DetailsAt(
493  InternalIndex(Tagged<Smi>(static_cast<Address>(b)).value())));
494  return details_a.dictionary_index() < details_b.dictionary_index();
495  }
496  Tagged<Dictionary> dict;
497 };
