75 /**
76  * Various helpers for skipping over V8 frames in a given stack.
77  *
78  * The unwinder API is only supported on the x64, ARM64 and ARM32 architectures.
79  */
80 class V8_EXPORT Unwinder {
81  public:
82  /**
83  * Attempt to unwind the stack to the most recent C++ frame. This function is
84  * signal-safe and does not access any V8 state and thus doesn't require an
85  * Isolate.
86  *
87  * The unwinder needs to know the location of the JS Entry Stub (a piece of
88  * code that is run when C++ code calls into generated JS code). This is used
89  * for edge cases where the current frame is being constructed or torn down
90  * when the stack sample occurs.
91  *
92  * The unwinder also needs the virtual memory range of all possible V8 code
93  * objects. There are two ranges required - the heap code range and the range
94  * for code embedded in the binary.
95  *
96  * Available on x64, ARM64 and ARM32.
97  *
98  * \param code_pages A list of all of the ranges in which V8 has allocated
99  * executable code. The caller should obtain this list by calling
100  * Isolate::CopyCodePages() during the same interrupt/thread suspension that
101  * captures the stack.
102  * \param register_state The current registers. This is an in-out param that
103  * will be overwritten with the register values after unwinding, on success.
104  * \param stack_base The resulting stack pointer and frame pointer values are
105  * bounds-checked against the stack_base and the original stack pointer value
106  * to ensure that they are valid locations in the given stack. If these values
107  * or any intermediate frame pointer values used during unwinding are ever out
108  * of these bounds, unwinding will fail.
109  *
110  * \return True on success.
111  */
112  static bool TryUnwindV8Frames(const JSEntryStubs& entry_stubs,
113  size_t code_pages_length,
114  const MemoryRange* code_pages,
115  RegisterState* register_state,
116  const void* stack_base);
117 
118  /**
119  * Whether the PC is within the V8 code range represented by code_pages.
120  *
121  * If this returns false, then calling UnwindV8Frames() with the same PC
122  * and unwind_state will always fail. If it returns true, then unwinding may
123  * (but not necessarily) be successful.
124  *
125  * Available on x64, ARM64 and ARM32
126  */
127  static bool PCIsInV8(size_t code_pages_length, const MemoryRange* code_pages,
128  void* pc);
129 };
