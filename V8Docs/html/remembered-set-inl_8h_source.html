18 template <typename Callback>
19 SlotCallbackResult UpdateTypedSlotHelper::UpdateTypedSlot(
20  WritableJitAllocation& jit_allocation, Heap* heap, SlotType slot_type,
21  Address addr, Callback callback) {
22  switch (slot_type) {
23  case SlotType::kCodeEntry: {
24  WritableRelocInfo rinfo(jit_allocation, addr, RelocInfo::CODE_TARGET);
25  return UpdateCodeTarget(&rinfo, callback);
26  }
27  case SlotType::kConstPoolCodeEntry: {
28  return UpdateCodeEntry(addr, callback);
29  }
30  case SlotType::kEmbeddedObjectCompressed: {
31  WritableRelocInfo rinfo(jit_allocation, addr,
32  RelocInfo::COMPRESSED_EMBEDDED_OBJECT);
33  return UpdateEmbeddedPointer(heap, &rinfo, callback);
34  }
35  case SlotType::kEmbeddedObjectFull: {
36  WritableRelocInfo rinfo(jit_allocation, addr,
37  RelocInfo::FULL_EMBEDDED_OBJECT);
38  return UpdateEmbeddedPointer(heap, &rinfo, callback);
39  }
40  case SlotType::kConstPoolEmbeddedObjectCompressed: {
41  Tagged<HeapObject> old_target = Cast<HeapObject>(
42  Tagged<Object>(V8HeapCompressionScheme::DecompressTagged(
43  base::Memory<Tagged_t>(addr))));
44  Tagged<HeapObject> new_target = old_target;
45  SlotCallbackResult result = callback(FullMaybeObjectSlot(&new_target));
46  DCHECK(!HasWeakHeapObjectTag(new_target));
47  if (new_target != old_target) {
48  jit_allocation.WriteValue<Tagged_t>(
49  addr, V8HeapCompressionScheme::CompressObject(new_target.ptr()));
50  }
51  return result;
52  }
53  case SlotType::kConstPoolEmbeddedObjectFull: {
54  Tagged<HeapObject> old_target =
55  Cast<HeapObject>(Tagged<Object>(base::Memory<Address>(addr)));
56  Tagged<HeapObject> new_target = old_target;
57  SlotCallbackResult result = callback(FullMaybeObjectSlot(&new_target));
58  if (new_target != old_target) {
59  jit_allocation.WriteValue(addr, new_target.ptr());
60  }
61  return result;
62  }
63  case SlotType::kCleared:
64  break;
65  }
66  UNREACHABLE();
67 }

69 Tagged<HeapObject> UpdateTypedSlotHelper::GetTargetObject(Heap* heap,
70  SlotType slot_type,
71  Address addr) {
72  switch (slot_type) {
73  case SlotType::kCodeEntry: {
74  RelocInfo rinfo(addr, RelocInfo::CODE_TARGET);
75  return InstructionStream::FromTargetAddress(rinfo.target_address());
76  }
77  case SlotType::kConstPoolCodeEntry: {
78  return InstructionStream::FromEntryAddress(addr);
79  }
80  case SlotType::kEmbeddedObjectCompressed: {
81  RelocInfo rinfo(addr, RelocInfo::COMPRESSED_EMBEDDED_OBJECT);
82  return rinfo.target_object(heap->isolate());
83  }
84  case SlotType::kEmbeddedObjectFull: {
85  RelocInfo rinfo(addr, RelocInfo::FULL_EMBEDDED_OBJECT);
86  return rinfo.target_object(heap->isolate());
87  }
88  case SlotType::kConstPoolEmbeddedObjectCompressed: {
89  Address full = V8HeapCompressionScheme::DecompressTagged(
90  base::Memory<Tagged_t>(addr));
91  return Cast<HeapObject>(Tagged<Object>(full));
92  }
93  case SlotType::kConstPoolEmbeddedObjectFull: {
94  FullHeapObjectSlot slot(addr);
95  return (*slot).GetHeapObjectAssumeStrong(heap->isolate());
96  }
97  case SlotType::kCleared:
98  break;
99  }
100  UNREACHABLE();
101 }
