188 AtomicGuard::AtomicGuard(AtomicMutex* atomic, bool is_blocking)
189  : atomic_(atomic), is_success_(false) {
190  do {
191  bool expected = false;
192  // We have to use the strong version here for the case where is_blocking
193  // is false, and we will only attempt the exchange once.
194  is_success_ = atomic->compare_exchange_strong(expected, true);
195  } while (is_blocking && !is_success_);
196 }
197 
198 AtomicGuard::~AtomicGuard() {
199  if (!is_success_) return;
200  atomic_->store(false);
201 }

217 void SamplerManager::AddSampler(Sampler* sampler) {
218  AtomicGuard atomic_guard(&samplers_access_counter_);
219  DCHECK(sampler->IsActive());
220  int thread_id = sampler->platform_data()->vm_tid();
221  auto it = sampler_map_.find(thread_id);
222  if (it == sampler_map_.end()) {
223  SamplerList samplers;
224  samplers.push_back(sampler);
225  sampler_map_.emplace(thread_id, std::move(samplers));
226  } else {
227  SamplerList& samplers = it->second;
228  auto sampler_it = std::find(samplers.begin(), samplers.end(), sampler);
229  if (sampler_it == samplers.end()) samplers.push_back(sampler);
230  }
231 }
232 
233 void SamplerManager::RemoveSampler(Sampler* sampler) {
234  AtomicGuard atomic_guard(&samplers_access_counter_);
235  DCHECK(sampler->IsActive());
236  int thread_id = sampler->platform_data()->vm_tid();
237  auto it = sampler_map_.find(thread_id);
238  DCHECK_NE(it, sampler_map_.end());
239  SamplerList& samplers = it->second;
240  samplers.erase(std::remove(samplers.begin(), samplers.end(), sampler),
241  samplers.end());
242  if (samplers.empty()) {
243  sampler_map_.erase(it);
244  }
245 }

247 void SamplerManager::DoSample(const v8::RegisterState& state) {
248  AtomicGuard atomic_guard(&samplers_access_counter_, false);
249  // TODO(petermarshall): Add stat counters for the bailouts here.
250  if (!atomic_guard.is_success()) return;
251  int thread_id = base::OS::GetCurrentThreadId();
252  auto it = sampler_map_.find(thread_id);
253  if (it == sampler_map_.end()) return;
254  SamplerList& samplers = it->second;
255 
256  for (Sampler* sampler : samplers) {
257  if (!sampler->ShouldRecordSample()) continue;
258  Isolate* isolate = sampler->isolate();
259  // We require a fully initialized and entered isolate.
260  if (isolate == nullptr || !isolate->IsInUse()) continue;
261  sampler->SampleStack(state);
262  }
263 }

328 class SignalHandler {
329  public:
330  static void IncreaseSamplerCount() {
331  base::RecursiveMutexGuard lock_guard(mutex_.Pointer());
332  if (++client_count_ == 1) Install();
333  }
334 
335  static void DecreaseSamplerCount() {
336  base::RecursiveMutexGuard lock_guard(mutex_.Pointer());
337  if (--client_count_ == 0) Restore();
338  }
339 
340  static bool Installed() {
341  // mutex_ will also be used in Sampler::DoSample to guard the state below.
342  base::RecursiveMutexGuard lock_guard(mutex_.Pointer());
343  return signal_handler_installed_;
344  }
345 
346  static v8::base::RecursiveMutex* mutex() { return mutex_.Pointer(); }
347 
348  private:
349  static void Install() {
350  struct sigaction sa;
351  sa.sa_sigaction = &HandleProfilerSignal;
352  sigemptyset(&sa.sa_mask);
353 #if V8_OS_QNX
354  sa.sa_flags = SA_SIGINFO | SA_ONSTACK;
355 #else
356  sa.sa_flags = SA_RESTART | SA_SIGINFO | SA_ONSTACK;
357 #endif
358  signal_handler_installed_ =
359  (sigaction(SIGPROF, &sa, &old_signal_handler_) == 0);
360  }
361 
362  static void Restore() {
363  if (signal_handler_installed_) {
364  signal_handler_installed_ = false;
365 #if V8_OS_AIX || V8_TARGET_ARCH_S390X
366  // On Aix, IBMi & zLinux SIGPROF can sometimes arrive after the
367  // default signal handler is restored, resulting in intermittent test
368  // failure when profiling is enabled (https://crbug.com/v8/12952)
369  base::OS::Sleep(base::TimeDelta::FromMicroseconds(10));
370 #endif
371  sigaction(SIGPROF, &old_signal_handler_, nullptr);
372  }
373  }

376  static void HandleProfilerSignal(int signal, siginfo_t* info, void* context);
377 
378  // Protects the process wide state below.
379  static base::LazyRecursiveMutex mutex_;
380  static int client_count_;
381  static bool signal_handler_installed_;
382  static struct sigaction old_signal_handler_;
383 };

406 void SignalHandler::FillRegisterState(void* context, RegisterState* state) {
407  // Extracting the sample from the context is extremely machine dependent.
408  ucontext_t* ucontext = reinterpret_cast<ucontext_t*>(context);
409 #if !(V8_OS_OPENBSD || V8_OS_ZOS || \
410  (V8_OS_LINUX && (V8_HOST_ARCH_S390X || V8_HOST_ARCH_PPC64)))
411  mcontext_t& mcontext = ucontext->uc_mcontext;
412 #elif V8_OS_ZOS
413  __mcontext_t_* mcontext = reinterpret_cast<__mcontext_t_*>(context);
414 #endif
415 #if V8_OS_LINUX
416 #if V8_HOST_ARCH_IA32
417  state->pc = reinterpret_cast<void*>(mcontext.gregs[REG_EIP]);
418  state->sp = reinterpret_cast<void*>(mcontext.gregs[REG_ESP]);
419  state->fp = reinterpret_cast<void*>(mcontext.gregs[REG_EBP]);
420 #elif V8_HOST_ARCH_X64
421  state->pc = reinterpret_cast<void*>(mcontext.gregs[REG_RIP]);
422  state->sp = reinterpret_cast<void*>(mcontext.gregs[REG_RSP]);
423  state->fp = reinterpret_cast<void*>(mcontext.gregs[REG_RBP]);
424 #elif V8_HOST_ARCH_ARM
425 #if V8_LIBC_GLIBC && !V8_GLIBC_PREREQ(2, 4)
426  // Old GLibc ARM versions used a gregs[] array to access the register
427  // values from mcontext_t.
428  state->pc = reinterpret_cast<void*>(mcontext.gregs[R15]);
429  state->sp = reinterpret_cast<void*>(mcontext.gregs[R13]);
430  state->fp = reinterpret_cast<void*>(mcontext.gregs[R11]);
431  state->lr = reinterpret_cast<void*>(mcontext.gregs[R14]);
432 #else
433  state->pc = reinterpret_cast<void*>(mcontext.arm_pc);
434  state->sp = reinterpret_cast<void*>(mcontext.arm_sp);
435  state->fp = reinterpret_cast<void*>(mcontext.arm_fp);
436  state->lr = reinterpret_cast<void*>(mcontext.arm_lr);
437 #endif // V8_LIBC_GLIBC && !V8_GLIBC_PREREQ(2, 4)
438 #elif V8_HOST_ARCH_ARM64
439  state->pc = reinterpret_cast<void*>(mcontext.pc);
440  state->sp = reinterpret_cast<void*>(mcontext.sp);
441  // FP is an alias for x29.
442  state->fp = reinterpret_cast<void*>(mcontext.regs[29]);
443  // LR is an alias for x30.
444  state->lr = reinterpret_cast<void*>(mcontext.regs[30]);
445 #elif V8_HOST_ARCH_MIPS64
446  state->pc = reinterpret_cast<void*>(mcontext.pc);
447  state->sp = reinterpret_cast<void*>(mcontext.gregs[29]);
448  state->fp = reinterpret_cast<void*>(mcontext.gregs[30]);
449 #elif V8_HOST_ARCH_LOONG64
450  state->pc = reinterpret_cast<void*>(mcontext.__pc);
451  state->sp = reinterpret_cast<void*>(mcontext.__gregs[3]);
452  state->fp = reinterpret_cast<void*>(mcontext.__gregs[22]);
453 #elif V8_HOST_ARCH_PPC64
454 #if V8_LIBC_GLIBC
455  state->pc = reinterpret_cast<void*>(ucontext->uc_mcontext.regs->nip);
456  state->sp = reinterpret_cast<void*>(ucontext->uc_mcontext.regs->gpr[PT_R1]);
457  state->fp = reinterpret_cast<void*>(ucontext->uc_mcontext.regs->gpr[PT_R31]);
458  state->lr = reinterpret_cast<void*>(ucontext->uc_mcontext.regs->link);
459 #else
460  // Some C libraries, notably Musl, define the regs member as a void pointer
461  state->pc = reinterpret_cast<void*>(ucontext->uc_mcontext.gp_regs[32]);
462  state->sp = reinterpret_cast<void*>(ucontext->uc_mcontext.gp_regs[1]);
463  state->fp = reinterpret_cast<void*>(ucontext->uc_mcontext.gp_regs[31]);
464  state->lr = reinterpret_cast<void*>(ucontext->uc_mcontext.gp_regs[36]);
465 #endif
466 #elif V8_HOST_ARCH_S390X
467  state->pc = reinterpret_cast<void*>(ucontext->uc_mcontext.psw.addr);
468  state->sp = reinterpret_cast<void*>(ucontext->uc_mcontext.gregs[15]);
469  state->fp = reinterpret_cast<void*>(ucontext->uc_mcontext.gregs[11]);
470  state->lr = reinterpret_cast<void*>(ucontext->uc_mcontext.gregs[14]);
471 #elif V8_HOST_ARCH_RISCV64 || V8_HOST_ARCH_RISCV32
472  // Spec CH.25 RISC-V Assembly Programmers Handbook
473  state->pc = reinterpret_cast<void*>(mcontext.__gregs[REG_PC]);
474  state->sp = reinterpret_cast<void*>(mcontext.__gregs[REG_SP]);
475  state->fp = reinterpret_cast<void*>(mcontext.__gregs[REG_S0]);
476  state->lr = reinterpret_cast<void*>(mcontext.__gregs[REG_RA]);
477 #endif // V8_HOST_ARCH_*
478 
479 #elif V8_OS_ZOS
480  state->pc = reinterpret_cast<void*>(mcontext->__mc_psw);
481  state->sp = reinterpret_cast<void*>(mcontext->__mc_gr[15]);
482  state->fp = reinterpret_cast<void*>(mcontext->__mc_gr[11]);
483  state->lr = reinterpret_cast<void*>(mcontext->__mc_gr[14]);
484 #elif V8_OS_IOS
485 
486 #if V8_TARGET_ARCH_ARM64

488  state->pc = reinterpret_cast<void*>(mcontext->__ss.__pc);
489  state->sp = reinterpret_cast<void*>(mcontext->__ss.__sp);
490  state->fp = reinterpret_cast<void*>(mcontext->__ss.__fp);
491 #elif V8_TARGET_ARCH_X64
492  // Building for the iOS simulator.

494  state->sp = reinterpret_cast<void*>(mcontext->__ss.__rsp);
495  state->fp = reinterpret_cast<void*>(mcontext->__ss.__rbp);

497 #error Unexpected iOS target architecture.
498 #endif // V8_TARGET_ARCH_ARM64
499 
500 #elif V8_OS_DARWIN
501 #if V8_HOST_ARCH_X64

503  state->sp = reinterpret_cast<void*>(mcontext->__ss.__rsp);
504  state->fp = reinterpret_cast<void*>(mcontext->__ss.__rbp);
505 #elif V8_HOST_ARCH_IA32
506  state->pc = reinterpret_cast<void*>(mcontext->__ss.__eip);
507  state->sp = reinterpret_cast<void*>(mcontext->__ss.__esp);

613 void Sampler::DoSample() {
614  HANDLE profiled_thread = platform_data()->profiled_thread();
615  if (profiled_thread == nullptr) return;
616 
617  const DWORD kSuspendFailed = static_cast<DWORD>(-1);
618  if (SuspendThread(profiled_thread) == kSuspendFailed) return;
619 
620  // Context used for sampling the register state of the profiled thread.
621  CONTEXT context;
622  memset(&context, 0, sizeof(context));
623  context.ContextFlags = CONTEXT_FULL;
624  if (GetThreadContext(profiled_thread, &context) != 0) {
625  v8::RegisterState state;
626 #if V8_HOST_ARCH_X64
627  state.pc = reinterpret_cast<void*>(context.Rip);
628  state.sp = reinterpret_cast<void*>(context.Rsp);
629  state.fp = reinterpret_cast<void*>(context.Rbp);
630 #elif V8_HOST_ARCH_ARM64
631  state.pc = reinterpret_cast<void*>(context.Pc);
632  state.sp = reinterpret_cast<void*>(context.Sp);
633  state.fp = reinterpret_cast<void*>(context.Fp);
634 #else
635  state.pc = reinterpret_cast<void*>(context.Eip);
636  state.sp = reinterpret_cast<void*>(context.Esp);
637  state.fp = reinterpret_cast<void*>(context.Ebp);
638 #endif
639  SampleStack(state);
640  }
641  ResumeThread(profiled_thread);
642 }

646 void Sampler::DoSample() {
647  zx_handle_t profiled_thread = platform_data()->profiled_thread();
648  if (profiled_thread == ZX_HANDLE_INVALID) return;
649 
650  zx_handle_t suspend_token = ZX_HANDLE_INVALID;
651  if (zx_task_suspend_token(profiled_thread, &suspend_token) != ZX_OK) return;
652 
653  // Wait for the target thread to become suspended, or to exit.
654  // TODO(wez): There is currently no suspension count for threads, so there
655  // is a risk that some other caller resumes the thread in-between our suspend
656  // and wait calls, causing us to miss the SUSPENDED signal. We apply a 100ms
657  // deadline to protect against hanging the sampler thread in this case.
658  zx_signals_t signals = 0;
659  zx_status_t suspended = zx_object_wait_one(
660  profiled_thread, ZX_THREAD_SUSPENDED | ZX_THREAD_TERMINATED,
661  zx_deadline_after(ZX_MSEC(100)), &signals);
662  if (suspended != ZX_OK || (signals & ZX_THREAD_SUSPENDED) == 0) {
663  zx_handle_close(suspend_token);
664  return;
665  }
666 
667  // Fetch a copy of its "general register" states.
668  zx_thread_state_general_regs_t thread_state = {};
669  if (zx_thread_read_state(profiled_thread, ZX_THREAD_STATE_GENERAL_REGS,
670  &thread_state, sizeof(thread_state)) == ZX_OK) {
671  v8::RegisterState state;
672 #if V8_HOST_ARCH_X64
673  state.pc = reinterpret_cast<void*>(thread_state.rip);
674  state.sp = reinterpret_cast<void*>(thread_state.rsp);
675  state.fp = reinterpret_cast<void*>(thread_state.rbp);
676 #elif V8_HOST_ARCH_ARM64
677  state.pc = reinterpret_cast<void*>(thread_state.pc);
678  state.sp = reinterpret_cast<void*>(thread_state.sp);
679  state.fp = reinterpret_cast<void*>(thread_state.r[29]);
680 #endif
681  SampleStack(state);
682  }
683 
684  zx_handle_close(suspend_token);
685 }

689 #undef ZX_THREAD_STATE_GENERAL_REGS
690 #endif
