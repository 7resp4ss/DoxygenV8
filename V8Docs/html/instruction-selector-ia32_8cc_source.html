83 // Copied from x64, dropped kWord64 constant support.
84 bool MatchScaledIndex(InstructionSelector* selector, OpIndex node,
85  OpIndex* index, int* scale, bool* power_of_two_plus_one) {
86  DCHECK_NOT_NULL(index);
87  DCHECK_NOT_NULL(scale);
88 
89  auto MatchScaleConstant = [](const Operation& op, int& scale,
90  bool* plus_one) {
91  const ConstantOp* constant = op.TryCast<ConstantOp>();
92  if (constant == nullptr) return false;
93  if (constant->kind != ConstantOp::Kind::kWord32) return false;
94 
95  uint64_t value = constant->integral();
96  if (plus_one) *plus_one = false;
97  if (value == 1) return (scale = 0), true;
98  if (value == 2) return (scale = 1), true;
99  if (value == 4) return (scale = 2), true;
100  if (value == 8) return (scale = 3), true;
101  if (plus_one == nullptr) return false;
102  *plus_one = true;
103  if (value == 3) return (scale = 1), true;
104  if (value == 5) return (scale = 2), true;
105  if (value == 9) return (scale = 3), true;
106  return false;
107  };
108 
109  const Operation& op = selector->Get(node);
110  if (const WordBinopOp* binop = op.TryCast<WordBinopOp>()) {
111  if (binop->kind != WordBinopOp::Kind::kMul) return false;
112  if (MatchScaleConstant(selector->Get(binop->right()), *scale,
113  power_of_two_plus_one)) {
114  *index = binop->left();
115  return true;
116  }
117  if (MatchScaleConstant(selector->Get(binop->left()), *scale,
118  power_of_two_plus_one)) {
119  *index = binop->right();
120  return true;
121  }
122  return false;
123  } else if (const ShiftOp* shift = op.TryCast<ShiftOp>()) {
124  if (shift->kind != ShiftOp::Kind::kShiftLeft) return false;
125  int32_t scale_value;
126  if (selector->MatchIntegralWord32Constant(shift->right(), &scale_value)) {
127  if (scale_value < 0 || scale_value > 3) return false;
128  *index = shift->left();
129  *scale = static_cast<int>(scale_value);
130  if (power_of_two_plus_one) *power_of_two_plus_one = false;
131  return true;
132  }
133  }
134  return false;
135 }

137 std::optional<ScaledIndexMatch> TryMatchScaledIndex(
138  InstructionSelector* selector, OpIndex node,
139  bool allow_power_of_two_plus_one) {
140  ScaledIndexMatch match;
141  bool plus_one = false;
142  if (MatchScaledIndex(selector, node, &match.index, &match.scale,
143  allow_power_of_two_plus_one ? &plus_one : nullptr)) {
144  match.base = plus_one ? match.index : OpIndex{};
145  return match;
146  }
147  return std::nullopt;
148 }

150 // Copied verbatim from x64 (just renamed).
151 std::optional<BaseWithScaledIndexAndDisplacementMatch>
152 TryMatchBaseWithScaledIndexAndDisplacementForWordBinop(
153  InstructionSelector* selector, OpIndex left, OpIndex right) {
154  BaseWithScaledIndexAndDisplacementMatch result;
155  result.displacement_mode = kPositiveDisplacement;
156 
157  auto OwnedByAddressingOperand = [](OpIndex) {
158  // TODO(nicohartmann@): Consider providing this. For now we just allow
159  // everything to be covered regardless of other uses.
160  return true;
161  };
162 
163  // Check (S + ...)
164  if (MatchScaledIndex(selector, left, &result.index, &result.scale, nullptr) &&
165  OwnedByAddressingOperand(left)) {
166  result.displacement_mode = kPositiveDisplacement;
167 
168  // Check (S + (... binop ...))
169  if (const WordBinopOp* right_binop =
170  selector->Get(right).TryCast<WordBinopOp>()) {
171  // Check (S + (B - D))
172  if (right_binop->kind == WordBinopOp::Kind::kSub &&
173  OwnedByAddressingOperand(right)) {
174  if (!selector->MatchIntegralWord32Constant(right_binop->right(),
175  &result.displacement)) {
176  return std::nullopt;
177  }
178  result.base = right_binop->left();
179  result.displacement_mode = kNegativeDisplacement;
180  return result;
181  }
182  // Check (S + (... + ...))
183  if (right_binop->kind == WordBinopOp::Kind::kAdd &&
184  OwnedByAddressingOperand(right)) {
185  if (selector->MatchIntegralWord32Constant(right_binop->right(),
186  &result.displacement)) {
187  // (S + (B + D))
188  result.base = right_binop->left();
189  } else if (selector->MatchIntegralWord32Constant(
190  right_binop->left(), &result.displacement)) {
191  // (S + (D + B))
192  result.base = right_binop->right();
193  } else {
194  // Treat it as (S + B)
195  result.base = right;
196  result.displacement = 0;
197  }
198  return result;
199  }
200  }
201 
202  // Check (S + D)
203  if (selector->MatchIntegralWord32Constant(right, &result.displacement)) {
204  result.base = OpIndex{};
205  return result;
206  }
207 
208  // Treat it as (S + B)
209  result.base = right;
210  result.displacement = 0;
211  return result;
212  }
213 
214  // Check ((... + ...) + ...)
215  if (const WordBinopOp* left_add = selector->Get(left).TryCast<WordBinopOp>();
216  left_add && left_add->kind == WordBinopOp::Kind::kAdd &&
217  OwnedByAddressingOperand(left)) {
218  // Check ((S + ...) + ...)
219  if (MatchScaledIndex(selector, left_add->left(), &result.index,
220  &result.scale, nullptr)) {
221  result.displacement_mode = kPositiveDisplacement;
222  // Check ((S + D) + B)
223  if (selector->MatchIntegralWord32Constant(left_add->right(),
224  &result.displacement)) {
225  result.base = right;
226  return result;
227  }
228  // Check ((S + B) + D)
229  if (selector->MatchIntegralWord32Constant(right, &result.displacement)) {
230  result.base = left_add->right();
231  return result;
232  }
233  // Treat it as (B + B) and use index as right B.
234  result.base = left;
235  result.index = right;
236  result.scale = 0;
237  DCHECK_EQ(result.displacement, 0);
238  return result;
239  }
240  }
241 
242  DCHECK_EQ(result.index, OpIndex{});
243  DCHECK_EQ(result.scale, 0);
244  result.displacement_mode = kPositiveDisplacement;
245 
246  // Check (B + D)
247  if (selector->MatchIntegralWord32Constant(right, &result.displacement)) {
248  result.base = left;
249  return result;
250  }
251 
252  // Treat as (B + B) and use index as left B.
253  result.index = left;
254  result.base = right;
255  return result;
256 }

258 // Copied verbatim from x64 (just renamed).
259 std::optional<BaseWithScaledIndexAndDisplacementMatch>
260 TryMatchBaseWithScaledIndexAndDisplacement(InstructionSelector* selector,
261  OpIndex node) {
262  // The BaseWithIndexAndDisplacementMatcher canonicalizes the order of
263  // displacements and scale factors that are used as inputs, so instead of
264  // enumerating all possible patterns by brute force, checking for node
265  // clusters using the following templates in the following order suffices
266  // to find all of the interesting cases (S = index * scale, B = base
267  // input, D = displacement input):
268  //
269  // (S + (B + D))
270  // (S + (B + B))
271  // (S + D)
272  // (S + B)
273  // ((S + D) + B)
274  // ((S + B) + D)
275  // ((B + D) + B)
276  // ((B + B) + D)
277  // (B + D)
278  // (B + B)
279  BaseWithScaledIndexAndDisplacementMatch result;
280  result.displacement_mode = kPositiveDisplacement;
281 
282  const Operation& op = selector->Get(node);
283  if (const LoadOp* load = op.TryCast<LoadOp>()) {
284  result.base = load->base();
285  result.index = load->index().value_or_invalid();
286  result.scale = load->element_size_log2;
287  result.displacement = load->offset;
288  if (load->kind.tagged_base) result.displacement -= kHeapObjectTag;
289  return result;
290  } else if (const StoreOp* store = op.TryCast<StoreOp>()) {
291  result.base = store->base();
292  result.index = store->index().value_or_invalid();
293  result.scale = store->element_size_log2;
294  result.displacement = store->offset;
295  if (store->kind.tagged_base) result.displacement -= kHeapObjectTag;
296  return result;
297  } else if (op.Is<WordBinopOp>()) {
298  // Nothing to do here, fall into the case below.
299 #ifdef V8_ENABLE_WEBASSEMBLY
300  } else if (const Simd128LaneMemoryOp* lane_op =
301  op.TryCast<Simd128LaneMemoryOp>()) {
302  result.base = lane_op->base();
303  result.index = lane_op->index();
304  result.scale = 0;
305  result.displacement = 0;
306  if (lane_op->kind.tagged_base) result.displacement -= kHeapObjectTag;
307  return result;
308  } else if (const Simd128LoadTransformOp* load_transform =
309  op.TryCast<Simd128LoadTransformOp>()) {
310  result.base = load_transform->base();
311  result.index = load_transform->index();
312  DCHECK_EQ(load_transform->offset, 0);
313  result.scale = 0;
314  result.displacement = 0;
315  DCHECK(!load_transform->load_kind.tagged_base);
316  return result;
317 #endif // V8_ENABLE_WEBASSEMBLY
318  } else {
319  return std::nullopt;
320  }
321 
322  const WordBinopOp& binop = op.Cast<WordBinopOp>();
323  OpIndex left = binop.left();
324  OpIndex right = binop.right();
325  return TryMatchBaseWithScaledIndexAndDisplacementForWordBinop(selector, left,
326  right);
327 }

342  bool CanBeMemoryOperand(InstructionCode opcode, OpIndex node, OpIndex input,
343  int effect_level) {
344  if (!selector()->IsLoadOrLoadImmutable(input)) return false;
345  if (!selector()->CanCover(node, input)) return false;
346  if (effect_level != selector()->GetEffectLevel(input)) {
347  return false;
348  }
349  MachineRepresentation rep =
350  selector()->load_view(input).loaded_rep().representation();
351  switch (opcode) {
352  case kIA32And:
353  case kIA32Or:
354  case kIA32Xor:
355  case kIA32Add:
356  case kIA32Sub:
357  case kIA32Cmp:
358  case kIA32Test:
359  return rep == MachineRepresentation::kWord32 || IsAnyTagged(rep);
360  case kIA32Cmp16:
361  case kIA32Test16:
362  return rep == MachineRepresentation::kWord16;
363  case kIA32Cmp8:
364  case kIA32Test8:
365  return rep == MachineRepresentation::kWord8;
366  default:
367  break;
368  }
369  return false;
370  }

372  bool CanBeImmediate(OpIndex node) {
373  if (selector()->IsExternalConstant(node)) return true;
374  if (const ConstantOp* constant = Get(node).TryCast<ConstantOp>()) {
375  switch (constant->kind) {
376  case ConstantOp::Kind::kWord32:
377  case ConstantOp::Kind::kRelocatableWasmCall:
378  case ConstantOp::Kind::kRelocatableWasmStubCall:
379  case ConstantOp::Kind::kSmi:
380  return true;
381  case ConstantOp::Kind::kNumber:
382  return constant->number().get_bits() == 0;
383  default:
384  break;
385  }
386  }
387  return false;

408  AddressingMode GenerateMemoryOperandInputs(
409  OptionalOpIndex index, int scale, OpIndex base, int32_t displacement,
410  DisplacementMode displacement_mode, InstructionOperand inputs[],
411  size_t* input_count,
412  RegisterMode register_mode = RegisterMode::kRegister) {
413  AddressingMode mode = kMode_MRI;
414  if (displacement_mode == kNegativeDisplacement) {
415  displacement = base::bits::WraparoundNeg32(displacement);
416  }
417  if (base.valid()) {
418  if (const ConstantOp* constant = Get(base).TryCast<ConstantOp>()) {
419  if (constant->kind == ConstantOp::Kind::kWord32) {
420  displacement =
421  base::bits::WraparoundAdd32(displacement, constant->word32());
422  base = OpIndex{};
423  } else if (constant->kind == ConstantOp::Kind::kSmi) {
424  displacement = base::bits::WraparoundAdd32(
425  displacement, static_cast<int32_t>(constant->smi().ptr()));
426  base = OpIndex{};
427  }
428  }
429  }
430  if (base.valid()) {
431  inputs[(*input_count)++] = UseRegisterWithMode(base, register_mode);
432  if (index.valid()) {
433  DCHECK(scale >= 0 && scale <= 3);
434  inputs[(*input_count)++] =
435  UseRegisterWithMode(index.value(), register_mode);
436  if (displacement != 0) {
437  inputs[(*input_count)++] = TempImmediate(displacement);
438  static const AddressingMode kMRnI_modes[] = {kMode_MR1I, kMode_MR2I,
439  kMode_MR4I, kMode_MR8I};
440  mode = kMRnI_modes[scale];
441  } else {
442  static const AddressingMode kMRn_modes[] = {kMode_MR1, kMode_MR2,
443  kMode_MR4, kMode_MR8};
444  mode = kMRn_modes[scale];
445  }
446  } else {
447  if (displacement == 0) {
448  mode = kMode_MR;
449  } else {
450  inputs[(*input_count)++] = TempImmediate(displacement);
451  mode = kMode_MRI;
452  }
453  }
454  } else {
455  DCHECK(scale >= 0 && scale <= 3);
456  if (index.valid()) {
457  inputs[(*input_count)++] =
458  UseRegisterWithMode(index.value(), register_mode);
459  if (displacement != 0) {
460  inputs[(*input_count)++] = TempImmediate(displacement);
461  static const AddressingMode kMnI_modes[] = {kMode_MRI, kMode_M2I,
462  kMode_M4I, kMode_M8I};
463  mode = kMnI_modes[scale];
464  } else {
465  static const AddressingMode kMn_modes[] = {kMode_MR, kMode_M2,
466  kMode_M4, kMode_M8};
467  mode = kMn_modes[scale];
468  }
469  } else {
470  inputs[(*input_count)++] = TempImmediate(displacement);
471  return kMode_MI;
472  }
473  }
474  return mode;
475  }

477  AddressingMode GetEffectiveAddressMemoryOperand(
478  OpIndex node, InstructionOperand inputs[], size_t* input_count,
479  RegisterMode register_mode = RegisterMode::kRegister) {
480  const Operation& op = this->Get(node);
481  if (op.Is<LoadOp>() || op.Is<StoreOp>()) {
482  LoadStoreView load_or_store(op);
483  if (ExternalReference reference;
484  this->MatchExternalConstant(load_or_store.base, &reference) &&
485  !load_or_store.index.valid()) {
486  if (selector()->CanAddressRelativeToRootsRegister(reference)) {
487  const ptrdiff_t delta =
488  load_or_store.offset +
489  MacroAssemblerBase::RootRegisterOffsetForExternalReference(
490  selector()->isolate(), reference);
491  if (is_int32(delta)) {
492  inputs[(*input_count)++] =
493  TempImmediate(static_cast<int32_t>(delta));
494  return kMode_Root;
495  }
496  }
497  }
498  }
499 
500  auto m = TryMatchBaseWithScaledIndexAndDisplacement(selector(), node);
501  DCHECK(m.has_value());
502  if (m->base.valid() &&
503  this->Get(m->base).template Is<LoadRootRegisterOp>() &&
504  !m->index.valid()) {
505  DCHECK_EQ(m->scale, 0);
506  DCHECK(ValueFitsIntoImmediate(m->displacement));
507  inputs[(*input_count)++] =
508  UseImmediate(static_cast<int>(m->displacement));
509  return kMode_Root;
510  } else if (ValueFitsIntoImmediate(m->displacement)) {
511  return GenerateMemoryOperandInputs(m->index, m->scale, m->base,
512  m->displacement, m->displacement_mode,
513  inputs, input_count, register_mode);
514  } else if (!m->base.valid() &&
515  m->displacement_mode == kPositiveDisplacement) {
516  // The displacement cannot be an immediate, but we can use the
517  // displacement as base instead and still benefit from addressing
518  // modes for the scale.
519  UNIMPLEMENTED();
520  } else {
521  // TODO(nicohartmann@): Turn this into a `DCHECK` once we have some
522  // coverage.
523  CHECK_EQ(m->displacement, 0);
524  inputs[(*input_count)++] = UseRegisterWithMode(m->base, register_mode);
525  inputs[(*input_count)++] = UseRegisterWithMode(m->index, register_mode);
526  return kMode_MR1;
527  }
528  }

548 ArchOpcode GetLoadOpcode(LoadRepresentation load_rep) {
549  ArchOpcode opcode;
550  switch (load_rep.representation()) {
551  case MachineRepresentation::kFloat32:
552  opcode = kIA32Movss;
553  break;
554  case MachineRepresentation::kFloat64:
555  opcode = kIA32Movsd;
556  break;
557  case MachineRepresentation::kBit: // Fall through.
558  case MachineRepresentation::kWord8:
559  opcode = load_rep.IsSigned() ? kIA32Movsxbl : kIA32Movzxbl;
560  break;
561  case MachineRepresentation::kWord16:
562  opcode = load_rep.IsSigned() ? kIA32Movsxwl : kIA32Movzxwl;
563  break;
564  case MachineRepresentation::kTaggedSigned: // Fall through.
565  case MachineRepresentation::kTaggedPointer: // Fall through.
566  case MachineRepresentation::kTagged: // Fall through.
567  case MachineRepresentation::kWord32:
568  opcode = kIA32Movl;
569  break;
570  case MachineRepresentation::kSimd128:
571  opcode = kIA32Movdqu;
572  break;
573  case MachineRepresentation::kFloat16:
574  UNIMPLEMENTED();
575  case MachineRepresentation::kSimd256: // Fall through.
576  case MachineRepresentation::kCompressedPointer: // Fall through.
577  case MachineRepresentation::kCompressed: // Fall through.
578  case MachineRepresentation::kProtectedPointer: // Fall through.
579  case MachineRepresentation::kIndirectPointer: // Fall through.
580  case MachineRepresentation::kSandboxedPointer: // Fall through.
581  case MachineRepresentation::kWord64: // Fall through.
582  case MachineRepresentation::kMapWord: // Fall through.
583  case MachineRepresentation::kFloat16RawBits: // Fall through.
584  case MachineRepresentation::kNone:
585  UNREACHABLE();
586  }
587  return opcode;
588 }

805 void InstructionSelector::VisitLoadLane(OpIndex node) {
806  InstructionCode opcode;
807  int lane;
808  const Simd128LaneMemoryOp& load = Cast<Simd128LaneMemoryOp>(node);
809  lane = load.lane;
810  switch (load.lane_kind) {
811  case Simd128LaneMemoryOp::LaneKind::k8:
812  opcode = kIA32Pinsrb;
813  break;
814  case Simd128LaneMemoryOp::LaneKind::k16:
815  opcode = kIA32Pinsrw;
816  break;
817  case Simd128LaneMemoryOp::LaneKind::k32:
818  opcode = kIA32Pinsrd;
819  break;
820  case Simd128LaneMemoryOp::LaneKind::k64:
821  // pinsrq not available on IA32.
822  if (lane == 0) {
823  opcode = kIA32Movlps;
824  } else {
825  DCHECK_EQ(1, lane);
826  opcode = kIA32Movhps;
827  }
828  break;
829  }
830  // IA32 supports unaligned loads.
831  DCHECK(!load.kind.maybe_unaligned);
832  // Trap handler is not supported on IA32.
833  DCHECK(!load.kind.with_trap_handler);
834 
835  IA32OperandGenerator g(this);
836  InstructionOperand outputs[] = {IsSupported(AVX)
837  ? g.DefineAsRegister(node)
838  : g.DefineSameAsFirst(node)};
839  // Input 0 is value node, 1 is lane idx, and
840  // GetEffectiveAddressMemoryOperand uses up to 3 inputs. This ordering is
841  // consistent with other operations that use the same opcode.
842  InstructionOperand inputs[5];
843  size_t input_count = 0;
844 
845  inputs[input_count++] = g.UseRegister(load.value());
846  inputs[input_count++] = g.UseImmediate(lane);
847 
848  AddressingMode mode =
849  g.GetEffectiveAddressMemoryOperand(node, inputs, &input_count);
850  opcode |= AddressingModeField::encode(mode);
851 
852  DCHECK_GE(5, input_count);
853 
854  Emit(opcode, 1, outputs, input_count, inputs);

1011 void VisitStoreCommon(InstructionSelector* selector,
1012  const InstructionSelector::StoreView& store) {
1013  IA32OperandGenerator g(selector);
1014 
1015  OpIndex base = store.base();
1016  OptionalOpIndex index = store.index();
1017  OpIndex value = store.value();
1018  int32_t displacement = store.displacement();
1019  uint8_t element_size_log2 = store.element_size_log2();
1020  std::optional<AtomicMemoryOrder> atomic_order = store.memory_order();
1021  StoreRepresentation store_rep = store.stored_rep();
1022 
1023  WriteBarrierKind write_barrier_kind = store_rep.write_barrier_kind();
1024  MachineRepresentation rep = store_rep.representation();
1025  const bool is_seqcst =
1026  atomic_order && *atomic_order == AtomicMemoryOrder::kSeqCst;
1027 
1028  if (v8_flags.enable_unconditional_write_barriers && CanBeTaggedPointer(rep)) {
1029  write_barrier_kind = kFullWriteBarrier;
1030  }
1031 
1032  if (write_barrier_kind != kNoWriteBarrier &&
1033  !v8_flags.disable_write_barriers) {
1034  DCHECK(CanBeTaggedPointer(rep));
1035  AddressingMode addressing_mode;
1036  InstructionOperand inputs[4];
1037  size_t input_count = 0;
1038  addressing_mode = g.GenerateMemoryOperandInputs(
1039  index, element_size_log2, base, displacement,
1040  DisplacementMode::kPositiveDisplacement, inputs, &input_count,
1041  IA32OperandGenerator::RegisterMode::kUniqueRegister);
1042  DCHECK_LT(input_count, 4);
1043  inputs[input_count++] = g.UseUniqueRegister(value);
1044  RecordWriteMode record_write_mode =
1045  WriteBarrierKindToRecordWriteMode(write_barrier_kind);
1046  InstructionOperand temps[] = {g.TempRegister()};
1047  size_t const temp_count = arraysize(temps);
1048  InstructionCode code = is_seqcst ? kArchAtomicStoreWithWriteBarrier
1049  : kArchStoreWithWriteBarrier;
1050  code |= AddressingModeField::encode(addressing_mode);
1051  code |= RecordWriteModeField::encode(record_write_mode);
1052  selector->Emit(code, 0, nullptr, input_count, inputs, temp_count, temps);
1053  } else {
1054  InstructionOperand inputs[4];
1055  size_t input_count = 0;
1056  // To inform the register allocator that xchg clobbered its input.
1057  InstructionOperand outputs[1];
1058  size_t output_count = 0;
1059  ArchOpcode opcode;
1060  AddressingMode addressing_mode;
1061 
1062  if (is_seqcst) {
1063  // SeqCst stores emit XCHG instead of MOV, so encode the inputs as we
1064  // would for XCHG. XCHG can't encode the value as an immediate and has
1065  // fewer addressing modes available.
1066  if (rep == MachineRepresentation::kWord8 ||
1067  rep == MachineRepresentation::kBit) {
1068  inputs[input_count++] = g.UseFixed(value, edx);
1069  outputs[output_count++] = g.DefineAsFixed(store, edx);
1070  } else {
1071  inputs[input_count++] = g.UseUniqueRegister(value);
1072  outputs[output_count++] = g.DefineSameAsFirst(store);
1073  }
1074  addressing_mode = g.GetEffectiveAddressMemoryOperand(
1075  store, inputs, &input_count,
1076  IA32OperandGenerator::RegisterMode::kUniqueRegister);
1077  opcode = GetSeqCstStoreOpcode(rep);
1078  } else {
1079  // Release and non-atomic stores emit MOV.
1080  // https://www.cl.cam.ac.uk/~pes20/cpp/cpp0xmappings.html
1081  InstructionOperand val;
1082  if (g.CanBeImmediate(value)) {
1083  val = g.UseImmediate(value);
1084  } else if (!atomic_order && (rep == MachineRepresentation::kWord8 ||
1085  rep == MachineRepresentation::kBit)) {
1086  val = g.UseByteRegister(value);
1087  } else {
1088  val = g.UseUniqueRegister(value);
1089  }
1090  addressing_mode = g.GetEffectiveAddressMemoryOperand(
1091  store, inputs, &input_count,
1092  IA32OperandGenerator::RegisterMode::kUniqueRegister);
1093  inputs[input_count++] = val;
1094  opcode = GetStoreOpcode(rep);
1095  }
1096  InstructionCode code =
1097  opcode | AddressingModeField::encode(addressing_mode);
1098  selector->Emit(code, output_count, outputs, input_count, inputs);
1099  }

1165 // Shared routine for multiple binary operations.
1166 void VisitBinop(InstructionSelector* selector, OpIndex node,
1167  InstructionCode opcode, FlagsContinuation* cont) {
1168  IA32OperandGenerator g(selector);
1169 
1170  auto [left, right] = selector->Inputs<2>(node);
1171  InstructionOperand inputs[6];
1172  size_t input_count = 0;
1173  InstructionOperand outputs[1];
1174  size_t output_count = 0;
1175 
1176  // TODO(turbofan): match complex addressing modes.
1177  if (left == right) {
1178  // If both inputs refer to the same operand, enforce allocating a register
1179  // for both of them to ensure that we don't end up generating code like
1180  // this:
1181  //
1182  // mov eax, [ebp-0x10]
1183  // add eax, [ebp-0x10]
1184  // jo label
1185  InstructionOperand const input = g.UseRegister(left);
1186  inputs[input_count++] = input;
1187  inputs[input_count++] = input;
1188  } else if (g.CanBeImmediate(right)) {
1189  inputs[input_count++] = g.UseRegister(left);
1190  inputs[input_count++] = g.UseImmediate(right);
1191  } else {
1192  int effect_level = selector->GetEffectLevel(node, cont);
1193  if (selector->IsCommutative(node) && g.CanBeBetterLeftOperand(right) &&
1194  (!g.CanBeBetterLeftOperand(left) ||
1195  !g.CanBeMemoryOperand(opcode, node, right, effect_level))) {
1196  std::swap(left, right);
1197  }
1198  if (g.CanBeMemoryOperand(opcode, node, right, effect_level)) {
1199  inputs[input_count++] = g.UseRegister(left);
1200  AddressingMode addressing_mode =
1201  g.GetEffectiveAddressMemoryOperand(right, inputs, &input_count);
1202  opcode |= AddressingModeField::encode(addressing_mode);
1203  } else {
1204  inputs[input_count++] = g.UseRegister(left);
1205  inputs[input_count++] = g.Use(right);
1206  }
1207  }
1208 
1209  outputs[output_count++] = g.DefineSameAsFirst(node);
1210 
1211  DCHECK_NE(0u, input_count);
1212  DCHECK_EQ(1u, output_count);
1213  DCHECK_GE(arraysize(inputs), input_count);
1214  DCHECK_GE(arraysize(outputs), output_count);

1806 void InstructionSelector::EmitPrepareArguments(
1807  ZoneVector<PushParameter>* arguments, const CallDescriptor* call_descriptor,
1808  OpIndex node) {

1812  // Prepare for C function call.

1814  InstructionOperand temps[] = {g.TempRegister()};
1815  size_t const temp_count = arraysize(temps);
1816  Emit(kArchPrepareCallCFunction | MiscField::encode(static_cast<int>(
1817  call_descriptor->ParameterCount())),
1818  0, nullptr, 0, nullptr, temp_count, temps);
1819 
1820  // Poke any stack arguments.
1821  for (size_t n = 0; n < arguments->size(); ++n) {

1835  int effect_level = GetEffectLevel(node);

3389  // SSE can't take advantage of both operands in registers and needs

3407  no_same_as_first = true;
3408  // TODO(v8:9198): This doesn't strictly require a register, forcing the

3413  }

3431  imms[imm_count++] = blend_mask;
3432  }
3433  }
3434  } else if (wasm::SimdShuffle::TryMatch16x8Shuffle(shuffle, shuffle16x8)) {
3435  uint8_t blend_mask;
3436  if (wasm::SimdShuffle::TryMatchBlend(shuffle)) {
3437  opcode = kIA32S16x8Blend;
3438  blend_mask = wasm::SimdShuffle::PackBlend8(shuffle16x8);
3439  imms[imm_count++] = blend_mask;
3440  } else if (wasm::SimdShuffle::TryMatchSplat<8>(shuffle, &index)) {
3441  opcode = kIA32S16x8Dup;
3442  src0_needs_reg = false;
3443  imms[imm_count++] = index;

3447  no_same_as_first = true;

3452  imms[imm_count++] = mask_hi;
3453  if (!is_swizzle) imms[imm_count++] = blend_mask;

3456  opcode = kIA32S8x16Dup;
3457  no_same_as_first = use_avx;
3458  src0_needs_reg = true;
3459  imms[imm_count++] = index;
3460  }

3466  imms[imm_count++] = wasm::SimdShuffle::Pack4Lanes(shuffle + 4);
3467  imms[imm_count++] = wasm::SimdShuffle::Pack4Lanes(shuffle + 8);
3468  imms[imm_count++] = wasm::SimdShuffle::Pack4Lanes(shuffle + 12);

3473  // move instruction in the CodeGenerator.
3474  OpIndex input0 = view.input(0);
3475  InstructionOperand dst =

3477  // TODO(v8:9198): Use src0_needs_reg when we have memory alignment for SIMD.

3479  USE(src0_needs_reg);
3480

3486  // TODO(v8:9198): Use src1_needs_reg when we have memory alignment for SIMD.
3487  inputs[input_count++] = g.UseRegister(input1);
3488  USE(src1_needs_reg);
3489  }

3491  inputs[input_count++] = g.UseImmediate(imms[i]);
3492  }
