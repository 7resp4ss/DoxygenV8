12 // static
13 DirectHandle<WeakFixedArray> WasmExportWrapperCache::New(Isolate* isolate,
14  int capacity) {
15  // Probing depends on bit-masking the hashes, so the capacity needs to be
16  // a power of two.
17  DCHECK(base::bits::IsPowerOfTwo(capacity));
18  capacity = capacity * kSlotsPerEntry + kReservedSlots;
19  DirectHandle<WeakFixedArray> cache =
20  WeakFixedArray::New(isolate, capacity, AllocationType::kOld,
21  direct_handle(Smi::FromInt(kUnused), isolate));
22  cache->set(kNumUsedElementsIndex, Smi::zero());
23  return cache;
24 }

35 // static
36 Tagged<CodeWrapper> WasmExportWrapperCache::Get(Isolate* isolate,
37  CanonicalTypeIndex sig_index,
38  bool receiver_is_first_param) {
39  Tagged<WeakFixedArray> cache = isolate->heap()->js_to_wasm_wrappers();
40  uint32_t hash = Hash(sig_index, receiver_is_first_param);
41  uint32_t capacity = Capacity(cache);
42  if (capacity == 0) return {}; // Not found (cache is uninitialized).
43  uint32_t count = 1;
44  for (InternalIndex entry = FirstProbe(hash, capacity);;
45  entry = NextProbe(entry, count++, capacity)) {
46  DCHECK(count < capacity);
47  int index = ToArraySlot(entry);
48  Tagged<MaybeWeak<Object>> raw_key = cache->get(index);
49  DCHECK(raw_key.IsSmi());
50  int key = raw_key.ToSmi().value();
51  if (key == kUnused) return {}; // Not found.
52  DCHECK_GE(key, 0);
53  if (static_cast<uint32_t>(key) != hash) continue;
54  Tagged<MaybeWeak<Object>> value = cache->get(index + 1);
55  if (value.IsCleared()) return {}; // Not found.
56  return Cast<CodeWrapper>(value.GetHeapObjectAssumeWeak());
57  }
58 }

61 uint32_t WasmExportWrapperCache::Capacity(Tagged<WeakFixedArray> cache) {
62  // If we ever add more reserved slots, we'll have to subtract them properly
63  // rather than just shifting them out implicitly.
64  static_assert((kReservedSlots >> 1) == 0);
65  return cache->length() >> 1;
66 }

68 // static
69 Tagged<WeakFixedArray> WasmExportWrapperCache::EnsureCapacity(
70  Isolate* isolate) {
71  Tagged<WeakFixedArray> cache = isolate->heap()->js_to_wasm_wrappers();
72  if (cache->length() == 0) {
73  // Lazy initialization.
74  static constexpr int kInitialCapacity = 16;
75  DirectHandle<WeakFixedArray> new_cache = New(isolate, kInitialCapacity);
76  isolate->heap()->SetJSToWasmWrappers(*new_cache);
77  return *new_cache;
78  }
79  uint32_t capacity = Capacity(cache);
80  uint32_t used = cache->get(kNumUsedElementsIndex).ToSmi().value();
81  // We allow a maximum usage of 50% plus one entry, otherwise grow.
82  // It can happen that we find many cleared entries during growing, so the
83  // returned grown cache may be nearly empty. That's unlikely to happen
84  // often, and if it happens a few times that's not a problem.
85  if (used < capacity / 2) return cache;
86  DirectHandle<WeakFixedArray> old_cache(cache, isolate);
87  Tagged<WeakFixedArray> grown = *New(isolate, capacity * 2);
88  DisallowGarbageCollection no_gc;
89  cache = *old_cache;
90  int num_used_entries = 0;
91  for (int i = kReservedSlots; i < cache->length(); i += kSlotsPerEntry) {
92  Tagged<MaybeWeak<Object>> raw_key = cache->get(i);
93  DCHECK(raw_key.IsSmi());
94  int key = raw_key.ToSmi().value();
95  if (key == kUnused) continue;
96  Tagged<MaybeWeak<Object>> value = cache->get(i + 1);
97  if (value.IsCleared()) continue;
98  num_used_entries++;
99  PutInternal<false>(grown, key, value);
100  }
101  grown->set(kNumUsedElementsIndex, Smi::FromInt(num_used_entries));
102  isolate->heap()->SetJSToWasmWrappers(grown);
103  return grown;
104 }

106 // static
107 template <bool entry_may_exist>
108 void WasmExportWrapperCache::PutInternal(Tagged<WeakFixedArray> cache,
109  uint32_t hash,
110  Tagged<MaybeWeak<Object>> value) {
111  uint32_t capacity = Capacity(cache);
112  uint32_t count = 1;
113  for (InternalIndex entry = FirstProbe(hash, capacity);;
114  entry = NextProbe(entry, count++, capacity)) {
115  DCHECK(count < capacity);
116  int index = ToArraySlot(entry);
117  Tagged<MaybeWeak<Object>> raw_key = cache->get(index);
118  DCHECK(raw_key.IsSmi());
119  int current_key = raw_key.ToSmi().value();
120  if (current_key == kUnused) {
121  cache->set(index, Smi::FromInt(hash));
122  int used_elements = cache->get(kNumUsedElementsIndex).ToSmi().value();
123  cache->set(kNumUsedElementsIndex, Smi::FromInt(used_elements + 1));
124  } else if constexpr (entry_may_exist) {
125  if (static_cast<uint32_t>(current_key) != hash) continue;
126  // Key exists; value must be cleared, otherwise we shouldn't be
127  // trying to put a new value.
128  DCHECK(cache->get(index + 1).IsCleared());
129  } else {
130  DCHECK_NE(static_cast<uint32_t>(current_key), hash);
131  continue;
132  }
133 
134  // We get here when we've found the right entry (unused or cleared).
135  cache->set(index + 1, MakeWeak(value));
136  return;
137  }
138 }

142 // static
143 void WasmExportWrapperCache::Verify(Heap* heap) {
144  Tagged<WeakFixedArray> cache = heap->js_to_wasm_wrappers();
145  if (cache->length() == 0) return;
146  CHECK(cache->get(kNumUsedElementsIndex).IsSmi());
147  for (int i = kReservedSlots; i < cache->length(); i += kSlotsPerEntry) {
148  Tagged<MaybeObject> key = cache->get(i);
149  Tagged<MaybeObject> value = cache->get(i + 1);
150  CHECK(key.IsSmi());
151  if (value.IsSmi()) {
152  CHECK_EQ(key.ToSmi().value(), kUnused);
153  continue;
154  }
155  if (value.IsCleared()) continue;
156  CHECK(value.IsWeak());
157  CHECK(IsCodeWrapper(value.GetHeapObjectAssumeWeak()));
158  }
159 }
