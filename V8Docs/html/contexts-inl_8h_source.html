245 int Context::FunctionMapIndex(LanguageMode language_mode, FunctionKind kind,
246  bool has_shared_name) {
247  if (IsClassConstructor(kind)) {
248  // Like the strict function map, but with no 'name' accessor. 'name'
249  // needs to be the last property and it is added during instantiation,
250  // in case a static property with the same name exists"
251  return CLASS_FUNCTION_MAP_INDEX;
252  }
253 
254  int base = 0;
255  if (IsGeneratorFunction(kind)) {
256  CHECK_FOLLOWS2(GENERATOR_FUNCTION_MAP_INDEX,
257  GENERATOR_FUNCTION_WITH_NAME_MAP_INDEX);
258  CHECK_FOLLOWS2(ASYNC_GENERATOR_FUNCTION_MAP_INDEX,
259  ASYNC_GENERATOR_FUNCTION_WITH_NAME_MAP_INDEX);
260 
261  base = IsAsyncFunction(kind) ? ASYNC_GENERATOR_FUNCTION_MAP_INDEX
262  : GENERATOR_FUNCTION_MAP_INDEX;
263 
264  } else if (IsAsyncFunction(kind) || IsModuleWithTopLevelAwait(kind)) {
265  CHECK_FOLLOWS2(ASYNC_FUNCTION_MAP_INDEX,
266  ASYNC_FUNCTION_WITH_NAME_MAP_INDEX);
267 
268  base = ASYNC_FUNCTION_MAP_INDEX;
269 
270  } else if (IsStrictFunctionWithoutPrototype(kind)) {
271  CHECK_FOLLOWS2(STRICT_FUNCTION_WITHOUT_PROTOTYPE_MAP_INDEX,
272  METHOD_WITH_NAME_MAP_INDEX);
273 
274  base = STRICT_FUNCTION_WITHOUT_PROTOTYPE_MAP_INDEX;
275 
276  } else {
277  CHECK_FOLLOWS2(SLOPPY_FUNCTION_MAP_INDEX,
278  SLOPPY_FUNCTION_WITH_NAME_MAP_INDEX);
279  CHECK_FOLLOWS2(STRICT_FUNCTION_MAP_INDEX,
280  STRICT_FUNCTION_WITH_NAME_MAP_INDEX);
281 
282  base = is_strict(language_mode) ? STRICT_FUNCTION_MAP_INDEX
283  : SLOPPY_FUNCTION_MAP_INDEX;
284  }
285  int offset = static_cast<int>(!has_shared_name);
286  DCHECK_EQ(0, offset & ~1);
287 
288  return base + offset;
289 }
