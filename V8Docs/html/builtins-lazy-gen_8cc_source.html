62 void LazyBuiltinsAssembler::MaybeTailCallOptimizedCodeSlot(
63  TNode<JSFunction> function, TNode<FeedbackVector> feedback_vector) {
64  Label fallthrough(this), may_have_optimized_code(this),
65  maybe_needs_logging(this);
66 
67  TNode<Uint16T> flags =
68  LoadObjectField<Uint16T>(feedback_vector, FeedbackVector::kFlagsOffset);
69 
70  // Fall through if no optimization trigger or optimized code.
71  constexpr uint32_t kFlagMask =
72  FeedbackVector::FlagMaskForNeedsProcessingCheckFrom(
73  CodeKind::INTERPRETED_FUNCTION);
74  GotoIfNot(IsSetWord32(flags, kFlagMask), &fallthrough);
75 
76  GotoIfNot(
77  IsSetWord32(flags, FeedbackVector::kFlagsTieringStateIsAnyRequested),
78  &maybe_needs_logging);
79  GenerateTailCallToReturnedCode(Runtime::kCompileOptimized, function);
80 
81  BIND(&maybe_needs_logging);
82  {
83  GotoIfNot(IsSetWord32(flags, FeedbackVector::kFlagsLogNextExecution),
84  &may_have_optimized_code);
85  GenerateTailCallToReturnedCode(Runtime::kFunctionLogNextExecution,
86  function);
87  }
88 
89  BIND(&may_have_optimized_code);
90  {
91  Label heal_optimized_code_slot(this);
92  TNode<MaybeObject> maybe_optimized_code_entry = LoadMaybeWeakObjectField(
93  feedback_vector, FeedbackVector::kMaybeOptimizedCodeOffset);
94 
95  // Optimized code slot is a weak reference to Code object.
96  TNode<CodeWrapper> code_wrapper = CAST(GetHeapObjectAssumeWeak(
97  maybe_optimized_code_entry, &heal_optimized_code_slot));
98  TNode<Code> optimized_code =
99  LoadCodePointerFromObject(code_wrapper, CodeWrapper::kCodeOffset);
100 
101  // Check if the optimized code is marked for deopt. If it is, call the
102  // runtime to clear it.
103  GotoIf(IsMarkedForDeoptimization(optimized_code),
104  &heal_optimized_code_slot);
105 
106  // Optimized code is good, get it into the closure and link the closure into
107  // the optimized functions list, then tail call the optimized code.
108  StoreCodePointerField(function, JSFunction::kCodeOffset, optimized_code);
109  Comment("MaybeTailCallOptimizedCodeSlot:: GenerateTailCallToJSCode");
110  GenerateTailCallToJSCode(optimized_code, function);
111 
112  // Optimized code slot contains deoptimized code, or the code is cleared
113  // and tiering state hasn't yet been updated. Evict the code, update the
114  // state and re-enter the closure's code.
115  BIND(&heal_optimized_code_slot);
116  GenerateTailCallToReturnedCode(Runtime::kHealOptimizedCodeSlot, function);
117  }
118 
119  // Fall-through if the optimized code cell is clear and the tiering state is
120  // kNone.
121  BIND(&fallthrough);
122 }

126 void LazyBuiltinsAssembler::CompileLazy(TNode<JSFunction> function,
127  TNode<Context> context) {
128  // First lookup code, maybe we don't need to compile!
129  Label compile_function(this, Label::kDeferred);
130 
131  // Check the code object for the SFI. If SFI's code entry points to
132  // CompileLazy, then we need to lazy compile regardless of the function or
133  // tiering state.
134  TNode<SharedFunctionInfo> shared =
135  CAST(LoadObjectField(function, JSFunction::kSharedFunctionInfoOffset));
136  TVARIABLE(Uint16T, sfi_data_type);
137  TNode<Code> sfi_code =
138  GetSharedFunctionInfoCode(shared, &sfi_data_type, &compile_function);
139 
140  TNode<HeapObject> feedback_cell_value = LoadFeedbackCellValue(function);
141 
142  // If feedback cell isn't initialized, compile function
143  GotoIf(IsUndefined(feedback_cell_value), &compile_function);
144 
145  CSA_DCHECK(this, SafeNotEqual(sfi_code, HeapConstantNoHole(BUILTIN_CODE(
146  isolate(), CompileLazy))));
147  USE(sfi_code);
148 #ifndef V8_ENABLE_LEAPTIERING
149  // In the leaptiering case, the code is installed below, through the
150  // InstallSFICode runtime function.
151  StoreCodePointerField(function, JSFunction::kCodeOffset, sfi_code);
152 #endif // V8_ENABLE_LEAPTIERING
153 
154  Label maybe_use_sfi_code(this);
155  // If there is no feedback, don't check for optimized code.
156  GotoIf(HasInstanceType(feedback_cell_value, CLOSURE_FEEDBACK_CELL_ARRAY_TYPE),
157  &maybe_use_sfi_code);
158 
159  // If it isn't undefined or fixed array it must be a feedback vector.
160  CSA_DCHECK(this, IsFeedbackVector(feedback_cell_value));
161 
162 #ifndef V8_ENABLE_LEAPTIERING
163  // Is there a tiering state or optimized code in the feedback vector?
164  MaybeTailCallOptimizedCodeSlot(function, CAST(feedback_cell_value));
165 #endif // !V8_ENABLE_LEAPTIERING
166  Goto(&maybe_use_sfi_code);
167 
168  // At this point we have a candidate InstructionStream object. It's *not* a
169  // cached optimized InstructionStream object (we'd have tail-called it above).
170  // A usual case would be the InterpreterEntryTrampoline to start executing
171  // existing bytecode.
172  BIND(&maybe_use_sfi_code);
173 #ifdef V8_ENABLE_LEAPTIERING
174  // In the leaptiering case, we now simply install the code of the SFI on the
175  // function's dispatch table entry and call it. Installing the code is
176  // necessary as the dispatch table entry may still contain the CompileLazy
177  // builtin at this point (we can only update dispatch table code from C++).
178  CallRuntime(Runtime::kInstallSFICode, context, function);
179  GenerateTailCallToJSFunction(function);
180 #else
181  Label tailcall_code(this), baseline(this);
182  TVARIABLE(Code, code);
183 
184  // Check if we have baseline code.
185  GotoIf(InstanceTypeEqual(sfi_data_type.value(), CODE_TYPE), &baseline);
186 
187  code = sfi_code;
188  Goto(&tailcall_code);
189 
190  BIND(&baseline);
191  // Ensure we have a feedback vector.
192  code = Select<Code>(
193  IsFeedbackVector(feedback_cell_value), [=]() { return sfi_code; },
194  [=, this]() {
195  return CAST(CallRuntime(Runtime::kInstallBaselineCode,
196  Parameter<Context>(Descriptor::kContext),
197  function));
198  });
199  Goto(&tailcall_code);
200 
201  BIND(&tailcall_code);
202  GenerateTailCallToJSCode(code.value(), function);
203 #endif // V8_ENABLE_LEAPTIERING
204 
205  BIND(&compile_function);
206  CallRuntime(Runtime::kCompileLazy, context, function);
207  GenerateTailCallToJSFunction(function);
208 }

219 template <typename Function>
220 void LazyBuiltinsAssembler::TieringBuiltinImpl(const Function& Impl) {
221  auto function = Parameter<JSFunction>(Descriptor::kTarget);
222  auto context = Parameter<Context>(Descriptor::kContext);
223  auto argc = UncheckedParameter<Int32T>(Descriptor::kActualArgumentsCount);
224  auto new_target = Parameter<Object>(Descriptor::kNewTarget);
225 
226 #ifdef V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE
227  auto dispatch_handle =
228  UncheckedParameter<JSDispatchHandleT>(Descriptor::kDispatchHandle);
229 #else
230  CHECK(!V8_ENABLE_SANDBOX_BOOL);
231  auto dispatch_handle = LoadObjectField<JSDispatchHandleT>(
232  function, JSFunction::kDispatchHandleOffset);
233 #endif
234 
235  // Apply the actual tiering. This function must uninstall the tiering builtin.
236  Impl(context, function);
237 
238  // The dispatch handle of the function shouldn't change.
239  CSA_DCHECK(this,
240  Word32Equal(dispatch_handle,
241  LoadObjectField<JSDispatchHandleT>(
242  function, JSFunction::kDispatchHandleOffset)));
243 
244  // Load the code directly from the dispatch table to guarantee the signature
245  // of the code matches with the number of arguments passed when calling into
246  // this trampoline.
247  // TODO(saelo): consider removing the {code} parameter from TailCallJSCode
248  // entirely and only passing the dispatch_handle.
249  TNode<Code> code = LoadCodeObjectFromJSDispatchTable(dispatch_handle);
250  TailCallJSCode(code, context, function, new_target, argc, dispatch_handle);
251 }
