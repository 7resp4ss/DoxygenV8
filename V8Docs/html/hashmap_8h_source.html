279 template <typename Key, typename Value, typename MatchFun,
280  class AllocationPolicy>
281 Value TemplateHashMapImpl<Key, Value, MatchFun, AllocationPolicy>::Remove(
282  Entry* p) {
283  DCHECK_LE(impl_.map_ - 1, p);
284  DCHECK_LT(p, map_end());
285  Value value = p->value;
286  // To remove an entry we need to ensure that it does not create an empty
287  // entry that will cause the search for another entry to stop too soon. If all
288  // the entries between the entry to remove and the next empty slot have their
289  // initial position inside this interval, clearing the entry to remove will
290  // not break the search. If, while searching for the next empty entry, an
291  // entry is encountered which does not have its initial position between the
292  // entry to remove and the position looked at, then this entry can be moved to
293  // the place of the entry to remove without breaking the search for it. The
294  // entry made vacant by this move is now the entry to remove and the process
295  // starts over.
296  // Algorithm from http://en.wikipedia.org/wiki/Open_addressing.
297 
298  // This guarantees loop termination as there is at least one empty entry so
299  // eventually the removed entry will have an empty entry after it.
300  DCHECK(occupancy() < capacity());
301 
302  // p is the candidate entry to clear. q is used to scan forwards.
303  Entry* q = p; // Start at the entry to remove.
304  while (true) {
305  // Move q to the next entry.
306  q = q + 1;
307  if (q == map_end()) {
308  q = impl_.map_;
309  }
310 
311  // All entries between p and q have their initial position between p and q
312  // and the entry p can be cleared without breaking the search for these
313  // entries.
314  if (!q->exists()) {
315  break;
316  }
317 
318  // Find the initial position for the entry at position q.
319  Entry* r = impl_.map_ + (q->hash & (capacity() - 1));
320 
321  // If the entry at position q has its initial position outside the range
322  // between p and q it can be moved forward to position p and will still be
323  // found. There is now a new candidate entry for clearing.
324  if ((q > p && (r <= p || r > q)) || (q < p && (r <= p && r > q))) {
325  *p = *q;
326  p = q;
327  }
328  }
329 
330  // Clear the entry which is allowed to en emptied.
331  p->clear();
332  impl_.occupancy_--;
333  return value;
334 }

367 template <typename Key, typename Value, typename MatchFun,
368  class AllocationPolicy>
369 template <typename LookupKey>
370 typename TemplateHashMapImpl<Key, Value, MatchFun, AllocationPolicy>::Entry*
371 TemplateHashMapImpl<Key, Value, MatchFun, AllocationPolicy>::Probe(
372  const LookupKey& key, uint32_t hash) const {
373  hash &= 0x7FFFFFFF;
374  DCHECK(base::bits::IsPowerOfTwo(capacity()));
375  size_t i = hash & (capacity() - 1);
376  DCHECK(i < capacity());
377 
378  DCHECK(occupancy() < capacity()); // Guarantees loop termination.
379  Entry* map = impl_.map_;
380  while (map[i].exists() &&
381  !impl_.match()(hash, map[i].hash, key, map[i].key)) {
382  i = (i + 1) & (capacity() - 1);
383  }
384 
385  return &map[i];
386 }

388 template <typename Key, typename Value, typename MatchFun,
389  class AllocationPolicy>
390 typename TemplateHashMapImpl<Key, Value, MatchFun, AllocationPolicy>::Entry*
391 TemplateHashMapImpl<Key, Value, MatchFun, AllocationPolicy>::FillEmptyEntry(
392  Entry* entry, const Key& key, const Value& value, uint32_t hash) {
393  DCHECK(!entry->exists());
394 
395  new (entry) Entry(key, value, hash);
396  impl_.occupancy_++;
397 
398  // Grow the map if we reached >= 80% occupancy.
399  if (occupancy() + occupancy() / 4 >= capacity()) {
400  Resize();
401  entry = Probe(key, hash);
402  }
403 
404  return entry;
405 }

421 template <typename Key, typename Value, typename MatchFun,
422  class AllocationPolicy>
423 void TemplateHashMapImpl<Key, Value, MatchFun, AllocationPolicy>::Resize() {
424  Entry* old_map = impl_.map_;
425  uint32_t old_capacity = capacity();
426  uint32_t n = occupancy();
427 
428  // Allocate larger map.
429  Initialize(capacity() * 2);
430 
431  // Rehash all current entries.
432  for (Entry* entry = old_map; n > 0; entry++) {
433  if (entry->exists()) {
434  Entry* new_entry = Probe(entry->key, entry->hash);
435  new_entry =
436  FillEmptyEntry(new_entry, entry->key, entry->value, entry->hash);
437  n--;
438  }
439  }
440 
441  // Delete old map.
442  impl_.allocator().DeleteArray(old_map, old_capacity);
443 }

533 // A hash map for pointer keys and values with an STL-like interface.
534 template <class Key, class Value, class MatchFun, class AllocationPolicy>
535 class TemplateHashMap
536  : private TemplateHashMapImpl<void*, void*,
537  HashEqualityThenKeyMatcher<void*, MatchFun>,
538  AllocationPolicy> {
539  using Base = TemplateHashMapImpl<void*, void*,
540  HashEqualityThenKeyMatcher<void*, MatchFun>,
541  AllocationPolicy>;
542 
543  public:
544  static_assert(sizeof(Key*) == sizeof(void*));
545  static_assert(sizeof(Value*) == sizeof(void*));
546  struct value_type {
547  Key* first;
548  Value* second;
549  };
550 
551  class Iterator {
552  public:
553  Iterator& operator++() {
554  entry_ = map_->Next(entry_);
555  return *this;
556  }
557 
558  value_type* operator->() { return reinterpret_cast<value_type*>(entry_); }
559  bool operator==(const Iterator& other) { return entry_ == other.entry_; }
560 
561  private:
562  Iterator(const Base* map, typename Base::Entry* entry)
563  : map_(map), entry_(entry) {}
564 
565  const Base* map_;
566  typename Base::Entry* entry_;
567 
568  friend class TemplateHashMap;
569  };
570 
571  explicit TemplateHashMap(MatchFun match,
572  AllocationPolicy allocator = AllocationPolicy())
573  : Base(Base::kDefaultHashMapCapacity,
574  HashEqualityThenKeyMatcher<void*, MatchFun>(match), allocator) {}
575 
576  Iterator begin() const { return Iterator(this, this->Start()); }
577  Iterator end() const { return Iterator(this, nullptr); }
578  Iterator find(Key* key, bool insert = false) {
579  if (insert) {
580  return Iterator(this, this->LookupOrInsert(key, key->Hash()));
581  }
582  return Iterator(this, this->Lookup(key, key->Hash()));
