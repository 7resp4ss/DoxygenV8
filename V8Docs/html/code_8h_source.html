32 // Code is a container for data fields related to its associated
33 // {InstructionStream} object. Since {InstructionStream} objects reside on
34 // write-protected pages within the heap, its header fields need to be
35 // immutable. Every InstructionStream object has an associated Code object,
36 // but not every Code object has an InstructionStream (e.g. for builtins).
37 //
38 // Embedded builtins consist of on-heap Code objects, with an out-of-line body
39 // section. Accessors (e.g. InstructionStart), redirect to the off-heap area.
40 // Metadata table offsets remain relative to MetadataStart(), i.e. they point
41 // into the off-heap metadata section. The off-heap layout is described in
42 // detail in the EmbeddedData class, but at a high level one can assume a
43 // dedicated, out-of-line, instruction and metadata section for each embedded
44 // builtin:
45 //
46 // +--------------------------+ <-- InstructionStart()
47 // | off-heap instructions |
48 // | ... |
49 // +--------------------------+ <-- InstructionEnd()
50 //
51 // +--------------------------+ <-- MetadataStart() (MS)
52 // | off-heap metadata |
53 // | ... | <-- MS + handler_table_offset()
54 // | | <-- MS + constant_pool_offset()
55 // | | <-- MS + code_comments_offset()
56 // | | <-- MS + jump_table_info_offset()
57 // | | <-- MS + unwinding_info_offset()
58 // +--------------------------+ <-- MetadataEnd()
59 //
60 // When the sandbox is enabled, Code objects are allocated outside the sandbox
61 // and referenced through indirect pointers, so they need to inherit from
62 // ExposedTrustedObject.
63 class Code : public ExposedTrustedObject {

390 #define CODE_DATA_FIELDS(V) \
391  /* The deoptimization_data_or_interpreter_data field contains: */ \
392  /* - A DeoptimizationData for optimized code (maglev or turbofan) */ \
393  /* - A BytecodeArray or InterpreterData for baseline code */ \
394  /* - Smi::zero() for all other types of code (e.g. builtin) */ \
395  V(kDeoptimizationDataOrInterpreterDataOffset, kTaggedSize) \
396  /* This field contains: */ \
397  /* - A bytecode offset table (trusted byte array) for baseline code */ \
398  /* - A (possibly empty) source position table (trusted byte array) for */ \
399  /* most other types of code */ \
400  /* - Smi::zero() for embedded builtin code (in RO space) */ \
401  /* TODO(saelo) once we have a trusted RO space, we could instead use */ \
402  /* empty_trusted_byte_array to avoid using Smi::zero() at all. */ \
403  V(kPositionTableOffset, kTaggedSize) \
404  /* Strong pointer fields. */ \
405  V(kStartOfStrongFieldsOffset, 0) \
406  V(kWrapperOffset, kTaggedSize) \
407  V(kEndOfStrongFieldsWithMainCageBaseOffset, 0) \
408  /* The InstructionStream field is special: it uses code_cage_base. */ \
409  V(kInstructionStreamOffset, kTaggedSize) \
410  V(kEndOfStrongFieldsOffset, 0) \
411  /* Untagged data not directly visited by GC starts here. */ \
412  /* When the sandbox is off, the instruction_start field contains a raw */ \
413  /* pointer to the first instruction of this Code. */ \

519 // A Code object when used in situations where gc might be in progress. The
520 // underlying pointer is guaranteed to be a Code object.
521 //
522 // Semantics around Code and InstructionStream objects are quite delicate when
523 // GC is in progress and objects are currently being moved, because the
524 // tightly-coupled object pair {Code,InstructionStream} are conceptually
525 // treated as a single object in our codebase, and we frequently convert
526 // between the two. However, during GC, extra care must be taken when accessing
527 // the `Code::instruction_stream` and `InstructionStream::code` slots because
528 // they may contain forwarding pointers.
529 //
530 // This class a) clarifies at use sites that we're dealing with a Code object
531 // in a situation that requires special semantics, and b) safely implements
532 // related functions.
533 //
534 // Note that both the underlying Code object and the associated
535 // InstructionStream may be forwarding pointers, thus type checks and normal
536 // (checked) casts do not work on GcSafeCode.
537 class GcSafeCode : public HeapObject {
538  public:
539  // Use with care, this casts away knowledge that we're dealing with a
540  // special-semantics object.
541  inline Tagged<Code> UnsafeCastToCode() const;
542 
543  // Safe accessors (these just forward to Code methods).
544  inline Address instruction_start() const;
545  inline Address instruction_end() const;
546  inline bool is_builtin() const;
547  inline Builtin builtin_id() const;
548  inline CodeKind kind() const;
549  inline bool is_interpreter_trampoline_builtin() const;
550  inline bool is_baseline_trampoline_builtin() const;
551  inline bool is_baseline_leave_frame_builtin() const;
