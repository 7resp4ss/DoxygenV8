53 void FreeSpace::SetNext(const WritableFreeSpace& writable_free_space,
54  Tagged<FreeSpace> next) {
55  DCHECK(IsValid());
56 
57 #ifdef V8_EXTERNAL_CODE_SPACE
58  if (next.is_null()) {
59  writable_free_space.WriteHeaderSlot<Smi, offsetof(FreeSpace, next_)>(
60  Smi::zero(), kRelaxedStore);
61  return;
62  }
63  intptr_t diff_to_next = next.ptr() - ptr();
64  DCHECK(IsAligned(diff_to_next, kObjectAlignment));
65  writable_free_space.WriteHeaderSlot<Smi, offsetof(FreeSpace, next_)>(
66  Smi::FromIntptr(diff_to_next / kObjectAlignment), kRelaxedStore);
67 #else
68  writable_free_space.WriteHeaderSlot<Object, offsetof(FreeSpace, next_)>(
69  next, kRelaxedStore);
70 #endif // V8_EXTERNAL_CODE_SPACE
71 }
