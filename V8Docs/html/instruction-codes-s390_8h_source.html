407 // Addressing modes represent the "shape" of inputs to an instruction.
408 // Many instructions support multiple addressing modes. Addressing modes
409 // are encoded into the InstructionCode of the instruction and tell the
410 // code generator after register allocation which assembler method to call.
411 //
412 // We use the following local notation for addressing modes:
413 //
414 // R = register
415 // O = register or stack slot
416 // D = double register
417 // I = immediate (handle, external, int32)
418 // MRI = [register + immediate]
419 // MRR = [register + register]
420 #define TARGET_ADDRESSING_MODE_LIST(V) \
421  V(MR) /* [%r0 ] */ \
422  V(MRI) /* [%r0 + K] */ \
423  V(MRR) /* [%r0 + %r1 ] */ \
424  V(MRRI) /* [%r0 + %r1 + K] */ \
425  V(Root) /* [%r0 + K] */
