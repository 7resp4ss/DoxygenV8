110 // Improve the speed that we scan for an initial point where a non-anchored
111 // regexp can match by using a Boyer-Moore-like table. This is done by
112 // identifying non-greedy non-capturing loops in the nodes that eat any
113 // character one at a time. For example in the middle of the regexp
114 // /foo[\s\S]*?bar/ we find such a loop. There is also such a loop implicitly
115 // inserted at the start of any non-anchored regexp.
116 //
117 // When we have found such a loop we look ahead in the nodes to find the set of
118 // characters that can come at given distances. For example for the regexp
119 // /.?foo/ we know that there are at least 3 characters ahead of us, and the
120 // sets of characters that can occur are [any, [f, o], [o]]. We find a range in
121 // the lookahead info where the set of characters is reasonably constrained. In
122 // our example this is from index 1 to 2 (0 is not constrained). We can now
123 // look 3 characters ahead and if we don't find one of [f, o] (the union of
124 // [f, o] and [o]) then we can skip forwards by the range size (in this case 2).
125 //
126 // For Unicode input strings we do the same, but modulo 128.
127 //
128 // We also look at the first string fed to the regexp and use that to get a hint
129 // of the character frequencies in the inputs. This affects the assessment of
130 // whether the set of characters is 'reasonably constrained'.
131 //
132 // We also have another lookahead mechanism (called quick check in the code),
133 // which uses a wide load of multiple characters followed by a mask and compare
134 // to determine whether a match is possible at this point.
135 enum ContainedInLattice {
136  kNotYet = 0,
137  kLatticeIn = 1,
138  kLatticeOut = 2,
139  kLatticeUnknown = 3 // Can also mean both in and out.
140 };
141 
142 inline ContainedInLattice Combine(ContainedInLattice a, ContainedInLattice b) {
143  return static_cast<ContainedInLattice>(a | b);
144 }

226 // There are many ways to generate code for a node. This class encapsulates
227 // the current way we should be generating. In other words it encapsulates
228 // the current state of the code generator. The effect of this is that we
229 // generate code for paths that the matcher can take through the regular
230 // expression. A given node in the regexp can be code-generated several times
231 // as it can be part of several traces. For example for the regexp:
232 // /foo(bar|ip)baz/ the code to match baz will be generated twice, once as part
233 // of the foo-bar-baz trace and once as part of the foo-ip-baz trace. The code
234 // to match foo is generated only once (the traces have a common prefix). The
235 // code to store the capture is deferred and generated (twice) after the places
236 // where baz has been matched.

420  FrequencyCollator() : total_samples_(0) {

426  void CountCharacter(int character) {
427  int index = (character & RegExpMacroAssembler::kTableMask);
428  frequencies_[index].Increment();
429  total_samples_++;
430  }

432  // Does not measure in percent, but rather per-128 (the table size from the
433  // regexp macro assembler).
434  int Frequency(int in_character) {
435  DCHECK((in_character & RegExpMacroAssembler::kTableMask) == in_character);
436  if (total_samples_ < 1) return 1; // Division by zero.
437  int freq_in_per128 =
438  (frequencies_[in_character].counter() * 128) / total_samples_;
439  return freq_in_per128;
440  }
