20 bool ValidStructSubtypeDefinition(ModuleTypeIndex subtype_index,
21  ModuleTypeIndex supertype_index,
22  const WasmModule* module) {
23  const TypeDefinition& sub_def = module->type(subtype_index);
24  const TypeDefinition& super_def = module->type(supertype_index);
25  const StructType* sub_struct = sub_def.struct_type;
26  const StructType* super_struct = super_def.struct_type;
27 
28  if (sub_struct->field_count() < super_struct->field_count()) {
29  return false;
30  }
31 
32  for (uint32_t i = 0; i < super_struct->field_count(); i++) {
33  bool sub_mut = sub_struct->mutability(i);
34  bool super_mut = super_struct->mutability(i);
35  if (sub_mut != super_mut ||
36  (sub_mut && !EquivalentTypes(sub_struct->field(i),
37  super_struct->field(i), module)) ||
38  (!sub_mut &&
39  !IsSubtypeOf(sub_struct->field(i), super_struct->field(i), module))) {
40  return false;
41  }
42  }
43  if (sub_def.descriptor.valid()) {
44  // If a type has a descriptor, its supertype must either have no descriptor,
45  // or the supertype's descriptor must be a supertype of the subtype's
46  // descriptor.
47  if (super_def.descriptor.valid() &&
48  !IsHeapSubtypeOf(module->heap_type(sub_def.descriptor),
49  module->heap_type(super_def.descriptor), module)) {
50  return false;
51  }
52  } else {
53  // If a type has no descriptor, its supertype must not have one either.
54  if (super_def.descriptor.valid()) return false;
55  }
56  // A type and its supertype must either both be descriptors, or both not.
57  if (sub_def.describes.valid() != super_def.describes.valid()) {
58  return false;
59  }
60  return true;
61 }

63 bool ValidArraySubtypeDefinition(ModuleTypeIndex subtype_index,
64  ModuleTypeIndex supertype_index,
65  const WasmModule* module) {
66  const ArrayType* sub_array = module->type(subtype_index).array_type;
67  const ArrayType* super_array = module->type(supertype_index).array_type;
68  bool sub_mut = sub_array->mutability();
69  bool super_mut = super_array->mutability();
70 
71  return (sub_mut && super_mut &&
72  EquivalentTypes(sub_array->element_type(),
73  super_array->element_type(), module)) ||
74  (!sub_mut && !super_mut &&
75  IsSubtypeOf(sub_array->element_type(), super_array->element_type(),
76  module));
77 }

79 bool ValidFunctionSubtypeDefinition(ModuleTypeIndex subtype_index,
80  ModuleTypeIndex supertype_index,
81  const WasmModule* module) {
82  const FunctionSig* sub_func = module->type(subtype_index).function_sig;
83  const FunctionSig* super_func = module->type(supertype_index).function_sig;
84 
85  if (sub_func->parameter_count() != super_func->parameter_count() ||
86  sub_func->return_count() != super_func->return_count()) {
87  return false;
88  }
89 
90  for (uint32_t i = 0; i < sub_func->parameter_count(); i++) {
91  // Contravariance for params.
92  if (!IsSubtypeOf(super_func->parameters()[i], sub_func->parameters()[i],
93  module)) {
94  return false;
95  }
96  }
97  for (uint32_t i = 0; i < sub_func->return_count(); i++) {
98  // Covariance for returns.
99  if (!IsSubtypeOf(sub_func->returns()[i], super_func->returns()[i],
100  module)) {
101  return false;
102  }
103  }
104 
105  return true;
106 }

121 StandardType UpcastToStandardType(ValueTypeBase type) {
122  if (!type.has_index()) return type.standard_type();
123  switch (type.ref_type_kind()) {
124  case RefTypeKind::kStruct:
125  return StandardType::kStruct;
126  case RefTypeKind::kArray:
127  return StandardType::kArray;
128  case RefTypeKind::kFunction:
129  return StandardType::kFunc;
130  case RefTypeKind::kCont:
131  return StandardType::kCont;
132  case RefTypeKind::kOther:

137 // Format: subtype, supertype
138 #define FOREACH_SUBTYPING(V) /* force 80 cols */ \
139  /* anyref hierarchy */ \
140  V(Eq, Any) \
141  V(Struct, Eq) \
142  V(None, Struct) \
143  V(Array, Eq) \
144  V(None, Array) \
145  V(I31, Eq) \
146  V(None, I31) \
147  V(String, Any) \
148  V(None, String) \
149  /* funcref hierarchy */ \
150  V(NoFunc, Func) \
151  /* extern hierarchy */ \
152  V(ExternString, Extern) \
153  V(NoExtern, ExternString) \
154  /* exnref hierarchy */ \
155  V(NoExn, Exn) \
156  /* cont hierarchy */ \
157  V(NoCont, Cont)

159 static constexpr uint32_t kNumStandardTypes =
160  value_type_impl::kNumberOfStandardTypes;
161 
162 static constexpr std::array<GenericKind, kNumStandardTypes>
163  kValueTypeLookupMap = base::make_array<kNumStandardTypes>([](size_t i) {
164 #define ENTRY(name, ...) \
165  if (static_cast<size_t>(StandardType::k##name) == i) { \
166  return GenericKind::k##name; \
167  }

169 #undef ENTRY

175  return kValueTypeLookupMap[static_cast<size_t>(type)];

178 ////////////////////////////////////////////////////////////////////////////////
179 // Construct a dense index space for nontrivial subtypes, to save space in
180 // the 2D lookup maps we'll build for them.
181 
182 constexpr int NotInSubtypeRelation(StandardType type) {
183 #define CASE(sub, super) \
184  if (type == StandardType::k##sub || type == StandardType::k##super) return 0;
185  FOREACH_SUBTYPING(CASE)
186 #undef CASE
187  return 1;
188 }
189 static constexpr uint32_t kNumNonTrivial =
190 #define CASE(name, ...) (NotInSubtypeRelation(StandardType::k##name) ? 0 : 1) +
191  FOREACH_GENERIC_TYPE(CASE)
192 // We could include numeric types here, but they don't have nontrivial
193 // subtyping relations anyway.
194 #undef CASE
195  0;
196 enum class CondensedIndices : int {
197 // "kFooAdjustNext" is a dummy entry to control whether the next "kBar" will
198 // get its own unique value: when NotInSubtypeRelation(kFoo), then we'll end
199 // up with kFoo = N, kFooAdJustNext = N-1, kBar = N, and we'll never use
200 // kFoo later, so have effectively skipped it from the condensed space.
201 #define ENTRY(name, ...) \
202  k##name, \
203  k##name##AdjustNext = \
204  k##name - NotInSubtypeRelation(StandardType::k##name),
205  FOREACH_GENERIC_TYPE(ENTRY)
206 #undef ENTRY
207 };
208 static constexpr uint8_t kNotRelatedSentinel = 0xFF;
209 static_assert(kNumStandardTypes < kNotRelatedSentinel);
210 
211 constexpr uint8_t ComputeCondensedIndex(StandardType type) {
212  switch (type) {
213 #define BAILOUT(name, ...) case StandardType::k##name:
214  FOREACH_NUMERIC_VALUE_TYPE(BAILOUT)
215  return kNotRelatedSentinel;
216 #undef BAILOUT
217 #define CASE(name, ...) \
218  case StandardType::k##name: \
219  if (NotInSubtypeRelation(type)) return kNotRelatedSentinel; \
220  return static_cast<uint8_t>(CondensedIndices::k##name);
221  FOREACH_GENERIC_TYPE(CASE)
222 #undef CASE
223  }
224 }
225 constexpr StandardType ComputeStandardType(uint8_t condensed_index) {
226 #define CASE(name, ...) \
227  if (ComputeCondensedIndex(StandardType::k##name) == condensed_index) \
228  return StandardType::k##name;
229  FOREACH_GENERIC_TYPE(CASE)
230 #undef CASE
231  UNREACHABLE();
232 }
233 
234 static constexpr std::array<uint8_t, kNumStandardTypes>
235  kCondensedIndexLookupMap =
236  base::make_array<kNumStandardTypes>([](size_t i) {
237  return ComputeCondensedIndex(static_cast<StandardType>(i));
238  });
239 static constexpr std::array<uint8_t, kNumNonTrivial> kCondensedToStandardMap =
240  base::make_array<kNumNonTrivial>(
241  [](size_t i) { return static_cast<uint8_t>(ComputeStandardType(i)); });
242 
243 constexpr uint8_t CondensedIndex(StandardType type) {
244  return kCondensedIndexLookupMap[static_cast<size_t>(type)];
245 }
246 constexpr StandardType CondensedToStandard(uint8_t condensed) {
247  return static_cast<StandardType>(kCondensedToStandardMap[condensed]);
248 }

250 ////////////////////////////////////////////////////////////////////////////////
251 // Statically compute transitive subtype relations.
252 
253 // Inputs are "condensed".
254 constexpr bool ComputeIsSubtype(size_t sub, size_t super) {
255  if (sub == super) return true;
256 #define CASE(a, b) \
257  { \
258  size_t raw_a = static_cast<size_t>(CondensedIndices::k##a); \
259  size_t raw_b = static_cast<size_t>(CondensedIndices::k##b); \
260  if (sub == raw_a) { \
261  if (super == raw_b) return true; \
262  if (ComputeIsSubtype(raw_b, super)) return true; \
263  } \
264  }
265  FOREACH_SUBTYPING(CASE)
266 #undef CASE
267  return false;
268 }
269 
270 // A 2D map (type, type) -> bool indicating transitive subtype relationships.
271 // Keyed by "condensed" indices to save space.
272 static constexpr std::array<std::array<bool, kNumNonTrivial>, kNumNonTrivial>
273  kSubtypeLookupMap2 = base::make_array<kNumNonTrivial>([](size_t sub) {
274  return base::make_array<kNumNonTrivial>(
275  [sub](size_t super) { return ComputeIsSubtype(sub, super); });
276  });

278 // The "public" accessor for looking up subtyping of standard types.
279 constexpr bool SubtypeLookup(StandardType sub, StandardType super) {
280  // Note: this implementation strikes a compromise between time and space.
281  // We could put everything into the lookup map, but that would significantly
282  // increase its size (at the time of this writing: by about 4x), with
283  // mostly-sparse additional entries.
284  if (sub == StandardType::kBottom) return true;
285  if (super == StandardType::kTop) return true;
286  uint8_t sub_condensed = CondensedIndex(sub);
287  if (sub_condensed == kNotRelatedSentinel) return sub == super;
288  uint8_t super_condensed = CondensedIndex(super);
289  if (super_condensed == kNotRelatedSentinel) return false;
290  return kSubtypeLookupMap2[sub_condensed][super_condensed];
291 }

293 ////////////////////////////////////////////////////////////////////////////////
294 // Statically compute common ancestors.
295 
296 // Inputs are "condensed", result is a full StandardType because it might
297 // be kTop.
298 constexpr StandardType ComputeCommonAncestor(size_t t1, size_t t2) {
299  if (kSubtypeLookupMap2[t1][t2]) return CondensedToStandard(t2);
300  if (kSubtypeLookupMap2[t2][t1]) return CondensedToStandard(t1);
301 #define CASE(a, b) \
302  if (t1 == static_cast<size_t>(CondensedIndices::k##a)) { \
303  return ComputeCommonAncestor(static_cast<size_t>(CondensedIndices::k##b), \
304  t2); \
305  }
306  FOREACH_SUBTYPING(CASE)
307 #undef CASE
308  return StandardType::kTop;
309 }

311 // A 2D map (type, type) -> type caching the lowest common supertype.
312 // Keyed by "condensed" indices to save space.
313 static constexpr std::array<std::array<StandardType, kNumNonTrivial>,
314  kNumNonTrivial>
315  kCommonAncestorLookupMap = base::make_array<kNumNonTrivial>([](size_t sub) {
316  return base::make_array<kNumNonTrivial>(
317  [sub](size_t super) { return ComputeCommonAncestor(sub, super); });
318  });

320 // The "public" accessor for looking up common supertypes of standard types.
321 constexpr StandardType CommonAncestorLookup(StandardType t1, StandardType t2) {
322  if (t1 == StandardType::kBottom) return t2;
323  if (t2 == StandardType::kBottom) return t1;
324  if (t1 == StandardType::kTop) return t1;
325  if (t2 == StandardType::kTop) return t2;
326  uint8_t t1_condensed = CondensedIndex(t1);
327  if (t1_condensed == kNotRelatedSentinel) {
328  return t2 == t1 ? t1 : StandardType::kTop;
329  }
330  uint8_t t2_condensed = CondensedIndex(t2);
331  if (t2_condensed == kNotRelatedSentinel) return StandardType::kTop;
332  return kCommonAncestorLookupMap[t1_condensed][t2_condensed];
333 }

338 HeapType NullSentinelImpl(HeapType type) {
339  StandardType standard = UpcastToStandardType(type);
340  constexpr StandardType candidates[] = {
341 #define NULLTYPE(name, ...) StandardType::k##name,
342  FOREACH_NONE_TYPE(NULLTYPE)
343 #undef NULLTYPE
344  };
345  for (StandardType candidate : candidates) {
346  if (SubtypeLookup(candidate, standard)) {
347  return HeapType::Generic(ToGenericKind(candidate), type.is_shared());
348  }
349  }
350  if (type.is_string_view()) {
351  // TODO(12868): This special case reflects unresolved discussion. If string
352  // views aren't nullable, they shouldn't really show up here at all.
353  return HeapType::Generic(GenericKind::kNone, type.is_shared());
354  }
355  UNREACHABLE();
356 }

372 bool ValidSubtypeDefinition(ModuleTypeIndex subtype_index,
373  ModuleTypeIndex supertype_index,
374  const WasmModule* module) {
375  const TypeDefinition& subtype = module->type(subtype_index);
376  const TypeDefinition& supertype = module->type(supertype_index);
377  if (subtype.kind != supertype.kind) return false;
378  if (supertype.is_final) return false;
379  if (subtype.is_shared != supertype.is_shared) return false;
380  switch (subtype.kind) {
381  case TypeDefinition::kFunction:
382  return ValidFunctionSubtypeDefinition(subtype_index, supertype_index,
383  module);
384  case TypeDefinition::kStruct:
385  return ValidStructSubtypeDefinition(subtype_index, supertype_index,
386  module);
387  case TypeDefinition::kArray:
388  return ValidArraySubtypeDefinition(subtype_index, supertype_index,
389  module);
390  case TypeDefinition::kCont:
391  return ValidContinuationSubtypeDefinition(subtype_index, supertype_index,
392  module);
393  }

397 // Common parts of the implementation for ValueType and CanonicalValueType.
398 std::optional<bool> IsSubtypeOf_CommonImpl(ValueTypeBase subtype,
399  ValueTypeBase supertype) {
400  DCHECK(!subtype.is_numeric() && !supertype.is_numeric());
401 
402  if (subtype.is_shared() != supertype.is_shared()) return false;
403  if (subtype.is_bottom()) return true;
404  if (supertype.has_index()) {
405  if (subtype.has_index()) {
406  // Only exact types can possibly be subtypes of other exact types.
407  if (supertype.is_exact() && !subtype.is_exact()) return false;
408  // If both types are indexed, the specialized implementations need to
409  // take care of it.
410  return {};
411  }
412  // Subtype is generic. It can only be a subtype if it is a none-type.
413  if (!IsGenericSubtypeOfIndexedTypes(subtype)) return false;
414  }
415  return SubtypeLookup(UpcastToStandardType(subtype),
416  UpcastToStandardType(supertype));
417 }

421 V8_NOINLINE V8_EXPORT_PRIVATE bool IsSubtypeOfImpl(
422  HeapType subtype, HeapType supertype, const WasmModule* sub_module,
423  const WasmModule* super_module) {
424  DCHECK(subtype != supertype || sub_module != super_module);
425 
426  std::optional<bool> result = IsSubtypeOf_CommonImpl(subtype, supertype);
427  if (result.has_value()) return result.value();
428  DCHECK(subtype.has_index() && supertype.has_index());
429  ModuleTypeIndex sub_index = subtype.ref_index();
430  CanonicalTypeIndex super_canon =
431  super_module->canonical_type_id(supertype.ref_index());
432  // Comparing canonicalized type indices handles both different modules
433  // and different recgroups in the same module.
434  if (supertype.is_exact()) {
435  return sub_module->canonical_type_id(sub_index) == super_canon;
436  }
437  do {
438  if (sub_module->canonical_type_id(sub_index) == super_canon) return true;
439  sub_index = sub_module->supertype(sub_index);
440  } while (sub_index.valid());
441  return false;
442 }

444 V8_NOINLINE V8_EXPORT_PRIVATE bool IsSubtypeOfImpl(
445  ValueType subtype, ValueType supertype, const WasmModule* sub_module,
446  const WasmModule* super_module) {
447  // Note: it seems tempting to handle <top> and the numeric types
448  // with the lookup-based mechanism we'll ultimately call. But that would
449  // require guarding the checks for nullability and sharedness behind
450  // checks for being reftypes, so it would end up being more complicated,
451  // not less.
452  if (supertype.is_top()) return true;
453  if (subtype.is_numeric()) return subtype == supertype;
454  if (supertype.is_numeric()) return subtype.is_bottom();
455  if (subtype.is_nullable() && !supertype.is_nullable()) return false;
456  HeapType sub_heap = subtype.heap_type();
457  HeapType super_heap = supertype.heap_type();
458  if (sub_heap == super_heap && sub_module == super_module) return true;
459  return IsSubtypeOfImpl(sub_heap, super_heap, sub_module, super_module);
460 }

462 V8_NOINLINE V8_EXPORT_PRIVATE bool IsSubtypeOfImpl(
463  CanonicalValueType subtype, CanonicalValueType supertype) {
464  DCHECK_NE(subtype, supertype); // Caller has checked.
465  if (supertype.is_top()) return true;
466  if (subtype.is_numeric()) return false;
467  if (supertype.is_numeric()) return subtype.is_bottom();
468  if (subtype.is_nullable() && !supertype.is_nullable()) return false;
469 
470  std::optional<bool> result = IsSubtypeOf_CommonImpl(subtype, supertype);
471  if (result.has_value()) return result.value();
472  DCHECK(subtype.has_index() && supertype.has_index());
473  CanonicalTypeIndex sub_index = subtype.ref_index();
474  CanonicalTypeIndex super_index = supertype.ref_index();
475  // Can happen despite subtype != supertype, e.g. when nullability differs.
476  if (sub_index == super_index) return true;
477  if (supertype.is_exact()) return false;
478  return GetTypeCanonicalizer()->IsHeapSubtype(sub_index, super_index);
479 }

481 V8_NOINLINE bool EquivalentTypes(ValueType type1, ValueType type2,
482  const WasmModule* module) {
483  if (type1 == type2) return true;
484  if (!type1.has_index() || !type2.has_index()) return type1 == type2;
485  if (type1.nullability() != type2.nullability()) return false;
486  if (type1.is_exact() != type2.is_exact()) return false;
487 
488  DCHECK(type1.has_index() && module->has_type(type1.ref_index()) &&
489  type2.has_index() && module->has_type(type2.ref_index()));
490 
491  return EquivalentIndices(type1.ref_index(), type2.ref_index(), module);
492 }

494 V8_NOINLINE bool EquivalentTypes(ValueType type1, ValueType type2,
495  const WasmModule* module1,
496  const WasmModule* module2) {
497  if (type1 == type2 && module1 == module2) return true;
498  if (!type1.has_index() || !type2.has_index()) return type1 == type2;
499  if (type1.nullability() != type2.nullability()) return false;
500  if (type1.is_exact() != type2.is_exact()) return false;
501 
502  DCHECK(type1.has_index() && module1->has_type(type1.ref_index()) &&
503  type2.has_index() && module2->has_type(type2.ref_index()));
504 
505  return module1->canonical_type_id(type1.ref_index()) ==
506  module2->canonical_type_id(type2.ref_index());
507 }

509 namespace {
510 // Returns the least common ancestor of two type indices, as a type index in
511 // {module1}.
512 HeapType CommonAncestor(HeapType type1, HeapType type2,
513  const WasmModule* module) {
514  DCHECK(type1.has_index() && type2.has_index());
515  bool both_shared = type1.is_shared();
516  if (both_shared != type2.is_shared()) return HeapType{kWasmTop};
517 
518  ModuleTypeIndex type_index1 = type1.ref_index();
519  ModuleTypeIndex type_index2 = type2.ref_index();
520  {
521  int depth1 = GetSubtypingDepth(module, type_index1);
522  int depth2 = GetSubtypingDepth(module, type_index2);
523  while (depth1 > depth2) {
524  type_index1 = module->supertype(type_index1);
525  depth1--;
526  }
527  while (depth2 > depth1) {
528  type_index2 = module->supertype(type_index2);
529  depth2--;
530  }
531  }
532  DCHECK_NE(type_index1, kNoSuperType);
533  DCHECK_NE(type_index2, kNoSuperType);
534  while (type_index1 != kNoSuperType &&
535  !EquivalentIndices(type_index1, type_index2, module)) {
536  type_index1 = module->supertype(type_index1);
537  type_index2 = module->supertype(type_index2);
538  }
539  DCHECK_EQ(type_index1 == kNoSuperType, type_index2 == kNoSuperType);
540  RefTypeKind kind1 = type1.ref_type_kind();
541  if (type_index1 != kNoSuperType) {
542  DCHECK_EQ(kind1, type2.ref_type_kind());
543  return HeapType::Index(type_index1, both_shared, kind1);
544  }
545  // No indexed type was found as common ancestor, so we can treat both types
546  // as generic.
547  StandardType generic_ancestor = CommonAncestorLookup(
548  UpcastToStandardType(type1), UpcastToStandardType(type2));
549  return HeapType::Generic(ToGenericKind(generic_ancestor), both_shared);
550 }

552 // Returns the least common ancestor of an abstract heap type {type1}, and
553 // another heap type {type2}.
554 HeapType CommonAncestorWithAbstract(HeapType heap1, HeapType heap2) {
555  DCHECK(heap1.is_abstract_ref());
556  bool is_shared = heap1.is_shared();
557  if (is_shared != heap2.is_shared()) return HeapType{kWasmTop};
558 
559  // If {heap2} is an indexed type, then {heap1} could be a subtype of it if
560  // it is a none-type. In that case, {heap2} is the common ancestor.
561  std::optional<bool> is_sub = IsSubtypeOf_CommonImpl(heap1, heap2);
562  DCHECK(is_sub.has_value()); // Guaranteed by {heap1.is_abstract_ref()}.
563  if (is_sub.value()) return heap2;
564 
565  // Otherwise, we can treat {heap2} like its generic supertype (e.g. any
566  // indexed struct is "rounded up" to structref).
567  StandardType generic_ancestor = CommonAncestorLookup(
568  UpcastToStandardType(heap1), UpcastToStandardType(heap2));
569  return HeapType::Generic(ToGenericKind(generic_ancestor), is_shared);
570 }

572 Exactness UnionExactness(ValueType type1, ValueType type2,
573  const WasmModule* module) {
574  // The union of two types could be exact in these cases:
575  // - if both types are exact and the same type index
576  // - if one type is an exact indexed type, and the other is the matching
577  // none-type.
578  if (type1.is_exact()) {
579  if (type2.is_exact()) {
580  bool same =
581  EquivalentIndices(type1.ref_index(), type2.ref_index(), module);
582  return same ? Exactness::kExact : Exactness::kAnySubtype;
583  }
584  // Possibly compatible, actual subtyping check will follow.
585  if (IsNullSentinel(type2)) return Exactness::kExact;
586  } else if (type2.is_exact()) {
587  if (IsNullSentinel(type1)) return Exactness::kExact;
588  }
589  return Exactness::kAnySubtype;
590 }

594 V8_EXPORT_PRIVATE TypeInModule Union(ValueType type1, ValueType type2,
595  const WasmModule* module) {
596  if (type1 == kWasmTop || type2 == kWasmTop) return {kWasmTop, module};
597  if (type1 == kWasmBottom) return {type2, module};
598  if (type2 == kWasmBottom) return {type1, module};
599  if (!type1.is_ref() || !type2.is_ref()) {
600  return {type1 == type2 ? type1 : kWasmTop, module};
601  }
602  Nullability nullability =
603  type1.is_nullable() || type2.is_nullable() ? kNullable : kNonNullable;
604  Exactness exactness = UnionExactness(type1, type2, module);
605  HeapType heap1 = type1.heap_type();
606  HeapType heap2 = type2.heap_type();
607  if (heap1 == heap2) {
608  return {type1.AsNullable(nullability).AsExactIfIndexed(exactness), module};
609  }
610  HeapType result_type = kWasmBottom;
611  if (heap1.is_abstract_ref()) {
612  result_type = CommonAncestorWithAbstract(heap1, heap2);
613  } else if (heap2.is_abstract_ref()) {
614  result_type = CommonAncestorWithAbstract(heap2, heap1);
615  } else {
616  result_type = CommonAncestor(heap1, heap2, module);
617  }
618  // The type could only be kBottom if the input was kBottom but any kBottom
619  // HeapType should be "normalized" to kWasmBottom ValueType.
620  DCHECK_NE(result_type, kWasmBottom);
621  if (result_type.is_top()) return {kWasmTop, module};
622  return {ValueType::RefMaybeNull(result_type, nullability).AsExact(exactness),
623  module};
624 }

626 TypeInModule Intersection(ValueType type1, ValueType type2,
627  const WasmModule* module) {
628  if (type1 == kWasmTop) return {type2, module};
629  if (type2 == kWasmTop) return {type1, module};
630  if (!type1.is_ref() || !type2.is_ref()) {
631  return {type1 == type2 ? type1 : kWasmBottom, module};
632  }
633  Nullability nullability =
634  type1.is_nullable() && type2.is_nullable() ? kNullable : kNonNullable;
635  // non-nullable null type is not a valid type.
636  if (nullability == kNonNullable &&
637  (IsNullSentinel(type1) || IsNullSentinel(type2))) {
638  return {kWasmBottom, module};
639  }
640  if (IsHeapSubtypeOf(type1.heap_type(), type2.heap_type(), module)) {
641  return TypeInModule{type1.AsNullable(nullability), module};
642  }
643  if (IsHeapSubtypeOf(type2.heap_type(), type1.heap_type(), module)) {
644  return TypeInModule{type2.AsNullable(nullability), module};
645  }
646  if (nullability == kNonNullable) {
647  return {kWasmBottom, module};
648  }
649  // Check for common null representation.
650  ValueType null_type1 = ToNullSentinel({type1, module});
651  if (null_type1 == ToNullSentinel({type2, module})) {
652  return {null_type1, module};
653  }
654  return {kWasmBottom, module};
655 }

657 ValueType ToNullSentinel(TypeInModule type) {
658  HeapType null_heap = NullSentinelImpl(type.type.heap_type());
659  DCHECK(IsHeapSubtypeOf(null_heap, type.type.heap_type(), type.module));
660  return ValueType::RefNull(null_heap);
661 }
