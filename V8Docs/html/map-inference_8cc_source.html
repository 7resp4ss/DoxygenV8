17 MapInference::MapInference(JSHeapBroker* broker, Node* object, Effect effect)
18  : broker_(broker), object_(object) {
19  auto result =
20  NodeProperties::InferMapsUnsafe(broker_, object_, effect, &maps_);
21  maps_state_ = (result == NodeProperties::kUnreliableMaps)
22  ? kUnreliableDontNeedGuard
23  : kReliableOrGuarded;
24  DCHECK_EQ(maps_.is_empty(), result == NodeProperties::kNoMaps);
25 }

109 bool MapInference::RelyOnMapsPreferStability(

119 bool MapInference::RelyOnMapsHelper(CompilationDependencies* dependencies,
120  JSGraph* jsgraph, Effect* effect,
121  Control control,
122  const FeedbackSource& feedback) {
123  if (Safe()) return true;
124 
125  auto is_stable = [](MapRef map) { return map.is_stable(); };
126  if (dependencies != nullptr &&
127  std::all_of(maps_.begin(), maps_.end(), is_stable)) {
128  for (MapRef map : maps_) {
129  dependencies->DependOnStableMap(map);
130  }
131  SetGuarded();
132  return true;
133  } else if (feedback.IsValid()) {
134  InsertMapChecks(jsgraph, effect, control, feedback);
135  return true;
136  } else {
137  return false;
138  }
139 }
