25 void SourcePositionTable::SetPosition(int pc_offset, LineAndColumn pos,
26  int inlining_id) {
27  DCHECK_GE(pc_offset, 0);
28  DCHECK_GT(pos.line, 0); // The 1-based number of the source line.
29  // It's possible that we map multiple source positions to a pc_offset in
30  // optimized code. Usually these map to the same line, so there is no
31  // difference here as we only store line number and not line/col in the form
32  // of a script offset. Ignore any subsequent sets to the same offset.
33  if (!pc_offsets_to_lines_.empty() &&
34  pc_offsets_to_lines_.back().pc_offset == pc_offset) {
35  return;
36  }
37  // Check that we are inserting in ascending order, so that the vector remains
38  // sorted.
39  DCHECK(pc_offsets_to_lines_.empty() ||
40  pc_offsets_to_lines_.back().pc_offset < pc_offset);
41  if (pc_offsets_to_lines_.empty() ||
42  pc_offsets_to_lines_.back().line_number != pos.line ||
43  pc_offsets_to_lines_.back().column_number != pos.column ||
44  pc_offsets_to_lines_.back().inlining_id != inlining_id) {
45  pc_offsets_to_lines_.push_back(
46  {pc_offset, pos.line, pos.column, inlining_id});
47  }
48 }

224 size_t CodeEntry::EstimatedSize() const {
225  size_t estimated_size = 0;
226  if (rare_data_) {
227  estimated_size += sizeof(rare_data_.get());
228 
229  for (const auto& inline_entry : rare_data_->inline_entries_) {
230  estimated_size += inline_entry->EstimatedSize();
231  }
232  estimated_size += rare_data_->inline_entries_.size() *
233  sizeof(decltype(rare_data_->inline_entries_)::value_type);
234 
235  for (const auto& inline_stack_pair : rare_data_->inline_stacks_) {
236  estimated_size += inline_stack_pair.second.size() *
237  sizeof(decltype(inline_stack_pair.second)::value_type);
238  }
239  estimated_size +=
240  rare_data_->inline_stacks_.size() *
241  (sizeof(decltype(rare_data_->inline_stacks_)::key_type) +
242  sizeof(decltype(rare_data_->inline_stacks_)::value_type));
243 
244  estimated_size +=
245  rare_data_->deopt_inlined_frames_.capacity() *
246  sizeof(decltype(rare_data_->deopt_inlined_frames_)::value_type);
247  }
248 
249  if (line_info_) {
250  estimated_size += line_info_->Size();
251  }
252  return sizeof(*this) + estimated_size;
253 }

341 CpuProfileNode::SourceType ProfileNode::source_type() const {
342  // Handle metadata and VM state code entry types.
343  if (entry_ == CodeEntry::program_entry() ||
344  entry_ == CodeEntry::idle_entry() || entry_ == CodeEntry::gc_entry() ||
345  entry_ == CodeEntry::root_entry()) {
346  return CpuProfileNode::kInternal;
347  }
348  if (entry_ == CodeEntry::unresolved_entry())
349  return CpuProfileNode::kUnresolved;
350 
351  // Otherwise, resolve based on logger tag.
352  switch (entry_->code_tag()) {
353  case LogEventListener::CodeTag::kEval:
354  case LogEventListener::CodeTag::kScript:
355  case LogEventListener::CodeTag::kFunction:
356  return CpuProfileNode::kScript;
357  case LogEventListener::CodeTag::kBuiltin:
358  case LogEventListener::CodeTag::kHandler:
359  case LogEventListener::CodeTag::kBytecodeHandler:
360  case LogEventListener::CodeTag::kNativeFunction:
361  case LogEventListener::CodeTag::kNativeScript:
362  return CpuProfileNode::kBuiltin;
363  case LogEventListener::CodeTag::kCallback:
364  return CpuProfileNode::kCallback;
365  case LogEventListener::CodeTag::kRegExp:
366  case LogEventListener::CodeTag::kStub:
367  case LogEventListener::CodeTag::kLength:
368  return CpuProfileNode::kInternal;
369  }
370  return CpuProfileNode::kInternal;
371  UNREACHABLE();
372 }

419  entry->column = p->first.second;
420  entry->hit_count = p->second;
421  }

478 ProfileNode* ProfileTree::AddPathFromEnd(const std::vector<CodeEntry*>& path,
479  LineAndColumn pos, bool update_stats) {
480  ProfileNode* node = root_;
481  CodeEntry* last_entry = nullptr;
482  for (auto it = path.rbegin(); it != path.rend(); ++it) {
483  if (*it == nullptr) continue;
484  last_entry = *it;
485  node = node->FindOrAddChild(*it, LineAndColumn{});
486  }
487  if (last_entry && last_entry->has_deopt_info()) {
488  node->CollectDeoptInfo(last_entry);
489  }
490  if (update_stats) {
491  node->IncrementSelfTicks();
492  if (pos.line != v8::CpuProfileNode::kNoLineNumberInfo) {
493  node->IncrementLineAndColumnTicks(pos);
494  }
495  }
496  return node;
497 }

542 // Non-recursive implementation of a depth-first post-order tree traversal.
543 template <typename Callback>
544 void ProfileTree::TraverseDepthFirst(Callback* callback) {
545  std::vector<Position> stack;
546  stack.emplace_back(root_);
547  while (!stack.empty()) {
548  Position& current = stack.back();
549  if (current.has_current_child()) {
550  callback->BeforeTraversingChild(current.node, current.current_child());
551  stack.emplace_back(current.current_child());
552  } else {
553  callback->AfterAllChildrenTraversed(current.node);
554  if (stack.size() > 1) {
555  Position& parent = stack[stack.size() - 2];
556  callback->AfterChildTraversed(parent.node, current.node);
557  parent.next_child();
558  }
559  // Remove child from the stack.
560  stack.pop_back();
561  }
562  }
563 }

575 CpuProfile::CpuProfile(CpuProfiler* profiler, ProfilerId id, const char* title,
576  CpuProfilingOptions options,
577  std::unique_ptr<DiscardedSamplesDelegate> delegate)
578  : title_(title),
579  options_(std::move(options)),
580  delegate_(std::move(delegate)),
581  start_time_(base::TimeTicks::Now()),
582  top_down_(profiler->isolate(), profiler->code_entries()),
583  profiler_(profiler),
584  streaming_next_sample_(0),
585  id_(id) {
586  // The startTime timestamp is not converted to Perfetto's clock domain and
587  // will get out of sync with other timestamps Perfetto knows about, including
588  // the automatic trace event "ts" timestamp. startTime is included for
589  // backward compatibility with the tracing protocol but the value of "ts"
590  // should be used instead (it is recorded nearly immediately after).
591  auto value = TracedValue::Create();
592  value->SetDouble("startTime", start_time_.since_origin().InMicroseconds());
593  TRACE_EVENT_SAMPLE_WITH_ID1(TRACE_DISABLED_BY_DEFAULT("v8.cpu_profiler"),
594  "Profile", id_, "data", std::move(value));
595 
596  DisallowHeapAllocation no_gc;
597  if (delegate_) {
598  delegate_->SetId(id_);
599  }

691 void CpuProfile::StreamPendingTraceEvents() {
692  std::vector<const ProfileNode*> pending_nodes = top_down_.TakePendingNodes();
693  if (pending_nodes.empty() && samples_.empty()) return;
694  auto value = TracedValue::Create();
695 
696  if (!pending_nodes.empty() || streaming_next_sample_ != samples_.size()) {
697  value->BeginDictionary("cpuProfile");
698  if (!pending_nodes.empty()) {
699  value->BeginArray("nodes");
700  for (auto node : pending_nodes) {
701  value->BeginDictionary();
702  BuildNodeValue(node, value.get());
703  value->EndDictionary();
704  }
705  value->EndArray();
706  }
707  if (streaming_next_sample_ != samples_.size()) {
708  value->BeginArray("samples");
709  for (size_t i = streaming_next_sample_; i < samples_.size(); ++i) {
710  value->AppendInteger(samples_[i].node->id());
711  }
712  value->EndArray();
713  value->BeginDictionary("trace_ids");
714  for (size_t i = streaming_next_sample_; i < samples_.size(); ++i) {
715  if (!samples_[i].trace_id.has_value()) {
716  continue;
717  }
718  value->SetUnsignedInteger(
719  std::to_string(samples_[i].trace_id.value()).c_str(),
720  samples_[i].node->id());
721  }
722  value->EndDictionary();
723  }
724  value->EndDictionary();
725  }
726  if (streaming_next_sample_ != samples_.size()) {
727  // timeDeltas are computed within CLOCK_MONOTONIC. However, trace event
728  // "ts" timestamps are converted to CLOCK_BOOTTIME by Perfetto. To get
729  // absolute timestamps in CLOCK_BOOTTIME from timeDeltas, add them to
730  // the "ts" timestamp from the initial "Profile" trace event sent by
731  // CpuProfile::CpuProfile().
732  //
733  // Note that if the system is suspended and resumed while samples_ is
734  // captured, timeDeltas derived after resume will not be convertible to
735  // correct CLOCK_BOOTTIME time values (for instance, producing
736  // CLOCK_BOOTTIME time values in the middle of the suspended period).
737  value->BeginArray("timeDeltas");
738  base::TimeTicks lastTimestamp =
739  streaming_next_sample_ ? samples_[streaming_next_sample_ - 1].timestamp
740  : start_time();
741  for (size_t i = streaming_next_sample_; i < samples_.size(); ++i) {
742  value->AppendInteger(static_cast<int>(
743  (samples_[i].timestamp - lastTimestamp).InMicroseconds()));
744  lastTimestamp = samples_[i].timestamp;
745  }
746  value->EndArray();
747  bool has_non_zero_lines =
748  std::any_of(samples_.begin() + streaming_next_sample_, samples_.end(),
749  [](const SampleInfo& sample) {
750  return sample.line_and_column.line != 0;
751  });
752  if (has_non_zero_lines) {
753  value->BeginArray("lines");
754  for (size_t i = streaming_next_sample_; i < samples_.size(); ++i) {
755  value->AppendInteger(samples_[i].line_and_column.line);
756  }
757  value->EndArray();
758  value->BeginArray("columns");
759  for (size_t i = streaming_next_sample_; i < samples_.size(); ++i) {
760  value->AppendInteger(samples_[i].line_and_column.column);
761  }
762  value->EndArray();
763  }
764  streaming_next_sample_ = samples_.size();
765  }
766 
767  TRACE_EVENT_SAMPLE_WITH_ID1(TRACE_DISABLED_BY_DEFAULT("v8.cpu_profiler"),
768  "ProfileChunk", id_, "data", std::move(value));
769 }

771 void CpuProfile::FinishProfile() {
772  end_time_ = base::TimeTicks::Now();
773  // Stop tracking context movements after profiling stops.
774  context_filter_.set_native_context_address(kNullAddress);
775  StreamPendingTraceEvents();
776  auto value = TracedValue::Create();

778  // get out of sync with other timestamps Perfetto knows about, including the
779  // automatic trace event "ts" timestamp. endTime is included for backward

785  "ProfileChunk", id_, "data", std::move(value));
786 }

1014 CodeEntry* InstructionStreamMap::FindEntry(Address addr,
1015  Address* out_instruction_start) {
1016  // Note that an address may correspond to multiple CodeEntry objects. An
1017  // arbitrary selection is made (as per multimap spec) in the event of a

1023  Address end_address = start_address + it->second.size;

1027  return ret;

1030 void InstructionStreamMap::MoveCode(Address from, Address to) {
1031  if (from == to) return;
1032 
1033  auto range = code_map_.equal_range(from);
1034  // Instead of iterating until |range.second|, iterate the number of elements.
1035  // This is because the |range.second| may no longer be the element past the
1036  // end of the equal elements range after insertions.
1037  size_t distance = std::distance(range.first, range.second);
1038  auto it = range.first;
1039  while (distance--) {
1040  CodeEntryMapInfo& info = it->second;
1041  DCHECK(info.entry);
1042  DCHECK_EQ(info.entry->instruction_start(), from);
1043  info.entry->set_instruction_start(to);
1044 
1045  DCHECK(from + info.size <= to || to + info.size <= from);
1046  code_map_.emplace(to, info);
1047  it++;
1048  }
1049 
1050  code_map_.erase(range.first, it);
1051 }
