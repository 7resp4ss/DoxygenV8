49 void GeneratorBuiltinsAssembler::InnerResume(
50  CodeStubArguments* args, TNode<JSGeneratorObject> receiver,
51  TNode<Object> value, TNode<Context> context,
52  JSGeneratorObject::ResumeMode resume_mode, char const* const method_name) {
53  // Check if the {receiver} is running or already closed.
54  TNode<Smi> receiver_continuation =
55  LoadObjectField<Smi>(receiver, JSGeneratorObject::kContinuationOffset);
56  Label if_receiverisclosed(this, Label::kDeferred),
57  if_receiverisrunning(this, Label::kDeferred);
58  TNode<Smi> closed = SmiConstant(JSGeneratorObject::kGeneratorClosed);
59  GotoIf(SmiEqual(receiver_continuation, closed), &if_receiverisclosed);
60  DCHECK_LT(JSGeneratorObject::kGeneratorExecuting,
61  JSGeneratorObject::kGeneratorClosed);
62  GotoIf(SmiLessThan(receiver_continuation, closed), &if_receiverisrunning);
63 
64  // Remember the {resume_mode} for the {receiver}.
65  StoreObjectFieldNoWriteBarrier(receiver, JSGeneratorObject::kResumeModeOffset,
66  SmiConstant(resume_mode));
67 
68  // Resume the {receiver} using our trampoline.
69  // Close the generator if there was an exception.
70  TVARIABLE(Object, var_exception);
71  Label if_exception(this, Label::kDeferred), if_final_return(this);
72  TNode<JSAny> result;
73  {
74  compiler::ScopedExceptionHandler handler(this, &if_exception,
75  &var_exception);
76  result = CallBuiltin<JSAny>(Builtin::kResumeGeneratorTrampoline, context,
77  value, receiver);
78  }
79 
80  // If the generator is not suspended (i.e., its state is 'executing'),
81  // close it and wrap the return value in IteratorResult.
82  TNode<Smi> result_continuation =
83  LoadObjectField<Smi>(receiver, JSGeneratorObject::kContinuationOffset);
84 
85  // The generator function should not close the generator by itself, let's
86  // check it is indeed not closed yet.
87  CSA_DCHECK(this, SmiNotEqual(result_continuation, closed));
88 
89  TNode<Smi> executing = SmiConstant(JSGeneratorObject::kGeneratorExecuting);
90  GotoIf(SmiEqual(result_continuation, executing), &if_final_return);
91 
92  args->PopAndReturn(result);
93 
94  BIND(&if_final_return);
95  {
96  // Close the generator.
97  StoreObjectFieldNoWriteBarrier(
98  receiver, JSGeneratorObject::kContinuationOffset, closed);
99  // Return the wrapped result.
100  args->PopAndReturn(CallBuiltin<JSAny>(Builtin::kCreateIterResultObject,
101  context, result, TrueConstant()));
102  }
103 
104  BIND(&if_receiverisclosed);
105  {
106  // The {receiver} is closed already.
107  TNode<JSAny> builtin_result;
108  switch (resume_mode) {
109  case JSGeneratorObject::kNext:
110  builtin_result =
111  CallBuiltin<JSAny>(Builtin::kCreateIterResultObject, context,
112  UndefinedConstant(), TrueConstant());
113  break;
114  case JSGeneratorObject::kReturn:
115  builtin_result = CallBuiltin<JSAny>(Builtin::kCreateIterResultObject,
116  context, value, TrueConstant());
117  break;
118  case JSGeneratorObject::kThrow:
119  builtin_result = CallRuntime<JSAny>(Runtime::kThrow, context, value);
120  break;
121  case JSGeneratorObject::kRethrow:
122  // Currently only async generators use this mode.
123  UNREACHABLE();
124  }
125  args->PopAndReturn(builtin_result);
126  }
127 
128  BIND(&if_receiverisrunning);
129  { ThrowTypeError(context, MessageTemplate::kGeneratorRunning); }
130 
131  BIND(&if_exception);
132  {
133  StoreObjectFieldNoWriteBarrier(
134  receiver, JSGeneratorObject::kContinuationOffset, closed);
135  CallRuntime(Runtime::kReThrow, context, var_exception.value());
136  Unreachable();
137  }
138 }

228  // the inspector.
229  auto bytecode_offset =
230  SmiTag(UncheckedParameter<IntPtrT>(Descriptor::kBytecodeOffset));
231  // Avoid the write barrier by using the generic helper.
232  StoreObjectFieldNoWriteBarrier(
233  generator, JSGeneratorObject::kInputOrDebugPosOffset, bytecode_offset);

236  LoadJSGeneratorObjectParametersAndRegisters(generator);
237  auto parameters_and_registers_length =
238  LoadAndUntagFixedArrayBaseLength(parameters_and_registers);
239 
240  // Copy over the function parameters
241  auto parameter_base_index = IntPtrConstant(
242  interpreter::Register::FromParameterIndex(0).ToOperand() + 1);
243  CSA_CHECK(this,
244  UintPtrLessThan(parameter_count, parameters_and_registers_length));
245  auto parent_frame_pointer = LoadParentFramePointer();
246  BuildFastLoop<IntPtrT>(
247  IntPtrConstant(0), parameter_count,
248  [=, this](TNode<IntPtrT> index) {
249  auto reg_index = IntPtrAdd(parameter_base_index, index);
250  TNode<Object> value = LoadFullTagged(parent_frame_pointer,
251  TimesSystemPointerSize(reg_index));
252  UnsafeStoreFixedArrayElement(parameters_and_registers, index, value);
253  },
254  1, LoopUnrollingMode::kNo, IndexAdvanceMode::kPost);
255 
256  // Iterate over register file and write values into array.
257  // The mapping of register to array index must match that used in
258  // BytecodeGraphBuilder::VisitResumeGenerator.
259  auto register_base_index = IntPtrAdd(
260  parameter_count, IntPtrConstant(interpreter::Register(0).ToOperand()));

262  auto end_index = IntPtrAdd(parameter_count, register_count);
263  CSA_CHECK(this, UintPtrLessThan(end_index, parameters_and_registers_length));
264  BuildFastLoop<IntPtrT>(
265  parameter_count, end_index,
266  [=, this](TNode<IntPtrT> index) {
267  auto reg_index = IntPtrSub(register_base_index, index);

269  TimesSystemPointerSize(reg_index));
270  UnsafeStoreFixedArrayElement(parameters_and_registers, index, value);
271  },
272  1, LoopUnrollingMode::kNo, IndexAdvanceMode::kPost);
273 
274  // The return value is unused, defaulting to undefined.
275  Return(UndefinedConstant());

292  auto parameters_and_registers_length =
293  LoadAndUntagFixedArrayBaseLength(parameters_and_registers);
294  CSA_CHECK(this, UintPtrLessThan(end_index, parameters_and_registers_length));
295  auto parent_frame_pointer = LoadParentFramePointer();
296  BuildFastLoop<IntPtrT>(
297  parameter_count, end_index,
298  [=, this](TNode<IntPtrT> index) {
299  TNode<Object> value =
300  UnsafeLoadFixedArrayElement(parameters_and_registers, index);
301  auto reg_index = IntPtrSub(register_base_index, index);
302  StoreFullTaggedNoWriteBarrier(parent_frame_pointer,
303  TimesSystemPointerSize(reg_index), value);
304  UnsafeStoreFixedArrayElement(parameters_and_registers, index,
305  StaleRegisterConstant(),
306  SKIP_WRITE_BARRIER);
307  },
308  1, LoopUnrollingMode::kNo, IndexAdvanceMode::kPost);
