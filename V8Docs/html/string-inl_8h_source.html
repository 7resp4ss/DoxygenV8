48 class V8_NODISCARD SharedStringAccessGuardIfNeeded {
49  public:
50  // Creates no MutexGuard for the string access since it was
51  // called from the main thread.
52  explicit SharedStringAccessGuardIfNeeded(Isolate* isolate) {}
53 
54  // Creates a MutexGuard for the string access if it was called
55  // from a background thread.
56  explicit SharedStringAccessGuardIfNeeded(LocalIsolate* local_isolate) {
57  if (IsNeeded(local_isolate)) {
58  mutex_guard.emplace(local_isolate->internalized_string_access());
59  }
60  }
61 
62  // Slow version which gets the isolate from the String.
63  explicit SharedStringAccessGuardIfNeeded(Tagged<String> str) {
64  Isolate* isolate = GetIsolateIfNeeded(str);
65  if (isolate != nullptr) {
66  mutex_guard.emplace(isolate->internalized_string_access());
67  }
68  }
69 
70  SharedStringAccessGuardIfNeeded(Tagged<String> str,
71  LocalIsolate* local_isolate) {
72  if (IsNeeded(str, local_isolate)) {
73  mutex_guard.emplace(local_isolate->internalized_string_access());
74  }
75  }
76 
77  static SharedStringAccessGuardIfNeeded NotNeeded() {
78  return SharedStringAccessGuardIfNeeded();
79  }
80 
81  static bool IsNeeded(Tagged<String> str, LocalIsolate* local_isolate) {
82  return IsNeeded(local_isolate) && IsNeeded(str, false);
83  }
84 
85  static bool IsNeeded(Tagged<String> str, bool check_local_heap = true) {
86  if (check_local_heap) {
87  LocalHeap* local_heap = LocalHeap::Current();
88  if (!local_heap || local_heap->is_main_thread()) {
89  // Don't acquire the lock for the main thread.
90  return false;
91  }
92  }
93 
94  if (ReadOnlyHeap::Contains(str)) {
95  // Don't acquire lock for strings in ReadOnlySpace.
96  return false;
97  }
98 
99  return true;
100  }
101 
102  static bool IsNeeded(LocalIsolate* local_isolate) {
103  // TODO(leszeks): Remove the nullptr check for local_isolate.
104  return local_isolate && !local_isolate->heap()->is_main_thread();
105  }
106 
107  private:
108  // Default constructor and move constructor required for the NotNeeded()
109  // static constructor.
110  constexpr SharedStringAccessGuardIfNeeded() = default;
111  constexpr SharedStringAccessGuardIfNeeded(SharedStringAccessGuardIfNeeded&&)
112  V8_NOEXCEPT {
113  DCHECK(!mutex_guard.has_value());
114  }
115 
116  // Returns the Isolate from the String if we need it for the lock.
117  static Isolate* GetIsolateIfNeeded(Tagged<String> str) {
118  if (!IsNeeded(str)) return nullptr;
119 
120  Isolate* isolate;
121  if (!GetIsolateFromHeapObject(str, &isolate)) {
122  // If we can't get the isolate from the String, it must be read-only.
123  DCHECK(ReadOnlyHeap::Contains(str));
124  return nullptr;
125  }
126  return isolate;
127  }
128 
129  std::optional<base::MutexGuard> mutex_guard;
130 };

222 inline bool StringShape::IsValidFor(Tagged<String> string) const {
223  Tagged<Map> map = string->map(kAcquireLoad);
224 #if V8_STATIC_ROOTS_BOOL
225  if (map_ == map) return true;
226 #else
227  InstanceType type = map->instance_type();
228  if (type_ == type) return true;
229 #endif
230  if (!v8_flags.shared_string_table) return false;
231 
232  // If the shared string table is enabled, we may observe a concurrent
233  // conversion from shared to internalized. Make sure that the two shapes are
234  // compatible.
235 #if V8_STATIC_ROOTS_BOOL
236  // Since the two maps are not equal, one must be a shared string and the
237  // other an internalized string, in exactly that combination. All other
238  // properties (sequential vs external, one vs two byte) should be the same.
239  // The following transitions are the only possible ones -- in particular,
240  // shared uncached external strings cannot be internalized in-place.
241  Tagged_t before_map_val = V8HeapCompressionScheme::CompressObject(map_.ptr());
242  Tagged_t after_map_val = V8HeapCompressionScheme::CompressObject(map.ptr());
243  if (before_map_val == StaticReadOnlyRoot::kSharedSeqOneByteStringMap) {
244  return after_map_val == StaticReadOnlyRoot::kInternalizedOneByteStringMap;
245  }
246  if (before_map_val == StaticReadOnlyRoot::kSharedSeqTwoByteStringMap) {
247  return after_map_val == StaticReadOnlyRoot::kInternalizedTwoByteStringMap;
248  }
249  if (before_map_val == StaticReadOnlyRoot::kSharedExternalOneByteStringMap) {
250  return after_map_val ==
251  StaticReadOnlyRoot::kExternalInternalizedOneByteStringMap;
252  }
253  if (before_map_val == StaticReadOnlyRoot::kSharedExternalTwoByteStringMap) {
254  return after_map_val ==
255  StaticReadOnlyRoot::kExternalInternalizedTwoByteStringMap;
256  }
257  return false;
258 #else
259  // Since the two types are not equal, one must be a shared string and the
260  // other an internalized string, in exactly that combination. All other
261  // properties (sequential vs external, one vs two byte) should be the same,
262  // so the XOR of the two instance types should be precisely
263  // `kSharedStringTag | kNotInternalizedTag`.
264  static_assert(
265  (INTERNALIZED_ONE_BYTE_STRING_TYPE ^ SHARED_SEQ_ONE_BYTE_STRING_TYPE) ==
266  (kSharedStringTag | kNotInternalizedTag));
267  return (type_ ^ type) == (kSharedStringTag | kNotInternalizedTag);
268 #endif
269 }
270 #endif

290  static constexpr bool kHasAnyNullOpt =
291  std::disjunction_v<std::is_same<std::nullopt_t, Ts>...>;
292  using type =
293  std::conditional_t<kHasAnyNullOpt,
294  // If there is a nullopt, common_type_handle_nullopt is

305 V8_NOINLINE V8_PRESERVE_MOST bool TryReportUnreachable(Tagged<String> string,
306  Tagged<Map> map) {
307  thread_local int recursion = 0;
308  if (recursion > 0) {
309  // On a recursive failure, dispatch onto the empty string. This will
310  // likely cause out-of-bounds reads or potentially some other failure, but
311  // this is ok since we're already dying and it prevents stack overflow.
312  return false;
313  }
314  recursion++;
315  Isolate::Current()->PushStackTraceAndDie(
316  reinterpret_cast<void*>(string->ptr()),
317  reinterpret_cast<void*>(map->ptr()));
318  recursion--;
319  UNREACHABLE();
320 }

324 template <typename TDispatcher>
325 auto StringShape::DispatchToSpecificType(Tagged<String> string,
326  TDispatcher&& dispatcher) const {
327  // Figure out a common return type from the possible dispatcher overloads.
328  using TReturn = typename detail::common_type_handle_nullopt<
329  decltype(dispatcher(Tagged<SeqOneByteString>{})),
330  decltype(dispatcher(Tagged<SeqTwoByteString>{})),
331  decltype(dispatcher(Tagged<ExternalOneByteString>{})),
332  decltype(dispatcher(Tagged<ExternalTwoByteString>{})),
333  decltype(dispatcher(Tagged<ThinString>{})),
334  decltype(dispatcher(Tagged<ConsString>{})),
335  decltype(dispatcher(Tagged<SlicedString>{}))>::type;
336 
337  // The following code inlines the dispatcher calls with V8_INLINE_STATEMENT.
338  // This is so that this behaves, as far as the caller is concerned, like an
339  // inlined type switch.
340  DCHECK(IsValidFor(string));
341 
342 #if V8_STATIC_ROOTS_BOOL
343  // Check the string map ranges in dense increasing order, to avoid needing
344  // to subtract away the lower bound. Don't use the InstanceTypeChecker::IsFoo
345  // helpers, because clang doesn't realise it can avoid the subtraction.
346  Tagged_t map = V8HeapCompressionScheme::CompressObject(map_.ptr());
347 
348  using StringTypeRange = InstanceTypeChecker::kUniqueMapRangeOfStringType;
349  static_assert(StringTypeRange::kSeqString.first == 0);
350  if (map <= StringTypeRange::kSeqString.second) {
351  if ((map & InstanceTypeChecker::kStringMapEncodingMask) ==
352  InstanceTypeChecker::kOneByteStringMapBit) {
353  V8_INLINE_STATEMENT return static_cast<TReturn>(
354  dispatcher(UncheckedCast<SeqOneByteString>(string)));
355  } else {
356  V8_INLINE_STATEMENT return static_cast<TReturn>(
357  dispatcher(UncheckedCast<SeqTwoByteString>(string)));
358  }
359  }
360 
361  static_assert(StringTypeRange::kSeqString.second + Map::kSize ==
362  StringTypeRange::kExternalString.first);
363  if (map <= StringTypeRange::kExternalString.second) {
364  if ((map & InstanceTypeChecker::kStringMapEncodingMask) ==
365  InstanceTypeChecker::kOneByteStringMapBit) {
366  V8_INLINE_STATEMENT return static_cast<TReturn>(
367  dispatcher(UncheckedCast<ExternalOneByteString>(string)));
368  } else {
369  V8_INLINE_STATEMENT return static_cast<TReturn>(
370  dispatcher(UncheckedCast<ExternalTwoByteString>(string)));
371  }
372  }
373  static_assert(StringTypeRange::kExternalString.second + Map::kSize ==
374  StringTypeRange::kConsString.first);
375  if (map <= StringTypeRange::kConsString.second) {
376  V8_INLINE_STATEMENT return static_cast<TReturn>(
377  dispatcher(UncheckedCast<ConsString>(string)));
378  }
379 
380  static_assert(StringTypeRange::kConsString.second + Map::kSize ==
381  StringTypeRange::kSlicedString.first);
382  if (map <= StringTypeRange::kSlicedString.second) {
383  V8_INLINE_STATEMENT return static_cast<TReturn>(
384  dispatcher(UncheckedCast<SlicedString>(string)));
385  }
386 
387  static_assert(StringTypeRange::kSlicedString.second + Map::kSize ==
388  StringTypeRange::kThinString.first);
389  if (map <= StringTypeRange::kThinString.second) {
390  V8_INLINE_STATEMENT return static_cast<TReturn>(
391  dispatcher(UncheckedCast<ThinString>(string)));
392  }
393 
394  [[unlikely]] if (!TryReportUnreachable(string, map_)) {
395  return static_cast<TReturn>(dispatcher(
396  UncheckedCast<SeqOneByteString>(GetReadOnlyRoots().empty_string())));
397  }
398  UNREACHABLE();
399 #else
400  switch (type_ & kStringRepresentationAndEncodingMask) {
401  case kSeqStringTag | kOneByteStringTag:
402  V8_INLINE_STATEMENT return static_cast<TReturn>(
403  dispatcher(UncheckedCast<SeqOneByteString>(string)));
404  case kSeqStringTag | kTwoByteStringTag:
405  V8_INLINE_STATEMENT return static_cast<TReturn>(
406  dispatcher(UncheckedCast<SeqTwoByteString>(string)));
407  case kConsStringTag | kOneByteStringTag:
408  case kConsStringTag | kTwoByteStringTag:
409  V8_INLINE_STATEMENT return static_cast<TReturn>(
410  dispatcher(UncheckedCast<ConsString>(string)));
411  case kExternalStringTag | kOneByteStringTag:
412  V8_INLINE_STATEMENT return static_cast<TReturn>(
413  dispatcher(UncheckedCast<ExternalOneByteString>(string)));
414  case kExternalStringTag | kTwoByteStringTag:
415  V8_INLINE_STATEMENT return static_cast<TReturn>(
416  dispatcher(UncheckedCast<ExternalTwoByteString>(string)));
417  case kSlicedStringTag | kOneByteStringTag:
418  case kSlicedStringTag | kTwoByteStringTag:
419  V8_INLINE_STATEMENT return static_cast<TReturn>(
420  dispatcher(UncheckedCast<SlicedString>(string)));
421  case kThinStringTag | kOneByteStringTag:
422  case kThinStringTag | kTwoByteStringTag:
423  V8_INLINE_STATEMENT return static_cast<TReturn>(
424  dispatcher(UncheckedCast<ThinString>(string)));
425  default:
426  UNREACHABLE();
427  }
428  UNREACHABLE();
429 #endif
430 }

729 template <String::EqualityType kEqType, typename Char>
730 bool String::IsEqualToImpl(
731  base::Vector<const Char> str,
732  const SharedStringAccessGuardIfNeeded& access_guard) const {
733  size_t len = str.size();
734  switch (kEqType) {
735  case EqualityType::kWholeString:
736  if (static_cast<size_t>(length()) != len) return false;
737  break;
738  case EqualityType::kPrefix:
739  if (static_cast<size_t>(length()) < len) return false;
740  break;
741  case EqualityType::kNoLengthCheck:
742  DCHECK_EQ(length(), len);
743  break;
744  }
745 
746  DisallowGarbageCollection no_gc;
747 
748  int slice_offset = 0;
749  Tagged<String> string = this;
750  const Char* data = str.data();
751  while (true) {
752  auto ret = string->DispatchToSpecificType(absl::Overload{
753  [&](Tagged<SeqOneByteString> s) {
754  return CompareCharsEqual(
755  s->GetChars(no_gc, access_guard) + slice_offset, data, len);
756  },
757  [&](Tagged<SeqTwoByteString> s) {
758  return CompareCharsEqual(
759  s->GetChars(no_gc, access_guard) + slice_offset, data, len);
760  },
761  [&](Tagged<ExternalOneByteString> s) {
762  return CompareCharsEqual(s->GetChars() + slice_offset, data, len);
763  },
764  [&](Tagged<ExternalTwoByteString> s) {
765  return CompareCharsEqual(s->GetChars() + slice_offset, data, len);
766  },
767  [&](Tagged<SlicedString> s) {
768  slice_offset += s->offset();
769  string = s->parent();
770  return std::nullopt;
771  },
772  [&](Tagged<ConsString> s) {
773  // The ConsString path is more complex and rare, so call out to an
774  // out-of-line handler.
775  // Slices cannot refer to ConsStrings, so there cannot be a non-zero
776  // slice offset here.
777  DCHECK_EQ(slice_offset, 0);
778  return IsConsStringEqualToImpl<Char>(s, str, access_guard);
779  },
780  [&](Tagged<ThinString> s) {
781  string = s->actual();
782  return std::nullopt;
783  }});
784  if (ret) return ret.value();
785  }
786 }

788 // static
789 template <typename Char>
790 bool String::IsConsStringEqualToImpl(
791  Tagged<ConsString> string, base::Vector<const Char> str,
792  const SharedStringAccessGuardIfNeeded& access_guard) {
793  // Already checked the len in IsEqualToImpl. Check GE rather than EQ in case
794  // this is a prefix check.
795  DCHECK_GE(string->length(), str.size());
796 
797  ConsStringIterator iter(Cast<ConsString>(string));
798  base::Vector<const Char> remaining_str = str;
799  int offset;
800  for (Tagged<String> segment = iter.Next(&offset); !segment.is_null();
801  segment = iter.Next(&offset)) {
802  // We create the iterator without an offset, so we should never have a
803  // per-segment offset.
804  DCHECK_EQ(offset, 0);
805  // Compare the individual segment against the appropriate subvector of the
806  // remaining string.
807  size_t len = std::min<size_t>(segment->length(), remaining_str.size());
808  base::Vector<const Char> sub_str = remaining_str.SubVector(0, len);
809  if (!segment->IsEqualToImpl<EqualityType::kNoLengthCheck>(sub_str,
810  access_guard)) {
811  return false;
812  }
813  remaining_str += len;
814  if (remaining_str.empty()) break;
815  }
816  DCHECK_EQ(remaining_str.data(), str.end());
817  DCHECK_EQ(remaining_str.size(), 0);
818  return true;
819 }

851 // static
852 template <template <typename> typename HandleType>
853  requires(std::is_convertible_v<HandleType<String>, DirectHandle<String>>)
854 V8_EXPORT_PRIVATE HandleType<String> String::SlowFlatten(
855  Isolate* isolate, HandleType<ConsString> cons, AllocationType allocation) {
856  DCHECK(!cons->IsFlat());
857  DCHECK_NE(cons->second()->length(), 0); // Equivalent to !IsFlat.
858  DCHECK(!HeapLayout::InAnySharedSpace(*cons));
859 
860  bool is_one_byte_representation;
861  uint32_t length;
862 
863  {
864  DisallowGarbageCollection no_gc;
865  Tagged<ConsString> raw_cons = *cons;
866 
867  // TurboFan can create cons strings with empty first parts. Make sure the
868  // cons shape is canonicalized by the end of this function (either here, if
869  // returning early, or below). Note this case is very rare in practice.
870  if (V8_UNLIKELY(raw_cons->first()->length() == 0)) {
871  Tagged<String> second = raw_cons->second();
872  if (StringShape{second}.IsSequential()) {
873  raw_cons->set_first(second);
874  raw_cons->set_second(ReadOnlyRoots(isolate).empty_string());
875  DCHECK(raw_cons->IsFlat());
876  return HandleType<String>(second, isolate);
877  }
878  // Note that the remaining subtree may still be non-flat and we thus
879  // need to continue below.
880  }
881 
882  if (V8_LIKELY(allocation != AllocationType::kSharedOld)) {
883  if (!HeapLayout::InYoungGeneration(raw_cons)) {
884  allocation = AllocationType::kOld;
885  }
886  }
887  length = raw_cons->length();
888  is_one_byte_representation = cons->IsOneByteRepresentation();
889  }
890 
891  DCHECK_EQ(length, cons->length());
892  DCHECK_EQ(is_one_byte_representation, cons->IsOneByteRepresentation());
893  DCHECK(AllowGarbageCollection::IsAllowed());
894 
895  HandleType<SeqString> result;
896  if (is_one_byte_representation) {
897  HandleType<SeqOneByteString> flat =
898  isolate->factory()
899  ->NewRawOneByteString(length, allocation)
900  .ToHandleChecked();
901  // When the ConsString had a forwarding index, it is possible that it was
902  // transitioned to a ThinString (and eventually shortcutted to
903  // InternalizedString) during GC.
904  if constexpr (v8_flags.always_use_string_forwarding_table.value()) {
905  if (!IsConsString(*cons)) {
906  DCHECK(IsInternalizedString(*cons) || IsThinString(*cons));
907  return String::Flatten(isolate, cons, allocation);
908  }
909  }
910  DisallowGarbageCollection no_gc;
911  Tagged<ConsString> raw_cons = *cons;
912  WriteToFlat2(flat->GetChars(no_gc), raw_cons, 0, length,
913  SharedStringAccessGuardIfNeeded::NotNeeded(), no_gc);
914  raw_cons->set_first(*flat);
915  raw_cons->set_second(ReadOnlyRoots(isolate).empty_string());
916  result = flat;
917  } else {
918  HandleType<SeqTwoByteString> flat =
919  isolate->factory()
920  ->NewRawTwoByteString(length, allocation)
921  .ToHandleChecked();
922  // When the ConsString had a forwarding index, it is possible that it was
923  // transitioned to a ThinString (and eventually shortcutted to
924  // InternalizedString) during GC.
925  if constexpr (v8_flags.always_use_string_forwarding_table.value()) {
926  if (!IsConsString(*cons)) {
927  DCHECK(IsInternalizedString(*cons) || IsThinString(*cons));
928  return String::Flatten(isolate, cons, allocation);
929  }
930  }
931  DisallowGarbageCollection no_gc;
932  Tagged<ConsString> raw_cons = *cons;
933  WriteToFlat2(flat->GetChars(no_gc), raw_cons, 0, length,
934  SharedStringAccessGuardIfNeeded::NotNeeded(), no_gc);
935  raw_cons->set_first(*flat);
936  raw_cons->set_second(ReadOnlyRoots(isolate).empty_string());
937  result = flat;
938  }
939  DCHECK(result->IsFlat());
940  DCHECK(cons->IsFlat());

942 }

947 template <typename T, template <typename> typename HandleType>
948  requires(std::is_convertible_v<HandleType<T>, DirectHandle<String>>)
949 HandleType<String> String::Flatten(Isolate* isolate, HandleType<T> string,
950  AllocationType allocation) {
951  DisallowGarbageCollection no_gc; // Unhandlified code.
952  Tagged<String> s = *string;
953  StringShape shape(s);
954 
955  // Shortcut already-flat strings.
956  if (V8_LIKELY(shape.IsDirect())) return string;
957 
958  if (shape.IsCons()) {
959  DCHECK(!HeapLayout::InAnySharedSpace(s));
960  Tagged<ConsString> cons = Cast<ConsString>(s);
961  if (!cons->IsFlat()) {
962  AllowGarbageCollection yes_gc;

965  SlowFlatten(isolate, Cast<ConsString>(string), allocation);
966  DCHECK(result->IsFlat());

971  shape = StringShape(s);
972  }

975  s = Cast<ThinString>(s)->actual();
976  DCHECK(!IsConsString(s));
977  }

980  DCHECK(string->IsFlat()); // In-place flattened.
981  return HandleType<String>(s, isolate);

1742  uint32_t remaining_chars = n_chars;
1743  uint16_t last = unibrow::Utf16::kNoPreviousCharacter;
1744  while (HasMore() && remaining_chars-- != 0) {
1745  uint16_t character = GetNext();

1753  // in-sandbox memory corruption.

1759
