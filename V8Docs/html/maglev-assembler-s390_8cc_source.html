29 template <typename T>
30 void AllocateRaw(MaglevAssembler* masm, Isolate* isolate,
31  RegisterSnapshot register_snapshot, Register object,
32  T size_in_bytes, AllocationType alloc_type,
33  AllocationAlignment alignment) {
34  // TODO(victorgomes): Call the runtime for large object allocation.
35  // TODO(victorgomes): Support double alignment.
36  DCHECK(masm->allow_allocate());
37  DCHECK_EQ(alignment, kTaggedAligned);
38  if (v8_flags.single_generation) {
39  alloc_type = AllocationType::kOld;
40  }
41  ExternalReference top = SpaceAllocationTopAddress(isolate, alloc_type);
42  ExternalReference limit = SpaceAllocationLimitAddress(isolate, alloc_type);
43  ZoneLabelRef done(masm);
44  MaglevAssembler::TemporaryRegisterScope temps(masm);
45  Register scratch = temps.AcquireScratch();
46  // We are a bit short on registers, so we use the same register for {object}
47  // and {new_top}. Once we have defined {new_top}, we don't use {object} until
48  // {new_top} is used for the last time. And there (at the end of this
49  // function), we recover the original {object} from {new_top} by subtracting
50  // {size_in_bytes}.
51  Register new_top = object;
52  // Check if there is enough space.
53  __ LoadU64(object, __ ExternalReferenceAsOperand(top, scratch));
54  __ AddU64(object, size_in_bytes);
55  __ LoadU64(scratch, __ ExternalReferenceAsOperand(limit, scratch));
56  __ CmpU64(new_top, scratch);
57  // Otherwise call runtime.
58  __ JumpToDeferredIf(kUnsignedGreaterThanEqual, AllocateSlow<T>,
59  register_snapshot, object, AllocateBuiltin(alloc_type),
60  size_in_bytes, done);
61  // Store new top and tag object.
62  __ Move(__ ExternalReferenceAsOperand(top, scratch), new_top);
63  SubSizeAndTagObject(masm, object, size_in_bytes);
64  __ bind(*done);
65 }

84 void MaglevAssembler::OSRPrologue(Graph* graph) {
85  TemporaryRegisterScope temps(this);
86  Register scratch = temps.AcquireScratch();
87 
88  DCHECK(graph->is_osr());
89  CHECK(!graph->has_recursive_calls());
90 
91  uint32_t source_frame_size =
92  graph->min_maglev_stackslots_for_unoptimized_frame_size();
93 
94  if (v8_flags.debug_code) {
95  lgr(scratch, sp);
96  lay(scratch,
97  MemOperand(scratch, source_frame_size * kSystemPointerSize +
98  StandardFrameConstants::kFixedFrameSizeFromFp));
99  CmpU64(scratch, fp);
100  Assert(eq, AbortReason::kOsrUnexpectedStackSize);
101  }
102 
103  uint32_t target_frame_size =
104  graph->tagged_stack_slots() + graph->untagged_stack_slots();
105  CHECK_LE(source_frame_size, target_frame_size);
106 
107  if (source_frame_size < target_frame_size) {
108  ASM_CODE_COMMENT_STRING(this, "Growing frame for OSR");
109  uint32_t additional_tagged =
110  source_frame_size < graph->tagged_stack_slots()
111  ? graph->tagged_stack_slots() - source_frame_size
112  : 0;
113  if (additional_tagged) {
114  Move(scratch, 0);
115  }
116  for (size_t i = 0; i < additional_tagged; ++i) {
117  Push(scratch);
118  }
119  uint32_t size_so_far = source_frame_size + additional_tagged;
120  CHECK_LE(size_so_far, target_frame_size);
121  if (size_so_far < target_frame_size) {
122  lay(sp, MemOperand(
123  sp, -(target_frame_size - size_so_far) * kSystemPointerSize));
124  }
125  }
126 }

128 void MaglevAssembler::Prologue(Graph* graph) {
129  TemporaryRegisterScope temps(this);
130  temps.Include({r6, r8});
131  Register scratch = temps.AcquireScratch();
132  DCHECK(!graph->is_osr());
133 
134  BailoutIfDeoptimized(scratch);
135 
136  if (graph->has_recursive_calls()) {
137  bind(code_gen_state()->entry_label());
138  }
139 
140 #ifndef V8_ENABLE_LEAPTIERING
141  // Tiering support.
142  if (v8_flags.turbofan) {
143  using D = MaglevOptimizeCodeOrTailCallOptimizedCodeSlotDescriptor;
144  Register flags = D::GetRegisterParameter(D::kFlags);
145  Register feedback_vector = D::GetRegisterParameter(D::kFeedbackVector);
146  DCHECK(!AreAliased(feedback_vector, kJavaScriptCallArgCountRegister,
147  kJSFunctionRegister, kContextRegister,
148  kJavaScriptCallNewTargetRegister));
149  DCHECK(!temps.Available().has(flags));
150  DCHECK(!temps.Available().has(feedback_vector));
151  Move(feedback_vector,
152  compilation_info()->toplevel_compilation_unit()->feedback().object());
153  TailCallBuiltin(Builtin::kMaglevOptimizeCodeOrTailCallOptimizedCodeSlot,
154  LoadFeedbackVectorFlagsAndCheckIfNeedsProcessing(
155  flags, feedback_vector, CodeKind::MAGLEV));
156  }
157 #endif // !V8_ENABLE_LEAPTIERING
158 
159  EnterFrame(StackFrame::MAGLEV);
160  // Save arguments in frame.
161  // TODO(leszeks): Consider eliding this frame if we don't make any calls
162  // that could clobber these registers.
163  Push(kContextRegister);
164  Push(kJSFunctionRegister); // Callee's JS function.
165  Push(kJavaScriptCallArgCountRegister); // Actual argument count.
166 
167  // Initialize stack slots.
168  if (graph->tagged_stack_slots() > 0) {
169  ASM_CODE_COMMENT_STRING(this, "Initializing stack slots");
170  Move(scratch, 0);
171 
172  // Magic value. Experimentally, an unroll size of 8 doesn't seem any
173  // worse than fully unrolled pushes.
174  const int kLoopUnrollSize = 8;
175  int tagged_slots = graph->tagged_stack_slots();
176  if (tagged_slots < kLoopUnrollSize) {
177  // If the frame is small enough, just unroll the frame fill
178  // completely.
179  for (int i = 0; i < tagged_slots; ++i) {
180  Push(scratch);
181  }
182  } else {
183  // Extract the first few slots to round to the unroll size.
184  int first_slots = tagged_slots % kLoopUnrollSize;
185  for (int i = 0; i < first_slots; ++i) {
186  Push(scratch);
187  }
188  Register unroll_counter = temps.AcquireScratch();
189  Move(unroll_counter, tagged_slots / kLoopUnrollSize);
190  // We enter the loop unconditionally, so make sure we need to loop at
191  // least once.
192  DCHECK_GT(tagged_slots / kLoopUnrollSize, 0);
193  Label loop;
194  bind(&loop);
195  for (int i = 0; i < kLoopUnrollSize; ++i) {
196  Push(scratch);
197  }
198  SubS32(unroll_counter, Operand(1));
199  bgt(&loop);
200  }
201  }
202  if (graph->untagged_stack_slots() > 0) {
203  // Extend rsp by the size of the remaining untagged part of the frame,
204  // no need to initialise these.
205  lay(sp,
206  MemOperand(sp, -graph->untagged_stack_slots() * kSystemPointerSize));
207  }
208 }

231 void MaglevAssembler::StringFromCharCode(RegisterSnapshot register_snapshot,
232  Label* char_code_fits_one_byte,
233  Register result, Register char_code,
234  Register scratch,
235  CharCodeMaskMode mask_mode) {
236  AssertZeroExtended(char_code);
237  DCHECK_NE(char_code, scratch);
238  ZoneLabelRef done(this);
239  if (mask_mode == CharCodeMaskMode::kMustApplyMask) {
240  AndP(char_code, char_code, Operand(0xFFFF));
241  }
242  CmpU32(char_code, Operand(String::kMaxOneByteCharCode));
243  JumpToDeferredIf(
244  kUnsignedGreaterThan,
245  [](MaglevAssembler* masm, RegisterSnapshot register_snapshot,
246  ZoneLabelRef done, Register result, Register char_code,
247  Register scratch) {
248  // Be sure to save {char_code}. If it aliases with {result}, use
249  // the scratch register.
250  // TODO(victorgomes): This is probably not needed any more, because
251  // we now ensure that results registers don't alias with inputs/temps.
252  // Confirm, and drop this check.
253  if (char_code == result) {
254  __ Move(scratch, char_code);
255  char_code = scratch;
256  }
257  DCHECK(char_code != result);
258  DCHECK(!register_snapshot.live_tagged_registers.has(char_code));
259  register_snapshot.live_registers.set(char_code);
260  __ AllocateTwoByteString(register_snapshot, result, 1);
261  __ StoreU16(
262  char_code,
263  FieldMemOperand(result, OFFSET_OF_DATA_START(SeqTwoByteString)));
264  __ b(*done);
265  },
266  register_snapshot, done, result, char_code, scratch);
267  if (char_code_fits_one_byte != nullptr) {
268  bind(char_code_fits_one_byte);
269  }
270  LoadSingleCharacterString(result, char_code, scratch);
271  bind(*done);
272 }

274 void MaglevAssembler::StringCharCodeOrCodePointAt(
275  BuiltinStringPrototypeCharCodeOrCodePointAt::Mode mode,
276  RegisterSnapshot& register_snapshot, Register result, Register string,
277  Register index, Register instance_type, Register scratch2,
278  Label* result_fits_one_byte) {
279  ASM_CODE_COMMENT(this);
280  ZoneLabelRef done(this);
281  Label seq_string;
282  Label cons_string;
283  Label sliced_string;
284 
285  Label* deferred_runtime_call = MakeDeferredCode(
286  [](MaglevAssembler* masm,
287  BuiltinStringPrototypeCharCodeOrCodePointAt::Mode mode,
288  RegisterSnapshot register_snapshot, ZoneLabelRef done, Register result,
289  Register string, Register index) {
290  DCHECK(!register_snapshot.live_registers.has(result));
291  DCHECK(!register_snapshot.live_registers.has(string));
292  DCHECK(!register_snapshot.live_registers.has(index));
293  {
294  SaveRegisterStateForCall save_register_state(masm, register_snapshot);
295  __ SmiTag(index);
296  __ Push(string, index);
297  __ Move(kContextRegister, masm->native_context().object());
298  // This call does not throw nor can deopt.
299  if (mode ==
300  BuiltinStringPrototypeCharCodeOrCodePointAt::kCodePointAt) {
301  __ CallRuntime(Runtime::kStringCodePointAt);
302  } else {
303  DCHECK_EQ(mode,
304  BuiltinStringPrototypeCharCodeOrCodePointAt::kCharCodeAt);
305  __ CallRuntime(Runtime::kStringCharCodeAt);
306  }
307  save_register_state.DefineSafepoint();
308  __ SmiUntag(kReturnRegister0);
309  __ Move(result, kReturnRegister0);
310  }
311  __ b(*done);
312  },
313  mode, register_snapshot, done, result, string, index);
314 
315  // We might need to try more than one time for ConsString, SlicedString and
316  // ThinString.
317  Label loop;
318  bind(&loop);
319 
320  if (v8_flags.debug_code) {
321  // Check if {string} is a string.
322  AssertObjectTypeInRange(string, FIRST_STRING_TYPE, LAST_STRING_TYPE,
323  AbortReason::kUnexpectedValue);
324 
325  Register scratch = instance_type;
326 
327  LoadU32(scratch, FieldMemOperand(string, offsetof(String, length_)));
328  CmpS32(index, scratch);
329  Check(lt, AbortReason::kUnexpectedValue);
330  }
331 
332  // Get instance type.
333  LoadInstanceType(instance_type, string);
334 
335  {
336  TemporaryRegisterScope temps(this);
337  Register representation = temps.AcquireScratch();
338 
339  // TODO(victorgomes): Add fast path for external strings.
340  And(representation, instance_type, Operand(kStringRepresentationMask));
341  CmpS32(representation, Operand(kSeqStringTag));
342  beq(&seq_string);
343  And(representation, Operand(kConsStringTag));
344  beq(&cons_string);
345  CmpS32(representation, Operand(kSlicedStringTag));
346  beq(&sliced_string);
347  CmpS32(representation, Operand(kThinStringTag));
348  bne(deferred_runtime_call);
349  // Fallthrough to thin string.
350  }
351 
352  // Is a thin string.
353  {
354  LoadTaggedField(string,
355  FieldMemOperand(string, offsetof(ThinString, actual_)));
356  b(&loop);
357  }
358 
359  bind(&sliced_string);
360  {
361  TemporaryRegisterScope temps(this);
362  Register offset = temps.AcquireScratch();
363 
364  LoadAndUntagTaggedSignedField(offset, string,
365  offsetof(SlicedString, offset_));
366  LoadTaggedField(string, string, offsetof(SlicedString, parent_));
367  AddS32(index, index, offset);
368  b(&loop);
369  }
370 
371  bind(&cons_string);
372  {
373  // Reuse {instance_type} register here, since CompareRoot requires a scratch
374  // register as well.
375  Register second_string = instance_type;
376  LoadU64(second_string,
377  FieldMemOperand(string, offsetof(ConsString, second_)));
378  CompareRoot(second_string, RootIndex::kempty_string);
379  bne(deferred_runtime_call);
380  LoadTaggedField(string,
381  FieldMemOperand(string, offsetof(ConsString, first_)));
382  b(&loop); // Try again with first string.
383  }
384 
385  bind(&seq_string);
386  {
387  Label two_byte_string;
388  And(instance_type, Operand(kStringEncodingMask));
389  CmpS32(instance_type, Operand(kTwoByteStringTag));
390  beq(&two_byte_string);
391  // The result of one-byte string will be the same for both modes
392  // (CharCodeAt/CodePointAt), since it cannot be the first half of a
393  // surrogate pair.
394  SeqOneByteStringCharCodeAt(result, string, index);
395  b(result_fits_one_byte);
396 
397  bind(&two_byte_string);
398  // {instance_type} is unused from this point, so we can use as scratch.
399  Register scratch = instance_type;
400  ShiftLeftU64(scratch, index, Operand(1));
401  AddU64(scratch,
402  Operand(OFFSET_OF_DATA_START(SeqTwoByteString) - kHeapObjectTag));
403 
404  if (mode == BuiltinStringPrototypeCharCodeOrCodePointAt::kCharCodeAt) {
405  LoadU16(result, MemOperand(string, scratch));
406  } else {
407  DCHECK_EQ(mode,
408  BuiltinStringPrototypeCharCodeOrCodePointAt::kCodePointAt);
409  Register string_backup = string;
410  if (result == string) {
411  string_backup = scratch2;
412  Move(string_backup, string);
413  }
414  LoadU16(result, MemOperand(string, scratch));
415 
416  Register first_code_point = scratch;
417  And(first_code_point, result, Operand(0xfc00));
418  CmpS32(first_code_point, Operand(0xd800));
419  bne(*done);
420 
421  Register length = scratch;
422  LoadU32(length, FieldMemOperand(string, offsetof(String, length_)));
423  AddS32(index, index, Operand(1));
424  CmpS32(index, length);
425  bge(*done);
426 
427  Register second_code_point = scratch;
428  ShiftLeftU32(index, index, Operand(1));
429  AddU32(index,
430  Operand(OFFSET_OF_DATA_START(SeqTwoByteString) - kHeapObjectTag));
431  LoadU16(second_code_point, MemOperand(string_backup, index));
432 
433  // {index} is not needed at this point.
434  Register scratch2 = index;
435  And(scratch2, second_code_point, Operand(0xfc00));
436  CmpS32(scratch2, Operand(0xdc00));
437  bne(*done);
438 
439  int surrogate_offset = 0x10000 - (0xd800 << 10) - 0xdc00;
440  AddS32(second_code_point, second_code_point, Operand(surrogate_offset));
441  ShiftLeftU32(result, result, Operand(10));
442  AddS32(result, result, second_code_point);
443  }
444 
445  // Fallthrough.
446  }
447 
448  bind(*done);
449 
450  if (v8_flags.debug_code) {
451  // We make sure that the user of this macro is not relying in string and
452  // index to not be clobbered.
453  if (result != string) {
454  Move(string, 0xdeadbeef);
455  }
456  if (result != index) {
457  Move(index, 0xdeadbeef);
458  }
459  }
460 }

517 void MaglevAssembler::TryTruncateDoubleToInt32(Register dst, DoubleRegister src,
518  Label* fail) {
519  TemporaryRegisterScope temps(this);
520  DoubleRegister temp = temps.AcquireScratchDouble();
521  Label done;
522 
523  // Convert the input float64 value to int32.
524  ConvertDoubleToInt32(dst, src);
525 
526  // Convert that int32 value back to float64.
527  ConvertIntToDouble(temp, dst);
528 
529  // Check that the result of the float64->int32->float64 is equal to the input
530  // (i.e. that the conversion didn't truncate.
531  CmpF64(src, temp);
532  JumpIf(ne, fail);
533 
534  // Check if {input} is -0.
535  CmpS32(dst, Operand::Zero());
536  JumpIf(ne, &done);
537 
538  // In case of 0, we need to check the high bits for the IEEE -0 pattern.
539  {
540  MovDoubleToInt64(r1, src);
541  ShiftRightS64(r1, r1, Operand(63));
542  CmpS64(r1, Operand(0));
543  JumpIf(lt, fail);
544  }
545 
546  bind(&done);
547 }
