32 Handle<SharedFunctionInfo> FunctionTemplateInfo::GetOrCreateSharedFunctionInfo(
33  Isolate* isolate, DirectHandle<FunctionTemplateInfo> info,
34  MaybeDirectHandle<Name> maybe_name) {
35  Tagged<Object> current_info = info->shared_function_info();
36  if (IsSharedFunctionInfo(current_info)) {
37  return handle(Cast<SharedFunctionInfo>(current_info), isolate);
38  }
39  DirectHandle<Name> name;
40  DirectHandle<String> name_string;
41  if (maybe_name.ToHandle(&name) && IsString(*name)) {
42  name_string = Cast<String>(name);
43  } else if (IsString(info->class_name())) {
44  name_string = direct_handle(Cast<String>(info->class_name()), isolate);
45  } else {
46  name_string = isolate->factory()->empty_string();
47  }
48  FunctionKind function_kind;
49  if (info->remove_prototype()) {
50  function_kind = FunctionKind::kConciseMethod;
51  } else {
52  function_kind = FunctionKind::kNormalFunction;
53  }
54  Handle<SharedFunctionInfo> sfi =
55  isolate->factory()->NewSharedFunctionInfoForApiFunction(name_string, info,
56  function_kind);
57  DCHECK(sfi->IsApiFunction());
58  info->set_shared_function_info(*sfi);
59  return sfi;
60 }

62 bool FunctionTemplateInfo::IsTemplateFor(Tagged<Map> map) const {
63  RCS_SCOPE(
64  LocalHeap::Current() == nullptr
65  ? GetIsolateChecked()->counters()->runtime_call_stats()
66  : LocalIsolate::FromHeap(LocalHeap::Current())->runtime_call_stats(),
67  RuntimeCallCounterId::kIsTemplateFor);
68 
69  // There is a constraint on the object; check.
70  if (!IsJSObjectMap(map)) return false;
71 
72  if (v8_flags.experimental_embedder_instance_types) {
73  DCHECK_IMPLIES(allowed_receiver_instance_type_range_start() == 0,
74  allowed_receiver_instance_type_range_end() == 0);
75  if (base::IsInRange(map->instance_type(),
76  allowed_receiver_instance_type_range_start(),
77  allowed_receiver_instance_type_range_end())) {
78  return true;
79  }
80  }
81 
82  // Fetch the constructor function of the object.
83  Tagged<Object> cons_obj = map->GetConstructor();
84  Tagged<Object> type;
85  if (IsJSFunction(cons_obj)) {
86  Tagged<JSFunction> fun = Cast<JSFunction>(cons_obj);
87  if (!fun->shared()->IsApiFunction()) return false;
88  type = fun->shared()->api_func_data();
89  } else if (IsFunctionTemplateInfo(cons_obj)) {
90  type = Cast<FunctionTemplateInfo>(cons_obj);
91  } else {
92  return false;
93  }
94  DCHECK(IsFunctionTemplateInfo(type));
95  // Iterate through the chain of inheriting function templates to
96  // see if the required one occurs.
97  while (IsFunctionTemplateInfo(type)) {
98  if (type == *this) return true;
99  type = Cast<FunctionTemplateInfo>(type)->GetParentTemplate();
100  }
101  // Didn't find the required type in the inheritance chain.
102  return false;
103 }

105 bool FunctionTemplateInfo::IsLeafTemplateForApiObject(
106  Tagged<Object> object) const {
107  i::DisallowGarbageCollection no_gc;
108 
109  if (!IsJSApiObject(object)) {
110  return false;
111  }
112 
113  bool result = false;
114  Tagged<Map> map = Cast<HeapObject>(object)->map();
115  Tagged<Object> constructor_obj = map->GetConstructor();
116  if (IsJSFunction(constructor_obj)) {
117  Tagged<JSFunction> fun = Cast<JSFunction>(constructor_obj);
118  result = (*this == fun->shared()->api_func_data());
119  } else if (IsFunctionTemplateInfo(constructor_obj)) {
120  result = (*this == constructor_obj);
121  }
122  DCHECK_IMPLIES(result, IsTemplateFor(map));
123  return result;
124 }

154  Isolate* isolate, Tagged<Object> getter) {
155  DisallowGarbageCollection no_gc;
156  if (!IsFunctionTemplateInfo(getter)) {
157  if (!IsJSFunction(getter)) return {};
158  Tagged<SharedFunctionInfo> info = Cast<JSFunction>(getter)->shared();

201 DirectHandle<DictionaryTemplateInfo> DictionaryTemplateInfo::Create(
202  Isolate* isolate, const v8::MemorySpan<const std::string_view>& names) {
203  DirectHandle<FixedArray> property_names = isolate->factory()->NewFixedArray(
204  static_cast<int>(names.size()), AllocationType::kOld);
205  int index = 0;
206  uint32_t unused_array_index;
207  for (const std::string_view& name : names) {
208  DirectHandle<String> internalized_name =
209  isolate->factory()->InternalizeString(
210  base::Vector<const char>(name.data(), name.length()));
211  // Check that property name cannot be used as index.
212  CHECK(!internalized_name->AsArrayIndex(&unused_array_index));
213  property_names->set(index, *internalized_name);
214  ++index;
215  }
216  return isolate->factory()->NewDictionaryTemplateInfo(property_names);
217 }

221 DirectHandle<JSObject> CreateSlowJSObjectWithProperties(
222  Isolate* isolate, DirectHandle<FixedArray> property_names,
223  const MemorySpan<MaybeLocal<Value>>& property_values,
224  int num_properties_set) {
225  DirectHandle<JSObject> object = isolate->factory()->NewSlowJSObjectFromMap(

227  AllocationType::kYoung);
228  Handle<Object> properties = handle(object->raw_properties_or_hash(), isolate);
229  for (int i = 0; i < static_cast<int>(property_values.size()); ++i) {
230  Local<Value> property_value;
231  if (!property_values[i].ToLocal(&property_value)) {
232  continue;
233  }
234  properties = PropertyDictionary::Add(
235  isolate, Cast<PropertyDictionary>(properties),
236  Cast<String>(handle(property_names->get(i), isolate)),
237  Utils::OpenDirectHandle(*property_value), PropertyDetails::Empty());
238  }
239  object->set_raw_properties_or_hash(*properties);
240  return object;
241 }

245 // static
246 DirectHandle<JSObject> DictionaryTemplateInfo::NewInstance(
247  DirectHandle<NativeContext> context,
248  DirectHandle<DictionaryTemplateInfo> self,
249  const MemorySpan<MaybeLocal<Value>>& property_values) {
250  Isolate* isolate = Isolate::Current();
251  DirectHandle<FixedArray> property_names(self->property_names(), isolate);
252 
253  const int property_names_len = property_names->length();
254  CHECK_EQ(property_names_len, static_cast<int>(property_values.size()));
255  const int num_properties_set = static_cast<int>(std::count_if(
256  property_values.begin(), property_values.end(),
257  [](const auto& maybe_value) { return !maybe_value.IsEmpty(); }));
258 
259  if (V8_UNLIKELY(num_properties_set > JSObject::kMaxInObjectProperties)) {
260  return CreateSlowJSObjectWithProperties(
261  isolate, property_names, property_values, num_properties_set);
262  }
263 
264  const bool can_use_map_cache = num_properties_set == property_names_len;
265  MaybeDirectHandle<Map> maybe_cached_map;
266  if (V8_LIKELY(can_use_map_cache)) {
267  maybe_cached_map = TemplateInfo::ProbeInstantiationsCache<Map>(
268  isolate, context, self, TemplateInfo::CachingMode::kUnlimited);
269  }
270  DirectHandle<Map> cached_map;
271  if (V8_LIKELY(can_use_map_cache && maybe_cached_map.ToHandle(&cached_map))) {
272  DCHECK(!cached_map->is_dictionary_map());
273  bool can_use_cached_map = !cached_map->is_deprecated();
274  if (V8_LIKELY(can_use_cached_map)) {
275  // Verify that the cached map can be reused.
276  auto descriptors = handle(cached_map->instance_descriptors(), isolate);
277  for (int i = 0; i < static_cast<int>(property_values.size()); ++i) {
278  DirectHandle<Object> value =
279  Utils::OpenDirectHandle(*property_values[i].ToLocalChecked());
280  InternalIndex descriptor{static_cast<size_t>(i)};
281  const auto details = descriptors->GetDetails(descriptor);
282 
283  if (!Object::FitsRepresentation(*value, details.representation()) ||
284  !FieldType::NowContains(descriptors->GetFieldType(descriptor),
285  value)) {
286  can_use_cached_map = false;
287  break;
288  }
289  // Double representation means mutable heap number. In this case we need
290  // to allocate a new heap number to put in the dictionary.
291  if (details.representation().Equals(Representation::Double())) {
292  // We allowed coercion in `FitsRepresentation` above which means that
293  // we may deal with a Smi here.
294  property_values[i] =
295  ToApiHandle<v8::Object>(isolate->factory()->NewHeapNumber(
296  Object::NumberValue(Cast<Number>(*value))));
297  }
298  }
299  if (V8_LIKELY(can_use_cached_map)) {
300  // Create the object from the cached map.
301  CHECK(!cached_map->is_deprecated());
302  CHECK_EQ(context->object_function_prototype(), cached_map->prototype());
303  auto object = isolate->factory()->NewJSObjectFromMap(
304  cached_map, AllocationType::kYoung);
305  DisallowGarbageCollection no_gc;
306  for (int i = 0; i < static_cast<int>(property_values.size()); ++i) {
307  Local<Value> property_value = property_values[i].ToLocalChecked();
308  DirectHandle<Object> value = Utils::OpenDirectHandle(*property_value);
309  const FieldIndex index = FieldIndex::ForPropertyIndex(
310  *cached_map, i, Representation::Tagged());
311  object->FastPropertyAtPut(index, *value,
312  WriteBarrierMode::SKIP_WRITE_BARRIER);
313  }
314  return object;
315  }
316  }
317  // A cached map was either deprecated or the descriptors changed in
318  // incompatible ways. We clear the cached map and continue with the generic
319  // path.
320  TemplateInfo::UncacheTemplateInstantiation(
321  isolate, context, self, TemplateInfo::CachingMode::kUnlimited);
322  }
323 
324  // General case: We either don't have a cached map, or it is unusable for the
325  // values provided.
326  DirectHandle<Map> current_map = isolate->factory()->ObjectLiteralMapFromCache(
327  context, num_properties_set);
328  DirectHandle<JSObject> object =
329  isolate->factory()->NewJSObjectFromMap(current_map);
330  int current_property_index = 0;
331  for (int i = 0; i < static_cast<int>(property_values.size()); ++i) {
332  Local<Value> property_value;
333  if (!property_values[i].ToLocal(&property_value)) {
334  continue;
335  }
336  auto name = Cast<String>(handle(property_names->get(i), isolate));
337  DirectHandle<Object> value = Utils::OpenDirectHandle(*property_value);
338  constexpr PropertyAttributes attributes = PropertyAttributes::NONE;
339  constexpr PropertyConstness constness = PropertyConstness::kConst;
340  current_map = Map::TransitionToDataProperty(isolate, current_map, name,
341  value, attributes, constness,
342  StoreOrigin::kNamed);
343  if (current_map->is_dictionary_map()) {
344  return CreateSlowJSObjectWithProperties(
345  isolate, property_names, property_values, num_properties_set);
346  }
347  JSObject::MigrateToMap(isolate, object, current_map);
348  PropertyDetails details = current_map->GetLastDescriptorDetails(isolate);
349  object->WriteToField(InternalIndex(current_property_index), details,
350  *value);
351  current_property_index++;
352  }
353  if (V8_LIKELY(can_use_map_cache)) {
354  TemplateInfo::CacheTemplateInstantiation(
355  isolate, context, self, TemplateInfo::CachingMode::kUnlimited,
356  direct_handle(object->map(), isolate));
357  }
358  return object;
359 }
