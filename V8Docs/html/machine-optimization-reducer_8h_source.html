49 // ******************************** OVERVIEW ********************************
50 //
51 // The MachineOptimizationAssembler performs basic optimizations on low-level
52 // operations that can be performed on-the-fly, without requiring type analysis
53 // or analyzing uses. It largely corresponds to MachineOperatorReducer in
54 // sea-of-nodes Turbofan.
55 //
56 // These peephole optimizations are typically very local: they based on the
57 // immediate inputs of an operation, we try to constant-fold or strength-reduce
58 // the operation.
59 //
60 // Typical examples include:
61 //
62 // * Reducing `a == a` to `1`
63 //
64 // * Reducing `a + 0` to `a`
65 //
66 // * Reducing `a * 2^k` to `a << k`
67 //

78 // Represents an operation of the form `(source & mask) == masked_value`.
79 // where each bit set in masked_value also has to be set in mask.
80 struct BitfieldCheck {
81  V<Word> const source;
82  uint32_t const mask;
83  uint32_t const masked_value;
84  bool const truncate_from_64_bit;
85 
86  BitfieldCheck(V<Word> source, uint32_t mask, uint32_t masked_value,
87  bool truncate_from_64_bit)
88  : source(source),
89  mask(mask),
90  masked_value(masked_value),
91  truncate_from_64_bit(truncate_from_64_bit) {
92  CHECK_EQ(masked_value & ~mask, 0);
93  }
94 
95  static std::optional<BitfieldCheck> Detect(const OperationMatcher& matcher,
96  const Graph& graph,
97  V<Word> index) {
98  // There are two patterns to check for here:
99  // 1. Single-bit checks: `(val >> shift) & 1`, where:
100  // - the shift may be omitted, and/or
101  // - the result may be truncated from 64 to 32
102  // 2. Equality checks: `(val & mask) == expected`, where:
103  // - val may be truncated from 64 to 32 before masking (see
104  // ReduceWordEqualForConstantRhs)
105  const Operation& op = graph.Get(index);
106  if (const ComparisonOp* equal = op.TryCast<Opmask::kWord32Equal>()) {
107  if (const WordBinopOp* left_and =
108  graph.Get(equal->left<Word32>())
109  .TryCast<Opmask::kWord32BitwiseAnd>()) {
110  uint32_t mask;
111  uint32_t masked_value;
112  if (matcher.MatchIntegralWord32Constant(left_and->right(), &mask) &&
113  matcher.MatchIntegralWord32Constant(equal->right<Word32>(),
114  &masked_value)) {
115  if ((masked_value & ~mask) != 0) return std::nullopt;
116  if (const ChangeOp* truncate =
117  graph.Get(left_and->left())
118  .TryCast<Opmask::kTruncateWord64ToWord32>()) {
119  return BitfieldCheck{truncate->input<Word64>(), mask, masked_value,
120  true};
121  } else {
122  return BitfieldCheck{left_and->left(), mask, masked_value, false};
123  }
124  }
125  }
126  } else if (const ChangeOp* truncate =
127  op.TryCast<Opmask::kTruncateWord64ToWord32>()) {
128  return TryDetectShiftAndMaskOneBit<Word64>(matcher,
129  truncate->input<Word64>());
130  } else {
131  return TryDetectShiftAndMaskOneBit<Word32>(matcher, index);
132  }
133  return std::nullopt;
134  }
135 
136  std::optional<BitfieldCheck> TryCombine(const BitfieldCheck& other) {
137  if (source != other.source ||
138  truncate_from_64_bit != other.truncate_from_64_bit) {
139  return std::nullopt;
140  }
141  uint32_t overlapping_bits = mask & other.mask;
142  // It would be kind of strange to have any overlapping bits, but they can be
143  // allowed as long as they don't require opposite values in the same
144  // positions.
145  if ((masked_value & overlapping_bits) !=
146  (other.masked_value & overlapping_bits)) {
147  return std::nullopt;
148  }
149  return BitfieldCheck{source, mask | other.mask,
150  masked_value | other.masked_value,
151  truncate_from_64_bit};
152  }
153 
154  private:
155  template <typename WordType>
156  static std::optional<BitfieldCheck> TryDetectShiftAndMaskOneBit(
157  const OperationMatcher& matcher, V<Word> index) {
158  constexpr WordRepresentation Rep = V<WordType>::rep;
159  // Look for the pattern `(val >> shift) & 1`. The shift may be omitted.
160  V<WordType> value;
161  uint64_t constant;
162  if (matcher.MatchBitwiseAndWithConstant(index, &value, &constant, Rep) &&
163  constant == 1) {
164  V<WordType> input;
165  if (int shift_amount;
166  matcher.MatchConstantRightShift(value, &input, Rep, &shift_amount) &&
167  shift_amount >= 0 && shift_amount < 32) {
168  uint32_t mask = 1 << shift_amount;
169  return BitfieldCheck{input, mask, mask,
170  Rep == WordRepresentation::Word64()};
171  }
172  return BitfieldCheck{value, 1, 1, Rep == WordRepresentation::Word64()};
173  }
174  return std::nullopt;
175  }
176 };

178 template <typename T>
179 bool CanFoldFloatConstants(bool ensure_deterministic_nan, T k1, T k2) {
180  if (!ensure_deterministic_nan) return true;
181  // Operations involving NaN values have platform specific results.
182  return (!std::isnan(k1) && !std::isnan(k2)) &&
183  // Operations involving +Inf and -Inf values have platform specific
184  // results.
185  (!(std::isinf(k1) && std::isinf(k2) && k1 != k2));
186 }

208  V<Untagged> REDUCE(Change)(V<Untagged> input, ChangeOp::Kind kind,
209  ChangeOp::Assumption assumption,
210  RegisterRepresentation from,
211  RegisterRepresentation to) {
212  if (ShouldSkipOptimizationStep()) {
213  return Next::ReduceChange(input, kind, assumption, from, to);
214  }
215  using Kind = ChangeOp::Kind;
216  if (from == WordRepresentation::Word32()) {
217  input = TryRemoveWord32ToWord64Conversion(V<Word64>::Cast(input));
218  }
219  if (uint64_t value;
220  from.IsWord() && matcher_.MatchIntegralWordConstant(
221  input, WordRepresentation(from), &value)) {
222  using Rep = RegisterRepresentation;
223  switch (multi(kind, from, to)) {
224  case multi(Kind::kSignExtend, Rep::Word32(), Rep::Word64()):
225  return __ Word64Constant(int64_t{static_cast<int32_t>(value)});
226  case multi(Kind::kZeroExtend, Rep::Word32(), Rep::Word64()):
227  case multi(Kind::kBitcast, Rep::Word32(), Rep::Word64()):
228  return __ Word64Constant(uint64_t{static_cast<uint32_t>(value)});
229  case multi(Kind::kBitcast, Rep::Word32(), Rep::Float32()):
230  return __ Float32Constant(
231  i::Float32::FromBits(static_cast<uint32_t>(value)));
232  case multi(Kind::kBitcast, Rep::Word64(), Rep::Float64()):
233  return __ Float64Constant(i::Float64::FromBits(value));
234  case multi(Kind::kSignedToFloat, Rep::Word32(), Rep::Float64()):
235  return __ Float64Constant(
236  static_cast<double>(static_cast<int32_t>(value)));
237  case multi(Kind::kSignedToFloat, Rep::Word64(), Rep::Float64()):
238  return __ Float64Constant(
239  static_cast<double>(static_cast<int64_t>(value)));
240  case multi(Kind::kUnsignedToFloat, Rep::Word32(), Rep::Float64()):
241  return __ Float64Constant(
242  static_cast<double>(static_cast<uint32_t>(value)));
243  case multi(Kind::kTruncate, Rep::Word64(), Rep::Word32()):
244  return __ Word32Constant(static_cast<uint32_t>(value));
245  default:
246  break;
247  }
248  }
249  if (i::Float32 value; from == RegisterRepresentation::Float32() &&
250  matcher_.MatchFloat32Constant(input, &value)) {
251  if (kind == Kind::kFloatConversion &&
252  to == RegisterRepresentation::Float64()) {
253  return __ Float64Constant(value.get_scalar());
254  }
255  if (kind == Kind::kBitcast && to == WordRepresentation::Word32()) {
256  return __ Word32Constant(value.get_bits());
257  }
258  }
259  if (i::Float64 value; from == RegisterRepresentation::Float64() &&
260  matcher_.MatchFloat64Constant(input, &value)) {
261  if (kind == Kind::kFloatConversion &&
262  to == RegisterRepresentation::Float32()) {
263  return __ Float32Constant(DoubleToFloat32_NoInline(value.get_scalar()));
264  }
265  if (kind == Kind::kBitcast && to == WordRepresentation::Word64()) {
266  return __ Word64Constant(base::bit_cast<uint64_t>(value));
267  }
268  if (kind == Kind::kSignedFloatTruncateOverflowToMin) {
269  double truncated = std::trunc(value.get_scalar());
270  if (to == WordRepresentation::Word64()) {
271  int64_t result = std::numeric_limits<int64_t>::min();
272  if (truncated >= std::numeric_limits<int64_t>::min() &&
273  truncated <= kMaxDoubleRepresentableInt64) {
274  result = static_cast<int64_t>(truncated);
275  }
276  return __ Word64Constant(result);
277  }
278  if (to == WordRepresentation::Word32()) {
279  int32_t result = std::numeric_limits<int32_t>::min();
280  if (truncated >= std::numeric_limits<int32_t>::min() &&
281  truncated <= std::numeric_limits<int32_t>::max()) {
282  result = static_cast<int32_t>(truncated);
283  }
284  return __ Word32Constant(result);
285  }
286  }
287  if (kind == Kind::kJSFloatTruncate &&
288  to == WordRepresentation::Word32()) {
289  return __ Word32Constant(DoubleToInt32_NoInline(value.get_scalar()));
290  }
291  if (kind == Kind::kExtractHighHalf) {
292  DCHECK_EQ(to, RegisterRepresentation::Word32());
293  return __ Word32Constant(static_cast<uint32_t>(value.get_bits() >> 32));
294  }
295  if (kind == Kind::kExtractLowHalf) {
296  DCHECK_EQ(to, RegisterRepresentation::Word32());
297  return __ Word32Constant(static_cast<uint32_t>(value.get_bits()));
298  }
299  }
300  if (float value; from == RegisterRepresentation::Float32() &&
301  matcher_.MatchFloat32Constant(input, &value)) {
302  if (kind == Kind::kFloatConversion &&
303  to == RegisterRepresentation::Float64()) {
304  return __ Float64Constant(value);
305  }
306  }
307 
308  const Operation& input_op = matcher_.Get(input);
309  if (const ChangeOp* change_op = input_op.TryCast<ChangeOp>()) {
310  if (change_op->from == to && change_op->to == from &&
311  change_op->IsReversibleBy(kind, signalling_nan_possible)) {
312  return change_op->input();
313  }
314  }
315  return Next::ReduceChange(input, kind, assumption, from, to);
316  }

332  V<Any> REDUCE(TaggedBitcast)(V<Any> input, RegisterRepresentation from,
333  RegisterRepresentation to,
334  TaggedBitcastOp::Kind kind) {
335  if (ShouldSkipOptimizationStep()) {
336  return Next::ReduceTaggedBitcast(input, from, to, kind);
337  }
338  // A Tagged -> Untagged -> Tagged sequence can be short-cut.
339  // An Untagged -> Tagged -> Untagged sequence however cannot be removed,
340  // because the GC might have modified the pointer.
341  if (auto* input_bitcast = matcher_.TryCast<TaggedBitcastOp>(input)) {
342  if (all_of(input_bitcast->to, from) ==
343  RegisterRepresentation::WordPtr() &&
344  all_of(input_bitcast->from, to) == RegisterRepresentation::Tagged()) {
345  return input_bitcast->input();
346  }
347  }
348  // An Untagged -> Smi -> Untagged sequence can be short-cut.
349  if (auto* input_bitcast = matcher_.TryCast<TaggedBitcastOp>(input);
350  input_bitcast && to.IsWord() &&
351  (kind == TaggedBitcastOp::Kind::kSmi ||
352  input_bitcast->kind == TaggedBitcastOp::Kind::kSmi)) {
353  if (input_bitcast->from == to) return input_bitcast->input();
354  if (input_bitcast->from == RegisterRepresentation::Word32()) {
355  DCHECK_EQ(to, RegisterRepresentation::Word64());
356  return __ BitcastWord32ToWord64(input_bitcast->input());
357  }
358  DCHECK(input_bitcast->from == RegisterRepresentation::Word64() &&
359  to == RegisterRepresentation::Word32());
360  return __ TruncateWord64ToWord32(input_bitcast->input());
361  }
362  // Try to constant-fold TaggedBitcast from Word Constant to Word.
363  if (to.IsWord()) {
364  if (const ConstantOp* cst = matcher_.TryCast<ConstantOp>(input)) {
365  if (cst->kind == ConstantOp::Kind::kWord32 ||
366  cst->kind == ConstantOp::Kind::kWord64) {
367  if (to == RegisterRepresentation::Word64()) {
368  return __ Word64Constant(cst->integral());
369  } else {
370  DCHECK_EQ(to, RegisterRepresentation::Word32());
371  return __ Word32Constant(static_cast<uint32_t>(cst->integral()));
372  }
373  }
374  }
375  }
376  if (const ConstantOp* cst = matcher_.TryCast<ConstantOp>(input)) {
377  // Try to constant-fold Word constant -> Tagged (Smi).
378  if (cst->IsIntegral() && to == RegisterRepresentation::Tagged()) {
379  if (Smi::IsValid(cst->integral())) {
380  return __ SmiConstant(
381  i::Tagged<Smi>(static_cast<intptr_t>(cst->integral())));
382  }
383  }
384  // Try to constant-fold Smi -> Untagged.
385  if (cst->kind == ConstantOp::Kind::kSmi) {
386  if (to == RegisterRepresentation::Word32()) {
387  return __ Word32Constant(static_cast<uint32_t>(cst->smi().ptr()));
388  } else if (to == RegisterRepresentation::Word64()) {
389  return __ Word64Constant(static_cast<uint64_t>(cst->smi().ptr()));
390  }
391  }
392  }
393  return Next::ReduceTaggedBitcast(input, from, to, kind);
394  }

581  V<Float> REDUCE(FloatBinop)(V<Float> lhs, V<Float> rhs,
582  FloatBinopOp::Kind kind,
583  FloatRepresentation rep) {
584  if (ShouldSkipOptimizationStep()) {
585  return Next::ReduceFloatBinop(lhs, rhs, kind, rep);
586  }
587 
588  using Kind = FloatBinopOp::Kind;
589 
590  // Place constant on the right for commutative operators.
591  if (FloatBinopOp::IsCommutative(kind) && matcher_.Is<ConstantOp>(lhs) &&
592  !matcher_.Is<ConstantOp>(rhs) && !ensure_deterministic_nan) {
593  return ReduceFloatBinop(rhs, lhs, kind, rep);
594  }
595 
596  // constant folding
597  if (float k1, k2;
598  rep == FloatRepresentation::Float32() &&
599  matcher_.MatchFloat32Constant(lhs, &k1) &&
600  matcher_.MatchFloat32Constant(rhs, &k2) &&
601  detail::CanFoldFloatConstants(ensure_deterministic_nan, k1, k2)) {
602  switch (kind) {
603  case Kind::kAdd:
604  return __ Float32Constant(k1 + k2);
605  case Kind::kMul:
606  return __ Float32Constant(k1 * k2);
607  case Kind::kSub:
608  return __ Float32Constant(k1 - k2);
609  case Kind::kMin:
610  return __ Float32Constant(JSMin(k1, k2));
611  case Kind::kMax:
612  return __ Float32Constant(JSMax(k1, k2));
613  case Kind::kDiv:
614  return __ Float32Constant(k1 / k2);
615  case Kind::kPower:
616  return __ Float32Constant(internal::math::pow(k1, k2));
617  case Kind::kAtan2:
618  return __ Float32Constant(base::ieee754::atan2(k1, k2));
619  case Kind::kMod:
620  UNREACHABLE();
621  }
622  }
623  if (double k1, k2;
624  rep == FloatRepresentation::Float64() &&
625  matcher_.MatchFloat64Constant(lhs, &k1) &&
626  matcher_.MatchFloat64Constant(rhs, &k2) &&
627  detail::CanFoldFloatConstants(ensure_deterministic_nan, k1, k2)) {
628  switch (kind) {
629  case Kind::kAdd:
630  return __ Float64Constant(k1 + k2);
631  case Kind::kMul:
632  return __ Float64Constant(k1 * k2);
633  case Kind::kSub:
634  return __ Float64Constant(k1 - k2);
635  case Kind::kMin:
636  return __ Float64Constant(JSMin(k1, k2));
637  case Kind::kMax:
638  return __ Float64Constant(JSMax(k1, k2));
639  case Kind::kDiv:
640  return __ Float64Constant(k1 / k2);
641  case Kind::kMod:
642  return __ Float64Constant(Modulo(k1, k2));
643  case Kind::kPower:
644  return __ Float64Constant(math::pow(k1, k2));
645  case Kind::kAtan2:
646  return __ Float64Constant(base::ieee754::atan2(k1, k2));
647  }
648  }
649 
650  // All NaN folding is disabled for Wasm; architectures disagree on which
651  // input to take in case both are NaN, sometimes it even depends on the
652  // exact NaN which one is propagated as the result.
653  // Even though the Wasm spec allows propagating any NaN input, we want
654  // determinism here for differential fuzzing across compilers.
655  if (!ensure_deterministic_nan) {
656  // NaN <op> rhs => NaN (except for Kind::kPower).
657  // lhs <op> NaN => NaN.
658  if (i::Float32 nan;
659  (matcher_.MatchFloat32Constant(lhs, &nan) && nan.is_nan() &&
660  kind != Kind::kPower) ||
661  (matcher_.MatchFloat32Constant(rhs, &nan) && nan.is_nan())) {
662  // Return a quiet NaN since Wasm operations could have signalling NaN as
663  // input but not as output.
664  return __ Float32Constant(std::numeric_limits<float>::quiet_NaN());
665  }
666 
667  // Same for float64.
668  if (i::Float64 nan;
669  (matcher_.MatchFloat64Constant(lhs, &nan) && nan.is_nan() &&
670  kind != Kind::kPower) ||
671  (matcher_.MatchFloat64Constant(rhs, &nan) && nan.is_nan())) {
672  // Return a quiet NaN since Wasm operations could have signalling NaN as
673  // input but not as output.
674  return __ Float64Constant(std::numeric_limits<double>::quiet_NaN());
675  }
676  }
677 
678  if (matcher_.Is<ConstantOp>(rhs)) {
679  if (kind == Kind::kMul) {
680  // lhs * 1 => lhs
681  if (!signalling_nan_possible && matcher_.MatchFloat(rhs, 1.0)) {
682  return lhs;
683  }
684  // lhs * 2 => lhs + lhs
685  if (matcher_.MatchFloat(rhs, 2.0)) {
686  return __ FloatAdd(lhs, lhs, rep);
687  }
688  // lhs * -1 => -lhs
689  // Note: FloatNegate does not set the quiet bit so we cannot use this if
690  // signalling nans can happen.
691  if (!signalling_nan_possible && matcher_.MatchFloat(rhs, -1.0)) {
692  return __ FloatNegate(lhs, rep);
693  }
694  }
695 
696  if (kind == Kind::kDiv) {
697  // lhs / 1 => lhs
698  if (!signalling_nan_possible && matcher_.MatchFloat(rhs, 1.0)) {
699  return lhs;
700  }
701  // lhs / -1 => -lhs
702  // Note: FloatNegate does not set the quiet bit so we cannot use this if
703  // signalling nans can happen.
704  if (!signalling_nan_possible && matcher_.MatchFloat(rhs, -1.0)) {
705  return __ FloatNegate(lhs, rep);
706  }
707  // All reciprocals of non-denormal powers of two can be represented
708  // exactly, so division by power of two can be reduced to
709  // multiplication by reciprocal, with the same result.
710  // x / k => x * (1 / k)
711  if (rep == FloatRepresentation::Float32()) {
712  if (float k;
713  matcher_.MatchFloat32Constant(rhs, &k) && std::isnormal(k) &&
714  k != 0 && std::isfinite(k) &&
715  base::bits::IsPowerOfTwo(base::Double(k).Significand())) {
716  return __ FloatMul(lhs, __ FloatConstant(1.0 / k, rep), rep);
717  }
718  } else {
719  DCHECK_EQ(rep, FloatRepresentation::Float64());
720  if (double k;
721  matcher_.MatchFloat64Constant(rhs, &k) && std::isnormal(k) &&
722  k != 0 && std::isfinite(k) &&
723  base::bits::IsPowerOfTwo(base::Double(k).Significand())) {
724  return __ FloatMul(lhs, __ FloatConstant(1.0 / k, rep), rep);
725  }
726  }
727  }
728 
729  if (kind == Kind::kMod) {
730  // x % 0 => NaN
731  if (matcher_.MatchFloat(rhs, 0.0)) {
732  return __ FloatConstant(std::numeric_limits<double>::quiet_NaN(),
733  rep);
734  }
735  }
736 
737  if (kind == Kind::kSub) {
738  // lhs - +0.0 => lhs
739  if (!signalling_nan_possible && matcher_.MatchFloat(rhs, +0.0)) {
740  return lhs;
741  }
742  }
743 
744  if (kind == Kind::kPower) {
745  if (matcher_.MatchFloat(rhs, 0.0) || matcher_.MatchFloat(rhs, -0.0)) {
746  // lhs ** 0 ==> 1
747  return __ FloatConstant(1.0, rep);
748  }
749  if (matcher_.MatchFloat(rhs, 2.0)) {
750  // lhs ** 2 ==> lhs * lhs
751  return __ FloatMul(lhs, lhs, rep);
752  }
753  if (matcher_.MatchFloat(rhs, 0.5)) {
754  // lhs ** 0.5 ==> sqrt(lhs)
755  // (unless if lhs is -infinity)
756  Variable result = __ NewLoopInvariantVariable(rep);
757  IF (UNLIKELY(__ FloatLessThanOrEqual(
758  lhs, __ FloatConstant(-V8_INFINITY, rep), rep))) {
759  __ SetVariable(result, __ FloatConstant(V8_INFINITY, rep));
760  } ELSE {
761  __ SetVariable(
762  result,
763  __ FloatSqrt(__ FloatAdd(lhs, __ FloatConstant(0, rep), rep),
764  rep));
765  }
766 
767  return __ GetVariable(result);
768  }
769  }
770  }
771 
772  if (!signalling_nan_possible && kind == Kind::kSub &&
773  matcher_.MatchFloat(lhs, -0.0)) {
774  // -0.0 - round_down(-0.0 - y) => round_up(y)
775  if (V<Float> a, b, c;
776  FloatUnaryOp::IsSupported(FloatUnaryOp::Kind::kRoundUp, rep) &&
777  matcher_.MatchFloatRoundDown(rhs, &a, rep) &&
778  matcher_.MatchFloatSub(a, &b, &c, rep) &&
779  matcher_.MatchFloat(b, -0.0)) {
780  return __ FloatRoundUp(c, rep);
781  }
782  // -0.0 - rhs => -rhs
783  return __ FloatNegate(rhs, rep);
784  }
785 
786  return Next::ReduceFloatBinop(lhs, rhs, kind, rep);
787  }

789  V<Word> REDUCE(WordBinop)(V<Word> left, V<Word> right, WordBinopOp::Kind kind,
790  WordRepresentation rep) {
791  if (ShouldSkipOptimizationStep()) {
792  return Next::ReduceWordBinop(left, right, kind, rep);
793  }
794 
795  using Kind = WordBinopOp::Kind;
796 
797  DCHECK_EQ(rep, any_of(WordRepresentation::Word32(),
798  WordRepresentation::Word64()));
799  bool is_64 = rep == WordRepresentation::Word64();
800 
801  if (!is_64) {
802  left = TryRemoveWord32ToWord64Conversion(left);
803  right = TryRemoveWord32ToWord64Conversion(right);
804  }
805 
806  // Place constant on the right for commutative operators.
807  if (WordBinopOp::IsCommutative(kind) && matcher_.Is<ConstantOp>(left) &&
808  !matcher_.Is<ConstantOp>(right)) {
809  return ReduceWordBinop(right, left, kind, rep);
810  }
811  // constant folding
812  if (uint64_t k1, k2; matcher_.MatchIntegralWordConstant(left, rep, &k1) &&
813  matcher_.MatchIntegralWordConstant(right, rep, &k2)) {
814  switch (kind) {
815  case Kind::kAdd:
816  return __ WordConstant(k1 + k2, rep);
817  case Kind::kMul:
818  return __ WordConstant(k1 * k2, rep);
819  case Kind::kBitwiseAnd:
820  return __ WordConstant(k1 & k2, rep);
821  case Kind::kBitwiseOr:
822  return __ WordConstant(k1 | k2, rep);
823  case Kind::kBitwiseXor:
824  return __ WordConstant(k1 ^ k2, rep);
825  case Kind::kSub:
826  return __ WordConstant(k1 - k2, rep);
827  case Kind::kSignedMulOverflownBits:
828  return __ WordConstant(
829  is_64 ? base::bits::SignedMulHigh64(static_cast<int64_t>(k1),
830  static_cast<int64_t>(k2))
831  : base::bits::SignedMulHigh32(static_cast<int32_t>(k1),
832  static_cast<int32_t>(k2)),
833  rep);
834  case Kind::kUnsignedMulOverflownBits:
835  return __ WordConstant(
836  is_64 ? base::bits::UnsignedMulHigh64(k1, k2)
837  : base::bits::UnsignedMulHigh32(static_cast<uint32_t>(k1),
838  static_cast<uint32_t>(k2)),
839  rep);
840  case Kind::kSignedDiv:
841  return __ WordConstant(
842  is_64 ? base::bits::SignedDiv64(k1, k2)
843  : base::bits::SignedDiv32(static_cast<int32_t>(k1),
844  static_cast<int32_t>(k2)),
845  rep);
846  case Kind::kUnsignedDiv:
847  return __ WordConstant(
848  is_64 ? base::bits::UnsignedDiv64(k1, k2)
849  : base::bits::UnsignedDiv32(static_cast<uint32_t>(k1),
850  static_cast<uint32_t>(k2)),
851  rep);
852  case Kind::kSignedMod:
853  return __ WordConstant(
854  is_64 ? base::bits::SignedMod64(k1, k2)
855  : base::bits::SignedMod32(static_cast<int32_t>(k1),
856  static_cast<int32_t>(k2)),
857  rep);
858  case Kind::kUnsignedMod:
859  return __ WordConstant(
860  is_64 ? base::bits::UnsignedMod64(k1, k2)
861  : base::bits::UnsignedMod32(static_cast<uint32_t>(k1),
862  static_cast<uint32_t>(k2)),
863  rep);
864  }
865  }
866 
867  if (kind == WordBinopOp::Kind::kBitwiseAnd &&
868  rep == WordRepresentation::Word32()) {
869  if (auto right_bitfield = detail::BitfieldCheck::Detect(
870  matcher_, __ output_graph(), right)) {
871  if (auto left_bitfield = detail::BitfieldCheck::Detect(
872  matcher_, __ output_graph(), left)) {
873  if (auto combined_bitfield =
874  left_bitfield->TryCombine(*right_bitfield)) {
875  V<Word> source = combined_bitfield->source;
876  if (combined_bitfield->truncate_from_64_bit) {
877  source = __ TruncateWord64ToWord32(V<Word64>::Cast(source));
878  }
879  return __ Word32Equal(__ Word32BitwiseAnd(V<Word32>::Cast(source),
880  combined_bitfield->mask),
881  combined_bitfield->masked_value);
882  }
883  }
884  }
885  }
886 
887  if (uint64_t right_value;
888  matcher_.MatchIntegralWordConstant(right, rep, &right_value)) {
889  // TODO(jkummerow): computing {right_value_signed} could probably be
890  // handled by the 4th argument to {MatchIntegralWordConstant}.
891  int64_t right_value_signed =
892  is_64 ? static_cast<int64_t>(right_value)
893  : int64_t{static_cast<int32_t>(right_value)};
894  // (a <op> k1) <op> k2 => a <op> (k1 <op> k2)
895  if (V<Word> a, k1;
896  WordBinopOp::IsAssociative(kind) &&
897  matcher_.MatchWordBinop<Word>(left, &a, &k1, kind, rep) &&
898  matcher_.Is<ConstantOp>(k1)) {
899  V<Word> k2 = right;
900  // This optimization allows to do constant folding of `k1` and `k2`.
901  // However, if (a <op> k1) has to be calculated anyways, then constant
902  // folding does not save any calculations during runtime, and it may
903  // increase register pressure because it extends the lifetime of `a`.
904  // Therefore we do the optimization only when `left = (a <op k1)` has no
905  // other uses.
906  if (matcher_.Get(left).saturated_use_count.IsZero()) {
907  return ReduceWordBinop(a, ReduceWordBinop(k1, k2, kind, rep), kind,
908  rep);
909  }
910  }
911  switch (kind) {
912  case Kind::kSub:
913  // left - k => left + -k
914  return ReduceWordBinop(left, __ WordConstant(-right_value, rep),
915  Kind::kAdd, rep);
916  case Kind::kAdd:
917  // left + 0 => left
918  if (right_value == 0) {
919  return left;
920  }
921  break;
922  case Kind::kBitwiseXor:
923  // left ^ 0 => left
924  if (right_value == 0) {
925  return left;
926  }
927  // left ^ 1 => left == 0 if left is 0 or 1
928  if (right_value == 1 && IsBit(left)) {
929  return __ Word32Equal(V<Word32>::Cast(left), 0);
930  }
931  // (x ^ -1) ^ -1 => x
932  {
933  V<Word> x, y;
934  int64_t k;
935  if (right_value_signed == -1 &&
936  matcher_.MatchBitwiseAnd(left, &x, &y, rep) &&
937  matcher_.MatchIntegralWordConstant(y, rep, &k) && k == -1) {
938  return x;
939  }
940  }
941  break;
942  case Kind::kBitwiseOr:
943  // left | 0 => left
944  if (right_value == 0) {
945  return left;
946  }
947  // left | -1 => -1
948  if (right_value_signed == -1) {
949  return right;
950  }
951  // (x & K1) | K2 => x | K2 if K2 has ones for every zero bit in K1.
952  // This case can be constructed by UpdateWord and UpdateWord32 in CSA.
953  {
954  V<Word> x, y;
955  uint64_t k1;
956  uint64_t k2 = right_value;
957  if (matcher_.MatchBitwiseAnd(left, &x, &y, rep) &&
958  matcher_.MatchIntegralWordConstant(y, rep, &k1) &&
959  (k1 | k2) == rep.MaxUnsignedValue()) {
960  return __ WordBitwiseOr(x, right, rep);
961  }
962  }
963  break;
964  case Kind::kMul:
965  // left * 0 => 0
966  if (right_value == 0) {
967  return __ WordConstant(0, rep);
968  }
969  // left * 1 => left
970  if (right_value == 1) {
971  return left;
972  }
973  // left * -1 => 0 - left
974  if (right_value_signed == -1) {
975  return __ WordSub(__ WordConstant(0, rep), left, rep);
976  }
977  // left * 2^k => left << k
978  if (base::bits::IsPowerOfTwo(right_value)) {
979  return __ ShiftLeft(left, base::bits::WhichPowerOfTwo(right_value),
980  rep);
981  }
982  break;
983  case Kind::kBitwiseAnd:
984  // left & -1 => left
985  if (right_value_signed == -1) {
986  return left;
987  }
988  // x & 0 => 0
989  if (right_value == 0) {
990  return right;
991  }
992 
993  if (right_value == 1) {
994  // (x + x) & 1 => 0
995  V<Word> left_ignore_extensions =
996  IsWord32ConvertedToWord64(left)
997  ? UndoWord32ToWord64Conversion(left)
998  : left;
999  if (V<Word> a, b;
1000  matcher_.MatchWordAdd(left_ignore_extensions, &a, &b,
1001  WordRepresentation::Word32()) &&
1002  a == b) {
1003  return __ WordConstant(0, rep);
1004  }
1005 
1006  // CMP & 1 => CMP
1007  if (IsBit(left_ignore_extensions)) {
1008  return left;
1009  }
1010 
1011  static_assert(kSmiTagMask == 1);
1012  // HeapObject & 1 => 1 ("& 1" is a Smi-check)
1013  if (TryMatchHeapObject(left)) {
1014  return __ WordConstant(1, rep);
1015  } else if (__ Get(left).template Is<Opmask::kSmiConstant>()) {
1016  return __ WordConstant(0, rep);
1017  }
1018  }
1019 
1020  // asm.js often benefits from these transformations, to optimize out
1021  // unnecessary memory access alignment masks. Conventions used in
1022  // the comments below:
1023  // x, y: arbitrary values
1024  // K, L, M: arbitrary constants
1025  // (-1 << K) == mask: the right-hand side of the bitwise AND.
1026  if (IsNegativePowerOfTwo(right_value_signed)) {
1027  uint64_t mask = right_value;
1028  int K = base::bits::CountTrailingZeros64(mask);
1029  V<Word> x, y;
1030  {
1031  int L;
1032  // (x << L) & (-1 << K)
1033  // => x << L iff L >= K
1034  if (matcher_.MatchConstantLeftShift(left, &x, rep, &L) &&
1035  L >= K) {
1036  return left;
1037  }
1038  }
1039 
1040  if (matcher_.MatchWordAdd(left, &x, &y, rep)) {
1041  uint64_t L; // L == (M << K) iff (L & mask) == L.
1042 
1043  // (x + (M << K)) & (-1 << K)
1044  // => (x & (-1 << K)) + (M << K)
1045  if (matcher_.MatchIntegralWordConstant(y, rep, &L) &&
1046  (L & mask) == L) {
1047  return __ WordAdd(__ WordBitwiseAnd(x, right, rep),
1048  __ WordConstant(L, rep), rep);
1049  }
1050 
1051  // (x1 * (M << K) + y) & (-1 << K)
1052  // => x1 * (M << K) + (y & (-1 << K))
1053  V<Word> x1, x2, y1, y2;
1054  if (matcher_.MatchWordMul(x, &x1, &x2, rep) &&
1055  matcher_.MatchIntegralWordConstant(x2, rep, &L) &&
1056  (L & mask) == L) {
1057  return __ WordAdd(x, __ WordBitwiseAnd(y, right, rep), rep);
1058  }
1059  // Same as above with swapped order:
1060  // (x + y1 * (M << K)) & (-1 << K)
1061  // => (x & (-1 << K)) + y1 * (M << K)
1062  if (matcher_.MatchWordMul(y, &y1, &y2, rep) &&
1063  matcher_.MatchIntegralWordConstant(y2, rep, &L) &&
1064  (L & mask) == L) {
1065  return __ WordAdd(__ WordBitwiseAnd(x, right, rep), y, rep);
1066  }
1067 
1068  // ((x1 << K) + y) & (-1 << K)
1069  // => (x1 << K) + (y & (-1 << K))
1070  int K2;
1071  if (matcher_.MatchConstantLeftShift(x, &x1, rep, &K2) &&
1072  K2 == K) {
1073  return __ WordAdd(x, __ WordBitwiseAnd(y, right, rep), rep);
1074  }
1075  // Same as above with swapped order:
1076  // (x + (y1 << K)) & (-1 << K)
1077  // => (x & (-1 << K)) + (y1 << K)
1078  if (matcher_.MatchConstantLeftShift(y, &y1, rep, &K2) &&
1079  K2 == K) {
1080  return __ WordAdd(__ WordBitwiseAnd(x, right, rep), y, rep);
1081  }
1082  } else if (matcher_.MatchWordMul(left, &x, &y, rep)) {
1083  // (x * (M << K)) & (-1 << K) => x * (M << K)
1084  uint64_t L; // L == (M << K) iff (L & mask) == L.
1085  if (matcher_.MatchIntegralWordConstant(y, rep, &L) &&
1086  (L & mask) == L) {
1087  return left;
1088  }
1089  }
1090  }
1091  break;
1092  case WordBinopOp::Kind::kSignedDiv:
1093  return ReduceSignedDiv(left, right_value_signed, rep);
1094  case WordBinopOp::Kind::kUnsignedDiv:
1095  return ReduceUnsignedDiv(left, right_value, rep);
1096  case WordBinopOp::Kind::kSignedMod:
1097  // left % 0 => 0
1098  // left % 1 => 0
1099  // left % -1 => 0
1100  if (right_value_signed == any_of(0, 1, -1)) {
1101  return __ WordConstant(0, rep);
1102  }
1103  if (right_value_signed != rep.MinSignedValue()) {
1104  right_value_signed = Abs(right_value_signed);
1105  }
1106  // left % 2^n => ((left + m) & (2^n - 1)) - m
1107  // where m = (left >> bits-1) >>> bits-n
1108  // This is a branch-free version of the following:
1109  // left >= 0 ? left & (2^n - 1)
1110  // : ((left + (2^n - 1)) & (2^n - 1)) - (2^n - 1)
1111  // Adding and subtracting (2^n - 1) before and after the bitwise-and
1112  // keeps the result congruent modulo 2^n, but shifts the resulting
1113  // value range to become -(2^n - 1) ... 0.
1114  if (base::bits::IsPowerOfTwo(right_value_signed)) {
1115  uint32_t bits = rep.bit_width();
1116  uint32_t n = base::bits::WhichPowerOfTwo(right_value_signed);
1117  V<Word> m = __ ShiftRightLogical(
1118  __ ShiftRightArithmetic(left, bits - 1, rep), bits - n, rep);
1119  return __ WordSub(
1120  __ WordBitwiseAnd(__ WordAdd(left, m, rep),
1121  __ WordConstant(right_value_signed - 1, rep),
1122  rep),
1123  m, rep);
1124  }
1125  // The `IntDiv` with a constant right-hand side will be turned into a
1126  // multiplication, avoiding the expensive integer division.
1127  return __ WordSub(
1128  left, __ WordMul(__ IntDiv(left, right, rep), right, rep), rep);
1129  case WordBinopOp::Kind::kUnsignedMod:
1130  // left % 0 => 0
1131  // left % 1 => 0
1132  if (right_value == 0 || right_value == 1) {
1133  return __ WordConstant(0, rep);
1134  }
1135  // x % 2^n => x & (2^n - 1)
1136  if (base::bits::IsPowerOfTwo(right_value)) {
1137  return __ WordBitwiseAnd(
1138  left, __ WordConstant(right_value - 1, rep), rep);
1139  }
1140  // The `UintDiv` with a constant right-hand side will be turned into a
1141  // multiplication, avoiding the expensive integer division.
1142  return __ WordSub(
1143  left, __ WordMul(right, __ UintDiv(left, right, rep), rep), rep);
1144  case WordBinopOp::Kind::kSignedMulOverflownBits:
1145  case WordBinopOp::Kind::kUnsignedMulOverflownBits:
1146  break;
1147  }
1148  }
1149 
1150  if (kind == Kind::kAdd) {
1151  V<Word> x, y, zero;
1152  // (0 - x) + y => y - x
1153  if (matcher_.MatchWordSub(left, &zero, &x, rep) &&
1154  matcher_.MatchZero(zero)) {
1155  y = right;
1156  return __ WordSub(y, x, rep);
1157  }
1158  // x + (0 - y) => x - y
1159  if (matcher_.MatchWordSub(right, &zero, &y, rep) &&
1160  matcher_.MatchZero(zero)) {
1161  x = left;
1162  return __ WordSub(x, y, rep);
1163  }
1164  }
1165 
1166  // 0 / right => 0
1167  // 0 % right => 0
1168  if (matcher_.MatchZero(left) &&
1169  kind == any_of(Kind::kSignedDiv, Kind::kUnsignedDiv, Kind::kUnsignedMod,
1170  Kind::kSignedMod)) {
1171  return __ WordConstant(0, rep);
1172  }
1173 
1174  if (left == right) {
1175  V<Word> x = left;
1176  switch (kind) {
1177  // x & x => x
1178  // x | x => x
1179  case WordBinopOp::Kind::kBitwiseAnd:
1180  case WordBinopOp::Kind::kBitwiseOr:
1181  return x;
1182  // x ^ x => 0
1183  // x - x => 0
1184  // x % x => 0
1185  case WordBinopOp::Kind::kBitwiseXor:
1186  case WordBinopOp::Kind::kSub:
1187  case WordBinopOp::Kind::kSignedMod:
1188  case WordBinopOp::Kind::kUnsignedMod:
1189  return __ WordConstant(0, rep);
1190  // x / x => x != 0
1191  case WordBinopOp::Kind::kSignedDiv:
1192  case WordBinopOp::Kind::kUnsignedDiv: {
1193  V<Word> zero = __ WordConstant(0, rep);
1194  V<Word32> result = __ Word32Equal(__ Equal(left, zero, rep), 0);
1195  return __ ZeroExtendWord32ToRep(result, rep);
1196  }
1197  case WordBinopOp::Kind::kAdd:
1198  case WordBinopOp::Kind::kMul:
1199  case WordBinopOp::Kind::kSignedMulOverflownBits:
1200  case WordBinopOp::Kind::kUnsignedMulOverflownBits:
1201  break;
1202  }
1203  }
1204 
1205  if (std::optional<V<Word>> ror = TryReduceToRor(left, right, kind, rep)) {
1206  return *ror;
1207  }
1208 
1209  return Next::ReduceWordBinop(left, right, kind, rep);
1210  }

1231  std::optional<V<Word>> TryReduceToRor(V<Word> left, V<Word> right,
1232  WordBinopOp::Kind kind,
1233  WordRepresentation rep) {
1234  // Recognize rotation, we are matcher_.Matching and transforming as follows
1235  // (assuming kWord32, kWord64 is handled correspondingly):
1236  // x << y | x >>> (32 - y) => x ror (32 - y)
1237  // x << (32 - y) | x >>> y => x ror y
1238  // x << y ^ x >>> (32 - y) => x ror (32 - y) if 1 <= y < 32
1239  // x << (32 - y) ^ x >>> y => x ror y if 1 <= y < 32
1240  // (As well as the commuted forms.)
1241  // Note the side condition for XOR: the optimization doesn't hold for
1242  // an effective rotation amount of 0.
1243 
1244  if (!(kind == any_of(WordBinopOp::Kind::kBitwiseOr,
1245  WordBinopOp::Kind::kBitwiseXor))) {
1246  return {};
1247  }
1248 
1249  const ShiftOp* high = matcher_.TryCast<ShiftOp>(left);
1250  if (!high) return {};
1251  const ShiftOp* low = matcher_.TryCast<ShiftOp>(right);
1252  if (!low) return {};
1253 
1254  if (low->kind == ShiftOp::Kind::kShiftLeft) {
1255  std::swap(low, high);
1256  }
1257  if (high->kind != ShiftOp::Kind::kShiftLeft ||
1258  low->kind != ShiftOp::Kind::kShiftRightLogical) {
1259  return {};
1260  }
1261  V<Word> x = high->left();
1262  if (low->left() != x) return {};
1263 
1264  if (uint64_t k1, k2;
1265  matcher_.MatchIntegralWordConstant(high->right(), rep, &k1) &&
1266  matcher_.MatchIntegralWordConstant(low->right(), rep, &k2) &&
1267  k1 + k2 == rep.bit_width() && k1 >= 0 && k2 >= 0) {
1268  if (k1 == 0 || k2 == 0) {
1269  if (kind == WordBinopOp::Kind::kBitwiseXor) {
1270  return __ WordConstant(0, rep);
1271  } else {
1272  DCHECK_EQ(kind, WordBinopOp::Kind::kBitwiseOr);
1273  return x;
1274  }
1275  }
1276  return __ RotateRight(x, low->right(), rep);
1277  }
1278 
1279  if (kind == WordBinopOp::Kind::kBitwiseXor) {
1280  // Can't guarantee that rotation amount is not 0.
1281  return {};
1282  }
1283  DCHECK_EQ(kind, WordBinopOp::Kind::kBitwiseOr);
1284 
1285  V<Word> a, b;
1286  uint64_t k;
1287  if (matcher_.MatchWordSub(high->right(), &a, &b, rep) &&
1288  matcher_.MatchIntegralWordConstant(a, rep, &k) && b == low->right() &&
1289  k == rep.bit_width()) {
1290  return __ RotateRight(x, b, rep);
1291  } else if (matcher_.MatchWordSub(low->right(), &a, &b, rep) &&
1292  matcher_.MatchIntegralWordConstant(a, rep, &k) &&
1293  b == high->right() && k == rep.bit_width()) {
1294  return __ RotateRight(x, low->right(), rep);
1295  }
1296 
1297  return {};
1298  }

1352  }

1356  // left - 0 => (left, false)
1357  if (kind == any_of(Kind::kSignedAdd, Kind::kSignedSub) &&

1360  }

1401  if (V<Word32> x;
1402  matcher_.MatchConstantShiftRightArithmeticShiftOutZeros(
1403  t1, &x, WordRepresentation::Word32(), &amount) &&
1404  amount == 1) {
1405  if (int32_t k; matcher_.MatchIntegralWord32Constant(tk, &k)) {
1406  switch (bitwise_op_kind) {
1407  case WordBinopOp::Kind::kBitwiseAnd:
1408  case WordBinopOp::Kind::kBitwiseOr:
1409  case WordBinopOp::Kind::kBitwiseXor:
1410  // If the topmost two bits are not identical then retagging
1411  // the smi could cause an overflow, so we do not optimize that
1412  // here.
1413  if (((k >> 31) & 0b1) == ((k >> 30) & 0b1)) {
1414  return __ Tuple(__ WordBinop(x, __ Word32Constant(k << 1),
1415  bitwise_op_kind,
1416  WordRepresentation::Word32()),
1417  __ Word32Constant(0));
1418  }
1419  break;
1420  default:
1421  break;
1422  }
1423  }
1424  }
1425  }
1426  }
1427 
1428  return Next::ReduceOverflowCheckedBinop(left, right, kind, rep);

1762  // (x >> K) << L => x if K == L

2684  V<Word> ReduceSignedDiv(V<Word> left, int64_t right, WordRepresentation rep) {
2685  // left / -1 => 0 - left
2686  if (right == -1) {
2687  return __ WordSub(__ WordConstant(0, rep), left, rep);
2688  }
2689  // left / 0 => 0
2690  if (right == 0) {
2691  return __ WordConstant(0, rep);
2692  }
2693  // left / 1 => left
2694  if (right == 1) {
2695  return left;
2696  }
2697  // left / MinSignedValue => left == MinSignedValue
2698  if (right == rep.MinSignedValue()) {
2699  V<Word32> equal_op = __ Equal(left, __ WordConstant(right, rep), rep);
2700  if (rep == WordRepresentation::Word64()) {
2701  return __ ChangeUint32ToUint64(equal_op);
2702  } else {
2703  return equal_op;
2704  }
2705  }
2706  // left / -right => -(left / right)
2707  if (right < 0) {
2708  DCHECK_NE(right, rep.MinSignedValue());
2709  return __ WordSub(__ WordConstant(0, rep),
2710  ReduceSignedDiv(left, Abs(right), rep), rep);
2711  }
2712 
2713  V<Word> quotient = left;
2714  if (base::bits::IsPowerOfTwo(right)) {
2715  uint32_t shift = base::bits::WhichPowerOfTwo(right);
2716  DCHECK_GT(shift, 0);
