37 void FinalizeBuiltin(
38  Builtin builtin,
39  std::array<BuiltinEffects, Builtins::kBuiltinCount>& builtins_effects) {
40  BuiltinEffects& effects = builtins_effects[static_cast<size_t>(builtin)];
41 
42  if (effects.is_finalized) return;
43 
44  if (effects.can_allocate == BuiltinAllocateEffect::kUninitialized) {
45  // This builtin wasn't compiled with Turbofan; we assume that it can
46  // allocate.
47  // TODO(dmercadier): for CPP builtins, we could have a mechanism similar as
48  // Runtime functions: annotate them with can_allocate/cant_allocate and
49  // enforce this with a DisallowGarbageCollection scope.
50  effects.can_allocate = BuiltinAllocateEffect::kYes;
51  effects.is_finalized = true;
52  return;
53  }
54 
55  if (effects.can_allocate != BuiltinAllocateEffect::kMaybeWithBuiltinCall) {
56  effects.is_finalized = true;
57  return;
58  }
59 
60  for (Builtin callee : effects.callees) {
61  BuiltinEffects& callee_effects =
62  builtins_effects[static_cast<size_t>(callee)];
63  if (!callee_effects.is_finalized) {
64  // Note that we rely on builtins not recursviely calling each others. So
65  // far, this always hold. If it ever stops holding, then we'll have
66  // unbounded recursion here, and we can then think about fixing it.
67  FinalizeBuiltin(callee, builtins_effects);
68  }
69  if (callee_effects.can_allocate == BuiltinAllocateEffect::kYes) {
70  effects.can_allocate = BuiltinAllocateEffect::kYes;
71  effects.is_finalized = true;
72  return;
73  }
74  }
75 
76  // None of the callees can allocate.
77  effects.can_allocate = BuiltinAllocateEffect::kNo;
78  effects.is_finalized = true;
79 }

81 void BuiltinsEffectsAnalyzer::Finalize() {
82  CHECK(!is_finalized_);
83 
84  for (size_t i = 0; i < builtins_effects_.size(); i++) {
85  FinalizeBuiltin(static_cast<Builtin>(i), builtins_effects_);
86  CHECK(builtins_effects_[i].is_finalized);
87  }
88 
89  is_finalized_ = true;
90 }
