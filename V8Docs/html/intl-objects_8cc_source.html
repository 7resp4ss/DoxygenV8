123 bool ToUpperFastASCII(base::Vector<const uint16_t> src,
124  DirectHandle<SeqOneByteString> result) {
125  // Do a faster loop for the case where all the characters are ASCII.
126  uint16_t ored = 0;
127  int32_t index = 0;
128  for (const uint16_t* it = src.begin(); it != src.end(); ++it) {
129  uint16_t ch = *it;
130  ored |= ch;
131  result->SeqOneByteStringSet(index++, ToAsciiUpper(ch));
132  }
133  return !(ored & ~0x7F);
134 }

138 template <typename Char>
139 bool ToUpperOneByte(base::Vector<const Char> src, uint8_t* dest,
140  int* sharp_s_count) {
141  // Still pretty-fast path for the input with non-ASCII Latin-1 characters.
142 
143  // There are two special cases.
144  // 1. U+00B5 and U+00FF are mapped to a character beyond U+00FF.
145  // 2. Lower case sharp-S converts to "SS" (two characters)
146  *sharp_s_count = 0;
147  for (auto it = src.begin(); it != src.end(); ++it) {
148  uint8_t ch = AsOneByte(*it);
149  if (V8_UNLIKELY(ch == sharp_s)) {
150  ++(*sharp_s_count);
151  continue;
152  }
153  if (V8_UNLIKELY(ch == 0xB5 || ch == 0xFF)) {
154  // Since this upper-cased character does not fit in an 8-bit string, we
155  // need to take the 16-bit path.
156  return false;
157  }
158  *dest++ = ToLatin1Upper(ch);
159  }
160 
161  return true;
162 }

219 icu::UnicodeString Intl::ToICUUnicodeString(Isolate* isolate,
220  DirectHandle<String> string,
221  int offset) {
222  DCHECK(string->IsFlat());
223  DisallowGarbageCollection no_gc;
224  std::unique_ptr<base::uc16[]> sap;
225  // Short one-byte strings can be expanded on the stack to avoid allocating a
226  // temporary buffer.
227  constexpr unsigned int kShortStringSize = 80;
228  UChar short_string_buffer[kShortStringSize];
229  const UChar* uchar_buffer = nullptr;
230  const String::FlatContent& flat = string->GetFlatContent(no_gc);
231  // We read the length from the heap, so it may be untrusted (in the sandbox
232  // attacker model) and we therefore need to use an unsigned int here when
233  // comparing it against the kShortStringSize.
234  uint32_t length = string->length();
235  // As the length is untrusted, we should also double-check the offset here.
236  // Technically it's fine for the offset to go out-of-bounds since it will
237  // only cause an OOB read (not write) outside the sandbox, but we still want
238  // to avoid these issues. An alternative would be for this function to
239  // allocate temporary buffers inside the sandbox, then there would be no need
240  // for additional bounds checks here (since we're using 32-bit sizes/offsets).
241  SBXCHECK_LE(offset, length);
242  if (flat.IsOneByte() && length <= kShortStringSize) {
243  CopyChars(short_string_buffer, flat.ToOneByteVector().begin(), length);
244  uchar_buffer = short_string_buffer;
245  } else {
246  uchar_buffer = GetUCharBufferFromFlat(flat, &sap, length);
247  }
248  return icu::UnicodeString(uchar_buffer + offset, length - offset);
249 }

271 MaybeHandle<String> LocaleConvertCase(Isolate* isolate, DirectHandle<String> s,
272  bool is_to_upper, const char* lang) {
273  auto case_converter = is_to_upper ? u_strToUpper : u_strToLower;
274  uint32_t src_length = s->length();
275  uint32_t dest_length = src_length;
276  UErrorCode status;
277  Handle<SeqTwoByteString> result;
278  std::unique_ptr<base::uc16[]> sap;
279 
280  if (dest_length == 0) return isolate->factory()->empty_string();
281 
282  // This is not a real loop. It'll be executed only once (no overflow) or
283  // twice (overflow).
284  for (int i = 0; i < 2; ++i) {
285  // Case conversion can increase the string length (e.g. sharp-S => SS) so
286  // that we have to handle RangeError exceptions here.
287  ASSIGN_RETURN_ON_EXCEPTION(
288  isolate, result, isolate->factory()->NewRawTwoByteString(dest_length));
289  DisallowGarbageCollection no_gc;
290  DCHECK(s->IsFlat());
291  String::FlatContent flat = s->GetFlatContent(no_gc);
292  const UChar* src = GetUCharBufferFromFlat(flat, &sap, src_length);
293  status = U_ZERO_ERROR;
294  dest_length =
295  case_converter(reinterpret_cast<UChar*>(result->GetChars(no_gc)),
296  dest_length, src, src_length, lang, &status);
297  if (status != U_BUFFER_OVERFLOW_ERROR) break;
298  }
299 
300  // In most cases, the output will fill the destination buffer completely
301  // leading to an unterminated string (U_STRING_NOT_TERMINATED_WARNING).
302  // Only in rare cases, it'll be shorter than the destination buffer and
303  // |result| has to be truncated.
304  DCHECK(U_SUCCESS(status));
305  if (V8_LIKELY(status == U_STRING_NOT_TERMINATED_WARNING)) {
306  DCHECK(dest_length == result->length());
307  return result;
308  }
309  DCHECK(dest_length < result->length());
310  return SeqString::Truncate(isolate, result, dest_length);
311 }

315 // A stripped-down version of ConvertToLower that can only handle flat one-byte
316 // strings and does not allocate. Note that {src} could still be, e.g., a
317 // one-byte sliced string with a two-byte parent string.
318 // Called from TF builtins.
319 Tagged<String> Intl::ConvertOneByteToLower(Tagged<String> src,
320  Tagged<String> dst) {
321  DCHECK_EQ(src->length(), dst->length());
322  DCHECK(src->IsOneByteRepresentation());
323  DCHECK(src->IsFlat());
324  DCHECK(IsSeqOneByteString(dst));
325 
326  DisallowGarbageCollection no_gc;
327 
328  const int length = src->length();
329  String::FlatContent src_flat = src->GetFlatContent(no_gc);
330  uint8_t* dst_data = Cast<SeqOneByteString>(dst)->GetChars(no_gc);
331 
332  if (src_flat.IsOneByte()) {
333  const uint8_t* src_data = src_flat.ToOneByteVector().begin();
334 
335  int ascii_prefix_length = FastAsciiConvert<unibrow::ToLowercase>(
336  reinterpret_cast<char*>(dst_data),
337  reinterpret_cast<const char*>(src_data), length);
338 
339  if (ascii_prefix_length == length) return dst;
340 
341  // If not ASCII, we keep the result up to index_to_first_unprocessed and
342  // process the rest.
343  for (int index = ascii_prefix_length; index < length; ++index) {
344  dst_data[index] = ToLatin1Lower(src_data[index]);
345  }
346  } else {
347  DCHECK(src_flat.IsTwoByte());
348  int index_to_first_unprocessed = FindFirstUpperOrNonAscii(src, length);
349  if (index_to_first_unprocessed == length) return src;
350 
351  const uint16_t* src_data = src_flat.ToUC16Vector().begin();
352  CopyChars(dst_data, src_data, index_to_first_unprocessed);
353  for (int index = index_to_first_unprocessed; index < length; ++index) {
354  // Truncating cast of two-byte src character to one-byte value. For valid
355  // cases (where a one-byte sliced string points to a two-byte parent) this
356  // will not lose any information, but we need to truncate anyway to
357  // avoid undefined behavior if the parent string is corrupted.
358  dst_data[index] = ToLatin1Lower(AsOneByte(src_data[index]));
359  }
360  }
361 
362  return dst;
363 }

365 MaybeHandle<String> Intl::ConvertToLower(Isolate* isolate,
366  DirectHandle<String> s) {
367  if (!s->IsOneByteRepresentation()) {
368  // Use a slower implementation for strings with characters beyond U+00FF.
369  return LocaleConvertCase(isolate, s, false, "");
370  }
371 
372  int length = s->length();
373 
374  // We depend here on the invariant that the length of a Latin1
375  // string is invariant under ToLowerCase, and the result always
376  // fits in the Latin1 range in the *root locale*. It does not hold
377  // for ToUpperCase even in the root locale.
378 
379  // Scan the string for uppercase and non-ASCII characters for strings
380  // shorter than a machine-word without any memory allocation overhead.
381  // TODO(jshin): Apply this to a longer input by breaking FastAsciiConvert()
382  // to two parts, one for scanning the prefix with no change and the other for
383  // handling ASCII-only characters.
384 
385  bool is_short = length < static_cast<int>(sizeof(uintptr_t));
386  if (is_short) {
387  bool is_lower_ascii = FindFirstUpperOrNonAscii(*s, length) == length;
388  if (is_lower_ascii) return indirect_handle(s, isolate);
389  }
390 
391  DirectHandle<SeqOneByteString> result =
392  isolate->factory()->NewRawOneByteString(length).ToHandleChecked();
393 
394  return handle(Intl::ConvertOneByteToLower(*s, *result), isolate);
395 }

397 MaybeDirectHandle<String> Intl::ConvertToUpper(Isolate* isolate,
398  DirectHandle<String> s) {
399  uint32_t length = s->length();
400  if (s->IsOneByteRepresentation() && length > 0) {
401  uint32_t prefix;
402  {
403  DisallowGarbageCollection no_gc;
404  String::FlatContent flat = s->GetFlatContent(no_gc);
405  prefix = FastAsciiCasePrefixLength<unibrow::ToUppercase>(
406  reinterpret_cast<const char*>(flat.ToOneByteVector().begin()),
407  length);
408  if (prefix == length) {
409  return indirect_handle(s, isolate);
410  }
411  }
412  Handle<SeqOneByteString> result =
413  isolate->factory()->NewRawOneByteString(length).ToHandleChecked();
414 
415  DCHECK(s->IsFlat());
416  int sharp_s_count;
417  bool is_result_single_byte;
418  {
419  DisallowGarbageCollection no_gc;
420  String::FlatContent flat = s->GetFlatContent(no_gc);
421  uint8_t* dest = result->GetChars(no_gc);
422  if (flat.IsOneByte()) {
423  base::Vector<const uint8_t> src = flat.ToOneByteVector();
424  std::memcpy(result->GetChars(no_gc), src.begin(), prefix);
425  uint32_t ascii_prefix_length =
426  FastAsciiConvert<unibrow::ToUppercase>(
427  reinterpret_cast<char*>(result->GetChars(no_gc)) + prefix,
428  reinterpret_cast<const char*>(src.begin()) + prefix,
429  length - prefix) +
430  prefix;
431  if (ascii_prefix_length == length) return result;
432  // If not ASCII, we keep the result up to index_to_first_unprocessed and
433  // process the rest.
434  is_result_single_byte =
435  ToUpperOneByte(src.SubVector(ascii_prefix_length, length),
436  dest + ascii_prefix_length, &sharp_s_count);
437  } else {
438  DCHECK(flat.IsTwoByte());
439  base::Vector<const uint16_t> src = flat.ToUC16Vector();
440  if (ToUpperFastASCII(src, result)) return result;
441  is_result_single_byte = ToUpperOneByte(src, dest, &sharp_s_count);
442  }
443  }
444 
445  // Go to the full Unicode path if there are characters whose uppercase
446  // is beyond the Latin-1 range (cannot be represented in OneByteString).
447  if (V8_UNLIKELY(!is_result_single_byte)) {
448  return LocaleConvertCase(isolate, s, true, "");
449  }
450 
451  if (sharp_s_count == 0) return result;
452 
453  // We have sharp_s_count sharp-s characters, but the result is still
454  // in the Latin-1 range.
455  ASSIGN_RETURN_ON_EXCEPTION(
456  isolate, result,
457  isolate->factory()->NewRawOneByteString(length + sharp_s_count));
458  DisallowGarbageCollection no_gc;
459  String::FlatContent flat = s->GetFlatContent(no_gc);
460  if (flat.IsOneByte()) {
461  ToUpperWithSharpS(flat.ToOneByteVector(), result);
462  } else {
463  ToUpperWithSharpS(flat.ToUC16Vector(), result);
464  }
465 
466  return result;
467  }
468 
469  return LocaleConvertCase(isolate, s, true, "");
470 }

590 bool ValidateResource(const icu::Locale locale, const char* path,
591  const char* key) {
592  bool result = false;
593  UErrorCode status = U_ZERO_ERROR;
594  UResourceBundle* bundle = ures_open(path, locale.getName(), &status);
595  if (bundle != nullptr && status == U_ZERO_ERROR) {
596  if (key == nullptr) {
597  result = true;
598  } else {
599  UResourceBundle* key_bundle =
600  ures_getByKey(bundle, key, nullptr, &status);
601  result = key_bundle != nullptr && (status == U_ZERO_ERROR);
602  ures_close(key_bundle);
603  }
604  }
605  ures_close(bundle);
606  if (!result) {
607  if ((locale.getCountry()[0] != '\0') && (locale.getScript()[0] != '\0')) {
608  // Fallback to try without country.
609  std::string without_country(locale.getLanguage());
610  without_country = without_country.append("-").append(locale.getScript());
611  return ValidateResource(without_country.c_str(), path, key);
612  } else if ((locale.getCountry()[0] != '\0') ||
613  (locale.getScript()[0] != '\0')) {
614  // Fallback to try with only language.
615  std::string language(locale.getLanguage());
616  return ValidateResource(language.c_str(), path, key);
617  }
618  }
619  return result;
620 }

734 // static
735 Maybe<std::string> Intl::ValidateAndCanonicalizeUnicodeLocaleId(
736  Isolate* isolate, std::string_view locale_in) {
737  if (!IsStructurallyValidLanguageTag(locale_in)) {
738  THROW_NEW_ERROR_RETURN_VALUE(
739  isolate,
740  NewRangeError(MessageTemplate::kInvalidLanguageTag,
741  isolate->factory()->NewStringFromAsciiChecked(locale_in)),
742  Nothing<std::string>());
743  }
744 
745  std::string locale(locale_in);
746 
747  // Optimize for the most common case: a 2-letter language code in the
748  // canonical form/lowercase that is not one of the deprecated codes
749  // (in, iw, ji, jw). Don't check for ~70 of 3-letter deprecated language
750  // codes. Instead, let them be handled by ICU in the slow path. However,
751  // fast-track 'fil' (3-letter canonical code).
752  if ((IsTwoLetterLanguage(locale) && !IsDeprecatedOrLegacyLanguage(locale)) ||
753  locale == "fil") {
754  return Just(locale);
755  }
756 
757  // Because per BCP 47 2.1.1 language tags are case-insensitive, lowercase
758  // the input before any more check.
759  std::transform(locale.begin(), locale.end(), locale.begin(), ToAsciiLower);
760 
761  // // ECMA 402 6.2.3
762  // TODO(jshin): uloc_{for,to}LanguageTag can fail even for a structurally
763  // valid language tag if it's too long (much longer than 100 chars). Even if
764  // we allocate a longer buffer, ICU will still fail if it's too long. Either
765  // propose to Ecma 402 to put a limit on the locale length or change ICU to
766  // handle long locale names better. See
767  // https://unicode-org.atlassian.net/browse/ICU-13417
768  UErrorCode error = U_ZERO_ERROR;
769  // uloc_forLanguageTag checks the structural validity. If the input BCP47
770  // language tag is parsed all the way to the end, it indicates that the input
771  // is structurally valid. Due to a couple of bugs, we can't use it
772  // without Chromium patches or ICU 62 or earlier.
773  icu::Locale icu_locale = icu::Locale::forLanguageTag(locale.c_str(), error);
774 
775  if (U_FAILURE(error) || icu_locale.isBogus()) {
776  THROW_NEW_ERROR_RETURN_VALUE(
777  isolate,
778  NewRangeError(
779  MessageTemplate::kInvalidLanguageTag,
780  isolate->factory()->NewStringFromAsciiChecked(locale.c_str())),
781  Nothing<std::string>());
782  }
783 
784  // Use LocaleBuilder to validate locale.
785  icu_locale = icu::LocaleBuilder().setLocale(icu_locale).build(error);
786  icu_locale.canonicalize(error);
787  if (U_FAILURE(error) || icu_locale.isBogus()) {
788  THROW_NEW_ERROR_RETURN_VALUE(
789  isolate,
790  NewRangeError(
791  MessageTemplate::kInvalidLanguageTag,
792  isolate->factory()->NewStringFromAsciiChecked(locale.c_str())),
793  Nothing<std::string>());
794  }
795  Maybe<std::string> maybe_to_language_tag = Intl::ToLanguageTag(icu_locale);
796  if (maybe_to_language_tag.IsNothing()) {
797  THROW_NEW_ERROR_RETURN_VALUE(
798  isolate,
799  NewRangeError(
800  MessageTemplate::kInvalidLanguageTag,
801  isolate->factory()->NewStringFromAsciiChecked(locale.c_str())),
802  Nothing<std::string>());
803  }
804 
805  return maybe_to_language_tag;
806 }

808 Maybe<std::vector<std::string>> Intl::CanonicalizeLocaleList(
809  Isolate* isolate, DirectHandle<Object> locales,
810  bool only_return_one_result) {
811  // 1. If locales is undefined, then
812  if (IsUndefined(*locales, isolate)) {
813  // 1a. Return a new empty List.
814  return Just(std::vector<std::string>());
815  }
816  // 2. Let seen be a new empty List.
817  std::vector<std::string> seen;
818  // 3. If Type(locales) is String or locales has an [[InitializedLocale]]
819  // internal slot, then
820  if (IsJSLocale(*locales)) {
821  // Since this value came from JSLocale, which is already went though the
822  // CanonializeLanguageTag process once, therefore there are no need to
823  // call CanonializeLanguageTag again.
824  seen.push_back(JSLocale::ToString(Cast<JSLocale>(locales)));
825  return Just(seen);
826  }
827  if (IsString(*locales)) {
828  // 3a. Let O be CreateArrayFromList( locales ).
829  // Instead of creating a one-element array and then iterating over it,
830  // we inline the body of the iteration:
831  std::string canonicalized_tag;
832  if (!CanonicalizeLanguageTag(isolate, locales).To(&canonicalized_tag)) {
833  return Nothing<std::vector<std::string>>();
834  }
835  seen.push_back(canonicalized_tag);
836  return Just(seen);
837  }
838  // 4. Else,
839  // 4a. Let O be ? ToObject(locales).
840  DirectHandle<JSReceiver> o;
841  ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate, o,
842  Object::ToObject(isolate, locales),
843  Nothing<std::vector<std::string>>());
844  // 5. Let len be ? ToLength(? Get(O, "length")).
845  DirectHandle<Object> length_obj;
846  ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate, length_obj,
847  Object::GetLengthFromArrayLike(isolate, o),
848  Nothing<std::vector<std::string>>());
849  // TODO(jkummerow): Spec violation: strictly speaking, we have to iterate
850  // up to 2^53-1 if {length_obj} says so. Since cases above 2^32 probably
851  // don't happen in practice (and would be very slow if they do), we'll keep
852  // the code simple for now by using a saturating to-uint32 conversion.
853  double raw_length = Object::NumberValue(*length_obj);
854  uint32_t len =
855  raw_length >= kMaxUInt32 ? kMaxUInt32 : static_cast<uint32_t>(raw_length);
856  // 6. Let k be 0.
857  // 7. Repeat, while k < len
858  for (uint32_t k = 0; k < len; k++) {
859  // 7a. Let Pk be ToString(k).
860  // 7b. Let kPresent be ? HasProperty(O, Pk).
861  LookupIterator it(isolate, o, k);
862  Maybe<bool> maybe_found = JSReceiver::HasProperty(&it);
863  MAYBE_RETURN(maybe_found, Nothing<std::vector<std::string>>());
864  // 7c. If kPresent is true, then
865  if (!maybe_found.FromJust()) continue;
866  // 7c i. Let kValue be ? Get(O, Pk).
867  DirectHandle<Object> k_value;
868  ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate, k_value, Object::GetProperty(&it),
869  Nothing<std::vector<std::string>>());
870  // 7c ii. If Type(kValue) is not String or Object, throw a TypeError
871  // exception.
872  // 7c iii. If Type(kValue) is Object and kValue has an [[InitializedLocale]]
873  // internal slot, then
874  std::string canonicalized_tag;
875  if (IsJSLocale(*k_value)) {
876  // 7c iii. 1. Let tag be kValue.[[Locale]].
877  canonicalized_tag = JSLocale::ToString(Cast<JSLocale>(k_value));
878  // 7c iv. Else,
879  } else {
880  // 7c iv 1. Let tag be ? ToString(kValue).
881  // 7c v. If IsStructurallyValidLanguageTag(tag) is false, throw a
882  // RangeError exception.
883  // 7c vi. Let canonicalizedTag be CanonicalizeLanguageTag(tag).
884  if (!CanonicalizeLanguageTag(isolate, k_value).To(&canonicalized_tag)) {
885  return Nothing<std::vector<std::string>>();
886  }
887  }
888  // 7c vi. If canonicalizedTag is not an element of seen, append
889  // canonicalizedTag as the last element of seen.
890  if (std::find(seen.begin(), seen.end(), canonicalized_tag) == seen.end()) {
891  seen.push_back(canonicalized_tag);
892  }
893  // 7d. Increase k by 1. (See loop header.)
894  // Optimization: some callers only need one result.
895  if (only_return_one_result) return Just(seen);
896  }
897  // 8. Return seen.
898  return Just(seen);
899 }

1070 // print("constexpr uint8_t %s[256] = {" % array_name, end = "")

1074 // print("\n ", end = "")

1080 // l1s = []

1150 template <class Char1T, class Char2T>
1151 bool FastCompareFlatString(const Char1T* lhs, const Char2T* rhs, int length,
1152  FastCompareStringsData* d) {
1153  for (int i = 0; i < length; i++) {
1154  const Char1T l = lhs[i];
1155  const Char2T r = rhs[i];
1156  if (!CanFastCompare(l) || !CanFastCompare(r)) {
1157  d->processed_until = i;
1158  return false;
1159  }
1160  UCollationResult l1_result =
1161  ToUCollationResult(kCollationWeightsL1[l] - kCollationWeightsL1[r]);
1162  if (l1_result != UCollationResult::UCOL_EQUAL) {
1163  d->has_diff = true;
1164  d->first_diff_at = i;
1165  d->processed_until = i;
1166  d->l1_result = l1_result;
1167  return true;
1168  }
1169  if (l != r && d->l3_result == UCollationResult::UCOL_EQUAL) {
1170  // Collapse the two-pass algorithm into one: if we find a difference in
1171  // L1 weights, that is our result. If not, use the first L3 weight
1172  // difference.
1173  UCollationResult l3_result =
1174  ToUCollationResult(kCollationWeightsL3[l] - kCollationWeightsL3[r]);
1175  d->l3_result = l3_result;
1176  if (!d->has_diff) {
1177  d->has_diff = true;
1178  d->first_diff_at = i;
1179  }
1180  }
1181  }
1182  d->processed_until = length;
1183  return true;
1184 }

1371 std::optional<UCollationResult> TryFastCompareStrings(
1372  Isolate* isolate, const icu::Collator& icu_collator,
1373  DirectHandle<String> string1, DirectHandle<String> string2,
1374  int* processed_until_out) {
1375  // TODO(jgruber): We could avoid the flattening (done by the caller) as well
1376  // by implementing comparison through string iteration. This has visible
1377  // performance benefits (e.g. 7% on CDJS) but complicates the code. Consider
1378  // doing this in the future.
1379  DCHECK(string1->IsFlat());
1380  DCHECK(string2->IsFlat());
1381 
1382  *processed_until_out = 0;
1383 
1384 #ifdef DEBUG
1385  // Checked by the caller, see CompareStringsOptionsFor.
1386  SLOW_DCHECK(CollatorAllowsFastComparison(icu_collator));
1387  USE(CollatorAllowsFastComparison);
1388 #endif // DEBUG
1389 
1390  DCHECK(!SharedStringAccessGuardIfNeeded::IsNeeded(*string1));
1391  DCHECK(!SharedStringAccessGuardIfNeeded::IsNeeded(*string2));
1392 
1393  const int length1 = string1->length();
1394  const int length2 = string2->length();
1395  int common_length = std::min(length1, length2);
1396 
1397  FastCompareStringsData d;
1398  DisallowGarbageCollection no_gc;
1399  const String::FlatContent& flat1 = string1->GetFlatContent(no_gc);
1400  const String::FlatContent& flat2 = string2->GetFlatContent(no_gc);
1401  if (!FastCompareStringFlatContent(flat1, flat2, common_length, &d)) {
1402  DCHECK_EQ(d.l1_result, UCollationResult::UCOL_EQUAL);
1403  return d.FastCompareFailed(processed_until_out);
1404  }
1405 
1406  // The result is only valid if the last processed character is not followed
1407  // by a unicode combining character (we are overly strict and restrict to
1408  // ASCII).
1409  if (!CharIsAsciiOrOutOfBounds(flat1, length1, d.processed_until + 1) ||
1410  !CharIsAsciiOrOutOfBounds(flat2, length2, d.processed_until + 1)) {
1411  return d.FastCompareFailed(processed_until_out);
1412  }
1413 
1414  if (d.l1_result != UCollationResult::UCOL_EQUAL) {
1415  return d.l1_result;
1416  }
1417 
1418  // Strings are L1-equal up to their common length, length differences win.
1419  UCollationResult length_result = ToUCollationResult(length1 - length2);
1420  if (length_result != UCollationResult::UCOL_EQUAL) {
1421  // Strings of different lengths may still compare as equal if the longer
1422  // string has a fully ignored suffix, e.g. "a" vs. "a\u{1}".
1423  if (!CharCanFastCompareOrOutOfBounds(flat1, length1, common_length) ||
1424  !CharCanFastCompareOrOutOfBounds(flat2, length2, common_length)) {
1425  return d.FastCompareFailed(processed_until_out);
1426  }
1427  return length_result;
1428  }
1429 
1430  // L1-equal and same length, the L3 result wins.
1431  return d.l3_result;
1432 }

1861  const std::string& locale) {
1862  // 1. Let candidate be locale.
1863  std::string candidate = locale;

1867  // 2.a. If availableLocales contains an element equal to candidate, return
1868  // candidate.

1870  return candidate;
1871  }

1874  // (U+002D) within candidate. If that character does not occur, return
1875  // undefined.

1877  if (pos == std::string::npos) {
1878  return std::string();

1880

1882  // decrease pos by 2.
1883  if (pos >= 2 && candidate[pos - 2] == '-') {
1884  pos -= 2;
1885  }
1886 
1887  // 2.d. Let candidate be the substring of candidate from position 0,
1888  // inclusive, to position pos, exclusive.
1889  candidate = candidate.substr(0, pos);
1890  }

1898 // Returns a struct containing a bcp47 tag without unicode extensions
1899 // and the removed unicode extensions.
1900 //
1901 // For example, given 'en-US-u-co-emoji' returns 'en-US' and
1902 // 'u-co-emoji'.
1903 ParsedLocale ParseBCP47Locale(const std::string& locale) {
1904  size_t length = locale.length();
1905  ParsedLocale parsed_locale;
1906 
1907  // Privateuse or grandfathered locales have no extension sequences.
1908  if ((length > 1) && (locale[1] == '-')) {
1909  // Check to make sure that this really is a grandfathered or
1910  // privateuse extension. ICU can sometimes mess up the
1911  // canonicalization.
1912  DCHECK(locale[0] == 'x' || locale[0] == 'i');
1913  parsed_locale.no_extensions_locale = locale;
1914  return parsed_locale;
1915  }
1916 
1917  size_t unicode_extension_start = locale.find("-u-");
1918 
1919  // No unicode extensions found.
1920  if (unicode_extension_start == std::string::npos) {
1921  parsed_locale.no_extensions_locale = locale;
1922  return parsed_locale;
1923  }
1924 
1925  size_t private_extension_start = locale.find("-x-");
1926 
1927  // Unicode extensions found within privateuse subtags don't count.
1928  if (private_extension_start != std::string::npos &&
1929  private_extension_start < unicode_extension_start) {
1930  parsed_locale.no_extensions_locale = locale;
1931  return parsed_locale;
1932  }
1933 
1934  const std::string beginning = locale.substr(0, unicode_extension_start);
1935  size_t unicode_extension_end = length;
1936  DCHECK_GT(length, 2);
1937 
1938  // Find the end of the extension production as per the bcp47 grammar
1939  // by looking for '-' followed by 2 chars and then another '-'.
1940  for (size_t i = unicode_extension_start + 1; i < length - 2; i++) {
1941  if (locale[i] != '-') continue;
1942 
1943  if (locale[i + 2] == '-') {
1944  unicode_extension_end = i;
1945  break;
1946  }
1947 
1948  i += 2;
1949  }
1950 
1951  const std::string end = locale.substr(unicode_extension_end);
1952  parsed_locale.no_extensions_locale = beginning + end;
1953  parsed_locale.extension = locale.substr(
1954  unicode_extension_start, unicode_extension_end - unicode_extension_start);
1955  return parsed_locale;
1956 }

2034 // The BestFitMatcher abstract operation compares requestedLocales, which must
2035 // be a List as returned by CanonicalizeLocaleList, against the locales in
2036 // availableLocales and determines the best available language to meet the
2037 // request. The algorithm is implementation dependent, but should produce
2038 // results that a typical user of the requested locales would perceive
2039 // as at least as good as those produced by the LookupMatcher abstract
2040 // operation. Options specified through Unicode locale extension sequences must
2041 // be ignored by the algorithm. Information about such subsequences is returned
2042 // separately. The abstract operation returns a record with a [[locale]] field,
2043 // whose value is the language tag of the selected locale, which must be an
2044 // element of availableLocales. If the language tag of the request locale that
2045 // led to the selected locale contained a Unicode locale extension sequence,
2046 // then the returned record also contains an [[extension]] field whose value is
2047 // the first Unicode locale extension sequence within the request locale
2048 // language tag.
2049 std::string BestFitMatcher(Isolate* isolate,
2050  const std::set<std::string>& available_locales,
2051  const std::vector<std::string>& requested_locales) {
2052  UErrorCode status = U_ZERO_ERROR;
2053  Iterator iter(requested_locales.cbegin(), requested_locales.cend());
2054  std::string bestfit = BuildLocaleMatcher(isolate, available_locales, &status)
2055  .getBestMatchResult(iter, status)
2056  .makeResolvedLocale(status)
2057  .toLanguageTag<std::string>(status);
2058  DCHECK(U_SUCCESS(status));
2059  return bestfit;
2060 }
