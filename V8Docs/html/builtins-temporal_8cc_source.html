148 // Like TEMPORAL_GET, but gets from an underlying Rust function
149 // rust_field is the name of the field with the Rust type. rust_getter is the
150 // name of the getter on the rust side (ideally the same as `field`). cvt is
151 // conversion code that converts `value` into the final returned JS Handle (use
152 // one of the macros below)
153 #define TEMPORAL_GET_RUST(T, rust_field, METHOD, js_field, rust_getter, cvt) \
154  BUILTIN(Temporal##T##Prototype##METHOD) { \
155  HandleScope scope(isolate); \
156  CHECK_RECEIVER(JSTemporal##T, obj, \
157  "Temporal." #T ".prototype." #js_field); \
158  auto value = obj->rust_field()->raw()->rust_getter(); \
159  cvt \
160  }
161 
162 #define CONVERT_INTEGER64 return *isolate->factory()->NewNumberFromInt64(value);
163 #define CONVERT_SMI return Smi::FromInt(value);
164 #define CONVERT_BOOLEAN return *isolate->factory()->ToBoolean(value);
165 #define CONVERT_DOUBLE return *isolate->factory()->NewNumber(value);
166 #define CONVERT_ASCII_STRING \
167  return *isolate->factory()->NewStringFromAsciiChecked(value);
168 
169 // converts nullopt to undefined
170 #define CONVERT_NULLABLE_INTEGER \
171  if (value.has_value()) { \
172  return *isolate->factory()->NewNumber(value.value()); \
173  } else { \
174  return *isolate->factory()->undefined_value(); \
175  }
176 
177 // temporal_rs returns errors in a couple spots where it should return
178 // `undefined`
179 #define CONVERT_FALLIBLE_INTEGER_AS_NULLABLE \
180  if (value.is_ok()) { \
181  return *isolate->factory()->NewNumber(std::move(value).ok().value()); \
182  } else { \
183  return *isolate->factory()->undefined_value(); \
184  }
185 
186 #define TEMPORAL_GET_NUMBER_AFTER_DIVID(T, M, field, scale, name) \
187  BUILTIN(Temporal##T##Prototype##M) { \
188  HandleScope scope(isolate); \
189  CHECK_RECEIVER(JSTemporal##T, handle, \
190  "get Temporal." #T ".prototype." #name); \
191  DirectHandle<BigInt> value; \
192  ASSIGN_RETURN_FAILURE_ON_EXCEPTION( \
193  isolate, value, \
194  BigInt::Divide(isolate, direct_handle(handle->field(), isolate), \
195  BigInt::FromUint64(isolate, scale))); \
196  DirectHandle<Object> number = BigInt::ToNumber(isolate, value); \
197  DCHECK(std::isfinite(Object::NumberValue(*number))); \
198  return *number; \
199  }
200 
201 #define TEMPORAL_GET_BIGINT_AFTER_DIVID(T, M, field, scale, name) \
202  BUILTIN(Temporal##T##Prototype##M) { \
203  HandleScope scope(isolate); \
204  CHECK_RECEIVER(JSTemporal##T, handle, \
205  "get Temporal." #T ".prototype." #name); \

207  isolate, \
208  BigInt::Divide(isolate, direct_handle(handle->field(), isolate), \
209  BigInt::FromUint64(isolate, scale))); \
210  }
