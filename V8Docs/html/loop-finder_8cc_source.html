31 // Update the `parent_loops_` of all of the blocks that are inside of the loop
32 // that starts on `header`.
33 LoopFinder::LoopInfo LoopFinder::VisitLoop(const Block* header) {
34  Block* backedge = header->LastPredecessor();
35  DCHECK(backedge->LastOperation(*input_graph_).Is<GotoOp>());
36  DCHECK_EQ(backedge->LastOperation(*input_graph_).Cast<GotoOp>().destination,
37  header);
38  DCHECK_GE(backedge->index().id(), header->index().id());
39 
40  LoopInfo info;
41  // The header is skipped by the while-loop below, so we initialize {info} with
42  // the `op_count` from {header}, and a `block_count` of 1 (= the header).
43  info.op_count = header->OpCountUpperBound();
44  info.start = header;
45  info.end = backedge;
46  info.block_count = 1;
47  CollectLoopInfo(header, &info);
48 
49  queue_.clear();
50  queue_.push_back(backedge);
51  while (!queue_.empty()) {
52  const Block* curr = queue_.back();
53  queue_.pop_back();
54  if (curr == header) continue;
55  if (loop_headers_[curr->index()] != nullptr) {
56  const Block* curr_parent = loop_headers_[curr->index()];
57  if (curr_parent == header) {
58  // If {curr}'s parent is already marked as being {header}, then we've
59  // already visited {curr}.
60  continue;
61  } else {
62  // If {curr}'s parent is not {header}, then {curr} is part of an inner
63  // loop. We should continue the search on the loop header: the
64  // predecessors of {curr} will all be in this inner loop.
65  queue_.push_back(curr_parent);
66  info.has_inner_loops = true;
67  continue;
68  }
69  }
70  info.block_count++;
71  info.op_count += curr->OpCountUpperBound();
72  loop_headers_[curr->index()] = header;
73  CollectLoopInfo(curr, &info);
74  const Block* pred_start = curr->LastPredecessor();
75  if (curr->IsLoop()) {
76  // Skipping the backedge of inner loops since we don't want to visit inner
77  // loops now (they should already have been visited).
78  DCHECK_NOT_NULL(pred_start);
79  pred_start = pred_start->NeighboringPredecessor();
80  info.has_inner_loops = true;
81  }
82  for (const Block* pred : NeighboringPredecessorIterable(pred_start)) {
83  queue_.push_back(pred);
84  }
85  }
86 
87  return info;
88 }

90 ZoneSet<const Block*, LoopFinder::BlockCmp> LoopFinder::GetLoopBody(
91  const Block* loop_header) {
92  DCHECK(!GetLoopInfo(loop_header).has_inner_loops);
93  ZoneSet<const Block*, BlockCmp> body(phase_zone_);
94  body.insert(loop_header);
95 
96  ZoneVector<const Block*> queue(phase_zone_);
97  queue.push_back(loop_header->LastPredecessor());
98  while (!queue.empty()) {
99  const Block* curr = queue.back();
100  queue.pop_back();
101  if (body.find(curr) != body.end()) continue;
102  body.insert(curr);
103  for (const Block* pred = curr->LastPredecessor(); pred != nullptr;
104  pred = pred->NeighboringPredecessor()) {
105  if (pred == loop_header) continue;
106  queue.push_back(pred);
107  }
108  }
109 
110  return body;
111 }
