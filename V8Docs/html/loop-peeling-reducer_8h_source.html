33 // LoopPeeling "peels" the first iteration of innermost loops (= it extracts the
34 // first iteration from the loop). The goal of this is mainly to hoist checks
35 // out of the loop (such as Smi-checks, type-checks, bound-checks, etc).
36 
37 template <class Next>
38 class LoopPeelingReducer : public Next {
39  public:
40  TURBOSHAFT_REDUCER_BOILERPLATE(LoopPeeling)
41 
42 #if defined(__clang__)
43  // LoopUnrolling and LoopPeeling shouldn't be performed in the same phase, see
44  // the comment in pipeline.cc where LoopUnrolling is triggered.
45  static_assert(
46  !reducer_list_contains<ReducerList, LoopUnrollingReducer>::value);
47 #endif

49  V<None> REDUCE_INPUT_GRAPH(Goto)(V<None> ig_idx, const GotoOp& gto) {
50  // Note that the "ShouldSkipOptimizationStep" is placed in the part of
51  // this Reduce method triggering the peeling rather than at the begining.
52  // This is because the backedge skipping is not an optimization but a
53  // mandatory lowering when peeling is being performed.
54  LABEL_BLOCK(no_change) { return Next::ReduceInputGraphGoto(ig_idx, gto); }
55 
56  const Block* dst = gto.destination;
57  if (dst->IsLoop() && !gto.is_backedge && CanPeelLoop(dst)) {
58  if (ShouldSkipOptimizationStep()) goto no_change;
59  PeelFirstIteration(dst);
60  return {};
61  } else if (IsEmittingPeeledIteration() && dst == current_loop_header_) {
62  // We skip the backedge of the loop: PeelFirstIeration will instead emit a
63  // forward edge to the non-peeled header.
64  return {};
65  }
66 
67  goto no_change;
68  }
69 
70  V<None> REDUCE_INPUT_GRAPH(JSStackCheck)(V<None> ig_idx,
71  const JSStackCheckOp& stack_check) {
72  if (ShouldSkipOptimizationStep() || !IsEmittingPeeledIteration()) {
73  return Next::ReduceInputGraphJSStackCheck(ig_idx, stack_check);
74  }
75 
76  // We remove the stack check of the peeled iteration.
77  return V<None>::Invalid();
78  }
79 
80 #if V8_ENABLE_WEBASSEMBLY
81  V<None> REDUCE_INPUT_GRAPH(WasmStackCheck)(
82  V<None> ig_idx, const WasmStackCheckOp& stack_check) {
83  if (ShouldSkipOptimizationStep() || !IsEmittingPeeledIteration()) {
84  return Next::ReduceInputGraphWasmStackCheck(ig_idx, stack_check);
85  }
86 
87  // We remove the stack check of the peeled iteration.
88  return V<None>::Invalid();
89  }
90 #endif

92  OpIndex REDUCE_INPUT_GRAPH(Phi)(OpIndex ig_idx, const PhiOp& phi) {
93  if (!IsEmittingUnpeeledBody() ||
94  __ current_input_block() != current_loop_header_) {
95  return Next::ReduceInputGraphPhi(ig_idx, phi);
96  }
97 
98  // The 1st input of the loop phis of the unpeeled loop header should be the
99  // 2nd input of the original loop phis, since with the peeling, they
100  // actually come from the backedge of the peeled iteration.
101  return __ PendingLoopPhi(
102  __ MapToNewGraph(phi.input(PhiOp::kLoopPhiBackEdgeIndex)), phi.rep);
103  }
104 
105  private:
106  static constexpr int kMaxSizeForPeeling = 1000;
107  enum class PeelingStatus {
108  kNotPeeling,
109  kEmittingPeeledLoop,
110  kEmittingUnpeeledBody
111  };
112 
113  void PeelFirstIteration(const Block* header) {
114  TRACE("LoopPeeling: peeling loop at " << header->index());
115  DCHECK_EQ(peeling_, PeelingStatus::kNotPeeling);
116  ScopedModification<PeelingStatus> scope(&peeling_,
117  PeelingStatus::kEmittingPeeledLoop);
118  current_loop_header_ = header;
119 
120  // Emitting the peeled iteration.
121  auto loop_body = loop_finder_.GetLoopBody(header);
122  // Note that this call to CloneSubGraph will not emit the backedge because
123  // we'll skip it in ReduceInputGraphGoto (above). The next CloneSubGraph
124  // call will start with a forward Goto to the header (like all
125  // CloneSubGraphs do), and will end by emitting the backedge, because this
126  // time {peeling_} won't be EmittingPeeledLoop, and the backedge Goto will
127  // thus be emitted.
128  TRACE("> Emitting peeled iteration");
129  __ CloneSubGraph(loop_body, /* keep_loop_kinds */ false);
130 
131  if (__ generating_unreachable_operations()) {
132  // While peeling, we realized that the 2nd iteration of the loop is not
133  // reachable.
134  TRACE("> Second iteration is not reachable, stopping now");
135  return;
136  }
137 
138  // We now emit the regular unpeeled loop.
139  peeling_ = PeelingStatus::kEmittingUnpeeledBody;
140  TRACE("> Emitting unpeeled loop body");
141  __ CloneSubGraph(loop_body, /* keep_loop_kinds */ true,
142  /* is_loop_after_peeling */ true);
143  }
144 
145  bool CanPeelLoop(const Block* header) {
146  TRACE("LoopPeeling: considering " << header->index());
147  if (IsPeeling()) {
148  TRACE("> Cannot peel because we're already peeling a loop");
149  return false;
150  }
151  auto info = loop_finder_.GetLoopInfo(header);
152  if (info.has_inner_loops) {
153  TRACE("> Cannot peel because it has inner loops");
154  return false;
155  }
156  if (info.op_count > kMaxSizeForPeeling) {
157  TRACE("> Cannot peel because it contains too many operations");
158  return false;
159  }
160  return true;
161  }
162 
163  bool IsPeeling() const {
164  return IsEmittingPeeledIteration() || IsEmittingUnpeeledBody();
165  }
166  bool IsEmittingPeeledIteration() const {
167  return peeling_ == PeelingStatus::kEmittingPeeledLoop;
168  }
169  bool IsEmittingUnpeeledBody() const {
170  return peeling_ == PeelingStatus::kEmittingUnpeeledBody;
171  }
172 
173  PeelingStatus peeling_ = PeelingStatus::kNotPeeling;
174  const Block* current_loop_header_ = nullptr;
175 
176  LoopFinder loop_finder_{__ phase_zone(), &__ modifiable_input_graph(),
177  LoopFinder::Config{}};
178  JSHeapBroker* broker_ = __ data() -> broker();
179 };
