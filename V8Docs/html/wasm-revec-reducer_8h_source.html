221  indexes_[0] = a;

225  OpIndex operator[](int i) const { return indexes_[i]; }
226 
227  bool operator==(const NodeGroup& other) const {
228  return indexes_[0] == other.indexes_[0] && indexes_[1] == other.indexes_[1];
229  }
230  bool operator!=(const NodeGroup& other) const {
231  return indexes_[0] != other.indexes_[0] || indexes_[1] != other.indexes_[1];

233 
234  const OpIndex* begin() const { return indexes_; }
235  const OpIndex* end() const { return indexes_ + kSize; }
236 
237  private:
238  OpIndex indexes_[kSize];

245 // A PackNode consists of a fixed number of isomorphic simd128 nodes which can
246 // execute in parallel and convert to a 256-bit simd node later. The nodes in a
247 // PackNode must satisfy that they can be scheduled in the same basic block and
248 // are mutually independent.
249 class PackNode : public NON_EXPORTED_BASE(ZoneObject) {
250  public:
251  enum NodeType {
252  kDefault, // Nodes are naturally packed without special attributes.
253  kForcePackNode, // Nodes do not satisfy some packing rule, but can be
254  // forcely coalesced with a Pack128To256 operation. E.g.
255  // inconsecutive loads. In x64, we can use the vinsertf128
256  // instruction to forcely coalescing two 128-bit values.
257  kShufflePackNode, // Nodes are Simd128Shuffle operations with specific
258  // info.
259  kBundlePackNode, // Nodes representing a i8x16/i16x8 to f32x4 conversion.
260  kIntersectPackNode, // One or more nodes already packed by an existing
261  // PackNode.
262  };
263 
264  explicit PackNode(Zone* zone, const NodeGroup& node_group,
265  NodeType node_type = kDefault)
266  : nodes_(node_group),
267  revectorized_node_(),
268  operands_(zone),
269  node_type_(node_type) {}
270  const NodeGroup& nodes() const { return nodes_; }
271  bool IsSame(const NodeGroup& node_group) const {
272  return nodes_ == node_group;
273  }
274  bool IsSame(const PackNode& other) const { return nodes_ == other.nodes_; }
275  V<Simd256> RevectorizedNode() const { return revectorized_node_; }
276  void SetRevectorizedNode(V<Simd256> node) { revectorized_node_ = node; }
277 
278  bool IsDefaultPackNode() const { return node_type_ == kDefault; }
279  bool IsForcePackNode() const { return node_type_ == kForcePackNode; }
280  bool IsShufflePackNode() const { return node_type_ == kShufflePackNode; }
281  bool IsBundlePackNode() const { return node_type_ == kBundlePackNode; }
282  // We will force-pack nodes for both ForcePackNode and IntersectPackNode.
283  bool is_force_packing() const {
284  return node_type_ == kForcePackNode || node_type_ == kIntersectPackNode;
285  }
286 
287  ForcePackNode* AsForcePackNode() {
288  DCHECK(IsForcePackNode());
289  return reinterpret_cast<ForcePackNode*>(this);
290  }
291  ShufflePackNode* AsShufflePackNode() {
292  DCHECK(IsShufflePackNode());
293  return reinterpret_cast<ShufflePackNode*>(this);
294  }
295  BundlePackNode* AsBundlePackNode() {
296  DCHECK(IsBundlePackNode());
297  return reinterpret_cast<BundlePackNode*>(this);
298  }
299 
300  PackNode* GetOperand(int index) const {
301  DCHECK_LT(index, operands_.size());
302  DCHECK(operands_[index]);
303  return operands_[index];
304  }
305 
306  void SetOperand(int index, PackNode* pnode) {
307  DCHECK_GE(index, 0);
308  if (operands_.size() < static_cast<size_t>(index + 1)) {
309  operands_.resize(index + 1);
310  }
311  operands_[index] = pnode;
312  }
313 
314  ZoneVector<PackNode*>::size_type GetOperandsSize() const {
315  return operands_.size();
316  }
317 
318  void Print(Graph* graph) const;
319 
320  private:
321  friend class ForcePackNode;
322  NodeGroup nodes_;
323  V<Simd256> revectorized_node_;
324  ZoneVector<PackNode*> operands_;
325  NodeType node_type_;
326 };

668  // The op is already emitted during emitting force pack node input trees.

997  // Skip revectorized node.

1000  switch (kind) {
1001  case ShufflePackNode::SpecificInfo::Kind::kS256Load32Transform:
1002  case ShufflePackNode::SpecificInfo::Kind::kS256Load64Transform: {
1003  const bool is_32 =
1004  kind == ShufflePackNode::SpecificInfo::Kind::kS256Load32Transform;
1005 
1006  const OpIndex load_index =
1007  op.input(pnode->info().splat_index() >> (is_32 ? 2 : 1));
1008  const LoadOp& load =
1009  __ input_graph().Get(load_index).template Cast<LoadOp>();
1010

1013  const int offset = splat_index + load.offset;
1014

1016  __ IntPtrConstant(offset));
1017 
1018  V<WordPtr> index = load.index().has_value()
1019  ? __ MapToNewGraph(load.index().value())
1020  : __ IntPtrConstant(0);
1021

1023  is_32 ? Simd256LoadTransformOp::TransformKind::k32Splat
1024  : Simd256LoadTransformOp::TransformKind::k64Splat;
1025  og_index = __ Simd256LoadTransform(base, index, load.kind,
1026  transform_kind, 0);
1027  pnode->SetRevectorizedNode(og_index);
1028  break;
1029  }
1030  case ShufflePackNode::SpecificInfo::Kind::kS256Load8x8U: {
1031  const Simd128ShuffleOp& op0 = __ input_graph()
1032  .Get(pnode -> nodes()[0])
1033  .template Cast<Simd128ShuffleOp>();
1034 
1035  V<Simd128> load_transform_idx =
1036  __ input_graph()
1037  .Get(op0.left())
1038  .template Is<Simd128LoadTransformOp>()
1039  ? op0.left()
1040  : op0.right();
1041  const Simd128LoadTransformOp& load_transform =
1042  __ input_graph()
1043  .Get(load_transform_idx)
1044  .template Cast<Simd128LoadTransformOp>();
1045  DCHECK_EQ(load_transform.transform_kind,
1046  Simd128LoadTransformOp::TransformKind::k64Zero);
1047  V<WordPtr> base = __ MapToNewGraph(load_transform.base());
1048  V<WordPtr> index = __ MapToNewGraph(load_transform.index());
1049  og_index = __ Simd256LoadTransform(
1050  base, index, load_transform.load_kind,
1051  Simd256LoadTransformOp::TransformKind::k8x8U,
1052  load_transform.offset);
1053  pnode->SetRevectorizedNode(og_index);
1054  break;
1055  }
1056 #ifdef V8_TARGET_ARCH_X64

1151  void ReduceInputsOfOp(OpIndex cur_index, OpIndex op_index) {
1152  // Reduce all the operations of op_index's input tree, which should be
1153  // bigger than the cur_index. The traversal is done in a DFS manner
1154  // to make sure all inputs are emitted before the use.
1155  const Block* current_input_block = Asm().current_input_block();
1156  std::stack<OpIndex> inputs;
1157  ZoneUnorderedSet<OpIndex> visited(Asm().phase_zone());
1158  inputs.push(op_index);
1159 
1160  while (!inputs.empty()) {
1161  OpIndex idx = inputs.top();
1162  if (visited.find(idx) != visited.end()) {
1163  inputs.pop();
1164  continue;
1165  }
1166 
1167  const Operation& op = __ input_graph().Get(idx);
1168  bool has_unvisited_inputs = false;
1169  for (OpIndex input : op.inputs()) {
1170  if (input > cur_index && visited.find(input) == visited.end()) {
1171  inputs.push(input);
1172  has_unvisited_inputs = true;
1173  }
1174  }
1175 
1176  if (!has_unvisited_inputs) {
1177  inputs.pop();
1178  visited.insert(idx);
1179 
1180  // op_index will be reduced later.
1181  if (idx == op_index) continue;
1182 
1183  DCHECK(!Asm().input_graph().Get(idx).template Is<PhiOp>());
1184  Asm().template VisitOpAndUpdateMapping<false>(idx, current_input_block);
1185  }
1186  }

1189  template <typename Op, typename Continuation>
1190  void ReduceForceOrIntersectPackNode(PackNode* pnode, const OpIndex ig_index,
1191  OpIndex* og_index) {
1192  std::array<OpIndex, 2> v;
1193  DCHECK_EQ(pnode->nodes().size(), 2);
1194  // The operation order in pnode is determined by the store or reduce
1195  // seed when build the SLPTree. It is not quaranteed to align with
1196  // the visiting order in each basic block from input graph. E.g. we
1197  // can have a block including {a1, a2, b1, b2} operations, and the
1198  // SLPTree can be pnode1: (a2, a1), pnode2: (b1, b2) if a2 is input
1199  // of b1, and a1 is input of b2.
1200  for (int i = 0; i < static_cast<int>(pnode->nodes().size()); i++) {
1201  OpIndex cur_index = pnode->nodes()[i];
1202  if ((*og_index).valid() && cur_index == ig_index) {
1203  v[i] = *og_index;
1204  } else {
1205  // The current index maybe already reduced by the IntersectPackNode.
1206  v[i] = __ template MapToNewGraph<true>(cur_index);
1207  }
1208 
1209  if (v[i].valid()) continue;
1210 
1211  if (cur_index != ig_index) {
1212  ReduceInputsOfOp(ig_index, cur_index);
1213  }
1214  const Op& op = Asm().input_graph().Get(cur_index).template Cast<Op>();
1215  v[i] = Continuation{this}.ReduceInputGraph(cur_index, op);
1216 
1217  if (cur_index == ig_index) {
1218  *og_index = v[i];
1219  } else {
1220  // We have to create the mapping as cur_index may exist in other
1221  // IntersectPackNode and reduce again.
1222  __ CreateOldToNewMapping(cur_index, v[i]);
1223  }
1224  }
1225 
1226  OpIndex revec_index = __ SimdPack128To256(v[0], v[1]);
