32 // Destructively intersects the right map into the left map, such that the
33 // left map is mutated to become the result of the intersection. Values that
34 // are in both maps are passed to the merging function to be merged with each
35 // other -- again, the LHS here is expected to be mutated.
36 template <typename Key, typename Value,
37  typename MergeFunc = std::equal_to<Value>>
38 void DestructivelyIntersect(ZoneMap<Key, Value>& lhs_map,
39  const ZoneMap<Key, Value>& rhs_map,
40  MergeFunc&& func = MergeFunc()) {
41  // Walk the two maps in lock step. This relies on the fact that ZoneMaps are
42  // sorted.
43  typename ZoneMap<Key, Value>::iterator lhs_it = lhs_map.begin();
44  typename ZoneMap<Key, Value>::const_iterator rhs_it = rhs_map.begin();
45  while (lhs_it != lhs_map.end() && rhs_it != rhs_map.end()) {
46  if (lhs_it->first < rhs_it->first) {
47  // Remove from LHS elements that are not in RHS.
48  lhs_it = lhs_map.erase(lhs_it);
49  } else if (rhs_it->first < lhs_it->first) {
50  // Skip over elements that are only in RHS.
51  ++rhs_it;
52  } else {
53  // Apply the merge function to the values of the two iterators. If the
54  // function returns false, remove the value.
55  bool keep_value = func(lhs_it->second, rhs_it->second);
56  if (keep_value) {
57  ++lhs_it;
58  } else {
59  lhs_it = lhs_map.erase(lhs_it);
60  }
61  ++rhs_it;
62  }
63  }
64  // If we haven't reached the end of LHS by now, then we have reached the end
65  // of RHS, and the remaining items are therefore not in RHS. Remove them.
66  if (lhs_it != lhs_map.end()) {
67  lhs_map.erase(lhs_it, lhs_map.end());
68  }
69 }

169  // Mutate this node info by merging in another node info, with the result
170  // being a node info that is the subset of information valid in both inputs.
171  void MergeWith(const NodeInfo& other, Zone* zone,
172  bool& any_merged_map_is_unstable) {
173  UnionType(other.type_);
174  alternative_.MergeWith(other.alternative_);
175  if (possible_maps_are_known_) {
176  if (other.possible_maps_are_known_) {
177  // Map sets are the set of _possible_ maps, so on a merge we need to
178  // _union_ them together (i.e. intersect the set of impossible maps).
179  // Remember whether _any_ of these merges observed unstable maps.
180  possible_maps_.Union(other.possible_maps_, zone);
181  } else {
182  possible_maps_.clear();
183  possible_maps_are_known_ = false;
184  }
185  }
186 
187  any_map_is_unstable_ = possible_maps_are_known_ &&
188  (any_map_is_unstable_ || other.any_map_is_unstable_);
189  any_merged_map_is_unstable =
190  any_merged_map_is_unstable || any_map_is_unstable_;

347  NodeType GetType(compiler::JSHeapBroker* broker, ValueNode* node) const {
348  // We first check the KnownNodeAspects in order to return the most precise
349  // type possible.
350  auto info = TryGetInfoFor(node);
351  if (info == nullptr) {
352  // If this node has no NodeInfo (or not known type in its NodeInfo), we
353  // fall back to its static type.
354  return node->GetStaticType(broker);
355  }
356  NodeType actual_type = info->type();
357  if (auto phi = node->TryCast<Phi>()) {
358  actual_type = IntersectType(actual_type, phi->type());
359  }
360 #ifdef DEBUG
361  NodeType static_type = node->GetStaticType(broker);
362  if (!NodeTypeIs(actual_type, static_type)) {
363  // In case we needed a numerical alternative of a smi value, the type
364  // must generalize. In all other cases the node info type should reflect
365  // the actual type.
366  DCHECK(static_type == NodeType::kSmi &&
367  actual_type == NodeType::kNumber &&
368  !TryGetInfoFor(node)->alternative().has_none());
369  }
370 #endif // DEBUG
371  return actual_type;
372  }

407  NodeType type, NodeType* old_type) {
408  NodeType static_type = node->GetStaticType(broker);
409  if (old_type) *old_type = static_type;

413  if (NodeTypeIs(known_info->type(), type)) return true;

418  if (NodeTypeIsUnstable(type)) {

420  any_map_for_any_node_is_unstable = true;
421  }
422  return false;
