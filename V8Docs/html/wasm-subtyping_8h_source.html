41 // Checks if {subtype}, defined in {module1}, is a subtype of {supertype},
42 // defined in {module2}.
43 // Subtyping between value types is described by the following rules
44 // (structural subtyping):
45 // - numeric types are subtype-related iff they are equal.
46 // - (ref null ht1) <: (ref null ht2) iff ht1 <: ht2.
47 // - (ref ht1) <: (ref null? ht2) iff ht1 <: ht2.
48 // - rtt1 <: rtt2 iff rtt1 ~ rtt2.
49 // For heap types, the following subtyping rules hold:
50 // - The abstract heap types form the following type hierarchies:
51 //
52 // any func extern
53 // / \ | |
54 // eq \ nofunc noextern
55 // / | \ \
56 // i31 array struct string
57 // \___|______|_____/
58 // |
59 // none
60 //
61 // - All functions are subtypes of func.
62 // - All structs are subtypes of struct.
63 // - All arrays are subtypes of array.
64 // - An indexed heap type h1 is a subtype of indexed heap type h2 if h2 is
65 // transitively an explicit canonical supertype of h1.
66 // Note that {any} includes references introduced by the host which belong to
67 // none of any's subtypes (e.g. JS objects).
68 V8_INLINE bool IsSubtypeOf(ValueType subtype, ValueType supertype,
69  const WasmModule* sub_module,
70  const WasmModule* super_module) {
71  if (subtype == supertype && sub_module == super_module) return true;
72  return IsSubtypeOfImpl(subtype, supertype, sub_module, super_module);
73 }

104 // Checks whether {subtype_index} is valid as a declared subtype of
105 // {supertype_index}.
106 // - Both type must be of the same kind (function, struct, or array).
107 // - Both type must have the same {is_shared} flag.
108 // - Structs: Subtype must have at least as many fields as supertype,
109 // covariance for respective immutable fields, equivalence for respective
110 // mutable fields.
111 // - Arrays: subtyping of respective element types for immutable arrays,
112 // equivalence of element types for mutable arrays.
113 // - Functions: equal number of parameter and return types. Contravariance for
114 // respective parameter types, covariance for respective return types.
115 V8_EXPORT_PRIVATE bool ValidSubtypeDefinition(ModuleTypeIndex subtype_index,
116  ModuleTypeIndex supertype_index,
117  const WasmModule* module);
