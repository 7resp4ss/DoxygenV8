38 void FinalizationRegistryCleanupTask::RunInternal() {
39  Isolate* isolate = heap_->isolate();
40  SlowAssertNoActiveJavaScript();
41 
42  TRACE_EVENT_CALL_STATS_SCOPED(isolate, "v8",
43  "V8.FinalizationRegistryCleanupTask");
44 
45  // First clear that the task is posted in case of early returns below.
46  heap_->set_is_finalization_registry_cleanup_task_posted(false);
47 
48  HandleScope handle_scope(isolate);
49  DirectHandle<JSFinalizationRegistry> finalization_registry;
50  // There could be no dirty FinalizationRegistries. When a context is disposed
51  // by the embedder, its FinalizationRegistries are removed from the dirty
52  // list.
53  if (!heap_->DequeueDirtyJSFinalizationRegistry().ToHandle(
54  &finalization_registry)) {
55  return;
56  }
57  finalization_registry->set_scheduled_for_cleanup(false);
58 
59  // Since FinalizationRegistry cleanup callbacks are scheduled by V8, enter the
60  // FinalizationRegistry's context.
61  DirectHandle<NativeContext> native_context(
62  finalization_registry->native_context(), isolate);
63  v8::Context::Scope context_scope(v8::Utils::ToLocal(native_context));
64  v8::Isolate* v8_isolate = reinterpret_cast<v8::Isolate*>(isolate);
65  v8::TryCatch catcher(v8_isolate);
66  catcher.SetVerbose(true);
67  std::unique_ptr<MicrotasksScope> microtasks_scope;
68  MicrotaskQueue* microtask_queue =
69  finalization_registry->native_context()->microtask_queue();
70  if (!microtask_queue) microtask_queue = isolate->default_microtask_queue();
71  if (microtask_queue &&
72  microtask_queue->microtasks_policy() == v8::MicrotasksPolicy::kScoped) {
73  // InvokeFinalizationRegistryCleanupFromTask will call into V8 API methods,
74  // so we need a valid microtasks scope on the stack to avoid running into
75  // the CallDepthScope check.
76  microtasks_scope.reset(new v8::MicrotasksScope(
77  v8_isolate, microtask_queue, v8::MicrotasksScope::kDoNotRunMicrotasks));
78  }
79 
80  // Exceptions are reported via the message handler. This is ensured by the
81  // verbose TryCatch.
82  //
83  // Cleanup is interrupted if there is an exception. The HTML spec calls for a
84  // microtask checkpoint after each cleanup task, so the task should return
85  // after an exception so the host can perform a microtask checkpoint. In case
86  // of exception, check if the FinalizationRegistry still needs cleanup
87  // and should be requeued.
88  //
89  // TODO(syg): Implement better scheduling for finalizers.
90  InvokeFinalizationRegistryCleanupFromTask(native_context,
91  finalization_registry);
92  if (finalization_registry->NeedsCleanup() &&
93  !finalization_registry->scheduled_for_cleanup()) {
94  auto nop = [](Tagged<HeapObject>, ObjectSlot, Tagged<Object>) {};
95  heap_->EnqueueDirtyJSFinalizationRegistry(*finalization_registry, nop);
96  }
97 
98  // Repost if there are remaining dirty FinalizationRegistries.
99  heap_->PostFinalizationRegistryCleanupTaskIfNeeded();
100 }
