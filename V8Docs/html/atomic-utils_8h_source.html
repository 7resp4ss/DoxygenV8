156  // Atomically sets bits selected by the mask to the given value.
157  // Returns false if the bits are already set as needed.
158  template <typename T>
159  static bool Release_SetBits(T* addr, T bits, T mask) {
160  static_assert(sizeof(T) <= sizeof(AtomicStorageType));
161  DCHECK_EQ(bits & ~mask, static_cast<T>(0));
162  T old_value = Relaxed_Load(addr);
163  T new_value, old_value_before_cas;
164  do {
165  if ((old_value & mask) == bits) return false;
166  new_value = (old_value & ~mask) | bits;
167  old_value_before_cas = old_value;
168  old_value = Release_CompareAndSwap(addr, old_value, new_value);
169  } while (old_value != old_value_before_cas);
170  return true;
171  }
172 
173  // Atomically sets bits selected by the mask to the given value.
174  // Returns false if the bits are already set as needed.
175  template <typename T>
176  static bool Relaxed_SetBits(T* addr, T bits, T mask) {
177  static_assert(sizeof(T) <= sizeof(AtomicStorageType));
178  DCHECK_EQ(bits & ~mask, static_cast<T>(0));
179  T old_value = Relaxed_Load(addr);
180  T new_value, old_value_before_cas;
181  do {
182  if ((old_value & mask) == bits) return false;
183  new_value = (old_value & ~mask) | bits;
184  old_value_before_cas = old_value;
185  old_value = Relaxed_CompareAndSwap(addr, old_value, new_value);
186  } while (old_value != old_value_before_cas);
187  return true;

267 template <typename T>
268 inline void CheckedIncrement(
269  std::atomic<T>* number, T amount,
270  std::memory_order order = std::memory_order_seq_cst)
271  requires std::is_unsigned_v<T>
272 {
273  const T old = number->fetch_add(amount, order);
274  DCHECK_GE(old + amount, old);
275  USE(old);
276 }

278 template <typename T>
279 inline void CheckedDecrement(
280  std::atomic<T>* number, T amount,
281  std::memory_order order = std::memory_order_seq_cst)
282  requires std::is_unsigned_v<T>
283 {
284  const T old = number->fetch_sub(amount, order);
285  DCHECK_GE(old, amount);
