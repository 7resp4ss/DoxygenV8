30 #ifdef __arm__
31 bool OS::ArmUsingHardFloat() {
32  // GCC versions 4.6 and above define __ARM_PCS or __ARM_PCS_VFP to specify
33  // the Floating Point ABI used (PCS stands for Procedure Call Standard).
34  // We use these as well as a couple of other defines to statically determine
35  // what FP ABI used.
36  // GCC versions 4.4 and below don't support hard-fp.
37  // GCC versions 4.5 may support hard-fp without defining __ARM_PCS or
38  // __ARM_PCS_VFP.
39 
40 #define GCC_VERSION \
41  (__GNUC__ * 10000 + __GNUC_MINOR__ * 100 + __GNUC_PATCHLEVEL__)
42 #if GCC_VERSION >= 40600 && !defined(__clang__)
43 #if defined(__ARM_PCS_VFP)
44  return true;
45 #else
46  return false;
47 #endif
48 
49 #elif GCC_VERSION < 40500 && !defined(__clang__)
50  return false;
51 
52 #else
53 #if defined(__ARM_PCS_VFP)
54  return true;
55 #elif defined(__ARM_PCS) || defined(__SOFTFP__) || defined(__SOFTFP) || \
56  !defined(__VFP_FP__)
57  return false;
58 #else
59 #error \
60  "Your version of compiler does not report the FP ABI compiled for." \
61  "Please report it on this issue" \
62  "http://code.google.com/p/v8/issues/detail?id=2140"
63 
64 #endif
65 #endif
66 #undef GCC_VERSION
67 }
68 #endif // def __arm__

96 int OS::ActivationFrameAlignment() {
97 #if V8_TARGET_ARCH_ARM
98  // On EABI ARM targets this is required for fp correctness in the
99  // runtime system.
100  return 8;
101 #elif V8_TARGET_ARCH_S390X
102  return 8;
103 #else
104  // Otherwise we just assume 16 byte alignment, i.e.:
105  // - With gcc 4.4 the tree vectorization optimizer can generate code
106  // that requires 16 byte alignment such as movdqa on x86.
107  // - Mac OS X, PPC and Solaris (64-bit) activation frames must
108  // be 16 byte-aligned; see "Mac OS X ABI Function Call Guide"
109  return 16;
110 #endif
111 }

125 void* Allocate(void* address, size_t size, OS::MemoryPermission access) {
126  int prot_flags;
127  switch (access) {
128  case OS::MemoryPermission::kNoAccess:
129  prot_flags = PROT_NONE;
130  break;
131  case OS::MemoryPermission::kReadWrite:
132  prot_flags = PROT_READ | PROT_WRITE;
133  break;
134  default:
135  SB_LOG(ERROR) << "The requested memory allocation access is not"
136  " implemented for Starboard: "
137  << static_cast<int>(access);
138  return nullptr;
139  }
140  void* result = mmap(nullptr, size, prot_flags, MAP_PRIVATE | MAP_ANON, -1, 0);
141  if (result == MAP_FAILED) {
142  return nullptr;
143  }
144  return result;
145 }

147 // static
148 void* OS::Allocate(void* address, size_t size, size_t alignment,
149  MemoryPermission access) {
150  size_t page_size = AllocatePageSize();
151  DCHECK_EQ(0, size % page_size);
152  DCHECK_EQ(0, alignment % page_size);
153  address = AlignedAddress(address, alignment);
154  // Add the maximum misalignment so we are guaranteed an aligned base address.
155  size_t request_size = size + (alignment - page_size);
156  request_size = RoundUp(request_size, OS::AllocatePageSize());
157  void* result = base::Allocate(address, request_size, access);
158  if (result == nullptr) return nullptr;
159 
160  // Unmap memory allocated before the aligned base address.
161  uint8_t* base = static_cast<uint8_t*>(result);
162  uint8_t* aligned_base = reinterpret_cast<uint8_t*>(
163  RoundUp(reinterpret_cast<uintptr_t>(base), alignment));
164  if (aligned_base != base) {
165  DCHECK_LT(base, aligned_base);
166  size_t prefix_size = static_cast<size_t>(aligned_base - base);
167  Free(base, prefix_size);
168  request_size -= prefix_size;
169  }
170  // Unmap memory allocated after the potentially unaligned end.
171  if (size != request_size) {
172  DCHECK_LT(size, request_size);
173  size_t suffix_size = request_size - size;
174  Free(aligned_base + size, suffix_size);
175  request_size -= suffix_size;
176  }
177 
178  DCHECK_EQ(size, request_size);
179  return static_cast<void*>(aligned_base);
180 }

192 // static
193 bool OS::SetPermissions(void* address, size_t size, MemoryPermission access) {
194  int new_protection;
195  switch (access) {
196  case OS::MemoryPermission::kNoAccess:
197  new_protection = PROT_NONE;
198  break;
199  case OS::MemoryPermission::kRead:
200  new_protection = PROT_READ;
201  case OS::MemoryPermission::kReadWrite:
202  new_protection = PROT_READ | PROT_WRITE;
203  break;
204  case OS::MemoryPermission::kReadExecute:
205 #if SB_CAN(MAP_EXECUTABLE_MEMORY)
206  new_protection = PROT_READ | PROT_EXEC;
207 #else
208  UNREACHABLE();
209 #endif
210  break;
211  default:
212  // All other types are not supported by Starboard.
213  return false;
214  }
215  return mprotect(address, size, new_protection) == 0;
216 }

443  // SbTimeZoneGetCurrent returns an offset west of Greenwich, which has the
444  // opposite sign V8 expects.
445  // The starboard function returns offset in minutes. We convert to return
446  // value in milliseconds.
447  return SbTimeZoneGetCurrent() * 60.0 * msPerSecond * (-1);

449  double DaylightSavingsOffset(double time_ms) override {
450  int64_t posix_microseconds = starboard::CurrentPosixTime();
451  EzTimeValue value = {
452  posix_microseconds / TimeConstants::kMicroSecondsPerSecond,
453  (int32_t)(posix_microseconds % TimeConstants::kMicroSecondsPerSecond)
454  };
455  EzTimeExploded ez_exploded;
456  bool result =
457  EzTimeValueExplode(&value, kEzTimeZoneLocal, &ez_exploded, NULL);
458  return ez_exploded.tm_isdst > 0 ? 3600 * msPerSecond : 0;
459  }
