117 wasm::ValueType WasmGlobalObject::type() const {
118  // Various consumers of ValueKind (e.g. ValueKind::name()) use the raw enum
119  // value as index into a global array. As such, if the index is corrupted
120  // (which must be assumed, as it comes from within the sandbox), this can
121  // lead to out-of-bounds reads outside the sandbox. While these are not
122  // technically sandbox violations, we should still try to avoid them to keep
123  // fuzzers happy. This SBXCHECK accomplishes that.
124  wasm::ValueType type = wasm::ValueType::FromRawBitField(raw_type());
125  SBXCHECK(type.is_valid());
126  return type;
127 }

645 DirectHandle<Object> WasmObject::ReadValueAt(Isolate* isolate,
646  DirectHandle<HeapObject> obj,
647  wasm::CanonicalValueType type,
648  uint32_t offset) {
649  Address field_address = obj->GetFieldAddress(offset);
650  switch (type.kind()) {
651  case wasm::kI8: {
652  int8_t value = base::Memory<int8_t>(field_address);
653  return direct_handle(Smi::FromInt(value), isolate);
654  }
655  case wasm::kI16: {
656  int16_t value = base::Memory<int16_t>(field_address);
657  return direct_handle(Smi::FromInt(value), isolate);
658  }
659  case wasm::kI32: {
660  int32_t value = base::Memory<int32_t>(field_address);
661  return isolate->factory()->NewNumberFromInt(value);
662  }
663  case wasm::kI64: {
664  int64_t value = base::ReadUnalignedValue<int64_t>(field_address);
665  return BigInt::FromInt64(isolate, value);
666  }
667  case wasm::kF16: {
668  uint16_t value = base::Memory<uint16_t>(field_address);
669  return isolate->factory()->NewNumber(fp16_ieee_to_fp32_value(value));
670  }
671  case wasm::kF32: {
672  float value = base::Memory<float>(field_address);
673  return isolate->factory()->NewNumber(value);
674  }
675  case wasm::kF64: {
676  double value = base::ReadUnalignedValue<double>(field_address);
677  return isolate->factory()->NewNumber(value);
678  }
679  case wasm::kS128:
680  // TODO(v8:11804): implement
681  UNREACHABLE();
682 
683  case wasm::kRef:
684  case wasm::kRefNull: {
685  ObjectSlot slot(field_address);
686  return direct_handle(slot.load(isolate), isolate);
687  }
688 
689  case wasm::kVoid:
690  case wasm::kTop:
691  case wasm::kBottom:
692  UNREACHABLE();
693  }
694 }
