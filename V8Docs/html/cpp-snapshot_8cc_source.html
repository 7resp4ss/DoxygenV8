106  public:
107  // Objects can either be hidden/visible, or depend on some other nodes while

127  // processed, see also IsPending() below.
128  bool IsVisited() const { return visited_; }
129 
130  // Pending objects are currently being processed as part of the same SCC.
131  bool IsPending() const { return pending_; }

133  bool IsVisibleNotDependent() {
134  auto v = GetVisibility();

136  return v == Visibility::kVisible;

140  CHECK_EQ(Visibility::kVisible, GetVisibility());
141  DCHECK_NULL(node_);

146  CHECK_EQ(Visibility::kVisible, GetVisibility());
147  return node_;

149 
150  protected:
151  const void* key_;
152  // State count keeps track of node processing order. It is used to create only
153  // dependencies on ancestors in the sub graph which ensures that there will be
154  // no cycles in dependencies.
155  const size_t state_count_;
156 
157  Visibility visibility_;
158  StateBase* visibility_dependency_ = nullptr;
159  EmbedderNode* node_;
160  bool visited_;
161  bool pending_ = false;
162 
163  Visibility GetVisibility() {
164  FollowDependencies();
165  return visibility_;
166  }
167 
168  StateBase* FollowDependencies() {
169  if (visibility_ != Visibility::kDependentVisibility) {
170  CHECK_NULL(visibility_dependency_);
171  return this;
172  }
173  StateBase* current = this;
174  std::vector<StateBase*> dependencies;
175  while (current->visibility_dependency_ &&
176  current->visibility_dependency_ != current) {
177  DCHECK_EQ(Visibility::kDependentVisibility, current->visibility_);
178  dependencies.push_back(current);
179  current = current->visibility_dependency_;
180  }
181  auto new_visibility = Visibility::kDependentVisibility;
182  auto* new_visibility_dependency = current;
183  if (current->visibility_ == Visibility::kVisible) {
184  new_visibility = Visibility::kVisible;
185  new_visibility_dependency = nullptr;
186  } else if (!IsPending()) {
187  DCHECK(IsVisited());
188  // The object was not visible (above case). Having a dependency on itself
189  // or null means no visible object was found.
190  new_visibility = Visibility::kHidden;
191  new_visibility_dependency = nullptr;
192  }
193  current->visibility_ = new_visibility;
194  current->visibility_dependency_ = new_visibility_dependency;
195  for (auto* state : dependencies) {
196  state->visibility_ = new_visibility;
197  state->visibility_dependency_ = new_visibility_dependency;
198  }
199  return current;
200  }
201 
202  friend class State;
203 };

225  void MarkDependentVisibility(StateBase* dependency) {
226  // Follow and update dependencies as much as possible.
227  dependency = dependency->FollowDependencies();
228  DCHECK(dependency->IsVisited());
229  if (visibility_ == StateBase::Visibility::kVisible) {
230  // Already visible, no dependency needed.
231  DCHECK_NULL(visibility_dependency_);
232  return;
233  }
234  if (dependency->visibility_ == Visibility::kVisible) {
235  // Simple case: Dependency is visible.
236  visibility_ = Visibility::kVisible;
237  visibility_dependency_ = nullptr;
238  return;
239  }
240  if ((visibility_dependency_ &&
241  (visibility_dependency_->state_count_ > dependency->state_count_)) ||
242  (!visibility_dependency_ &&
243  (state_count_ > dependency->state_count_))) {
244  // Only update when new state_count_ < original state_count_. This
245  // ensures that we pick an ancestor as dependency and not a child which
246  // is guaranteed to converge to an answer.
247  //
248  // Dependency is now
249  // a) either pending with unknown visibility (same call chain), or
250  // b) not pending and has defined visibility.
251  //
252  // It's not possible to point to a state that is not pending but has
253  // dependent visibility because dependencies are updated to the top-most
254  // dependency at the beginning of method.
255  if (dependency->IsPending()) {
256  visibility_ = Visibility::kDependentVisibility;
257  visibility_dependency_ = dependency;
258  } else {
259  CHECK_NE(Visibility::kDependentVisibility, dependency->visibility_);
260  if (dependency->visibility_ == Visibility::kVisible) {
261  visibility_ = Visibility::kVisible;
262  visibility_dependency_ = nullptr;
263  }
264  }
265  }
266  }

403 // The following implements a snapshotting algorithm for C++ objects that also
404 // filters strongly-connected components (SCCs) of only "hidden" objects that
405 // are not (transitively) referencing any non-hidden objects.
406 //
407 // C++ objects come in two versions.
408 // a. Named objects that have been assigned a name through NameProvider.
409 // b. Unnamed objects, that are potentially hidden if the build configuration
410 // requires Oilpan to hide such names. Hidden objects have their name
411 // set to NameProvider::kHiddenName.
412 //
413 // The main challenge for the algorithm is to avoid blowing up the final object
414 // graph with hidden nodes that do not carry information. For that reason, the
415 // algorithm filters SCCs of only hidden objects, e.g.:
416 // ... -> (object) -> (object) -> (hidden) -> (hidden)
417 // In this case the (hidden) objects are filtered from the graph. The trickiest
418 // part is maintaining visibility state for objects referencing other objects
419 // that are currently being processed.
420 //
421 // Main algorithm idea (two passes):
422 // 1. First pass marks all non-hidden objects and those that transitively reach
423 // non-hidden objects as visible. Details:
424 // - Iterate over all objects.
425 // - If object is non-hidden mark it as visible and also mark parent as
426 // visible if needed.
427 // - If object is hidden, traverse children as DFS to find non-hidden
428 // objects. Post-order process the objects and mark those objects as
429 // visible that have child nodes that are visible themselves.
430 // - Maintain an epoch counter (StateStorage::state_count_) to allow
431 // deferring the visibility decision to other objects in the same SCC. This
432 // is similar to the "lowlink" value in Tarjan's algorithm for SCC.
433 // - After the first pass it is guaranteed that all deferred visibility
434 // decisions can be resolved.
435 // 2. Second pass adds nodes and edges for all visible objects.
436 // - Upon first checking the visibility state of an object, all deferred
437 // visibility states are resolved.
438 //
439 // For practical reasons, the recursion is transformed into an iteration. We do
440 // do not use plain Tarjan's algorithm to avoid another pass over all nodes to
441 // create SCCs.

487  void AddEdge(State& parent, const HeapObjectHeader& header,
488  std::string_view edge_name) {
489  DCHECK(parent.IsVisibleNotDependent());
490  auto& current = states_.GetExistingState(header);
491  if (!current.IsVisibleNotDependent()) return;
492 
493  // Both states are visible. Create nodes in case this is the first edge
494  // created for any of them.
495  if (!parent.get_node()) {
496  parent.set_node(AddNode(*parent.header()));
497  }
498  if (!current.get_node()) {
499  current.set_node(AddNode(header));
500  }
501 
502  if (!edge_name.empty()) {
503  graph_.AddEdge(parent.get_node(), current.get_node(),
504  parent.get_node()->InternalizeEdgeName(edge_name));
505  } else {
506  graph_.AddEdge(parent.get_node(), current.get_node());
507  }

510  void AddEdge(State& parent, const TracedReferenceBase& ref,
511  std::string_view edge_name) {
512  DCHECK(parent.IsVisibleNotDependent());
513  v8::Local<v8::Data> v8_data =
514  ref.Get(reinterpret_cast<v8::Isolate*>(cpp_heap_.isolate()));
515  if (v8_data.IsEmpty()) return;
516 
517  if (!parent.get_node()) {
518  parent.set_node(AddNode(*parent.header()));
519  }
520  auto* v8_node = graph_.V8Node(v8_data);
521  if (!edge_name.empty()) {
522  graph_.AddEdge(parent.get_node(), v8_node,
523  parent.get_node()->InternalizeEdgeName(edge_name));
524  } else {
525  graph_.AddEdge(parent.get_node(), v8_node);
526  }
527 
528  // Don't merge nodes if edges have a name.
529  if (!edge_name.empty()) return;
530 
531  // Try to extract the back reference. If the back reference matches
532  // `parent`, then the nodes are merged.
533  void* back_reference_object = ExtractEmbedderDataBackref(
534  reinterpret_cast<v8::internal::Isolate*>(cpp_heap_.isolate()),
535  cpp_heap_, v8_data);
536  if (!back_reference_object) return;
537  // Only JS objects have back references.
538  DCHECK(v8_data->IsValue() && v8_data.As<v8::Value>()->IsObject());
539 
540  auto& back_header = HeapObjectHeader::FromObject(back_reference_object);
541  auto& back_state = states_.GetExistingState(back_header);
542 
543  // If the back reference doesn't point to the same header, just return. In
544  // such a case we have stand-alone references to a wrapper.
545  if (parent.header() != back_state.header()) return;
546 
547  // Back reference points to parents header. In this case, the nodes should
548  // be merged and query the detachedness state of the embedder.
549  if (!back_state.get_node()) {
550  back_state.set_node(AddNode(back_header));
551  }
552  back_state.get_node()->SetWrapperNode(v8_node);
553 
554  auto* profiler = reinterpret_cast<Isolate*>(cpp_heap_.isolate())
555  ->heap()
556  ->heap_profiler();
557  if (profiler->HasGetDetachednessCallback()) {
558  back_state.get_node()->SetDetachedness(
559  profiler->GetDetachedness(v8_data.As<v8::Value>(), 0));
560  }
561  }

646  WeakContainerScope weak_container_scope(*this, container_header);

648  graph_builder_.VisitWeakContainerForVisibility(container_header);
649 
650  if (!weak_desc.callback) {
651  // Weak container does not contribute to liveness.
652  return;
653  }
654  // Heap snapshot is always run after a GC so we know there are no dead
655  // entries in the container.
656  if (object) {
657  // The container will itself be traced strongly via the regular Visit()
658  // handling that iterates over all live objects. The visibility visitor
659  // will thus see (because of strongly treating the container):
660  // 1. the container itself;
661  // 2. for each {key} in container: container->key;
662  // 3. for each {key, value} in container: key->value;
663  //
664  // In case the visitor is used stand-alone, we trace through the container
665  // here to create the same state as we would when the container is traced
666  // separately.
667  container_header.TraceImpl(this);
668  }

857 void CppGraphBuilderImpl::VisitForVisibility(State* parent,
858  const HeapObjectHeader& header) {
859  auto& current = states_.GetOrCreateState(header);
860 
861  if (current.IsVisited()) {
862  // Avoid traversing into already visited subgraphs and just update the state
863  // based on a previous result.
864  if (parent) {
865  parent->MarkDependentVisibility(&current);
866  }
867  return;
868  }
869 
870  current.MarkVisited();
871  if (header.GetName().name_was_hidden) {
872  current.MarkPending();
873  workstack_.push_back(std::unique_ptr<WorkstackItemBase>{
874  new VisitationItem(parent, current)});
875  } else {
876  // No need to mark/unmark pending as the node is immediately processed.
877  current.MarkVisible();
878  // In case the names are visible, the graph is not traversed in this phase.
879  // Explicitly trace one level to handle weak containers.
880  WeakVisitor weak_visitor(*this);
881  header.TraceImpl(&weak_visitor);
882  if (parent) {
883  // Eagerly update a parent object as its visibility state is now fixed.
884  parent->MarkVisible();
885  }
886  }

1021 void CppGraphBuilderImpl::Run() {
1022  // Sweeping from a previous GC might still be running, in which case not all
1023  // pages have been returned to spaces yet.
1024  cpp_heap_.sweeper().FinishIfRunning();
1025  cppgc::subtle::DisallowGarbageCollectionScope no_gc(
1026  cpp_heap_.GetHeapHandle());
1027  // First pass: Figure out which objects should be included in the graph -- see
1028  // class-level comment on CppGraphBuilder.
1029  LiveObjectsForVisibilityIterator visitor(*this);
1030  visitor.Traverse(cpp_heap_.raw_heap());
1031  // Second pass: Add graph nodes for objects that must be shown.
1032  states_.ForAllStates([this](StateBase* state_base) {
1033  // No roots have been created so far, so all StateBase objects are State.
1034  State& state = *static_cast<State*>(state_base);
1035 
1036  if (!state.IsVisibleNotDependent()) {
1037  graph_.AddNativeSize(state.header()->AllocatedSize());
1038  return;
1039  }
1040 
1041  // Emit no edges for the contents of the weak containers. For both, fully
1042  // weak and ephemeron containers, the contents should be retained from
1043  // somewhere else.
1044  if (state.IsWeakContainer()) return;
1045 
1046  ParentScope parent_scope(state);
1047  GraphBuildingVisitor object_visitor(*this, parent_scope);
1048  if (!state.header()->IsInConstruction()) {
1049  // TODO(mlippautz): Handle in-construction objects.
1050  state.header()->TraceImpl(&object_visitor);
1051  }
1052  state.ForAllEphemeronEdges([this, &state](const HeapObjectHeader& value) {
1053  AddEdge(state, value, kEphemeronEdgeName);
1054  });
1055  object_visitor.set_edge_name(kEphemeronEdgeName);
1056  state.ForAllEagerEphemeronEdges(
1057  [&object_visitor](const void* value, cppgc::TraceCallback callback) {
1058  callback(&object_visitor, value);
1059  });
1060  });
1061  // Add roots.
1062  {
1063  ParentScope parent_scope(
1064  states_.CreateRootState(AddRootNode("C++ Persistent roots")));
1065  GraphBuildingRootVisitor root_object_visitor(*this, parent_scope);
1066  cpp_heap_.GetStrongPersistentRegion().Iterate(root_object_visitor);
1067  }
1068  {
1069  ParentScope parent_scope(states_.CreateRootState(
1070  AddRootNode("C++ CrossThreadPersistent roots")));
1071  GraphBuildingRootVisitor root_object_visitor(*this, parent_scope);
1072  cppgc::internal::PersistentRegionLock guard;
1073  cpp_heap_.GetStrongCrossThreadPersistentRegion().Iterate(
1074  root_object_visitor);
1075  }
1076  // Only add stack roots in case the callback is not run from generating a
1077  // snapshot without stack. This avoids adding false-positive edges when
1078  // conservatively scanning the stack.
1079  if (cpp_heap_.isolate()->heap()->IsGCWithMainThreadStack()) {
1080  ParentScope parent_scope(
1081  states_.CreateRootState(AddRootNode("C++ native stack roots")));
1082  GraphBuildingRootVisitor root_object_visitor(*this, parent_scope);
1083  GraphBuildingStackVisitor stack_visitor(*this, cpp_heap_,
1084  root_object_visitor);
1085  cpp_heap_.stack()->IteratePointersUntilMarker(&stack_visitor);
1086  }
1087  // Connect each `CppHeapExternalObject` to its corresponding cpp object.
1088  for (Tagged<CppHeapExternalObject> object : cpp_heap_external_objects_) {
1089  AddEdgeForCppHeapExternalObject(object);
1090  }
1091 }
