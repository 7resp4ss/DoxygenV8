27 TNode<Object> BinaryOpAssembler::Generate_AddWithFeedback(
28  const LazyNode<Context>& context, TNode<Object> lhs, TNode<Object> rhs,
29  TNode<UintPtrT> slot_id, const LazyNode<HeapObject>& maybe_feedback_vector,
30  UpdateFeedbackMode update_feedback_mode, bool rhs_known_smi) {
31  // Shared entry for floating point addition.
32  Label do_fadd(this), if_lhsisnotnumber(this, Label::kDeferred),
33  check_rhsisoddball(this, Label::kDeferred),
34  call_with_oddball_feedback(this), call_with_any_feedback(this),
35  call_add_stub(this), end(this), bigint(this, Label::kDeferred),
36  bigint64(this);
37  TVARIABLE(Float64T, var_fadd_lhs);
38  TVARIABLE(Float64T, var_fadd_rhs);
39  TVARIABLE(Smi, var_type_feedback);
40  TVARIABLE(Object, var_result);
41 
42  // Check if the {lhs} is a Smi or a HeapObject.
43  Label if_lhsissmi(this);
44  // If rhs is known to be an Smi we want to fast path Smi operation. This is
45  // for AddSmi operation. For the normal Add operation, we want to fast path
46  // both Smi and Number operations, so this path should not be marked as
47  // Deferred.
48  Label if_lhsisnotsmi(this,
49  rhs_known_smi ? Label::kDeferred : Label::kNonDeferred);
50  Branch(TaggedIsNotSmi(lhs), &if_lhsisnotsmi, &if_lhsissmi);
51 
52  auto IsAdditiveSafeIntegerFeedback = [&](TNode<Float64T> value) {
53  return Select<BoolT>(
54  IsAdditiveSafeIntegerFeedbackEnabled(),
55  [&] { return IsAdditiveSafeInteger(value); },
56  [&] { return BoolConstant(false); });
57  };
58 
59  BIND(&if_lhsissmi);
60  {
61  Comment("lhs is Smi");
62  TNode<Smi> lhs_smi = CAST(lhs);
63  if (!rhs_known_smi) {
64  // Check if the {rhs} is also a Smi.
65  Label if_rhsissmi(this), if_rhsisnotsmi(this);
66  Branch(TaggedIsSmi(rhs), &if_rhsissmi, &if_rhsisnotsmi);
67 
68  BIND(&if_rhsisnotsmi);
69  {
70  // Check if the {rhs} is a HeapNumber.
71  TNode<HeapObject> rhs_heap_object = CAST(rhs);
72  GotoIfNot(IsHeapNumber(rhs_heap_object), &check_rhsisoddball);
73 
74  var_fadd_lhs = SmiToFloat64(lhs_smi);
75  var_fadd_rhs = LoadHeapNumberValue(rhs_heap_object);
76  var_type_feedback = SelectSmiConstant(
77  IsAdditiveSafeIntegerFeedback(var_fadd_rhs.value()),
78  BinaryOperationFeedback::kAdditiveSafeInteger,
79  BinaryOperationFeedback::kNumber);
80  Goto(&do_fadd);
81  }
82 
83  BIND(&if_rhsissmi);
84  }
85 
86  {
87  Comment("perform smi operation");
88  // If rhs is known to be an Smi we want to fast path Smi operation. This
89  // is for AddSmi operation. For the normal Add operation, we want to fast
90  // path both Smi and Number operations, so this path should not be marked
91  // as Deferred.
92  TNode<Smi> rhs_smi = CAST(rhs);
93  Label if_overflow(this,
94  rhs_known_smi ? Label::kDeferred : Label::kNonDeferred);
95  TNode<Smi> smi_result = TrySmiAdd(lhs_smi, rhs_smi, &if_overflow);
96  // Not overflowed.
97  {
98  var_type_feedback = SmiConstant(BinaryOperationFeedback::kSignedSmall);
99  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(),
100  slot_id, update_feedback_mode);
101  var_result = smi_result;
102  Goto(&end);
103  }
104 
105  BIND(&if_overflow);
106  {
107  var_fadd_lhs = SmiToFloat64(lhs_smi);
108  var_fadd_rhs = SmiToFloat64(rhs_smi);
109  var_type_feedback =
110  SelectSmiConstant(IsAdditiveSafeIntegerFeedbackEnabled(),
111  BinaryOperationFeedback::kAdditiveSafeInteger,
112  BinaryOperationFeedback::kNumber);
113  Goto(&do_fadd);
114  }
115  }
116  }
117 
118  BIND(&if_lhsisnotsmi);
119  {
120  // Check if {lhs} is a HeapNumber.
121  TNode<HeapObject> lhs_heap_object = CAST(lhs);
122  GotoIfNot(IsHeapNumber(lhs_heap_object), &if_lhsisnotnumber);
123 
124  if (!rhs_known_smi) {
125  // Check if the {rhs} is Smi.
126  Label if_rhsissmi(this), if_rhsisnotsmi(this);
127  Branch(TaggedIsSmi(rhs), &if_rhsissmi, &if_rhsisnotsmi);
128 
129  BIND(&if_rhsisnotsmi);
130  {
131  // Check if the {rhs} is a HeapNumber.
132  TNode<HeapObject> rhs_heap_object = CAST(rhs);
133  GotoIfNot(IsHeapNumber(rhs_heap_object), &check_rhsisoddball);
134 
135  var_fadd_lhs = LoadHeapNumberValue(lhs_heap_object);
136  var_fadd_rhs = LoadHeapNumberValue(rhs_heap_object);
137  var_type_feedback = SmiConstant(BinaryOperationFeedback::kNumber);
138  GotoIfNot(IsAdditiveSafeIntegerFeedback(var_fadd_lhs.value()),
139  &do_fadd);
140  GotoIfNot(IsAdditiveSafeIntegerFeedback(var_fadd_rhs.value()),
141  &do_fadd);
142  var_type_feedback =
143  SmiConstant(BinaryOperationFeedback::kAdditiveSafeInteger);
144  Goto(&do_fadd);
145  }
146 
147  BIND(&if_rhsissmi);
148  }
149  {
150  var_fadd_lhs = LoadHeapNumberValue(lhs_heap_object);
151  var_fadd_rhs = SmiToFloat64(CAST(rhs));
152  var_type_feedback =
153  SelectSmiConstant(IsAdditiveSafeIntegerFeedback(var_fadd_lhs.value()),
154  BinaryOperationFeedback::kAdditiveSafeInteger,
155  BinaryOperationFeedback::kNumber);
156  Goto(&do_fadd);
157  }
158  }
159 
160  BIND(&do_fadd);
161  {
162  TNode<Float64T> value =
163  Float64Add(var_fadd_lhs.value(), var_fadd_rhs.value());
164  TNode<HeapNumber> result = AllocateHeapNumberWithValue(value);
165  var_result = result;
166 
167  Label AdditiveSafeInteger_overflow_check_done(this);
168  GotoIfNot(IsAdditiveSafeIntegerFeedbackEnabled(),
169  &AdditiveSafeInteger_overflow_check_done);
170  {
171  GotoIfNot(
172  SmiEqual(var_type_feedback.value(),
173  SmiConstant(BinaryOperationFeedback::kAdditiveSafeInteger)),
174  &AdditiveSafeInteger_overflow_check_done);
175  GotoIf(IsAdditiveSafeIntegerFeedback(value),
176  &AdditiveSafeInteger_overflow_check_done);
177  var_type_feedback = SmiConstant(BinaryOperationFeedback::kNumber);
178  Goto(&AdditiveSafeInteger_overflow_check_done);
179  }
180  BIND(&AdditiveSafeInteger_overflow_check_done);
181 
182  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(), slot_id,
183  update_feedback_mode);
184  Goto(&end);
185  }
186 
187  BIND(&if_lhsisnotnumber);
188  {
189  // No checks on rhs are done yet. We just know lhs is not a number or Smi.
190  Label if_lhsisoddball(this), if_lhsisnotoddball(this);
191  TNode<Uint16T> lhs_instance_type = LoadInstanceType(CAST(lhs));
192  TNode<BoolT> lhs_is_oddball =
193  InstanceTypeEqual(lhs_instance_type, ODDBALL_TYPE);
194  Branch(lhs_is_oddball, &if_lhsisoddball, &if_lhsisnotoddball);
195 
196  BIND(&if_lhsisoddball);
197  {
198  GotoIf(TaggedIsSmi(rhs), &call_with_oddball_feedback);
199 
200  // Check if {rhs} is a HeapNumber.
201  Branch(IsHeapNumber(CAST(rhs)), &call_with_oddball_feedback,
202  &check_rhsisoddball);
203  }
204 
205  BIND(&if_lhsisnotoddball);
206  {
207  // Check if the {rhs} is a smi, and exit the string and bigint check early
208  // if it is.
209  GotoIf(TaggedIsSmi(rhs), &call_with_any_feedback);
210  TNode<HeapObject> rhs_heap_object = CAST(rhs);
211 
212  Label lhs_is_string(this), lhs_is_bigint(this);
213  GotoIf(IsStringInstanceType(lhs_instance_type), &lhs_is_string);
214  GotoIf(IsBigIntInstanceType(lhs_instance_type), &lhs_is_bigint);
215  Goto(&call_with_any_feedback);
216 
217  BIND(&lhs_is_bigint);
218  {
219  GotoIfNot(IsBigInt(rhs_heap_object), &call_with_any_feedback);
220  if (Is64()) {
221  GotoIfLargeBigInt(CAST(lhs), &bigint);
222  GotoIfLargeBigInt(CAST(rhs), &bigint);
223  Goto(&bigint64);
224  } else {
225  Goto(&bigint);
226  }
227  }
228 
229  BIND(&lhs_is_string);
230  {
231  Label lhs_is_string_rhs_is_not_string(this);
232 
233  TNode<Uint16T> rhs_instance_type = LoadInstanceType(rhs_heap_object);
234 
235  // Fast path where both {lhs} and {rhs} are strings. Since {lhs} is a
236  // string we no longer need an Oddball check.
237  GotoIfNot(IsStringInstanceType(rhs_instance_type),
238  &lhs_is_string_rhs_is_not_string);
239 
240  var_type_feedback = SmiConstant(BinaryOperationFeedback::kString);
241  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(),
242  slot_id, update_feedback_mode);
243  var_result =
244  CallBuiltin(Builtin::kStringAdd_CheckNone, context(), lhs, rhs);
245 
246  Goto(&end);
247 
248  BIND(&lhs_is_string_rhs_is_not_string);
249 
250  GotoIfNot(IsStringWrapper(rhs_heap_object), &call_with_any_feedback);
251 
252  // lhs is a string and rhs is a string wrapper.
253 
254  TNode<PropertyCell> to_primitive_protector =
255  StringWrapperToPrimitiveProtectorConstant();
256  GotoIf(TaggedEqual(LoadObjectField(to_primitive_protector,
257  PropertyCell::kValueOffset),
258  SmiConstant(Protectors::kProtectorInvalid)),
259  &call_with_any_feedback);
260 
261  var_type_feedback =
262  SmiConstant(BinaryOperationFeedback::kStringOrStringWrapper);
263  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(),
264  slot_id, update_feedback_mode);
265  TNode<String> rhs_string =
266  CAST(LoadJSPrimitiveWrapperValue(CAST(rhs_heap_object)));
267  var_result = CallBuiltin(Builtin::kStringAdd_CheckNone, context(), lhs,
268  rhs_string);
269  Goto(&end);
270  }
271  }
272  }
273  // TODO(v8:12199): Support "string wrapper + string" concatenation too.
274 
275  BIND(&check_rhsisoddball);
276  {
277  // Check if rhs is an oddball. At this point we know lhs is either a
278  // Smi or number or oddball and rhs is not a number or Smi.
279  TNode<Uint16T> rhs_instance_type = LoadInstanceType(CAST(rhs));
280  TNode<BoolT> rhs_is_oddball =
281  InstanceTypeEqual(rhs_instance_type, ODDBALL_TYPE);
282  GotoIf(rhs_is_oddball, &call_with_oddball_feedback);
283  Goto(&call_with_any_feedback);
284  }
285 
286  if (Is64()) {
287  BIND(&bigint64);
288  {
289  // Both {lhs} and {rhs} are of BigInt type and can fit in 64-bit
290  // registers.
291  Label if_overflow(this);
292  TVARIABLE(UintPtrT, lhs_raw);
293  TVARIABLE(UintPtrT, rhs_raw);
294  BigIntToRawBytes(CAST(lhs), &lhs_raw, &lhs_raw);
295  BigIntToRawBytes(CAST(rhs), &rhs_raw, &rhs_raw);
296  var_result = BigIntFromInt64(
297  TryIntPtrAdd(UncheckedCast<IntPtrT>(lhs_raw.value()),
298  UncheckedCast<IntPtrT>(rhs_raw.value()), &if_overflow));
299 
300  var_type_feedback = SmiConstant(BinaryOperationFeedback::kBigInt64);
301  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(),
302  slot_id, update_feedback_mode);
303  Goto(&end);
304 
305  BIND(&if_overflow);
306  Goto(&bigint);
307  }
308  }
309 
310  BIND(&bigint);
311  {
312  // Both {lhs} and {rhs} are of BigInt type.
313  Label bigint_too_big(this);
314  var_result = CallBuiltin(Builtin::kBigIntAddNoThrow, context(), lhs, rhs);
315  // Check for sentinel that signals BigIntTooBig exception.
316  GotoIf(TaggedIsSmi(var_result.value()), &bigint_too_big);
317 
318  var_type_feedback = SmiConstant(BinaryOperationFeedback::kBigInt);
319  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(), slot_id,
320  update_feedback_mode);
321  Goto(&end);
322 
323  BIND(&bigint_too_big);
324  {
325  // Update feedback to prevent deopt loop.
326  UpdateFeedback(SmiConstant(BinaryOperationFeedback::kAny),
327  maybe_feedback_vector(), slot_id, update_feedback_mode);
328  ThrowRangeError(context(), MessageTemplate::kBigIntTooBig);
329  }
330  }
331 
332  BIND(&call_with_oddball_feedback);
333  {
334  var_type_feedback = SmiConstant(BinaryOperationFeedback::kNumberOrOddball);
335  Goto(&call_add_stub);
336  }
337 
338  BIND(&call_with_any_feedback);
339  {
340  var_type_feedback = SmiConstant(BinaryOperationFeedback::kAny);
341  Goto(&call_add_stub);
342  }
343 
344  BIND(&call_add_stub);
345  {
346  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(), slot_id,
347  update_feedback_mode);
348  var_result = CallBuiltin(Builtin::kAdd, context(), lhs, rhs);
349  Goto(&end);
350  }
351 
352  BIND(&end);
353  return var_result.value();
354 }

377 TNode<Object>
378 BinaryOpAssembler::Generate_AddStringConstantInternalizeWithFeedback(
379  const LazyNode<Context>& context, TNode<Object> lhs, TNode<Object> rhs,
380  TNode<UintPtrT> slot_id, const LazyNode<HeapObject>& maybe_feedback_vector,
381  UpdateFeedbackMode update_feedback_mode, bool rhs_known_smi,
382  AddStringConstantAndInternalizeVariant as_variant) {
383  using ASVariant = AddStringConstantAndInternalizeVariant;
384  Label call_with_any_feedback(this), call_add_stub(this),
385  slow(this, Label::kDeferred), end(this);
386  TVARIABLE(Smi, var_type_feedback, SmiConstant(BinaryOperationFeedback::kAny));
387  TVARIABLE(Object, var_result);
388 
389  TNode<Object> const_str_operand =
390  as_variant == ASVariant::kLhsIsStringConstant ? lhs : rhs;
391  TNode<Object> other_operand =
392  as_variant == ASVariant::kLhsIsStringConstant ? rhs : lhs;
393 
394  // Either lhs or rhs is an internalized string constant. Since our cache is
395  // stored in the feedback vector and is thus specific to a particular
396  // bytecode, we can ignore the constant and use the other parameter as the
397  // cache key instead. This is our optimization:
398  //
399  // Instead of: return Internalize(lhs + rhs)
400  // .. we do: return cache[rhs] // .. (or lhs) if possible.
401  USE(const_str_operand);
402  CSA_DCHECK(this, Word32BinaryNot(TaggedIsSmi(const_str_operand)));
403  CSA_DCHECK(this, IsInternalizedStringInstanceType(
404  LoadInstanceType(CAST(const_str_operand))));
405 
406  // We need the cache slot for our optimization. Without it, delegate to the
407  // generic Add logic.
408  {
409  Label next(this);
410  GotoIfNot(IsUndefined(maybe_feedback_vector()), &next);
411  // TODO(jgruber): This could be a builtin call. Add a builtin that expects
412  // a possibly-undefined feedback vector.
413  var_result = Generate_AddWithFeedback(context, lhs, rhs, slot_id,
414  maybe_feedback_vector,
415  update_feedback_mode, rhs_known_smi);
416  Goto(&end);
417  BIND(&next);
418  }
419  TNode<FeedbackVector> fv = CAST(maybe_feedback_vector());
420 
421  // Ensure other_operand is internalized so we can query the cache.
422  // TODO(jgruber): We could TryInternalize in CSA.
423  {
424  Label set_feedback_to_any_then_slow(this), next(this);
425  GotoIf(TaggedIsSmi(other_operand), &set_feedback_to_any_then_slow);
426  TNode<Uint16T> itype = LoadInstanceType(CAST(other_operand));
427  GotoIfNot(IsStringInstanceType(itype), &set_feedback_to_any_then_slow);
428 
429  var_type_feedback = SmiConstant(BinaryOperationFeedback::kString);
430  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(), slot_id,
431  update_feedback_mode);
432  GotoIfNot(IsInternalizedStringInstanceType(itype), &slow);
433  Goto(&next);
434 
435  BIND(&set_feedback_to_any_then_slow);
436  var_type_feedback = SmiConstant(BinaryOperationFeedback::kAny);
437  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(), slot_id,
438  update_feedback_mode);
439  Goto(&slow);
440 
441  BIND(&next);
442  }
443  TNode<String> other_operand_internalized = CAST(other_operand);
444 
445  // Get or create the cache object.
446  TNode<SimpleNameDictionary> cache;
447  {
448  Label next(this);
449  static constexpr int kSlotOffset =
450  kAdd_StringConstant_Internalize_CacheSlotOffset;
451  TVARIABLE(Object, var_maybe_cache);
452  var_maybe_cache = CAST(LoadFeedbackVectorSlot(
453  fv, UintPtrAdd(slot_id, UintPtrConstant(kSlotOffset))));
454  GotoIf(
455  TaggedNotEqual(var_maybe_cache.value(), UninitializedSymbolConstant()),
456  &next);
457  var_maybe_cache = AllocateSimpleNameDictionary(1);
458  Goto(&next);
459  BIND(&next);
460  cache = CAST(var_maybe_cache.value());
461  }
462 
463  {
464  // TODO(jgruber): Fill the cache in CSA.
465  TVARIABLE(IntPtrT, var_entry);
466  Label index_found(this, {&var_entry});
467  NameDictionaryLookup<SimpleNameDictionary>(
468  cache, other_operand_internalized, &index_found, &var_entry, &slow,
469  LookupMode::kFindExisting);
470  BIND(&index_found);
471  var_result =
472  LoadValueByKeyIndex<SimpleNameDictionary>(cache, var_entry.value());
473  Goto(&end);
474  }
475 
476  BIND(&slow);
477  {
478  auto target = as_variant == ASVariant::kLhsIsStringConstant
479  ? Runtime::kStringAdd_LhsIsStringConstant_Internalize
480  : Runtime::kStringAdd_RhsIsStringConstant_Internalize;
481  var_result =
482  CallRuntime(target, context(), lhs, rhs, maybe_feedback_vector(),
483  IntPtrToTaggedIndex(Signed(slot_id)));
484  Goto(&end);
485  }
486 
487  BIND(&end);
488  return var_result.value();
489 }

491 TNode<Object> BinaryOpAssembler::Generate_BinaryOperationWithFeedback(
492  const LazyNode<Context>& context, TNode<Object> lhs, TNode<Object> rhs,
493  TNode<UintPtrT> slot_id, const LazyNode<HeapObject>& maybe_feedback_vector,
494  const SmiOperation& smiOperation, const FloatOperation& floatOperation,
495  Operation op, UpdateFeedbackMode update_feedback_mode, bool rhs_known_smi) {
496  Label do_float_operation(this), end(this), call_stub(this),
497  check_rhsisoddball(this, Label::kDeferred), call_with_any_feedback(this),
498  if_lhsisnotnumber(this, Label::kDeferred),
499  if_both_bigint(this, Label::kDeferred), if_both_bigint64(this);
500  TVARIABLE(Float64T, var_float_lhs);
501  TVARIABLE(Float64T, var_float_rhs);
502  TVARIABLE(Smi, var_type_feedback);
503  TVARIABLE(Object, var_result);
504 
505  Label if_lhsissmi(this);
506  // If rhs is known to be an Smi (in the SubSmi, MulSmi, DivSmi, ModSmi
507  // bytecode handlers) we want to fast path Smi operation. For the normal
508  // operation, we want to fast path both Smi and Number operations, so this
509  // path should not be marked as Deferred.
510  Label if_lhsisnotsmi(this,
511  rhs_known_smi ? Label::kDeferred : Label::kNonDeferred);
512  Branch(TaggedIsNotSmi(lhs), &if_lhsisnotsmi, &if_lhsissmi);
513 
514  // Check if the {lhs} is a Smi or a HeapObject.
515  BIND(&if_lhsissmi);
516  {
517  Comment("lhs is Smi");
518  TNode<Smi> lhs_smi = CAST(lhs);
519  if (!rhs_known_smi) {
520  // Check if the {rhs} is also a Smi.
521  Label if_rhsissmi(this), if_rhsisnotsmi(this);
522  Branch(TaggedIsSmi(rhs), &if_rhsissmi, &if_rhsisnotsmi);
523 
524  BIND(&if_rhsisnotsmi);
525  {
526  // Check if {rhs} is a HeapNumber.
527  TNode<HeapObject> rhs_heap_object = CAST(rhs);
528  GotoIfNot(IsHeapNumber(rhs_heap_object), &check_rhsisoddball);
529 
530  // Perform a floating point operation.
531  var_float_lhs = SmiToFloat64(lhs_smi);
532  var_float_rhs = LoadHeapNumberValue(rhs_heap_object);
533  Goto(&do_float_operation);
534  }
535 
536  BIND(&if_rhsissmi);
537  }
538 
539  {
540  Comment("perform smi operation");
541  var_result = smiOperation(lhs_smi, CAST(rhs), &var_type_feedback);
542  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(),
543  slot_id, update_feedback_mode);
544  Goto(&end);
545  }
546  }
547 
548  BIND(&if_lhsisnotsmi);
549  {
550  Comment("lhs is not Smi");
551  // Check if the {lhs} is a HeapNumber.
552  TNode<HeapObject> lhs_heap_object = CAST(lhs);
553  GotoIfNot(IsHeapNumber(lhs_heap_object), &if_lhsisnotnumber);
554 
555  if (!rhs_known_smi) {
556  // Check if the {rhs} is a Smi.
557  Label if_rhsissmi(this), if_rhsisnotsmi(this);
558  Branch(TaggedIsSmi(rhs), &if_rhsissmi, &if_rhsisnotsmi);
559 
560  BIND(&if_rhsisnotsmi);
561  {
562  // Check if the {rhs} is a HeapNumber.
563  TNode<HeapObject> rhs_heap_object = CAST(rhs);
564  GotoIfNot(IsHeapNumber(rhs_heap_object), &check_rhsisoddball);
565 
566  // Perform a floating point operation.
567  var_float_lhs = LoadHeapNumberValue(lhs_heap_object);
568  var_float_rhs = LoadHeapNumberValue(rhs_heap_object);
569  Goto(&do_float_operation);
570  }
571 
572  BIND(&if_rhsissmi);
573  }
574 
575  {
576  // Perform floating point operation.
577  var_float_lhs = LoadHeapNumberValue(lhs_heap_object);
578  var_float_rhs = SmiToFloat64(CAST(rhs));
579  Goto(&do_float_operation);
580  }
581  }
582 
583  BIND(&do_float_operation);
584  {
585  var_type_feedback = SmiConstant(BinaryOperationFeedback::kNumber);
586  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(), slot_id,
587  update_feedback_mode);
588  TNode<Float64T> lhs_value = var_float_lhs.value();
589  TNode<Float64T> rhs_value = var_float_rhs.value();
590  TNode<Float64T> value = floatOperation(lhs_value, rhs_value);
591  var_result = AllocateHeapNumberWithValue(value);
592  Goto(&end);
593  }
594 
595  BIND(&if_lhsisnotnumber);
596  {
597  // No checks on rhs are done yet. We just know lhs is not a number or Smi.
598  Label if_left_bigint(this), if_left_oddball(this);
599  TNode<Uint16T> lhs_instance_type = LoadInstanceType(CAST(lhs));
600  GotoIf(IsBigIntInstanceType(lhs_instance_type), &if_left_bigint);
601  TNode<BoolT> lhs_is_oddball =
602  InstanceTypeEqual(lhs_instance_type, ODDBALL_TYPE);
603  Branch(lhs_is_oddball, &if_left_oddball, &call_with_any_feedback);
604 
605  BIND(&if_left_oddball);
606  {
607  Label if_rhsissmi(this), if_rhsisnotsmi(this);
608  Branch(TaggedIsSmi(rhs), &if_rhsissmi, &if_rhsisnotsmi);
609 
610  BIND(&if_rhsissmi);
611  {
612  var_type_feedback =
613  SmiConstant(BinaryOperationFeedback::kNumberOrOddball);
614  Goto(&call_stub);
615  }
616 
617  BIND(&if_rhsisnotsmi);
618  {
619  // Check if {rhs} is a HeapNumber.
620  GotoIfNot(IsHeapNumber(CAST(rhs)), &check_rhsisoddball);
621 
622  var_type_feedback =
623  SmiConstant(BinaryOperationFeedback::kNumberOrOddball);
624  Goto(&call_stub);
625  }
626  }
627 
628  BIND(&if_left_bigint);
629  {
630  GotoIf(TaggedIsSmi(rhs), &call_with_any_feedback);
631  GotoIfNot(IsBigInt(CAST(rhs)), &call_with_any_feedback);
632  if (IsBigInt64OpSupported(this, op)) {
633  GotoIfLargeBigInt(CAST(lhs), &if_both_bigint);
634  GotoIfLargeBigInt(CAST(rhs), &if_both_bigint);
635  Goto(&if_both_bigint64);
636  } else {
637  Goto(&if_both_bigint);
638  }
639  }
640  }
641 
642  BIND(&check_rhsisoddball);
643  {
644  // Check if rhs is an oddball. At this point we know lhs is either a
645  // Smi or number or oddball and rhs is not a number or Smi.
646  TNode<Uint16T> rhs_instance_type = LoadInstanceType(CAST(rhs));
647  TNode<BoolT> rhs_is_oddball =
648  InstanceTypeEqual(rhs_instance_type, ODDBALL_TYPE);
649  GotoIfNot(rhs_is_oddball, &call_with_any_feedback);
650 
651  var_type_feedback = SmiConstant(BinaryOperationFeedback::kNumberOrOddball);
652  Goto(&call_stub);
653  }
654 
655  if (IsBigInt64OpSupported(this, op)) {
656  BIND(&if_both_bigint64);
657  var_type_feedback = SmiConstant(BinaryOperationFeedback::kBigInt64);
658  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(), slot_id,
659  update_feedback_mode);
660 
661  TVARIABLE(UintPtrT, lhs_raw);
662  TVARIABLE(UintPtrT, rhs_raw);
663  BigIntToRawBytes(CAST(lhs), &lhs_raw, &lhs_raw);
664  BigIntToRawBytes(CAST(rhs), &rhs_raw, &rhs_raw);
665 
666  switch (op) {
667  case Operation::kSubtract: {
668  var_result = BigIntFromInt64(TryIntPtrSub(
669  UncheckedCast<IntPtrT>(lhs_raw.value()),
670  UncheckedCast<IntPtrT>(rhs_raw.value()), &if_both_bigint));
671  Goto(&end);
672  break;
673  }
674  case Operation::kMultiply: {
675  var_result = BigIntFromInt64(TryIntPtrMul(
676  UncheckedCast<IntPtrT>(lhs_raw.value()),
677  UncheckedCast<IntPtrT>(rhs_raw.value()), &if_both_bigint));
678  Goto(&end);
679  break;
680  }
681  case Operation::kDivide: {
682  // No need to check overflow because INT_MIN is excluded
683  // from the range of small BigInts.
684  Label if_div_zero(this);
685  var_result = BigIntFromInt64(TryIntPtrDiv(
686  UncheckedCast<IntPtrT>(lhs_raw.value()),
687  UncheckedCast<IntPtrT>(rhs_raw.value()), &if_div_zero));
688  Goto(&end);
689 
690  BIND(&if_div_zero);
691  {
692  // Update feedback to prevent deopt loop.
693  UpdateFeedback(SmiConstant(BinaryOperationFeedback::kAny),
694  maybe_feedback_vector(), slot_id,
695  update_feedback_mode);
696  ThrowRangeError(context(), MessageTemplate::kBigIntDivZero);
697  }
698  break;
699  }
700  case Operation::kModulus: {
701  Label if_div_zero(this);
702  var_result = BigIntFromInt64(TryIntPtrMod(
703  UncheckedCast<IntPtrT>(lhs_raw.value()),
704  UncheckedCast<IntPtrT>(rhs_raw.value()), &if_div_zero));
705  Goto(&end);
706 
707  BIND(&if_div_zero);
708  {
709  // Update feedback to prevent deopt loop.
710  UpdateFeedback(SmiConstant(BinaryOperationFeedback::kAny),
711  maybe_feedback_vector(), slot_id,
712  update_feedback_mode);
713  ThrowRangeError(context(), MessageTemplate::kBigIntDivZero);
714  }
715  break;
716  }
717  default:
718  UNREACHABLE();
719  }
720  }
721 
722  BIND(&if_both_bigint);
723  {
724  var_type_feedback = SmiConstant(BinaryOperationFeedback::kBigInt);
725  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(), slot_id,
726  update_feedback_mode);
727  switch (op) {
728  case Operation::kSubtract: {
729  var_result =
730  CallBuiltin(Builtin::kBigIntSubtractNoThrow, context(), lhs, rhs);
731 
732  // Check for sentinel that signals BigIntTooBig exception.
733  GotoIfNot(TaggedIsSmi(var_result.value()), &end);
734 
735  // Update feedback to prevent deopt loop.
736  UpdateFeedback(SmiConstant(BinaryOperationFeedback::kAny),
737  maybe_feedback_vector(), slot_id, update_feedback_mode);
738  ThrowRangeError(context(), MessageTemplate::kBigIntTooBig);
739  break;
740  }
741  case Operation::kMultiply: {
742  Label termination_requested(this, Label::kDeferred);
743  var_result =
744  CallBuiltin(Builtin::kBigIntMultiplyNoThrow, context(), lhs, rhs);
745 
746  GotoIfNot(TaggedIsSmi(var_result.value()), &end);
747 
748  // Check for sentinel that signals TerminationRequested exception.
749  GotoIf(TaggedEqual(var_result.value(), SmiConstant(1)),
750  &termination_requested);
751 
752  // Handles BigIntTooBig exception.
753  // Update feedback to prevent deopt loop.
754  UpdateFeedback(SmiConstant(BinaryOperationFeedback::kAny),
755  maybe_feedback_vector(), slot_id, update_feedback_mode);
756  ThrowRangeError(context(), MessageTemplate::kBigIntTooBig);
757 
758  BIND(&termination_requested);
759  TerminateExecution(context());
760  break;
761  }
762  case Operation::kDivide: {
763  Label termination_requested(this, Label::kDeferred);
764  var_result =
765  CallBuiltin(Builtin::kBigIntDivideNoThrow, context(), lhs, rhs);
766 
767  GotoIfNot(TaggedIsSmi(var_result.value()), &end);
768 
769  // Check for sentinel that signals TerminationRequested exception.
770  GotoIf(TaggedEqual(var_result.value(), SmiConstant(1)),
771  &termination_requested);
772 
773  // Handles BigIntDivZero exception.
774  // Update feedback to prevent deopt loop.
775  UpdateFeedback(SmiConstant(BinaryOperationFeedback::kAny),
776  maybe_feedback_vector(), slot_id, update_feedback_mode);
777  ThrowRangeError(context(), MessageTemplate::kBigIntDivZero);
778 
779  BIND(&termination_requested);
780  TerminateExecution(context());
781  break;
782  }
783  case Operation::kModulus: {
784  Label termination_requested(this, Label::kDeferred);
785  var_result =
786  CallBuiltin(Builtin::kBigIntModulusNoThrow, context(), lhs, rhs);
787 
788  GotoIfNot(TaggedIsSmi(var_result.value()), &end);
789 
790  // Check for sentinel that signals TerminationRequested exception.
791  GotoIf(TaggedEqual(var_result.value(), SmiConstant(1)),
792  &termination_requested);
793 
794  // Handles BigIntDivZero exception.
795  // Update feedback to prevent deopt loop.
796  UpdateFeedback(SmiConstant(BinaryOperationFeedback::kAny),
797  maybe_feedback_vector(), slot_id, update_feedback_mode);
798  ThrowRangeError(context(), MessageTemplate::kBigIntDivZero);
799 
800  BIND(&termination_requested);
801  TerminateExecution(context());
802  break;
803  }
804  case Operation::kExponentiate: {
805  // TODO(panq): replace the runtime with builtin once it is implemented.
806  var_result =
807  CallRuntime(Runtime::kBigIntExponentiate, context(), lhs, rhs);
808  Goto(&end);
809  break;
810  }
811  default:
812  UNREACHABLE();
813  }
814  }
815 
816  BIND(&call_with_any_feedback);
817  {
818  var_type_feedback = SmiConstant(BinaryOperationFeedback::kAny);
819  Goto(&call_stub);
820  }
821 
822  BIND(&call_stub);
823  {
824  UpdateFeedback(var_type_feedback.value(), maybe_feedback_vector(), slot_id,
825  update_feedback_mode);
826  TNode<Object> result;
827  switch (op) {
828  case Operation::kSubtract:
829  result = CallBuiltin(Builtin::kSubtract, context(), lhs, rhs);
830  break;
831  case Operation::kMultiply:
832  result = CallBuiltin(Builtin::kMultiply, context(), lhs, rhs);
833  break;
834  case Operation::kDivide:
835  result = CallBuiltin(Builtin::kDivide, context(), lhs, rhs);
836  break;
837  case Operation::kModulus:
838  result = CallBuiltin(Builtin::kModulus, context(), lhs, rhs);
839  break;
840  case Operation::kExponentiate:
841  result = CallBuiltin(Builtin::kExponentiate, context(), lhs, rhs);
842  break;
843  default:
844  UNREACHABLE();
845  }
846  var_result = result;
847  Goto(&end);
848  }
849 
850  BIND(&end);
851  return var_result.value();
852 }

989 TNode<Object> BinaryOpAssembler::Generate_BitwiseBinaryOpWithOptionalFeedback(
990  Operation bitwise_op, TNode<Object> left, TNode<Object> right,
991  const LazyNode<Context>& context, TNode<UintPtrT>* slot,
992  const LazyNode<HeapObject>* maybe_feedback_vector,
993  UpdateFeedbackMode update_feedback_mode) {
994  TVARIABLE(Object, result);
995  TVARIABLE(Smi, var_left_feedback);
996  TVARIABLE(Smi, var_right_feedback);
997  TVARIABLE(Word32T, var_left_word32);
998  TVARIABLE(Word32T, var_right_word32);
999  TVARIABLE(BigInt, var_left_bigint);
1000  Label done(this);
1001  Label if_left_number(this), do_number_op(this);
1002  Label if_left_bigint(this), if_left_bigint64(this);
1003  Label if_left_number_right_bigint(this, Label::kDeferred);
1004 
1005  FeedbackValues feedback_values =
1006  slot ? FeedbackValues{&var_left_feedback, maybe_feedback_vector, slot,
1007  update_feedback_mode}
1008  : FeedbackValues();
1009 
1010  TaggedToWord32OrBigIntWithFeedback(
1011  context(), left, &if_left_number, &var_left_word32, &if_left_bigint,
1012  IsBigInt64OpSupported(this, bitwise_op) ? &if_left_bigint64 : nullptr,
1013  &var_left_bigint, feedback_values);
1014 
1015  BIND(&if_left_number);
1016  feedback_values.var_feedback = slot ? &var_right_feedback : nullptr;
1017  TaggedToWord32OrBigIntWithFeedback(
1018  context(), right, &do_number_op, &var_right_word32,
1019  &if_left_number_right_bigint, nullptr, nullptr, feedback_values);
1020 
1021  BIND(&if_left_number_right_bigint);
1022  {
1023  if (slot) {
1024  // Ensure that the feedback is updated before we throw.
1025  UpdateFeedback(SmiConstant(BinaryOperationFeedback::kAny),
1026  (*maybe_feedback_vector)(), *slot, update_feedback_mode);
1027  }
1028  ThrowTypeError(context(), MessageTemplate::kBigIntMixedTypes);
1029  }
1030 
1031  BIND(&do_number_op);
1032  {
1033  result = BitwiseOp(var_left_word32.value(), var_right_word32.value(),
1034  bitwise_op);
1035 
1036  if (slot) {
1037  TNode<Smi> result_type = SelectSmiConstant(
1038  TaggedIsSmi(result.value()), BinaryOperationFeedback::kSignedSmall,
1039  BinaryOperationFeedback::kNumber);
1040  TNode<Smi> input_feedback =
1041  SmiOr(var_left_feedback.value(), var_right_feedback.value());
1042  TNode<Smi> feedback = SmiOr(result_type, input_feedback);
1043  UpdateFeedback(feedback, (*maybe_feedback_vector)(), *slot,
1044  update_feedback_mode);
1045  }
1046  Goto(&done);
1047  }
1048 
1049  // BigInt cases.
1050  {
1051  TVARIABLE(BigInt, var_right_bigint);
1052  Label if_both_bigint(this), if_both_bigint64(this);
1053  Label if_bigint_mix(this, Label::kDeferred);
1054 
1055  BIND(&if_left_bigint);
1056  TaggedToBigInt(context(), right, &if_bigint_mix, &if_both_bigint, nullptr,
1057  &var_right_bigint, slot ? &var_right_feedback : nullptr);
1058 
1059  if (IsBigInt64OpSupported(this, bitwise_op)) {
1060  BIND(&if_left_bigint64);
1061  TaggedToBigInt(context(), right, &if_bigint_mix, &if_both_bigint,
1062  &if_both_bigint64, &var_right_bigint,
1063  slot ? &var_right_feedback : nullptr);
1064 
1065  BIND(&if_both_bigint64);
1066  if (slot) {
1067  // {feedback} is Any if {left} or {right} is non-number.
1068  TNode<Smi> feedback =
1069  SmiOr(var_left_feedback.value(), var_right_feedback.value());
1070  UpdateFeedback(feedback, (*maybe_feedback_vector)(), *slot,
1071  update_feedback_mode);
1072  }
1073 
1074  TVARIABLE(UintPtrT, left_raw);
1075  TVARIABLE(UintPtrT, right_raw);
1076  BigIntToRawBytes(var_left_bigint.value(), &left_raw, &left_raw);
1077  BigIntToRawBytes(var_right_bigint.value(), &right_raw, &right_raw);
1078 
1079  switch (bitwise_op) {
1080  case Operation::kBitwiseAnd: {
1081  result = BigIntFromInt64(UncheckedCast<IntPtrT>(
1082  WordAnd(left_raw.value(), right_raw.value())));
1083  Goto(&done);
1084  break;
1085  }
1086  case Operation::kBitwiseOr: {
1087  result = BigIntFromInt64(UncheckedCast<IntPtrT>(
1088  WordOr(left_raw.value(), right_raw.value())));
1089  Goto(&done);
1090  break;
1091  }
1092  case Operation::kBitwiseXor: {
1093  result = BigIntFromInt64(UncheckedCast<IntPtrT>(
1094  WordXor(left_raw.value(), right_raw.value())));
1095  Goto(&done);
1096  break;
1097  }
1098  default:
1099  UNREACHABLE();
1100  }
1101  }
1102 
1103  BIND(&if_both_bigint);
1104  {
1105  if (slot) {
1106  // Ensure that the feedback is updated even if the runtime call below
1107  // would throw.
1108  TNode<Smi> feedback =
1109  SmiOr(var_left_feedback.value(), var_right_feedback.value());
1110  UpdateFeedback(feedback, (*maybe_feedback_vector)(), *slot,
1111  update_feedback_mode);
