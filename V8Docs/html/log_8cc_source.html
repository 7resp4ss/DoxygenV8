129 const char* ComputeMarker(Tagged<SharedFunctionInfo> shared,
130  Tagged<AbstractCode> code) {
131  PtrComprCageBase cage_base = GetPtrComprCageBase(shared);
132  CodeKind kind = code->kind(cage_base);
133  // We record interpreter trampoline builtin copies as having the
134  // "interpreted" marker.
135  if (v8_flags.interpreted_frames_native_stack && kind == CodeKind::BUILTIN &&
136  code->has_instruction_stream(cage_base) &&
137  code->builtin_id(cage_base) == Builtin::kInterpreterEntryTrampoline) {
138  kind = CodeKind::INTERPRETED_FUNCTION;
139  }
140  if (shared->all_optimization_disabled() &&
141  kind == CodeKind::INTERPRETED_FUNCTION) {
142  return "";
143  }
144  return CodeKindToMarker(
145  kind, IsCode(code) && code->GetCode()->is_context_specialized());
146 }

165  void Reset() { utf8_pos_ = 0; }

167  void Init(CodeTag tag) {
168  Reset();

173  void AppendName(Tagged<Name> name) {
174  if (IsString(name)) {
175  AppendString(Cast<String>(name));
176  } else {
177  Tagged<Symbol> symbol = Cast<Symbol>(name);
178  AppendBytes("symbol(");
179  if (!IsUndefined(symbol->description())) {
180  AppendBytes("\"");
181  AppendString(Cast<String>(symbol->description()));
182  AppendBytes("\" ");
183  }
184  AppendBytes("hash ");
185  AppendHex(symbol->hash());
186  AppendByte(')');
187  }
188  }
189 
190  void AppendString(Tagged<String> str) {
191  if (str.is_null()) return;
192  size_t length = 0;
193  std::unique_ptr<char[]> c_str = str->ToCString(&length);
194  AppendBytes(c_str.get(), length);
195  }
196 
197  void AppendBytes(const char* bytes, size_t size) {
198  size = std::min(size, kUtf8BufferSize - utf8_pos_);
199  MemCopy(utf8_buffer_ + utf8_pos_, bytes, size);
200  utf8_pos_ += size;
201  }
202 
203  void AppendBytes(const char* bytes) {
204  size_t len = strlen(bytes);
205  DCHECK_GE(kMaxInt, len);
206  AppendBytes(bytes, static_cast<int>(len));
207  }
208 
209  void AppendByte(char c) {
210  if (utf8_pos_ >= kUtf8BufferSize) return;
211  utf8_buffer_[utf8_pos_++] = c;
212  }
213 
214  void AppendInt(int n) {
215  if (utf8_pos_ >= kUtf8BufferSize) return;
216  size_t space = kUtf8BufferSize - utf8_pos_;
217  base::Vector<char> buffer(utf8_buffer_ + utf8_pos_, space);
218  int size = SNPrintF(buffer, "%d", n);
219  if (size > 0 && utf8_pos_ + size <= kUtf8BufferSize) {
220  utf8_pos_ += size;
221  }
222  }
223 
224  void AppendHex(uint32_t n) {
225  if (utf8_pos_ >= kUtf8BufferSize) return;
226  size_t space = kUtf8BufferSize - utf8_pos_;
227  base::Vector<char> buffer(utf8_buffer_ + utf8_pos_, space);
228  int size = SNPrintF(buffer, "%x", n);
229  if (size > 0 && utf8_pos_ + size <= kUtf8BufferSize) {
230  utf8_pos_ += size;
231  }
232  }
233 
234  const char* get() { return utf8_buffer_; }
235  size_t size() const { return utf8_pos_; }
236 
237  private:
238  static const size_t kUtf8BufferSize = 4096;
239  static const size_t kUtf16BufferSize = kUtf8BufferSize;
240 
241  size_t utf8_pos_;
242  char utf8_buffer_[kUtf8BufferSize];
243 };

438 void PerfBasicLogger::WriteLogRecordedBuffer(uintptr_t address, size_t size,
439  const char* name,
440  size_t name_length) {
441  // Linux perf expects hex literals without a leading 0x, while some
442  // implementations of printf might prepend one when using the %p format
443  // for pointers, leading to wrongly formatted JIT symbols maps. On the other
444  // hand, Android's simpleperf does expect a leading 0x.
445  //
446  // Instead, we use V8PRIxPTR format string and cast pointer to uintpr_t,
447  // so that we have control over the exact output format.
448  int int_name_length = static_cast<int>(name_length);
449 #ifdef V8_OS_ANDROID
450  base::OS::FPrint(perf_output_handle_, "0x%" V8PRIxPTR " 0x%zx %.*s\n",
451  address, size, int_name_length, name);
452 #else
453  base::OS::FPrint(perf_output_handle_, "%" V8PRIxPTR " %zx %.*s\n", address,
454  size, int_name_length, name);
455 #endif
456 }

1058 // The Profiler samples pc and sp values for the main thread.
1059 // Each sample is appended to a circular buffer.
1060 // An independent thread removes data and writes it to the log.
1061 // This design minimizes the time spent in the sampler.
1062 //
1063 class Profiler : public base::Thread {
1064  public:
1065  explicit Profiler(Isolate* isolate);
1066  void Engage();
1067  void Disengage();
1068 
1069  // Inserts collected profiling data into buffer.
1070  void Insert(TickSample* sample) {
1071  if (Succ(head_) == static_cast<int>(base::Acquire_Load(&tail_))) {
1072  base::Relaxed_Store(&overflow_, true);
1073  } else {
1074  buffer_[head_] = *sample;
1075  head_ = Succ(head_);
1076  buffer_semaphore_.Signal(); // Tell we have an element.
1077  }
1078  }
1079 
1080  void Run() override;
1081 
1082  private:
1083  // Waits for a signal and removes profiling data.
1084  bool Remove(TickSample* sample) {
1085  buffer_semaphore_.Wait(); // Wait for an element.
1086  *sample = buffer_[base::Relaxed_Load(&tail_)];
1087  bool result = base::Relaxed_Load(&overflow_);
1088  base::Release_Store(
1089  &tail_, static_cast<base::Atomic32>(Succ(base::Relaxed_Load(&tail_))));
1090  base::Relaxed_Store(&overflow_, false);
1091  return result;
1092  }
1093 
1094  // Returns the next index in the cyclic buffer.
1095  int Succ(int index) { return (index + 1) % kBufferSize; }
1096 
1097  Isolate* isolate_;
1098  // Cyclic buffer for communicating profiling samples
1099  // between the signal handler and the worker thread.
1100  static const int kBufferSize = 128;
1101  TickSample buffer_[kBufferSize]; // Buffer storage.
1102  int head_; // Index to the buffer head.
1103  base::Atomic32 tail_; // Index to the buffer tail.
1104  base::Atomic32 overflow_; // Tell whether a buffer overflow has occurred.
1105  // Semaphore used for buffer synchronization.
1106  base::Semaphore buffer_semaphore_;
1107 
1108  // Tells whether worker thread should continue running.
1109  base::Atomic32 running_;
1110 };

1384 // We log source code information in the form:
1385 //
1386 // code-source-info <addr>,<script>,<start>,<end>,<pos>,<inline-pos>,<fns>
1387 //
1388 // where
1389 // <addr> is code object address
1390 // <script> is script id
1391 // <start> is the starting position inside the script
1392 // <end> is the end position inside the script
1393 // <pos> is source position table encoded in the string,
1394 // it is a sequence of C<code-offset>O<script-offset>[I<inlining-id>]
1395 // where
1396 // <code-offset> is the offset within the code object
1397 // <script-offset> is the position within the script
1398 // <inlining-id> is the offset in the <inlining> table
1399 // <inlining> table is a sequence of strings of the form
1400 // F<function-id>O<script-offset>[I<inlining-id>]
1401 // where
1402 // <function-id> is an index into the <fns> function table
1403 // <fns> is the function table encoded as a sequence of strings
1404 // S<shared-function-info-address>

1406 void V8FileLogger::LogSourceCodeInformation(
1407  DirectHandle<AbstractCode> code, DirectHandle<SharedFunctionInfo> shared) {
1408  PtrComprCageBase cage_base(isolate_);
1409  Tagged<Object> script_object = shared->script(cage_base);
1410  if (!IsScript(script_object, cage_base)) return;
1411  Tagged<Script> script = Cast<Script>(script_object);
1412  EnsureLogScriptSource(script);
1413 
1414  if (!v8_flags.log_source_position) return;
1415  VMStateIfMainThread<LOGGING> state(isolate_);
1416  MSG_BUILDER();
1417  msg << "code-source-info" << V8FileLogger::kNext
1418  << reinterpret_cast<void*>(code->InstructionStart(cage_base))
1419  << V8FileLogger::kNext << script->id() << V8FileLogger::kNext
1420  << shared->StartPosition() << V8FileLogger::kNext << shared->EndPosition()
1421  << V8FileLogger::kNext;
1422  // TODO(v8:11429): Clean-up baseline-replated code in source position
1423  // iteration.
1424  bool hasInlined = false;
1425  if (code->kind(cage_base) != CodeKind::BASELINE) {
1426  SourcePositionTableIterator iterator(
1427  code->SourcePositionTable(isolate_, *shared));
1428  for (; !iterator.done(); iterator.Advance()) {
1429  SourcePosition pos = iterator.source_position();
1430  msg << "C" << iterator.code_offset() << "O" << pos.ScriptOffset();
1431  if (pos.isInlined()) {
1432  msg << "I" << pos.InliningId();
1433  hasInlined = true;
1434  }
1435  }
1436  }
1437  msg << V8FileLogger::kNext;
1438  int maxInlinedId = -1;
1439  if (hasInlined) {
1440  Tagged<TrustedPodArray<InliningPosition>> inlining_positions =
1441  Cast<DeoptimizationData>(Cast<Code>(code)->deoptimization_data())
1442  ->InliningPositions();
1443  for (int i = 0; i < inlining_positions->length(); i++) {
1444  InliningPosition inlining_pos = inlining_positions->get(i);
1445  msg << "F";
1446  if (inlining_pos.inlined_function_id != -1) {
1447  msg << inlining_pos.inlined_function_id;
1448  if (inlining_pos.inlined_function_id > maxInlinedId) {
1449  maxInlinedId = inlining_pos.inlined_function_id;
1450  }
1451  }
1452  SourcePosition pos = inlining_pos.position;
1453  msg << "O" << pos.ScriptOffset();
1454  if (pos.isInlined()) {
1455  msg << "I" << pos.InliningId();
1456  }
1457  }
1458  }
1459  msg << V8FileLogger::kNext;
1460  if (hasInlined) {
1461  Tagged<DeoptimizationData> deopt_data =
1462  Cast<DeoptimizationData>(Cast<Code>(code)->deoptimization_data());
1463  msg << std::hex;
1464  for (int i = 0; i <= maxInlinedId; i++) {
1465  msg << "S"
1466  << reinterpret_cast<void*>(
1467  deopt_data->GetInlinedFunction(i).address());
1468  }
1469  msg << std::dec;
1470  }
1471  msg.WriteToLogFile();
1472 }

2093  compiled_funcs;

2096  [](const std::pair<Tagged<SharedFunctionInfo>, Tagged<AbstractCode>>& p) {
2097  return base::hash_combine(p.first.address(), p.second.address());
2098  };

2100  std::pair<Tagged<SharedFunctionInfo>, Tagged<AbstractCode>>,
2101  decltype(hash)>
2102  seen(8, hash);
2103

2105  if (auto [iter, inserted] = seen.emplace(sfi, c); inserted)
2106  compiled_funcs.emplace_back(handle(sfi, isolate), handle(c, isolate));
2107  };

2110  for (Tagged<HeapObject> obj = iterator.Next(); !obj.is_null();

2112  if (IsSharedFunctionInfo(obj)) {
2113  Tagged<SharedFunctionInfo> sfi = Cast<SharedFunctionInfo>(obj);

2132  // TODO(jarin) This leaves out deoptimized code that might still be on the

2134  // only on a type feedback vector. We should make this more precise.
