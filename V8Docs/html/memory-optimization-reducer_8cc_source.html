40 void MemoryAnalyzer::Run() {
41  block_states[current_block] = BlockState{};
42  BlockIndex end = BlockIndex(input_graph.block_count());
43  while (current_block < end) {
44  state = *block_states[current_block];
45  auto operations_range =
46  input_graph.operations(input_graph.Get(current_block));
47  // Set the next block index here already, to allow it to be changed if
48  // needed.
49  current_block = BlockIndex(current_block.id() + 1);
50  for (const Operation& op : operations_range) {
51  Process(op);
52  }
53  }
54 }

56 void MemoryAnalyzer::Process(const Operation& op) {
57  if (ShouldSkipOperation(op)) {
58  return;
59  }
60 
61  if (auto* alloc = op.TryCast<AllocateOp>()) {
62  ProcessAllocation(*alloc);
63  return;
64  }
65  if (auto* store = op.TryCast<StoreOp>()) {
66  ProcessStore(*store);
67  return;
68  }
69  if (op.Effects().can_allocate) {
70  state = BlockState();
71  }
72  if (op.IsBlockTerminator()) {
73  ProcessBlockTerminator(op);
74  }
75 }

77 // Update the successor block states based on the state of the current block.
78 // For loop backedges, we need to re-start the analysis from the loop header
79 // unless the backedge state is unchanged.
80 void MemoryAnalyzer::ProcessBlockTerminator(const Operation& terminator) {
81  if (auto* goto_op = terminator.TryCast<GotoOp>()) {
82  if (input_graph.IsLoopBackedge(*goto_op)) {
83  std::optional<BlockState>& target_state =
84  block_states[goto_op->destination->index()];
85  BlockState old_state = *target_state;
86  MergeCurrentStateIntoSuccessor(goto_op->destination);
87  if (old_state != *target_state) {
88  // We can never fold allocations inside of the loop into an
89  // allocation before the loop, since this leads to unbounded
90  // allocation size. An unknown `reserved_size` will prevent adding
91  // allocations inside of the loop.
92  target_state->reserved_size = std::nullopt;
93  // Redo the analysis from the beginning of the loop.
94  current_block = goto_op->destination->index();
95  }
96  return;
97  } else if (goto_op->destination->IsLoop()) {
98  // Look ahead to detect allocating loops earlier, avoiding a wrong
99  // speculation resulting in processing the loop twice.
100  for (const Operation& op :
101  input_graph.operations(*goto_op->destination)) {
102  if (op.Effects().can_allocate && !ShouldSkipOperation(op)) {
103  state = BlockState();
104  break;
105  }
106  }
107  }
108  }
109  for (Block* successor : SuccessorBlocks(terminator)) {
110  MergeCurrentStateIntoSuccessor(successor);
111  }
112 }

114 // We try to merge the new allocation into a previous dominating allocation.
115 // We also allow folding allocations across blocks, as long as there is a
116 // dominating relationship.
117 void MemoryAnalyzer::ProcessAllocation(const AllocateOp& alloc) {
118  if (ShouldSkipOptimizationStep()) return;
119  std::optional<uint64_t> new_size;
120  if (auto* size =
121  input_graph.Get(alloc.size()).template TryCast<ConstantOp>()) {
122  new_size = size->integral();
123  }
124  // If the new allocation has a static size and is of the same type, then we
125  // can fold it into the previous allocation unless the folded allocation would
126  // exceed `kMaxRegularHeapObjectSize`.
127  if (allocation_folding == AllocationFolding::kDoAllocationFolding &&
128  state.last_allocation && new_size.has_value() &&
129  state.reserved_size.has_value() &&
130  alloc.type == state.last_allocation->type &&
131  alloc.type != AllocationType::kSharedOld &&
132  *new_size <= kMaxRegularHeapObjectSize - *state.reserved_size) {
133  state.reserved_size =
134  static_cast<uint32_t>(*state.reserved_size + *new_size);
135  folded_into[&alloc] = state.last_allocation;
136  uint32_t& max_reserved_size = reserved_size[state.last_allocation];
137  max_reserved_size = std::max(max_reserved_size, *state.reserved_size);
138  return;
139  }
140  state.last_allocation = &alloc;
141  state.reserved_size = std::nullopt;
142  if (new_size.has_value() && *new_size <= kMaxRegularHeapObjectSize) {
143  state.reserved_size = static_cast<uint32_t>(*new_size);
144  }
145  // We might be re-visiting the current block. In this case, we need to remove
146  // an allocation that can no longer be folded.
147  reserved_size.erase(&alloc);
148  folded_into.erase(&alloc);
149 }

151 void MemoryAnalyzer::ProcessStore(const StoreOp& store) {
152  V<None> store_op_index = input_graph.Index(store);
153  if (SkipWriteBarrier(store)) {
154  skipped_write_barriers.insert(store_op_index);
155  } else {
156  // We might be re-visiting the current block. In this case, we need to
157  // still update the information.
158  DCHECK_NE(store.write_barrier, WriteBarrierKind::kAssertNoWriteBarrier);
159  skipped_write_barriers.erase(store_op_index);
160  }
161 }

163 void MemoryAnalyzer::MergeCurrentStateIntoSuccessor(const Block* successor) {
164  std::optional<BlockState>& target_state = block_states[successor->index()];
165  if (!target_state.has_value()) {
166  target_state = state;
167  return;
168  }
169  // All predecessors need to have the same last allocation for us to continue
170  // folding into it.
171  if (target_state->last_allocation != state.last_allocation) {
172  target_state = BlockState();
173  return;
174  }
175  // We take the maximum allocation size of all predecessors. If the size is
176  // unknown because it is dynamic, we remember the allocation to eliminate
177  // write barriers.
178  if (target_state->reserved_size.has_value() &&
179  state.reserved_size.has_value()) {
180  target_state->reserved_size =
181  std::max(*target_state->reserved_size, *state.reserved_size);
182  } else {
183  target_state->reserved_size = std::nullopt;
184  }
185 }
