38 void* BoundedPageAllocator::AllocatePages(size_t size, size_t alignment,
39  PageAllocator::Permission access,
40  PageAllocator::AllocationHint hint) {
41  MutexGuard guard(&mutex_);
42  DCHECK(IsAligned(alignment, region_allocator_.page_size()));
43  DCHECK(IsAligned(alignment, allocate_page_size_));
44 
45  Address address = RegionAllocator::kAllocationFailure;
46 
47  Address hint_address = reinterpret_cast<Address>(hint.Address());
48  if (hint_address && IsAligned(hint_address, alignment) &&
49  region_allocator_.contains(hint_address, size)) {
50  if (region_allocator_.AllocateRegionAt(hint_address, size)) {
51  address = hint_address;
52  }
53  }
54 
55  const RegionAllocator::AllocationStrategy allocation_strategy =
56  hint.MayGrow() ? RegionAllocator::AllocationStrategy::kLargestFit
57  : RegionAllocator::AllocationStrategy::kFirstFit;
58 
59  if (address == RegionAllocator::kAllocationFailure) {
60  if (alignment <= allocate_page_size_) {
61  // TODO(ishell): Consider using randomized version here.
62  address = region_allocator_.AllocateRegion(size, allocation_strategy);
63  } else {
64  address = region_allocator_.AllocateAlignedRegion(size, alignment);
65  }
66  }
67 
68  if (address == RegionAllocator::kAllocationFailure) {
69  allocation_status_ = AllocationStatus::kRanOutOfReservation;
70  return nullptr;
71  }
72 
73  void* ptr = reinterpret_cast<void*>(address);
74  // It's assumed that free regions are in kNoAccess/kNoAccessWillJitLater
75  // state.
76  if (access == PageAllocator::kNoAccess ||
77  access == PageAllocator::kNoAccessWillJitLater) {
78  allocation_status_ = AllocationStatus::kSuccess;
79  return ptr;
80  }
81 
82  if (page_initialization_mode_ == PageInitializationMode::kRecommitOnly) {
83  if (page_allocator_->RecommitPages(ptr, size, access)) {
84  allocation_status_ = AllocationStatus::kSuccess;
85  return ptr;
86  }
87  } else {
88  if (page_allocator_->SetPermissions(ptr, size, access)) {
89  allocation_status_ = AllocationStatus::kSuccess;
90  return ptr;
91  }
92  }
93 
94  // This most likely means that we ran out of memory.
95  CHECK_EQ(region_allocator_.FreeRegion(address), size);
96  allocation_status_ = AllocationStatus::kFailedToCommit;
97  return nullptr;
98 }

100 bool BoundedPageAllocator::AllocatePagesAt(Address address, size_t size,
101  PageAllocator::Permission access) {
102  MutexGuard guard(&mutex_);
103 
104  DCHECK(IsAligned(address, allocate_page_size_));
105  DCHECK(IsAligned(size, allocate_page_size_));
106 
107  DCHECK(region_allocator_.contains(address, size));
108 
109  if (!region_allocator_.AllocateRegionAt(address, size)) {
110  allocation_status_ = AllocationStatus::kHintedAddressTakenOrNotFound;
111  return false;
112  }
113 
114  void* ptr = reinterpret_cast<void*>(address);
115  if (!page_allocator_->SetPermissions(ptr, size, access)) {
116  // This most likely means that we ran out of memory.
117  CHECK_EQ(region_allocator_.FreeRegion(address), size);
118  allocation_status_ = AllocationStatus::kFailedToCommit;
119  return false;
120  }
121 
122  allocation_status_ = AllocationStatus::kSuccess;
123  return true;
124 }

126 bool BoundedPageAllocator::ResizeAllocationAt(
127  void* address, size_t old_size, size_t new_size,
128  PageAllocator::Permission access) {
129  MutexGuard guard(&mutex_);
130 
131  const Address address_at = reinterpret_cast<Address>(address);
132  DCHECK(IsAligned(old_size, commit_page_size_));
133  DCHECK(IsAligned(new_size, commit_page_size_));
134 
135  if (new_size < old_size) {
136  // Shrinking is not supported at the moment.
137  return false;
138  } else if (new_size == old_size) {
139  // Nothing to do in this case.
140  return true;
141  }
142 
143  DCHECK_LT(old_size, new_size);
144 
145  const Address allocated_old_size = RoundUp(old_size, allocate_page_size_);
146  const Address allocated_new_size = RoundUp(new_size, allocate_page_size_);
147 
148  if (allocated_old_size < allocated_new_size) {
149  if (!region_allocator_.TryGrowRegion(address_at, allocated_new_size)) {
150  allocation_status_ = AllocationStatus::kHintedAddressTakenOrNotFound;
151  return false;
152  }
153  }
154 
155  if (!page_allocator_->SetPermissions(
156  reinterpret_cast<void*>(address_at + old_size), new_size - old_size,
157  access)) {
158  if (allocated_old_size < allocated_new_size) {
159  // This most likely means that we ran out of memory.
160  CHECK_EQ(region_allocator_.TrimRegion(address_at, allocated_old_size),
161  allocated_new_size - allocated_old_size);
162  }
163 
164  allocation_status_ = AllocationStatus::kFailedToCommit;
165  return false;
166  }
167 
168  allocation_status_ = AllocationStatus::kSuccess;
169  return true;
170 }

172 bool BoundedPageAllocator::ReserveForSharedMemoryMapping(void* ptr,
173  size_t size) {
174  MutexGuard guard(&mutex_);
175 
176  Address address = reinterpret_cast<Address>(ptr);
177  DCHECK(IsAligned(address, allocate_page_size_));
178  DCHECK(IsAligned(size, commit_page_size_));
179 
180  DCHECK(region_allocator_.contains(address, size));
181 
182  // Region allocator requires page size rather than commit size so just over-
183  // allocate there since any extra space couldn't be used anyway.
184  size_t region_size = RoundUp(size, allocate_page_size_);
185  if (!region_allocator_.AllocateRegionAt(
186  address, region_size, RegionAllocator::RegionState::kExcluded)) {
187  allocation_status_ = AllocationStatus::kHintedAddressTakenOrNotFound;
188  return false;
189  }
190 
191  const bool success = page_allocator_->SetPermissions(
192  ptr, size, PageAllocator::Permission::kNoAccess);
193  if (success) {
194  allocation_status_ = AllocationStatus::kSuccess;
195  } else {
196  allocation_status_ = AllocationStatus::kFailedToCommit;
197  }
198  return success;
199 }

201 bool BoundedPageAllocator::FreePages(void* raw_address, size_t size) {
202  // Careful: we are not locked here, do not touch BoundedPageAllocator
203  // metadata.
204  bool success;
205  Address address = reinterpret_cast<Address>(raw_address);
206 
207  // The operations below can be expensive, don't hold the lock while they
208  // happen. There is still potentially contention in the kernel, but at least
209  // we don't need to hold the V8-side lock.
210  if (page_initialization_mode_ ==
211  PageInitializationMode::kAllocatedPagesMustBeZeroInitialized) {
212  DCHECK_NE(page_freeing_mode_, PageFreeingMode::kDiscard);
213  // When we are required to return zero-initialized pages, we decommit the
214  // pages here, which will cause any wired pages to be removed by the OS.
215  success = page_allocator_->DecommitPages(raw_address, size);
216  } else {
217  switch (page_freeing_mode_) {
218  case PageFreeingMode::kMakeInaccessible:
219  DCHECK_EQ(page_initialization_mode_,
220  PageInitializationMode::kAllocatedPagesCanBeUninitialized);
221  success = page_allocator_->SetPermissions(raw_address, size,
222  PageAllocator::kNoAccess);
223  break;
224 
225  case PageFreeingMode::kDiscard:
226  success = page_allocator_->DiscardSystemPages(raw_address, size);
227  break;
228  }
229  }
230 
231  MutexGuard guard(&mutex_);
232  CHECK_EQ(size, region_allocator_.FreeRegion(address));
233 
234  return success;
235 }

237 bool BoundedPageAllocator::ReleasePages(void* raw_address, size_t size,
238  size_t new_size) {
239  Address address = reinterpret_cast<Address>(raw_address);
240  DCHECK(IsAligned(address, allocate_page_size_));
241 
242  DCHECK_LT(new_size, size);
243  DCHECK(IsAligned(size - new_size, commit_page_size_));
244 
245  // This must be held until the page permissions are updated.
246  MutexGuard guard(&mutex_);
247 
248  // Check if we freed any allocatable pages by this release.
249  size_t allocated_size = RoundUp(size, allocate_page_size_);
250  size_t new_allocated_size = RoundUp(new_size, allocate_page_size_);
251 
252 #ifdef DEBUG
253  {
254  // There must be an allocated region at given |address| of a size not
255  // smaller than |size|.
256  DCHECK_EQ(allocated_size, region_allocator_.CheckRegion(address));
257  }
258 #endif
259 
260  if (new_allocated_size < allocated_size) {
261  region_allocator_.TrimRegion(address, new_allocated_size);
262  }
263 
264  // Keep the region in "used" state just uncommit some pages.
265  void* free_address = reinterpret_cast<void*>(address + new_size);
266  size_t free_size = size - new_size;
267  if (page_initialization_mode_ ==
268  PageInitializationMode::kAllocatedPagesMustBeZeroInitialized) {
269  DCHECK_NE(page_freeing_mode_, PageFreeingMode::kDiscard);
270  // See comment in FreePages().
271  return (page_allocator_->DecommitPages(free_address, free_size));
272  }
273  if (page_freeing_mode_ == PageFreeingMode::kMakeInaccessible) {
274  DCHECK_EQ(page_initialization_mode_,
275  PageInitializationMode::kAllocatedPagesCanBeUninitialized);
276  return page_allocator_->SetPermissions(free_address, free_size,
277  PageAllocator::kNoAccess);
278  }
279  CHECK_EQ(page_freeing_mode_, PageFreeingMode::kDiscard);
280  return page_allocator_->DiscardSystemPages(free_address, free_size);
281 }
