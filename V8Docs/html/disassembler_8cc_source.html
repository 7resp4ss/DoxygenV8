68 void V8NameConverter::InitExternalRefsCache() const {
69  ExternalReferenceTable* external_reference_table =
70  isolate_->external_reference_table();
71  if (!external_reference_table->is_initialized()) return;
72 
73  base::AddressRegion addressable_region =
74  isolate_->root_register_addressable_region();
75  Address isolate_root = isolate_->isolate_root();
76 
77  for (uint32_t i = 0; i < ExternalReferenceTable::kSize; i++) {
78  Address address = external_reference_table->address(i);
79  if (addressable_region.contains(address)) {
80  int offset = static_cast<int>(address - isolate_root);
81  const char* name = external_reference_table->name(i);
82  directly_accessed_external_refs_.insert({offset, name});
83  }
84  }
85 }

87 const char* V8NameConverter::NameOfAddress(uint8_t* pc) const {
88  if (!code_.is_null()) {
89  const char* name =
90  isolate_ ? isolate_->builtins()->Lookup(reinterpret_cast<Address>(pc))
91  : nullptr;
92 
93  if (name != nullptr) {
94  SNPrintF(v8_buffer_, "%p (%s)", static_cast<void*>(pc), name);
95  return v8_buffer_.begin();
96  }
97 
98  int offs = static_cast<int>(reinterpret_cast<Address>(pc) -
99  code_.instruction_start());
100  // print as code offset, if it seems reasonable
101  if (0 <= offs && offs < code_.instruction_size()) {
102  SNPrintF(v8_buffer_, "%p <+0x%x>", static_cast<void*>(pc), offs);
103  return v8_buffer_.begin();
104  }
105 
106 #if V8_ENABLE_WEBASSEMBLY
107  if (auto* wasm_code = wasm::GetWasmCodeManager()->LookupCode(
108  isolate_, reinterpret_cast<Address>(pc))) {
109  SNPrintF(v8_buffer_, "%p (%s)", static_cast<void*>(pc),
110  wasm::GetWasmCodeKindAsString(wasm_code->kind()));
111  return v8_buffer_.begin();
112  }
113 #endif // V8_ENABLE_WEBASSEMBLY
114  }
115 
116  return disasm::NameConverter::NameOfAddress(pc);
117 }

125 const char* V8NameConverter::RootRelativeName(int offset) const {
126  if (isolate_ == nullptr) return nullptr;
127 
128  const int kRootsTableStart = IsolateData::roots_table_offset();
129  const unsigned kRootsTableSize = sizeof(RootsTable);
130  const int kExtRefsTableStart = IsolateData::external_reference_table_offset();
131  const unsigned kExtRefsTableSize = ExternalReferenceTable::kSizeInBytes;
132  const int kBuiltinTier0TableStart = IsolateData::builtin_tier0_table_offset();
133  const unsigned kBuiltinTier0TableSize =
134  Builtins::kBuiltinTier0Count * kSystemPointerSize;
135  const int kBuiltinTableStart = IsolateData::builtin_table_offset();
136  const unsigned kBuiltinTableSize =
137  Builtins::kBuiltinCount * kSystemPointerSize;
138 
139  if (static_cast<unsigned>(offset - kRootsTableStart) < kRootsTableSize) {
140  uint32_t offset_in_roots_table = offset - kRootsTableStart;
141 
142  // Fail safe in the unlikely case of an arbitrary root-relative offset.
143  if (offset_in_roots_table % kSystemPointerSize != 0) return nullptr;
144 
145  RootIndex root_index =
146  static_cast<RootIndex>(offset_in_roots_table / kSystemPointerSize);
147 
148  SNPrintF(v8_buffer_, "root (%s)", RootsTable::name(root_index));
149  return v8_buffer_.begin();
150  } else if (static_cast<unsigned>(offset - kExtRefsTableStart) <
151  kExtRefsTableSize) {
152  uint32_t offset_in_extref_table = offset - kExtRefsTableStart;
153 
154  // Fail safe in the unlikely case of an arbitrary root-relative offset.
155  if (offset_in_extref_table % ExternalReferenceTable::kEntrySize != 0) {
156  return nullptr;
157  }
158 
159  // Likewise if the external reference table is uninitialized.
160  if (!isolate_->external_reference_table()->is_initialized()) {
161  return nullptr;
162  }
163 
164  SNPrintF(v8_buffer_, "external reference (%s)",
165  isolate_->external_reference_table()->NameFromOffset(
166  offset_in_extref_table));
167  return v8_buffer_.begin();
168  } else if (static_cast<unsigned>(offset - kBuiltinTier0TableStart) <
169  kBuiltinTier0TableSize) {
170  uint32_t offset_in_builtins_table = (offset - kBuiltinTier0TableStart);
171 
172  Builtin builtin =
173  Builtins::FromInt(offset_in_builtins_table / kSystemPointerSize);
174  const char* name = Builtins::name(builtin);
175  SNPrintF(v8_buffer_, "builtin (%s)", name);
176  return v8_buffer_.begin();
177  } else if (static_cast<unsigned>(offset - kBuiltinTableStart) <
178  kBuiltinTableSize) {
179  uint32_t offset_in_builtins_table = (offset - kBuiltinTableStart);
180 
181  Builtin builtin =
182  Builtins::FromInt(offset_in_builtins_table / kSystemPointerSize);
183  const char* name = Builtins::name(builtin);
184  SNPrintF(v8_buffer_, "builtin (%s)", name);
185  return v8_buffer_.begin();
186  } else {
187  // It must be a direct access to one of the external values.
188  if (directly_accessed_external_refs_.empty()) {
189  InitExternalRefsCache();
190  }
191 
192  auto iter = directly_accessed_external_refs_.find(offset);
193  if (iter != directly_accessed_external_refs_.end()) {
194  SNPrintF(v8_buffer_, "external value (%s)", iter->second);
195  return v8_buffer_.begin();
196  }
197  return nullptr;
198  }
199 }

209 static void PrintRelocInfo(std::ostringstream& out, Isolate* isolate,
210  const ExternalReferenceEncoder* ref_encoder,
211  std::ostream& os, CodeReference host,
212  RelocInfo* relocinfo, bool first_reloc_info = true) {
213  // Indent the printing of the reloc info.
214  int padding = kRelocInfoPosition;
215  if (first_reloc_info) {
216  // The first reloc info is printed after the disassembled instruction.
217  padding -= std::min(padding, static_cast<int>(out.tellp()));
218  } else {
219  // Additional reloc infos are printed on separate lines.
220  DumpBuffer(os, out);
221  }
222  std::fill_n(std::ostream_iterator<char>(out), padding, ' ');
223 
224  RelocInfo::Mode rmode = relocinfo->rmode();
225  if (rmode == RelocInfo::DEOPT_SCRIPT_OFFSET) {
226  out << " ;; debug: deopt position, script offset '"
227  << static_cast<int>(relocinfo->data()) << "'";
228  } else if (rmode == RelocInfo::DEOPT_INLINING_ID) {
229  out << " ;; debug: deopt position, inlining id '"
230  << static_cast<int>(relocinfo->data()) << "'";
231  } else if (rmode == RelocInfo::DEOPT_REASON) {
232  DeoptimizeReason reason = static_cast<DeoptimizeReason>(relocinfo->data());
233  out << " ;; debug: deopt reason '" << DeoptimizeReasonToString(reason)
234  << "'";
235  } else if (rmode == RelocInfo::DEOPT_ID) {
236  out << " ;; debug: deopt index " << static_cast<int>(relocinfo->data());
237  } else if (rmode == RelocInfo::DEOPT_NODE_ID) {
238 #ifdef DEBUG
239  out << " ;; debug: deopt node id "
240  << static_cast<uint32_t>(relocinfo->data());
241 #else // DEBUG
242  UNREACHABLE();
243 #endif // DEBUG
244  } else if (RelocInfo::IsEmbeddedObjectMode(rmode)) {
245  HeapStringAllocator allocator;
246  StringStream accumulator(&allocator);
247  ShortPrint(relocinfo->target_object(isolate), &accumulator);
248  std::unique_ptr<char[]> obj_name = accumulator.ToCString();
249  const bool is_compressed = RelocInfo::IsCompressedEmbeddedObject(rmode);
250  out << " ;; " << (is_compressed ? "(compressed) " : "")
251  << "object: " << obj_name.get();
252  } else if (rmode == RelocInfo::EXTERNAL_REFERENCE) {
253  Address address = relocinfo->target_external_reference();
254  const char* reference_name =
255  ref_encoder
256  ? ref_encoder->NameOfAddress(isolate, address)
257  : ExternalReferenceTable::NameOfIsolateIndependentAddress(
258  address, IsolateGroup::current()->external_ref_table());
259  out << " ;; external reference (" << reference_name << ")";
260  } else if (rmode == RelocInfo::JS_DISPATCH_HANDLE) {
261 #ifdef V8_ENABLE_LEAPTIERING
262  out << " ;; js dispatch handle:0x" << std::hex
263  << relocinfo->js_dispatch_handle();
264  Tagged<Code> code = IsolateGroup::current()->js_dispatch_table()->GetCode(
265  relocinfo->js_dispatch_handle());
266  CodeKind kind = code->kind();
267  if (code->is_builtin()) {
268  out << " Builtin::" << Builtins::name(code->builtin_id());
269  } else {
270  out << " " << CodeKindToString(kind);
271  }
272 #else
273  UNREACHABLE();
274 #endif
275  } else if (RelocInfo::IsCodeTargetMode(rmode)) {
276  out << " ;; code:";
277  Tagged<Code> code =
278  isolate->heap()->FindCodeForInnerPointer(relocinfo->target_address());
279  CodeKind kind = code->kind();
280  if (code->is_builtin()) {
281  out << " Builtin::" << Builtins::name(code->builtin_id());
282  } else {
283  out << " " << CodeKindToString(kind);
284  }
285 #if V8_ENABLE_WEBASSEMBLY
286  } else if (RelocInfo::IsWasmStubCall(rmode) && host.is_wasm_code()) {
287  // Host is isolate-independent, try wasm native module instead.
288  const char* runtime_stub_name = Builtins::name(
289  host.as_wasm_code()->native_module()->GetBuiltinInJumptableSlot(
290  relocinfo->wasm_stub_call_address()));
291  out << " ;; wasm stub: " << runtime_stub_name;
292 #endif // V8_ENABLE_WEBASSEMBLY
293  } else {
294  out << " ;; " << RelocInfo::RelocModeName(rmode);
295  }
296 }

298 static int DecodeIt(Isolate* isolate, ExternalReferenceEncoder* ref_encoder,
299  std::ostream& os, CodeReference code,
300  const V8NameConverter& converter, uint8_t* begin,
301  uint8_t* end, Address current_pc, size_t range_limit) {
302  CHECK(!code.is_null());
303  v8::base::EmbeddedVector<char, 128> decode_buffer;
304  std::ostringstream out;
305  uint8_t* pc = begin;
306  disasm::Disassembler d(converter,
307  disasm::Disassembler::kContinueOnUnimplementedOpcode);
308  RelocIterator rit(code);
309  CodeCommentsIterator cit(code.code_comments(), code.code_comments_size());
310 
311  std::unique_ptr<JumpTableInfoIterator> table_info_it = nullptr;
312  if constexpr (V8_JUMP_TABLE_INFO_BOOL) {
313  if (code.is_code() && code.as_code()->has_jump_table_info()) {
314  table_info_it = std::make_unique<JumpTableInfoIterator>(
315  code.as_code()->jump_table_info(),
316  code.as_code()->jump_table_info_size());
317 #if V8_ENABLE_WEBASSEMBLY
318  } else if (code.is_wasm_code() &&
319  code.as_wasm_code()->has_jump_table_info()) {
320  table_info_it = std::make_unique<JumpTableInfoIterator>(
321  code.as_wasm_code()->jump_table_info(),
322  code.as_wasm_code()->jump_table_info_size());
323 #endif // V8_ENABLE_WEBASSEMBLY
324  }
325  }
326 
327  int constants = -1; // no constants being decoded at the start
328 
329  while (pc < end) {
330  // First decode instruction so that we know its length.
331  uint8_t* prev_pc = pc;
332  bool decoding_constant_pool = constants > 0;
333  if (decoding_constant_pool) {
334  SNPrintF(
335  decode_buffer, "%08x constant",
336  base::ReadUnalignedValue<int32_t>(reinterpret_cast<Address>(pc)));
337  constants--;
338  pc += 4;
339  } else {
340  int num_const = d.ConstantPoolSizeAt(pc);
341  if (num_const >= 0) {
342  SNPrintF(
343  decode_buffer, "%08x constant pool begin (num_const = %d)",
344  base::ReadUnalignedValue<int32_t>(reinterpret_cast<Address>(pc)),
345  num_const);
346  constants = num_const;
347  pc += 4;
348  } else if (!rit.done() &&
349  rit.rinfo()->pc() == reinterpret_cast<Address>(pc) &&
350  rit.rinfo()->rmode() == RelocInfo::INTERNAL_REFERENCE) {
351  // A raw pointer embedded in code stream.
352  uint8_t* ptr =
353  base::ReadUnalignedValue<uint8_t*>(reinterpret_cast<Address>(pc));
354  SNPrintF(decode_buffer, "%08" V8PRIxPTR " jump table entry %4zu",
355  reinterpret_cast<intptr_t>(ptr),
356  static_cast<size_t>(ptr - begin));
357  pc += sizeof(ptr);
358  } else if (table_info_it && table_info_it->HasCurrent() &&
359  table_info_it->GetPCOffset() ==
360  static_cast<uint32_t>(pc - begin)) {
361  int32_t target_pc_offset = table_info_it->GetTarget();
362  static_assert(sizeof(target_pc_offset) ==
363  JumpTableInfoEntry::kTargetSize);
364  SNPrintF(decode_buffer, "jump table entry %08x", target_pc_offset);
365  pc += JumpTableInfoEntry::kTargetSize;
366  table_info_it->Next();
367  } else {
368  decode_buffer[0] = '\0';
369  pc += d.InstructionDecode(decode_buffer, pc);
370  }
371  }
372 
373  Address pc_address = reinterpret_cast<Address>(pc);
374  if (range_limit != 0) {
375  if (pc_address > current_pc + range_limit) break;
376  if (pc_address <= current_pc - range_limit) continue;
377  }
378 
379  // Collect RelocInfo for this instruction (prev_pc .. pc-1)
380  std::vector<const char*> comments;
381  std::vector<Address> pcs;
382  std::vector<RelocInfo::Mode> rmodes;
383  std::vector<intptr_t> datas;
384  while (!rit.done() && rit.rinfo()->pc() < reinterpret_cast<Address>(pc)) {
385  // Collect all data.
386  pcs.push_back(rit.rinfo()->pc());
387  rmodes.push_back(rit.rinfo()->rmode());
388  datas.push_back(rit.rinfo()->data());
389  rit.next();
390  }
391  while (cit.HasCurrent()) {
392  Address cur = cit.GetPCOffset();
393  if (cur >= static_cast<Address>(pc - begin)) break;
394  if (range_limit == 0 ||
395  cur + range_limit > current_pc - reinterpret_cast<Address>(begin)) {
396  comments.push_back(cit.GetComment());
397  }
398  cit.Next();
399  }
400 
401  // Comments.
402  for (size_t i = 0; i < comments.size(); i++) {
403  if (v8_flags.log_colour) {
404  out << "\033[34m";
405  }
406  out << " " << comments[i];
407  if (v8_flags.log_colour) {
408  out << "\033[;m";
409  }
410  DumpBuffer(os, out);
411  }
412 
413  // Instruction address and instruction offset.
414  if (v8_flags.log_colour &&
415  reinterpret_cast<Address>(prev_pc) == current_pc) {
416  // If this is the given "current" pc, make it yellow and bold.
417  out << "\033[33;1m";
418  }
419  out << static_cast<void*>(prev_pc) << " " << std::setw(4) << std::hex
420  << prev_pc - begin << " ";
421 
422  // Instruction.
423  out << decode_buffer.begin();
424 
425  // Print all the reloc info for this instruction which are not comments.
426  for (size_t i = 0; i < pcs.size(); i++) {
427  // Put together the reloc info.
428  const CodeReference& host = code;
429  Address constant_pool =
430  host.is_null() ? kNullAddress : host.constant_pool();
431  if (host.is_code()) {
432  RelocInfo relocinfo(pcs[i], rmodes[i], datas[i], constant_pool);
433  bool first_reloc_info = (i == 0);
434  PrintRelocInfo(out, isolate, ref_encoder, os, code, &relocinfo,
435  first_reloc_info);
436  }
437  }
438 
439  // If this is a constant pool load and we haven't found any RelocInfo
440  // already, check if we can find some RelocInfo for the target address in
441  // the constant pool.
442  // Make sure we're also not currently in the middle of decoding a constant
443  // pool itself, rather than a contant pool load. Since it can store any
444  // bytes, a constant could accidentally match with the bit-pattern checked
445  // by IsInConstantPool() below.
446  if (pcs.empty() && !code.is_null() && !decoding_constant_pool) {
447  RelocInfo dummy_rinfo(reinterpret_cast<Address>(prev_pc),
448  RelocInfo::NO_INFO);
449  if (dummy_rinfo.IsInConstantPool()) {
450  Address constant_pool_entry_address =
451  dummy_rinfo.constant_pool_entry_address();
452  RelocIterator reloc_it(code);
453  while (!reloc_it.done()) {
454  if (reloc_it.rinfo()->IsInConstantPool() &&
455  (reloc_it.rinfo()->constant_pool_entry_address() ==
456  constant_pool_entry_address)) {
457  PrintRelocInfo(out, isolate, ref_encoder, os, code,
458  reloc_it.rinfo());
459  break;
460  }
461  reloc_it.next();
462  }
463  }
464  }
465 
466  if (v8_flags.log_colour &&
467  reinterpret_cast<Address>(prev_pc) == current_pc) {
468  out << "\033[m";
469  }
470 
471  DumpBuffer(os, out);
472  }
473 
474  // Emit comments following the last instruction (if any).
475  while (cit.HasCurrent()) {
476  Address cur = cit.GetPCOffset();
477  if (range_limit == 0 ||
478  cur + range_limit == current_pc - reinterpret_cast<Address>(begin)) {
479  out << " " << cit.GetComment();
480  DumpBuffer(os, out);
481  }
482  cit.Next();
483  }
484 
485  return static_cast<int>(pc - begin);
486 }

488 int Disassembler::Decode(Isolate* isolate, std::ostream& os, uint8_t* begin,
489  uint8_t* end, CodeReference code, Address current_pc,
490  size_t range_limit) {
491  DCHECK_WITH_MSG(v8_flags.text_is_readable,
492  "Builtins disassembly requires a readable .text section");
493  V8NameConverter v8NameConverter(isolate, code);
494  if (isolate) {
495  // We have an isolate, so support external reference names from V8 and
496  // embedder.
497  SealHandleScope shs(isolate);
498  DisallowGarbageCollection no_alloc;
499  ExternalReferenceEncoder ref_encoder(isolate);
500  return DecodeIt(isolate, &ref_encoder, os, code, v8NameConverter, begin,
501  end, current_pc, range_limit);
502  } else {
503  // No isolate => isolate-independent code. Only V8 External references
504  // available.
505  return DecodeIt(nullptr, nullptr, os, code, v8NameConverter, begin, end,
506  current_pc, range_limit);
507  }
508 }
