25 DirectHandle<Object> NormalizeReceiver(Isolate* isolate,
26  DirectHandle<Object> receiver) {
27  // Convert calls on global objects to be calls on the global
28  // receiver instead to avoid having a 'this' pointer which refers
29  // directly to a global object.
30  if (IsJSGlobalObject(*receiver)) {
31  return direct_handle(Cast<JSGlobalObject>(receiver)->global_proxy(),
32  isolate);
33  }
34  return receiver;
35 }

166 DirectHandle<Code> JSEntry(Isolate* isolate, Execution::Target execution_target,
167  bool is_construct) {
168  if (is_construct) {
169  DCHECK_EQ(Execution::Target::kCallable, execution_target);
170  return BUILTIN_CODE(isolate, JSConstructEntry);
171  } else if (execution_target == Execution::Target::kCallable) {
172  DCHECK(!is_construct);

175  DCHECK(!is_construct);

181 MaybeDirectHandle<Context> NewScriptContext(
182  Isolate* isolate, DirectHandle<JSFunction> function,
183  DirectHandle<FixedArray> host_defined_options) {
184  // TODO(cbruni, 1244145): Use passed in host_defined_options.
185  // Creating a script context is a side effect, so abort if that's not
186  // allowed.
187  if (isolate->should_check_side_effects()) {
188  isolate->Throw(*isolate->factory()->NewEvalError(
189  MessageTemplate::kNoSideEffectDebugEvaluate));
190  return MaybeDirectHandle<Context>();
191  }
192  SaveAndSwitchContext save(isolate, function->context());
193  Tagged<SharedFunctionInfo> sfi = function->shared();
194  Handle<Script> script(Cast<Script>(sfi->script()), isolate);
195  DirectHandle<ScopeInfo> scope_info(sfi->scope_info(), isolate);
196  // Make sure this is the first time we're running this function.
197  CHECK(IsNativeContext(function->context()));
198  DirectHandle<NativeContext> native_context(
199  Cast<NativeContext>(function->context()), isolate);
200  DirectHandle<JSGlobalObject> global_object(native_context->global_object(),
201  isolate);
202  Handle<ScriptContextTable> script_context(
203  native_context->script_context_table(), isolate);
204 
205  // Find name clashes.
206  for (auto name_it : ScopeInfo::IterateLocalNames(scope_info)) {
207  Handle<String> name(name_it->name(), isolate);
208  VariableMode mode = scope_info->ContextLocalMode(name_it->index());
209  VariableLookupResult lookup;
210  if (script_context->Lookup(name, &lookup)) {
211  if (IsLexicalVariableMode(mode) || IsLexicalVariableMode(lookup.mode)) {
212  DirectHandle<Context> context(script_context->get(lookup.context_index),
213  isolate);
214  // If we are trying to redeclare a REPL-mode let as a let, REPL-mode
215  // const as a const, REPL-mode using as a using and REPL-mode await
216  // using as an await using allow it.
217  if (!((mode == lookup.mode && IsLexicalVariableMode(mode)) &&
218  scope_info->IsReplModeScope() &&
219  context->scope_info()->IsReplModeScope())) {
220  // ES#sec-globaldeclarationinstantiation 5.b:
221  // If envRec.HasLexicalDeclaration(name) is true, throw a SyntaxError
222  // exception.
223  MessageLocation location(script, 0, 1);
224  isolate->ThrowAt(isolate->factory()->NewSyntaxError(
225  MessageTemplate::kVarRedeclaration, name),
226  &location);
227  return MaybeDirectHandle<Context>();
228  }
229  }
230  }
231 
232  if (IsLexicalVariableMode(mode)) {
233  LookupIterator lookup_it(isolate, global_object, name, global_object,
234  LookupIterator::OWN_SKIP_INTERCEPTOR);
235  Maybe<PropertyAttributes> maybe =
236  JSReceiver::GetPropertyAttributes(&lookup_it);
237  // Can't fail since the we looking up own properties on the global object
238  // skipping interceptors.
239  CHECK(!maybe.IsNothing());
240  if ((maybe.FromJust() & DONT_DELETE) != 0) {
241  // ES#sec-globaldeclarationinstantiation 5.a:
242  // If envRec.HasVarDeclaration(name) is true, throw a SyntaxError
243  // exception.
244  // ES#sec-globaldeclarationinstantiation 5.d:
245  // If hasRestrictedGlobal is true, throw a SyntaxError exception.
246  MessageLocation location(script, 0, 1);
247  isolate->ThrowAt(isolate->factory()->NewSyntaxError(
248  MessageTemplate::kVarRedeclaration, name),
249  &location);
250  return MaybeDirectHandle<Context>();
251  }
252 
253  JSGlobalObject::InvalidatePropertyCell(global_object, name);
254  }
255  }
256 
257  DirectHandle<Context> result =
258  isolate->factory()->NewScriptContext(native_context, scope_info);
259 
260  result->Initialize(isolate);
261  // In REPL mode, we are allowed to add/modify let/const/using/await using
262  // variables. We use the previous defined script context for those.
263  const bool ignore_duplicates = scope_info->IsReplModeScope();
264  DirectHandle<ScriptContextTable> new_script_context_table =
265  ScriptContextTable::Add(isolate, script_context, result,
266  ignore_duplicates);
267  native_context->synchronized_set_script_context_table(
268  *new_script_context_table);
269  return result;
270 }

272 V8_WARN_UNUSED_RESULT MaybeHandle<Object> Invoke(Isolate* isolate,
273  const InvokeParams& params) {
274  RCS_SCOPE(isolate, RuntimeCallCounterId::kInvoke);
275  DCHECK(!IsJSGlobalObject(*params.receiver));
276  DCHECK_LE(params.args.size(), FixedArray::kMaxLength);
277  DCHECK(!isolate->has_exception());
278  // Runtime code must be able to get the "current" isolate from TLS, and this
279  // must equal the isolate we execute in.
280  DCHECK_EQ(isolate, Isolate::TryGetCurrent());
281 
282 #if V8_ENABLE_WEBASSEMBLY
283  // If we have PKU support for Wasm, ensure that code is currently write
284  // protected for this thread.
285  DCHECK_IMPLIES(wasm::GetWasmCodeManager()->HasMemoryProtectionKeySupport(),
286  !wasm::GetWasmCodeManager()->MemoryProtectionKeyWritable());
287 #endif // V8_ENABLE_WEBASSEMBLY
288 
289 #ifdef USE_SIMULATOR
290  // Simulators use separate stacks for C++ and JS. JS stack overflow checks
291  // are performed whenever a JS function is called. However, it can be the case
292  // that the C++ stack grows faster than the JS stack, resulting in an overflow
293  // there. Add a check here to make that less likely.
294  StackLimitCheck check(isolate);
295  if (check.HasOverflowed()) {
296  isolate->StackOverflow();
297  isolate->ReportPendingMessages(params.message_handling ==
298  Execution::MessageHandling::kReport);
299  return MaybeHandle<Object>();
300  }
301 #endif
302 
303  // api callbacks can be called directly, unless we want to take the detour
304  // through JS to set up a frame for break-at-entry.
305  if (IsJSFunction(*params.target)) {
306  auto function = Cast<JSFunction>(params.target);
307  if ((!params.is_construct || IsConstructor(*function)) &&
308  function->shared()->IsApiFunction() &&
309  !function->shared()->BreakAtEntry(isolate)) {
310  SaveAndSwitchContext save(isolate, function->context());
311  DCHECK(IsJSGlobalObject(function->context()->global_object()));
312 
313  DirectHandle<Object> receiver = params.is_construct
314  ? isolate->factory()->the_hole_value()
315  : params.receiver;
316  DirectHandle<FunctionTemplateInfo> fun_data(
317  function->shared()->api_func_data(), isolate);
318  auto value = Builtins::InvokeApiFunction(
319  isolate, params.is_construct, fun_data, receiver, params.args,
320  Cast<HeapObject>(params.new_target));
321  bool has_exception = value.is_null();
322  DCHECK_EQ(has_exception, isolate->has_exception());
323  if (has_exception) {
324  isolate->ReportPendingMessages(params.message_handling ==
325  Execution::MessageHandling::kReport);
326  return MaybeHandle<Object>();
327  } else {
328  isolate->clear_pending_message();
329  }
330  return value;
331  }
332 #ifdef DEBUG
333  if (function->shared()->is_script()) {
334  DCHECK(params.IsScript());
335  DCHECK(IsJSGlobalProxy(*params.receiver));
336  DCHECK_EQ(params.args.size(), 1);
337  DCHECK(IsFixedArray(*params.args[0]));
338  } else {
339  DCHECK(!params.IsScript());
340  }
341 #endif
342  // Set up a ScriptContext when running scripts that need it.
343  if (function->shared()->needs_script_context()) {
344  DirectHandle<Context> context;
345  DirectHandle<FixedArray> host_defined_options =
346  const_cast<InvokeParams&>(params).GetAndResetHostDefinedOptions();
347  if (!NewScriptContext(isolate, function, host_defined_options)
348  .ToHandle(&context)) {
349  isolate->ReportPendingMessages(params.message_handling ==
350  Execution::MessageHandling::kReport);
351  return MaybeHandle<Object>();
352  }
353 
354  // We mutate the context if we allocate a script context. This is
355  // guaranteed to only happen once in a native context since scripts will
356  // always produce name clashes with themselves.
357  function->set_context(*context);
358  }
359  }
360 
361  // Entering JavaScript.
362  VMState<JS> state(isolate);
363  if (!AllowJavascriptExecution::IsAllowed(isolate)) {
364  GRACEFUL_FATAL("Invoke in DisallowJavascriptExecutionScope");
365  }
366  if (!ThrowOnJavascriptExecution::IsAllowed(isolate)) {
367  isolate->ThrowIllegalOperation();
368  isolate->ReportPendingMessages(params.message_handling ==
369  Execution::MessageHandling::kReport);
370  return MaybeHandle<Object>();
371  }
372  if (!DumpOnJavascriptExecution::IsAllowed(isolate)) {
373  V8::GetCurrentPlatform()->DumpWithoutCrashing();
374  return isolate->factory()->undefined_value();
375  }
376  isolate->IncrementJavascriptExecutionCounter();
377 
378  if (params.execution_target == Execution::Target::kCallable) {
379  DirectHandle<NativeContext> context = isolate->native_context();
380  if (!IsUndefined(context->script_execution_callback(), isolate)) {
381  v8::Context::AbortScriptExecutionCallback callback =
382  v8::ToCData<v8::Context::AbortScriptExecutionCallback,
383  kApiAbortScriptExecutionCallbackTag>(
384  isolate, context->script_execution_callback());
385  v8::Isolate* api_isolate = reinterpret_cast<v8::Isolate*>(isolate);
386  v8::Local<v8::Context> api_context = v8::Utils::ToLocal(context);
387  callback(api_isolate, api_context);
388  DCHECK(!isolate->has_exception());
389  // Always throw an exception to abort execution, if callback exists.
390  isolate->ThrowIllegalOperation();
391  return MaybeHandle<Object>();
392  }
393  }
394 
395  // Placeholder for return value.
396  Tagged<Object> value;
397  DirectHandle<Code> code =
398  JSEntry(isolate, params.execution_target, params.is_construct);
399  {
400  // Save and restore context around invocation.
401  SaveContext save(isolate);
402 
403  if (v8_flags.clear_exceptions_on_js_entry) isolate->clear_exception();
404 
405  if (params.execution_target == Execution::Target::kCallable) {
406  // clang-format off
407  // {new_target}, {target}, {receiver}, return value: tagged pointers
408  // {argv}: pointer to array of tagged pointers
409  using JSEntryFunction = GeneratedCode<Address(
410  Address root_register_value, Address new_target, Address target,
411  Address receiver, intptr_t argc, Address** argv)>;
412  // clang-format on
413  JSEntryFunction stub_entry =
414  JSEntryFunction::FromAddress(isolate, code->instruction_start());
415 
416  Address orig_func = (*params.new_target).ptr();
417  Address func = (*params.target).ptr();
418  Address recv = (*params.receiver).ptr();
419 
420  int argc = static_cast<int>(params.args.size());
421 #ifdef V8_ENABLE_DIRECT_HANDLE
422  // TODO(42203211): Store the arguments to indirect handles because
423  // generated code still expects them in indirect handles. A fresh handle
424  // scope is introduced for these handles, which is then sealed. When this
425  // is eventually removed, sealing can go back together with saving the
426  // context for both branches.
427  HandleScope scope_for_conversion(isolate);
428  std::vector<IndirectHandle<Object>> args(argc);
429  for (int i = 0; i < argc; ++i)
430  args[i] = indirect_handle(params.args[i], isolate);
431  Address** argv = reinterpret_cast<Address**>(args.data());
432 #else
433  Address** argv = const_cast<Address**>(
434  reinterpret_cast<Address* const*>(params.args.data()));
435 #endif
436 
437  // Block the allocation of handles without explicit handle scopes.
438  SealHandleScope shs(isolate);
439 
440  RCS_SCOPE(isolate, RuntimeCallCounterId::kJS_Execution);
441  value = Tagged<Object>(
442  stub_entry.Call(isolate->isolate_data()->isolate_root(), orig_func,
443  func, recv, JSParameterCount(argc), argv));
444  } else {
445  DCHECK_EQ(Execution::Target::kRunMicrotasks, params.execution_target);
446 
447  // clang-format off
448  // return value: tagged pointers
449  // {microtask_queue}: pointer to a C++ object
450  using JSEntryFunction = GeneratedCode<Address(
451  Address root_register_value, MicrotaskQueue* microtask_queue)>;
452  // clang-format on
453  JSEntryFunction stub_entry =
454  JSEntryFunction::FromAddress(isolate, code->instruction_start());
455 
456  // Block the allocation of handles without explicit handle scopes.
457  SealHandleScope shs(isolate);
458 
459  RCS_SCOPE(isolate, RuntimeCallCounterId::kJS_Execution);
460  value = Tagged<Object>(stub_entry.Call(
461  isolate->isolate_data()->isolate_root(), params.microtask_queue));
462  }
463  }
464 
465 #ifdef VERIFY_HEAP
466  if (v8_flags.verify_heap) {
467  Object::ObjectVerify(value, isolate);
468  }
469 #endif
470 
471  // Update the pending exception flag and return the value.
472  bool has_exception = IsException(value, isolate);
473  DCHECK_EQ(has_exception, isolate->has_exception());
474  if (has_exception) {
475  isolate->ReportPendingMessages(params.message_handling ==
476  Execution::MessageHandling::kReport);
477  return MaybeHandle<Object>();
478  } else {
479  isolate->clear_pending_message();
480  }
481 
482  return Handle<Object>(value, isolate);
483 }

485 MaybeDirectHandle<Object> InvokeWithTryCatch(Isolate* isolate,
486  const InvokeParams& params) {
487  DCHECK_IMPLIES(v8_flags.strict_termination_checks,
488  !isolate->is_execution_terminating());
489  MaybeDirectHandle<Object> maybe_result;
490  if (params.exception_out != nullptr) {
491  *params.exception_out = {};
492  }
493 
494  // Enter a try-block while executing the JavaScript code. To avoid
495  // duplicate error printing it must be non-verbose. Also, to avoid
496  // creating message objects during stack overflow we shouldn't
497  // capture messages.
498  v8::TryCatch catcher(reinterpret_cast<v8::Isolate*>(isolate));
499  catcher.SetVerbose(false);
500  catcher.SetCaptureMessage(false);
501 
502  maybe_result = Invoke(isolate, params);
503 
504  if (V8_LIKELY(!maybe_result.is_null())) {
505  DCHECK(!isolate->has_exception());
506  return maybe_result;
507  }
508 
509  DCHECK(isolate->has_exception());
510  if (isolate->is_execution_terminating()) {
511  return maybe_result;
512  }
513 
514  if (params.exception_out != nullptr) {
515  DCHECK(catcher.HasCaught());
516  *params.exception_out = v8::Utils::OpenDirectHandle(*catcher.Exception());
517  }
518 
519  return maybe_result;
520 }

621 void Execution::CallWasm(Isolate* isolate, DirectHandle<Code> wrapper_code,
622  WasmCodePointer wasm_call_target,
623  DirectHandle<Object> object_ref, Address packed_args) {
624  // Runtime code must be able to get the "current" isolate from TLS, and this
625  // must equal the isolate we execute in.
626  DCHECK_EQ(isolate, Isolate::TryGetCurrent());
627 
628  using WasmEntryStub = GeneratedCode<Address(
629  Address target, Address object_ref, Address argv, Address c_entry_fp)>;
630  WasmEntryStub stub_entry =
631  WasmEntryStub::FromAddress(isolate, wrapper_code->instruction_start());
632 
633  // Save and restore context around invocation and block the
634  // allocation of handles without explicit handle scopes.
635  SaveContext save(isolate);
636  SealHandleScope shs(isolate);
637 
638  Address saved_c_entry_fp = *isolate->c_entry_fp_address();
639  Address saved_js_entry_sp = *isolate->js_entry_sp_address();
640  if (saved_js_entry_sp == kNullAddress) {
641  *isolate->js_entry_sp_address() = GetCurrentStackPosition();
642  }
643  StackHandlerMarker stack_handler;
644  stack_handler.next = isolate->thread_local_top()->handler_;
645 #ifdef V8_USE_ADDRESS_SANITIZER
646  stack_handler.padding = GetCurrentStackPosition();
647 #else
648  stack_handler.padding = 0;
649 #endif
650  isolate->thread_local_top()->handler_ =
651  reinterpret_cast<Address>(&stack_handler);
652  trap_handler::SetThreadInWasm();
653 
654  {
655  RCS_SCOPE(isolate, RuntimeCallCounterId::kJS_Execution);
656  static_assert(compiler::CWasmEntryParameters::kCodeEntry == 0);
657  static_assert(compiler::CWasmEntryParameters::kObjectRef == 1);
658  static_assert(compiler::CWasmEntryParameters::kArgumentsBuffer == 2);
659  static_assert(compiler::CWasmEntryParameters::kCEntryFp == 3);
660  Address result =
661  stub_entry.CallSandboxed(wasm_call_target.value(), (*object_ref).ptr(),
662  packed_args, saved_c_entry_fp);
663  if (result != kNullAddress) isolate->set_exception(Tagged<Object>(result));
664  }
665 
666  // If there was an exception, then the thread-in-wasm flag is cleared
667  // already.
668  if (trap_handler::IsThreadInWasm()) {
669  trap_handler::ClearThreadInWasm();
670  }
671  isolate->thread_local_top()->handler_ = stack_handler.next;
672  if (saved_js_entry_sp == kNullAddress) {
673  *isolate->js_entry_sp_address() = saved_js_entry_sp;
674  }
675  *isolate->c_entry_fp_address() = saved_c_entry_fp;
676 }
