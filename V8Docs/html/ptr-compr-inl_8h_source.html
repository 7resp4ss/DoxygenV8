113 Address V8HeapCompressionSchemeImpl<Cage>::DecompressTagged(
114  Tagged_t raw_value) {
115 #ifdef V8_COMPRESS_POINTERS
116  Address cage_base = base();
117 #ifdef V8_COMPRESS_POINTERS_IN_MULTIPLE_CAGES
118  DCHECK_WITH_MSG(cage_base != kNullAddress,

131 template <typename Cage>
132 template <typename ProcessPointerCallback>
133 void V8HeapCompressionSchemeImpl<Cage>::ProcessIntermediatePointers(
134  Address raw_value, ProcessPointerCallback callback) {
135  // If pointer compression is enabled, we may have random compressed pointers
136  // on the stack that may be used for subsequent operations.
137  // Extract, decompress and trace both halfwords.
138  Address decompressed_low =
139  V8HeapCompressionSchemeImpl<Cage>::DecompressTagged(
140  static_cast<Tagged_t>(raw_value));
141  callback(decompressed_low);
142  Address decompressed_high =
143  V8HeapCompressionSchemeImpl<Cage>::DecompressTagged(
144  static_cast<Tagged_t>(raw_value >> (sizeof(Tagged_t) * CHAR_BIT)));
145  callback(decompressed_high);
146 }

215 // static
216 Address ExternalCodeCompressionScheme::DecompressTagged(Tagged_t raw_value) {
217  Address cage_base = base();
218 #ifdef V8_COMPRESS_POINTERS_IN_MULTIPLE_CAGES
219  DCHECK_WITH_MSG(cage_base != kNullAddress,
220  "ExternalCodeCompressionScheme::base is not initialized for "
221  "current thread");
222 #endif // V8_COMPRESS_POINTERS_IN_MULTIPLE_CAGES
223  V8_ASSUME((cage_base & kMinExpectedOSPageSizeMask) == cage_base);
224 
225  Address diff = static_cast<Address>(static_cast<uint32_t>(raw_value)) -
226  static_cast<Address>(static_cast<uint32_t>(cage_base));
227  // The cage base value was chosen such that it's less or equal than any
228  // pointer in the cage, thus if we got a negative diff then it means that
229  // the decompressed value is off by 4GB.
230  if (static_cast<intptr_t>(diff) < 0) {
231  diff += size_t{4} * GB;
232  }
233  DCHECK(is_uint32(diff));
234  Address result = cage_base + diff;
235  DCHECK_EQ(static_cast<uint32_t>(result), raw_value);
236  return result;
237 }
