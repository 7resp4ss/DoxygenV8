35 void Disassemble(base::Vector<const uint8_t> wire_bytes,
36  v8::debug::DisassemblyCollector* collector,
37  std::vector<int>* function_body_offsets) {
38  std::unique_ptr<OffsetsProvider> offsets = AllocateOffsetsProvider();
39  ModuleResult result =
40  DecodeWasmModuleForDisassembler(wire_bytes, offsets.get());
41  MultiLineStringBuilder out;
42  AccountingAllocator allocator;
43  if (result.failed()) {
44  WasmError error = result.error();
45  out << "Decoding error: " << error.message() << " at offset "
46  << error.offset();
47  out.ToDisassemblyCollector(collector);
48  return;
49  }
50  const WasmModule* module = result.value().get();
51  NamesProvider names(module, wire_bytes);
52  ModuleWireBytes module_bytes(wire_bytes);
53  ModuleDisassembler md(out, module, &names, module_bytes, &allocator,
54  std::move(offsets), function_body_offsets);
55  md.PrintModule({0, 2}, v8_flags.wasm_disassembly_max_mb);
56  out.ToDisassemblyCollector(collector);
57 }

120 // Returns the log2 of the alignment, e.g. "4" means 2<<4 == 16 bytes.
121 // This is the same format as used in .wasm binary modules.
122 uint32_t GetDefaultAlignment(WasmOpcode opcode) {
123  switch (opcode) {
124  case kExprS128LoadMem:
125  case kExprS128StoreMem:
126  return 4;
127  case kExprS128Load8x8S:
128  case kExprS128Load8x8U:
129  case kExprS128Load16x4S:
130  case kExprS128Load16x4U:
131  case kExprS128Load32x2S:
132  case kExprS128Load32x2U:
133  case kExprS128Load64Splat:
134  case kExprS128Load64Zero:
135  case kExprS128Load64Lane:
136  case kExprS128Store64Lane:
137  return 3;
138  case kExprS128Load32Splat:
139  case kExprS128Load32Zero:
140  case kExprS128Load32Lane:
141  case kExprS128Store32Lane:
142  return 2;
143  case kExprS128Load16Splat:
144  case kExprS128Load16Lane:
145  case kExprS128Store16Lane:
146  return 1;
147  case kExprS128Load8Splat:
148  case kExprS128Load8Lane:
149  case kExprS128Store8Lane:
150  return 0;
151 
152 #define CASE(Opcode, ...) \
153  case kExpr##Opcode: \
154  return GetLoadType(kExpr##Opcode).size_log_2();
155  FOREACH_LOAD_MEM_OPCODE(CASE)
156 #undef CASE
157 #define CASE(Opcode, ...) \
158  case kExpr##Opcode: \
159  return GetStoreType(kExpr##Opcode).size_log_2();
160  FOREACH_STORE_MEM_OPCODE(CASE)
161 #undef CASE
162 
163 #define CASE(Opcode, Type) \
164  case kExpr##Opcode: \
165  return ElementSizeLog2Of(MachineType::Type().representation());
166  ATOMIC_OP_LIST(CASE)
167  ATOMIC_STORE_OP_LIST(CASE)
168 #undef CASE
169 
170  default:
171  UNREACHABLE();
172  }
173 }

217 void FunctionBodyDisassembler::DecodeAsWat(MultiLineStringBuilder& out,
218  Indentation indentation,
219  FunctionHeader include_header,
220  uint32_t* first_instruction_offset) {
221  out_ = &out;
222  int base_indentation = indentation.current();
223  // Print header.
224  if (include_header == kPrintHeader) {
225  out << indentation << "(func ";
226  names_->PrintFunctionName(out, func_index_, NamesProvider::kDevTools);
227  PrintSignatureOneLine(out, sig_, func_index_, names_, true,
228  kIndicesAsComments);
229  out.NextLine(pc_offset());
230  } else {
231  out.set_current_line_bytecode_offset(pc_offset());
232  }
233  indentation.increase();
234 
235  // Decode and print locals.
236  uint32_t locals_length = DecodeLocals(pc_);
237  if (failed()) {
238  // TODO(jkummerow): Improve error handling.
239  out << "Failed to decode locals\n";
240  return;
241  }
242  for (uint32_t i = static_cast<uint32_t>(sig_->parameter_count());
243  i < num_locals_; i++) {
244  out << indentation << "(local ";
245  names_->PrintLocalName(out, func_index_, i);
246  out << " ";
247  names_->PrintValueType(out, local_type(i));
248  out << ")";
249  out.NextLine(pc_offset());
250  }
251  consume_bytes(locals_length);
252  out.set_current_line_bytecode_offset(pc_offset());
253  if (first_instruction_offset) *first_instruction_offset = pc_offset();
254 
255  // Main loop.
256  while (pc_ < end_ && ok()) {
257  WasmOpcode opcode = GetOpcode();
258  current_opcode_ = opcode; // Some immediates need to know this.
259 
260  // Deal with indentation.
261  if (opcode == kExprEnd || opcode == kExprElse || opcode == kExprCatch ||
262  opcode == kExprCatchAll || opcode == kExprDelegate) {
263  if (indentation.current() >= base_indentation) {
264  indentation.decrease();
265  }
266  }
267  out << indentation;
268  if (opcode == kExprElse || opcode == kExprCatch ||
269  opcode == kExprCatchAll || opcode == kExprBlock || opcode == kExprIf ||
270  opcode == kExprLoop || opcode == kExprTry || opcode == kExprTryTable) {
271  indentation.increase();
272  }
273 
274  // Print the opcode and its immediates.
275  if (opcode == kExprEnd) {
276  if (indentation.current() < base_indentation) {
277  out << ";; Unexpected end byte";
278  } else if (indentation.current() == base_indentation) {
279  out << ")"; // End of the function.
280  } else {
281  out << "end";
282  const LabelInfo& label = label_stack_.back();
283  if (label.start != nullptr) {
284  out << " ";
285  out.write(label.start, label.length);
286  }
287  label_stack_.pop_back();
288  }
289  } else {
290  out << WasmOpcodes::OpcodeName(opcode);
291  }
292  if (opcode == kExprBlock || opcode == kExprIf || opcode == kExprLoop ||
293  opcode == kExprTry || opcode == kExprTryTable) {
294  // Create the LabelInfo now to get the correct offset, but only push it
295  // after printing the immediates because the immediates don't see the new
296  // label yet.
297  LabelInfo label(out.line_number(), out.length(),
298  label_occurrence_index_++);
299  pc_ += PrintImmediatesAndGetLength(out);
300  label_stack_.push_back(label);
301  } else {
302  pc_ += PrintImmediatesAndGetLength(out);
303  }
304 
305  out.NextLine(pc_offset());
306  }
307 
308  if (pc_ != end_) {
309  // TODO(jkummerow): Improve error handling.
310  out << "Beyond end of code";
311  }
312 }

314 void FunctionBodyDisassembler::DecodeGlobalInitializer(StringBuilder& out) {
315  while (pc_ < end_) {
316  WasmOpcode opcode = GetOpcode();
317  current_opcode_ = opcode; // Some immediates need to know this.
318  // Don't print the final "end".
319  if (opcode == kExprEnd && pc_ + 1 == end_) break;
320  uint32_t length;
321  out << " (" << WasmOpcodes::OpcodeName(opcode);
322  length = PrintImmediatesAndGetLength(out);
323  out << ")";
324  pc_ += length;
325  }

334 void FunctionBodyDisassembler::PrintHexNumber(StringBuilder& out,
335  uint64_t number) {
336  constexpr size_t kBufferSize = sizeof(number) * 2 + 2; // +2 for "0x".
337  char buffer[kBufferSize];
338  char* end = buffer + kBufferSize;
339  char* ptr = end;
340  do {
341  *(--ptr) = kHexChars[number & 0xF];
342  number >>= 4;
343  } while (number > 0);
344  *(--ptr) = 'x';
345  *(--ptr) = '0';
346  size_t length = static_cast<size_t>(end - ptr);
347  char* output = out.allocate(length);
348  memcpy(output, ptr, length);

360  void PrintDepthAsLabel(int imm_depth) {
361  out_ << " ";
362  size_t label_start_position = out_.length();
363  int depth = imm_depth;
364  if (owner_->current_opcode_ == kExprDelegate) depth++;
365  // Be robust: if the module is invalid, print what we got.
366  if (depth < 0 || depth >= static_cast<int>(owner_->label_stack_.size())) {
367  out_ << imm_depth;
368  return;
369  }
370  // If the label's name has already been determined and backpatched, just
371  // copy it here.
372  LabelInfo& label_info = owner_->label_info(depth);
373  if (label_info.start != nullptr) {
374  out_.write(label_info.start, label_info.length);
375  return;

378  names()->PrintLabelName(out_, owner_->func_index_,
379  label_info.name_section_index,
380  owner_->label_generation_index_++);

382  owner_->out_->PatchLabel(label_info, out_.start() + label_start_position);

588  void F32Const(ImmF32Immediate& imm) {
589  float f = imm.value;
590  if (f == 0) {
591  out_ << (1 / f < 0 ? " -0.0" : " 0.0");
592  } else if (std::isinf(f)) {
593  out_ << (f > 0 ? " inf" : " -inf");
594  } else if (std::isnan(f)) {
595  uint32_t bits = base::bit_cast<uint32_t>(f);
596  uint32_t payload = bits & 0x7F'FFFFu;
597  uint32_t signbit = bits >> 31;
598  if (payload == 0x40'0000u) {
599  out_ << (signbit == 1 ? " -nan" : " nan");
600  } else {
601  out_ << (signbit == 1 ? " -nan:" : " +nan:");
602  owner_->PrintHexNumber(out_, payload);
603  }
604  } else {
605  std::ostringstream o;
606  // TODO(dlehmann): Change to `std::format` (C++20) or to `std::to_chars`
607  // (C++17) once available, so that `0.1` isn't printed as `0.100000001`
608  // any more.
609  o << std::setprecision(std::numeric_limits<float>::max_digits10) << f;
610  out_ << " " << o.str();
611  }

614  void F64Const(ImmF64Immediate& imm) {
615  double d = imm.value;
616  if (d == 0) {
617  out_ << (1 / d < 0 ? " -0.0" : " 0.0");
618  } else if (std::isinf(d)) {
619  out_ << (d > 0 ? " inf" : " -inf");
620  } else if (std::isnan(d)) {
621  uint64_t bits = base::bit_cast<uint64_t>(d);
622  uint64_t payload = bits & 0xF'FFFF'FFFF'FFFFull;
623  uint64_t signbit = bits >> 63;
624  if (payload == 0x8'0000'0000'0000ull) {
625  out_ << (signbit == 1 ? " -nan" : " nan");
626  } else {
627  out_ << (signbit == 1 ? " -nan:" : " +nan:");
628  owner_->PrintHexNumber(out_, payload);

845 void ModuleDisassembler::PrintModule(Indentation indentation, size_t max_mb) {
846  // 0. General infrastructure.
847  // We don't store import/export information on {WasmTag} currently.
848  size_t num_tags = module_->tags.size();
849  std::vector<bool> exported_tags(num_tags, false);
850  for (const WasmExport& ex : module_->export_table) {
851  if (ex.kind == kExternalTag) exported_tags[ex.index] = true;
852  }
853 
854  // I. Module name.
855  out_ << indentation << "(module";
856  if (module_->name.is_set()) {
857  out_ << " $";
858  const uint8_t* name_start = start_ + module_->name.offset();
859  out_.write(name_start, module_->name.length());
860  }
861  indentation.increase();
862 
863  // II. Types
864  uint32_t recgroup_index = 0;
865  OffsetsProvider::RecGroup recgroup = offsets_->recgroup(recgroup_index++);
866  bool in_explicit_recgroup = false;
867  for (uint32_t i = 0; i < module_->types.size(); i++) {
868  // No need to check {recgroup.valid()}, as the comparison will simply
869  // never be true otherwise.
870  while (i == recgroup.start_type_index) {
871  out_.NextLine(recgroup.offset);
872  out_ << indentation << "(rec";
873  if V8_UNLIKELY (recgroup.end_type_index == i) {
874  // Empty recgroup.
875  out_ << ")";
876  DCHECK(!in_explicit_recgroup);
877  recgroup = offsets_->recgroup(recgroup_index++);
878  continue;
879  } else {
880  in_explicit_recgroup = true;
881  indentation.increase();
882  break;
883  }
884  }
885  if (kSkipFunctionTypesInTypeSection &&
886  module_->has_signature(ModuleTypeIndex{i}) && !in_explicit_recgroup) {
887  continue;
888  }
889  PrintTypeDefinition(i, indentation, kIndicesAsComments);
890  if (in_explicit_recgroup && i == recgroup.end_type_index - 1) {
891  in_explicit_recgroup = false;
892  indentation.decrease();
893  // The end of a recgroup is implicit in the wire bytes, so repeat the
894  // previous line's offset for it.
895  uint32_t offset = out_.current_line_bytecode_offset();
896  out_.NextLine(offset);
897  out_ << indentation << ")";
898  recgroup = offsets_->recgroup(recgroup_index++);
899  }
900  }
901  while (recgroup.valid()) {
902  // There could be empty recgroups at the end of the type section.
903  DCHECK_GE(recgroup.start_type_index, module_->types.size());
904  DCHECK_EQ(recgroup.start_type_index, recgroup.end_type_index);
905  out_.NextLine(recgroup.offset);
906  out_ << indentation << "(rec)";
907  recgroup = offsets_->recgroup(recgroup_index++);
908  }
909 
910  // III. Imports
911  for (uint32_t i = 0; i < module_->import_table.size(); i++) {
912  const WasmImport& import = module_->import_table[i];
913  out_.NextLine(offsets_->import_offset(i));
914  out_ << indentation;
915  switch (import.kind) {
916  case kExternalTable: {
917  out_ << "(table ";
918  names_->PrintTableName(out_, import.index, kIndicesAsComments);
919  const WasmTable& table = module_->tables[import.index];
920  if (table.exported) PrintExportName(kExternalTable, import.index);
921  PrintImportName(import);
922  PrintTable(table);
923  break;
924  }
925  case kExternalFunction: {
926  out_ << "(func ";
927  names_->PrintFunctionName(out_, import.index, NamesProvider::kDevTools,
928  kIndicesAsComments);
929  const WasmFunction& func = module_->functions[import.index];
930  if (func.exported) PrintExportName(kExternalFunction, import.index);
931  PrintImportName(import);
932  PrintSignatureOneLine(out_, func.sig, import.index, names_, false);
933  break;
934  }
935  case kExternalGlobal: {
936  out_ << "(global ";
937  names_->PrintGlobalName(out_, import.index, kIndicesAsComments);
938  const WasmGlobal& global = module_->globals[import.index];
939  if (global.exported) PrintExportName(kExternalGlobal, import.index);
940  PrintImportName(import);
941  PrintGlobal(global);
942  break;
943  }
944  case kExternalMemory:
945  out_ << "(memory ";
946  names_->PrintMemoryName(out_, import.index, kIndicesAsComments);
947  if (module_->memories[import.index].exported) {
948  PrintExportName(kExternalMemory, 0);
949  }
950  PrintImportName(import);
951  PrintMemory(module_->memories[import.index]);
952  break;
953  case kExternalTag:
954  out_ << "(tag ";
955  names_->PrintTagName(out_, import.index, kIndicesAsComments);
956  PrintImportName(import);
957  if (exported_tags[import.index]) {
958  PrintExportName(kExternalTag, import.index);
959  }
960  PrintTagSignature(module_->tags[import.index].sig);
961  break;
962  }
963  out_ << ")";
964  }
965 
966  // IV. Tables
967  for (uint32_t i = module_->num_imported_tables; i < module_->tables.size();
968  i++) {
969  const WasmTable& table = module_->tables[i];
970  DCHECK(!table.imported);
971  out_.NextLine(offsets_->table_offset(i));
972  out_ << indentation << "(table ";
973  names_->PrintTableName(out_, i, kIndicesAsComments);
974  if (table.exported) PrintExportName(kExternalTable, i);
975  PrintTable(table);
976  out_ << ")";
977  }
978 
979  // V. Memories
980  uint32_t num_memories = static_cast<uint32_t>(module_->memories.size());
981  for (uint32_t memory_index = 0; memory_index < num_memories; ++memory_index) {
982  const WasmMemory& memory = module_->memories[memory_index];
983  if (memory.imported) continue;
984  out_.NextLine(offsets_->memory_offset());
985  out_ << indentation << "(memory ";
986  names_->PrintMemoryName(out_, memory_index, kIndicesAsComments);
987  if (memory.exported) PrintExportName(kExternalMemory, memory_index);
988  PrintMemory(memory);
989  out_ << ")";
990  }
991 
992  // VI.Tags
993  for (uint32_t i = module_->num_imported_tags; i < module_->tags.size(); i++) {
994  const WasmTag& tag = module_->tags[i];
995  out_.NextLine(offsets_->tag_offset(i));
996  out_ << indentation << "(tag ";
997  names_->PrintTagName(out_, i, kIndicesAsComments);
998  if (exported_tags[i]) PrintExportName(kExternalTag, i);
999  PrintTagSignature(tag.sig);
1000  out_ << ")";
1001  }
1002 
1003  // VII. String literals
1004  size_t num_strings = module_->stringref_literals.size();
1005  for (uint32_t i = 0; i < num_strings; i++) {
1006  const WasmStringRefLiteral lit = module_->stringref_literals[i];
1007  out_.NextLine(offsets_->string_offset(i));
1008  out_ << indentation << "(string \"";
1009  PrintString(lit.source);
1010  out_ << '"';
1011  if (kIndicesAsComments) out_ << " (;" << i << ";)";
1012  out_ << ")";
1013  }
1014 
1015  // VIII. Globals
1016  for (uint32_t i = module_->num_imported_globals; i < module_->globals.size();
1017  i++) {
1018  const WasmGlobal& global = module_->globals[i];
1019  DCHECK(!global.imported);
1020  out_.NextLine(offsets_->global_offset(i));
1021  out_ << indentation << "(global ";
1022  names_->PrintGlobalName(out_, i, kIndicesAsComments);
1023  if (global.exported) PrintExportName(kExternalGlobal, i);
1024  PrintGlobal(global);
1025  PrintInitExpression(global.init, global.type);
1026  out_ << ")";
1027  }
1028 
1029  // IX. Start
1030  if (module_->start_function_index >= 0) {
1031  out_.NextLine(offsets_->start_offset());
1032  out_ << indentation << "(start ";
1033  names_->PrintFunctionName(out_, module_->start_function_index,
1034  NamesProvider::kDevTools);
1035  out_ << ")";
1036  }
1037 
1038  // X. Elements
1039  for (uint32_t i = 0; i < module_->elem_segments.size(); i++) {
1040  const WasmElemSegment& elem = module_->elem_segments[i];
1041  out_.NextLine(offsets_->element_offset(i));
1042  out_ << indentation << "(elem ";
1043  names_->PrintElementSegmentName(out_, i, kIndicesAsComments);
1044  if (elem.status == WasmElemSegment::kStatusDeclarative) {
1045  out_ << " declare";
1046  } else if (elem.status == WasmElemSegment::kStatusActive) {
1047  if (elem.table_index != 0) {
1048  out_ << " (table ";
1049  names_->PrintTableName(out_, elem.table_index);
1050  out_ << ")";
1051  }
1052  PrintInitExpression(elem.offset, kWasmI32);
1053  }
1054  out_ << " ";
1055  if (elem.shared) out_ << "shared ";
1056  names_->PrintValueType(out_, elem.type);
1057 
1058  WasmDetectedFeatures unused_detected_features;
1059  ModuleDecoderImpl decoder(
1060  WasmEnabledFeatures::All(), wire_bytes_.module_bytes(),
1061  ModuleOrigin::kWasmOrigin, &unused_detected_features);
1062  decoder.consume_bytes(elem.elements_wire_bytes_offset);
1063  for (size_t j = 0; j < elem.element_count; j++) {
1064  ConstantExpression entry = decoder.consume_element_segment_entry(
1065  const_cast<WasmModule*>(module_), elem);
1066  PrintInitExpression(entry, elem.type);
1067  }
1068  out_ << ")";
1069  }
1070 
1071  // For the FunctionBodyDisassembler, we flip the convention: {NextLine} is
1072  // now called *after* printing something, instead of before.
1073  if (out_.length() != 0) out_.NextLine(0);
1074 
1075  // XI. Code / function bodies.
1076  if (function_body_offsets_ != nullptr) {
1077  size_t num_defined_functions =
1078  module_->functions.size() - module_->num_imported_functions;
1079  function_body_offsets_->reserve(num_defined_functions * 2);
1080  }
1081  for (uint32_t i = module_->num_imported_functions;
1082  i < module_->functions.size(); i++) {
1083  const WasmFunction* func = &module_->functions[i];
1084  out_.set_current_line_bytecode_offset(func->code.offset());
1085  out_ << indentation << "(func ";
1086  names_->PrintFunctionName(out_, i, NamesProvider::kDevTools,
1087  kIndicesAsComments);
1088  if (func->exported) PrintExportName(kExternalFunction, i);
1089  PrintSignatureOneLine(out_, func->sig, i, names_, true, kIndicesAsComments);
1090  out_.NextLine(func->code.offset());
1091  bool shared = module_->type(func->sig_index).is_shared;
1092  WasmDetectedFeatures detected;
1093  base::Vector<const uint8_t> code = wire_bytes_.GetFunctionBytes(func);
1094  FunctionBodyDisassembler d(&zone_, module_, i, shared, &detected, func->sig,
1095  code.begin(), code.end(), func->code.offset(),
1096  wire_bytes_, names_);
1097  uint32_t first_instruction_offset;
1098  d.DecodeAsWat(out_, indentation, FunctionBodyDisassembler::kSkipHeader,
1099  &first_instruction_offset);
1100  if (function_body_offsets_ != nullptr) {
1101  function_body_offsets_->push_back(first_instruction_offset);
1102  function_body_offsets_->push_back(d.pc_offset());
1103  }
1104  if (out_.ApproximateSizeMB() > max_mb) {
1105  out_ << "<truncated...>";
1106  return;
1107  }
1108  }
1109 
1110  // XII. Data
1111  for (uint32_t i = 0; i < module_->data_segments.size(); i++) {
1112  const WasmDataSegment& data = module_->data_segments[i];
1113  out_.set_current_line_bytecode_offset(offsets_->data_offset(i));
1114  out_ << indentation << "(data";
1115  if (!kSkipDataSegmentNames) {
1116  out_ << " ";
1117  names_->PrintDataSegmentName(out_, i, kIndicesAsComments);
1118  }
1119  if (data.shared) out_ << " shared";
1120  if (data.active) {
1121  ValueType type = module_->memories[data.memory_index].is_memory64()
1122  ? kWasmI64
1123  : kWasmI32;
1124  PrintInitExpression(data.dest_addr, type);
1125  }
1126  out_ << " \"";
1127  PrintString(data.source);
1128  out_ << "\")";
1129  out_.NextLine(0);
1130 
1131  if (out_.ApproximateSizeMB() > max_mb) {
1132  out_ << "<truncated...>";
1133  return;
1134  }
1135  }
1136 
1137  indentation.decrease();
1138  out_.set_current_line_bytecode_offset(
1139  static_cast<uint32_t>(wire_bytes_.length()));
1140  out_ << indentation << ")"; // End of the module.
1141  out_.NextLine(0);
1142 }

1188  ValueType expected_type) {
1189  switch (init.kind()) {
1190  case ConstantExpression::Kind::kEmpty:
1191  break;
1192  case ConstantExpression::Kind::kI32Const:
1193  out_ << " (i32.const " << init.i32_value() << ")";
1194  break;
1195  case ConstantExpression::Kind::kRefNull:
1196  out_ << " (ref.null ";
1197  names_->PrintHeapType(out_, init.type());
1198  out_ << ")";
1199  break;
1200  case ConstantExpression::Kind::kRefFunc:

1203  out_ << ")";
