137 uint32_t CodeGenerator::GetStackCheckOffset() {
138  if (!frame_access_state()->has_frame()) {
139  DCHECK_EQ(max_unoptimized_frame_height_, 0);
140  DCHECK_EQ(max_pushed_argument_count_, 0);
141  return 0;
142  }
143 
144  size_t incoming_parameter_count =
145  linkage_->GetIncomingDescriptor()->ParameterSlotCount();
146  DCHECK(is_int32(incoming_parameter_count));
147  int32_t optimized_frame_height =
148  static_cast<int32_t>(incoming_parameter_count) * kSystemPointerSize +
149  frame()->GetTotalFrameSlotCount() * kSystemPointerSize;
150  DCHECK(is_int32(max_unoptimized_frame_height_));
151  int32_t signed_max_unoptimized_frame_height =
152  static_cast<int32_t>(max_unoptimized_frame_height_);
153 
154  // The offset is either the delta between the optimized frames and the
155  // interpreted frame, or the maximal number of bytes pushed to the stack
156  // while preparing for function calls, whichever is bigger.
157  uint32_t frame_height_delta = static_cast<uint32_t>(std::max(
158  signed_max_unoptimized_frame_height - optimized_frame_height, 0));
159  uint32_t max_pushed_argument_bytes =
160  static_cast<uint32_t>(max_pushed_argument_count_ * kSystemPointerSize);
161  return std::max(frame_height_delta, max_pushed_argument_bytes);
162 }

164 CodeGenerator::CodeGenResult CodeGenerator::AssembleDeoptimizerCall(
165  DeoptimizationExit* exit) {
166  int deoptimization_id = exit->deoptimization_id();
167  if (deoptimization_id > Deoptimizer::kMaxNumberOfEntries) {
168  return kTooManyDeoptimizationBailouts;
169  }
170 
171  DeoptimizeKind deopt_kind = exit->kind();
172  DeoptimizeReason deoptimization_reason = exit->reason();
173  Label* jump_deoptimization_entry_label =
174  &jump_deoptimization_entry_labels_[static_cast<int>(deopt_kind)];
175  if (info()->source_positions() ||
176  AlwaysPreserveDeoptReason(deoptimization_reason)) {
177  masm()->RecordDeoptReason(deoptimization_reason, exit->node_id(),
178  exit->pos(), deoptimization_id);
179  }
180 
181  if (deopt_kind == DeoptimizeKind::kLazy ||
182  deopt_kind == DeoptimizeKind::kLazyAfterFastCall) {
183  ++lazy_deopt_count_;
184  masm()->BindExceptionHandler(exit->label());
185  } else {
186  ++eager_deopt_count_;
187  masm()->bind(exit->label());
188  }
189  Builtin target = Deoptimizer::GetDeoptimizationEntry(deopt_kind);
190  masm()->CallForDeoptimization(target, deoptimization_id, exit->label(),
191  deopt_kind, exit->continue_label(),
192  jump_deoptimization_entry_label);
193 
194  exit->set_emitted();
195 
196  return kSuccess;
197 }

203 void CodeGenerator::AssembleCode() {
204  OptimizedCompilationInfo* info = this->info();
205  auto call_descriptor = linkage()->GetIncomingDescriptor();
206 
207  // Compute incoming parameter count for code using JS linkage. This will
208  // ultimately set the parameter count on the resulting Code object.
209  if (call_descriptor->IsJSFunctionCall()) {
210  parameter_count_ = call_descriptor->ParameterSlotCount();
211  if (Builtins::IsBuiltinId(info->builtin())) {
212  CHECK_EQ(parameter_count_,
213  Builtins::GetStackParameterCount(info->builtin()));
214  } else if (info->has_bytecode_array()) {
215  CHECK_EQ(parameter_count_, info->bytecode_array()->parameter_count());
216  }
217  }
218 
219  // Open a frame scope to indicate that there is a frame on the stack. The
220  // MANUAL indicates that the scope shouldn't actually generate code to set up
221  // the frame (that is done in AssemblePrologue).
222  FrameScope frame_scope(masm(), StackFrame::MANUAL);
223 
224  if (info->source_positions()) {
225  AssembleSourcePosition(start_source_position());
226  }
227  offsets_info_.code_start_register_check = masm()->pc_offset();
228 
229  masm()->CodeEntry();
230 
231 #ifdef V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
232  // TODO(saelo): should there also be a info->IsJS()?
233  if (v8_flags.debug_code &&
234  (call_descriptor->IsJSFunctionCall() || info->IsWasm())) {
235  masm()->RecordComment("-- Prologue: check sandboxing mode --");
236  masm()->AssertInSandboxedExecutionMode();
237  }
238 #endif
239 
240  // Check that {kJavaScriptCallCodeStartRegister} has been set correctly.
241  if (v8_flags.debug_code && info->called_with_code_start_register()) {
242  masm()->RecordComment("-- Prologue: check code start register --");
243  AssembleCodeStartRegisterCheck();
244  }
245 
246 #ifdef V8_ENABLE_LEAPTIERING
247  // Check that {kJavaScriptCallDispatchHandleRegister} has been set correctly.
248  if (v8_flags.debug_code && call_descriptor->IsJSFunctionCall()) {
249  masm()->RecordComment("-- Prologue: check dispatch handle register --");
250  AssembleDispatchHandleRegisterCheck();
251  }
252 #endif
253 
254  offsets_info_.deopt_check = masm()->pc_offset();
255  // We want to bailout only from JS functions, which are the only ones
256  // that are optimized.
257  if (info->IsOptimizing()) {
258  DCHECK(call_descriptor->IsJSFunctionCall());
259  masm()->RecordComment("-- Prologue: check for deoptimization --");
260  BailoutIfDeoptimized();
261  }
262 
263  // Define deoptimization literals for all inlined functions.
264  DCHECK_EQ(0u, deoptimization_literals_.size());
265  for (OptimizedCompilationInfo::InlinedFunctionHolder& inlined :
266  info->inlined_functions()) {
267  if (!inlined.shared_info.equals(info->shared_info())) {
268  int index = DefineDeoptimizationLiteral(
269  DeoptimizationLiteral(inlined.shared_info));
270  inlined.RegisterInlinedFunctionId(index);
271  }
272  }
273  inlined_function_count_ = deoptimization_literals_.size();
274 
275  unwinding_info_writer_.SetNumberOfInstructionBlocks(
276  instructions()->InstructionBlockCount());
277 
278  if (info->trace_turbo_json()) {
279  block_starts_.assign(instructions()->instruction_blocks().size(), -1);
280  instr_starts_.assign(instructions()->instructions().size(), {});
281  }
282  // Assemble instructions in assembly order.
283  offsets_info_.blocks_start = masm()->pc_offset();
284  for (const InstructionBlock* block : instructions()->ao_blocks()) {
285  // Align loop headers on vendor recommended boundaries.
286  if (block->ShouldAlignLoopHeader()) {
287  masm()->LoopHeaderAlign();
288  } else if (block->ShouldAlignCodeTarget()) {
289  masm()->CodeTargetAlign();
290  }
291 
292  if (info->trace_turbo_json()) {
293  block_starts_[block->rpo_number().ToInt()] = masm()->pc_offset();
294  }
295  // Bind a label for a block.
296  current_block_ = block->rpo_number();
297  unwinding_info_writer_.BeginInstructionBlock(masm()->pc_offset(), block);
298  if (v8_flags.code_comments && !block->omitted_by_jump_threading()) {
299  std::ostringstream buffer;
300  buffer << "-- B" << block->rpo_number().ToInt() << " start";
301  if (block->IsDeferred()) buffer << " (deferred)";
302  if (!block->needs_frame()) buffer << " (no frame)";
303  if (block->must_construct_frame()) buffer << " (construct frame)";
304  if (block->must_deconstruct_frame()) buffer << " (deconstruct frame)";
305 
306  if (block->IsLoopHeader()) {
307  buffer << " (loop up to " << block->loop_end().ToInt() << ")";
308  }
309  if (block->loop_header().IsValid()) {
310  buffer << " (in loop " << block->loop_header().ToInt() << ")";
311  }
312  buffer << " --";
313  masm()->RecordComment(buffer.str().c_str(), SourceLocation());
314  }
315 
316  frame_access_state()->MarkHasFrame(block->needs_frame());
317 
318  masm()->bind(GetLabel(current_block_));
319 
320 #ifdef V8_ENABLE_CONTROL_FLOW_INTEGRITY
321  if (block->IsTableSwitchTarget()) {
322  masm()->JumpTarget();
323  }
324 #endif
325 
326  if (block->must_construct_frame()) {
327  AssembleConstructFrame();
328  // We need to setup the root register after we assemble the prologue, to
329  // avoid clobbering callee saved registers in case of C linkage and
330  // using the roots.
331  // TODO(mtrofin): investigate how we can avoid doing this repeatedly.
332  if (call_descriptor->InitializeRootRegister()) {
333  masm()->InitializeRootRegister();
334  }
335  }
336 #ifdef CAN_USE_RVV_INSTRUCTIONS
337  // RVV uses VectorUnit to emit vset{i}vl{i}, reducing the static and dynamic
338  // overhead of the vset{i}vl{i} instruction. However there are some jumps
339  // back between blocks. the Rvv instruction may get an incorrect vtype. so
340  // here VectorUnit needs to be cleared to ensure that the vtype is correct
341  // within the block.
342  masm()->VU.clear();
343 #endif
344  if (V8_EMBEDDED_CONSTANT_POOL_BOOL && !block->needs_frame()) {
345  ConstantPoolUnavailableScope constant_pool_unavailable(masm());
346  result_ = AssembleBlock(block);
347  } else {
348  result_ = AssembleBlock(block);
349  }
350  if (result_ != kSuccess) return;
351  unwinding_info_writer_.EndInstructionBlock(block);
352  }
353 
354  // Assemble all out-of-line code.
355  offsets_info_.out_of_line_code = masm()->pc_offset();
356  if (ools_) {
357  masm()->RecordComment("-- Out of line code --");
358  for (OutOfLineCode* ool = ools_; ool; ool = ool->next()) {
359  masm()->bind(ool->entry());
360  ool->Generate();
361  if (ool->exit()->is_bound()) masm()->jmp(ool->exit());
362  }
363  }
364 
365  // This nop operation is needed to ensure that the trampoline is not
366  // confused with the pc of the call before deoptimization.
367  // The test regress/regress-259 is an example of where we need it.
368  masm()->nop();
369 
370  // For some targets, we must make sure that constant and veneer pools are
371  // emitted before emitting the deoptimization exits.
372  PrepareForDeoptimizationExits(&deoptimization_exits_);
373 
374  deopt_exit_start_offset_ = masm()->pc_offset();
375 
376  // Assemble deoptimization exits.
377  offsets_info_.deoptimization_exits = masm()->pc_offset();
378  int last_updated = 0;
379  // We sort the deoptimization exits here so that the lazy ones will be visited
380  // last. We need this as lazy deopts might need additional instructions.
381  auto cmp = [](const DeoptimizationExit* a, const DeoptimizationExit* b) {
382  // The deoptimization exits are sorted so that lazy deopt exits appear after
383  // eager deopts.
384  int a_kind_val = a->kind() == DeoptimizeKind::kEager ? 0 : 1;
385  int b_kind_val = b->kind() == DeoptimizeKind::kEager ? 0 : 1;
386 
387  DCHECK_IMPLIES(a_kind_val == 1,
388  a->kind() == DeoptimizeKind::kLazy ||
389  a->kind() == DeoptimizeKind::kLazyAfterFastCall);
390  DCHECK_IMPLIES(b_kind_val == 1,
391  b->kind() == DeoptimizeKind::kLazy ||
392  b->kind() == DeoptimizeKind::kLazyAfterFastCall);
393 
394  if (a_kind_val != b_kind_val) {
395  return a_kind_val < b_kind_val;
396  }
397  return a->pc_offset() < b->pc_offset();
398  };
399  std::sort(deoptimization_exits_.begin(), deoptimization_exits_.end(), cmp);
400 
401  {
402 #ifdef V8_TARGET_ARCH_PPC64
403  v8::internal::Assembler::BlockTrampolinePoolScope block_trampoline_pool(
404  masm());
405 #endif
406  for (DeoptimizationExit* exit : deoptimization_exits_) {
407  if (exit->emitted()) continue;
408  exit->set_deoptimization_id(next_deoptimization_id_++);
409  result_ = AssembleDeoptimizerCall(exit);
410  if (result_ != kSuccess) return;
411 
412  // UpdateDeoptimizationInfo expects lazy deopts to be visited in pc_offset
413  // order, which is always the case since they are added to
414  // deoptimization_exits_ in that order, and the optional sort operation
415  // above preserves that order.
416  if (exit->kind() == DeoptimizeKind::kLazy ||
417  exit->kind() == DeoptimizeKind::kLazyAfterFastCall) {
418  int trampoline_pc = exit->label()->pos();
419  last_updated = safepoints()->UpdateDeoptimizationInfo(
420  exit->pc_offset(), trampoline_pc, last_updated,
421  exit->deoptimization_id());
422  }
423  }
424  }
425 
426  offsets_info_.pools = masm()->pc_offset();
427  // TODO(jgruber): Move all inlined metadata generation into a new,
428  // architecture-independent version of FinishCode. Currently, this includes
429  // the safepoint table, handler table, constant pool, and code comments, in
430  // that order.
431  FinishCode();
432 
433  offsets_info_.jump_tables = masm()->pc_offset();
434  // Emit the jump tables.
435  if (jump_tables_) {
436  masm()->Align(kSystemPointerSize);
437  for (JumpTable* table = jump_tables_; table; table = table->next()) {
438  masm()->bind(table->label());
439  AssembleJumpTable(table->targets());
440  }
441  }
442 
443  // The PerfJitLogger logs code up until here, excluding the safepoint
444  // table. Resolve the unwinding info now so it is aware of the same code
445  // size as reported by perf.
446  unwinding_info_writer_.Finish(masm()->pc_offset());
447 
448  // Final alignment before starting on the metadata section.
449  masm()->Align(InstructionStream::kMetadataAlignment);
450 
451  safepoints()->Emit(masm(), frame()->GetTotalFrameSlotCount());
452 
453  // Emit the exception handler table.
454  if (!handlers_.empty()) {
455  handler_table_offset_ = HandlerTable::EmitReturnTableStart(masm());
456  for (size_t i = 0; i < handlers_.size(); ++i) {
457  int pos = handlers_[i].handler != nullptr ? handlers_[i].handler->pos()
458  : HandlerTable::kLazyDeopt;
459  HandlerTable::EmitReturnEntry(masm(), handlers_[i].pc_offset, pos);
460  }
461  }
462 
463  masm()->MaybeEmitOutOfLineConstantPool();
464  masm()->FinalizeJumpOptimizationInfo();
465 
466  result_ = kSuccess;
467 }

508 MaybeHandle<Code> CodeGenerator::FinalizeCode() {
509  if (result_ != kSuccess) {
510  masm()->AbortedCodeGeneration();
511  return {};
512  }
513 
514  // Allocate the source position table.
515  Handle<TrustedByteArray> source_positions =
516  source_position_table_builder_.ToSourcePositionTable(isolate());
517 
518  // Allocate and install the code.
519  CodeDesc desc;
520  masm()->GetCode(isolate()->main_thread_local_isolate(), &desc, safepoints(),
521  handler_table_offset_);
522 
523 #if defined(V8_OS_WIN64)
524  if (Builtins::IsBuiltinId(info_->builtin())) {
525  isolate_->SetBuiltinUnwindData(info_->builtin(), masm()->GetUnwindInfo());
526  }
527 #endif // V8_OS_WIN64
528 
529  if (unwinding_info_writer_.eh_frame_writer()) {
530  unwinding_info_writer_.eh_frame_writer()->GetEhFrame(&desc);
531  }
532 
533  Factory::CodeBuilder builder(isolate(), desc, info()->code_kind());
534  builder.set_builtin(info()->builtin())
535  .set_inlined_bytecode_size(info()->inlined_bytecode_size())
536  .set_parameter_count(parameter_count_)
537  .set_source_position_table(source_positions)
538  .set_is_turbofanned()
539  .set_stack_slots(frame()->GetTotalFrameSlotCount())
540  .set_profiler_data(info()->profiler_data())
541  .set_osr_offset(info()->osr_offset());
542 
543  if (info()->function_context_specializing()) {
544  builder.set_is_context_specialized();
545  }
546 
547  if (CodeKindUsesDeoptimizationData(info()->code_kind())) {
548  builder.set_deoptimization_data(GenerateDeoptimizationData());
549  DCHECK(info()->has_bytecode_array() ||
550  info()->code_kind() == CodeKind::WASM_FUNCTION);
551  }
552 
553  MaybeHandle<Code> maybe_code = builder.TryBuild();
554 
555  Handle<Code> code;
556  if (!maybe_code.ToHandle(&code)) {
557  masm()->AbortedCodeGeneration();
558  return {};
559  }
560 
561  LOG_CODE_EVENT(isolate(), CodeLinePosInfoRecordEvent(
562  code->instruction_start(), *source_positions,
563  JitCodeEvent::JIT_CODE));
564 
565  return code;
566 }

575 void CodeGenerator::RecordSafepoint(ReferenceMap* references, int pc_offset) {
576  auto safepoint = safepoints()->DefineSafepoint(masm(), pc_offset);
577 
578  for (int tagged : frame()->tagged_slots()) {
579  safepoint.DefineTaggedStackSlot(tagged);
580  }
581 
582  int frame_header_offset = frame()->GetFixedSlotCount();
583  for (const InstructionOperand& operand : references->reference_operands()) {
584  if (operand.IsStackSlot()) {
585  int index = LocationOperand::cast(operand).index();
586  DCHECK_LE(0, index);
587  // We might index values in the fixed part of the frame (i.e. the
588  // closure pointer or the context pointer); these are not spill slots
589  // and therefore don't work with the SafepointTable currently, but
590  // we also don't need to worry about them, since the GC has special
591  // knowledge about those fields anyway.
592  if (index < frame_header_offset) continue;
593  safepoint.DefineTaggedStackSlot(index);

638 void CodeGenerator::GetPushCompatibleMoves(Instruction* instr,
639  PushTypeFlags push_type,
640  ZoneVector<MoveOperands*>* pushes) {
641  static constexpr int first_push_compatible_index =
642  kReturnAddressStackSlotCount;
643  pushes->clear();
644  for (int i = Instruction::FIRST_GAP_POSITION;
645  i <= Instruction::LAST_GAP_POSITION; ++i) {
646  Instruction::GapPosition inner_pos =
647  static_cast<Instruction::GapPosition>(i);
648  ParallelMove* parallel_move = instr->GetParallelMove(inner_pos);
649  if (parallel_move != nullptr) {
650  for (auto move : *parallel_move) {
651  InstructionOperand source = move->source();
652  InstructionOperand destination = move->destination();
653  // If there are any moves from slots that will be overridden by pushes,
654  // then the full gap resolver must be used since optimization with
655  // pushes don't participate in the parallel move and might clobber
656  // values needed for the gap resolve.
657  if (source.IsAnyStackSlot() && LocationOperand::cast(source).index() >=
658  first_push_compatible_index) {
659  pushes->clear();
660  return;
661  }
662  // TODO(danno): Right now, only consider moves from the FIRST gap for
663  // pushes. Theoretically, we could extract pushes for both gaps (there
664  // are cases where this happens), but the logic for that would also have
665  // to check to make sure that non-memory inputs to the pushes from the
666  // LAST gap don't get clobbered in the FIRST gap.
667  if (i == Instruction::FIRST_GAP_POSITION) {
668  if (destination.IsStackSlot() &&
669  LocationOperand::cast(destination).index() >=
670  first_push_compatible_index) {
671  int index = LocationOperand::cast(destination).index();
672  if (IsValidPush(source, push_type)) {
673  if (index >= static_cast<int>(pushes->size())) {
674  pushes->resize(index + 1);
675  }
676  (*pushes)[index] = move;
677  }
678  }
679  }
680  }
681  }
682  }
683 
684  // For now, only support a set of continuous pushes at the end of the list.
685  size_t push_count_upper_bound = pushes->size();
686  size_t push_begin = push_count_upper_bound;
687  for (auto move : base::Reversed(*pushes)) {
688  if (move == nullptr) break;
689  push_begin--;
690  }
691  size_t push_count = pushes->size() - push_begin;
692  std::copy(pushes->begin() + push_begin,
693  pushes->begin() + push_begin + push_count, pushes->begin());
694  pushes->resize(push_count);
695 }

745 RpoNumber CodeGenerator::ComputeBranchInfo(BranchInfo* branch,
746  FlagsCondition condition,
747  Instruction* instr) {
748  // Assemble a branch after this instruction.
749  InstructionOperandConverter i(this, instr);
750  RpoNumber true_rpo =
751  i.InputRpo(instr->InputCount() - kBranchEndOffsetOfTrueBlock);
752  RpoNumber false_rpo =
753  i.InputRpo(instr->InputCount() - kBranchEndOffsetOfFalseBlock);
754 
755  if (true_rpo == false_rpo) {
756  return true_rpo;
757  }
758  if (IsNextInAssemblyOrder(true_rpo) || instructions()
759  ->InstructionBlockAt(false_rpo)
760  ->IsLoopHeaderInAssemblyOrder()) {
761  // true block is next, can fall through if condition negated.
762  // false block is loop header, can save one jump if condition negated.
763  std::swap(true_rpo, false_rpo);
764  condition = NegateFlagsCondition(condition);
765  }
766  branch->condition = condition;
767  branch->true_label = GetLabel(true_rpo);
768  branch->false_label = GetLabel(false_rpo);
769  branch->hinted = static_cast<bool>(BranchHintField::decode(instr->opcode()));
770  branch->fallthru = IsNextInAssemblyOrder(false_rpo);
771  return RpoNumber::Invalid();
772 }

774 CodeGenerator::CodeGenResult CodeGenerator::AssembleInstruction(
775  int instruction_index, const InstructionBlock* block) {
776  Instruction* instr = instructions()->InstructionAt(instruction_index);
777  if (info()->trace_turbo_json()) {
778  instr_starts_[instruction_index].gap_pc_offset = masm()->pc_offset();
779  }
780  int first_unused_stack_slot;
781  FlagsMode mode = FlagsModeField::decode(instr->opcode());
782  if (mode != kFlags_trap) {
783  AssembleSourcePosition(instr);
784  }
785  bool adjust_stack =
786  GetSlotAboveSPBeforeTailCall(instr, &first_unused_stack_slot);
787  if (adjust_stack) AssembleTailCallBeforeGap(instr, first_unused_stack_slot);
788  if (instr->opcode() == kArchNop && block->successors().empty() &&
789  block->code_end() - block->code_start() == 1) {
790  // When the frame-less dummy end block in Turbofan contains a Phi node,
791  // don't attempt to access spill slots.
792  // TODO(dmercadier): When the switch to Turboshaft is complete, this
793  // will no longer be required.
794  } else {
795  AssembleGaps(instr);
796  }
797  if (adjust_stack) AssembleTailCallAfterGap(instr, first_unused_stack_slot);
798  DCHECK_IMPLIES(
799  block->must_deconstruct_frame(),
800  instr != instructions()->InstructionAt(block->last_instruction_index()) ||
801  instr->IsRet() || instr->IsJump());
802  if (instr->IsJump() && block->must_deconstruct_frame()) {
803  AssembleDeconstructFrame();
804  }
805  if (info()->trace_turbo_json()) {
806  instr_starts_[instruction_index].arch_instr_pc_offset = masm()->pc_offset();
807  }
808  // Assemble architecture-specific code for the instruction.
809  CodeGenResult result = AssembleArchInstruction(instr);
810  if (result != kSuccess) return result;
811 
812  if (info()->trace_turbo_json()) {
813  instr_starts_[instruction_index].condition_pc_offset = masm()->pc_offset();
814  }
815 
816  FlagsCondition condition = FlagsConditionField::decode(instr->opcode());
817  switch (mode) {
818  case kFlags_branch:
819  case kFlags_conditional_branch: {
820  if (mode == kFlags_conditional_branch) {
821  InstructionOperandConverter i(this, instr);
822  condition = static_cast<FlagsCondition>(
823  i.ToConstant(instr->InputAt(instr->InputCount() -
824  kConditionalBranchEndOffsetOfCondition))
825  .ToInt64());
826  }
827  BranchInfo branch;
828  RpoNumber target = ComputeBranchInfo(&branch, condition, instr);
829  if (target.IsValid()) {
830  // redundant branch.
831  if (!IsNextInAssemblyOrder(target)) {
832  AssembleArchJump(target);
833  }
834  return kSuccess;
835  }
836  if (mode == kFlags_branch) {
837  // Assemble architecture-specific branch.
838  AssembleArchBranch(instr, &branch);
839  } else {
840  AssembleArchConditionalBranch(instr, &branch);
841  }
842  break;
843  }
844  case kFlags_deoptimize: {
845  // Assemble a conditional eager deoptimization after this instruction.
846  InstructionOperandConverter i(this, instr);
847  size_t frame_state_offset =
848  DeoptFrameStateOffsetField::decode(instr->opcode());
849  size_t immediate_args_count =
850  DeoptImmedArgsCountField::decode(instr->opcode());
851  DeoptimizationExit* const exit = AddDeoptimizationExit(
852  instr, frame_state_offset, immediate_args_count);
853  BranchInfo branch;
854  branch.condition = condition;
855  branch.true_label = exit->label();
856  branch.false_label = exit->continue_label();
857  branch.hinted = true;
858  branch.fallthru = true;
859  AssembleArchDeoptBranch(instr, &branch);
860  masm()->bind(exit->continue_label());
861  break;
862  }
863  case kFlags_set: {
864  // Assemble a boolean materialization after this instruction.
865  AssembleArchBoolean(instr, condition);
866  break;
867  }
868  case kFlags_conditional_set: {
869  // Assemble a conditional boolean materialization after this instruction.
870  AssembleArchConditionalBoolean(instr);
871  break;
872  }
873  case kFlags_select: {
874  AssembleArchSelect(instr, condition);
875  break;
876  }
877  case kFlags_trap: {
878 #if V8_ENABLE_WEBASSEMBLY
879  AssembleArchTrap(instr, condition);
880  break;
881 #else
882  UNREACHABLE();
883 #endif // V8_ENABLE_WEBASSEMBLY
884  }
885  case kFlags_none: {
886  break;
887  }
888  }
889 
890  return kSuccess;
891 }
