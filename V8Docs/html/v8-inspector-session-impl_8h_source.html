153  // On each call to "dispatchProtocolMessage", the session turns the weakThis
154  // reference into a strong one, so nested run loops are not able to fully
155  // deconstruct the V8 session until we return from the
156  // "dispatchProtocolMessage" call (i.e. no freed "this" remains on the stack).
157  class KeepSessionAliveScope {
158  public:
159  explicit KeepSessionAliveScope(const V8InspectorSessionImpl& session)

161 
162  private:

164  };
