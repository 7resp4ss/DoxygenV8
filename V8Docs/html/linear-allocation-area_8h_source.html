16 // A linear allocation area to allocate objects from.
17 //
18 // Invariant that must hold at all times:
19 // start <= top <= limit
20 class LinearAllocationArea final {
21  public:
22  LinearAllocationArea() = default;
23  LinearAllocationArea(Address top, Address limit)
24  : start_(top), top_(top), limit_(limit) {
25  Verify();
26  }
27 
28  void Reset(Address top, Address limit) {
29  start_ = top;
30  top_ = top;
31  limit_ = limit;
32  Verify();
33  }
34 
35  void ResetStart() { start_ = top_; }
36 
37  V8_INLINE bool CanIncrementTop(size_t bytes) const {
38  Verify();
39  return (top_ + bytes) <= limit_;
40  }
41 
42  V8_INLINE Address IncrementTop(size_t bytes) {
43  Address old_top = top_;
44  top_ += bytes;
45  Verify();
46  return old_top;
47  }
48 
49  V8_INLINE bool DecrementTopIfAdjacent(Address new_top, size_t bytes) {
50  Verify();
51  if ((new_top + bytes) == top_) {
52  top_ = new_top;
53  if (start_ > top_) {
54  ResetStart();
55  }
56  Verify();
57  return true;
58  }
59  return false;
60  }
61 
62  V8_INLINE void SetLimit(Address limit) {
63  limit_ = limit;
64  Verify();
65  }

68  Verify();

70  }

73  return top_;
74  }

78  }

84  void Verify() const {
85 #ifdef DEBUG
86  SLOW_DCHECK(start_ <= top_);
87  SLOW_DCHECK(top_ <= limit_);
88  if (V8_COMPRESS_POINTERS_8GB_BOOL) {
89  SLOW_DCHECK(IsAligned(top_, kObjectAlignment8GbHeap));
90  } else {
91  SLOW_DCHECK(IsAligned(top_, kObjectAlignment));
92  }
93 #endif // DEBUG
94  }
95 
96  static constexpr int kSize = 3 * kSystemPointerSize;
97 
98  private:
99  // The start of the LAB. Initially coincides with `top_`. As top is moved
100  // ahead, the area [start_, top_[ denotes a range of new objects. This range
101  // is reset with `ResetStart()`.
102  Address start_ = kNullAddress;
103  // The top of the LAB that is used for allocation.
104  Address top_ = kNullAddress;
105  // Limit of the LAB the denotes the end of the valid range for allocation.
106  Address limit_ = kNullAddress;
107 };
