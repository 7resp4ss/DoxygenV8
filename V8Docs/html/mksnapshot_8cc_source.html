38  void WriteSnapshot(v8::StartupData blob) const {
39  // TODO(crbug/633159): if we crash before the files have been fully created,
40  // we end up with a corrupted snapshot file. The build step would succeed,
41  // but the build target is unusable. Ideally we would write out temporary
42  // files and only move them to the final destination as last step.
43  v8::base::Vector<const uint8_t> blob_vector(
44  reinterpret_cast<const uint8_t*>(blob.data), blob.raw_size);
45  MaybeWriteSnapshotFile(blob_vector);
46  MaybeWriteStartupBlob(blob_vector);
47  }

49  private:
50  void MaybeWriteStartupBlob(v8::base::Vector<const uint8_t> blob) const {
51  if (!snapshot_blob_path_) return;
52 
53  FILE* fp = GetFileDescriptorOrDie(snapshot_blob_path_);
54  size_t written = fwrite(blob.begin(), 1, blob.length(), fp);
55  v8::base::Fclose(fp);
56  if (written != static_cast<size_t>(blob.length())) {
57  i::PrintF("Writing snapshot file failed.. Aborting.\n");
58  remove(snapshot_blob_path_);
59  exit(1);
60  }
61  }

63  void MaybeWriteSnapshotFile(v8::base::Vector<const uint8_t> blob) const {
64  if (!snapshot_cpp_path_) return;
65 
66  FILE* fp = GetFileDescriptorOrDie(snapshot_cpp_path_);
67 
68  WriteSnapshotFilePrefix(fp);
69  WriteSnapshotFileData(fp, blob);
70  WriteSnapshotFileSuffix(fp);
71 
72  v8::base::Fclose(fp);
73  }

99  static void WriteSnapshotFileData(FILE* fp,
100  v8::base::Vector<const uint8_t> blob) {
101  fprintf(
102  fp,
103  "alignas(kPointerAlignment) static const uint8_t blob_data[] = {\n");
104  WriteBinaryContentsAsCArray(fp, blob);
105  fprintf(fp, "};\n");
106  fprintf(fp, "static const int blob_size = %d;\n", blob.length());
107  fprintf(fp, "static const v8::StartupData blob =\n");
108  fprintf(fp, "{ (const char*) blob_data, blob_size };\n");
109  }

111  static void WriteBinaryContentsAsCArray(
112  FILE* fp, v8::base::Vector<const uint8_t> blob) {
113  for (int i = 0; i < blob.length(); i++) {
114  if ((i & 0x1F) == 0x1F) fprintf(fp, "\n");
115  if (i > 0) fprintf(fp, ",");
116  fprintf(fp, "%u", static_cast<unsigned char>(blob.at(i)));
117  }
118  fprintf(fp, "\n");
119  }

121  static FILE* GetFileDescriptorOrDie(const char* filename) {
122  FILE* fp = v8::base::OS::FOpen(filename, "wb");
123  if (fp == nullptr) {
124  i::PrintF("Unable to open file \"%s\" for writing.\n", filename);
125  exit(1);

134 std::unique_ptr<char[]> GetExtraCode(char* filename, const char* description) {
135  if (filename == nullptr || strlen(filename) == 0) return nullptr;
136  ::printf("Loading script for %s: %s\n", description, filename);
137  FILE* file = v8::base::OS::FOpen(filename, "rb");
138  if (file == nullptr) {
139  fprintf(stderr, "Failed to open '%s': errno %d\n", filename, errno);
140  exit(1);
141  }
142  fseek(file, 0, SEEK_END);
143  size_t size = ftell(file);
144  rewind(file);
145  char* chars = new char[size + 1];
146  chars[size] = '\0';
147  for (size_t i = 0; i < size;) {
148  size_t read = fread(&chars[i], 1, size - i, file);
149  if (ferror(file)) {
150  fprintf(stderr, "Failed to read '%s': errno %d\n", filename, errno);
151  exit(1);
152  }
153  i += read;
154  }
155  v8::base::Fclose(file);
156  return std::unique_ptr<char[]>(chars);
157 }

199 using CounterMap = std::map<std::string, int>;
200 CounterMap* counter_map_ = nullptr;

202 void MaybeSetCounterFunction(v8::Isolate* isolate) {
203  // If --native-code-counters is on then we enable all counters to make
204  // sure we generate code to increment them from the snapshot.
205  //
206  // Note: For the sake of the mksnapshot, the counter function must only
207  // return distinct addresses for each counter s.t. the serializer can properly
208  // distinguish between them. In theory it should be okay to just return an
209  // incremented int value each time this function is called, but we play it
210  // safe and return a real distinct memory location tied to every counter name.
211  if (i::v8_flags.native_code_counters) {
212  counter_map_ = new CounterMap();
213  isolate->SetCounterFunction([](const char* name) -> int* {
214  auto map_entry = counter_map_->find(name);
215  if (map_entry == counter_map_->end()) {
216  counter_map_->emplace(name, 0);
217  }
218  return &counter_map_->at(name);
219  });
220  }
221 }

225 int main(int argc, char** argv) {
226  v8::base::EnsureConsoleOutput();
227 
228  // Make mksnapshot runs predictable to create reproducible snapshots.
229  i::v8_flags.predictable = true;
230 
231  // Disable ICs globally in mksnapshot to avoid problems with Code handlers.
232  // See https://crbug.com/345280736.
233  // TODO(jgruber): Re-enable once a better fix is available.
234  i::v8_flags.use_ic = false;
235 
236  // Print the usage if an error occurs when parsing the command line
237  // flags or if the help flag is set.
238  using HelpOptions = i::FlagList::HelpOptions;
239  std::string usage = "Usage: " + std::string(argv[0]) +
240  " [--startup-src=file]" + " [--startup-blob=file]" +
241  " [--embedded-src=file]" + " [--embedded-variant=label]" +
242  " [--static-roots-src=file]" + " [--target-arch=arch]" +
243  " [--target-os=os] [extras]\n\n";
244  int result = i::FlagList::SetFlagsFromCommandLine(
245  &argc, argv, true, HelpOptions(HelpOptions::kExit, usage.c_str()));
246  if (result > 0 || (argc > 3)) {
247  i::PrintF(stdout, "%s", usage.c_str());
248  return result;
249  }
250 
251  i::CpuFeatures::Probe(true);
252  v8::V8::InitializeICUDefaultLocation(argv[0]);
253  std::unique_ptr<v8::Platform> platform = v8::platform::NewDefaultPlatform();
254  v8::V8::InitializePlatform(platform.get());
255  v8::V8::Initialize();
256 
257  {
258  SnapshotFileWriter snapshot_writer;
259  snapshot_writer.SetSnapshotFile(i::v8_flags.startup_src);
260  snapshot_writer.SetStartupBlobFile(i::v8_flags.startup_blob);
261 
262  i::EmbeddedFileWriter embedded_writer;
263  embedded_writer.SetEmbeddedFile(i::v8_flags.embedded_src);
264  embedded_writer.SetEmbeddedVariant(i::v8_flags.embedded_variant);
265  embedded_writer.SetTargetArch(i::v8_flags.target_arch);
266  embedded_writer.SetTargetOs(i::v8_flags.target_os);
267 
268  std::unique_ptr<char[]> embed_script =
269  GetExtraCode(argc >= 2 ? argv[1] : nullptr, "embedding");
270  std::unique_ptr<char[]> warmup_script =
271  GetExtraCode(argc >= 3 ? argv[2] : nullptr, "warm up");
272 
273  v8::StartupData blob;
274  {
275  v8::Isolate* isolate = v8::Isolate::Allocate();
276 
277  MaybeSetCounterFunction(isolate);
278 
279  // The isolate contains data from builtin compilation that needs
280  // to be written out if builtins are embedded.
281  i::Isolate* i_isolate = reinterpret_cast<i::Isolate*>(isolate);
282  i_isolate->RegisterEmbeddedFileWriter(&embedded_writer);
283 
284  std::unique_ptr<v8::ArrayBuffer::Allocator> array_buffer_allocator(
285  v8::ArrayBuffer::Allocator::NewDefaultAllocator());
286  v8::Isolate::CreateParams create_params;
287  create_params.array_buffer_allocator = array_buffer_allocator.get();
288 
289  // Set code range such that relative jumps for builtins to
290  // builtin calls in the snapshot are possible.
291  size_t code_range_size_mb =
292  i::kMaximalCodeRangeSize == 0
293  ? i::kMaxPCRelativeCodeRangeInMB
294  : std::min(i::kMaximalCodeRangeSize / i::MB,
295  i::kMaxPCRelativeCodeRangeInMB);
296  create_params.constraints.set_code_range_size_in_bytes(
297  code_range_size_mb * i::MB);
298 
299  {
300  v8::SnapshotCreator creator(isolate, create_params);
301 
302  blob = CreateSnapshotDataBlob(creator, embed_script.get());
303 
304  WriteEmbeddedFile(&embedded_writer);
305 
306 #if V8_STATIC_ROOTS_GENERATION_BOOL
307  if (i::v8_flags.static_roots_src) {
308  i::StaticRootsTableGen::write(i_isolate,
309  i::v8_flags.static_roots_src);
310  }
311 #endif
312 
313  if (i::v8_flags.builtins_effects_src) {
314  DCHECK_NOT_NULL(i_isolate->builtins_effects_analyzer());
315  i_isolate->builtins_effects_analyzer()->Write(
316  i::v8_flags.builtins_effects_src);
317  }
318  if (i_isolate->builtins_effects_analyzer() != nullptr) {
319  i_isolate->builtins_effects_analyzer()->TearDown(i_isolate);
320  }
321  }
322  isolate->Dispose();
323  }
324 
325  if (warmup_script) {
326  v8::StartupData cold = blob;
327  blob = WarmUpSnapshotDataBlob(cold, warmup_script.get());
328  delete[] cold.data;
329  }
330 
331  delete counter_map_;
332 
333  CHECK(blob.data);
334  snapshot_writer.WriteSnapshot(blob);
335  delete[] blob.data;
336  }
337 
338  v8::V8::Dispose();
339  v8::V8::DisposePlatform();
340  return 0;
341 }
