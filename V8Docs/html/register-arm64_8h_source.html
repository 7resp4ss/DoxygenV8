169  // These assertions ensure that the size and type of the register are as
170  // described. They do not consider the number of lanes that make up a vector.
171  // So, for example, Is8B() implies IsD(), and Is1D() implies IsD, but IsD()
172  // does not imply Is1D() or Is8B().
173  // Check the number of lanes, ie. the format of the vector, using methods such

220  static constexpr bool IsValidRegister(int code, int size) {
221  return (size == kWRegSizeInBits || size == kXRegSizeInBits) &&
222  (code < kNumberOfRegisters || code == kSPRegInternalCode);

224 
225  static constexpr bool IsValidVRegister(int code, int size) {
226  return (size == kBRegSizeInBits || size == kHRegSizeInBits ||
227  size == kSRegSizeInBits || size == kDRegSizeInBits ||
228  size == kQRegSizeInBits) &&
229  code < kNumberOfVRegisters;
230  }
231 
232  static constexpr bool IsValid(int code, int size, RegisterType type) {
233  return (type == kRegister && IsValidRegister(code, size)) ||
234  (type == kVRegister && IsValidVRegister(code, size));
235  }
236 
237  static constexpr bool IsNone(int code, int size, RegisterType type) {

300 // Functions for handling NEON vector format information.
301 enum VectorFormat {
302  kFormatUndefined = 0xffffffff,
303  kFormat8B = NEON_8B,
304  kFormat16B = NEON_16B,
305  kFormat4H = NEON_4H,
306  kFormat8H = NEON_8H,
307  kFormat2S = NEON_2S,
308  kFormat4S = NEON_4S,
309  kFormat1D = NEON_1D,
310  kFormat2D = NEON_2D,
311 
312  // Scalar formats. We add the scalar bit to distinguish between scalar and
313  // vector enumerations; the bit is always set in the encoding of scalar ops
314  // and always clear for vector ops. Although kFormatD and kFormat1D appear
315  // to be the same, their meaning is subtly different. The first is a scalar
316  // operation, the second a vector operation that only affects one lane.
317  kFormatB = NEON_B | NEONScalar,
318  kFormatH = NEON_H | NEONScalar,
319  kFormatS = NEON_S | NEONScalar,
320  kFormatD = NEON_D | NEONScalar,
321 
322  kFormat1Q = 0xfffffffd
323 };
