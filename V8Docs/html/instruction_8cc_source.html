92 bool InstructionOperand::InterferesWith(const InstructionOperand& other) const {
93  const bool combine_fp_aliasing = kFPAliasing == AliasingKind::kCombine &&
94  this->IsFPLocationOperand() &&
95  other.IsFPLocationOperand();
96  const bool stack_slots = this->IsAnyStackSlot() && other.IsAnyStackSlot();
97  if (!combine_fp_aliasing && !stack_slots) {
98  return EqualsCanonicalized(other);
99  }
100  const LocationOperand& loc = *LocationOperand::cast(this);
101  const LocationOperand& other_loc = LocationOperand::cast(other);
102  MachineRepresentation rep = loc.representation();
103  MachineRepresentation other_rep = other_loc.representation();
104  LocationOperand::LocationKind kind = loc.location_kind();
105  LocationOperand::LocationKind other_kind = other_loc.location_kind();
106  if (kind != other_kind) return false;
107 
108  if (combine_fp_aliasing && !stack_slots) {
109  if (rep == other_rep) return EqualsCanonicalized(other);
110  DCHECK_EQ(kind, LocationOperand::REGISTER);
111  // FP register-register interference.
112  return GetRegConfig()->AreAliases(rep, loc.register_code(), other_rep,
113  other_loc.register_code());
114  }
115 
116  DCHECK(stack_slots);
117  int num_slots =
118  AlignedSlotAllocator::NumSlotsForWidth(ElementSizeInBytes(rep));
119  int num_slots_other =
120  AlignedSlotAllocator::NumSlotsForWidth(ElementSizeInBytes(other_rep));
121  const bool complex_stack_slot_interference =
122  (num_slots > 1 || num_slots_other > 1);
123  if (!complex_stack_slot_interference) {
124  return EqualsCanonicalized(other);
125  }
126 
127  // Complex multi-slot operand interference:
128  // - slots of different FP reps can alias because the gap resolver may break a
129  // move into 2 or 4 equivalent smaller moves,
130  // - stack layout can be rearranged for tail calls
131  DCHECK_EQ(LocationOperand::STACK_SLOT, kind);
132  int index_hi = loc.index();
133  int index_lo =
134  index_hi -
135  AlignedSlotAllocator::NumSlotsForWidth(ElementSizeInBytes(rep)) + 1;
136  int other_index_hi = other_loc.index();
137  int other_index_lo =
138  other_index_hi -
139  AlignedSlotAllocator::NumSlotsForWidth(ElementSizeInBytes(other_rep)) + 1;
140  return other_index_hi >= index_lo && index_hi >= other_index_lo;
141 }

143 bool LocationOperand::IsCompatible(LocationOperand* op) {
144  if (IsRegister() || IsStackSlot()) {
145  return op->IsRegister() || op->IsStackSlot();
146  } else if (kFPAliasing != AliasingKind::kCombine) {
147  // A backend may choose to generate the same instruction sequence regardless
148  // of the FP representation. As a result, we can relax the compatibility and
149  // allow a Double to be moved in a Float for example. However, this is only
150  // allowed if registers do not overlap.
151  return (IsFPRegister() || IsFPStackSlot()) &&
152  (op->IsFPRegister() || op->IsFPStackSlot());
153  } else if (IsFloatRegister() || IsFloatStackSlot()) {
154  return op->IsFloatRegister() || op->IsFloatStackSlot();
155  } else if (IsDoubleRegister() || IsDoubleStackSlot()) {
156  return op->IsDoubleRegister() || op->IsDoubleStackSlot();
157  } else {
158  return (IsSimd128Register() || IsSimd128StackSlot()) &&
159  (op->IsSimd128Register() || op->IsSimd128StackSlot());
160  }
161 }

335 void ParallelMove::PrepareInsertAfter(
336  MoveOperands* move, ZoneVector<MoveOperands*>* to_eliminate) const {
337  bool no_aliasing = kFPAliasing != AliasingKind::kCombine ||
338  !move->destination().IsFPLocationOperand();
339  MoveOperands* replacement = nullptr;
340  MoveOperands* eliminated = nullptr;
341  for (MoveOperands* curr : *this) {
342  if (curr->IsEliminated()) continue;
343  if (curr->destination().EqualsCanonicalized(move->source())) {
344  // We must replace move's source with curr's destination in order to
345  // insert it into this ParallelMove.
346  DCHECK(!replacement);
347  replacement = curr;
348  if (no_aliasing && eliminated != nullptr) break;
349  } else if (curr->destination().InterferesWith(move->destination())) {
350  // We can eliminate curr, since move overwrites at least a part of its
351  // destination, implying its value is no longer live.
352  eliminated = curr;
353  to_eliminate->push_back(curr);
354  if (no_aliasing && replacement != nullptr) break;
355  }
356  }
357  if (replacement != nullptr) move->set_source(replacement->source());
358 }

717 static InstructionBlock* InstructionBlockFor(Zone* zone,
718  const BasicBlock* block) {
719  bool is_handler =
720  !block->empty() && block->front()->opcode() == IrOpcode::kIfException;
721  InstructionBlock* instr_block = zone->New<InstructionBlock>(
722  zone, GetRpo(block), GetRpo(block->loop_header()), GetLoopEndRpo(block),
723  GetRpo(block->dominator()), block->deferred(), is_handler);
724  // Map successors and predecessors
725  instr_block->successors().reserve(block->SuccessorCount());
726  for (BasicBlock* successor : block->successors()) {
727  instr_block->successors().push_back(GetRpo(successor));
728  }
729  instr_block->predecessors().reserve(block->PredecessorCount());
730  for (BasicBlock* predecessor : block->predecessors()) {
731  instr_block->predecessors().push_back(GetRpo(predecessor));
732  }
733  return instr_block;
734 }

736 static InstructionBlock* InstructionBlockFor(
737  Zone* zone, const turboshaft::Graph& graph, const turboshaft::Block* block,
738  const turboshaft::Block* loop_header) {
739  bool is_handler =
740  block->FirstOperation(graph).Is<turboshaft::CatchBlockBeginOp>();
741  bool deferred = block->get_custom_data(
742  turboshaft::Block::CustomDataKind::kDeferredInSchedule);
743  InstructionBlock* instr_block = zone->New<InstructionBlock>(
744  zone, GetRpo(block), GetRpo(loop_header), GetLoopEndRpo(block),
745  GetRpo(block->GetDominator()), deferred, is_handler);
746  // Map successors and predecessors.
747  base::SmallVector<turboshaft::Block*, 4> succs =
748  turboshaft::SuccessorBlocks(block->LastOperation(graph));
749  instr_block->successors().reserve(succs.size());
750  for (const turboshaft::Block* successor : succs) {
751  instr_block->successors().push_back(GetRpo(successor));
752  }
753  instr_block->predecessors().reserve(block->PredecessorCount());
754  for (const turboshaft::Block* predecessor = block->LastPredecessor();
755  predecessor; predecessor = predecessor->NeighboringPredecessor()) {
756  instr_block->predecessors().push_back(GetRpo(predecessor));
757  }
758  std::reverse(instr_block->predecessors().begin(),
759  instr_block->predecessors().end());
760  return instr_block;
761 }

905 void InstructionSequence::ComputeAssemblyOrder() {
906  int ao = 0;
907  RpoNumber invalid = RpoNumber::Invalid();
908 
909  ao_blocks_ = zone()->AllocateArray<InstructionBlocks>(1);
910  new (ao_blocks_) InstructionBlocks(zone());
911  ao_blocks_->reserve(instruction_blocks_->size());
912 
913  // Place non-deferred blocks.
914  for (InstructionBlock* const block : *instruction_blocks_) {
915  DCHECK_NOT_NULL(block);
916  if (block->IsDeferred()) continue; // skip deferred blocks.
917  if (block->ao_number() != invalid) continue; // loop rotated.
918  if (block->IsLoopHeader()) {
919  bool header_align = true;
920  if (v8_flags.turbo_loop_rotation) {
921  // Perform loop rotation for non-deferred loops.
922  InstructionBlock* loop_end =
923  instruction_blocks_->at(block->loop_end().ToSize() - 1);
924  if (!loop_end->IsDeferred() && /* Ignore deferred loop ends */
925  loop_end->SuccessorCount() == 1 && /* ends with goto */
926  loop_end != block /* not a degenerate infinite loop */) {
927  // If the last block has an unconditional jump back to the header,
928  // then move it to be in front of the header in the assembly order.
929  DCHECK_EQ(block->rpo_number(), loop_end->successors()[0]);
930  loop_end->set_ao_number(RpoNumber::FromInt(ao++));
931  ao_blocks_->push_back(loop_end);
932  // This block will be the new machine-level loop header, so align
933  // this block instead of the loop header block.
934  loop_end->set_loop_header_alignment(true);
935  header_align = false;
936  }
937  }
938  block->set_loop_header_alignment(header_align);
939  }
940  if (block->loop_header().IsValid() && block->IsTableSwitchTarget()) {
941  block->set_code_target_alignment(true);
942  }
943  block->set_ao_number(RpoNumber::FromInt(ao++));
944  ao_blocks_->push_back(block);
945  }
946  // Add all leftover (deferred) blocks.
947  for (InstructionBlock* const block : *instruction_blocks_) {
948  if (block->ao_number() == invalid) {
949  block->set_ao_number(RpoNumber::FromInt(ao++));
950  ao_blocks_->push_back(block);
951  }
952  }
953  DCHECK_EQ(instruction_blocks_->size(), ao);
954 }
