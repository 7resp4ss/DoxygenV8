47 std::optional<Builtin> TryGetBuiltinId(const ConstantOp* target,
48  JSHeapBroker* broker) {
49  if (!target) return std::nullopt;
50  if (target->kind != ConstantOp::Kind::kHeapObject) return std::nullopt;
51  // TODO(nicohartmann@): For builtin compilation we don't have a broker. We
52  // could try to access the heap directly instead.
53  if (broker == nullptr) return std::nullopt;
54  UnparkedScopeIfNeeded scope(broker);
55  AllowHandleDereference allow_handle_dereference;
56  HeapObjectRef ref = MakeRef(broker, target->handle());
57  if (ref.IsCode()) {
58  CodeRef code = ref.AsCode();
59  if (code.object()->is_builtin()) {
60  return code.object()->builtin_id();
61  }
62  }
63  return std::nullopt;
64 }

75 void ValidateOpInputRep(
76  const Graph& graph, OpIndex input,
77  std::initializer_list<RegisterRepresentation> expected_reps,
78  const Operation* checked_op, std::optional<size_t> projection_index) {
79  const Operation& input_op = graph.Get(input);
80  base::Vector<const RegisterRepresentation> input_reps =
81  input_op.outputs_rep();
82  RegisterRepresentation input_rep;
83  if (projection_index) {
84  if (*projection_index < input_reps.size()) {
85  input_rep = input_reps[*projection_index];
86  } else {
87  std::cerr << "Turboshaft operation ";
88  if (checked_op) {
89  std::cerr << *checked_op << " ";
90  }
91  std::cerr << "has input #" << input << ":" << input_op
92  << " with wrong arity.\n";
93  std::cerr << "Input has results " << PrintCollection(input_reps)
94  << ", but expected at least " << (*projection_index + 1)
95  << " results.\n";
96  UNREACHABLE();
97  }
98  } else if (input_reps.size() == 1) {
99  input_rep = input_reps[0];
100  } else {
101  std::cerr << "Turboshaft operation ";
102  if (checked_op) {
103  std::cerr << *checked_op << " ";
104  }
105  std::cerr << "has input #" << input << ":" << input_op
106  << " with wrong arity.\n";
107  std::cerr << "Expected a single output but found " << input_reps.size()
108  << ".\n";
109  UNREACHABLE();
110  }
111  for (RegisterRepresentation expected_rep : expected_reps) {
112  if (input_rep.AllowImplicitRepresentationChangeTo(
113  expected_rep, graph.IsCreatedFromTurbofan())) {
114  return;
115  }
116  }
117  std::cerr << "Turboshaft operation ";
118  if (checked_op) {
119  std::cerr << *checked_op << " ";
120  }
121  std::cerr << "has input #" << input << ":" << input_op
122  << " with wrong representation.\n";
123  std::cerr << "Expected " << (expected_reps.size() > 1 ? "one of " : "")
124  << PrintCollection(expected_reps).WithoutBrackets() << " but found "
125  << input_rep << ".\n";
126  std::cout << "Input: " << graph.Get(input) << "\n";
127  UNREACHABLE();
128 }
