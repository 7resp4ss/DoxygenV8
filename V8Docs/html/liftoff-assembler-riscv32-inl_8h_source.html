16 // Liftoff Frames.
17 //
18 // slot Frame
19 // +--------------------+---------------------------
20 // n+4 | optional padding slot to keep the stack 16 byte aligned.
21 // n+3 | parameter n |
22 // ... | ... |
23 // 4 | parameter 1 | or parameter 2
24 // 3 | parameter 0 | or parameter 1
25 // 2 | (result address) | or parameter 0
26 // -----+--------------------+---------------------------
27 // 1 | return addr (ra) |
28 // 0 | previous frame (fp)|
29 // -----+--------------------+ <-- frame ptr (fp)
30 // -1 | StackFrame::WASM |
31 // -2 | instance |
32 // -3 | feedback vector|
33 // -----+--------------------+---------------------------
34 // -4 | slot 0 | ^
35 // -5 | slot 1 | |
36 // | | Frame slots
37 // | | |
38 // | | v
39 // | optional padding slot to keep the stack 16 byte aligned.
40 // -----+--------------------+ <-- stack ptr (sp)
41 //

57 inline MemOperand GetMemOp(LiftoffAssembler* assm, Register addr,
58  Register offset, uintptr_t offset_imm,
59  unsigned shift_amount = 0) {
60  DCHECK_NE(addr, kScratchReg2);
61  DCHECK_NE(offset, kScratchReg2);
62  if (is_uint31(offset_imm)) {
63  int32_t offset_imm32 = static_cast<int32_t>(offset_imm);
64  if (offset == no_reg) return MemOperand(addr, offset_imm32);
65  if (shift_amount != 0) {
66  assm->CalcScaledAddress(kScratchReg2, addr, offset, shift_amount);
67  } else {
68  assm->AddWord(kScratchReg2, offset, addr);
69  }
70  return MemOperand(kScratchReg2, offset_imm32);
71  }
72  // Offset immediate does not fit in 31 bits.
73  assm->li(kScratchReg2, offset_imm);
74  assm->AddWord(kScratchReg2, kScratchReg2, addr);
75  if (offset != no_reg) {
76  if (shift_amount != 0) {
77  assm->CalcScaledAddress(kScratchReg2, kScratchReg2, offset, shift_amount);
78  } else {
79  assm->AddWord(kScratchReg2, kScratchReg2, offset);
80  }
81  }
82  return MemOperand(kScratchReg2, 0);
83 }

85 inline void Load(LiftoffAssembler* assm, LiftoffRegister dst, Register base,
86  int32_t offset, ValueKind kind) {
87  MemOperand src(base, offset);
88 
89  switch (kind) {
90  case kI32:
91  case kRef:
92  case kRefNull:
93  assm->Lw(dst.gp(), src);
94  break;
95  case kI64:
96  assm->Lw(dst.low_gp(),
97  MemOperand(base, offset + liftoff::kLowWordOffset));
98  assm->Lw(dst.high_gp(),
99  MemOperand(base, offset + liftoff::kHighWordOffset));
100  break;
101  case kF32:
102  assm->LoadFloat(dst.fp(), src);
103  break;
104  case kF64:
105  assm->LoadDouble(dst.fp(), src);
106  break;
107  case kS128:{
108  assm->VU.set(kScratchReg, E8, m1);
109  Register src_reg = src.offset() == 0 ? src.rm() : kScratchReg;
110  if (src.offset() != 0) {
111  assm->AddWord(src_reg, src.rm(), src.offset());
112  }
113  assm->vl(dst.fp().toV(), src_reg, 0, E8);
114  break;
115  }
116  default:
117  UNREACHABLE();
118  }
119 }

121 inline void Store(LiftoffAssembler* assm, Register base, int32_t offset,
122  LiftoffRegister src, ValueKind kind) {
123  MemOperand dst(base, offset);
124  switch (kind) {
125  case kI32:
126  case kRefNull:
127  case kRef:
128  assm->Sw(src.gp(), dst);
129  break;
130  case kI64:
131  assm->Sw(src.low_gp(),
132  MemOperand(base, offset + liftoff::kLowWordOffset));
133  assm->Sw(src.high_gp(),
134  MemOperand(base, offset + liftoff::kHighWordOffset));
135  break;
136  case kF32:
137  assm->StoreFloat(src.fp(), dst);
138  break;
139  case kF64:
140  assm->StoreDouble(src.fp(), dst);
141  break;
142  case kS128:{
143  assm->VU.set(kScratchReg, E8, m1);
144  Register dst_reg = dst.offset() == 0 ? dst.rm() : kScratchReg;
145  if (dst.offset() != 0) {
146  assm->AddWord(kScratchReg, dst.rm(), dst.offset());
147  }
148  assm->vs(src.fp().toV(), dst_reg, 0, VSew::E8);
149  break;
150  }
151  default:
152  UNREACHABLE();

156 inline void push(LiftoffAssembler* assm, LiftoffRegister reg, ValueKind kind) {
157  switch (kind) {
158  case kI32:
159  case kRefNull:
160  case kRef:
161  assm->addi(sp, sp, -kSystemPointerSize);
162  assm->Sw(reg.gp(), MemOperand(sp, 0));
163  break;
164  case kI64:
165  assm->Push(reg.high_gp(), reg.low_gp());
166  break;
167  case kF32:
168  assm->addi(sp, sp, -kSystemPointerSize);
169  assm->StoreFloat(reg.fp(), MemOperand(sp, 0));
170  break;
171  case kF64:
172  assm->addi(sp, sp, -kDoubleSize);
173  assm->StoreDouble(reg.fp(), MemOperand(sp, 0));
174  break;
175  case kS128:{
176  assm->VU.set(kScratchReg, E8, m1);
177  assm->addi(sp, sp, -kSystemPointerSize * 4);
178  assm->vs(reg.fp().toV(), sp, 0, VSew::E8);
179  break;
180  }
181  default:
182  UNREACHABLE();

257 void LiftoffAssembler::StoreTaggedPointer(Register dst_addr,
258  Register offset_reg,
259  int32_t offset_imm, Register src,
260  LiftoffRegList pinned,
261  uint32_t* protected_store_pc,
262  SkipWriteBarrier skip_write_barrier) {
263  static_assert(kTaggedSize == kInt32Size);
264  UseScratchRegisterScope temps{this};
265  Register actual_offset_reg = offset_reg;
266  if (offset_reg != no_reg && offset_imm != 0) {
267  if (cache_state()->is_used(LiftoffRegister(offset_reg))) {
268  // The code below only needs a scratch register if the {MemOperand} given
269  // to {str} has an offset outside the uint12 range. After doing the
270  // addition below we will not pass an immediate offset to {str} though, so
271  // we can use the scratch register here.
272  actual_offset_reg = temps.Acquire();
273  }
274  Add32(actual_offset_reg, offset_reg, Operand(offset_imm));
275  }
276  MemOperand dst_op = MemOperand(kScratchReg, 0);
277  if (actual_offset_reg == no_reg) {
278  dst_op = MemOperand(dst_addr, offset_imm);
279  } else {
280  AddWord(kScratchReg, dst_addr, actual_offset_reg);
281  dst_op = MemOperand(kScratchReg, 0);
282  }
283  auto trapper = [protected_store_pc](int offset) {
284  if (protected_store_pc) *protected_store_pc = static_cast<uint32_t>(offset);
285  };
286  StoreWord(src, dst_op, trapper);
287  if (protected_store_pc) {
288  DCHECK(InstructionAt(*protected_store_pc)->IsStore());
289  }
290 
291  if (skip_write_barrier || v8_flags.disable_write_barriers) return;
292 
293  // The write barrier.
294  Label exit;
295  CheckPageFlag(dst_addr, MemoryChunk::kPointersFromHereAreInterestingMask,
296  kZero, &exit);
297  JumpIfSmi(src, &exit);
298  CheckPageFlag(src, MemoryChunk::kPointersToHereAreInterestingMask, eq, &exit);
299  CallRecordWriteStubSaveRegisters(
300  dst_addr,
301  actual_offset_reg == no_reg ? Operand(offset_imm)
302  : Operand(actual_offset_reg),
303  SaveFPRegsMode::kSave, StubCallMode::kCallWasmRuntimeStub);
304  bind(&exit);
305 }

315 void LiftoffAssembler::Load(LiftoffRegister dst, Register src_addr,
316  Register offset_reg, uintptr_t offset_imm,
317  LoadType type, uint32_t* protected_load_pc,
318  bool /* is_load_mem */, bool /* i64_offset */,
319  bool needs_shift) {
320  unsigned shift_amount = needs_shift ? type.size_log_2() : 0;
321  MemOperand src_op =
322  liftoff::GetMemOp(this, src_addr, offset_reg, offset_imm, shift_amount);
323  auto trapper = [protected_load_pc](int offset) {
324  if (protected_load_pc) *protected_load_pc = static_cast<uint32_t>(offset);
325  };
326  switch (type.value()) {
327  case LoadType::kI32Load8U:
328  Lbu(dst.gp(), src_op, trapper);
329  break;
330  case LoadType::kI64Load8U:
331  Lbu(dst.low_gp(), src_op, trapper);
332  mv(dst.high_gp(), zero_reg);
333  break;
334  case LoadType::kI32Load8S:
335  Lb(dst.gp(), src_op, trapper);
336  break;
337  case LoadType::kI64Load8S:
338  Lb(dst.low_gp(), src_op, trapper);
339  srai(dst.high_gp(), dst.low_gp(), 31);
340  break;
341  case LoadType::kI32Load16U:
342  Lhu(dst.gp(), src_op, trapper);
343  break;
344  case LoadType::kI64Load16U:
345  Lhu(dst.low_gp(), src_op, trapper);
346  mv(dst.high_gp(), zero_reg);
347  break;
348  case LoadType::kI32Load16S:
349  Lh(dst.gp(), src_op, trapper);
350  break;
351  case LoadType::kI64Load16S:
352  Lh(dst.low_gp(), src_op, trapper);
353  srai(dst.high_gp(), dst.low_gp(), 31);
354  break;
355  case LoadType::kI64Load32U:
356  Lw(dst.low_gp(), src_op, trapper);
357  mv(dst.high_gp(), zero_reg);
358  break;
359  case LoadType::kI64Load32S:
360  Lw(dst.low_gp(), src_op, trapper);
361  srai(dst.high_gp(), dst.low_gp(), 31);
362  break;
363  case LoadType::kI32Load:
364  Lw(dst.gp(), src_op, trapper);
365  break;
366  case LoadType::kI64Load: {
367  Lw(dst.low_gp(), src_op, trapper);
368  src_op = liftoff::GetMemOp(this, src_addr, offset_reg,
369  offset_imm + kSystemPointerSize);
370  Lw(dst.high_gp(), src_op);
371  } break;
372  case LoadType::kF32Load:
373  LoadFloat(dst.fp(), src_op, trapper);
374  break;
375  case LoadType::kF64Load:
376  LoadDouble(dst.fp(), src_op, trapper);
377  break;
378  case LoadType::kS128Load: {
379  VU.set(kScratchReg, E8, m1);
380  Register src_reg = src_op.offset() == 0 ? src_op.rm() : kScratchReg;
381  if (src_op.offset() != 0) {
382  AddWord(src_reg, src_op.rm(), src_op.offset());
383  }
384  trapper(pc_offset());
385  vl(dst.fp().toV(), src_reg, 0, E8);
386  break;
387  }
388  case LoadType::kF32LoadF16:
389  UNIMPLEMENTED();
390  break;
391  default:
392  UNREACHABLE();
393  }
394  if (protected_load_pc) {
395  DCHECK(InstructionAt(*protected_load_pc)->IsLoad());
396  }
397 
398 #if defined(V8_TARGET_BIG_ENDIAN)
399  if (is_load_mem) {
400  pinned.set(src_op.rm());
401  liftoff::ChangeEndiannessLoad(this, dst, type, pinned);
402  }
403 #endif
404 }

406 void LiftoffAssembler::Store(Register dst_addr, Register offset_reg,
407  uintptr_t offset_imm, LiftoffRegister src,
408  StoreType type, LiftoffRegList pinned,
409  uint32_t* protected_store_pc, bool is_store_mem,
410  bool i64_offset) {
411  MemOperand dst_op = liftoff::GetMemOp(this, dst_addr, offset_reg, offset_imm);
412 
413 #if defined(V8_TARGET_BIG_ENDIAN)
414  if (is_store_mem) {
415  pinned.set(dst_op.rm());
416  LiftoffRegister tmp = GetUnusedRegister(src.reg_class(), pinned);
417  // Save original value.
418  Move(tmp, src, type.value_type());
419 
420  src = tmp;
421  pinned.set(tmp);
422  liftoff::ChangeEndiannessStore(this, src, type, pinned);
423  }
424 #endif
425  auto trapper = [protected_store_pc](int offset) {
426  if (protected_store_pc) *protected_store_pc = static_cast<uint32_t>(offset);
427  };
428  switch (type.value()) {
429  case StoreType::kI32Store8:
430  Sb(src.gp(), dst_op, trapper);
431  break;
432  case StoreType::kI64Store8:
433  Sb(src.low_gp(), dst_op, trapper);
434  break;
435  case StoreType::kI32Store16:
436  Sh(src.gp(), dst_op, trapper);
437  break;
438  case StoreType::kI64Store16:
439  Sh(src.low_gp(), dst_op, trapper);
440  break;
441  case StoreType::kI32Store:
442  Sw(src.gp(), dst_op, trapper);
443  break;
444  case StoreType::kI64Store32:
445  Sw(src.low_gp(), dst_op, trapper);
446  break;
447  case StoreType::kI64Store: {
448  Sw(src.low_gp(), dst_op, trapper);
449  dst_op = liftoff::GetMemOp(this, dst_addr, offset_reg,
450  offset_imm + kSystemPointerSize);
451  Sw(src.high_gp(), dst_op, trapper);
452  break;
453  }
454  case StoreType::kF32Store:
455  StoreFloat(src.fp(), dst_op, trapper);
456  break;
457  case StoreType::kF64Store:
458  StoreDouble(src.fp(), dst_op, trapper);
459  break;
460  case StoreType::kS128Store: {
461  VU.set(kScratchReg, E8, m1);
462  Register dst_reg = dst_op.offset() == 0 ? dst_op.rm() : kScratchReg;
463  if (dst_op.offset() != 0) {
464  AddWord(kScratchReg, dst_op.rm(), dst_op.offset());
465  }
466  trapper(pc_offset());
467  vs(src.fp().toV(), dst_reg, 0, VSew::E8);
468  break;
469  }
470  default:
471  UNREACHABLE();
472  }
473  if (protected_store_pc) {
474  DCHECK(InstructionAt(*protected_store_pc)->IsStore());
475  }
476 }

481 inline Register CalculateActualAddress(LiftoffAssembler* lasm,
482  UseScratchRegisterScope& temps,
483  Register addr_reg, Register offset_reg,
484  uintptr_t offset_imm,
485  Register result_reg = no_reg) {
486  if (offset_reg == no_reg && offset_imm == 0) {
487  if (result_reg == addr_reg || result_reg == no_reg) return addr_reg;
488  lasm->mv(result_reg, addr_reg);
489  return result_reg;
490  }
491  if (result_reg == no_reg) result_reg = temps.Acquire();
492  if (offset_reg == no_reg) {
493  lasm->AddWord(result_reg, addr_reg, Operand(offset_imm));
494  } else {
495  lasm->AddWord(result_reg, addr_reg, Operand(offset_reg));
496  if (offset_imm != 0)
497  lasm->AddWord(result_reg, result_reg, Operand(offset_imm));
498  }
499  return result_reg;
500 }

502 enum class Binop { kAdd, kSub, kAnd, kOr, kXor, kExchange };
503 inline void AtomicBinop64(LiftoffAssembler* lasm, Register dst_addr,
504  Register offset_reg, uintptr_t offset_imm,
505  LiftoffRegister value, LiftoffRegister result,
506  StoreType type, Binop op) {
507  ASM_CODE_COMMENT(lasm);
508  FrameScope scope(lasm, StackFrame::MANUAL);
509  RegList c_params = {kCArgRegs[0], kCArgRegs[1], kCArgRegs[2]};
510  RegList result_list = {result.low_gp(), result.high_gp()};
511  // Result registers does not need to be pushed.
512  __ MultiPush(c_params - result_list);
513  UseScratchRegisterScope temps(lasm);
514  liftoff::CalculateActualAddress(lasm, temps, dst_addr, offset_reg, offset_imm,
515  kScratchReg);
516  __ Mv(kCArgRegs[1], value.low_gp());
517  __ Mv(kCArgRegs[2], value.high_gp());
518  __ Mv(kCArgRegs[0], kScratchReg);
519  __ MultiPush(kJSCallerSaved - c_params - result_list);
520  __ PrepareCallCFunction(3, 0, kScratchReg);
521  ExternalReference extern_func_ref;
522  switch (op) {
523  case Binop::kAdd:
524  extern_func_ref = ExternalReference::atomic_pair_add_function();
525  break;
526  case Binop::kSub:
527  extern_func_ref = ExternalReference::atomic_pair_sub_function();
528  break;
529  case Binop::kAnd:
530  extern_func_ref = ExternalReference::atomic_pair_and_function();
531  break;
532  case Binop::kOr:
533  extern_func_ref = ExternalReference::atomic_pair_or_function();
534  break;
535  case Binop::kXor:
536  extern_func_ref = ExternalReference::atomic_pair_xor_function();
537  break;
538  case Binop::kExchange:
539  extern_func_ref = ExternalReference::atomic_pair_exchange_function();
540  break;
541  default:
542  UNREACHABLE();
543  }
544  __ CallCFunction(extern_func_ref, 3, 0);
545  __ MultiPop(kJSCallerSaved - c_params - result_list);
546  __ Mv(result.low_gp(), kReturnRegister0);
547  __ Mv(result.high_gp(), kReturnRegister1);
548  __ MultiPop(c_params - result_list);
549  return;
550 }

552 inline void AtomicBinop(LiftoffAssembler* lasm, Register dst_addr,
553  Register offset_reg, uintptr_t offset_imm,
554  LiftoffRegister value, LiftoffRegister result,
555  StoreType type, Binop op) {
556  LiftoffRegList pinned{dst_addr, value, result};
557  if (offset_reg != no_reg) pinned.set(offset_reg);
558  Register store_result = pinned.set(__ GetUnusedRegister(kGpReg, pinned)).gp();
559 
560  // Make sure that {result} is unique.
561  Register result_reg = no_reg;
562  Register value_reg = no_reg;
563  bool change_result = false;
564  switch (type.value()) {
565  case StoreType::kI64Store8:
566  case StoreType::kI64Store16:
567  __ LoadConstant(result.high(), WasmValue(0));
568  result_reg = result.low_gp();
569  value_reg = value.low_gp();
570  break;
571  case StoreType::kI32Store8:
572  case StoreType::kI32Store16:
573  result_reg = result.gp();
574  value_reg = value.gp();
575  break;
576  default:
577  UNREACHABLE();
578  }
579  if (result_reg == value_reg || result_reg == dst_addr ||
580  result_reg == offset_reg) {
581  result_reg = __ GetUnusedRegister(kGpReg, pinned).gp();
582  change_result = true;
583  }
584 
585  UseScratchRegisterScope temps(lasm);
586  Register actual_addr = liftoff::CalculateActualAddress(
587  lasm, temps, dst_addr, offset_reg, offset_imm);
588 
589  // Allocate an additional {temp} register to hold the result that should be
590  // stored to memory. Note that {temp} and {store_result} are not allowed to be
591  // the same register.
592  Register temp = temps.Acquire();
593 
594  Label retry;
595  __ bind(&retry);
596  switch (type.value()) {
597  case StoreType::kI64Store8:
598  case StoreType::kI32Store8:
599  __ lbu(result_reg, actual_addr, 0);
600  __ sync();
601  break;
602  case StoreType::kI64Store16:
603  case StoreType::kI32Store16:
604  __ lhu(result_reg, actual_addr, 0);
605  __ sync();
606  break;
607  case StoreType::kI64Store32:
608  case StoreType::kI32Store:
609  __ lr_w(true, false, result_reg, actual_addr);
610  break;
611  default:
612  UNREACHABLE();
613  }
614 
615  switch (op) {
616  case Binop::kAdd:
617  __ add(temp, result_reg, value_reg);
618  break;
619  case Binop::kSub:
620  __ sub(temp, result_reg, value_reg);
621  break;
622  case Binop::kAnd:
623  __ and_(temp, result_reg, value_reg);
624  break;
625  case Binop::kOr:
626  __ or_(temp, result_reg, value_reg);
627  break;
628  case Binop::kXor:
629  __ xor_(temp, result_reg, value_reg);
630  break;
631  case Binop::kExchange:
632  __ mv(temp, value_reg);
633  break;
634  }
635  switch (type.value()) {
636  case StoreType::kI64Store8:
637  case StoreType::kI32Store8:
638  __ sync();
639  __ sb(temp, actual_addr, 0);
640  __ sync();
641  __ mv(store_result, zero_reg);
642  break;
643  case StoreType::kI64Store16:
644  case StoreType::kI32Store16:
645  __ sync();
646  __ sh(temp, actual_addr, 0);
647  __ sync();
648  __ mv(store_result, zero_reg);
649  break;
650  case StoreType::kI64Store32:
651  case StoreType::kI32Store:
652  __ sc_w(false, true, store_result, actual_addr, temp);
653  break;
654  default:
655  UNREACHABLE();
656  }
657 
658  __ bnez(store_result, &retry);
659  if (change_result) {
660  switch (type.value()) {
661  case StoreType::kI64Store8:
662  case StoreType::kI64Store16:
663  case StoreType::kI64Store32:
664  __ mv(result.low_gp(), result_reg);
665  break;
666  case StoreType::kI32Store8:
667  case StoreType::kI32Store16:
668  case StoreType::kI32Store:
669  __ mv(result.gp(), result_reg);
670  break;
671  default:
672  UNREACHABLE();
673  }
674  }
675 }

991 void LiftoffAssembler::AtomicCompareExchange(
992  Register dst_addr, Register offset_reg, uintptr_t offset_imm,
993  LiftoffRegister expected, LiftoffRegister new_value, LiftoffRegister result,
994  StoreType type, bool i64_offset) {
995  ASM_CODE_COMMENT(this);
996  LiftoffRegList pinned{dst_addr, expected, new_value, result};
997  if (offset_reg != no_reg) pinned.set(offset_reg);
998 
999  if (type.value() == StoreType::kI64Store) {
1000  UseScratchRegisterScope temps(this);
1001  Register actual_addr = liftoff::CalculateActualAddress(
1002  this, temps, dst_addr, offset_reg, offset_imm, kScratchReg);
1003  FrameScope scope(this, StackFrame::MANUAL);
1004  // NOTE:
1005  // a0~a4 are caller-saved registers and also used
1006  // to pass parameters for C functions.
1007  RegList c_params = {kCArgRegs[0], kCArgRegs[1], kCArgRegs[2], kCArgRegs[3],
1008  a4};
1009  RegList result_list = {result.low_gp(), result.high_gp()};
1010  MultiPush(c_params - result_list);
1011 
1012  Mv(a1, expected.low_gp());
1013  Mv(a2, expected.high_gp());
1014  Mv(a3, new_value.low_gp());
1015  Mv(a4, new_value.high_gp());
1016  Mv(a0, actual_addr);
1017 
1018  MultiPush(kJSCallerSaved - c_params - result_list);
1019  PrepareCallCFunction(5, 0, kScratchReg);
1020  CallCFunction(ExternalReference::atomic_pair_compare_exchange_function(), 5,
1021  0);
1022  MultiPop(kJSCallerSaved - c_params - result_list);
1023  Mv(kScratchReg, kReturnRegister1);
1024  Mv(result.low_gp(), kReturnRegister0);
1025  Mv(result.high_gp(), kScratchReg);
1026  MultiPop(c_params - result_list);
1027  return;
1028  }
1029  // Make sure that {result} is unique.
1030  switch (type.value()) {
1031  case StoreType::kI64Store8:
1032  case StoreType::kI64Store16:
1033  case StoreType::kI64Store32:
1034  LoadConstant(result.high(), WasmValue(0));
1035  result = result.low();
1036  new_value = new_value.low();
1037  expected = expected.low();
1038  break;
1039  case StoreType::kI32Store8:
1040  case StoreType::kI32Store16:
1041  case StoreType::kI32Store:
1042  break;
1043  default:
1044  UNREACHABLE();
1045  }
1046 
1047  UseScratchRegisterScope temps(this);
1048  Register actual_addr = liftoff::CalculateActualAddress(
1049  this, temps, dst_addr, offset_reg, offset_imm, kScratchReg);
1050 
1051  Register temp0 = pinned.set(GetUnusedRegister(kGpReg, pinned)).gp();
1052  Register temp1 = pinned.set(GetUnusedRegister(kGpReg, pinned)).gp();
1053  Register temp2 = pinned.set(GetUnusedRegister(kGpReg, pinned)).gp();
1054 
1055  if (type.value() != StoreType::kI32Store &&
1056  type.value() != StoreType::kI64Store32) {
1057  And(temp1, actual_addr, 0x3);
1058  SubWord(temp0, actual_addr, Operand(temp1));
1059  SllWord(temp1, temp1, 3);
1060  }
1061  Label retry;
1062  Label done;
1063  bind(&retry);
1064  switch (type.value()) {
1065  case StoreType::kI64Store8:
1066  case StoreType::kI32Store8:
1067  lr_w(true, true, temp2, temp0);
1068  ExtractBits(result.gp(), temp2, temp1, 8, false);
1069  ExtractBits(temp2, expected.gp(), zero_reg, 8, false);
1070  Branch(&done, ne, temp2, Operand(result.gp()));
1071  InsertBits(temp2, new_value.gp(), temp1, 8);
1072  sc_w(true, true, temp2, temp0, temp2);
1073  break;
1074  case StoreType::kI64Store16:
1075  case StoreType::kI32Store16:
1076  lr_w(true, true, temp2, temp0);
1077  ExtractBits(result.gp(), temp2, temp1, 16, false);
1078  ExtractBits(temp2, expected.gp(), zero_reg, 16, false);
1079  Branch(&done, ne, temp2, Operand(result.gp()));
1080  InsertBits(temp2, new_value.gp(), temp1, 16);
1081  sc_w(true, true, temp2, temp0, temp2);
1082  break;
1083  case StoreType::kI64Store32:
1084  case StoreType::kI32Store:
1085  lr_w(true, true, result.gp(), actual_addr);
1086  Branch(&done, ne, result.gp(), Operand(expected.gp()));
1087  sc_w(true, true, temp2, actual_addr, new_value.gp());
1088  break;
1089  default:
1090  UNREACHABLE();
1091  }
1092  bnez(temp2, &retry);
1093  bind(&done);
1094 }

1118 void LiftoffAssembler::MoveStackValue(uint32_t dst_offset, uint32_t src_offset,
1119  ValueKind kind) {
1120  DCHECK_NE(dst_offset, src_offset);
1121 
1122  MemOperand src = liftoff::GetStackSlot(src_offset);
1123  MemOperand dst = liftoff::GetStackSlot(dst_offset);
1124  switch (kind) {
1125  case kI32:
1126  Lw(kScratchReg, src);
1127  Sw(kScratchReg, dst);
1128  break;
1129  case kI64:
1130  case kRef:
1131  case kRefNull:
1132  Lw(kScratchReg, src);
1133  Sw(kScratchReg, dst);
1134  src = liftoff::GetStackSlot(src_offset - 4);
1135  dst = liftoff::GetStackSlot(dst_offset - 4);
1136  Lw(kScratchReg, src);
1137  Sw(kScratchReg, dst);
1138  break;
1139  case kF32:
1140  LoadFloat(kScratchDoubleReg, src);
1141  StoreFloat(kScratchDoubleReg, dst);
1142  break;
1143  case kF64:
1144  MacroAssembler::LoadDouble(kScratchDoubleReg, src);
1145  MacroAssembler::StoreDouble(kScratchDoubleReg, dst);
1146  break;
1147  case kS128: {
1148  VU.set(kScratchReg, E8, m1);
1149  Register src_reg = src.offset() == 0 ? src.rm() : kScratchReg;
1150  if (src.offset() != 0) {
1151  MacroAssembler::AddWord(src_reg, src.rm(), src.offset());
1152  }
1153  vl(kSimd128ScratchReg, src_reg, 0, E8);
1154  Register dst_reg = dst.offset() == 0 ? dst.rm() : kScratchReg;
1155  if (dst.offset() != 0) {
1156  AddWord(kScratchReg, dst.rm(), dst.offset());
1157  }
1158  vs(kSimd128ScratchReg, dst_reg, 0, VSew::E8);
1159  break;

1520 inline void Emit64BitShiftOperation(
1521  LiftoffAssembler* assm, LiftoffRegister dst, LiftoffRegister src,
1522  Register amount,
1523  void (MacroAssembler::*emit_shift)(Register, Register, Register, Register,
1524  Register, Register, Register)) {
1525  LiftoffRegList pinned{dst, src, amount};
1526 
1527  // If some of destination registers are in use, get another, unused pair.
1528  // That way we prevent overwriting some input registers while shifting.
1529  // Do this before any branch so that the cache state will be correct for
1530  // all conditions.
1531  Register amount_capped =
1532  pinned.set(assm->GetUnusedRegister(kGpReg, pinned).gp());
1533  assm->And(amount_capped, amount, Operand(63));
1534  if (liftoff::IsRegInRegPair(dst, amount) || dst.overlaps(src)) {
1535  // Do the actual shift.
1536  LiftoffRegister tmp = assm->GetUnusedRegister(kGpRegPair, pinned);
1537  (assm->*emit_shift)(tmp.low_gp(), tmp.high_gp(), src.low_gp(),
1538  src.high_gp(), amount_capped, kScratchReg,
1539  kScratchReg2);
1540 
1541  // Place result in destination register.
1542  assm->MacroAssembler::Move(dst.high_gp(), tmp.high_gp());
1543  assm->MacroAssembler::Move(dst.low_gp(), tmp.low_gp());
1544  } else {
1545  (assm->*emit_shift)(dst.low_gp(), dst.high_gp(), src.low_gp(),
1546  src.high_gp(), amount_capped, kScratchReg,
1547  kScratchReg2);
1548  }

1890 void LiftoffAssembler::emit_i64_set_cond(Condition cond, Register dst,
1891  LiftoffRegister lhs,
1892  LiftoffRegister rhs) {
1893  ASM_CODE_COMMENT(this);
1894  Label low, cont;
1895 
1896  // For signed i64 comparisons, we still need to use unsigned comparison for
1897  // the low word (the only bit carrying signedness information is the MSB in
1898  // the high word).
1899  Condition unsigned_cond = liftoff::cond_make_unsigned(cond);
1900 
1901  Register tmp = dst;
1902  if (liftoff::IsRegInRegPair(lhs, dst) || liftoff::IsRegInRegPair(rhs, dst)) {
1903  tmp = GetUnusedRegister(kGpReg, LiftoffRegList{dst, lhs, rhs}).gp();
1904  }
1905 
1906  // Write 1 initially in tmp register.
1907  MacroAssembler::li(tmp, 1);
1908 
1909  // If high words are equal, then compare low words, else compare high.
1910  Branch(&low, eq, lhs.high_gp(), Operand(rhs.high_gp()));
1911 
1912  Branch(&cont, cond, lhs.high_gp(), Operand(rhs.high_gp()));
1913  mv(tmp, zero_reg);
1914  Branch(&cont);
1915 
1916  bind(&low);
1917  if (unsigned_cond == cond) {
1918  Branch(&cont, cond, lhs.low_gp(), Operand(rhs.low_gp()));
1919  mv(tmp, zero_reg);
1920  } else {
1921  Label lt_zero;
1922  Branch(&lt_zero, lt, lhs.high_gp(), Operand(zero_reg));
1923  Branch(&cont, unsigned_cond, lhs.low_gp(), Operand(rhs.low_gp()));
1924  mv(tmp, zero_reg);
1925  Branch(&cont);
1926  bind(&lt_zero);
1927  Branch(&cont, unsigned_cond, rhs.low_gp(), Operand(lhs.low_gp()));
1928  mv(tmp, zero_reg);
1929  Branch(&cont);
1930  }
1931  bind(&cont);
1932  // Move result to dst register if needed.
1933  MacroAssembler::Move(dst, tmp);
1934 }
