36 IteratorRecord IteratorBuiltinsAssembler::GetIterator(TNode<Context> context,
37  TNode<JSAny> object,
38  TNode<Object> method) {
39  Label if_not_callable(this, Label::kDeferred), if_callable(this);
40  GotoIf(TaggedIsSmi(method), &if_not_callable);
41  Branch(IsCallable(CAST(method)), &if_callable, &if_not_callable);
42 
43  BIND(&if_not_callable);
44  CallRuntime(Runtime::kThrowIteratorError, context, object);
45  Unreachable();
46 
47  BIND(&if_callable);
48  {
49  TNode<JSAny> iterator = Call(context, method, object);
50 
51  Label get_next(this), if_notobject(this, Label::kDeferred);
52  GotoIf(TaggedIsSmi(iterator), &if_notobject);
53  Branch(JSAnyIsNotPrimitive(CAST(iterator)), &get_next, &if_notobject);
54 
55  BIND(&if_notobject);
56  CallRuntime(Runtime::kThrowSymbolIteratorInvalid, context);
57  Unreachable();
58 
59  BIND(&get_next);
60  TNode<JSAny> next =
61  GetProperty(context, iterator, factory()->next_string());
62  return IteratorRecord{TNode<JSReceiver>::UncheckedCast(iterator), next};
63  }

66 TNode<JSReceiver> IteratorBuiltinsAssembler::IteratorStep(
67  TNode<Context> context, const IteratorRecord& iterator, Label* if_done,
68  std::optional<TNode<Map>> fast_iterator_result_map) {
69  DCHECK_NOT_NULL(if_done);
70  // 1. a. Let result be ? Invoke(iterator, "next", « »).
71  TNode<JSAny> result = Call(context, iterator.next, iterator.object);
72 
73  // 3. If Type(result) is not Object, throw a TypeError exception.
74  Label if_notobject(this, Label::kDeferred), return_result(this);
75  GotoIf(TaggedIsSmi(result), &if_notobject);
76  TNode<JSAnyNotSmi> heap_object_result = CAST(result);
77  TNode<Map> result_map = LoadMap(heap_object_result);
78  GotoIfNot(JSAnyIsNotPrimitiveMap(result_map), &if_notobject);
79 
80  // IteratorComplete
81  // 2. Return ToBoolean(? Get(iterResult, "done")).
82  IteratorComplete(context, heap_object_result, if_done,
83  fast_iterator_result_map);
84  Goto(&return_result);
85 
86  BIND(&if_notobject);
87  CallRuntime(Runtime::kThrowIteratorResultNotAnObject, context, result);
88  Unreachable();
89 
90  BIND(&return_result);
91  return CAST(heap_object_result);
92 }

94 void IteratorBuiltinsAssembler::IteratorComplete(
95  TNode<Context> context, const TNode<JSAnyNotSmi> iterator, Label* if_done,
96  std::optional<TNode<Map>> fast_iterator_result_map) {
97  DCHECK_NOT_NULL(if_done);
98 
99  Label return_result(this);
100 
101  TNode<Map> result_map = LoadMap(iterator);
102 
103  if (fast_iterator_result_map) {
104  // Fast iterator result case:
105  Label if_generic(this);
106 
107  // 4. Return result.
108  GotoIfNot(TaggedEqual(result_map, *fast_iterator_result_map), &if_generic);
109 
110  // 2. Return ToBoolean(? Get(iterResult, "done")).
111  TNode<Object> done =
112  LoadObjectField(iterator, JSIteratorResult::kDoneOffset);
113  BranchIfToBooleanIsTrue(done, if_done, &return_result);
114 
115  BIND(&if_generic);
116  }
117 
118  // Generic iterator result case:
119  {
120  // 2. Return ToBoolean(? Get(iterResult, "done")).
121  TNode<JSAny> done =
122  GetProperty(context, iterator, factory()->done_string());
123  BranchIfToBooleanIsTrue(done, if_done, &return_result);
124  }
125 
126  BIND(&return_result);
127 }

130  TNode<Context> context, TNode<JSReceiver> result,
131  std::optional<TNode<Map>> fast_iterator_result_map) {
132  Label exit(this);
133  TVARIABLE(JSAny, var_value);
134  if (fast_iterator_result_map) {
135  // Fast iterator result case:
136  Label if_generic(this);
137  TNode<Map> map = LoadMap(result);
138  GotoIfNot(TaggedEqual(map, *fast_iterator_result_map), &if_generic);
139  var_value = CAST(LoadObjectField(result, JSIteratorResult::kValueOffset));
140  Goto(&exit);
141 
142  BIND(&if_generic);
143  }
144

146  var_value = GetProperty(context, result, factory()->value_string());
147  Goto(&exit);
148 
149  BIND(&exit);
150  return var_value.value();

161 void IteratorBuiltinsAssembler::Iterate(
162  TNode<Context> context, TNode<JSAny> iterable, TNode<Object> iterable_fn,
163  std::function<void(TNode<Object>)> func,
164  std::initializer_list<compiler::CodeAssemblerVariable*> merged_variables) {
165  Label done(this);
166 
167  IteratorRecord iterator_record = GetIterator(context, iterable, iterable_fn);
168 
169  Label if_exception(this, Label::kDeferred);
170  TVARIABLE(Object, var_exception);
171 
172  Label loop_start(this, merged_variables);
173  Goto(&loop_start);
174 
175  BIND(&loop_start);
176  {
177  TNode<JSReceiver> next = IteratorStep(context, iterator_record, &done);
178  TNode<Object> next_value = IteratorValue(context, next);
179 
180  {
181  compiler::ScopedExceptionHandler handler(this, &if_exception,
182  &var_exception);
183  func(next_value);
184  }
185 
186  Goto(&loop_start);
187  }
188 
189  BIND(&if_exception);
190  {
191  TNode<HeapObject> message = GetPendingMessage();
192  SetPendingMessage(TheHoleConstant());
193  IteratorCloseOnException(context, iterator_record.object);
194  CallRuntime(Runtime::kReThrowWithMessage, context, var_exception.value(),
195  message);
196  Unreachable();
197  }
198 
199  BIND(&done);
200 }

217 void IteratorBuiltinsAssembler::FillFixedArrayFromIterable(
218  TNode<Context> context, TNode<JSAny> iterable, TNode<Object> iterator_fn,
219  GrowableFixedArray* values) {
220  // 1. Let iteratorRecord be ? GetIterator(items, method) (handled by Iterate).
221 
222  // 2. Let values be a new empty List.
223 
224  // The GrowableFixedArray has already been created. It's ok if we do this step
225  // out of order, since creating an empty List is not observable.
226 
227  // 3. Let next be true. (handled by Iterate)
228  // 4. Repeat, while next is not false (handled by Iterate)
229  Iterate(context, iterable, iterator_fn,
230  [&values](TNode<Object> value) {
231  // Handled by Iterate:
232  // a. Set next to ? IteratorStep(iteratorRecord).
233  // b. If next is not false, then
234  // i. Let nextValue be ? IteratorValue(next).
235 
236  // ii. Append nextValue to the end of the List values.
237  values->Push(value);
238  },
239  {values->var_array(), values->var_capacity(), values->var_length()});
240 }

283 TNode<FixedArray> IteratorBuiltinsAssembler::StringListFromIterable(
284  TNode<Context> context, TNode<JSAny> iterable) {
285  Label done(this);
286  GrowableFixedArray list(state());
287  // 1. If iterable is undefined, then
288  // a. Return a new empty List.
289  GotoIf(IsUndefined(iterable), &done);
290 
291  // 2. Let iteratorRecord be ? GetIterator(items) (handled by Iterate).
292 
293  // 3. Let list be a new empty List.
294 
295  // 4. Let next be true (handled by Iterate).
296  // 5. Repeat, while next is not false (handled by Iterate).
297  Iterate(
298  context, iterable,
299  [&](TNode<Object> next_value) {
300  // Handled by Iterate:
301  // a. Set next to ? IteratorStep(iteratorRecord).
302  // b. If next is not false, then
303  // i. Let nextValue be ? IteratorValue(next).
304 
305  // ii. If Type(nextValue) is not String, then
306  Label if_isnotstringtype(this, Label::kDeferred), loop_body_end(this);
307  GotoIf(TaggedIsSmi(next_value), &if_isnotstringtype);
308  TNode<Uint16T> next_value_type = LoadInstanceType(CAST(next_value));
309  GotoIfNot(IsStringInstanceType(next_value_type), &if_isnotstringtype);
310 
311  // iii. Append nextValue to the end of the List list.
312  list.Push(next_value);
313 
314  Goto(&loop_body_end);
315 
316  // 5.b.ii
317  BIND(&if_isnotstringtype);
318  {
319  // 1. Let error be ThrowCompletion(a newly created TypeError object).
320 
321  CallRuntime(Runtime::kThrowTypeError, context,
322  SmiConstant(MessageTemplate::kIterableYieldedNonString),
323  next_value);
324  // 2. Return ? IteratorClose(iteratorRecord, error). (handled by
325  // Iterate).
326  Unreachable();
327  }
328 
329  BIND(&loop_body_end);
330  },
331  {list.var_array(), list.var_length(), list.var_capacity()});
332  Goto(&done);
333 
334  BIND(&done);
335  // 6. Return list.
336  return list.ToFixedArray();
337 }

346 // This builtin always returns a new JSArray and is thus safe to use even in the
347 // presence of code that may call back into user-JS. This builtin will take the
348 // fast path if the iterable is a fast array and the Array prototype and the
349 // Symbol.iterator is untouched. The fast path skips the iterator and copies the
350 // backing store to the new array. Note that if the array has holes, the holes
351 // will be copied to the new array, which is inconsistent with the behavior of
352 // an actual iteration, where holes should be replaced with undefined (if the
353 // prototype has no elements). To maintain the correct behavior for holey
354 // arrays, use the builtins IterableToList or IterableToListWithSymbolLookup or
355 // IterableToListConvertHoles.
356 TF_BUILTIN(IterableToListMayPreserveHoles, IteratorBuiltinsAssembler) {
357  auto context = Parameter<Context>(Descriptor::kContext);
358  auto iterable = Parameter<Object>(Descriptor::kIterable);
359  auto iterator_fn = Parameter<Object>(Descriptor::kIteratorFn);
360 
361  Label slow_path(this);
362 
363  GotoIfNot(IsFastJSArrayWithNoCustomIteration(context, iterable), &slow_path);
364 
365  // The fast path will copy holes to the new array.
366  TailCallBuiltin(Builtin::kCloneFastJSArray, context, iterable);
367 
368  BIND(&slow_path);
369  TailCallBuiltin(Builtin::kIterableToList, context, iterable, iterator_fn);

395 void IteratorBuiltinsAssembler::FastIterableToList(
396  TNode<Context> context, TNode<JSAny> iterable,
397  TVariable<JSArray>* var_result, Label* slow) {
398  Label done(this), check_string(this), check_map(this), check_set(this);
399 
400  // Always call the `next()` builtins when the debugger is
401  // active, to ensure we capture side-effects correctly.
402  GotoIf(IsDebugActive(), slow);
403 
404  GotoIfNot(
405  Word32Or(IsFastJSArrayWithNoCustomIteration(context, iterable),
406  IsFastJSArrayForReadWithNoCustomIteration(context, iterable)),
407  &check_string);
408 
409  // Fast path for fast JSArray.
410  *var_result = CAST(
411  CallBuiltin(Builtin::kCloneFastJSArrayFillingHoles, context, iterable));
412  Goto(&done);
413 
414  BIND(&check_string);
415  {
416  Label string_maybe_fast_call(this);
417  StringBuiltinsAssembler string_assembler(state());
418  string_assembler.BranchIfStringPrimitiveWithNoCustomIteration(
419  iterable, context, &string_maybe_fast_call, &check_map);
420 
421  BIND(&string_maybe_fast_call);
422  const TNode<Uint32T> length = LoadStringLengthAsWord32(CAST(iterable));
423  // Use string length as conservative approximation of number of codepoints.
424  GotoIf(
425  Uint32GreaterThan(length, Uint32Constant(JSArray::kMaxFastArrayLength)),
426  slow);
427  *var_result = CAST(CallBuiltin(Builtin::kStringToList, context, iterable));
428  Goto(&done);
429  }
430 
431  BIND(&check_map);
432  {
433  Label map_fast_call(this);
434  BranchIfIterableWithOriginalKeyOrValueMapIterator(
435  state(), iterable, context, &map_fast_call, &check_set);
436 
437  BIND(&map_fast_call);
438  *var_result =
439  CAST(CallBuiltin(Builtin::kMapIteratorToList, context, iterable));
440  Goto(&done);
441  }
442 
443  BIND(&check_set);
444  {
445  Label set_fast_call(this);
446  BranchIfIterableWithOriginalValueSetIterator(state(), iterable, context,
447  &set_fast_call, slow);
448 
449  BIND(&set_fast_call);
450  *var_result =
451  CAST(CallBuiltin(Builtin::kSetOrSetIteratorToList, context, iterable));
452  Goto(&done);
453  }
454 
455  BIND(&done);
456 }

465 // This builtin loads the property Symbol.iterator as the iterator, and has fast
466 // paths for fast arrays, for primitive strings, for sets and set iterators, and
467 // for map iterators. These fast paths will only be taken if Symbol.iterator and
468 // the Iterator prototype are not modified in a way that changes the original
469 // iteration behavior.
470 // * In case of fast holey arrays, holes will be converted to undefined to
471 // reflect iteration semantics. Note that replacement by undefined is only
472 // correct when the NoElements protector is valid.
473 // * In case of map/set iterators, there is an additional requirement that the
474 // iterator is not partially consumed. To be spec-compliant, after spreading
475 // the iterator is set to be exhausted.

478  auto iterable = Parameter<JSAny>(Descriptor::kIterable);

481 
482  GotoIfForceSlowPath(&slow_path);
483 
484  TVARIABLE(JSArray, var_result);
485  FastIterableToList(context, iterable, &var_result, &slow_path);
486  Return(var_result.value());
487 
488  BIND(&slow_path);
489  {
490  TNode<Object> iterator_fn = GetIteratorMethod(context, iterable);
491  TailCallBuiltin(Builtin::kIterableToList, context, iterable, iterator_fn);
