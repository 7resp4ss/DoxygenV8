21 class SweepIdentityNodes {
22  public:
23  void PreProcessGraph(Graph* graph) {}
24  void PostProcessGraph(Graph* graph) {}
25  void PostProcessBasicBlock(BasicBlock* block) {}
26  BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
27  return BlockProcessResult::kContinue;
28  }
29  void PostPhiProcessing() {}
30  ProcessResult Process(NodeBase* node, const ProcessingState& state) {
31  for (int i = 0; i < node->input_count(); i++) {
32  Input& input = node->input(i);
33  while (input.node() && input.node()->Is<Identity>()) {
34  node->change_input(i, input.node()->input(0).node());
35  }
36  }
37  // While visiting the deopt info, the iterator will clear the identity nodes
38  // automatically.
39  if (node->properties().can_lazy_deopt()) {
40  node->lazy_deopt_info()->ForEachInput([&](ValueNode* node) {});
41  }
42  if (node->properties().can_eager_deopt()) {
43  node->eager_deopt_info()->ForEachInput([&](ValueNode* node) {});
44  }
45  return ProcessResult::kContinue;
46  }

49 // Optimizations involving loops which cannot be done at graph building time.
50 // Currently mainly loop invariant code motion.
51 class LoopOptimizationProcessor {
52  public:
53  explicit LoopOptimizationProcessor(MaglevCompilationInfo* info)
54  : zone(info->zone()) {
55  was_deoptimized =
56  info->toplevel_compilation_unit()->feedback().was_once_deoptimized();
57  }
58 
59  void PreProcessGraph(Graph* graph) {}
60  void PostPhiProcessing() {}
61 
62  void PostProcessBasicBlock(BasicBlock* block) {}
63  BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
64  current_block = block;
65  if (current_block->is_loop()) {
66  loop_effects = current_block->state()->loop_effects();
67  if (loop_effects) return BlockProcessResult::kContinue;
68  } else {
69  // TODO(olivf): Some dominance analysis would allow us to keep loop
70  // effects longer than just the first block of the loop.
71  loop_effects = nullptr;
72  }
73  return BlockProcessResult::kSkip;
74  }
75 
76  bool IsLoopPhi(Node* input) {
77  DCHECK(current_block->is_loop());
78  if (auto phi = input->TryCast<Phi>()) {
79  if (phi->is_loop_phi() && phi->merge_state() == current_block->state()) {
80  return true;
81  }
82  }
83  return false;
84  }
85 
86  bool CanHoist(Node* candidate) {
87  DCHECK_EQ(candidate->input_count(), 1);
88  DCHECK(current_block->is_loop());
89  ValueNode* input = candidate->input(0).node();
90  DCHECK(!IsLoopPhi(input));
91  // For hoisting an instruction we need:
92  // * A unique loop entry block.
93  // * Inputs live before the loop (i.e., not defined inside the loop).
94  // * No hoisting over checks (done eagerly by clearing loop_effects).
95  // TODO(olivf): We should enforce loops having a unique entry block at graph
96  // building time.
97  if (current_block->predecessor_count() != 2) return false;
98  BasicBlock* loop_entry = current_block->predecessor_at(0);
99  if (loop_entry->successors().size() != 1) {
100  return false;
101  }
102  if (IsConstantNode(input->opcode())) return true;
103  return input->owner() != current_block;
104  }
105 
106  ProcessResult Process(LoadTaggedFieldForContextSlotNoCells* ltf,
107  const ProcessingState& state) {
108  DCHECK(loop_effects);
109  ValueNode* object = ltf->object_input().node();
110  if (IsLoopPhi(object)) {
111  return ProcessResult::kContinue;
112  }
113  auto key = std::tuple{object, ltf->offset()};
114  if (!loop_effects->may_have_aliasing_contexts &&
115  !loop_effects->unstable_aspects_cleared &&
116  !loop_effects->context_slot_written.count(key) && CanHoist(ltf)) {
117  return ProcessResult::kHoist;
118  }
119  return ProcessResult::kContinue;
120  }
121 
122  ProcessResult Process(LoadTaggedFieldForProperty* ltf,
123  const ProcessingState& state) {
124  return ProcessNamedLoad(ltf, ltf->object_input().node(), ltf->name());
125  }
126 
127  ProcessResult Process(StringLength* len, const ProcessingState& state) {
128  return ProcessNamedLoad(
129  len, len->object_input().node(),
130  KnownNodeAspects::LoadedPropertyMapKey::StringLength());
131  }
132 
133  ProcessResult Process(LoadTypedArrayLength* len,
134  const ProcessingState& state) {
135  return ProcessNamedLoad(
136  len, len->receiver_input().node(),
137  KnownNodeAspects::LoadedPropertyMapKey::TypedArrayLength());
138  }
139 
140  ProcessResult ProcessNamedLoad(Node* load, ValueNode* object,
141  KnownNodeAspects::LoadedPropertyMapKey name) {
142  DCHECK(!load->properties().can_deopt());
143  if (!loop_effects) return ProcessResult::kContinue;
144  if (IsLoopPhi(object)) {
145  return ProcessResult::kContinue;
146  }
147  if (!loop_effects->unstable_aspects_cleared &&
148  !loop_effects->keys_cleared.count(name) &&
149  !loop_effects->objects_written.count(object) && CanHoist(load)) {
150  return ProcessResult::kHoist;
151  }
152  return ProcessResult::kContinue;
153  }
154 
155  ProcessResult Process(CheckMaps* maps, const ProcessingState& state) {
156  DCHECK(loop_effects);
157  // Hoisting a check out of a loop can cause it to trigger more than actually
158  // needed (i.e., if the loop is executed 0 times). This could lead to
159  // deoptimization loops as there is no feedback to learn here. Thus, we
160  // abort this optimization if the function deoptimized previously. Also, if
161  // hoisting of this check fails we need to abort (and not continue) to
162  // ensure we are not hoisting other instructions over it.
163  if (was_deoptimized) return ProcessResult::kSkipBlock;
164  ValueNode* object = maps->receiver_input().node();
165  if (IsLoopPhi(object)) {
166  return ProcessResult::kSkipBlock;
167  }
168  if (!loop_effects->unstable_aspects_cleared && CanHoist(maps)) {
169  if (auto j = current_block->predecessor_at(0)
170  ->control_node()
171  ->TryCast<CheckpointedJump>()) {
172  maps->SetEagerDeoptInfo(zone, j->eager_deopt_info()->top_frame(),
173  maps->eager_deopt_info()->feedback_to_update());
174  return ProcessResult::kHoist;
175  }
176  }
177  return ProcessResult::kSkipBlock;
178  }
179 
180  template <typename NodeT>
181  ProcessResult Process(NodeT* node, const ProcessingState& state) {
182  // Ensure we are not hoisting over checks.
183  if (node->properties().can_eager_deopt()) {
184  loop_effects = nullptr;
185  return ProcessResult::kSkipBlock;
186  }
187  return ProcessResult::kContinue;
188  }
189 
190  void PostProcessGraph(Graph* graph) {}
191 
192  Zone* zone;
193  BasicBlock* current_block;
194  const LoopEffects* loop_effects;
195  bool was_deoptimized;
196 };

204  public:

212  template <typename NodeT>
213  ProcessResult Process(NodeT* node, const ProcessingState& state) {
214  if constexpr (IsValueNode(Node::opcode_of<NodeT>) &&
215  (!NodeT::kProperties.is_required_when_unused() ||
216  std::is_same_v<ArgumentsElements, NodeT>)) {
217  if (!node->is_used()) {
218  if (!node->unused_inputs_were_visited()) {
219  DropInputUses(node);
220  }
221  return ProcessResult::kRemove;
222  }
223  }
224 
225  if constexpr (CanBeStoreToNonEscapedObject<NodeT>()) {
226  if (node->input(0).node()->template Is<InlinedAllocation>()) {
227  stores_to_allocations_.push_back(node);
228  }
229  }
230 
231  return ProcessResult::kContinue;
232  }

234 #ifdef DEBUG

236  if (!v8_flags.maglev_untagged_phis) {
237  // These nodes are removed in the phi representation selector, if we are
238  // running without it. Just remove it here.
239  return ProcessResult::kRemove;
240  }
241  UNREACHABLE();
242  }

245  void PostProcessGraph(Graph* graph) {
246  RunEscapeAnalysis(graph);
247  DropUseOfValueInStoresToCapturedAllocations();
248  }

253  void EscapeAllocation(Graph* graph, InlinedAllocation* alloc,
254  Graph::SmallAllocationVector& deps) {
255  if (alloc->HasBeenAnalysed() && alloc->HasEscaped()) return;
256  alloc->SetEscaped();
257  for (auto dep : deps) {
258  EscapeAllocation(graph, dep,
259  graph->allocations_escape_map().find(dep)->second);
260  }
261  }

263  void VerifyEscapeAnalysis(Graph* graph) {
264 #ifdef DEBUG
265  for (const auto& it : graph->allocations_escape_map()) {
266  auto* alloc = it.first;
267  DCHECK(alloc->HasBeenAnalysed());
268  if (alloc->HasEscaped()) {
269  for (auto* dep : it.second) {
270  DCHECK(dep->HasEscaped());
271  }
272  }
273  }
274 #endif // DEBUG
275  }
276 
277  void RunEscapeAnalysis(Graph* graph) {
278  for (auto& it : graph->allocations_escape_map()) {
279  auto* alloc = it.first;
280  if (alloc->HasBeenAnalysed()) continue;
281  // Check if all its uses are non escaping.
282  if (alloc->IsEscaping()) {
283  // Escape this allocation and all its dependencies.
284  EscapeAllocation(graph, alloc, it.second);
285  } else {
286  // Try to capture the allocation. This can still change if a escaped
287  // allocation has this value as one of its dependencies.
288  alloc->SetElided();
289  }
290  }
291  // Check that we've reached a fixpoint.
292  VerifyEscapeAnalysis(graph);
293  }
294 
295  void DropUseOfValueInStoresToCapturedAllocations() {
296  for (Node* node : stores_to_allocations_) {
297  InlinedAllocation* alloc =
298  node->input(0).node()->Cast<InlinedAllocation>();
299  // Since we don't analyze if allocations will escape until a fixpoint,
300  // this could drop an use of an allocation and turn it non-escaping.
301  if (alloc->HasBeenElided()) {
302  // Skip first input.
303  for (int i = 1; i < node->input_count(); i++) {
304  DropInputUses(node->input(i));
305  }
306  }
307  }
308  }
309 
310  void DropInputUses(Input& input) {
311  ValueNode* input_node = input.node();
312  if (input_node->properties().is_required_when_unused() &&
313  !input_node->Is<ArgumentsElements>())
314  return;
315  input_node->remove_use();
316  if (!input_node->is_used() && !input_node->unused_inputs_were_visited()) {
317  DropInputUses(input_node);
318  }
319  }
320 
321  void DropInputUses(ValueNode* node) {
322  for (Input& input : *node) {
323  DropInputUses(input);
324  }
325  DCHECK(!node->properties().can_eager_deopt());
326  DCHECK(!node->properties().can_lazy_deopt());
327  node->mark_unused_inputs_visited();
328  }

331 class DeadNodeSweepingProcessor {
332  public:
333  void PreProcessGraph(Graph* graph) {
334  if (graph->has_graph_labeller()) {
335  labeller_ = graph->graph_labeller();
336  }
337  }
338  void PostProcessGraph(Graph* graph) {}
339  void PostProcessBasicBlock(BasicBlock* block) {}
340  BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
341  return BlockProcessResult::kContinue;
342  }
343  void PostPhiProcessing() {}
344 
345  ProcessResult Process(AllocationBlock* node, const ProcessingState& state) {
346  // Note: this need to be done before ValueLocationConstraintProcessor, since
347  // it access the allocation offsets.
348  int size = 0;
349  for (auto alloc : node->allocation_list()) {
350  if (alloc->HasEscaped()) {
351  alloc->set_offset(size);
352  size += alloc->size();
353  }
354  }
355  // ... and update its size.
356  node->set_size(size);
357  // If size is zero, then none of the inlined allocations have escaped, we
358  // can remove the allocation block.
359  if (size == 0) return ProcessResult::kRemove;
360  return ProcessResult::kContinue;
361  }
362 
363  ProcessResult Process(InlinedAllocation* node, const ProcessingState& state) {
364  // Remove inlined allocation that became non-escaping.
365  if (!node->HasEscaped()) {
366  if (v8_flags.trace_maglev_escape_analysis) {
367  std::cout << "* Removing allocation node "
368  << PrintNodeLabel(labeller_, node) << std::endl;
369  }
370  return ProcessResult::kRemove;
371  }
372  return ProcessResult::kContinue;
373  }
374 
375  template <typename NodeT>
376  ProcessResult Process(NodeT* node, const ProcessingState& state) {
377  if constexpr (IsValueNode(Node::opcode_of<NodeT>) &&
378  (!NodeT::kProperties.is_required_when_unused() ||
379  std::is_same_v<ArgumentsElements, NodeT>)) {
380  if (!node->is_used()) {
381  return ProcessResult::kRemove;
382  }
383  return ProcessResult::kContinue;
384  }
385 
386  if constexpr (CanBeStoreToNonEscapedObject<NodeT>()) {
387  if (InlinedAllocation* object =
388  node->input(0).node()->template TryCast<InlinedAllocation>()) {
389  if (!object->HasEscaped()) {
390  if (v8_flags.trace_maglev_escape_analysis) {
391  std::cout << "* Removing store node "
392  << PrintNodeLabel(labeller_, node) << " to allocation "
393  << PrintNodeLabel(labeller_, object) << std::endl;
394  }
395  return ProcessResult::kRemove;
396  }
397  }
398  }
399  return ProcessResult::kContinue;
400  }
401 
402  private:
403  MaglevGraphLabeller* labeller_ = nullptr;
404 };
