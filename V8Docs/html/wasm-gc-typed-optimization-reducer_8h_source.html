21 // The WasmGCTypedOptimizationReducer infers type information based on the input
22 // graph and reduces type checks and casts based on that information.
23 //
24 // This is done in two steps:
25 // 1) The WasmGCTypeAnalyzer infers the types based on the input graph, e.g.:
26 // func (param anyref) (result i32)
27 // local.get 0
28 // ref.test $MyType
29 // if // local 0 is known to be (a subtype of) $MyType
30 // local.get 0
31 // ref.cast $MyType // the input of this cast is a subtype of $MyType
32 // // it can be removed during reduction
33 // struct.get $MyType 0
34 // return
35 // end // local 0 is still anyref
36 // i32.const 0
37 //
38 // 2) The WasmGCTypedOptimizationReducer reduces the graph to a new graph
39 // potentially removing, simplifying (e.g. replacing a cast with a null
40 // check) or refining (setting the from type to a more specific type) type
41 // operations.

157  V<Object> REDUCE_INPUT_GRAPH(WasmTypeCast)(V<Object> op_idx,
158  const WasmTypeCastOp& cast_op) {
159  LABEL_BLOCK(no_change) {
160  return Next::ReduceInputGraphWasmTypeCast(op_idx, cast_op);
161  }
162  if (ShouldSkipOptimizationStep()) goto no_change;
163 
164  wasm::ValueType type = analyzer_.GetInputTypeOrSentinelType(op_idx);
165  if (type.is_uninhabited()) {
166  // We are either already in unreachable code (then this instruction isn't
167  // even emitted) or the type analyzer inferred that this instruction will
168  // always trap. In either case emitting an unconditional trap to increase
169  // the chances of logic errors just leading to wrong behaviors but not
170  // resulting in security issues.
171  __ TrapIf(1, TrapId::kTrapIllegalCast);
172  __ Unreachable();
173  return OpIndex::Invalid();
174  }
175  if (type != wasm::ValueType()) {
176  CHECK(!type.is_uninhabited());
177  CHECK(wasm::IsSameTypeHierarchy(type.heap_type(),
178  cast_op.config.to.heap_type(), module_));
179  bool to_nullable = cast_op.config.to.is_nullable();
180  if (wasm::IsHeapSubtypeOf(type.heap_type(), cast_op.config.to.heap_type(),
181  module_)) {
182  if (to_nullable || type.is_non_nullable()) {
183  // The inferred type is already as specific as the cast target, the
184  // cast is guaranteed to always succeed and can therefore be removed.
185  return __ MapToNewGraph(cast_op.object());
186  } else {
187  // The inferred heap type is already as specific as the cast target,
188  // but the source can be nullable and the target cannot be, so a null
189  // check is still required.
190  return __ AssertNotNull(__ MapToNewGraph(cast_op.object()), type,
191  TrapId::kTrapIllegalCast);
192  }
193  }
194  if (wasm::HeapTypesUnrelated(type.heap_type(),
195  cast_op.config.to.heap_type(), module_)) {
196  // A cast between unrelated types can only succeed if the argument is
197  // null. Otherwise, it always fails.
198  V<Word32> non_trapping_condition =
199  type.is_nullable() && to_nullable ? __ IsNull(__ MapToNewGraph(
200  cast_op.object()),
201  type)
202  : __ Word32Constant(0);
203  __ TrapIfNot(non_trapping_condition, TrapId::kTrapIllegalCast);
204  if (!to_nullable) {
205  __ Unreachable();
206  }
207  return __ MapToNewGraph(cast_op.object());
208  }
209 
210  // If the cast resulted in an uninhabitable type, the analyzer should have
211  // returned a sentinel (bottom) type as {type}.
212  CHECK(!wasm::Intersection(type, cast_op.config.to, module_)
213  .type.is_uninhabited());
214 
215  // The cast cannot be replaced. Still, we can refine the source type, so
216  // that the lowering could potentially skip null or smi checks.
217  wasm::ValueType from_type =
218  wasm::Intersection(type, cast_op.config.from, module_).type;
219  DCHECK(!from_type.is_uninhabited());
220  WasmTypeCheckConfig config{from_type, cast_op.config.to,
221  cast_op.config.exactness};
222  return __ WasmTypeCast(__ MapToNewGraph(cast_op.object()),
223  __ MapToNewGraph(cast_op.rtt()), config);
224  }
225  goto no_change;
226  }

228  V<Word32> REDUCE_INPUT_GRAPH(WasmTypeCheck)(
229  V<Word32> op_idx, const WasmTypeCheckOp& type_check) {
230  LABEL_BLOCK(no_change) {
231  return Next::ReduceInputGraphWasmTypeCheck(op_idx, type_check);
232  }
233  if (ShouldSkipOptimizationStep()) goto no_change;
234 
235  wasm::ValueType type = analyzer_.GetInputTypeOrSentinelType(op_idx);
236  if (type.is_uninhabited()) {
237  __ Unreachable();
238  return OpIndex::Invalid();
239  }
240  if (type != wasm::ValueType()) {
241  CHECK(!type.is_uninhabited());
242  CHECK(wasm::IsSameTypeHierarchy(
243  type.heap_type(), type_check.config.to.heap_type(), module_));
244  bool to_nullable = type_check.config.to.is_nullable();
245  if (wasm::IsHeapSubtypeOf(type.heap_type(),
246  type_check.config.to.heap_type(), module_)) {
247  if (to_nullable || type.is_non_nullable()) {
248  // The inferred type is guaranteed to be a subtype of the checked
249  // type.
250  return __ Word32Constant(1);
251  } else {
252  // The inferred type is guaranteed to be a subtype of the checked
253  // type if it is not null.
254  return __ Word32Equal(
255  __ IsNull(__ MapToNewGraph(type_check.object()), type), 0);
256  }
257  }
258  if (wasm::HeapTypesUnrelated(type.heap_type(),
259  type_check.config.to.heap_type(), module_)) {
260  if (to_nullable && type.is_nullable()) {
261  return __ IsNull(__ MapToNewGraph(type_check.object()), type);
262  } else {
263  return __ Word32Constant(0);
264  }
265  }
266 
267  // If there isn't a type that matches our known input type and the
268  // type_check.config.to type, the type check always fails.
269  wasm::ValueType true_type =
270  wasm::Intersection(type, type_check.config.to, module_).type;
271  if (true_type.is_uninhabited()) {
272  return __ Word32Constant(0);
273  }
274 
275  // The check cannot be replaced. Still, we can refine the source type, so
276  // that the lowering could potentially skip null or smi checks.
277  wasm::ValueType from_type =
278  wasm::Intersection(type, type_check.config.from, module_).type;
279  DCHECK(!from_type.is_uninhabited());
280  WasmTypeCheckConfig config{from_type, type_check.config.to,
281  type_check.config.exactness};
282  return __ WasmTypeCheck(__ MapToNewGraph(type_check.object()),
283  __ MapToNewGraph(type_check.rtt()), config);
284  }
285  goto no_change;
286  }
