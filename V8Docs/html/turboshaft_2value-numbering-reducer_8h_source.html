28 // Value numbering removes redundant nodes from the graph. A simple example
29 // could be:
30 //
31 // x = a + b
32 // y = a + b
33 // z = x * y
34 //
35 // Is simplified to
36 //
37 // x = a + b
38 // z = x * x
39 //
40 // It works by storing previously seen nodes in a hashmap, and when visiting a
41 // new node, we check to see if it's already in the hashmap. If yes, then we
42 // return the old node. If not, then we keep the new one (and add it into the
43 // hashmap). A high-level pseudo-code would be:
44 //
45 // def VisitOp(op):
46 // if op in hashmap:
47 // return hashmap.get(op)
48 // else:
49 // hashmap.add(op)
50 // return op
51 //
52 // We implemented our own hashmap (to have more control, it should become
53 // clearer why by the end of this explanation). When there is a collision, we
54 // look at the next index (and the next one if there is yet another collision,
55 // etc). While not the fastest approach, it has the advantage of not requiring
56 // any dynamic memory allocation (besides the initial table, and the resizing).
57 //
58 // For the approach describe above (the pseudocode and the paragraph before it)
59 // to be correct, a node should only be replaced by a node defined in blocks
60 // that dominate the current block. Thus, this reducer relies on the fact that
61 // OptimizationPhases that iterate the graph dominator order. Then, when going
62 // down the dominator tree, we add nodes to the hashmap, and when going back up
63 // the dominator tree, we remove nodes from the hashmap.
64 //
65 // In order to efficiently remove all the nodes of a given block from the
66 // hashmap, we maintain a linked-list of hashmap entries per block (this way, we
67 // don't have to iterate the wole hashmap). Note that, in practice, we think in
68 // terms of "depth" rather than "block", and we thus have one linked-list per
69 // depth of the dominator tree. The heads of those linked lists are stored in
70 // the vector {depths_heads_}. The linked lists are then implemented in-place in
71 // the hashtable entries, thanks to the `depth_neighboring_entry` field of the
72 // `Entry` structure.
73 // To remove all of the entries from a given linked list, we iterate the entries
74 // in the linked list, setting all of their `hash` field to 0 (we prevent hashes
75 // from being equal to 0, in order to detect empty entries: their hash is 0).

123  template <typename Op>
124  static constexpr bool CanBeGVNed() {
125  constexpr Opcode opcode = operation_to_opcode_v<Op>;
126  /* Throwing operations have a non-trivial lowering, so they don't work
127  * with value numbering. */
128  if constexpr (MayThrow(opcode)) return false;
129  if constexpr (opcode == Opcode::kCatchBlockBegin) {
130  /* CatchBlockBegin are never interesting to GVN, but additionally
131  * split-edge can transform CatchBlockBeginOp into PhiOp, which means
132  * that there is no guarantee here than {result} is indeed a
133  * CatchBlockBegin. */
134  return false;
135  }
136  if constexpr (opcode == Opcode::kComment) {
137  /* We don't want to GVN comments. */
138  return false;
139  }
140  return true;
141  }

143 #define EMIT_OP(Name) \
144  template <class... Args> \
145  OpIndex Reduce##Name(Args... args) { \
146  OpIndex next_index = Asm().output_graph().next_operation_index(); \
147  USE(next_index); \
148  OpIndex result = Next::Reduce##Name(args...); \
149  UpdateEpoch<Name##Op>(next_index); \
150  if (ShouldSkipOptimizationStep()) return result; \
151  if constexpr (!CanBeGVNed<Name##Op>()) return result; \
152  DCHECK_EQ(next_index, result); \
153  return AddOrFind<Name##Op>(result); \
154  }

158  template <typename Op>
159  void UpdateEpoch(OpIndex idx) {
160  if constexpr (std::is_same_v<Op, RetainOp>) {
161  // RetainOp has CanWrite property but doesn't actually write.
162  return;
163  }
164  if (std::optional<OpEffects> effects = Op::EffectsIfStatic()) {
165  if (effects.value().can_write()) {
166  current_epoch_++;
167  }
168  } else {
169  const Op& op = Asm().Get(idx).template Cast<Op>();
170  if (op.Effects().can_write()) {
171  current_epoch_++;
172  }
173  }
174  }

177  RecordOldEpochAndSetNew(block);

184  void RecordOldEpochAndSetNew(Block* block) {
185  // Recording epoch for previous block.
186  if (current_block_ != nullptr) {
187  block_end_epochs_[current_block_->index()] = current_epoch_;
188  }
189 
190  int max_predecessor_epoch = 0;
191  for (Block* pred : block->Predecessors()) {
192  if (block_end_epochs_[pred->index()] != 0) {
193  max_predecessor_epoch = std::max<int>(max_predecessor_epoch,
194  block_end_epochs_[pred->index()]);
195  }
196  }
197  current_epoch_ = max_predecessor_epoch;
198  current_block_ = block;
199 
200  if (block->IsLoop()) {
201  // We assume that the backedge might invalidate everything that requires
202  // an epoch check. So, we increment the current_epoch, thus guaranteeing
203  // that we won't replace any CanRead operation in the loop by an operation
204  // from before the loop.
205  current_epoch_++;
206  }
207  }

209  // Resets {table_} up to the first dominator of {block} that it contains.
210  void ResetToBlock(Block* block) {
211  Block* target = block->GetDominator();
212  while (!dominator_path_.empty() && target != nullptr &&
213  dominator_path_.back() != target) {
214  if (dominator_path_.back()->Depth() > target->Depth()) {
215  ClearCurrentDepthEntries();
216  } else if (dominator_path_.back()->Depth() < target->Depth()) {
217  target = target->GetDominator();
218  } else {
219  // {target} and {dominator_path.back} have the same depth but are not
220  // equal, so we go one level up for both.
221  ClearCurrentDepthEntries();
222  target = target->GetDominator();
223  }
224  }
225  }

258  template <class Op>
259  OpIndex AddOrFind(OpIndex op_idx) {
260  if (is_disabled()) return op_idx;
261 
262  const Op& op = Asm().output_graph().Get(op_idx).template Cast<Op>();
263  if (std::is_same_v<Op, PendingLoopPhiOp> || op.IsBlockTerminator() ||
264  (!CanGVN(op.Effects()) && !std::is_same_v<Op, DeoptimizeIfOp>)) {
265  // GVNing DeoptimizeIf is safe, despite the fact that it has the CanDeopt
266  // property, which implies CanLeaveCurrentFunction, which is generally not
267  // safe to GVN.
268  return op_idx;
269  }
270  RehashIfNeeded();
271 
272  size_t hash;
273  Entry* entry = Find(op, &hash);
274  if (entry->IsEmpty()) {
275  // {op} is not present in the state, inserting it.
276  *entry = Entry{op_idx, Asm().current_block()->index(), hash,
277  depths_heads_.back(), current_epoch_};
278  depths_heads_.back() = entry;
279  ++entry_count_;
280  return op_idx;
281  } else {
282  // {op} is already present, removing it from the graph and returning the
283  // previous one.
284  Next::RemoveLast(op_idx);
285  return entry->value;
286  }
287  }

289  template <class Op>
290  Entry* Find(const Op& op, size_t* hash_ret = nullptr) {
291  bool needs_epoch_check = op.Effects().can_read_mutable_memory();
292  constexpr bool same_block_only = std::is_same_v<Op, PhiOp>;
293  size_t hash = ComputeHash<same_block_only>(op, needs_epoch_check);
294  size_t start_index = hash & mask_;
295  for (size_t i = start_index;; i = NextEntryIndex(i)) {
296  Entry& entry = table_[i];
297  if (entry.IsEmpty()) {
298  // We didn't find {op} in {table_}. Returning where it could be
299  // inserted.
300  if (hash_ret) *hash_ret = hash;
301  return &entry;
302  }
303  if (entry.hash == hash) {
304  const Operation& entry_op = Asm().output_graph().Get(entry.value);
305  if (entry_op.Is<Op>() &&
306  (!same_block_only ||
307  entry.block == Asm().current_block()->index()) &&
308  (!needs_epoch_check || (current_epoch_ == entry.epoch)) &&
309  entry_op.Cast<Op>().EqualsForGVN(op)) {
310  return &entry;
311  }
312  }
313  // Making sure that we don't have an infinite loop.
314  DCHECK_NE(start_index, NextEntryIndex(i));
315  }
316  }

318  // Remove all of the Entries of the current depth.
319  void ClearCurrentDepthEntries() {
320  for (Entry* entry = depths_heads_.back(); entry != nullptr;) {
321  entry->hash = 0;
322  Entry* next_entry = entry->depth_neighboring_entry;
323  entry->depth_neighboring_entry = nullptr;
324  entry = next_entry;
325  --entry_count_;
326  }
327  depths_heads_.pop_back();
328  dominator_path_.pop_back();
329  }

331  // If the table is too full, double its size and re-insert the old entries.
332  void RehashIfNeeded() {
333  if (V8_LIKELY(table_.size() - (table_.size() / 4) > entry_count_)) return;
334  base::Vector<Entry> new_table = table_ =
335  Asm().phase_zone()->template NewVector<Entry>(table_.size() * 2);
336  size_t mask = mask_ = table_.size() - 1;
337 
338  for (size_t depth_idx = 0; depth_idx < depths_heads_.size(); depth_idx++) {
339  // It's important to fill the new hash by inserting data in increasing
340  // depth order, in order to avoid holes when later calling
341  // ClearCurrentDepthEntries. Consider for instance:
342  //
343  // ---+------+------+------+----
344  // | a1 | a2 | a3 |
345  // ---+------+------+------+----
346  //
347  // Where a1, a2 and a3 have the same hash. By construction, we know that
348  // depth(a1) <= depth(a2) <= depth(a3). If, when re-hashing, we were to
349  // insert them in another order, say:
350  //
351  // ---+------+------+------+----
352  // | a3 | a1 | a2 |
353  // ---+------+------+------+----
354  //
355  // Then, when we'll call ClearCurrentDepthEntries to remove entries from
356  // a3's depth, we'll get this:
357  //
358  // ---+------+------+------+----
359  // | null | a1 | a2 |
360  // ---+------+------+------+----
361  //
362  // And, when looking if a1 is in the hash, we'd find a "null" where we
363  // expect it, and assume that it's not present. If, instead, we always
364  // conserve the increasing depth order, then when removing a3, we'd get:
365  //
366  // ---+------+------+------+----
367  // | a1 | a2 | null |
368  // ---+------+------+------+----
369  //
370  // Where we can still find a1 and a2.
371  Entry* entry = depths_heads_[depth_idx];
372  depths_heads_[depth_idx] = nullptr;
373 
374  while (entry != nullptr) {
375  for (size_t i = entry->hash & mask;; i = NextEntryIndex(i)) {
376  if (new_table[i].hash == 0) {
377  new_table[i] = *entry;
378  Entry* next_entry = entry->depth_neighboring_entry;
379  new_table[i].depth_neighboring_entry = depths_heads_[depth_idx];
380  depths_heads_[depth_idx] = &new_table[i];
381  entry = next_entry;
382  break;
383  }
384  }
385  }
386  }
387  }

389  template <bool same_block_only, class Op>
390  size_t ComputeHash(const Op& op, bool needs_epoch_check) {
391  size_t hash = op.hash_value();
392  if (same_block_only) {
393  hash = fast_hash_combine(Asm().current_block()->index(), hash);
394  }
395  if (needs_epoch_check) {
396  hash = fast_hash_combine(current_epoch_, hash);
397  }
398  if (V8_UNLIKELY(hash == 0)) return 1;
399  return hash;
400  }
