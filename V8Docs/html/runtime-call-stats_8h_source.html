568 
569  inline void Start(RuntimeCallCounter* counter, RuntimeCallTimer* parent) {
570  DCHECK(!IsStarted());
571  counter_ = counter;
572  parent_.SetValue(parent);
573  if (TracingFlags::runtime_stats.load(std::memory_order_relaxed) ==
574  v8::tracing::TracingCategoryObserver::ENABLED_BY_SAMPLING) {
575  return;
576  }

578  if (parent) parent->Pause(now);

584

586  if (!IsStarted()) return parent();
587  base::TimeTicks now = RuntimeCallTimer::Now();
588  Pause(now);
589  counter_->Increment();
590  CommitTimeToCounter();
591 
592  RuntimeCallTimer* parent_timer = parent();
593  if (parent_timer) {
594  parent_timer->Resume(now);
595  }
596  return parent_timer;
597  }

609  start_ticks_ = base::TimeTicks();
610  }

613  DCHECK(!IsStarted());
614  start_ticks_ = now;
