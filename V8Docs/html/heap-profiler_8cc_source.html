55 std::vector<v8::Local<v8::Value>> HeapProfiler::GetDetachedJSWrapperObjects() {
56  heap()->CollectAllAvailableGarbage(GarbageCollectionReason::kHeapProfiler);
57 
58  std::vector<v8::Local<v8::Value>> js_objects_found;
59  HeapObjectIterator iterator(heap());
60  for (Tagged<HeapObject> obj = iterator.Next(); !obj.is_null();
61  obj = iterator.Next()) {
62  if (HeapLayout::InCodeSpace(obj)) continue;
63  if (!IsJSApiWrapperObject(obj)) continue;
64  // Ensure object is wrappable, otherwise GetDetachedness() can crash
65  CppHeapObjectWrapper wrapper = CppHeapObjectWrapper(Cast<JSObject>(obj));
66  if (!wrapper.GetCppHeapWrappable(isolate(), kAnyCppHeapPointer)) continue;
67 
68  v8::Local<v8::Value> data(
69  Utils::ToLocal(direct_handle(Cast<JSObject>(obj), isolate())));
70  v8::EmbedderGraph::Node::Detachedness detachedness =
71  GetDetachedness(data, 0);
72 
73  if (detachedness != v8::EmbedderGraph::Node::Detachedness::kDetached)
74  continue;
75 
76  js_objects_found.push_back(data);
77  }
78 
79  return js_objects_found;
80 }

133 HeapSnapshot* HeapProfiler::TakeSnapshot(
134  const v8::HeapProfiler::HeapSnapshotOptions options) {
135  is_taking_snapshot_ = true;
136  HeapSnapshot* result =
137  new HeapSnapshot(this, options.snapshot_mode, options.numerics_mode);
138 
139  // We need a stack marker here to allow deterministic passes over the stack.
140  // The garbage collection and the filling of references in GenerateSnapshot
141  // should scan the same part of the stack.
142  heap()->stack().SetMarkerIfNeededAndCallback([this, &options, &result]() {
143  std::optional<CppClassNamesAsHeapObjectNameScope> use_cpp_class_name;
144  if (result->expose_internals() && heap()->cpp_heap()) {
145  use_cpp_class_name.emplace(heap()->cpp_heap());
146  }
147 
148  HeapSnapshotGenerator generator(result, options.control,
149  options.global_object_name_resolver, heap(),
150  options.stack_state);
151  if (!generator.GenerateSnapshot()) {
152  delete result;
153  result = nullptr;
154  } else {
155  snapshots_.emplace_back(result);
156  }
157  });
158  ids_->RemoveDeadEntries();
159  if (native_move_listener_) {
160  native_move_listener_->StartListening();
161  }
162  is_tracking_object_moves_ = true;
163  heap()->isolate()->UpdateLogObjectRelocation();
164  is_taking_snapshot_ = false;
165 
166  return result;
167 }

169 // Precondition: only call this if you have just completed a full GC cycle.
170 void HeapProfiler::WriteSnapshotToDiskAfterGC(HeapSnapshotMode snapshot_mode) {
171  // We need to set a stack marker for the stack walk performed by the
172  // snapshot generator to work.
173  heap()->stack().SetMarkerIfNeededAndCallback([this, snapshot_mode]() {
174  int64_t time = V8::GetCurrentPlatform()->CurrentClockTimeMilliseconds();
175  std::string filename = "v8-heap-" + std::to_string(time) + ".heapsnapshot";
176  v8::HeapProfiler::HeapSnapshotOptions options;
177  std::unique_ptr<HeapSnapshot> result(
178  new HeapSnapshot(this, snapshot_mode, options.numerics_mode));
179  HeapSnapshotGenerator generator(result.get(), options.control,
180  options.global_object_name_resolver, heap(),
181  options.stack_state);
182  if (!generator.GenerateSnapshotAfterGC()) return;
183  i::FileOutputStream stream(filename.c_str());
184  HeapSnapshotJSONSerializer serializer(result.get());
185  serializer.Serialize(&stream);
186  PrintF("Wrote heap snapshot to %s.\n", filename.c_str());
187  });
188 }

335 void HeapProfiler::QueryObjects(DirectHandle<Context> context,
336  v8::QueryObjectPredicate* predicate,
337  std::vector<v8::Global<v8::Object>>* objects) {
338  // We need a stack marker here to allow deterministic passes over the stack.
339  // The garbage collection and the two object heap iterators should scan the
340  // same part of the stack.
341  heap()->stack().SetMarkerIfNeededAndCallback([this, predicate, objects]() {
342  {
343  HandleScope handle_scope(isolate());
344  std::vector<Handle<JSTypedArray>> on_heap_typed_arrays;
345  CombinedHeapObjectIterator heap_iterator(
346  heap(), HeapObjectIterator::kFilterUnreachable);
347  for (Tagged<HeapObject> heap_obj = heap_iterator.Next();
348  !heap_obj.is_null(); heap_obj = heap_iterator.Next()) {
349  if (IsFeedbackVector(heap_obj)) {
350  Cast<FeedbackVector>(heap_obj)->ClearSlots(isolate());
351  } else if (IsJSTypedArray(heap_obj) &&
352  Cast<JSTypedArray>(heap_obj)->is_on_heap()) {
353  // Cannot call typed_array->GetBuffer() here directly because it may
354  // trigger GC. Defer that call by collecting the object in a vector.
355  on_heap_typed_arrays.push_back(
356  handle(Cast<JSTypedArray>(heap_obj), isolate()));
357  }
358  }
359  for (auto& typed_array : on_heap_typed_arrays) {
360  // Convert the on-heap typed array into off-heap typed array, so that
361  // its ArrayBuffer becomes valid and can be returned in the result.
362  typed_array->GetBuffer(isolate());
363  }
364  }
365  // We should return accurate information about live objects, so we need to
366  // collect all garbage first.
367  heap()->CollectAllAvailableGarbage(GarbageCollectionReason::kHeapProfiler);
368  CombinedHeapObjectIterator heap_iterator(
369  heap(), HeapObjectIterator::kFilterUnreachable);
370  PtrComprCageBase cage_base(isolate());
371  for (Tagged<HeapObject> heap_obj = heap_iterator.Next();
372  !heap_obj.is_null(); heap_obj = heap_iterator.Next()) {
373  if (!IsJSObject(heap_obj, cage_base) ||
374  IsJSExternalObject(heap_obj, cage_base))
375  continue;
376  v8::Local<v8::Object> v8_obj(
377  Utils::ToLocal(direct_handle(Cast<JSObject>(heap_obj), isolate())));
378  if (!predicate->Filter(v8_obj)) continue;
379  objects->emplace_back(reinterpret_cast<v8::Isolate*>(isolate()), v8_obj);
380  }
381  });
382 }
