37 StackMemory::StackMemory() : owned_(true) {
38  static std::atomic<int> next_id(1);
39  id_ = next_id.fetch_add(1);
40  size_t kJsStackSizeKB = v8_flags.wasm_stack_switching_stack_size;
41  // v8_flags.stack_size is a size of the central stack and maximum
42  // size of a secondary stack to grow.
43  const size_t size_limit = v8_flags.stack_size;
44  PageAllocator* allocator = GetPlatformPageAllocator();
45  auto page_size = allocator->AllocatePageSize();
46  size_t initial_size = std::min<size_t>(
47  size_limit * KB,
48  kJsStackSizeKB * KB + SimulatorStack::JSStackLimitMargin());
49  first_segment_ =
50  new StackSegment(RoundUp(initial_size, page_size) / page_size);
51  active_segment_ = first_segment_;
52  size_ = first_segment_->size_;
53  limit_ = first_segment_->limit_;
54  if (v8_flags.trace_wasm_stack_switching) {
55  PrintF("Allocate stack #%d (limit: %p, base: %p, size: %zu)\n", id_, limit_,
56  limit_ + size_, size_);
57  }
58 }

66 StackMemory::StackSegment::StackSegment(size_t pages) {
67  DCHECK_GE(pages, 1);
68  PageAllocator* allocator = GetPlatformPageAllocator();
69  size_t page_size = allocator->AllocatePageSize();
70  size_ = pages * page_size;
71  // Reserve one guard page before and after the stack memory.
72  limit_ = static_cast<uint8_t*>(allocator->AllocatePages(
73  nullptr, size_ + 2 * page_size, allocator->AllocatePageSize(),
74  PageAllocator::kNoAccess));
75  if (limit_ == nullptr || !SetPermissions(allocator, limit_ + page_size, size_,
76  PageAllocator::kReadWrite)) {
77  V8::FatalProcessOutOfMemory(nullptr,
78  "StackMemory::StackSegment::StackSegment");
79  }
80  limit_ += page_size;
81 }

100 bool StackMemory::Grow(Address current_fp, size_t min_size) {
101  DCHECK(owned_);
102  while (V8_UNLIKELY(active_segment_->next_segment_ != nullptr &&
103  active_segment_->next_segment_->size_ < min_size)) {
104  // If the next segment is too small to fit the evicted frame, remove it.
105  StackSegment* to_delete = active_segment_->next_segment_;
106  active_segment_->next_segment_ =
107  active_segment_->next_segment_->next_segment_;
108  if (active_segment_->next_segment_ != nullptr) {
109  active_segment_->next_segment_->prev_segment_ = active_segment_;
110  }
111  delete to_delete;
112  }
113  if (active_segment_->next_segment_ != nullptr) {
114  active_segment_ = active_segment_->next_segment_;
115  } else {
116  PageAllocator* allocator = GetPlatformPageAllocator();
117  auto page_size = allocator->AllocatePageSize();
118  const size_t size_limit = RoundUp(v8_flags.stack_size * KB, page_size);
119  DCHECK_GE(size_limit, size_);
120  size_t room_to_grow = RoundDown(size_limit - size_, page_size);
121  min_size = RoundUp(min_size, page_size);
122  if (room_to_grow < min_size) {
123  if (v8_flags.trace_wasm_stack_switching) {
124  PrintF("Stack #%d reached the grow limit %zu bytes\n", id_, size_limit);
125  }
126  return false;
127  }
128  size_t new_size =
129  std::clamp(2 * active_segment_->size_, min_size, room_to_grow);
130  DCHECK_EQ(new_size % page_size, 0);
131  auto new_segment = new StackSegment(new_size / page_size);
132  new_segment->prev_segment_ = active_segment_;
133  active_segment_->next_segment_ = new_segment;
134  active_segment_ = new_segment;
135  }
136  active_segment_->old_fp = current_fp;
137  size_ += active_segment_->size_;
138  if (v8_flags.trace_wasm_stack_switching) {
139  PrintF("Grow stack #%d by %zu bytes (limit: %p, base: %p)\n", id_,
140  active_segment_->size_, active_segment_->limit_,
141  active_segment_->limit_ + active_segment_->size_);
142  }
143  return true;
144 }

146 Address StackMemory::Shrink() {
147  DCHECK(owned_);
148  DCHECK_NE(active_segment_->prev_segment_, nullptr);
149  Address old_fp = active_segment_->old_fp;
150  size_ -= active_segment_->size_;
151  active_segment_->old_fp = 0;
152  active_segment_ = active_segment_->prev_segment_;
153  if (v8_flags.trace_wasm_stack_switching) {
154  PrintF("Shrink stack #%d (limit: %p, base: %p)\n", id_,
155  active_segment_->limit_,
156  active_segment_->limit_ + active_segment_->size_);
157  }
158  return old_fp;
159 }

161 void StackMemory::ShrinkTo(Address stack_address) {
162  DCHECK_NOT_NULL(active_segment_);
163  while (active_segment_) {
164  if (stack_address <= active_segment_->base() &&
165  stack_address >= reinterpret_cast<Address>(active_segment_->limit_)) {

184 std::unique_ptr<StackMemory> StackPool::GetOrAllocate() {
185  while (size_ > kMaxSize) {
186  size_ -= freelist_.back()->allocated_size();
187  freelist_.pop_back();
188  }
189  std::unique_ptr<StackMemory> stack;
190  if (freelist_.empty()) {
191  stack = StackMemory::New();
192  } else {
193  stack = std::move(freelist_.back());
194  freelist_.pop_back();
195  size_ -= stack->allocated_size();
196  }
197 #if DEBUG
198  constexpr uint8_t kZapValue = 0xab;
199  stack->FillWith(kZapValue);
200 #endif
201  return stack;
202 }

204 void StackPool::Add(std::unique_ptr<StackMemory> stack) {
205  // Add the stack to the pool regardless of kMaxSize, because the stack might
206  // still be in use by the unwinder.
207  // Shrink the freelist lazily when we get the next stack instead.
208  size_ += stack->allocated_size();
209  stack->Reset();
210  freelist_.push_back(std::move(stack));
211 }
