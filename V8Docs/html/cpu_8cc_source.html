67 static V8_INLINE void __cpuid(int cpu_info[4], int info_type) {
68 // Clear ecx to align with __cpuid() of MSVC:
69 // https://msdn.microsoft.com/en-us/library/hskdteyh.aspx
70 #if defined(__i386__) && defined(__pic__)
71  // Make sure to preserve ebx, which contains the pointer
72  // to the GOT in case we're generating PIC.
73  __asm__ volatile(
74  "mov %%ebx, %%edi\n\t"
75  "cpuid\n\t"
76  "xchg %%edi, %%ebx\n\t"
77  : "=a"(cpu_info[0]), "=D"(cpu_info[1]), "=c"(cpu_info[2]),
78  "=d"(cpu_info[3])
79  : "a"(info_type), "c"(0));
80 #else
81  __asm__ volatile("cpuid \n\t"
82  : "=a"(cpu_info[0]), "=b"(cpu_info[1]), "=c"(cpu_info[2]),
83  "=d"(cpu_info[3])
84  : "a"(info_type), "c"(0));
85 #endif // defined(__i386__) && defined(__pic__)
86 }

88 static V8_INLINE void __cpuidex(int cpu_info[4], int info_type,
89  int sub_info_type) {
90 // Gather additional information about the processor.
91 // Set the value of the ECX register to sub_info_type before it generates the
92 // cpuid instruction, align with __cpuidex() of MSVC:
93 // https://msdn.microsoft.com/en-us/library/hskdteyh.aspx
94 #if defined(__i386__) && defined(__pic__)
95  // Make sure to preserve ebx, which contains the pointer
96  // to the GOT in case we're generating PIC.
97  __asm__ volatile(
98  "mov %%ebx, %%edi\n\t"
99  "cpuid\n\t"
100  "xchg %%edi, %%ebx\n\t"
101  : "=a"(cpu_info[0]), "=D"(cpu_info[1]), "=c"(cpu_info[2]),
102  "=d"(cpu_info[3])
103  : "a"(info_type), "c"(sub_info_type));
104 #else
105  __asm__ volatile("cpuid \n\t"
106  : "=a"(cpu_info[0]), "=b"(cpu_info[1]), "=c"(cpu_info[2]),
107  "=d"(cpu_info[3])
108  : "a"(info_type), "c"(sub_info_type));
109 #endif // defined(__i386__) && defined(__pic__)
110 }

199 static std::tuple<uint64_t, uint64_t> ReadELFHWCaps() {
200  uint64_t hwcap = 0;
201  uint64_t hwcap2 = 0;
202 #if V8_OS_LINUX && defined(AT_HWCAP)
203  hwcap = static_cast<uint64_t>(getauxval(AT_HWCAP));
204 #if defined(AT_HWCAP2)
205  hwcap2 = static_cast<uint64_t>(getauxval(AT_HWCAP2));
206 #endif // AT_HWCAP2
207 #else
208  // Read the ELF HWCAP flags by parsing /proc/self/auxv.
209  // If getauxval is not available, the kernel/libc is also not new enough to
210  // expose hwcap2.
211  FILE* fp = base::Fopen("/proc/self/auxv", "r");
212  if (fp != nullptr) {
213  struct {
214  uint32_t tag;
215  uint32_t value;
216  } entry;
217  for (;;) {
218  size_t n = fread(&entry, sizeof(entry), 1, fp);
219  if (n == 0 || (entry.tag == 0 && entry.value == 0)) {
220  break;
221  }
222  if (entry.tag == AT_HWCAP) {
223  hwcap = entry.value;
224  break;
225  }
226  }
227  base::Fclose(fp);
228  }
229 #endif
230  return std::make_tuple(hwcap, hwcap2);
231 }

235 // Extract the information exposed by the kernel via /proc/cpuinfo.
236 class CPUInfo final {
237  public:
238  CPUInfo() : datalen_(0) {
239  // Get the size of the cpuinfo file by reading it until the end. This is
240  // required because files under /proc do not always return a valid size
241  // when using fseek(0, SEEK_END) + ftell(). Nor can the be mmap()-ed.
242  static const char PATHNAME[] = "/proc/cpuinfo";
243  FILE* fp = base::Fopen(PATHNAME, "r");
244  if (fp != nullptr) {
245  for (;;) {
246  char buffer[256];
247  size_t n = fread(buffer, 1, sizeof(buffer), fp);
248  if (n == 0) {
249  break;
250  }
251  datalen_ += n;
252  }
253  base::Fclose(fp);
254  }
255 
256  // Read the contents of the cpuinfo file.
257  data_ = new char[datalen_ + 1];
258  fp = base::Fopen(PATHNAME, "r");
259  if (fp != nullptr) {
260  for (size_t offset = 0; offset < datalen_; ) {
261  size_t n = fread(data_ + offset, 1, datalen_ - offset, fp);
262  if (n == 0) {
263  break;
264  }
265  offset += n;
266  }
267  base::Fclose(fp);
268  }
269 
270  // Zero-terminate the data.
271  data_[datalen_] = '\0';
272  }
273 
274  ~CPUInfo() {
275  delete[] data_;
276  }
277 
278  // Extract the content of a the first occurrence of a given field in
279  // the content of the cpuinfo file and return it as a heap-allocated
280  // string that must be freed by the caller using delete[].
281  // Return nullptr if not found.
282  char* ExtractField(const char* field) const {
283  DCHECK_NOT_NULL(field);
284 
285  // Look for first field occurrence, and ensure it starts the line.
286  size_t fieldlen = strlen(field);
287  char* p = data_;
288  for (;;) {
289  p = strstr(p, field);
290  if (p == nullptr) {
291  return nullptr;
292  }
293  if (p == data_ || p[-1] == '\n') {
294  break;
295  }
296  p += fieldlen;
297  }
298 
299  // Skip to the first colon followed by a space.
300  p = strchr(p + fieldlen, ':');
301  if (p == nullptr || !isspace(p[1])) {
302  return nullptr;
303  }
304  p += 2;
305 
306  // Find the end of the line.
307  char* q = strchr(p, '\n');
308  if (q == nullptr) {
309  q = data_ + datalen_;
310  }
311 
312  // Copy the line into a heap-allocated buffer.
313  size_t len = q - p;
314  char* result = new char[len + 1];
315  if (result != nullptr) {
316  memcpy(result, p, len);
317  result[len] = '\0';
318  }
319  return result;
320  }
321 
322  private:
323  char* data_;
324  size_t datalen_;
325 };

327 // Checks that a space-separated list of items contains one given 'item'.
328 static bool HasListItem(const char* list, const char* item) {
329  ssize_t item_len = strlen(item);
330  const char* p = list;
331  if (p != nullptr) {
332  while (*p != '\0') {
333  // Skip whitespace.
334  while (isspace(*p)) ++p;
335 
336  // Find end of current list item.
337  const char* q = p;
338  while (*q != '\0' && !isspace(*q)) ++q;
339 
340  if (item_len == q - p && memcmp(p, item, item_len) == 0) {
341  return true;
342  }
343 
344  // Skip to next item.
345  p = q;
346  }
347  }
348  return false;
349 }

475  // __cpuid with an InfoType argument of 0 returns the number of
476  // valid Ids in CPUInfo[0] and the CPU identification string in
477  // the other three array elements. The CPU identification string is
478  // not in linear order. The code below arranges the information
479  // in a human readable form. The human readable order is CPUInfo[1] |
480  // CPUInfo[3] | CPUInfo[2]. CPUInfo[2] and CPUInfo[3] are swapped
481  // before using memcpy to copy these three array elements to cpu_string.
482  __cpuid(cpu_info, 0);
483  unsigned num_ids = cpu_info[0];
484  std::swap(cpu_info[2], cpu_info[3]);
485  memcpy(vendor_, cpu_info + 1, 12);
486  vendor_[12] = '\0';
487 
488  // Interpret CPU feature information.
489  if (num_ids > 0) {
490  __cpuid(cpu_info, 1);
491 
492  int cpu_info70[4] = {0};
493  int cpu_info71[4] = {0};
494  if (num_ids >= 7) {
495  __cpuid(cpu_info70, 7);
496  // Check the maximum input value for supported leaf 7 sub-leaves
497  if (cpu_info70[0] >= 1) {
498  __cpuidex(cpu_info71, 7, 1);
499  }
500  }
501 
502  stepping_ = cpu_info[0] & 0xF;
503  model_ = ((cpu_info[0] >> 4) & 0xF) + ((cpu_info[0] >> 12) & 0xF0);
504  family_ = (cpu_info[0] >> 8) & 0xF;
505  type_ = (cpu_info[0] >> 12) & 0x3;
506  ext_model_ = (cpu_info[0] >> 16) & 0xF;
507  ext_family_ = (cpu_info[0] >> 20) & 0xFF;
508  has_fpu_ = (cpu_info[3] & 0x00000001) != 0;
509  has_cmov_ = (cpu_info[3] & 0x00008000) != 0;
510  has_mmx_ = (cpu_info[3] & 0x00800000) != 0;
511  has_sse_ = (cpu_info[3] & 0x02000000) != 0;
512  has_sse2_ = (cpu_info[3] & 0x04000000) != 0;
513  has_sse3_ = (cpu_info[2] & 0x00000001) != 0;
514  has_ssse3_ = (cpu_info[2] & 0x00000200) != 0;
515  has_sse41_ = (cpu_info[2] & 0x00080000) != 0;
516  has_sse42_ = (cpu_info[2] & 0x00100000) != 0;
517  has_popcnt_ = (cpu_info[2] & 0x00800000) != 0;
518  has_osxsave_ = (cpu_info[2] & 0x08000000) != 0;
519  has_avx_ = (cpu_info[2] & 0x10000000) != 0;
520  has_avx2_ = (cpu_info70[1] & 0x00000020) != 0;
521  has_avx_vnni_ = (cpu_info71[0] & 0x00000010) != 0;
522  has_avx_vnni_int8_ = (cpu_info71[3] & 0x00000020) != 0;
523  has_fma3_ = (cpu_info[2] & 0x00001000) != 0;
524  has_f16c_ = (cpu_info[2] & 0x20000000) != 0;
525  // CET shadow stack feature flag. See
526  // https://en.wikipedia.org/wiki/CPUID#EAX=7,_ECX=0:_Extended_Features
527  has_cetss_ = (cpu_info70[2] & 0x00000080) != 0;
528  // "Hypervisor Present Bit: Bit 31 of ECX of CPUID leaf 0x1."
529  // See https://lwn.net/Articles/301888/
530  // This is checking for any hypervisor. Hypervisors may choose not to
531  // announce themselves. Hypervisors trap CPUID and sometimes return
532  // different results to underlying hardware.
533  is_running_in_vm_ = (cpu_info[2] & 0x80000000) != 0;
534 
535  if (family_ == 0x6) {
536  switch (model_) {
537  case 0x1C: // SLT
538  case 0x26:
539  case 0x36:
540  case 0x27:
541  case 0x35:
542  case 0x37: // SLM
543  case 0x4A:
544  case 0x4D:
545  case 0x4C: // AMT
546  case 0x6E:
547  is_atom_ = true;
548  }
549 
550  // CPUs that are affected by Intel JCC erratum:
551  // https://www.intel.com/content/dam/support/us/en/documents/processors/mitigations-jump-conditional-code-erratum.pdf
552  switch (model_) {
553  case 0x4E:
554  has_intel_jcc_erratum_ = stepping_ == 0x3;
555  break;
556  case 0x55:
557  has_intel_jcc_erratum_ = stepping_ == 0x4 || stepping_ == 0x7;
558  break;
559  case 0x5E:
560  has_intel_jcc_erratum_ = stepping_ == 0x3;
561  break;
562  case 0x8E:
563  has_intel_jcc_erratum_ = stepping_ == 0x9 || stepping_ == 0xA ||
564  stepping_ == 0xB || stepping_ == 0xC;
565  break;
566  case 0x9E:
567  has_intel_jcc_erratum_ = stepping_ == 0x9 || stepping_ == 0xA ||
568  stepping_ == 0xB || stepping_ == 0xD;
569  break;
570  case 0xA6:
571  has_intel_jcc_erratum_ = stepping_ == 0x0;
572  break;
573  case 0xAE:
574  has_intel_jcc_erratum_ = stepping_ == 0xA;
575  break;
576  }
577  }
578  }
579 
580  // There are separate feature flags for VEX-encoded GPR instructions.
581  if (num_ids >= 7) {
582  __cpuid(cpu_info, 7);
583  has_bmi1_ = (cpu_info[1] & 0x00000008) != 0;
584  has_bmi2_ = (cpu_info[1] & 0x00000100) != 0;
585  }
586 
587  // Query extended IDs.
588  __cpuid(cpu_info, 0x80000000);
589  unsigned num_ext_ids = cpu_info[0];
590

592  if (num_ext_ids > 0x80000000) {
593  __cpuid(cpu_info, 0x80000001);

596  has_sahf_ = (cpu_info[2] & 0x00000001) != 0;
