37 // There are several kinds of ObjectData values.
38 //
39 // kSmi: The underlying V8 object is a Smi and the data is an instance of the
40 // base class (ObjectData), i.e. it's basically just the handle. Because the
41 // object is a Smi, it's safe to access the handle in order to extract the
42 // number value, and AsSmi() does exactly that.
43 //
44 // kBackgroundSerializedHeapObject: The underlying V8 object is a HeapObject
45 // and the data is an instance of the corresponding (most-specific) subclass,
46 // e.g. JSFunctionData, which provides serialized information about the
47 // object. Allows serialization from the background thread.
48 //
49 // kUnserializedHeapObject: The underlying V8 object is a HeapObject and the
50 // data is an instance of the base class (ObjectData), i.e. it basically
51 // carries no information other than the handle.
52 //
53 // kNeverSerializedHeapObject: The underlying V8 object is a (potentially
54 // mutable) HeapObject and the data is an instance of ObjectData. Its handle
55 // must be persistent so that the GC can update it at a safepoint. Via this
56 // handle, the object can be accessed concurrently to the main thread.
57 //
58 // kUnserializedReadOnlyHeapObject: The underlying V8 object is a read-only
59 // HeapObject and the data is an instance of ObjectData. For
60 // ReadOnlyHeapObjects, it is OK to access heap even from off-thread, so
61 // these objects need not be serialized.
62 enum ObjectDataKind {
63  kSmi,
64  kBackgroundSerializedHeapObject,
65  kUnserializedHeapObject,
66  kNeverSerializedHeapObject,
67  kUnserializedReadOnlyHeapObject
68 };

110  kind == kBackgroundSerializedHeapObject);
111  DCHECK_IMPLIES(kind == kUnserializedReadOnlyHeapObject,

221 bool PropertyCellData::Cache(JSHeapBroker* broker) {
222  if (serialized()) return true;
223 
224  TraceScope tracer(broker, this, "PropertyCellData::Serialize");
225  auto cell = Cast<PropertyCell>(object());
226 
227  // While this code runs on a background thread, the property cell might
228  // undergo state transitions via calls to PropertyCell::Transition. These
229  // transitions follow a certain protocol on which we rely here to ensure that
230  // we only report success when we can guarantee consistent data. A key
231  // property is that after transitioning from cell type A to B (A != B), there
232  // will never be a transition back to A, unless A is kConstant and the new
233  // value is the hole (i.e. the property cell was invalidated, which is a final
234  // state).
235 
236  PropertyDetails property_details = cell->property_details(kAcquireLoad);
237 
238  Handle<Object> value =
239  broker->CanonicalPersistentHandle(cell->value(kAcquireLoad));
240  if (broker->ObjectMayBeUninitialized(value)) {
241  DCHECK(!broker->IsMainThread());
242  return false;
243  }
244 
245  {
246  PropertyDetails property_details_again =
247  cell->property_details(kAcquireLoad);
248  if (property_details != property_details_again) {
249  DCHECK(!broker->IsMainThread());
250  return false;
251  }
252  }
253 
254  if (property_details.cell_type() == PropertyCellType::kInTransition) {
255  DCHECK(!broker->IsMainThread());
256  return false;
257  }
258 
259  ObjectData* value_data = broker->TryGetOrCreateData(value);
260  if (value_data == nullptr) {
261  DCHECK(!broker->IsMainThread());
262  return false;
263  }
264 
265  PropertyCell::CheckDataIsCompatible(property_details, *value);
266 
267  DCHECK(!serialized());
268  property_details_ = property_details;
269  value_ = value_data;
270  DCHECK(serialized());
271  return true;
272 }

308 std::optional<Tagged<Object>> GetOwnFastConstantDataPropertyFromHeap(
309  JSHeapBroker* broker, JSObjectRef holder, Representation representation,
310  FieldIndex field_index) {
311  std::optional<Tagged<Object>> constant;
312  {
313  DisallowGarbageCollection no_gc;
314  PtrComprCageBase cage_base = broker->cage_base();
315 
316  // This check to ensure the live map is the same as the cached map to
317  // to protect us against reads outside the bounds of the heap. This could
318  // happen if the Ref was created in a prior GC epoch, and the object
319  // shrunk in size. It might end up at the edge of a heap boundary. If
320  // we see that the map is the same in this GC epoch, we are safe.
321  Tagged<Map> map = holder.object()->map(cage_base, kAcquireLoad);
322  if (*holder.map(broker).object() != map) {
323  TRACE_BROKER_MISSING(broker, "Map changed for " << holder);
324  return {};
325  }
326 
327  if (field_index.is_inobject()) {
328  constant =
329  holder.object()->RawInobjectPropertyAt(cage_base, map, field_index);
330  if (!constant.has_value()) {
331  TRACE_BROKER_MISSING(
332  broker, "Constant field in " << holder << " is unsafe to read");
333  return {};
334  }
335  } else {
336  Tagged<Object> raw_properties_or_hash =
337  holder.object()->raw_properties_or_hash(cage_base, kRelaxedLoad);
338  // Ensure that the object is safe to inspect.
339  if (broker->ObjectMayBeUninitialized(raw_properties_or_hash)) {
340  return {};
341  }
342  if (!IsPropertyArray(raw_properties_or_hash, cage_base)) {
343  TRACE_BROKER_MISSING(
344  broker,
345  "Expected PropertyArray for backing store in " << holder << ".");
346  return {};
347  }
348  Tagged<PropertyArray> properties =
349  Cast<PropertyArray>(raw_properties_or_hash);
350  const int array_index = field_index.outobject_array_index();
351  if (array_index < properties->length(kAcquireLoad)) {
352  constant = properties->get(array_index);
353  } else {
354  TRACE_BROKER_MISSING(
355  broker, "Backing store for " << holder << " not long enough.");
356  return {};
357  }
358  }
359  // We might read the uninitialized sentinel, if we race with the main
360  // thread adding a new property to the object (having set the map, but not
361  // yet initialised the property value). Since this is a tight race, it won't
362  // happen very often, so we can just abort the load.
363  // TODO(leszeks): We could instead sleep/yield and spin the load, since the
364  // timing on this is tight enough that we wouldn't delay the compiler thread
365  // by much.
366  if (IsUninitialized(constant.value())) {
367  TRACE_BROKER_MISSING(broker, "Read uninitialized property.");
368  return {};
369  }
370 
371  // {constant} needs to pass the gc predicate before we can introspect on it.
372  if (broker->ObjectMayBeUninitialized(constant.value())) return {};
373 
374  // Ensure that {constant} matches the {representation} we expect for the
375  // field.
376  if (!Object::FitsRepresentation(*constant, representation, false)) {
377  const char* repString = IsSmi(*constant) ? "Smi"
378  : IsHeapNumber(*constant) ? "HeapNumber"
379  : "HeapObject";
380  TRACE_BROKER_MISSING(broker, "Mismatched representation for "
381  << holder << ". Expected "
382  << representation << ", but object is a "
383  << repString);
384  return {};
385  }
386  }
387  return constant;
388 }

596 int InstanceSizeWithMinSlack(JSHeapBroker* broker, MapRef map) {
597  // This operation is split into two phases (1. map collection, 2. map
598  // processing). This is to avoid having to take two locks
599  // (full_transition_array_access and map_updater_access) at once and thus
600  // having to deal with related deadlock issues.
601  ZoneVector<IndirectHandle<Map>> maps(broker->zone());
602  maps.push_back(map.object());
603 
604  {
605  DisallowGarbageCollection no_gc;
606 
607  // Has to be an initial map.
608  DCHECK(IsUndefined(map.object()->GetBackPointer(), broker->isolate()));
609 
610  static constexpr bool kConcurrentAccess = true;
611  TransitionsAccessor(broker->isolate(), *map.object(), kConcurrentAccess)
612  .TraverseTransitionTree([&](Tagged<Map> m) {
613  maps.push_back(broker->CanonicalPersistentHandle(m));
614  });
615  }
616 
617  // The lock is needed for UnusedPropertyFields and InstanceSizeFromSlack.
618  JSHeapBroker::MapUpdaterGuardIfNeeded mumd_scope(broker);
619 
620  int slack = std::numeric_limits<int>::max();
621  for (DirectHandle<Map> m : maps) {
622  slack = std::min(slack, m->UnusedPropertyFields());
623  }
624 
625  return map.object()->InstanceSizeFromSlack(slack);
626 }

630 // IMPORTANT: Keep this sync'd with JSFunctionData::IsConsistentWithHeapState.
631 void JSFunctionData::Cache(JSHeapBroker* broker) {
632  DCHECK(!serialized_);
633 
634  TraceScope tracer(broker, this, "JSFunctionData::Cache");
635  DirectHandle<JSFunction> function = Cast<JSFunction>(object());
636 
637  // This function may run on the background thread and thus must be individual
638  // fields in a thread-safe manner. Consistency between fields is *not*
639  // guaranteed here, instead we verify it in `IsConsistentWithHeapState`,
640  // called during job finalization. Relaxed loads are thus okay: we're
641  // guaranteed to see an initialized JSFunction object, and after
642  // initialization fields remain in a valid state.
643 
644  ContextRef context =
645  MakeRefAssumeMemoryFence(broker, function->context(kRelaxedLoad));
646  context_ = context.data();

1044 ObjectData* JSHeapBroker::TryGetOrCreateData(IndirectHandle<Object> object,
1045  GetOrCreateDataFlags flags) {
1046  RefsMap::Entry* entry = refs_->Lookup(object.address());
1047  if (entry != nullptr) return entry->value;
1048 
1049  if (mode() == JSHeapBroker::kDisabled) {
1050  entry = refs_->LookupOrInsert(object.address());
1051  ObjectData** storage = &entry->value;
1052  if (*storage == nullptr) {
1053  entry->value = zone()->New<ObjectData>(
1054  this, storage, object,
1055  IsSmi(*object) ? kSmi : kUnserializedHeapObject);
1056  }
1057  return *storage;
1058  }
1059 
1060  CHECK(mode() == JSHeapBroker::kSerializing ||
1061  mode() == JSHeapBroker::kSerialized);
1062 
1063  ObjectData* object_data;
1064  if (IsSmi(*object)) {
1065  entry = refs_->LookupOrInsert(object.address());
1066  return zone()->New<ObjectData>(this, &entry->value, object, kSmi);
1067  }
1068 
1069  DCHECK(!IsSmi(*object));
1070 
1071  const bool crash_on_error = (flags & kCrashOnError) != 0;
1072 
1073  if ((flags & kAssumeMemoryFence) == 0 &&
1074  ObjectMayBeUninitialized(Cast<HeapObject>(*object))) {
1075  TRACE_BROKER_MISSING(this, "Object may be uninitialized " << *object);
1076  CHECK_WITH_MSG(!crash_on_error, "Ref construction failed");

1078  }

1081  entry = refs_->LookupOrInsert(object.address());
1082  return zone()->New<ObjectData>(this, &entry->value, object,
1083  kUnserializedReadOnlyHeapObject);
1084  }

1087  Cast<HeapObject>(*object)->map()->instance_type();

1249  return false;

2084 std::optional<Tagged<Object>> JSObjectRef::GetOwnConstantElementFromHeap(
2085  JSHeapBroker* broker, Tagged<FixedArrayBase> elements,
2086  ElementsKind elements_kind, uint32_t index) const {
2087  DCHECK_LE(index, JSObject::kMaxElementIndex);
2088 
2089  DirectHandle<JSObject> holder = object();
2090 
2091  // This block is carefully constructed to avoid Ref creation and access since
2092  // this method may be called after the broker has retired.
2093  // The relaxed `length` read is safe to use in this case since:
2094  // - TryGetOwnConstantElement (below) only detects a constant for JSArray
2095  // holders if the array is frozen.
2096  // - Frozen arrays can't change length.
2097  // - We've already seen the corresponding map (when this JSObjectRef was
2098  // created);
2099  // - The release-load of that map ensures we read the newest value
2100  // of `length` below.
2101  if (i::IsJSArray(*holder)) {
2102  Tagged<Object> array_length_obj =
2103  Cast<JSArray>(*holder)->length(broker->isolate(), kRelaxedLoad);
2104  if (!i::IsSmi(array_length_obj)) {
2105  // Can't safely read into HeapNumber objects without atomic semantics
2106  // (relaxed would be sufficient due to the guarantees above).
2107  return {};
2108  }
2109  uint32_t array_length;
2110  if (!Object::ToArrayLength(array_length_obj, &array_length)) {
2111  return {};
2112  }
2113  // See also ElementsAccessorBase::GetMaxIndex.
2114  if (index >= array_length) return {};
2115  }
2116 
2117  Tagged<Object> maybe_element;
2118  auto result = ConcurrentLookupIterator::TryGetOwnConstantElement(
2119  &maybe_element, broker->isolate(), broker->local_isolate(), *holder,
2120  elements, elements_kind, index);
2121 
2122  if (result == ConcurrentLookupIterator::kGaveUp) {
2123  TRACE_BROKER_MISSING(broker, "JSObject::GetOwnConstantElement on "
2124  << *this << " at index " << index);
2125  return {};
2126  } else if (result == ConcurrentLookupIterator::kNotPresent) {
2127  return {};
2128  }
2129 
2130  DCHECK_EQ(result, ConcurrentLookupIterator::kPresent);
2131  return maybe_element;
2132 }
