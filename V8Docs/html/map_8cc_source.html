513 MaybeHandle<Map> Map::CopyWithField(Isolate* isolate, DirectHandle<Map> map,
514  DirectHandle<Name> name,
515  DirectHandle<FieldType> type,
516  PropertyAttributes attributes,
517  PropertyConstness constness,
518  Representation representation,
519  TransitionFlag flag) {
520  DCHECK(map->instance_descriptors(isolate)
521  ->Search(*name, map->NumberOfOwnDescriptors())
522  .is_not_found());
523 
524  // Ensure the descriptor array does not get too big.
525  if (map->NumberOfOwnDescriptors() >= kMaxNumberOfDescriptors) {
526  return MaybeHandle<Map>();
527  }
528 
529  // Compute the new index for new field.
530  int index = map->NextFreePropertyIndex();
531 
532  if (map->instance_type() == JS_CONTEXT_EXTENSION_OBJECT_TYPE) {
533  constness = PropertyConstness::kMutable;
534  representation = Representation::Tagged();
535  type = FieldType::Any(isolate);
536  } else {
537  Map::GeneralizeIfCanHaveTransitionableFastElementsKind(
538  isolate, map->instance_type(), &representation, &type);
539  }
540 
541  MaybeObjectDirectHandle wrapped_type = WrapFieldType(type);
542 
543  Descriptor d = Descriptor::DataField(name, index, attributes, constness,
544  representation, wrapped_type);
545  Handle<Map> new_map = Map::CopyAddDescriptor(isolate, map, &d, flag);
546  new_map->AccountAddedPropertyField();
547  return new_map;
548 }

580 bool Map::InstancesNeedRewriting(Tagged<Map> target,
581  int target_number_of_fields,
582  int target_inobject, int target_unused,
583  int* old_number_of_fields,
584  ConcurrencyMode cmode) const {
585  // If fields were added (or removed), rewrite the instance.
586  *old_number_of_fields = NumberOfFields(cmode);
587  DCHECK(target_number_of_fields >= *old_number_of_fields);
588  if (target_number_of_fields != *old_number_of_fields) return true;
589 
590  // If smi descriptors were replaced by double descriptors, rewrite.
591  Tagged<DescriptorArray> old_desc = IsConcurrent(cmode)
592  ? instance_descriptors(kAcquireLoad)
593  : instance_descriptors();
594  Tagged<DescriptorArray> new_desc =
595  IsConcurrent(cmode) ? target->instance_descriptors(kAcquireLoad)
596  : target->instance_descriptors();
597  for (InternalIndex i : IterateOwnDescriptors()) {
598  if (new_desc->GetDetails(i).representation().IsDouble() !=
599  old_desc->GetDetails(i).representation().IsDouble()) {
600  return true;
601  }
602  }
603 
604  // If no fields were added, and no inobject properties were removed, setting
605  // the map is sufficient.
606  if (target_inobject == GetInObjectProperties()) return false;
607  // In-object slack tracking may have reduced the object size of the new map.
608  // In that case, succeed if all existing fields were inobject, and they still
609  // fit within the new inobject size.
610  DCHECK(target_inobject < GetInObjectProperties());
611  if (target_number_of_fields <= target_inobject) {
612  DCHECK(target_number_of_fields + target_unused == target_inobject);
613  return false;
614  }
615  // Otherwise, properties will need to be moved to the backing store.
616  return true;
617 }

619 int Map::NumberOfFields(ConcurrencyMode cmode) const {
620  Tagged<DescriptorArray> descriptors = IsConcurrent(cmode)
621  ? instance_descriptors(kAcquireLoad)
622  : instance_descriptors();
623  int result = 0;
624  for (InternalIndex i : IterateOwnDescriptors()) {
625  if (descriptors->GetDetails(i).location() == PropertyLocation::kField)
626  result++;
627  }
628  return result;

676 // proper sharing of descriptor arrays.

678  Tagged<DescriptorArray> new_descriptors) {
679  PtrComprCageBase cage_base(isolate);
680  // Don't overwrite the empty descriptor array or initial map's descriptors.

682  IsUndefined(GetBackPointer(cage_base), isolate)) {

784 Tagged<Map> Map::TryReplayPropertyTransitions(Isolate* isolate,
785  Tagged<Map> old_map,
786  ConcurrencyMode cmode) {
787  DisallowGarbageCollection no_gc;
788 
789  const int root_nof = NumberOfOwnDescriptors();
790  const int old_nof = old_map->NumberOfOwnDescriptors();
791  // TODO(jgruber,chromium:1239009): The main thread should use non-atomic
792  // reads, but this currently leads to odd behavior (see the linked bug).
793  // Investigate and fix this properly. Also below and in called functions.
794  Tagged<DescriptorArray> old_descriptors =
795  old_map->instance_descriptors(isolate, kAcquireLoad);
796 
797  Tagged<Map> new_map = *this;
798  for (InternalIndex i : InternalIndex::Range(root_nof, old_nof)) {
799  PropertyDetails old_details = old_descriptors->GetDetails(i);
800  Tagged<Map> transition =
801  TransitionsAccessor(isolate, new_map, IsConcurrent(cmode))
802  .SearchTransition(old_descriptors->GetKey(i), old_details.kind(),
803  old_details.attributes());
804  if (transition.is_null()) return Map();
805  new_map = transition;
806  Tagged<DescriptorArray> new_descriptors =
807  new_map->instance_descriptors(isolate, kAcquireLoad);
808 
809  PropertyDetails new_details = new_descriptors->GetDetails(i);
810  DCHECK_EQ(old_details.kind(), new_details.kind());
811  DCHECK_EQ(old_details.attributes(), new_details.attributes());
812  if (!IsGeneralizableTo(old_details.constness(), new_details.constness())) {
813  return Map();
814  }
815  DCHECK(IsGeneralizableTo(old_details.location(), new_details.location()));
816  if (!old_details.representation().fits_into(new_details.representation())) {
817  return Map();
818  }
819  if (new_details.location() == PropertyLocation::kField) {
820  if (new_details.kind() == PropertyKind::kData) {
821  Tagged<FieldType> new_type = new_descriptors->GetFieldType(i);
822  DCHECK_EQ(PropertyKind::kData, old_details.kind());
823  DCHECK_EQ(PropertyLocation::kField, old_details.location());
824  Tagged<FieldType> old_type = old_descriptors->GetFieldType(i);
825  if (!FieldType::NowIs(old_type, new_type)) {
826  return Map();
827  }
828  } else {
829  DCHECK_EQ(PropertyKind::kAccessor, new_details.kind());
830 #ifdef DEBUG
831  Tagged<FieldType> new_type = new_descriptors->GetFieldType(i);
832  DCHECK(IsAny(new_type));
833 #endif
834  UNREACHABLE();
835  }
836  } else {
837  DCHECK_EQ(PropertyLocation::kDescriptor, new_details.location());
838  if (old_details.location() == PropertyLocation::kField ||
839  old_descriptors->GetStrongValue(i) !=
840  new_descriptors->GetStrongValue(i)) {
841  return Map();
842  }
843  }
844  }
845  if (new_map->NumberOfOwnDescriptors() != old_nof) return Map();
846  return new_map;
847 }

861 void Map::EnsureDescriptorSlack(Isolate* isolate, DirectHandle<Map> map,
862  int slack) {
863  // Only supports adding slack to owned descriptors.
864  CHECK(map->owns_descriptors());
865 
866  DirectHandle<DescriptorArray> descriptors(map->instance_descriptors(isolate),
867  isolate);
868  int old_size = map->NumberOfOwnDescriptors();
869  if (slack <= descriptors->number_of_slack_descriptors()) return;
870 
871  DirectHandle<DescriptorArray> new_descriptors =
872  DescriptorArray::CopyUpTo(isolate, descriptors, old_size, slack);
873 
874  DisallowGarbageCollection no_gc;
875  if (old_size == 0) {
876  map->UpdateDescriptors(isolate, *new_descriptors,
877  map->NumberOfOwnDescriptors());
878  return;
879  }
880 
881  // If the source descriptors had an enum cache we copy it. This ensures
882  // that the maps to which we push the new descriptor array back can rely
883  // on a cache always being available once it is set. If the map has more
884  // enumerated descriptors than available in the original cache, the cache
885  // will be lazily replaced by the extended cache when needed.
886  new_descriptors->CopyEnumCacheFrom(*descriptors);
887 
888  // Replace descriptors by new_descriptors in all maps that share it. The old
889  // descriptors will not be trimmed in the mark-compactor, we need to mark
890  // all its elements.
891 #ifndef V8_DISABLE_WRITE_BARRIERS
892  WriteBarrier::ForDescriptorArray(*descriptors,
893  descriptors->number_of_descriptors());
894 #endif
895 
896  // Update the descriptors from {map} (inclusive) until the initial map
897  // (exclusive). In the case that {map} is the initial map, update it.
898  map->UpdateDescriptors(isolate, *new_descriptors,
899  map->NumberOfOwnDescriptors());
900  Tagged<Object> next = map->GetBackPointer();
901  if (IsUndefined(next, isolate)) return;
902 
903  Tagged<Map> current = Cast<Map>(next);
904  while (current->instance_descriptors(isolate) == *descriptors) {
905  next = current->GetBackPointer();
906  if (IsUndefined(next, isolate)) break;
907  current->UpdateDescriptors(isolate, *new_descriptors,
908  current->NumberOfOwnDescriptors());
909  current = Cast<Map>(next);
910  }
911 }

994  MapHandlesSpan candidates,
995  ConcurrencyMode cmode) {

1625 // Creates transition tree starting from |split_map| and adding all descriptors
1626 // starting from descriptor with index |split_map|.NumberOfOwnDescriptors().
1627 // The way how it is done is tricky because of GC and special descriptors
1628 // marking logic.
1629 Handle<Map> Map::AddMissingTransitions(
1630  Isolate* isolate, DirectHandle<Map> split_map,
1631  DirectHandle<DescriptorArray> descriptors) {
1632  DCHECK(descriptors->IsSortedNoDuplicates());
1633  int split_nof = split_map->NumberOfOwnDescriptors();
1634  int nof_descriptors = descriptors->number_of_descriptors();
1635  CHECK_LT(split_nof, nof_descriptors);
1636 
1637  // Start with creating last map which will own full descriptors array.
1638  // This is necessary to guarantee that GC will mark the whole descriptor
1639  // array if any of the allocations happening below fail.
1640  // Number of unused properties is temporarily incorrect and the layout
1641  // descriptor could unnecessarily be in slow mode but we will fix after
1642  // all the other intermediate maps are created.
1643  // Also the last map might have interesting symbols, we temporarily set
1644  // the flag and clear it right before the descriptors are installed. This
1645  // makes heap verification happy and ensures the flag ends up accurate.
1646  Handle<Map> last_map = CopyDropDescriptors(isolate, split_map);
1647  last_map->InitializeDescriptors(isolate, *descriptors);
1648  last_map->SetInObjectUnusedPropertyFields(0);
1649  last_map->set_may_have_interesting_properties(true);
1650 
1651  // During creation of intermediate maps we violate descriptors sharing
1652  // invariant since the last map is not yet connected to the transition tree
1653  // we create here. But it is safe because GC never trims map's descriptors
1654  // if there are no dead transitions from that map and this is exactly the
1655  // case for all the intermediate maps we create here.
1656  DirectHandle<Map> map = split_map;
1657  for (InternalIndex i : InternalIndex::Range(split_nof, nof_descriptors - 1)) {
1658  DirectHandle<Map> new_map = CopyDropDescriptors(isolate, map);
1659  // Force connection of these maps to prevent split_map being a root map to
1660  // be treated as detached.
1661  InstallDescriptors(isolate, map, new_map, i, descriptors,
1662  /* force_connect */ true);
1663  DCHECK_EQ(*new_map->GetBackPointer(), *map);
1664  map = new_map;
1665  }
1666  map->NotifyLeafMapLayoutChange(isolate);
1667  last_map->set_may_have_interesting_properties(false);
1668  InstallDescriptors(isolate, map, last_map, InternalIndex(nof_descriptors - 1),
1669  descriptors);
1670  return last_map;
1671 }
