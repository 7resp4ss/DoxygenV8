24 OptimizedCompilationInfo::OptimizedCompilationInfo(
25  Zone* zone, Isolate* isolate, IndirectHandle<SharedFunctionInfo> shared,
26  IndirectHandle<JSFunction> closure, CodeKind code_kind,
27  BytecodeOffset osr_offset)
28  : isolate_unsafe_(isolate),
29  code_kind_(code_kind),
30  osr_offset_(osr_offset),
31  zone_(zone),
32  optimization_id_(isolate->NextOptimizationId()) {
33  DCHECK_EQ(*shared, closure->shared());
34  DCHECK(shared->is_compiled());
35  DCHECK_IMPLIES(is_osr(), IsOptimizing());
36  bytecode_array_ = handle(shared->GetBytecodeArray(isolate), isolate);
37  shared_info_ = shared;
38  closure_ = closure;
39  canonical_handles_ = std::make_unique<CanonicalHandlesMap>(
40  isolate->heap(), ZoneAllocationPolicy(zone));
41 
42  // Collect source positions for optimized code when profiling or if debugger
43  // is active, to be able to get more precise source positions at the price of
44  // more memory consumption.
45  if (isolate->NeedsDetailedOptimizedCodeLineInfo()) {
46  // We might not have source positions if collection fails (e.g. because we
47  // run out of stack space).
48  if (bytecode_array_->HasSourcePositionTable()) {
49  set_source_positions();
50  }
51  }
52 
53  SetTracingFlags(shared->PassesFilter(v8_flags.trace_turbo_filter));
54  ConfigureFlags();
55 
56  if (isolate->node_observer()) {
57  SetNodeObserver(isolate->node_observer());
58  }
59 }

79 void OptimizedCompilationInfo::ConfigureFlags() {
80  if (v8_flags.turbo_inline_js_wasm_calls) set_inline_js_wasm_calls();
81 
82  if (v8_flags.cet_compatible) {
83  set_shadow_stack_compliant_lazy_deopt();
84  }
85 
86  switch (code_kind_) {
87  case CodeKind::TURBOFAN_JS:
88  set_called_with_code_start_register();
89  set_switch_jump_table();
90  if (v8_flags.analyze_environment_liveness) {
91  set_analyze_environment_liveness();
92  }
93  if (v8_flags.turbo_splitting) set_splitting();
94  break;
95  case CodeKind::BYTECODE_HANDLER:
96  set_called_with_code_start_register();
97 #ifdef V8_ENABLE_BUILTIN_JUMP_TABLE_SWITCH
98  set_switch_jump_table();
99 #endif // V8_ENABLE_BUILTIN_JUMP_TABLE_SWITCH
100  if (v8_flags.turbo_splitting) set_splitting();
101  if (v8_flags.enable_allocation_folding) set_allocation_folding();
102  break;
103  case CodeKind::BUILTIN:
104 #ifdef V8_ENABLE_BUILTIN_JUMP_TABLE_SWITCH
105  set_switch_jump_table();
106 #endif // V8_ENABLE_BUILTIN_JUMP_TABLE_SWITCH
107  [[fallthrough]];
108  case CodeKind::FOR_TESTING:
109  if (v8_flags.turbo_splitting) set_splitting();
110  if (v8_flags.enable_allocation_folding) set_allocation_folding();
111 #if ENABLE_GDB_JIT_INTERFACE && DEBUG
112  set_source_positions();
113 #endif // ENABLE_GDB_JIT_INTERFACE && DEBUG
114  break;
115  case CodeKind::WASM_FUNCTION:
116  case CodeKind::WASM_TO_CAPI_FUNCTION:
117  set_switch_jump_table();
118  break;
119  case CodeKind::C_WASM_ENTRY:
120  case CodeKind::JS_TO_WASM_FUNCTION:
121  case CodeKind::WASM_TO_JS_FUNCTION:
122  break;
123  case CodeKind::BASELINE:
124  case CodeKind::MAGLEV:
125  case CodeKind::INTERPRETED_FUNCTION:
126  case CodeKind::REGEXP:
127  UNREACHABLE();
128  }
129 }

138 void OptimizedCompilationInfo::ReopenAndCanonicalizeHandlesInNewScope(
139  Isolate* isolate) {
140  if (!shared_info_.is_null()) {
141  shared_info_ = CanonicalHandle(*shared_info_, isolate);
142  }
143  if (!bytecode_array_.is_null()) {
144  bytecode_array_ = CanonicalHandle(*bytecode_array_, isolate);
145  }
146  if (!closure_.is_null()) {
147  closure_ = CanonicalHandle(*closure_, isolate);
148  }
149  DCHECK(code_.is_null());
150 }

152 void OptimizedCompilationInfo::AbortOptimization(BailoutReason reason) {
153  DCHECK_NE(reason, BailoutReason::kNoReason);
154  if (bailout_reason_ == BailoutReason::kNoReason) {
155  bailout_reason_ = reason;
156  }
157  if (IsTerminalBailoutReasonForTurbofan(reason)) {
158  set_disable_future_optimization();
159  }

168 std::unique_ptr<char[]> OptimizedCompilationInfo::GetDebugName() const {
169  if (!shared_info().is_null()) {
170  return shared_info()->DebugNameCStr();
171  }
172  base::Vector<const char> name_vec = debug_name_;
173  if (name_vec.empty()) name_vec = base::ArrayVector("unknown");
174  std::unique_ptr<char[]> name(new char[name_vec.length() + 1]);
175  memcpy(name.get(), name_vec.begin(), name_vec.length());
176  name[name_vec.length()] = '\0';
177  return name;
178 }

180 StackFrame::Type OptimizedCompilationInfo::GetOutputStackFrameType() const {
181  switch (code_kind()) {
182  case CodeKind::FOR_TESTING:
183  case CodeKind::BYTECODE_HANDLER:
184  case CodeKind::BUILTIN:
185  return StackFrame::STUB;
186 #if V8_ENABLE_WEBASSEMBLY
187  case CodeKind::WASM_FUNCTION:
188  return StackFrame::WASM;
189  case CodeKind::WASM_TO_CAPI_FUNCTION:
190  return StackFrame::WASM_EXIT;
191  case CodeKind::JS_TO_WASM_FUNCTION:
192  return StackFrame::JS_TO_WASM;
193  case CodeKind::WASM_TO_JS_FUNCTION:
194  return StackFrame::WASM_TO_JS;
195  case CodeKind::C_WASM_ENTRY:
196  return StackFrame::C_WASM_ENTRY;
197 #endif // V8_ENABLE_WEBASSEMBLY
198  default:
199  UNIMPLEMENTED();
200  }
201 }
