175 // static
176 void NodeProperties::ReplaceUses(Node* node, Node* value, Node* effect,
177  Node* success, Node* exception) {
178  // Requires distinguishing between value, effect and control edges.
179  for (Edge edge : node->use_edges()) {
180  if (IsControlEdge(edge)) {
181  if (edge.from()->opcode() == IrOpcode::kIfSuccess) {
182  DCHECK_NOT_NULL(success);
183  edge.UpdateTo(success);
184  } else if (edge.from()->opcode() == IrOpcode::kIfException) {
185  DCHECK_NOT_NULL(exception);
186  edge.UpdateTo(exception);
187  } else {
188  DCHECK_NOT_NULL(success);
189  edge.UpdateTo(success);
190  }
191  } else if (IsEffectEdge(edge)) {
192  DCHECK_NOT_NULL(effect);
193  edge.UpdateTo(effect);
194  } else {
195  DCHECK_NOT_NULL(value);
196  edge.UpdateTo(value);
197  }
198  }
199 }

259 // static
260 void NodeProperties::CollectControlProjections(Node* node, Node** projections,
261  size_t projection_count) {
262 #ifdef DEBUG
263  DCHECK_LE(static_cast<int>(projection_count), node->UseCount());
264  std::memset(projections, 0, sizeof(*projections) * projection_count);
265 #endif
266  size_t if_value_index = 0;
267  for (Edge const edge : node->use_edges()) {
268  if (!IsControlEdge(edge)) continue;
269  Node* use = edge.from();
270  size_t index;
271  switch (use->opcode()) {
272  case IrOpcode::kIfTrue:
273  DCHECK_EQ(IrOpcode::kBranch, node->opcode());
274  index = 0;
275  break;
276  case IrOpcode::kIfFalse:
277  DCHECK_EQ(IrOpcode::kBranch, node->opcode());
278  index = 1;
279  break;
280  case IrOpcode::kIfSuccess:
281  DCHECK(!node->op()->HasProperty(Operator::kNoThrow));
282  index = 0;
283  break;
284  case IrOpcode::kIfException:
285  DCHECK(!node->op()->HasProperty(Operator::kNoThrow));
286  index = 1;
287  break;
288  case IrOpcode::kIfValue:
289  DCHECK_EQ(IrOpcode::kSwitch, node->opcode());
290  index = if_value_index++;
291  break;
292  case IrOpcode::kIfDefault:
293  DCHECK_EQ(IrOpcode::kSwitch, node->opcode());
294  index = projection_count - 1;
295  break;
296  default:
297  continue;
298  }
299  DCHECK_LT(if_value_index, projection_count);
300  DCHECK_LT(index, projection_count);
301  DCHECK_NULL(projections[index]);
302  projections[index] = use;
303  }
304 #ifdef DEBUG
305  for (size_t index = 0; index < projection_count; ++index) {
306  DCHECK_NOT_NULL(projections[index]);
307  }
308 #endif
309 }

311 // static
312 MachineRepresentation NodeProperties::GetProjectionType(
313  Node const* projection) {
314  size_t index = ProjectionIndexOf(projection->op());
315  Node* input = projection->InputAt(0);
316  switch (input->opcode()) {
317  case IrOpcode::kInt32AddWithOverflow:
318  case IrOpcode::kInt32SubWithOverflow:
319  case IrOpcode::kInt32MulWithOverflow:
320  case IrOpcode::kInt32AbsWithOverflow:
321  CHECK_LE(index, static_cast<size_t>(1));
322  return index == 0 ? MachineRepresentation::kWord32
323  : MachineRepresentation::kBit;
324  case IrOpcode::kInt64AddWithOverflow:
325  case IrOpcode::kInt64SubWithOverflow:
326  case IrOpcode::kInt64MulWithOverflow:
327  case IrOpcode::kInt64AbsWithOverflow:
328  CHECK_LE(index, static_cast<size_t>(1));
329  return index == 0 ? MachineRepresentation::kWord64
330  : MachineRepresentation::kBit;
331  case IrOpcode::kTryTruncateFloat64ToInt32:
332  case IrOpcode::kTryTruncateFloat64ToUint32:
333  CHECK_LE(index, static_cast<size_t>(1));
334  return index == 0 ? MachineRepresentation::kWord32
335  : MachineRepresentation::kBit;
336  case IrOpcode::kTryTruncateFloat32ToInt64:
337  case IrOpcode::kTryTruncateFloat64ToInt64:
338  case IrOpcode::kTryTruncateFloat64ToUint64:
339  case IrOpcode::kTryTruncateFloat32ToUint64:
340  CHECK_LE(index, static_cast<size_t>(1));
341  return index == 0 ? MachineRepresentation::kWord64
342  : MachineRepresentation::kBit;
343  case IrOpcode::kCall: {
344  auto call_descriptor = CallDescriptorOf(input->op());
345  return call_descriptor->GetReturnType(index).representation();
346  }
347  case IrOpcode::kInt32PairAdd:
348  case IrOpcode::kInt32PairSub:
349  case IrOpcode::kWord32AtomicPairLoad:
350  case IrOpcode::kWord32AtomicPairAdd:
351  case IrOpcode::kWord32AtomicPairSub:
352  case IrOpcode::kWord32AtomicPairAnd:
353  case IrOpcode::kWord32AtomicPairOr:
354  case IrOpcode::kWord32AtomicPairXor:
355  case IrOpcode::kWord32AtomicPairExchange:
356  case IrOpcode::kWord32AtomicPairCompareExchange:
357  CHECK_LE(index, static_cast<size_t>(1));
358  return MachineRepresentation::kWord32;
359  default:
360  return MachineRepresentation::kNone;
361  }
362 }

366  for (;;) {

381 // static
382 OptionalMapRef NodeProperties::GetJSCreateMap(JSHeapBroker* broker,
383  Node* receiver) {
384  DCHECK(receiver->opcode() == IrOpcode::kJSCreate ||
385  receiver->opcode() == IrOpcode::kJSCreateArray);
386  HeapObjectMatcher mtarget(GetValueInput(receiver, 0));
387  HeapObjectMatcher mnewtarget(GetValueInput(receiver, 1));
388  if (mtarget.HasResolvedValue() && mnewtarget.HasResolvedValue() &&
389  mnewtarget.Ref(broker).IsJSFunction()) {
390  ObjectRef target = mtarget.Ref(broker);
391  JSFunctionRef newtarget = mnewtarget.Ref(broker).AsJSFunction();
392  if (newtarget.map(broker).has_prototype_slot() &&
393  newtarget.has_initial_map(broker)) {
394  MapRef initial_map = newtarget.initial_map(broker);
395  if (initial_map.GetConstructor(broker).equals(target)) {
396  DCHECK(target.AsJSFunction().map(broker).is_constructor());
397  DCHECK(newtarget.map(broker).is_constructor());
398  return initial_map;
399  }
400  }
401  }
402  return std::nullopt;
403 }

405 // static
406 NodeProperties::InferMapsResult NodeProperties::InferMapsUnsafe(
407  JSHeapBroker* broker, Node* receiver, Effect effect,
408  ZoneRefSet<Map>* maps_out) {
409  HeapObjectMatcher m(receiver);
410  if (m.HasResolvedValue()) {
411  HeapObjectRef ref = m.Ref(broker);
412  // We don't use ICs for the Array.prototype and the Object.prototype
413  // because the runtime has to be able to intercept them properly, so
414  // we better make sure that TurboFan doesn't outsmart the system here
415  // by storing to elements of either prototype directly.
416  //
417  // TODO(bmeurer): This can be removed once the Array.prototype and
418  // Object.prototype have NO_ELEMENTS elements kind.
419  if (!ref.IsJSObject() ||
420  !broker->IsArrayOrObjectPrototype(ref.AsJSObject())) {
421  if (ref.map(broker).is_stable()) {
422  // The {receiver_map} is only reliable when we install a stability
423  // code dependency.
424  *maps_out = ZoneRefSet<Map>{ref.map(broker)};
425  return kUnreliableMaps;
426  }
427  }
428  }
429  InferMapsResult result = kReliableMaps;
430  while (true) {
431 #ifdef DEBUG
432  InferMapsResult prev_result = result;
433  bool write_might_change_aliasing_maps = true;
434 #define WRITE_CANNOT_CHANGE_ALIASING_MAPS() \
435  write_might_change_aliasing_maps = false
436 #else
437 // Swallow the semicolon.
438 #define WRITE_CANNOT_CHANGE_ALIASING_MAPS() ((void)0)
439 #endif
440 
441  switch (effect->opcode()) {
442  case IrOpcode::kTypeGuard: {
443  DCHECK_EQ(1, effect->op()->EffectInputCount());
444  effect = NodeProperties::GetEffectInput(effect);
445  continue;
446  }
447 
448  case IrOpcode::kMapGuard: {
449  Node* const object = GetValueInput(effect, 0);
450  if (IsSame(receiver, object)) {
451  *maps_out = MapGuardMapsOf(effect->op());
452  return result;
453  }
454  break;
455  }
456  case IrOpcode::kCheckMaps: {
457  Node* const object = GetValueInput(effect, 0);
458  if (IsSame(receiver, object)) {
459  *maps_out = CheckMapsParametersOf(effect->op()).maps();
460  return result;
461  }
462  // This is subtle: CheckMaps with TryMigrateInstance _can_ write maps,
463  // but it can only transition away from deprecated maps, which should
464  // never be inferable maps, since we eagerly try to find the migration
465  // target of deprecated maps when we see them. It could be that a
466  // previous CheckMaps or other map inference node contains deprecated
467  // maps as of this check, because there could have been a concurrent
468  // deprecation, and this means that we can't CHECK this invariant here;
469  // but, then compilation would fail during finalization thanks to a
470  // CheckNoDeprecatedMaps check.
471  WRITE_CANNOT_CHANGE_ALIASING_MAPS();
472  break;
473  }
474  case IrOpcode::kTransitionElementsKindOrCheckMap: {
475  Node* const object = GetValueInput(effect, 0);
476  if (IsSame(receiver, object)) {
477  *maps_out = ZoneRefSet<Map>{
478  ElementsTransitionWithMultipleSourcesOf(effect->op()).target()};
479  return result;
480  }
481  // `receiver` and `object` might alias, so
482  // TransitionElementsKindOrCheckMaps might change receiver's map.
483  result = kUnreliableMaps;
484  break;
485  }
486  case IrOpcode::kJSCreate: {
487  if (IsSame(receiver, effect)) {
488  OptionalMapRef initial_map = GetJSCreateMap(broker, receiver);
489  if (initial_map.has_value()) {
490  *maps_out = ZoneRefSet<Map>{initial_map.value()};
491  return result;
492  }
493  // We reached the allocation of the {receiver}.
494  return kNoMaps;
495  }
496  result = kUnreliableMaps; // JSCreate can have side-effect.
497  break;
498  }
499  case IrOpcode::kJSCreatePromise: {
500  if (IsSame(receiver, effect)) {
501  *maps_out = ZoneRefSet<Map>{broker->target_native_context()
502  .promise_function(broker)
503  .initial_map(broker)};
504  return result;
505  }
506  WRITE_CANNOT_CHANGE_ALIASING_MAPS();
507  break;
508  }
509  case IrOpcode::kStoreField: {
510  // We only care about StoreField of maps.
511  Node* const object = GetValueInput(effect, 0);
512  FieldAccess const& access = FieldAccessOf(effect->op());
513  if (access.base_is_tagged == kTaggedBase &&
514  access.offset == HeapObject::kMapOffset) {
515  if (IsSame(receiver, object)) {
516  Node* const value = GetValueInput(effect, 1);
517  HeapObjectMatcher m2(value);
518  if (m2.HasResolvedValue()) {
519  *maps_out = ZoneRefSet<Map>{m2.Ref(broker).AsMap()};
520  return result;
521  }
522  }
523  // Without alias analysis we cannot tell whether this
524  // StoreField[map] affects {receiver} or not.
525  result = kUnreliableMaps;
526  } else {
527  WRITE_CANNOT_CHANGE_ALIASING_MAPS();
528  }
529  break;
530  }
531  case IrOpcode::kJSStoreMessage:
532  case IrOpcode::kJSStoreModule:
533  case IrOpcode::kStoreElement:
534  case IrOpcode::kStoreTypedElement: {
535  // These never change the map of objects.
536  WRITE_CANNOT_CHANGE_ALIASING_MAPS();
537  break;
538  }
539  case IrOpcode::kFinishRegion: {
540  // FinishRegion renames the output of allocations, so we need
541  // to update the {receiver} that we are looking for, if the
542  // {receiver} matches the current {effect}.
543  if (IsSame(receiver, effect)) receiver = GetValueInput(effect, 0);
544  // This never change the map of objects.
545  WRITE_CANNOT_CHANGE_ALIASING_MAPS();
546  break;
547  }
548  case IrOpcode::kEffectPhi: {
549  Node* control = GetControlInput(effect);
550  if (control->opcode() != IrOpcode::kLoop) {
551  DCHECK(control->opcode() == IrOpcode::kDead ||
552  control->opcode() == IrOpcode::kMerge);
553  return kNoMaps;
554  }
555 
556  // Continue search for receiver map outside the loop. Since operations
557  // inside the loop may change the map, the result is unreliable.
558  effect = GetEffectInput(effect, 0);
559  result = kUnreliableMaps;
560  continue;
561  }
562  default: {
563  DCHECK_EQ(1, effect->op()->EffectOutputCount());
564  if (effect->op()->EffectInputCount() != 1) {
565  // Didn't find any appropriate CheckMaps node.
566  return kNoMaps;
567  }
568  if (!effect->op()->HasProperty(Operator::kNoWrite)) {
569  // Without alias/escape analysis we cannot tell whether this
570  // {effect} affects {receiver} or not.
571  result = kUnreliableMaps;
572  }
573  break;
574  }
575  }
576 
577 #undef WRITE_CANNOT_CHANGE_ALIASING_MAPS
578 #ifdef DEBUG
579  if (effect->op()->HasProperty(Operator::kNoWrite) ||
580  !write_might_change_aliasing_maps) {
581  DCHECK_EQ(result, prev_result);
582  } else {
583  DCHECK_EQ(result, kUnreliableMaps);
584  }
585 #endif
586 
587  // Stop walking the effect chain once we hit the definition of
588  // the {receiver} along the {effect}s.
589  if (IsSame(receiver, effect)) return kNoMaps;
590 
591  // Continue with the next {effect}.
592  DCHECK_EQ(1, effect->op()->EffectInputCount());
593  effect = NodeProperties::GetEffectInput(effect);
594  }
595 }

611 // static
612 bool NodeProperties::CanBePrimitive(JSHeapBroker* broker, Node* receiver,
613  Effect effect) {
614  switch (receiver->opcode()) {
615 #define CASE(Opcode) case IrOpcode::k##Opcode:
616  JS_CONSTRUCT_OP_LIST(CASE)
617  JS_CREATE_OP_LIST(CASE)
618 #undef CASE
619  case IrOpcode::kCheckReceiver:
620  case IrOpcode::kConvertReceiver:
621  case IrOpcode::kJSGetSuperConstructor:
622  case IrOpcode::kJSToObject:
623  return false;
624  case IrOpcode::kHeapConstant: {
625  HeapObjectRef value = HeapObjectMatcher(receiver).Ref(broker);
626  return value.map(broker).IsPrimitiveMap();
627  }
628  default: {
629  MapInference inference(broker, receiver, effect);
630  return !inference.HaveMaps() ||
631  !inference.AllOfInstanceTypesAreJSReceiver();
632  }
633  }
634 }

636 // static
637 bool NodeProperties::CanBeNullOrUndefined(JSHeapBroker* broker, Node* receiver,
638  Effect effect) {
639  if (CanBePrimitive(broker, receiver, effect)) {
640  switch (receiver->opcode()) {
641  case IrOpcode::kCheckInternalizedString:
642  case IrOpcode::kCheckNumber:
643  case IrOpcode::kCheckNumberFitsInt32:
644  case IrOpcode::kCheckSmi:
645  case IrOpcode::kCheckString:
646  case IrOpcode::kCheckSymbol:
647  case IrOpcode::kJSToLength:
648  case IrOpcode::kJSToName:
649  case IrOpcode::kJSToNumber:
650  case IrOpcode::kJSToNumberConvertBigInt:
651  case IrOpcode::kJSToNumeric:
652  case IrOpcode::kJSToString:
653  case IrOpcode::kToBoolean:
654  return false;
655  case IrOpcode::kHeapConstant: {
656  HeapObjectRef value = HeapObjectMatcher(receiver).Ref(broker);
