27 uint32_t SharedFunctionInfo::Hash() {
28  // Hash SharedFunctionInfo based on its start position and script id. Note: we
29  // don't use the function's literal id since getting that is slow for compiled
30  // functions.
31  int start_pos = StartPosition();
32  int script_id = IsScript(script()) ? Cast<Script>(script())->id() : 0;
33  return static_cast<uint32_t>(base::hash_combine(start_pos, script_id));
34 }

36 void SharedFunctionInfo::Init(ReadOnlyRoots ro_roots, int unique_id) {
37  DisallowGarbageCollection no_gc;
38 
39  // Set the function data to the "illegal" builtin. Ideally we'd use some sort
40  // of "uninitialized" marker here, but it's cheaper to use a valid builtin and
41  // avoid having to do uninitialized checks elsewhere.
42  set_builtin_id(Builtin::kIllegal);
43 
44  // Set the name to the no-name sentinel, this can be updated later.
45  set_name_or_scope_info(SharedFunctionInfo::kNoSharedNameSentinel,
46  kReleaseStore, SKIP_WRITE_BARRIER);
47 
48  // Generally functions won't have feedback, unless they have been created
49  // from a FunctionLiteral. Those can just reset this field to keep the
50  // SharedFunctionInfo in a consistent state.
51  set_raw_outer_scope_info_or_feedback_metadata(ro_roots.the_hole_value(),
52  SKIP_WRITE_BARRIER);
53  set_script(ro_roots.undefined_value(), kReleaseStore, SKIP_WRITE_BARRIER);
54  set_function_literal_id(kInvalidInfoId, kRelaxedStore);
55  set_unique_id(unique_id);
56 
57  // Set integer fields (smi or int, depending on the architecture).
58  set_length(0);
59  set_internal_formal_parameter_count(JSParameterCount(0));
60  set_expected_nof_properties(0);
61  set_raw_function_token_offset(0);
62 
63  // All flags default to false or 0, except ConstructAsBuiltinBit just because
64  // we're using the kIllegal builtin.
65  set_flags(ConstructAsBuiltinBit::encode(true), kRelaxedStore);
66  set_flags2(0);
67 
68  UpdateFunctionMapIndex();
69 
70  set_age(0);
71 
72  clear_padding();
73 }

75 Tagged<Code> SharedFunctionInfo::GetCode(Isolate* isolate) const {
76  // ======
77  // NOTE: This chain of checks MUST be kept in sync with the equivalent CSA
78  // GetSharedFunctionInfoCode method in code-stub-assembler.cc.
79  // ======
80 
81  Tagged<Object> data = GetTrustedData(isolate);
82  if (data != Smi::zero()) {
83  DCHECK(HasTrustedData());
84 
85  if (IsBytecodeArray(data)) {
86  // Having a bytecode array means we are a compiled, interpreted function.
87  DCHECK(HasBytecodeArray());
88  return isolate->builtins()->code(Builtin::kInterpreterEntryTrampoline);
89  }
90  if (IsCode(data)) {
91  // Having baseline Code means we are a compiled, baseline function.
92  DCHECK(HasBaselineCode());
93  return Cast<Code>(data);
94  }
95  if (IsInterpreterData(data)) {
96  Tagged<Code> code = InterpreterTrampoline(isolate);
97  DCHECK(IsCode(code));
98  DCHECK(code->is_interpreter_trampoline_builtin());
99  return code;
100  }
101  if (IsUncompiledData(data)) {
102  // Having uncompiled data (with or without scope) means we need to
103  // compile.
104  DCHECK(HasUncompiledData(isolate));
105  return isolate->builtins()->code(Builtin::kCompileLazy);
106  }
107 #if V8_ENABLE_WEBASSEMBLY
108  if (IsWasmExportedFunctionData(data)) {
109  // Having a WasmExportedFunctionData means the code is in there.
110  DCHECK(HasWasmExportedFunctionData(isolate));
111  return wasm_exported_function_data()->wrapper_code(isolate);
112  }
113  if (IsWasmJSFunctionData(data)) {
114  return wasm_js_function_data()->wrapper_code(isolate);
115  }
116  if (IsWasmCapiFunctionData(data)) {
117  return wasm_capi_function_data()->wrapper_code(isolate);
118  }
119 #endif // V8_ENABLE_WEBASSEMBLY
120  } else {
121  DCHECK(HasUntrustedData());
122  data = GetUntrustedData();
123 
124  if (IsSmi(data)) {
125  // Holding a Smi means we are a builtin.
126  DCHECK(HasBuiltinId());
127  return isolate->builtins()->code(builtin_id());
128  }
129  if (IsFunctionTemplateInfo(data)) {
130  // Having a function template info means we are an API function.
131  DCHECK(IsApiFunction());
132  return isolate->builtins()->code(Builtin::kHandleApiCallOrConstruct);
133  }
134 #if V8_ENABLE_WEBASSEMBLY
135  if (IsAsmWasmData(data)) {
136  // Having AsmWasmData means we are an asm.js/wasm function.
137  DCHECK(HasAsmWasmData());
138  return isolate->builtins()->code(Builtin::kInstantiateAsmJs);
139  }
140  if (IsWasmResumeData(data)) {
141  if (static_cast<wasm::OnResume>(wasm_resume_data()->on_resume()) ==
142  wasm::OnResume::kContinue) {
143  return isolate->builtins()->code(Builtin::kWasmResume);
144  } else {
145  return isolate->builtins()->code(Builtin::kWasmReject);
146  }
147  }
148 #endif // V8_ENABLE_WEBASSEMBLY
149  }
150 
151  UNREACHABLE();
152 }

180 void SharedFunctionInfo::SetScript(IsolateForSandbox isolate,
181  ReadOnlyRoots roots,
182  Tagged<HeapObject> script_object,
183  int function_literal_id,
184  bool reset_preparsed_scope_data) {
185  DisallowGarbageCollection no_gc;
186 
187  if (script() == script_object) return;
188 
189  if (reset_preparsed_scope_data &&
190  HasUncompiledDataWithPreparseData(isolate)) {
191  ClearPreparseData(isolate);
192  }
193 
194  // Add shared function info to new script's list. If a collection occurs,
195  // the shared function info may be temporarily in two lists.
196  // This is okay because the gc-time processing of these lists can tolerate
197  // duplicates.
198  if (IsScript(script_object)) {
199  DCHECK(!IsScript(script()));
200  Tagged<Script> script = Cast<Script>(script_object);
201  Tagged<WeakFixedArray> list = script->infos();
202 #ifdef DEBUG
203  DCHECK_LT(function_literal_id, list->length());
204  Tagged<MaybeObject> maybe_object = list->get(function_literal_id);
205  Tagged<HeapObject> heap_object;
206  if (maybe_object.GetHeapObjectIfWeak(&heap_object)) {
207  DCHECK_EQ(heap_object, *this);
208  }
209 #endif
210  list->set(function_literal_id, MakeWeak(Tagged(*this)));
211  } else {
212  DCHECK(IsScript(script()));
213 
214  // Remove shared function info from old script's list.
215  Tagged<Script> old_script = Cast<Script>(script());
216 
217  // Due to liveedit, it might happen that the old_script doesn't know
218  // about the SharedFunctionInfo, so we have to guard against that.
219  Tagged<WeakFixedArray> infos = old_script->infos();
220  if (function_literal_id < infos->length()) {
221  Tagged<MaybeObject> raw = old_script->infos()->get(function_literal_id);
222  Tagged<HeapObject> heap_object;
223  if (raw.GetHeapObjectIfWeak(&heap_object) && heap_object == *this) {
224  old_script->infos()->set(function_literal_id, roots.undefined_value());
225  }
226  }
227  }
228 
229  // Finally set new script.
230  set_script(script_object, kReleaseStore);
231 }

233 void SharedFunctionInfo::CopyFrom(Tagged<SharedFunctionInfo> other,
234  IsolateForSandbox isolate) {
235  if (other->HasTrustedData()) {
236  SetTrustedData(Cast<ExposedTrustedObject>(other->GetTrustedData(isolate)));
237  } else {
238  SetUntrustedData(other->GetUntrustedData());
239  }
240 
241  PtrComprCageBase cage_base = GetPtrComprCageBase(*this);
242  set_name_or_scope_info(other->name_or_scope_info(cage_base, kAcquireLoad),
243  kReleaseStore);
244  set_outer_scope_info_or_feedback_metadata(
245  other->outer_scope_info_or_feedback_metadata(cage_base));
246  set_script(other->script(cage_base, kAcquireLoad), kReleaseStore);
247 
248  set_length(other->length());
249  set_formal_parameter_count(other->formal_parameter_count());
250  set_function_token_offset(other->function_token_offset());
251  set_expected_nof_properties(other->expected_nof_properties());
252  set_flags2(other->flags2());
253  set_flags(other->flags(kRelaxedLoad), kRelaxedStore);
254  set_function_literal_id(other->function_literal_id(kRelaxedLoad),
255  kRelaxedStore);
256  set_unique_id(other->unique_id());
257  set_age(0);
258 
259 #if DEBUG
260  // This should now be byte-for-byte identical to the input except for the age
261  // field (could be reset concurrently). Compare content before age field now:
262  DCHECK_EQ(memcmp(reinterpret_cast<void*>(address()),
263  reinterpret_cast<void*>(other.address()),
264  SharedFunctionInfo::kAgeOffset),
265  0);
266  // Compare content after age field.
267  constexpr Address kPastAgeOffset =
268  SharedFunctionInfo::kAgeOffset + SharedFunctionInfo::kAgeSize;
269  DCHECK_EQ(memcmp(reinterpret_cast<void*>(address() + kPastAgeOffset),
270  reinterpret_cast<void*>(other.address() + kPastAgeOffset),
271  SharedFunctionInfo::kSize - kPastAgeOffset),
272  0);
273 #endif
274 }

280 // Needs to be kept in sync with Scope::UniqueIdInScript and
281 // ScopeInfo::UniqueIdInScript.
282 int SharedFunctionInfo::UniqueIdInScript() const {
283  // Script scopes start "before" the script to avoid clashing with a scope that
284  // starts on character 0.
285  if (function_literal_id(kRelaxedLoad) == kFunctionLiteralIdTopLevel)
286  return -2;
287  // Wrapped functions start before the function body, but after the script
288  // start, to avoid clashing with a scope starting on character 0.
289  if (syntax_kind() == FunctionSyntaxKind::kWrapped) return -1;
290  // Default constructors have the same start position as their parent class
291  // scope. Use the next char position to distinguish this scope.
292  return StartPosition() + IsDefaultConstructor(kind());
293 }

410 void SharedFunctionInfo::DiscardCompiled(
411  Isolate* isolate, DirectHandle<SharedFunctionInfo> shared_info) {
412  DCHECK(shared_info->CanDiscardCompiled());
413 
414  Handle<String> inferred_name_val(shared_info->inferred_name(), isolate);
415  int start_position = shared_info->StartPosition();
416  int end_position = shared_info->EndPosition();
417 
418  MaybeDirectHandle<UncompiledData> data;
419  if (!shared_info->HasUncompiledDataWithPreparseData(isolate)) {
420  // Create a new UncompiledData, without pre-parsed scope.
421  data = isolate->factory()->NewUncompiledDataWithoutPreparseData(
422  inferred_name_val, start_position, end_position);
423  }
424 
425  // If the GC runs after changing one but not both fields below, it could see
426  // the SharedFunctionInfo in an unexpected state.
427  DisallowGarbageCollection no_gc;
428 
429  shared_info->DiscardCompiledMetadata(isolate);
430 
431  // Replace compiled data with a new UncompiledData object.
432  if (shared_info->HasUncompiledDataWithPreparseData(isolate)) {
433  // If this is uncompiled data with a pre-parsed scope data, we can just
434  // clear out the scope data and keep the uncompiled data.
435  shared_info->ClearPreparseData(isolate);
436  DCHECK(data.is_null());
437  } else {
438  // Update the function data to point to the UncompiledData without preparse
439  // data created above. Use the raw function data setter to avoid validity
440  // checks, since we're performing the unusual task of decompiling.
441  shared_info->SetTrustedData(*data.ToHandleChecked());
442  }
443 }

477  builder.AppendString(
478  DirectHandle<String>(Cast<String>(args->get(i)), isolate));

480  builder.AppendCStringLiteral(") {\n");
481  builder.AppendString(source);
482  builder.AppendCStringLiteral("\n}");
483  return indirect_handle(builder.Finish().ToHandleChecked(), isolate);

488 // Output the source code without any allocation in the heap.
489 std::ostream& operator<<(std::ostream& os, const SourceCodeOf& v) {
490  const Tagged<SharedFunctionInfo> s = v.value;
491  // For some native functions there is no source.
492  if (!s->HasSourceCode()) return os << "<No Source>";
493 
494  // Get the source for the script which this function came from.
495  // Don't use Cast<String> because we don't want more assertion errors while
496  // we are already creating a stack dump.
497  Tagged<String> script_source =
498  UncheckedCast<String>(Cast<Script>(s->script())->source());
499 
500  if (!s->is_toplevel()) {
501  os << "function ";
502  Tagged<String> name = s->Name();
503  if (name->length() > 0) {
504  name->PrintUC16(os);
505  }
506  }
507 
508  int len = s->EndPosition() - s->StartPosition();
509  if (len <= v.max_length || v.max_length < 0) {
510  script_source->PrintUC16(os, s->StartPosition(), s->EndPosition());
511  return os;
512  } else {
513  script_source->PrintUC16(os, s->StartPosition(),
514  s->StartPosition() + v.max_length);
515  return os << "...\n";
516  }

542 template <typename IsolateT>
543 void SharedFunctionInfo::InitFromFunctionLiteral(IsolateT* isolate,
544  FunctionLiteral* lit,
545  bool is_toplevel) {
546  DCHECK(!IsScopeInfo(
547  lit->shared_function_info()->name_or_scope_info(kAcquireLoad)));
548  {
549  DisallowGarbageCollection no_gc;
550  Tagged<SharedFunctionInfo> raw_sfi = *lit->shared_function_info();
551  DCHECK_EQ(raw_sfi->function_literal_id(kRelaxedLoad),
552  lit->function_literal_id());
553  // When adding fields here, make sure DeclarationScope::AnalyzePartially is
554  // updated accordingly.
555  raw_sfi->set_internal_formal_parameter_count(
556  JSParameterCount(lit->parameter_count()));
557  raw_sfi->SetFunctionTokenPosition(lit->function_token_position(),
558  lit->start_position());
559  raw_sfi->set_syntax_kind(lit->syntax_kind());
560  raw_sfi->set_allows_lazy_compilation(lit->AllowsLazyCompilation());
561  raw_sfi->set_language_mode(lit->language_mode());
562  // FunctionKind must have already been set.
563  DCHECK(lit->kind() == raw_sfi->kind());
564  DCHECK_IMPLIES(lit->requires_instance_members_initializer(),
565  IsClassConstructor(lit->kind()));
566  raw_sfi->set_requires_instance_members_initializer(
567  lit->requires_instance_members_initializer());
568  DCHECK_IMPLIES(lit->class_scope_has_private_brand(),
569  IsClassConstructor(lit->kind()));
570  raw_sfi->set_class_scope_has_private_brand(
571  lit->class_scope_has_private_brand());
572  DCHECK_IMPLIES(lit->has_static_private_methods_or_accessors(),
573  IsClassConstructor(lit->kind()));
574  raw_sfi->set_has_static_private_methods_or_accessors(
575  lit->has_static_private_methods_or_accessors());
576 
577  raw_sfi->set_is_toplevel(is_toplevel);
578  DCHECK(IsTheHole(raw_sfi->outer_scope_info()));
579  Scope* outer_scope = lit->scope()->GetOuterScopeWithContext();
580  if (outer_scope && (!is_toplevel || !outer_scope->is_script_scope())) {
581  raw_sfi->set_outer_scope_info(*outer_scope->scope_info());
582  raw_sfi->set_private_name_lookup_skips_outer_class(
