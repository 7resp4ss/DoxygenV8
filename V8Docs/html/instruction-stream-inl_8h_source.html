42 Tagged<InstructionStream> InstructionStream::Initialize(
43  Tagged<HeapObject> self, Tagged<Map> map, uint32_t body_size,
44  int constant_pool_offset, Tagged<TrustedByteArray> reloc_info) {
45  {
46  WritableJitAllocation writable_allocation =
47  ThreadIsolation::RegisterInstructionStreamAllocation(
48  self.address(), InstructionStream::SizeFor(body_size));
49  CHECK_EQ(InstructionStream::SizeFor(body_size), writable_allocation.size());
50 
51  writable_allocation.WriteHeaderSlot<Map, kMapOffset>(map, kRelaxedStore);
52 
53  writable_allocation.WriteHeaderSlot<uint32_t, kBodySizeOffset>(body_size);
54 
55  if constexpr (V8_EMBEDDED_CONSTANT_POOL_BOOL) {
56  writable_allocation.WriteHeaderSlot<int, kConstantPoolOffsetOffset>(
57  kHeaderSize + constant_pool_offset);
58  }
59 
60  // During the Code initialization process, InstructionStream::code is
61  // briefly unset (the Code object has not been allocated yet). In this state
62  // it is only visible through heap iteration.
63  writable_allocation.WriteHeaderSlot<Smi, kCodeOffset>(Smi::zero(),
64  kReleaseStore);
65 
66  DCHECK(!HeapLayout::InYoungGeneration(reloc_info));
67  writable_allocation.WriteProtectedPointerHeaderSlot<TrustedByteArray,
68  kRelocationInfoOffset>(
69  reloc_info, kRelaxedStore);
70 
71  // Clear header padding
72  writable_allocation.ClearBytes(kUnalignedSize,
73  kHeaderSize - kUnalignedSize);
74  // Clear trailing padding.
75  writable_allocation.ClearBytes(kHeaderSize + body_size,
76  TrailingPaddingSizeFor(body_size));
77  }
78 
79  Tagged<InstructionStream> istream = Cast<InstructionStream>(self);
80 
81  // We want to keep the code minimal that runs with write access to a JIT
82  // allocation, so trigger the write barriers after the WritableJitAllocation
83  // went out of scope.
84  SLOW_DCHECK(!WriteBarrier::IsRequired(istream, map));
85  CONDITIONAL_PROTECTED_POINTER_WRITE_BARRIER(*istream, kRelocationInfoOffset,
86  reloc_info, UPDATE_WRITE_BARRIER);
87 
88  return istream;
89 }

91 // Copy from compilation artifacts stored in CodeDesc to the target on-heap
92 // objects.
93 //
94 // Note this is quite convoluted for historical reasons. The CodeDesc buffer
95 // contains instructions, a part of inline metadata, and the relocation info.
96 // Additionally, the unwinding_info is stored in a separate buffer
97 // `desc.unwinding_info`. In this method, we copy all these parts into the
98 // final on-heap representation.
99 //
100 // The off-heap representation:
101 //
102 // CodeDesc.buffer:
103 //
104 // +-------------------
105 // | instructions
106 // +-------------------
107 // | inline metadata
108 // | .. safepoint table
109 // | .. handler table
110 // | .. constant pool
111 // | .. code comments
112 // +-------------------
113 // | reloc info
114 // +-------------------
115 //
116 // CodeDesc.unwinding_info: .. the unwinding info.
117 //
118 // This is transformed into the on-heap representation, where
119 // InstructionStream contains all instructions and inline metadata, and a
120 // pointer to the relocation info byte array.
121 void InstructionStream::Finalize(Tagged<Code> code,
122  Tagged<TrustedByteArray> reloc_info,
123  CodeDesc desc, Heap* heap) {
124  DisallowGarbageCollection no_gc;
125  std::optional<WriteBarrierPromise> promise;
126 
127  // Copy the relocation info first before we unlock the Jit allocation.
128  // TODO(sroettger): reloc info should live in protected memory.
129  DCHECK_EQ(reloc_info->length(), desc.reloc_size);
130  CopyBytes(reloc_info->begin(), desc.buffer + desc.reloc_offset,
131  static_cast<size_t>(desc.reloc_size));
132 
133  {
134  WritableJitAllocation writable_allocation =
135  ThreadIsolation::LookupJitAllocation(
136  address(), InstructionStream::SizeFor(body_size()),
137  ThreadIsolation::JitAllocationType::kInstructionStream, true);
138 
139  // Copy code and inline metadata.
140  static_assert(InstructionStream::kOnHeapBodyIsContiguous);
141  writable_allocation.CopyCode(kHeaderSize, desc.buffer,
142  static_cast<size_t>(desc.instr_size));
143  writable_allocation.CopyData(kHeaderSize + desc.instr_size,
144  desc.unwinding_info,
145  static_cast<size_t>(desc.unwinding_info_size));
146  DCHECK_EQ(desc.body_size(), desc.instr_size + desc.unwinding_info_size);
147  DCHECK_EQ(code->body_size(),
148  code->instruction_size() + code->metadata_size());
149 
150  promise.emplace(RelocateFromDesc(writable_allocation, heap, desc,
151  code->constant_pool(), no_gc));
152 
153  // Publish the code pointer after the istream has been fully initialized.
154  writable_allocation.WriteProtectedPointerHeaderSlot<Code, kCodeOffset>(
155  code, kReleaseStore);
156  }
157 
158  // Trigger the write barriers after we dropped the JIT write permissions.
159  RelocateFromDescWriteBarriers(heap, desc, code->constant_pool(), *promise,
160  no_gc);
161  CONDITIONAL_PROTECTED_POINTER_WRITE_BARRIER(*this, kCodeOffset, code,
162  UPDATE_WRITE_BARRIER);
163 
164  code->FlushICache();
165 }
