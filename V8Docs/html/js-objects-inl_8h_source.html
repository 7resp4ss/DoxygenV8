178 template <typename TSlot>
179 void JSObject::EnsureCanContainElements(Isolate* isolate,
180  DirectHandle<JSObject> object,
181  TSlot objects, uint32_t count,
182  EnsureElementsMode mode) {
183  static_assert(std::is_same_v<TSlot, FullObjectSlot> ||
184  std::is_same_v<TSlot, ObjectSlot>,
185  "Only ObjectSlot and FullObjectSlot are expected here");
186  ElementsKind current_kind = object->GetElementsKind();
187  ElementsKind target_kind = current_kind;
188  {
189  DisallowGarbageCollection no_gc;
190  DCHECK(mode != ALLOW_COPIED_DOUBLE_ELEMENTS);
191  bool is_holey = IsHoleyElementsKind(current_kind);
192  if (current_kind == HOLEY_ELEMENTS) return;
193  Tagged<Object> the_hole = GetReadOnlyRoots().the_hole_value();
194  for (uint32_t i = 0; i < count; ++i, ++objects) {
195  Tagged<Object> current = *objects;
196  if (current == the_hole) {
197  is_holey = true;
198  target_kind = GetHoleyElementsKind(target_kind);
199 #ifdef V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
200  } else if (IsUndefined(current)) {
201  if (mode == ALLOW_CONVERTED_DOUBLE_ELEMENTS) {
202  if (IsSmiElementsKind(target_kind)) {
203  target_kind = HOLEY_DOUBLE_ELEMENTS;
204  } else if (target_kind == PACKED_DOUBLE_ELEMENTS) {
205  target_kind = HOLEY_DOUBLE_ELEMENTS;
206  } else {
207  DCHECK(target_kind == PACKED_ELEMENTS ||
208  target_kind == HOLEY_ELEMENTS ||
209  target_kind == HOLEY_DOUBLE_ELEMENTS);
210  }
211  } else if (is_holey) {
212  if (IsSmiElementsKind(target_kind)) {
213  target_kind = HOLEY_ELEMENTS;
214  break;
215  }
216  } else {
217  target_kind = PACKED_ELEMENTS;
218  }
219 #endif // V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
220  } else if (!IsSmi(current)) {
221  if (mode == ALLOW_CONVERTED_DOUBLE_ELEMENTS && IsNumber(current)) {
222  if (IsSmiElementsKind(target_kind)) {
223  if (is_holey) {
224  target_kind = HOLEY_DOUBLE_ELEMENTS;
225  } else {
226  target_kind = PACKED_DOUBLE_ELEMENTS;
227  }
228  }
229  } else if (is_holey) {
230  target_kind = HOLEY_ELEMENTS;
231  break;
232  } else {
233  target_kind = PACKED_ELEMENTS;
234  }
235  }
236  }
237  }
238  if (target_kind != current_kind) {
239  TransitionElementsKind(isolate, object, target_kind);
240  }
241 }

243 void JSObject::EnsureCanContainElements(Isolate* isolate,
244  DirectHandle<JSObject> object,
245  DirectHandle<FixedArrayBase> elements,
246  uint32_t length,
247  EnsureElementsMode mode) {
248  ReadOnlyRoots roots = GetReadOnlyRoots();
249  if (elements->map() != roots.fixed_double_array_map()) {
250  DCHECK(elements->map() == roots.fixed_array_map() ||
251  elements->map() == roots.fixed_cow_array_map());
252  if (mode == ALLOW_COPIED_DOUBLE_ELEMENTS) {
253  mode = DONT_ALLOW_DOUBLE_ELEMENTS;
254  }
255  ObjectSlot objects = Cast<FixedArray>(elements)->RawFieldOfFirstElement();
256  EnsureCanContainElements(isolate, object, objects, length, mode);
257  return;
258  }
259 
260  DCHECK(mode == ALLOW_COPIED_DOUBLE_ELEMENTS);
261  if (object->GetElementsKind() == HOLEY_SMI_ELEMENTS) {
262  TransitionElementsKind(isolate, object, HOLEY_DOUBLE_ELEMENTS);
263  } else if (object->GetElementsKind() == PACKED_SMI_ELEMENTS) {
264  auto double_array = Cast<FixedDoubleArray>(elements);
265  for (uint32_t i = 0; i < length; ++i) {
266  if (double_array->is_the_hole(i)) {
267  TransitionElementsKind(isolate, object, HOLEY_DOUBLE_ELEMENTS);
268  return;
269  }
270  }
271  TransitionElementsKind(isolate, object, PACKED_DOUBLE_ELEMENTS);
272  }
273 }

425 std::optional<Tagged<Object>> JSObject::RawInobjectPropertyAt(
426  PtrComprCageBase cage_base, Tagged<Map> original_map,
427  FieldIndex index) const {
428  CHECK(index.is_inobject());
429 
430  // This method implements a "snapshot" protocol to protect against reading out
431  // of bounds of an object. It's used to access a fast in-object property from
432  // a background thread with no locking. That caller does have the guarantee
433  // that a garbage collection cannot happen during its query. However, it must
434  // contend with the main thread altering the object in heavy ways through
435  // object migration. Specifically, the object can get smaller. Initially, this
436  // may seem benign, because object migration fills the freed-up space with
437  // FillerMap words which, even though they offer wrong values, are at
438  // least tagged values.
439 
440  // However, there is an additional danger. Sweeper threads may discover the
441  // filler words and offer that space to the main thread for allocation. Should
442  // a HeapNumber be allocated into that space while we're reading a property at
443  // that location (from our out-of-date information), we risk interpreting a
444  // double value as a pointer. This must be prevented.
445  //
446  // We do this by:
447  //
448  // a) Reading the map first
449  // b) Reading the property with acquire semantics (but do not inspect it!)
450  // c) Re-read the map with acquire semantics.
451  //
452  // Only if the maps match can the property be inspected. It may have a "wrong"
453  // value, but it will be within the bounds of the objects instance size as
454  // given by the map and it will be a valid Smi or object pointer.
455  Tagged<Object> maybe_tagged_object =
456  TaggedField<Object>::Acquire_Load(cage_base, *this, index.offset());
457  if (original_map != map(cage_base, kAcquireLoad)) return {};
458  return maybe_tagged_object;
459 }

498 void JSObject::WriteToField(InternalIndex descriptor, PropertyDetails details,
499  Tagged<Object> value) {
500  DCHECK_EQ(PropertyLocation::kField, details.location());
501  DCHECK_EQ(PropertyKind::kData, details.kind());
502  DisallowGarbageCollection no_gc;
503  FieldIndex index = FieldIndex::ForDetails(map(), details);
504  if (details.representation().IsDouble()) {
505  // Manipulating the signaling NaN used for the hole and uninitialized
506  // double field sentinel in C++, e.g. with base::bit_cast or
507  // value()/set_value(), will change its value on ia32 (the x87 stack is used
508  // to return values and stores to the stack silently clear the signalling
509  // bit).
510  uint64_t bits;
511  if (IsSmi(value)) {
512  bits = base::bit_cast<uint64_t>(static_cast<double>(Smi::ToInt(value)));
513  } else if (IsUninitialized(value)) {
514  bits = kHoleNanInt64;
515  } else {
516  DCHECK(IsHeapNumber(value));
517  bits = Cast<HeapNumber>(value)->value_as_bits();
518  }
519  auto box = Cast<HeapNumber>(RawFastPropertyAt(index));
520  box->set_value_as_bits(bits);
521  } else {
522  FastPropertyAtPut(index, value);
523  }
524 }

588 void JSObject::InitializeBody(Tagged<Map> map, int start_offset,
589  bool is_slack_tracking_in_progress,
590  MapWord filler_map,
591  Tagged<Object> undefined_filler) {
592  int size = map->instance_size();
593  int offset = start_offset;
594 
595  // embedder data slots need to be initialized separately
596  if (MayHaveEmbedderFields(map)) {
597  int embedder_field_start = GetEmbedderFieldsStartOffset(map);
598  int embedder_field_count = GetEmbedderFieldCount(map);
599 
600  // fill start with references to the undefined value object
601  DCHECK_LE(offset, embedder_field_start);
602  while (offset < embedder_field_start) {
603  WRITE_FIELD(*this, offset, undefined_filler);
604  offset += kTaggedSize;
605  }
606 
607  // initialize embedder data slots
608  DCHECK_EQ(offset, embedder_field_start);
609  for (int i = 0; i < embedder_field_count; i++) {
610  // TODO(v8): consider initializing embedded data slots with Smi::zero().
611  EmbedderDataSlot(Tagged<JSObject>(*this), i).Initialize(undefined_filler);
612  offset += kEmbedderDataSlotSize;
613  }
614  } else {
615  DCHECK_EQ(0, GetEmbedderFieldCount(map));
616  }
617 
618  DCHECK_LE(offset, size);
619  if (is_slack_tracking_in_progress) {
620  int end_of_pre_allocated_offset =
621  size - (map->UnusedPropertyFields() * kTaggedSize);
622  DCHECK_LE(kHeaderSize, end_of_pre_allocated_offset);
623  DCHECK_LE(offset, end_of_pre_allocated_offset);
624  // fill pre allocated slots with references to the undefined value object
625  while (offset < end_of_pre_allocated_offset) {
626  WRITE_FIELD(*this, offset, undefined_filler);
627  offset += kTaggedSize;
628  }
629  // fill the remainder with one word filler objects (ie just a map word)
630  while (offset < size) {
631  Tagged<Object> fm = Tagged<Object>(filler_map.ptr());
632  WRITE_FIELD(*this, offset, fm);
633  offset += kTaggedSize;
634  }
635  } else {
636  while (offset < size) {
637  // fill everything with references to the undefined value object
638  WRITE_FIELD(*this, offset, undefined_filler);
639  offset += kTaggedSize;
640  }
641  }
642 }
