406 // Addressing modes represent the "shape" of inputs to an instruction.
407 // Many instructions support multiple addressing modes. Addressing modes
408 // are encoded into the InstructionCode of the instruction and tell the
409 // code generator after register allocation which assembler method to call.
410 //
411 // We use the following local notation for addressing modes:
412 //
413 // R = register
414 // O = register or stack slot
415 // D = double register
416 // I = immediate (handle, external, int32)
417 // MRI = [register + immediate]
418 // MRR = [register + register]
419 #define TARGET_ADDRESSING_MODE_LIST(V) \
420  V(MRI) /* [%r0 + K] */ \
421  V(MRR) /* [%r0 + %r1] */ \
422  V(Operand2_R_LSL_I) /* %r0 LSL K */ \
423  V(Operand2_R_LSR_I) /* %r0 LSR K */ \
424  V(Operand2_R_ASR_I) /* %r0 ASR K */ \
425  V(Operand2_R_ROR_I) /* %r0 ROR K */ \
426  V(Operand2_R_UXTB) /* %r0 UXTB (unsigned extend byte) */ \
427  V(Operand2_R_UXTH) /* %r0 UXTH (unsigned extend halfword) */ \
428  V(Operand2_R_SXTB) /* %r0 SXTB (signed extend byte) */ \
429  V(Operand2_R_SXTH) /* %r0 SXTH (signed extend halfword) */ \
430  V(Operand2_R_SXTW) /* %r0 SXTW (signed extend word) */ \
431  V(Root) /* [%rr + K] */
