105  // Returns the maximum time delta, which should be greater than any reasonable
106  // time delta we might compare it to. Adding or subtracting the maximum time
107  // delta to a time or another time delta has an undefined result.
108  static constexpr TimeDelta Max();
109 
110  // Returns the minimum time delta, which should be less than than any
111  // reasonable time delta we might compare it to. Adding or subtracting the
112  // minimum time delta to a time or another time delta has an undefined result.
113  static constexpr TimeDelta Min();

116  constexpr bool IsZero() const { return delta_ == 0; }

119  constexpr bool IsMax() const {
120  return delta_ == std::numeric_limits<int64_t>::max();
121  }

123  return delta_ == std::numeric_limits<int64_t>::min();
124  }
125

229 // TimeBase--------------------------------------------------------------------
230 
231 // Provides value storage and comparison/math operations common to all time
232 // classes. Each subclass provides for strong type-checking to ensure
233 // semantically meaningful comparison/math of time values from the same clock
234 // source or timeline.
235 template <class TimeClass>
236 class TimeBase : public TimeConstants {
237  public:
238 #if V8_OS_WIN
239  // To avoid overflow in QPC to Microseconds calculations, since we multiply
240  // by kMicrosecondsPerSecond, then the QPC value should not exceed
241  // (2^63 - 1) / 1E6. If it exceeds that threshold, we divide then multiply.
242  static constexpr int64_t kQPCOverflowThreshold = INT64_C(0x8637BD05AF7);
243 #endif
244 
245  // Returns true if this object has not been initialized.
246  //
247  // Warning: Be careful when writing code that performs math on time values,
248  // since it's possible to produce a valid "zero" result that should not be
249  // interpreted as a "null" value.
250  constexpr bool IsNull() const { return us_ == 0; }
251 
252  // Returns the maximum/minimum times, which should be greater/less than any
253  // reasonable time with which we might compare it.
254  static TimeClass Max() {
255  return TimeClass(std::numeric_limits<int64_t>::max());
256  }
257  static TimeClass Min() {
258  return TimeClass(std::numeric_limits<int64_t>::min());
259  }
260 
261  // Returns true if this object represents the maximum/minimum time.
262  constexpr bool IsMax() const {
263  return us_ == std::numeric_limits<int64_t>::max();
264  }
265  constexpr bool IsMin() const {
266  return us_ == std::numeric_limits<int64_t>::min();
267  }
268 
269  // For serializing only. Use FromInternalValue() to reconstitute. Please don't
270  // use this and do arithmetic on it, as it is more error prone than using the
271  // provided operators.
272  int64_t ToInternalValue() const { return us_; }
273 
274  // The amount of time since the origin (or "zero") point. This is a syntactic
275  // convenience to aid in code readability, mainly for debugging/testing use
276  // cases.
277  //
278  // Warning: While the Time subclass has a fixed origin point, the origin for
279  // the other subclasses can vary each time the application is restarted.
280  constexpr TimeDelta since_origin() const {
281  return TimeDelta::FromMicroseconds(us_);
282  }
283 
284  TimeClass& operator=(TimeClass other) {
285  us_ = other.us_;
286  return *(static_cast<TimeClass*>(this));
287  }
288 
289  // Compute the difference between two times.
290  TimeDelta operator-(TimeClass other) const {
291  return TimeDelta::FromMicroseconds(us_ - other.us_);
292  }
293 
294  // Return a new time modified by some delta.
295  TimeClass operator+(TimeDelta delta) const {
296  return TimeClass(bits::SignedSaturatedAdd64(delta.delta_, us_));
297  }
298  TimeClass operator-(TimeDelta delta) const {
299  return TimeClass(-bits::SignedSaturatedSub64(delta.delta_, us_));
300  }
301 
302  // Modify by some time delta.
303  TimeClass& operator+=(TimeDelta delta) {
304  return static_cast<TimeClass&>(*this = (*this + delta));
305  }
306  TimeClass& operator-=(TimeDelta delta) {
307  return static_cast<TimeClass&>(*this = (*this - delta));
308  }
309 
310  constexpr auto operator<=>(const TimeBase&) const = default;
311 
312  // Converts an integer value representing TimeClass to a class. This is used
313  // when deserializing a |TimeClass| structure, using a value known to be
314  // compatible. It is not provided as a constructor because the integer type
315  // may be unclear from the perspective of a caller.
316  static TimeClass FromInternalValue(int64_t us) { return TimeClass(us); }
317 
318  protected:
319  explicit constexpr TimeBase(int64_t us) : us_(us) {}
320 
321  // Time value in a microsecond timebase.
322  int64_t us_;
323 };

343

347  // For timing sensitive unittests, this function should be used.
348  static Time NowFromSystemTime();
349 
350  // Returns the time for epoch in Unix-like system (Jan 1, 1970).
351  static Time UnixEpoch() { return Time(0); }
352 
353  // Converts to/from POSIX time specs.
354  static Time FromTimespec(struct timespec ts);
355  struct timespec ToTimespec() const;
356 
357  // Converts to/from POSIX time values.
358  static Time FromTimeval(struct timeval tv);
359  struct timeval ToTimeval() const;
360 
361  // Converts to/from Windows file times.
362  static Time FromFiletime(struct _FILETIME ft);
363  struct _FILETIME ToFiletime() const;
364 
365  // Converts to/from the Javascript convention for times, a number of
366  // milliseconds since the epoch:

382 // -----------------------------------------------------------------------------
383 // TimeTicks
384 //
385 // This class represents an abstract time that is most of the time incrementing
386 // for use in measuring time durations. It is internally represented in
387 // microseconds. It can not be converted to a human-readable time, but is
388 // guaranteed not to decrease (if the user changes the computer clock,
389 // Time::Now() may actually decrease or jump). But note that TimeTicks may
390 // "stand still", for example if the computer suspended.
391 
392 class V8_BASE_EXPORT TimeTicks final
393  : public time_internal::TimeBase<TimeTicks> {
394  public:
395  constexpr TimeTicks() : TimeBase(0) {}
396 
397  // Platform-dependent tick count representing "right now." When
398  // IsHighResolution() returns false, the resolution of the clock could be as
399  // coarse as ~15.6ms. Otherwise, the resolution should be no worse than one
400  // microsecond.
401  // This method never returns a null TimeTicks.
402  static TimeTicks Now();
403 
404  // Returns true if the high-resolution clock is working on this system.
405  static bool IsHighResolution();
406 
407  static constexpr TimeTicks FromMsTicksForTesting(int64_t ticks) {
408  return TimeTicks(ticks * kMicrosecondsPerMillisecond);
409  }
410 
411  private:
412  friend class time_internal::TimeBase<TimeTicks>;
413 
414  // Please use Now() to create a new object. This is for internal use
415  // and testing. Ticks are in microseconds.
416  explicit constexpr TimeTicks(int64_t ticks) : TimeBase(ticks) {}
417 };
418 
419 inline TimeTicks operator+(const TimeDelta& delta, const TimeTicks& ticks) {
