57  // LocalSweeper holds local data structures required for sweeping and is used
58  // to initiate sweeping and promoted page iteration on multiple threads. Each
59  // thread should holds its own LocalSweeper. Once sweeping is done, all
60  // LocalSweepers should be finalized on the main thread.
61  //
62  // LocalSweeper is not thread-safe and should not be concurrently by several
63  // threads. The exceptions to this rule are allocations during parallel
64  // evacuation and from concurrent allocators. In practice the data structures
65  // in LocalSweeper are only actively used for new space sweeping. Since
66  // parallel evacuators and concurrent allocators never try to allocate in new
67  // space, they will never contribute to new space sweeping and thus can use
68  // the main thread's local sweeper without risk of data races.
69  class LocalSweeper final {
70  public:
71  explicit LocalSweeper(Sweeper* sweeper) : sweeper_(sweeper) {
72  DCHECK_NOT_NULL(sweeper_);
73  }

343 template <typename ShouldYieldCallback>
344 bool Sweeper::LocalSweeper::ContributeAndWaitForPromotedPagesIterationImpl(
345  ShouldYieldCallback should_yield_callback) {
346  if (!sweeper_->sweeping_in_progress()) return true;
347  if (!sweeper_->IsIteratingPromotedPages()) return true;
348  if (!ParallelIteratePromotedPagesImpl(should_yield_callback)) return false;
349  base::MutexGuard guard(
350  &sweeper_->promoted_pages_iteration_notification_mutex_);
351  // Check again that iteration is not yet finished.
352  if (!sweeper_->IsIteratingPromotedPages()) return true;
353  if (should_yield_callback()) {
354  return false;
355  }
356  sweeper_->promoted_pages_iteration_notification_variable_.Wait(
357  &sweeper_->promoted_pages_iteration_notification_mutex_);
358  return true;
359 }
360 
361 template <typename ShouldYieldCallback>
362 bool Sweeper::LocalSweeper::ParallelIteratePromotedPagesImpl(
363  ShouldYieldCallback should_yield_callback) {
364  while (!should_yield_callback()) {
365  MutablePageMetadata* chunk = sweeper_->GetPromotedPageSafe();
366  if (chunk == nullptr) return true;
367  ParallelIteratePromotedPage(chunk);
368  }
369  return false;
370 }
