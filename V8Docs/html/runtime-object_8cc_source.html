25 MaybeDirectHandle<Object> Runtime::GetObjectProperty(
26  Isolate* isolate, DirectHandle<JSAny> lookup_start_object,
27  DirectHandle<Object> key, DirectHandle<JSAny> receiver, bool* is_found) {
28  if (receiver.is_null()) {
29  receiver = lookup_start_object;
30  }
31  if (IsNullOrUndefined(*lookup_start_object, isolate)) {
32  ErrorUtils::ThrowLoadFromNullOrUndefined(isolate, lookup_start_object, key);
33  return MaybeDirectHandle<Object>();
34  }
35 
36  bool success = false;
37  PropertyKey lookup_key(isolate, key, &success);
38  if (!success) return MaybeDirectHandle<Object>();
39  LookupIterator it =
40  LookupIterator(isolate, receiver, lookup_key, lookup_start_object);
41 
42  MaybeDirectHandle<Object> result = Object::GetProperty(&it);
43  if (result.is_null()) {
44  return result;
45  }
46  if (is_found) {
47  *is_found = it.state() != LookupIterator::TYPED_ARRAY_INDEX_NOT_FOUND &&
48  it.state() != LookupIterator::NOT_FOUND;
49  }
50 
51  return result;
52 }

54 MaybeDirectHandle<Object> Runtime::HasProperty(Isolate* isolate,
55  DirectHandle<Object> object,
56  DirectHandle<Object> key) {
57  // Check that {object} is actually a receiver.
58  if (!IsJSReceiver(*object)) {
59  THROW_NEW_ERROR(
60  isolate,
61  NewTypeError(MessageTemplate::kInvalidInOperatorUse, key, object));
62  }
63  DirectHandle<JSReceiver> receiver = Cast<JSReceiver>(object);
64 
65  // Convert the {key} to a name.
66  DirectHandle<Name> name;
67  ASSIGN_RETURN_ON_EXCEPTION(isolate, name, Object::ToName(isolate, key));
68 
69  // Lookup the {name} on {receiver}.
70  Maybe<bool> maybe = JSReceiver::HasProperty(isolate, receiver, name);
71  if (maybe.IsNothing()) return MaybeDirectHandle<Object>();
72  return isolate->factory()->ToBoolean(maybe.FromJust());
73 }

160 RUNTIME_FUNCTION(Runtime_ObjectHasOwnProperty) {
161  HandleScope scope(isolate);
162  DirectHandle<Object> property = args.at(1);
163 
164  // TODO(ishell): To improve performance, consider performing the to-string
165  // conversion of {property} before calling into the runtime.
166  bool success;
167  PropertyKey key(isolate, property, &success);
168  if (!success) return ReadOnlyRoots(isolate).exception();
169 
170  DirectHandle<JSAny> object = args.at<JSAny>(0);
171 
172  if (IsJSModuleNamespace(*object)) {
173  LookupIterator it(isolate, object, key, LookupIterator::OWN);
174  PropertyDescriptor desc;
175  Maybe<bool> result = JSReceiver::GetOwnPropertyDescriptor(&it, &desc);
176  if (!result.IsJust()) return ReadOnlyRoots(isolate).exception();
177  return isolate->heap()->ToBoolean(result.FromJust());
178 
179  } else if (IsJSObject(*object)) {
180  DirectHandle<JSObject> js_obj = Cast<JSObject>(object);
181  // Fast case: either the key is a real named property or it is not
182  // an array index and there are no interceptors or hidden
183  // prototypes.
184  // TODO(jkummerow): Make JSReceiver::HasOwnProperty fast enough to
185  // handle all cases directly (without this custom fast path).
186  {
187  LookupIterator::Configuration c = LookupIterator::OWN_SKIP_INTERCEPTOR;
188  LookupIterator it(isolate, js_obj, key, js_obj, c);
189  Maybe<bool> maybe = JSReceiver::HasProperty(&it);
190  if (maybe.IsNothing()) return ReadOnlyRoots(isolate).exception();
191  DCHECK(!isolate->has_exception());
192  if (maybe.FromJust()) return ReadOnlyRoots(isolate).true_value();
193  }
194 
195  Tagged<Map> map = js_obj->map();
196  if (!IsJSGlobalProxyMap(map) &&
197  (key.is_element() && key.index() <= JSObject::kMaxElementIndex
198  ? !map->has_indexed_interceptor()
199  : !map->has_named_interceptor())) {
200  return ReadOnlyRoots(isolate).false_value();
201  }
202 
203  // Slow case.
204  LookupIterator it(isolate, js_obj, key, js_obj, LookupIterator::OWN);
205  Maybe<bool> maybe = JSReceiver::HasProperty(&it);
206  if (maybe.IsNothing()) return ReadOnlyRoots(isolate).exception();
207  DCHECK(!isolate->has_exception());
208  return isolate->heap()->ToBoolean(maybe.FromJust());
209 
210  } else if (IsJSProxy(*object)) {
211  LookupIterator it(isolate, object, key, Cast<JSProxy>(object),
212  LookupIterator::OWN);
213  Maybe<PropertyAttributes> attributes =
214  JSReceiver::GetPropertyAttributes(&it);
215  if (attributes.IsNothing()) return ReadOnlyRoots(isolate).exception();
216  return isolate->heap()->ToBoolean(attributes.FromJust() != ABSENT);
217 
218  } else if (IsString(*object)) {
219  return isolate->heap()->ToBoolean(
220  key.is_element()
221  ? key.index() < static_cast<size_t>(Cast<String>(*object)->length())
222  : key.name()->Equals(ReadOnlyRoots(isolate).length_string()));
223  } else if (IsNullOrUndefined(*object, isolate)) {
224  THROW_NEW_ERROR_RETURN_FAILURE(
225  isolate, NewTypeError(MessageTemplate::kUndefinedOrNullToObject));
226  }
227 
228  return ReadOnlyRoots(isolate).false_value();
229 }

306 RUNTIME_FUNCTION(Runtime_AddPrivateBrand) {
307  HandleScope scope(isolate);
308  DCHECK_EQ(args.length(), 4);
309  DirectHandle<JSReceiver> receiver = args.at<JSReceiver>(0);
310  DirectHandle<Symbol> brand = args.at<Symbol>(1);
311  DirectHandle<Context> context = args.at<Context>(2);
312  int depth = args.smi_value_at(3);
313  DCHECK(brand->is_private_name());
314 
315  LookupIterator it(isolate, receiver, brand, LookupIterator::OWN);
316 
317  if (it.IsFound()) {
318  THROW_NEW_ERROR_RETURN_FAILURE(
319  isolate,
320  NewTypeError(MessageTemplate::kInvalidPrivateBrandReinitialization,
321  brand));
322  }
323 
324  PropertyAttributes attributes =
325  static_cast<PropertyAttributes>(DONT_ENUM | DONT_DELETE | READ_ONLY);
326 
327  // Look for the context in |depth| in the context chain to store it
328  // in the instance with the brand variable as key, which is needed by
329  // the debugger for retrieving names of private methods.
330  DCHECK_GE(depth, 0);
331  for (; depth > 0; depth--) {
332  context = direct_handle(
333  Cast<Context>(context->GetNoCell(Context::PREVIOUS_INDEX)), isolate);
334  }
335  DCHECK_EQ(context->scope_info()->scope_type(), ScopeType::CLASS_SCOPE);
336  Maybe<bool> added_brand = Object::AddDataProperty(
337  &it, context, attributes, Just(kThrowOnError), StoreOrigin::kMaybeKeyed);
338  // Objects in shared space are fixed shape, so private symbols cannot be
339  // added.
340  if (V8_UNLIKELY(IsAlwaysSharedSpaceJSObject(*receiver))) {
341  CHECK(added_brand.IsNothing());
342  return ReadOnlyRoots(isolate).exception();
343  }
344  CHECK(added_brand.IsJust());
345  return *receiver;
346 }

641  // the global proxy object can return a valid result even though the
642  // global proxy object never has properties. This is the case
643  // because the global proxy object forwards everything to its hidden
644  // prototype including own lookups.
645  //
646  // Additionally, we need to make sure that we do not cache results
647  // for objects that require access checks.

650  // below; and speed up subsequent conversion to index.
651  uint32_t index;
652  if (IsString(*key_obj) && Cast<String>(*key_obj)->AsArrayIndex(&index)) {
653  key_obj = isolate->factory()->NewNumberFromUint(index);
654  }

656  DirectHandle<JSObject> lookup_start_object =
657  Cast<JSObject>(lookup_start_obj);
658  if (!IsJSGlobalProxy(*lookup_start_object) &&
659  !IsAccessCheckNeeded(*lookup_start_object) && IsName(*key_obj)) {
660  DirectHandle<Name> key = Cast<Name>(key_obj);
661  key_obj = key = isolate->factory()->InternalizeName(key);
662

665  // Attempt dictionary lookup.

667  Cast<JSGlobalObject>(*lookup_start_object)

670  if (entry.is_found()) {

676  }

682  lookup_start_object->property_dictionary_swiss();

686  return dictionary->ValueAt(entry);

704  // become PACKED_DOUBLE_ELEMENTS.

1407  Isolate* isolate, DirectHandle<JSReceiver> receiver,
