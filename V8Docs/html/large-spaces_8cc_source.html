108 AllocationResult OldLargeObjectSpace::AllocateRaw(LocalHeap* local_heap,
109  int object_size,
110  Executability executable,
111  AllocationHint hint) {
112  object_size = ALIGN_TO_ALLOCATION_ALIGNMENT(object_size);
113  DCHECK_IMPLIES(identity() == SHARED_LO_SPACE,
114  !allocation_counter_.HasAllocationObservers());
115  DCHECK_IMPLIES(identity() == SHARED_LO_SPACE,
116  pending_object() == kNullAddress);
117 
118  // Check if we want to force a GC before growing the old space further.
119  // If so, fail the allocation.
120  if (!heap()->ShouldExpandOldGenerationOnSlowAllocation(
121  local_heap, AllocationOrigin::kRuntime) ||
122  !heap()->CanExpandOldGeneration(object_size)) {
123  return AllocationResult::Failure();
124  }
125 
126  heap()->StartIncrementalMarkingIfAllocationLimitIsReached(
127  local_heap, heap()->GCFlagsForIncrementalMarking(),
128  kGCCallbackScheduleIdleGarbageCollection);
129 
130  LargePageMetadata* page = AllocateLargePage(object_size, executable, hint);
131  if (page == nullptr) return AllocationResult::Failure();
132  Tagged<HeapObject> object = page->GetObject();
133  if (local_heap->is_main_thread() && identity() != SHARED_LO_SPACE) {
134  UpdatePendingObject(object);
135  }
136  if (v8_flags.sticky_mark_bits ||
137  heap()->incremental_marking()->black_allocation()) {
138  heap()->marking_state()->TryMarkAndAccountLiveBytes(object, object_size);
139  }
140  DCHECK_IMPLIES(heap()->incremental_marking()->black_allocation(),
141  heap()->marking_state()->IsMarked(object));
142  page->Chunk()->InitializationMemoryFence();
143  heap()->NotifyOldGenerationExpansion(local_heap, identity(), page);
144 
145  if (local_heap->is_main_thread() && identity() != SHARED_LO_SPACE) {
146  AdvanceAndInvokeAllocationObservers(object.address(),
147  static_cast<size_t>(object_size));
148  }
149  return AllocationResult::FromObject(object);
150 }

152 LargePageMetadata* LargeObjectSpace::AllocateLargePage(int object_size,
153  Executability executable,
154  AllocationHint hint) {
155  base::MutexGuard expansion_guard(heap_->heap_expansion_mutex());
156 
157  if (identity() != NEW_LO_SPACE &&
158  !heap()->IsOldGenerationExpansionAllowed(object_size, expansion_guard)) {
159  return nullptr;
160  }
161 
162  LargePageMetadata* page = heap()->memory_allocator()->AllocateLargePage(
163  this, object_size, executable, hint);
164  if (page == nullptr) return nullptr;
165  DCHECK_GE(page->area_size(), static_cast<size_t>(object_size));
166 
167  {
168  base::RecursiveMutexGuard guard(&allocation_mutex_);
169  AddPage(page, object_size);
170  }
171 
172  return page;
173 }

183  MemoryChunk* chunk = page->Chunk();
184  DCHECK_EQ(page->owner_identity(), NEW_LO_SPACE);
185  DCHECK(chunk->IsLargePage());
186  DCHECK(chunk->IsFlagSet(MemoryChunk::FROM_PAGE));

189  static_cast<LargeObjectSpace*>(page->owner())->RemovePage(page);
190  chunk->ClearFlagNonExecutable(MemoryChunk::FROM_PAGE);

193  AddPage(page, static_cast<size_t>(page->GetObject()->Size(cage_base)));

196 void LargeObjectSpace::AddPage(LargePageMetadata* page, size_t object_size) {
197  size_ += static_cast<int>(page->size());
198  AccountCommitted(page->size());
199  objects_size_ += object_size;
200  page_count_++;
201  memory_chunk_list_.PushBack(page);
202  page->set_owner(this);
203  ForAll<ExternalBackingStoreType>(
204  [this, page](ExternalBackingStoreType type, int index) {
205  IncrementExternalBackingStoreBytes(
206  type, page->ExternalBackingStoreBytes(type));
207  });
208 }

210 void LargeObjectSpace::RemovePage(LargePageMetadata* page) {
211  size_ -= static_cast<int>(page->size());
212  AccountUncommitted(page->size());
213  page_count_--;
214  memory_chunk_list_.Remove(page);
215  page->set_owner(nullptr);
216  ForAll<ExternalBackingStoreType>(
217  [this, page](ExternalBackingStoreType type, int index) {
218  DecrementExternalBackingStoreBytes(
219  type, page->ExternalBackingStoreBytes(type));
220  });
221 }

223 void LargeObjectSpace::ShrinkPageToObjectSize(LargePageMetadata* page,
224  Tagged<HeapObject> object,
225  size_t object_size) {
226  MemoryChunk* chunk = page->Chunk();
227 #ifdef DEBUG
228  PtrComprCageBase cage_base(heap()->isolate());
229  DCHECK_EQ(object, page->GetObject());
230  DCHECK_EQ(object_size, page->GetObject()->Size(cage_base));
231  DCHECK_EQ(chunk->executable(), NOT_EXECUTABLE);
232 #endif // DEBUG
233 
234  const size_t used_committed_size =
235  ::RoundUp(chunk->Offset(object.address()) + object_size,
236  MemoryAllocator::GetCommitPageSize());
237 
238  // Object shrunk since last GC.
239  if (object_size < page->area_size()) {
240  page->ClearOutOfLiveRangeSlots(object.address() + object_size);
241  const Address new_area_end = page->area_start() + object_size;
242 
243  // Object shrunk enough that we can even free some OS pages.
244  if (used_committed_size < page->size()) {
245  const size_t bytes_to_free = page->size() - used_committed_size;
246  heap()->memory_allocator()->PartialFreeMemory(
247  page, chunk->address() + used_committed_size, bytes_to_free,
248  new_area_end);
249  size_ -= bytes_to_free;
250  AccountUncommitted(bytes_to_free);
251  } else {
252  // Can't free OS page but keep object area up-to-date.
253  page->set_area_end(new_area_end);
254  }
255  }
256 
257  DCHECK_EQ(used_committed_size, page->size());
258  DCHECK_EQ(object_size, page->area_size());
259 }

261 void LargeObjectSpace::UpdateAccountingAfterResizingObject(
262  size_t old_object_size, size_t new_object_size) {
263  DCHECK_GE(new_object_size, old_object_size);
264  size_t old_size_committed =
265  ::RoundUp(old_object_size, MemoryAllocator::GetCommitPageSize());
266  size_t new_size_committed =
267  ::RoundUp(new_object_size, MemoryAllocator::GetCommitPageSize());
268  DCHECK_GE(new_size_committed, old_size_committed);
269 
270  size_ += new_size_committed - old_size_committed;
271  AccountCommitted(new_size_committed - old_size_committed);
272  objects_size_.fetch_add(new_object_size - old_object_size);
273 }

299 #ifdef VERIFY_HEAP
300 // We do not assume that the large object iterator works, because it depends
301 // on the invariants we are checking during verification.
302 void LargeObjectSpace::Verify(Isolate* isolate,
303  SpaceVerificationVisitor* visitor) const {
304  size_t external_backing_store_bytes[static_cast<int>(
305  ExternalBackingStoreType::kNumValues)] = {0};
306 
307  PtrComprCageBase cage_base(isolate);
308  for (const LargePageMetadata* chunk = first_page(); chunk != nullptr;
309  chunk = chunk->next_page()) {
310  visitor->VerifyPage(chunk);
311 
312  // Each chunk contains an object that starts at the large object page's
313  // object area start.
314  Tagged<HeapObject> object = chunk->GetObject();
315  PageMetadata* page = PageMetadata::FromHeapObject(object);
316  CHECK(object.address() == page->area_start());
317 
318  // Only certain types may be in the large object space:
319 #define V(Name) Is##Name(object, cage_base) ||
320  const bool is_valid_lo_space_object =
321  DYNAMICALLY_SIZED_HEAP_OBJECT_LIST(V) false;
322 #undef V
323  if (!is_valid_lo_space_object) {
324  i::Print(object);
325  FATAL("Found invalid Object (instance_type=%i) in large object space.",
326  object->map(cage_base)->instance_type());
327  }
328 
329  // Invoke visitor on each object.
330  visitor->VerifyObject(object);
331 
332  ForAll<ExternalBackingStoreType>(
333  [chunk, &external_backing_store_bytes](ExternalBackingStoreType type,
334  int index) {
335  external_backing_store_bytes[index] +=
336  chunk->ExternalBackingStoreBytes(type);
337  });
338 
339  visitor->VerifyPageDone(chunk);
340  }
341  ForAll<ExternalBackingStoreType>(
342  [this, external_backing_store_bytes](ExternalBackingStoreType type,
343  int index) {
344  CHECK_EQ(external_backing_store_bytes[index],
345  ExternalBackingStoreBytes(type));
346  });
347 }
348 #endif

374 AllocationResult NewLargeObjectSpace::AllocateRaw(LocalHeap* local_heap,
375  int object_size,
376  AllocationHint hint) {
377  object_size = ALIGN_TO_ALLOCATION_ALIGNMENT(object_size);
378  DCHECK(local_heap->is_main_thread());
379  // Do not allocate more objects if promoting the existing object would exceed
380  // the old generation capacity.
381  if (!heap()->CanExpandOldGeneration(SizeOfObjects())) {
382  return AllocationResult::Failure();
383  }
384 
385  // Allocation for the first object must succeed independent from the capacity.
386  if (SizeOfObjects() > 0 && static_cast<size_t>(object_size) > Available()) {
387  if (!heap()->ShouldExpandYoungGenerationOnSlowAllocation(object_size)) {
388  return AllocationResult::Failure();
389  }
390  }
391 
392  LargePageMetadata* page =
393  AllocateLargePage(object_size, NOT_EXECUTABLE, hint);
394  if (page == nullptr) return AllocationResult::Failure();
395 
396  // The size of the first object may exceed the capacity.
397  capacity_ = std::max(capacity_, SizeOfObjects());
398 
399  Tagged<HeapObject> result = page->GetObject();
400  MemoryChunk* chunk = page->Chunk();
401  chunk->SetFlagNonExecutable(MemoryChunk::TO_PAGE);
402  UpdatePendingObject(result);
403  if (v8_flags.minor_ms) {
404  page->ClearLiveness();
405  }
406  chunk->InitializationMemoryFence();
407  DCHECK(chunk->IsLargePage());
408  DCHECK_EQ(page->owner_identity(), NEW_LO_SPACE);
409  AdvanceAndInvokeAllocationObservers(result.address(),
410  static_cast<size_t>(object_size));
411  return AllocationResult::FromObject(result);
412 }

430 void NewLargeObjectSpace::FreeDeadObjects(
431  const std::function<bool(Tagged<HeapObject>)>& is_dead) {
432  DCHECK(!heap()->incremental_marking()->IsMarking());
433  size_t surviving_object_size = 0;
434  PtrComprCageBase cage_base(heap()->isolate());
435  std::vector<LargePageMetadata*> pages_to_pool;
436  if (v8_flags.large_page_pool) {
437  pages_to_pool.reserve(memory_chunk_list().size());
438  }
439  for (auto it = begin(); it != end();) {
440  LargePageMetadata* page = *it;
441  it++;
442  Tagged<HeapObject> object = page->GetObject();
443  if (is_dead(object)) {
444  RemovePage(page);
445  if (v8_flags.large_page_pool) {
446  pages_to_pool.push_back(page);
447  } else {
448  heap()->memory_allocator()->Free(
449  MemoryAllocator::FreeMode::kImmediately, page);
450  }
451  } else {
452  surviving_object_size += static_cast<size_t>(object->Size(cage_base));
453  }
454  }
455  // Right-trimming does not update the objects_size_ counter. We are lazily
456  // updating it after every GC.
457  objects_size_ = surviving_object_size;
458 
459  if (v8_flags.large_page_pool && !pages_to_pool.empty()) {
460  heap()->memory_allocator()->FreeLargePagesPooled(std::move(pages_to_pool));
461  }
462 }
