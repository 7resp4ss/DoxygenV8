42  void SampleStack(const v8::RegisterState& regs) override {
43  Isolate* isolate = reinterpret_cast<Isolate*>(this->isolate());
44  if (isolate->was_locker_ever_used() &&
45  (!isolate->thread_manager()->IsLockedByThread(
46  perThreadData_->thread_id()) ||
47  perThreadData_->thread_state() != nullptr)) {
48  ProfilerStats::Instance()->AddReason(
49  ProfilerStats::Reason::kIsolateNotLocked);
50  return;
51  }
52  TickSample* sample = processor_->StartTickSample();
53  if (sample == nullptr) {
54  ProfilerStats::Instance()->AddReason(
55  ProfilerStats::Reason::kTickBufferFull);
56  return;
57  }
58  // Every bailout up until here resulted in a dropped sample. From now on,
59  // the sample is created in the buffer.
60  sample->Init(isolate, regs, TickSample::kIncludeCEntryFrame,
61  /* update_stats */ true,
62  /* use_simulator_reg_state */ true, processor_->period());
63  if (is_counting_samples_ && !sample->timestamp.IsNull()) {
64  if (sample->state == JS) ++js_sample_count_;
65  if (sample->state == EXTERNAL) ++external_sample_count_;
66  }
67  processor_->FinishTickSample();
68  }
69 
70  private:
71  SamplingEventsProcessor* processor_;
72  Isolate::PerIsolateThreadData* perThreadData_;
73 };

149 void ProfilerEventsProcessor::AddDeoptStack(Address from, int fp_to_sp_delta) {
150  TickSampleEventRecord record(last_code_event_id_);
151  RegisterState regs;
152  Address fp = isolate_->c_entry_fp(isolate_->thread_local_top());
153  regs.sp = reinterpret_cast<void*>(fp - fp_to_sp_delta);
154  regs.fp = reinterpret_cast<void*>(fp);
155  regs.pc = reinterpret_cast<void*>(from);
156  record.sample.Init(isolate_, regs, TickSample::kSkipCEntryFrame, false,
157  false);
158  ticks_from_vm_buffer_.Enqueue(record);
159 }

161 void ProfilerEventsProcessor::AddCurrentStack(
162  bool update_stats, const std::optional<uint64_t> trace_id) {
163  TickSampleEventRecord record(last_code_event_id_);
164  RegisterState regs;
165  StackFrameIterator it(isolate_, isolate_->thread_local_top());
166  if (!it.done()) {
167  StackFrame* frame = it.frame();
168  regs.sp = reinterpret_cast<void*>(frame->sp());
169  regs.fp = reinterpret_cast<void*>(frame->fp());
170  regs.pc = reinterpret_cast<void*>(frame->pc());
171  }
172  record.sample.Init(isolate_, regs, TickSample::kSkipCEntryFrame, update_stats,
173  false, base::TimeDelta(), trace_id);
174  ticks_from_vm_buffer_.Enqueue(record);
175 }

196 bool ProfilerEventsProcessor::ProcessCodeEvent() {
197  CodeEventsContainer record;
198  if (events_buffer_.Dequeue(&record)) {
199  if (record.generic.type == CodeEventRecord::Type::kNativeContextMove) {
200  NativeContextMoveEventRecord& nc_record =
201  record.NativeContextMoveEventRecord_;
202  profiles_->UpdateNativeContextAddressForCurrentProfiles(
203  nc_record.from_address, nc_record.to_address);
204  } else {
205  code_observer_->CodeEventHandlerInternal(record);
206  }
207  last_processed_code_event_id_ = record.generic.order;
208  return true;
209  }
210  return false;
211 }

213 void ProfilerEventsProcessor::CodeEventHandler(
214  const CodeEventsContainer& evt_rec) {
215  switch (evt_rec.generic.type) {
216  case CodeEventRecord::Type::kCodeCreation:
217  case CodeEventRecord::Type::kCodeMove:
218  case CodeEventRecord::Type::kCodeDisableOpt:
219  case CodeEventRecord::Type::kCodeDelete:
220  case CodeEventRecord::Type::kNativeContextMove:
221  Enqueue(evt_rec);
222  break;
223  case CodeEventRecord::Type::kCodeDeopt: {
224  const CodeDeoptEventRecord* rec = &evt_rec.CodeDeoptEventRecord_;
225  Address pc = rec->pc;
226  int fp_to_sp_delta = rec->fp_to_sp_delta;
227  Enqueue(evt_rec);
228  AddDeoptStack(pc, fp_to_sp_delta);
229  break;
230  }
231  case CodeEventRecord::Type::kNoEvent:
232  case CodeEventRecord::Type::kReportBuiltin:
233  UNREACHABLE();
234  }
235 }

251 ProfilerEventsProcessor::SampleProcessingResult
252 SamplingEventsProcessor::ProcessOneSample() {
253  TickSampleEventRecord record1;
254  if (ticks_from_vm_buffer_.Peek(&record1) &&
255  (record1.order == last_processed_code_event_id_)) {
256  TickSampleEventRecord record;
257  ticks_from_vm_buffer_.Dequeue(&record);
258  SymbolizeAndAddToProfiles(&record);
259  return OneSampleProcessed;
260  }
261 
262  const TickSampleEventRecord* record = ticks_buffer_.Peek();
263  if (record == nullptr) {
264  if (ticks_from_vm_buffer_.IsEmpty()) return NoSamplesInQueue;
265  return FoundSampleForNextCodeEvent;
266  }
267  if (record->order != last_processed_code_event_id_) {
268  return FoundSampleForNextCodeEvent;
269  }
270  SymbolizeAndAddToProfiles(record);
271  ticks_buffer_.Remove();
272  return OneSampleProcessed;
273 }

275 void SamplingEventsProcessor::Run() {
276  // Set the current isolate such that trusted pointer tables etc are available.
277  SetCurrentIsolateScope isolate_scope(isolate_);
278  base::MutexGuard guard(&running_mutex_);
279  while (running_.load(std::memory_order_relaxed)) {
280  base::TimeTicks nextSampleTime = base::TimeTicks::Now() + period_;
281  base::TimeTicks now;
282  SampleProcessingResult result;
283  // Keep processing existing events until we need to do next sample
284  // or the ticks buffer is empty.
285  do {
286  result = ProcessOneSample();
287  if (result == FoundSampleForNextCodeEvent) {
288  // All ticks of the current last_processed_code_event_id_ are
289  // processed, proceed to the next code event.
290  ProcessCodeEvent();
291  }
292  now = base::TimeTicks::Now();
293  } while (result != NoSamplesInQueue && now < nextSampleTime);
294 
295  if (nextSampleTime > now) {
296 #if V8_OS_WIN
297  if (use_precise_sampling_ &&
298  nextSampleTime - now < base::TimeDelta::FromMilliseconds(100)) {
299  if (precise_sleep_timer_.IsInitialized()) {
300  precise_sleep_timer_.Sleep(nextSampleTime - now);
301  } else {
302  // Do not use Sleep on Windows as it is very imprecise, with up to
303  // 16ms jitter, which is unacceptable for short profile intervals.
304  while (base::TimeTicks::Now() < nextSampleTime) {
305  }
306  }
307  } else // NOLINT
308 #else
309  USE(use_precise_sampling_);
310 #endif // V8_OS_WIN
311  {
312  // Allow another thread to interrupt the delay between samples in the
313  // event of profiler shutdown.
314  while (now < nextSampleTime &&
315  running_cond_.WaitFor(&running_mutex_, nextSampleTime - now)) {
316  // If true was returned, we got interrupted before the timeout
317  // elapsed. If this was not due to a change in running state, a
318  // spurious wakeup occurred (thus we should continue to wait).
319  if (!running_.load(std::memory_order_relaxed)) {
320  break;
321  }
322  now = base::TimeTicks::Now();
323  }
324  }
325  }
326 
327  // Schedule next sample.
328  sampler_->DoSample();
329  }
330 
331  // Process remaining tick events.
332  do {
333  SampleProcessingResult result;
334  do {
335  result = ProcessOneSample();
336  } while (result == OneSampleProcessed);
337  } while (ProcessCodeEvent());
338 }

421 void ProfilerCodeObserver::LogBuiltins() {
422  Builtins* builtins = isolate_->builtins();
423  DCHECK(builtins->is_initialized());
424  for (Builtin builtin = Builtins::kFirst; builtin <= Builtins::kLast;
425  ++builtin) {
426 #if V8_ENABLE_WEBASSEMBLY
427  // We add the embedded data entry below.
428  if (builtin == Builtin::kWasmToJsWrapperCSA) continue;
429 #endif
430  CodeEventsContainer evt_rec(CodeEventRecord::Type::kReportBuiltin);
431  ReportBuiltinEventRecord* rec = &evt_rec.ReportBuiltinEventRecord_;
432  Tagged<Code> code = builtins->code(builtin);
433  rec->instruction_start = code->instruction_start();
434  rec->instruction_size = code->instruction_size();
435  rec->builtin = builtin;
436  CodeEventHandlerInternal(evt_rec);
437  }
438 
439 #if V8_ENABLE_WEBASSEMBLY
440  // We can call the WasmToJS wrapper from the embedded blob
441  CodeEventsContainer evt_rec(CodeEventRecord::Type::kReportBuiltin);
442  ReportBuiltinEventRecord* rec = &evt_rec.ReportBuiltinEventRecord_;
443  rec->instruction_start =
444  Builtins::EmbeddedEntryOf(Builtin::kWasmToJsWrapperCSA);
445  rec->instruction_size =
446  EmbeddedData::FromBlob().InstructionSizeOf(Builtin::kWasmToJsWrapperCSA);
447  rec->builtin = Builtin::kWasmToJsWrapperCSA;
448  CodeEventHandlerInternal(evt_rec);
449 #endif
450 }

486  void RemoveProfiler(Isolate* isolate, CpuProfiler* profiler) {
487  base::MutexGuard lock(&mutex_);
488  auto range = profilers_.equal_range(isolate);
489  for (auto it = range.first; it != range.second; ++it) {
490  if (it->second != profiler) continue;
491  profilers_.erase(it);
492  return;
493  }

498  const std::optional<uint64_t> trace_id) {
499  base::MutexGuard lock(&mutex_);
500  auto range = profilers_.equal_range(isolate);
501  for (auto it = range.first; it != range.second; ++it) {
502  it->second->CollectSample(trace_id);
503  }
504  }

506  size_t GetAllProfilersMemorySize(Isolate* isolate) {
507  base::MutexGuard lock(&mutex_);
508  size_t estimated_memory = 0;
509  auto range = profilers_.equal_range(isolate);
510  for (auto it = range.first; it != range.second; ++it) {
511  estimated_memory += it->second->GetEstimatedMemoryUsage();
512  }
513  return estimated_memory;
514  }

516  private:
