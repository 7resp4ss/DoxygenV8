268 // The modes possibly affected by apply must be in kApplyMask.
269 void WritableRelocInfo::apply(intptr_t delta) {
270  if (IsCodeTarget(rmode_) || IsNearBuiltinEntry(rmode_) ||
271  IsWasmStubCall(rmode_)) {
272  jit_allocation_.WriteUnalignedValue(
273  pc_, ReadUnalignedValue<int32_t>(pc_) - static_cast<int32_t>(delta));
274  } else if (IsInternalReference(rmode_)) {
275  // Absolute code pointer inside code object moves with the code object.
276  jit_allocation_.WriteUnalignedValue(
277  pc_, ReadUnalignedValue<Address>(pc_) + delta);
278  }
279 }

373 void WritableRelocInfo::set_target_object(Tagged<HeapObject> target,
374  ICacheFlushMode icache_flush_mode) {
375  DCHECK(IsCodeTarget(rmode_) || IsEmbeddedObjectMode(rmode_));
376  if (IsCompressedEmbeddedObject(rmode_)) {
377  DCHECK(COMPRESS_POINTERS_BOOL);
378  // We must not compress pointers to objects outside of the main pointer
379  // compression cage as we wouldn't be able to decompress them with the
380  // correct cage base.
381  DCHECK_IMPLIES(V8_ENABLE_SANDBOX_BOOL, !HeapLayout::InTrustedSpace(target));
382  DCHECK_IMPLIES(V8_EXTERNAL_CODE_SPACE_BOOL,
383  !HeapLayout::InCodeSpace(target));
384  Tagged_t tagged = V8HeapCompressionScheme::CompressObject(target.ptr());
385  jit_allocation_.WriteUnalignedValue(pc_, tagged);
386  } else {
387  DCHECK(IsFullEmbeddedObject(rmode_));
388  jit_allocation_.WriteUnalignedValue(pc_, target.ptr());
389  }
390  if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
391  FlushInstructionCache(pc_, sizeof(Address));
392  }
393 }
