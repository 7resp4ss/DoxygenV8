51 // Find the frame pointer of the interpreter frame on the stack.
52 Address FindInterpreterEntryFramePointer(Isolate* isolate) {
53  StackFrameIterator it(isolate, isolate->thread_local_top());
54  // On top: C entry stub.
55  DCHECK_EQ(StackFrame::EXIT, it.frame()->type());
56  it.Advance();
57  // Next: the wasm interpreter entry.
58  DCHECK_EQ(StackFrame::WASM_INTERPRETER_ENTRY, it.frame()->type());
59  return it.frame()->fp();
60 }

64 RUNTIME_FUNCTION(Runtime_WasmRunInterpreter) {
65  DCHECK_EQ(3, args.length());
66  HandleScope scope(isolate);
67  DirectHandle<WasmInstanceObject> instance = args.at<WasmInstanceObject>(0);
68  DirectHandle<WasmTrustedInstanceData> trusted_data(
69  instance->trusted_data(isolate), isolate);
70  int32_t func_index = NumberToInt32(args[1]);
71  DirectHandle<Object> arg_buffer_obj = args.at(2);
72 
73  // The arg buffer is the raw pointer to the caller's stack. It looks like a
74  // Smi (lowest bit not set, as checked by IsSmi), but is no valid Smi. We just
75  // cast it back to the raw pointer.
76  CHECK(!IsHeapObject(*arg_buffer_obj));
77  CHECK(IsSmi(*arg_buffer_obj));
78  Address arg_buffer = (*arg_buffer_obj).ptr();
79 
80  // Reserve buffers for argument and return values.
81  DCHECK_GT(trusted_data->module()->functions.size(), func_index);
82  const wasm::FunctionSig* sig =
83  trusted_data->module()->functions[func_index].sig;
84  DCHECK_GE(kMaxInt, sig->parameter_count());
85  int num_params = static_cast<int>(sig->parameter_count());
86  std::vector<wasm::WasmValue> wasm_args(num_params);
87  DCHECK_GE(kMaxInt, sig->return_count());
88  int num_returns = static_cast<int>(sig->return_count());
89  std::vector<wasm::WasmValue> wasm_rets(num_returns);
90 
91  // Set the current isolate's context.
92  isolate->set_context(trusted_data->native_context());
93 
94  // Make sure the WasmInterpreterObject and InterpreterHandle for this instance
95  // exist.
96  DirectHandle<Tuple2> interpreter_object =
97  WasmTrustedInstanceData::GetOrCreateInterpreterObject(instance);
98  wasm::InterpreterHandle* interpreter_handle =
99  wasm::GetOrCreateInterpreterHandle(isolate, interpreter_object);
100 
101  if (wasm::WasmBytecode::ContainsSimd(sig)) {
102  wasm::ClearThreadInWasmScope clear_wasm_flag(isolate);
103 
104  interpreter_handle->SetTrapFunctionIndex(func_index);
105  isolate->Throw(*isolate->factory()->NewTypeError(
106  MessageTemplate::kWasmTrapJSTypeError));
107  return ReadOnlyRoots(isolate).exception();
108  }
109 
110  Address frame_pointer = FindInterpreterEntryFramePointer(isolate);
111 
112  // If there are Ref arguments or return values, we store their pointers into
113  // an array of bytes so we need to disable GC until they are unpacked by the
114  // callee.
115  {
116  DisallowHeapAllocation no_gc;
117 
118  // Copy the arguments for the {arg_buffer} into a vector of {WasmValue}.
119  // This also boxes reference types into handles, which needs to happen
120  // before any methods that could trigger a GC are being called.
121  Address arg_buf_ptr = arg_buffer;
122  for (int i = 0; i < num_params; ++i) {
123 #define CASE_ARG_TYPE(type, ctype) \
124  case wasm::type: \
125  DCHECK_EQ(wasm::ValueTypes::ElementSizeInBytes(sig->GetParam(i)), \
126  sizeof(ctype)); \
127  wasm_args[i] = \
128  wasm::WasmValue(base::ReadUnalignedValue<ctype>(arg_buf_ptr)); \
129  arg_buf_ptr += sizeof(ctype); \
130  break;
131 
132  wasm::ValueType value_type = sig->GetParam(i);
133  wasm::ValueKind kind = value_type.kind();
134  switch (kind) {
135  CASE_ARG_TYPE(kWasmI32.kind(), uint32_t)
136  CASE_ARG_TYPE(kWasmI64.kind(), uint64_t)
137  CASE_ARG_TYPE(kWasmF32.kind(), float)
138  CASE_ARG_TYPE(kWasmF64.kind(), double)
139 #undef CASE_ARG_TYPE
140  case wasm::kRef:
141  case wasm::kRefNull: {
142  DCHECK_EQ(wasm::ValueTypes::ElementSizeInBytes(sig->GetParam(i)),
143  kSystemPointerSize);
144  // MarkCompactCollector::RootMarkingVisitor requires ref slots to be
145  // 64-bit aligned.
146  arg_buf_ptr += (arg_buf_ptr & 0x04);
147 
148  DirectHandle<Object> ref(
149  base::ReadUnalignedValue<Tagged<Object>>(arg_buf_ptr), isolate);
150 
151  const wasm::WasmInterpreterRuntime* wasm_runtime =
152  interpreter_handle->interpreter()->GetWasmRuntime();
153  ref = wasm_runtime->JSToWasmObject(ref, value_type);
154  if (isolate->has_exception()) {
155  interpreter_handle->SetTrapFunctionIndex(func_index);
156  return ReadOnlyRoots(isolate).exception();
157  }
158 
159  if ((value_type != wasm::kWasmExternRef &&
160  value_type != wasm::kWasmNullExternRef) &&
161  IsNull(*ref, isolate)) {
162  ref = isolate->factory()->wasm_null();
163  }
164 
165  wasm_args[i] = wasm::WasmValue(ref, wasm::kWasmAnyRef);
166  arg_buf_ptr += kSystemPointerSize;
167  break;
168  }
169  case wasm::kWasmS128.kind():
170  default:
171  UNREACHABLE();
172  }
173  }
174 
175  // Run the function in the interpreter. Note that neither the
176  // {WasmInterpreterObject} nor the {InterpreterHandle} have to exist,
177  // because interpretation might have been triggered by another Isolate
178  // sharing the same WasmEngine.
179  bool success = WasmInterpreterObject::RunInterpreter(
180  isolate, frame_pointer, instance, func_index, wasm_args, wasm_rets);
181 
182  // Early return on failure.
183  if (!success) {
184  DCHECK(isolate->has_exception());
185  return ReadOnlyRoots(isolate).exception();
186  }
187 
188  // Copy return values from the vector of {WasmValue} into {arg_buffer}. This
189  // also un-boxes reference types from handles into raw pointers.
190  arg_buf_ptr = arg_buffer;
191 
192  for (int i = 0; i < num_returns; ++i) {
193 #define CASE_RET_TYPE(type, ctype) \
194  case wasm::type: \
195  DCHECK_EQ(wasm::ValueTypes::ElementSizeInBytes(sig->GetReturn(i)), \
196  sizeof(ctype)); \
197  base::WriteUnalignedValue<ctype>(arg_buf_ptr, wasm_rets[i].to<ctype>()); \
198  arg_buf_ptr += sizeof(ctype); \
199  break;
200 
201  switch (sig->GetReturn(i).kind()) {
202  CASE_RET_TYPE(kWasmI32.kind(), uint32_t)
203  CASE_RET_TYPE(kWasmI64.kind(), uint64_t)
204  CASE_RET_TYPE(kWasmF32.kind(), float)
205  CASE_RET_TYPE(kWasmF64.kind(), double)
206 #undef CASE_RET_TYPE
207  case wasm::kRef:
208  case wasm::kRefNull: {
209  DCHECK_EQ(wasm::ValueTypes::ElementSizeInBytes(sig->GetReturn(i)),
210  kSystemPointerSize);
211  DirectHandle<Object> ref = wasm_rets[i].to_ref();
212  // Note: WasmToJSObject(ref) already called in ContinueExecution or
213  // CallExternalJSFunction.
214 
215  // Make sure ref slots are 64-bit aligned.
216  arg_buf_ptr += (arg_buf_ptr & 0x04);
217  base::WriteUnalignedValue<Tagged<Object>>(arg_buf_ptr, *ref);
218  arg_buf_ptr += kSystemPointerSize;
219  break;
220  }
221  case wasm::kWasmS128.kind():
222  default:
223  UNREACHABLE();
224  }
225  }
226 
227  return ReadOnlyRoots(isolate).undefined_value();
228  }
229 }

242 V8_EXPORT_PRIVATE InterpreterHandle* GetOrCreateInterpreterHandle(
243  Isolate* isolate, DirectHandle<Tuple2> interpreter_object) {
244  DirectHandle<Object> handle(
245  WasmInterpreterObject::get_interpreter_handle(*interpreter_object),
246  isolate);
247  if (IsUndefined(*handle, isolate)) {
248  // Use the maximum stack size to estimate the maximum size of the
249  // interpreter. The interpreter keeps its own stack internally, and the size
250  // of the stack should dominate the overall size of the interpreter. We
251  // multiply by '2' to account for the growing strategy for the backing store
252  // of the stack.
253  size_t interpreter_size = v8_flags.stack_size * KB * 2;
254  handle = Managed<InterpreterHandle>::From(
255  isolate, interpreter_size,
256  std::make_shared<InterpreterHandle>(isolate, interpreter_object));
257  WasmInterpreterObject::set_interpreter_handle(*interpreter_object, *handle);
258  }
259 
260  return Cast<Managed<InterpreterHandle>>(handle)->raw();
261 }

263 // A helper for an entry in an indirect function table (IFT).
264 // The underlying storage in the instance is used by generated code to
265 // call functions indirectly at runtime.
266 // Each entry has the following fields:
267 // - implicit_arg = A WasmTrustedInstanceData or a WasmImportData.
268 // - sig_id = signature id of function.
269 // - target = entrypoint to Wasm code or import wrapper code.
270 // - function_index = function index, if a Wasm function, or
271 // WasmDispatchTable::kInvalidFunctionIndex otherwise.
272 class IndirectFunctionTableEntry {
273  public:
274  inline IndirectFunctionTableEntry(DirectHandle<WasmInstanceObject>,
275  int table_index, int entry_index);
276 
277  inline Tagged<Object> implicit_arg() const {
278  return table_->implicit_arg(index_);
279  }
280  inline wasm::CanonicalTypeIndex sig_id() const { return table_->sig(index_); }
281  inline WasmCodePointer target() const { return table_->target(index_); }
282  inline uint32_t function_index() const {
283  return table_->function_index(index_);
284  }
285 
286  private:
287  DirectHandle<WasmDispatchTable> const table_;
288  int const index_;
289 };

574 // Unpack the values encoded in the given exception. The exception values are
575 // pushed onto the operand stack.
576 void WasmInterpreterRuntime::UnpackException(
577  uint32_t* sp, const WasmTag& tag, DirectHandle<Object> exception_object,
578  uint32_t first_param_slot_index, uint32_t first_param_ref_stack_index) {
579  DirectHandle<FixedArray> encoded_values =
580  Cast<FixedArray>(WasmExceptionPackage::GetExceptionValues(
581  isolate_, Cast<WasmExceptionPackage>(exception_object)));
582  // Decode the exception values from the given exception package and push
583  // them onto the operand stack. This encoding has to be in sync with other
584  // backends so that exceptions can be passed between them.
585  const WasmTagSig* sig = tag.sig;
586  uint32_t encoded_index = 0;
587  uint32_t* p = sp + first_param_slot_index;
588  for (size_t i = 0; i < sig->parameter_count(); ++i) {
589 #ifdef V8_ENABLE_DRUMBRAKE_TRACING
590  if (v8_flags.trace_drumbrake_execution) {
591  TracePush(sig->GetParam(i).kind(), static_cast<uint32_t>(p - sp));
592  }
593 #endif // V8_ENABLE_DRUMBRAKE_TRACING
594 
595  WasmValue value;
596  switch (sig->GetParam(i).kind()) {
597  case kI32: {
598  uint32_t u32 = 0;
599  DecodeI32ExceptionValue(encoded_values, &encoded_index, &u32);
600  base::WriteUnalignedValue<uint32_t>(reinterpret_cast<Address>(p), u32);
601  p += sizeof(uint32_t) / kSlotSize;
602  break;
603  }
604  case kF32: {
605  uint32_t f32_bits = 0;
606  DecodeI32ExceptionValue(encoded_values, &encoded_index, &f32_bits);
607  float f32 = Float32::FromBits(f32_bits).get_scalar();
608  base::WriteUnalignedValue<float>(reinterpret_cast<Address>(p), f32);
609  p += sizeof(float) / kSlotSize;
610  break;
611  }
612  case kI64: {
613  uint64_t u64 = 0;
614  DecodeI64ExceptionValue(encoded_values, &encoded_index, &u64);
615  base::WriteUnalignedValue<uint64_t>(reinterpret_cast<Address>(p), u64);
616  p += sizeof(uint64_t) / kSlotSize;
617  break;
618  }
619  case kF64: {
620  uint64_t f64_bits = 0;
621  DecodeI64ExceptionValue(encoded_values, &encoded_index, &f64_bits);
622  float f64 = Float64::FromBits(f64_bits).get_scalar();
623  base::WriteUnalignedValue<double>(reinterpret_cast<Address>(p), f64);
624  p += sizeof(double) / kSlotSize;
625  break;
626  }
627  case kS128: {
628  int32x4 s128 = {0, 0, 0, 0};
629  uint32_t* vals = reinterpret_cast<uint32_t*>(s128.val);
630  DecodeI32ExceptionValue(encoded_values, &encoded_index, &vals[0]);
631  DecodeI32ExceptionValue(encoded_values, &encoded_index, &vals[1]);
632  DecodeI32ExceptionValue(encoded_values, &encoded_index, &vals[2]);
633  DecodeI32ExceptionValue(encoded_values, &encoded_index, &vals[3]);
634  base::WriteUnalignedValue<Simd128>(reinterpret_cast<Address>(p),
635  Simd128(s128));
636  p += sizeof(Simd128) / kSlotSize;
637  break;
638  }
639  case kRef:
640  case kRefNull: {
641  DirectHandle<Object> ref(encoded_values->get(encoded_index++),
642  isolate_);
643  if (sig->GetParam(i).value_type_code() == wasm::kFuncRefCode &&
644  i::IsNull(*ref, isolate_)) {
645  ref = isolate_->factory()->wasm_null();
646  }
647  StoreWasmRef(first_param_ref_stack_index++, ref);
648  base::WriteUnalignedValue<WasmRef>(reinterpret_cast<Address>(p), ref);
649  p += sizeof(WasmRef) / kSlotSize;
650  break;
651  }
652  default:
653  UNREACHABLE();
654  }
655  }
656  DCHECK_EQ(WasmExceptionPackage::GetEncodedSize(&tag), encoded_index);
657 }

712 // Handle a thrown exception. Returns whether the exception was handled inside
713 // of wasm. Unwinds the interpreted stack accordingly.
714 WasmInterpreterThread::ExceptionHandlingResult
715 WasmInterpreterRuntime::HandleException(uint32_t* sp,
716  const uint8_t*& current_code) {
717  DCHECK_IMPLIES(current_code, current_frame_.current_function_);
718  DCHECK_IMPLIES(!current_code, !current_frame_.current_function_);
719  DCHECK(isolate_->has_exception());
720 
721  bool catchable = current_frame_.current_function_ &&
722  isolate_->is_catchable_by_wasm(isolate_->exception());
723  if (catchable) {
724  HandleScope scope(isolate_);
725  DirectHandle<Object> exception(isolate_->exception(), isolate_);
726  Tagged<WasmTrustedInstanceData> trusted_data =
727  *wasm_trusted_instance_data();
728 
729  // We might need to allocate a new FixedArray<Object> to store the caught
730  // exception.
731  DCHECK(AllowHeapAllocation::IsAllowed());
732 
733  size_t current_code_offset =
734  current_code - current_frame_.current_function_->GetCode();
735  const WasmEHData::TryBlock* try_block =
736  current_frame_.current_function_->GetTryBlock(current_code_offset);
737  while (try_block) {
738  for (const auto& catch_handler : try_block->catch_handlers) {
739  if (catch_handler.tag_index < 0) {
740  // Catch all.
741  current_code = current_frame_.current_function_->GetCode() +
742  catch_handler.code_offset;
743  current_frame_.SetCaughtException(
744  isolate_, catch_handler.catch_block_index, exception);
745  isolate_->clear_exception();
746  return WasmInterpreterThread::HANDLED;
747  } else if (IsWasmExceptionPackage(*exception, isolate_)) {
748  // The exception was thrown by Wasm code and it's wrapped in a
749  // WasmExceptionPackage.
750  DirectHandle<Object> caught_tag =
751  WasmExceptionPackage::GetExceptionTag(
752  isolate_, Cast<WasmExceptionPackage>(exception));
753  DirectHandle<Object> expected_tag(
754  trusted_data->tags_table()->get(catch_handler.tag_index),
755  isolate_);
756  DCHECK(IsWasmExceptionTag(*expected_tag));
757  // Determines whether the given exception has a tag matching the
758  // expected tag for the given index within the exception table of the
759  // current instance.
760  if (expected_tag.is_identical_to(caught_tag)) {
761  current_code = current_frame_.current_function_->GetCode() +
762  catch_handler.code_offset;
763  DCHECK_LT(catch_handler.tag_index, module_->tags.size());
764  const WasmTag& tag = module_->tags[catch_handler.tag_index];
765  auto exception_payload_slot_offsets =
766  current_frame_.current_function_
767  ->GetExceptionPayloadStartSlotOffsets(
768  catch_handler.catch_block_index);
769  UnpackException(
770  sp, tag, exception,
771  exception_payload_slot_offsets.first_param_slot_offset,
772  exception_payload_slot_offsets.first_param_ref_stack_index);
773  current_frame_.SetCaughtException(
774  isolate_, catch_handler.catch_block_index, exception);
775  isolate_->clear_exception();
776  return WasmInterpreterThread::HANDLED;
777  }
778  } else {
779  // Check for the special case where the tag is WebAssembly.JSTag and
780  // the exception is not a WebAssembly.Exception. In this case the
781  // exception is caught and pushed on the operand stack.
782  // Only perform this check if the tag signature is the same as
783  // the JSTag signature, i.e. a single externref, otherwise we know
784  // statically that it cannot be the JSTag.
785  DCHECK_LT(catch_handler.tag_index, module_->tags.size());
786  const WasmTagSig* sig = module_->tags[catch_handler.tag_index].sig;
787  if (sig->return_count() != 0 || sig->parameter_count() != 1 ||
788  (sig->GetParam(0).kind() != kRefNull &&
789  sig->GetParam(0).kind() != kRef)) {
790  continue;
791  }
792 
793  DirectHandle<JSObject> js_tag_object(
794  isolate_->native_context()->wasm_js_tag(), isolate_);
795  DirectHandle<WasmTagObject> wasm_tag_object(
796  Cast<WasmTagObject>(*js_tag_object), isolate_);
797  DirectHandle<Object> caught_tag(wasm_tag_object->tag(), isolate_);
798  DirectHandle<Object> expected_tag(
799  trusted_data->tags_table()->get(catch_handler.tag_index),
800  isolate_);
801  if (!expected_tag.is_identical_to(caught_tag)) {
802  continue;
803  }
804 
805  current_code = current_frame_.current_function_->GetCode() +
806  catch_handler.code_offset;
807  // Push exception on the operand stack.
808  auto exception_payload_slot_offsets =
809  current_frame_.current_function_
810  ->GetExceptionPayloadStartSlotOffsets(
811  catch_handler.catch_block_index);
812  StoreWasmRef(
813  exception_payload_slot_offsets.first_param_ref_stack_index,
814  exception);
815  base::WriteUnalignedValue<WasmRef>(
816  reinterpret_cast<Address>(
817  sp + exception_payload_slot_offsets.first_param_slot_offset),
818  exception);
819 
820  current_frame_.SetCaughtException(
821  isolate_, catch_handler.catch_block_index, exception);
822  isolate_->clear_exception();
823  return WasmInterpreterThread::HANDLED;
824  }
825  }
826  try_block =
827  current_frame_.current_function_->GetParentTryBlock(try_block);
828  }
829  }
830 
831  DCHECK_NOT_NULL(current_thread_);
832  current_thread_->Unwinding();
833  return WasmInterpreterThread::UNWOUND;
834 }

894 void WasmInterpreterRuntime::BeginExecution(
895  WasmInterpreterThread* thread, uint32_t func_index, Address frame_pointer,
896  uint8_t* interpreter_fp, uint32_t ref_stack_offset,
897  const std::vector<WasmValue>* argument_values) {
898  current_thread_ = thread;
899  start_function_index_ = func_index;
900 
901  thread->StartActivation(this, frame_pointer, interpreter_fp, current_frame_);
902 
903  current_frame_.current_function_ = nullptr;
904  current_frame_.previous_frame_ = nullptr;
905  current_frame_.current_bytecode_ = nullptr;
906  current_frame_.current_sp_ = interpreter_fp;
907  current_frame_.ref_array_current_sp_ = ref_stack_offset;
908  current_frame_.thread_ = thread;
909 #ifdef V8_ENABLE_DRUMBRAKE_TRACING
910  current_frame_.current_stack_start_args_ = thread->CurrentStackFrameStart();
911 #endif // V8_ENABLE_DRUMBRAKE_TRACING
912 
913  const FunctionSig* sig = module_->functions[func_index].sig;
914  size_t args_count = 0;
915  uint32_t rets_slots_size = 0;
916  uint32_t ref_rets_count = 0;
917  uint32_t ref_args_count = 0;
918  WasmBytecode* target_function = GetFunctionBytecode(func_index);
919  if (target_function) {
920  args_count = target_function->args_count();
921  rets_slots_size = target_function->rets_slots_size();
922  ref_rets_count = target_function->ref_rets_count();
923  ref_args_count = target_function->ref_args_count();
924  } else {
925  // We begin execution by calling an imported function.
926  args_count = sig->parameter_count();
927  rets_slots_size = WasmBytecode::RetsSizeInSlots(sig);
928  ref_rets_count = WasmBytecode::RefRetsCount(sig);
929  ref_args_count = WasmBytecode::RefArgsCount(sig);
930  }
931 
932  // Here GC is disabled, we cannot "resize" the reference_stack_ FixedArray
933  // before having created Handles for the Ref arguments passed in
934  // argument_values.
935  HandleScope handle_scope(isolate_); // Avoid leaking handles.
936 
937  DirectHandleVector<Object> ref_args(isolate_);
938  if (ref_args_count > 0) {
939  ref_args.reserve(ref_args_count);
940  }
941 
942  uint8_t* p = interpreter_fp + rets_slots_size * kSlotSize;
943 
944  // Check stack overflow.
945  const uint8_t* stack_limit = thread->StackLimitAddress();
946  if (V8_UNLIKELY(p + (ref_rets_count + ref_args_count) * sizeof(WasmRef) >=
947  stack_limit)) {
948  size_t additional_required_size =
949  p + (ref_rets_count + ref_args_count) * sizeof(WasmRef) - stack_limit;
950  if (!thread->ExpandStack(additional_required_size)) {
951  // TODO(paolosev@microsoft.com) - Calculate initial function offset.
952  ClearThreadInWasmScope clear_wasm_flag(isolate_);
953  SealHandleScope shs(isolate_);
954  isolate_->StackOverflow();
955  const pc_t trap_pc = 0;
956  SetTrap(TrapReason::kTrapUnreachable, trap_pc);
957  thread->FinishActivation();
958  return;
959  }
960  }
961 
962  if (argument_values) {
963  // We are being called from JS, arguments are passed in the
964  // {argument_values} vector.
965  for (size_t i = 0; i < argument_values->size(); i++) {
966  const WasmValue& value = (*argument_values)[i];
967  switch (value.type().kind()) {
968  case kI32:
969  base::WriteUnalignedValue<int32_t>(reinterpret_cast<Address>(p),
970  value.to<int32_t>());
971  p += sizeof(int32_t);
972  break;
973  case kI64:
974  base::WriteUnalignedValue<int64_t>(reinterpret_cast<Address>(p),
975  value.to<int64_t>());
976  p += sizeof(int64_t);
977  break;
978  case kF32:
979  base::WriteUnalignedValue<float>(reinterpret_cast<Address>(p),
980  value.to<float>());
981  p += sizeof(float);
982  break;
983  case kF64:
984  base::WriteUnalignedValue<double>(reinterpret_cast<Address>(p),
985  value.to<double>());
986  p += sizeof(double);
987  break;
988  case kRef:
989  case kRefNull:
990  ref_args.push_back(value.to_ref());
991  base::WriteUnalignedValue<uint64_t>(reinterpret_cast<Address>(p),
992  kSlotsZapValue);
993  p += sizeof(WasmRef);
994  break;
995  case kS128:
996  default:
997  UNREACHABLE();
998  }
999  }
1000  } else {
1001  // We are being called from Wasm, arguments are already in the stack.
1002  for (size_t i = 0; i < args_count; i++) {
1003  switch (sig->GetParam(i).kind()) {
1004  case kI32:
1005  p += sizeof(int32_t);
1006  break;
1007  case kI64:
1008  p += sizeof(int64_t);
1009  break;
1010  case kF32:
1011  p += sizeof(float);
1012  break;
1013  case kF64:
1014  p += sizeof(double);
1015  break;
1016  case kS128:
1017  p += sizeof(Simd128);
1018  break;
1019  case kRef:
1020  case kRefNull: {
1021  DirectHandle<Object> ref =
1022  base::ReadUnalignedValue<DirectHandle<Object>>(
1023  reinterpret_cast<Address>(p));
1024  ref_args.push_back(ref);
1025  p += sizeof(WasmRef);
1026  break;
1027  }
1028  default:
1029  UNREACHABLE();
1030  }
1031  }
1032  }
1033 
1034  {
1035  // Once we have read ref argument passed on the stack and we have stored
1036  // them into the ref_args vector of Handles, we can re-enable the GC.
1037  AllowHeapAllocation allow_gc;
1038 
1039  if (ref_rets_count + ref_args_count > 0) {
1040  // Reserve space for reference args and return values in the
1041  // reference_stack_.
1042  current_thread_->EnsureRefStackSpace(
1043  current_frame_.ref_array_current_sp_ + ref_rets_count +
1044  ref_args_count);
1045 
1046  uint32_t ref_stack_arg_index = ref_rets_count;
1047  for (uint32_t ref_arg_index = 0; ref_arg_index < ref_args_count;
1048  ref_arg_index++) {
1049  StoreWasmRef(ref_stack_arg_index++, ref_args[ref_arg_index]);
1050  }
1051  }
1052  }
1053 }

1055 void WasmInterpreterRuntime::ContinueExecution(WasmInterpreterThread* thread,
1056  bool called_from_js) {
1057  DCHECK_NE(start_function_index_, UINT_MAX);
1058 
1059  uint32_t start_function_index = start_function_index_;
1060  FrameState current_frame = current_frame_;
1061 
1062  const uint8_t* code = nullptr;
1063  const FunctionSig* sig = nullptr;
1064  const CanonicalSig* canonicalized_sig = nullptr;
1065  uint32_t return_count = 0;
1066  WasmBytecode* target_function = GetFunctionBytecode(start_function_index_);
1067  if (target_function) {
1068  sig = target_function->GetFunctionSignature();
1069  canonicalized_sig = target_function->GetCanonicalFunctionSignature();
1070  return_count = target_function->return_count();
1071  ExecuteFunction(code, start_function_index_, target_function->args_count(),
1072  0, 0, 0);
1073  } else {
1074  sig = module_->functions[start_function_index_].sig;
1075  CanonicalTypeIndex canonical_sig_index = module_->canonical_sig_id(
1076  module_->functions[start_function_index_].sig_index);
1077  canonicalized_sig =
1078  GetTypeCanonicalizer()->LookupFunctionSignature(canonical_sig_index);
1079  return_count = static_cast<uint32_t>(sig->return_count());
1080  ExecuteImportedFunction(code, start_function_index_,
1081  static_cast<uint32_t>(sig->parameter_count()), 0, 0,
1082  0);
1083  }
1084 
1085  // If there are Ref types in the set of result types defined in the function
1086  // signature, they are located from the first ref_stack_ slot of the current
1087  // Activation.
1088  uint32_t ref_result_slot_index = 0;
1089 
1090  if (state() == WasmInterpreterThread::State::RUNNING) {
1091  if (return_count > 0) {
1092  uint32_t* dst = reinterpret_cast<uint32_t*>(current_frame_.current_sp_);
1093 
1094  if (called_from_js) {
1095  // We are returning the results to a JS caller, we need to store them
1096  // into the {function_result_} vector and they will be retrieved via
1097  // {GetReturnValue}.
1098  function_result_.resize(return_count);
1099  for (size_t index = 0; index < return_count; index++) {
1100  CanonicalValueType ret_value_type =
1101  canonicalized_sig->GetReturn(index);
1102  switch (ret_value_type.kind()) {
1103  case kI32:
1104  function_result_[index] =
1105  WasmValue(base::ReadUnalignedValue<int32_t>(
1106  reinterpret_cast<Address>(dst)));
1107  dst += sizeof(uint32_t) / kSlotSize;
1108  break;
1109  case kI64:
1110  function_result_[index] =
1111  WasmValue(base::ReadUnalignedValue<int64_t>(
1112  reinterpret_cast<Address>(dst)));
1113  dst += sizeof(uint64_t) / kSlotSize;
1114  break;
1115  case kF32:
1116  function_result_[index] =
1117  WasmValue(base::ReadUnalignedValue<float>(
1118  reinterpret_cast<Address>(dst)));
1119  dst += sizeof(float) / kSlotSize;
1120  break;
1121  case kF64:
1122  function_result_[index] =
1123  WasmValue(base::ReadUnalignedValue<double>(
1124  reinterpret_cast<Address>(dst)));
1125  dst += sizeof(double) / kSlotSize;
1126  break;
1127  case kRef:
1128  case kRefNull: {
1129  DirectHandle<Object> ref =
1130  ExtractWasmRef(ref_result_slot_index++);
1131  ref = WasmToJSObject(ref);
1132  function_result_[index] = WasmValue(ref, ret_value_type);
1133  dst += sizeof(WasmRef) / kSlotSize;
1134  break;
1135  }
1136  case kS128:
1137  default:
1138  UNREACHABLE();
1139  }
1140  }
1141  } else {
1142  // We are returning the results on the stack
1143  for (size_t index = 0; index < return_count; index++) {
1144  switch (sig->GetReturn(index).kind()) {
1145  case kI32:
1146  dst += sizeof(uint32_t) / kSlotSize;
1147  break;
1148  case kI64:
1149  dst += sizeof(uint64_t) / kSlotSize;
1150  break;
1151  case kF32:
1152  dst += sizeof(float) / kSlotSize;
1153  break;
1154  case kF64:
1155  dst += sizeof(double) / kSlotSize;
1156  break;
1157  case kS128:
1158  dst += sizeof(Simd128) / kSlotSize;
1159  break;
1160  case kRef:
1161  case kRefNull: {
1162  // Make sure the ref result is termporarily stored in a stack
1163  // slot, to be retrieved by the caller.
1164  DirectHandle<Object> ref =
1165  ExtractWasmRef(ref_result_slot_index++);
1166  base::WriteUnalignedValue<WasmRef>(reinterpret_cast<Address>(dst),
1167  ref);
1168  dst += sizeof(WasmRef) / kSlotSize;
1169  break;

1221 // A tail call should not add an additional stack frame to the interpreter
1222 // stack. This is implemented by unwinding the current stack frame just before
1223 // the tail call.
1224 void WasmInterpreterRuntime::UnwindCurrentStackFrame(
1225  uint32_t* sp, uint32_t slot_offset, uint32_t rets_size, uint32_t args_size,
1226  uint32_t rets_refs, uint32_t args_refs, uint32_t ref_stack_fp_offset) {
1227  // At the moment of the call the interpreter stack is as in the diagram below.
1228  // A new interpreter frame for the callee function has been initialized, with
1229  // `R` slots to contain the R return values, followed by {args_size} slots to
1230  // contain the callee arguments.
1231  //
1232  // In order to unwind an interpreter stack frame we just copy the content of
1233  // the slots that contain the callee arguments into the caller stack frame,
1234  // just after the slots of the return values. Note that the return call is
1235  // invalid if the number and types of the return values of the callee function
1236  // do not exactly match the number and types of the return values of the
1237  // caller function. Instead, the number of types of the caller and callee
1238  // functions arguments can differ.
1239  //
1240  // The other slots in the caller frame, for const values and locals, will be
1241  // initialized later in ExecuteFunction().
1242  //
1243  // +----------------------+
1244  // | argA-1 | ^ ^
1245  // | ... | | | ->-----+
1246  // | ... | | | |
1247  // | arg0 | callee v |
1248  // | retR-1 | frame |
1249  // | ... | | |
1250  // | ret0 | v | copy
1251  // +----------------------+ (slot_offset) |
1252  // | ... | ^ V
1253  // | <stack slots> | | |
1254  // | <locals slots> | | |
1255  // | <const slots> | | ^ |
1256  // | argN-1 | caller | <------+
1257  // | ... | frame |
1258  // | arg0 | | v
1259  // | retR-1 | |
1260  // | ... | |
1261  // | ret0 | v
1262  // +----------------------+ (0)
1263 
1264  uint8_t* next_sp = reinterpret_cast<uint8_t*>(sp);
1265  uint8_t* prev_sp = next_sp + slot_offset;
1266  // Here {args_size} is the number of arguments expected by the function we are
1267  // calling, which can be different from the number of args of the caller
1268  // function.
1269  ::memmove(next_sp + rets_size, prev_sp, args_size);
1270 
1271  // If some of the argument-slots contain Ref values, we need to move them
1272  // accordingly, in the thread {reference_stack_}.
1273  if (rets_refs) {
1274  current_thread_->ClearRefStackValues(current_frame_.ref_array_current_sp_,
1275  rets_refs);
1276  }
1277  // Here {args_refs} is the number of reference args expected by the function
1278  // we are calling, which can be different from the number of reference args of
1279  // the caller function.
1280  for (uint32_t i = 0; i < args_refs; i++) {
1281  StoreWasmRef(rets_refs + i, ExtractWasmRef(ref_stack_fp_offset + i));
1282  }
1283  if (ref_stack_fp_offset > rets_refs + args_refs) {
1284  current_thread_->ClearRefStackValues(
1285  current_frame_.ref_array_current_sp_ + rets_refs + args_refs,
1286  ref_stack_fp_offset - rets_refs - args_refs);
1287  }
1288 }

1743 bool WasmInterpreterRuntime::CheckIndirectCallSignature(
1744  uint32_t table_index, uint32_t entry_index, uint32_t sig_index) const {
1745  const WasmTable& table = module_->tables[table_index];
1746  bool needs_type_check =
1747  !EquivalentTypes(table.type.AsNonNull(),
1748  ValueType::Ref(ModuleTypeIndex({sig_index}), false,
1749  RefTypeKind::kFunction),
1750  module_, module_);
1751  bool needs_null_check = table.type.is_nullable();
1752 
1753  // Copied from Liftoff.
1754  // We do both the type check and the null check by checking the signature,
1755  // so this shares most code. For the null check we then only check if the
1756  // stored signature is != -1.
1757  if (needs_type_check || needs_null_check) {
1758  const IndirectCallTable& dispatch_table =
1759  indirect_call_tables_[table_index];
1760  const wasm::CanonicalTypeIndex real_sig_id =
1761  dispatch_table[entry_index].sig_index;
1762  wasm::CanonicalTypeIndex canonical_sig_id =
1763  module_->canonical_sig_id(ModuleTypeIndex({sig_index}));
1764  if (!needs_type_check) {
1765  // Only check for -1 (nulled table entry).
1766  if (!real_sig_id.valid()) return false;
1767  } else if (!module_->types[sig_index].is_final) {
1768  if (real_sig_id == canonical_sig_id) return true;
1769  if (needs_null_check && !real_sig_id.valid()) return false;
1770 
1771  Tagged<Map> rtt = Tagged<Map>::cast(isolate_->heap()
1772  ->wasm_canonical_rtts()
1773  ->get(real_sig_id.index)
1774  .GetHeapObjectAssumeWeak());
1775  DirectHandle<Map> formal_rtt = RttCanon(sig_index);
1776  return SubtypeCheck(rtt, *formal_rtt, sig_index);
1777  } else {
1778  if (real_sig_id != canonical_sig_id) return false;
1779  }
1780  }
1781 
1782  return true;
1783 }

1858

1958  } else {
1959  thread->Run();
1960  }
1961  }
1962  }

1966  // frame.

2855 inline WasmInterpreterThread::State InterpreterHandle::RunExecutionLoop(
2856  WasmInterpreterThread* thread, bool called_from_js) {
2857  // If there were Ref values passed as arguments they have already been read
2858  // in BeginExecution(), so we can re-enable GC.
2859  AllowHeapAllocation allow_gc;
2860 
2861  bool finished = false;
2862  WasmInterpreterThread::State state = thread->state();
2863  if (state != WasmInterpreterThread::State::RUNNING) {
2864  return state;
2865  }
2866 
2867  while (!finished) {
2868  state = ContinueExecution(thread, called_from_js);
2869  switch (state) {
2870  case WasmInterpreterThread::State::FINISHED:
2871  case WasmInterpreterThread::State::RUNNING:
2872  // Perfect, just break the switch and exit the loop.
2873  finished = true;
2874  break;
2875  case WasmInterpreterThread::State::TRAPPED: {
2876  if (!isolate_->has_exception()) {
2877  // An exception handler was found, keep running the loop.
2878  if (!trap_handler::IsThreadInWasm()) {
2879  trap_handler::SetThreadInWasm();
2880  }
2881  break;
2882  }
2883  thread->Stop();
2884  [[fallthrough]];
2885  }
2886  case WasmInterpreterThread::State::STOPPED:
2887  // An exception happened, and the current activation was unwound
2888  // without hitting a local exception handler. All that remains to be
2889  // done is finish the activation and let the exception propagate.
2890  DCHECK(isolate_->has_exception());
2891  return state; // Either STOPPED or TRAPPED.
2892  case WasmInterpreterThread::State::EH_UNWINDING: {
2893  thread->Stop();
2894  return WasmInterpreterThread::State::STOPPED;
2895  }
2896  }
2897  }
2898  return state;
2899 }
