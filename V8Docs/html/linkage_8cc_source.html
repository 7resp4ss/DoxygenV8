96 int CallDescriptor::GetStackParameterDelta(
97  CallDescriptor const* tail_caller) const {
98  // In the IsTailCallForTierUp case, the callee has
99  // identical linkage and runtime arguments to the caller, thus the stack
100  // parameter delta is 0. We don't explicitly pass the runtime arguments as
101  // inputs to the TailCall node, since they already exist on the stack.
102  if (IsTailCallForTierUp()) return 0;
103 
104  // Add padding if necessary before computing the stack parameter delta.
105  int callee_slots_above_sp = AddArgumentPaddingSlots(GetOffsetToReturns());
106  int tail_caller_slots_above_sp =
107  AddArgumentPaddingSlots(tail_caller->GetOffsetToReturns());
108  int stack_param_delta = callee_slots_above_sp - tail_caller_slots_above_sp;
109  DCHECK(!ShouldPadArguments(stack_param_delta));
110  return stack_param_delta;
111 }

113 int CallDescriptor::GetOffsetToFirstUnusedStackSlot() const {
114  int offset = kFirstCallerSlotOffset;
115  for (size_t i = 0; i < InputCount(); ++i) {
116  LinkageLocation operand = GetInputLocation(i);
117  if (!operand.IsRegister()) {
118  DCHECK(operand.IsCallerFrameSlot());
119  int slot_offset = -operand.GetLocation();
120  offset = std::max(offset, slot_offset + operand.GetSizeInPointers());
121  }
122  }
123  return offset;
124 }

126 int CallDescriptor::GetOffsetToReturns() const {
127  // Find the return slot with the least offset relative to the callee.
128  int offset = kNoCallerSlotOffset;
129  for (size_t i = 0; i < ReturnCount(); ++i) {
130  LinkageLocation operand = GetReturnLocation(i);
131  if (!operand.IsRegister()) {
132  DCHECK(operand.IsCallerFrameSlot());
133  int slot_offset = -operand.GetLocation();
134  offset = std::min(offset, slot_offset);
135  }
136  }
137  // If there was a return slot, return the offset minus 1 slot.
138  if (offset != kNoCallerSlotOffset) {
139  return offset - 1;
140  }
141 
142  // Otherwise, return the first slot after the parameters area, including
143  // optional padding slots.
144  int last_argument_slot = GetOffsetToFirstUnusedStackSlot() - 1;
145  offset = AddArgumentPaddingSlots(last_argument_slot);
146 
147  DCHECK_IMPLIES(offset == 0, ParameterSlotCount() == 0);
148  return offset;
149 }

151 uint32_t CallDescriptor::GetTaggedParameterSlots() const {
152  uint32_t count = 0;
153  uint32_t untagged_count = 0;
154  uint32_t first_offset = kMaxInt;
155  for (size_t i = 0; i < InputCount(); ++i) {
156  LinkageLocation operand = GetInputLocation(i);
157  if (!operand.IsRegister()) {
158  if (operand.GetType().IsTagged()) {
159  ++count;
160  // Caller frame slots have negative indices and start at -1. Flip it
161  // back to a positive offset (to be added to the frame's SP to find the
162  // slot).
163  int slot_offset = -operand.GetLocation() - 1;
164  DCHECK_GE(slot_offset, 0);
165  first_offset =
166  std::min(first_offset, static_cast<uint32_t>(slot_offset));
167  } else {
168  untagged_count += operand.GetSizeInPointers();
169  }
170  }
171  }
172  if (count == 0) {
173  // If we don't have any tagged parameter slots, still initialize the offset
174  // to point past the untagged parameter slots, so that
175  // offset + count == stack slot count.
176  first_offset = untagged_count;
177  }
178  DCHECK(first_offset != kMaxInt);
179  return (first_offset << 16) | (count & 0xFFFFu);
180 }

182 bool CallDescriptor::CanTailCall(const CallDescriptor* callee) const {
183  if (ReturnCount() != callee->ReturnCount()) return false;
184  const int stack_returns_delta =
185  GetOffsetToReturns() - callee->GetOffsetToReturns();
186  for (size_t i = 0; i < ReturnCount(); ++i) {
187  if (GetReturnLocation(i).IsCallerFrameSlot() &&
188  callee->GetReturnLocation(i).IsCallerFrameSlot()) {
189  if (GetReturnLocation(i).AsCallerFrameSlot() + stack_returns_delta !=
190  callee->GetReturnLocation(i).AsCallerFrameSlot()) {
191  return false;
192  }
193  } else if (!LinkageLocation::IsSameLocation(GetReturnLocation(i),
194  callee->GetReturnLocation(i))) {
195  return false;
196  }
197  }
198  return true;
199 }

238 EncodedCSignature CallDescriptor::ToEncodedCSignature() const {
239  int parameter_count = static_cast<int>(ParameterCount());
240  EncodedCSignature sig(parameter_count);
241  CHECK_LT(parameter_count, EncodedCSignature::kInvalidParamCount);
242 
243  for (int i = 0; i < parameter_count; ++i) {
244  if (IsFloatingPoint(GetParameterType(i).representation())) {
245  sig.SetFloat(i);
246  }
247  }
248  if (ReturnCount() > 0) {
249  DCHECK_EQ(1, ReturnCount());
250  if (IsFloatingPoint(GetReturnType(0).representation())) {

253  sig.SetReturnFloat64();
254  } else {
255  sig.SetReturnFloat32();
256  }
257  }
258  }
259  return sig;

276 CallDescriptor* ReplaceTypeInCallDescriptorWith(
277  Zone* zone, const CallDescriptor* call_descriptor, size_t num_replacements,
278  MachineType from, MachineType to) {
279  // The last parameter may be the special callable parameter. In that case we
280  // have to preserve it as the last parameter, i.e. we allocate it in the new
281  // location signature again in the same register.
282  bool extra_callable_param =
283  (call_descriptor->GetInputLocation(call_descriptor->InputCount() - 1) ==
284  LinkageLocation::ForRegister(kJSFunctionRegister.code(),
285  MachineType::TaggedPointer()));
286 
287  size_t return_count = call_descriptor->ReturnCount();
288  // To recover the function parameter count, disregard the instance parameter,
289  // and the extra callable parameter if present.
290  size_t parameter_count =
291  call_descriptor->ParameterCount() - (extra_callable_param ? 2 : 1);
292 
293  // Precompute if the descriptor contains {from}.
294  bool needs_change = false;
295  for (size_t i = 0; !needs_change && i < return_count; i++) {
296  needs_change = call_descriptor->GetReturnType(i) == from;
297  }
298  for (size_t i = 1; !needs_change && i < parameter_count + 1; i++) {
299  needs_change = call_descriptor->GetParameterType(i) == from;
300  }
301  if (!needs_change) return const_cast<CallDescriptor*>(call_descriptor);
302 
303  std::vector<MachineType> reps;
304 
305  for (size_t i = 0, limit = return_count; i < limit; i++) {
306  MachineType initial_type = call_descriptor->GetReturnType(i);
307  if (initial_type == from) {
308  for (size_t j = 0; j < num_replacements; j++) reps.push_back(to);
309  return_count += num_replacements - 1;
310  } else {
311  reps.push_back(initial_type);
312  }
313  }
314 
315  // Disregard the instance (first) parameter.
316  for (size_t i = 1, limit = parameter_count + 1; i < limit; i++) {
317  MachineType initial_type = call_descriptor->GetParameterType(i);
318  if (initial_type == from) {
319  for (size_t j = 0; j < num_replacements; j++) reps.push_back(to);
320  parameter_count += num_replacements - 1;
321  } else {
322  reps.push_back(initial_type);
323  }
324  }
325 
326  MachineSignature sig(return_count, parameter_count, reps.data());
327 
328  int parameter_slots;
329  int return_slots;
330  LocationSignature* location_sig = BuildLocations(
331  zone, &sig, extra_callable_param, &parameter_slots, &return_slots);
332 
333  return zone->New<CallDescriptor>( // --
334  call_descriptor->kind(), // kind
335  call_descriptor->tag(), // tag
336  call_descriptor->GetInputType(0), // target MachineType
337  call_descriptor->GetInputLocation(0), // target location
338  location_sig, // location_sig
339  parameter_slots, // parameter slot count
340  call_descriptor->properties(), // properties
341  call_descriptor->CalleeSavedRegisters(), // callee-saved registers
342  call_descriptor->CalleeSavedFPRegisters(), // callee-saved fp regs
343  call_descriptor->flags(), // flags
344  call_descriptor->debug_name(), // debug name
345  call_descriptor->GetStackArgumentOrder(), // stack order
346  call_descriptor->AllocatableRegisters(), // allocatable registers
347  return_slots, // return slot count
348  call_descriptor->IsIndirectWasmFunctionCall()
349  ? call_descriptor->signature_hash()
350  : kInvalidWasmSignatureHash); // signature hash
351 }
352 } // namespace

385 bool Linkage::NeedsFrameStateInput(Runtime::FunctionId function) {
386  switch (function) {
387  // Most runtime functions need a FrameState. A few chosen ones that we know
388  // not to call into arbitrary JavaScript, not to throw, and not to lazily
389  // deoptimize are allowlisted here and can be called without a FrameState.
390  case Runtime::kAbort:
391  case Runtime::kAllocateInOldGeneration:
392  case Runtime::kCreateIterResultObject:
393  case Runtime::kGrowableSharedArrayBufferByteLength:
394  case Runtime::kIncBlockCounter:
395  case Runtime::kNewClosure:
396  case Runtime::kNewClosure_Tenured:
397  case Runtime::kNewFunctionContext:
398  case Runtime::kPushBlockContext:
399  case Runtime::kPushCatchContext:
400  case Runtime::kStringEqual:
401  case Runtime::kStringLessThan:
402  case Runtime::kStringLessThanOrEqual:
403  case Runtime::kStringGreaterThan:
404  case Runtime::kStringGreaterThanOrEqual:
405  case Runtime::kToFastProperties: // TODO(conradw): Is it safe?
406  case Runtime::kTraceEnter:
407  case Runtime::kTraceExit:
408  return false;
409 
410  // Some inline intrinsics are also safe to call without a FrameState.
411  case Runtime::kInlineCreateIterResultObject:
412  case Runtime::kInlineIncBlockCounter:
413  case Runtime::kInlineGeneratorClose:
414  case Runtime::kInlineGeneratorGetResumeMode:
415  case Runtime::kInlineCreateJSGeneratorObject:
416  return false;
417 
418  default:
419  break;
420  }
421 
422  // For safety, default to needing a FrameState unless allowlisted.
423  return true;
424 }

509 CallDescriptor* Linkage::GetJSCallDescriptor(Zone* zone, bool is_osr,
510  int js_parameter_count,
511  CallDescriptor::Flags flags,
512  Operator::Properties properties) {
513  const size_t return_count = 1;
514  const size_t context_count = 1;
515  const size_t new_target_count = 1;
516  const size_t num_args_count = 1;
517  const size_t dispatch_handle_count =
518  V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE_BOOL ? 1 : 0;
519  const size_t parameter_count = js_parameter_count + new_target_count +
520  num_args_count + dispatch_handle_count +
521  context_count;
522 
523  // The JSCallDescriptor must be compatible both with the interface descriptor
524  // of JS builtins and with the general JS calling convention (as defined by
525  // the JSTrampolineDescriptor). The JS builtin descriptors are already
526  // statically asserted to be compatible with the JS calling convention, so
527  // here we just ensure compatibility with the JS builtin descriptors.
528  DCHECK_EQ(parameter_count, kJSBuiltinBaseParameterCount + js_parameter_count);
529 
530  LocationSignature::Builder locations(zone, return_count, parameter_count);
531 
532  // All JS calls have exactly one return value.
533  locations.AddReturn(regloc(kReturnRegister0, MachineType::AnyTagged()));
534 
535  // All parameters to JS calls go on the stack.
536  for (int i = 0; i < js_parameter_count; i++) {
537  int spill_slot_index = -i - 1;
538  locations.AddParam(LinkageLocation::ForCallerFrameSlot(
539  spill_slot_index, MachineType::AnyTagged()));
540  }
541 
542  // Add JavaScript call new target value.
543  locations.AddParam(
544  regloc(kJavaScriptCallNewTargetRegister, MachineType::AnyTagged()));
545 
546  // Add JavaScript call argument count.
547  locations.AddParam(
548  regloc(kJavaScriptCallArgCountRegister, MachineType::Int32()));
549 
550 #ifdef V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE
551  // Add dispatch handle.
552  locations.AddParam(
553  regloc(kJavaScriptCallDispatchHandleRegister, MachineType::Int32()));
554 #endif
555 
556  // Add context.
557  locations.AddParam(regloc(kContextRegister, MachineType::AnyTagged()));
558 
559  // The target for JS function calls is the JSFunction object.
560  MachineType target_type = MachineType::AnyTagged();
561  // When entering into an OSR function from unoptimized code the JSFunction
562  // is not in a register, but it is on the stack in the marker spill slot.
563  // For kind == JSDescKind::kBuiltin, we should still use the regular
564  // kJSFunctionRegister, so that frame attribution for stack traces works.
565  LinkageLocation target_loc = is_osr
566  ? LinkageLocation::ForSavedCallerFunction()
567  : regloc(kJSFunctionRegister, target_type);
568  CallDescriptor::Kind descriptor_kind = CallDescriptor::kCallJSFunction;
569  return zone->New<CallDescriptor>( // --
570  descriptor_kind, // kind
571  kJSEntrypointTag, // tag
572  target_type, // target MachineType
573  target_loc, // target location
574  locations.Get(), // location_sig
575  js_parameter_count, // stack_parameter_count
576  properties, // properties
577  kNoCalleeSaved, // callee-saved
578  kNoCalleeSavedFp, // callee-saved fp
579  flags, // flags
580  "js-call"); // debug name
581 }

583 // TODO(turbofan): cache call descriptors for code stub calls.
584 // TODO(jgruber): Clean up stack parameter count handling. The descriptor
585 // already knows the formal stack parameter count and ideally only additional
586 // stack parameters should be passed into this method. All call-sites should
587 // be audited for correctness (e.g. many used to assume a stack parameter count
588 // of 0).
589 CallDescriptor* Linkage::GetStubCallDescriptor(
590  Zone* zone, const CallInterfaceDescriptor& descriptor,
591  int stack_parameter_count, CallDescriptor::Flags flags,
592  Operator::Properties properties, StubCallMode stub_mode) {
593  const int register_parameter_count = descriptor.GetRegisterParameterCount();
594  const int js_parameter_count =
595  register_parameter_count + stack_parameter_count;
596  const int context_count = descriptor.HasContextParameter() ? 1 : 0;
597  const size_t parameter_count =
598  static_cast<size_t>(js_parameter_count + context_count);
599 
600  DCHECK_GE(stack_parameter_count, descriptor.GetStackParameterCount());
601 
602  int return_count = descriptor.GetReturnCount();
603  LocationSignature::Builder locations(zone, return_count, parameter_count);
604 
605  // Add returns.
606  for (int i = 0; i < return_count; i++) {
607  MachineType type = descriptor.GetReturnType(static_cast<int>(i));
608  if (IsFloatingPoint(type.representation())) {
609  DoubleRegister reg = descriptor.GetDoubleRegisterReturn(i);
610  locations.AddReturn(regloc(reg, type));
611  } else {
612  Register reg = descriptor.GetRegisterReturn(i);
613  locations.AddReturn(regloc(reg, type));
614  }
615  }
616 
617  // Add parameters in registers and on the stack.
618  for (int i = 0; i < js_parameter_count; i++) {
619  if (i < register_parameter_count) {
620  // The first parameters go in registers.
621  MachineType type = descriptor.GetParameterType(i);
622  if (IsFloatingPoint(type.representation())) {
623  DoubleRegister reg = descriptor.GetDoubleRegisterParameter(i);
624  locations.AddParam(regloc(reg, type));
625  } else {
626  Register reg = descriptor.GetRegisterParameter(i);
627  locations.AddParam(regloc(reg, type));
628  }
629  } else {
630  // The rest of the parameters go on the stack.
631  int stack_slot = i - register_parameter_count - stack_parameter_count;
632  locations.AddParam(LinkageLocation::ForCallerFrameSlot(
633  stack_slot, i < descriptor.GetParameterCount()
634  ? descriptor.GetParameterType(i)
635  : MachineType::AnyTagged()));
636  }
637  }
638 
639  // Add context.
640  if (context_count) {
641  locations.AddParam(regloc(kContextRegister, MachineType::AnyTagged()));
642  }
643 
644  // The target for stub calls depends on the requested mode.
645  CallDescriptor::Kind kind;
646  MachineType target_type;
647  switch (stub_mode) {
648  case StubCallMode::kCallCodeObject:
649  kind = CallDescriptor::kCallCodeObject;
650  target_type = MachineType::AnyTagged();
651  break;
652 #if V8_ENABLE_WEBASSEMBLY
653  case StubCallMode::kCallWasmRuntimeStub:
654  kind = CallDescriptor::kCallWasmFunction;
655  target_type = MachineType::Pointer();
656  break;
657 #endif // V8_ENABLE_WEBASSEMBLY
658  case StubCallMode::kCallBuiltinPointer:
659  kind = CallDescriptor::kCallBuiltinPointer;
660  target_type = MachineType::AnyTagged();
661  break;
662  }
663 
664  RegList allocatable_registers = descriptor.allocatable_registers();
665  RegList callee_saved_registers = kNoCalleeSaved;
666  if (descriptor.CalleeSaveRegisters()) {
667  callee_saved_registers = allocatable_registers;
668  DCHECK(!callee_saved_registers.is_empty());
669  }
670  LinkageLocation target_loc = LinkageLocation::ForAnyRegister(target_type);
671  return zone->New<CallDescriptor>( // --
672  kind, // kind
673  descriptor.tag(), // tag
674  target_type, // target MachineType
675  target_loc, // target location
676  locations.Get(), // location_sig
677  stack_parameter_count, // stack_parameter_count
678  properties, // properties
679  callee_saved_registers, // callee-saved registers
680  kNoCalleeSavedFp, // callee-saved fp
681  CallDescriptor::kCanUseRoots | flags, // flags
682  descriptor.DebugName(), // debug name
683  descriptor.GetStackArgumentOrder(), // stack order
684  allocatable_registers);
685 }

689  Zone* zone, const CallInterfaceDescriptor& descriptor,
690  int stack_parameter_count) {
691  const int register_parameter_count = descriptor.GetRegisterParameterCount();
692  const int parameter_count = register_parameter_count + stack_parameter_count;
693 
694  DCHECK_EQ(descriptor.GetReturnCount(), 1);
695  LocationSignature::Builder locations(zone, 1, parameter_count);
696 
697  locations.AddReturn(regloc(kReturnRegister0, descriptor.GetReturnType(0)));
698 
699  // Add parameters in registers and on the stack.
700  for (int i = 0; i < parameter_count; i++) {
701  if (i < register_parameter_count) {
702  // The first parameters go in registers.
703  Register reg = descriptor.GetRegisterParameter(i);
704  MachineType type = descriptor.GetParameterType(i);
705  locations.AddParam(regloc(reg, type));
706  } else {
707  // The rest of the parameters go on the stack.
708  int stack_slot = i - register_parameter_count - stack_parameter_count;
709  locations.AddParam(LinkageLocation::ForCallerFrameSlot(
710  stack_slot, MachineType::AnyTagged()));
711  }
712  }
713 
714  // The target for interpreter dispatches is a code entry address.
715  MachineType target_type = MachineType::Pointer();
716  LinkageLocation target_loc = LinkageLocation::ForAnyRegister(target_type);
717  const CallDescriptor::Flags kFlags =
718  CallDescriptor::kCanUseRoots | CallDescriptor::kFixedTargetRegister;
719  return zone->New<CallDescriptor>( // --
720  CallDescriptor::kCallAddress, // kind
721  kBytecodeHandlerEntrypointTag, // tag
722  target_type, // target MachineType
723  target_loc, // target location
724  locations.Get(), // location_sig
725  stack_parameter_count, // stack_parameter_count
726  Operator::kNoProperties, // properties
727  kNoCalleeSaved, // callee-saved registers
728  kNoCalleeSavedFp, // callee-saved fp
729  kFlags, // flags
730  descriptor.DebugName());

734  CHECK(incoming_->IsJSFunctionCall());
735  int parameter_count_with_receiver =
736  static_cast<int>(incoming_->JSParameterCount());
737  int first_stack_slot =
738  OsrHelper::FirstStackSlotIndex(parameter_count_with_receiver - 1);
739 
740  if (index == kOsrContextSpillSlotIndex) {
741  int context_index =
742  Linkage::GetJSCallContextParamIndex(parameter_count_with_receiver);
743  return GetParameterLocation(context_index);

748  return LinkageLocation::ForCalleeFrameSlot(spill_index,
749  MachineType::AnyTagged());

751  // Parameter. Use the assigned location from the incoming call descriptor.
752  return GetParameterLocation(index);
