39 void DeoptimizationFrameTranslationPrintSingleOpcode(
40  std::ostream& os, TranslationOpcode opcode,
41  DeoptimizationFrameTranslation::Iterator& iterator,
42  Tagged<ProtectedDeoptimizationLiteralArray> protected_literal_array,
43  Tagged<DeoptimizationLiteralArray> literal_array) {
44  disasm::NameConverter converter;
45  switch (opcode) {
46  case TranslationOpcode::BEGIN_WITH_FEEDBACK:
47  case TranslationOpcode::BEGIN_WITHOUT_FEEDBACK:
48  case TranslationOpcode::MATCH_PREVIOUS_TRANSLATION: {
49  iterator.NextOperand(); // Skip the lookback distance.
50  int frame_count = iterator.NextOperand();
51  int jsframe_count = iterator.NextOperand();
52  os << "{frame count=" << frame_count
53  << ", js frame count=" << jsframe_count << "}";
54  break;
55  }
56 
57  case TranslationOpcode::INTERPRETED_FRAME_WITH_RETURN:
58  case TranslationOpcode::INTERPRETED_FRAME_WITHOUT_RETURN: {
59  int bytecode_offset = iterator.NextOperand();
60  int shared_info_id = iterator.NextOperand();
61  int bytecode_array_id = iterator.NextOperand();
62  unsigned height = iterator.NextOperandUnsigned();
63  int return_value_offset = 0;
64  int return_value_count = 0;
65  if (opcode == TranslationOpcode::INTERPRETED_FRAME_WITH_RETURN) {
66  DCHECK_EQ(TranslationOpcodeOperandCount(opcode), 6);
67  return_value_offset = iterator.NextOperand();
68  return_value_count = iterator.NextOperand();
69  } else {
70  DCHECK_EQ(TranslationOpcodeOperandCount(opcode), 4);
71  }
72  Tagged<Object> shared_info = literal_array->get(shared_info_id);
73  Tagged<Object> bytecode_array =
74  protected_literal_array->get(bytecode_array_id);
75  os << "{bytecode_offset=" << bytecode_offset << ", function="
76  << Cast<SharedFunctionInfo>(shared_info)->DebugNameCStr().get()
77  << ", bytecode=" << Brief(bytecode_array) << ", height=" << height
78  << ", retval=@" << return_value_offset << "(#" << return_value_count
79  << ")}";
80  break;
81  }

85  DCHECK_EQ(TranslationOpcodeOperandCount(opcode), 3);

89  unsigned height = iterator.NextOperandUnsigned();
90  os << "{bailout_id=" << bailout_id << ", function="

556 Tagged<Object> TranslatedValue::GetRawValue() const {
557  // If we have a value, return it.
558  if (materialization_state() == kFinished) {
559  int smi;
560  if (IsHeapNumber(*storage_) &&
561  DoubleToSmiInteger(Object::NumberValue(*storage_), &smi)) {
562  return Smi::FromInt(smi);
563  }
564  return *storage_;
565  }
566 
567  // Otherwise, do a best effort to get the value without allocation.
568  switch (kind()) {
569  case kTagged: {
570  Tagged<Object> object = raw_literal();
571  if (IsSlicedString(object)) {
572  // If {object} is a sliced string of length smaller than
573  // SlicedString::kMinLength, then trim the underlying SeqString and
574  // return it. This assumes that such sliced strings are only built by
575  // the fast string builder optimization of Turbofan's
576  // StringBuilderOptimizer/EffectControlLinearizer.
577  Tagged<SlicedString> string = Cast<SlicedString>(object);
578  if (string->length() < SlicedString::kMinLength) {
579  Tagged<String> backing_store = string->parent();
580  CHECK(IsSeqString(backing_store));
581 
582  // Creating filler at the end of the backing store if needed.
583  int string_size =
584  IsSeqOneByteString(backing_store)
585  ? SeqOneByteString::SizeFor(backing_store->length())
586  : SeqTwoByteString::SizeFor(backing_store->length());
587  int needed_size = IsSeqOneByteString(backing_store)
588  ? SeqOneByteString::SizeFor(string->length())
589  : SeqTwoByteString::SizeFor(string->length());
590  if (needed_size < string_size) {
591  Address new_end = backing_store.address() + needed_size;
592  isolate()->heap()->CreateFillerObjectAt(
593  new_end, (string_size - needed_size));
594  }
595 
596  // Updating backing store's length, effectively trimming it.
597  backing_store->set_length(string->length());
598 
599  // Zeroing the padding bytes of {backing_store}.
600  SeqString::DataAndPaddingSizes sz =
601  Cast<SeqString>(backing_store)->GetDataAndPaddingSizes();
602  auto padding =
603  reinterpret_cast<char*>(backing_store.address() + sz.data_size);
604  for (int i = 0; i < sz.padding_size; ++i) {
605  padding[i] = 0;
606  }
607 
608  // Overwriting {string} with a filler, so that we don't leave around a
609  // potentially-too-small SlicedString.
610  isolate()->heap()->CreateFillerObjectAt(string.address(),
611  sizeof(SlicedString));
612 
613  return backing_store;
614  }
615  }
616  return object;
617  }
618 
619  case kInt32: {
620  bool is_smi = Smi::IsValid(int32_value());
621  if (is_smi) {
622  return Smi::FromInt(int32_value());
623  }
624  break;
625  }
626 
627  case kInt64: {
628  bool is_smi = (int64_value() >= static_cast<int64_t>(Smi::kMinValue) &&
629  int64_value() <= static_cast<int64_t>(Smi::kMaxValue));
630  if (is_smi) {
631  return Smi::FromIntptr(static_cast<intptr_t>(int64_value()));
632  }
633  break;
634  }
635 
636  case kInt64ToBigInt:
637  // Return the arguments marker.
638  break;
639 
640  case kUint32: {
641  bool is_smi = (uint32_value() <= static_cast<uintptr_t>(Smi::kMaxValue));
642  if (is_smi) {
643  return Smi::FromInt(static_cast<int32_t>(uint32_value()));
644  }
645  break;
646  }
647 
648  case kBoolBit: {
649  if (uint32_value() == 0) {
650  return ReadOnlyRoots(isolate()).false_value();
651  } else {
652  CHECK_EQ(1U, uint32_value());
653  return ReadOnlyRoots(isolate()).true_value();
654  }
655  }
656 
657  case kFloat: {
658  int smi;
659  if (DoubleToSmiInteger(float_value().get_scalar(), &smi)) {
660  return Smi::FromInt(smi);
661  }
662  break;
663  }
664 
665  case kHoleyDouble:
666  if (double_value().is_hole_nan()
667 #ifdef V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
668  || double_value().is_undefined_nan()
669 #endif // V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
670  ) {
671  // Hole NaNs and undefined NaNs that made it to here represent the
672  // undefined value.
673  return ReadOnlyRoots(isolate()).undefined_value();
674  }
675  // If this is not the hole nan, then this is a normal double value, so
676  // fall through to that.
677  [[fallthrough]];
678 
679  case kDouble: {
680  int smi;
681  if (DoubleToSmiInteger(double_value().get_scalar(), &smi)) {
682  return Smi::FromInt(smi);
683  }
684  break;
685  }
686 
687  default:
688  break;
689  }
690 
691  // If we could not get the value without allocation, return the arguments
692  // marker.
693  return ReadOnlyRoots(isolate()).arguments_marker();
694 }

702 Handle<Object> TranslatedValue::GetValue() {
703  Handle<Object> value(GetRawValue(), isolate());
704  if (materialization_state() == kFinished) return value;
705 
706  if (IsSmi(*value)) {
707  // Even though stored as a Smi, this number might instead be needed as a
708  // HeapNumber when materializing a JSObject with a field of HeapObject
709  // representation. Since we don't have this information available here, we
710  // just always allocate a HeapNumber and later extract the Smi again if we
711  // don't need a HeapObject.
712  set_initialized_storage(
713  isolate()->factory()->NewHeapNumber(Object::NumberValue(*value)));
714  return value;
715  }
716 
717  if (*value != ReadOnlyRoots(isolate()).arguments_marker()) {
718  set_initialized_storage(Cast<HeapObject>(value));
719  return storage_;
720  }
721 
722  // Otherwise we have to materialize.
723 
724  if (kind() == TranslatedValue::kCapturedObject ||
725  kind() == TranslatedValue::kDuplicatedObject) {
726  // We need to materialize the object (or possibly even object graphs).
727  // To make the object verifier happy, we materialize in two steps.
728 
729  // 1. Allocate storage for reachable objects. This makes sure that for
730  // each object we have allocated space on heap. The space will be
731  // a byte array that will be later initialized, or a fully
732  // initialized object if it is safe to allocate one that will
733  // pass the verifier.
734  container_->EnsureObjectAllocatedAt(this);
735 
736  // 2. Initialize the objects. If we have allocated only byte arrays
737  // for some objects, we now overwrite the byte arrays with the
738  // correct object fields. Note that this phase does not allocate
739  // any new objects, so it does not trigger the object verifier.
740  return container_->InitializeObjectAt(this);
741  }
742 
743  if (kind() == TranslatedValue::kCapturedStringConcat) {
744  // We need to materialize the string concatenation.
745  return container_->ResolveStringConcat(this);
746  }
747 
748  double number = 0;
749  Handle<HeapObject> heap_object;
750  switch (kind()) {
751  case TranslatedValue::kInt32:
752  number = int32_value();
753  heap_object = isolate()->factory()->NewHeapNumber(number);
754  break;
755  case TranslatedValue::kInt64:
756  number = int64_value();
757  heap_object = isolate()->factory()->NewHeapNumber(number);
758  break;
759  case TranslatedValue::kInt64ToBigInt:
760  heap_object = BigInt::FromInt64(isolate(), int64_value());
761  break;
762  case TranslatedValue::kUint64ToBigInt:
763  heap_object = BigInt::FromUint64(isolate(), uint64_value());
764  break;
765  case TranslatedValue::kUint32:
766  number = uint32_value();
767  heap_object = isolate()->factory()->NewHeapNumber(number);
768  break;
769  case TranslatedValue::kFloat:
770  number = float_value().get_scalar();
771  heap_object = isolate()->factory()->NewHeapNumber(number);
772  break;
773  case TranslatedValue::kDouble:
774  // We shouldn't have hole values by now, so treat holey double as normal
775  // doubles.
776  case TranslatedValue::kHoleyDouble:
777  number = double_value().get_scalar();
778  heap_object = isolate()->factory()->NewHeapNumber(number);
779  break;
780  default:
781  UNREACHABLE();
782  }
783  DCHECK(!IsSmiDouble(number) || kind() == TranslatedValue::kInt64ToBigInt ||
784  kind() == TranslatedValue::kUint64ToBigInt);
785  set_initialized_storage(heap_object);
786  return storage_;
787 }

1275 // based on dynamic information in the optimized frame.
1276 void TranslatedState::CreateArgumentsElementsTranslatedValues(
1277  int frame_index, Address input_frame_pointer, CreateArgumentsType type,
1278  FILE* trace_file) {
1279  TranslatedFrame& frame = frames_[frame_index];
1280  int length =
1281  type == CreateArgumentsType::kRestParameter
1282  ? std::max(0, actual_argument_count_ - formal_parameter_count_)
1283  : actual_argument_count_;

1285  int value_index = static_cast<int>(frame.values_.size());
1286  if (trace_file != nullptr) {

1289  }
1290 
1291  object_positions_.push_back({frame_index, value_index});

1293  this, length + OFFSET_OF_DATA_START(FixedArray) / kTaggedSize,
1294  object_index));
1295

1318  ? stack_frame_pointer_
1319  : input_frame_pointer;

1321  CommonFrameConstants::kFixedFrameSizeAboveFp +
