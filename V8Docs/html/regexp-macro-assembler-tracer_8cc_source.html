175 class PrintablePrinter {
176  public:
177  explicit PrintablePrinter(base::uc16 character) : character_(character) {}
178 
179  const char* operator*() {
180  if (character_ >= ' ' && character_ <= '~') {
181  buffer_[0] = '(';
182  buffer_[1] = static_cast<char>(character_);
183  buffer_[2] = ')';
184  buffer_[3] = '\0';
185  } else {
186  buffer_[0] = '\0';
187  }
188  return &buffer_[0];
189  }
190 
191  private:
192  base::uc16 character_;
193  char buffer_[4];
194 };

372 void RegExpMacroAssemblerTracer::SkipUntilBitInTable(
373  int cp_offset, Handle<ByteArray> table, Handle<ByteArray> nibble_table,
374  int advance_by) {
375  PrintF("SkipUntilBitInTable(cp_offset=%d, advance_by=%d\n ", cp_offset,
376  advance_by);
377  for (int i = 0; i < kTableSize; i++) {
378  PrintF("%c", table->get(i) != 0 ? 'X' : '.');
379  if (i % 32 == 31 && i != kTableMask) {
380  PrintF("\n ");
381  }
382  }
383  static_assert(kTableSize == 128);
384  static constexpr int kRows = 16;
385  static_assert(kRows * kBitsPerByte == kTableSize);
386  if (!nibble_table.is_null()) {
387  PrintF("\n");
388  PrintF(" +----------------\n");
389  PrintF(" |");
390  for (int j = 0; j < kBitsPerByte; j++) {
391  PrintF(" %x", j);
392  }
393  PrintF("\n--+----------------");
394  for (int i = 0; i < kRows; i++) {
395  int r = nibble_table->get(i);
396  PrintF("\n%x |", i);
397  for (int j = 0; j < kBitsPerByte; j++) {
398  PrintF(" %c", (r & (1 << j)) == 0 ? '.' : 'X');
399  }
400  }
401  }
402  PrintF(");\n");
403  assembler_->SkipUntilBitInTable(cp_offset, table, nibble_table, advance_by);
404 }
