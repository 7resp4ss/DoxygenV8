73  MaybeReduceResult& operator=(const MaybeReduceResult&) V8_NOEXCEPT = default;

75  ValueNode* value() const {
76  DCHECK(HasValue());
77  return payload_.GetPointerWithKnownPayload(kDoneWithValue);
78  }
79  bool HasValue() const { return kind() == kDoneWithValue; }
80 
81  // Either DoneWithValue, DoneWithoutValue or DoneWithAbort.
82  bool IsDone() const { return !IsFail(); }
83 
84  // MaybeReduceResult failed.
85  bool IsFail() const { return kind() == kFail; }
86 
87  // Done with a ValueNode.
88  bool IsDoneWithValue() const { return HasValue(); }
89 
90  // Done without producing a ValueNode.
91  bool IsDoneWithoutValue() const { return kind() == kDoneWithoutValue; }
92 
93  // Done with an abort (unconditional deopt, infinite loop in an inlined
94  // function, etc)
95  bool IsDoneWithAbort() const { return kind() == kDoneWithAbort; }
96 
97  Kind kind() const { return payload_.GetPayload(); }
98 
99  inline ReduceResult Checked();
100 
101  base::PointerWithPayload<ValueNode, Kind, 3> GetPayload() const {
102  return payload_;
103  }
104 
105  protected:
106  explicit MaybeReduceResult(Kind kind) : payload_(kind) {}
107  explicit MaybeReduceResult(
108  base::PointerWithPayload<ValueNode, Kind, 3> payload)
109  : payload_(payload) {}
110  base::PointerWithPayload<ValueNode, Kind, 3> payload_;

300  source_position_iterator_.code_offset() < entrypoint_) {

617  // means we would deoptimize to the caller and then probably end up in the
618  // same maglev osr code again, before reaching the turbofan OSR code in the

681  // Splits incoming critical edges and labels predecessors.
682  void ProcessMergePointPredecessors(
683  MergePointInterpreterFrameState& merge_state,
684  BasicBlockRef& jump_targets) {
685  // TODO(olivf): Support allocation folding across control flow.
686  DCHECK_EQ(current_allocation_block_, nullptr);
687 
688  // Merges aren't simple fallthroughs, so we should reset state which is
689  // cached directly on the builder instead of on the merge states.
690  ResetBuilderCachedState();
691 
692  if (merge_state.is_loop()) {
693  DCHECK_EQ(merge_state.predecessors_so_far(),
694  merge_state.predecessor_count() - 1);
695  } else {
696  DCHECK_EQ(merge_state.predecessors_so_far(),
697  merge_state.predecessor_count());
698  }
699 
700  if (merge_state.predecessor_count() == 1) return;
701 
702  // Set up edge-split.
703  int predecessor_index = merge_state.predecessor_count() - 1;
704  if (merge_state.is_loop()) {
705  // For loops, the JumpLoop block hasn't been generated yet, and so isn't
706  // in the list of jump targets. IT's the last predecessor, so drop the
707  // index by one.
708  DCHECK(merge_state.is_unmerged_loop());
709  predecessor_index--;
710  }
711  BasicBlockRef* old_jump_targets = jump_targets.Reset();
712  while (old_jump_targets != nullptr) {
713  BasicBlock* predecessor = merge_state.predecessor_at(predecessor_index);
714  CHECK(predecessor);
715  ControlNode* control = predecessor->control_node();
716  if (control->Is<ConditionalControlNode>()) {
717  // CreateEmptyBlock automatically registers itself with the offset.
718  predecessor = CreateEdgeSplitBlock(jump_targets, predecessor);
719  // Set the old predecessor's (the conditional block) reference to
720  // point to the new empty predecessor block.
721  old_jump_targets =
722  old_jump_targets->SetToBlockAndReturnNext(predecessor);
723  merge_state.set_predecessor_at(predecessor_index, predecessor);
724  } else {
725  // Re-register the block in the offset's ref list.
726  old_jump_targets = old_jump_targets->MoveToRefList(&jump_targets);
727  }
728  // We only set the predecessor id after splitting critical edges, to make
729  // sure the edge split blocks pick up the correct predecessor index.
730  predecessor->set_predecessor_id(predecessor_index--);
731  }
732  DCHECK_EQ(predecessor_index, -1);
733  RegisterPhisWithGraphLabeller(merge_state);
734  }

860  void VisitSingleBytecode() {

870

872  if (V8_UNLIKELY(merge_state != nullptr)) {
873  bool preserve_known_node_aspects = in_optimistic_peeling_iteration() &&
874  loop_headers_to_peel_.Contains(offset);
875  if (merge_state->is_resumable_loop()) {
876  current_for_in_state.enum_cache_indices = nullptr;
877  }
878  if (current_block_ != nullptr) {
879  DCHECK(!preserve_known_node_aspects);
880  // TODO(leszeks): Re-evaluate this DCHECK, we might hit it if the only
881  // bytecodes in this basic block were only register juggling.
882  // DCHECK(!node_buffer().empty());

884  if (merge_state->is_loop() && !merge_state->is_resumable_loop() &&
885  need_checkpointed_loop_entry()) {
886  predecessor =
887  FinishBlock<CheckpointedJump>({}, &jump_targets_[offset]);
888  } else {
889  predecessor = FinishBlock<Jump>({}, &jump_targets_[offset]);
890  }
891  merge_state->Merge(this, *compilation_unit_, current_interpreter_frame_,
892  predecessor);
893  }
894  if (v8_flags.trace_maglev_graph_building) {
895  auto detail = merge_state->is_exception_handler() ? "exception handler"
896  : merge_state->is_loop() ? "loop header"
897  : "merge";

899  << ") at "
900  << compilation_unit()->shared_function_info().object()
901  << "==" << std::endl;
902  PrintVirtualObjects();
903  }
904 
905  if (V8_UNLIKELY(merge_state->is_exception_handler())) {
906  CHECK_EQ(predecessor_count(offset), 0);
907  // If we have no reference to this block, then the exception handler is
908  // dead.
909  if (!jump_targets_[offset].has_ref() ||
910  !merge_state->exception_handler_was_used()) {
911  MarkBytecodeDead();
912  return;
913  }
914  ProcessMergePointAtExceptionHandlerStart(offset);
915  } else if (merge_state->is_unmerged_unreachable_loop()) {
916  // We encountered a loop header that is only reachable by the JumpLoop
917  // back-edge, but the bytecode_analysis didn't notice upfront. This can
918  // e.g. be a loop that is entered on a dead fall-through.
919  static_assert(kLoopsMustBeEnteredThroughHeader);
920  MarkBytecodeDead();
921  return;
922  } else {
923  ProcessMergePoint(offset, preserve_known_node_aspects);
924  }
925 
926  if (is_loop_effect_tracking_enabled() && merge_state->is_loop()) {
927  BeginLoopEffects(offset);
928  }
929  // We pass nullptr for the `predecessor` argument of StartNewBlock because
930  // this block is guaranteed to have a merge_state_, and hence to not have
931  // a `predecessor_` field.
932  StartNewBlock(offset, /*predecessor*/ nullptr);
933  } else if (V8_UNLIKELY(current_block_ == nullptr)) {
934  // If we don't have a current block, the bytecode must be dead (because of
935  // some earlier deopt). Mark this bytecode dead too and return.
936  // TODO(leszeks): Merge these two conditions by marking dead states with
937  // a sentinel value.
938  if (predecessor_count(offset) == 1) {
939  CHECK_NULL(merge_state);
940  CHECK(bytecode_analysis().IsLoopHeader(offset));
941  } else {
942  CHECK_EQ(predecessor_count(offset), 0);
943  }
944  MarkBytecodeDead();
945  return;
946  }
947 
948  // Handle exceptions if we have a table.
949  if (bytecode().handler_table_size() > 0) {
950  // Pop all entries where offset >= end.
951  while (IsInsideTryBlock()) {
952  HandlerTableEntry& entry = catch_block_stack_.top();
953  if (offset < entry.end) break;
954  catch_block_stack_.pop();
955  }
956  // Push new entries from interpreter handler table where offset >= start
957  // && offset < end.
958  HandlerTable table(*bytecode().object());
959  while (next_handler_table_index_ < table.NumberOfRangeEntries()) {
960  int start = table.GetRangeStart(next_handler_table_index_);
961  if (offset < start) break;
962  int end = table.GetRangeEnd(next_handler_table_index_);
963  if (offset >= end) {
964  next_handler_table_index_++;
965  continue;
966  }
967  int handler = table.GetRangeHandler(next_handler_table_index_);
968  catch_block_stack_.push({end, handler});
969  DCHECK_NOT_NULL(merge_states_[handler]);
970  next_handler_table_index_++;
971  }
972  }
973 
974  DCHECK_NOT_NULL(current_block_);
975 #ifdef DEBUG
976  // Clear new nodes for the next VisitFoo
977  new_nodes_.clear();
978 #endif
979 
980  if (iterator_.current_bytecode() == interpreter::Bytecode::kJumpLoop &&
981  iterator_.GetJumpTargetOffset() < entrypoint_) {
982  static_assert(kLoopsMustBeEnteredThroughHeader);
983  CHECK(EmitUnconditionalDeopt(DeoptimizeReason::kOSREarlyExit)
984  .IsDoneWithAbort());
985  MarkBytecodeDead();
986  return;
987  }

1221  void AttachExceptionHandlerInfo(NodeT* node) {
1222  if constexpr (NodeT::kProperties.can_throw()) {
1223  CatchBlockDetails catch_block = GetCurrentTryCatchBlock();

1227  // matching catch but we'll lazy deopt if we ever throw.

1229  ExceptionHandlerInfo(ExceptionHandlerInfo::kLazyDeopt);

1233  if (is_non_eager_inlining_enabled()) {
1234  // Ensure that we always have the handler of inline call
1235  // candidates.
1236  current_block_->AddExceptionHandler(
1237  node->exception_handler_info());
1238  }
1239  }
1240  return;
1241  }
1242 
1243  DCHECK_IMPLIES(!IsInsideTryBlock(), is_inline());
1244  if (catch_block.block_already_exists) {
1245  DCHECK(!IsInsideTryBlock());

1253  // block doesn't exist yet, use the ref-list mechanism.
1254  new (node->exception_handler_info()) ExceptionHandlerInfo(

1256  }

1259  DCHECK(!node->exception_handler_info()->ShouldLazyDeopt());
1260 
1261  current_block_->AddExceptionHandler(node->exception_handler_info());

1264  // Merge the current state into the handler state.
1265  auto state = GetCatchBlockFrameState();

1267  state->MergeThrow(this, compilation_unit_,

1269  current_interpreter_frame_.virtual_objects());
1270  }
1271  } else {

3086  void CalculatePredecessorCounts() {
3087  // Add 1 after the end of the bytecode so we can always write to the offset
3088  // after the last bytecode.
3089  uint32_t array_length = bytecode().length() + 1;
3090  predecessor_count_ = zone()->AllocateArray<uint32_t>(array_length);
3091  MemsetUint32(predecessor_count_, 0, entrypoint_);
3092  MemsetUint32(predecessor_count_ + entrypoint_, 1,
3093  array_length - entrypoint_);
3094 
3095  const int max_peelings = v8_flags.maglev_optimistic_peeled_loops ? 2 : 1;
3096  // We count jumps from peeled loops to outside of the loop twice.
3097  bool is_loop_peeling_iteration = false;
3098  std::optional<int> peeled_loop_end;
3099  interpreter::BytecodeArrayIterator iterator(bytecode().object());
3100  for (iterator.AdvanceTo(entrypoint_); !iterator.done();
3101  iterator.Advance()) {
3102  interpreter::Bytecode bytecode = iterator.current_bytecode();
3103  if (allow_loop_peeling_ &&
3104  bytecode_analysis().IsLoopHeader(iterator.current_offset())) {
3105  const compiler::LoopInfo& loop_info =
3106  bytecode_analysis().GetLoopInfoFor(iterator.current_offset());
3107  // Generators use irreducible control flow, which makes loop peeling too
3108  // complicated.
3109  int size = loop_info.loop_end() - loop_info.loop_start();
3110  if (loop_info.innermost() && !loop_info.resumable() &&
3111  iterator.next_offset() < loop_info.loop_end() &&
3112  size < v8_flags.maglev_loop_peeling_max_size &&
3113  size + graph_->total_peeled_bytecode_size() <
3114  v8_flags.maglev_loop_peeling_max_size_cumulative) {
3115  DCHECK(!is_loop_peeling_iteration);
3116  graph_->add_peeled_bytecode_size(size);
3117  is_loop_peeling_iteration = true;
3118  loop_headers_to_peel_.Add(iterator.current_offset());
3119  peeled_loop_end = bytecode_analysis().GetLoopEndOffsetForInnermost(
3120  iterator.current_offset());
3121  }
3122  }
3123  if (interpreter::Bytecodes::IsJump(bytecode)) {
3124  if (is_loop_peeling_iteration &&
3125  bytecode == interpreter::Bytecode::kJumpLoop) {
3126  DCHECK_EQ(iterator.next_offset(), peeled_loop_end);
3127  is_loop_peeling_iteration = false;
3128  peeled_loop_end = {};
3129  }
3130  if (iterator.GetJumpTargetOffset() < entrypoint_) {
3131  static_assert(kLoopsMustBeEnteredThroughHeader);
3132  if (predecessor_count(iterator.GetJumpTargetOffset()) == 1) {
3133  // We encountered a JumpLoop whose loop header is not reachable
3134  // otherwise. This loop is either dead or the JumpLoop will bail
3135  // with DeoptimizeReason::kOSREarlyExit.
3136  InitializePredecessorCount(iterator.GetJumpTargetOffset(), 0);
3137  }
3138  } else {
3139  UpdatePredecessorCount(iterator.GetJumpTargetOffset(), 1);
3140  }
3141  if (is_loop_peeling_iteration &&
3142  iterator.GetJumpTargetOffset() >= *peeled_loop_end) {
3143  // Jumps from within the peeled loop to outside need to be counted
3144  // twice, once for the peeled and once for the regular loop body.
3145  UpdatePredecessorCount(iterator.GetJumpTargetOffset(), max_peelings);
3146  }
3147  if (!interpreter::Bytecodes::IsConditionalJump(bytecode)) {
3148  UpdatePredecessorCount(iterator.next_offset(), -1);
3149  }
3150  } else if (interpreter::Bytecodes::IsSwitch(bytecode)) {
3151  for (auto offset : iterator.GetJumpTableTargetOffsets()) {
3152  UpdatePredecessorCount(offset.target_offset, 1);
3153  }
3154  } else if (interpreter::Bytecodes::Returns(bytecode) ||
3155  interpreter::Bytecodes::UnconditionallyThrows(bytecode)) {
3156  UpdatePredecessorCount(iterator.next_offset(), -1);
3157  // Collect inline return jumps in the slot after the last bytecode.
3158  if (is_inline() && interpreter::Bytecodes::Returns(bytecode)) {
3159  UpdatePredecessorCount(array_length - 1, 1);
3160  if (is_loop_peeling_iteration) {
3161  UpdatePredecessorCount(array_length - 1, max_peelings);
3162  }
3163  }
3164  }
3165  // TODO(leszeks): Also consider handler entries (the bytecode analysis)
3166  // will do this automatically I guess if we merge this into that.
3167  }
3168  if (!is_inline()) {
3169  DCHECK_EQ(0, predecessor_count(bytecode().length()));
3170  }
3171  }
