34 bool CanInlinePropertyAccess(MapRef map, AccessMode access_mode) {
35  // We can inline property access to prototypes of all primitives, except
36  // the special Oddball ones that have no wrapper counterparts (i.e. Null,
37  // Undefined and TheHole).
38  // We can only inline accesses to dictionary mode holders if the access is a
39  // load and the holder is a prototype. The latter ensures a 1:1
40  // relationship between the map and the object (and therefore the property
41  // dictionary).
42  static_assert(ODDBALL_TYPE == LAST_PRIMITIVE_HEAP_OBJECT_TYPE);
43  if (IsBooleanMap(*map.object())) return true;
44  if (map.instance_type() < LAST_PRIMITIVE_HEAP_OBJECT_TYPE) return true;
45  if (IsJSObjectMap(*map.object())) {
46  if (map.is_dictionary_map()) {
47  if (!V8_DICT_PROPERTY_CONST_TRACKING_BOOL) return false;
48  return access_mode == AccessMode::kLoad &&
49  map.object()->is_prototype_map();
50  }
51  return !map.object()->has_named_interceptor() &&
52  // TODO(verwaest): Allowlist contexts to which we have access.
53  !map.is_access_check_needed();
54  }
55 #if V8_ENABLE_WEBASSEMBLY
56  if (IsWasmObjectMap(*map.object())) {
57  DCHECK(!map.object()->has_named_interceptor());
58  DCHECK(!map.is_access_check_needed());
59  // Accesses to Wasm objects all go through the prototype chain, but
60  // we can't express that in this helper function.
61  return true;
62  }
63 #endif // V8_ENABLE_WEBASSEMBLY
64  return false;
65 }

307 bool PropertyAccessInfo::Merge(PropertyAccessInfo const* that,
308  AccessMode access_mode, Zone* zone) {
309  if (kind_ != that->kind_) return false;
310  if (!OptionalRefEquals(holder_, that->holder_)) return false;
311 
312  switch (kind_) {
313  case kInvalid:
314  DCHECK_EQ(that->kind_, kInvalid);
315  return true;
316 
317  case kDataField:
318  case kFastDataConstant: {
319  // Check if we actually access the same field (we use the
320  // GetFieldAccessStubKey method here just like the ICs do
321  // since that way we only compare the relevant bits of the
322  // field indices).
323  if (field_index_.GetFieldAccessStubKey() !=
324  that->field_index_.GetFieldAccessStubKey()) {
325  return false;
326  }
327 
328  switch (access_mode) {
329  case AccessMode::kHas:
330  case AccessMode::kLoad: {
331  if (!field_representation_.Equals(that->field_representation_)) {
332  if (field_representation_.IsDouble() ||
333  that->field_representation_.IsDouble()) {
334  return false;
335  }
336  field_representation_ = Representation::Tagged();
337  }
338  if (!OptionalRefEquals(field_map_, that->field_map_)) {
339  field_map_ = {};
340  }
341  break;
342  }
343  case AccessMode::kStore:
344  case AccessMode::kStoreInLiteral:
345  case AccessMode::kDefine: {
346  // For stores, the field map and field representation information
347  // must match exactly, otherwise we cannot merge the stores. We
348  // also need to make sure that in case of transitioning stores,
349  // the transition targets match.
350  if (!OptionalRefEquals(field_map_, that->field_map_) ||
351  !field_representation_.Equals(that->field_representation_) ||
352  !OptionalRefEquals(transition_map_, that->transition_map_)) {
353  return false;
354  }
355  break;
356  }
357  }
358 
359  field_type_ = Type::Union(field_type_, that->field_type_, zone);
360  AppendVector(&lookup_start_object_maps_, that->lookup_start_object_maps_);
361  AppendVector(&unrecorded_dependencies_, that->unrecorded_dependencies_);
362  return true;
363  }
364 
365  case kDictionaryProtoAccessorConstant:
366  case kFastAccessorConstant: {
367  // Check if we actually access the same constant.
368  if (!OptionalRefEquals(constant_, that->constant_)) return false;
369 
370  DCHECK(unrecorded_dependencies_.empty());
371  DCHECK(that->unrecorded_dependencies_.empty());
372  AppendVector(&lookup_start_object_maps_, that->lookup_start_object_maps_);
373  return true;
374  }
375 
376  case kDictionaryProtoDataConstant: {
377  DCHECK_EQ(AccessMode::kLoad, access_mode);
378  if (dictionary_index_ != that->dictionary_index_) return false;
379  AppendVector(&lookup_start_object_maps_, that->lookup_start_object_maps_);
380  return true;
381  }
382 
383  case kNotFound:
384  case kStringLength:
385  case kStringWrapperLength: {
386  DCHECK(unrecorded_dependencies_.empty());
387  DCHECK(that->unrecorded_dependencies_.empty());
388  AppendVector(&lookup_start_object_maps_, that->lookup_start_object_maps_);
389  return true;
390  }
391  case kTypedArrayLength:
392  DCHECK_EQ(lookup_start_object_maps_.size(), 1);
393  DCHECK_EQ(that->lookup_start_object_maps_.size(), 1);
394  return lookup_start_object_maps_[0] == that->lookup_start_object_maps_[0];
395  case kModuleExport:
396  return false;
397  }
398 }

408 bool AccessInfoFactory::ObjectMayHaveElements(JSObjectRef obj,
409  MapRef map) const {
410  if (IsCustomElementsReceiverInstanceType(map.instance_type())) return true;
411  // Ruled out by non-"custom elements" receiver type:
412  DCHECK(!map.has_indexed_interceptor());
413  if (map.is_extensible()) return true;
414  // When the map is not extensible (as we just checked), an empty elements
415  // array will always remain empty.
416  OptionalFixedArrayBaseRef elements = obj.elements(broker(), kRelaxedLoad);
417  if (!elements.has_value()) return true;
418  return (elements.value() != broker()->empty_fixed_array() &&
419  elements.value() != broker()->empty_slow_element_dictionary());
420 }

422 bool AccessInfoFactory::ObjectMayHaveOwnProperties(JSObjectRef obj,
423  MapRef map) const {
424  if (IsSpecialReceiverInstanceType(map.instance_type())) return true;
425  // Ruled out by non-"special" receiver type:
426  DCHECK(!map.has_named_interceptor());
427  if (map.is_extensible()) return true;
428  // When the map is not extensible (as we just checked), an empty property
429  // backing store will always remain empty.
430  if (map.is_dictionary_map()) {
431  if (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
432  Tagged<SwissNameDictionary> dict =
433  obj.object()->property_dictionary_swiss();
434  return dict->NumberOfElements() > 0;
435  } else {
436  Tagged<NameDictionary> dict = obj.object()->property_dictionary();
437  return dict->NumberOfElements() > 0;
438  }
439  } else {
440  return map.NumberOfOwnDescriptors() > 0;
441  }
442 }

444 std::optional<ElementAccessInfo> AccessInfoFactory::ComputeElementAccessInfo(
445  MapRef map, AccessMode access_mode) const {
446  if (map.CanInlineElementAccess()) {
447  return ElementAccessInfo({{map}, zone()}, map.elements_kind(), zone());
448  }
449 #if V8_ENABLE_WEBASSEMBLY
450  if (map.IsWasmObjectMap() &&
451  (access_mode == AccessMode::kLoad || access_mode == AccessMode::kStore)) {
452  // See if there is a Proxy on the prototype chain that will handle
453  // element accesses.
454  HeapObjectRef prototype = map.prototype(broker());
455  bool prototypes_may_have_properties = false;
456  while (true) {
457  if (prototype.IsNull()) return {};
458  MapRef proto_map = prototype.map(broker());
459  if (proto_map.is_access_check_needed()) return {};
460  if (proto_map.IsJSObjectMap()) {
461  // Any regular prototypes on the chain must be guaranteed not to have
462  // or later acquire any elements. We need to require this for both
463  // indexed and named mode.
464  if (ObjectMayHaveElements(prototype.AsJSObject(), proto_map)) return {};
465  // If the Proxy trap we'll find later (see reads of
466  // {prototypes_may_have_properties}) supports named properties, we'll
467  // need to know whether prototypes contained any named properties.
468  prototypes_may_have_properties =
469  prototypes_may_have_properties ||
470  ObjectMayHaveOwnProperties(prototype.AsJSObject(), proto_map);
471  // This prototype is okay. Continue walking the chain.
472  prototype = proto_map.prototype(broker());
473  continue;
474  }
475  if (proto_map.IsJSProxyMap()) {
476  JSProxyRef proxy = prototype.AsJSProxy();
477  if (proxy.is_revocable()) return {};
478 
479  // Check the "handler". We do this first because the specific trap
480  // function we find determines what to require from the "target".
481  OptionalObjectRef maybe_handler = proxy.GetHandler(broker());
482  if (!maybe_handler.has_value()) return {};
483  if (!maybe_handler->IsJSObject()) return {};
484  JSObjectRef handler = maybe_handler.value().AsJSObject();
485  MapRef handler_map = handler.map(broker());
486  // We could implement support for dictionary-mode handlers, but it's
487  // probably not worth it.
488  if (handler_map.is_dictionary_map()) return {};
489  NameRef trap_name = access_mode == AccessMode::kLoad
490  ? broker()->get_string()
491  : broker()->set_string();
492  PropertyAccessInfo trap_info = broker()->GetPropertyAccessInfo(
493  handler_map, trap_name, AccessMode::kLoad);
494  if (!trap_info.IsFastDataConstant()) return {};
495  Representation repr = trap_info.field_representation();
496  if (!repr.IsHeapObject() && !repr.IsTagged()) return {};
497  OptionalObjectRef trap = handler.GetOwnFastConstantDataProperty(
498  broker(), repr, trap_info.field_index(), dependencies());
499  if (!trap.has_value()) return {};
500  ObjectRef trap_value = trap.value();
501 
502  // Check that the trap is a Wasm function. We require that for spec
503  // compliance reasons: the JS-to-Wasm value conversion of parameters
504  // hides the fact that we're not emitting toString conversion operations
505  // for the property keys, which is in particular helpful when the
506  // Wasm function wants integer indices anyway.
507  if (!trap_value.IsHeapObject()) return {};
508  // Support for other callables is not implemented.
509  if (!trap_value.AsHeapObject().IsJSFunction()) return {};
510 
511  SharedFunctionInfoRef sfi = trap_value.AsJSFunction().shared(broker());
512  Tagged<Object> trusted_data =
513  sfi.object()->GetTrustedData(broker()->local_isolate_or_isolate());
514  if (!IsWasmExportedFunctionData(trusted_data)) return {};
515  Tagged<WasmExportedFunctionData> wasm_data =
516  Cast<WasmExportedFunctionData>(trusted_data);
517  // Supporting receiver-is-first-param mode would require passing
518  // the Proxy's handler to the eventual building of the Call node.
519  if (wasm_data->receiver_is_first_param()) return {};
520  const wasm::CanonicalSig* wasm_signature = wasm_data->sig();
521  if (wasm_signature->parameter_count() < 2) return {};
522  wasm::CanonicalValueType key_type = wasm_signature->GetParam(1);
523 
524  // Check the "target".
525  OptionalObjectRef maybe_target = proxy.GetTarget(broker());
526  if (!maybe_target.has_value()) return {};
527  if (!maybe_target->IsJSObject()) return {};
528  JSObjectRef target = maybe_target.value().AsJSObject();
529  MapRef target_map = target.map(broker());
530  if (ObjectMayHaveElements(target, target_map)) return {};
531  if (key_type == wasm::kWasmExternRef) {
532  if (prototypes_may_have_properties) return {};
533  if (ObjectMayHaveOwnProperties(target, target_map)) return {};
534  } else if (key_type != wasm::kWasmI32) {
535  return {};
536  }
537  if (!target_map.prototype(broker()).IsNull()) return {};
538 
539  // Finally: all good!
540  bool string_keys = key_type == wasm::kWasmExternRef;
541  return ElementAccessInfo(map, trap_value, target, string_keys, zone());
542  }
543  // Nothing else can occur on prototype chains.
544  UNREACHABLE();
545  }
546  }
547 #endif // V8_ENABLE_WEBASSEMBLY
548  return {};
549 }

551 bool AccessInfoFactory::ComputeElementAccessInfos(
552  ElementAccessFeedback const& feedback,
553  ZoneVector<ElementAccessInfo>* access_infos) const {
554  AccessMode access_mode = feedback.keyed_mode().access_mode();
555  if (access_mode == AccessMode::kLoad || access_mode == AccessMode::kHas) {
556  // For polymorphic loads of similar elements kinds (i.e. all tagged or all
557  // double), always use the "worst case" code without a transition. This is
558  // much faster than transitioning the elements to the worst case, trading a
559  // TransitionElementsKind for a CheckMaps, avoiding mutation of the array.
560  std::optional<ElementAccessInfo> access_info =
561  ConsolidateElementLoad(feedback);
562  if (access_info.has_value()) {
563  access_infos->push_back(*access_info);
564  return true;
565  }
566  }
567 
568  for (auto const& group : feedback.transition_groups()) {
569  DCHECK(!group.empty());
570  OptionalMapRef target = group.front();
571  std::optional<ElementAccessInfo> access_info =
572  ComputeElementAccessInfo(target.value(), access_mode);
573  if (!access_info.has_value()) return false;
574 
575  for (size_t i = 1; i < group.size(); ++i) {
576  OptionalMapRef map_ref = group[i];
577  if (!map_ref.has_value()) continue;
578  access_info->AddTransitionSource(map_ref.value());
579  }
580  access_infos->push_back(*access_info);
581  }
582  return true;
583 }

585 PropertyAccessInfo AccessInfoFactory::ComputeDataFieldAccessInfo(
586  MapRef receiver_map, MapRef map, NameRef name, OptionalJSObjectRef holder,
587  InternalIndex descriptor, AccessMode access_mode) const {
588  DCHECK(descriptor.is_found());
589  // TODO(jgruber,v8:7790): Use DescriptorArrayRef instead.
590  DirectHandle<DescriptorArray> descriptors =
591  map.instance_descriptors(broker()).object();
592  PropertyDetails const details = descriptors->GetDetails(descriptor);
593  int index = descriptors->GetFieldIndex(descriptor);
594  Representation details_representation = details.representation();
595  if (details_representation.IsNone()) {
596  // The ICs collect feedback in PREMONOMORPHIC state already,
597  // but at this point the {receiver_map} might still contain
598  // fields for which the representation has not yet been
599  // determined by the runtime. So we need to catch this case
600  // here and fall back to use the regular IC logic instead.
601  return Invalid();
602  }
603  FieldIndex field_index = FieldIndex::ForPropertyIndex(*map.object(), index,
604  details_representation);
605  // Private brands are used when loading private methods, which are stored in a
606  // BlockContext, an internal object.
607  Type field_type = name.object()->IsPrivateBrand() ? Type::OtherInternal()
608  : Type::NonInternal();
609  OptionalMapRef field_map;
610 
611  ZoneVector<CompilationDependency const*> unrecorded_dependencies(zone());
612 
613  Handle<FieldType> descriptors_field_type =
614  broker()->CanonicalPersistentHandle(
615  descriptors->GetFieldType(descriptor));
616  OptionalObjectRef descriptors_field_type_ref =
617  TryMakeRef<Object>(broker(), descriptors_field_type);
618  if (!descriptors_field_type_ref.has_value()) return Invalid();
619 
620  // Note: FindFieldOwner may be called multiple times throughout one
621  // compilation. This is safe since its result is fixed for a given map and
622  // descriptor.
623  MapRef field_owner_map = map.FindFieldOwner(broker(), descriptor);
624 
625  if (details_representation.IsSmi()) {
626  field_type = Type::SignedSmall();
627  unrecorded_dependencies.push_back(
628  dependencies()->FieldRepresentationDependencyOffTheRecord(
629  map, field_owner_map, descriptor, details_representation));
630  } else if (details_representation.IsDouble()) {
631  field_type = type_cache_->kFloat64;
632  unrecorded_dependencies.push_back(
633  dependencies()->FieldRepresentationDependencyOffTheRecord(
634  map, field_owner_map, descriptor, details_representation));
635  } else if (details_representation.IsHeapObject()) {
636  if (IsNone(*descriptors_field_type)) {
637  // Cleared field-types are pre-monomorphic states. The field type was
638  // garbge collected and we need to record an updated type.
639  static_assert(FieldType::kFieldTypesCanBeClearedOnGC);
640  switch (access_mode) {
641  case AccessMode::kStore:
642  case AccessMode::kStoreInLiteral:
643  case AccessMode::kDefine:
644  return Invalid();
645  case AccessMode::kLoad:
646  case AccessMode::kHas:
647  break;
648  }
649  }
650  unrecorded_dependencies.push_back(
651  dependencies()->FieldRepresentationDependencyOffTheRecord(
652  map, field_owner_map, descriptor, details_representation));
653  if (IsClass(*descriptors_field_type)) {
654  // Remember the field map, and try to infer a useful type.
655  OptionalMapRef maybe_field_map =
656  TryMakeRef(broker(), FieldType::AsClass(*descriptors_field_type));
657  if (!maybe_field_map.has_value()) return Invalid();
658  field_map = maybe_field_map;
659  // field_type can only be inferred from field_map if it is stable and we
660  // add a stability dependency. This happens on use in the access builder.
661  }
662  } else {
663  CHECK(details_representation.IsTagged());
664  }
665  // TODO(turbofan): We may want to do this only depending on the use
666  // of the access info.
667  unrecorded_dependencies.push_back(
668  dependencies()->FieldTypeDependencyOffTheRecord(
669  map, field_owner_map, descriptor,
670  descriptors_field_type_ref.value()));
671 
672  PropertyConstness constness =
673  map.GetPropertyDetails(broker_, descriptor).constness();
674  switch (constness) {
675  case PropertyConstness::kMutable:
676  return PropertyAccessInfo::DataField(
677  broker(), zone(), receiver_map, std::move(unrecorded_dependencies),
678  field_index, details_representation, field_type, field_owner_map,
679  field_map, holder, {});
680  case PropertyConstness::kConst:
681  auto constness_dep = dependencies()->FieldConstnessDependencyOffTheRecord(
682  map, field_owner_map, descriptor);
683  unrecorded_dependencies.push_back(constness_dep);
684  return PropertyAccessInfo::FastDataConstant(
685  zone(), receiver_map, std::move(unrecorded_dependencies), field_index,
686  details_representation, field_type, field_owner_map, field_map,
687  holder, {});
688  }
689  UNREACHABLE();
690 }

696 PropertyAccessInfo AccessorAccessInfoHelper(
697  Isolate* isolate, Zone* zone, JSHeapBroker* broker,
698  const AccessInfoFactory* ai_factory, MapRef receiver_map, NameRef name,
699  MapRef holder_map, OptionalJSObjectRef holder, AccessMode access_mode,
700  AccessorsObjectGetter get_accessors) {
701  if (holder_map.instance_type() == JS_MODULE_NAMESPACE_TYPE) {
702  DCHECK(holder_map.object()->is_prototype_map());
703  DirectHandle<PrototypeInfo> proto_info = broker->CanonicalPersistentHandle(
704  Cast<PrototypeInfo>(holder_map.object()->prototype_info()));
705  DirectHandle<JSModuleNamespace> module_namespace =
706  broker->CanonicalPersistentHandle(
707  Cast<JSModuleNamespace>(proto_info->module_namespace()));
708  Handle<Cell> cell = broker->CanonicalPersistentHandle(
709  Cast<Cell>(module_namespace->module()->exports()->Lookup(
710  isolate, name.object(),
711  Smi::ToInt(Object::GetHash(*name.object())))));
712  if (IsAnyStore(access_mode)) {
713  // ES#sec-module-namespace-exotic-objects-set-p-v-receiver
714  // ES#sec-module-namespace-exotic-objects-defineownproperty-p-desc
715  //
716  // Storing to a module namespace object is always an error or a no-op in
717  // JS.
718  return PropertyAccessInfo::Invalid(zone);
719  }
720  if (IsTheHole(cell->value(kRelaxedLoad), isolate)) {
721  // This module has not been fully initialized yet.
722  return PropertyAccessInfo::Invalid(zone);
723  }
724  OptionalCellRef cell_ref = TryMakeRef(broker, cell);
725  if (!cell_ref.has_value()) {
726  return PropertyAccessInfo::Invalid(zone);
727  }
728  return PropertyAccessInfo::ModuleExport(zone, receiver_map,
729  cell_ref.value());
730  }
731  if (access_mode == AccessMode::kHas) {
732  // kHas is not supported for dictionary mode objects.
733  DCHECK(!holder_map.is_dictionary_map());
734 
735  // HasProperty checks don't call getter/setters, existence is sufficient.
736  return PropertyAccessInfo::FastAccessorConstant(zone, receiver_map, holder,
737  {}, {});
738  }
739  DirectHandle<Object> maybe_accessors = get_accessors();
740  if (!IsAccessorPair(*maybe_accessors)) {
741  return PropertyAccessInfo::Invalid(zone);
742  }
743  DirectHandle<AccessorPair> accessors = Cast<AccessorPair>(maybe_accessors);
744  Handle<Object> accessor = broker->CanonicalPersistentHandle(
745  access_mode == AccessMode::kLoad ? accessors->getter(kAcquireLoad)
746  : accessors->setter(kAcquireLoad));
747 
748  OptionalObjectRef accessor_ref = TryMakeRef(broker, accessor);
749  if (!accessor_ref.has_value()) return PropertyAccessInfo::Invalid(zone);
750 
751  OptionalJSObjectRef api_holder_ref;
752  if (!IsJSFunction(*accessor)) {
753  CallOptimization optimization(broker->local_isolate_or_isolate(), accessor);
754  if (!optimization.is_simple_api_call() ||
755  optimization.IsCrossContextLazyAccessorPair(
756  *broker->target_native_context().object(), *holder_map.object())) {
757  return PropertyAccessInfo::Invalid(zone);
758  }
759  if (DEBUG_BOOL && holder.has_value()) {
760  std::optional<Tagged<NativeContext>> holder_creation_context =
761  holder->object()->GetCreationContext();
762  CHECK(holder_creation_context.has_value());
763  CHECK_EQ(*broker->target_native_context().object(),
764  holder_creation_context.value());
765  }
766 
767  CallOptimization::HolderLookup holder_lookup;
768  Handle<JSObject> api_holder = broker->CanonicalPersistentHandle(
769  optimization.LookupHolderOfExpectedType(
770  broker->local_isolate_or_isolate(), receiver_map.object(),
771  &holder_lookup));
772  if (holder_lookup == CallOptimization::kHolderNotFound) {
773  return PropertyAccessInfo::Invalid(zone);
774  }
775  DCHECK_IMPLIES(holder_lookup == CallOptimization::kHolderIsReceiver,
776  api_holder.is_null());
777  DCHECK_IMPLIES(holder_lookup == CallOptimization::kHolderFound,
778  !api_holder.is_null());
779 
780  if (!api_holder.is_null()) {
781  api_holder_ref = TryMakeRef(broker, api_holder);
782  if (!api_holder_ref.has_value()) return PropertyAccessInfo::Invalid(zone);
783  }
784  }
785  if (access_mode == AccessMode::kLoad) {
786  std::optional<Tagged<Name>> cached_property_name =
787  FunctionTemplateInfo::TryGetCachedPropertyName(isolate, *accessor);
788  if (cached_property_name.has_value()) {
789  OptionalNameRef cached_property_name_ref =
790  TryMakeRef(broker, cached_property_name.value());
791  if (cached_property_name_ref.has_value()) {
792  PropertyAccessInfo access_info = ai_factory->ComputePropertyAccessInfo(
793  holder_map, cached_property_name_ref.value(), access_mode);
794  if (!access_info.IsInvalid()) return access_info;
795  }
796  }
797  }
798 
799  if (holder_map.is_dictionary_map()) {
800  CHECK(!api_holder_ref.has_value());
801  return PropertyAccessInfo::DictionaryProtoAccessorConstant(
802  zone, receiver_map, holder, accessor_ref.value(), api_holder_ref, name);
803  } else {
804  return PropertyAccessInfo::FastAccessorConstant(
805  zone, receiver_map, holder, accessor_ref.value(), api_holder_ref);
806  }
807 }

857 bool AccessInfoFactory::TryLoadPropertyDetails(
858  MapRef map, OptionalJSObjectRef maybe_holder, NameRef name,
859  InternalIndex* index_out, PropertyDetails* details_out) const {
860  if (map.is_dictionary_map()) {
861  DCHECK(V8_DICT_PROPERTY_CONST_TRACKING_BOOL);
862  DCHECK(map.object()->is_prototype_map());
863 
864  DisallowGarbageCollection no_gc;
865 
866  if (!maybe_holder.has_value()) {
867  // TODO(v8:11457) In this situation, we have a dictionary mode prototype
868  // as a receiver. Consider other means of obtaining the holder in this
869  // situation.
870 
871  // Without the holder, we can't get the property details.
872  return false;
873  }
874 
875  DirectHandle<JSObject> holder = maybe_holder->object();
876  if (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
877  Tagged<SwissNameDictionary> dict = holder->property_dictionary_swiss();
878  *index_out = dict->FindEntry(isolate(), name.object());
879  if (index_out->is_found()) {
880  *details_out = dict->DetailsAt(*index_out);
881  }
882  } else {
883  Tagged<NameDictionary> dict = holder->property_dictionary();
884  *index_out = dict->FindEntry(isolate(), name.object());
885  if (index_out->is_found()) {
886  *details_out = dict->DetailsAt(*index_out);
887  }
888  }
889 #if V8_ENABLE_WEBASSEMBLY
890  } else if (InstanceTypeChecker::IsWasmObject(map.instance_type())) {
891  DCHECK(index_out->is_not_found());
892  return true; // Skip to prototypes.
893 #endif // V8_ENABLE_WEBASSEMBLY
894  } else {
895  Tagged<DescriptorArray> descriptors =
896  *map.instance_descriptors(broker()).object();
897  *index_out = descriptors->Search(*name.object(), *map.object(), true);
898  if (index_out->is_found()) {
899  *details_out = descriptors->GetDetails(*index_out);
900  }
901  }
902 
903  return true;
904 }

906 PropertyAccessInfo AccessInfoFactory::ComputePropertyAccessInfo(
907  MapRef map, NameRef name, AccessMode access_mode) const {
908  CHECK(name.IsUniqueName());
909 
910  // Dictionary property const tracking is unsupported with concurrent inlining.
911  CHECK(!V8_DICT_PROPERTY_CONST_TRACKING_BOOL);
912 
913  JSHeapBroker::MapUpdaterGuardIfNeeded mumd_scope(broker());
914 
915  if (access_mode == AccessMode::kHas && !IsJSReceiverMap(*map.object())) {
916  return Invalid();
917  }
918 
919  // Check if it is safe to inline property access for the {map}.
920  if (!CanInlinePropertyAccess(map, access_mode)) {
921  return Invalid();
922  }
923 
924  // We support fast inline cases for certain JSObject getters.
925  if (access_mode == AccessMode::kLoad || access_mode == AccessMode::kHas) {
926  PropertyAccessInfo access_info = LookupSpecialFieldAccessor(map, name);
927  if (!access_info.IsInvalid()) return access_info;
928  }
929 
930  // Only relevant if V8_DICT_PROPERTY_CONST_TRACKING enabled.
931  bool dictionary_prototype_on_chain = false;
932  bool fast_mode_prototype_on_chain = false;
933 
934  // Remember the receiver map. We use {map} as loop variable.
935  MapRef receiver_map = map;
936  OptionalJSObjectRef holder;
937 
938  // Perform the implicit ToObject for primitives here.
939  // Implemented according to ES6 section 7.3.2 GetV (V, P).
940  // Note: Keep sync'd with
941  // CompilationDependencies::DependOnStablePrototypeChains.
942  if (receiver_map.IsPrimitiveMap()) {
943  OptionalJSFunctionRef constructor =
944  broker()->target_native_context().GetConstructorFunction(broker(),
945  receiver_map);
946  if (!constructor.has_value()) return Invalid();
947  map = constructor->initial_map(broker());
948  DCHECK(!map.IsPrimitiveMap());
949  }
950 
951  while (true) {
952  PropertyDetails details = PropertyDetails::Empty();
953  InternalIndex index = InternalIndex::NotFound();
954  if (!TryLoadPropertyDetails(map, holder, name, &index, &details)) {
955  return Invalid();
956  }
957 
958  if (index.is_found()) {
959  if (IsAnyStore(access_mode)) {
960  DCHECK(!map.is_dictionary_map());
961 
962  // Don't bother optimizing stores to read-only properties.
963  if (details.IsReadOnly()) return Invalid();
964 
965  if (details.kind() == PropertyKind::kData && holder.has_value()) {
966  // This is a store to a property not found on the receiver but on a
967  // prototype. According to ES6 section 9.1.9 [[Set]], we need to
968  // create a new data property on the receiver. We can still optimize
969  // if such a transition already exists.
970  return LookupTransition(receiver_map, name, holder, NONE);
971  }
972  }
973 
974  if (IsDefiningStore(access_mode)) {
975  if (details.attributes() != PropertyAttributes::NONE ||
976  details.kind() != PropertyKind::kData) {
977  // We should store the property with WEC attributes, but that's not
978  // the attributes of the property that we found. We just bail out and
979  // let the runtime figure out what to do (which probably requires
980  // changing the object's map).
981  // Same for accessor case - we must reconfigure property to a data
982  // property.
983  return Invalid();
984  }
985  }
986 
987  if (map.is_dictionary_map()) {
988  DCHECK(V8_DICT_PROPERTY_CONST_TRACKING_BOOL);
989 
990  if (fast_mode_prototype_on_chain) {
991  // TODO(v8:11248) While the work on dictionary mode prototypes is in
992  // progress, we may still see fast mode objects on the chain prior to
993  // reaching a dictionary mode prototype holding the property . Due to
994  // this only being an intermediate state, we don't stupport these kind
995  // of heterogenous prototype chains.
996  return Invalid();
997  }
998 
999  // TryLoadPropertyDetails only succeeds if we know the holder.
1000  return ComputeDictionaryProtoAccessInfo(
1001  receiver_map, name, holder.value(), index, access_mode, details);
1002  }
1003 
1004  if (dictionary_prototype_on_chain) {
1005  // If V8_DICT_PROPERTY_CONST_TRACKING_BOOL was disabled, then a
1006  // dictionary prototype would have caused a bailout earlier.
1007  DCHECK(V8_DICT_PROPERTY_CONST_TRACKING_BOOL);
1008 
1009  // TODO(v8:11248) We have a fast mode holder, but there was a dictionary
1010  // mode prototype earlier on the chain. Note that seeing a fast mode
1011  // prototype even though V8_DICT_PROPERTY_CONST_TRACKING is enabled
1012  // should only be possible while the implementation of dictionary mode
1013  // prototypes is work in progress. Eventually, enabling
1014  // V8_DICT_PROPERTY_CONST_TRACKING will guarantee that all prototypes
1015  // are always in dictionary mode, making this case unreachable. However,
1016  // due to the complications of checking dictionary mode prototypes for
1017  // modification, we don't attempt to support dictionary mode prototypes
1018  // occuring before a fast mode holder on the chain.
1019  return Invalid();
1020  }
1021 
1022  if (access_mode == AccessMode::kLoad && holder.has_value()) {
1023  PropertyAccessInfo access_info = LookupSpecialFieldAccessorInHolder(
1024  receiver_map, name, *holder, details, index);
1025  if (!access_info.IsInvalid()) return access_info;
1026  }
1027 
1028  if (details.location() == PropertyLocation::kField) {
1029  if (details.kind() == PropertyKind::kData) {
1030  return ComputeDataFieldAccessInfo(receiver_map, map, name, holder,
1031  index, access_mode);
1032  } else {
1033  DCHECK_EQ(PropertyKind::kAccessor, details.kind());
1034  // TODO(turbofan): Add support for general accessors?
1035  return Invalid();
1036  }
1037  } else {
1038  DCHECK_EQ(PropertyLocation::kDescriptor, details.location());
1039  DCHECK_EQ(PropertyKind::kAccessor, details.kind());
1040  return ComputeAccessorDescriptorAccessInfo(receiver_map, name, map,
1041  holder, index, access_mode);
1042  }
1043 
1044  UNREACHABLE();
1045  }
1046 
1047  // The property wasn't found on {map}. Look on the prototype if appropriate.
1048  DCHECK(!index.is_found());
1049 
1050  // Don't search on the prototype chain for special indices in case of
1051  // integer indexed exotic objects (see ES6 section 9.4.5).
1052  if (IsJSTypedArrayMap(*map.object()) && name.IsString()) {
1053  StringRef name_str = name.AsString();
1054  SharedStringAccessGuardIfNeeded access_guard(
1055  *name_str.object(), broker()->local_isolate_or_isolate());
1056  if (IsSpecialIndex(*name_str.object(), access_guard)) return Invalid();
1057  }
1058 
1059  // Don't search on the prototype when storing in literals, or performing a
1060  // Define operation
1061  if (access_mode == AccessMode::kStoreInLiteral ||
1062  access_mode == AccessMode::kDefine) {
1063  PropertyAttributes attrs = NONE;
1064  if (name.object()->IsPrivate()) {
1065  // When PrivateNames are added to an object, they are by definition
1066  // non-enumerable.
1067  attrs = DONT_ENUM;
1068  }
1069  return LookupTransition(receiver_map, name, holder, attrs);
1070  }
1071 
1072  // Don't lookup private symbols on the prototype chain.
1073  if (name.object()->IsPrivate()) {
1074  return Invalid();
1075  }
1076 
1077  if (V8_DICT_PROPERTY_CONST_TRACKING_BOOL && holder.has_value()) {
1078  // At this point, we are past the first loop iteration.
1079  DCHECK(holder->object()->map()->is_prototype_map());
1080  DCHECK(!holder->map(broker()).equals(receiver_map));
1081 
1082  fast_mode_prototype_on_chain =
1083  fast_mode_prototype_on_chain || !map.is_dictionary_map();
1084  dictionary_prototype_on_chain =
1085  dictionary_prototype_on_chain || map.is_dictionary_map();
1086  }
1087 
1088  // Walk up the prototype chain.
1089  // Load the map's prototype's map to guarantee that every time we use it,
1090  // we use the same Map.
1091  HeapObjectRef prototype = map.prototype(broker());
1092 
1093  MapRef map_prototype_map = prototype.map(broker());
1094  if (!IsJSObjectMap(*map_prototype_map.object())) {
1095  // Don't allow proxies on the prototype chain.
1096  if (!prototype.IsNull()) {
1097  DCHECK(IsJSProxy(*prototype.object()) ||
1098  IsWasmObject(*prototype.object()));
1099  return Invalid();
1100  }
1101 
1102  DCHECK(prototype.IsNull());
1103 
1104  if (dictionary_prototype_on_chain) {
1105  // TODO(v8:11248) See earlier comment about
1106  // dictionary_prototype_on_chain. We don't support absent properties
1107  // with dictionary mode prototypes on the chain, either. This is again
1108  // just due to how we currently deal with dependencies for dictionary
1109  // properties during finalization.
1110  return Invalid();
1111  }
1112 
1113  // Store to property not found on the receiver or any prototype, we need
1114  // to transition to a new data property.
1115  // Implemented according to ES6 section 9.1.9 [[Set]] (P, V, Receiver)
1116  if (access_mode == AccessMode::kStore) {
1117  return LookupTransition(receiver_map, name, holder, NONE);
1118  }
1119 
1120  // The property was not found (access returns undefined or throws
1121  // depending on the language mode of the load operation.
1122  // Implemented according to ES6 section 9.1.8 [[Get]] (P, Receiver)
1123  return PropertyAccessInfo::NotFound(zone(), receiver_map, holder);
1124  }
1125 
1126  CHECK(prototype.IsJSObject());
1127  holder = prototype.AsJSObject();
1128  map = map_prototype_map;
1129 
1130  if (!CanInlinePropertyAccess(map, access_mode)) {
1131  return Invalid();
1132  }
1133 
1134  // Successful lookup on prototype chain needs to guarantee that all the
1135  // prototypes up to the holder have stable maps, except for dictionary-mode
1136  // prototypes. We currently do this by taking a
1137  // DependOnStablePrototypeChains dependency in the caller.
1138  //
1139  // TODO(jgruber): This is brittle and easy to miss. Consider a refactor
1140  // that moves the responsibility of taking the dependency into
1141  // AccessInfoFactory.
1142  }
1143  UNREACHABLE();
1144 }
