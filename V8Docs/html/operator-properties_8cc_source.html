36  case IrOpcode::kJSCloneObject:

284 // static
285 bool OperatorProperties::IsBasicBlockBegin(const Operator* op) {
286  Operator::Opcode const opcode = op->opcode();
287  return opcode == IrOpcode::kStart || opcode == IrOpcode::kEnd ||
288  opcode == IrOpcode::kDead || opcode == IrOpcode::kLoop ||
289  opcode == IrOpcode::kMerge || opcode == IrOpcode::kIfTrue ||
290  opcode == IrOpcode::kIfFalse || opcode == IrOpcode::kIfSuccess ||
291  opcode == IrOpcode::kIfException || opcode == IrOpcode::kIfValue ||
292  opcode == IrOpcode::kIfDefault;
293 }
