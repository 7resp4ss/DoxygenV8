234 std::unique_ptr<char[]> FunctionLiteral::GetDebugName() const {
235  const AstConsString* cons_string;
236  if (raw_name_ != nullptr && !raw_name_->IsEmpty()) {
237  cons_string = raw_name_;
238  } else if (raw_inferred_name_ != nullptr && !raw_inferred_name_->IsEmpty()) {
239  cons_string = raw_inferred_name_;
240  } else if (!shared_function_info_.is_null()) {
241  return shared_function_info_->inferred_name()->ToCString();
242  } else {
243  char* empty_str = new char[1];
244  empty_str[0] = 0;
245  return std::unique_ptr<char[]>(empty_str);

248  // TODO(rmcilroy): Deal with two-character strings.
249  std::vector<char> result_vec;

251  for (const AstRawString* string : strings) {
252  if (!string->is_one_byte()) break;

254  result_vec.push_back(string->raw_data()[i]);
255  }
256  }
257  std::unique_ptr<char[]> result(new char[result_vec.size() + 1]);
258  if (result_vec.size()) {
259  memcpy(result.get(), result_vec.data(), result_vec.size());
260  }
261  result[result_vec.size()] = '\0';
262  return result;

339 void ObjectLiteral::CalculateEmitStore(Zone* zone) {
340  const auto GETTER = ObjectLiteral::Property::GETTER;
341  const auto SETTER = ObjectLiteral::Property::SETTER;
342 
343  CustomMatcherZoneHashMap table(Literal::Match,
344  ZoneHashMap::kDefaultHashMapCapacity,
345  ZoneAllocationPolicy(zone));
346  for (int i = properties()->length() - 1; i >= 0; i--) {
347  ObjectLiteral::Property* property = properties()->at(i);
348  if (property->is_computed_name()) continue;
349  if (property->IsPrototype()) continue;
350  Literal* literal = property->key()->AsLiteral();
351  DCHECK(!literal->IsNullLiteral());
352 
353  uint32_t hash = literal->Hash();
354  ZoneHashMap::Entry* entry = table.LookupOrInsert(literal, hash);
355  if (entry->value == nullptr) {
356  entry->value = property;
357  } else {
358  // We already have a later definition of this property, so we don't need
359  // to emit a store for the current one.
360  //
361  // There are two subtleties here.
362  //
363  // (1) Emitting a store might actually be incorrect. For example, in {get
364  // foo() {}, foo: 42}, the getter store would override the data property
365  // (which, being a non-computed compile-time valued property, is already
366  // part of the initial literal object.
367  //
368  // (2) If the later definition is an accessor (say, a getter), and the
369  // current definition is a complementary accessor (here, a setter), then
370  // we still must emit a store for the current definition.
371 
372  auto later_kind =
373  static_cast<ObjectLiteral::Property*>(entry->value)->kind();
374  bool complementary_accessors =
375  (property->kind() == GETTER && later_kind == SETTER) ||
376  (property->kind() == SETTER && later_kind == GETTER);
377  if (!complementary_accessors) {
378  property->set_emit_store(false);
379  if (later_kind == GETTER || later_kind == SETTER) {
380  entry->value = property;
381  }
382  }
383  }
384  }
385 }

411 void ObjectLiteralBoilerplateBuilder::InitDepthAndFlags() {
412  if (is_initialized()) return;
413  bool is_simple = true;
414  bool has_seen_prototype = false;
415  bool needs_initial_allocation_site = false;
416  DepthKind depth_acc = kShallow;
417  uint32_t nof_properties = 0;
418  uint32_t elements = 0;
419  uint32_t max_element_index = 0;
420  for (int i = 0; i < properties()->length(); i++) {
421  ObjectLiteral::Property* property = properties()->at(i);
422  if (property->IsPrototype()) {
423  has_seen_prototype = true;
424  // __proto__:null has no side-effects and is set directly on the
425  // boilerplate.
426  if (property->IsNullPrototype()) {
427  set_has_null_protoype(true);
428  continue;
429  }
430  DCHECK(!has_null_prototype());
431  is_simple = false;
432  continue;
433  }
434  if (nof_properties == boilerplate_properties_) {
435  DCHECK(property->is_computed_name());
436  is_simple = false;
437  if (!has_seen_prototype) InitFlagsForPendingNullPrototype(i);
438  break;
439  }
440  DCHECK(!property->is_computed_name());
441 
442  MaterializedLiteral* literal = property->value()->AsMaterializedLiteral();
443  if (literal != nullptr) {
444  LiteralBoilerplateBuilder::InitDepthAndFlags(literal);
445  depth_acc = kNotShallow;
446  needs_initial_allocation_site |= literal->NeedsInitialAllocationSite();
447  }
448 
449  Literal* key = property->key()->AsLiteral();
450  Expression* value = property->value();
451 
452  bool is_compile_time_value = value->IsCompileTimeValue();
453  is_simple = is_simple && is_compile_time_value;
454 
455  // Keep track of the number of elements in the object literal and
456  // the largest element index. If the largest element index is
457  // much larger than the number of elements, creating an object
458  // literal with fast elements will be a waste of space.
459  uint32_t element_index = 0;
460  if (key->AsArrayIndex(&element_index)) {
461  max_element_index = std::max(element_index, max_element_index);
462  elements++;
463  } else {
464  DCHECK(key->IsPropertyName());
465  }
466 
467  nof_properties++;
468  }
469 
470  set_depth(depth_acc);
471  set_is_simple(is_simple);
472  set_needs_initial_allocation_site(needs_initial_allocation_site);
473  set_has_elements(elements > 0);
474  set_fast_elements((max_element_index <= 32) ||
475  ((2 * elements) >= max_element_index));
476 }

478 template <typename IsolateT>
479 void ObjectLiteralBoilerplateBuilder::BuildBoilerplateDescription(
480  IsolateT* isolate) {
481  if (!boilerplate_description_.is_null()) return;
482 
483  int index_keys = 0;
484  bool has_seen_proto = false;
485  for (int i = 0; i < properties()->length(); i++) {
486  ObjectLiteral::Property* property = properties()->at(i);
487  if (property->IsPrototype()) {
488  has_seen_proto = true;
489  continue;
490  }
491  if (property->is_computed_name()) continue;
492 
493  Literal* key = property->key()->AsLiteral();
494  if (!key->IsPropertyName()) index_keys++;
495  }
496 
497  Handle<ObjectBoilerplateDescription> boilerplate_description =
498  isolate->factory()->NewObjectBoilerplateDescription(
499  boilerplate_properties_, properties()->length(), index_keys,
500  has_seen_proto);
501 
502  int position = 0;
503  for (int i = 0; i < properties()->length(); i++) {
504  ObjectLiteral::Property* property = properties()->at(i);
505  if (property->IsPrototype()) continue;
506 
507  if (static_cast<uint32_t>(position) == boilerplate_properties_) {
508  DCHECK(property->is_computed_name());
509  break;
510  }
511  DCHECK(!property->is_computed_name());
512 
513  MaterializedLiteral* m_literal = property->value()->AsMaterializedLiteral();
514  if (m_literal != nullptr) {
515  BuildConstants(isolate, m_literal);
516  }
517 
518  // Add CONSTANT and COMPUTED properties to boilerplate. Use the
519  // 'uninitialized' Oddball for COMPUTED properties, the real value is filled
520  // in at runtime. The enumeration order is maintained.
521  Literal* key_literal = property->key()->AsLiteral();
522  uint32_t element_index = 0;
523  DirectHandle<Object> key =
524  key_literal->AsArrayIndex(&element_index)
525  ? isolate->factory()
526  ->template NewNumberFromUint<AllocationType::kOld>(
527  element_index)
528  : Cast<Object>(key_literal->AsRawPropertyName()->string());
529  DirectHandle<Object> value =
530  GetBoilerplateValue(property->value(), isolate);
531  boilerplate_description->set_key_value(position++, *key, *value);
532  }
533 
534  boilerplate_description->set_flags(EncodeLiteralType());
535 
536  boilerplate_description_ = boilerplate_description;
537 }

582 void ArrayLiteralBoilerplateBuilder::InitDepthAndFlags() {
583  if (is_initialized()) return;
584 
585  int constants_length =
586  first_spread_index_ >= 0 ? first_spread_index_ : values_->length();
587 
588  // Fill in the literals.
589  bool is_simple = first_spread_index_ < 0;
590  bool is_holey = false;
591  ElementsKind kind = FIRST_FAST_ELEMENTS_KIND;
592  DepthKind depth_acc = kShallow;
593  int array_index = 0;
594  for (; array_index < constants_length; array_index++) {
595  Expression* element = values_->at(array_index);
596  MaterializedLiteral* materialized_literal =
597  element->AsMaterializedLiteral();
598  if (materialized_literal != nullptr) {
599  LiteralBoilerplateBuilder::InitDepthAndFlags(materialized_literal);
600  depth_acc = kNotShallow;
601  }
602 
603  if (!element->IsCompileTimeValue()) {
604  is_simple = false;
605 
606  // Don't change kind here: non-compile time values resolve to an unknown
607  // elements kind, so we allow them to be considered as any one of them.
608 
609  // TODO(leszeks): It would be nice to DCHECK here that GetBoilerplateValue
610  // will return IsUninitialized, but that would require being on the main
611  // thread which we may not be.
612  } else {
613  Literal* literal = element->AsLiteral();
614 
615  if (!literal) {
616  // Only arrays and objects are compile-time values but not (primitive)
617  // literals.
618  DCHECK(element->IsObjectLiteral() || element->IsArrayLiteral());
619  kind = PACKED_ELEMENTS;
620  } else {
621  switch (literal->type()) {
622  case Literal::kTheHole:
623  is_holey = true;
624  // The hole is allowed in holey double arrays (and holey Smi
625  // arrays), so ignore it as far as is_all_number is concerned.
626  break;
627  case Literal::kHeapNumber:
628  if (kind == PACKED_SMI_ELEMENTS) kind = PACKED_DOUBLE_ELEMENTS;
629  DCHECK_EQ(kind,
630  GetMoreGeneralElementsKind(kind, PACKED_DOUBLE_ELEMENTS));
631  break;
632  case Literal::kSmi:
633  DCHECK_EQ(kind,
634  GetMoreGeneralElementsKind(kind, PACKED_SMI_ELEMENTS));
635  break;
636  case Literal::kBigInt:
637  case Literal::kString:
638  case Literal::kConsString:
639  case Literal::kBoolean:
640  case Literal::kUndefined:
641  case Literal::kNull:
642  kind = PACKED_ELEMENTS;
643  break;
644  }
645  }
646  }
647  }
648 
649  if (is_holey) {
650  kind = GetHoleyElementsKind(kind);
651  }
652 
653  set_depth(depth_acc);
654  set_is_simple(is_simple);
655  set_boilerplate_descriptor_kind(kind);
656 
657  // Array literals always need an initial allocation site to properly track
658  // elements transitions.
659  set_needs_initial_allocation_site(true);

713  // We shouldn't allocate after creating the boilerplate value.
714  DisallowGarbageCollection no_gc;
715 
716  if (IsTheHole(boilerplate_value, isolate)) {
717  DCHECK(IsHoleyElementsKind(kind));
