25 // ScopedList is a scope-lifetime list with a std::vector backing that can be
26 // reused between ScopedLists. Note that a ScopedList in an outer scope cannot
27 // add any entries if there is a ScopedList with the same backing in an inner
28 // scope.
29 template <typename T, typename TBacking = T>
30 class V8_NODISCARD ScopedList final {
31  // The backing can either be the same type as the list type, or, for pointers,
32  // we additionally allow a void* backing store.
33  static_assert((std::is_same_v<TBacking, T>) ||
34  (std::is_same_v<TBacking, void*> && std::is_pointer_v<T>),
35  "Incompatible combination of T and TBacking types");
36 
37  public:
38  explicit ScopedList(std::vector<TBacking>* buffer)
39  : buffer_(*buffer), start_(buffer->size()), end_(buffer->size()) {}
40 
41  ~ScopedList() { Rewind(); }
42 
43  void Rewind() {
44  DCHECK_EQ(buffer_.size(), end_);
45  buffer_.resize(start_);
46  end_ = start_;
47  }
48 
49  void MergeInto(ScopedList* parent) {
50  DCHECK_EQ(parent->end_, start_);
51  parent->end_ = end_;
52  start_ = end_;
53  DCHECK_EQ(0, length());
54  }
55 
56  int length() const { return static_cast<int>(end_ - start_); }
57 
58  const T& at(int i) const {
59  size_t index = start_ + i;
60  DCHECK_LE(start_, index);
61  DCHECK_LT(index, buffer_.size());
62  return *reinterpret_cast<T*>(&buffer_[index]);
63  }
64 
65  T& at(int i) {
66  size_t index = start_ + i;
67  DCHECK_LE(start_, index);
68  DCHECK_LT(index, buffer_.size());
69  return *reinterpret_cast<T*>(&buffer_[index]);
70  }

72  base::Vector<const T> ToConstVector() const {
73  T* data = reinterpret_cast<T*>(buffer_.data() + start_);
74  return base::Vector<const T>(data, length());
75  }
76 
77  void Add(const T& value) {
78  DCHECK_EQ(buffer_.size(), end_);
79  buffer_.push_back(value);
80  ++end_;
81  }

83  void AddAll(base::Vector<const T> list) {
84  DCHECK_EQ(buffer_.size(), end_);
85  buffer_.reserve(buffer_.size() + list.length());
86  for (int i = 0; i < list.length(); i++) {
87  buffer_.push_back(list.at(i));
88  }
89  end_ += list.length();
90  }

97  }

103  inline const_iterator end() const {
104  return reinterpret_cast<T*>(buffer_.data() + end_);
105  }

110  size_t end_;
111 };
