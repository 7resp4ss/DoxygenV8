533 // An ExternalReference represents a C++ address used in the generated
534 // code. All references to C++ functions and variables must be encapsulated
535 // in an ExternalReference instance. This is done in order to track the
536 // origin of all external references in the code so that they can be bound
537 // to the correct addresses when deserializing a heap.

541  enum Type {
542  // Builtin call.
543  // Address f(v8::internal::Arguments).
544  BUILTIN_CALL, // default

547  // ObjectPair f(v8::internal::Arguments).
548  BUILTIN_CALL_PAIR,
549 
550  // TODO(mslekova): Once FAST_C_CALL is supported in the simulator,

553 
554  // Builtin that takes float arguments and returns an int.
555  // int f(double, double).
556  BUILTIN_COMPARE_CALL,

558  // Builtin call that returns floating point.
559  // double f(double, double).
560  BUILTIN_FP_FP_CALL,
561 
562  // Builtin call that returns floating point.
563  // double f(double).
564  BUILTIN_FP_CALL,
565 
566  // Builtin call that returns floating point.
567  // double f(double, int).

569

575  // int f(double).
