38 // A DescriptorArray is a custom array that holds instance descriptors.
39 // It has the following layout:
40 // Header:
41 // [16:0 bits]: number_of_all_descriptors (including slack)
42 // [32:16 bits]: number_of_descriptors
43 // [64:32 bits]: raw_gc_state (used by GC)
44 // [kEnumCacheOffset]: enum cache
45 // Elements:
46 // [kHeaderSize + 0]: first key (and internalized String)
47 // [kHeaderSize + 1]: first descriptor details (see PropertyDetails)
48 // [kHeaderSize + 2]: first value for constants / Tagged<Smi>(1) when not
49 // used
50 // Slack:
51 // [kHeaderSize + number of descriptors * 3]: start of slack
52 // The "value" fields store either values or field types. A field type is either
53 // FieldType::None(), FieldType::Any() or a weak reference to a Map. All other
54 // references are strong.

278 // Custom DescriptorArray marking state for visitors that are allowed to write
279 // into the heap. The marking state uses DescriptorArray::raw_gc_state() as
280 // storage.
281 //
282 // The state essentially keeps track of 3 fields:
283 // 1. The collector epoch: The rest of the state is only valid if the epoch
284 // matches. If the epoch doesn't match, the other fields should be considered
285 // invalid. The epoch is necessary, as not all DescriptorArray objects are
286 // eventually trimmed in the atomic pause and thus available for resetting
287 // the state.
288 // 2. Number of already marked descriptors.
289 // 3. Delta of to be marked descriptors in this cycle. This must be 0 after
290 // marking is done.
291 class DescriptorArrayMarkingState final {
292  public:
293 #define BIT_FIELD_FIELDS(V, _) \
294  V(Epoch, unsigned, 2, _) \
295  V(Marked, uint16_t, 14, _) \
296  V(Delta, uint16_t, 16, _)
297  DEFINE_BIT_FIELDS(BIT_FIELD_FIELDS)
298 #undef BIT_FIELD_FIELDS
299  static_assert(Marked::kMax <= Delta::kMax);
300  static_assert(kMaxNumberOfDescriptors <= Marked::kMax);
301 
302  using DescriptorIndex = uint16_t;
303  using RawGCStateType = uint32_t;
304 
305  static constexpr RawGCStateType kInitialGCState = 0;
306 
307  static constexpr RawGCStateType GetFullyMarkedState(
308  unsigned epoch, DescriptorIndex number_of_descriptors) {
309  return NewState(epoch & Epoch::kMask, number_of_descriptors, 0);
310  }
311 
312  // Potentially updates the delta of to be marked descriptors. Returns true if
313  // the update was successful and the object should be processed via a marking
314  // visitor.
315  //
316  // The call issues and Acq/Rel barrier to allow synchronizing other state
317  // (e.g. value of descriptor slots) with it.
318  static inline bool TryUpdateIndicesToMark(unsigned gc_epoch,
319  Tagged<DescriptorArray> array,
320  DescriptorIndex index_to_mark);
321 
322  // Used from the visitor when processing a DescriptorArray. Returns a range of
323  // start and end descriptor indices. No processing is required for start ==
324  // end. The method signals the first invocation by returning start == 0, and
325  // end != 0.
326  static inline std::pair<DescriptorIndex, DescriptorIndex>
327  AcquireDescriptorRangeToMark(unsigned gc_epoch,
328  Tagged<DescriptorArray> array);
329 
330  private:
331  static constexpr RawGCStateType NewState(unsigned masked_epoch,
332  DescriptorIndex marked,
333  DescriptorIndex delta) {
334  return Epoch::encode(masked_epoch) | Marked::encode(marked) |
335  Delta::encode(delta);
336  }
337 
338  static bool SwapState(Tagged<DescriptorArray> array, RawGCStateType old_state,
339  RawGCStateType new_state) {
340  return static_cast<RawGCStateType>(base::AcquireRelease_CompareAndSwap(
341  reinterpret_cast<base::Atomic32*>(
342  FIELD_ADDR(array, DescriptorArray::kRawGcStateOffset)),
343  old_state, new_state)) == old_state;
344  }
345 };
