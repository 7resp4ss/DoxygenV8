156 // All heap objects have a Map that describes their structure.
157 // A Map contains information about:
158 // - Size information about the object
159 // - How to iterate over an object (for garbage collection)
160 //
161 // Map layout:
162 // +---------------+-------------------------------------------------+
163 // | _ Type _ | _ Description _ |
164 // +---------------+-------------------------------------------------+
165 // | TaggedPointer | map - Always a pointer to the MetaMap root |
166 // +---------------+-------------------------------------------------+
167 // | Int | The first int field |
168 // `---+----------+-------------------------------------------------+
169 // | Byte | [instance_size] |
170 // +----------+-------------------------------------------------+
171 // | Byte | If Map for a primitive type: |
172 // | | native context index for constructor fn |
173 // | | If Map for an Object type: |
174 // | | inobject properties start offset in words |
175 // +----------+-------------------------------------------------+
176 // | Byte | [used_or_unused_instance_size_in_words] |
177 // | | For JSObject in fast mode this byte encodes |
178 // | | the size of the object that includes only |
179 // | | the used property fields or the slack size |
180 // | | in properties backing store. |
181 // +----------+-------------------------------------------------+
182 // | Byte | [visitor_id] |
183 // +----+----------+-------------------------------------------------+
184 // | Int | The second int field |
185 // `---+----------+-------------------------------------------------+
186 // | Short | [instance_type] |
187 // +----------+-------------------------------------------------+
188 // | Byte | [bit_field] |
189 // | | - has_non_instance_prototype (bit 0) |
190 // | | - is_callable (bit 1) |
191 // | | - has_named_interceptor (bit 2) |
192 // | | - has_indexed_interceptor (bit 3) |
193 // | | - is_undetectable (bit 4) |
194 // | | - is_access_check_needed (bit 5) |
195 // | | - is_constructor (bit 6) |
196 // | | - has_prototype_slot (bit 7) |
197 // +----------+-------------------------------------------------+
198 // | Byte | [bit_field2] |
199 // | | - new_target_is_base (bit 0) |
200 // | | - is_immutable_proto (bit 1) |
201 // | | - elements_kind (bits 2..7) |
202 // +----+----------+-------------------------------------------------+
203 // | Int | [bit_field3] |
204 // | | - enum_length (bit 0..9) |
205 // | | - number_of_own_descriptors (bit 10..19) |
206 // | | - is_prototype_map (bit 20) |
207 // | | - is_dictionary_map (bit 21) |
208 // | | - owns_descriptors (bit 22) |
209 // | | - is_in_retained_map_list (bit 23) |
210 // | | - is_deprecated (bit 24) |
211 // | | - is_unstable (bit 25) |
212 // | | - is_migration_target (bit 26) |
213 // | | - is_extensible (bit 28) |
214 // | | - may_have_interesting_properties (bit 28) |
215 // | | - construction_counter (bit 29..31) |
216 // | | |
217 // +*****************************************************************+
218 // | Int | On systems with 64bit pointer types, there |
219 // | | is an unused 32bits after bit_field3 |
220 // +*****************************************************************+
221 // | TaggedPointer | [prototype] |
222 // +---------------+-------------------------------------------------+
223 // | TaggedPointer | [constructor_or_back_pointer_or_native_context] |
224 // +---------------+-------------------------------------------------+
225 // | TaggedPointer | [instance_descriptors] (if JS object) |
226 // | | [custom_descriptor] (if WasmStruct) |
227 // +---------------+-------------------------------------------------+
228 // | TaggedPointer | [dependent_code] |
229 // +---------------+-------------------------------------------------+
230 // | TaggedPointer | [prototype_validity_cell] |
231 // +---------------+-------------------------------------------------+
232 // | TaggedPointer | If Map is a prototype map: |
233 // | | [prototype_info] |
234 // | | Else: |
235 // | | [raw_transitions] |
236 // +---------------+-------------------------------------------------+

365  // Inobject slack tracking is the way to reclaim unused inobject space.
366  //
367  // The instance size is initially determined by adding some slack to
368  // expected_nof_properties (to allow for a few extra properties added
369  // after the constructor). There is no guarantee that the extra space
370  // will not be wasted.
371  //
372  // Here is the algorithm to reclaim the unused inobject space:
373  // - Detect the first constructor call for this JSFunction.
374  // When it happens enter the "in progress" state: initialize construction
375  // counter in the initial_map.
376  // - While the tracking is in progress initialize unused properties of a new
377  // object with one_pointer_filler_map instead of undefined_value (the "used"
378  // part is initialized with undefined_value as usual). This way they can
379  // be resized quickly and safely.
380  // - Once enough objects have been created compute the 'slack'
381  // (traverse the map transition tree starting from the
382  // initial_map and find the lowest value of unused_property_fields).
383  // - Traverse the transition tree again and decrease the instance size
384  // of every map. Existing objects will resize automatically (they are
385  // filled with one_pointer_filler_map). All further allocations will
386  // use the adjusted instance size.
387  // - SharedFunctionInfo's expected_nof_properties left unmodified since
388  // allocations made using different closures could actually create different
389  // kind of objects (see prototype inheritance pattern).
390  //
391  // Important: inobject slack tracking is not attempted during the snapshot
392  // creation.
393 
394  static const int kGenerousAllocationCount =
395  kSlackTrackingCounterStart - kSlackTrackingCounterEnd + 1;
