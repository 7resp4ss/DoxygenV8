44 // A helper template to get string from option into a enum.
45 // The enum in the enum_values is the corresponding value to the strings
46 // in the str_values. If the option does not contains name,
47 // default_value will be return.
48 template <typename T>
49 V8_WARN_UNUSED_RESULT static Maybe<T> GetStringOption(
50  Isolate* isolate, DirectHandle<JSReceiver> options, const char* name,
51  const char* method_name, const std::span<const std::string_view> str_values,
52  const std::span<const T> enum_values, T default_value) {
53  DCHECK_EQ(str_values.size(), enum_values.size());
54  DirectHandle<String> found_string;
55  Maybe<bool> found = GetStringOption(isolate, options, name, str_values,
56  method_name, &found_string);
57  MAYBE_RETURN(found, Nothing<T>());
58  if (found.FromJust()) {
59  for (size_t i = 0; i < str_values.size(); i++) {
60  if (found_string->IsEqualTo(str_values[i], isolate)) {
61  return Just(enum_values[i]);
62  }
63  }
64  UNREACHABLE();
65  }
66  return Just(default_value);
67 }

69 // A helper template to get string from option into a enum.
70 // The enum in the enum_values is the corresponding value to the strings
71 // in the str_values. If the option does not contains name,
72 // default_value will be return.
73 template <typename T>
74 V8_WARN_UNUSED_RESULT static Maybe<T> GetStringOrBooleanOption(
75  Isolate* isolate, DirectHandle<JSReceiver> options, const char* property,
76  const char* method, const std::span<const std::string_view> str_values,
77  const std::span<const T> enum_values, T true_value, T false_value,
78  T fallback_value) {
79  DCHECK_EQ(str_values.size(), enum_values.size());
80  Factory* factory = isolate->factory();
81  DirectHandle<String> property_str =
82  factory->NewStringFromAsciiChecked(property);
83 
84  // 1. Let value be ? Get(options, property).
85  DirectHandle<Object> value;
86  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
87  isolate, value,
88  Object::GetPropertyOrElement(isolate, options, property_str),
89  Nothing<T>());
90  // 2. If value is undefined, then return fallback.
91  if (IsUndefined(*value, isolate)) {
92  return Just(fallback_value);
93  }
94  // 3. If value is true, then return trueValue.
95  if (IsTrue(*value, isolate)) {
96  return Just(true_value);
97  }
98  // 4. Let valueBoolean be ToBoolean(value).
99  bool valueBoolean = Object::BooleanValue(*value, isolate);
100  // 5. If valueBoolean is false, then return valueBoolean.
101  if (!valueBoolean) {
102  return Just(false_value);
103  }
104 
105  DirectHandle<String> value_str;
106  // 6. Let value be ? ToString(value).
107  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
108  isolate, value_str, Object::ToString(isolate, value), Nothing<T>());
109  // 7. If value is *"true"* or *"false"*, return _fallback_.
110  if (String::Equals(isolate, value_str, factory->true_string()) ||
111  String::Equals(isolate, value_str, factory->false_string())) {
112  return Just(fallback_value);
113  }
114  // 8. If values does not contain an element equal to _value_, throw a
115  // *RangeError* exception.
116  // 9. Return value.
117  for (size_t i = 0; i < str_values.size(); i++) {
118  if (value_str->IsEqualTo(str_values[i], isolate)) {
119  return Just(enum_values[i]);
120  }
121  }
122 
123  THROW_NEW_ERROR_RETURN_VALUE(
124  isolate,
125  NewRangeError(MessageTemplate::kValueOutOfRange, value,
126  factory->NewStringFromAsciiChecked(method), property_str),
127  Nothing<T>());
128 }
