19 // The jump table info is a part of code metadata, used by the disassembler. The
20 // layout is:
21 //
22 // byte count content
23 // ----------------------------------------------------------------
24 // [Inline array of JumpTableInfoEntry in increasing pc_offset order]
25 // ┌ 4 pc_offset of entry as uint32_t
26 // └ 4 target of entry as int32_t
27 
28 struct JumpTableInfoEntry {
29  constexpr JumpTableInfoEntry(uint32_t pc_offset, int32_t target)
30  : pc_offset(pc_offset), target(target) {}
31  uint32_t pc_offset;
32  int32_t target;
33 
34  static constexpr int kPCOffsetSize = kUInt32Size;
35  static constexpr int kTargetSize = kInt32Size;
36  static constexpr int kSize = kPCOffsetSize + kTargetSize;
37 };
38 static_assert(sizeof(JumpTableInfoEntry) == JumpTableInfoEntry::kSize);
