62 double Type::Min() const {
63  DCHECK(this->Is(Number()));
64  DCHECK(!this->Is(NaN()));
65  if (this->IsBitset()) return BitsetType::Min(this->AsBitset());
66  if (this->IsUnion()) {
67  double min = +V8_INFINITY;
68  for (int i = 1, n = AsUnion()->Length(); i < n; ++i) {
69  min = std::min(min, AsUnion()->Get(i).Min());
70  }
71  Type bitset = AsUnion()->Get(0);
72  if (!bitset.Is(NaN())) min = std::min(min, bitset.Min());
73  return min;
74  }
75  if (this->IsRange()) return this->AsRange()->Min();
76  DCHECK(this->IsOtherNumberConstant());
77  return this->AsOtherNumberConstant()->Value();
78 }

80 double Type::Max() const {
81  DCHECK(this->Is(Number()));
82  DCHECK(!this->Is(NaN()));
83  if (this->IsBitset()) return BitsetType::Max(this->AsBitset());
84  if (this->IsUnion()) {
85  double max = -V8_INFINITY;
86  for (int i = 1, n = this->AsUnion()->Length(); i < n; ++i) {
87  max = std::max(max, this->AsUnion()->Get(i).Max());
88  }
89  Type bitset = this->AsUnion()->Get(0);
90  if (!bitset.Is(NaN())) max = std::max(max, bitset.Max());
91  return max;
92  }
93  if (this->IsRange()) return this->AsRange()->Max();
94  DCHECK(this->IsOtherNumberConstant());
95  return this->AsOtherNumberConstant()->Value();
96 }

101 // The largest bitset subsumed by this type.
102 Type::bitset Type::BitsetGlb() const {
103  DisallowGarbageCollection no_gc;
104  // Fast case.
105  if (IsBitset()) {
106  return AsBitset();
107  } else if (IsUnion()) {
108  SLOW_DCHECK(AsUnion()->Wellformed());
109  return AsUnion()->Get(0).BitsetGlb() |
110  AsUnion()->Get(1).BitsetGlb(); // Shortcut.
111  } else if (IsRange()) {
112  bitset glb = BitsetType::Glb(AsRange()->Min(), AsRange()->Max());
113  return glb;
114  } else {
115  return BitsetType::kNone;
116  }
117 }

119 // The smallest bitset subsuming this type, possibly not a proper one.
120 Type::bitset Type::BitsetLub() const {
121  DisallowGarbageCollection no_gc;
122  if (IsBitset()) return AsBitset();
123  if (IsUnion()) {
124  // Take the representation from the first element, which is always
125  // a bitset.
126  bitset lub = AsUnion()->Get(0).BitsetLub();
127  for (int i = 0, n = AsUnion()->Length(); i < n; ++i) {
128  // Other elements only contribute their semantic part.
129  lub |= AsUnion()->Get(i).BitsetLub();
130  }
131  return lub;
132  }
133  if (IsHeapConstant()) return AsHeapConstant()->Lub();
134  if (IsOtherNumberConstant()) {
135  return AsOtherNumberConstant()->Lub();
136  }
137  if (IsRange()) return AsRange()->Lub();
138  if (IsTuple()) return BitsetType::kOtherInternal;
139 #if V8_ENABLE_WEBASSEMBLY
140  if (IsWasm()) return static_cast<const WasmType*>(ToTypeBase())->Lub();
141 #endif
142  UNREACHABLE();
143 }

461 Type::bitset BitsetType::ExpandInternals(Type::bitset bits) {
462  DCHECK_IMPLIES(bits & kOtherString, (bits & kString) == kString);
463  DisallowGarbageCollection no_gc;
464  if (!(bits & kPlainNumber)) return bits; // Shortcut.
465  const Boundary* boundaries = Boundaries();
466  for (size_t i = 0; i < BoundariesSize(); ++i) {
467  DCHECK(BitsetType::Is(boundaries[i].internal, boundaries[i].external));

473 Type::bitset BitsetType::Lub(double min, double max) {
474  DisallowGarbageCollection no_gc;
475  bitset lub = kNone;
476  const Boundary* mins = Boundaries();
477 
478  for (size_t i = 1; i < BoundariesSize(); ++i) {
479  if (min < mins[i].min) {
480  lub |= mins[i - 1].internal;
481  if (max < mins[i].min) return lub;
482  }
483  }
484  return lub | mins[BoundariesSize() - 1].internal;

489 Type::bitset BitsetType::Glb(double min, double max) {
490  DisallowGarbageCollection no_gc;
491  bitset glb = kNone;
492  const Boundary* mins = Boundaries();
493 
494  // If the range does not touch 0, the bound is empty.
495  if (max < -1 || min > 0) return glb;
496 
497  for (size_t i = 1; i + 1 < BoundariesSize(); ++i) {
498  if (min <= mins[i].min) {
499  if (max + 1 < mins[i + 1].min) break;
500  glb |= mins[i].external;
501  }
502  }
503  // OtherNumber also contains float numbers, so it can never be
504  // in the greatest lower bound.
505  return glb & ~(kOtherNumber);
506 }

508 double BitsetType::Min(bitset bits) {
509  DisallowGarbageCollection no_gc;
510  DCHECK(Is(bits, kNumber));
511  DCHECK(!Is(bits, kNaN));
512  const Boundary* mins = Boundaries();
513  bool mz = bits & kMinusZero;
514  for (size_t i = 0; i < BoundariesSize(); ++i) {
515  if (Is(mins[i].internal, bits)) {
516  return mz ? std::min(0.0, mins[i].min) : mins[i].min;
517  }
518  }
519  DCHECK(mz);
520  return 0;
521 }

523 double BitsetType::Max(bitset bits) {
524  DisallowGarbageCollection no_gc;
525  DCHECK(Is(bits, kNumber));
526  DCHECK(!Is(bits, kNaN));
527  const Boundary* mins = Boundaries();
528  bool mz = bits & kMinusZero;
529  if (BitsetType::Is(mins[BoundariesSize() - 1].internal, bits)) {
530  return +V8_INFINITY;
531  }
532  for (size_t i = BoundariesSize() - 1; i-- > 0;) {
533  if (Is(mins[i].internal, bits)) {
534  return mz ? std::max(0.0, mins[i + 1].min - 1) : mins[i + 1].min - 1;
535  }
536  }
537  DCHECK(mz);
538  return 0;
539 }

559 bool Type::SimplyEquals(Type that) const {
560  DisallowGarbageCollection no_gc;
561  if (this->IsHeapConstant()) {
562  return that.IsHeapConstant() &&
563  this->AsHeapConstant()->Value().address() ==
564  that.AsHeapConstant()->Value().address();
565  }
566  if (this->IsOtherNumberConstant()) {
567  return that.IsOtherNumberConstant() &&
568  this->AsOtherNumberConstant()->Value() ==
569  that.AsOtherNumberConstant()->Value();
570  }
571  if (this->IsRange()) {
572  if (that.IsHeapConstant() || that.IsOtherNumberConstant()) return false;
573  }
574  if (this->IsTuple()) {
575  if (!that.IsTuple()) return false;
576  const TupleType* this_tuple = this->AsTuple();
577  const TupleType* that_tuple = that.AsTuple();
578  if (this_tuple->Arity() != that_tuple->Arity()) {
579  return false;
580  }
581  for (int i = 0, n = this_tuple->Arity(); i < n; ++i) {

589 // Check if [this] <= [that].
590 bool Type::SlowIs(Type that) const {
591  DisallowGarbageCollection no_gc;
592 
593  // Fast bitset cases
594  if (that.IsBitset()) {
595  return BitsetType::Is(this->BitsetLub(), that.AsBitset());
596  }
597 
598  if (this->IsBitset()) {
599  return BitsetType::Is(this->AsBitset(), that.BitsetGlb());
600  }

606  }
607  return true;
608  }

611  if (that.IsUnion()) {
612  for (int i = 0, n = that.AsUnion()->Length(); i < n; ++i) {
613  if (this->Is(that.AsUnion()->Get(i))) return true;
614  if (i > 1 && this->IsRange()) return false; // Shortcut.
615  }
616  return false;
617  }
618 
619  if (that.IsRange()) {
620  return this->IsRange() && Contains(that.AsRange(), this->AsRange());
621  }
622  if (this->IsRange()) return false;

624 #ifdef V8_ENABLE_WEBASSEMBLY
625  if (this->IsWasm()) {

629  return wasm::IsSubtypeOf(this_type.type, that_type.type, this_type.module,
630  that_type.module);
631  }
632 #endif
633

637 // Check if [this] and [that] overlap.
638 bool Type::Maybe(Type that) const {
639  DisallowGarbageCollection no_gc;
640 
641  if (BitsetType::IsNone(this->BitsetLub() & that.BitsetLub())) return false;
642 
643  // (T1 \/ ... \/ Tn) overlaps T if (T1 overlaps T) \/ ... \/ (Tn overlaps T)
644  if (this->IsUnion()) {
645  for (int i = 0, n = this->AsUnion()->Length(); i < n; ++i) {
646  if (this->AsUnion()->Get(i).Maybe(that)) return true;
647  }
648  return false;
649  }
650 
651  // T overlaps (T1 \/ ... \/ Tn) if (T overlaps T1) \/ ... \/ (T overlaps Tn)
652  if (that.IsUnion()) {
653  for (int i = 0, n = that.AsUnion()->Length(); i < n; ++i) {
654  if (this->Maybe(that.AsUnion()->Get(i))) return true;
655  }
656  return false;
657  }
658 
659  if (this->IsBitset() && that.IsBitset()) return true;
660 
661  if (this->IsRange()) {
662  if (that.IsRange()) {
663  return Overlap(this->AsRange(), that.AsRange());
664  }
665  if (that.IsBitset()) {
666  bitset number_bits = BitsetType::NumberBits(that.AsBitset());
667  if (number_bits == BitsetType::kNone) {
668  return false;
669  }
670  double min = std::max(BitsetType::Min(number_bits), this->Min());
671  double max = std::min(BitsetType::Max(number_bits), this->Max());
672  return min <= max;
673  }
674  }
675  if (that.IsRange()) {
676  return that.Maybe(*this); // This case is handled above.
677  }
678 
679  if (this->IsBitset() || that.IsBitset()) return true;
680 
681  return this->SimplyEquals(that);
682 }

694 bool UnionType::Wellformed() const {
695  DisallowGarbageCollection no_gc;
696  // This checks the invariants of the union representation:
697  // 1. There are at least two elements.
698  // 2. The first element is a bitset, no other element is a bitset.
699  // 3. At most one element is a range, and it must be the second one.
700  // 4. No element is itself a union.
701  // 5. No element (except the bitset) is a subtype of any other.
702  // 6. If there is a range, then the bitset type does not contain
703  // plain number bits.
704  DCHECK_LE(2, this->Length()); // (1)
705  DCHECK(this->Get(0).IsBitset()); // (2a)
706 
707  for (int i = 0; i < this->Length(); ++i) {
708  if (i != 0) DCHECK(!this->Get(i).IsBitset()); // (2b)
709  if (i != 1) DCHECK(!this->Get(i).IsRange()); // (3)
710  DCHECK(!this->Get(i).IsUnion()); // (4)
711  for (int j = 0; j < this->Length(); ++j) {
712  if (i != j && i != 0) DCHECK(!this->Get(i).Is(this->Get(j))); // (5)
713  }
714  }
715  DCHECK(!this->Get(1).IsRange() ||
716  (BitsetType::NumberBits(this->Get(0).AsBitset()) ==
717  BitsetType::kNone)); // (6)
718  return true;
719 }

724 Type Type::Intersect(Type type1, Type type2, Zone* zone) {
725  // Fast case: bit sets.
726  if (type1.IsBitset() && type2.IsBitset()) {
727  return NewBitset(type1.AsBitset() & type2.AsBitset());
728  }
729 
730  // Fast case: top or bottom types.
731  if (type1.IsNone() || type2.IsAny()) return type1; // Shortcut.
732  if (type2.IsNone() || type1.IsAny()) return type2; // Shortcut.
733 
734  // Semi-fast case.
735  if (type1.Is(type2)) return type1;
736  if (type2.Is(type1)) return type2;
737 
738  // Slow case: create union.
739 
740  // Semantic subtyping check - this is needed for consistency with the
741  // semi-fast case above.
742  if (type1.Is(type2)) {
743  type2 = Any();
744  } else if (type2.Is(type1)) {
745  type1 = Any();
746  }
747 
748  bitset bits = type1.BitsetGlb() & type2.BitsetGlb();
749  int size1 = type1.IsUnion() ? type1.AsUnion()->Length() : 1;
750  int size2 = type2.IsUnion() ? type2.AsUnion()->Length() : 1;
751  int size;
752  if (base::bits::SignedAddOverflow32(size1, size2, &size)) return Any();
753  if (base::bits::SignedAddOverflow32(size, 2, &size)) return Any();
754  UnionType* result = UnionType::New(size, zone);
755  size = 0;
756 
757  // Deal with bitsets.
758  result->Set(size++, NewBitset(bits));
759 
760  RangeType::Limits lims = RangeType::Limits::Empty();
761  size = IntersectAux(type1, type2, result, size, &lims, zone);
762 
763  // If the range is not empty, then insert it into the union and
764  // remove the number bits from the bitset.
765  if (!lims.IsEmpty()) {
766  size = UpdateRange(Type::Range(lims, zone), result, size, zone);
767 
768  // Remove the number bits.
769  bitset number_bits = BitsetType::NumberBits(bits);
770  bits &= ~number_bits;
771  result->Set(0, NewBitset(bits));
772  }
773  return NormalizeUnion(result, size, zone);
774 }

776 int Type::UpdateRange(Type range, UnionType* result, int size, Zone* zone) {
777  if (size == 1) {
778  result->Set(size++, range);
779  } else {
780  // Make space for the range.
781  result->Set(size++, result->Get(1));
782  result->Set(1, range);
783  }
784 
785  // Remove any components that just got subsumed.
786  for (int i = 2; i < size;) {
787  if (result->Get(i).Is(range)) {
788  result->Set(i, result->Get(--size));
789  } else {
790  ++i;
791  }
792  }
793  return size;
794 }

814 int Type::IntersectAux(Type lhs, Type rhs, UnionType* result, int size,
815  RangeType::Limits* lims, Zone* zone) {
816  if (lhs.IsUnion()) {
817  for (int i = 0, n = lhs.AsUnion()->Length(); i < n; ++i) {
818  size = IntersectAux(lhs.AsUnion()->Get(i), rhs, result, size, lims, zone);
819  }
820  return size;
821  }
822  if (rhs.IsUnion()) {
823  for (int i = 0, n = rhs.AsUnion()->Length(); i < n; ++i) {
824  size = IntersectAux(lhs, rhs.AsUnion()->Get(i), result, size, lims, zone);
825  }
826  return size;
827  }
828 
829  if (BitsetType::IsNone(lhs.BitsetLub() & rhs.BitsetLub())) return size;
830 
831  if (lhs.IsRange()) {
832  if (rhs.IsBitset()) {
833  RangeType::Limits lim = IntersectRangeAndBitset(lhs, rhs, zone);
834 
835  if (!lim.IsEmpty()) {
836  *lims = RangeType::Limits::Union(lim, *lims);
837  }
838  return size;
839  }
840  if (rhs.IsRange()) {
841  RangeType::Limits lim = RangeType::Limits::Intersect(
842  RangeType::Limits(lhs.AsRange()), RangeType::Limits(rhs.AsRange()));
843  if (!lim.IsEmpty()) {
844  *lims = RangeType::Limits::Union(lim, *lims);
845  }
846  }
847  return size;
848  }
849  if (rhs.IsRange()) {
850  // This case is handled symmetrically above.
851  return IntersectAux(rhs, lhs, result, size, lims, zone);
852  }
853  if (lhs.IsBitset() || rhs.IsBitset()) {
854  return AddToUnion(lhs.IsBitset() ? rhs : lhs, result, size, zone);
855  }
856  if (lhs.SimplyEquals(rhs)) {
857  return AddToUnion(lhs, result, size, zone);
858  }
859  return size;
860 }

862 // Make sure that we produce a well-formed range and bitset:
863 // If the range is non-empty, the number bits in the bitset should be
864 // clear. Moreover, if we have a canonical range (such as Signed32),
865 // we want to produce a bitset rather than a range.
866 Type Type::NormalizeRangeAndBitset(Type range, bitset* bits, Zone* zone) {
867  // Fast path: If the bitset does not mention numbers, we can just keep the
868  // range.
869  bitset number_bits = BitsetType::NumberBits(*bits);
870  if (number_bits == 0) {
871  return range;
872  }
873 
874  // If the range is semantically contained within the bitset, return None and
875  // leave the bitset untouched.
876  bitset range_lub = range.BitsetLub();
877  if (BitsetType::Is(range_lub, *bits)) {
878  return None();
879  }
880 
881  // Slow path: reconcile the bitset range and the range.
882  double bitset_min = BitsetType::Min(number_bits);
883  double bitset_max = BitsetType::Max(number_bits);
884 
885  double range_min = range.Min();
886  double range_max = range.Max();
887 
888  // Remove the number bits from the bitset, they would just confuse us now.
889  // NOTE: bits contains OtherNumber iff bits contains PlainNumber, in which
890  // case we already returned after the subtype check above.
891  *bits &= ~number_bits;
892 
893  if (range_min <= bitset_min && range_max >= bitset_max) {
894  // Bitset is contained within the range, just return the range.
895  return range;
896  }
897 
898  if (bitset_min < range_min) {
899  range_min = bitset_min;
900  }
901  if (bitset_max > range_max) {
902  range_max = bitset_max;
903  }
904  return Type::Range(range_min, range_max, zone);
905 }

956 Type Type::Union(Type type1, Type type2, Zone* zone) {
957  // Fast case: bit sets.
958  if (type1.IsBitset() && type2.IsBitset()) {
959  return NewBitset(type1.AsBitset() | type2.AsBitset());
960  }
961 
962  // Fast case: top or bottom types.
963  if (type1.IsAny() || type2.IsNone()) return type1;
964  if (type2.IsAny() || type1.IsNone()) return type2;
965 
966  // Semi-fast case.
967  if (type1.Is(type2)) return type2;
968  if (type2.Is(type1)) return type1;
969 
970  // Slow case: create union.
971  int size1 = type1.IsUnion() ? type1.AsUnion()->Length() : 1;
972  int size2 = type2.IsUnion() ? type2.AsUnion()->Length() : 1;
973  int size;
974  if (base::bits::SignedAddOverflow32(size1, size2, &size)) return Any();
975  if (base::bits::SignedAddOverflow32(size, 2, &size)) return Any();
976  UnionType* result = UnionType::New(size, zone);
977  size = 0;
978 
979  // Compute the new bitset.
980  bitset new_bitset = type1.BitsetGlb() | type2.BitsetGlb();
981 
982  // Deal with ranges.
983  Type range = None();
984  Type range1 = type1.GetRange();
985  Type range2 = type2.GetRange();
986  if (range1 != nullptr && range2 != nullptr) {
987  RangeType::Limits lims =
988  RangeType::Limits::Union(RangeType::Limits(range1.AsRange()),
989  RangeType::Limits(range2.AsRange()));
990  Type union_range = Type::Range(lims, zone);
991  range = NormalizeRangeAndBitset(union_range, &new_bitset, zone);
992  } else if (range1 != nullptr) {
993  range = NormalizeRangeAndBitset(range1, &new_bitset, zone);
994  } else if (range2 != nullptr) {
995  range = NormalizeRangeAndBitset(range2, &new_bitset, zone);
996  }
997  Type bits = NewBitset(new_bitset);
998  result->Set(size++, bits);
999  if (!range.IsNone()) result->Set(size++, range);
1000 
1001  size = AddToUnion(type1, result, size, zone);
1002  size = AddToUnion(type2, result, size, zone);
1003  return NormalizeUnion(result, size, zone);
1004 }

1006 // Add [type] to [result] unless [type] is bitset, range, or already subsumed.
