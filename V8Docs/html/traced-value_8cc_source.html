32 void EscapeAndAppendString(const char* value, std::string* result) {
33  *result += '"';
34  while (*value) {
35  unsigned char c = *value++;
36  switch (c) {
37  case '\b':
38  *result += "\\b";
39  break;
40  case '\f':
41  *result += "\\f";
42  break;
43  case '\n':
44  *result += "\\n";
45  break;
46  case '\r':
47  *result += "\\r";
48  break;
49  case '\t':
50  *result += "\\t";
51  break;
52  case '\"':
53  *result += "\\\"";
54  break;
55  case '\\':
56  *result += "\\\\";
57  break;
58  default:
59  if (c < '\x20' || c == '\x7F') {
60  char number_buffer[8];
61  base::OS::SNPrintF(number_buffer, arraysize(number_buffer), "\\u%04X",
62  static_cast<unsigned>(c));
63  *result += number_buffer;
64  } else {
65  *result += c;
66  }
67  }
68  }
69  *result += '"';
70 }
