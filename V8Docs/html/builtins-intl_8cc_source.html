238 template <class T>
239 Tagged<Object> LegacyFormatConstructor(BuiltinArguments args, Isolate* isolate,
240  v8::Isolate::UseCounterFeature feature,
241  DirectHandle<JSAny> constructor,
242  const char* method_name) {
243  isolate->CountUsage(feature);
244  DirectHandle<JSReceiver> new_target;
245  // 1. If NewTarget is undefined, let newTarget be the active
246  // function object, else let newTarget be NewTarget.
247  if (IsUndefined(*args.new_target(), isolate)) {
248  new_target = args.target();
249  } else {
250  new_target = Cast<JSReceiver>(args.new_target());
251  }
252 
253  // [[Construct]]
254  DirectHandle<JSFunction> target = args.target();
255  DirectHandle<Object> locales = args.atOrUndefined(isolate, 1);
256  DirectHandle<Object> options = args.atOrUndefined(isolate, 2);
257 
258  // 2. Let format be ? OrdinaryCreateFromConstructor(newTarget,
259  // "%<T>Prototype%", ...).
260  DirectHandle<Map> map;
261  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
262  isolate, map, JSFunction::GetDerivedMap(isolate, target, new_target));
263 
264  // 3. Perform ? Initialize<T>(Format, locales, options).
265  DirectHandle<T> format;
266  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
267  isolate, format, T::New(isolate, map, locales, options, method_name));
268  // 4. Let this be the this value.
269  if (IsUndefined(*args.new_target(), isolate)) {
270  DirectHandle<JSAny> receiver = args.receiver();
271  // 5. If NewTarget is undefined and ? OrdinaryHasInstance(%<T>%, this)
272  // is true, then Look up the intrinsic value that has been stored on
273  // the context.
274  DirectHandle<Object> ordinary_has_instance_obj;
275  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
276  isolate, ordinary_has_instance_obj,
277  Object::OrdinaryHasInstance(isolate, constructor, receiver));
278  if (Object::BooleanValue(*ordinary_has_instance_obj, isolate)) {
279  if (!IsJSReceiver(*receiver)) {
280  THROW_NEW_ERROR_RETURN_FAILURE(
281  isolate, NewTypeError(MessageTemplate::kIncompatibleMethodReceiver,
282  isolate->factory()->NewStringFromAsciiChecked(
283  method_name),
284  receiver));
285  }
286  DirectHandle<JSReceiver> rec = Cast<JSReceiver>(receiver);
287  // a. Perform ? DefinePropertyOrThrow(this,
288  // %Intl%.[[FallbackSymbol]], PropertyDescriptor{ [[Value]]: format,
289  // [[Writable]]: false, [[Enumerable]]: false, [[Configurable]]: false }).
290  PropertyDescriptor desc;
291  desc.set_value(format);
292  desc.set_writable(false);
293  desc.set_enumerable(false);
294  desc.set_configurable(false);
295  Maybe<bool> success = JSReceiver::DefineOwnProperty(
296  isolate, rec, isolate->factory()->intl_fallback_symbol(), &desc,
297  Just(kThrowOnError));
298  MAYBE_RETURN(success, ReadOnlyRoots(isolate).exception());
299  CHECK(success.FromJust());
300  // b. b. Return this.
301  return *receiver;
302  }
303  }
304  // 6. Return format.
305  return *format;
306 }
