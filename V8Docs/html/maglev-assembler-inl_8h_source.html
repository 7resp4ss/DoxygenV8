220 template <typename Function, typename... Args>
221 inline Label* MaglevAssembler::MakeDeferredCode(Function&& deferred_code_gen,
222  Args&&... args) {
223  using FunctionPointer =
224  typename detail::FunctionArgumentsTupleHelper<Function>::FunctionPointer;
225  static_assert(
226  std::is_invocable_v<FunctionPointer, MaglevAssembler*,
227  decltype(detail::CopyForDeferred(
228  std::declval<MaglevCompilationInfo*>(),
229  std::declval<Args>()))...>,
230  "Parameters of deferred_code_gen function should match arguments into "
231  "MakeDeferredCode");
232 
233  TemporaryRegisterScope scratch_scope(this);
234  using DeferredCodeInfoT = detail::DeferredCodeInfoImpl<Function>;
235  DeferredCodeInfoT* deferred_code =
236  compilation_info()->zone()->New<DeferredCodeInfoT>(
237  compilation_info(), scratch_scope.CopyForDefer(), deferred_code_gen,
238  std::forward<Args>(args)...);
239 
240 #ifdef DEBUG
241  deferred_code->set_allow_call(allow_deferred_call_);
242  deferred_code->set_allow_allocate(allow_allocate_);
243 #endif // DEBUG
244 
245  code_gen_state()->PushDeferredCode(deferred_code);
246  return &deferred_code->deferred_code_label;
247 }

265 inline void AllocateSlow(MaglevAssembler* masm,
266  RegisterSnapshot register_snapshot, Register object,
267  Builtin builtin, T size_in_bytes, ZoneLabelRef done) {
268  // Remove {object} from snapshot, since it is the returned allocated
269  // HeapObject.

273  SaveRegisterStateForCall save_register_state(masm, register_snapshot);
274  using D = AllocateDescriptor;
275  masm->Move(D::GetRegisterParameter(D::kRequestedSize), size_in_bytes);

279  }
280  masm->Jump(*done);

402 inline void MaglevAssembler::Branch(Condition condition, Label* if_true,
403  Label::Distance true_distance,
404  bool fallthrough_when_true, Label* if_false,
405  Label::Distance false_distance,
406  bool fallthrough_when_false) {
407  if (fallthrough_when_false) {
408  if (fallthrough_when_true) {
409  // If both paths are a fallthrough, do nothing.
410  DCHECK_EQ(if_true, if_false);
411  return;
412  }
413  // Jump over the false block if true, otherwise fall through into it.
414  JumpIf(condition, if_true, true_distance);
415  } else {
416  // Jump to the false block if true.
417  JumpIf(NegateCondition(condition), if_false, false_distance);
418  // Jump to the true block if it's not the next block.
419  if (!fallthrough_when_true) {
420  Jump(if_true, true_distance);
421  }
422  }
423 }

625  std::tuple<Args&&...> args_tuple{std::forward<Args>(args)...};
626 
627  // If there is a context, the first argument is the context parameter. Use
628  // the remaining args as the actual arguments. We pass the context first
629  // instead of last to avoid ambiguity around dealing with on-stack
630  // arguments.
631  constexpr size_t context_args = Descriptor::HasContextParameter() ? 1 : 0;
632  static_assert(context_args <= std::tuple_size_v<decltype(args_tuple)>,
633  "Not enough arguments passed in to builtin (are you missing a "
634  "context argument?)");
635  auto args_tuple_without_context = base::tuple_drop<context_args>(args_tuple);
636  CheckArgs<Descriptor>(masm, args_tuple_without_context);
637 
638  // Split args into register and stack args.
639  static_assert(Descriptor::GetRegisterParameterCount() <=
640  std::tuple_size_v<decltype(args_tuple_without_context)>,
641  "Not enough arguments passed in to builtin (are you missing a "
642  "context argument?)");
643  auto register_args =
644  base::tuple_head<Descriptor::GetRegisterParameterCount()>(
645  args_tuple_without_context);
646  auto stack_args = base::tuple_drop<Descriptor::GetRegisterParameterCount()>(
647  args_tuple_without_context);
648 
649  // Split stack args into fixed and variable.
650  static_assert(
651  Descriptor::GetStackParameterCount() <=
652  std::tuple_size_v<decltype(stack_args)>,
653  "Not enough stack arguments passed in to builtin (are you missing a "
654  "context argument?)");
655  auto fixed_stack_args =
656  base::tuple_head<Descriptor::GetStackParameterCount()>(stack_args);
657  auto vararg_stack_args =
658  base::tuple_drop<Descriptor::GetStackParameterCount()>(stack_args);
659 
660  if constexpr (!Descriptor::AllowVarArgs()) {
661  static_assert(std::tuple_size_v<decltype(vararg_stack_args)> == 0,
662  "Too many arguments passed in to builtin that expects no "
663  "vararg stack arguments");
664  }
665 
666  // First push stack arguments (if any), since some of these may be in
667  // registers and we don't want to clobber them. This supports any thing
668  // `masm->Push` supports, including iterator ranges, so the tuple size may be
669  // smaller than the number of arguments actually pushed. We push fixed and
670  // vararg stack arguments separately, so that there's an appropriate amount
671  // of padding between them.
672  if (Descriptor::kStackArgumentOrder == StackArgumentOrder::kDefault) {
673  PushArgumentsForBuiltin<Descriptor>(
674  masm, std::forward<decltype(fixed_stack_args)>(fixed_stack_args));
675  PushArgumentsForBuiltin<Descriptor>(
676  masm, std::forward<decltype(vararg_stack_args)>(vararg_stack_args));
677  } else {
678  PushArgumentsForBuiltin<Descriptor>(
679  masm, std::forward<decltype(vararg_stack_args)>(vararg_stack_args));
680  PushArgumentsForBuiltin<Descriptor>(
681  masm, std::forward<decltype(fixed_stack_args)>(fixed_stack_args));
682  }
683 
684 // Then, set register arguments.
685 // TODO(leszeks): Use the parallel move helper to do register moves, instead
686 // of detecting clobbering.
687 #ifdef DEBUG
688  RegList written_registers = {};
689  DoubleRegList written_double_registers = {};
690 #endif // DEBUG
691 
692  base::tuple_for_each_with_index(register_args, [&](auto&& arg, auto index) {
693  using Arg = decltype(arg);
694  static_assert(index < Descriptor::GetRegisterParameterCount());
695 
696  // Make sure the argument wasn't clobbered by any previous write.
697  DCHECK(!ClobberedBy(written_registers, arg));
698  DCHECK(!ClobberedBy(written_double_registers, arg));
699 
700  static constexpr bool use_double_register =
701  IsFloatingPoint(Descriptor::GetParameterType(index).representation());
702  if constexpr (use_double_register) {
703  DoubleRegister target = Descriptor::GetDoubleRegisterParameter(index);
704  if constexpr (std::is_same_v<Input, std::decay_t<Arg>>) {
705  DCHECK_EQ(target, arg.AssignedDoubleRegister());
706  USE(target);
707  } else {
708  masm->Move(target, std::forward<Arg>(arg));
709  }
710 #ifdef DEBUG
711  written_double_registers.set(target);
712 #endif // DEBUG
713  } else {
714  Register target = Descriptor::GetRegisterParameter(index);
715  if constexpr (std::is_same_v<Input, std::decay_t<Arg>>) {
716  DCHECK_EQ(target, arg.AssignedGeneralRegister());
717  USE(target);
718  } else {
719  masm->Move(target, std::forward<Arg>(arg));
720  }
721 #ifdef DEBUG
722  written_registers.set(target);
723 #endif // DEBUG
724  }
725 
726  // TODO(leszeks): Support iterator range for register args.
727  });
728 
729  // Set the context last (to avoid clobbering).
730  if constexpr (Descriptor::HasContextParameter()) {
731  auto&& context = std::get<0>(args_tuple);
732  DCHECK(!ClobberedBy(written_registers, context));
733  DCHECK(!ClobberedBy(written_double_registers, context));
734  DCHECK(MachineTypeMatches(MachineType::AnyTagged(), context));
735 
736  if constexpr (std::is_same_v<Input, std::decay_t<decltype(context)>>) {
737  DCHECK_EQ(Descriptor::ContextRegister(),
738  context.AssignedGeneralRegister());
739  } else {
740  // Don't allow raw Register here, force materialisation from a constant.
741  // This is because setting parameters could have clobbered the register.
742  // TODO(leszeks): Include the context register in the parallel moves
743  // described above.
744  static_assert(!std::is_same_v<Register, std::decay_t<decltype(context)>>);
745  masm->Move(Descriptor::ContextRegister(), context);
746  }
747  }
748 }
