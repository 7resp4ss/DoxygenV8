298 // The InstructionCode is an opaque, target-specific integer that encodes what
299 // code to emit for an instruction in the code generator. It is not interesting
300 // to the register allocator, as the inputs and flags on the instructions
301 // specify everything of interest.
302 using InstructionCode = uint32_t;
303 
304 // Helpers for encoding / decoding InstructionCode into the fields needed
305 // for code generation. We encode the instruction, addressing mode, flags, and
306 // other information into a single InstructionCode which is stored as part of
307 // the instruction.
308 //
309 // All instructions have the first five fields, using up the lower 22 bits.
310 // The remaining 10 bits are accessible in the MiscField, which the other
311 // instructions types can overlay specific data:
312 // -- Generic
313 // Field | Bits
314 // ArchOpcode | 9
315 // AddressingMode | 5
316 // FlagsMode | 3
317 // FlagsCondition | 5
318 // Misc | 10
319 //
320 // So, the following instruction types use the MiscField in the following ways:
321 // -- Atomics
322 // Field | Bits
323 // AtomicWidth | 2
324 // AtomicMemoryOrder | 2
325 // AtomicStoreRecordWriteMode | 4
326 // AccessMode | 2
327 //
328 // -- Write barriers
329 // Field | Bits
330 // RecordWriteMode | 4
331 // Undefined | 4
332 // AccessMode | 2
333 //
334 // -- X64 vectors
335 // Field | Bits
336 // LaneSize | 2
337 // VectorLength | 2
338 // Undefined | 4
339 // AccessMode | 2
340 //
341 // -- Everyone else vectors
342 // Field | Bits
343 // LaneSize | 8
344 // AccessMode | 2
345 //
346 // -- Deopts
347 // Field | Bits
348 // DeoptImmedArgsCount | 2
349 // DeoptFrameStateOffset | 8
350 //
351 // -- Non-deopt branches
352 // Field | Bits
353 // StackCheck | 2
354 // BranchHint | 1
355 // Undefined | 7
356 //
357 using ArchOpcodeField = base::BitField<ArchOpcode, 0, 9>;
358 static_assert(ArchOpcodeField::is_valid(kLastArchOpcode),
359  "All opcodes must fit in the 9-bit ArchOpcodeField.");
360 using AddressingModeField = ArchOpcodeField::Next<AddressingMode, 5>;
361 static_assert(
362  AddressingModeField::is_valid(kLastAddressingMode),
363  "All addressing modes must fit in the 5-bit AddressingModeField.");
364 using FlagsModeField = AddressingModeField::Next<FlagsMode, 3>;
365 using FlagsConditionField = FlagsModeField::Next<FlagsCondition, 5>;
366 
367 // AtomicWidthField is used for the various Atomic opcodes. Only used on 64bit
368 // architectures. All atomic instructions on 32bit architectures are assumed to
369 // be 32bit wide.
370 using AtomicWidthField = FlagsConditionField::Next<AtomicWidth, 2>;
371 // AtomicMemoryOrderField is used for the various Atomic opcodes. This field is
372 // not used on all architectures. It is used on architectures where the codegen
373 // for kSeqCst and kAcqRel differ only by emitting fences.
374 using AtomicMemoryOrderField = AtomicWidthField::Next<AtomicMemoryOrder, 2>;
375 using AtomicStoreRecordWriteModeField =
376  AtomicMemoryOrderField::Next<RecordWriteMode, 4>;
377 
378 // Write modes for writes with barrier.
379 using RecordWriteModeField = FlagsConditionField::Next<RecordWriteMode, 3>;
380 
381 // LaneSizeField and AccessModeField are helper types to encode/decode a lane
382 // size, an access mode, or both inside the overlapping MiscField.
383 #ifdef V8_TARGET_ARCH_X64
384 enum LaneSize { kL8 = 0, kL16 = 1, kL32 = 2, kL64 = 3 };
385 enum VectorLength { kV128 = 0, kV256 = 1, kV512 = 3 };
386 using LaneSizeField = FlagsConditionField::Next<LaneSize, 2>;
387 using VectorLengthField = LaneSizeField::Next<VectorLength, 2>;
388 #else
389 using LaneSizeField = FlagsConditionField::Next<int, 8>;
390 #endif // V8_TARGET_ARCH_X64
391 
392 // Denotes whether the instruction needs to emit an accompanying landing pad for
393 // the trap handler.
394 using AccessModeField =
395  AtomicStoreRecordWriteModeField::Next<MemoryAccessMode, 2>;
396 
397 // Since AccessModeField is defined in terms of atomics, this assert ensures it
398 // does not overlap with other fields it is used with.
399 static_assert(AtomicStoreRecordWriteModeField::kLastUsedBit >=
400  RecordWriteModeField::kLastUsedBit);
401 #ifdef V8_TARGET_ARCH_X64
402 static_assert(AtomicStoreRecordWriteModeField::kLastUsedBit >=
403  VectorLengthField::kLastUsedBit);
404 #else
405 static_assert(AtomicStoreRecordWriteModeField::kLastUsedBit >=
406  LaneSizeField::kLastUsedBit);
407 #endif

409 // TODO(turbofan): {HasMemoryAccessMode} is currently only used to guard
410 // decoding (in CodeGenerator and InstructionScheduler). Encoding (in
411 // InstructionSelector) is not yet guarded. There are in fact instructions for
412 // which InstructionSelector does set a MemoryAccessMode but CodeGenerator
413 // doesn't care to consume it (e.g. kArm64LdrDecompressTaggedSigned). This is
414 // scary. {HasMemoryAccessMode} does not include these instructions, so they can
415 // be easily found by guarding encoding.
416 inline bool HasMemoryAccessMode(ArchOpcode opcode) {
417 #if defined(TARGET_ARCH_OPCODE_WITH_MEMORY_ACCESS_MODE_LIST)
418  switch (opcode) {
419 #define CASE(Name) \
420  case k##Name: \
421  return true;
422  COMMON_ARCH_OPCODE_WITH_MEMORY_ACCESS_MODE_LIST(CASE)
423  TARGET_ARCH_OPCODE_WITH_MEMORY_ACCESS_MODE_LIST(CASE)
424 #undef CASE
425  default:
426  return false;
427  }
428 #else
429  return false;
430 #endif
431 }
