18 // Just a wrapper around calls to unify CallOp and TailCallOp.
19 struct CallView {
20  V<CallTarget> target;
21  base::Vector<const OpIndex> arguments;
22  const TSCallDescriptor* descriptor;
23 
24  explicit CallView(const Operation& op) {
25  switch (op.opcode) {
26  case Opcode::kCall: {
27  const CallOp& call = op.Cast<CallOp>();
28  this->target = call.callee();
29  this->arguments = call.arguments();
30  this->descriptor = call.descriptor;
31  return;
32  }
33  case Opcode::kTailCall: {
34  const TailCallOp& call = op.Cast<TailCallOp>();
35  this->target = call.callee();
36  this->arguments = call.arguments();
37  this->descriptor = call.descriptor;
38  return;
39  }
40 
41  default:
42  UNREACHABLE();
43  }
44  }
45 };

47 std::optional<Builtin> TryGetBuiltin(V<Any> node, Graph& graph) {
48  if (const ConstantOp* target_obj_cst =
49  graph.Get(node).TryCast<Opmask::kHeapConstant>()) {
50  IndirectHandle<HeapObject> target_obj = target_obj_cst->handle();
51  if (IsCode(*target_obj)) {
52  IndirectHandle<Code> target_code = Cast<Code>(target_obj);
53  if (target_code->is_builtin()) {
54  return target_code->builtin_id();
55  }
56  }
57  }
58  return {};
59 }

61 void AnalyzeCall(CallView call, std::unordered_set<Builtin>& called_builtins,
62  BuiltinAllocateEffect* can_allocate, Graph& graph,
63  Builtin current_builtin) {
64  if (*can_allocate == BuiltinAllocateEffect::kYes) {
65  // No need to bother with allocations through builtin calls: this builtin
66  // allocates directly.
67  return;
68  }
69 
70  V<CallTarget> target = call.target;
71  if (std::optional<Builtin> builtin = TryGetBuiltin(target, graph)) {
72  if (*builtin == current_builtin) {
73  // Recursive call, no need to record anything.
74  return;
75  } else if (std::optional<Runtime::FunctionId> runtime_function_id =
76  call.descriptor->descriptor->runtime_function_id()) {
77  // We special case Abort, which can allocate, but always result from
78  // something unexpected already having happened, and never resumes
79  // execution afterwards. This could lead to the GC seeing uninitialized
80  // memory, which is unfortunate, but it's better this way rather than
81  // having over-conservative effects because we try to terminate execution
82  // cleanly through Abort rather than just segfaulting abruptly.
83  if (Runtime::kCanTriggerGC[runtime_function_id.value()] &&
84  runtime_function_id.value() != Runtime::kAbort) {
85  *can_allocate = BuiltinAllocateEffect::kYes;
86  }
87  return;
88  } else {
89  called_builtins.insert(*builtin);
90  *can_allocate = BuiltinAllocateEffect::kMaybeWithBuiltinCall;
91  return;
92  }
93  } else if (graph.Get(target).Is<Opmask::kExternalConstant>()) {
94  // External functions cannot allocate without conservative stack scanning.
95  return;
96  }
97 
98  *can_allocate = BuiltinAllocateEffect::kYes;
99 }

101 void CsaEffectsComputationPhase::Run(PipelineData* data, Zone* temp_zone) {
102  // We need to unpark the local heap so that we can look if HeapConstants are
103  // builtin code or not.
104  UnparkedScopeIfNeeded scope(data->broker());
105 
106  Builtin builtin = data->info()->builtin();
107 
108  BuiltinAllocateEffect can_allocate = BuiltinAllocateEffect::kNo;
109  std::unordered_set<Builtin> called_builtins;
110  for (const Operation& op : data->graph().AllOperations()) {
111  switch (op.opcode) {
112  case Opcode::kAllocate:
113  can_allocate = BuiltinAllocateEffect::kYes;
114  break;
115 
116  case Opcode::kCall:
117  case Opcode::kTailCall:
118  AnalyzeCall(CallView{op}, called_builtins, &can_allocate, data->graph(),
119  builtin);
120  break;
121 
122  default:
123  break;
124  }
125  }
126 
127  data->isolate()->builtins_effects_analyzer()->RegisterInitialEffects(
128  builtin, can_allocate, called_builtins);
129 }
