53 struct IntrinsicFunctionIdentifier {
54  IntrinsicFunctionIdentifier(const unsigned char* data, const int length)
55  : data_(data), length_(length) {}
56 
57  static bool Match(void* key1, void* key2) {
58  const IntrinsicFunctionIdentifier* lhs =
59  static_cast<IntrinsicFunctionIdentifier*>(key1);
60  const IntrinsicFunctionIdentifier* rhs =
61  static_cast<IntrinsicFunctionIdentifier*>(key2);
62  if (lhs->length_ != rhs->length_) return false;
63  return CompareCharsEqual(lhs->data_, rhs->data_, rhs->length_);
64  }
65 
66  uint32_t Hash() {
67  return StringHasher::HashSequentialString<uint8_t>(
68  data_, length_, v8::internal::kZeroHashSeed);
69  }
70 
71  const unsigned char* data_;
72  const int length_;
73 };

75 void InitializeIntrinsicFunctionNames() {
76  base::CustomMatcherHashMap* function_name_map =
77  new base::CustomMatcherHashMap(IntrinsicFunctionIdentifier::Match);
78  for (size_t i = 0; i < arraysize(kIntrinsicFunctions); ++i) {
79  const Runtime::Function* function = &kIntrinsicFunctions[i];
80  IntrinsicFunctionIdentifier* identifier = new IntrinsicFunctionIdentifier(
81  reinterpret_cast<const unsigned char*>(function->name),
82  static_cast<int>(strlen(function->name)));
83  base::HashMap::Entry* entry =
84  function_name_map->InsertNew(identifier, identifier->Hash());
85  entry->value = const_cast<Runtime::Function*>(function);
86  }
87  kRuntimeFunctionNameMap = function_name_map;
88 }

196 bool Runtime::IsEnabledForFuzzing(FunctionId id) {
197  CHECK(v8_flags.fuzzing);
198 
199  // In general, all runtime functions meant for testing should also be exposed
200  // to the fuzzers. That way, the fuzzers are able to import and mutate
201  // regression tests that use those functions. Internal runtime functions
202  // (which are e.g. only called from other builtins, etc.) should not directly
203  // be exposed as they are not meant to be called directly from JavaScript.
204  // However, exceptions exist: some test functions cannot be used for certain
205  // types of fuzzing (e.g. differential fuzzing), or would cause false
206  // positive crashes and therefore should not be exposed to fuzzers at all.
207 
208  // For differential fuzzing, only a handful of functions are allowed,
209  // everything else is disabled. Many runtime functions are unsuited for
210  // differential fuzzing as they for example expose internal engine state
211  // (e.g. functions such as %HasFastProperties). To avoid having to maintain a
212  // large denylist of such functions, we instead use an allowlist for
213  // differential fuzzing.
214  bool is_differential_fuzzing =
215  v8_flags.allow_natives_for_differential_fuzzing;
216  if (is_differential_fuzzing) {
217  switch (id) {
218  case Runtime::kArrayBufferDetach:
219  case Runtime::kDeoptimizeFunction:
220  case Runtime::kDeoptimizeNow:
221  case Runtime::kDisableOptimizationFinalization:
222  case Runtime::kEnableCodeLoggingForTesting:
223  case Runtime::kFinalizeOptimization:
224  case Runtime::kGetUndetectable:
225  case Runtime::kNeverOptimizeFunction:
226  case Runtime::kOptimizeFunctionOnNextCall:
227  case Runtime::kOptimizeMaglevOnNextCall:
228  case Runtime::kOptimizeOsr:
229  case Runtime::kPrepareFunctionForOptimization:
230  case Runtime::kPretenureAllocationSite:
231  case Runtime::kSetAllocationTimeout:
232  case Runtime::kSetForceSlowPath:
233  case Runtime::kSimulateNewspaceFull:
234  case Runtime::kWaitForBackgroundOptimization:
235  case Runtime::kSetBatterySaverMode:
236  case Runtime::kSetPriorityBestEffort:
237  case Runtime::kSetPriorityUserVisible:
238  case Runtime::kSetPriorityUserBlocking:
239  case Runtime::kIsEfficiencyModeEnabled:
240  case Runtime::kBaselineOsr:
241  case Runtime::kCompileBaseline:
242 #if V8_ENABLE_WEBASSEMBLY && V8_WASM_RANDOM_FUZZERS
243  case Runtime::kWasmGenerateRandomModule:
244 #endif // V8_ENABLE_WEBASSEMBLY && V8_WASM_RANDOM_FUZZERS
245 #if V8_ENABLE_WEBASSEMBLY
246  case Runtime::kWasmArray:
247  case Runtime::kWasmStruct:
248  case Runtime::kWasmTierUpFunction:
249  case Runtime::kWasmTriggerTierUpForTesting:
250 #endif // V8_ENABLE_WEBASSEMBLY
251  return true;
252 
253  default:
254  return false;
255  }
256  }
257 
258  // Runtime functions disabled for all/most types of fuzzing.
259  // Reasons for a function to be in this list include that it is not useful
260  // for fuzzing (e.g. %DebugPrint) or not fuzzing-safe and therefore would
261  // cause false-positive crashes (e.g. %AbortJS).
262  switch (id) {
263  case Runtime::kAbort:
264  case Runtime::kAbortCSADcheck:
265  case Runtime::kAbortJS:
266  case Runtime::kSystemBreak:
267  case Runtime::kBenchMaglev:
268  case Runtime::kBenchTurbofan:
269  case Runtime::kDebugPrint:
270  case Runtime::kDisassembleFunction:
271  case Runtime::kGetFunctionForCurrentFrame:
272  case Runtime::kGetCallable:
273  case Runtime::kGetAbstractModuleSource:
274  case Runtime::kTurbofanStaticAssert:
275  case Runtime::kClearFunctionFeedback:
276  case Runtime::kStringIsFlat:
277  case Runtime::kGetInitializerFunction:
278 #ifdef V8_ENABLE_WEBASSEMBLY
279  case Runtime::kWasmTraceEnter:
280  case Runtime::kWasmTraceExit:
281  case Runtime::kWasmTraceMemory:
282  case Runtime::kCheckIsOnCentralStack:
283  case Runtime::kSetWasmInstantiateControls:
284  case Runtime::kWasmNull:
285  case Runtime::kFreezeWasmLazyCompilation:
286  case Runtime::kDeserializeWasmModule:
287 #endif // V8_ENABLE_WEBASSEMBLY
288  // TODO(353685107): investigate whether these should be exposed to fuzzers.
289  case Runtime::kConstructDouble:
290  case Runtime::kConstructConsString:
291  case Runtime::kConstructSlicedString:
292  case Runtime::kConstructInternalizedString:
293  case Runtime::kConstructThinString:
294  // TODO(353971258): investigate whether this should be exposed to fuzzers.
295  case Runtime::kSerializeDeserializeNow:
296  // TODO(353928347): investigate whether this should be exposed to fuzzers.
297  case Runtime::kCompleteInobjectSlackTracking:
298  // TODO(354005312): investigate whether this should be exposed to fuzzers.
299  case Runtime::kShareObject:
300  // TODO(354310130): investigate whether this should be exposed to fuzzers.
301  case Runtime::kForceFlush:
302  return false;
303 
304  case Runtime::kLeakHole:
305  return v8_flags.hole_fuzzing;
306 
307  default:
308  break;
309  }
310 
311  // The default case: test functions are exposed, everything else is not.
312  switch (id) {
313 #define F(name, nargs, ressize, ...) case k##name:
314 #define I(name, nargs, ressize, ...) case kInline##name:
315  FOR_EACH_INTRINSIC_TEST(F, I)
316  IF_WASM(FOR_EACH_INTRINSIC_WASM_TEST, F, I)
317 #undef I
318 #undef F
319  return true;
320  default:
321  return false;
322  }
323 }

353 const Runtime::Function* Runtime::RuntimeFunctionTable(Isolate* isolate) {
354 #ifdef USE_SIMULATOR
355  // When running with the simulator we need to provide a table which has
356  // redirected runtime entry addresses.
357  if (!isolate->runtime_state()->redirected_intrinsic_functions()) {
358  size_t function_count = arraysize(kIntrinsicFunctions);
359  Function* redirected_functions = new Function[function_count];
360  memcpy(redirected_functions, kIntrinsicFunctions,
361  sizeof(kIntrinsicFunctions));
362  for (size_t i = 0; i < function_count; i++) {
363  ExternalReference redirected_entry =
364  ExternalReference::Create(static_cast<Runtime::FunctionId>(i));
365  redirected_functions[i].entry = redirected_entry.address();
366  }
367  isolate->runtime_state()->set_redirected_intrinsic_functions(
368  redirected_functions);
369  }
370 
371  return isolate->runtime_state()->redirected_intrinsic_functions();
372 #else
373  return kIntrinsicFunctions;
374 #endif
375 }
