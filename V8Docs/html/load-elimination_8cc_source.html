41 bool MayAlias(Node* a, Node* b) {
42  if (a != b) {
43  if (!NodeProperties::GetType(a).Maybe(NodeProperties::GetType(b))) {
44  return false;
45  } else if (IsRename(b)) {
46  return MayAlias(a, b->InputAt(0));
47  } else if (IsRename(a)) {
48  return MayAlias(a->InputAt(0), b);
49  } else if (b->opcode() == IrOpcode::kAllocate) {
50  switch (a->opcode()) {
51  case IrOpcode::kAllocate:
52  case IrOpcode::kHeapConstant:
53  case IrOpcode::kParameter:
54  return false;
55  default:
56  break;
57  }
58  } else if (a->opcode() == IrOpcode::kAllocate) {
59  switch (b->opcode()) {
60  case IrOpcode::kHeapConstant:
61  case IrOpcode::kParameter:
62  return false;
63  default:
64  break;
65  }
66  }
67  }
68  return true;
69 }

77 Reduction LoadElimination::Reduce(Node* node) {
78  if (v8_flags.trace_turbo_load_elimination) {
79  if (node->op()->EffectInputCount() > 0) {
80  PrintF(" visit #%d:%s", node->id(), node->op()->mnemonic());
81  if (node->op()->ValueInputCount() > 0) {
82  PrintF("(");
83  for (int i = 0; i < node->op()->ValueInputCount(); ++i) {
84  if (i > 0) PrintF(", ");
85  Node* const value = NodeProperties::GetValueInput(node, i);
86  PrintF("#%d:%s", value->id(), value->op()->mnemonic());
87  }
88  PrintF(")");
89  }
90  PrintF("\n");
91  for (int i = 0; i < node->op()->EffectInputCount(); ++i) {
92  Node* const effect = NodeProperties::GetEffectInput(node, i);
93  if (AbstractState const* const state = node_states_.Get(effect)) {
94  PrintF(" state[%i]: #%d:%s\n", i, effect->id(),
95  effect->op()->mnemonic());
96  state->Print();
97  } else {
98  PrintF(" no state[%i]: #%d:%s\n", i, effect->id(),
99  effect->op()->mnemonic());
100  }
101  }
102  }
103  }
104  switch (node->opcode()) {
105  case IrOpcode::kMapGuard:
106  return ReduceMapGuard(node);
107  case IrOpcode::kCheckMaps:
108  return ReduceCheckMaps(node);
109  case IrOpcode::kCompareMaps:
110  return ReduceCompareMaps(node);
111  case IrOpcode::kEnsureWritableFastElements:
112  return ReduceEnsureWritableFastElements(node);
113  case IrOpcode::kMaybeGrowFastElements:
114  return ReduceMaybeGrowFastElements(node);
115  case IrOpcode::kTransitionElementsKind:
116  return ReduceTransitionElementsKind(node);
117  case IrOpcode::kTransitionElementsKindOrCheckMap:
118  return ReduceTransitionElementsKindOrCheckMap(node);
119  case IrOpcode::kLoadField:
120  return ReduceLoadField(node, FieldAccessOf(node->op()));
121  case IrOpcode::kStoreField:
122  return ReduceStoreField(node, FieldAccessOf(node->op()));
123  case IrOpcode::kLoadElement:
124  return ReduceLoadElement(node);
125  case IrOpcode::kStoreElement:
126  return ReduceStoreElement(node);
127  case IrOpcode::kTransitionAndStoreElement:
128  return ReduceTransitionAndStoreElement(node);
129  case IrOpcode::kStoreTypedElement:
130  return ReduceStoreTypedElement(node);
131  case IrOpcode::kEffectPhi:
132  return ReduceEffectPhi(node);
133  case IrOpcode::kDead:
134  break;
135  case IrOpcode::kStart:
136  return ReduceStart(node);
137  default:
138  return ReduceOtherNode(node);
139  }
140  return NoChange();
141 }

171  Zone* zone) const {
172  for (Element const element : this->elements_) {
173  if (element.object == nullptr) continue;
174  if (MayAlias(object, element.object)) {
175  AbstractElements* that = zone->New<AbstractElements>(zone);

183  that->elements_[that->next_index_++] = element2;

185  }
186  that->next_index_ %= arraysize(elements_);
187  return that;
188  }
189  }
190  return this;

291 LoadElimination::AbstractField const* LoadElimination::AbstractField::KillConst(
292  Node* object, Zone* zone) const {
293  for (auto info1 : this->info_for_node_) {
294  if (info1.first->IsDead()) continue;
295  // If we previously recorded information about a const store on the given
296  // 'object', we might not have done it on the same node; e.g. we might now
297  // identify the object by a FinishRegion node, whereas the initial const
298  // store was performed on the Allocate node. We therefore remove information
299  // on all nodes that must alias with 'object'.
300  if (MustAlias(object, info1.first)) {
301  AbstractField* that = zone->New<AbstractField>(zone);
302  for (auto info2 : this->info_for_node_) {
303  if (!MustAlias(object, info2.first)) {
304  that->info_for_node_.insert(info2);
305  }
306  }
307  return that;
308  }
309  }
310  return this;
311 }

635 LoadElimination::AbstractState const*
636 LoadElimination::AbstractState::KillFields(Node* object, MaybeHandle<Name> name,
637  Zone* zone) const {
638  AliasStateInfo alias_info(this, object);
639  for (size_t i = 0;; ++i) {
640  if (i == fields_.size()) {
641  return this;
642  }
643  if (AbstractField const* this_field = this->fields_[i]) {
644  AbstractField const* that_field =
645  this_field->Kill(alias_info, name, zone);
646  if (that_field != this_field) {
647  AbstractState* that = zone->New<AbstractState>(*this);
648  that->fields_[i] = that_field;

677 LoadElimination::FieldInfo const* LoadElimination::AbstractState::LookupField(
678  Node* object, IndexRange index_range,
679  ConstFieldInfo const_field_info) const {
680  // Check if all the indices in {index_range} contain identical information.
681  // If not, a partially overlapping access has invalidated part of the value.
682  std::optional<LoadElimination::FieldInfo const*> result;
683  for (int index : index_range) {
684  LoadElimination::FieldInfo const* info = nullptr;
685  if (const_field_info.IsConst()) {
686  if (AbstractField const* this_field = const_fields_[index]) {
687  info = this_field->Lookup(object);
688  }
689  if (!(info && info->const_field_info == const_field_info)) return nullptr;
690  } else {
691  if (AbstractField const* this_field = fields_[index]) {
692  info = this_field->Lookup(object);
693  }
694  if (!info) return nullptr;
695  }
696  if (!result.has_value()) {
697  result = info;
698  } else if (**result != *info) {
699  // We detected inconsistent information for a field here.
700  // This can happen when incomplete alias information makes an unrelated
701  // write invalidate part of a field and then we re-combine this partial
702  // information.
703  // This is probably OK, but since it's rare, we better bail out here.
704  return nullptr;
705  }
706  }
707  return *result;
708 }

710 bool LoadElimination::AliasStateInfo::MayAlias(Node* other) const {
711  // If {object} is being initialized right here (indicated by {object} being
712  // an Allocate node instead of a FinishRegion node), we know that {other}
713  // can only alias with {object} if they refer to exactly the same node.
714  if (object_->opcode() == IrOpcode::kAllocate) {

719  return false;

721  // Decide aliasing based on maps (if available).

726  if (map != other_maps.at(0)) {
727  return false;
728  }
729  }
730  }
731  return true;

988  Node* effect = NodeProperties::GetEffectInput(node);
989  Node* control = NodeProperties::GetControlInput(node);

1008  state->LookupField(object, field_index, access.const_field_info);

1012  lookup_result =
1013  state->LookupField(object, field_index, ConstFieldInfo::None());

1019  if (IsCompatible(representation, lookup_result->representation) &&
1020  !replacement->IsDead()) {

1026  NodeProperties::GetType(node),
1027  NodeProperties::GetType(replacement), graph()->zone());
1028  replacement = effect =
1029  graph()->NewNode(common()->TypeGuard(replacement_type),
1030  replacement, effect, control);
1031  NodeProperties::SetType(replacement, replacement_type);
1032  }
1033  ReplaceWithValue(node, replacement, effect);
1034  return Replace(replacement);
1035  }
1036  }
1037  FieldInfo info(node, representation, access.name,
1038  access.const_field_info);
1039  state = state->AddField(object, field_index, info, zone());
1040  }
1041  }
1042  if (access.map.has_value()) {
1043  state = state->SetMaps(node, ZoneRefSet<Map>(*access.map), zone());

1048 Reduction LoadElimination::ReduceStoreField(Node* node,
1049  FieldAccess const& access) {
1050  Node* const object = NodeProperties::GetValueInput(node, 0);
1051  Node* const new_value = NodeProperties::GetValueInput(node, 1);
1052  Node* const effect = NodeProperties::GetEffectInput(node);
1053  AbstractState const* state = node_states_.Get(effect);
1054  if (state == nullptr) return NoChange();
1055  if (access.offset == HeapObject::kMapOffset &&
1056  access.base_is_tagged == kTaggedBase) {
1057  DCHECK(IsAnyTagged(access.machine_type.representation()));
1058  // Kill all potential knowledge about the {object}s map.
1059  state = state->KillMaps(object, zone());
1060  Type const new_value_type = NodeProperties::GetType(new_value);
1061  if (new_value_type.IsHeapConstant()) {
1062  // Record the new {object} map information.
1063  ZoneRefSet<Map> object_maps(
1064  new_value_type.AsHeapConstant()->Ref().AsMap());
1065  state = state->SetMaps(object, object_maps, zone());
1066  }
1067  } else {
1068  IndexRange field_index = FieldIndexOf(access);
1069  if (field_index != IndexRange::Invalid()) {
1070  bool is_const_store = access.const_field_info.IsConst();
1071  MachineRepresentation representation =
1072  access.machine_type.representation();
1073  FieldInfo const* lookup_result =
1074  state->LookupField(object, field_index, access.const_field_info);
1075 
1076  if (lookup_result &&
1077  (!is_const_store || V8_ENABLE_DOUBLE_CONST_STORE_CHECK_BOOL)) {
1078  // At runtime, we should never encounter
1079  // - any store replacing existing info with a different, incompatible
1080  // representation, nor
1081  // - two consecutive const stores, unless the latter is a store into
1082  // a literal.
1083  // However, we may see such code statically, so we guard against
1084  // executing it by emitting Unreachable.
1085  // TODO(gsps): Re-enable the double const store check even for
1086  // non-debug builds once we have identified other FieldAccesses
1087  // that should be marked mutable instead of const
1088  // (cf. JSCreateLowering::AllocateFastLiteral).
1089  bool incompatible_representation =
1090  !lookup_result->name.is_null() &&
1091  !IsCompatible(representation, lookup_result->representation);
1092  bool illegal_double_const_store =
1093  is_const_store && !access.is_store_in_literal;
1094  if (incompatible_representation || illegal_double_const_store) {
1095  Node* control = NodeProperties::GetControlInput(node);
1096  Node* unreachable =
1097  graph()->NewNode(common()->Unreachable(), effect, control);
1098  return Replace(unreachable);
1099  }
1100  if (lookup_result->value == new_value) {
1101  // This store is fully redundant.
1102  return Replace(effect);
1103  }
1104  }
1105 
1106  // Kill all potentially aliasing fields and record the new value.
1107  FieldInfo new_info(new_value, representation, access.name,
1108  access.const_field_info);
1109  if (is_const_store && access.is_store_in_literal) {
1110  // We only kill const information when there is a chance that we
1111  // previously stored information about the given const field (namely,
1112  // when we observe const stores to literals).
1113  state = state->KillConstField(object, field_index, zone());
1114  }
1115  state = state->KillField(object, field_index, access.name, zone());
1116  state = state->AddField(object, field_index, new_info, zone());
1117  if (is_const_store) {
1118  // For const stores, we track information in both the const and the
1119  // mutable world to guard against field accesses that should have
1120  // been marked const, but were not.
1121  new_info.const_field_info = ConstFieldInfo::None();
1122  state = state->AddField(object, field_index, new_info, zone());
1123  }
1124  } else {
1125  // Unsupported StoreField operator.
1126  state = state->KillFields(object, access.name, zone());
1127  }
1128  }
1129  return UpdateState(node, state);
1130 }

1363 LoadElimination::AbstractState const* LoadElimination::ComputeLoopState(
1364  Node* node, AbstractState const* state) const {
1365  Node* const control = NodeProperties::GetControlInput(node);
1366  struct TransitionElementsKindInfo {
1367  ElementsTransition transition;
1368  Node* object;
1369  };
1370  // Allocate zone data structures in a temporary zone with a lifetime limited
1371  // to this function to avoid blowing up the size of the stage-global zone.
1372  Zone temp_zone(zone()->allocator(), "Temporary scoped zone");
1373  ZoneVector<TransitionElementsKindInfo> element_transitions_(&temp_zone);
1374  ZoneQueue<Node*> queue(&temp_zone);
1375  ZoneSet<Node*> visited(&temp_zone);
1376  visited.insert(node);
1377  for (int i = 1; i < control->InputCount(); ++i) {
1378  queue.push(node->InputAt(i));
1379  }
1380  while (!queue.empty()) {
1381  Node* const current = queue.front();
1382  queue.pop();
1383  if (visited.find(current) == visited.end()) {
1384  visited.insert(current);
1385  if (!current->op()->HasProperty(Operator::kNoWrite)) {
1386  switch (current->opcode()) {
1387  case IrOpcode::kEnsureWritableFastElements: {
1388  Node* const object = NodeProperties::GetValueInput(current, 0);
1389  state = state->KillField(
1390  object, FieldIndexOf(JSObject::kElementsOffset, kTaggedSize),
1391  MaybeHandle<Name>(), zone());
1392  break;
1393  }
1394  case IrOpcode::kMaybeGrowFastElements: {
1395  Node* const object = NodeProperties::GetValueInput(current, 0);
1396  state = state->KillField(
1397  object, FieldIndexOf(JSObject::kElementsOffset, kTaggedSize),
1398  MaybeHandle<Name>(), zone());
1399  break;
1400  }
1401  case IrOpcode::kTransitionElementsKind: {
1402  ElementsTransition transition = ElementsTransitionOf(current->op());
1403  Node* const object = NodeProperties::GetValueInput(current, 0);
1404  ZoneRefSet<Map> object_maps;
1405  if (!state->LookupMaps(object, &object_maps) ||
1406  !ZoneRefSet<Map>(transition.target()).contains(object_maps)) {
1407  element_transitions_.push_back({transition, object});
1408  }
1409  break;
1410  }
1411  case IrOpcode::kTransitionElementsKindOrCheckMap: {
1412  ElementsTransitionWithMultipleSources transition =
1413  ElementsTransitionWithMultipleSourcesOf(current->op());
1414  Node* const object = NodeProperties::GetValueInput(current, 0);
1415  ZoneRefSet<Map> object_maps;
1416  MapRef target = transition.target();
1417  if (!state->LookupMaps(object, &object_maps) ||
1418  !ZoneRefSet<Map>(target).contains(object_maps)) {
1419  for (MapRef source : transition.sources()) {
1420  ElementsTransition::Mode mode =
1421  IsSimpleMapChangeTransition(source.elements_kind(),
1422  target.elements_kind())
1423  ? ElementsTransition::kFastTransition
1424  : ElementsTransition::kSlowTransition;
1425  element_transitions_.push_back(
1426  {ElementsTransition(mode, source, target), object});
1427  }
1428  }
1429  break;
1430  }
1431  case IrOpcode::kTransitionAndStoreElement: {
1432  Node* const object = NodeProperties::GetValueInput(current, 0);
1433  // Invalidate what we know about the {object}s map.
1434  state = state->KillMaps(object, zone());
1435  // Kill the elements as well.
1436  state = state->KillField(
1437  object, FieldIndexOf(JSObject::kElementsOffset, kTaggedSize),
1438  MaybeHandle<Name>(), zone());
1439  break;
1440  }
1441  case IrOpcode::kStoreField: {
1442  FieldAccess access = FieldAccessOf(current->op());
1443  state = ComputeLoopStateForStoreField(current, state, access);
1444  break;
1445  }
1446  case IrOpcode::kStoreElement: {
1447  Node* const object = NodeProperties::GetValueInput(current, 0);
1448  Node* const index = NodeProperties::GetValueInput(current, 1);
1449  state = state->KillElement(object, index, zone());
1450  break;
1451  }
1452  case IrOpcode::kCheckMaps:
1453  case IrOpcode::kStoreTypedElement: {
1454  // Doesn't affect anything we track with the state currently.
1455  break;
1456  }
1457  default:
1458  return state->KillAll(zone());
1459  }
1460  }
1461  for (int i = 0; i < current->op()->EffectInputCount(); ++i) {
1462  queue.push(NodeProperties::GetEffectInput(current, i));
1463  }
1464  }
1465  }
1466 
1467  // Finally, we apply the element transitions. For each transition, we will try
1468  // to only invalidate information about nodes that can have the transition's
1469  // source map. The trouble is that an object can be transitioned by some other
1470  // transition to the source map. In that case, the other transition will
1471  // invalidate the information, so we are mostly fine.
1472  //
1473  // The only bad case is
1474  //
1475  // mapA ---fast---> mapB ---slow---> mapC
1476  //
1477  // If we process the slow transition first on an object that has mapA, we will
1478  // ignore the transition because the object does not have its source map
1479  // (mapB). When we later process the fast transition, we invalidate the
1480  // object's map, but we keep the information about the object's elements. This
1481  // is wrong because the elements will be overwritten by the slow transition.
1482  //
1483  // Note that the slow-slow case is fine because either of the slow transition
1484  // will invalidate the elements field, so the processing order does not
1485  // matter.
1486  //
1487  // To handle the bad case properly, we first kill the maps using all
1488  // transitions. We kill the the fields later when all the transitions are
1489  // already reflected in the map information.
1490 
1491  for (const TransitionElementsKindInfo& t : element_transitions_) {
1492  AliasStateInfo alias_info(state, t.object, t.transition.source());
1493  state = state->KillMaps(alias_info, zone());
1494  }
1495  for (const TransitionElementsKindInfo& t : element_transitions_) {
1496  switch (t.transition.mode()) {
1497  case ElementsTransition::kFastTransition:
1498  break;
1499  case ElementsTransition::kSlowTransition: {
1500  AliasStateInfo alias_info(state, t.object, t.transition.source());
1501  state = state->KillField(
1502  alias_info, FieldIndexOf(JSObject::kElementsOffset, kTaggedSize),
1503  MaybeHandle<Name>(), zone());
1504  break;
1505  }
1506  }
1507  }
1508  return state;
1509 }
