25 bool CheckObjectComparisonAllowed(Address a, Address b) {
26  // LINT.IfChange(CheckObjectComparisonAllowed)
27  if (!HAS_STRONG_HEAP_OBJECT_TAG(a) || !HAS_STRONG_HEAP_OBJECT_TAG(b)) {
28  return true;
29  }
30  Tagged<HeapObject> obj_a = UncheckedCast<HeapObject>(Tagged<Object>(a));
31  Tagged<HeapObject> obj_b = UncheckedCast<HeapObject>(Tagged<Object>(b));
32  // This check might fail when we try to compare objects in different pointer
33  // compression cages (e.g. the one used by code space or trusted space) with
34  // each other. The main legitimate case when such "mixed" comparison could
35  // happen is comparing two AbstractCode objects. If that's the case one must
36  // use AbstractCode's == operator instead of Object's one or SafeEquals().
37  CHECK_EQ(HeapLayout::InCodeSpace(obj_a), HeapLayout::InCodeSpace(obj_b));
38 #ifdef V8_ENABLE_SANDBOX
39  CHECK_EQ(HeapLayout::InTrustedSpace(obj_a),
40  HeapLayout::InTrustedSpace(obj_b));
41 #endif
42  return true;
43  // LINT.ThenChange(src/codegen/code-stub-assembler.cc:CheckObjectComparisonAllowed)
44 }
45 #endif // defined(V8_EXTERNAL_CODE_SPACE) || defined(V8_ENABLE_SANDBOX)
