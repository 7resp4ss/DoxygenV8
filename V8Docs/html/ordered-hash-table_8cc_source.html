18 template <class Derived, int entrysize>
19 MaybeHandle<Derived> OrderedHashTable<Derived, entrysize>::Allocate(
20  Isolate* isolate, int capacity, AllocationType allocation) {
21  // Capacity must be a power of two, since we depend on being able
22  // to divide and multiple by 2 (kLoadFactor) to derive capacity
23  // from number of buckets. If we decide to change kLoadFactor
24  // to something other than 2, capacity should be stored as another
25  // field of this object.
26  capacity =
27  base::bits::RoundUpToPowerOfTwo32(std::max({kInitialCapacity, capacity}));
28  if (capacity > MaxCapacity()) {
29  // Throw RangeError with a generic message.
30  THROW_NEW_ERROR_RETURN_VALUE(
31  isolate,
32  NewRangeError(MessageTemplate::kCollectionGrowFailed,
33  isolate->factory()->empty_string()),
34  {});
35  }
36  int num_buckets = capacity / kLoadFactor;
37  Handle<FixedArray> backing_store = isolate->factory()->NewFixedArrayWithMap(
38  Derived::GetMap(isolate->roots_table()),
39  HashTableStartIndex() + num_buckets + (capacity * kEntrySize),
40  allocation);
41  Handle<Derived> table = Cast<Derived>(backing_store);
42  DisallowGarbageCollection no_gc;
43  Tagged<Derived> raw_table = *table;
44  for (int i = 0; i < num_buckets; ++i) {
45  raw_table->set(HashTableStartIndex() + i, Smi::FromInt(kNotFound));
46  }
47  raw_table->SetNumberOfBuckets(num_buckets);
48  raw_table->SetNumberOfElements(0);
49  raw_table->SetNumberOfDeletedElements(0);
50  return table;
51 }

73 template <class Derived, int entrysize>
74 template <template <typename> typename HandleType>
75  requires(std::is_convertible_v<HandleType<Derived>, DirectHandle<Derived>>)
76 HandleType<Derived>::MaybeType
77 OrderedHashTable<Derived, entrysize>::EnsureCapacityForAdding(
78  Isolate* isolate, HandleType<Derived> table) {
79  DCHECK(!table->IsObsolete());
80 
81  int nof = table->NumberOfElements();
82  int nod = table->NumberOfDeletedElements();
83  int capacity = table->Capacity();
84  if ((nof + nod) < capacity) return table;
85 
86  int new_capacity;
87  if (capacity == 0) {
88  // step from empty to minimum proper size
89  new_capacity = kInitialCapacity;
90  } else if (nod >= (capacity >> 1)) {
91  // Don't need to grow if we can simply clear out deleted entries instead.
92  // Note that we can't compact in place, though, so we always allocate
93  // a new table.
94  new_capacity = capacity;
95  } else {
96  new_capacity = capacity << 1;
97  }
98 
99  return Derived::Rehash(isolate, table, new_capacity);
100 }

147 template <class Derived, int entrysize>
148 InternalIndex OrderedHashTable<Derived, entrysize>::FindEntry(
149  Isolate* isolate, Tagged<Object> key) {
150  if (NumberOfElements() == 0) {
151  // This is not just an optimization but also ensures that we do the right
152  // thing if Capacity() == 0
153  return InternalIndex::NotFound();
154  }
155 
156  int raw_entry;
157  // This special cases for Smi, so that we avoid the HandleScope
158  // creation below.
159  if (IsSmi(key)) {
160  uint32_t hash = ComputeUnseededHash(Smi::ToInt(key));
161  raw_entry = HashToEntryRaw(hash & Smi::kMaxValue);
162  } else {
163  HandleScope scope(isolate);
164  Tagged<Object> hash = Object::GetHash(key);
165  // If the object does not have an identity hash, it was never used as a key
166  if (IsUndefined(hash, isolate)) return InternalIndex::NotFound();
167  raw_entry = HashToEntryRaw(Smi::ToInt(hash));
168  }
169 
170  // Walk the chain in the bucket to find the key.
171  while (raw_entry != kNotFound) {
172  Tagged<Object> candidate_key = KeyAt(InternalIndex(raw_entry));
173  if (Object::SameValueZero(candidate_key, key))
174  return InternalIndex(raw_entry);
175  raw_entry = NextChainEntryRaw(raw_entry);
176  }
177 
178  return InternalIndex::NotFound();
179 }

181 template <template <typename> typename HandleType>
182  requires(std::is_convertible_v<HandleType<OrderedHashSet>,
183  DirectHandle<OrderedHashSet>>)
184 HandleType<OrderedHashSet>::MaybeType OrderedHashSet::Add(
185  Isolate* isolate, HandleType<OrderedHashSet> table,
186  DirectHandle<Object> key) {
187  int hash;
188  {
189  DisallowGarbageCollection no_gc;
190  Tagged<Object> raw_key = *key;
191  Tagged<OrderedHashSet> raw_table = *table;
192  hash = Object::GetOrCreateHash(raw_key, isolate).value();
193  if (raw_table->NumberOfElements() > 0) {
194  int raw_entry = raw_table->HashToEntryRaw(hash);
195  // Walk the chain of the bucket and try finding the key.
196  while (raw_entry != kNotFound) {
197  Tagged<Object> candidate_key =
198  raw_table->KeyAt(InternalIndex(raw_entry));
199  // Do not add if we have the key already
200  if (Object::SameValueZero(candidate_key, raw_key)) return table;
201  raw_entry = raw_table->NextChainEntryRaw(raw_entry);
202  }
203  }
204  }
205 
206  typename HandleType<OrderedHashSet>::MaybeType table_candidate =
207  OrderedHashSet::EnsureCapacityForAdding(isolate, table);
208  if (!table_candidate.ToHandle(&table)) {
209  CHECK(isolate->has_exception());
210  return table_candidate;
211  }
212  DisallowGarbageCollection no_gc;
213  Tagged<OrderedHashSet> raw_table = *table;
214  // Read the existing bucket values.
215  int bucket = raw_table->HashToBucket(hash);
216  int previous_entry = raw_table->HashToEntryRaw(hash);
217  int nof = raw_table->NumberOfElements();
218  // Insert a new entry at the end,
219  int new_entry = nof + raw_table->NumberOfDeletedElements();
220  int new_index = raw_table->EntryToIndexRaw(new_entry);
221  raw_table->set(new_index, *key);
222  raw_table->set(new_index + kChainOffset, Smi::FromInt(previous_entry));
223  // and point the bucket to the new entry.
224  raw_table->set(HashTableStartIndex() + bucket, Smi::FromInt(new_entry));
225  raw_table->SetNumberOfElements(nof + 1);
226  return table;
227 }

236 Handle<FixedArray> OrderedHashSet::ConvertToKeysArray(
237  Isolate* isolate, Handle<OrderedHashSet> table, GetKeysConversion convert) {
238  int length = table->NumberOfElements();
239  int nof_buckets = table->NumberOfBuckets();
240  // Convert the dictionary to a linear list.
241  Handle<FixedArray> result = Cast<FixedArray>(table);
242  // From this point on table is no longer a valid OrderedHashSet.
243  result->set_map(isolate, ReadOnlyRoots(isolate).fixed_array_map());
244  for (int i = 0; i < length; i++) {
245  int index = HashTableStartIndex() + nof_buckets + (i * kEntrySize);
246  Tagged<Object> key = table->get(index);
247  uint32_t index_value;
248  if (convert == GetKeysConversion::kConvertToString) {
249  if (Object::ToArrayIndex(key, &index_value)) {
250  // Avoid trashing the number to string cache with numbers that
251  // are not likely to be needed.
252  bool use_cache = i < SmiStringCache::kMaxCapacity;
253  key = *isolate->factory()->Uint32ToString(index_value, use_cache);
254  } else {
255  CHECK(IsName(key));
256  }
257  } else if (convert == GetKeysConversion::kNoNumbers) {
258  DCHECK(!Object::ToArrayIndex(key, &index_value));
259  }
260  result->set(i, key);
261  }
262  return FixedArray::RightTrimOrEmpty(isolate, result, length);
263 }

282 template <class Derived, int entrysize>
283 template <template <typename> typename HandleType>
284  requires(std::is_convertible_v<HandleType<Derived>, DirectHandle<Derived>>)
285 HandleType<Derived>::MaybeType OrderedHashTable<Derived, entrysize>::Rehash(
286  Isolate* isolate, HandleType<Derived> table, int new_capacity) {
287  DCHECK(!table->IsObsolete());
288 
289  typename HandleType<Derived>::MaybeType new_table_candidate =
290  Derived::Allocate(isolate, new_capacity,
291  HeapLayout::InYoungGeneration(*table)
292  ? AllocationType::kYoung
293  : AllocationType::kOld);
294  DirectHandle<Derived> new_table;
295  if (!new_table_candidate.ToHandle(&new_table)) {
296  return new_table_candidate;
297  }
298  int new_buckets = new_table->NumberOfBuckets();
299  int new_entry = 0;
300  int removed_holes_index = 0;
301 
302  DisallowGarbageCollection no_gc;
303 
304  for (InternalIndex old_entry : table->IterateEntries()) {
305  int old_entry_raw = old_entry.as_int();
306  Tagged<Object> key = table->KeyAt(old_entry);
307  if (IsHashTableHole(key, isolate)) {
308  table->SetRemovedIndexAt(removed_holes_index++, old_entry_raw);
309  continue;
310  }
311 
312  Tagged<Object> hash = Object::GetHash(key);
313  int bucket = Smi::ToInt(hash) & (new_buckets - 1);
314  Tagged<Object> chain_entry = new_table->get(HashTableStartIndex() + bucket);
315  new_table->set(HashTableStartIndex() + bucket, Smi::FromInt(new_entry));
316  int new_index = new_table->EntryToIndexRaw(new_entry);
317  int old_index = table->EntryToIndexRaw(old_entry_raw);
318  for (int i = 0; i < entrysize; ++i) {
319  Tagged<Object> value = table->get(old_index + i);
320  new_table->set(new_index + i, value);
321  }
322  new_table->set(new_index + kChainOffset, chain_entry);
323  ++new_entry;
324  }
325 
326  DCHECK_EQ(table->NumberOfDeletedElements(), removed_holes_index);
327 
328  new_table->SetNumberOfElements(table->NumberOfElements());
329  if (table->NumberOfBuckets() > 0) {
330  // Don't try to modify the empty canonical table which lives in RO space.
331  table->SetNextTable(*new_table);
332  }
333 
334  return new_table_candidate;
335 }

940 template <class Derived>
941 Handle<Derived> SmallOrderedHashTable<Derived>::Rehash(Isolate* isolate,
942  Handle<Derived> table,
943  int new_capacity) {
944  DCHECK_GE(kMaxCapacity, new_capacity);
945 
946  Handle<Derived> new_table = SmallOrderedHashTable<Derived>::Allocate(
947  isolate, new_capacity,
948  HeapLayout::InYoungGeneration(*table) ? AllocationType::kYoung
949  : AllocationType::kOld);
950  int new_entry = 0;
951 
952  {
953  DisallowGarbageCollection no_gc;
954  for (InternalIndex old_entry : table->IterateEntries()) {
955  Tagged<Object> key = table->KeyAt(old_entry);
956  if (IsTheHole(key, isolate)) continue;
957 
958  int hash = Smi::ToInt(Object::GetHash(key));
959  int bucket = new_table->HashToBucket(hash);
960  int chain = new_table->GetFirstEntry(bucket);
961 
962  new_table->SetFirstEntry(bucket, new_entry);
963  new_table->SetNextEntry(new_entry, chain);
964 
965  for (int i = 0; i < Derived::kEntrySize; ++i) {
966  Tagged<Object> value = table->GetDataEntry(old_entry.as_int(), i);
967  new_table->SetDataEntry(new_entry, i, value);
968  }
969 
970  ++new_entry;
971  }
972 
973  new_table->SetNumberOfElements(table->NumberOfElements());
974  }
975  return new_table;
976 }

1009 template <class Derived>

1015  // Don't need to grow if we can simply clear out deleted entries instead.
