98  super_name));

131 // Gets |index|'th argument which may be a class constructor object, a class
132 // prototype object or a class method. In the latter case the following
133 // post-processing may be required:
134 // 1) set method's name to a concatenation of |name_prefix| and |key| if the
135 // method's shared function info indicates that method does not have a
136 // shared name.
137 template <typename Dictionary>
138 MaybeDirectHandle<Object> GetMethodAndSetName(Isolate* isolate,
139  RuntimeArguments& args,
140  Tagged<Smi> index,
141  DirectHandle<String> name_prefix,
142  Handle<Object> key) {
143  int int_index = index.value();
144 
145  // Class constructor and prototype values do not require post processing.
146  if (int_index < ClassBoilerplate::kFirstDynamicArgumentIndex) {
147  return args.at<Object>(int_index);
148  }
149 
150  DirectHandle<JSFunction> method = args.at<JSFunction>(int_index);
151 
152  if (!method->shared()->HasSharedName()) {
153  // TODO(ishell): method does not have a shared name at this point only if
154  // the key is a computed property name. However, the bytecode generator
155  // explicitly generates ToName bytecodes to ensure that the computed
156  // property name is properly converted to Name. So, we can actually be smart
157  // here and avoid converting Smi keys back to Name.
158  DirectHandle<Name> name = KeyToName<Dictionary>(isolate, key);
159  if (!JSFunction::SetName(isolate, method, name, name_prefix)) {
160  return MaybeDirectHandle<Object>();
161  }
162  }
163  return method;
164 }
165 
166 // Gets |index|'th argument which may be a class constructor object, a class
167 // prototype object or a class method.
168 // This is a simplified version of GetMethodAndSetName()
169 // function above that is used when it's guaranteed that the method has
170 // shared name.
171 Tagged<Object> GetMethodWithSharedName(Isolate* isolate, RuntimeArguments& args,
172  Tagged<Object> index) {
173  DisallowGarbageCollection no_gc;
174  int int_index = Smi::ToInt(index);
175 
176  // Class constructor and prototype values do not require post processing.
177  if (int_index < ClassBoilerplate::kFirstDynamicArgumentIndex) {
178  return args[int_index];
179  }
180 
181  DirectHandle<JSFunction> method = args.at<JSFunction>(int_index);
182  DCHECK(method->shared()->HasSharedName());
183  return *method;
184 }

186 template <typename Dictionary>
187 Handle<Dictionary> ShallowCopyDictionaryTemplate(
188  Isolate* isolate, DirectHandle<Dictionary> dictionary_template) {
189  Handle<Dictionary> dictionary =
190  Dictionary::ShallowCopy(isolate, dictionary_template);
191  // Clone all AccessorPairs in the dictionary.
192  for (InternalIndex i : dictionary->IterateEntries()) {
193  Tagged<Object> value = dictionary->ValueAt(i);
194  if (IsAccessorPair(value)) {
195  DirectHandle<AccessorPair> pair(Cast<AccessorPair>(value), isolate);
196  pair = AccessorPair::Copy(isolate, pair);
197  dictionary->ValueAtPut(i, *pair);
198  }
199  }
200  return dictionary;
201 }

203 template <typename Dictionary>
204 bool SubstituteValues(Isolate* isolate, DirectHandle<Dictionary> dictionary,
205  RuntimeArguments& args) {
206  // Replace all indices with proper methods.
207  ReadOnlyRoots roots(isolate);
208  for (InternalIndex i : dictionary->IterateEntries()) {
209  Tagged<Object> maybe_key = dictionary->KeyAt(i);
210  if (!Dictionary::IsKey(roots, maybe_key)) continue;
211  Handle<Object> key(maybe_key, isolate);
212  Handle<Object> value(dictionary->ValueAt(i), isolate);
213  if (IsAccessorPair(*value)) {
214  auto pair = Cast<AccessorPair>(value);
215  Tagged<Object> tmp = pair->getter();
216  if (IsSmi(tmp)) {
217  DirectHandle<Object> result;
218  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
219  isolate, result,
220  GetMethodAndSetName<Dictionary>(isolate, args, Cast<Smi>(tmp),
221  isolate->factory()->get_string(),
222  key),
223  false);
224  pair->set_getter(*result);
225  }
226  tmp = pair->setter();
227  if (IsSmi(tmp)) {
228  DirectHandle<Object> result;
229  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
230  isolate, result,
231  GetMethodAndSetName<Dictionary>(isolate, args, Cast<Smi>(tmp),
232  isolate->factory()->set_string(),
233  key),
234  false);
235  pair->set_setter(*result);
236  }
237  } else if (IsSmi(*value)) {
238  DirectHandle<Object> result;
239  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
240  isolate, result,
241  GetMethodAndSetName<Dictionary>(isolate, args, Cast<Smi>(*value),
242  isolate->factory()->empty_string(),
243  key),
244  false);
245  dictionary->ValueAtPut(i, *result);
246  }
247  }
248  return true;
249 }

272 bool AddDescriptorsByTemplate(
273  Isolate* isolate, DirectHandle<Map> map,
274  DirectHandle<DescriptorArray> descriptors_template,
275  DirectHandle<NumberDictionary> elements_dictionary_template,
276  DirectHandle<JSObject> receiver, RuntimeArguments& args) {
277  int nof_descriptors = descriptors_template->number_of_descriptors();
278 
279  DirectHandle<DescriptorArray> descriptors =
280  DescriptorArray::Allocate(isolate, nof_descriptors, 0);
281 
282  DirectHandle<NumberDictionary> elements_dictionary =
283  *elements_dictionary_template ==
284  ReadOnlyRoots(isolate).empty_slow_element_dictionary()
285  ? elements_dictionary_template
286  : ShallowCopyDictionaryTemplate(isolate,
287  elements_dictionary_template);
288 
289  // Count the number of properties that must be in the instance and
290  // create the property array to hold the constants.
291  int count = 0;
292  for (InternalIndex i : InternalIndex::Range(nof_descriptors)) {
293  PropertyDetails details = descriptors_template->GetDetails(i);
294  if (details.location() == PropertyLocation::kDescriptor &&
295  details.kind() == PropertyKind::kData) {
296  count++;
297  }
298  }
299  DirectHandle<PropertyArray> property_array =
300  isolate->factory()->NewPropertyArray(count);
301 
302  // Read values from |descriptors_template| and store possibly post-processed
303  // values into "instantiated" |descriptors| array.
304  int field_index = 0;
305  for (InternalIndex i : InternalIndex::Range(nof_descriptors)) {
306  Tagged<Object> value = descriptors_template->GetStrongValue(i);
307  if (IsAccessorPair(value)) {
308  DirectHandle<AccessorPair> pair = AccessorPair::Copy(
309  isolate, direct_handle(Cast<AccessorPair>(value), isolate));
310  value = *pair;
311  }
312  DisallowGarbageCollection no_gc;
313  Tagged<Name> name = descriptors_template->GetKey(i);
314  // TODO(v8:5799): consider adding a ClassBoilerplate flag
315  // "has_interesting_properties".
316  if (name->IsInteresting(isolate)) {
317  map->set_may_have_interesting_properties(true);
318  }
319  DCHECK(IsUniqueName(name));
320  PropertyDetails details = descriptors_template->GetDetails(i);
321  if (details.location() == PropertyLocation::kDescriptor) {
322  if (details.kind() == PropertyKind::kData) {
323  if (IsSmi(value)) {
324  value = GetMethodWithSharedName(isolate, args, value);
325  }
326  details = details.CopyWithRepresentation(
327  Object::OptimalRepresentation(value, isolate));
328  } else {
329  DCHECK_EQ(PropertyKind::kAccessor, details.kind());
330  if (IsAccessorPair(value)) {
331  Tagged<AccessorPair> pair = Cast<AccessorPair>(value);
332  Tagged<Object> tmp = pair->getter();
333  if (IsSmi(tmp)) {
334  pair->set_getter(GetMethodWithSharedName(isolate, args, tmp));
335  }
336  tmp = pair->setter();
337  if (IsSmi(tmp)) {
338  pair->set_setter(GetMethodWithSharedName(isolate, args, tmp));
339  }
340  }
341  }
342  } else {
343  UNREACHABLE();
344  }
345  DCHECK(Object::FitsRepresentation(value, details.representation()));
346  if (details.location() == PropertyLocation::kDescriptor &&
347  details.kind() == PropertyKind::kData) {
348  details =
349  PropertyDetails(details.kind(), details.attributes(),
350  PropertyLocation::kField, PropertyConstness::kConst,
351  details.representation(), field_index)
352  .set_pointer(details.pointer());
353 
354  property_array->set(field_index, value);
355  field_index++;
356  descriptors->Set(i, name, FieldType::Any(), details);
357  } else {
358  descriptors->Set(i, name, value, details);
359  }
360  }
361 
362  UpdateProtectors(isolate, receiver, descriptors_template);
363 
364  map->InitializeDescriptors(isolate, *descriptors);
365  if (elements_dictionary->NumberOfElements() > 0) {
366  if (!SubstituteValues<NumberDictionary>(isolate, elements_dictionary,
367  args)) {
368  return false;
369  }
370  map->set_elements_kind(DICTIONARY_ELEMENTS);
371  }
372 
373  // Atomically commit the changes.
374  receiver->set_map(isolate, *map, kReleaseStore);
375  if (elements_dictionary->NumberOfElements() > 0) {
376  receiver->set_elements(*elements_dictionary);
377  }
378  if (property_array->length() > 0) {
379  receiver->SetProperties(*property_array);
380  }
381  return true;
382 }

397 template <typename Dictionary>
398 bool AddDescriptorsByTemplate(
399  Isolate* isolate, DirectHandle<Map> map,
400  DirectHandle<Dictionary> properties_dictionary_template,
401  DirectHandle<NumberDictionary> elements_dictionary_template,
402  DirectHandle<FixedArray> computed_properties,
403  DirectHandle<JSObject> receiver, RuntimeArguments& args) {
404  int computed_properties_length = computed_properties->length();
405 
406  // Shallow-copy properties template.
407  Handle<Dictionary> properties_dictionary =
408  ShallowCopyDictionaryTemplate(isolate, properties_dictionary_template);
409  Handle<NumberDictionary> elements_dictionary =
410  ShallowCopyDictionaryTemplate(isolate, elements_dictionary_template);
411 
412  using ValueKind = ClassBoilerplate::ValueKind;
413  using ComputedEntryFlags = ClassBoilerplate::ComputedEntryFlags;
414 
415  // Merge computed properties with properties and elements dictionary
416  // templates.
417  int i = 0;
418  while (i < computed_properties_length) {
419  int flags = Smi::ToInt(computed_properties->get(i++));
420 
421  ValueKind value_kind = ComputedEntryFlags::ValueKindBits::decode(flags);
422  int key_index = ComputedEntryFlags::KeyIndexBits::decode(flags);
423  Tagged<Smi> value = Smi::FromInt(key_index + 1); // Value follows name.
424 
425  Handle<Object> key = args.at(key_index);
426  DCHECK(IsName(*key));
427  uint32_t element;
428  Handle<Name> name = Cast<Name>(key);
429  if (name->AsArrayIndex(&element)) {
430  ClassBoilerplate::AddToElementsTemplate(
431  isolate, elements_dictionary, element, key_index, value_kind, value);
432 
433  } else {
434  name = isolate->factory()->InternalizeName(name);
435  ClassBoilerplate::AddToPropertiesTemplate(
436  isolate, properties_dictionary, name, key_index, value_kind, value);
437  if (name->IsInteresting(isolate)) {
438  // TODO(pthier): Add flags to swiss dictionaries.
439  if constexpr (!V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
440  properties_dictionary->set_may_have_interesting_properties(true);
441  }
442  }
443  }
444  }
445 
446  // Replace all indices with proper methods.
447  if (!SubstituteValues<Dictionary>(isolate, properties_dictionary, args)) {
448  return false;
449  }
450 
451  UpdateProtectors(isolate, receiver,
452  DirectHandle<Dictionary>(properties_dictionary));
453 
454  if (elements_dictionary->NumberOfElements() > 0) {
455  if (!SubstituteValues<NumberDictionary>(isolate, elements_dictionary,
456  args)) {
457  return false;
458  }
459  map->set_elements_kind(DICTIONARY_ELEMENTS);
460  }
461 
462  // Atomically commit the changes.
463  receiver->set_map(isolate, *map, kReleaseStore);
464  receiver->set_raw_properties_or_hash(*properties_dictionary, kRelaxedStore);
465  if (elements_dictionary->NumberOfElements() > 0) {
466  receiver->set_elements(*elements_dictionary);
467  }
468  return true;
469 }

482 bool InitClassPrototype(Isolate* isolate,
483  DirectHandle<ClassBoilerplate> class_boilerplate,
484  DirectHandle<JSObject> prototype,
485  DirectHandle<JSPrototype> prototype_parent,
486  DirectHandle<JSFunction> constructor,
487  RuntimeArguments& args) {
488  DirectHandle<Map> map(prototype->map(), isolate);
489  map = Map::CopyDropDescriptors(isolate, map);
490  map->set_is_prototype_map(true);
491  Map::SetPrototype(isolate, map, prototype_parent);
492  isolate->UpdateProtectorsOnSetPrototype(prototype, prototype_parent);
493  constructor->set_prototype_or_initial_map(*prototype, kReleaseStore);
494  map->SetConstructor(*constructor);
495  DirectHandle<FixedArray> computed_properties(
496  class_boilerplate->instance_computed_properties(), isolate);
497  DirectHandle<NumberDictionary> elements_dictionary_template(
498  Cast<NumberDictionary>(class_boilerplate->instance_elements_template()),
499  isolate);
500 
501  Handle<Object> properties_template(
502  class_boilerplate->instance_properties_template(), isolate);
503 
504  if (IsDescriptorArray(*properties_template)) {
505  auto descriptors_template = Cast<DescriptorArray>(properties_template);
506 
507  // The size of the prototype object is known at this point.
508  // So we can create it now and then add the rest instance methods to the
509  // map.
510  return AddDescriptorsByTemplate(isolate, map, descriptors_template,
511  elements_dictionary_template, prototype,
512  args);
513  } else {
514  map->set_is_dictionary_map(true);
515  map->set_is_migration_target(false);
516  map->set_may_have_interesting_properties(true);
517  map->set_construction_counter(Map::kNoSlackTracking);
518 
519  DirectHandle<PropertyDictionary> properties_dictionary_template =
520  Cast<PropertyDictionary>(properties_template);
521  return AddDescriptorsByTemplate(
522  isolate, map, properties_dictionary_template,
523  elements_dictionary_template, computed_properties, prototype, args);
524  }
525 }

527 bool InitClassConstructor(Isolate* isolate,
528  DirectHandle<ClassBoilerplate> class_boilerplate,
529  DirectHandle<JSPrototype> constructor_parent,
530  DirectHandle<JSFunction> constructor,
531  RuntimeArguments& args) {
532  DirectHandle<Map> map(constructor->map(), isolate);
533  map = Map::CopyDropDescriptors(isolate, map);
534  DCHECK(map->is_prototype_map());
535 
536  if (!constructor_parent.is_null()) {
537  // Set map's prototype without enabling prototype setup mode for superclass
538  // because it does not make sense.
539  Map::SetPrototype(isolate, map, constructor_parent, false);
540  // Ensure that setup mode will never be enabled for superclass.
541  JSObject::MakePrototypesFast(constructor_parent, kStartAtReceiver, isolate);
542  }
543 
544  DirectHandle<NumberDictionary> elements_dictionary_template(
545  Cast<NumberDictionary>(class_boilerplate->static_elements_template()),
546  isolate);
547  DirectHandle<FixedArray> computed_properties(
548  class_boilerplate->static_computed_properties(), isolate);
549 
550  Handle<Object> properties_template(
551  class_boilerplate->static_properties_template(), isolate);
552 
553  if (IsDescriptorArray(*properties_template)) {
554  auto descriptors_template = Cast<DescriptorArray>(properties_template);
555 
556  return AddDescriptorsByTemplate(isolate, map, descriptors_template,
557  elements_dictionary_template, constructor,
558  args);
559  } else {
560  map->set_is_dictionary_map(true);
561  map->InitializeDescriptors(isolate,
562  ReadOnlyRoots(isolate).empty_descriptor_array());
563  map->set_is_migration_target(false);
564  map->set_may_have_interesting_properties(true);
565  map->set_construction_counter(Map::kNoSlackTracking);
566 
567  if constexpr (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
568  DirectHandle<SwissNameDictionary> properties_dictionary_template =
569  Cast<SwissNameDictionary>(properties_template);
570 
571  return AddDescriptorsByTemplate(
572  isolate, map, properties_dictionary_template,
573  elements_dictionary_template, computed_properties, constructor, args);
574  } else {
575  DirectHandle<NameDictionary> properties_dictionary_template =
576  Cast<NameDictionary>(properties_template);
577  return AddDescriptorsByTemplate(
578  isolate, map, properties_dictionary_template,
579  elements_dictionary_template, computed_properties, constructor, args);
580  }
581  }
582 }

584 MaybeDirectHandle<Object> DefineClass(
585  Isolate* isolate, DirectHandle<ClassBoilerplate> class_boilerplate,
586  DirectHandle<Object> super_class, DirectHandle<JSFunction> constructor,
587  RuntimeArguments& args) {
588  DirectHandle<JSPrototype> prototype_parent;
589  DirectHandle<JSPrototype> constructor_parent;
590 
591  if (IsTheHole(*super_class, isolate)) {
592  prototype_parent = isolate->initial_object_prototype();
593  } else {
594  if (IsNull(*super_class, isolate)) {
595  prototype_parent = isolate->factory()->null_value();
596  } else if (IsConstructor(*super_class)) {
597  DCHECK(!IsJSFunction(*super_class) ||
598  !IsResumableFunction(
599  Cast<JSFunction>(super_class)->shared()->kind()));
600  DirectHandle<Object> maybe_prototype_parent;
601  ASSIGN_RETURN_ON_EXCEPTION(
602  isolate, maybe_prototype_parent,
603  Runtime::GetObjectProperty(isolate, Cast<JSAny>(super_class),
604  isolate->factory()->prototype_string()));
605  if (!TryCast(maybe_prototype_parent, &prototype_parent)) {
606  THROW_NEW_ERROR(
607  isolate, NewTypeError(MessageTemplate::kPrototypeParentNotAnObject,
608  maybe_prototype_parent));
609  }
610  // Create new handle to avoid |constructor_parent| corruption because of
611  // |super_class| handle value overwriting via storing to
612  // args[ClassBoilerplate::kPrototypeArgumentIndex] below.
613  constructor_parent =
614  direct_handle(Cast<JSPrototype>(*super_class), isolate);
615  } else {
616  THROW_NEW_ERROR(isolate,
617  NewTypeError(MessageTemplate::kExtendsValueNotConstructor,
618  super_class));
619  }
620  }
621 
622  DirectHandle<JSObject> prototype = CreateClassPrototype(isolate);
623  DCHECK_EQ(*constructor, args[ClassBoilerplate::kConstructorArgumentIndex]);
624  // Temporarily change ClassBoilerplate::kPrototypeArgumentIndex for the
625  // subsequent calls, but use a scope to make sure to change it back before
626  // returning, to not corrupt the caller's argument frame (in particular, for
627  // the interpreter, to not clobber the register frame).
628  RuntimeArguments::ChangeValueScope set_prototype_value_scope(
629  isolate, &args, ClassBoilerplate::kPrototypeArgumentIndex, *prototype);
630 
631  if (!InitClassConstructor(isolate, class_boilerplate, constructor_parent,
632  constructor, args) ||
