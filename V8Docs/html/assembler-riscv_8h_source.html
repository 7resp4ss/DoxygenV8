103 
104  // Register.
105  V8_INLINE explicit Operand(Register rm) : rm_(rm) {}
106 
107  // Return true if this is a register operand.
108  V8_INLINE bool is_reg() const { return rm_.is_valid(); }
109 
110  inline intptr_t immediate_for_heap_number_request() const {
111  DCHECK(rmode() == RelocInfo::FULL_EMBEDDED_OBJECT);
112  return value_.immediate;
113  }
114 
115  inline intptr_t immediate() const {
116  DCHECK(!is_reg());
117  DCHECK(!IsHeapNumberRequest());
118  return value_.immediate;
119  }
120 
121  bool IsImmediate() const { return !rm_.is_valid(); }
122 
123  HeapNumberRequest heap_number_request() const {
124  DCHECK(IsHeapNumberRequest());
125  return value_.heap_number_request;
126  }
127 
128  bool IsHeapNumberRequest() const {
129  DCHECK_IMPLIES(is_heap_number_request_, IsImmediate());
130  DCHECK_IMPLIES(is_heap_number_request_,
131  rmode_ == RelocInfo::FULL_EMBEDDED_OBJECT ||
132  rmode_ == RelocInfo::CODE_TARGET);
133  return is_heap_number_request_;
134  }
135 
136  Register rm() const { return rm_; }
137 
138  RelocInfo::Mode rmode() const { return rmode_; }
139 
140  private:
141  Register rm_;
142  union Value {
143  Value() {}
144  HeapNumberRequest heap_number_request; // if is_heap_number_request_
145  intptr_t immediate; // otherwise

148  RelocInfo::Mode rmode_;
