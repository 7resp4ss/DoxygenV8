57 // Save the result of two uint64_t subtraction.
58 class OffsetDiff {
59  public:
60  OffsetDiff(uint64_t offset1, uint64_t offset2) {
61  if (offset1 >= offset2) {
62  abs_diff_ = offset1 - offset2;
63  negative_ = false;
64  } else {
65  abs_diff_ = offset2 - offset1;
66  negative_ = true;
67  }
68  }
69 
70  bool operator==(int64_t other) const {
71  return (negative_ == (other < 0)) &&
72  (abs_diff_ == static_cast<uint64_t>(abs(other)));
73  }
74  bool negative() const { return negative_; }
75 
76  private:
77  uint64_t abs_diff_; // abs(a-b)
78  bool negative_; // sign(a-b)
79 };

81 // This class is the wrapper for StoreOp/LoadOp, which is helpful to calcualte
82 // the relative offset between two StoreOp/LoadOp.
83 template <typename Op,
84  typename = std::enable_if_t<
85  std::is_same_v<Op, StoreOp> || std::is_same_v<Op, LoadOp> ||
86  std::is_same_v<Op, Simd128LoadTransformOp>>>
87 class StoreLoadInfo {
88  public:
89  StoreLoadInfo(const Graph* graph, const Op* op)
90  : op_(op), offset_(op->offset) {
91  base_ = &graph->Get(op->base());
92  if constexpr (std::is_same_v<Op, Simd128LoadTransformOp>) {
93  DCHECK_EQ(offset_, 0);
94  const WordBinopOp* add_op = base_->TryCast<WordBinopOp>();
95  if (!add_op || add_op->kind != WordBinopOp::Kind::kAdd ||
96  add_op->rep != WordRepresentation::Word64()) {
97  SetInvalid();
98  return;
99  }
100  base_ = &graph->Get(add_op->left());
101  const ConstantOp* const_op =
102  graph->Get(add_op->right()).TryCast<ConstantOp>();
103  if (!const_op) {
104  SetInvalid();
105  return;
106  }
107  offset_ = const_op->word64();
108  index_ = &(graph->Get(op->index()));
109  } else {
110  if (!op->index().has_value()) {
111  SetInvalid();
112  return;
113  }
114  index_ = &(graph->Get(op->index().value()));
115  }
116 
117  if (const ChangeOp* change_op = index_->TryCast<ChangeOp>()) {
118  if (change_op->kind != ChangeOp::Kind::kZeroExtend) {
119  TRACE("ChangeOp kind not supported for revectorization\n");
120  SetInvalid();
121  return;
122  }
123 
124  index_ = &graph->Get(change_op->input());
125  // If index_ is constant, add the constant to offset_ and set index_ to
126  // nullptr
127  if (const ConstantOp* const_op = index_->TryCast<ConstantOp>()) {
128  DCHECK_EQ(const_op->kind, ConstantOp::Kind::kWord32);
129  // Exceed uint32 limits.
130  if (offset_ > std::numeric_limits<uint32_t>::max() -
131  static_cast<uint32_t>(const_op->word32())) {
132  SetInvalid();
133  return;
134  }
135  offset_ += static_cast<uint32_t>(const_op->word32());
136  index_ = nullptr;
137  }
138  } else { // memory64
139  if (const ConstantOp* const_op = index_->TryCast<ConstantOp>()) {
140  DCHECK_EQ(const_op->kind, ConstantOp::Kind::kWord64);
141  // Exceed uint64 limits.
142  if (offset_ > std::numeric_limits<uint64_t>::max() -
143  static_cast<uint64_t>(const_op->word64())) {
144  SetInvalid();
145  return;
146  }
147  offset_ += static_cast<uint64_t>(const_op->word64());
148  index_ = nullptr;
149  }
150  }
151  }
152 
153  std::optional<OffsetDiff> operator-(const StoreLoadInfo<Op>& rhs) const {
154  DCHECK(IsValid() && rhs.IsValid());
155  bool calculatable = base_ == rhs.base_ && index_ == rhs.index_;
156 
157  if constexpr (std::is_same_v<Op, Simd128LoadTransformOp>) {
158  calculatable &= (op_->load_kind == rhs.op_->load_kind &&
159  op_->transform_kind == rhs.op_->transform_kind);
160  } else {
161  calculatable &= (op_->kind == rhs.op_->kind);
162  }
163 
164  if constexpr (std::is_same_v<Op, StoreOp>) {
165  // TODO(v8:12716) If one store has a full write barrier and the other has
166  // no write barrier, consider combine them with a full write barrier.
167  calculatable &= (op_->write_barrier == rhs.op_->write_barrier);
168  }
169 
170  if (calculatable) {
171  return OffsetDiff(offset_, rhs.offset_);
172  }
173  return {};
174  }
175 
176  bool IsValid() const { return op_ != nullptr; }
177 
178  const Operation* index() const { return index_; }
179  uint64_t offset() const { return offset_; }
180  const Op* op() const { return op_; }
181 
182  private:
183  void SetInvalid() { op_ = nullptr; }
184 
185  const Op* op_;
186  const Operation* base_ = nullptr;
187  const Operation* index_ = nullptr;
188  uint64_t offset_;
189 };

294 bool SLPTree::HasInputDependencies(const NodeGroup& node_group) {
295  DCHECK_EQ(node_group.size(), 2);
296  if (node_group[0] == node_group[1]) return false;
297  OpIndex start, end;
298  if (node_group[0] < node_group[1]) {
299  start = node_group[0];
300  end = node_group[1];
301  } else {
302  start = node_group[1];
303  end = node_group[0];
304  }
305  // Do BFS from the end node and see if there is a path to the start node.
306  ZoneQueue<OpIndex> to_visit(phase_zone_);
307  to_visit.push(end);
308  while (!to_visit.empty()) {
309  OpIndex to_visit_node = to_visit.front();
310  Operation& op = graph_.Get(to_visit_node);
311  to_visit.pop();
312  for (OpIndex input : op.inputs()) {
313  if (input == start) {
314  return true;
315  } else if (input > start) {
316  // We should ensure that there is no back edge.
317  DCHECK_LT(input, to_visit_node);
318  to_visit.push(input);
319  }
320  }
321  }
322  return false;
323 }

336 PackNode* SLPTree::NewForcePackNode(const NodeGroup& node_group,
337  ForcePackNode::ForcePackType type,
338  const Graph& graph) {
339  // Currently we only support force packing two nodes.
340  DCHECK_EQ(node_group.size(), 2);
341  // We should guarantee that the one node in the NodeGroup does not rely on the
342  // result of the other. Because it is costly to force pack such candidates.
343  // For example, we have four nodes {A, B, C, D} which are connected by input
344  // edges: A <-- B <-- C <-- D. If {B} and {D} are already packed into a
345  // PackNode and we want to force pack {A} and {C}, we need to duplicate {B}
346  // and the result will be {A, B, C}, {B, D}. This increase the cost of
347  // ForcePack so currently we do not support it.
348  if (HasInputDependencies(node_group)) {
349  TRACE("ForcePackNode %s(#%d, #%d) failed due to input dependencies.\n",
350  GetSimdOpcodeName(graph_.Get(node_group[0])).c_str(),
351  node_group[0].id(), node_group[1].id());
352  return nullptr;
353  }
354 
355  TRACE("ForcePackNode %s(#%d, #%d)\n",
356  GetSimdOpcodeName(graph_.Get(node_group[0])).c_str(),
357  node_group[0].id(), node_group[1].id());
358  ForcePackNode* pnode =
359  phase_zone_->New<ForcePackNode>(phase_zone_, node_group, type);
360  for (OpIndex node : node_group) {
361  node_to_packnode_[node] = pnode;

478 ShufflePackNode* SLPTree::Try256ShuffleMatchLoad8x8U(
479  const NodeGroup& node_group, const uint8_t* shuffle0,
480  const uint8_t* shuffle1) {
481  uint8_t shuffle_copy0[kSimd128Size];
482  uint8_t shuffle_copy1[kSimd128Size];
483 
484  V<Simd128> op_idx0 = node_group[0];
485  V<Simd128> op_idx1 = node_group[1];
486  const Simd128ShuffleOp& op0 = graph_.Get(op_idx0).Cast<Simd128ShuffleOp>();
487  const Simd128ShuffleOp& op1 = graph_.Get(op_idx1).Cast<Simd128ShuffleOp>();
488 
489  if (op0.kind != Simd128ShuffleOp::Kind::kI8x16 ||
490  op1.kind != Simd128ShuffleOp::Kind::kI8x16) {
491  return nullptr;
492  }
493 
494  if (op0.left() == op0.right() || op1.left() == op1.right()) {
495  // Here shuffle couldn't be swizzle
496  return nullptr;
497  }
498 
499  CopyChars(shuffle_copy0, shuffle0, kSimd128Size);
500  CopyChars(shuffle_copy1, shuffle1, kSimd128Size);
501 
502  bool need_swap, is_swizzle;
503 
504 #define CANONICALIZE_SHUFFLE(n) \
505  wasm::SimdShuffle::CanonicalizeShuffle(false, shuffle_copy##n, &need_swap, \
506  &is_swizzle); \
507  if (is_swizzle) { \
508  /* Here shuffle couldn't be swizzle*/ \
509  return nullptr; \
510  } \
511  V<Simd128> shuffle##n##_left_idx = need_swap ? op##n.right() : op##n.left(); \
512  V<Simd128> shuffle##n##_right_idx = need_swap ? op##n.left() : op##n.right();
513 
514  CANONICALIZE_SHUFFLE(0);
515  CANONICALIZE_SHUFFLE(1);
516 
517 #undef CANONICALIZE_SHUFFLE
518  if (shuffle0_left_idx != shuffle1_left_idx) {
519  // Not the same left
520  return nullptr;
521  }
522 
523  const Simd128LoadTransformOp* load_transform =
524  graph_.Get(shuffle0_left_idx).TryCast<Simd128LoadTransformOp>();
525 
526  if (!load_transform) {
527  // shuffle left is not Simd128LoadTransformOp
528  return nullptr;
529  }
530 
531  Simd128ConstantOp* shuffle0_const =
532  graph_.Get(shuffle0_right_idx).TryCast<Simd128ConstantOp>();
533  Simd128ConstantOp* shuffle1_const =
534  graph_.Get(shuffle1_right_idx).TryCast<Simd128ConstantOp>();
535 
536  if (!shuffle0_const || !shuffle1_const || !shuffle0_const->IsZero() ||
537  !shuffle1_const->IsZero()) {
538  // Shuffle right is not zero
539  return nullptr;
540  }
541 
542  if (load_transform->transform_kind ==
543  Simd128LoadTransformOp::TransformKind::k64Zero) {
544  /*
545  should look like this:
546  shuffle0 = 0,x,x,x, 1,x,x,x 2,x,x,x 3,x,x,x
547  shuffle1 = 4,x,x,x, 5,x,x,x 6,x,x,x 7,x,x,x
548  x >= 16
549  */
550 
551  for (int i = 0; i < kSimd128Size / 4; ++i) {
552  if (shuffle_copy0[i * 4] != i || shuffle_copy1[i * 4] != i + 4) {
553  // not match
554  return nullptr;
555  }
556 
557  if (shuffle_copy0[i * 4 + 1] < kSimd128Size ||
558  shuffle_copy0[i * 4 + 2] < kSimd128Size ||
559  shuffle_copy0[i * 4 + 3] < kSimd128Size ||
560  shuffle_copy1[i * 4 + 1] < kSimd128Size ||
561  shuffle_copy1[i * 4 + 2] < kSimd128Size ||
562  shuffle_copy1[i * 4 + 3] < kSimd128Size) {
563  // not match
564  return nullptr;
565  }
566  }
567  TRACE("match load extend 8x8->32x8\n");
568  return NewShufflePackNode(
569  node_group, ShufflePackNode::SpecificInfo::Kind::kS256Load8x8U);
570  }
571  return nullptr;
572 }

666 // (local.get 0))))))

669 // (local.get 7))))

674 // (i8x16.extract_lane_u 3
675 // (local.get 7))))
676 std::optional<SLPTree::ExtendIntToF32x4Info>
677 SLPTree::TryGetExtendIntToF32x4Info(OpIndex index) {
678  OpIndex current = index;
679  LaneExtendInfo lane_extend_info[4];
680 
681  // Get information for lane 1 to lane 3
682  for (int lane_index = 3; lane_index > 0; lane_index--) {
683  const Simd128ReplaceLaneOp* replace_lane =
684  graph_.Get(current)
685  .TryCast<turboshaft::Opmask::kSimd128ReplaceLaneF32x4>();
686  if (!replace_lane) {
687  TRACE("Mismatch in replace lane\n");
688  return {};
689  }
690  const ChangeOp* change =
691  graph_.Get(replace_lane->new_lane()).TryCast<ChangeOp>();
692  if (!change) {
693  TRACE("Mismatch in type convert\n");
694  return {};
695  }
696  const Simd128ExtractLaneOp* extract_lane =
697  graph_.Get(change->input()).TryCast<Simd128ExtractLaneOp>();
698  if (!extract_lane) {
699  TRACE("Mismatch in extract lane\n");
700  return {};
701  }
702  lane_extend_info[lane_index].replace_lane_index = replace_lane->lane;
703  lane_extend_info[lane_index].change_kind = change->kind;
704  lane_extend_info[lane_index].extract_from = extract_lane->input();
705  lane_extend_info[lane_index].extract_kind = extract_lane->kind;
706  lane_extend_info[lane_index].extract_lane_index = extract_lane->lane;
707 
708  current = replace_lane->into();
709  }
710 
711  // Get information for lane 0(splat)
712  const Simd128SplatOp* splat = graph_.Get(current).TryCast<Simd128SplatOp>();
713  if (!splat) {
714  TRACE("Mismatch in splat\n");
715  return {};
716  }
717  const ChangeOp* change = graph_.Get(splat->input()).TryCast<ChangeOp>();
718  if (!change) {
719  TRACE("Mismatch in splat type convert\n");
720  return {};
721  }
722  const Simd128ExtractLaneOp* extract_lane =
723  graph_.Get(change->input()).TryCast<Simd128ExtractLaneOp>();
724  if (!extract_lane) {
725  TRACE("Mismatch in splat extract lane\n");
726  return {};
727  }
728  lane_extend_info[0].replace_lane_index = 0;
729  lane_extend_info[0].change_kind = change->kind;
730  lane_extend_info[0].extract_from = extract_lane->input();
731  lane_extend_info[0].extract_kind = extract_lane->kind;
732  lane_extend_info[0].extract_lane_index = extract_lane->lane;
733 
734  // Pattern matching for f32x4.convert_i32x4(i32x4.extract_lane)
735  for (int i = 0; i < 4; i++) {
736  if (lane_extend_info[i].replace_lane_index != i) {
737  return {};
738  }
739  if (lane_extend_info[i].change_kind != lane_extend_info[0].change_kind ||
740  (lane_extend_info[i].change_kind != ChangeOp::Kind::kSignedToFloat &&
741  lane_extend_info[i].change_kind != ChangeOp::Kind::kUnsignedToFloat)) {
742  return {};
743  }
744  if (lane_extend_info[i].extract_from != lane_extend_info[0].extract_from) {
745  return {};
746  }
747  if (lane_extend_info[i].extract_kind != lane_extend_info[0].extract_kind ||
748  (lane_extend_info[i].extract_kind !=
749  Simd128ExtractLaneOp::Kind::kI8x16S &&
750  lane_extend_info[i].extract_kind !=
751  Simd128ExtractLaneOp::Kind::kI8x16U &&
752  lane_extend_info[i].extract_kind !=
753  Simd128ExtractLaneOp::Kind::kI16x8S &&
754  lane_extend_info[i].extract_kind !=
755  Simd128ExtractLaneOp::Kind::kI16x8U)) {
756  return {};
757  }
758  if (lane_extend_info[i].extract_lane_index !=
759  lane_extend_info[0].extract_lane_index + i) {
760  return {};
761  }
762  }
763 
764  ExtendIntToF32x4Info info;
765  info.extend_from = lane_extend_info[0].extract_from;
766  info.start_lane = lane_extend_info[0].extract_lane_index;
767  if (lane_extend_info[0].extract_kind == Simd128ExtractLaneOp::Kind::kI8x16S ||
768  lane_extend_info[0].extract_kind == Simd128ExtractLaneOp::Kind::kI8x16U) {
769  info.lane_size = 1;
770  } else {
771  info.lane_size = 2;
772  }
773  info.is_sign_extract =
774  lane_extend_info[0].extract_kind == Simd128ExtractLaneOp::Kind::kI8x16S ||
775  lane_extend_info[0].extract_kind == Simd128ExtractLaneOp::Kind::kI16x8S;
776  info.is_sign_convert =
777  lane_extend_info[0].change_kind == ChangeOp::Kind::kSignedToFloat;
778 
779  return info;

820 bool SLPTree::IsSideEffectFree(OpIndex first, OpIndex second) {
821  DCHECK_LE(first.offset(), second.offset());
822  if (first == second) return true;
823  OpEffects effects = graph().Get(second).Effects();
824  OpIndex prev_node = graph().PreviousIndex(second);
825  while (prev_node != first) {
826  OpEffects prev_effects = graph().Get(prev_node).Effects();
827  if ((graph().Get(second).IsProtectedLoad() &&
828  graph().Get(prev_node).IsProtectedLoad())
829  ? CannotSwapProtectedLoads(prev_effects, effects)

831  TRACE("break side effect %d, %d\n", prev_node.id(), second.id());
832  return false;
833  }
834  prev_node = graph().PreviousIndex(prev_node);
835  }
836  return true;

923 PackNode* SLPTree::BuildTreeRec(const NodeGroup& node_group,
924  unsigned recursion_depth) {
925  DCHECK_EQ(node_group.size(), 2);
926 
927  OpIndex node0 = node_group[0];
928  OpIndex node1 = node_group[1];
929  Operation& op0 = graph_.Get(node0);
930  Operation& op1 = graph_.Get(node1);
931 
932  if (recursion_depth == RecursionMaxDepth) {
933  TRACE("Failed due to max recursion depth!\n");
934  return nullptr;
935  }
936 
937  if (!CanBePacked(node_group)) {
938  return nullptr;
939  }
940 
941  // Check if this is a duplicate of another entry.
942  bool is_intersected = false;
943  // For revisited node_group, we only need to match from node0.
944  if (PackNode* pnode = GetPackNode(node0)) {
945  const Operation& op = graph_.Get(node0);
946  if (pnode->IsSame(node_group)) {
947  TRACE("Perfect diamond merge at #%d,%s\n", node0.id(),
948  GetSimdOpcodeName(op).c_str());
949  return pnode;
950  }
951 
952  // TODO(yolanda): Support other intersect PackNode e.g. overlapped loads.
953  if (!pnode->IsForcePackNode() || recursion_depth < 1) {
954  TRACE("Unsupported partial overlap at #%d,%s!\n", node0.id(),
955  GetSimdOpcodeName(op).c_str());
956  return nullptr;
957  }
958 
959  // Match intersect packnodes from current tree.
960  if (auto intersect_packnodes = GetIntersectPackNodes(node0)) {
961  for (auto intersect_pnode : *intersect_packnodes) {
962  if (intersect_pnode->IsSame(node_group)) {
963  TRACE("Perfect diamond merge at intersect pack node #%d,%s, #%d\n",
964  node0.id(), GetSimdOpcodeName(op).c_str(), node1.id());
965  return intersect_pnode;
966  }
967  }
968  }
969 
970  // Match intersect packnodes from analyzer
971  if (auto intersect_packnodes = analyzer_->GetIntersectPackNodes(node0)) {
972  for (auto intersect_pnode : *intersect_packnodes) {
973  if (intersect_pnode->IsSame(node_group)) {
974  TRACE("Perfect diamond merge at intersect pack node #%d,%s, #%d\n",
975  node0.id(), GetSimdOpcodeName(op).c_str(), node1.id());
976  return intersect_pnode;
977  }
978  }
979  }
980 
981  is_intersected = true;
982  TRACE("Partial overlap at #%d,%s!\n", node0.id(),
983  GetSimdOpcodeName(op).c_str());
984  }
985 
986  // Catch overlapped PackNode on the other nodes.
987  if (!is_intersected) {
988  for (int i = 1; i < static_cast<int>(node_group.size()); i++) {
989  const OpIndex op_idx = node_group[i];
990  const Operation& op = graph_.Get(op_idx);
991  if (auto pnode = GetPackNode(op_idx)) {
992  if (!pnode->IsForcePackNode()) {
993  TRACE("Unsupported partial overlap at #%d,%s!\n", op_idx.id(),
994  GetSimdOpcodeName(op).c_str());
995  return nullptr;
996  }
997  } else if (!GetIntersectPackNodes(op_idx) &&
998  !analyzer_->GetIntersectPackNodes(op_idx)) {
999  continue;
1000  }
1001 
1002  is_intersected = true;
1003  TRACE("Partial overlap at #%d,%s!\n", op_idx.id(),
1004  GetSimdOpcodeName(op).c_str());
1005  break;
1006  }
1007  }
1008 
1009  if (is_intersected) {
1010  TRACE("Create IntersectPackNode due to partial overlap!\n");
1011  PackNode* pnode = NewIntersectPackNode(node_group);
1012  return pnode;
1013  }
1014 
1015  int value_in_count = op0.input_count;
1016 
1017  switch (op0.opcode) {
1018  case Opcode::kSimd128Constant: {
1019  PackNode* p = NewPackNode(node_group);
1020  return p;
1021  }
1022 
1023  case Opcode::kSimd128LoadTransform: {
1024  const Simd128LoadTransformOp& transform_op0 =
1025  op0.Cast<Simd128LoadTransformOp>();
1026  const Simd128LoadTransformOp& transform_op1 =
1027  op1.Cast<Simd128LoadTransformOp>();
1028  StoreLoadInfo<Simd128LoadTransformOp> info0(&graph_, &transform_op0);
1029  StoreLoadInfo<Simd128LoadTransformOp> info1(&graph_, &transform_op1);
1030  if (!info0.IsValid() || !info1.IsValid()) {
1031  return NewForcePackNode(node_group, ForcePackNode::kGeneral, graph_);
1032  }
1033  auto stride = info1 - info0;
1034  if (IsLoadSplat(transform_op0)) {
1035  TRACE("Simd128LoadTransform: LoadSplat\n");
1036  if (IsSplat(node_group) ||
1037  (stride.has_value() && stride.value() == 0)) {
1038  return NewPackNode(node_group);
1039  }
1040  return NewForcePackNode(node_group, ForcePackNode::kGeneral, graph_);
1041  } else if (IsLoadExtend(transform_op0)) {
1042  TRACE("Simd128LoadTransform: LoadExtend\n");
1043  if (stride.has_value()) {
1044  const OffsetDiff value = stride.value();
1045  if (value == kSimd128Size / 2) {
1046  return NewPackNode(node_group);
1047  } else if (value == 0) {
1048  return NewForcePackNode(node_group, ForcePackNode::kSplat, graph_);
1049  }
1050  }
1051  return NewForcePackNode(node_group, ForcePackNode::kGeneral, graph_);
1052  } else {
1053  TRACE("Load Transfrom k64Zero/k32Zero!\n");
1054  DCHECK(transform_op0.transform_kind ==
1055  Simd128LoadTransformOp::TransformKind::k32Zero ||
1056  transform_op0.transform_kind ==
1057  Simd128LoadTransformOp::TransformKind::k64Zero);
1058  if (stride.has_value() && stride.value() == 0) {
1059  return NewForcePackNode(node_group, ForcePackNode::kSplat, graph_);
1060  }
1061  return NewForcePackNode(node_group, ForcePackNode::kGeneral, graph_);
1062  }
1063  }
1064 
1065  case Opcode::kLoad: {
1066  TRACE("Load leaf node\n");
1067  const LoadOp& load0 = op0.Cast<LoadOp>();
1068  const LoadOp& load1 = op1.Cast<LoadOp>();
1069  if (load0.loaded_rep != MemoryRepresentation::Simd128() ||
1070  load1.loaded_rep != MemoryRepresentation::Simd128()) {
1071  TRACE("Failed due to non-simd load representation!\n");
1072  return nullptr;
1073  }
1074  StoreLoadInfo<LoadOp> info0(&graph_, &load0);
1075  StoreLoadInfo<LoadOp> info1(&graph_, &load1);
1076  if (info0.IsValid() && info1.IsValid()) {
1077  auto stride = info1 - info0;
1078  if (stride.has_value()) {
1079  if (const OffsetDiff value = stride.value(); value == kSimd128Size) {
1080  // TODO(jiepan) Sort load
1081  return NewPackNode(node_group);
1082  } else if (value == 0) {
1083  return NewForcePackNode(node_group, ForcePackNode::kSplat, graph_);
1084  }
1085  }
1086  }
1087  return NewForcePackNode(node_group, ForcePackNode::kGeneral, graph_);
1088  }
1089  case Opcode::kStore: {
1090  TRACE("Added a vector of stores.\n");
1091  // input: base, value, [index]
1092  PackNode* pnode = NewPackNodeAndRecurs(node_group, 1, 1, recursion_depth);
1093  return pnode;
1094  }
1095  case Opcode::kPhi: {
1096  TRACE("Added a vector of phi nodes.\n");
1097  const PhiOp& phi = graph().Get(node0).Cast<PhiOp>();
1098  if (phi.rep != RegisterRepresentation::Simd128() ||
1099  op0.input_count != op1.input_count) {
1100  TRACE("Failed due to invalid phi\n");
1101  return nullptr;
1102  }
1103  PackNode* pnode =
1104  NewPackNodeAndRecurs(node_group, 0, value_in_count, recursion_depth);
1105  return pnode;
1106  }
1107  case Opcode::kSimd128Unary: {
1108 #define UNARY_CASE(op_128, not_used) case Simd128UnaryOp::Kind::k##op_128:
1109 #define UNARY_SIGN_EXTENSION_CASE(op_low, not_used1, op_high) \
1110  case Simd128UnaryOp::Kind::k##op_low: { \
1111  if (const Simd128UnaryOp* unop1 = \
1112  op1.TryCast<Opmask::kSimd128##op_high>(); \
1113  unop1 && op0.Cast<Simd128UnaryOp>().input() == unop1->input()) { \
1114  return NewPackNode(node_group); \
1115  } \
1116  [[fallthrough]]; \
1117  } \
1118  case Simd128UnaryOp::Kind::k##op_high: { \
1119  if (op1.Cast<Simd128UnaryOp>().kind == op0.Cast<Simd128UnaryOp>().kind) { \
1120  auto force_pack_type = \
1121  node0 == node1 ? ForcePackNode::kSplat : ForcePackNode::kGeneral; \
1122  return NewForcePackNode(node_group, force_pack_type, graph_); \
1123  } else { \
1124  return nullptr; \
1125  } \
1126  }
1127  switch (op0.Cast<Simd128UnaryOp>().kind) {
1128  SIMD256_UNARY_SIGN_EXTENSION_OP(UNARY_SIGN_EXTENSION_CASE)
1129  SIMD256_UNARY_SIMPLE_OP(UNARY_CASE) {
1130  TRACE("Added a vector of Unary\n");
1131  PackNode* pnode = NewPackNodeAndRecurs(node_group, 0, value_in_count,
1132  recursion_depth);
1133  return pnode;
1134  }
1135  default: {
1136  TRACE("Unsupported Simd128Unary: %s\n",
1137  GetSimdOpcodeName(op0).c_str());
1138  return nullptr;
1139  }
1140  }
1141 #undef UNARY_CASE
1142 #undef UNARY_SIGN_EXTENSION_CASE
1143  }
1144  case Opcode::kSimd128Binop: {
1145 #define BINOP_CASE(op_128, not_used) case Simd128BinopOp::Kind::k##op_128:
1146 #define BINOP_SIGN_EXTENSION_CASE(op_low, not_used1, op_high) \
1147  case Simd128BinopOp::Kind::k##op_low: { \

1377 void WasmRevecAnalyzer::ProcessBlock(const Block& block) {
1378  StoreInfoSet simd128_stores(phase_zone_);
1379  for (const Operation& op : base::Reversed(graph_.operations(block))) {
1380  if (const StoreOp* store_op = op.TryCast<StoreOp>()) {

1382  StoreLoadInfo<StoreOp> info(&graph_, store_op);
1383  if (info.IsValid()) {
1384  simd128_stores.insert(info);
1385  }
1386  }
1387  }
1388  // Try to find reduce op which can be used as revec seeds.
1389  if (IsSupportedReduceSeed(op)) {
1390  const Simd128BinopOp& binop = op.Cast<Simd128BinopOp>();

1392  V<Simd128> right_index = binop.right();

1394  const Operation& right_op = graph_.Get(right_index);
1395

1397  IsSameOpAndKind(left_op, right_op)) {
1398  reduce_seeds_.push_back({left_index, right_index});
1399  }
1400  }
1401  }
1402 
1403  if (simd128_stores.size() >= 2) {
1404  for (auto it = std::next(simd128_stores.cbegin()),
1405  end = simd128_stores.cend();
1406  it != end;) {
1407  const StoreLoadInfo<StoreOp>& info0 = *std::prev(it);
1408  const StoreLoadInfo<StoreOp>& info1 = *it;
1409  auto diff = info1 - info0;
1410

1412  const OffsetDiff value = diff.value();
1413  DCHECK_EQ(value.negative(), false);
1414  if (value == kSimd128Size) {
1415  store_seeds_.push_back(
1416  {graph_.Index(*info0.op()), graph_.Index(*info1.op())});
1417  if (std::distance(it, end) < 2) {
1418  break;
1419  }
1420  std::advance(it, 2);
1421  continue;
1422  }
1423  }
1424  it++;
1425  }

1490 bool WasmRevecAnalyzer::DecideVectorize() {
1491  TRACE("Enter %s\n", __func__);
1492  int save = 0, cost = 0;
1493  ForEach(
1494  [&](PackNode const* pnode) {
1495  const NodeGroup& nodes = pnode->nodes();
1496  // An additional store is emitted in case of OOB trap at the higher
1497  // 128-bit address. Thus no save if the store at lower address is
1498  // executed first. Return directly as we dont need to check external use
1499  // for stores.
1500  if (graph_.Get(nodes[0]).opcode == Opcode::kStore) {
1501  if (nodes[0] > nodes[1]) save++;
1502  return;
1503  }
1504 
1505  if (pnode->IsForcePackNode()) {
1506  cost++;
1507  return;
1508  }
1509 
1510  // Splat nodes will not cause a saving as it simply extends itself.
1511  if (!IsSplat(nodes)) {
1512  save++;
1513  }
1514 
1515 #ifdef V8_TARGET_ARCH_X64
1516  // On x64 platform, we dont emit extract for lane 0 as the source ymm
1517  // register is alias to the corresponding xmm register in lower 128-bit.
1518  for (int i = 1; i < static_cast<int>(nodes.size()); i++) {
1519  if (nodes[i] == nodes[0]) continue;
1520 #else
1521  for (int i = 0; i < static_cast<int>(nodes.size()); i++) {
1522  if (i > 0 && nodes[i] == nodes[0]) continue;
1523 #endif // V8_TARGET_ARCH_X64
1524 
1525  for (auto use : use_map_->uses(nodes[i])) {
1526  if (!GetPackNode(use) || GetPackNode(use)->is_force_packing()) {
1527  TRACE("External use edge: (%d:%s) -> (%d:%s)\n", use.id(),
1528  OpcodeName(graph_.Get(use).opcode), nodes[i].id(),
1529  OpcodeName(graph_.Get(nodes[i]).opcode));
1530  ++cost;
1531 
1532  // We only need one Extract node and all other uses can share.
1533  break;
1534  }
1535  }
1536  }
1537  },
1538  revectorizable_node_);
1539 
1540  ForEach(
1541  [&](PackNode const* pnode) {
1542  // We always generate SimdPack128To256Op for IntersectPackNode.
1543  cost++;
1544  return;
1545  },
1546  revectorizable_intersect_node_);
1547 
1548  TRACE("Save: %d, cost: %d\n", save, cost);
1549  return save > cost;
1550 }
