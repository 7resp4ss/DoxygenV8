139 // A Handle provides a reference to an object that survives relocation by
140 // the garbage collector.
141 //
142 // Handles are only valid within a HandleScope. When a handle is created
143 // for an object a cell is allocated in the current HandleScope.
144 //
145 // Also note that Handles do not provide default equality comparison or hashing
146 // operators on purpose. Such operators would be misleading, because intended
147 // semantics is ambiguous between Handle location and object identity. Instead
148 // use either {is_identical_to} or {location} explicitly.
149 template <typename T>
150 class Handle final : public HandleBase {
151  public:
152  // Handles denote strong references.
153  static_assert(!is_maybe_weak_v<T>);
154 
155  V8_INLINE Handle() : HandleBase(nullptr) {}
156 
157  V8_INLINE explicit Handle(Address* location) : HandleBase(location) {
158  // TODO(jkummerow): Runtime type check here as a SLOW_DCHECK?
159  }
160 
161  V8_INLINE Handle(Tagged<T> object, Isolate* isolate);
162  V8_INLINE Handle(Tagged<T> object, LocalIsolate* isolate);
163  V8_INLINE Handle(Tagged<T> object, LocalHeap* local_heap);
164 
165  // Allocate a new handle for the object.
166  V8_INLINE static Handle<T> New(Tagged<T> object, Isolate* isolate);
167 
168  // Constructor for handling automatic up casting.
169  // Ex. Handle<JSFunction> can be passed when Handle<Object> is expected.
170  template <typename S>
171  V8_INLINE Handle(Handle<S> handle)
172  requires(is_subtype_v<S, T>)
173  : HandleBase(handle) {}
174 
175  // Access a member of the T object referenced by this handle.
176  //
177  // This is actually a double dereference -- first it dereferences the Handle
178  // pointing to a Tagged<T>, and then continues through Tagged<T>::operator->.
179  // This means that this is only permitted for Tagged<T> with an operator->,
180  // i.e. for on-heap object T.
181  V8_INLINE Tagged<T> operator->() const {
182  // For non-HeapObjects, there's no on-heap object to dereference, so
183  // disallow using operator->.
184  //
185  // If you got an error here and want to access the Tagged<T>, use
186  // operator* -- e.g. for `Tagged<Smi>::value()`, use `(*handle).value()`.
187  static_assert(
188  is_subtype_v<T, HeapObject>,
189  "This handle does not reference a heap object. Use `(*handle).foo`.");
190  return **this;
191  }
192 
193  V8_INLINE Tagged<T> operator*() const {
194  // This static type check also fails for forward class declarations. We
195  // check on access instead of on construction to allow Handles to forward
196  // declared types.
197  static_assert(is_taggable_v<T>, "static type violation");
198  // Direct construction of Tagged from address, without a type check, because
199  // we rather trust Handle<T> to contain a T than include all the respective
200  // -inl.h headers for SLOW_DCHECKs.
201  SLOW_DCHECK(IsDereferenceAllowed());
202  return Tagged<T>(*location());
203  }
204 
205  // Consider declaring values that contain empty handles as
206  // MaybeHandle to force validation before being used as handles.
207  static const Handle<T> null() { return Handle<T>(); }
208 
209  // Location equality.
210  bool equals(Handle<T> other) const { return address() == other.address(); }
211 
212  // Patches this Handle's value, in-place, with a new value. All indirect
213  // handles with the same location will see this update.
214  void PatchValue(Tagged<T> new_value) {
215  SLOW_DCHECK(location_ != nullptr && IsDereferenceAllowed());
216  *location_ = new_value.ptr();
217  }
218 
219  // Provide function object for location equality comparison.
220  struct equal_to {
221  V8_INLINE bool operator()(Handle<T> lhs, Handle<T> rhs) const {
222  return lhs.equals(rhs);
223  }
224  };
225 
226  // Provide function object for location hashing.
227  struct hash {
228  V8_INLINE size_t operator()(Handle<T> const& handle) const {
229  return base::hash<Address>()(handle.address());
230  }
231  };
232 
233  using MaybeType = MaybeHandle<T>;
234 
235  private:
236  // Handles of different classes are allowed to access each other's location_.
237  template <typename>
238  friend class Handle;
239  // MaybeHandle is allowed to access location_.
240  template <typename>
241  friend class MaybeHandle;
242  // Casts are allowed to access location_.
243  template <typename To, typename From>
244  friend inline IndirectHandle<To> UncheckedCast(IndirectHandle<From> value);
245 };

251 // A stack-allocated class that governs a number of local handles.
252 // After a handle scope has been created, all local handles will be
253 // allocated within that handle scope until either the handle scope is
254 // deleted or another handle scope is created. If there is already a
255 // handle scope and a new one is created, all allocations will take
256 // place in the new handle scope until it is deleted. After that,
257 // new handles will again be allocated in the original handle scope.
258 //
259 // After the handle scope of a local handle has been deleted the
260 // garbage collector will no longer track the object stored in the
261 // handle and may deallocate it. The behavior of accessing a handle
262 // for which the handle scope has been deleted is undefined.

264  public:
265  explicit V8_INLINE HandleScope(Isolate* isolate);
266  inline HandleScope(HandleScope&& other) V8_NOEXCEPT;
267  HandleScope(const HandleScope&) = delete;
268  HandleScope& operator=(const HandleScope&) = delete;
269 
270  // Allow placement new.
271  void* operator new(size_t size, void* storage) {
272  return ::operator new(size, storage);
273  }
274 
275  // Prevent heap allocation or illegal handle scopes.
276  void* operator new(size_t size) = delete;
277  void operator delete(void* size_t) = delete;
278 
279  V8_INLINE ~HandleScope();
280 
281  inline HandleScope& operator=(HandleScope&& other) V8_NOEXCEPT;
282 
283  // Counts the number of allocated handles.
284  V8_EXPORT_PRIVATE static int NumberOfHandles(Isolate* isolate);
285 
286  // Creates a new handle with the given value.

292  static Address current_next_address(Isolate* isolate);

296  // Closes the HandleScope (invalidating all handles
297  // created in the scope of the HandleScope) and returns
298  // a Handle backed by the parent scope holding the
299  // value of the argument handle.
300  //
301  // TODO(42203211): When direct handles are enabled, the version with

329

336  V8_EXPORT_PRIVATE static void ZapRange(Address* start, Address* end,

495 // ----------------------------------------------------------------------------
496 // A DirectHandle provides a reference to an object without an intermediate
497 // pointer.
498 //
499 // A DirectHandle is a simple wrapper around a tagged pointer to a heap object
500 // or a SMI. Its methods are symmetrical with Handle, so that Handles can be
501 // easily migrated.
502 //
503 // DirectHandles are intended to be used with conservative stack scanning, as
504 // they do not provide a mechanism for keeping an object alive across a garbage
505 // collection.
506 //
507 // Further motivation is explained in the design doc:
508 // https://docs.google.com/document/d/1uRGYQM76vk1fc_aDqDH3pm2qhaJtnK2oyzeVng4cS6I/
509 template <typename T>
510 class DirectHandle : public DirectHandleBase {
511  public:
512  // Handles denote strong references.
513  static_assert(!is_maybe_weak_v<T>);
514 
515  V8_INLINE DirectHandle() : DirectHandle(kTaggedNullAddress) {}
516 
517  V8_INLINE DirectHandle(Tagged<T> object, Isolate* isolate)
518  : DirectHandle(object) {}
519  V8_INLINE DirectHandle(Tagged<T> object, LocalIsolate* isolate)
520  : DirectHandle(object) {}
521  V8_INLINE DirectHandle(Tagged<T> object, LocalHeap* local_heap)
522  : DirectHandle(object) {}
523 
524  V8_INLINE static DirectHandle FromAddress(Address object) {
525  return DirectHandle(object);
526  }
527  V8_INLINE static DirectHandle FromSlot(Address* slot) {
528  return FromAddress(slot != nullptr ? *slot : kTaggedNullAddress);
529  }
530 
531  V8_INLINE static DirectHandle<T> New(Tagged<T> object, Isolate* isolate) {
532  return DirectHandle<T>(object);
533  }
534 
535  // Constructor for handling automatic up casting.
536  // Ex. DirectHandle<JSFunction> can be passed when DirectHandle<Object> is
537  // expected.
538  template <typename S, typename = std::enable_if_t<is_subtype_v<S, T>>>
539  V8_INLINE DirectHandle(DirectHandle<S> handle) : DirectHandle(handle.obj_) {}
540 
541  template <typename S, typename = std::enable_if_t<is_subtype_v<S, T>>>
542  V8_INLINE DirectHandle(Handle<S> handle)
543  : DirectHandle(handle.location() != nullptr ? *handle.location()
544  : kTaggedNullAddress) {}
545 
546  V8_INLINE Tagged<T> operator->() const {
547  if constexpr (is_subtype_v<T, HeapObject>) {
548  return **this;
549  } else {
550  // For non-HeapObjects, there's no on-heap object to dereference, so
551  // disallow using operator->.
552  //
553  // If you got an error here and want to access the Tagged<T>, use
554  // operator* -- e.g. for `Tagged<Smi>::value()`, use `(*handle).value()`.
555  static_assert(
556  false,
557  "This handle does not reference a heap object. Use `(*handle).foo`.");
558  }
559  }
560 
561  V8_INLINE Tagged<T> operator*() const {
562  // This static type check also fails for forward class declarations. We
563  // check on access instead of on construction to allow DirectHandles to
564  // forward declared types.
565  static_assert(is_taggable_v<T>, "static type violation");
566  // Direct construction of Tagged from address, without a type check, because
567  // we rather trust DirectHandle<T> to contain a T than include all the
568  // respective -inl.h headers for SLOW_DCHECKs.
569  SLOW_DCHECK(IsDereferenceAllowed());
570  return Tagged<T>(address());
571  }
572 
573  // Consider declaring values that contain empty handles as
574  // MaybeDirectHandle to force validation before being used as handles.
575  V8_INLINE static const DirectHandle<T> null() { return DirectHandle<T>(); }
576 
577  // Address equality.
578  bool equals(DirectHandle<T> other) const {
579  return address() == other.address();
580  }
581 
582  // Sets this DirectHandle's value. This is equivalent to handle assignment,
583  // except for the check that is equivalent to that performed in
584  // Handle<T>::PatchValue.
585  // TODO(42203211): Calls to this method will eventually be replaced by direct
586  // handle assignments, when the migration to direct handles is complete.
587  void SetValue(Tagged<T> new_value) {
588  SLOW_DCHECK(obj_ != kTaggedNullAddress && IsDereferenceAllowed());
589  obj_ = new_value.ptr();
590  }
591 
592  using MaybeType = MaybeDirectHandle<T>;
593 
594  private:
595  // DirectHandles of different classes are allowed to access each other's
596  // obj_.
597  template <typename>
598  friend class DirectHandle;
599  // MaybeDirectHandle is allowed to access obj_.
600  template <typename>
601  friend class MaybeDirectHandle;
602  friend class DirectHandleUnchecked<T>;
603  // Casts are allowed to access obj_.
604  template <typename To, typename From>
605  friend inline DirectHandle<To> UncheckedCast(DirectHandle<From> value);
606 
607  V8_INLINE explicit DirectHandle(Tagged<T> object);
608  V8_INLINE explicit DirectHandle(Address object) : DirectHandleBase(object) {}
609 
610  explicit DirectHandle(no_checking_tag do_not_check)
611  : DirectHandleBase(kTaggedNullAddress, do_not_check) {}
612  explicit DirectHandle(const DirectHandle<T>& other,
613  no_checking_tag do_not_check)
614  : DirectHandleBase(other.obj_, do_not_check) {}
615 };
