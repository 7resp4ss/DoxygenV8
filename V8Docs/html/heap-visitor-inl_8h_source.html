117 template <typename ConcreteVisitor>
118 size_t HeapVisitor<ConcreteVisitor>::Visit(Tagged<Map> map,
119  Tagged<HeapObject> object,
120  MaybeObjectSize maybe_object_size) {
121  if constexpr (ConcreteVisitor::UsePrecomputedObjectSize()) {
122  DCHECK_EQ(maybe_object_size.AssumeSize(), object->SizeFromMap(map));
123  static_assert(!ConcreteVisitor::EnableConcurrentVisitation());
124  } else {
125  DCHECK(maybe_object_size.IsNone());
126  }
127 
128  ConcreteVisitor* visitor = static_cast<ConcreteVisitor*>(this);
129  switch (map->visitor_id()) {
130 #define CASE(TypeName) \
131  case kVisit##TypeName: \
132  /* If this DCHECK fails, it means that the object type wasn't added \
133  * to the TRUSTED_VISITOR_ID_LIST. \
134  * Note: This would normally be just !IsTrustedObject(obj), however we \
135  * might see trusted objects here before they've been migrated to trusted \
136  * space, hence the second condition. */ \
137  DCHECK(!InstanceTypeChecker::IsTrustedObject(map) || \
138  !HeapLayout::InTrustedSpace(object)); \
139  return visitor->Visit##TypeName( \
140  map, ConcreteVisitor::template Cast<TypeName>(object, heap_), \
141  maybe_object_size);
142  TYPED_VISITOR_ID_LIST(CASE)
143  TYPED_VISITOR_WITH_SLACK_ID_LIST(CASE)
144  TORQUE_VISITOR_ID_LIST(CASE)
145 #undef CASE
146 #define CASE(TypeName) \
147  case kVisit##TypeName: \
148  DCHECK(InstanceTypeChecker::IsTrustedObject(map)); \
149  /* Trusted objects are protected from modifications by an attacker as \
150  * they are located outside of the sandbox. However, an attacker can \
151  * still craft their own fake trusted objects inside the sandbox. In \
152  * this case, bad things might happen if these objects are then \
153  * processed by e.g. an object visitor as they will typically assume \
154  * that these objects are trustworthy. The following check defends \
155  * against that by ensuring that the object is outside of the sandbox. \
156  * See also crbug.com/c/1505089. */ \
157  SBXCHECK(OutsideSandboxOrInReadonlySpace(object)); \
158  return visitor->Visit##TypeName( \
159  map, ConcreteVisitor::template Cast<TypeName>(object, heap_), \
160  maybe_object_size);
161  TRUSTED_VISITOR_ID_LIST(CASE)
162 #undef CASE
163  case kVisitShortcutCandidate:
164  return visitor->VisitShortcutCandidate(
165  map, ConcreteVisitor::template Cast<ConsString>(object, heap_),
166  maybe_object_size);
167  case kVisitJSObjectFast:
168  return visitor->VisitJSObjectFast(
169  map, ConcreteVisitor::template Cast<JSObject>(object, heap_),
170  maybe_object_size);
171  case kVisitJSApiObject:
172  return visitor->VisitJSApiObject(
173  map, ConcreteVisitor::template Cast<JSObject>(object, heap_),
174  maybe_object_size);
175  case kVisitCppHeapExternalObject:
176  return visitor->VisitCppHeapExternalObject(
177  map,
178  ConcreteVisitor::template Cast<CppHeapExternalObject>(object, heap_),
179  maybe_object_size);
180  case kVisitStruct:
181  return visitor->VisitStruct(map, object, maybe_object_size);
182  case kVisitFiller:
183  return visitor->VisitFiller(map, object, maybe_object_size);
184  case kVisitFreeSpace:
185  return visitor->VisitFreeSpace(map, i::Cast<FreeSpace>(object),
186  maybe_object_size);
187  case kDataOnlyVisitorIdCount:
188  case kVisitorIdCount:
189  UNREACHABLE();
190  }
191  // TODO(chromium:327992715): Remove once we have some clarity why execution
192  // can reach this point.
193  {
194  Isolate* isolate;
195  if (GetIsolateFromHeapObject(object, &isolate)) {
196  isolate->PushParamsAndDie(
197  reinterpret_cast<void*>(object.ptr()),
198  reinterpret_cast<void*>(map.ptr()),
199  reinterpret_cast<void*>(static_cast<intptr_t>(map->visitor_id())));
200  }
201  }
202  UNREACHABLE();
203 }

332 template <typename ConcreteVisitor>
333 template <typename T, typename TBodyDescriptor>
334 size_t HeapVisitor<ConcreteVisitor>::VisitJSObjectSubclass(
335  Tagged<Map> map, Tagged<T> object, MaybeObjectSize maybe_object_size) {
336  // JSObject types are subject to slack tracking. At the end of slack tracking
337  // a Map's instance size is adjusted properly. Since this changes the instance
338  // size, we cannot DCHECK that `SizeFromMap()` is consistent with
339  // `TBodyDescriptor::SizeOf()` as that would require taking a snapshot of the
340  // Map.
341 
342  ConcreteVisitor* visitor = static_cast<ConcreteVisitor*>(this);
343  visitor->template VisitMapPointerIfNeeded<VisitorId::kVisitJSObject>(object);
344 
345  const size_t size = ConcreteVisitor::UsePrecomputedObjectSize()
346  ? maybe_object_size.AssumeSize()
347  : TBodyDescriptor::SizeOf(map, object);
348 
349  int visitation_size = static_cast<int>(size);
350 
351  if (!ConcreteVisitor::ShouldVisitFullJSObject()) {
352  // It is important to visit only the used field and ignore the slack fields
353  // because the slack fields may be trimmed concurrently and we don't want to
354  // find fillers (slack) during pointer visitation.
355  const int used_size = map->UsedInstanceSize();
356  DCHECK_LE(used_size, size);
357  DCHECK_GE(used_size, JSObject::GetHeaderSize(map));
358 
359  visitation_size = used_size;
360  }
361 
362  TBodyDescriptor::IterateBody(map, object, visitation_size, visitor);
363 
364  return size;
365 }

367 template <typename ConcreteVisitor>
368 template <VisitorId visitor_id, typename T, typename TBodyDescriptor>
369 size_t HeapVisitor<ConcreteVisitor>::VisitWithBodyDescriptor(
370  Tagged<Map> map, Tagged<T> object, MaybeObjectSize maybe_object_size) {
371  // If you see the following DCHECK fail, then the size computation of
372  // BodyDescriptor doesn't match the size return via obj.Size(). This is
373  // problematic as the GC requires those sizes to match for accounting reasons.
374  // The fix likely involves adding a padding field in the object definitions.
375  //
376  // We can only perform this check for types that do not support right trimming
377  // when running concurrently. `RefineAllocatedBytesAfterSweeping()` ensures
378  // that we only see sizes that get smaller during marking.
379 #ifdef DEBUG
380  if (!SupportsRightTrim<visitor_id>() ||
381  !ConcreteVisitor::EnableConcurrentVisitation()) {
382  DCHECK_EQ(object->SizeFromMap(map), TBodyDescriptor::SizeOf(map, object));
383  }
384 #endif // DEBUG
385  DCHECK(!map->IsInobjectSlackTrackingInProgress());
386 
387  ConcreteVisitor* visitor = static_cast<ConcreteVisitor*>(this);
388  visitor->template VisitMapPointerIfNeeded<visitor_id>(object);
389  const int size = ConcreteVisitor::UsePrecomputedObjectSize()
390  ? static_cast<int>(maybe_object_size.AssumeSize())
391  : TBodyDescriptor::SizeOf(map, object);
392  TBodyDescriptor::IterateBody(map, object, size, visitor);
393  return size;
394 }

458 template <typename ConcreteVisitor>
459 template <typename T>
460 size_t ConcurrentHeapVisitor<ConcreteVisitor>::VisitStringLocked(
461  Tagged<T> object) {
462  ConcreteVisitor* visitor = static_cast<ConcreteVisitor*>(this);
463  ObjectLockGuard guard(object);
464  // The object has been locked. At this point shared read access is
465  // guaranteed but we must re-read the map and check whether the string has
466  // transitioned.
467  Tagged<Map> map = object->map();
468  int size;
469  switch (map->visitor_id()) {
470 #define UNSAFE_STRING_TRANSITION_TARGET_CASE(VisitorIdType, TypeName) \
471  case kVisit##VisitorIdType: \
472  visitor \
473  ->template VisitMapPointerIfNeeded<VisitorId::kVisit##VisitorIdType>( \
474  object); \
475  size = ObjectTraits<TypeName>::BodyDescriptor::SizeOf(map, object); \
476  ObjectTraits<TypeName>::BodyDescriptor::IterateBody( \
477  map, UncheckedCast<TypeName>(object), size, visitor); \
478  break;
479 
480  UNSAFE_STRING_TRANSITION_TARGETS(UNSAFE_STRING_TRANSITION_TARGET_CASE)
481 #undef UNSAFE_STRING_TRANSITION_TARGET_CASE
482  default:
483  UNREACHABLE();
484  }
485  return size;
486 }
