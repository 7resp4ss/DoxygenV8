73 // static
74 void ThreadIsolation::Initialize(
75  ThreadIsolatedAllocator* thread_isolated_allocator) {
76 #if DEBUG
77  trusted_data_.initialized = true;
78 #endif
79 
80  bool enable = thread_isolated_allocator != nullptr;
81 
82  if (v8_flags.jitless) {
83  enable = v8_flags.force_memory_protection_keys;
84  }
85 
86  DCHECK_IMPLIES(v8_flags.force_memory_protection_keys,
87  thread_isolated_allocator != nullptr);
88 
89 #ifdef THREAD_SANITIZER
90  // TODO(sroettger): with TSAN enabled, we get crashes because
91  // SetDefaultPermissionsForAllKeysInSignalHandler gets called while a
92  // RwxMemoryWriteScope is active. It seems that tsan's ProcessPendingSignals
93  // doesn't restore the pkru value after executing the signal handler.
94  enable = false;
95 #endif
96 
97 #if V8_HAS_PKU_JIT_WRITE_PROTECT
98  if (!v8_flags.memory_protection_keys ||
99  !base::MemoryProtectionKey::HasMemoryProtectionKeyAPIs()) {
100  DCHECK_IMPLIES(v8_flags.force_memory_protection_keys,
101  v8_flags.memory_protection_keys);
102  DCHECK_IMPLIES(v8_flags.force_memory_protection_keys,
103  base::MemoryProtectionKey::HasMemoryProtectionKeyAPIs());
104  enable = false;
105  }
106 #endif
107 
108  if (enable) {
109  trusted_data_.allocator = thread_isolated_allocator;
110 #if V8_HAS_PKU_JIT_WRITE_PROTECT
111  trusted_data_.pkey = trusted_data_.allocator->Pkey();
112 
113  // We need to inform our MemoryProtectionKey class that there was an
114  // externally-allocated pkey that we will be using. This is necessary for
115  // signal handlers to have access to memory protected with this key. For
116  // more details see https://crbug.com/416209124.
117  base::MemoryProtectionKey::RegisterExternallyAllocatedKey(
118  trusted_data_.pkey);
119 #endif
120  }
121 
122  {
123  // We need to allocate the memory for jit page tracking even if we don't
124  // enable the ThreadIsolation protections.
125  CFIMetadataWriteScope write_scope("Initialize thread isolation.");
126  ConstructNew(&trusted_data_.jit_pages_mutex_);
127  ConstructNew(&trusted_data_.jit_pages_);
128  }
129 
130  CHECK_IMPLIES(v8_flags.force_memory_protection_keys, enable);
131 
132  if (!enable) {
133  return;
134  }
135 
136 #if V8_HAS_PKU_JIT_WRITE_PROTECT
137  // Check that our compile time assumed page size that we use for padding was
138  // large enough.
139  CHECK_GE(THREAD_ISOLATION_ALIGN_SZ,
140  GetPlatformPageAllocator()->CommitPageSize());
141 
142  // TODO(sroettger): make this immutable once there's OS support.
143  bool success = base::MemoryProtectionKey::SetPermissionsAndKey(
144  {reinterpret_cast<Address>(&trusted_data_), sizeof(trusted_data_)},
145  v8::PageAllocator::Permission::kRead,
146  base::MemoryProtectionKey::kDefaultProtectionKey);
147  CHECK_IMPLIES(v8_flags.force_memory_protection_keys, success);
148 #endif
149 }

180 // static
181 std::optional<ThreadIsolation::JitPageReference>
182 ThreadIsolation::TryLookupJitPageLocked(Address addr, size_t size) {
183  trusted_data_.jit_pages_mutex_->AssertHeld();
184 
185  Address end = addr + size;
186  CHECK_GT(end, addr);
187 
188  // upper_bound gives us an iterator to the position after address.
189  auto it = trusted_data_.jit_pages_->upper_bound(addr);
190 
191  // The previous page should be the one we're looking for.
192  if (it == trusted_data_.jit_pages_->begin()) {
193  return {};
194  }
195 
196  it--;
197 
198  JitPageReference jit_page(it->second, it->first);
199 
200  // If the address is not in the range of the jit page, return.
201  if (jit_page.End() <= addr) {
202  return {};
203  }
204 
205  if (jit_page.End() >= end) {
206  return jit_page;
207  }
208 
209  // It's possible that the allocation spans multiple pages, merge them.
210  auto to_delete_start = ++it;
211  for (; jit_page.End() < end && it != trusted_data_.jit_pages_->end(); it++) {
212  {
213  JitPageReference next_page(it->second, it->first);
214  CHECK_EQ(next_page.Address(), jit_page.End());
215  jit_page.Merge(next_page);
216  }
217  Delete(it->second);
218  }
219 
220  trusted_data_.jit_pages_->erase(to_delete_start, it);
221 
222  if (jit_page.End() < end) {
223  return {};
224  }
225 
226  return jit_page;
227 }

239 template <class T>
240 void CheckForRegionOverlap(const T& map, Address addr, size_t size) {
241  // The data is untrusted from the pov of CFI, so we check that there's no
242  // overlaps with existing regions etc.
243  CHECK_GE(addr + size, addr);
244 
245  // Find an entry in the map with key > addr
246  auto it = map.upper_bound(addr);
247  bool is_begin = it == map.begin();
248  bool is_end = it == map.end();
249 
250  // Check for overlap with the next entry
251  if (!is_end) {
252  Address next_addr = it->first;
253  Address offset = next_addr - addr;
254  CHECK_LE(size, offset);
255  }
256 
257  // Check the previous entry for overlap
258  if (!is_begin) {
259  it--;
260  Address prev_addr = it->first;
261  const typename T::value_type::second_type& prev_entry = it->second;
262  Address offset = addr - prev_addr;
263  CHECK_LE(GetSize(prev_entry), offset);
264  }
265 }

267 template <typename Iterator>
268 bool AllocationIsBehindRange(Address range_start, Address range_size,
269  const Iterator& it) {
270  Address range_end = range_start + range_size;
271  Address allocation_start = it->first;
272  Address allocation_size = it->second.Size();
273  Address allocation_end = allocation_start + allocation_size;
274 
275  if (allocation_start >= range_end) return true;
276 
277  CHECK_LE(allocation_end, range_end);
278  return false;
279 }

295 void ThreadIsolation::JitPageReference::Shrink(class JitPage* tail) {
296  jit_page_->size_ -= tail->size_;
297  // Move all allocations that are out of bounds.
298  auto it = jit_page_->allocations_.lower_bound(End());
299  tail->allocations_.insert(it, jit_page_->allocations_.end());
300  jit_page_->allocations_.erase(it, jit_page_->allocations_.end());

307 void ThreadIsolation::JitPageReference::Merge(JitPageReference& next) {
308  DCHECK_EQ(End(), next.Address());
309  jit_page_->size_ += next.jit_page_->size_;

315 ThreadIsolation::JitAllocation&
316 ThreadIsolation::JitPageReference::RegisterAllocation(base::Address addr,
317  size_t size,
318  JitAllocationType type) {
319  // The data is untrusted from the pov of CFI, so the checks are security
320  // sensitive.
321  CHECK_GE(addr, address_);
322  base::Address offset = addr - address_;
323  base::Address end_offset = offset + size;
324  CHECK_GT(end_offset, offset);
325  CHECK_GT(jit_page_->size_, offset);
326  CHECK_GE(jit_page_->size_, end_offset);
327 
328  CheckForRegionOverlap(jit_page_->allocations_, addr, size);
329  return jit_page_->allocations_.emplace(addr, JitAllocation(size, type))
330  .first->second;
331 }

371 void ThreadIsolation::JitPageReference::UnregisterAllocationsExcept(
372  base::Address start, size_t size, const std::vector<base::Address>& keep) {
373  // TODO(sroettger): check that the page is not in use (scan shadow stacks).
374  JitPage::AllocationMap keep_allocations;
375 
376  auto keep_before = jit_page_->allocations_.lower_bound(start);
377  auto keep_after = jit_page_->allocations_.lower_bound(start + size);
378 
379  // keep all allocations before the start address.
380  if (keep_before != jit_page_->allocations_.begin()) {
381  keep_before--;
382  keep_allocations.insert(jit_page_->allocations_.begin(), keep_before);
383  }
384 
385  // from the start address, keep only allocations passed in the vector
386  auto keep_iterator = keep.begin();
387  for (auto it = keep_before; it != keep_after; it++) {
388  if (keep_iterator == keep.end()) break;
389  if (it->first == *keep_iterator) {
390  keep_allocations.emplace_hint(keep_allocations.end(), it->first,
391  it->second);
392  keep_iterator++;
393  }
394  }
395  CHECK_EQ(keep_iterator, keep.end());
396 
397  // keep all allocations after the region
398  keep_allocations.insert(keep_after, jit_page_->allocations_.end());
399 
400  jit_page_->allocations_.swap(keep_allocations);
401 }

430 void ThreadIsolation::UnregisterJitPage(Address address, size_t size) {
431  // TODO(sroettger): merge the write scopes higher up.
432  CFIMetadataWriteScope write_scope("Removing executable memory.");
433 
434  JitPage* to_delete;
435  {
436  base::MutexGuard guard(trusted_data_.jit_pages_mutex_);
437  JitPageReference jit_page = LookupJitPageLocked(address, size);
438 
439  // We're merging jit pages together, so potentially split them back up
440  // if we're only freeing a subrange.
441 
442  Address to_free_end = address + size;
443  Address jit_page_end = jit_page.Address() + jit_page.Size();
444 
445  if (to_free_end < jit_page_end) {
446  // There's a tail after the page that we release. Shrink the page and
447  // add the tail to the map.
448  size_t tail_size = jit_page_end - to_free_end;
449  JitPage* tail;
450  ConstructNew(&tail, tail_size);
451  jit_page.Shrink(tail);
452  trusted_data_.jit_pages_->emplace(to_free_end, tail);
453  }
454 
455  DCHECK_EQ(to_free_end, jit_page.Address() + jit_page.Size());
456 
457  if (address == jit_page.Address()) {
458  // We remove the start of the region, just remove it from the map.
459  to_delete = jit_page.JitPage();
460  trusted_data_.jit_pages_->erase(address);
461  } else {
462  // Otherwise, we need to shrink the region.
463  DCHECK_GT(address, jit_page.Address());
464  JitPage* tail;
465  ConstructNew(&tail, size);
466  jit_page.Shrink(tail);
467  to_delete = tail;
468  }
469  }
470  Delete(to_delete);
471 }

511 void ThreadIsolation::RegisterJitAllocations(Address start,
512  const std::vector<size_t>& sizes,
513  JitAllocationType type) {
514  CFIMetadataWriteScope write_scope("Register bulk allocations.");
515

555 ThreadIsolation::JitPageReference ThreadIsolation::SplitJitPageLocked(
556  Address addr, size_t size) {
557  trusted_data_.jit_pages_mutex_->AssertHeld();
558 
559  JitPageReference jit_page = LookupJitPageLocked(addr, size);
560 
561  // Split the JitPage into upto three pages.
562  size_t head_size = addr - jit_page.Address();
563  size_t tail_size = jit_page.Size() - size - head_size;
564  if (tail_size > 0) {
565  JitPage* tail;
566  ConstructNew(&tail, tail_size);
567  jit_page.Shrink(tail);
568  trusted_data_.jit_pages_->emplace(addr + size, tail);
569  }
570  if (head_size > 0) {
571  JitPage* mid;
572  ConstructNew(&mid, size);
573  jit_page.Shrink(mid);
574  trusted_data_.jit_pages_->emplace(addr, mid);
575  return JitPageReference(mid, addr);
576  }
577 
578  return jit_page;
579 }

581 std::pair<ThreadIsolation::JitPageReference, ThreadIsolation::JitPageReference>
582 ThreadIsolation::SplitJitPages(Address addr1, size_t size1, Address addr2,
583  size_t size2) {
584  if (addr1 > addr2) {
585  auto reversed_pair = SplitJitPages(addr2, size2, addr1, size1);
586  return {std::move(reversed_pair.second), std::move(reversed_pair.first)};
587  }
588  // Make sure there's no overlap. SplitJitPageLocked will do additional checks
589  // that the sizes don't overflow.
590  CHECK_LE(addr1 + size1, addr2);
591 
592  base::MutexGuard guard(trusted_data_.jit_pages_mutex_);
593  return {SplitJitPageLocked(addr1, size1), SplitJitPageLocked(addr2, size2)};
594 }

642 // static
643 bool ThreadIsolation::CanLookupStartOfJitAllocationAt(Address inner_pointer) {
644  CFIMetadataWriteScope write_scope("CanLookupStartOfJitAllocationAt");
645 
646  // Try to lock the pages mutex and the mutex of the page itself to prevent
647  // potential dead locks. The profiler can try to do a lookup from a signal
648  // handler. If that signal handler runs while the thread locked one of these
649  // mutexes, it would result in a dead lock.
650  if (!trusted_data_.jit_pages_mutex_->TryLock()) {
651  return false;
652  }
653  MutexUnlocker pages_mutex_unlocker(*trusted_data_.jit_pages_mutex_);
654 
655  // upper_bound gives us an iterator to the position after address.
656  auto it = trusted_data_.jit_pages_->upper_bound(inner_pointer);
657 
658  // The previous page should be the one we're looking for.
659  if (it == trusted_data_.jit_pages_->begin()) {
660  return {};
661  }
662  it--;
663 
664  JitPage* jit_page = it->second;
665  if (jit_page->mutex_.TryLock()) {
666  jit_page->mutex_.Unlock();
667  return true;
668  }
669  return false;
670 }
