25 // We use ValueType instances constructed from canonical type indices, so we
26 // can't let them get bigger than what we have storage space for.
27 // We could raise this limit using unused bits in the ValueType, but that
28 // doesn't seem urgent, as we have no evidence of the current limit being
29 // an actual limitation in practice.

38 // A singleton class, responsible for isorecursive canonicalization of wasm
39 // types.
40 // A recursive group is a subsequence of types explicitly marked in the type
41 // section of a wasm module. Identical recursive groups have to be canonicalized
42 // to a single canonical group. Respective types in two identical groups are
43 // considered identical for all purposes.
44 // Two groups are considered identical if they have the same shape, and all
45 // type indices referenced in the same position in both groups reference:
46 // - identical types, if those do not belong to the rec. group,
47 // - types in the same relative position in the group, if those belong to the
48 // rec. group.

205  // Support for hashing of recursion groups, where type indexes have to be
206  // hashed relative to the recursion group.
207  struct CanonicalHashing {
208  base::Hasher hasher;
209  const RecursionGroupRange recgroup;
210 
211  explicit CanonicalHashing(RecursionGroupRange recgroup)
212  : recgroup{recgroup} {}
213 
214  uint32_t MakeGroupRelative(CanonicalTypeIndex index) {
215  uint32_t is_relative = recgroup.Contains(index) ? 1 : 0;
216  uint32_t relative = index.index - is_relative * recgroup.first.index;
217  return (relative << 1) | is_relative;
218  }
219 
220  void Add(CanonicalType type) {
221  // Add several pieces of information in a single number, for faster
222  // hashing.
223  uint32_t supertype_index = MakeGroupRelative(type.supertype);
224  static_assert(kMaxCanonicalTypes <= kMaxUInt32 >> 3);
225  uint32_t metadata =
226  (supertype_index << 2) | (type.is_shared << 1) | (type.is_final << 0);
227  hasher.Add(metadata);
228  switch (type.kind) {
229  case CanonicalType::kFunction:
230  Add(*type.function_sig);
231  break;
232  case CanonicalType::kStruct: {
233  uint32_t descriptor_index = MakeGroupRelative(type.descriptor);
234  uint32_t describes_index = MakeGroupRelative(type.describes);
235  // {MakeGroupRelative} returns a value at most 1 bit bigger than
236  // the max canonical index, which currently fits into 20 bits. So
237  // by shifting left by 11, we're not truncating anything. (And even
238  // if we did, it's just a hash, collisions are OK.)
239  uint32_t desc = (descriptor_index << 11) ^ (describes_index);
240  hasher.Add(desc);
241  Add(*type.struct_type);
242  break;
243  }
244  case CanonicalType::kArray:
245  Add(*type.array_type);
246  break;
247  case CanonicalType::kCont:
248  Add(*type.cont_type);
249  break;
250  }
251  }
252 
253  void Add(CanonicalValueType value_type) {
254  if (value_type.has_index() && recgroup.Contains(value_type.ref_index())) {
255  // For relative indexed types, add their nullability, exactness, and
256  // the relative index to the hash.
257  // Shift the relative index by {kMaxCanonicalTypes} to map it to a
258  // different index space (note that collisions in hashing are OK
259  // though).
260  static_assert(kMaxCanonicalTypes <= kMaxUInt32 / 2);
261  // TODO(403372470): Add the 'exact' bit.
262  hasher.Add((value_type.is_exact() << 1) | value_type.is_nullable());
263  hasher.Add((value_type.ref_index().index - recgroup.first.index) +
264  kMaxCanonicalTypes);
265  } else {
266  hasher.Add(value_type);
267  }
268  }
269 
270  void Add(const CanonicalSig& sig) {
271  hasher.Add(sig.parameter_count());
272  for (CanonicalValueType type : sig.all()) Add(type);
273  }
274 
275  void Add(const CanonicalStructType& struct_type) {
276  hasher.AddRange(struct_type.mutabilities());
277  for (const ValueTypeBase& field : struct_type.fields()) {
278  Add(CanonicalValueType{field});
279  }
280  }
281 
282  void Add(const CanonicalArrayType& array_type) {
283  hasher.Add(array_type.mutability());
284  Add(array_type.element_type());
285  }
286 
287  void Add(const CanonicalContType& cont_type) {
288  CanonicalTypeIndex cont_index = cont_type.contfun_typeindex();
289 
290  if (recgroup.Contains(cont_index)) {
291  hasher.Add(cont_index.index - recgroup.first.index +
292  kMaxCanonicalTypes);
293  } else {
294  hasher.Add(cont_index.index); // Not relative to this rec group
295  }
296  }
297 
298  size_t hash() const { return hasher.hash(); }
299  };

301  // Support for equality checking of recursion groups, where type indexes have
302  // to be compared relative to their respective recursion group.
303  struct CanonicalEquality {
304  // Recursion group bounds for LHS and RHS.
305  const RecursionGroupRange recgroup1;
306  const RecursionGroupRange recgroup2;
307 
308  CanonicalEquality(RecursionGroupRange recgroup1,
309  RecursionGroupRange recgroup2)
310  : recgroup1{recgroup1}, recgroup2{recgroup2} {}
311 
312  bool EqualTypeIndex(CanonicalTypeIndex index1,
313  CanonicalTypeIndex index2) const {
314  const bool relative_index = recgroup1.Contains(index1);
315  if (relative_index != recgroup2.Contains(index2)) return false;
316  if (relative_index) {
317  // Compare relative type indexes within the respective recgroups.
318  uint32_t rel_type1 = index1.index - recgroup1.first.index;
319  uint32_t rel_type2 = index2.index - recgroup2.first.index;
320  if (rel_type1 != rel_type2) return false;
321  } else if (index1 != index2) {
322  return false;
323  }
324  return true;
325  }
326 
327  bool EqualType(const CanonicalType& type1,
328  const CanonicalType& type2) const {
329  if (!EqualTypeIndex(type1.supertype, type2.supertype)) return false;
330  if (type1.is_final != type2.is_final) return false;
331  if (type1.is_shared != type2.is_shared) return false;
332  switch (type1.kind) {
333  case CanonicalType::kFunction:
334  return type2.kind == CanonicalType::kFunction &&
335  EqualSig(*type1.function_sig, *type2.function_sig);
336  case CanonicalType::kStruct:
337  return type2.kind == CanonicalType::kStruct &&
338  EqualTypeIndex(type1.descriptor, type2.descriptor) &&
339  EqualTypeIndex(type1.describes, type2.describes) &&
340  EqualStructType(*type1.struct_type, *type2.struct_type);
341  case CanonicalType::kArray:
342  return type2.kind == CanonicalType::kArray &&
343  EqualArrayType(*type1.array_type, *type2.array_type);
344  case CanonicalType::kCont:
345  return type2.kind == CanonicalType::kCont &&
346  EqualContType(*type1.cont_type, *type2.cont_type);
347  }
348  }
349 
350  bool EqualTypes(base::Vector<const CanonicalType> types1,
351  base::Vector<const CanonicalType> types2) const {
352  return std::equal(types1.begin(), types1.end(), types2.begin(),
353  types2.end(),
354  std::bind_front(&CanonicalEquality::EqualType, this));
355  }
356 
357  bool EqualValueType(CanonicalValueType type1,
358  CanonicalValueType type2) const {
359  const bool indexed = type1.has_index();
360  if (indexed != type2.has_index()) return false;
361  if (indexed) {
362  return type1.is_equal_except_index(type2) &&
363  EqualTypeIndex(type1.ref_index(), type2.ref_index());
364  }
365  return type1 == type2;
366  }
367 
368  bool EqualSig(const CanonicalSig& sig1, const CanonicalSig& sig2) const {
369  if (sig1.parameter_count() != sig2.parameter_count()) return false;
370  return std::equal(
371  sig1.all().begin(), sig1.all().end(), sig2.all().begin(),
372  sig2.all().end(),
373  std::bind_front(&CanonicalEquality::EqualValueType, this));
374  }
375 
376  bool EqualStructType(const CanonicalStructType& type1,
377  const CanonicalStructType& type2) const {
378  return
379  // Compare fields, including a check that the size is the same.
380  std::equal(
381  type1.fields().begin(), type1.fields().end(),
382  type2.fields().begin(), type2.fields().end(),
383  std::bind_front(&CanonicalEquality::EqualValueType, this)) &&
384  // Compare mutabilities, skipping the check for the size.
385  std::equal(type1.mutabilities().begin(), type1.mutabilities().end(),
386  type2.mutabilities().begin());
387  }
388 
389  bool EqualArrayType(const CanonicalArrayType& type1,
390  const CanonicalArrayType& type2) const {
391  return type1.mutability() == type2.mutability() &&
392  EqualValueType(type1.element_type(), type2.element_type());
393  }
394 
395  bool EqualContType(const CanonicalContType& type1,
396  const CanonicalContType& type2) const {
397  return EqualTypeIndex(type1.contfun_typeindex(),
398  type2.contfun_typeindex());
399  }

402  struct CanonicalGroup {
403  CanonicalGroup(Zone* zone, size_t size, CanonicalTypeIndex first)
404  : types(zone->AllocateVector<CanonicalType>(size)), first(first) {
405  // size >= 2; otherwise a `CanonicalSingletonGroup` should have been used.
406  DCHECK_LE(2, size);
407  }
408 
409  bool operator==(const CanonicalGroup& other) const {
410  CanonicalTypeIndex last{first.index +
411  static_cast<uint32_t>(types.size() - 1)};
412  CanonicalTypeIndex other_last{
413  other.first.index + static_cast<uint32_t>(other.types.size() - 1)};
414  CanonicalEquality equality{{first, last}, {other.first, other_last}};
415  return equality.EqualTypes(types, other.types);
416  }
417 
418  size_t hash_value() const {
419  CanonicalTypeIndex last{first.index +
420  static_cast<uint32_t>(types.size()) - 1};
421  CanonicalHashing hasher{{first, last}};
422  for (CanonicalType t : types) {
423  hasher.Add(t);
424  }
425  return hasher.hash();
426  }

429  const base::Vector<CanonicalType> types;
430  const CanonicalTypeIndex first;

449  // Conceptually a vector of CanonicalType. Modification generally requires
450  // synchronization, read-only access can be done without locking.
451  class CanonicalTypeVector {
452  static constexpr uint32_t kSegmentSize = 1024;
453  static constexpr uint32_t kNumSegments =
454  (kMaxCanonicalTypes + kSegmentSize - 1) / kSegmentSize;
455  static_assert(kSegmentSize * kNumSegments >= kMaxCanonicalTypes);
456  static_assert(
457  kNumSegments <= 1024,
458  "Reconsider this data structures when increasing kMaxCanonicalTypes");
459 
460  public:
461  const CanonicalType* operator[](CanonicalTypeIndex index) const {
462  uint32_t segment_idx = index.index / kSegmentSize;
463  // Only check against the static constant here; uninitialized segments are
464  // {nullptr}, so accessing them crashes.
465  SBXCHECK_GT(kNumSegments, segment_idx);
466  Segment* segment = segments_[segment_idx].load(std::memory_order_relaxed);
467  const CanonicalType* type = (*segment)[index.index % kSegmentSize];
468  // DCHECK is sufficient as returning {nullptr} is safe.
469  DCHECK_NOT_NULL(type);
470  return type;
471  }
472 
473  const CanonicalType* operator[](CanonicalValueType type) const {
474  DCHECK(type.has_index());
475  return (*this)[type.ref_index()];
476  }
477 
478  void reserve(uint32_t size, Zone* zone) {
479  DCHECK_GE(kMaxCanonicalTypes, size);
480  uint32_t segment_idx = size / kSegmentSize;
481  // Stop on the first segment (iterating backwards) which already exists.
482  while (!segments_[segment_idx].load(std::memory_order_relaxed)) {
483  segments_[segment_idx].store(zone->New<Segment>(),
484  std::memory_order_relaxed);
485  if (segment_idx-- == 0) break;
486  }
487  }
488 
489  void set(CanonicalTypeIndex index, const CanonicalType* type) {
490  // No checking needed, this is only executed after CheckMaxCanonicalIndex.
491  uint32_t segment_idx = index.index / kSegmentSize;
492  Segment* segment = segments_[segment_idx].load(std::memory_order_relaxed);
493  segment->set(index.index % kSegmentSize, type);
494  }
495 
496  void ClearForTesting() {
497  for (uint32_t i = 0; i < kNumSegments; ++i) {
498  if (segments_[i].load(std::memory_order_relaxed) == nullptr) break;
499  segments_[i].store(nullptr, std::memory_order_relaxed);
500  }
501  }
502 
503  const CanonicalTypeIndex FindIndex_Slow(const CanonicalSig* sig) const {
504  for (uint32_t i = 0; i < kNumSegments; ++i) {
505  Segment* segment = segments_[i].load(std::memory_order_relaxed);
506  // If callers have a CanonicalSig* to pass into this function, the
507  // type canonicalizer must know about this sig, hence we must find it
508  // before hitting a `nullptr` segment.
509  DCHECK_NOT_NULL(segment);
510  for (uint32_t k = 0; k < kSegmentSize; ++k) {
511  const CanonicalType* type = (*segment)[k];
512  // Again: We expect to find the signature before hitting uninitialized
513  // slots.
514  DCHECK_NOT_NULL(type);
515  if (type->kind == CanonicalType::kFunction &&
516  type->function_sig == sig) {
517  return CanonicalTypeIndex{i * kSegmentSize + k};
518  }
519  }
520  }
521  UNREACHABLE();
522  }
