938 // Prefixed opcodes are encoded as 1 prefix byte, followed by LEB encoded
939 // opcode bytes. We internally encode them as {WasmOpcode} as follows:
940 // 1) non-prefixed opcodes use the opcode itself as {WasmOpcode} enum value;
941 // 2) prefixed opcodes in [0, 0xff] use {(prefix << 8) | opcode};
942 // 3) prefixed opcodes in [0x100, 0xfff] use {(prefix << 12) | opcode} (this is
943 // only used for relaxed simd so far).
944 //
945 // This encoding is bijective (i.e. a one-to-one mapping in both directions).
946 // The used opcode ranges are:
947 // 1) [0, 0xff] -> no prefix, 8 bits opcode
948 // 2) [0xfb00, 0xfe00] -> prefix shifted by 8 bits, and 8 bits opcode
949 // 3) [0xfd100, 0xfdfff] -> prefix shifted by 12 bits, and 12 bits opcode
950 // (only [0xfd100, 0xfd1ff] used so far)
951 //
952 // This allows to compute back the prefix and the non-prefixed opcode from each
953 // WasmOpcode, see {WasmOpcodes::ExtractPrefix} and
954 // {ExtractPrefixedOpcodeBytes} (for testing).

956 // Declare expression opcodes.
957 #define DECLARE_NAMED_ENUM(name, opcode, ...) kExpr##name = opcode,
958  FOREACH_OPCODE(DECLARE_NAMED_ENUM)
959 #undef DECLARE_NAMED_ENUM
960 #define DECLARE_PREFIX(name, opcode) k##name##Prefix = opcode,
961  FOREACH_PREFIX(DECLARE_PREFIX)
962 #undef DECLARE_PREFIX
963 };
