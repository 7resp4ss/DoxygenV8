66 //
67 // This is the central hub for dispatching to the various compilers within V8.
68 // Logic for which compiler to choose and how to wire compilation results into
69 // the object heap should be kept inside this class.

71 // General strategy: Scripts are translated into anonymous functions w/o
72 // parameters which then can be executed. If the source code contains other
73 // functions, they might be compiled and allocated as part of the compilation
74 // of the source code or deferred for lazy compilation at a later point.

289 // A base class for compilation jobs intended to run concurrent to the main
290 // thread. The current state of the job can be checked using {state()}.
291 class V8_EXPORT_PRIVATE CompilationJob {
292  public:
293  enum Status { SUCCEEDED, FAILED, RETRY_ON_MAIN_THREAD };
294  enum class State {
295  kReadyToPrepare,
296  kReadyToExecute,
297  kReadyToFinalize,
298  kSucceeded,
299  kFailed,
300  };
301 
302  explicit CompilationJob(State initial_state) : state_(initial_state) {}
303  virtual ~CompilationJob() = default;
304 
305  State state() const { return state_; }
306 
307  protected:
308  V8_WARN_UNUSED_RESULT Status UpdateState(Status status, State next_state) {
309  switch (status) {
310  case SUCCEEDED:
311  state_ = next_state;
312  break;
313  case FAILED:
314  state_ = State::kFailed;
315  break;
316  case RETRY_ON_MAIN_THREAD:
317  // Don't change the state, we'll re-try on the main thread.
318  break;
319  }
320  return status;
321  }
322 
323  private:
324  State state_;
325 };

327 // A base class for unoptimized compilation jobs.
328 //
329 // The job is split into two phases which are called in sequence on
330 // different threads and with different limitations:
331 // 1) ExecuteJob: Runs concurrently. No heap allocation or handle derefs.
332 // 2) FinalizeJob: Runs on main thread. No dependency changes.
333 //
334 // Either of phases can either fail or succeed.
