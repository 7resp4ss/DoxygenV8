76  void PrintBit(const char* name, bool value) {
77  if (!value) return;
78  PrintSeparator();
79  os_ << "{";

81  os_ << AsUC16(static_cast<unsigned char>(*p));

85  void PrintPositive(const char* name, int value) {
86  if (value < 0) return;
87  PrintSeparator();
88  os_ << "{" << name << "|" << value << "}";

135 void DotPrinterImpl::VisitText(TextNode* that) {
136  Zone* zone = that->zone();
137  os_ << " n" << that << " [label=\"";
138  for (int i = 0; i < that->elements()->length(); i++) {
139  if (i > 0) os_ << " ";
140  TextElement elm = that->elements()->at(i);
141  switch (elm.text_type()) {
142  case TextElement::ATOM: {
143  base::Vector<const base::uc16> data = elm.atom()->data();
144  for (int j = 0; j < data.length(); j++) {
145  os_ << AsUC32(data[j]);
146  }
147  break;
148  }
149  case TextElement::CLASS_RANGES: {
150  RegExpClassRanges* node = elm.class_ranges();
151  os_ << "[";
152  if (node->is_negated()) os_ << "^";
153  for (int j = 0; j < node->ranges(zone)->length(); j++) {
154  CharacterRange range = node->ranges(zone)->at(j);
155  os_ << AsUC32(range.from()) << "-" << AsUC32(range.to());
156  if (j > 5) {
157  os_ << "...";
158  break;
159  }
160  }
161  os_ << "]";
162  break;
163  }
164  default:
165  UNREACHABLE();
166  }
167  if (i > 40) {
168  os_ << "...";
169  break;
170  }
171  }
172  os_ << "\", shape=box, peripheries=2];\n";
173  PrintAttributes(that);
174  os_ << " n" << that << " -> n" << that->on_success() << ";\n";
175  Visit(that->on_success());
176 }

217 void DotPrinterImpl::VisitAction(ActionNode* that) {
218  os_ << " n" << that << " [";
219  switch (that->action_type_) {
220  case ActionNode::SET_REGISTER_FOR_LOOP:
221  os_ << "label=\"$" << that->register_from() << ":=" << that->value()
222  << "\", shape=octagon";
223  break;
224  case ActionNode::INCREMENT_REGISTER:
225  os_ << "label=\"$" << that->register_from() << "++\", shape=octagon";
226  break;
227  case ActionNode::CLEAR_POSITION:
228  os_ << "label=\"$" << that->register_from()
229  << ":=$pos c\", shape=octagon";
230  break;
231  case ActionNode::RESTORE_POSITION:
232  os_ << "label=\"$" << that->register_from()
233  << ":=$pos r\", shape=octagon";
234  break;
235  case ActionNode::BEGIN_POSITIVE_SUBMATCH:
236  os_ << "label=\"$" << that->data_.u_submatch.current_position_register
237  << ":=$pos,begin-positive\", shape=septagon";
238  break;
239  case ActionNode::BEGIN_NEGATIVE_SUBMATCH:
240  os_ << "label=\"$" << that->data_.u_submatch.current_position_register
241  << ":=$pos,begin-negative\", shape=septagon";
242  break;
243  case ActionNode::POSITIVE_SUBMATCH_SUCCESS:
244  os_ << "label=\"escape\", shape=septagon";
245  break;
246  case ActionNode::EMPTY_MATCH_CHECK:
247  os_ << "label=\"$" << that->data_.u_empty_match_check.start_register
248  << "=$pos?,$" << that->data_.u_empty_match_check.repetition_register
249  << "<" << that->data_.u_empty_match_check.repetition_limit
250  << "?\", shape=septagon";
251  break;
252  case ActionNode::CLEAR_CAPTURES: {
253  os_ << "label=\"clear $" << that->register_from() << " to $"
254  << that->register_to() << "\", shape=septagon";
255  break;
256  }
257  case ActionNode::MODIFY_FLAGS: {
258  os_ << "label=\"flags $" << that->flags() << "\", shape=septagon";
259  break;
260  }
261  }
262  os_ << "];\n";
263  PrintAttributes(that);
264  RegExpNode* successor = that->on_success();
265  os_ << " n" << that << " -> n" << successor << ";\n";
266  Visit(successor);
267 }
