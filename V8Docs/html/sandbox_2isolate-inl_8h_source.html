85 ExternalPointerTable::Space*
86 IsolateForPointerCompression::GetExternalPointerTableSpaceFor(
87  ExternalPointerTagRange tag_range, Address host) {
88  DCHECK(!tag_range.IsEmpty());
89 
90  if (V8_UNLIKELY(IsSharedExternalPointerType(tag_range))) {
91  DCHECK(!ReadOnlyHeap::Contains(host));
92  return isolate_->shared_external_pointer_space();
93  }
94 
95  if (V8_UNLIKELY(IsMaybeReadOnlyExternalPointerType(tag_range) &&
96  ReadOnlyHeap::Contains(host))) {
97  return isolate_->heap()->read_only_external_pointer_space();
98  }
99 
100  if (HeapLayout::InYoungGeneration(HeapObject::FromAddress(host))) {
101  return isolate_->heap()->young_external_pointer_space();
102  }
103 
104  return isolate_->heap()->old_external_pointer_space();
105 }
