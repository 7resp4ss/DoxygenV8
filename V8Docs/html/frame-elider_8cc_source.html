33 void FrameElider::MarkBlocks() {
34  for (InstructionBlock* block : instruction_blocks()) {
35  if (block->needs_frame()) continue;
36  for (int i = block->code_start(); i < block->code_end(); ++i) {
37  const Instruction* instr = InstructionAt(i);
38  if (instr->IsCall() || instr->IsDeoptimizeCall() ||
39  instr->arch_opcode() == ArchOpcode::kArchStackPointerGreaterThan ||
40  instr->arch_opcode() == ArchOpcode::kArchFramePointer) {
41  block->mark_needs_frame();
42  break;
43  }
44  if (instr->arch_opcode() == ArchOpcode::kArchStackSlot &&
45  ((instr->InputAt(0)->IsImmediate() &&
46  code_->GetImmediate(ImmediateOperand::cast(instr->InputAt(0)))
47  .ToInt32() > 0) ||
48  is_wasm_to_js_)) {
49  // We shouldn't allow accesses to the stack below the current stack
50  // pointer (indicated by positive slot indices).
51  // This is in particular because signal handlers (which could, of
52  // course, be triggered at any point in time) will overwrite this
53  // memory.
54  // Additionally wasm-to-JS code always requires a frame to address
55  // stack slots, because the stack pointer may switch to the central
56  // stack at the beginning of the code.
57  block->mark_needs_frame();
58  break;
59  }
60  }
61  }
62 }

69 void FrameElider::MarkDeConstruction() {
70  for (InstructionBlock* block : instruction_blocks()) {
71  if (block->needs_frame()) {
72  // Special case: The start block needs a frame.
73  if (block->predecessors().empty()) {
74  block->mark_must_construct_frame();
75  if (block->SuccessorCount() == 0) {
76  // We only have a single block, so the block also needs to be marked
77  // to deconstruct the frame.
78  const Instruction* last =
79  InstructionAt(block->last_instruction_index());
80  // The only cases when we need to deconstruct are ret and jump.
81  if (last->IsRet() || last->IsJump()) {
82  block->mark_must_deconstruct_frame();
83  }
84  }
85  }
86  // Find "frame -> no frame" transitions, inserting frame
87  // deconstructions.
88  for (RpoNumber& succ : block->successors()) {
89  if (!InstructionBlockAt(succ)->needs_frame()) {
90  DCHECK_EQ(1U, block->SuccessorCount());
91  const Instruction* last =
92  InstructionAt(block->last_instruction_index());
93  if (last->IsTailCall() || last->IsDeoptimizeCall()) {
94  // We need to keep the frame if we exit the block through any
95  // of these.
96  continue;
97  }
98  // The only cases when we need to deconstruct are ret and jump.
99  DCHECK(last->IsRet() || last->IsJump());
100  block->mark_must_deconstruct_frame();
101  }
102  }
103  if (block->SuccessorCount() == 0) {
104  const Instruction* last =
105  InstructionAt(block->last_instruction_index());
106  // The only cases when we need to deconstruct are ret and jump.
107  if (last->IsRet() || last->IsJump()) {
108  block->mark_must_deconstruct_frame();
109  }
110  }
111  } else {
112  // Find "no frame -> frame" transitions, inserting frame constructions.
113  for (RpoNumber& succ : block->successors()) {
114  if (InstructionBlockAt(succ)->needs_frame()) {
115  DCHECK_NE(1U, block->SuccessorCount());
116  InstructionBlockAt(succ)->mark_must_construct_frame();
117  }
118  }
119  }
120  }
121 }

139 bool FrameElider::PropagateIntoBlock(InstructionBlock* block) {
140  // Already marked, nothing to do...
141  if (block->needs_frame()) return false;
142 
143  // Turbofan does have an empty dummy end block, which we need to ignore here.
144  // However, Turboshaft does not have such a block.
145  if (has_dummy_end_block_) {
146  // Never mark the dummy end node, otherwise we might incorrectly decide to
147  // put frame deconstruction code there later,
148  if (block->successors().empty()) return false;
149  }
150 
151  // Propagate towards the end ("downwards") if there is a predecessor needing
152  // a frame, but don't "bleed" from deferred code to non-deferred code.
153  for (RpoNumber& pred : block->predecessors()) {
154  if (InstructionBlockAt(pred)->needs_frame() &&
155  (!InstructionBlockAt(pred)->IsDeferred() || block->IsDeferred())) {
156  block->mark_needs_frame();
157  return true;
158  }
159  }
160 
161  // Propagate towards start ("upwards")
162  bool need_frame_successors = false;
163  if (block->SuccessorCount() == 1) {
164  // For single successors, propagate the needs_frame information.
165  need_frame_successors =
166  InstructionBlockAt(block->successors()[0])->needs_frame();
167  } else {
168  // For multiple successors, each successor must only have a single
169  // predecessor (because the graph is in edge-split form), so each successor
170  // can independently create/dismantle a frame if needed. Given this
171  // independent control, only propagate needs_frame if all non-deferred
172  // blocks need a frame.
173  for (RpoNumber& succ : block->successors()) {
174  InstructionBlock* successor_block = InstructionBlockAt(succ);
175  DCHECK_EQ(1, successor_block->PredecessorCount());
176  if (!successor_block->IsDeferred()) {
177  if (successor_block->needs_frame()) {
178  need_frame_successors = true;
179  } else {
180  return false;
181  }
182  }
183  }
184  }
185  if (need_frame_successors) {
186  block->mark_needs_frame();
187  return true;
188  } else {
189  return false;
190  }
191 }
