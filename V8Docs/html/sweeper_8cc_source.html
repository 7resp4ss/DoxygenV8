56  bool ConcurrentSweepSpace(AllocationSpace identity, JobDelegate* delegate) {
57  DCHECK(IsValidSweepingSpace(identity));
58  DCHECK_NE(NEW_SPACE, identity);
59  while (!delegate->ShouldYield()) {
60  PageMetadata* page = sweeper_->GetSweepingPageSafe(identity);

66  return false;
67  }

108 class Sweeper::MajorSweeperJob final : public JobTask {
109  private:
110  // Major sweeping jobs don't sweep new space.
111  static constexpr int kNumberOfMajorSweepingSpaces =
112  kNumberOfSweepingSpaces - 1;
113 
114  public:
115  static constexpr int kMaxTasks = kNumberOfMajorSweepingSpaces;
116 
117  MajorSweeperJob(Isolate* isolate, Sweeper* sweeper)
118  : sweeper_(sweeper),
119  concurrent_sweepers(
120  sweeper_->major_sweeping_state_.concurrent_sweepers()),
121  tracer_(isolate->heap()->tracer()),
122  trace_id_(sweeper_->major_sweeping_state_.background_trace_id()) {
123  DCHECK_LE(concurrent_sweepers.size(), kMaxTasks);
124  }
125 
126  ~MajorSweeperJob() override = default;
127 
128  MajorSweeperJob(const MajorSweeperJob&) = delete;
129  MajorSweeperJob& operator=(const MajorSweeperJob&) = delete;
130 
131  void Run(JobDelegate* delegate) final {
132  DCHECK_IMPLIES(
133  delegate->IsJoiningThread(),
134  sweeper_->heap_->IsMainThread() ||
135  (!sweeper_->heap_->isolate()->is_shared_space_isolate() &&
136  sweeper_->heap_->isolate()
137  ->shared_space_isolate()
138  ->heap()
139  ->IsMainThread()));
140  RunImpl(delegate,
141  delegate->IsJoiningThread() && sweeper_->heap_->IsMainThread());
142  }
143 
144  size_t GetMaxConcurrency(size_t worker_count) const override {
145  static constexpr int kPagePerTask = 2;
146  return std::min<size_t>(
147  concurrent_sweepers.size(),
148  worker_count +
149  (sweeper_->ConcurrentMajorSweepingPageCount() + kPagePerTask - 1) /
150  kPagePerTask);
151  }
152 
153  private:
154  void RunImpl(JobDelegate* delegate, bool is_main_thread) {
155  // Set the current isolate such that trusted pointer tables etc are
156  // available and the cage base is set correctly for multi-cage mode.
157  SetCurrentIsolateScope isolate_scope(sweeper_->heap_->isolate());
158 
159  DCHECK(sweeper_->major_sweeping_in_progress());
160  const int offset = delegate->GetTaskId();
161  DCHECK_LT(offset, concurrent_sweepers.size());
162  ConcurrentMajorSweeper& concurrent_sweeper = concurrent_sweepers[offset];
163  TRACE_GC_EPOCH_WITH_FLOW(
164  tracer_, sweeper_->GetTracingScope(OLD_SPACE, is_main_thread),
165  is_main_thread ? ThreadKind::kMain : ThreadKind::kBackground, trace_id_,
166  TRACE_EVENT_FLAG_FLOW_IN);
167  for (int i = 0; i < kNumberOfMajorSweepingSpaces; i++) {
168  const AllocationSpace space_id = static_cast<AllocationSpace>(
169  FIRST_SWEEPABLE_SPACE + 1 +
170  ((offset + i) % kNumberOfMajorSweepingSpaces));
171  DCHECK_LE(FIRST_SWEEPABLE_SPACE, space_id);
172  DCHECK_LE(space_id, LAST_SWEEPABLE_SPACE);
173  DCHECK_NE(NEW_SPACE, space_id);
174  if (!concurrent_sweeper.ConcurrentSweepSpace(space_id, delegate)) return;
175  }
176  }
177 
178  Sweeper* const sweeper_;
179  std::vector<ConcurrentMajorSweeper>& concurrent_sweepers;
180  GCTracer* const tracer_;
181  const uint64_t trace_id_;
182 };

305  DCHECK(!HasValidJob());
306  DCHECK(in_progress_);
307  if (v8_flags.concurrent_sweeping &&

309  auto job =

365 bool Sweeper::LocalSweeper::ParallelSweepSpace(AllocationSpace identity,
366  SweepingMode sweeping_mode,
367  uint32_t max_pages) {
368  uint32_t pages_swept = 0;
369  bool found_usable_pages = false;
370  PageMetadata* page = nullptr;
371  while ((page = sweeper_->GetSweepingPageSafe(identity)) != nullptr) {
372  ParallelSweepPage(page, identity, sweeping_mode);
373  if (!page->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE)) {
374  found_usable_pages = true;
375 #if DEBUG
376  } else {
377  // All remaining pages are also marked with NEVER_ALLOCATE_ON_PAGE.
378  base::MutexGuard guard(&sweeper_->mutex_);
379  int space_index = GetSweepSpaceIndex(identity);
380  Sweeper::SweepingList& sweeping_list =
381  sweeper_->sweeping_list_[space_index];
382  DCHECK(std::all_of(sweeping_list.begin(), sweeping_list.end(),
383  [](const PageMetadata* p) {
384  return p->Chunk()->IsFlagSet(
385  MemoryChunk::NEVER_ALLOCATE_ON_PAGE);
386  }));
387 #endif // DEBUG
388  }
389  if (++pages_swept >= max_pages) break;
390  }
391  return found_usable_pages;
392 }

394 void Sweeper::LocalSweeper::ParallelSweepPage(PageMetadata* page,
395  AllocationSpace identity,
396  SweepingMode sweeping_mode) {
397  DCHECK(IsValidSweepingSpace(identity));
398 
399  DCHECK(!page->SweepingDone());
400 
401  {
402  base::MutexGuard guard(page->mutex());
403  DCHECK(!page->SweepingDone());
404  DCHECK_EQ(PageMetadata::ConcurrentSweepingState::kPendingSweeping,
405  page->concurrent_sweeping_state());
406  page->set_concurrent_sweeping_state(
407  PageMetadata::ConcurrentSweepingState::kInProgress);
408  const FreeSpaceTreatmentMode free_space_treatment_mode =
409  heap::ShouldZapGarbage() ? FreeSpaceTreatmentMode::kZapFreeSpace
410  : FreeSpaceTreatmentMode::kIgnoreFreeSpace;
411  DCHECK_IMPLIES(identity == NEW_SPACE,
412  !sweeper_->minor_sweeping_state_.should_reduce_memory());
413  sweeper_->RawSweep(
414  page, free_space_treatment_mode, sweeping_mode,
415  identity == NEW_SPACE
416  ? false
417  : sweeper_->major_sweeping_state_.should_reduce_memory());
418  sweeper_->AddSweptPage(page, identity);
419  DCHECK(page->SweepingDone());
420  }

593 // Atomically zap the specified area.
594 V8_INLINE void AtomicZapBlock(Address addr, size_t size_in_bytes) {
595  static_assert(sizeof(Tagged_t) == kTaggedSize);
596  static constexpr Tagged_t kZapTagged = static_cast<Tagged_t>(kZapValue);
597  DCHECK(IsAligned(addr, kTaggedSize));
598  DCHECK(IsAligned(size_in_bytes, kTaggedSize));
599  const size_t size_in_tagged = size_in_bytes / kTaggedSize;
600  Tagged_t* current_addr = reinterpret_cast<Tagged_t*>(addr);
601  for (size_t i = 0; i < size_in_tagged; ++i) {
602  base::AsAtomicPtr(current_addr++)
603  ->store(kZapTagged, std::memory_order_relaxed);
604  }
605 }

607 enum class ZappingMode { kNone, kCreateFillers, kCreateFillersAndZap };
608 ZappingMode ShouldZapDeadObjectsOnPage() {
609  if (heap::ShouldZapGarbage() || v8_flags.track_gc_object_stats) {
610  // We need to zap and create fillers on promoted pages when
611  // --track-gc-object-stats is enabled because it expects all dead objects to
612  // still be valid objects. Dead object on promoted pages may otherwise
613  // contain invalid old-to-new references to pages that are gone or were
614  // already reallocated.
615  return ZappingMode::kCreateFillersAndZap;
616  }
617  // Conservative stack scanning requires fillers over all dead objects
618  // otherwise a false reference found on stack may result in resurrecting a
619  // dead object.
620  return v8_flags.conservative_stack_scanning ? ZappingMode::kCreateFillers
621  : ZappingMode::kNone;
622 }

705 V8_INLINE bool ComparePagesForSweepingOrder(const PageMetadata* a,
706  const PageMetadata* b) {
707  // Prioritize pages that can be allocated on.
708  if (a->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE) !=
709  b->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE))
710  return a->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE);
711  // We sort in descending order of live bytes, i.e., ascending order of
712  // free bytes, because GetSweepingPageSafe returns pages in reverse order.
713  // This works automatically for black allocated pages, since we set live bytes
714  // for them to the area size.
715  return a->live_bytes() > b->live_bytes();
716 }

1027 void Sweeper::ZeroOrDiscardUnusedMemory(PageMetadata* page, Address addr,
1028  size_t size) {
1029  if (size < sizeof(FreeSpace)) {
1030  return;
1031  }
1032 
1033  const Address unused_start = addr + sizeof(FreeSpace);
1034  DCHECK(page->ContainsLimit(unused_start));
1035  const Address unused_end = addr + size;
1036  DCHECK(page->ContainsLimit(unused_end));
1037 
1038  std::optional<RwxMemoryWriteScope> scope;
1039  if (page->Chunk()->executable()) {
1040  scope.emplace("For zeroing unused memory.");
1041  }
1042  const std::optional<base::AddressRegion> discard_area =
1043  ComputeDiscardMemoryArea(unused_start, unused_end);
1044 
1045 #if !defined(V8_OS_WIN)
1046  constexpr bool kDiscardEmptyPages = true;
1047 #else
1048  // Discarding memory on Windows does not decommit the memory and does not
1049  // contribute to reduce the memory footprint. On the other hand, these
1050  // calls become expensive the more memory is allocated in the system and
1051  // can result in hangs. Thus, it is better to not discard on Windows.
1052  constexpr bool kDiscardEmptyPages = false;
1053 #endif // !defined(V8_OS_WIN)
1054 
1055  if (kDiscardEmptyPages && discard_area) {
1056  {
1057  v8::PageAllocator* page_allocator =
1058  heap_->memory_allocator()->page_allocator(page->owner_identity());
1059  DiscardSealedMemoryScope discard_scope("Discard unused memory");
1060  CHECK(page_allocator->DiscardSystemPages(
1061  reinterpret_cast<void*>(discard_area->begin()),
1062  discard_area->size()));
1063  }
1064 
1065  if (v8_flags.zero_unused_memory) {
1066  // Now zero unused memory right before and after the discarded OS pages to
1067  // help with OS page compression.
1068  memset(reinterpret_cast<void*>(unused_start), 0,
1069  discard_area->begin() - unused_start);
1070  memset(reinterpret_cast<void*>(discard_area->end()), 0,
1071  unused_end - discard_area->end());
1072  }
1073  } else if (v8_flags.zero_unused_memory) {
1074  // Unused memory does not span a full OS page. Simply clear all of the
1075  // unused memory. This helps with OS page compression.
1076  memset(reinterpret_cast<void*>(unused_start), 0, unused_end - unused_start);
1077  }
1078 }

1080 V8_INLINE void Sweeper::CleanupRememberedSetEntriesForFreedMemory(
1081  Address free_start, Address free_end, PageMetadata* page,
1082  bool record_free_ranges, TypedSlotSet::FreeRangesMap* free_ranges_map,
1083  SweepingMode sweeping_mode) {

1085  if (sweeping_mode == SweepingMode::kEagerDuringGC) {

1087  // empty after a full GC, so we do not need to remove from it after the full
1088  // GC. However, we wouldn't even be allowed to do that, since the main
1089  // thread then owns the old-to-new remembered set. Removing from it from a
1090  // sweeper thread would race with the main thread.

1099  SlotSet::KEEP_EMPTY_BUCKETS);

1101  SlotSet::KEEP_EMPTY_BUCKETS);

1104  DCHECK_NULL(page->slot_set<TRUSTED_TO_TRUSTED>());
1105  }
1106

1108  // during and after a full GC.
1109  RememberedSet<OLD_TO_SHARED>::RemoveRange(page, free_start, free_end,
1110  SlotSet::KEEP_EMPTY_BUCKETS);

1116  free_ranges_map->insert(std::pair<uint32_t, uint32_t>(
1117  static_cast<uint32_t>(chunk->Offset(free_start)),
1118  static_cast<uint32_t>(chunk->Offset(free_end))));
1119  }
