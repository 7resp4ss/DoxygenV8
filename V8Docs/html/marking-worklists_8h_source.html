169 template <AccessMode mode>
170 void MarkingWorklists::ExternalMarkingWorklist::Push(HeapObjectHeader* object) {
171  DCHECK_NOT_NULL(object);
172  ConditionalMutexGuard<mode> guard(&lock_);
173  objects_.insert(object);
174 }

176 template <AccessMode mode>
177 bool MarkingWorklists::ExternalMarkingWorklist::Contains(
178  HeapObjectHeader* object) {

181 }

183 template <AccessMode mode>
184 std::unordered_set<HeapObjectHeader*>
185 MarkingWorklists::ExternalMarkingWorklist::Extract() {
186  ConditionalMutexGuard<mode> guard(&lock_);
187  std::unordered_set<HeapObjectHeader*> extracted;
188  std::swap(extracted, objects_);
189  DCHECK(objects_.empty());
190  return extracted;
191 }
