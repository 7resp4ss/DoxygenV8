128 // As defined in
129 // https://docs.microsoft.com/en-us/cpp/build/x64-calling-convention?view=vs-2019#parameter-passing,
130 // Windows calling convention doesn't differentiate between GP and FP params
131 // when counting how many of them should be placed in registers. That's why
132 // we use the same counter {i} for both types here.
133 // MIPS is the same, as defined in
134 // https://techpubs.jurassic.nl/manuals/0630/developer/Mpro_n32_ABI/sgi_html/ch02.html#id52620.
135 void BuildParameterLocations(const MachineSignature* msig,
136  size_t kFPParamRegisterCount,
137  size_t kParamRegisterCount,
138  const DoubleRegister* kFPParamRegisters,
139  const v8::internal::Register* kParamRegisters,
140  LocationSignature::Builder* out_locations) {
141 #ifdef STACK_SHADOW_WORDS
142  int stack_offset = STACK_SHADOW_WORDS;
143 #else
144  int stack_offset = 0;
145 #endif
146  CHECK_EQ(kFPParamRegisterCount, kParamRegisterCount);
147 
148  for (size_t i = 0; i < msig->parameter_count(); i++) {
149  MachineType type = msig->GetParam(i);
150  bool spill = (i >= kParamRegisterCount);
151  if (spill) {
152  out_locations->AddParam(
153  LinkageLocation::ForCallerFrameSlot(-1 - stack_offset, type));
154  stack_offset++;
155  } else {
156  if (IsFloatingPoint(type.representation())) {
157  out_locations->AddParam(
158  LinkageLocation::ForRegister(kFPParamRegisters[i].code(), type));
159  } else {
160  out_locations->AddParam(
161  LinkageLocation::ForRegister(kParamRegisters[i].code(), type));
162  }
163  }
164  }
165 }
166 #elif defined(V8_TARGET_ARCH_LOONG64)
167 // As defined in
168 // https://loongson.github.io/LoongArch-Documentation/LoongArch-ELF-ABI-EN.html#_procedure_calling_convention
169 // Loongarch calling convention uses GP to pass floating-point arguments when no
170 // FP is available.
171 void BuildParameterLocations(const MachineSignature* msig,
172  size_t kFPParamRegisterCount,
173  size_t kParamRegisterCount,
174  const DoubleRegister* kFPParamRegisters,
175  const v8::internal::Register* kParamRegisters,
176  LocationSignature::Builder* out_locations) {
177 #ifdef STACK_SHADOW_WORDS
178  int stack_offset = STACK_SHADOW_WORDS;
179 #else
180  int stack_offset = 0;
181 #endif
182  size_t num_params = 0;
183  size_t num_fp_params = 0;
184  for (size_t i = 0; i < msig->parameter_count(); i++) {
185  MachineType type = msig->GetParam(i);
186  if (IsFloatingPoint(type.representation())) {
187  if (num_fp_params < kFPParamRegisterCount) {
188  out_locations->AddParam(LinkageLocation::ForRegister(
189  kFPParamRegisters[num_fp_params].code(), type));
190  ++num_fp_params;
191  } else if (num_params < kParamRegisterCount) {
192  // ForNullRegister represents a floating-point param that should be put
193  // into the GPR, and reg_code is the the negative of encoding of the
194  // GPR, and the maximum is -4.
195  out_locations->AddParam(LinkageLocation::ForNullRegister(
196  -kParamRegisters[num_params].code(), type));
197  ++num_params;
198  } else {
199  out_locations->AddParam(
200  LinkageLocation::ForCallerFrameSlot(-1 - stack_offset, type));
201  stack_offset++;
202  }
203  } else {
204  if (num_params < kParamRegisterCount) {
205  out_locations->AddParam(LinkageLocation::ForRegister(
206  kParamRegisters[num_params].code(), type));
207  ++num_params;
208  } else {
209  out_locations->AddParam(
210  LinkageLocation::ForCallerFrameSlot(-1 - stack_offset, type));
211  stack_offset++;
212  }
213  }
214  }
215 }
216 #else
217 // As defined in https://www.agner.org/optimize/calling_conventions.pdf,
218 // Section 7, Linux and Mac place parameters in consecutive registers,
219 // differentiating between GP and FP params. That's why we maintain two
220 // separate counters here. This also applies to Arm systems following
221 // the AAPCS and Windows on Arm.
222 void BuildParameterLocations(const MachineSignature* msig,
223  size_t kFPParamRegisterCount,
224  size_t kParamRegisterCount,
225  const DoubleRegister* kFPParamRegisters,
226  const v8::internal::Register* kParamRegisters,
227  LocationSignature::Builder* out_locations) {
228 #ifdef STACK_SHADOW_WORDS
229  int stack_offset = STACK_SHADOW_WORDS;
230 #else
231  int stack_offset = 0;
232 #endif
233  size_t num_params = 0;
234  size_t num_fp_params = 0;
235  for (size_t i = 0; i < msig->parameter_count(); i++) {
236  MachineType type = msig->GetParam(i);
237  bool spill = IsFloatingPoint(type.representation())
238  ? (num_fp_params >= kFPParamRegisterCount)
239  : (num_params >= kParamRegisterCount);
240  if (spill) {
241  out_locations->AddParam(
242  LinkageLocation::ForCallerFrameSlot(-1 - stack_offset, type));
243  stack_offset++;
244  } else {
245  if (IsFloatingPoint(type.representation())) {
246  out_locations->AddParam(LinkageLocation::ForRegister(
247  kFPParamRegisters[num_fp_params].code(), type));
248  ++num_fp_params;
249  } else {
250  out_locations->AddParam(LinkageLocation::ForRegister(
251  kParamRegisters[num_params].code(), type));
252  ++num_params;
253  }
254  }
255  }
256 }

261 CallDescriptor* Linkage::GetSimplifiedCDescriptor(
262  Zone* zone, const MachineSignature* msig, CallDescriptor::Flags flags,
263  Operator::Properties properties) {
264 #ifdef UNSUPPORTED_C_LINKAGE
265  // This method should not be called on unknown architectures.
266  FATAL("requested C call descriptor on unsupported architecture");
267  return nullptr;
268 #endif
269 
270  DCHECK(properties == Operator::kNoThrow || properties == Operator::kPure);
271  DCHECK_LE(msig->parameter_count(), static_cast<size_t>(kMaxCParameters));
272 
273  LocationSignature::Builder locations(zone, msig->return_count(),
274  msig->parameter_count());
275 
276 #ifndef V8_ENABLE_FP_PARAMS_IN_C_LINKAGE
277  // Check the types of the signature.
278  for (size_t i = 0; i < msig->parameter_count(); i++) {
279  MachineType type = msig->GetParam(i);
280  CHECK(!IsFloatingPoint(type.representation()));
281  }
282 
283  // Check the return types.
284  for (size_t i = 0; i < locations.return_count_; i++) {
285  MachineType type = msig->GetReturn(i);
286  CHECK(!IsFloatingPoint(type.representation()));
287  }
288 #endif
289 
290  CHECK_GE(2, locations.return_count_);
291  if (locations.return_count_ > 0) {
292 #ifdef FP_RETURN_REGISTER
293  const v8::internal::DoubleRegister kFPReturnRegister = FP_RETURN_REGISTER;
294  auto reg = IsFloatingPoint(msig->GetReturn(0).representation())
295  ? kFPReturnRegister.code()
296  : kReturnRegister0.code();
297 #else
298  auto reg = kReturnRegister0.code();
299 #endif
300  // TODO(chromium:1052746): Use the correctly sized register here (e.g. "al"
301  // if the return type is kBit), so we don't have to use a hacky bitwise AND
302  // elsewhere.
303  locations.AddReturn(LinkageLocation::ForRegister(reg, msig->GetReturn(0)));
304  }
305 
306  if (locations.return_count_ > 1) {
307  DCHECK(!IsFloatingPoint(msig->GetReturn(0).representation()));
308 
309  locations.AddReturn(LinkageLocation::ForRegister(kReturnRegister1.code(),
310  msig->GetReturn(1)));
311  }
312 
313 #ifdef PARAM_REGISTERS
314  const v8::internal::Register kParamRegisters[] = {PARAM_REGISTERS};
315  const int kParamRegisterCount = static_cast<int>(arraysize(kParamRegisters));
316 #else
317  const v8::internal::Register* kParamRegisters = nullptr;
318  const int kParamRegisterCount = 0;
319 #endif
320 
321 #ifdef FP_PARAM_REGISTERS
322  const DoubleRegister kFPParamRegisters[] = {FP_PARAM_REGISTERS};
323  const size_t kFPParamRegisterCount = arraysize(kFPParamRegisters);
324 #else
325  const DoubleRegister* kFPParamRegisters = nullptr;
326  const size_t kFPParamRegisterCount = 0;
327 #endif
328 
329  // Add register and/or stack parameter(s).
330  BuildParameterLocations(msig, kFPParamRegisterCount, kParamRegisterCount,
331  kFPParamRegisters, kParamRegisters, &locations);
332 
333  const RegList kCalleeSaveRegisters = {CALLEE_SAVE_REGISTERS};
334  const DoubleRegList kCalleeSaveFPRegisters = {CALLEE_SAVE_FP_REGISTERS};
335 
336  // The target for C calls is always an address (i.e. machine pointer).
337  MachineType target_type = MachineType::Pointer();
338  LinkageLocation target_loc = LinkageLocation::ForAnyRegister(target_type);
339  flags |= CallDescriptor::kNoAllocate;
340 
341  // TODO(saelo): here we probably want to use a c-call specific tag.
342  return zone->New<CallDescriptor>( // --
343  CallDescriptor::kCallAddress, // kind
344  kDefaultCodeEntrypointTag, // tag
345  target_type, // target MachineType
346  target_loc, // target location
347  locations.Get(), // location_sig
348  0, // stack_parameter_count
349  properties, // properties
350  kCalleeSaveRegisters, // callee-saved registers
351  kCalleeSaveFPRegisters, // callee-saved fp regs
352  flags, "c-call");
353 }
