19 // Arguments object passed to C++ builtins.
20 class BuiltinArguments : public JavaScriptArguments {
21  public:
22  BuiltinArguments(int length, Address* arguments)
23  : Arguments(length, arguments) {
24  // Check we have at least the receiver.
25  DCHECK_LE(1, this->length());
26  DCHECK(Tagged<Object>((*at(0)).ptr()).IsObject());
27  }
28 
29  // Zero index states for receiver.
30  Tagged<Object> operator[](int index) const {
31  DCHECK_LT(index, length());
32  return Tagged<Object>(*address_of_arg_at(index + kArgsIndex));
33  }
34 
35  // Zero index states for receiver.
36  template <class S = Object>
37  Handle<S> at(int index) const {
38  DCHECK_LT(index, length());
39  return Handle<S>(address_of_arg_at(index + kArgsIndex));
40  }
41 
42  // Zero index states for receiver.
43  inline void set_at(int index, Tagged<Object> value) {
44  DCHECK_LT(index, length());
45  *address_of_arg_at(index + kArgsIndex) = value.ptr();
46  }
47 
48  // Note: this should return the address after the receiver,
49  // even when length() == 1.
50  inline Address* address_of_first_argument() const {
51  return address_of_arg_at(kFirstArgsIndex);
52  }
53 
54  static constexpr int kNewTargetIndex = 0;
55  static constexpr int kTargetIndex = 1;
56  static constexpr int kArgcIndex = 2;
57  // TODO(ishell): this padding is required only on arm64.
58  static constexpr int kPaddingIndex = 3;
59 
60  static constexpr int kNumExtraArgs = 4;
61  static constexpr int kNumExtraArgsWithReceiver = 5;
62 
63  static constexpr int kArgsIndex = kNumExtraArgs;
64  static constexpr int kReceiverIndex = kArgsIndex;
65  static constexpr int kFirstArgsIndex = kArgsIndex + 1; // Skip receiver.
66  // Index of the receiver argument in JS arguments array returned by
67  // |address_of_first_argument()|.
68  static constexpr int kReceiverArgsIndex = kArgsIndex - kFirstArgsIndex;
69 
70  // Zero index states for receiver.
71  inline Handle<Object> atOrUndefined(Isolate* isolate, int index) const;
72  inline Handle<JSAny> receiver() const;
73  inline Handle<JSFunction> target() const;
74  inline Handle<HeapObject> new_target() const;
75 
76  // Gets the total number of arguments including the receiver (but
77  // excluding extra arguments).
78  int length() const { return Arguments::length() - kNumExtraArgs; }
79 };

106 // ----------------------------------------------------------------------------
107 // Support macro for defining builtins in C++.
108 // ----------------------------------------------------------------------------
109 //
110 // A builtin function is defined by writing:
111 //
112 // BUILTIN(name) {
113 // ...
114 // }
115 //
116 // In the body of the builtin function the arguments can be accessed
117 // through the BuiltinArguments object args.
118 // TODO(cbruni): add global flag to check whether any tracing events have been
119 // enabled.
120 #define BUILTIN_RCS(name) \
121  V8_WARN_UNUSED_RESULT static Tagged<Object> Builtin_Impl_##name( \
122  BuiltinArguments args, Isolate* isolate); \
123  \
124  V8_NOINLINE static Address Builtin_Impl_Stats_##name( \
125  int args_length, Address* args_object, Isolate* isolate) { \
126  BuiltinArguments args(args_length, args_object); \
127  RCS_SCOPE(isolate, RuntimeCallCounterId::kBuiltin_##name); \
128  TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.runtime"), \
129  "V8.Builtin_" #name); \
130  return BUILTIN_CONVERT_RESULT(Builtin_Impl_##name(args, isolate)); \
131  } \
132  \
133  V8_WARN_UNUSED_RESULT Address Builtin_##name( \
134  int args_length, Address* args_object, Isolate* isolate) { \
135  DCHECK(isolate->context().is_null() || IsContext(isolate->context())); \
136  if (V8_UNLIKELY(TracingFlags::is_runtime_stats_enabled())) { \
137  return Builtin_Impl_Stats_##name(args_length, args_object, isolate); \
138  } \
139  BuiltinArguments args(args_length, args_object); \
140  CHECK_BUILTIN_SANDBOXING_MODE() \
141  return BUILTIN_CONVERT_RESULT(Builtin_Impl_##name(args, isolate)); \
142  } \
143  \
144  V8_WARN_UNUSED_RESULT static Tagged<Object> Builtin_Impl_##name( \
145  BuiltinArguments args, Isolate* isolate)
146 
147 #define BUILTIN_NO_RCS(name) \
148  V8_WARN_UNUSED_RESULT static Tagged<Object> Builtin_Impl_##name( \
149  BuiltinArguments args, Isolate* isolate); \
150  \
151  V8_WARN_UNUSED_RESULT Address Builtin_##name( \
152  int args_length, Address* args_object, Isolate* isolate) { \
153  DCHECK(isolate->context().is_null() || IsContext(isolate->context())); \
154  BuiltinArguments args(args_length, args_object); \
155  CHECK_BUILTIN_SANDBOXING_MODE() \

157  } \
158  \

169 #define CHECK_RECEIVER(Type, name, method) \
170  if (!Is##Type(*args.receiver())) { \
171  THROW_NEW_ERROR_RETURN_FAILURE( \
172  isolate, \
173  NewTypeError(MessageTemplate::kIncompatibleMethodReceiver, \
174  isolate->factory()->NewStringFromAsciiChecked(method), \
175  args.receiver())); \
176  } \
177  auto name = Cast<Type>(args.receiver())
178 
179 // Throws a TypeError for {method} if the receiver is not coercible to Object,
180 // or converts the receiver to a String otherwise and assigns it to a new var
181 // with the given {name}.
182 #define TO_THIS_STRING(name, method) \
183  if (IsNullOrUndefined(*args.receiver(), isolate)) { \
184  THROW_NEW_ERROR_RETURN_FAILURE( \
185  isolate, \
186  NewTypeError(MessageTemplate::kCalledOnNullOrUndefined, \
187  isolate->factory()->NewStringFromAsciiChecked(method))); \
188  } \
189  DirectHandle<String> name; \
190  ASSIGN_RETURN_FAILURE_ON_EXCEPTION( \
191  isolate, name, Object::ToString(isolate, args.receiver()))
