136 void PerfJitLogger::OpenJitDumpFile() {
137  // Open the perf JIT dump file.
138  perf_output_handle_ = nullptr;
139 
140  size_t bufferSize = strlen(v8_flags.perf_prof_path) +
141  sizeof(kFilenameFormatString) + kFilenameBufferPadding;
142  base::ScopedVector<char> perf_dump_name(bufferSize);
143  int size = SNPrintF(perf_dump_name, kFilenameFormatString,
144  v8_flags.perf_prof_path.value(), process_id_);
145  CHECK_NE(size, -1);
146 
147  int fd = open(perf_dump_name.begin(), O_CREAT | O_TRUNC | O_RDWR, 0666);
148  if (fd == -1) return;
149 
150  // If --perf-prof-delete-file is given, unlink the file right after opening
151  // it. This keeps the file handle to the file valid. This only works on Linux,
152  // which is the only platform supported for --perf-prof anyway.
153  if (v8_flags.perf_prof_delete_file)
154  CHECK_EQ(0, unlink(perf_dump_name.begin()));
155 
156  // On Linux, call OpenMarkerFile so that perf knows about the file path via
157  // an MMAP record.
158  // On macOS, don't call OpenMarkerFile because samply has already detected
159  // the file path during the call to `open` above (it interposes `open` with
160  // a preloaded library), and because the mmap call can be slow.
161 #if V8_OS_DARWIN
162  marker_address_ = nullptr;
163 #else
164  marker_address_ = OpenMarkerFile(fd);
165  if (marker_address_ == nullptr) return;
166 #endif
167 
168  perf_output_handle_ = fdopen(fd, "w+");
169  if (perf_output_handle_ == nullptr) return;
170 
171  setvbuf(perf_output_handle_, nullptr, _IOFBF, kLogBufferSize);
172 }

174 void PerfJitLogger::CloseJitDumpFile() {
175  if (perf_output_handle_ == nullptr) return;
176  base::Fclose(perf_output_handle_);
177  perf_output_handle_ = nullptr;
178 }

180 void* PerfJitLogger::OpenMarkerFile(int fd) {
181  long page_size = sysconf(_SC_PAGESIZE); // NOLINT(runtime/int)
182  if (page_size == -1) return nullptr;
183 
184  // Mmap the file so that there is a mmap record in the perf_data file.
185  //
186  // The map must be PROT_EXEC to ensure it is not ignored by perf record.
187  void* marker_address =
188  mmap(nullptr, page_size, PROT_READ | PROT_EXEC, MAP_PRIVATE, fd, 0);
189  return (marker_address == MAP_FAILED) ? nullptr : marker_address;
190 }
191 
192 void PerfJitLogger::CloseMarkerFile(void* marker_address) {
193  if (marker_address == nullptr) return;
194  long page_size = sysconf(_SC_PAGESIZE); // NOLINT(runtime/int)
195  if (page_size == -1) return;
196  munmap(marker_address, page_size);
197 }

226 void PerfJitLogger::LogRecordedBuffer(
227  Tagged<AbstractCode> abstract_code,
228  MaybeDirectHandle<SharedFunctionInfo> maybe_sfi, const char* name,
229  size_t length) {
230  DisallowGarbageCollection no_gc;
231  if (v8_flags.perf_basic_prof_only_functions) {
232  CodeKind code_kind = abstract_code->kind(isolate_);
233  if (!CodeKindIsJSFunction(code_kind)) {
234  return;
235  }
236  }
237 
238  base::LockGuard<base::RecursiveMutex> guard_file(GetFileMutex().Pointer());
239 
240  if (perf_output_handle_ == nullptr) return;
241 
242  // We only support non-interpreted functions.
243  if (!IsCode(abstract_code, isolate_)) return;
244  Tagged<Code> code = Cast<Code>(abstract_code);
245 
246  // Debug info has to be emitted first.
247  DirectHandle<SharedFunctionInfo> sfi;
248  if (v8_flags.perf_prof && maybe_sfi.ToHandle(&sfi)) {
249  // TODO(herhut): This currently breaks for js2wasm/wasm2js functions.
250  CodeKind kind = code->kind();
251  if (kind != CodeKind::JS_TO_WASM_FUNCTION &&
252  kind != CodeKind::WASM_TO_JS_FUNCTION) {
253  DCHECK_IMPLIES(IsScript(sfi->script()),
254  Cast<Script>(sfi->script())->has_line_ends());
255  LogWriteDebugInfo(code, sfi);
256  }
257  }
258 
259  const char* code_name = name;
260  uint8_t* code_pointer = reinterpret_cast<uint8_t*>(code->instruction_start());
261 
262  // Unwinding info comes right after debug info.
263  if (v8_flags.perf_prof_unwinding_info) LogWriteUnwindingInfo(code);
264 
265  WriteJitCodeLoadEntry(code_pointer, code->instruction_size(), code_name,
266  length);
267 }

283 void PerfJitLogger::WriteJitCodeLoadEntry(const uint8_t* code_pointer,
284  uint32_t code_size, const char* name,
285  size_t name_length) {
286  PerfJitCodeLoad code_load;
287  code_load.event_ = PerfJitCodeLoad::kLoad;
288  code_load.size_ =
289  static_cast<uint32_t>(sizeof(code_load) + name_length + 1 + code_size);
290  code_load.time_stamp_ = GetTimestamp();
291  code_load.process_id_ = static_cast<uint32_t>(process_id_);
292  code_load.thread_id_ = static_cast<uint32_t>(base::OS::GetCurrentThreadId());
293  code_load.vma_ = reinterpret_cast<uint64_t>(code_pointer);
294  code_load.code_address_ = reinterpret_cast<uint64_t>(code_pointer);
295  code_load.code_size_ = code_size;
296  code_load.code_id_ = code_index_;
297 
298  code_index_++;
299 
300  LogWriteBytes(reinterpret_cast<const char*>(&code_load), sizeof(code_load));
301  LogWriteBytes(name, name_length);
302  LogWriteBytes(kStringTerminator, sizeof(kStringTerminator));
303  LogWriteBytes(reinterpret_cast<const char*>(code_pointer), code_size);
304 }

347 void PerfJitLogger::LogWriteDebugInfo(Tagged<Code> code,
348  DirectHandle<SharedFunctionInfo> shared) {
349  // Line ends of all scripts have been initialized prior to this.
350  DisallowGarbageCollection no_gc;
351  // The WasmToJS wrapper stubs have source position entries.
352  Tagged<SharedFunctionInfo> raw_shared = *shared;
353  if (!raw_shared->HasSourceCode()) return;
354 
355  PerfJitCodeDebugInfo debug_info;
356  uint32_t size = sizeof(debug_info);
357 
358  Tagged<TrustedByteArray> source_position_table =
359  code->SourcePositionTable(isolate_, raw_shared);
360  // Compute the entry count and get the names of all scripts.
361  // Avoid additional work if the script name is repeated. Multiple script
362  // names only occur for cross-script inlining.
363  uint32_t entry_count = 0;
364  Tagged<Object> last_script = Smi::zero();
365  size_t last_script_name_size = 0;
366  std::vector<base::Vector<const char>> script_names;
367  for (SourcePositionTableIterator iterator(source_position_table);
368  !iterator.done(); iterator.Advance()) {
369  SourcePositionInfo info(GetSourcePositionInfo(isolate_, code, shared,
370  iterator.source_position()));
371  Tagged<Object> current_script = *info.script;
372  if (current_script != last_script) {
373  std::unique_ptr<char[]> name_storage;
374  auto name = GetScriptName(raw_shared->script(), &name_storage, no_gc);
375  script_names.push_back(name);
376  // Add the size of the name after each entry.
377  last_script_name_size = name.size() + sizeof(kStringTerminator);
378  size += last_script_name_size;
379  last_script = current_script;
380  } else {
381  DCHECK_LT(0, last_script_name_size);
382  size += last_script_name_size;
383  }
384  entry_count++;
385  }
386  if (entry_count == 0) return;
387 
388  debug_info.event_ = PerfJitCodeLoad::kDebugInfo;
389  debug_info.time_stamp_ = GetTimestamp();
390  debug_info.address_ = code->instruction_start();
391  debug_info.entry_count_ = entry_count;
392 
393  // Add the sizes of fixed parts of entries.
394  size += entry_count * sizeof(PerfJitDebugEntry);
395 
396  int padding = ((size + 7) & (~7)) - size;
397  debug_info.size_ = size + padding;
398  LogWriteBytes(reinterpret_cast<const char*>(&debug_info), sizeof(debug_info));
399 
400  Address code_start = code->instruction_start();
401 
402  last_script = Smi::zero();
403  int script_names_index = 0;
404  for (SourcePositionTableIterator iterator(source_position_table);
405  !iterator.done(); iterator.Advance()) {
406  SourcePositionInfo info(GetSourcePositionInfo(isolate_, code, shared,
407  iterator.source_position()));
408  PerfJitDebugEntry entry;
409  // The entry point of the function will be placed straight after the ELF
410  // header when processed by "perf inject". Adjust the position addresses
411  // accordingly.
412  entry.address_ = code_start + iterator.code_offset() + kElfHeaderSize;
413  entry.line_number_ = info.line + 1;
414  entry.column_ = info.column + 1;
415  LogWriteBytes(reinterpret_cast<const char*>(&entry), sizeof(entry));
416  Tagged<Object> current_script = *info.script;
417  auto name_string = script_names[script_names_index];
418  LogWriteBytes(name_string.begin(), name_string.size());
419  LogWriteBytes(kStringTerminator, sizeof(kStringTerminator));
420  if (current_script != last_script) {
421  if (last_script != Smi::zero()) script_names_index++;
422  last_script = current_script;
423  }
424  }
425  char padding_bytes[8] = {0};
426  LogWriteBytes(padding_bytes, padding);
427 }

429 #if V8_ENABLE_WEBASSEMBLY
430 void PerfJitLogger::LogWriteDebugInfo(const wasm::WasmCode* code) {
431  if (code->IsAnonymous()) {
432  return;
433  }
434 
435  wasm::WasmModuleSourceMap* source_map =
436  code->native_module()->GetWasmSourceMap();
437  wasm::WireBytesRef code_ref =
438  code->native_module()->module()->functions[code->index()].code;
439  uint32_t code_offset = code_ref.offset();
440  uint32_t code_end_offset = code_ref.end_offset();
441 
442  uint32_t entry_count = 0;
443  uint32_t size = 0;
444 
445  if (!source_map || !source_map->IsValid() ||
446  !source_map->HasSource(code_offset, code_end_offset)) {
447  return;
448  }
449 
450  for (SourcePositionTableIterator iterator(code->source_positions());
451  !iterator.done(); iterator.Advance()) {
452  uint32_t offset = iterator.source_position().ScriptOffset() + code_offset;
453  if (!source_map->HasValidEntry(code_offset, offset)) continue;
454  entry_count++;
455  size += source_map->GetFilename(offset).size() + 1;
456  }
457 
458  if (entry_count == 0) return;
459 
460  PerfJitCodeDebugInfo debug_info;
461 
462  debug_info.event_ = PerfJitCodeLoad::kDebugInfo;
463  debug_info.time_stamp_ = GetTimestamp();
464  debug_info.address_ =
465  reinterpret_cast<uintptr_t>(code->instructions().begin());
466  debug_info.entry_count_ = entry_count;
467 
468  size += sizeof(debug_info);
469  // Add the sizes of fixed parts of entries.
470  size += entry_count * sizeof(PerfJitDebugEntry);
471 
472  int padding = ((size + 7) & (~7)) - size;
473  debug_info.size_ = size + padding;
474  LogWriteBytes(reinterpret_cast<const char*>(&debug_info), sizeof(debug_info));
475 
476  uintptr_t code_begin =
477  reinterpret_cast<uintptr_t>(code->instructions().begin());
478 
479  for (SourcePositionTableIterator iterator(code->source_positions());
480  !iterator.done(); iterator.Advance()) {
481  uint32_t offset = iterator.source_position().ScriptOffset() + code_offset;
482  if (!source_map->HasValidEntry(code_offset, offset)) continue;
483  PerfJitDebugEntry entry;
484  // The entry point of the function will be placed straight after the ELF
485  // header when processed by "perf inject". Adjust the position addresses
486  // accordingly.
487  entry.address_ = code_begin + iterator.code_offset() + kElfHeaderSize;
488  entry.line_number_ =
489  static_cast<int>(source_map->GetSourceLine(offset)) + 1;
490  entry.column_ = 1;
491  LogWriteBytes(reinterpret_cast<const char*>(&entry), sizeof(entry));
492  std::string name_string = source_map->GetFilename(offset);
493  LogWriteBytes(name_string.c_str(), name_string.size());
494  LogWriteBytes(kStringTerminator, sizeof(kStringTerminator));
495  }
496 
497  char padding_bytes[8] = {0};
498  LogWriteBytes(padding_bytes, padding);
499 }

502 void PerfJitLogger::LogWriteUnwindingInfo(Tagged<Code> code) {
503  PerfJitCodeUnwindingInfo unwinding_info_header;
504  unwinding_info_header.event_ = PerfJitCodeLoad::kUnwindingInfo;
505  unwinding_info_header.time_stamp_ = GetTimestamp();
506  unwinding_info_header.eh_frame_hdr_size_ = EhFrameConstants::kEhFrameHdrSize;
507 
508  if (code->has_unwinding_info()) {
509  unwinding_info_header.unwinding_size_ = code->unwinding_info_size();
510  unwinding_info_header.mapped_size_ = unwinding_info_header.unwinding_size_;
511  } else {
512  unwinding_info_header.unwinding_size_ = EhFrameConstants::kEhFrameHdrSize;
513  unwinding_info_header.mapped_size_ = 0;
514  }
515 
516  int content_size = static_cast<int>(sizeof(unwinding_info_header) +
517  unwinding_info_header.unwinding_size_);
518  int padding_size = RoundUp(content_size, 8) - content_size;
519  unwinding_info_header.size_ = content_size + padding_size;
520 
521  LogWriteBytes(reinterpret_cast<const char*>(&unwinding_info_header),
522  sizeof(unwinding_info_header));
523 
524  if (code->has_unwinding_info()) {
525  LogWriteBytes(reinterpret_cast<const char*>(code->unwinding_info_start()),
526  code->unwinding_info_size());
527  } else {
528  OFStream perf_output_stream(perf_output_handle_);
529  EhFrameWriter::WriteEmptyEhFrame(perf_output_stream);
530  }
531 
532  char padding_bytes[] = "\0\0\0\0\0\0\0\0";
533  DCHECK_LT(padding_size, static_cast<int>(sizeof(padding_bytes)));
534  LogWriteBytes(padding_bytes, padding_size);
535 }

543 void PerfJitLogger::LogWriteHeader() {
544  DCHECK_NOT_NULL(perf_output_handle_);

546

548  header.version_ = PerfJitHeader::kVersion;
549  header.size_ = sizeof(header);
