15 // Enum to specify if `+0` and `-0` should be treated as the same value.
16 enum IdentifyZeros : uint8_t {
17  // `+0` and `-0` should be treated as the same value.
18  kIdentifyZeros,
19  // `+0` and `-0` should be treated as different values.
20  kDistinguishZeros
21 };

198 // The {UseInfo} class is used to describe a use of an input of a node.
199 //
200 // This information is used in two different ways, based on the phase:
201 //
202 // 1. During propagation, the use info is used to inform the input node
203 // about what part of the input is used (we call this truncation) and what
204 // is the preferred representation. For conversions that will require
205 // checks, we also keep track of whether a minus zero check is needed.
206 //
207 // 2. During lowering, the use info is used to properly convert the input
208 // to the preferred representation. The preferred representation might be
209 // insufficient to do the conversion (e.g. word32->float64 conv), so we also
210 // need the signedness information to produce the correct value.
211 // Additionally, use info may contain {CheckParameters} which contains
212 // information for the deoptimizer such as a CallIC on which speculation
213 // should be disallowed if the check fails.
214 class UseInfo {
215  public:
216  UseInfo(MachineRepresentation representation, Truncation truncation,
217  TypeCheckKind type_check = TypeCheckKind::kNone,
218  const FeedbackSource& feedback = FeedbackSource())
219  : representation_(representation),
220  truncation_(truncation),
221  type_check_(type_check),
222  feedback_(feedback) {}
223  static UseInfo TruncatingWord32() {
224  return UseInfo(MachineRepresentation::kWord32, Truncation::Word32());
225  }
226 
227  static UseInfo TruncatingWord64() {
228  return UseInfo(MachineRepresentation::kWord64, Truncation::Word64());
229  }

231  DCHECK(Is64());
