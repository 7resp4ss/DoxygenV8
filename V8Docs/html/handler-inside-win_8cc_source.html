7 // This file implements the out of bounds trap handler for
8 // WebAssembly. Exception handlers are notoriously difficult to get
9 // right, and getting it wrong can lead to security
10 // vulnerabilities. In order to minimize this risk, here are some
11 // rules to follow.
12 //
13 // 1. Do not introduce any new external dependencies. This file needs
14 // to be self contained so it is easy to audit everything that a
15 // trap handler might do.

17 // 2. Any changes must be reviewed by someone from the crash reporting
18 // or security team. See OWNERS for suggested reviewers.
19 //

44 // The below struct needed to access the offset in the Thread Environment Block
45 // to see if the thread local storage for the thread has been allocated yet.
46 //
47 // The ThreadLocalStorage pointer is located 12 pointers into the TEB (i.e. at
48 // offset 0x58 for 64-bit platforms, and 0x2c for 32-bit platforms). This is
49 // true for x64, x86, ARM, and ARM64 platforms (see the header files in the SDK
50 // named ksamd64.inc, ks386.inc, ksarm.h, and ksarm64.h respectively).
51 //
52 // These offsets are baked into compiled binaries, so can never be changed for
53 // backwards compatibility reasons.
54 struct TEB {
55  PVOID reserved[11];
56  PVOID thread_local_storage_pointer;
57 };

66 bool TryHandleWasmTrap(EXCEPTION_POINTERS* exception) {
67  // VectoredExceptionHandlers need extreme caution. Do as little as possible
68  // to determine if the exception should be handled or not. Exceptions can be
69  // thrown very early in a threads life, before the thread has even completed
70  // initializing. As a demonstrative example, there was a bug (#8966) where an
71  // exception would be raised before the thread local copy of the
72  // "__declspec(thread)" variables had been allocated, the handler tried to
73  // access the thread-local "g_thread_in_wasm_code", which would then raise
74  // another exception, and an infinite loop ensued.
75 
76  // First ensure this is an exception type of interest
77  if (exception->ExceptionRecord->ExceptionCode != EXCEPTION_ACCESS_VIOLATION) {
78  return false;
79  }
80 
81  // See if thread-local storage for __declspec(thread) variables has been
82  // allocated yet. This pointer is initially null in the TEB until the
83  // loader has completed allocating the memory for thread_local variables
84  // and copy constructed their initial values. (Note: Any functions that
85  // need to run to initialize values may not have run yet, but that is not
86  // the case for any thread_locals used here).
87  TEB* pteb = reinterpret_cast<TEB*>(NtCurrentTeb());
88  if (!pteb->thread_local_storage_pointer) return false;
89 
90  // Now safe to run more advanced logic, which may access thread_locals
91  // Ensure the faulting thread was actually running Wasm code.
92  if (!IsThreadInWasm()) return false;
93 
94  // Clear g_thread_in_wasm_code, primarily to protect against nested faults.
95  // The only path that resets the flag to true is if we find a landing pad (in
96  // which case this function returns true). Otherwise we leave the flag unset
97  // since we do not return to wasm code.
98  g_thread_in_wasm_code = false;
99 
100  const EXCEPTION_RECORD* record = exception->ExceptionRecord;
101 
102  uintptr_t fault_addr = reinterpret_cast<uintptr_t>(record->ExceptionAddress);
103 
104 #ifdef V8_TRAP_HANDLER_VIA_SIMULATOR
105  // Only handle signals triggered by the load in {ProbeMemory}.
106  if (fault_addr != reinterpret_cast<uintptr_t>(&ProbeMemory)) return false;
107 
108  // The simulated ip will be in the second parameter register (%rdx).
109  uintptr_t simulated_ip = exception->ContextRecord->Rdx;
110  if (!IsFaultAddressCovered(simulated_ip)) return false;
111 
112  exception->ContextRecord->Rax = gLandingPad;
113  // The fault_address that is set in non-simulator builds here is set in the
114  // simulator directly.
115  // Continue at the memory probing continuation.
116  exception->ContextRecord->Rip =
117  reinterpret_cast<uintptr_t>(&probe_memory_continuation);
118 #else
119  if (!IsFaultAddressCovered(fault_addr)) return false;
120 
121  TH_DCHECK(gLandingPad != 0);
122  // Tell the caller to return to the landing pad.
123 #if V8_HOST_ARCH_X64
124  exception->ContextRecord->Rip = gLandingPad;
125  exception->ContextRecord->R10 = fault_addr;
126 #elif V8_HOST_ARCH_ARM64
127  exception->ContextRecord->Pc = gLandingPad;
128  exception->ContextRecord->X16 = fault_addr;
129 #else
130 #error Unsupported architecture
131 #endif // V8_HOST_ARCH_X64
132 #endif // V8_TRAP_HANDLER_VIA_SIMULATOR
133  // We will return to wasm code, so restore the g_thread_in_wasm_code flag.
134  g_thread_in_wasm_code = true;
135  return true;
136 }

138 LONG HandleWasmTrap(EXCEPTION_POINTERS* exception) {
139  if (TryHandleWasmTrap(exception)) {
140  return EXCEPTION_CONTINUE_EXECUTION;
141  }
142  return EXCEPTION_CONTINUE_SEARCH;
