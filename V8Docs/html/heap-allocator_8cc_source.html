28 void HeapAllocator::Setup(LinearAllocationArea* new_allocation_info,
29  LinearAllocationArea* old_allocation_info) {
30  for (int i = FIRST_SPACE; i <= LAST_SPACE; ++i) {
31  spaces_[i] = heap_->space(i);
32  }
33 
34  if ((heap_->new_space() || v8_flags.sticky_mark_bits) &&
35  local_heap_->is_main_thread()) {
36  new_space_allocator_.emplace(
37  local_heap_,
38  v8_flags.sticky_mark_bits
39  ? static_cast<SpaceWithLinearArea*>(heap_->sticky_space())
40  : static_cast<SpaceWithLinearArea*>(heap_->new_space()),
41  MainAllocator::IsNewGeneration::kYes, new_allocation_info);
42  }
43 
44  old_space_allocator_.emplace(local_heap_, heap_->old_space(),
45  MainAllocator::IsNewGeneration::kNo,
46  old_allocation_info);
47 
48  trusted_space_allocator_.emplace(local_heap_, heap_->trusted_space(),
49  MainAllocator::IsNewGeneration::kNo);
50  code_space_allocator_.emplace(local_heap_, heap_->code_space(),
51  MainAllocator::IsNewGeneration::kNo);
52 
53  if (heap_->isolate()->has_shared_space()) {
54  shared_space_allocator_.emplace(local_heap_,
55  heap_->shared_allocation_space(),
56  MainAllocator::IsNewGeneration::kNo);
57  shared_lo_space_ = heap_->shared_lo_allocation_space();
58 
59  shared_trusted_space_allocator_.emplace(
60  local_heap_, heap_->shared_trusted_allocation_space(),
61  MainAllocator::IsNewGeneration::kNo);
62  shared_trusted_lo_space_ = heap_->shared_trusted_lo_allocation_space();
63  }
64 }

70 AllocationResult HeapAllocator::AllocateRawLargeInternal(
71  int size_in_bytes, AllocationType allocation, AllocationOrigin origin,
72  AllocationAlignment alignment, AllocationHint hint) {
73  DCHECK_GT(size_in_bytes, heap_->MaxRegularHeapObjectSize(allocation));
74  switch (allocation) {
75  case AllocationType::kYoung:
76  return new_lo_space()->AllocateRaw(local_heap_, size_in_bytes, hint);
77  case AllocationType::kOld:
78  return lo_space()->AllocateRaw(local_heap_, size_in_bytes, hint);
79  case AllocationType::kCode:
80  return code_lo_space()->AllocateRaw(local_heap_, size_in_bytes, hint);
81  case AllocationType::kSharedOld:
82  return shared_lo_space()->AllocateRaw(local_heap_, size_in_bytes, hint);
83  case AllocationType::kTrusted:
84  return trusted_lo_space()->AllocateRaw(local_heap_, size_in_bytes, hint);
85  case AllocationType::kSharedTrusted:
86  return shared_trusted_lo_space()->AllocateRaw(local_heap_, size_in_bytes,
87  hint);
88  case AllocationType::kMap:
89  case AllocationType::kReadOnly:
90  case AllocationType::kSharedMap:
91  UNREACHABLE();
92  }
93 }

127 void HeapAllocator::CollectGarbage(
128  AllocationType allocation, PerformHeapLimitCheck perform_heap_limit_check) {
129  if (IsSharedAllocationType(allocation)) {
130  heap_->CollectGarbageShared(local_heap_,
131  GarbageCollectionReason::kAllocationFailure);
132  } else if (local_heap_->is_main_thread()) {
133  // On the main thread we can directly start the GC.
134  AllocationSpace space_to_gc = AllocationTypeToGCSpace(allocation);
135  heap_->CollectGarbage(space_to_gc,
136  GarbageCollectionReason::kAllocationFailure,
137  kNoGCCallbackFlags, perform_heap_limit_check);
138  } else {
139  // Request GC from main thread.
140  heap_->CollectGarbageFromAnyThread(local_heap_);
141  }
142 }

166 bool HeapAllocator::TryResizeLargeObject(Tagged<HeapObject> object,
167  size_t old_object_size,
168  size_t new_object_size) {
169  if (V8_UNLIKELY(!v8_flags.resize_large_object)) {
170  return false;
171  }
172 
173  PageMetadata* page = PageMetadata::FromHeapObject(object);
174  Space* space = page->owner();
175  if (space->identity() != NEW_LO_SPACE && space->identity() != LO_SPACE) {
176  return false;
177  }
178  DCHECK(page->IsLargePage());
179  DCHECK_EQ(page->area_size(), old_object_size);
180  CHECK_GT(new_object_size, old_object_size);
181  if (!heap_->memory_allocator()->ResizeLargePage(
182  LargePageMetadata::cast(page), old_object_size, new_object_size)) {
183  if (V8_UNLIKELY(v8_flags.trace_resize_large_object)) {
184  heap_->isolate()->PrintWithTimestamp(
185  "resizing large object failed: allocation could not be extended\n");
186  }
187  return false;
188  }
189 
190  LargeObjectSpace* large_space = static_cast<LargeObjectSpace*>(page->owner());
191  large_space->UpdateAccountingAfterResizingObject(old_object_size,
192  new_object_size);
193  return true;
194 }

397 void HeapAllocator::UpdateAllocationTimeout() {
398  if (v8_flags.random_gc_interval > 0) {
399  const int new_timeout = heap_->isolate()->fuzzer_rng()->NextInt(
400  v8_flags.random_gc_interval + 1);
401  // Reset the allocation timeout, but make sure to allow at least a few
402  // allocations after a collection. The reason for this is that we have a lot
403  // of allocation sequences and we assume that a garbage collection will
404  // allow the subsequent allocation attempts to go through.
405  constexpr int kFewAllocationsHeadroom = 6;
406  int timeout = std::max(kFewAllocationsHeadroom, new_timeout);
407  SetAllocationTimeout(timeout);
408  DCHECK(allocation_timeout_.has_value());
409  return;
410  }
411 
412  int timeout = allocation_gc_interval_.load(std::memory_order_relaxed);
413  SetAllocationTimeout(timeout);
414 }

416 bool HeapAllocator::ReachedAllocationTimeout() {
417  DCHECK(allocation_timeout_.has_value());
418 
419  if (heap_->always_allocate() || local_heap_->IsRetryOfFailedAllocation()) {
420  return false;
421  }
422 
423  allocation_timeout_ = std::max(0, allocation_timeout_.value() - 1);
424  return allocation_timeout_.value() <= 0;
