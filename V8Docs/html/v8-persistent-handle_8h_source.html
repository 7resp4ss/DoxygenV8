539 template <class T>
540 template <class S>
541 Global<T>& Global<T>::operator=(Global<S>&& rhs) {
542  static_assert(std::is_base_of_v<T, S>, "type check");
543  if (this != &rhs) {
544  this->Reset();
545  if (!rhs.IsEmpty()) {
546  this->slot() = rhs.slot();
547  api_internal::MoveGlobalReference(&rhs.slot(), &this->slot());
548  rhs.Clear();
549  }
550  }
551  return *this;
