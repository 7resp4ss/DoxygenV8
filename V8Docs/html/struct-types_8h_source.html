58  // Returns the offset of this field in the runtime representation of the
59  // object, from the start of the object fields (disregarding the object
60  // header).
61  uint32_t field_offset(uint32_t index) const {
62  DCHECK_LT(index, field_count());
63  if (index == 0) {
64  return is_descriptor() ? kTaggedSize : 0;
65  }
66  DCHECK(offsets_initialized_);
67  return field_offsets_[index - 1];
68  }
69  uint32_t total_fields_size() const {
70  if (field_count() == 0) {
71  return is_descriptor() ? kTaggedSize : 0;
72  }
73  return field_offsets_[field_count() - 1];
74  }

76  uint32_t Align(uint32_t offset, uint32_t alignment, bool is_shared) {
77  return RoundUp(
78  offset,
79  std::min(alignment,
80  static_cast<uint32_t>(is_shared ? kDoubleSize : kTaggedSize)));
81  }

83  void InitializeOffsets() {
84  if (field_count() == 0) return;
85  DCHECK(!offsets_initialized_);
86  if (is_descriptor() && is_shared()) {
87  // TODO(42204563, 403372470): Implement shared custom descriptors and
88  // update the offset calculation for the first field.
89  UNIMPLEMENTED();
90  }
91  uint32_t offset = is_descriptor() ? kTaggedSize : 0;
92  offset += field(0).value_kind_size();
93  // Optimization: we track the last gap that was introduced by alignment,
94  // and place any sufficiently-small fields in it.
95  // It's important that the algorithm that assigns offsets to fields is
96  // subtyping-safe, i.e. two lists of fields with a common prefix must
97  // always compute the same offsets for the fields in this common prefix.
98  uint32_t gap_position = 0;
99  uint32_t gap_size = 0;
100  for (uint32_t i = 1; i < field_count(); i++) {
101  uint32_t field_size = field(i).value_kind_size();
102  if (field_size <= gap_size) {
103  uint32_t aligned_gap = Align(gap_position, field_size, is_shared());
104  uint32_t gap_before = aligned_gap - gap_position;
105  uint32_t aligned_gap_size = gap_size - gap_before;
106  if (field_size <= aligned_gap_size) {
107  field_offsets_[i - 1] = aligned_gap;
108  uint32_t gap_after = aligned_gap_size - field_size;
109  if (gap_before > gap_after) {
110  // Keep old {gap_position}.
111  gap_size = gap_before;
112  } else {
113  gap_position = aligned_gap + field_size;
114  gap_size = gap_after;
115  }
116  continue; // Successfully placed the field in the gap.
117  }
118  }
119  uint32_t old_offset = offset;
120  offset = Align(offset, field_size, is_shared());
121  uint32_t gap = offset - old_offset;
122  if (gap > gap_size) {
123  gap_size = gap;
124  gap_position = old_offset;
125  }
126  field_offsets_[i - 1] = offset;
127  offset += field_size;
128  }
129  offset = RoundUp(offset, kTaggedSize);
130  field_offsets_[field_count() - 1] = offset;
131 #if DEBUG
132  offsets_initialized_ = true;
133 #endif
134  }

144  // For incrementally building StructTypes.
145  template <class Subclass, class ValueTypeSubclass>
146  class BuilderImpl {
147  public:
148  enum ComputeOffsets : bool {
149  kComputeOffsets = true,
150  kUseProvidedOffsets = false
151  };
152 
153  BuilderImpl(Zone* zone, uint32_t field_count, bool is_descriptor,
154  bool is_shared)
155  : zone_(zone),
156  field_count_(field_count),
157  is_descriptor_(is_descriptor),
158  is_shared_(is_shared),
159  cursor_(0),
160  field_offsets_(zone_->AllocateArray<uint32_t>(field_count_)),
161  buffer_(zone->AllocateArray<ValueTypeSubclass>(
162  static_cast<int>(field_count))),
163  mutabilities_(
164  zone->AllocateArray<bool>(static_cast<int>(field_count))) {}
165 
166  void AddField(ValueTypeSubclass type, bool mutability,
167  uint32_t offset = 0) {
168  DCHECK_LT(cursor_, field_count_);
169  if (cursor_ > 0) {
170  field_offsets_[cursor_ - 1] = offset;
171  } else {
172  // offset == 0 could mean that we'll compute the offsets later,
173  // or that this is the first field's offset being copied over from
174  // another struct type.
175  DCHECK(offset == 0 || (is_descriptor_ && offset == kTaggedSize));
176  }
177  mutabilities_[cursor_] = mutability;
178  buffer_[cursor_++] = type;
179  }
180 
181  void set_total_fields_size(uint32_t size) {
182  if (field_count_ == 0) {
183  DCHECK_EQ(is_descriptor_ ? kTaggedSize : 0, size);
184  return;
185  }
186  field_offsets_[field_count_ - 1] = size;
187  }
188 
189  Subclass* Build(ComputeOffsets compute_offsets = kComputeOffsets) {
190  DCHECK_EQ(cursor_, field_count_);
191  Subclass* result =
192  zone_->New<Subclass>(field_count_, field_offsets_, buffer_,
193  mutabilities_, is_descriptor_, is_shared_);
194  if (compute_offsets == kComputeOffsets) {
195  result->InitializeOffsets();
196  } else {
197 #if DEBUG
198  bool offsets_specified = true;
199  for (uint32_t i = 0; i < field_count_; i++) {
200  if (field_offsets_[i] == 0) {
201  offsets_specified = false;
202  break;
203  }
204  }
205  result->offsets_initialized_ = offsets_specified;
206 #endif
207  }
208  return result;
209  }
210 
211  private:
212  Zone* const zone_;
213  const uint32_t field_count_;
214  const bool is_descriptor_;
215  const bool is_shared_;
216  uint32_t cursor_;
217  uint32_t* field_offsets_;
218  ValueTypeSubclass* const buffer_;
219  bool* const mutabilities_;
220  };
