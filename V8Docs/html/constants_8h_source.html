143 inline bool IsConstexprName(const std::string& name) {
144  return name.substr(0, std::strlen(CONSTEXPR_TYPE_PREFIX)) ==
145  CONSTEXPR_TYPE_PREFIX;
146 }
147 
148 inline std::string GetNonConstexprName(const std::string& name) {
149  if (!IsConstexprName(name)) return name;
150  return name.substr(std::strlen(CONSTEXPR_TYPE_PREFIX));
151 }
152 
153 inline std::string GetConstexprName(const std::string& name) {
154  if (IsConstexprName(name)) return name;
155  return CONSTEXPR_TYPE_PREFIX + name;
156 }
