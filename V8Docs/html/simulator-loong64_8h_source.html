284  // Only arguments up to 64 bits in size are supported.
285  class CallArgument {
286  public:
287  template <typename T>
288  explicit CallArgument(T argument) {
289  bits_ = 0;
290  DCHECK(sizeof(argument) <= sizeof(bits_));
291  bits_ = ConvertArg(argument);
292  type_ = GP_ARG;
293  }
294 
295  explicit CallArgument(double argument) {
296  DCHECK(sizeof(argument) == sizeof(bits_));
297  memcpy(&bits_, &argument, sizeof(argument));
298  type_ = FP_ARG;
299  }
300 
301  explicit CallArgument(float argument) {
302  // Make the FPU register a NaN to try to trap errors if the callee expects
303  // a double. If it expects a float, the callee should ignore the top word.
304  double sNaN = std::numeric_limits<double>::signaling_NaN();
305  DCHECK(sizeof(sNaN) == sizeof(bits_));
306  memcpy(&bits_, &sNaN, sizeof(sNaN));
307 
308  // Write the float payload to the FPU register.
309  DCHECK(sizeof(argument) <= sizeof(bits_));
310  memcpy(&bits_, &argument, sizeof(argument));
311  type_ = FP_ARG;
312  }

319  bool IsEnd() const { return type_ == NO_ARG; }

412  // "Probe" if an address range can be read. This is currently implemented
413  // by doing a 1-byte read of the last accessed byte, since the assumption is
414  // that if the last byte is accessible, also all lower bytes are accessible
415  // (which holds true for Wasm).
416  // Returns true if the access was successful, false if the access raised a
417  // signal which was then handled by the trap handler (also see
418  // {trap_handler::ProbeMemory}). If the access raises a signal which is not
419  // handled by the trap handler (e.g. because the current PC is not registered
420  // as a protected instruction), the signal will propagate and make the process
421  // crash. If no trap handler is available, this always returns true.
