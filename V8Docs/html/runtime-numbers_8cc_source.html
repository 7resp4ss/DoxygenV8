26 RUNTIME_FUNCTION(Runtime_StringParseInt) {
27  HandleScope handle_scope(isolate);
28  DCHECK_EQ(2, args.length());
29  Handle<Object> string = args.at(0);
30  Handle<Object> radix = args.at(1);
31 
32  // Convert {string} to a String first, and flatten it.
33  Handle<String> subject;
34  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, subject,
35  Object::ToString(isolate, string));
36  subject = String::Flatten(isolate, subject);
37 
38  // Convert {radix} to Int32.
39  if (!IsNumber(*radix)) {
40  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, radix,
41  Object::ToNumber(isolate, radix));
42  }
43  int radix32 = DoubleToInt32(Object::NumberValue(*radix));
44  if (radix32 != 0 && (radix32 < 2 || radix32 > 36)) {
45  return ReadOnlyRoots(isolate).nan_value();
46  }
47 
48  double result = StringToInt(isolate, subject, radix32);
49  return *isolate->factory()->NewNumber(result);
50 }
