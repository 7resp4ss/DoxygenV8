152 template <bool do_callback>
153 class V8_NODISCARD CallDepthScope {
154  public:
155  CallDepthScope(i::Isolate* isolate, Local<Context> context)
156  : isolate_(isolate), saved_context_(isolate->context(), isolate_) {
157  isolate_->thread_local_top()->IncrementCallDepth<do_callback>(this);
158  i::Tagged<i::NativeContext> env = *Utils::OpenDirectHandle(*context);
159  isolate->set_context(env);
160 
161  if (do_callback) isolate_->FireBeforeCallEnteredCallback();
162  }
163  explicit CallDepthScope(i::Isolate* isolate) : isolate_(isolate) {
164  isolate_->thread_local_top()->IncrementCallDepth<do_callback>(this);
165 
166  if (do_callback) isolate_->FireBeforeCallEnteredCallback();
167  }
168  ~CallDepthScope() {
169  i::MicrotaskQueue* microtask_queue =
170  i::Cast<i::NativeContext>(isolate_->context())
171  ->microtask_queue(isolate_);
172 
173  isolate_->thread_local_top()->DecrementCallDepth(this);
174  // Clear the exception when exiting V8 to avoid memory leaks.
175  // Also clear termination exceptions iff there's no TryCatch handler.
176  // TODO(verwaest): Drop this once we propagate exceptions to external
177  // TryCatch on Throw. This should be debug-only.
178  if (isolate_->thread_local_top()->CallDepthIsZero() &&
179  (isolate_->thread_local_top()->try_catch_handler_ == nullptr ||
180  !isolate_->is_execution_terminating())) {
181  isolate_->clear_internal_exception();
182  }
183  if (do_callback) isolate_->FireCallCompletedCallback(microtask_queue);
184 #ifdef DEBUG
185  if (do_callback) {
186  if (microtask_queue && microtask_queue->microtasks_policy() ==
187  v8::MicrotasksPolicy::kScoped) {
188  DCHECK(microtask_queue->GetMicrotasksScopeDepth() ||
189  !microtask_queue->DebugMicrotasksScopeDepthIsZero());
190  }
191  }
192  DCHECK(CheckKeptObjectsClearedAfterMicrotaskCheckpoint(microtask_queue));
193 #endif
194 
195  if (!saved_context_.is_null()) isolate_->set_context(*saved_context_);
196  }
197 
198  CallDepthScope(const CallDepthScope&) = delete;
199  CallDepthScope& operator=(const CallDepthScope&) = delete;
200 
201  private:
202 #ifdef DEBUG
203  bool CheckKeptObjectsClearedAfterMicrotaskCheckpoint(
204  i::MicrotaskQueue* microtask_queue) {
205  bool did_perform_microtask_checkpoint =
206  isolate_->thread_local_top()->CallDepthIsZero() && do_callback &&
207  microtask_queue &&
208  microtask_queue->microtasks_policy() == MicrotasksPolicy::kAuto &&
209  !isolate_->is_execution_terminating();
210  return !did_perform_microtask_checkpoint ||
211  IsUndefined(isolate_->heap()->weak_refs_keep_during_job(), isolate_);
212  }
213 #endif
214 
215  i::Isolate* const isolate_;
216  i::Handle<i::Context> saved_context_;
217 
218  i::Address previous_stack_height_;
219 
220  friend class i::ThreadLocalTop;
221 
222  DISALLOW_NEW_AND_DELETE()
223 };

425 template <typename T>
426 void CopySmiElementsToTypedBuffer(T* dst, uint32_t length,
427  i::Tagged<i::FixedArray> elements) {
428  for (uint32_t i = 0; i < length; ++i) {
429  double value = i::Object::NumberValue(
430  i::Cast<i::Smi>(elements->get(static_cast<int>(i))));
431  // TODO(mslekova): Avoid converting back-and-forth when possible, e.g
432  // avoid int->double->int conversions to boost performance.
433  dst[i] = i::ConvertDouble<T>(value);
434  }
435 }

438 void CopyDoubleElementsToTypedBuffer(T* dst, uint32_t length,
439  i::Tagged<i::FixedDoubleArray> elements) {

442  // TODO(mslekova): There are certain cases, e.g. double->double, in which
443  // we could do a memcpy directly.

445  }

448 template <CTypeInfo::Identifier type_info_id, typename T>
449 bool CopyAndConvertArrayToCppBuffer(Local<Array> src, T* dst,
450  uint32_t max_length) {
451  static_assert(
452  std::is_same_v<T, typename i::CTypeInfoTraits<
453  CTypeInfo(type_info_id).GetType()>::ctype>,
454  "Type mismatch between the expected CTypeInfo::Type and the destination "
455  "array");
456 
457  uint32_t length = src->Length();
458  if (length == 0) {
459  // Early return here to avoid a cast error below, as the EmptyFixedArray
460  // cannot be cast to a FixedDoubleArray.
461  return true;
462  }
463  if (length > max_length) {
464  return false;
465  }
466 
467  i::DisallowGarbageCollection no_gc;
468  i::Tagged<i::JSArray> obj = *Utils::OpenDirectHandle(*src);
469  if (i::Object::IterationHasObservableEffects(obj)) {
470  // The array has a custom iterator.
471  return false;
472  }
473 
474  i::Tagged<i::FixedArrayBase> elements = obj->elements();
475  switch (obj->GetElementsKind()) {
476  case i::PACKED_SMI_ELEMENTS:
477  CopySmiElementsToTypedBuffer(dst, length,
478  i::Cast<i::FixedArray>(elements));
479  return true;
480  case i::PACKED_DOUBLE_ELEMENTS:
481  CopyDoubleElementsToTypedBuffer(dst, length,
482  i::Cast<i::FixedDoubleArray>(elements));
483  return true;
484  default:
485  return false;
486  }
487 }
