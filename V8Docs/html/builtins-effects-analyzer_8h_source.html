14 // BuiltinsEffectsAnalyzer is used when compiling builtins to compute the side
15 // effects of builtins. Currently, it is only used to compute which builtins can
16 // allocate.
17 //
18 // When compiling each builtins, CsaEffectsComputationPhase computes the effects
19 // of this builtin and records them in BuiltinsEffectsAnalyzer by calling
20 // RegisterInitialEffects. At this stage, we don't have a definitive answer for
21 // all builtins, since some builtins call other builtins that may not yet be
22 // compiled and whose effects have yet to be computed.
23 //
24 // Once we've done this for all builtins, we call
25 // BuiltinsEffectsAnalyzer::Finalize to take into account calls to other
26 // builtins (eg, if builtin A doesn't directly allocate but it calls builtin B
27 // which can allocate, then A can allocate).
28 //
29 // Finally, BuiltinsEffectsAnalyzer::Write generates a C++ file containing a
30 // `bool CanAllocate(Builtin)` function, which is used in Turboshaft and Maglev
31 // to DCHECK that correct CanAllocate effects are used.
32 
33 enum class BuiltinAllocateEffect {
34  kUninitialized,
35  kNo, // This builtin cannot allocate.
36  kYes, // This builtin can allocate.
37  kMaybeWithBuiltinCall // This builtin cannot directly allocate, but it calls
38  // other builtins (and we don't know yet whether those
39  // other builtins allocate or not).
40 };
41 
42 struct BuiltinEffects {
43  BuiltinAllocateEffect can_allocate = BuiltinAllocateEffect::kUninitialized;
44  std::unordered_set<Builtin> callees;
45  bool is_finalized = false;
46 };
