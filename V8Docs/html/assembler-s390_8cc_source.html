63 static bool supportsCPUFeature(const char* feature) {
64 #if V8_OS_ZOS
65  // TODO(gabylb): zos - use cpu_init() and cpu_supports() to test support of
66  // z/OS features when the current compiler supports them.
67  // Currently the only feature to be checked is Vector Extension Facility
68  // ("vector128" on z/OS, "vx" on LoZ) - hence the assert in case that changed.
69  assert(strcmp(feature, "vx") == 0);
70  return __is_vxf_available();
71 #else
72  static std::set<std::string>& features = *new std::set<std::string>();
73  static std::set<std::string>& all_available_features =
74  *new std::set<std::string>({"iesan3", "zarch", "stfle", "msa", "ldisp",
75  "eimm", "dfp", "etf3eh", "highgprs", "te",
76  "vx"});
77  if (features.empty()) {
78 #if V8_HOST_ARCH_S390X
79 
80 #ifndef HWCAP_S390_VX
81 #define HWCAP_S390_VX 2048
82 #endif
83 #define CHECK_AVAILABILITY_FOR(mask, value) \
84  if (f & mask) features.insert(value);
85 
86  // initialize feature vector
87  uint64_t f = getauxval(AT_HWCAP);
88  CHECK_AVAILABILITY_FOR(HWCAP_S390_ESAN3, "iesan3")
89  CHECK_AVAILABILITY_FOR(HWCAP_S390_ZARCH, "zarch")
90  CHECK_AVAILABILITY_FOR(HWCAP_S390_STFLE, "stfle")
91  CHECK_AVAILABILITY_FOR(HWCAP_S390_MSA, "msa")
92  CHECK_AVAILABILITY_FOR(HWCAP_S390_LDISP, "ldisp")
93  CHECK_AVAILABILITY_FOR(HWCAP_S390_EIMM, "eimm")
94  CHECK_AVAILABILITY_FOR(HWCAP_S390_DFP, "dfp")
95  CHECK_AVAILABILITY_FOR(HWCAP_S390_ETF3EH, "etf3eh")
96  CHECK_AVAILABILITY_FOR(HWCAP_S390_HIGH_GPRS, "highgprs")
97  CHECK_AVAILABILITY_FOR(HWCAP_S390_TE, "te")
98  CHECK_AVAILABILITY_FOR(HWCAP_S390_VX, "vx")
99 #else
100  // import all features
101  features.insert(all_available_features.begin(),
102  all_available_features.end());
103 #endif
104  }
105  USE(all_available_features);
106  return features.find(feature) != features.end();
107 #endif // !V8_OS_ZOS
108 }

113 // Check whether Store Facility STFLE instruction is available on the platform.
114 // Instruction returns a bit vector of the enabled hardware facilities.
115 static bool supportsSTFLE() {
116 #if V8_OS_ZOS
117  return __is_stfle_available();
118 #elif V8_HOST_ARCH_S390X
119  static bool read_tried = false;
120  static uint32_t auxv_hwcap = 0;
121 
122  if (!read_tried) {
123  // Open the AUXV (auxiliary vector) pseudo-file
124  int fd = open("/proc/self/auxv", O_RDONLY);
125 
126  read_tried = true;
127  if (fd != -1) {
128  static Elf64_auxv_t buffer[16];
129  Elf64_auxv_t* auxv_element;
130  int bytes_read = 0;
131  while (bytes_read >= 0) {
132  // Read a chunk of the AUXV
133  bytes_read = read(fd, buffer, sizeof(buffer));
134  // Locate and read the platform field of AUXV if it is in the chunk
135  for (auxv_element = buffer;
136  auxv_element + sizeof(auxv_element) <= buffer + bytes_read &&
137  auxv_element->a_type != AT_NULL;
138  auxv_element++) {
139  // We are looking for HWCAP entry in AUXV to search for STFLE support
140  if (auxv_element->a_type == AT_HWCAP) {
141  /* Note: Both auxv_hwcap and buffer are static */
142  auxv_hwcap = auxv_element->a_un.a_val;
143  goto done_reading;
144  }
145  }
146  }
147  done_reading:
148  close(fd);
149  }
150  }
151 
152  // Did not find result
153  if (0 == auxv_hwcap) {
154  return false;
155  }
156 
157  // HWCAP_S390_STFLE is defined to be 4 in include/asm/elf.h. Currently
158  // hardcoded in case that include file does not exist.
159  const uint32_t _HWCAP_S390_STFLE = 4;
160  return (auxv_hwcap & _HWCAP_S390_STFLE);
161 #else
162  // STFLE is not available on non-s390 hosts
163  return false;
164 #endif
165 }

175 void CpuFeatures::ProbeImpl(bool cross_compile) {
176  supported_ |= CpuFeaturesImpliedByCompiler();
177  icache_line_size_ = 256;
178 
179  // Only use statically determined features for cross compile (snapshot).
180  if (cross_compile) return;
181 
182 #ifdef DEBUG
183  initialized_ = true;
184 #endif
185 
186  static bool performSTFLE = supportsSTFLE();
187 
188 // Need to define host, as we are generating inlined S390 assembly to test
189 // for facilities.
190 #if V8_HOST_ARCH_S390X
191  if (performSTFLE) {
192  // STFLE D(B) requires:
193  // GPR0 to specify # of double words to update minus 1.
194  // i.e. GPR0 = 0 for 1 doubleword
195  // D(B) to specify to memory location to store the facilities bits
196  // The facilities we are checking for are:
197  // Bit 45 - Distinct Operands for instructions like ARK, SRK, etc.
198  // As such, we require only 1 double word
199  int64_t facilities[4] = {0L};
200 #if V8_OS_ZOS
201  int64_t reg0 = 3;
202  asm volatile(" stfle %0" : "=m"(facilities), __ZL_NR("+", r0)(reg0)::"cc");
203 #else
204  int16_t reg0;
205  // LHI sets up GPR0
206  // STFLE is specified as .insn, as opcode is not recognized.
207  // We register the instructions kill r0 (LHI) and the CC (STFLE).
208  asm volatile(
209  "lhi %%r0,3\n"
210  ".insn s,0xb2b00000,%0\n"
211  : "=Q"(facilities), "=r"(reg0)
212  :
213  : "cc", "r0");
214 #endif // V8_OS_ZOS
215 
216  uint64_t one = static_cast<uint64_t>(1);
217  // Test for Distinct Operands Facility - Bit 45
218  if (facilities[0] & (one << (63 - 45))) {
219  supported_ |= (1u << DISTINCT_OPS);
220  }
221  // Test for General Instruction Extension Facility - Bit 34
222  if (facilities[0] & (one << (63 - 34))) {
223  supported_ |= (1u << GENERAL_INSTR_EXT);
224  }
225  // Test for Floating Point Extension Facility - Bit 37
226  if (facilities[0] & (one << (63 - 37))) {
227  supported_ |= (1u << FLOATING_POINT_EXT);
228  }
229  // Test for Vector Facility - Bit 129
230  if (facilities[2] & (one << (63 - (129 - 128))) &&
231  supportsCPUFeature("vx")) {
232  supported_ |= (1u << VECTOR_FACILITY);
233  }
234  // Test for Vector Enhancement Facility 1 - Bit 135
235  if (facilities[2] & (one << (63 - (135 - 128))) &&
236  supportsCPUFeature("vx")) {
237  supported_ |= (1u << VECTOR_ENHANCE_FACILITY_1);
238  }
239  // Test for Vector Enhancement Facility 2 - Bit 148
240  if (facilities[2] & (one << (63 - (148 - 128))) &&
241  supportsCPUFeature("vx")) {
242  supported_ |= (1u << VECTOR_ENHANCE_FACILITY_2);
243  }
244  // Test for Vector Enhancement Facility 3 - Bit 198
245  if (facilities[3] & (one << (63 - (198 - 192))) &&
246  supportsCPUFeature("vx")) {
247  supported_ |= (1u << VECTOR_ENHANCE_FACILITY_3);
248  }
249  // Test for Miscellaneous Instruction Extension Facility 2 - Bit 58
250  if (facilities[0] & (1lu << (63 - 58))) {
251  supported_ |= (1u << MISC_INSTR_EXT2);
252  }
253  // Test for Miscellaneous Instruction Extension Facility 4 - Bit 84
254  if (facilities[1] & (one << (63 - (84 - 64)))) {
255  supported_ |= (1u << MISC_INSTR_EXT4);
256  }
257  }
258 #else
259  // All distinct ops instructions can be simulated
260  supported_ |= (1u << DISTINCT_OPS);
261  // RISBG can be simulated
262  supported_ |= (1u << GENERAL_INSTR_EXT);
263  supported_ |= (1u << FLOATING_POINT_EXT);
264  supported_ |= (1u << MISC_INSTR_EXT2);
265  supported_ |= (1u << MISC_INSTR_EXT4);
266  USE(performSTFLE); // To avoid assert
267  USE(supportsCPUFeature);
268  supported_ |= (1u << VECTOR_FACILITY);
269  supported_ |= (1u << VECTOR_ENHANCE_FACILITY_1);
270  supported_ |= (1u << VECTOR_ENHANCE_FACILITY_2);
271  supported_ |= (1u << VECTOR_ENHANCE_FACILITY_3);
272 #endif
273  supported_ |= (1u << FPU);
274 
275  // Set a static value on whether Simd is supported.
276  // This variable is only used for certain archs to query SupportWasmSimd128()
277  // at runtime in builtins using an extern ref. Other callers should use
278  // CpuFeatures::SupportWasmSimd128().
279  CpuFeatures::supports_wasm_simd_128_ = CpuFeatures::SupportsWasmSimd128();
280 }

387 void Assembler::GetCode(LocalIsolate* isolate, CodeDesc* desc,
388  SafepointTableBuilderBase* safepoint_table_builder,
389  int handler_table_offset) {
390  // As a crutch to avoid having to add manual Align calls wherever we use a
391  // raw workflow to create Code objects (mostly in tests), add another Align
392  // call here. It does no harm - the end of the Code object is aligned to the
393  // (larger) kCodeAlignment anyways.
394  // TODO(jgruber): Consider moving responsibility for proper alignment to
395  // metadata table builders (safepoint, handler, constant pool, code
396  // comments).
397  DataAlign(InstructionStream::kMetadataAlignment);
398 
399  EmitRelocations();
400 
401  int code_comments_size = WriteCodeComments();
402 
403  AllocateAndInstallRequestedHeapNumbers(isolate);
404 
405  // Set up code descriptor.
406  // TODO(jgruber): Reconsider how these offsets and sizes are maintained up to
407  // this point to make CodeDesc initialization less fiddly.
408 
409  static constexpr int kConstantPoolSize = 0;
410  static constexpr int kBuiltinJumpTableInfoSize = 0;
411  const int instruction_size = pc_offset();
412  const int builtin_jump_table_info_offset =
413  instruction_size - kBuiltinJumpTableInfoSize;
414  const int code_comments_offset =
415  builtin_jump_table_info_offset - code_comments_size;
416  const int constant_pool_offset = code_comments_offset - kConstantPoolSize;
417  const int handler_table_offset2 = (handler_table_offset == kNoHandlerTable)
418  ? constant_pool_offset
419  : handler_table_offset;
420  const int safepoint_table_offset =
421  (safepoint_table_builder == kNoSafepointTable)
422  ? handler_table_offset2
423  : safepoint_table_builder->safepoint_table_offset();
424  const int reloc_info_offset =
425  static_cast<int>(reloc_info_writer.pos() - buffer_->start());
426  CodeDesc::Initialize(desc, this, safepoint_table_offset,
427  handler_table_offset2, constant_pool_offset,
428  code_comments_offset, builtin_jump_table_info_offset,
429  reloc_info_offset);
430 }

458 // Labels refer to positions in the (to be) generated code.
459 // There are bound, linked, and unused labels.
460 //
461 // Bound labels refer to known positions in the already
462 // generated code. pos() is the position the label refers to.
463 //
464 // Linked labels refer to unknown positions in the code
465 // to be generated; pos() is the position of the last
466 // instruction using the label.
467 
468 // The link chain is terminated by a negative code position (must be aligned)
469 const int kEndOfChain = -4;
470 
471 // Returns the target address of the relative instructions, typically
472 // of the form: pos + imm (where immediate is in # of halfwords for
473 // BR* and LARL).
474 int Assembler::target_at(int pos) {
475  SixByteInstr instr = instr_at(pos);
476  // check which type of branch this is 16 or 26 bit offset
477  Opcode opcode = Instruction::S390OpcodeValue(buffer_start_ + pos);
478 
479  if (BRC == opcode || BRCT == opcode || BRCTG == opcode || BRXH == opcode) {
480  int16_t imm16 = SIGN_EXT_IMM16((instr & kImm16Mask));
481  imm16 <<= 1; // immediate is in # of halfwords
482  if (imm16 == 0) return kEndOfChain;
483  return pos + imm16;
484  } else if (LLILF == opcode || BRCL == opcode || LARL == opcode ||
485  BRASL == opcode || LGRL == opcode) {
486  int32_t imm32 =
487  static_cast<int32_t>(instr & (static_cast<uint64_t>(0xFFFFFFFF)));
488  if (LLILF != opcode)
489  imm32 <<= 1; // BR* + LARL treat immediate in # of halfwords
490  if (imm32 == 0) return kEndOfChain;
491  return pos + imm32;
492  } else if (BRXHG == opcode) {
493  // offset is in bits 16-31 of 48 bit instruction
494  instr = instr >> 16;
495  int16_t imm16 = SIGN_EXT_IMM16((instr & kImm16Mask));
496  imm16 <<= 1; // immediate is in # of halfwords
497  if (imm16 == 0) return kEndOfChain;
498  return pos + imm16;
499  }
500 
501  // Unknown condition
502  DCHECK(false);
503  return -1;
504 }

506 // Update the target address of the current relative instruction.
507 void Assembler::target_at_put(int pos, int target_pos, bool* is_branch) {
508  SixByteInstr instr = instr_at(pos);
509  Opcode opcode = Instruction::S390OpcodeValue(buffer_start_ + pos);
510 
511  if (is_branch != nullptr) {
512  *is_branch =
513  (opcode == BRC || opcode == BRCT || opcode == BRCTG || opcode == BRCL ||
514  opcode == BRASL || opcode == BRXH || opcode == BRXHG);
515  }
516 
517  if (BRC == opcode || BRCT == opcode || BRCTG == opcode || BRXH == opcode) {
518  int16_t imm16 = target_pos - pos;
519  instr &= (~0xFFFF);
520  DCHECK(is_int16(imm16));
521  instr_at_put<FourByteInstr>(pos, instr | (imm16 >> 1));
522  return;
523  } else if (BRCL == opcode || LARL == opcode || BRASL == opcode ||
524  LGRL == opcode) {
525  // Immediate is in # of halfwords
526  int32_t imm32 = target_pos - pos;
527  instr &= (~static_cast<uint64_t>(0xFFFFFFFF));
528  instr_at_put<SixByteInstr>(pos, instr | (imm32 >> 1));
529  return;
530  } else if (LLILF == opcode) {
531  DCHECK(target_pos == kEndOfChain || target_pos >= 0);
532  // Emitted label constant, not part of a branch.
533  // Make label relative to InstructionStream pointer of generated
534  // InstructionStream object.
535  int32_t imm32 =
536  target_pos + (InstructionStream::kHeaderSize - kHeapObjectTag);
537  instr &= (~static_cast<uint64_t>(0xFFFFFFFF));
538  instr_at_put<SixByteInstr>(pos, instr | imm32);
539  return;
540  } else if (BRXHG == opcode) {
541  // Immediate is in bits 16-31 of 48 bit instruction
542  int32_t imm16 = target_pos - pos;
543  instr &= (0xFFFF0000FFFF); // clear bits 16-31
544  imm16 &= 0xFFFF; // clear high halfword
545  imm16 <<= 16;
546  // Immediate is in # of halfwords
547  instr_at_put<SixByteInstr>(pos, instr | (imm16 >> 1));
548  return;
549  }
550  DCHECK(false);
551 }

553 // Returns the maximum number of bits given instruction can address.
554 int Assembler::max_reach_from(int pos) {
555  Opcode opcode = Instruction::S390OpcodeValue(buffer_start_ + pos);
556  // Check which type of instr. In theory, we can return
557  // the values below + 1, given offset is # of halfwords
558  if (BRC == opcode || BRCT == opcode || BRCTG == opcode || BRXH == opcode ||
559  BRXHG == opcode) {
560  return 16;
561  } else if (LLILF == opcode || BRCL == opcode || LARL == opcode ||
562  BRASL == opcode || LGRL == opcode) {

572 void Assembler::bind_to(Label* L, int pos) {
573  DCHECK(0 <= pos && pos <= pc_offset()); // must have a valid binding position
574  bool is_branch = false;
575  while (L->is_linked()) {
576  int fixup_pos = L->pos();
577 #ifdef DEBUG
578  int32_t offset = pos - fixup_pos;
579  int maxReach = max_reach_from(fixup_pos);
580 #endif
581  next(L); // call next before overwriting link with target at fixup_pos
582  DCHECK(is_intn(offset, maxReach));
583  target_at_put(fixup_pos, pos, &is_branch);
584  }
585  L->bind_to(pos);
586 
587  // Keep track of the last bound label so we don't eliminate any instructions
588  // before a bound label.
589  if (pos > last_bound_pos_) last_bound_pos_ = pos;
590 }

610  if (L->is_bound()) {
611  position = L->pos();
612  } else {
613  if (L->is_linked()) {
614  position = L->pos(); // L's link
615  } else {

617  // However, using self to mark the first reference
618  // should avoid most instances of branch offset overflow. See
619  // target_at() for where this is converted back to kEndOfChain.

769 void Assembler::GrowBuffer(int needed) {
770  DCHECK_EQ(buffer_start_, buffer_->start());
771 
772  // Compute new buffer size.
773  int old_size = buffer_->size();
774  int new_size = std::min(2 * old_size, old_size + 1 * MB);
775  int space = buffer_space() + (new_size - old_size);
776  new_size += (space < needed) ? needed - space : 0;
777 
778  // Some internal data structures overflow for very large buffers,
779  // they must ensure that kMaximalBufferSize is not too large.
780  if (new_size > kMaximalBufferSize) {
781  V8::FatalProcessOutOfMemory(nullptr, "Assembler::GrowBuffer");
782  }
783 
784  // Set up new buffer.
785  std::unique_ptr<AssemblerBuffer> new_buffer = buffer_->Grow(new_size);
786  DCHECK_EQ(new_size, new_buffer->size());
787  uint8_t* new_start = new_buffer->start();
788 
789  // Copy the data.
790  intptr_t pc_delta = new_start - buffer_start_;
791  intptr_t rc_delta = (new_start + new_size) - (buffer_start_ + old_size);
792  size_t reloc_size = (buffer_start_ + old_size) - reloc_info_writer.pos();
793  MemMove(new_start, buffer_start_, pc_offset());
794  MemMove(reloc_info_writer.pos() + rc_delta, reloc_info_writer.pos(),
795  reloc_size);
796 
797  // Switch buffers.
798  buffer_ = std::move(new_buffer);
799  buffer_start_ = new_start;
800  pc_ += pc_delta;
801  reloc_info_writer.Reposition(reloc_info_writer.pos() + rc_delta,
802  reloc_info_writer.last_pc() + pc_delta);
803 
804  // None of our relocation types are pc relative pointing outside the code
805  // buffer nor pc absolute pointing inside the code buffer, so there is no need
806  // to relocate any emitted relocation entries.
807 }

854 void Assembler::EmitRelocations() {
855  EnsureSpaceFor(relocations_.size() * kMaxRelocSize);
856 
857  for (std::vector<DeferredRelocInfo>::iterator it = relocations_.begin();
858  it != relocations_.end(); it++) {
859  RelocInfo::Mode rmode = it->rmode();
860  Address pc = reinterpret_cast<Address>(buffer_start_) + it->position();
861  RelocInfo rinfo(pc, rmode, it->data());
862 
863  // Fix up internal references now that they are guaranteed to be bound.
864  if (RelocInfo::IsInternalReference(rmode)) {
865  // Jump table entry
866  Address pos = Memory<Address>(pc);
867  Memory<Address>(pc) = reinterpret_cast<Address>(buffer_start_) + pos;
868  } else if (RelocInfo::IsInternalReferenceEncoded(rmode)) {
869  // mov sequence
870  Address pos = target_address_at(pc, 0);
871  set_target_address_at(pc, 0,
872  reinterpret_cast<Address>(buffer_start_) + pos,
873  nullptr, SKIP_ICACHE_FLUSH);
874  }
875 
876  reloc_info_writer.Write(&rinfo);
877  }
