79  // Moves all nodes after |node| to the resulting ZoneVector, while keeping all
80  // nodes before |node| in the basic block. |node| itself is dropped.
81  ZoneVector<Node*> Split(Node* node, Zone* zone) {
82  size_t split = 0;
83  for (; split < nodes_.size(); split++) {
84  if (nodes_[split] == node) break;
85  }
86  DCHECK_NE(split, nodes_.size());
87  size_t after_split = split + 1;
88  ZoneVector<Node*> result(nodes_.size() - after_split, zone);
89  for (size_t i = 0; i < result.size(); i++) {
90  result[i] = nodes_[i + after_split];
91  }
92  nodes_.resize(split);
93  return result;
94  }

189  template <typename Func>
190  void ForEachPredecessor(Func&& functor) const {
191  if (type_ == kEdgeSplit || type_ == kOther) {
192  BasicBlock* predecessor_block = predecessor();
193  if (predecessor_block) {
194  functor(predecessor_block);
195  }
196  } else {
197  for (int i = 0; i < predecessor_count(); i++) {
198  functor(predecessor_at(i));
199  }
200  }
201  }
202 
203  template <typename Func>
204  bool ForAllPredecessors(Func&& functor) const {
205  if (type_ == kEdgeSplit || type_ == kOther) {
206  BasicBlock* predecessor_block = predecessor();
207  if (predecessor_block) {
208  if (!functor(predecessor_block)) {
209  return false;
210  }
211  }
212  } else {
213  for (int i = 0; i < predecessor_count(); i++) {
214  if (!functor(predecessor_at(i))) {
215  return false;
216  }
217  }
218  }
219  return true;
220  }

222  template <typename Func>
223  static void ForEachSuccessorFollowing(ControlNode* control, Func&& functor) {
224  if (auto unconditional_control =
225  control->TryCast<UnconditionalControlNode>()) {
226  functor(unconditional_control->target());
227  } else if (auto branch = control->TryCast<BranchControlNode>()) {
228  functor(branch->if_true());
229  functor(branch->if_false());
230  } else if (auto switch_node = control->TryCast<Switch>()) {
231  for (int i = 0; i < switch_node->size(); i++) {
232  functor(switch_node->targets()[i].block_ptr());
233  }
234  if (switch_node->has_fallthrough()) {
235  functor(switch_node->fallthrough());
236  }
237  }
238  }

240  template <typename Func>
241  void ForEachSuccessor(Func&& functor) const {
242  ControlNode* control = control_node();
243  ForEachSuccessorFollowing(control, functor);
244  }

262  // If the basic block is an empty (unnecessary) block containing only an
263  // unconditional jump to the successor block, return the successor block.
264  BasicBlock* ComputeRealJumpTarget() {
265  BasicBlock* current = this;
266  while (true) {
267  if (!current->nodes_.empty() || current->is_loop() ||
268  current->is_exception_handler_block() ||
269  current->HasPhisOrRegisterMerges()) {
270  break;
271  }
272  Jump* control = current->control_node()->TryCast<Jump>();
273  if (!control) {
274  break;
275  }
276  BasicBlock* next = control->target();
277  if (next->HasPhisOrRegisterMerges()) {
278  break;
279  }
280  current = next;
281  }
282  return current;
283  }

301  private:
302  bool HasPhisOrRegisterMerges() const {
303  if (!has_state()) {
304  return false;
305  }
306  if (has_phi()) {
307  return true;
308  }
309  bool has_register_merge = false;
310 #ifdef V8_ENABLE_MAGLEV
311  if (!state()->register_state().is_initialized()) {
312  // This can happen when the graph has disconnected blocks; bail out and
313  // don't jump thread them.
314  return true;
315  }
316 
317  state()->register_state().ForEachGeneralRegister(
318  [&](Register reg, RegisterState& state) {
319  ValueNode* node;
320  RegisterMerge* merge;
321  if (LoadMergeState(state, &node, &merge)) {
322  has_register_merge = true;
323  }
324  });
325  state()->register_state().ForEachDoubleRegister(
326  [&](DoubleRegister reg, RegisterState& state) {
327  ValueNode* node;
328  RegisterMerge* merge;
329  if (LoadMergeState(state, &node, &merge)) {
330  has_register_merge = true;
331  }
332  });
333 #endif // V8_ENABLE_MAGLEV
334  return has_register_merge;
335  }

364 inline base::SmallVector<BasicBlock*, 2> BasicBlock::successors() const {
365  ControlNode* control = control_node();
366  if (auto unconditional_control =
367  control->TryCast<UnconditionalControlNode>()) {
368  return {unconditional_control->target()};
369  } else if (auto branch = control->TryCast<BranchControlNode>()) {
370  return {branch->if_true(), branch->if_false()};
371  } else if (auto switch_node = control->TryCast<Switch>()) {
372  base::SmallVector<BasicBlock*, 2> succs;
373  for (int i = 0; i < switch_node->size(); i++) {
374  succs.push_back(switch_node->targets()[i].block_ptr());
375  }
376  if (switch_node->has_fallthrough()) {
377  succs.push_back(switch_node->fallthrough());
378  }
379  return succs;
380  } else {
381  return base::SmallVector<BasicBlock*, 2>();
382  }
