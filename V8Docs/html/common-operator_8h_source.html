47 inline BranchHint NegateBranchHint(BranchHint hint) {
48  switch (hint) {
49  case BranchHint::kNone:
50  return hint;
51  case BranchHint::kTrue:
52  return BranchHint::kFalse;
53  case BranchHint::kFalse:
54  return BranchHint::kTrue;
55  }
56  UNREACHABLE();
57 }

275 // Used to define a sparse set of inputs. This can be used to efficiently encode
276 // nodes that can have a lot of inputs, but where many inputs can have the same
277 // value.
278 class SparseInputMask final {
279  public:
280  using BitMaskType = uint32_t;
281 
282  // The mask representing a dense input set.
283  static const BitMaskType kDenseBitMask = 0x0;
284  // The bits representing the end of a sparse input set.
285  static const BitMaskType kEndMarker = 0x1;
286  // The mask for accessing a sparse input entry in the bitmask.
287  static const BitMaskType kEntryMask = 0x1;
288 
289  // The number of bits in the mask, minus one for the end marker.
290  static const int kMaxSparseInputs = (sizeof(BitMaskType) * kBitsPerByte - 1);
291 
292  // An iterator over a node's sparse inputs.
293  class InputIterator final {
294  public:
295  InputIterator() = default;
296  InputIterator(BitMaskType bit_mask, Node* parent);
297 
298  Node* parent() const { return parent_; }
299  int real_index() const { return real_index_; }
300 
301  // Advance the iterator to the next sparse input. Only valid if the iterator
302  // has not reached the end.
303  void Advance();
304 
305  // Get the current sparse input's real node value. Only valid if the
306  // current sparse input is real.
307  Node* GetReal() const;
308 
309  // Advance to the next real value or the end. Only valid if the iterator is
310  // not dense. Returns the number of empty values that were skipped. This can
311  // return 0 and in that case, it does not advance.
312  size_t AdvanceToNextRealOrEnd();
313 
314  // Get the current sparse input, returning either a real input node if
315  // the current sparse input is real, or the given {empty_value} if the
316  // current sparse input is empty.
317  Node* Get(Node* empty_value) const {
318  return IsReal() ? GetReal() : empty_value;
319  }
320 
321  // True if the current sparse input is a real input node.
322  bool IsReal() const;
323 
324  // True if the current sparse input is an empty value.
325  bool IsEmpty() const { return !IsReal(); }
326 
327  // True if the iterator has reached the end of the sparse inputs.
328  bool IsEnd() const;
329 
330  private:
331  BitMaskType bit_mask_;
332  Node* parent_;
333  int real_index_;
334  };
335 
336  explicit SparseInputMask(BitMaskType bit_mask) : bit_mask_(bit_mask) {}
337 
338  // Provides a SparseInputMask representing a dense input set.
339  static SparseInputMask Dense() { return SparseInputMask(kDenseBitMask); }
340 
341  BitMaskType mask() const { return bit_mask_; }
342 
343  bool IsDense() const { return bit_mask_ == SparseInputMask::kDenseBitMask; }
344 
345  // Counts how many real values are in the sparse array. Only valid for
346  // non-dense masks.
347  int CountReal() const;
348 
349  // Returns an iterator over the sparse inputs of {node}.
350  InputIterator IterateOverInputs(Node* node);
351 
352  private:
353  //
354  // The sparse input mask has a bitmask specifying if the node's inputs are
355  // represented sparsely. If the bitmask value is 0, then the inputs are dense;
356  // otherwise, they should be interpreted as follows:
357  //
358  // * The bitmask represents which values are real, with 1 for real values
359  // and 0 for empty values.
360  // * The inputs to the node are the real values, in the order of the 1s from
361  // least- to most-significant.
362  // * The top bit of the bitmask is a guard indicating the end of the values,
363  // whether real or empty (and is not representative of a real input
364  // itself). This is used so that we don't have to additionally store a
365  // value count.
366  //
367  // So, for N 1s in the bitmask, there are N - 1 inputs into the node.
368  BitMaskType bit_mask_;
369 };

421 // The ArgumentsElementsState and ArgumentsLengthState can describe the layout
422 // for backing stores of arguments objects of various types:
423 //
424 // +------------------------------------+
425 // - kUnmappedArguments: | arg0, ... argK-1, argK, ... argN-1 | {length:N}
426 // +------------------------------------+
427 // +------------------------------------+
428 // - kMappedArguments: | hole, ... hole, argK, ... argN-1 | {length:N}
429 // +------------------------------------+
430 // +------------------+
431 // - kRestParameter: | argK, ... argN-1 | {length:N-K}
432 // +------------------+
433 //
434 // Here {K} represents the number for formal parameters of the active function,
435 // whereas {N} represents the actual number of arguments passed at runtime.
436 // Note that {N < K} can happen and causes {K} to be capped accordingly.
437 //
438 // Also note that it is possible for an arguments object of {kMappedArguments}
439 // type to carry a backing store of {kUnappedArguments} type when {K == 0}.
440 using ArgumentsStateType = CreateArgumentsType;
441

765  public:

767  DCHECK_EQ(IrOpcode::kStart, node->opcode());

770  // The receiver is counted as part of formal parameters.
771  static constexpr int kReceiverOutputCount = 1;
772  // These outputs are in addition to formal parameters.
773  static constexpr int kExtraOutputCount =
774  4 + V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE_BOOL;
775 
776  // Takes the formal parameter count of the current function (including
777  // receiver) and returns the number of value outputs of the start node.
778  static constexpr int OutputArityForFormalParameterCount(int argc) {
779  constexpr int kClosure = 1;
780  constexpr int kNewTarget = 1;
781  constexpr int kArgCount = 1;
782  constexpr int kDispatchHandle =
783  V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE_BOOL ? 1 : 0;
784  constexpr int kContext = 1;
785  static_assert(kClosure + kNewTarget + kArgCount + kDispatchHandle +
786  kContext ==
787  kExtraOutputCount);
788  // Checking related linkage methods here since they rely on Start node
789  // layout.
790  DCHECK_EQ(-1, Linkage::kJSCallClosureParamIndex);
791  DCHECK_EQ(argc + 0, Linkage::GetJSCallNewTargetParamIndex(argc));
792  DCHECK_EQ(argc + 1, Linkage::GetJSCallArgCountParamIndex(argc));
793 #ifdef V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE
794  DCHECK_EQ(argc + 2, Linkage::GetJSCallDispatchHandleParamIndex(argc));
795  DCHECK_EQ(argc + 3, Linkage::GetJSCallContextParamIndex(argc));
796 #else
797  DCHECK_EQ(argc + 2, Linkage::GetJSCallContextParamIndex(argc));
798 #endif
799  return argc + kClosure + kNewTarget + kArgCount + kDispatchHandle +
800  kContext;
801  }
802 
803  int FormalParameterCount() const {
804  DCHECK_GE(node()->op()->ValueOutputCount(),
805  kExtraOutputCount + kReceiverOutputCount);
806  return node()->op()->ValueOutputCount() - kExtraOutputCount;
807  }
808 
809  int FormalParameterCountWithoutReceiver() const {
810  DCHECK_GE(node()->op()->ValueOutputCount(),
811  kExtraOutputCount + kReceiverOutputCount);
812  return node()->op()->ValueOutputCount() - kExtraOutputCount -
813  kReceiverOutputCount;
814  }
815 
816  // Note these functions don't return the index of the Start output; instead
817  // they return the index assigned to the Parameter node.
818  // TODO(jgruber): Consider unifying the two.
819  int NewTargetParameterIndex() const {
820  return Linkage::GetJSCallNewTargetParamIndex(FormalParameterCount());
821  }
822  int ArgCountParameterIndex() const {
823  return Linkage::GetJSCallArgCountParamIndex(FormalParameterCount());
824  }
825 #ifdef V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE
826  int DispatchHandleOutputIndex() const {
827  return Linkage::GetJSCallDispatchHandleParamIndex(FormalParameterCount());
828  }
829 #endif
830  int ContextParameterIndex() const {
831  return Linkage::GetJSCallContextParamIndex(FormalParameterCount());
832  }
833 
834  // TODO(jgruber): Remove this function and use
835  // Linkage::GetJSCallContextParamIndex instead. This currently doesn't work
836  // because tests don't create valid Start nodes - for example, they may add
837  // only two context outputs (and not the closure, new target, argc). Once
838  // tests are fixed, remove this function.
839  int ContextParameterIndex_MaybeNonStandardLayout() const {
840  // The context is always the last parameter to a JavaScript function, and
841  // {Parameter} indices start at -1, so value outputs of {Start} look like
842  // this: closure, receiver, param0, ..., paramN, context.
843  //
844  // TODO(jgruber): This function is called from spots that operate on
845  // CSA/Torque graphs; Start node layout appears to be different there.
846  // These should be unified to avoid confusion. Once done, enable this
847  // DCHECK: DCHECK_EQ(LastOutputIndex(), ContextOutputIndex());
848  return node()->op()->ValueOutputCount() - 2;
849  }
850  int LastParameterIndex_MaybeNonStandardLayout() const {
851  return ContextParameterIndex_MaybeNonStandardLayout();
852  }
853 
854  // Unlike ContextParameterIndex_MaybeNonStandardLayout above, these return
855  // output indices (and not the index assigned to a Parameter).
856  int NewTargetOutputIndex() const {
857  // Indices assigned to parameters are off-by-one (Parameters indices start
858  // at -1). TODO(jgruber): Consider starting at 0.
859  return Linkage::GetJSCallNewTargetParamIndex(FormalParameterCount()) + 1;
860  }
861  int ArgCountOutputIndex() const {
862  // Indices assigned to parameters are off-by-one (Parameters indices start
863  // at -1). TODO(jgruber): Consider starting at 0.
864  return Linkage::GetJSCallArgCountParamIndex(FormalParameterCount()) + 1;
865  }
866  int ContextOutputIndex() const {
867  // Indices assigned to parameters are off-by-one (Parameters indices start
868  // at -1). TODO(jgruber): Consider starting at 0.
869  return Linkage::GetJSCallContextParamIndex(FormalParameterCount()) + 1;
870  }
871  int LastOutputIndex() const { return ContextOutputIndex(); }
872 };
