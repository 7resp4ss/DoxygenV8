20 int BytecodeArray::SourcePosition(int offset) const {
21  int position = 0;
22  if (!HasSourcePositionTable()) return position;
23  for (SourcePositionTableIterator it(
24  source_position_table(kAcquireLoad),
25  SourcePositionTableIterator::kJavaScriptOnly,
26  SourcePositionTableIterator::kDontSkipFunctionEntry);
27  !it.done() && it.code_offset() <= offset; it.Advance()) {
28  position = it.source_position().ScriptOffset();
29  }
30  return position;
31 }

111 void BytecodeArray::Disassemble(Handle<BytecodeArray> handle,
112  std::ostream& os) {
113  DisallowGarbageCollection no_gc;
114 
115  os << "Parameter count " << handle->parameter_count() << "\n";
116  os << "Register count " << handle->register_count() << "\n";
117  os << "Frame size " << handle->frame_size() << "\n";
118 
119  Address base_address = handle->GetFirstBytecodeAddress();
120  SourcePositionTableIterator source_positions(handle->SourcePositionTable());
121 
122  interpreter::BytecodeArrayIterator iterator(handle);
123  while (!iterator.done()) {
124  if (!source_positions.done() &&
125  iterator.current_offset() == source_positions.code_offset()) {
126  os << std::setw(5) << source_positions.source_position().ScriptOffset();
127  if (source_positions.is_breakable()) {
128  os << (source_positions.is_statement() ? " S> " : " E> ");
129  } else {
130  os << (source_positions.is_statement() ? " s> " : " e> ");
131  }
132  source_positions.Advance();
133  } else {
134  os << " ";
135  }
136  Address current_address = base_address + iterator.current_offset();
137  os << reinterpret_cast<const void*>(current_address) << " @ "
138  << std::setw(4) << iterator.current_offset() << " : ";
139  interpreter::BytecodeDecoder::Decode(
140  os, reinterpret_cast<uint8_t*>(current_address));
141  if (interpreter::Bytecodes::IsJump(iterator.current_bytecode())) {
142  Address jump_target = base_address + iterator.GetJumpTargetOffset();
143  os << " (" << reinterpret_cast<void*>(jump_target) << " @ "
144  << iterator.GetJumpTargetOffset() << ")";
145  }
146  if (interpreter::Bytecodes::IsSwitch(iterator.current_bytecode())) {
147  os << " {";
148  bool first_entry = true;
149  for (interpreter::JumpTableTargetOffset entry :
150  iterator.GetJumpTableTargetOffsets()) {
151  if (first_entry) {
152  first_entry = false;
153  } else {
154  os << ",";
155  }
156  os << " " << entry.case_value << ": @" << entry.target_offset;
157  }
158  os << " }";
159  }
160  os << std::endl;
161  iterator.Advance();
162  }
163 
164  os << "Constant pool (size = " << handle->constant_pool()->length() << ")\n";
165 #ifdef OBJECT_PRINT
166  if (handle->constant_pool()->length() > 0) {
167  Print(handle->constant_pool(), os);
168  }
169 #endif
170 
171  os << "Handler Table (size = " << handle->handler_table()->length() << ")\n";
172 #ifdef ENABLE_DISASSEMBLER
173  if (handle->handler_table()->length() > 0) {
174  HandlerTable table(*handle);
175  table.HandlerTableRangePrint(os);
176  }
177 #endif
178 
179  Tagged<TrustedByteArray> source_position_table =
180  handle->SourcePositionTable();
181  os << "Source Position Table (size = " << source_position_table->length()
182  << ")\n";
183 #ifdef OBJECT_PRINT
184  if (source_position_table->length() > 0) {
185  os << Brief(source_position_table) << std::endl;
186  }
187 #endif
188 }
