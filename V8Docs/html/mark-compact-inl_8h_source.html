36 void MarkCompactCollector::MarkRootObject(
37  Root root, Tagged<HeapObject> obj,
38  MarkingHelper::WorklistTarget target_worklist) {
39  DCHECK(ReadOnlyHeap::Contains(obj) || heap_->Contains(obj));
40  MarkingHelper::TryMarkAndPush(heap_, local_marking_worklists_.get(),
41  marking_state_, target_worklist, obj);
42  if (V8_UNLIKELY(in_conservative_stack_scanning_)) {
43  DCHECK_EQ(root, Root::kStackRoots);
44  MemoryChunk* chunk = MemoryChunk::FromHeapObject(obj);
45  if (chunk->IsEvacuationCandidate()) {
46  DCHECK(!chunk->InYoungGeneration());
47  ReportAbortedEvacuationCandidateDueToFlags(
48  PageMetadata::cast(chunk->Metadata()), chunk);
49  } else if (chunk->InYoungGeneration() && !chunk->IsLargePage()) {
50  DCHECK(chunk->IsToPage());
51  if (!chunk->IsQuarantined()) {
52  chunk->SetFlagNonExecutable(MemoryChunk::IS_QUARANTINED);
53  }
54  }
55  }
56 }

70 template <typename THeapObjectSlot>
71 void MarkCompactCollector::RecordSlot(MemoryChunk* source_chunk,
72  THeapObjectSlot slot,
73  Tagged<HeapObject> target) {
74  MemoryChunk* target_chunk = MemoryChunk::FromHeapObject(target);
75  if (target_chunk->IsEvacuationCandidate()) {
76  MutablePageMetadata* source_page =
77  MutablePageMetadata::cast(source_chunk->Metadata());
78  if (target_chunk->IsFlagSet(MemoryChunk::IS_EXECUTABLE)) {
79  // TODO(377724745): currently needed because flags are untrusted.
80  SBXCHECK(!InsideSandbox(target_chunk->address()));
81  RememberedSet<TRUSTED_TO_CODE>::Insert<AccessMode::ATOMIC>(
82  source_page, source_chunk->Offset(slot.address()));
83  } else if (source_chunk->IsFlagSet(MemoryChunk::IS_TRUSTED) &&
84  target_chunk->IsFlagSet(MemoryChunk::IS_TRUSTED)) {
85  // TODO(377724745): currently needed because flags are untrusted.
86  SBXCHECK(!InsideSandbox(target_chunk->address()));
87  RememberedSet<TRUSTED_TO_TRUSTED>::Insert<AccessMode::ATOMIC>(
88  source_page, source_chunk->Offset(slot.address()));
89  } else if (V8_LIKELY(!target_chunk->InWritableSharedSpace()) ||
90  source_page->heap()->isolate()->is_shared_space_isolate()) {
91  DCHECK_EQ(source_page->heap(), target_chunk->GetHeap());
92  RememberedSet<OLD_TO_OLD>::Insert<AccessMode::ATOMIC>(
93  source_page, source_chunk->Offset(slot.address()));
94  } else {
95  // DCHECK here that we only don't record in case of local->shared
96  // references in a client GC.
97  DCHECK(!source_page->heap()->isolate()->is_shared_space_isolate());
98  DCHECK(target_chunk->GetHeap()->isolate()->is_shared_space_isolate());
99  DCHECK(target_chunk->InWritableSharedSpace());
100  }
101  }
102 }

122 void RootMarkingVisitor::MarkObjectByPointer(Root root, FullObjectSlot p) {
123  Tagged<Object> object = *p;
124 #ifdef V8_ENABLE_DIRECT_HANDLE
125  if (object.ptr() == kTaggedNullAddress) return;
126 #endif
127  if (!IsHeapObject(object)) return;
128  Tagged<HeapObject> heap_object = Cast<HeapObject>(object);
129  const auto target_worklist =
130  MarkingHelper::ShouldMarkObject(collector_->heap(), heap_object);
131  if (!target_worklist) {
132  return;
133  }
134  collector_->MarkRootObject(root, heap_object, target_worklist.value());
135 }
