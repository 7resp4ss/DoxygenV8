44 Reduction JSGenericLowering::Reduce(Node* node) {
45  switch (node->opcode()) {
46 #define DECLARE_CASE(x, ...) \
47  case IrOpcode::k##x: \
48  if constexpr (IrOpcode::k##x == IrOpcode::kJSStackCheck) { \
49  /* We don't want to lower stack checks so early, and we don't want to \
50  * return `Changed` for them, otherwise we'll loop infinitely trying to \
51  * lower them. */ \
52  return NoChange(); \
53  } \
54  Lower##x(node); \
55  break;
56  JS_OP_LIST(DECLARE_CASE)
57 #undef DECLARE_CASE
58  default:
59  // Nothing to see.
60  return NoChange();
61  }
62  return Changed(node);
63 }

197 // The megamorphic load/store builtin can be used as a performance optimization
198 // in some cases - unlike the full builtin, the megamorphic builtin does fewer
199 // checks and does not collect feedback.
200 bool ShouldUseMegamorphicAccessBuiltin(FeedbackSource const& source,
201  OptionalNameRef name, AccessMode mode,
202  JSHeapBroker* broker) {
203  ProcessedFeedback const& feedback =
204  broker->GetFeedbackForPropertyAccess(source, mode, name);
205 
206  if (feedback.kind() == ProcessedFeedback::kElementAccess) {
207  return feedback.AsElementAccess().transition_groups().empty();
208  } else if (feedback.kind() == ProcessedFeedback::kNamedAccess) {
209  return feedback.AsNamedAccess().maps().empty();
210  } else if (feedback.kind() == ProcessedFeedback::kInsufficient) {
211  return false;
212  }
213  UNREACHABLE();
214 }
