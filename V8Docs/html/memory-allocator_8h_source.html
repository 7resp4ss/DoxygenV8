122  // Returns an indication of whether a pointer is in a space that has
123  // been allocated by this MemoryAllocator. It is conservative, allowing
124  // false negatives (i.e., if a pointer is outside the allocated space, it may
125  // return false) but not false positives (i.e., if a pointer is inside the
126  // allocated space, it will definitely return false).
127  V8_INLINE bool IsOutsideAllocatedSpace(Address address) const {
128  return IsOutsideAllocatedSpace(address, NOT_EXECUTABLE) &&
129  IsOutsideAllocatedSpace(address, EXECUTABLE);
130  }
131  V8_INLINE bool IsOutsideAllocatedSpace(Address address,
132  Executability executable) const {
133  switch (executable) {
134  case NOT_EXECUTABLE:
135  return address < lowest_not_executable_ever_allocated_ ||
136  address >= highest_not_executable_ever_allocated_;
137  case EXECUTABLE:
138  return address < lowest_executable_ever_allocated_ ||
139  address >= highest_executable_ever_allocated_;
140  }
141  }

299  void UpdateAllocatedSpaceLimits(Address low, Address high,
300  Executability executable) {
301  // The use of atomic primitives does not guarantee correctness (wrt.
302  // desired semantics) by default. The loop here ensures that we update the
303  // values only if they did not change in between.
304  Address ptr;
305  switch (executable) {
306  case NOT_EXECUTABLE:
307  ptr = lowest_not_executable_ever_allocated_.load(
308  std::memory_order_relaxed);
309  while ((low < ptr) &&
310  !lowest_not_executable_ever_allocated_.compare_exchange_weak(
311  ptr, low, std::memory_order_acq_rel)) {
312  }
313  ptr = highest_not_executable_ever_allocated_.load(
314  std::memory_order_relaxed);
315  while ((high > ptr) &&
316  !highest_not_executable_ever_allocated_.compare_exchange_weak(
317  ptr, high, std::memory_order_acq_rel)) {
318  }
319  break;
320  case EXECUTABLE:
321  ptr = lowest_executable_ever_allocated_.load(std::memory_order_relaxed);
322  while ((low < ptr) &&
323  !lowest_executable_ever_allocated_.compare_exchange_weak(
324  ptr, low, std::memory_order_acq_rel)) {
325  }
326  ptr =
327  highest_executable_ever_allocated_.load(std::memory_order_relaxed);
328  while ((high > ptr) &&
329  !highest_executable_ever_allocated_.compare_exchange_weak(
330  ptr, high, std::memory_order_acq_rel)) {
331  }
332  break;
333  }
334  }
