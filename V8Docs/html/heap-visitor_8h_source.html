185 // The base class for visitors that need to dispatch on object type. The default
186 // behavior of all visit functions is to iterate body of the given object using
187 // the BodyDescriptor of the object.
188 //
189 // The visit functions return the size of the object.
190 //
191 // This class is intended to be used in the following way:
192 //
193 // class SomeVisitor : public HeapVisitor<SomeVisitor> {
194 // ...
195 // }
