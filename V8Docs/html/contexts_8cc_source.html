30 Handle<ScriptContextTable> ScriptContextTable::New(Isolate* isolate,
31  int capacity,
32  AllocationType allocation) {
33  DCHECK_GE(capacity, 0);
34  DCHECK_LE(capacity, kMaxCapacity);
35 
36  auto names = NameToIndexHashTable::New(isolate, 16);
37 
38  std::optional<DisallowGarbageCollection> no_gc;
39  Handle<ScriptContextTable> result =
40  Allocate(isolate, capacity, &no_gc, allocation);
41  result->set_length(0, kReleaseStore);
42  result->set_names_to_context_index(*names);
43  ReadOnlyRoots roots{isolate};
44  MemsetTagged(result->RawFieldOfFirstElement(), roots.undefined_value(),
45  capacity);
46  return result;
47 }

51 // Adds local names from `script_context` to the hash table.
52 Handle<NameToIndexHashTable> AddLocalNamesFromContext(
53  Isolate* isolate, Handle<NameToIndexHashTable> names_table,
54  DirectHandle<Context> script_context, bool ignore_duplicates,
55  int script_context_index) {
56  ReadOnlyRoots roots(isolate);
57  DirectHandle<ScopeInfo> scope_info(script_context->scope_info(), isolate);
58  int local_count = scope_info->ContextLocalCount();
59  names_table = names_table->EnsureCapacity(isolate, names_table, local_count);
60 
61  for (auto it : ScopeInfo::IterateLocalNames(scope_info)) {
62  DirectHandle<Name> name(it->name(), isolate);
63  if (ignore_duplicates) {
64  int32_t hash = NameToIndexShape::Hash(roots, *name);
65  if (names_table->FindEntry(isolate, roots, *name, hash).is_found()) {
66  continue;
67  }
68  }
69  names_table = NameToIndexHashTable::Add(isolate, names_table, name,
70  script_context_index);
71  }
72 
73  return names_table;
74 }

78 Handle<ScriptContextTable> ScriptContextTable::Add(
79  Isolate* isolate, Handle<ScriptContextTable> table,
80  DirectHandle<Context> script_context, bool ignore_duplicates) {
81  DCHECK(script_context->IsScriptContext());
82 
83  int old_length = table->length(kAcquireLoad);
84  int new_length = old_length + 1;
85  DCHECK_LE(0, old_length);
86 
87  Handle<ScriptContextTable> result = table;
88  int old_capacity = table->capacity();
89  DCHECK_LE(old_length, old_capacity);
90  if (old_length == old_capacity) {
91  int new_capacity = NewCapacityForIndex(old_length, old_capacity);
92  auto new_table = New(isolate, new_capacity);
93  new_table->set_length(old_length, kReleaseStore);
94  new_table->set_names_to_context_index(table->names_to_context_index());
95  CopyElements(isolate, *new_table, 0, *table, 0, old_length);
96  result = new_table;
97  }
98 
99  Handle<NameToIndexHashTable> names_table(result->names_to_context_index(),
100  isolate);
101  names_table = AddLocalNamesFromContext(isolate, names_table, script_context,
102  ignore_duplicates, old_length);
103  result->set_names_to_context_index(*names_table);
104 
105  result->set(old_length, *script_context, kReleaseStore);
106  result->set_length(new_length, kReleaseStore);
107  return result;
108 }

120 bool ScriptContextTable::Lookup(DirectHandle<String> name,
121  VariableLookupResult* result) {
122  DisallowGarbageCollection no_gc;
123  int index = names_to_context_index()->Lookup(*name);
124  if (index == -1) return false;
125  DCHECK_LE(0, index);
126  DCHECK_LT(index, length(kAcquireLoad));
127  Tagged<Context> context = get(index);
128  DCHECK(context->IsScriptContext());
129  int slot_index = context->scope_info()->ContextSlotIndex(*name, result);
130  if (slot_index < 0) return false;
131  result->context_index = index;
132  result->slot_index = slot_index;
133  return true;
134 }

206 /**
207  * Lookups a property in an object environment, taking the unscopables into
208  * account. This is used For HasBinding spec algorithms for ObjectEnvironment.
209  */
210 static Maybe<bool> UnscopableLookup(LookupIterator* it, bool is_with_context) {
211  Isolate* isolate = it->isolate();
212 
213  Maybe<bool> found = JSReceiver::HasProperty(it);
214  if (!is_with_context || found.IsNothing() || !found.FromJust()) return found;
215 
216  DirectHandle<Object> unscopables;
217  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
218  isolate, unscopables,
219  JSReceiver::GetProperty(isolate, Cast<JSReceiver>(it->GetReceiver()),
220  isolate->factory()->unscopables_symbol()),
221  Nothing<bool>());
222  if (!IsJSReceiver(*unscopables)) return Just(true);
223  DirectHandle<Object> blocklist;
224  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
225  isolate, blocklist,
226  JSReceiver::GetProperty(isolate, Cast<JSReceiver>(unscopables),
227  it->name()),
228  Nothing<bool>());
229  return Just(!Object::BooleanValue(*blocklist, isolate));
230 }

237 // static
238 Handle<Object> Context::Lookup(Handle<Context> context, Handle<String> name,
239  ContextLookupFlags flags, int* index,
240  PropertyAttributes* attributes,
241  InitializationFlag* init_flag,
242  VariableMode* variable_mode,
243  bool* is_sloppy_function_name) {
244  Isolate* isolate = Isolate::Current();
245 
246  bool follow_context_chain = (flags & FOLLOW_CONTEXT_CHAIN) != 0;
247  bool has_seen_debug_evaluate_context = false;
248  *index = kNotFound;
249  *attributes = ABSENT;
250  *init_flag = kCreatedInitialized;
251  *variable_mode = VariableMode::kVar;
252  if (is_sloppy_function_name != nullptr) {
253  *is_sloppy_function_name = false;
254  }
255 
256  if (v8_flags.trace_contexts) {
257  PrintF("Context::Lookup(");
258  ShortPrint(*name);
259  PrintF(")\n");
260  }
261 
262  do {
263  if (v8_flags.trace_contexts) {
264  PrintF(" - looking in context %p",
265  reinterpret_cast<void*>(context->ptr()));
266  if (context->IsScriptContext()) PrintF(" (script context)");
267  if (IsNativeContext(*context)) PrintF(" (native context)");
268  if (context->IsDebugEvaluateContext()) PrintF(" (debug context)");
269  PrintF("\n");
270  }
271 
272  // 1. Check global objects, subjects of with, and extension objects.
273  DCHECK_IMPLIES(context->IsEvalContext() && context->has_extension(),
274  IsTheHole(context->extension(), isolate));
275  if ((IsNativeContext(*context) || context->IsWithContext() ||
276  context->IsFunctionContext() || context->IsBlockContext()) &&
277  context->has_extension() && !context->extension_receiver().is_null()) {
278  Handle<JSReceiver> object(context->extension_receiver(), isolate);
279 
280  if (IsNativeContext(*context)) {
281  DisallowGarbageCollection no_gc;
282  if (v8_flags.trace_contexts) {
283  PrintF(" - trying other script contexts\n");
284  }
285  // Try other script contexts.
286  Tagged<ScriptContextTable> script_contexts =
287  context->native_context()->script_context_table();
288  VariableLookupResult r;
289  if (script_contexts->Lookup(name, &r)) {
290  Tagged<Context> script_context =
291  script_contexts->get(r.context_index);
292  if (v8_flags.trace_contexts) {
293  PrintF("=> found property in script context %d: %p\n",
294  r.context_index,
295  reinterpret_cast<void*>(script_context.ptr()));
296  }
297  *index = r.slot_index;
298  *variable_mode = r.mode;
299  *init_flag = r.init_flag;
300  *attributes = GetAttributesForMode(r.mode);
301  return handle(script_context, isolate);
302  }
303  }
304 
305  // Context extension objects needs to behave as if they have no
306  // prototype. So even if we want to follow prototype chains, we need
307  // to only do a local lookup for context extension objects.
308  Maybe<PropertyAttributes> maybe = Nothing<PropertyAttributes>();
309  if ((flags & FOLLOW_PROTOTYPE_CHAIN) == 0 ||
310  IsJSContextExtensionObject(*object)) {
311  maybe = JSReceiver::GetOwnPropertyAttributes(isolate, object, name);
312  } else {
313  // A with context will never bind "this", but debug-eval may look into
314  // a with context when resolving "this". Other synthetic variables such
315  // as new.target may be resolved as VariableMode::kDynamicLocal due to
316  // bug v8:5405 , skipping them here serves as a workaround until a more
317  // thorough fix can be applied.
318  // TODO(v8:5405): Replace this check with a DCHECK when resolution of
319  // of synthetic variables does not go through this code path.
320  if (ScopeInfo::VariableIsSynthetic(*name)) {
321  maybe = Just(ABSENT);
322  } else {
323  LookupIterator it(isolate, object, name, object);
324  Maybe<bool> found = UnscopableLookup(&it, context->IsWithContext());
325  if (found.IsNothing()) {
326  maybe = Nothing<PropertyAttributes>();
327  } else {
328  // Luckily, consumers of |maybe| only care whether the property
329  // was absent or not, so we can return a dummy |NONE| value
330  // for its attributes when it was present.
331  maybe = Just(found.FromJust() ? NONE : ABSENT);
332  }
333  }
334  }
335 
336  if (maybe.IsNothing()) return Handle<Object>();
337  DCHECK(!isolate->has_exception());
338  *attributes = maybe.FromJust();
339 
340  if (maybe.FromJust() != ABSENT) {
341  if (v8_flags.trace_contexts) {
342  PrintF("=> found property in context object %p\n",
343  reinterpret_cast<void*>(object->ptr()));
344  }
345  return object;
346  }
347  }
348 
349  // 2. Check the context proper if it has slots.
350  if (context->IsFunctionContext() || context->IsBlockContext() ||
351  context->IsScriptContext() || context->IsEvalContext() ||
352  context->IsModuleContext() || context->IsCatchContext()) {
353  DisallowGarbageCollection no_gc;
354  // Use serialized scope information of functions and blocks to search
355  // for the context index.
356  Tagged<ScopeInfo> scope_info = context->scope_info();
357  VariableLookupResult lookup_result;
358  int slot_index = scope_info->ContextSlotIndex(*name, &lookup_result);
359  DCHECK(slot_index < 0 || slot_index >= MIN_CONTEXT_SLOTS);
360  if (slot_index >= 0) {
361  // Re-direct lookup to the ScriptContextTable in case we find a hole in
362  // a REPL script context. REPL scripts allow re-declaration of
363  // script-level let bindings. The value itself is stored in the script
364  // context of the first script that declared a variable, all other
365  // script contexts will contain 'the hole' for that particular name.
366  if (scope_info->IsReplModeScope() &&
367  context->IsElementTheHole(slot_index)) {
368  context = Handle<Context>(context->previous(), isolate);
369  continue;
370  }
371 
372  if (v8_flags.trace_contexts) {
373  PrintF("=> found local in context slot %d (mode = %hhu)\n",
374  slot_index, static_cast<uint8_t>(lookup_result.mode));
375  }
376  *index = slot_index;
377  *variable_mode = lookup_result.mode;
378  *init_flag = lookup_result.init_flag;
379  *attributes = GetAttributesForMode(lookup_result.mode);
380  return context;
381  }
382 
383  // Check the slot corresponding to the intermediate context holding
384  // only the function name variable. It's conceptually (and spec-wise)
385  // in an outer scope of the function's declaration scope.
386  if (follow_context_chain && context->IsFunctionContext()) {
387  int function_index = scope_info->FunctionContextSlotIndex(*name);
388  if (function_index >= 0) {
389  if (v8_flags.trace_contexts) {
390  PrintF("=> found intermediate function in context slot %d\n",
391  function_index);
392  }
393  *index = function_index;
394  *attributes = READ_ONLY;
395  *init_flag = kCreatedInitialized;
396  *variable_mode = VariableMode::kConst;
397  if (is_sloppy_function_name != nullptr &&
398  is_sloppy(scope_info->language_mode())) {
399  *is_sloppy_function_name = true;
400  }
401  return context;
402  }
403  }
404 
405  // Lookup variable in module imports and exports.
406  if (context->IsModuleContext()) {
407  VariableMode mode;
408  InitializationFlag flag;
409  MaybeAssignedFlag maybe_assigned_flag;
410  int cell_index =
411  scope_info->ModuleIndex(*name, &mode, &flag, &maybe_assigned_flag);
412  if (cell_index != 0) {
413  if (v8_flags.trace_contexts) {
414  PrintF("=> found in module imports or exports\n");
415  }
416  *index = cell_index;
417  *variable_mode = mode;
418  *init_flag = flag;
419  *attributes = SourceTextModuleDescriptor::GetCellIndexKind(
420  cell_index) == SourceTextModuleDescriptor::kExport
421  ? GetAttributesForMode(mode)
422  : READ_ONLY;
423  return handle(context->module(), isolate);
424  }
425  }
426  } else if (context->IsDebugEvaluateContext()) {
427  has_seen_debug_evaluate_context = true;
428 
429  // Check materialized locals.
430  Tagged<Object> ext = context->get(EXTENSION_INDEX, kRelaxedLoad);
431  if (IsJSReceiver(ext)) {
432  Handle<JSReceiver> extension(Cast<JSReceiver>(ext), isolate);
433  LookupIterator it(isolate, extension, name, extension);
434  Maybe<bool> found = JSReceiver::HasProperty(&it);
435  if (found.FromMaybe(false)) {
436  *attributes = NONE;
437  return extension;
438  }
439  }
440 
441  // Check the original context, but do not follow its context chain.
442  Tagged<Object> obj = context->get(WRAPPED_CONTEXT_INDEX, kRelaxedLoad);
443  if (IsContext(obj)) {
444  Handle<Context> wrapped_context(Cast<Context>(obj), isolate);
445  Handle<Object> result =
446  Context::Lookup(wrapped_context, name, DONT_FOLLOW_CHAINS, index,
447  attributes, init_flag, variable_mode);
448  if (!result.is_null()) return result;
449  }
450  }
451 
452  // 3. Prepare to continue with the previous (next outermost) context.
453  if (IsNativeContext(*context)) break;
454 
455  // In case we saw any DebugEvaluateContext, we'll need to check the block
456  // list before we can advance to properly "shadow" stack-allocated
457  // variables.
458  // Note that this implicitly skips the block list check for the
459  // "wrapped" context lookup for DebugEvaluateContexts. In that case
460  // `has_seen_debug_evaluate_context` will always be false.
461  if (has_seen_debug_evaluate_context &&
462  IsEphemeronHashTable(isolate->heap()->locals_block_list_cache())) {
463  DirectHandle<ScopeInfo> scope_info =
464  direct_handle(context->scope_info(), isolate);
465  Tagged<Object> maybe_outer_block_list =
466  isolate->LocalsBlockListCacheGet(scope_info);
467  if (IsStringSet(maybe_outer_block_list) &&
468  Cast<StringSet>(maybe_outer_block_list)->Has(isolate, name)) {
469  if (v8_flags.trace_contexts) {
470  PrintF(" - name is blocklisted. Aborting.\n");
471  }
472  break;
473  }
474  }
475 
476  context = Handle<Context>(context->previous(), isolate);
477  } while (follow_context_chain);
478 
479  if (v8_flags.trace_contexts) {
480  PrintF("=> no property/slot found\n");
481  }
482  return Handle<Object>::null();
483 }

486 std::optional<int32_t> DoubleFitsInInt32(double value) {
487  constexpr double int32_min = std::numeric_limits<int32_t>::min();
488  constexpr double int32_max = std::numeric_limits<int32_t>::max();
489  // Check -0.0 first.
490  if (value == 0.0 && std::signbit(value)) return {};
491  double trunc_value = std::trunc(value);
492  if (int32_min <= value && value <= int32_max && value == trunc_value) {
493  return static_cast<int32_t>(trunc_value);
494  }
495  return {};
496 }

525 // static
526 DirectHandle<Object> Context::Get(DirectHandle<Context> context, int index,
527  Isolate* isolate) {
528  DirectHandle<Object> value =
529  handle(context->get(index, kRelaxedLoad), isolate);
530  if (!Is<ContextCell>(value)) {
531  return value;
532  }
533  DCHECK(context->HasContextCells());
534  DirectHandle<ContextCell> cell = Cast<ContextCell>(value);
535  switch (cell->state()) {
536  case ContextCell::kConst:
537  case ContextCell::kSmi:
538  return handle(cell->tagged_value(), isolate);
539  case ContextCell::kInt32:
540  if (Smi::IsValid(cell->int32_value())) {
541  return handle(Smi::FromInt(cell->int32_value()), isolate);
542  }
543  return isolate->factory()->NewHeapNumber(
544  static_cast<double>(cell->int32_value()));
545  case ContextCell::kFloat64:
546  return isolate->factory()->NewHeapNumber(cell->float64_value());
547  case ContextCell::kDetached:
548  UNREACHABLE();
549  }
550  UNREACHABLE();
551 }

553 // static
554 void Context::Set(DirectHandle<Context> context, int index,
555  DirectHandle<Object> new_value, Isolate* isolate) {
556  DirectHandle<Object> old_value(context->get(index, kRelaxedLoad), isolate);
557  if (!context->HasContextCells()) {
558  context->set(index, *new_value);
559  return;
560  }
561 
562  if (IsTheHole(*old_value, isolate)) {
563  // Setting the initial value.
564  DirectHandle<ContextCell> cell =
565  isolate->factory()->NewContextCell(Cast<JSAny>(new_value));
566  context->set(index, *cell);
567  return;
568  }
569 
570  if (!Is<ContextCell>(old_value)) {
571  context->set(index, *new_value);
572  return;
573  }
574 
575  if (IsUndefinedContextCell(*old_value, isolate)) {
576  if (IsUndefined(*new_value)) return;
577  if (IsTheHole(*new_value)) {
578  // This can happened in let-variable in function contexts.
579  context->set(index, *new_value);
580  return;
581  }
582  DirectHandle<ContextCell> cell =
583  isolate->factory()->NewContextCell(Cast<JSAny>(new_value));
584  context->set(index, *cell);
585  return;
586  }
587 
588  DirectHandle<ContextCell> cell = Cast<ContextCell>(old_value);
589  switch (cell->state()) {
590  case ContextCell::kConst:
591  // If we are assigning the same value, the property won't change.
592  if (cell->tagged_value() == *new_value) {
593  return;
594  }
595  // If both values are HeapNumbers with the same double value, the property
596  // won't change either.
597  if (Is<HeapNumber>(cell->tagged_value()) && Is<HeapNumber>(*new_value)) {
598  double old_number = Cast<HeapNumber>(cell->tagged_value())->value();
599  double new_number = Cast<HeapNumber>(*new_value)->value();
600  if (old_number == new_number && old_number != 0) {
601  return;
602  }
603  }
604  NotifyContextCellStateWillChange(cell, isolate);
605  if (Is<Smi>(*new_value)) {
606  cell->set_smi_value(Cast<Smi>(*new_value));
607  cell->set_state(ContextCell::kSmi);
608  } else if (IsHeapNumber(*new_value)) {
609  TransitionContextCellToUntagged(Cast<HeapNumber>(*new_value), cell);
610  cell->clear_tagged_value();
611  } else {
612  context->set(index, *new_value);
613  cell->clear_tagged_value();
614  cell->set_state(ContextCell::kDetached);
615  }
616  return;
617  case ContextCell::kSmi:
618  if (IsSmi(*new_value)) {
619  cell->set_smi_value(Cast<Smi>(*new_value));
620  cell->set_state(ContextCell::kSmi);
621  } else {
622  NotifyContextCellStateWillChange(cell, isolate);
623  if (IsHeapNumber(*new_value)) {
624  TransitionContextCellToUntagged(Cast<HeapNumber>(*new_value), cell);

779  DirectHandle<Object> parent) {

781  DCHECK(isolate->HasContextPromiseHooks());
782  int contextSlot;
783 
784  switch (type) {
785  case PromiseHookType::kInit:
786  contextSlot = PROMISE_HOOK_INIT_FUNCTION_INDEX;
787  break;
788  case PromiseHookType::kResolve:
789  contextSlot = PROMISE_HOOK_RESOLVE_FUNCTION_INDEX;
790  break;
791  case PromiseHookType::kBefore:
792  contextSlot = PROMISE_HOOK_BEFORE_FUNCTION_INDEX;
793  break;
794  case PromiseHookType::kAfter:
795  contextSlot = PROMISE_HOOK_AFTER_FUNCTION_INDEX;
796  break;
797  default:
798  UNREACHABLE();
799  }
800 
801  DirectHandle<Object> hook(isolate->native_context()->GetNoCell(contextSlot),
802  isolate);
803  if (IsUndefined(*hook)) return;
804 
805  size_t argc = type == PromiseHookType::kInit ? 2 : 1;
806  DirectHandle<Object> argv[2] = {Cast<Object>(promise), parent};
807 
808  DirectHandle<Object> receiver = isolate->global_proxy();
809 
810  StackLimitCheck check(isolate);
811  bool failed = false;
812  if (check.HasOverflowed()) {
813  isolate->StackOverflow();
814  failed = true;
815  } else {
816  failed = Execution::Call(isolate, hook, receiver, {argv, argc}).is_null();
817  }
818  if (failed) {
819  DCHECK(isolate->has_exception());
820  DirectHandle<Object> exception(isolate->exception(), isolate);
821 
822  MessageLocation* no_location = nullptr;
823  DirectHandle<JSMessageObject> message =
824  isolate->CreateMessageOrAbort(exception, no_location);
825  MessageHandler::ReportMessage(isolate, no_location, message);
826 
827  isolate->clear_exception();
828  }
