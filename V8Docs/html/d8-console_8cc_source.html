19 void WriteToFile(const char* prefix, FILE* file, Isolate* isolate,
20  const debug::ConsoleCallArguments& args) {
21  if (prefix) fprintf(file, "%s: ", prefix);
22  for (int i = 0; i < args.Length(); i++) {
23  HandleScope handle_scope(isolate);
24  if (i > 0) fprintf(file, " ");
25 
26  Local<Value> arg = args[i];
27  Local<String> str_obj;
28 
29  if (arg->IsSymbol()) arg = Local<Symbol>::Cast(arg)->Description(isolate);
30  if (!arg->ToString(isolate->GetCurrentContext()).ToLocal(&str_obj)) return;
31 
32  v8::String::Utf8Value str(isolate, str_obj);
33  size_t n = fwrite(*str, sizeof(**str), str.length(), file);
34  if (n != str.length()) {
35  printf("Error in fwrite\n");
36  base::OS::ExitProcess(1);
37  }
38  }
39  fprintf(file, "\n");
40  // Flush the file to avoid output to pile up in a buffer. Console output is
41  // often used for timing, so it should appear as soon as the code is executed.
42  fflush(file);
43 }

47 std::optional<std::string> GetTimerLabel(
48  const debug::ConsoleCallArguments& args) {
49  if (args.Length() == 0) return "default";
50  Isolate* isolate = args.GetIsolate();
51  v8::TryCatch try_catch(isolate);
52  v8::String::Utf8Value label(isolate, args[0]);
53  if (*label == nullptr) return std::nullopt;
54  return std::string(*label, label.length());
55 }

118 void D8Console::ProfileEnd(const debug::ConsoleCallArguments& args,
119  const v8::debug::ConsoleContext&) {
120  if (!profiler_) return;
121  CpuProfile* profile = profiler_->StopProfiling(String::Empty(isolate_));
122  profiler_active_ = false;
123  if (!profile) return;
124  if (Shell::HasOnProfileEndListener(isolate_)) {
125  i::StringOutputStream out;
126  profile->Serialize(&out);
127  Shell::TriggerOnProfileEndListener(isolate_, out.str());
128  } else {
129  i::FileOutputStream out(kCpuProfileOutputFilename);
130  profile->Serialize(&out);
131  }
132  profile->Delete();
133 }

145 void D8Console::TimeLog(const debug::ConsoleCallArguments& args,
146  const v8::debug::ConsoleContext&) {
147  if (i::v8_flags.correctness_fuzzer_suppressions) return;
148  std::optional label = GetTimerLabel(args);
149  if (!label.has_value()) return;
150  auto it = timers_.find(label.value());
151  if (it == timers_.end()) {
152  printf("console.timeLog: Timer '%s' does not exist\n",
153  label.value().c_str());
154  return;
155  }
156  base::TimeDelta delta = base::TimeTicks::Now() - it->second;
157  printf("console.timeLog: %s, %f\n", label.value().c_str(),
158  delta.InMillisecondsF());
159 }
