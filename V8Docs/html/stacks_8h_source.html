35  // We track the state below to prevent stack corruptions under the sandbox
36  // security model.
37  // Assuming that the external pointer to the jump buffer has been corrupted
38  // and replaced with a different jump buffer, we check its state before
39  // resuming it to verify that it is not Active or Retired.
40  // The distinction between Suspended and Inactive may not be strictly
41  // necessary since we currently always pass a single JS value in the return
42  // register across stacks (either the Promise, the result of the Promise, or
43  // the result of the export). However adding a state does not cost anything
44  // and is more robust against potential changes in the calling conventions.
45  enum StackState : int32_t {
46  Active, // The (unique) active stack. The jump buffer is invalid in that
47  // state.
48  Suspended, // A stack suspended by WasmSuspend.
49  Inactive, // A parent/ancestor of the active stack. In other words, a
50  // stack that either called or resumed a suspendable stack.
51  Retired // A finished stack. The jump buffer is invalid in that state.
52  };
53  StackState state;
54 };

66  ~StackMemory();
67  void* jslimit() const;
68  Address base() const {
69  Address memory_limit = active_segment_
70  ? active_segment_->base()
71  : reinterpret_cast<Address>(limit_ + size_);
72 #ifdef USE_SIMULATOR
73  // To perform runtime calls with different signatures, some simulators
74  // prepare a fixed number of arguments which is an upper bound of the actual
75  // parameter count. The extra stack slots contain arbitrary data and are
76  // never used, but with stack-switching this can happen close to the stack
77  // start, so we need to reserve a safety gap to ensure that the addresses
78  // are at least mapped to prevent a crash.
79  constexpr int kStackBaseSafetyOffset = 20 * kSystemPointerSize;
80 #else
81  constexpr int kStackBaseSafetyOffset = 0;
82 #endif
83  return memory_limit - kStackBaseSafetyOffset;
84  }
85  bool IsValidContinuation(Tagged<WasmContinuationObject> cont);
86  JumpBuffer* jmpbuf() { return &jmpbuf_; }
87  bool Contains(Address addr) {
88  if (!owned_) {
89  return reinterpret_cast<Address>(jslimit()) <= addr && addr < base();
90  }
91  for (auto segment = first_segment_; segment;
92  segment = segment->next_segment_) {
93  if (reinterpret_cast<Address>(segment->limit_) <= addr &&
94  addr < segment->base()) {
95  return true;
96  }
97  if (segment == active_segment_) break;
98  }
99  return false;
100  }
101  int id() { return id_; }
102  bool IsActive() { return jmpbuf_.state == JumpBuffer::Active; }
103  void set_index(size_t index) { index_ = index; }
104  size_t index() { return index_; }
105  size_t allocated_size() {
106  size_t size = 0;
107  auto segment = first_segment_;
108  while (segment) {
109  size += segment->size_;
110  segment = segment->next_segment_;
111  }
112  return size;
113  }
114  void FillWith(uint8_t value) {
115  auto segment = first_segment_;
116  while (segment) {
117  memset(segment->limit_, value, segment->size_);
118  segment = segment->next_segment_;
119  }
120  }

147  struct StackSwitchInfo {
148  // Source FP and target SP of the frame that switched to the central stack.
149  // The source FP is in the secondary stack, the target SP is in the central
150  // stack.
151  // The stack cannot be suspended while it is on the central stack, so there
152  // can be at most one switch for a given stack.
153  Address source_fp = kNullAddress;
154  Address target_sp = kNullAddress;
155  bool has_value() const { return source_fp != kNullAddress; }
156  };
157  const StackSwitchInfo& stack_switch_info() const {
158  return stack_switch_info_;
159  }
160  void set_stack_switch_info(Address fp, Address sp) {
161  stack_switch_info_ = {fp, sp};
162  }
163  void clear_stack_switch_info() {
164  stack_switch_info_.source_fp = kNullAddress;
165  }

167  static int JSStackLimitMarginKB() {
168  if (v8_flags.experimental_wasm_growable_stacks) {
169  // The limiting factor for this margin is the stack space used by outgoing
170  // stack parameters in wasm. They can take up to 16KB (1000 simd
171  // parameters, minus register parameters) and are not taken into account
172  // by stack checks.
173  // TODO(42204615): look into changing the stack check to take outgoing
174  // stack parameters into account.
175  static_assert(kMaxValueTypeSize == 16);
176  static_assert(kV8MaxWasmFunctionParams == 1000);
177  return 20;
178  } else {
179  return DEBUG_BOOL ? 80 : 40;
180  }
181  }
