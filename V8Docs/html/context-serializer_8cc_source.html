24 // During serialization, puts the native context into a state understood by the
25 // serializer (e.g. by clearing lists of InstructionStream objects). After
26 // serialization, the original state is restored.
27 class V8_NODISCARD SanitizeNativeContextScope final {
28  public:
29  SanitizeNativeContextScope(Isolate* isolate,
30  Tagged<NativeContext> native_context,
31  bool allow_active_isolate_for_testing,
32  const DisallowGarbageCollection& no_gc)
33  : native_context_(native_context), no_gc_(no_gc) {
34 #ifdef DEBUG
35  if (!allow_active_isolate_for_testing) {
36  // Microtasks.
37  MicrotaskQueue* microtask_queue = native_context_->microtask_queue();
38  DCHECK_EQ(0, microtask_queue->size());
39  DCHECK(!microtask_queue->HasMicrotasksSuppressions());
40  DCHECK_EQ(0, microtask_queue->GetMicrotasksScopeDepth());
41  DCHECK(microtask_queue->DebugMicrotasksScopeDepthIsZero());
42  }
43 #endif
44  microtask_queue_external_pointer_ =
45  native_context
46  ->RawExternalPointerField(NativeContext::kMicrotaskQueueOffset,
47  kNativeContextMicrotaskQueueTag)
48  .GetAndClearContentForSerialization(no_gc);
49  }
50 
51  ~SanitizeNativeContextScope() {
52  // Restore saved fields.
53  native_context_
54  ->RawExternalPointerField(NativeContext::kMicrotaskQueueOffset,
55  kNativeContextMicrotaskQueueTag)
56  .RestoreContentAfterSerialization(microtask_queue_external_pointer_,
57  no_gc_);
58  }
59 
60  private:
61  Tagged<NativeContext> native_context_;
62  ExternalPointerSlot::RawContent microtask_queue_external_pointer_;
63  const DisallowGarbageCollection& no_gc_;
64 };

83 void ContextSerializer::Serialize(Tagged<Context>* o,
84  const DisallowGarbageCollection& no_gc) {
85  context_ = *o;
86  DCHECK(IsNativeContext(context_));
87 
88  // Upon deserialization, references to the global proxy and its map will be
89  // replaced.
90  reference_map()->AddAttachedReference(context_->global_proxy());
91  reference_map()->AddAttachedReference(context_->global_proxy()->map());
92 
93  // The bootstrap snapshot has a code-stub context. When serializing the
94  // context snapshot, it is chained into the weak context list on the isolate
95  // and it's next context pointer may point to the code-stub context. Clear
96  // it before serializing, it will get re-added to the context list
97  // explicitly when it's loaded.
98  // TODO(v8:10416): These mutations should not observably affect the running
99  // context.
100  context_->SetNoCell(Context::NEXT_CONTEXT_LINK,
101  ReadOnlyRoots(isolate()).undefined_value());
102  DCHECK(!IsUndefined(context_->global_object()));
103  // Reset math random cache to get fresh random numbers.
104  MathRandom::ResetContext(context_);
105 
106  SanitizeNativeContextScope sanitize_native_context(
107  isolate(), context_->native_context(), allow_active_isolate_for_testing(),
108  no_gc);
109 
110  VisitRootPointer(Root::kStartupObjectCache, nullptr, FullObjectSlot(o));
111  SerializeDeferredObjects();
112 
113  // Add section for embedder-serialized embedder fields.
114  if (!embedder_fields_sink_.data()->empty()) {
115  sink_.Put(kEmbedderFieldsData, "embedder fields data");
116  sink_.Append(embedder_fields_sink_);
117  sink_.Put(kSynchronize, "Finished with embedder fields data");
118  }
119 
120  // Add section for embedder-serializer API wrappers.
121  if (!api_wrapper_sink_.data()->empty()) {
122  sink_.Put(kApiWrapperFieldsData, "api wrapper fields data");
123  sink_.Append(api_wrapper_sink_);
124  sink_.Put(kSynchronize, "Finished with api wrapper fields data");
125  }
126 
127  Pad();
128 }

159 void ContextSerializer::SerializeObjectImpl(Handle<HeapObject> obj,
160  SlotType slot_type) {
161  DCHECK(!ObjectIsBytecodeHandler(*obj)); // Only referenced in dispatch table.
162 
163  if (!allow_active_isolate_for_testing()) {
164  // When serializing a snapshot intended for real use, we should not end up
165  // at another native context.
166  // But in test scenarios there is no way to avoid this. Since we only
167  // serialize a single context in these cases, and this context does not
168  // have to be executable, we can simply ignore this.
169  DCHECK_IMPLIES(IsNativeContext(*obj), *obj == context_);
170  }
171 
172  {
173  DisallowGarbageCollection no_gc;
174  Tagged<HeapObject> raw = *obj;
175  if (SerializeHotObject(raw)) return;
176  if (SerializeRoot(raw)) return;
177  if (SerializeBackReference(raw)) return;
178  if (SerializeReadOnlyObjectReference(raw, &sink_)) return;
179  }
180 
181  if (startup_serializer_->SerializeUsingSharedHeapObjectCache(&sink_, obj)) {
182  return;
183  }
184 
185  if (ShouldBeInTheStartupObjectCache(*obj)) {
186  startup_serializer_->SerializeUsingStartupObjectCache(&sink_, obj);
187  return;
188  }
189 
190  // Pointers from the context snapshot to the objects in the startup snapshot
191  // should go through the root array or through the startup object cache.
192  // If this is not the case you may have to add something to the root array.
193  DCHECK(!startup_serializer_->ReferenceMapContains(obj));
194  // All the internalized strings that the context snapshot needs should be
195  // either in the root table or in the shared heap object cache.
196  DCHECK(!IsInternalizedString(*obj));
197  // Function and object templates are not context specific.
198  DCHECK(!IsTemplateInfo(*obj));
199 
200  InstanceType instance_type = obj->map()->instance_type();
201  if (InstanceTypeChecker::IsFeedbackVector(instance_type)) {
202  // Clear literal boilerplates and feedback.
203  Cast<FeedbackVector>(obj)->ClearSlots(isolate());
204  } else if (InstanceTypeChecker::IsJSObject(instance_type)) {
205  Handle<JSObject> js_obj = Cast<JSObject>(obj);
206  int embedder_fields_count = js_obj->GetEmbedderFieldCount();
207  if (embedder_fields_count > 0) {
208  DCHECK(!js_obj->NeedsRehashing(cage_base()));
209  v8::Local<v8::Object> api_obj = v8::Utils::ToLocal(js_obj);
210  v8::SerializeInternalFieldsCallback user_callback =
211  serialize_embedder_fields_.js_object_callback;
212  SerializeObjectWithEmbedderFields(js_obj, embedder_fields_count,
213  InternalFieldSerializeWrapper,
214  user_callback, api_obj);
215  if (IsJSApiWrapperObject(*js_obj)) {
216  SerializeApiWrapperFields(js_obj);
217  }
218  return;
219  }
220  if (InstanceTypeChecker::IsJSFunction(instance_type)) {
221  DisallowGarbageCollection no_gc;
222  // Unconditionally reset the JSFunction to its SFI's code, since we can't
223  // serialize optimized code anyway.
224  Tagged<JSFunction> closure = Cast<JSFunction>(*obj);
225  if (closure->shared()->HasBytecodeArray()) {
226  closure->SetInterruptBudget(isolate(), BudgetModification::kReset);
227  }
228  closure->ResetIfCodeFlushed(isolate());
229  if (closure->is_compiled(isolate())) {
230  if (closure->shared()->HasBaselineCode()) {
231  closure->shared()->FlushBaselineCode();
232  }
233  Tagged<Code> sfi_code = closure->shared()->GetCode(isolate());
234  if (!sfi_code.SafeEquals(closure->code(isolate()))) {
235  closure->UpdateCode(isolate(), sfi_code);
236  }
237  }
238  }
239  } else if (InstanceTypeChecker::IsEmbedderDataArray(instance_type) &&
240  !allow_active_isolate_for_testing()) {
241  DCHECK_EQ(*obj, context_->embedder_data());
242  Handle<EmbedderDataArray> embedder_data = Cast<EmbedderDataArray>(obj);
243  int embedder_fields_count = embedder_data->length();
244  if (embedder_data->length() > 0) {
245  DirectHandle<Context> context_handle(context_, isolate());
246  v8::Local<v8::Context> api_obj =
247  v8::Utils::ToLocal(Cast<NativeContext>(context_handle));
248  v8::SerializeContextDataCallback user_callback =
249  serialize_embedder_fields_.context_callback;
250  SerializeObjectWithEmbedderFields(embedder_data, embedder_fields_count,
251  ContextDataSerializeWrapper,
252  user_callback, api_obj);
253  return;
254  }
255  }
256 
257  CheckRehashability(*obj);
258 
259  // Object has not yet been serialized. Serialize it here.
260  ObjectSerializer serializer(this, obj, &sink_);
261  serializer.Serialize(slot_type);
262  if (InstanceTypeChecker::IsJSApiWrapperObject(instance_type)) {
263  SerializeApiWrapperFields(Cast<JSObject>(obj));
264  }
265 }

267 bool ContextSerializer::ShouldBeInTheStartupObjectCache(Tagged<HeapObject> o) {
268  // We can't allow scripts to be part of the context snapshot because they
269  // contain a unique ID, and deserializing several context snapshots containing
270  // script would cause dupes.
271  return IsName(o) || IsScript(o) || IsSharedFunctionInfo(o) ||
272  IsHeapNumber(o) || IsCode(o) || IsInstructionStream(o) ||
273  IsScopeInfo(o) || IsAccessorInfo(o) || IsTemplateInfo(o) ||
274  IsClassPositions(o) ||
275  o->map() == ReadOnlyRoots(isolate()).fixed_cow_array_map();
276 }

278 bool ContextSerializer::ShouldBeInTheSharedObjectCache(Tagged<HeapObject> o) {
279  // v8_flags.shared_string_table may be true during deserialization, so put
280  // internalized strings into the shared object snapshot.
281  return IsInternalizedString(o);
282 }

288 void ContextSerializer::SerializeApiWrapperFields(
289  DirectHandle<JSObject> js_object) {
290  DCHECK(IsJSApiWrapperObject(*js_object));
291  auto* cpp_heap_pointer =
292  CppHeapObjectWrapper(*js_object)
293  .GetCppHeapWrappable(isolate(), kAnyCppHeapPointer);
294  const auto& callback_data = serialize_embedder_fields_.api_wrapper_callback;
295  if (callback_data.callback == nullptr && cpp_heap_pointer == nullptr) {
296  // No need to serialize anything as empty handles or handles pointing to
297  // null objects will be preserved.
298  return;
299  }
300  DCHECK_NOT_NULL(callback_data.callback);
301  const auto data = callback_data.callback(
302  v8::Utils::ToLocal(js_object), cpp_heap_pointer, callback_data.data);
303  if (DataIsEmpty(data)) {
304  return;
305  }
306  const SerializerReference* reference =
307  reference_map()->LookupReference(*js_object);
308  DCHECK_NOT_NULL(reference);
309  DCHECK(reference->is_back_reference());
310  api_wrapper_sink_.Put(kNewObject, "api wrapper field holder");
311  api_wrapper_sink_.PutUint30(reference->back_ref_index(), "BackRefIndex");
312  api_wrapper_sink_.PutUint30(data.raw_size, "api wrapper raw field data size");
313  api_wrapper_sink_.PutRaw(reinterpret_cast<const uint8_t*>(data.data),
314  data.raw_size, "api wrapper raw field data");
315 }

317 template <typename V8Type, typename UserSerializerWrapper,
318  typename UserCallback, typename ApiObjectType>
319 void ContextSerializer::SerializeObjectWithEmbedderFields(
320  Handle<V8Type> data_holder, int embedder_fields_count,
321  UserSerializerWrapper wrapper, UserCallback user_callback,
322  ApiObjectType api_obj) {
323  DisallowGarbageCollection no_gc;
324  CHECK_GT(embedder_fields_count, 0);
325  DisallowJavascriptExecution no_js(isolate());
326  DisallowCompilation no_compile(isolate());
327 
328  auto raw_obj = *data_holder;
329 
330  std::vector<EmbedderDataSlot::RawData> original_embedder_values;
331  std::vector<StartupData> serialized_data;
332  std::vector<bool> should_clear_slot;
333 
334  // 1) Iterate embedder fields. Hold onto the original value of the fields.
335  // Ignore references to heap objects since these are to be handled by the
336  // serializer. For aligned pointers, call the serialize callback. Hold
337  // onto the result.
338  for (int i = 0; i < embedder_fields_count; i++) {
339  EmbedderDataSlot slot(raw_obj, i);
340  original_embedder_values.emplace_back(slot.load_raw(isolate(), no_gc));
341  Tagged<Object> object = slot.load_tagged();
342  if (IsHeapObject(object)) {
343  DCHECK(IsValidHeapObject(isolate()->heap(), Cast<HeapObject>(object)));
344  serialized_data.push_back({nullptr, 0});
345  should_clear_slot.push_back(false);
346  } else {
347  StartupData data =
348  wrapper(i, object == Smi::zero(), user_callback, api_obj);
349  serialized_data.push_back(data);
350  bool clear_slot =
351  !DataIsEmpty(data) || slot.MustClearDuringSerialization(no_gc);
352  should_clear_slot.push_back(clear_slot);
353  }
354  }
355 
356  // 2) Prevent embedder fields that are not V8 objects from ending up in the
357  // blob. This is done separately to step 1 so as to not interleave with
358  // embedder callbacks.
359  for (int i = 0; i < embedder_fields_count; i++) {
360  if (should_clear_slot[i]) {
361  EmbedderDataSlot(raw_obj, i).store_raw(isolate(), kNullAddress, no_gc);
362  }
363  }
364 
365  // 3) Serialize the object. References from embedder fields to heap objects or
366  // smis are serialized regularly.
367  {
368  AllowGarbageCollection allow_gc;
369  ObjectSerializer(this, data_holder, &sink_).Serialize(SlotType::kAnySlot);
370  // Reload raw pointer.
371  raw_obj = *data_holder;
372  }
373 
374  // 4) Obtain back reference for the serialized object.
375  const SerializerReference* reference =
376  reference_map()->LookupReference(raw_obj);
377  DCHECK_NOT_NULL(reference);
378  DCHECK(reference->is_back_reference());
379 
380  // 5) Write data returned by the embedder callbacks into a separate sink,
381  // headed by the back reference. Restore the original embedder fields.
382  for (int i = 0; i < embedder_fields_count; i++) {
383  StartupData data = serialized_data[i];
384  if (!should_clear_slot[i]) continue;
385  // Restore original values from cleared fields.
386  EmbedderDataSlot(raw_obj, i)
387  .store_raw(isolate(), original_embedder_values[i], no_gc);
388  if (DataIsEmpty(data)) continue;
389  embedder_fields_sink_.Put(kNewObject, "embedder field holder");
390  embedder_fields_sink_.PutUint30(reference->back_ref_index(),
391  "BackRefIndex");
392  embedder_fields_sink_.PutUint30(i, "embedder field index");
393  embedder_fields_sink_.PutUint30(data.raw_size, "embedder fields data size");
394  embedder_fields_sink_.PutRaw(reinterpret_cast<const uint8_t*>(data.data),
395  data.raw_size, "embedder fields data");
396  delete[] data.data;
397  }
398 
399  // 6) The content of the separate sink is appended eventually to the default
400  // sink. The ensures that during deserialization, we call the deserializer
401  // callback at the end, and can guarantee that the deserialized objects are
402  // in a consistent state. See ContextSerializer::Serialize.
403 }

405 void ContextSerializer::CheckRehashability(Tagged<HeapObject> obj) {
406  if (!can_be_rehashed_) return;
407  if (!obj->NeedsRehashing(cage_base())) return;
408  if (obj->CanBeRehashed(cage_base())) return;
409  can_be_rehashed_ = false;
410 }
