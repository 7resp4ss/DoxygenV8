98 void MessageHandler::ReportMessage(Isolate* isolate, const MessageLocation* loc,
99  DirectHandle<JSMessageObject> message) {
100  v8::Local<v8::Message> api_message_obj = v8::Utils::MessageToLocal(message);
101 
102  if (api_message_obj->ErrorLevel() != v8::Isolate::kMessageError) {
103  ReportMessageNoExceptions(isolate, loc, message, v8::Local<v8::Value>());
104  return;
105  }
106 
107  // We are calling into embedder's code which can throw exceptions.
108  // Thus we need to save current exception state, reset it to the clean one
109  // and ignore scheduled exceptions callbacks can throw.
110 
111  // We pass the exception object into the message handler callback though.
112  DirectHandle<Object> exception = isolate->factory()->undefined_value();
113  if (isolate->has_exception()) {
114  exception = direct_handle(isolate->exception(), isolate);
115  }
116 
117  Isolate::ExceptionScope exception_scope(isolate);
118  isolate->clear_pending_message();
119 
120  // Turn the exception on the message into a string if it is an object.
121  if (IsJSObject(message->argument())) {
122  HandleScope scope(isolate);
123  DirectHandle<Object> argument(message->argument(), isolate);
124 
125  MaybeDirectHandle<Object> maybe_stringified;
126  DirectHandle<Object> stringified;
127  // Make sure we don't leak uncaught internally generated Error objects.
128  if (IsJSError(*argument)) {
129  maybe_stringified = Object::NoSideEffectsToString(isolate, argument);
130  } else {
131  v8::TryCatch catcher(reinterpret_cast<v8::Isolate*>(isolate));
132  catcher.SetVerbose(false);
133  catcher.SetCaptureMessage(false);
134 
135  maybe_stringified = Object::ToString(isolate, argument);
136  }
137 
138  if (!maybe_stringified.ToHandle(&stringified)) {
139  isolate->clear_pending_message();
140  stringified = isolate->factory()->exception_string();
141  }
142  message->set_argument(*stringified);
143  }
144 
145  v8::Local<v8::Value> api_exception_obj = v8::Utils::ToLocal(exception);
146  ReportMessageNoExceptions(isolate, loc, message, api_exception_obj);
147 }

149 void MessageHandler::ReportMessageNoExceptions(
150  Isolate* isolate, const MessageLocation* loc, DirectHandle<Object> message,
151  v8::Local<v8::Value> api_exception_obj) {
152  v8::Local<v8::Message> api_message_obj = v8::Utils::MessageToLocal(message);
153  int error_level = api_message_obj->ErrorLevel();
154 
155  DirectHandle<ArrayList> global_listeners =
156  isolate->factory()->message_listeners();
157  int global_length = global_listeners->length();
158  if (global_length == 0) {
159  DefaultMessageReport(isolate, loc, message);
160  } else {
161  for (int i = 0; i < global_length; i++) {
162  HandleScope scope(isolate);
163  if (IsUndefined(global_listeners->get(i), isolate)) continue;
164  Tagged<FixedArray> listener = Cast<FixedArray>(global_listeners->get(i));
165  Tagged<Foreign> callback_obj = Cast<Foreign>(listener->get(0));
166  int32_t message_levels =
167  static_cast<int32_t>(Smi::ToInt(listener->get(2)));
168  if (!(message_levels & error_level)) {
169  continue;
170  }
171  v8::MessageCallback callback =
172  v8::ToCData<v8::MessageCallback, kMessageListenerTag>(isolate,
173  callback_obj);
174  DirectHandle<Object> callback_data(listener->get(1), isolate);
175  {
176  RCS_SCOPE(isolate, RuntimeCallCounterId::kMessageListenerCallback);
177  // Do not allow exceptions to propagate.
178  v8::TryCatch try_catch(reinterpret_cast<v8::Isolate*>(isolate));
179  callback(api_message_obj, IsUndefined(*callback_data, isolate)
180  ? api_exception_obj
181  : v8::Utils::ToLocal(callback_data));
182  }
183  }
184  }

203 // Convert the raw frames as written by Isolate::CaptureSimpleStackTrace into
204 // a JSArray of JSCallSite objects.
205 MaybeDirectHandle<JSArray> GetStackFrames(Isolate* isolate,
206  DirectHandle<FixedArray> frames) {
207  int frame_count = frames->length();
208  DirectHandle<JSFunction> constructor = isolate->callsite_function();
209  DirectHandle<FixedArray> sites =
210  isolate->factory()->NewFixedArray(frame_count);
211  for (int i = 0; i < frame_count; ++i) {
212  DirectHandle<CallSiteInfo> frame(Cast<CallSiteInfo>(frames->get(i)),
213  isolate);
214  DirectHandle<JSObject> site;
215  ASSIGN_RETURN_ON_EXCEPTION(isolate, site,
216  JSObject::New(constructor, constructor,
217  Handle<AllocationSite>::null()));
218  RETURN_ON_EXCEPTION(
219  isolate, JSObject::SetOwnPropertyIgnoreAttributes(
220  site, isolate->factory()->call_site_info_symbol(), frame,
221  DONT_ENUM));
222  sites->set(i, *site);
223  }
224 
225  return isolate->factory()->NewJSArrayWithElements(sites);
226 }

228 MaybeDirectHandle<Object> AppendErrorString(Isolate* isolate,
229  DirectHandle<Object> error,
230  IncrementalStringBuilder* builder) {
231  v8::TryCatch try_catch(reinterpret_cast<v8::Isolate*>(isolate));
232  try_catch.SetVerbose(false);
233  try_catch.SetCaptureMessage(false);
234  MaybeDirectHandle<String> err_str = ErrorUtils::ToString(
235  isolate, Cast<Object>(error),
236  ErrorUtils::ToStringMessageSource::kPreferOriginalMessage);
237  if (err_str.is_null()) {
238  // Error.toString threw. Try to return a string representation of the thrown
239  // exception instead.
240 
241  DCHECK(isolate->has_exception());
242  if (isolate->is_execution_terminating()) {
243  return {};
244  }
245  DirectHandle<Object> exception(isolate->exception(), isolate);
246  try_catch.Reset();
247 
248  err_str = ErrorUtils::ToString(
249  isolate, exception,
250  ErrorUtils::ToStringMessageSource::kPreferOriginalMessage);
251  if (err_str.is_null()) {
252  // Formatting the thrown exception threw again, give up.
253  DCHECK(isolate->has_exception());
254  if (isolate->is_execution_terminating()) return {};
255  builder->AppendCStringLiteral("<error>");
256  } else {
257  // Formatted thrown exception successfully, append it.
258  builder->AppendCStringLiteral("<error: ");
259  builder->AppendString(err_str.ToHandleChecked());
260  builder->AppendCharacter('>');
261  }
262  } else {
263  builder->AppendString(err_str.ToHandleChecked());
264  }
265 
266  return error;
267 }

288 MaybeDirectHandle<Object> ErrorUtils::FormatStackTrace(
289  Isolate* isolate, DirectHandle<JSObject> error,
290  DirectHandle<Object> raw_stack) {
291  if (v8_flags.correctness_fuzzer_suppressions) {
292  return isolate->factory()->empty_string();
293  }
294  DCHECK(IsFixedArray(*raw_stack));
295  auto elems = Cast<FixedArray>(raw_stack);
296 
297  const bool in_recursion = isolate->formatting_stack_trace();
298  const bool has_overflowed = i::StackLimitCheck{isolate}.HasOverflowed();
299  DirectHandle<NativeContext> error_context;
300  if (!in_recursion && !has_overflowed &&
301  error->GetCreationContext(isolate).ToHandle(&error_context)) {
302  if (isolate->HasPrepareStackTraceCallback()) {
303  PrepareStackTraceScope scope(isolate);
304 
305  DirectHandle<JSArray> sites;
306  ASSIGN_RETURN_ON_EXCEPTION(isolate, sites,
307  GetStackFrames(isolate, elems));
308 
309  DirectHandle<Object> result;
310  ASSIGN_RETURN_ON_EXCEPTION(
311  isolate, result,
312  isolate->RunPrepareStackTraceCallback(error_context, error, sites));
313  return result;
314  } else {
315  DirectHandle<JSFunction> global_error(error_context->error_function(),
316  isolate);
317 
318  // If there's a user-specified "prepareStackTrace" function, call it on
319  // the frames and use its result.
320 
321  DirectHandle<Object> prepare_stack_trace;
322  ASSIGN_RETURN_ON_EXCEPTION(
323  isolate, prepare_stack_trace,
324  JSFunction::GetProperty(isolate, global_error, "prepareStackTrace"));
325 
326  if (IsJSFunction(*prepare_stack_trace)) {
327  PrepareStackTraceScope scope(isolate);
328 
329  isolate->CountUsage(v8::Isolate::kErrorPrepareStackTrace);
330 
331  DirectHandle<JSArray> sites;
332  ASSIGN_RETURN_ON_EXCEPTION(isolate, sites,
333  GetStackFrames(isolate, elems));
334 
335  constexpr int argc = 2;
336  std::array<DirectHandle<Object>, argc> args;
337  if (V8_UNLIKELY(IsJSGlobalObject(*error))) {
338  // Pass global proxy instead of global object.
339  args[0] = direct_handle(Cast<JSGlobalObject>(*error)->global_proxy(),
340  isolate);
341  } else {
342  args[0] = error;
343  }
344  args[1] = sites;
345 
346  DirectHandle<Object> result;
347 
348  ASSIGN_RETURN_ON_EXCEPTION(
349  isolate, result,
350  Execution::Call(isolate, prepare_stack_trace, global_error,
351  base::VectorOf(args)));
352  return result;
353  }
354  }
355  }
356 
357  // Otherwise, run our internal formatting logic.
358  IncrementalStringBuilder builder(isolate);
359 
360  RETURN_ON_EXCEPTION(isolate, AppendErrorString(isolate, error, &builder));
361 
362  for (int i = 0; i < elems->length(); ++i) {
363  builder.AppendCStringLiteral("\n at ");
364 
365  DirectHandle<CallSiteInfo> frame(Cast<CallSiteInfo>(elems->get(i)),
366  isolate);
367 
368  v8::TryCatch try_catch(reinterpret_cast<v8::Isolate*>(isolate));
369  SerializeCallSiteInfo(isolate, frame, &builder);
370 
371  if (isolate->has_exception()) {
372  // CallSite.toString threw. Parts of the current frame might have been
373  // stringified already regardless. Still, try to append a string
374  // representation of the thrown exception.
375 
376  DirectHandle<Object> exception(isolate->exception(), isolate);
377  try_catch.Reset();
378 
379  MaybeDirectHandle<String> exception_string =
380  ErrorUtils::ToString(isolate, exception);
381  if (exception_string.is_null()) {
382  // Formatting the thrown exception threw again, give up.
383 
384  builder.AppendCStringLiteral("<error>");
385  } else {
386  // Formatted thrown exception successfully, append it.
387  builder.AppendCStringLiteral("<error: ");
388  builder.AppendString(exception_string.ToHandleChecked());
389  builder.AppendCStringLiteral("<error>");
390  }
391  }
392  }
393 
394  return builder.Finish();
395 }

397 DirectHandle<String> MessageFormatter::Format(
398  Isolate* isolate, MessageTemplate index,
399  base::Vector<const DirectHandle<Object>> args) {
400  constexpr size_t kMaxArgs = 3;
401  DirectHandle<String> arg_strings[kMaxArgs];
402  DCHECK_LE(args.size(), kMaxArgs);
403  for (size_t i = 0; i < args.size(); ++i) {
404  DCHECK(!args[i].is_null());
405  arg_strings[i] = Object::NoSideEffectsToString(isolate, args[i]);
406  }
407  v8::TryCatch try_catch(reinterpret_cast<v8::Isolate*>(isolate));
408  try_catch.SetVerbose(false);
409  try_catch.SetCaptureMessage(false);
410  MaybeHandle<String> maybe_result_string = MessageFormatter::TryFormat(
411  isolate, index, base::VectorOf(arg_strings, args.size()));
412  Handle<String> result_string;
413  if (!maybe_result_string.ToHandle(&result_string)) {
414  DCHECK(isolate->has_exception());
415  return isolate->factory()->InternalizeString(
416  base::StaticCharVector("<error>"));
417  }
418  // A string that has been obtained from JS code in this way is
419  // likely to be a complicated ConsString of some sort. We flatten it
420  // here to improve the efficiency of converting it to a C string and
421  // other operations that are likely to take place (see GetLocalizedMessage
422  // for example).
423  return String::Flatten(isolate, result_string);
424 }

440  Isolate* isolate, MessageTemplate index,
441  base::Vector<const DirectHandle<String>> args) {
442  const char* template_string = TemplateString(index);
443

447  static constexpr MessageTemplate kTemplatesWithMismatchedArguments[] = {
448  MessageTemplate::kConstAssign,
449  MessageTemplate::kConstructorNotReceiver,

461  MessageTemplate::kInvalidLhsInPrefixOp,
462  MessageTemplate::kInvalidPrivateBrandReinitialization,
463  MessageTemplate::kInvalidPrivateFieldReinitialization,
464  MessageTemplate::kInvalidPrivateMemberWrite,
465  MessageTemplate::kInvalidRegExpExecResult,
466  MessageTemplate::kInvalidTimeValue,

469  MessageTemplate::kIteratorReduceNoInitial,
470  MessageTemplate::kJsonParseShortString,

473  MessageTemplate::kJsonParseUnexpectedTokenShortString,
474  MessageTemplate::kJsonParseUnexpectedTokenStartStringWithContext,
475  MessageTemplate::kJsonParseUnexpectedTokenSurroundStringWithContext,
476  MessageTemplate::kMustBePositive,
477  MessageTemplate::kNotIterable,
478  MessageTemplate::kNotTypedArray,
479  MessageTemplate::kProxyNonObject,
480  MessageTemplate::kProxyPrivate,
481  MessageTemplate::kProxyRevoked,
482  MessageTemplate::kProxyTrapReturnedFalsishFor,
483  MessageTemplate::kReduceNoInitial,
484  MessageTemplate::kSpreadIteratorSymbolNonCallable,
485  MessageTemplate::kSymbolIteratorInvalid,
486  MessageTemplate::kTopLevelAwaitStalled,
487  MessageTemplate::kUndefinedOrNullToObject,
488  MessageTemplate::kUnexpectedStrictReserved,
489  MessageTemplate::kUnexpectedTokenIdentifier,
490  MessageTemplate::kWeakRefsCleanupMustBeCallable};
491 
492  base::Vector<const DirectHandle<String>> remaining_args = args;
493  for (const char* c = template_string; *c != '\0'; c++) {
494  if (*c == '%') {
495  // %% results in verbatim %.
496  if (*(c + 1) == '%') {
497  c++;
498  builder.AppendCharacter('%');
499  } else {
500  // TODO(14386): Remove this fallback.
501  if (remaining_args.empty()) {
502  if (std::count(std::begin(kTemplatesWithMismatchedArguments),
503  std::end(kTemplatesWithMismatchedArguments), index)) {
504  builder.AppendCString("undefined");
505  } else {
506  FATAL("Missing argument to template (got %zu): %s", args.size(),
507  template_string);
508  }
509  } else {
510  DirectHandle<String> arg = remaining_args[0];
511  remaining_args += 1;
512  builder.AppendString(arg);
513  }
514  }
515  } else {
516  builder.AppendCharacter(*c);
517  }
518  }
519  if (!remaining_args.empty() &&
520  std::count(std::begin(kTemplatesWithMismatchedArguments),
521  std::end(kTemplatesWithMismatchedArguments), index) == 0) {
522  FATAL("Too many arguments to template (expected %zu, got %zu): %s",
523  args.size() - remaining_args.size(), args.size(), template_string);
524  }
525 
526  return indirect_handle(builder.Finish(), isolate);
527 }

549 MaybeHandle<JSObject> ErrorUtils::Construct(
550  Isolate* isolate, DirectHandle<JSFunction> target,
551  DirectHandle<Object> new_target, DirectHandle<Object> message,
552  DirectHandle<Object> options, FrameSkipMode mode,
553  DirectHandle<Object> caller, StackTraceCollection stack_trace_collection) {
554  if (v8_flags.correctness_fuzzer_suppressions) {
555  // Abort range errors in correctness fuzzing, as their causes differ
556  // across correctness-fuzzing scenarios.
557  if (target.is_identical_to(isolate->range_error_function())) {
558  FATAL("Aborting on range error");
559  }
560  // Ignore error messages in correctness fuzzing, because the spec leaves
561  // room for undefined behavior.
562  message = isolate->factory()->InternalizeUtf8String(
563  "Message suppressed for fuzzers (--correctness-fuzzer-suppressions)");
564  }
565 
566  // 1. If NewTarget is undefined, let newTarget be the active function object,
567  // else let newTarget be NewTarget.
568  DirectHandle<JSReceiver> new_target_recv = IsJSReceiver(*new_target)
569  ? Cast<JSReceiver>(new_target)
570  : Cast<JSReceiver>(target);
571 
572  // 2. Let O be ? OrdinaryCreateFromConstructor(newTarget, "%ErrorPrototype%",
573  // [[ErrorData]] ).
574  Handle<JSObject> err;
575  ASSIGN_RETURN_ON_EXCEPTION(isolate, err,
576  JSObject::New(target, new_target_recv, {}));
577 
578  // 3. If message is not undefined, then
579  // a. Let msg be ? ToString(message).
580  // b. Let msgDesc be the PropertyDescriptor{[[Value]]: msg, [[Writable]]:
581  // true, [[Enumerable]]: false, [[Configurable]]: true}.
582  // c. Perform ! DefinePropertyOrThrow(O, "message", msgDesc).
583  // 4. Return O.
584  if (!IsUndefined(*message, isolate)) {
585  DirectHandle<String> msg_string;
586  ASSIGN_RETURN_ON_EXCEPTION(isolate, msg_string,
587  Object::ToString(isolate, message));
588  RETURN_ON_EXCEPTION(isolate, JSObject::SetOwnPropertyIgnoreAttributes(
589  err, isolate->factory()->message_string(),
590  msg_string, DONT_ENUM));
591 
592  if (v8_flags.use_original_message_for_stack_trace) {
593  RETURN_ON_EXCEPTION(isolate,
594  JSObject::SetOwnPropertyIgnoreAttributes(
595  err, isolate->factory()->error_message_symbol(),
596  msg_string, DONT_ENUM));
597  }
598  }
599 
600  if (!IsUndefined(*options, isolate)) {
601  // If Type(options) is Object and ? HasProperty(options, "cause") then
602  // a. Let cause be ? Get(options, "cause").
603  // b. Perform ! CreateNonEnumerableDataPropertyOrThrow(O, "cause", cause).
604  DirectHandle<Name> cause_string = isolate->factory()->cause_string();
605  if (IsJSReceiver(*options)) {
606  DirectHandle<JSReceiver> js_options = Cast<JSReceiver>(options);
607  Maybe<bool> has_cause =
608  JSObject::HasProperty(isolate, js_options, cause_string);
609  if (has_cause.IsNothing()) {
610  DCHECK((isolate)->has_exception());
611  return MaybeHandle<JSObject>();
612  }
613  if (has_cause.ToChecked()) {
614  DirectHandle<Object> cause;
615  ASSIGN_RETURN_ON_EXCEPTION(
616  isolate, cause,
617  JSObject::GetProperty(isolate, js_options, cause_string));
618  RETURN_ON_EXCEPTION(isolate, JSObject::SetOwnPropertyIgnoreAttributes(
619  err, cause_string, cause, DONT_ENUM));
620  }
621  }
622  }
623 
624  switch (stack_trace_collection) {
625  case StackTraceCollection::kEnabled:
626  RETURN_ON_EXCEPTION(isolate,
627  isolate->CaptureAndSetErrorStack(

658 // ES6 section 19.5.3.4 Error.prototype.toString ( )
659 MaybeHandle<String> ErrorUtils::ToString(Isolate* isolate,
660  DirectHandle<Object> receiver,
661  ToStringMessageSource message_source) {
662  // 1. Let O be the this value.
663  // 2. If Type(O) is not Object, throw a TypeError exception.
664  DirectHandle<JSReceiver> recv;
665  if (!TryCast<JSReceiver>(receiver, &recv)) {
666  THROW_NEW_ERROR(isolate,
667  NewTypeError(MessageTemplate::kIncompatibleMethodReceiver,
668  isolate->factory()->NewStringFromAsciiChecked(
669  "Error.prototype.toString"),
670  receiver));
671  }
672  // 3. Let name be ? Get(O, "name").
673  // 4. If name is undefined, let name be "Error"; otherwise let name be
674  // ? ToString(name).
675  DirectHandle<String> name_key = isolate->factory()->name_string();
676  Handle<String> name_default = isolate->factory()->Error_string();
677  Handle<String> name;
678  ASSIGN_RETURN_ON_EXCEPTION(
679  isolate, name,
680  GetStringPropertyOrDefault(isolate, recv, name_key, name_default));
681 
682  // 5. Let msg be ? Get(O, "message").
683  // 6. If msg is undefined, let msg be the empty String; otherwise let msg be
684  // ? ToString(msg).
685  Handle<String> msg;
686  Handle<String> msg_default = isolate->factory()->empty_string();
687  if (message_source == ToStringMessageSource::kPreferOriginalMessage) {
688  // V8-specific extension for Error.stack: Use the original message with
689  // which the Error constructor was called. This keeps Error.stack consistent
690  // w.r.t. "message" property changes regardless of the time when Error.stack
691  // is accessed the first time.
692  //
693  // If |recv| was not constructed with %Error%, use the "message" property.
694  LookupIterator it(isolate, LookupIterator::PROTOTYPE_CHAIN_SKIP_INTERCEPTOR,
695  recv, isolate->factory()->error_message_symbol());
696  Handle<Object> result = JSReceiver::GetDataProperty(&it);
697  if (it.IsFound() && IsUndefined(*result, isolate)) {
698  msg = msg_default;
699  } else if (it.IsFound()) {
700  ASSIGN_RETURN_ON_EXCEPTION(isolate, msg,
701  Object::ToString(isolate, result));
702  }
703  }
704 
705  if (msg.is_null()) {
706  DirectHandle<String> msg_key = isolate->factory()->message_string();
707  ASSIGN_RETURN_ON_EXCEPTION(
708  isolate, msg,
709  GetStringPropertyOrDefault(isolate, recv, msg_key, msg_default));
710  }
711 
712  // 7. If name is the empty String, return msg.
713  // 8. If msg is the empty String, return name.
714  if (name->length() == 0) return msg;
715  if (msg->length() == 0) return name;
716 
717  // 9. Return the result of concatenating name, the code unit 0x003A (COLON),
718  // the code unit 0x0020 (SPACE), and msg.
719  IncrementalStringBuilder builder(isolate);
720  builder.AppendString(name);
721  builder.AppendCStringLiteral(": ");
722  builder.AppendString(msg);
723 
724  Handle<String> result;
725  ASSIGN_RETURN_ON_EXCEPTION(isolate, result,
726  indirect_handle(builder.Finish(), isolate));
727  return result;
728 }

1007 Tagged<Object> ErrorUtils::ThrowLoadFromNullOrUndefined(
1008  Isolate* isolate, DirectHandle<Object> object,
1009  MaybeDirectHandle<Object> key) {
1010  DCHECK(IsNullOrUndefined(*object));
1011 
1012  MaybeDirectHandle<String> maybe_property_name;
1013 
1014  // Try to extract the property name from the given key, if any.
1015  DirectHandle<Object> key_handle;
1016  if (key.ToHandle(&key_handle)) {
1017  if (IsString(*key_handle)) {
1018  maybe_property_name = Cast<String>(key_handle);
1019  } else {
1020  maybe_property_name =
1021  Object::NoSideEffectsToMaybeString(isolate, key_handle);
1022  }
1023  }
1024 
1025  DirectHandle<String> callsite;
1026 
1027  // Inline the RenderCallSite logic here so that we can additionally access the
1028  // destructuring property.
1029  bool location_computed = false;
1030  bool is_destructuring = false;
1031  MessageLocation location;
1032  if (ComputeLocation(isolate, &location)) {
1033  location_computed = true;
1034 
1035  UnoptimizedCompileFlags flags = UnoptimizedCompileFlags::ForFunctionCompile(
1036  isolate, *location.shared());
1037  flags.set_is_reparse(true);
1038  UnoptimizedCompileState compile_state;
1039  ReusableUnoptimizedCompileState reusable_state(isolate);
1040  ParseInfo info(isolate, flags, &compile_state, &reusable_state);
1041  if (parsing::ParseAny(&info, location.shared(), isolate,
1042  parsing::ReportStatisticsMode::kNo)) {
1043  info.ast_value_factory()->Internalize(isolate);
1044  CallPrinter printer(isolate, location.shared()->IsUserJavaScript());
1045  DirectHandle<String> str =
1046  printer.Print(info.literal(), location.start_pos());
1047 
1048  int pos = -1;
1049  is_destructuring = printer.destructuring_assignment() != nullptr;
1050 
1051  if (is_destructuring) {
1052  // If we don't have one yet, try to extract the property name from the
1053  // destructuring property in the AST.
1054  ObjectLiteralProperty* destructuring_prop =
1055  printer.destructuring_prop();
1056  if (maybe_property_name.is_null() && destructuring_prop != nullptr &&
1057  destructuring_prop->key()->IsPropertyName()) {
1058  maybe_property_name = destructuring_prop->key()
1059  ->AsLiteral()
1060  ->AsRawPropertyName()
1061  ->string();
1062  // Change the message location to point at the property name.
1063  pos = destructuring_prop->key()->position();
1064  }
1065  if (maybe_property_name.is_null()) {
1066  // Change the message location to point at the destructured value.
1067  pos = printer.destructuring_assignment()->value()->position();
1068  }
1069 
1070  // If we updated the pos to a valid pos, rewrite the location.
1071  if (pos != -1) {
1072  location = MessageLocation(location.script(), pos, pos + 1,
1073  location.shared());
1074  }
1075  }
1076 
1077  if (str->length() > 0) callsite = str;
1078  }
1079  }
1080 
1081  if (callsite.is_null()) {
1082  callsite = BuildDefaultCallSite(isolate, object);
1083  }

1086  DirectHandle<String> property_name;
1087  if (is_destructuring) {
1088  if (maybe_property_name.ToHandle(&property_name)) {
1089  error = isolate->factory()->NewTypeError(
1090  MessageTemplate::kNonCoercibleWithProperty, property_name, callsite,
1091  object);
1092  } else {
1093  error = isolate->factory()->NewTypeError(MessageTemplate::kNonCoercible,
1094  callsite, object);
1095  }
1096  } else {
1097  if (!key.ToHandle(&key_handle) ||
1098  !maybe_property_name.ToHandle(&property_name)) {
1099  error = isolate->factory()->NewTypeError(
1100  MessageTemplate::kNonObjectPropertyLoad, object);
1101  } else if (*key_handle == ReadOnlyRoots(isolate).iterator_symbol()) {
1102  error = NewIteratorError(isolate, object);
1103  } else {
1104  error = isolate->factory()->NewTypeError(
1105  MessageTemplate::kNonObjectPropertyLoadWithProperty, object,
1106  property_name);
1107  }
1108  }
1109 
1110  if (location_computed) {
1111  isolate->ThrowAt(error, &location);
1112  } else {
1113  isolate->Throw(*error);
1114  }
1115  return ReadOnlyRoots(isolate).exception();
1116 }
