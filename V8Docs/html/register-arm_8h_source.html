116 // Representation of a list of non-overlapping VFP registers. This list
117 // represents the data layout of VFP registers as a bitfield:
118 // S registers cover 1 bit
119 // D registers cover 2 bits
120 // Q registers cover 4 bits
121 //
122 // This way, we make sure no registers in the list ever overlap. However, a list
123 // may represent multiple different sets of registers,
124 // e.g. [d0 s2 s3] <=> [s0 s1 d1].
125 using VfpRegList = uint64_t;
126 
127 // Single word VFP register.
128 class SwVfpRegister : public RegisterBase<SwVfpRegister, kSwVfpAfterLast> {
129  public:
130  static constexpr int kSizeInBytes = 4;
131 
132  static void split_code(int reg_code, int* vm, int* m) {
133  DCHECK(from_code(reg_code).is_valid());
134  *m = reg_code & 0x1;
135  *vm = reg_code >> 1;
136  }
137  void split_code(int* vm, int* m) const { split_code(code(), vm, m); }
138  VfpRegList ToVfpRegList() const {
139  // Each bit in the list corresponds to a S register.
140  return uint64_t{0x1} << code();
141  }
142 
143  private:
144  friend class RegisterBase;
145  explicit constexpr SwVfpRegister(int code) : RegisterBase(code) {}
146 };

161 // Double word VFP register.
162 class DwVfpRegister : public RegisterBase<DwVfpRegister, kDoubleAfterLast> {
163  public:

171  static void split_code(int reg_code, int* vm, int* m) {
172  DCHECK(from_code(reg_code).is_valid());
173  *m = (reg_code & 0x10) >> 4;
174  *vm = reg_code & 0x0F;
175  }
176  void split_code(int* vm, int* m) const { split_code(code(), vm, m); }
177  VfpRegList ToVfpRegList() const {
178  // A D register overlaps two S registers.
179  return uint64_t{0x3} << (code() * 2);
180  }
181

184  friend class LowDwVfpRegister;
185  explicit constexpr DwVfpRegister(int code) : RegisterBase(code) {}
186 };
