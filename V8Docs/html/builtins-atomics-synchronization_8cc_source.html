13 std::optional<base::TimeDelta> GetTimeoutDelta(
14  DirectHandle<Object> timeout_obj) {
15  double ms = Object::NumberValue(*timeout_obj);
16  if (!std::isnan(ms)) {
17  if (ms < 0) ms = 0;
18  if (ms <= static_cast<double>(std::numeric_limits<int64_t>::max())) {
19  return base::TimeDelta::FromMilliseconds(static_cast<int64_t>(ms));
20  }
21  }
22  return std::nullopt;
23 }

78  isolate->factory()->NewStringFromAsciiChecked(
79  method_name)));
80  }

82  DirectHandle<Object> result;
83  {
84  JSAtomicsMutex::LockGuard lock_guard(isolate, js_mutex);
85  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
86  isolate, result,
87  Execution::Call(isolate, run_under_lock,
88  isolate->factory()->undefined_value(), {}));
89  }
