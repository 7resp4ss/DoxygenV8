25 // Abstraction over the stack. Supports handling of:
26 // - native stack;
27 // - ASAN/MSAN;
28 // - SafeStack: https://releases.llvm.org/10.0.0/tools/clang/docs/SafeStack.html
29 //
30 // Stacks grow down, so throughout this class "start" refers the highest
31 // address of the stack, and top/marker the lowest.
32 //
33 // TODO(chromium:1056170): Consider adding a component that keeps track
34 // of relevant GC stack regions where interesting pointers can be found.

125  // Stack segments that may contain pointers and should be scanned.
126  struct Segment {
127  // The start and top of the stack. It must be sp <= top <= start.
128  // The top pointer is generally a marker that signals the end of the
129  // interesting stack region in which on-heap pointers can be found.
130  const void* start = nullptr;
131  const void* top = nullptr;
132 
133 #ifdef V8_USE_ADDRESS_SANITIZER
134  // The start of ASAN's fake stack.
135  const void* asan_fake_stack = nullptr;
136 #endif // V8_USE_ADDRESS_SANITIZER
137 
138 #ifdef V8_USE_SAFE_STACK
139  // Start and top for the unsafe stack that is used in clang with
140  // -fsanitizer=safe-stack.
141  // It must be unsafe_sp <= unsafe_stack_top <= unsafe_stack_start.
142  // Notice that the terms "start" and "top" have here a different meaning in
143  // the terminology used in this feature's documentation.
144  const void* unsafe_stack_start = nullptr;
145  const void* unsafe_stack_top = nullptr;
146 #endif // V8_USE_SAFE_STACK
147 
148  Segment() = default;
149  Segment(const void* stack_start, const void* stack_top)
150  : start(stack_start), top(stack_top) {
151 #ifdef V8_USE_ADDRESS_SANITIZER
152  asan_fake_stack = __asan_get_current_fake_stack();
153 #endif // V8_USE_ADDRESS_SANITIZER
154 #ifdef V8_USE_SAFE_STACK
155  unsafe_stack_start = __builtin___get_unsafe_stack_top();
156  unsafe_stack_top = __builtin___get_unsafe_stack_ptr();
157 #endif // V8_USE_SAFE_STACK
158  }
159 
160 #ifdef DEBUG
161  bool Contains(const void* address) const {
162  return (start >= address) && (address >= top);
163  }
164 #endif
165  };

175  template <typename Callback>
176  static void SetMarkerAndCallbackImpl(Stack* stack, void* argument,
177  const void* stack_end) {
178  Segment previous_segment = stack->current_segment_;
179  stack->current_segment_.top = stack_end;
180 #ifdef V8_USE_SAFE_STACK
181  stack->current_segment_.unsafe_stack_top =
182  __builtin___get_unsafe_stack_ptr();
183 #endif // V8_USE_SAFE_STACK
184  Callback* callback = static_cast<Callback*>(argument);
185  (*callback)();
186  stack->current_segment_ = previous_segment;
187  }

189  template <typename Callback>
190  static void SetMarkerForBackgroundThreadAndCallbackImpl(
191  Stack* stack, void* argument, const void* stack_end) {
192  DCHECK(IsOnCurrentStack(stack_end));
193  auto [thread, callback] =
194  *static_cast<std::pair<ThreadId, Callback*>*>(argument);
195  auto& background_stacks = stack->background_stacks_;
196  Segment previous_segment;
197  {
198  v8::base::MutexGuard guard(&stack->lock_);
199  if (auto it = background_stacks.find(thread);
200  it != background_stacks.end()) {
201  previous_segment = it->second;
202  DCHECK_NOT_NULL(previous_segment.top);
203  } else {
204  DCHECK_NULL(previous_segment.top);
205  }
206  // This implicitly uses the current values (if applicable) for:
207  // - asan_fake_start
208  // - unsafe stack start
209  // - unsafe stack top
210  background_stacks[thread] =
211  Segment(v8::base::Stack::GetStackStart(), stack_end);
212  }
213  (*callback)();
214  {
215  v8::base::MutexGuard guard(&stack->lock_);
216  if (previous_segment.top)
217  background_stacks[thread] = previous_segment;
218  else
219  background_stacks.erase(thread);
220  }
221  }
