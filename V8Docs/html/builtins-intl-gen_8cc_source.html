50  // Jumps to {target} if the first two characters of {seq_string} equal
51  // {pattern} ignoring case.
52  void JumpIfStartsWithIgnoreCase(TNode<SeqOneByteString> seq_string,
53  const char* pattern, Label* target) {
54  size_t effective_offset =
55  OFFSET_OF_DATA_START(SeqOneByteString) - kHeapObjectTag;
56  TNode<Uint16T> raw =
57  Load<Uint16T>(seq_string, IntPtrConstant(effective_offset));
58  DCHECK_EQ(strlen(pattern), 2);
59 #if V8_TARGET_BIG_ENDIAN
60  int raw_pattern = (pattern[0] << 8) + pattern[1];
61 #else
62  int raw_pattern = pattern[0] + (pattern[1] << 8);
63 #endif
64  GotoIf(Word32Equal(Word32Or(raw, Int32Constant(0x2020)),
65  Int32Constant(raw_pattern)),
66  target);
67  }

69  TNode<BoolT> IsNonAlpha(TNode<Uint8T> character) {
70  return Uint32GreaterThan(

72  Int32Constant('z' - 'a'));

124 void IntlBuiltinsAssembler::ToLowerCaseImpl(
125  TNode<String> string, TNode<Object> maybe_locales, TNode<Context> context,
126  ToLowerCaseKind kind, std::function<void(TNode<JSAny>)> ReturnFct) {
127  Label call_c(this), return_string(this), runtime(this, Label::kDeferred);
128 
129  // Unpack strings if possible, and bail to runtime unless we get a one-byte
130  // flat string.
131  ToDirectStringAssembler to_direct(
132  state(), string, ToDirectStringAssembler::kDontUnpackSlicedStrings);
133  to_direct.TryToDirect(&runtime);
134 
135  if (kind == ToLowerCaseKind::kToLocaleLowerCase) {
136  Label fast(this), check_locale(this);
137  // Check for fast locales.
138  GotoIf(IsUndefined(maybe_locales), &fast);
139  // Passing a Smi as locales requires performing a ToObject conversion
140  // followed by reading the length property and the "indexed" properties of
141  // it until a valid locale is found.
142  GotoIf(TaggedIsSmi(maybe_locales), &runtime);
143  GotoIfNot(IsString(CAST(maybe_locales)), &runtime);
144  GotoIfNot(IsSeqOneByteString(CAST(maybe_locales)), &runtime);
145  TNode<SeqOneByteString> locale = CAST(maybe_locales);
146  TNode<Uint32T> locale_length = LoadStringLengthAsWord32(locale);
147  GotoIf(Int32LessThan(locale_length, Int32Constant(2)), &runtime);
148  GotoIf(IsNonAlpha(GetChar(locale, 0)), &runtime);
149  GotoIf(IsNonAlpha(GetChar(locale, 1)), &runtime);
150  GotoIf(Word32Equal(locale_length, Int32Constant(2)), &check_locale);
151  GotoIf(Word32NotEqual(locale_length, Int32Constant(5)), &runtime);
152  GotoIf(Word32NotEqual(GetChar(locale, 2), Int32Constant('-')), &runtime);
153  GotoIf(IsNonAlpha(GetChar(locale, 3)), &runtime);
154  GotoIf(IsNonAlpha(GetChar(locale, 4)), &runtime);
155  Goto(&check_locale);
156 
157  Bind(&check_locale);
158  JumpIfStartsWithIgnoreCase(locale, "az", &runtime);
159  JumpIfStartsWithIgnoreCase(locale, "el", &runtime);
160  JumpIfStartsWithIgnoreCase(locale, "lt", &runtime);
161  JumpIfStartsWithIgnoreCase(locale, "tr", &runtime);
162  Goto(&fast);
163 
164  Bind(&fast);
165  }
166 
167  // Early exit on empty string.
168  const TNode<Uint32T> length = LoadStringLengthAsWord32(string);
169  GotoIf(Word32Equal(length, Uint32Constant(0)), &return_string);
170 
171  const TNode<BoolT> is_one_byte = to_direct.IsOneByte();
172  GotoIfNot(is_one_byte, &runtime);
173 
174  // For short strings, do the conversion in CSA through the lookup table.
175 
176  const TNode<String> dst = AllocateSeqOneByteString(length);
177 
178  const int kMaxShortStringLength = 24; // Determined empirically.
179  GotoIf(Uint32GreaterThan(length, Uint32Constant(kMaxShortStringLength)),
180  &call_c);
181 
182  {
183  const TNode<IntPtrT> dst_ptr = PointerToSeqStringData(dst);
184  TVARIABLE(IntPtrT, var_cursor, IntPtrConstant(0));
185 
186  const TNode<IntPtrT> start_address =
187  ReinterpretCast<IntPtrT>(to_direct.PointerToData(&call_c));
188  const TNode<IntPtrT> end_address =
189  Signed(IntPtrAdd(start_address, ChangeUint32ToWord(length)));
190 
191  const TNode<ExternalReference> to_lower_table_addr =
192  ExternalConstant(ExternalReference::intl_to_latin1_lower_table());
193 
194  TVARIABLE(Word32T, var_did_change, Int32Constant(0));
195 
196  VariableList push_vars({&var_cursor, &var_did_change}, zone());
197  BuildFastLoop<IntPtrT>(
198  push_vars, start_address, end_address,
199  [&](TNode<IntPtrT> current) {
200  TNode<Uint8T> c = Load<Uint8T>(current);
201  TNode<Uint8T> lower =
202  Load<Uint8T>(to_lower_table_addr, ChangeInt32ToIntPtr(c));
203  StoreNoWriteBarrier(MachineRepresentation::kWord8, dst_ptr,
204  var_cursor.value(), lower);
205 
206  var_did_change =
207  Word32Or(Word32NotEqual(c, lower), var_did_change.value());
208 
209  Increment(&var_cursor);
210  },
211  kCharSize, LoopUnrollingMode::kNo, IndexAdvanceMode::kPost);
212 
213  // Return the original string if it remained unchanged in order to preserve
214  // e.g. internalization and private symbols (such as the preserved object
215  // hash) on the source string.
216  GotoIfNot(var_did_change.value(), &return_string);
217 
218  ReturnFct(dst);
219  }
220 
221  // Call into C for case conversion. The signature is:
222  // String ConvertOneByteToLower(String src, String dst);
223  BIND(&call_c);
224  {
225  const TNode<String> src = to_direct.string();
226 
227  const TNode<ExternalReference> function_addr =
228  ExternalConstant(ExternalReference::intl_convert_one_byte_to_lower());
229 
230  MachineType type_tagged = MachineType::AnyTagged();
231 
232  const TNode<String> result = CAST(CallCFunction(
233  function_addr, type_tagged, std::make_pair(type_tagged, src),
234  std::make_pair(type_tagged, dst)));
235 
236  ReturnFct(result);
237  }
238 
239  BIND(&return_string);
240  ReturnFct(string);
241 
242  BIND(&runtime);
243  if (kind == ToLowerCaseKind::kToLocaleLowerCase) {
244  ReturnFct(CallRuntime<JSAny>(Runtime::kStringToLocaleLowerCase, context,
245  string, maybe_locales));
246  } else {
247  DCHECK_EQ(kind, ToLowerCaseKind::kToLowerCase);
248  ReturnFct(CallRuntime<JSAny>(Runtime::kStringToLowerCaseIntl,
249  NoContextConstant(), string));
250  }
251 }

253 void IntlBuiltinsAssembler::ListFormatCommon(TNode<Context> context,
254  TNode<Int32T> argc,
255  Runtime::FunctionId format_func_id,
256  const char* method_name) {
257  CodeStubArguments args(this, argc);
258 
259  // Label has_list(this);
260  // 1. Let lf be this value.
261  // 2. If Type(lf) is not Object, throw a TypeError exception.
262  TNode<Object> receiver = args.GetReceiver();
263 
264  // 3. If lf does not have an [[InitializedListFormat]] internal slot, throw a
265  // TypeError exception.
266  ThrowIfNotInstanceType(context, receiver, JS_LIST_FORMAT_TYPE, method_name);
267  TNode<JSListFormat> list_format = CAST(receiver);
268 
269  TNode<Object> list = args.GetOptionalArgumentValue(0);
270  {
271  // 4. Let stringList be ? StringListFromIterable(list).
272  TNode<Object> string_list =
273  CallBuiltin(Builtin::kStringListFromIterable, context, list);
274 
275  // 6. Return ? FormatList(lf, stringList).
276  args.PopAndReturn(
277  CallRuntime<JSAny>(format_func_id, context, list_format, string_list));
278  }
279 }
