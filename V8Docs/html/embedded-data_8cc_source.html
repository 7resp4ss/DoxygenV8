16 Builtin EmbeddedData::TryLookupCode(Address address) const {
17  if (!IsInCodeRange(address)) return Builtin::kNoBuiltinId;
18 
19  // Note: Addresses within the padding section between builtins (i.e. within
20  // start + size <= address < start + padded_size) are interpreted as belonging
21  // to the preceding builtin.
22  uint32_t offset =
23  static_cast<uint32_t>(address - reinterpret_cast<Address>(RawCode()));
24 
25  const struct BuiltinLookupEntry* start =
26  BuiltinLookupEntry(static_cast<ReorderedBuiltinIndex>(0));
27  const struct BuiltinLookupEntry* end = start + kTableSize;
28  const struct BuiltinLookupEntry* desc =
29  std::upper_bound(start, end, offset,
30  [](uint32_t o, const struct BuiltinLookupEntry& desc) {
31  return o < desc.end_offset;
32  });
33  Builtin builtin = static_cast<Builtin>(desc->builtin_id);
34  DCHECK_LT(address,
35  InstructionStartOf(builtin) + PaddedInstructionSizeOf(builtin));
36  DCHECK_GE(address, InstructionStartOf(builtin));
37  return builtin;
38 }

52 bool OffHeapInstructionStream::TryGetAddressForHashing(
53  Isolate* isolate, Address address, uint32_t* hashable_address) {
54  // Mksnapshot calls this while the embedded blob is not available yet.
55  if (isolate->embedded_blob_code() == nullptr) return false;
56  DCHECK_NOT_NULL(Isolate::CurrentEmbeddedBlobCode());
57 
58  EmbeddedData d = EmbeddedData::FromBlob(isolate);
59  if (d.IsInCodeRange(address)) {
60  *hashable_address = d.AddressForHashing(address);
61  return true;
62  }
63 
64  if (isolate->is_short_builtin_calls_enabled()) {
65  d = EmbeddedData::FromBlob();
66  if (d.IsInCodeRange(address)) {
67  *hashable_address = d.AddressForHashing(address);
68  return true;
69  }
70  }
71  return false;

74 // static
75 Builtin OffHeapInstructionStream::TryLookupCode(Isolate* isolate,
76  Address address) {
77  // Mksnapshot calls this while the embedded blob is not available yet.
78  if (isolate->embedded_blob_code() == nullptr) return Builtin::kNoBuiltinId;
79  DCHECK_NOT_NULL(Isolate::CurrentEmbeddedBlobCode());
80 
81  Builtin builtin = EmbeddedData::FromBlob(isolate).TryLookupCode(address);
82 
83  if (isolate->is_short_builtin_calls_enabled() &&
84  !Builtins::IsBuiltinId(builtin)) {
85  builtin = EmbeddedData::FromBlob().TryLookupCode(address);
86  }
87 
88 #ifdef V8_COMPRESS_POINTERS_IN_SHARED_CAGE
89  if (V8_SHORT_BUILTIN_CALLS_BOOL && !Builtins::IsBuiltinId(builtin)) {
90  // When shared pointer compression cage is enabled and it has the embedded
91  // code blob copy then it could have been used regardless of whether the
92  // isolate uses it or knows about it or not (see
93  // InstructionStream::OffHeapInstructionStart()).
94  // So, this blob has to be checked too.
95  CodeRange* code_range = IsolateGroup::current()->GetCodeRange();
96  if (code_range && code_range->embedded_blob_code_copy() != nullptr) {
97  builtin = EmbeddedData::FromBlob(code_range).TryLookupCode(address);
98  }
99  }
100 #endif
101  return builtin;
102 }

104 // static
105 void OffHeapInstructionStream::CreateOffHeapOffHeapInstructionStream(
106  Isolate* isolate, uint8_t** code, uint32_t* code_size, uint8_t** data,
107  uint32_t* data_size) {
108  // Create the embedded blob from scratch using the current Isolate's heap.
109  EmbeddedData d = EmbeddedData::NewFromIsolate(isolate);
110 
111  // Allocate the backing store that will contain the embedded blob in this
112  // Isolate. The backing store is on the native heap, *not* on V8's garbage-
113  // collected heap.
114  v8::PageAllocator* page_allocator = v8::internal::GetPlatformPageAllocator();
115  const uint32_t alignment =
116  static_cast<uint32_t>(page_allocator->AllocatePageSize());
117 
118  void* const requested_allocation_code_address =
119  AlignedAddress(isolate->heap()->GetRandomMmapAddr(), alignment);
120  const uint32_t allocation_code_size = RoundUp(d.code_size(), alignment);
121  uint8_t* allocated_code_bytes = static_cast<uint8_t*>(
122  AllocatePages(page_allocator, allocation_code_size, alignment,
123  PageAllocator::kReadWrite,
124  v8::PageAllocator::AllocationHint().WithAddress(
125  requested_allocation_code_address)));
126  CHECK_NOT_NULL(allocated_code_bytes);
127 
128  void* const requested_allocation_data_address =
129  AlignedAddress(isolate->heap()->GetRandomMmapAddr(), alignment);
130  const uint32_t allocation_data_size = RoundUp(d.data_size(), alignment);
131  uint8_t* allocated_data_bytes = static_cast<uint8_t*>(
132  AllocatePages(page_allocator, allocation_data_size, alignment,
133  PageAllocator::kReadWrite,
134  v8::PageAllocator::AllocationHint().WithAddress(
135  requested_allocation_data_address)));
136  CHECK_NOT_NULL(allocated_data_bytes);
137 
138  // Copy the embedded blob into the newly allocated backing store. Switch
139  // permissions to read-execute since builtin code is immutable from now on
140  // and must be executable in case any JS execution is triggered.
141  //
142  // Once this backing store is set as the current_embedded_blob, V8 cannot tell
143  // the difference between a 'real' embedded build (where the blob is embedded
144  // in the binary) and what we are currently setting up here (where the blob is
145  // on the native heap).
146  std::memcpy(allocated_code_bytes, d.code(), d.code_size());
147  if (v8_flags.experimental_flush_embedded_blob_icache) {
148  FlushInstructionCache(allocated_code_bytes, d.code_size());
149  }
150  CHECK(SetPermissions(page_allocator, allocated_code_bytes,
151  allocation_code_size, PageAllocator::kReadExecute));
152 
153  std::memcpy(allocated_data_bytes, d.data(), d.data_size());
154  CHECK(SetPermissions(page_allocator, allocated_data_bytes,
155  allocation_data_size, PageAllocator::kRead));
156 
157  *code = allocated_code_bytes;
158  *code_size = d.code_size();
159  *data = allocated_data_bytes;
160  *data_size = d.data_size();
161 
162  d.Dispose();
163 }

177 void FinalizeEmbeddedCodeTargets(Isolate* isolate, EmbeddedData* blob) {
178  static const int kRelocMask =
179  RelocInfo::ModeMask(RelocInfo::CODE_TARGET) |
180  RelocInfo::ModeMask(RelocInfo::RELATIVE_CODE_TARGET);
181 
182  static_assert(Builtins::kAllBuiltinsAreIsolateIndependent);
183  for (Builtin builtin = Builtins::kFirst; builtin <= Builtins::kLast;
184  ++builtin) {
185  Tagged<Code> code = isolate->builtins()->code(builtin);
186  RelocIterator on_heap_it(code, kRelocMask);
187  RelocIterator off_heap_it(blob, code, kRelocMask);
188 
189 #if defined(V8_TARGET_ARCH_X64) || defined(V8_TARGET_ARCH_ARM64) || \
190  defined(V8_TARGET_ARCH_ARM) || defined(V8_TARGET_ARCH_IA32) || \
191  defined(V8_TARGET_ARCH_S390X) || defined(V8_TARGET_ARCH_RISCV64) || \
192  defined(V8_TARGET_ARCH_LOONG64) || defined(V8_TARGET_ARCH_RISCV32)
193  // On these platforms we emit relative builtin-to-builtin
194  // jumps for isolate independent builtins in the snapshot. This fixes up the
195  // relative jumps to the right offsets in the snapshot.
196  // See also: InstructionStream::IsIsolateIndependent.
197  while (!on_heap_it.done()) {
198  DCHECK(!off_heap_it.done());
199 
200  RelocInfo* rinfo = on_heap_it.rinfo();
201  DCHECK_EQ(rinfo->rmode(), off_heap_it.rinfo()->rmode());
202  Tagged<Code> target_code =
203  Code::FromTargetAddress(rinfo->target_address());
204  CHECK(Builtins::IsIsolateIndependentBuiltin(target_code));
205 
206  // Do not emit write-barrier for off-heap writes.
207  off_heap_it.rinfo()->set_off_heap_target_address(
208  blob->InstructionStartOf(target_code->builtin_id()));
209 
210  on_heap_it.next();
211  off_heap_it.next();
212  }
213  DCHECK(off_heap_it.done());
214 #else
215  // Architectures other than x64 and arm/arm64 do not use pc-relative calls
216  // and thus must not contain embedded code targets. Instead, we use an
217  // indirection through the root register.
218  CHECK(on_heap_it.done());
219  CHECK(off_heap_it.done());
220 #endif
221  }
222 }

224 void EnsureRelocatable(Tagged<Code> code) {
225  if (code->relocation_size() == 0) return;
226 
227  // On some architectures (arm) the builtin might have a non-empty reloc
228  // info containing a CONST_POOL entry. These entries don't have to be
229  // updated when InstructionStream object is relocated, so it's safe to drop
230  // the reloc info alltogether. If it wasn't the case then we'd have to store
231  // it in the metadata.
232  for (RelocIterator it(code); !it.done(); it.next()) {
233  CHECK_EQ(it.rinfo()->rmode(), RelocInfo::CONST_POOL);
234  }
235 }

239 // static
240 EmbeddedData EmbeddedData::NewFromIsolate(Isolate* isolate) {
241  Builtins* builtins = isolate->builtins();
242 
243  // Store instruction stream lengths and offsets.
244  std::vector<struct LayoutDescription> layout_descriptions(kTableSize);
245  std::vector<struct BuiltinLookupEntry> offset_descriptions(kTableSize);
246 
247  bool saw_unsafe_builtin = false;
248  uint32_t raw_code_size = 0;
249  uint32_t raw_data_size = 0;
250  static_assert(Builtins::kAllBuiltinsAreIsolateIndependent);
251 
252  std::vector<Builtin> reordered_builtins;
253  if (v8_flags.reorder_builtins &&
254  BuiltinsCallGraph::Get()->all_hash_matched()) {
255  DCHECK(v8_flags.turbo_profiling_input.value());
256  // TODO(ishell, v8:13938): avoid the binary size overhead for non-mksnapshot
257  // binaries.
258  BuiltinsSorter sorter;
259  std::vector<uint32_t> builtin_sizes;
260  for (Builtin i = Builtins::kFirst; i <= Builtins::kLast; ++i) {
261  Tagged<Code> code = builtins->code(i);
262  uint32_t instruction_size =
263  static_cast<uint32_t>(code->instruction_size());
264  uint32_t padding_size = PadAndAlignCode(instruction_size);
265  builtin_sizes.push_back(padding_size);
266  }
267  reordered_builtins = sorter.SortBuiltins(
268  v8_flags.turbo_profiling_input.value(), builtin_sizes);
269  CHECK_EQ(reordered_builtins.size(), Builtins::kBuiltinCount);
270  }
271 
272  for (ReorderedBuiltinIndex embedded_index = 0;
273  embedded_index < Builtins::kBuiltinCount; embedded_index++) {
274  Builtin builtin;
275  if (reordered_builtins.empty()) {
276  builtin = static_cast<Builtin>(embedded_index);
277  } else {
278  builtin = reordered_builtins[embedded_index];
279  }
280  Tagged<Code> code = builtins->code(builtin);
281 
282  // Sanity-check that the given builtin is isolate-independent.
283  if (!code->IsIsolateIndependent(isolate)) {
284  saw_unsafe_builtin = true;
285  fprintf(stderr, "%s is not isolate-independent.\n",
286  Builtins::name(builtin));
287  }
288 
289  uint32_t instruction_size = static_cast<uint32_t>(code->instruction_size());
290  DCHECK_EQ(0, raw_code_size % kCodeAlignment);
291  {
292  // We use builtin id as index in layout_descriptions.
293  const int builtin_id = static_cast<int>(builtin);
294  struct LayoutDescription& layout_desc = layout_descriptions[builtin_id];
295  layout_desc.instruction_offset = raw_code_size;
296  layout_desc.instruction_length = instruction_size;
297  layout_desc.metadata_offset = raw_data_size;
298  }
299  // Align the start of each section.
300  raw_code_size += PadAndAlignCode(instruction_size);
301  raw_data_size += PadAndAlignData(code->metadata_size());
302 
303  {
304  // We use embedded index as index in offset_descriptions.
305  struct BuiltinLookupEntry& offset_desc =
306  offset_descriptions[embedded_index];
307  offset_desc.end_offset = raw_code_size;
308  offset_desc.builtin_id = static_cast<uint32_t>(builtin);
309  }
310  }
311  CHECK_WITH_MSG(
312  !saw_unsafe_builtin,
313  "One or more builtins marked as isolate-independent either contains "
314  "isolate-dependent code or aliases the off-heap trampoline register. "
315  "If in doubt, ask jgruber@");
316 
317  // Allocate space for the code section, value-initialized to 0.
318  static_assert(RawCodeOffset() == 0);
319  const uint32_t blob_code_size = RawCodeOffset() + raw_code_size;
320  uint8_t* const blob_code = new uint8_t[blob_code_size]();
321 
322  // Allocate space for the data section, value-initialized to 0.
323  static_assert(
324  IsAligned(FixedDataSize(), InstructionStream::kMetadataAlignment));
325  const uint32_t blob_data_size = FixedDataSize() + raw_data_size;
326  uint8_t* const blob_data = new uint8_t[blob_data_size]();
327 
328  // Initially zap the entire blob, effectively padding the alignment area
329  // between two builtins with int3's (on x64/ia32).
330  ZapCode(reinterpret_cast<Address>(blob_code), blob_code_size);
331 
332  // Hash relevant parts of the Isolate's heap and store the result.
333  {
334  static_assert(IsolateHashSize() == kSizetSize);
335  const size_t hash = isolate->HashIsolateForEmbeddedBlob();
336  std::memcpy(blob_data + IsolateHashOffset(), &hash, IsolateHashSize());
337  }
338 
339  // Write the layout_descriptions tables.
340  DCHECK_EQ(LayoutDescriptionTableSize(),
341  sizeof(layout_descriptions[0]) * layout_descriptions.size());
342  std::memcpy(blob_data + LayoutDescriptionTableOffset(),
343  layout_descriptions.data(), LayoutDescriptionTableSize());
344 
345  // Write the builtin_offset_descriptions tables.
346  DCHECK_EQ(BuiltinLookupEntryTableSize(),
347  sizeof(offset_descriptions[0]) * offset_descriptions.size());
348  std::memcpy(blob_data + BuiltinLookupEntryTableOffset(),
349  offset_descriptions.data(), BuiltinLookupEntryTableSize());
350 
351  // .. and the variable-size data section.
352  uint8_t* const raw_metadata_start = blob_data + RawMetadataOffset();
353  static_assert(Builtins::kAllBuiltinsAreIsolateIndependent);
354  for (Builtin builtin = Builtins::kFirst; builtin <= Builtins::kLast;
355  ++builtin) {
356  Tagged<Code> code = builtins->code(builtin);
357  uint32_t offset =
358  layout_descriptions[static_cast<int>(builtin)].metadata_offset;
359  uint8_t* dst = raw_metadata_start + offset;
360  DCHECK_LE(RawMetadataOffset() + offset + code->metadata_size(),
361  blob_data_size);
362  std::memcpy(dst, reinterpret_cast<uint8_t*>(code->metadata_start()),
363  code->metadata_size());
364  }
365  CHECK_IMPLIES(
366  kMaxPCRelativeCodeRangeInMB,
367  static_cast<size_t>(raw_code_size) <= kMaxPCRelativeCodeRangeInMB * MB);
368 
369  // .. and the variable-size code section.
370  uint8_t* const raw_code_start = blob_code + RawCodeOffset();
371  static_assert(Builtins::kAllBuiltinsAreIsolateIndependent);
372  for (Builtin builtin = Builtins::kFirst; builtin <= Builtins::kLast;
373  ++builtin) {
374  Tagged<Code> code = builtins->code(builtin);
375  uint32_t offset =
376  layout_descriptions[static_cast<int>(builtin)].instruction_offset;
377  uint8_t* dst = raw_code_start + offset;
378  DCHECK_LE(RawCodeOffset() + offset + code->instruction_size(),
379  blob_code_size);
380  std::memcpy(dst, reinterpret_cast<uint8_t*>(code->instruction_start()),
381  code->instruction_size());
382  }
383 
384  EmbeddedData d(blob_code, blob_code_size, blob_data, blob_data_size);
385 
386  // Fix up call targets that point to other embedded builtins.
387  FinalizeEmbeddedCodeTargets(isolate, &d);
388 
389  // Hash the blob and store the result.
390  {
391  static_assert(EmbeddedBlobDataHashSize() == kSizetSize);
392  const size_t data_hash = d.CreateEmbeddedBlobDataHash();
393  std::memcpy(blob_data + EmbeddedBlobDataHashOffset(), &data_hash,
394  EmbeddedBlobDataHashSize());
395 
396  static_assert(EmbeddedBlobCodeHashSize() == kSizetSize);
397  const size_t code_hash = d.CreateEmbeddedBlobCodeHash();
398  std::memcpy(blob_data + EmbeddedBlobCodeHashOffset(), &code_hash,
399  EmbeddedBlobCodeHashSize());
400 
401  DCHECK_EQ(data_hash, d.CreateEmbeddedBlobDataHash());
402  DCHECK_EQ(data_hash, d.EmbeddedBlobDataHash());
403  DCHECK_EQ(code_hash, d.CreateEmbeddedBlobCodeHash());
404  DCHECK_EQ(code_hash, d.EmbeddedBlobCodeHash());
405  }
406 
407  if (DEBUG_BOOL) {
408  for (Builtin builtin = Builtins::kFirst; builtin <= Builtins::kLast;
409  ++builtin) {
410  Tagged<Code> code = builtins->code(builtin);
411  CHECK_EQ(d.InstructionSizeOf(builtin), code->instruction_size());
412  }
413  }
414 
415  // Ensure that InterpreterEntryTrampolineForProfiling is relocatable.
416  // See v8_flags.interpreted_frames_native_stack for details.
417  EnsureRelocatable(
418  builtins->code(Builtin::kInterpreterEntryTrampolineForProfiling));
419 
420  if (v8_flags.serialization_statistics) d.PrintStatistics();
421 
422  return d;
423 }

425 size_t EmbeddedData::CreateEmbeddedBlobDataHash() const {
426  static_assert(EmbeddedBlobDataHashOffset() == 0);
427  static_assert(EmbeddedBlobCodeHashOffset() == EmbeddedBlobDataHashSize());
428  static_assert(IsolateHashOffset() ==

434  data_size_ - kFirstHashedDataOffset);
435  return Checksum(payload);

450 void EmbeddedData::PrintStatistics() const {
451  DCHECK(v8_flags.serialization_statistics);
452 
453  constexpr int kCount = Builtins::kBuiltinCount;
454  int sizes[kCount];
455  static_assert(Builtins::kAllBuiltinsAreIsolateIndependent);
456  for (int i = 0; i < kCount; i++) {
457  sizes[i] = InstructionSizeOf(Builtins::FromInt(i));
458  }
459 
460  // Sort for percentiles.
461  std::sort(&sizes[0], &sizes[kCount]);
462 
463  const int k50th = kCount * 0.5;
464  const int k75th = kCount * 0.75;
465  const int k90th = kCount * 0.90;
466  const int k99th = kCount * 0.99;
467 
468  PrintF("EmbeddedData:\n");
469  PrintF(" Total size: %d\n",
470  static_cast<int>(code_size() + data_size()));
471  PrintF(" Data size: %d\n", static_cast<int>(data_size()));
472  PrintF(" Code size: %d\n", static_cast<int>(code_size()));
473  PrintF(" Instruction size (50th percentile): %d\n", sizes[k50th]);
474  PrintF(" Instruction size (75th percentile): %d\n", sizes[k75th]);
475  PrintF(" Instruction size (90th percentile): %d\n", sizes[k90th]);
476  PrintF(" Instruction size (99th percentile): %d\n", sizes[k99th]);
477  PrintF("\n");
