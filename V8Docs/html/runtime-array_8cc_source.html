37 RUNTIME_FUNCTION(Runtime_NewArray) {
38  HandleScope scope(isolate);
39  DCHECK_LE(3, args.length());
40  int const argc = args.length() - 3;
41  // argv points to the arguments constructed by the JavaScript call.
42  JavaScriptArguments argv(argc, args.address_of_arg_at(0));
43  DirectHandle<JSFunction> constructor = args.at<JSFunction>(argc);
44  DirectHandle<JSReceiver> new_target = args.at<JSReceiver>(argc + 1);
45  Handle<HeapObject> type_info = args.at<HeapObject>(argc + 2);
46  // TODO(bmeurer): Use MaybeHandle to pass around the AllocationSite.
47  DirectHandle<AllocationSite> site = IsAllocationSite(*type_info)
48  ? Cast<AllocationSite>(type_info)
49  : Handle<AllocationSite>::null();
50 
51  Factory* factory = isolate->factory();
52 
53  // If called through new, new.target can be:
54  // - a subclass of constructor,
55  // - a proxy wrapper around constructor, or
56  // - the constructor itself.
57  // If called through Reflect.construct, it's guaranteed to be a constructor by
58  // REFLECT_CONSTRUCT_PREPARE.
59  DCHECK(IsConstructor(*new_target));
60 
61  bool holey = false;
62  bool can_use_type_feedback = !site.is_null();
63  bool can_inline_array_constructor = true;
64  if (argv.length() == 1) {
65  DirectHandle<Object> argument_one = argv.at<Object>(0);
66  if (IsSmi(*argument_one)) {
67  int value = Cast<Smi>(*argument_one).value();
68  if (value < 0 ||
69  JSArray::SetLengthWouldNormalize(isolate->heap(), value)) {
70  // the array is a dictionary in this case.
71  can_use_type_feedback = false;
72  } else if (value != 0) {
73  holey = true;
74  if (value >= JSArray::kInitialMaxFastElementArray) {
75  can_inline_array_constructor = false;
76  }
77  }
78  } else {
79  // Non-smi length argument produces a dictionary
80  can_use_type_feedback = false;
81  }
82  }
83 
84  DirectHandle<Map> initial_map;
85  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
86  isolate, initial_map,
87  JSFunction::GetDerivedMap(isolate, constructor, new_target));
88 
89  ElementsKind to_kind = can_use_type_feedback ? site->GetElementsKind()
90  : initial_map->elements_kind();
91  if (holey && !IsHoleyElementsKind(to_kind)) {
92  to_kind = GetHoleyElementsKind(to_kind);
93  // Update the allocation site info to reflect the advice alteration.
94  if (!site.is_null()) site->SetElementsKind(to_kind);
95  }
96 
97  // We should allocate with an initial map that reflects the allocation site
98  // advice. Therefore we use AllocateJSObjectFromMap instead of passing
99  // the constructor.
100  initial_map = Map::AsElementsKind(isolate, initial_map, to_kind);
101 
102  // If we don't care to track arrays of to_kind ElementsKind, then
103  // don't emit a memento for them.
104  DirectHandle<AllocationSite> allocation_site;
105  if (AllocationSite::ShouldTrack(to_kind)) {
106  allocation_site = site;
107  }
108 
109  DirectHandle<JSArray> array = Cast<JSArray>(factory->NewJSObjectFromMap(
110  initial_map, AllocationType::kYoung, allocation_site));
111 
112  factory->NewJSArrayStorage(
113  array, 0, 0, ArrayStorageAllocationMode::DONT_INITIALIZE_ARRAY_ELEMENTS);
114 
115  ElementsKind old_kind = array->GetElementsKind();
116  RETURN_FAILURE_ON_EXCEPTION(
117  isolate, ArrayConstructInitializeElements(isolate, array, &argv));
118  // If we have an allocation site, and the array constructor can't be inlined,
119  // mark this on the allocation site. If there's no allocation site yet, the
120  // optimized code will eventually optimistically try to inline and worst case
121  // will deopt and set the allocation site itself, or set the CallIC disable
122  // speculation bit.
123  if (!site.is_null()) {
124  if ((old_kind != array->GetElementsKind() || !can_use_type_feedback ||
125  !can_inline_array_constructor)) {
126  // The arguments passed in caused a transition. This kind of complexity
127  // can't be dealt with in the inlined optimized array constructor case.
128  // We must mark the allocationsite as un-inlinable.
129  site->SetDoNotInlineCall();
130  }
131  }
132 
133  return *array;
134 }

146 // GrowArrayElements grows fast kind elements and returns a sentinel Smi if the
147 // object was normalized or if the key is negative.
148 RUNTIME_FUNCTION(Runtime_GrowArrayElements) {
149  HandleScope scope(isolate);
150  DCHECK_EQ(2, args.length());
151  DirectHandle<JSObject> object = args.at<JSObject>(0);
152  DirectHandle<Object> key = args.at(1);
153  ElementsKind kind = object->GetElementsKind();
154  CHECK(IsFastElementsKind(kind));
155  uint32_t index;
156  if (IsSmi(*key)) {
157  int value = Smi::ToInt(*key);
158  if (value < 0) return Smi::zero();
159  index = static_cast<uint32_t>(value);
160  } else {
161  CHECK(IsHeapNumber(*key));
162  double value = Cast<HeapNumber>(*key)->value();
163  if (value < 0 || value > std::numeric_limits<uint32_t>::max()) {
164  return Smi::zero();
165  }
166  index = static_cast<uint32_t>(value);
167  }
168 
169  uint32_t capacity = static_cast<uint32_t>(object->elements()->length());
170 
171  if (index >= capacity) {
172  bool has_grown;
173  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
174  isolate, has_grown,
175  object->GetElementsAccessor()->GrowCapacity(isolate, object, index));
176  if (!has_grown) {
177  return Smi::zero();
178  }
179  }
180 
181  return object->elements();
182 }

210 RUNTIME_FUNCTION(Runtime_ArrayIncludes_Slow) {
211  HandleScope shs(isolate);
212  DCHECK_EQ(3, args.length());
213  DirectHandle<Object> search_element = args.at(1);
214  Handle<Object> from_index = args.at(2);
215 
216  // Let O be ? ToObject(this value).
217  DirectHandle<JSReceiver> object;
218  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
219  isolate, object,
220  Object::ToObject(isolate, Handle<Object>(args[0], isolate)));
221 
222  // Let len be ? ToLength(? Get(O, "length")).
223  int64_t len;
224  {
225  if (object->map()->instance_type() == JS_ARRAY_TYPE) {
226  uint32_t len32 = 0;
227  bool success =
228  Object::ToArrayLength(Cast<JSArray>(*object)->length(), &len32);
229  DCHECK(success);
230  USE(success);
231  len = len32;
232  } else {
233  DirectHandle<Object> len_;
234  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
235  isolate, len_,
236  Object::GetProperty(isolate, object,
237  isolate->factory()->length_string()));
238 
239  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, len_,
240  Object::ToLength(isolate, len_));
241  len = static_cast<int64_t>(Object::NumberValue(*len_));
242  DCHECK_EQ(len, Object::NumberValue(*len_));
243  }
244  }
245 
246  if (len == 0) return ReadOnlyRoots(isolate).false_value();
247 
248  // Let n be ? ToInteger(fromIndex). (If fromIndex is undefined, this step
249  // produces the value 0.)
250  int64_t index = 0;
251  if (!IsUndefined(*from_index, isolate)) {
252  double start_from;
253  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
254  isolate, start_from, Object::IntegerValue(isolate, from_index));
255 
256  if (start_from >= len) return ReadOnlyRoots(isolate).false_value();
257  if (V8_LIKELY(std::isfinite(start_from))) {
258  if (start_from < 0) {
259  index = static_cast<int64_t>(std::max<double>(start_from + len, 0));
260  } else {
261  index = start_from;
262  }
263  }
264 
265  DCHECK_GE(index, 0);
266  }
267 
268  // If the receiver is not a special receiver type, and the length is a valid
269  // element index, perform fast operation tailored to specific ElementsKinds.
270  if (!IsSpecialReceiverMap(object->map()) &&
271  len <= JSObject::kMaxElementCount &&
272  JSObject::PrototypeHasNoElements(isolate, Cast<JSObject>(*object))) {
273  DirectHandle<JSObject> obj = Cast<JSObject>(object);
274  ElementsAccessor* elements = obj->GetElementsAccessor();
275  Maybe<bool> result =
276  elements->IncludesValue(isolate, obj, search_element, index, len);
277  MAYBE_RETURN(result, ReadOnlyRoots(isolate).exception());
278  return *isolate->factory()->ToBoolean(result.FromJust());
279  }
280 
281  // Otherwise, perform slow lookups for special receiver types.
282  for (; index < len; ++index) {
283  HandleScope iteration_hs(isolate);
284 
285  // Let elementK be the result of ? Get(O, ! ToString(k)).
286  DirectHandle<Object> element_k;
287  {
288  PropertyKey key(isolate, static_cast<double>(index));
289  LookupIterator it(isolate, object, key);
290  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, element_k,
291  Object::GetProperty(&it));
292  }
293 
294  // If SameValueZero(searchElement, elementK) is true, return true.
295  if (Object::SameValueZero(*search_element, *element_k)) {
296  return ReadOnlyRoots(isolate).true_value();
297  }
298  }
299  return ReadOnlyRoots(isolate).false_value();
300 }

302 RUNTIME_FUNCTION(Runtime_ArrayIndexOf) {
303  HandleScope hs(isolate);
304  DCHECK_EQ(3, args.length());
305  DirectHandle<Object> search_element = args.at(1);
306  Handle<Object> from_index = args.at(2);
307 
308  // Let O be ? ToObject(this value).
309  DirectHandle<JSReceiver> object;
310  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
311  isolate, object,
312  Object::ToObject(isolate, args.at(0), "Array.prototype.indexOf"));
313 
314  // Let len be ? ToLength(? Get(O, "length")).
315  int64_t len;
316  {
317  if (IsJSArray(*object)) {
318  uint32_t len32 = 0;
319  bool success =
320  Object::ToArrayLength(Cast<JSArray>(*object)->length(), &len32);
321  DCHECK(success);
322  USE(success);
323  len = len32;
324  } else {
325  DirectHandle<Object> len_;
326  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
327  isolate, len_,
328  Object::GetProperty(isolate, object,
329  isolate->factory()->length_string()));
330 
331  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, len_,
332  Object::ToLength(isolate, len_));
333  len = static_cast<int64_t>(Object::NumberValue(*len_));
334  DCHECK_EQ(len, Object::NumberValue(*len_));
335  }
336  }
337 
338  if (len == 0) return Smi::FromInt(-1);
339 
340  // Let n be ? ToInteger(fromIndex). (If fromIndex is undefined, this step
341  // produces the value 0.)
342  int64_t start_from;
343  {
344  double fp;
345  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
346  isolate, fp, Object::IntegerValue(isolate, from_index));
347  if (fp > len) return Smi::FromInt(-1);
348  if (V8_LIKELY(fp >=
349  static_cast<double>(std::numeric_limits<int64_t>::min()))) {
350  DCHECK(fp < static_cast<double>(std::numeric_limits<int64_t>::max()));
351  start_from = static_cast<int64_t>(fp);
352  } else {
353  start_from = std::numeric_limits<int64_t>::min();
354  }
355  }
356 
357  int64_t index;
358  if (start_from >= 0) {
359  index = start_from;
360  } else {
361  index = len + start_from;
362  if (index < 0) {
363  index = 0;
364  }
365  }
366 
367  // If the receiver is not a special receiver type, and the length fits
368  // uint32_t, perform fast operation tailored to specific ElementsKinds.
369  if (!IsSpecialReceiverMap(object->map()) && len <= kMaxUInt32 &&
370  JSObject::PrototypeHasNoElements(isolate, Cast<JSObject>(*object))) {
371  DirectHandle<JSObject> obj = Cast<JSObject>(object);
372  ElementsAccessor* elements = obj->GetElementsAccessor();
373  Maybe<int64_t> result = elements->IndexOfValue(isolate, obj, search_element,
374  static_cast<uint32_t>(index),
375  static_cast<uint32_t>(len));
376  MAYBE_RETURN(result, ReadOnlyRoots(isolate).exception());
377  return *isolate->factory()->NewNumberFromInt64(result.FromJust());
378  }
379 
380  // Otherwise, perform slow lookups for special receiver types.
381  for (; index < len; ++index) {
382  HandleScope iteration_hs(isolate);
383  // Let elementK be the result of ? Get(O, ! ToString(k)).
384  DirectHandle<Object> element_k;
385  {
386  PropertyKey key(isolate, static_cast<double>(index));
387  LookupIterator it(isolate, object, key);
388  Maybe<bool> present = JSReceiver::HasProperty(&it);
389  MAYBE_RETURN(present, ReadOnlyRoots(isolate).exception());
390  if (!present.FromJust()) continue;
391  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, element_k,
392  Object::GetProperty(&it));
393  if (Object::StrictEquals(*search_element, *element_k)) {
394  return *isolate->factory()->NewNumberFromInt64(index);
395  }
396  }
397  }
398  return Smi::FromInt(-1);
399 }
