40 // Calls to Wasm imports are handled in several different ways, depending on the
41 // type of the target function/callable and whether the signature matches the
42 // argument arity.
43 enum class ImportCallKind : uint8_t {
44  kLinkError, // static Wasm->Wasm type error
45  kRuntimeTypeError, // runtime Wasm->JS type error
46  kWasmToCapi, // fast Wasm->C-API call
47  kWasmToJSFastApi, // fast Wasm->JS Fast API C call
48  kWasmToWasm, // fast Wasm->Wasm call
49  kJSFunction, // fast Wasm->JS call
50  // For everything else, there's the call builtin.
51  kUseCallBuiltin
52 };
