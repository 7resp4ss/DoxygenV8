119 template <bool capture_raw, bool unicode>
120 base::uc32 Scanner::ScanHexNumber(int expected_length) {
121  DCHECK_LE(expected_length, 4); // prevent overflow
122 
123  int begin = source_pos() - 2;
124  base::uc32 x = 0;
125  for (int i = 0; i < expected_length; i++) {
126  int d = base::HexValue(c0_);
127  if (d < 0) {
128  ReportScannerError(Location(begin, begin + expected_length + 2),
129  unicode
130  ? MessageTemplate::kInvalidUnicodeEscapeSequence
131  : MessageTemplate::kInvalidHexEscapeSequence);
132  return Invalid();
133  }
134  x = x * 16 + d;
135  Advance<capture_raw>();
136  }
137 
138  return x;

162 Token::Value Scanner::Next() {
163  // Rotate through tokens.
164  TokenDesc* previous = current_;
165  current_ = next_;
166  // Either we already have the next token lined up, in which case next_next_
167  // simply becomes next_. In that case we use current_ as new next_next_ and
168  // clear its token to indicate that it wasn't scanned yet. Otherwise we use
169  // current_ as next_ and scan into it, leaving next_next_ uninitialized.
170  if (V8_LIKELY(next_next().token == Token::kUninitialized)) {
171  DCHECK(next_next_next().token == Token::kUninitialized);
172  next_ = previous;
173  // User 'previous' instead of 'next_' because for some reason the compiler
174  // thinks 'next_' could be modified before the entry into Scan.
175  previous->after_line_terminator = false;
176  Scan(previous);
177  } else {
178  next_ = next_next_;
179 
180  if (V8_LIKELY(next_next_next().token == Token::kUninitialized)) {
181  next_next_ = previous;
182  } else {
183  next_next_ = next_next_next_;
184  next_next_next_ = previous;
185  }
186 
187  previous->token = Token::kUninitialized;
188  DCHECK_NE(Token::kUninitialized, current().token);
189  }
190  return current().token;
191 }

193 Token::Value Scanner::PeekAhead() {
194  DCHECK(next().token != Token::kDiv);
195  DCHECK(next().token != Token::kAssignDiv);
196 
197  if (next_next().token != Token::kUninitialized) {
198  return next_next().token;
199  }
200  TokenDesc* temp = next_;
201  next_ = next_next_;
202  next().after_line_terminator = false;
203  Scan();
204  next_next_ = next_;
205  next_ = temp;
206  return next_next().token;
207 }

209 Token::Value Scanner::PeekAheadAhead() {
210  if (next_next_next().token != Token::kUninitialized) {
211  return next_next_next().token;
212  }
213  // PeekAhead() must be called first in order to call PeekAheadAhead().
214  DCHECK(next_next().token != Token::kUninitialized);
215  TokenDesc* temp = next_;
216  TokenDesc* temp_next = next_next_;
217  next_ = next_next_next_;

219  Scan();
220  next_next_next_ = next_;
221  next_next_ = temp_next;
222  next_ = temp;
223  return next_next_next().token;
224 }

255 void ProcessPerFunctionCompileHints(const base::Vector<const uint8_t>& data,
256  int current_position,
257  std::vector<int>& positions) {
258  // Compile hints are relative to the position of the comment end.
259  int last_position = current_position;
260  size_t pos = 0;
261  const char* char_data = reinterpret_cast<const char*>(data.data());
262  while (pos < static_cast<size_t>(data.length())) {
263  int32_t delta = base::VLQBase64Decode(char_data, data.length(), &pos);
264  if (delta == std::numeric_limits<int32_t>::min()) {
265  // Invalid data, bail out and clear the data we read so far. (Not using
266  // the data until the invalid portion is consistent with 2-byte data not
267  // being handled at all.)
268  positions.clear();
269  return;
270  }
271  last_position += delta;
272  positions.push_back(last_position);
273  }
274  positions.shrink_to_fit();
275 }

279 void Scanner::TryToParseMagicComment(base::uc32 hash_or_at_sign) {
280  // Magic comments are of the form: //[#@]\s<name>=\s*<value>\s*.* and this
281  // function will just return if it cannot parse a magic comment.
282  DCHECK(!IsWhiteSpaceOrLineTerminator(kEndOfInput));
283  if (!IsWhiteSpace(c0_)) return;
284  Advance();
285  LiteralBuffer name;
286  name.Start();
287 
288  while (c0_ != kEndOfInput && !IsWhiteSpaceOrLineTerminator(c0_) &&
289  c0_ != '=') {
290  name.AddChar(c0_);
291  Advance();
292  }
293  if (!name.is_one_byte()) return;
294  base::Vector<const uint8_t> name_literal = name.one_byte_literal();
295  LiteralBuffer* value;
296  LiteralBuffer per_function_compile_hints_value;
297  if (name_literal == base::StaticOneByteVector("sourceURL")) {
298  value = &source_url_;
299  } else if (name_literal == base::StaticOneByteVector("sourceMappingURL")) {
300  value = &source_mapping_url_;
301  DCHECK(hash_or_at_sign == '#' || hash_or_at_sign == '@');
302  saw_source_mapping_url_magic_comment_at_sign_ = hash_or_at_sign == '@';
303  } else if (!saw_non_comment_ &&
304  name_literal ==
305  base::StaticOneByteVector("allFunctionsCalledOnLoad") &&
306  hash_or_at_sign == '#' && c0_ != '=') {
307  saw_magic_comment_compile_hints_all_ = true;
308  } else if (name_literal ==
309  base::StaticOneByteVector("functionsCalledOnLoad") &&
310  hash_or_at_sign == '#') {
311  value = &per_function_compile_hints_value;
312  } else if (name_literal == base::StaticOneByteVector("debugId") &&
313  hash_or_at_sign == '#') {
314  value = &debug_id_;
315  } else {
316  return;
317  }
318  if (c0_ != '=')
319  return;
320  value->Start();
321  Advance();
322  while (IsWhiteSpace(c0_)) {
323  Advance();
324  }
325  while (c0_ != kEndOfInput && !unibrow::IsLineTerminator(c0_)) {
326  if (IsWhiteSpace(c0_)) {
327  break;
328  }
329  value->AddChar(c0_);
330  Advance();
331  }
332  // Allow whitespace at the end.
333  while (c0_ != kEndOfInput && !unibrow::IsLineTerminator(c0_)) {
334  if (!IsWhiteSpace(c0_)) {
335  value->Start();
336  break;
337  }
338  Advance();
339  }
340  if (value == &per_function_compile_hints_value &&
341  per_function_compile_hints_value.is_one_byte()) {
342  base::Vector<const uint8_t> value_literal =
343  per_function_compile_hints_value.one_byte_literal();
344  per_function_compile_hint_positions_.clear();
345  per_function_compile_hint_positions_idx_ = 0;
346  ProcessPerFunctionCompileHints(value_literal, source_pos(),
347  per_function_compile_hint_positions_);
348  }
349 }

351 bool Scanner::HasPerFunctionCompileHint(int position) {
352  // Allow off-by-<slack> in the compile hints positions, to account for adding
353  // newlines at the end of the comment, function positions being off-by-one,
354  // etc.
355  const int kSlack = 3;
356  while (per_function_compile_hint_positions_idx_ <
357  per_function_compile_hint_positions_.size() &&
358  per_function_compile_hint_positions_
359  [per_function_compile_hint_positions_idx_] <
360  position - kSlack) {
361  ++per_function_compile_hint_positions_idx_;
362  }
363  if (per_function_compile_hint_positions_idx_ >=
364  per_function_compile_hint_positions_.size()) {
365  return false;
366  }
367  int hint_position = per_function_compile_hint_positions_
368  [per_function_compile_hint_positions_idx_];
369  return hint_position >= position - kSlack &&
370  hint_position <= position + kSlack;
371 }

373 Token::Value Scanner::SkipMultiLineComment() {
374  DCHECK_EQ(c0_, '*');
375 
376  // Until we see the first newline, check for * and newline characters.
377  if (!next().after_line_terminator) {
378  do {
379  AdvanceUntil([](base::uc32 c0) {
380  if (V8_UNLIKELY(static_cast<uint32_t>(c0) > kMaxAscii)) {
381  return unibrow::IsLineTerminator(c0);
382  }
383  uint8_t char_flags = character_scan_flags[c0];
384  return MultilineCommentCharacterNeedsSlowPath(char_flags);
385  });
386 
387  while (c0_ == '*') {
388  Advance();
389  if (c0_ == '/') {
390  Advance();
391  return Token::kWhitespace;
392  }
393  }
394 
395  if (unibrow::IsLineTerminator(c0_)) {
396  next().after_line_terminator = true;
397  break;
398  }
399  } while (c0_ != kEndOfInput);
400  }
401 
402  // After we've seen newline, simply try to find '*/'.
403  while (c0_ != kEndOfInput) {
404  AdvanceUntil([](base::uc32 c0) { return c0 == '*'; });
405 
406  while (c0_ == '*') {
407  Advance();
408  if (c0_ == '/') {
409  Advance();
410  return Token::kWhitespace;
411  }
412  }
413  }
414 
415  return Token::kIllegal;
416 }

451 void Scanner::SeekForward(int pos) {
452  // After this call, we will have the token at the given position as
453  // the "next" token. The "current" token will be invalid.
454  if (pos == next().location.beg_pos) return;
455  int current_pos = source_pos();
456  DCHECK_EQ(next().location.end_pos, current_pos);
457  // Positions inside the lookahead token aren't supported.
458  DCHECK(pos >= current_pos);
459  if (pos != current_pos) {
460  source_->Seek(pos);
461  Advance();
462  // This function is only called to seek to the location
463  // of the end of a function (at the "}" token). It doesn't matter
464  // whether there was a line terminator in the part we skip.
465  next().after_line_terminator = false;
466  }
467  Scan();
468 }

470 template <bool capture_raw>
471 bool Scanner::ScanEscape() {
472  base::uc32 c = c0_;
473  Advance<capture_raw>();
474 
475  // Skip escaped newlines.
476  DCHECK(!unibrow::IsLineTerminator(kEndOfInput));
477  if (!capture_raw && unibrow::IsLineTerminator(c)) {
478  // Allow escaped CR+LF newlines in multiline string literals.
479  if (IsCarriageReturn(c) && IsLineFeed(c0_)) Advance();
480  return true;
481  }
482 
483  switch (c) {
484  case 'b' : c = '\b'; break;
485  case 'f' : c = '\f'; break;
486  case 'n' : c = '\n'; break;
487  case 'r' : c = '\r'; break;
488  case 't' : c = '\t'; break;
489  case 'u' : {
490  c = ScanUnicodeEscape<capture_raw>();
491  if (IsInvalid(c)) return false;
492  break;
493  }
494  case 'v':
495  c = '\v';
496  break;
497  case 'x': {
498  c = ScanHexNumber<capture_raw>(2);
499  if (IsInvalid(c)) return false;
500  break;
501  }
502  case '0':
503  case '1':
504  case '2':
505  case '3':
506  case '4':
507  case '5':
508  case '6':
509  case '7':
510  c = ScanOctalEscape<capture_raw>(c, 2);
511  break;
512  case '8':
513  case '9':
514  // '\8' and '\9' are disallowed in strict mode.
515  // Reuse the octal error state to propagate the error.
516  octal_pos_ = Location(source_pos() - 2, source_pos() - 1);
517  octal_message_ = capture_raw ? MessageTemplate::kTemplate8Or9Escape
518  : MessageTemplate::kStrict8Or9Escape;
519  break;
520  }
521 
522  // Other escaped characters are interpreted as their non-escaped version.
523  AddLiteralChar(c);
524  return true;
525 }

527 template <bool capture_raw>
528 base::uc32 Scanner::ScanOctalEscape(base::uc32 c, int length) {
529  DCHECK('0' <= c && c <= '7');
530  base::uc32 x = c - '0';
531  int i = 0;
532  for (; i < length; i++) {
533  int d = c0_ - '0';
534  if (d < 0 || d > 7) break;
535  int nx = x * 8 + d;
536  if (nx >= 256) break;
537  x = nx;
538  Advance<capture_raw>();
539  }
540  // Anything except '\0' is an octal escape sequence, illegal in strict mode.
541  // Remember the position of octal escape sequences so that an error
542  // can be reported later (in strict mode).
543  // We don't report the error immediately, because the octal escape can
544  // occur before the "use strict" directive.
545  if (c != '0' || i > 0 || IsNonOctalDecimalDigit(c0_)) {
546  octal_pos_ = Location(source_pos() - i - 1, source_pos() - 1);
547  octal_message_ = capture_raw ? MessageTemplate::kTemplateOctalLiteral
548  : MessageTemplate::kStrictOctalEscape;
549  }
550  return x;
551 }

553 Token::Value Scanner::ScanString() {
554  base::uc32 quote = c0_;
555 
556  next().literal_chars.Start();
557  while (true) {
558  AdvanceUntil([this](base::uc32 c0) {
559  if (V8_UNLIKELY(static_cast<uint32_t>(c0) > kMaxAscii)) {
560  if (V8_UNLIKELY(unibrow::IsStringLiteralLineTerminator(c0))) {
561  return true;
562  }
563  AddLiteralChar(c0);
564  return false;
565  }
566  uint8_t char_flags = character_scan_flags[c0];
567  if (MayTerminateString(char_flags)) return true;
568  AddLiteralChar(c0);
569  return false;
570  });
571 
572  while (c0_ == '\\') {
573  Advance();
574  // TODO(verwaest): Check whether we can remove the additional check.
575  if (V8_UNLIKELY(c0_ == kEndOfInput || !ScanEscape<false>())) {
576  return Token::kIllegal;
577  }
578  }
579 
580  if (c0_ == quote) {
581  Advance();
582  return Token::kString;
583  }
584 
585  if (V8_UNLIKELY(c0_ == kEndOfInput ||
586  unibrow::IsStringLiteralLineTerminator(c0_))) {
587  return Token::kIllegal;
588  }
589 
590  AddLiteralChar(c0_);
591  }
592 }

611 Token::Value Scanner::ScanTemplateSpan() {
612  // When scanning a TemplateSpan, we are looking for the following construct:
613  // kTemplateSpan ::
614  // ` LiteralChars* ${
615  // | } LiteralChars* ${
616  //
617  // kTemplateTail ::
618  // ` LiteralChars* `
619  // | } LiteralChar* `
620  //
621  // A kTemplateSpan should always be followed by an Expression, while a
622  // kTemplateTail terminates a TemplateLiteral and does not need to be
623  // followed by an Expression.
624 
625  // These scoped helpers save and restore the original error state, so that we
626  // can specially treat invalid escape sequences in templates (which are
627  // handled by the parser).
628  ErrorState scanner_error_state(&scanner_error_, &scanner_error_location_);
629  ErrorState octal_error_state(&octal_message_, &octal_pos_);
630 
631  Token::Value result = Token::kTemplateSpan;
632  next().literal_chars.Start();
633  next().raw_literal_chars.Start();
634  const bool capture_raw = true;
635  while (true) {
636  base::uc32 c = c0_;
637  if (c == '`') {
638  Advance(); // Consume '`'
639  result = Token::kTemplateTail;
640  break;
641  } else if (c == '$' && Peek() == '{') {
642  Advance(); // Consume '$'
643  Advance(); // Consume '{'
644  break;
645  } else if (c == '\\') {
646  Advance(); // Consume '\\'
647  DCHECK(!unibrow::IsLineTerminator(kEndOfInput));
648  if (capture_raw) AddRawLiteralChar('\\');
649  if (unibrow::IsLineTerminator(c0_)) {
650  // The TV of LineContinuation :: \ LineTerminatorSequence is the empty
651  // code unit sequence.
652  base::uc32 lastChar = c0_;
653  Advance();
654  if (lastChar == '\r') {
655  // Also skip \n.
656  if (c0_ == '\n') Advance();
657  lastChar = '\n';
658  }
659  if (capture_raw) AddRawLiteralChar(lastChar);
660  } else {
661  bool success = ScanEscape<capture_raw>();
662  USE(success);
663  DCHECK_EQ(!success, has_error());
664  // For templates, invalid escape sequence checking is handled in the
665  // parser.
666  scanner_error_state.MoveErrorTo(next_);
667  octal_error_state.MoveErrorTo(next_);
668  }
669  } else if (c == kEndOfInput) {
670  // Unterminated template literal
671  break;
672  } else {
673  Advance(); // Consume c.
674  // The TRV of LineTerminatorSequence :: <CR> is the CV 0x000A.

855 Token::Value Scanner::ScanNumber(bool seen_period) {
856  DCHECK(IsDecimalDigit(c0_)); // the first digit of the number or the fraction
857 
858  NumberKind kind = DECIMAL;
859 
860  next().literal_chars.Start();
861  bool at_start = !seen_period;
862  int start_pos = source_pos(); // For reporting octal positions.
863  if (seen_period) {
864  // we have already seen a decimal point of the float
865  AddLiteralChar('.');
866  if (c0_ == '_') {
867  return Token::kIllegal;
868  }
869  // we know we have at least one digit
870  if (!ScanDecimalDigits(true)) return Token::kIllegal;
871  } else {
872  // if the first character is '0' we must check for octals and hex
873  if (c0_ == '0') {
874  AddLiteralCharAdvance();
875 
876  // either 0, 0exxx, 0Exxx, 0.xxx, a hex number, a binary number or
877  // an octal number.
878  if (AsciiAlphaToLower(c0_) == 'x') {
879  AddLiteralCharAdvance();
880  kind = HEX;
881  if (!ScanHexDigits()) return Token::kIllegal;
882  } else if (AsciiAlphaToLower(c0_) == 'o') {
883  AddLiteralCharAdvance();
884  kind = OCTAL;
885  if (!ScanOctalDigits()) return Token::kIllegal;
886  } else if (AsciiAlphaToLower(c0_) == 'b') {
887  AddLiteralCharAdvance();
888  kind = BINARY;
889  if (!ScanBinaryDigits()) return Token::kIllegal;
890  } else if (IsOctalDigit(c0_)) {
891  kind = IMPLICIT_OCTAL;
892  if (!ScanImplicitOctalDigits(start_pos, &kind)) {
893  return Token::kIllegal;
894  }
895  if (kind == DECIMAL_WITH_LEADING_ZERO) {
896  at_start = false;
897  }
898  } else if (IsNonOctalDecimalDigit(c0_)) {
899  kind = DECIMAL_WITH_LEADING_ZERO;
900  } else if (c0_ == '_') {
901  ReportScannerError(Location(source_pos(), source_pos() + 1),
902  MessageTemplate::kZeroDigitNumericSeparator);
903  return Token::kIllegal;
904  }
905  }
906 
907  // Parse decimal digits and allow trailing fractional part.
908  if (IsDecimalNumberKind(kind)) {
909  bool allow_numeric_separator = kind != DECIMAL_WITH_LEADING_ZERO;
910  // This is an optimization for parsing Decimal numbers as Smi's.
911  if (at_start) {
912  uint64_t value = 0;
913  // scan subsequent decimal digits
914  if (!ScanDecimalAsSmi(&value, allow_numeric_separator)) {
915  return Token::kIllegal;
916  }
917 
918  if (next().literal_chars.one_byte_literal().length() <= 10 &&
919  value <= Smi::kMaxValue && c0_ != '.' && !IsIdentifierStart(c0_)) {
920  next().smi_value = static_cast<uint32_t>(value);
921 
922  if (kind == DECIMAL_WITH_LEADING_ZERO) {
923  octal_pos_ = Location(start_pos, source_pos());
924  octal_message_ = MessageTemplate::kStrictDecimalWithLeadingZero;
925  }
926  return Token::kSmi;
927  }
928  }
929 
930  if (!ScanDecimalDigits(allow_numeric_separator)) {
931  return Token::kIllegal;
932  }
933  if (c0_ == '.') {
934  seen_period = true;
935  AddLiteralCharAdvance();
936  if (c0_ == '_') {
937  return Token::kIllegal;
938  }
939  if (!ScanDecimalDigits(true)) return Token::kIllegal;
940  }
941  }
942  }
943 
944  bool is_bigint = false;
945  if (c0_ == 'n' && !seen_period && IsValidBigIntKind(kind)) {
946  // Check that the literal is within our limits for BigInt length.
947  // For simplicity, use 4 bits per character to calculate the maximum
948  // allowed literal length.
949  static const int kMaxBigIntCharacters = BigInt::kMaxLengthBits / 4;
950  int length = source_pos() - start_pos - (kind != DECIMAL ? 2 : 0);
951  if (length > kMaxBigIntCharacters) {
952  ReportScannerError(Location(start_pos, source_pos()),
953  MessageTemplate::kBigIntTooBig);
954  return Token::kIllegal;
955  }
956 
957  is_bigint = true;
958  Advance();
959  } else if (AsciiAlphaToLower(c0_) == 'e') {
960  // scan exponent, if any
961  DCHECK_NE(kind, HEX); // 'e'/'E' must be scanned as part of the hex number
962 
963  if (!IsDecimalNumberKind(kind)) return Token::kIllegal;
964 
965  // scan exponent
966  AddLiteralCharAdvance();
967

969  }

972  // not be an identifier start or a decimal digit; see ECMA-262
973  // section 7.8.3, page 17 (note that we read only one decimal digit
974  // if the value is 0).
975  if (IsDecimalDigit(c0_) || IsIdentifierStart(c0_)) {
976  return Token::kIllegal;
977  }

980  octal_pos_ = Location(start_pos, source_pos());
981  octal_message_ = MessageTemplate::kStrictDecimalWithLeadingZero;
982  }
983 
984  next().number_kind = kind;
985  return is_bigint ? Token::kBigInt : Token::kNumber;

1063 bool Scanner::ScanRegExpPattern() {
1064  DCHECK_EQ(Token::kUninitialized, next_next().token);
1065  DCHECK(next().token == Token::kDiv || next().token == Token::kAssignDiv);
1066 
1067  // Scan: ('/' | '/=') RegularExpressionBody '/' RegularExpressionFlags
1068  bool in_character_class = false;
1069 
1070  // Scan regular expression body: According to ECMA-262, 3rd, 7.8.5,
1071  // the scanner should pass uninterpreted bodies to the RegExp
1072  // constructor.
1073  next().literal_chars.Start();
1074  if (next().token == Token::kAssignDiv) {
1075  AddLiteralChar('=');
1076  }
1077 
1078  while (c0_ != '/' || in_character_class) {
1079  if (c0_ == kEndOfInput || unibrow::IsLineTerminator(c0_)) {
1080  return false;
1081  }
1082  if (c0_ == '\\') { // Escape sequence.
1083  AddLiteralCharAdvance();

1085  return false;
1086  }
1087  AddLiteralCharAdvance();
1088  // If the escape allows more characters, i.e., \x??, \u????, or \c?,
1089  // only "safe" characters are allowed (letters, digits, underscore),
1090  // otherwise the escape isn't valid and the invalid character has
1091  // its normal meaning. I.e., we can just continue scanning without
1092  // worrying whether the following characters are part of the escape
1093  // or not, since any '/', '\\' or '[' is guaranteed to not be part
1094  // of the escape sequence.
1095  } else { // Unescaped character.
1096  if (c0_ == '[') in_character_class = true;
1097  if (c0_ == ']') in_character_class = false;
1098  AddLiteralCharAdvance();
1099  }
1100  }
1101  Advance(); // consume '/'
1102 
1103  next().token = Token::kRegExpLiteral;
1104  return true;
1105 }
