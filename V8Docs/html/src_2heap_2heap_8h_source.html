251  uint64_t UpdateAmount(int64_t delta) {
252  const uint64_t amount_before =
253  total_.fetch_add(delta, std::memory_order_relaxed);
254  CHECK_GE(static_cast<int64_t>(amount_before), -delta);
255  return amount_before + delta;
256  }

267  uint64_t AllocatedSinceMarkCompact() const {
268  uint64_t total_bytes = total();
269  uint64_t low_since_mark_compact_bytes = low_since_mark_compact();
270 
271  if (total_bytes <= low_since_mark_compact_bytes) {
272  return 0;
273  }
274  return total_bytes - low_since_mark_compact_bytes;
275  }

279  total_.store(value, std::memory_order_relaxed);
