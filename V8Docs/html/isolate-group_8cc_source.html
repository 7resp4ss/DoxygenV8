70 struct PtrComprCageReservationParams
71  : public VirtualMemoryCage::ReservationParams {
72  PtrComprCageReservationParams() {
73  page_allocator = GetPlatformPageAllocator();
74 
75  reservation_size = kPtrComprCageReservationSize;
76  base_alignment = kPtrComprCageBaseAlignment;
77 
78  // Simplify BoundedPageAllocator's life by configuring it to use same page
79  // size as the Heap will use (MemoryChunk::kPageSize).
80  page_size =
81  RoundUp(size_t{1} << kPageSizeBits, page_allocator->AllocatePageSize());
82  requested_start_hint = RoundDown(
83  reinterpret_cast<Address>(page_allocator->GetRandomMmapAddr()),
84  base_alignment);
85 
86 #if V8_OS_FUCHSIA && !V8_EXTERNAL_CODE_SPACE
87  // If external code space is not enabled then executable pages (e.g. copied
88  // builtins, and JIT pages) will fall under the pointer compression range.
89  // Under Fuchsia that means the entire range must be allocated as JITtable.
90  permissions = PageAllocator::Permission::kNoAccessWillJitLater;
91 #else
92  permissions = PageAllocator::Permission::kNoAccess;
93 #endif
94  page_initialization_mode =
95  base::PageInitializationMode::kAllocatedPagesCanBeUninitialized;
96  page_freeing_mode = base::PageFreeingMode::kMakeInaccessible;
97  }
98 };
99 #endif // V8_COMPRESS_POINTERS

286  DCHECK_EQ(isolate->isolate_group(), this);
287  base::MutexGuard guard(&mutex_);
288 
289  const bool inserted = isolates_.insert(isolate).second;
290  CHECK(inserted);

304  DCHECK(isolate->owns_shareable_data_);
305  }

309 void IsolateGroup::RemoveIsolate(Isolate* isolate) {
310  base::MutexGuard guard(&mutex_);
311 
312  if (isolates_.size() == 1) {
313  read_only_artifacts_.reset();
314 
315  optimizing_compile_task_executor_->Stop();
316 
317  // We are removing the last isolate from the group. If this group has a
318  // shared heap, the last isolate has to be the shared space isolate.
319  DCHECK_EQ(has_shared_space_isolate(), isolate->is_shared_space_isolate());
320 
321  if (isolate->is_shared_space_isolate()) {
322  CHECK_EQ(isolate, shared_space_isolate_);
323  shared_space_isolate_ = nullptr;
324  }
325  } else {
326  // The shared space isolate needs to be removed last.
327  DCHECK(!isolate->is_shared_space_isolate());
328  }
329 
330  CHECK_EQ(isolates_.erase(isolate), 1);
331 
332  if (main_isolate_ == isolate) {
333  if (isolates_.empty()) {
334  main_isolate_ = nullptr;
335  } else {
336  main_isolate_ = *isolates_.begin();
337  }
338  }

379 void SandboxedArrayBufferAllocator::LazyInitialize(Sandbox* sandbox) {
380  base::MutexGuard guard(&mutex_);
381  if (is_initialized()) {
382  return;
383  }
384  CHECK(sandbox->is_initialized());
385  sandbox_ = sandbox;
386  constexpr size_t max_backing_memory_size = 8ULL * GB;
387  constexpr size_t min_backing_memory_size = 1ULL * GB;
388  size_t backing_memory_size = max_backing_memory_size;
389  Address backing_memory_base = 0;
390  while (!backing_memory_base &&
391  backing_memory_size >= min_backing_memory_size) {
392  backing_memory_base = sandbox_->address_space()->AllocatePages(
393  VirtualAddressSpace::kNoHint, backing_memory_size, kChunkSize,
394  PagePermissions::kNoAccess);
395  if (!backing_memory_base) {
396  backing_memory_size /= 2;
397  }
398  }
399  if (!backing_memory_base) {
400  V8::FatalProcessOutOfMemory(
401  nullptr, "Could not reserve backing memory for ArrayBufferAllocators");
402  }
403  DCHECK(IsAligned(backing_memory_base, kChunkSize));
404 
405  region_alloc_ = std::make_unique<base::RegionAllocator>(
406  backing_memory_base, backing_memory_size, kAllocationGranularity);
407  end_of_accessible_region_ = region_alloc_->begin();
408 
409  // Install an on-merge callback to discard or decommit unused pages.
410  region_alloc_->set_on_merge_callback([this](Address start, size_t size) {
411  mutex_.AssertHeld();
412  Address end = start + size;
413  if (end == region_alloc_->end() &&
414  start <= end_of_accessible_region_ - kChunkSize) {
415  // Can shrink the accessible region.
416  Address new_end_of_accessible_region = RoundUp(start, kChunkSize);
417  size_t size_to_decommit =
418  end_of_accessible_region_ - new_end_of_accessible_region;
419  if (!sandbox_->address_space()->DecommitPages(
420  new_end_of_accessible_region, size_to_decommit)) {
421  V8::FatalProcessOutOfMemory(nullptr, "SandboxedArrayBufferAllocator()");
422  }
423  end_of_accessible_region_ = new_end_of_accessible_region;
424  } else if (size >= 2 * kChunkSize) {
425  // Can discard pages. The pages stay accessible, so the size of the
426  // accessible region doesn't change.
427  Address chunk_start = RoundUp(start, kChunkSize);
428  Address chunk_end = RoundDown(start + size, kChunkSize);
429  if (!sandbox_->address_space()->DiscardSystemPages(
430  chunk_start, chunk_end - chunk_start)) {
431  V8::FatalProcessOutOfMemory(nullptr, "SandboxedArrayBufferAllocator()");
432  }
433  }
434  });
435 }

446 void* SandboxedArrayBufferAllocator::Allocate(size_t length) {
447  base::MutexGuard guard(&mutex_);
448 
449  length = RoundUp(length, kAllocationGranularity);
450  Address region = region_alloc_->AllocateRegion(length);
451  if (region == base::RegionAllocator::kAllocationFailure) return nullptr;
452 
453  // Check if the memory is inside the accessible region. If not, grow it.
454  Address end = region + length;
455  size_t length_to_memset = length;
456  if (end > end_of_accessible_region_) {
457  Address new_end_of_accessible_region = RoundUp(end, kChunkSize);
458  size_t size = new_end_of_accessible_region - end_of_accessible_region_;
459  if (!sandbox_->address_space()->SetPagePermissions(
460  end_of_accessible_region_, size, PagePermissions::kReadWrite)) {
461  if (!region_alloc_->FreeRegion(region)) {
462  V8::FatalProcessOutOfMemory(
463  nullptr, "SandboxedArrayBufferAllocator::Allocate()");
464  }
465  return nullptr;
466  }
467 
468  // The pages that were inaccessible are guaranteed to be zeroed, so only
469  // memset until the previous end of the accessible region.
470  length_to_memset = end_of_accessible_region_ - region;
471  end_of_accessible_region_ = new_end_of_accessible_region;
472  }
473 
474  void* mem = reinterpret_cast<void*>(region);
475  memset(mem, 0, length_to_memset);
476  return mem;
477 }
