299 // Uses a map as backing storage when sparsely, or a vector when densely
300 // populated. Requires {Value} to implement `bool is_set()` to identify
301 // uninitialized objects.
302 template <class Value>
303 class AdaptiveMap {
304  public:
305  // The technical limitation here is that index+1 must not overflow. Since
306  // we have significantly lower maximums on anything that can be named,
307  // we can have a tighter limit here to reject useless entries early.
308  static constexpr uint32_t kMaxKey = 10'000'000;
309  static_assert(kMaxKey < std::numeric_limits<uint32_t>::max());
310 
311  AdaptiveMap() : map_(new MapType()) {}
312 
313  explicit AdaptiveMap(const AdaptiveMap&) = delete;
314  AdaptiveMap& operator=(const AdaptiveMap&) = delete;
315 
316  AdaptiveMap(AdaptiveMap&& other) V8_NOEXCEPT { *this = std::move(other); }
317 
318  AdaptiveMap& operator=(AdaptiveMap&& other) V8_NOEXCEPT {
319  mode_ = other.mode_;
320  vector_.swap(other.vector_);
321  map_.swap(other.map_);
322  return *this;
323  }
324 
325  void FinishInitialization();
326 
327  bool is_set() const { return mode_ != kInitializing; }
328 
329  void Put(uint32_t key, const Value& value) {
330  DCHECK(mode_ == kInitializing);
331  DCHECK_LE(key, kMaxKey);
332  map_->insert(std::make_pair(key, value));
333  }
334 
335  void Put(uint32_t key, Value&& value) {
336  DCHECK(mode_ == kInitializing);
337  DCHECK_LE(key, kMaxKey);
338  map_->insert(std::make_pair(key, std::move(value)));
339  }
340 
341  const Value* Get(uint32_t key) const {
342  if (mode_ == kDense) {
343  if (key >= vector_.size()) return nullptr;
344  if (!vector_[key].is_set()) return nullptr;
345  return &vector_[key];
346  } else {
347  DCHECK(mode_ == kSparse || mode_ == kInitializing);
348  auto it = map_->find(key);
349  if (it == map_->end()) return nullptr;
350  return &it->second;
351  }
352  }
353 
354  bool Has(uint32_t key) const {
355  if (mode_ == kDense) {
356  return key < vector_.size() && vector_[key].is_set();
357  } else {
358  DCHECK(mode_ == kSparse || mode_ == kInitializing);
359  return map_->find(key) != map_->end();
360  }
361  }
362 
363  size_t EstimateCurrentMemoryConsumption() const;
364 
365  private:
366  static constexpr uint32_t kLoadFactor = 4;
367  using MapType = std::map<uint32_t, Value>;
368  enum Mode { kDense, kSparse, kInitializing };
369 
370  Mode mode_{kInitializing};
371  std::vector<Value> vector_;
372  std::unique_ptr<MapType> map_;
373 };

493  public:
494  struct PolymorphicCase {

499  static CallSiteFeedback CreateMegamorphic() {
500  CallSiteFeedback feedback;
501  feedback.is_megamorphic_ = true;
502  DCHECK(!feedback.is_invalid());
503  DCHECK(!feedback.is_monomorphic());
504  DCHECK(!feedback.is_polymorphic());
505  DCHECK(feedback.is_megamorphic());
506  return feedback;
507  }
508 
509  // Regular constructor: uninitialized/unknown, monomorphic, or
510  // polymorphic.
511  CallSiteFeedback() : index_or_count_(-1), frequency_or_ool_(0) {}
512  CallSiteFeedback(int function_index, int call_count)
513  : index_or_count_(function_index), frequency_or_ool_(call_count) {}
514  CallSiteFeedback(PolymorphicCase* polymorphic_cases, int num_cases)
515  : index_or_count_(-num_cases),
516  frequency_or_ool_(reinterpret_cast<intptr_t>(polymorphic_cases)) {}
517 
518  // Copying and assignment: prefer moving, as it's cheaper.
519  // The code below makes sure external polymorphic storage is copied and/or
520  // freed as appropriate.
521  CallSiteFeedback(const CallSiteFeedback& other) V8_NOEXCEPT { *this = other; }
522  CallSiteFeedback(CallSiteFeedback&& other) V8_NOEXCEPT { *this = other; }
523  CallSiteFeedback& operator=(const CallSiteFeedback& other) V8_NOEXCEPT {
524  index_or_count_ = other.index_or_count_;
525  if (other.is_polymorphic()) {
526  int num_cases = other.num_cases();
527  PolymorphicCase* polymorphic = new PolymorphicCase[num_cases];
528  for (int i = 0; i < num_cases; i++) {
529  polymorphic[i].function_index = other.function_index(i);
530  polymorphic[i].absolute_call_frequency = other.call_count(i);
531  }
532  frequency_or_ool_ = reinterpret_cast<intptr_t>(polymorphic);
533  } else {
534  frequency_or_ool_ = other.frequency_or_ool_;
535  }
536  has_non_inlineable_targets_ = other.has_non_inlineable_targets_;
537  is_megamorphic_ = other.is_megamorphic_;
538  return *this;
539  }
540  CallSiteFeedback& operator=(CallSiteFeedback&& other) V8_NOEXCEPT {
541  if (this != &other) {
542  index_or_count_ = other.index_or_count_;
543  frequency_or_ool_ = other.frequency_or_ool_;
544  other.frequency_or_ool_ = 0;
545  }
546  has_non_inlineable_targets_ = other.has_non_inlineable_targets_;
547  is_megamorphic_ = other.is_megamorphic_;
548  return *this;
549  }
550 
551  ~CallSiteFeedback() {
552  if (is_polymorphic()) delete[] polymorphic_storage();
553  }
554 
555  int num_cases() const {
556  if (is_monomorphic()) return 1;
557  if (is_invalid() || is_megamorphic()) return 0;
558  return -index_or_count_;
559  }
560  int function_index(int i) const {
561  DCHECK(!is_invalid() && !is_megamorphic());
562  if (is_monomorphic()) return index_or_count_;
563  return polymorphic_storage()[i].function_index;
564  }
565  int call_count(int i) const {
566  DCHECK(!is_invalid() && !is_megamorphic());
567  if (is_monomorphic()) return static_cast<int>(frequency_or_ool_);
568  return polymorphic_storage()[i].absolute_call_frequency;
569  }
570  bool has_non_inlineable_targets() const {
571  return has_non_inlineable_targets_;
572  }
573  void set_has_non_inlineable_targets(bool has_non_inlineable_targets) {
574  has_non_inlineable_targets_ = has_non_inlineable_targets;
575  }
576 
577  bool is_megamorphic() const { return is_megamorphic_; }
578 
579  private:
580  bool is_monomorphic() const { return index_or_count_ >= 0; }
581  bool is_polymorphic() const { return index_or_count_ <= -2; }
582  bool is_invalid() const { return index_or_count_ == -1 && !is_megamorphic_; }
583  const PolymorphicCase* polymorphic_storage() const {
584  DCHECK(is_polymorphic());
585  return reinterpret_cast<PolymorphicCase*>(frequency_or_ool_);
586  }
587 
588  int index_or_count_;
589  bool has_non_inlineable_targets_ = false;
590  bool is_megamorphic_ = false;
591  intptr_t frequency_or_ool_;
592 };

626 struct TypeFeedbackStorage {
627  std::unordered_map<uint32_t, FunctionTypeFeedback> feedback_for_function;
628  std::unordered_map<uint32_t, uint32_t> deopt_count_for_function;
629  // Accesses to {feedback_for_function} and {deopt_count_for_function} are
630  // guarded by this mutex. Multiple reads are allowed (shared lock), but only
631  // exclusive writes. Currently known users of the mutex are:
632  // - LiftoffCompiler: writes {call_targets}.
633  // - TransitiveTypeFeedbackProcessor: reads {call_targets},
634  // writes {feedback_vector}, reads {feedback_vector.size()}.
635  // - TriggerTierUp: increments {tierup_priority}.
636  // - WasmGraphBuilder: reads {feedback_vector}.
637  // - Feedback vector allocation: reads {call_targets.size()}.
638  // - PGO ProfileGenerator: reads everything.
