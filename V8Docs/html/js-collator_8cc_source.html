87 // static
88 DirectHandle<JSObject> JSCollator::ResolvedOptions(
89  Isolate* isolate, DirectHandle<JSCollator> collator) {
90  DirectHandle<JSObject> options =
91  isolate->factory()->NewJSObject(isolate->object_function());
92 
93  icu::Collator* icu_collator = collator->icu_collator()->raw();
94  DCHECK_NOT_NULL(icu_collator);
95 
96  UErrorCode status = U_ZERO_ERROR;
97  bool numeric =
98  icu_collator->getAttribute(UCOL_NUMERIC_COLLATION, status) == UCOL_ON;
99  DCHECK(U_SUCCESS(status));
100 
101  const char* case_first = nullptr;
102  status = U_ZERO_ERROR;
103  switch (icu_collator->getAttribute(UCOL_CASE_FIRST, status)) {
104  case UCOL_LOWER_FIRST:
105  case_first = "lower";
106  break;
107  case UCOL_UPPER_FIRST:
108  case_first = "upper";
109  break;
110  default:
111  case_first = "false";
112  }
113  DCHECK(U_SUCCESS(status));
114 
115  const char* sensitivity = nullptr;
116  status = U_ZERO_ERROR;
117  switch (icu_collator->getAttribute(UCOL_STRENGTH, status)) {
118  case UCOL_PRIMARY: {
119  DCHECK(U_SUCCESS(status));
120  status = U_ZERO_ERROR;
121  // case level: true + s1 -> case, s1 -> base.
122  if (UCOL_ON == icu_collator->getAttribute(UCOL_CASE_LEVEL, status)) {
123  sensitivity = "case";
124  } else {
125  sensitivity = "base";
126  }
127  DCHECK(U_SUCCESS(status));
128  break;
129  }
130  case UCOL_SECONDARY:
131  sensitivity = "accent";
132  break;
133  case UCOL_TERTIARY:
134  sensitivity = "variant";
135  break;
136  case UCOL_QUATERNARY:
137  // We shouldn't get quaternary and identical from ICU, but if we do
138  // put them into variant.
139  sensitivity = "variant";
140  break;
141  default:
142  sensitivity = "variant";
143  }
144  DCHECK(U_SUCCESS(status));
145 
146  status = U_ZERO_ERROR;
147  bool ignore_punctuation = icu_collator->getAttribute(UCOL_ALTERNATE_HANDLING,
148  status) == UCOL_SHIFTED;
149  DCHECK(U_SUCCESS(status));
150 
151  status = U_ZERO_ERROR;
152 
153  icu::Locale icu_locale(icu_collator->getLocale(ULOC_VALID_LOCALE, status));
154  DCHECK(U_SUCCESS(status));
155 
156  const char* collation = "default";
157  const char* usage = "sort";
158  const char* collation_key = "co";
159  status = U_ZERO_ERROR;
160  std::string collation_value =
161  icu_locale.getUnicodeKeywordValue<std::string>(collation_key, status);
162 
163  std::string locale;
164  if (U_SUCCESS(status)) {
165  if (collation_value == "search") {
166  usage = "search";
167 
168  // Search is disallowed as a collation value per spec. Let's
169  // use `default`, instead.
170  //
171  // https://tc39.github.io/ecma402/#sec-properties-of-intl-collator-instances
172  collation = "default";
173 
174  // We clone the icu::Locale because we don't want the
175  // icu_collator to be affected when we remove the collation key
176  // below.
177  icu::Locale new_icu_locale = icu_locale;
178 
179  // The spec forbids the search as a collation value in the
180  // locale tag, so let's filter it out.
181  status = U_ZERO_ERROR;
182  new_icu_locale.setUnicodeKeywordValue(collation_key, nullptr, status);
183  DCHECK(U_SUCCESS(status));
184 
185  locale = Intl::ToLanguageTag(new_icu_locale).FromJust();
186  } else {
187  collation = collation_value.c_str();
188  locale = Intl::ToLanguageTag(icu_locale).FromJust();
189  }
190  } else {
191  locale = Intl::ToLanguageTag(icu_locale).FromJust();
192  }
193 
194  // 5. For each row of Table 2, except the header row, in table order, do
195  // ...
196  // Table 2: Resolved Options of Collator Instances
197  // Internal Slot Property Extension Key
198  // [[Locale] "locale"
199  // [[Usage] "usage"
200  // [[Sensitivity]] "sensitivity"
201  // [[IgnorePunctuation]] "ignorePunctuation"
202  // [[Collation]] "collation"
203  // [[Numeric]] "numeric" kn
204  // [[CaseFirst]] "caseFirst" kf
205 
206  // If the collator return the locale differ from what got requested, we stored
207  // it in the collator->locale. Otherwise, we just use the one from the
208  // collator.
209  if (collator->locale()->length() != 0) {
210  // Get the locale from collator->locale() since we know in some cases
211  // collator won't be able to return the requested one, such as zh_CN.
212  DirectHandle<String> locale_from_collator(collator->locale(), isolate);
213  Maybe<bool> maybe = JSReceiver::CreateDataProperty(
214  isolate, options, isolate->factory()->locale_string(),
215  locale_from_collator, Just(kDontThrow));
216  DCHECK(maybe.FromJust());
217  USE(maybe);
218  } else {
219  // Just return from the collator for most of the cases that we can recover
220  // from the collator.
221  CreateDataPropertyForOptions(
222  isolate, options, isolate->factory()->locale_string(), locale.c_str());
223  }
224 
225  CreateDataPropertyForOptions(isolate, options,
226  isolate->factory()->usage_string(), usage);
227  CreateDataPropertyForOptions(
228  isolate, options, isolate->factory()->sensitivity_string(), sensitivity);
229  CreateDataPropertyForOptions(isolate, options,
230  isolate->factory()->ignorePunctuation_string(),
231  ignore_punctuation);
232  CreateDataPropertyForOptions(
233  isolate, options, isolate->factory()->collation_string(), collation);
234  CreateDataPropertyForOptions(isolate, options,
235  isolate->factory()->numeric_string(), numeric);
236  CreateDataPropertyForOptions(
237  isolate, options, isolate->factory()->caseFirst_string(), case_first);
238  return options;
239 }

243 CaseFirst ToCaseFirst(const char* str) {
244  if (strcmp(str, "upper") == 0) return CaseFirst::kUpper;
245  if (strcmp(str, "lower") == 0) return CaseFirst::kLower;
246  if (strcmp(str, "false") == 0) return CaseFirst::kFalse;
247  return CaseFirst::kUndefined;
248 }

250 UColAttributeValue ToUColAttributeValue(CaseFirst case_first) {
251  switch (case_first) {
252  case CaseFirst::kUpper:
253  return UCOL_UPPER_FIRST;
254  case CaseFirst::kLower:
255  return UCOL_LOWER_FIRST;
256  case CaseFirst::kFalse:
257  case CaseFirst::kUndefined:
258  return UCOL_OFF;
259  }
260 }

280 // static
281 MaybeHandle<JSCollator> JSCollator::New(Isolate* isolate, DirectHandle<Map> map,
282  DirectHandle<Object> locales,
283  DirectHandle<Object> options_obj,
284  const char* service) {
285  // 1. Let requestedLocales be ? CanonicalizeLocaleList(locales).
286  Maybe<std::vector<std::string>> maybe_requested_locales =
287  Intl::CanonicalizeLocaleList(isolate, locales);
288  MAYBE_RETURN(maybe_requested_locales, Handle<JSCollator>());
289  std::vector<std::string> requested_locales =
290  maybe_requested_locales.FromJust();
291 
292  // 2. Set options to ? CoerceOptionsToObject(options).
293  DirectHandle<JSReceiver> options;
294  ASSIGN_RETURN_ON_EXCEPTION(
295  isolate, options, CoerceOptionsToObject(isolate, options_obj, service));
296 
297  // 4. Let usage be ? GetOption(options, "usage", "string", « "sort",
298  // "search" », "sort").
299  Maybe<Usage> maybe_usage = GetStringOption<Usage>(
300  isolate, options, "usage", service,
301  std::to_array<const std::string_view>({"sort", "search"}),
302  std::array{Usage::SORT, Usage::SEARCH}, Usage::SORT);
303  MAYBE_RETURN(maybe_usage, MaybeHandle<JSCollator>());
304  Usage usage = maybe_usage.FromJust();
305 
306  // 9. Let matcher be ? GetOption(options, "localeMatcher", "string",
307  // « "lookup", "best fit" », "best fit").
308  // 10. Set opt.[[localeMatcher]] to matcher.
309  Maybe<Intl::MatcherOption> maybe_locale_matcher =
310  Intl::GetLocaleMatcher(isolate, options, service);
311  MAYBE_RETURN(maybe_locale_matcher, MaybeHandle<JSCollator>());
312  Intl::MatcherOption matcher = maybe_locale_matcher.FromJust();
313 
314  // x. Let _collation_ be ? GetOption(_options_, *"collation"*, *"string"*,
315  // *undefined*, *undefined*).
316  DirectHandle<String> collation_str;
317  Maybe<bool> maybe_collation =
318  GetStringOption(isolate, options, "collation",
319  std::span<std::string_view>(), service, &collation_str);
320  MAYBE_RETURN(maybe_collation, MaybeHandle<JSCollator>());
321 
322  // Unfortunately needs to be a std::string because of Intl::IsValidCollation
323  std::string collation_stdstr;
324  // x. If _collation_ is not *undefined*, then
325  if (maybe_collation.FromJust()) {
326  collation_stdstr = collation_str->ToStdString();
327  // 1. If _collation_ does not match the Unicode Locale Identifier `type`
328  // nonterminal, throw a *RangeError* exception.
329  if (!JSLocale::Is38AlphaNumList(collation_stdstr)) {
330  THROW_NEW_ERROR_RETURN_VALUE(
331  isolate,
332  NewRangeError(MessageTemplate::kInvalid,
333  isolate->factory()->collation_string(), collation_str),
334  MaybeHandle<JSCollator>());
335  }
336  }
337  // x. Set _opt_.[[co]] to _collation_.
338 
339  // 11. Let numeric be ? GetOption(options, "numeric", "boolean",
340  // undefined, undefined).
341  // 12. If numeric is not undefined, then
342  // a. Let numeric be ! ToString(numeric).
343  //
344  // Note: We omit the ToString(numeric) operation as it's not
345  // observable. GetBoolOption returns a Boolean and
346  // ToString(Boolean) is not side-effecting.
347  //
348  // 13. Set opt.[[kn]] to numeric.
349  bool numeric;
350  Maybe<bool> found_numeric =
351  GetBoolOption(isolate, options, "numeric", service, &numeric);
352  MAYBE_RETURN(found_numeric, MaybeHandle<JSCollator>());
353 
354  // 14. Let caseFirst be ? GetOption(options, "caseFirst", "string",
355  // « "upper", "lower", "false" », undefined).
356  Maybe<CaseFirst> maybe_case_first = GetCaseFirst(isolate, options, service);
357  MAYBE_RETURN(maybe_case_first, MaybeHandle<JSCollator>());
358  CaseFirst case_first = maybe_case_first.FromJust();
359 
360  // The relevant unicode extensions accepted by Collator as specified here:
361  // https://tc39.github.io/ecma402/#sec-intl-collator-internal-slots
362  //
363  // 16. Let relevantExtensionKeys be %Collator%.[[RelevantExtensionKeys]].
364 
365  // 17. Let r be ResolveLocale(%Collator%.[[AvailableLocales]],
366  // requestedLocales, opt, %Collator%.[[RelevantExtensionKeys]],
367  // localeData).
368  Intl::ResolvedLocale r;
369  if (!Intl::ResolveLocale(isolate, JSCollator::GetAvailableLocales(),
370  requested_locales, matcher, {"co", "kn", "kf"})
371  .To(&r)) {
372  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
373  }
374 
375  // 18. Set collator.[[Locale]] to r.[[locale]].
376  icu::Locale icu_locale = r.icu_locale;
377  DCHECK(!icu_locale.isBogus());
378 
379  // 19. Let collation be r.[[co]].
380  UErrorCode status = U_ZERO_ERROR;
381  if (maybe_collation.FromJust()) {
382  auto co_extension_it = r.extensions.find("co");
383  if (co_extension_it != r.extensions.end() &&
384  co_extension_it->second != collation_stdstr) {
385  icu_locale.setUnicodeKeywordValue("co", nullptr, status);
386  DCHECK(U_SUCCESS(status));
387  }
388  }
389 
390  // 5. Set collator.[[Usage]] to usage.
391  //
392  // 6. If usage is "sort", then
393  // a. Let localeData be %Collator%.[[SortLocaleData]].
394  // 7. Else,
395  // a. Let localeData be %Collator%.[[SearchLocaleData]].
396  //
397  // The Intl spec doesn't allow us to use "search" as an extension
398  // value for collation as per:
399  // https://tc39.github.io/ecma402/#sec-intl-collator-internal-slots
400  //
401  // But the only way to pass the value "search" for collation from
402  // the options object to ICU is to use the 'co' extension keyword.
403  //
404  // This will need to be filtered out when creating the
405  // resolvedOptions object.
406  if (usage == Usage::SEARCH) {
407  UErrorCode set_status = U_ZERO_ERROR;
408  icu_locale.setUnicodeKeywordValue("co", "search", set_status);
409  DCHECK(U_SUCCESS(set_status));
410  } else {
411  if (maybe_collation.FromJust() &&
412  Intl::IsValidCollation(icu_locale, collation_stdstr)) {
413  icu_locale.setUnicodeKeywordValue("co", collation_stdstr, status);
414  DCHECK(U_SUCCESS(status));
415  }
416  }
417 
418  // 20. If collation is null, let collation be "default".
419  // 21. Set collator.[[Collation]] to collation.
420  //
421  // We don't store the collation value as per the above two steps
422  // here. The collation value can be looked up from icu::Collator on
423  // demand, as part of Intl.Collator.prototype.resolvedOptions.
424 
425  std::unique_ptr<icu::Collator> icu_collator(
426  icu::Collator::createInstance(icu_locale, status));
427  if (U_FAILURE(status) || icu_collator == nullptr) {
428  status = U_ZERO_ERROR;
429  // Remove extensions and try again.
430  icu::Locale no_extension_locale(icu_locale.getBaseName());
431  icu_collator.reset(
432  icu::Collator::createInstance(no_extension_locale, status));
433 
434  if (U_FAILURE(status) || icu_collator == nullptr) {
435  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
436  }
437  }
438  DCHECK(U_SUCCESS(status));
439 
440  icu::Locale collator_locale(
441  icu_collator->getLocale(ULOC_VALID_LOCALE, status));
442 
443  // 22. If relevantExtensionKeys contains "kn", then
444  // a. Set collator.[[Numeric]] to ! SameValue(r.[[kn]], "true").
445  //
446  // If the numeric value is passed in through the options object,
447  // then we use it. Otherwise, we check if the numeric value is
448  // passed in through the unicode extensions.
449  status = U_ZERO_ERROR;
450  if (found_numeric.FromJust()) {
451  SetNumericOption(icu_collator.get(), numeric);
452  } else {
453  auto kn_extension_it = r.extensions.find("kn");
454  if (kn_extension_it != r.extensions.end()) {
455  SetNumericOption(icu_collator.get(), (kn_extension_it->second == "true"));
456  }
457  }
458 
459  // 23. If relevantExtensionKeys contains "kf", then
460  // a. Set collator.[[CaseFirst]] to r.[[kf]].
461  //
462  // If the caseFirst value is passed in through the options object,
463  // then we use it. Otherwise, we check if the caseFirst value is
464  // passed in through the unicode extensions.
465  if (case_first != CaseFirst::kUndefined) {
466  SetCaseFirstOption(icu_collator.get(), case_first);
467  } else {
468  auto kf_extension_it = r.extensions.find("kf");
469  if (kf_extension_it != r.extensions.end()) {
470  SetCaseFirstOption(icu_collator.get(),
471  ToCaseFirst(kf_extension_it->second.c_str()));
472  }
473  }
474 
475  // Normalization is always on, by the spec. We are free to optimize
476  // if the strings are already normalized (but we don't have a way to tell
477  // that right now).
478  status = U_ZERO_ERROR;
479  icu_collator->setAttribute(UCOL_NORMALIZATION_MODE, UCOL_ON, status);
480  DCHECK(U_SUCCESS(status));
481 
482  // 24. Let sensitivity be ? GetOption(options, "sensitivity",
483  // "string", « "base", "accent", "case", "variant" », undefined).
484  Maybe<Sensitivity> maybe_sensitivity = GetStringOption<Sensitivity>(
485  isolate, options, "sensitivity", service,
486  std::to_array<const std::string_view>(
487  {"base", "accent", "case", "variant"}),
488  std::array{Sensitivity::kBase, Sensitivity::kAccent, Sensitivity::kCase,
489  Sensitivity::kVariant},
490  Sensitivity::kUndefined);
491  MAYBE_RETURN(maybe_sensitivity, MaybeHandle<JSCollator>());
492  Sensitivity sensitivity = maybe_sensitivity.FromJust();
493 
494  // 25. If sensitivity is undefined, then
495  if (sensitivity == Sensitivity::kUndefined) {
496  // 25. a. If usage is "sort", then
497  if (usage == Usage::SORT) {
498  // 25. a. i. Let sensitivity be "variant".
499  sensitivity = Sensitivity::kVariant;
500  }
501  }
502  // 26. Set collator.[[Sensitivity]] to sensitivity.
503  switch (sensitivity) {
504  case Sensitivity::kBase:
505  icu_collator->setStrength(icu::Collator::PRIMARY);
506  break;
507  case Sensitivity::kAccent:
508  icu_collator->setStrength(icu::Collator::SECONDARY);
509  break;
510  case Sensitivity::kCase:
511  icu_collator->setStrength(icu::Collator::PRIMARY);
512  status = U_ZERO_ERROR;
513  icu_collator->setAttribute(UCOL_CASE_LEVEL, UCOL_ON, status);
514  DCHECK(U_SUCCESS(status));
515  break;
516  case Sensitivity::kVariant:
517  icu_collator->setStrength(icu::Collator::TERTIARY);
518  break;
519  case Sensitivity::kUndefined:
520  break;
521  }
522 
523  // 27.Let ignorePunctuation be ? GetOption(options,
524  // "ignorePunctuation", "boolean", undefined, false).
525  bool ignore_punctuation = false;
526  Maybe<bool> found_ignore_punctuation = GetBoolOption(
527  isolate, options, "ignorePunctuation", service, &ignore_punctuation);
528  MAYBE_RETURN(found_ignore_punctuation, MaybeHandle<JSCollator>());
529 
530  // 28. Set collator.[[IgnorePunctuation]] to ignorePunctuation.
531 
532  // Note: The following implementation does not strictly follow the spec text
533  // due to https://github.com/tc39/ecma402/issues/832
534  // If the ignorePunctuation is not defined, instead of fall back
535  // to default false, we just depend on ICU to default based on the
536  // built in locale collation rule, which in "th" locale that is true
537  // but false on other locales.
538  if (found_ignore_punctuation.FromJust()) {
539  status = U_ZERO_ERROR;
540  icu_collator->setAttribute(
541  UCOL_ALTERNATE_HANDLING,
542  ignore_punctuation ? UCOL_SHIFTED : UCOL_NON_IGNORABLE, status);
543  DCHECK(U_SUCCESS(status));
544  }
545 
546  DirectHandle<Managed<icu::Collator>> managed_collator =
547  Managed<icu::Collator>::From(isolate, 0, std::move(icu_collator));
548 
549  // We only need to do so if it is different from the collator would return.
550  DirectHandle<String> locale_str =
551  isolate->factory()->NewStringFromAsciiChecked(
552  (collator_locale != icu_locale) ? r.locale.c_str() : "");
553  // Now all properties are ready, so we can allocate the result object.
554  Handle<JSCollator> collator =
555  Cast<JSCollator>(isolate->factory()->NewFastOrSlowJSObjectFromMap(map));
556  DisallowGarbageCollection no_gc;
557  collator->set_icu_collator(*managed_collator);
558  collator->set_locale(*locale_str);
559 
560  // 29. Return collator.
561  return collator;
562 }

566 class CollatorAvailableLocales {
567  public:
568  CollatorAvailableLocales() {
569  int32_t num_locales = 0;
570  const icu::Locale* icu_available_locales =
571  icu::Collator::getAvailableLocales(num_locales);
572  std::vector<std::string> locales;
573  locales.reserve(num_locales);
574  for (int32_t i = 0; i < num_locales; ++i) {
575  locales.push_back(
576  Intl::ToLanguageTag(icu_available_locales[i]).FromJust());
577  }
578 #define U_ICUDATA_COLL U_ICUDATA_NAME U_TREE_SEPARATOR_STRING "coll"
579  set_ = Intl::BuildLocaleSet(locales, U_ICUDATA_COLL, nullptr);
580 #undef U_ICUDATA_COLL
581  }
582  virtual ~CollatorAvailableLocales() = default;
583  const std::set<std::string>& Get() const { return set_; }
584 
585  private:
586  std::set<std::string> set_;
587 };
