27 // The intention of the base::iterator_range class is to encapsulate two
28 // iterators so that the range defined by the iterators can be used like
29 // a regular STL container (actually only a subset of the full container
30 // functionality is available usually).
31 template <typename ForwardIterator>
32 class iterator_range {
33  public:
34  using iterator = ForwardIterator;
35  using const_iterator = ForwardIterator;
36  using pointer = typename std::iterator_traits<iterator>::pointer;
37  using reference = typename std::iterator_traits<iterator>::reference;
38  using value_type = typename std::iterator_traits<iterator>::value_type;
39  using difference_type =
40  typename std::iterator_traits<iterator>::difference_type;
41 
42  iterator_range() : begin_(), end_() {}
43  iterator_range(ForwardIterator begin, ForwardIterator end)
44  : begin_(begin), end_(end) {}
45 
46  iterator begin() const { return begin_; }
47  iterator end() const { return end_; }
48  const_iterator cbegin() const { return begin_; }
49  const_iterator cend() const { return end_; }
50  auto rbegin() const { return std::make_reverse_iterator(end_); }
51  auto rend() const { return std::make_reverse_iterator(begin_); }
52 
53  bool empty() const { return cbegin() == cend(); }
54 
55  // Random Access iterators only.
56  reference operator[](difference_type n) { return begin()[n]; }
57  difference_type size() const { return cend() - cbegin(); }
58 
59  private:
60  const_iterator const begin_;
61  const_iterator const end_;
62 };

89 // {Reversed} returns a container adapter usable in a range-based "for"
90 // statement for iterating a reversible container in reverse order.
91 //
92 // Example:
93 //
94 // std::vector<int> v = ...;
95 // for (int i : base::Reversed(v)) {
96 // // iterates through v from back to front
97 // }
98 //
99 // The signature avoids binding to temporaries (T&& / const T&) on purpose. The
100 // lifetime of a temporary would not extend to a range-based for loop using it.
101 template <typename T>
102 auto Reversed(T& t) {
103  return make_iterator_range(std::rbegin(t), std::rend(t));
104 }
105 
106 // This overload of `Reversed` is safe even when the argument is a temporary,
107 // because we rely on the wrapped iterators instead of the `iterator_range`
108 // object itself.
109 template <typename T>
110 auto Reversed(const iterator_range<T>& t) {
111  return make_iterator_range(std::rbegin(t), std::rend(t));
112 }

114 // {IterateWithoutLast} returns a container adapter usable in a range-based
115 // "for" statement for iterating all elements without the last in a forward
116 // order. It performs a check whether the container is empty.
117 //
118 // Example:
119 //
120 // std::vector<int> v = ...;
121 // for (int i : base::IterateWithoutLast(v)) {
122 // // iterates through v front to --back
123 // }
124 //
125 // The signature avoids binding to temporaries, see the remark in {Reversed}.
126 template <typename T>
127 auto IterateWithoutLast(T& t) {
128  DCHECK_NE(std::begin(t), std::end(t));
129  auto new_end = std::end(t);
130  return make_iterator_range(std::begin(t), --new_end);
131 }
132 
133 template <typename T>
134 auto IterateWithoutLast(const iterator_range<T>& t) {
135  iterator_range<T> range_copy = {t.begin(), t.end()};
136  return IterateWithoutLast(range_copy);
137 }

155 // TupleIterator is an iterator wrapping around multiple iterators. It is use by
156 // the `zip` function below to iterate over multiple containers at once.
157 template <class... Iterators>
158 class TupleIterator
159  : public base::iterator<
160  std::bidirectional_iterator_tag,
161  std::tuple<typename std::iterator_traits<Iterators>::reference...>> {
162  public:
163  using value_type =
164  std::tuple<typename std::iterator_traits<Iterators>::reference...>;
165 
166  explicit TupleIterator(Iterators... its) : its_(its...) {}
167 
168  TupleIterator& operator++() {
169  std::apply([](auto&... iterators) { (++iterators, ...); }, its_);
170  return *this;
171  }
172 
173  template <class Other>
174  bool operator!=(const Other& other) const {
175  return not_equal_impl(other, std::index_sequence_for<Iterators...>{});
176  }
177 
178  value_type operator*() const {
179  return std::apply(
180  [](auto&... this_iterators) { return value_type{*this_iterators...}; },
181  its_);
182  }
183 
184  private:
185  template <class Other, size_t... indices>
186  bool not_equal_impl(const Other& other,
187  std::index_sequence<indices...>) const {
188  return (... || (std::get<indices>(its_) != std::get<indices>(other.its_)));
189  }
190 
191  std::tuple<Iterators...> its_;
192 };

194 // `zip` creates an iterator_range from multiple containers. It can be used to
195 // iterate over multiple containers at once. For instance:
196 //
197 // std::vector<int> arr = { 2, 4, 6 };
198 // std::set<double> set = { 3.5, 4.5, 5.5 };
199 // for (auto [i, d] : base::zip(arr, set)) {
200 // std::cout << i << " and " << d << std::endl;
201 // }
202 //
203 // Prints "2 and 3.5", "4 and 4.5" and "6 and 5.5".
204 template <class... Containers>
205 auto zip(Containers&... containers) {
206  using TupleIt =
207  TupleIterator<decltype(std::declval<Containers>().begin())...>;
208  return base::make_iterator_range(TupleIt(containers.begin()...),
209  TupleIt(containers.end()...));
210 }
