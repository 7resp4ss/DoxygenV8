210 TF_BUILTIN(FastNewClosure, ConstructorBuiltinsAssembler) {
211  auto shared_function_info =
212  Parameter<SharedFunctionInfo>(Descriptor::kSharedFunctionInfo);
213  auto feedback_cell = Parameter<FeedbackCell>(Descriptor::kFeedbackCell);
214  auto context = Parameter<Context>(Descriptor::kContext);
215 
216  // Bump the closure counter encoded in the {feedback_cell}s map.
217  {
218  const TNode<Map> feedback_cell_map = LoadMap(feedback_cell);
219  Label no_closures(this), one_closure(this), cell_done(this);
220 
221  GotoIf(IsNoClosuresCellMap(feedback_cell_map), &no_closures);
222  GotoIf(IsOneClosureCellMap(feedback_cell_map), &one_closure);
223  CSA_DCHECK(this, IsManyClosuresCellMap(feedback_cell_map),
224  feedback_cell_map, feedback_cell);
225  Goto(&cell_done);
226 
227  BIND(&no_closures);
228  StoreMapNoWriteBarrier(feedback_cell, RootIndex::kOneClosureCellMap);
229  Goto(&cell_done);
230 
231  BIND(&one_closure);
232 #ifdef V8_ENABLE_LEAPTIERING
233  // The transition from one to many closures under leap tiering requires
234  // making sure that the dispatch_handle's code isn't context specialized for
235  // the single closure. This is handled in the runtime.
236  //
237  // TODO(leszeks): We could fast path this for the case where the dispatch
238  // handle either doesn't contain any code, or that code isn't context
239  // specialized.
240  TailCallRuntime(Runtime::kNewClosure, context, shared_function_info,
241  feedback_cell);
242 #else
243  StoreMapNoWriteBarrier(feedback_cell, RootIndex::kManyClosuresCellMap);
244  Goto(&cell_done);
245 #endif // V8_ENABLE_LEAPTIERING
246 
247  BIND(&cell_done);
248  }
249 
250  // The calculation of |function_map_index| must be in sync with
251  // SharedFunctionInfo::function_map_index().
252  TNode<Uint32T> flags = LoadObjectField<Uint32T>(
253  shared_function_info, SharedFunctionInfo::kFlagsOffset);
254  const TNode<IntPtrT> function_map_index = Signed(IntPtrAdd(
255  DecodeWordFromWord32<SharedFunctionInfo::FunctionMapIndexBits>(flags),
256  IntPtrConstant(Context::FIRST_FUNCTION_MAP_INDEX)));
257  CSA_DCHECK(this, UintPtrLessThanOrEqual(
258  function_map_index,
259  IntPtrConstant(Context::LAST_FUNCTION_MAP_INDEX)));
260 
261  // Get the function map in the current native context and set that
262  // as the map of the allocated object.
263  const TNode<NativeContext> native_context = LoadNativeContext(context);
264  const TNode<Map> function_map =
265  CAST(LoadContextElementNoCell(native_context, function_map_index));
266 
267  // Create a new closure from the given function info in new space
268  TNode<IntPtrT> instance_size_in_bytes =
269  TimesTaggedSize(LoadMapInstanceSizeInWords(function_map));
270  TNode<HeapObject> result = Allocate(instance_size_in_bytes);
271  StoreMapNoWriteBarrier(result, function_map);
272  InitializeJSObjectBodyNoSlackTracking(result, function_map,
273  instance_size_in_bytes,
274  JSFunction::kSizeWithoutPrototype);
275 
276  // Initialize the rest of the function.
277  StoreObjectFieldRoot(result, JSObject::kPropertiesOrHashOffset,
278  RootIndex::kEmptyFixedArray);
279  StoreObjectFieldRoot(result, JSObject::kElementsOffset,
280  RootIndex::kEmptyFixedArray);
281  {
282  // Set function prototype if necessary.
283  Label done(this), init_prototype(this);
284  Branch(IsFunctionWithPrototypeSlotMap(function_map), &init_prototype,
285  &done);
286 
287  BIND(&init_prototype);
288  StoreObjectFieldRoot(result, JSFunction::kPrototypeOrInitialMapOffset,
289  RootIndex::kTheHoleValue);
290  Goto(&done);
291  BIND(&done);
292  }
293 
294  static_assert(JSFunction::kSizeWithoutPrototype == 7 * kTaggedSize);
295  StoreObjectFieldNoWriteBarrier(result, JSFunction::kFeedbackCellOffset,
296  feedback_cell);
297  StoreObjectFieldNoWriteBarrier(result, JSFunction::kSharedFunctionInfoOffset,
298  shared_function_info);
299  StoreObjectFieldNoWriteBarrier(result, JSFunction::kContextOffset, context);
300 #ifdef V8_ENABLE_LEAPTIERING
301  TNode<JSDispatchHandleT> dispatch_handle = LoadObjectField<JSDispatchHandleT>(
302  feedback_cell, FeedbackCell::kDispatchHandleOffset);
303  CSA_DCHECK(this,
304  Word32NotEqual(dispatch_handle,
305  Int32Constant(kNullJSDispatchHandle.value())));
306  StoreObjectFieldNoWriteBarrier(result, JSFunction::kDispatchHandleOffset,
307  dispatch_handle);
308 #else
309  TNode<Code> lazy_builtin =
310  HeapConstantNoHole(BUILTIN_CODE(isolate(), CompileLazy));
311  StoreCodePointerField(result, JSFunction::kCodeOffset, lazy_builtin);
312 #endif // V8_ENABLE_LEAPTIERING
313  Return(result);
314 }

348 TNode<JSObject> ConstructorBuiltinsAssembler::FastNewObject(
349  TNode<Context> context, TNode<JSFunction> target,
350  TNode<JSReceiver> new_target, Label* call_runtime) {
351  // Verify that the new target is a JSFunction.
352  Label end(this);
353  TNode<JSFunction> new_target_func =
354  HeapObjectToJSFunctionWithPrototypeSlot(new_target, call_runtime);
355  // Fast path.
356 
357  // Load the initial map and verify that it's in fact a map.
358  TNode<Object> initial_map_or_proto =
359  LoadJSFunctionPrototypeOrInitialMap(new_target_func);
360  GotoIf(TaggedIsSmi(initial_map_or_proto), call_runtime);
361  GotoIf(DoesntHaveInstanceType(CAST(initial_map_or_proto), MAP_TYPE),
362  call_runtime);
363  TNode<Map> initial_map = CAST(initial_map_or_proto);
364 
365  // Fall back to runtime if the target differs from the new target's
366  // initial map constructor.
367  TNode<Object> new_target_constructor = LoadObjectField(
368  initial_map, Map::kConstructorOrBackPointerOrNativeContextOffset);
369  GotoIf(TaggedNotEqual(target, new_target_constructor), call_runtime);
370 
371  TVARIABLE(HeapObject, properties);
372 
373  Label instantiate_map(this), allocate_properties(this);
374  GotoIf(IsDictionaryMap(initial_map), &allocate_properties);
375  {
376  properties = EmptyFixedArrayConstant();
377  Goto(&instantiate_map);
378  }
379  BIND(&allocate_properties);
380  {
381  properties =
382  AllocatePropertyDictionary(PropertyDictionary::kInitialCapacity);
383  Goto(&instantiate_map);
384  }
385 
386  BIND(&instantiate_map);
387  return AllocateJSObjectFromMap(initial_map, properties.value(), std::nullopt,
388  AllocationFlag::kNone, kWithSlackTracking);
389 }

391 TNode<Context> ConstructorBuiltinsAssembler::FastNewFunctionContext(
392  TNode<ScopeInfo> scope_info, TNode<Uint32T> slots, TNode<Context> context,
393  ScopeType scope_type, ContextMode context_mode) {
394  TNode<IntPtrT> slots_intptr = Signed(ChangeUint32ToWord(slots));
395  TNode<IntPtrT> size = ElementOffsetFromIndex(slots_intptr, PACKED_ELEMENTS,
396  Context::kTodoHeaderSize);
397 
398  // Create a new closure from the given function info in new space
399  TNode<Context> function_context =
400  UncheckedCast<Context>(AllocateInNewSpace(size));
401 
402  TNode<NativeContext> native_context = LoadNativeContext(context);
403  Context::Field index;
404  switch (scope_type) {
405  case EVAL_SCOPE:
406  index = Context::EVAL_CONTEXT_MAP_INDEX;
407  break;
408  case FUNCTION_SCOPE:
409  index = Context::FUNCTION_CONTEXT_MAP_INDEX;
410  break;
411  default:
412  UNREACHABLE();
413  }
414  TNode<Map> map = CAST(LoadContextElementNoCell(native_context, index));
415  // Set up the header.
416  StoreMapNoWriteBarrier(function_context, map);
417  TNode<IntPtrT> min_context_slots = IntPtrConstant(Context::MIN_CONTEXT_SLOTS);
418  // TODO(ishell): for now, length also includes MIN_CONTEXT_SLOTS.
419  TNode<IntPtrT> length = IntPtrAdd(slots_intptr, min_context_slots);
420  StoreObjectFieldNoWriteBarrier(function_context, Context::kLengthOffset,
421  SmiTag(length));
422  StoreObjectFieldNoWriteBarrier(function_context, Context::kScopeInfoOffset,
423  scope_info);
424  StoreObjectFieldNoWriteBarrier(function_context, Context::kPreviousOffset,
425  context);
426 
427  // Initialize the varrest of the slots to undefined.
428  TNode<Object> undefined = UndefinedConstant();
429  if (context_mode == ContextMode::kHasContextCells) {
430  TVARIABLE(IntPtrT, offset, IntPtrConstant(Context::kTodoHeaderSize));
431  TNode<IntPtrT> local_count = LoadScopeInfoContextLocalCount(scope_info);
432 
433  Label extension_field_done(this);
434  GotoIfNot(LoadScopeInfoHasExtensionField(scope_info),
435  &extension_field_done);
436  StoreObjectFieldNoWriteBarrier(function_context, Context::kExtensionOffset,
437  undefined);
438  offset = IntPtrAdd(offset.value(), IntPtrConstant(kTaggedSize));
439  Goto(&extension_field_done);
440  BIND(&extension_field_done);
441 
442  CodeStubAssembler::VariableList vars({&offset}, zone());
443  BuildFastLoop<IntPtrT>(
444  vars, IntPtrConstant(0), local_count,
445  [&](TNode<IntPtrT> index) {
446  TNode<Object> init_value =
447  GetFunctionContextSlotInitialValue(scope_info, index);
448  StoreObjectFieldNoWriteBarrier(function_context, offset.value(),
449  init_value);
450  offset = IntPtrAdd(offset.value(), IntPtrConstant(kTaggedSize));
451  },
452  1, LoopUnrollingMode::kYes, IndexAdvanceMode::kPost);
453 
454  Label done(this);
455  GotoIf(IntPtrEqual(offset.value(), size), &done);
456  // We store another undefined for function variables that are stored in the
457  // last slot of the context.
458  StoreObjectFieldNoWriteBarrier(function_context, offset.value(), undefined);
459  // Check that this is indeed the last slot.
460  CSA_DCHECK(this, IntPtrEqual(
461  IntPtrAdd(offset.value(), IntPtrConstant(kTaggedSize)),
462  size));
463  Goto(&done);
464 
465  BIND(&done);
466  return function_context;
467  } else {
468  DCHECK_EQ(context_mode, ContextMode::kNoContextCells);
469  TNode<IntPtrT> start_offset = IntPtrConstant(Context::kTodoHeaderSize);
470  CodeStubAssembler::VariableList vars(0, zone());
471  BuildFastLoop<IntPtrT>(
472  vars, start_offset, size,
473  [=, this](TNode<IntPtrT> offset) {
474  StoreObjectFieldNoWriteBarrier(function_context, offset, undefined);
475  },
476  kTaggedSize, LoopUnrollingMode::kYes, IndexAdvanceMode::kPost);
477  return function_context;
478  }
479 }

481 TNode<JSRegExp> ConstructorBuiltinsAssembler::CreateRegExpLiteral(
482  TNode<HeapObject> maybe_feedback_vector, TNode<TaggedIndex> slot,
483  TNode<Object> pattern, TNode<Smi> flags, TNode<Context> context) {
484  Label call_runtime(this, Label::kDeferred), end(this);
485 
486  GotoIf(IsUndefined(maybe_feedback_vector), &call_runtime);
487 
488  TVARIABLE(JSRegExp, result);
489  TNode<FeedbackVector> feedback_vector = CAST(maybe_feedback_vector);
490  TNode<Object> literal_site =
491  CAST(LoadFeedbackVectorSlot(feedback_vector, slot));
492  GotoIfNot(HasBoilerplate(literal_site), &call_runtime);
493  {
494  static_assert(JSRegExp::kDataOffset == JSObject::kHeaderSize);
495  static_assert(JSRegExp::kSourceOffset ==
496  JSRegExp::kDataOffset + kTaggedSize);
497  static_assert(JSRegExp::kFlagsOffset ==
498  JSRegExp::kSourceOffset + kTaggedSize);
499  static_assert(JSRegExp::kHeaderSize ==
500  JSRegExp::kFlagsOffset + kTaggedSize);
501  static_assert(JSRegExp::kLastIndexOffset == JSRegExp::kHeaderSize);
502  DCHECK_EQ(JSRegExp::Size(), JSRegExp::kLastIndexOffset + kTaggedSize);
503 
504  TNode<RegExpBoilerplateDescription> boilerplate = CAST(literal_site);
505  TNode<HeapObject> new_object = Allocate(JSRegExp::Size());
506 
507  // Initialize Object fields.
508  TNode<JSFunction> regexp_function = CAST(LoadContextElementNoCell(
509  LoadNativeContext(context), Context::REGEXP_FUNCTION_INDEX));
510  TNode<Map> initial_map = CAST(LoadObjectField(
511  regexp_function, JSFunction::kPrototypeOrInitialMapOffset));
512  StoreMapNoWriteBarrier(new_object, initial_map);
513  // Initialize JSReceiver fields.
514  StoreObjectFieldRoot(new_object, JSReceiver::kPropertiesOrHashOffset,
515  RootIndex::kEmptyFixedArray);
516  // Initialize JSObject fields.
517  StoreObjectFieldRoot(new_object, JSObject::kElementsOffset,
518  RootIndex::kEmptyFixedArray);
519  // Initialize JSRegExp fields.
520  StoreTrustedPointerField(
521  new_object, JSRegExp::kDataOffset, kRegExpDataIndirectPointerTag,
522  CAST(LoadTrustedPointerFromObject(
523  boilerplate, RegExpBoilerplateDescription::kDataOffset,
524  kRegExpDataIndirectPointerTag)));
525  StoreObjectFieldNoWriteBarrier(
526  new_object, JSRegExp::kSourceOffset,
527  LoadObjectField(boilerplate,
528  RegExpBoilerplateDescription::kSourceOffset));
529  StoreObjectFieldNoWriteBarrier(
530  new_object, JSRegExp::kFlagsOffset,
531  LoadObjectField(boilerplate,
532  RegExpBoilerplateDescription::kFlagsOffset));
533  StoreObjectFieldNoWriteBarrier(
534  new_object, JSRegExp::kLastIndexOffset,
535  SmiConstant(JSRegExp::kInitialLastIndexValue));
536 
537  result = CAST(new_object);
538  Goto(&end);
539  }
540 
541  BIND(&call_runtime);
542  {
543  result = CAST(CallRuntime(Runtime::kCreateRegExpLiteral, context,
544  maybe_feedback_vector, slot, pattern, flags));
545  Goto(&end);
546  }
547 
548  BIND(&end);
549  return result.value();
550 }

552 TNode<JSArray> ConstructorBuiltinsAssembler::CreateShallowArrayLiteral(
553  TNode<FeedbackVector> feedback_vector, TNode<TaggedIndex> slot,
554  TNode<Context> context, AllocationSiteMode allocation_site_mode,

630 TNode<HeapObject> ConstructorBuiltinsAssembler::CreateShallowObjectLiteral(
631  TNode<AllocationSite> allocation_site, TNode<JSObject> boilerplate,
632  Label* call_runtime, bool bailout_if_dictionary_properties) {
633  TNode<Map> boilerplate_map = LoadMap(boilerplate);
634  CSA_DCHECK(this, IsJSObjectMap(boilerplate_map));
635 
636  TVARIABLE(HeapObject, var_properties);
637  {
638  TNode<Uint32T> bit_field_3 = LoadMapBitField3(boilerplate_map);
639  GotoIf(IsSetWord32<Map::Bits3::IsDeprecatedBit>(bit_field_3), call_runtime);
640  // Directly copy over the property store for dict-mode boilerplates.
641  Label if_dictionary(this), if_fast(this), done(this);
642  Branch(IsSetWord32<Map::Bits3::IsDictionaryMapBit>(bit_field_3),
643  &if_dictionary, &if_fast);
644  BIND(&if_dictionary);
645  {
646  if (bailout_if_dictionary_properties) {
647  Goto(call_runtime);
648  } else {
649  Comment("Copy dictionary properties");
650  if (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
651  var_properties =
652  CopySwissNameDictionary(CAST(LoadSlowProperties(boilerplate)));
653  } else {
654  var_properties = CopyNameDictionary(
655  CAST(LoadSlowProperties(boilerplate)), call_runtime);
656  }
657  // Slow objects have no in-object properties.
658  Goto(&done);
659  }
660  }
661  BIND(&if_fast);
662  {
663  // TODO(cbruni): support copying out-of-object properties.
664  // (CreateObjectFromSlowBoilerplate needs to handle them, too.)
665  TNode<HeapObject> boilerplate_properties =
666  LoadFastProperties(boilerplate);
667  GotoIfNot(IsEmptyFixedArray(boilerplate_properties), call_runtime);
668  var_properties = EmptyFixedArrayConstant();
669  Goto(&done);
670  }
671  BIND(&done);
672  }
673 
674  TVARIABLE(FixedArrayBase, var_elements);
675  {
676  // Copy the elements backing store, assuming that it's flat.
677  Label if_empty_fixed_array(this), if_copy_elements(this), done(this);
678  TNode<FixedArrayBase> boilerplate_elements = LoadElements(boilerplate);
679  Branch(IsEmptyFixedArray(boilerplate_elements), &if_empty_fixed_array,
680  &if_copy_elements);
681 
682  BIND(&if_empty_fixed_array);
683  var_elements = boilerplate_elements;
684  Goto(&done);
685 
686  BIND(&if_copy_elements);
687  CSA_DCHECK(this, Word32BinaryNot(
688  IsFixedCOWArrayMap(LoadMap(boilerplate_elements))));
689  auto flags = ExtractFixedArrayFlag::kAllFixedArrays;
690  var_elements = CloneFixedArray(boilerplate_elements, flags);
691  Goto(&done);
692  BIND(&done);
693  }
694 
695  // Ensure new-space allocation for a fresh JSObject so we can skip write
696  // barriers when copying all object fields.
697  static_assert(JSObject::kMaxInstanceSize < kMaxRegularHeapObjectSize);
698  TNode<IntPtrT> instance_size =
699  TimesTaggedSize(LoadMapInstanceSizeInWords(boilerplate_map));
700  TNode<IntPtrT> aligned_instance_size =
701  AlignToAllocationAlignment(instance_size);
702  TNode<IntPtrT> allocation_size = instance_size;
703  bool needs_allocation_memento = v8_flags.allocation_site_pretenuring;
704  if (needs_allocation_memento) {
705  DCHECK(V8_ALLOCATION_SITE_TRACKING_BOOL);
706  // Prepare for inner-allocating the AllocationMemento.
707  allocation_size = IntPtrAdd(aligned_instance_size,
708  IntPtrConstant(ALIGN_TO_ALLOCATION_ALIGNMENT(
709  sizeof(AllocationMemento))));
710  }
711 
712  TNode<HeapObject> copy =
713  UncheckedCast<HeapObject>(AllocateInNewSpace(allocation_size));
714  {
715  Comment("Initialize Literal Copy");
716  // Initialize Object fields.
717  StoreMapNoWriteBarrier(copy, boilerplate_map);
718  StoreObjectFieldNoWriteBarrier(copy, JSObject::kPropertiesOrHashOffset,
719  var_properties.value());
720  StoreObjectFieldNoWriteBarrier(copy, JSObject::kElementsOffset,
721  var_elements.value());
722  }
723 
724  // Initialize the AllocationMemento before potential GCs due to heap number
725  // allocation when copying the in-object properties.
726  if (needs_allocation_memento) {
727  InitializeAllocationMemento(copy, aligned_instance_size, allocation_site);
728  }
729 
730  {
731  // Copy over in-object properties.
732  Label continue_with_write_barrier(this), done_init(this);
733  TVARIABLE(IntPtrT, offset, IntPtrConstant(JSObject::kHeaderSize));
734  {
735  Comment("Copy in-object properties fast");
736  Label continue_fast(this, &offset);
737  Branch(IntPtrEqual(offset.value(), instance_size), &done_init,
738  &continue_fast);
739  BIND(&continue_fast);
740  TNode<Object> field = LoadObjectField(boilerplate, offset.value());
741  Label store_field(this);
742  GotoIf(TaggedIsSmi(field), &store_field);
743  // TODO(leszeks): Read the field descriptor to decide if this heap
744  // number is mutable or not.
745  GotoIf(IsHeapNumber(CAST(field)), &continue_with_write_barrier);
746  Goto(&store_field);
747  BIND(&store_field);
748  StoreObjectFieldNoWriteBarrier(copy, offset.value(), field);
749  offset = IntPtrAdd(offset.value(), IntPtrConstant(kTaggedSize));
750  Branch(WordNotEqual(offset.value(), instance_size), &continue_fast,
751  &done_init);
752  }
753 
754  // Continue initializing the literal after seeing the first sub-object
755  // potentially causing allocation. In this case we prepare the new literal
756  // by copying all pending fields over from the boilerplate and emit full
757  // write barriers from here on.
758  BIND(&continue_with_write_barrier);
759  {
760  Comment("Copy in-object properties slow");
761  BuildFastLoop<IntPtrT>(
762  offset.value(), instance_size,
763  [=, this](TNode<IntPtrT> offset) {
764  // TODO(ishell): value decompression is not necessary here.
765  TNode<Object> field = LoadObjectField(boilerplate, offset);
766  StoreObjectFieldNoWriteBarrier(copy, offset, field);
767  },
768  kTaggedSize, LoopUnrollingMode::kYes, IndexAdvanceMode::kPost);
769  CopyMutableHeapNumbersInObject(copy, offset.value(), instance_size);
770  Goto(&done_init);
771  }
772  BIND(&done_init);
773  }
774  return copy;
775 }
