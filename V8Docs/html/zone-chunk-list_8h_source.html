21 // A zone-backed hybrid of a vector and a linked list. Use it if you need a
22 // collection that
23 // * needs to grow indefinitely,
24 // * will mostly grow at the back, but may sometimes grow in front as well
25 // (preferably in batches),
26 // * needs to have very low overhead,
27 // * offers forward- and backwards-iteration,
28 // * offers relatively fast seeking,
29 // * offers bidirectional iterators,
30 // * can be rewound without freeing the backing store,
31 // * can be split and joined again efficiently.
32 // This list will maintain a doubly-linked list of chunks. When a chunk is
33 // filled up, a new one gets appended. New chunks appended at the end will
34 // grow in size up to a certain limit to avoid over-allocation and to keep
35 // the zone clean. Chunks may be partially filled. In particular, chunks may
36 // be empty after rewinding, such that they can be reused when inserting
37 // again at a later point in time.

150  // Initial empty state.

153  } else if (empty()) {

165  bool in_empty_tail = false;
166  for (Chunk* chunk = front_; chunk != nullptr; chunk = chunk->next_) {
167  // Chunks from `front_` to `last_nonempty_` (inclusive) are non-empty.
168  DCHECK_EQ(in_empty_tail, chunk->empty());
169  size_check += chunk->size();
170 
171  if (chunk == last_nonempty_) {
172  in_empty_tail = true;
173  }
174  }
175  DCHECK_EQ(size_check, size());
176  }
177  }
178 #endif

228  void Advance(uint32_t amount) {
229  static_assert(!backwards, "Advance only works on forward iterators");
230 
231 #ifdef DEBUG
232  ZoneChunkListIterator clone(*this);
233  for (uint32_t i = 0; i < amount; ++i) {
234  ++clone;
235  }
236 #endif
237 
238  CHECK(!base::bits::UnsignedAddOverflow32(position_, amount, &position_));
239  while (position_ > 0 && position_ >= current_->position_) {
240  auto overshoot = position_ - current_->position_;
241  current_ = current_->next_;
242  position_ = overshoot;
243 
244  DCHECK(position_ == 0 || current_);
245  }
246 
247 #ifdef DEBUG
248  DCHECK_EQ(clone, *this);
249 #endif
250  }

255  static ZoneChunkListIterator Begin(ChunkList* list) {
256  // Forward iterator:
257  if (!backwards) return ZoneChunkListIterator(list->front_, 0);
258 
259  // Backward iterator:
260  if (list->empty()) return End(list);
261 
262  DCHECK(!list->last_nonempty_->empty());
263  return ZoneChunkListIterator(list->last_nonempty_,
264  list->last_nonempty_->position_ - 1);
265  }
266 
267  static ZoneChunkListIterator End(ChunkList* list) {
268  // Backward iterator:
269  if (backwards) return ZoneChunkListIterator(nullptr, 0);
270 
271  // Forward iterator:
272  if (list->empty()) return Begin(list);
273 
274  // NOTE: Decrementing `end()` is not supported if `last_nonempty_->next_`
275  // is nullptr (in that case `Move` will crash on dereference).
276  return ZoneChunkListIterator(list->last_nonempty_->next_, 0);

284  template <bool move_backward>
285  void Move() {
286  if (move_backward) {
287  // Move backwards.
288  if (position_ == 0) {
289  current_ = current_->previous_;
290  position_ = current_ ? current_->position_ - 1 : 0;
291  } else {
292  --position_;
293  }
294  } else {
295  // Move forwards.
296  ++position_;
297  if (position_ >= current_->position_) {
298  current_ = current_->next_;
299  position_ = 0;
300  }
301  }
302  }
303 
304  Chunk* current_;
305  uint32_t position_;

336 template <typename T>
337 void ZoneChunkList<T>::push_back(const T& item) {
338  if (last_nonempty_ == nullptr) {
339  // Initially empty chunk list.
340  front_ = NewChunk(kInitialChunkCapacity);
341  last_nonempty_ = front_;
342  } else if (last_nonempty_->full()) {
343  // If there is an empty chunk following, reuse that, otherwise allocate.
344  if (last_nonempty_->next_ == nullptr) {
345  Chunk* chunk = NewChunk(NextChunkCapacity(last_nonempty_->capacity_));
346  last_nonempty_->next_ = chunk;
347  chunk->previous_ = last_nonempty_;
348  }
349  last_nonempty_ = last_nonempty_->next_;
350  DCHECK(!last_nonempty_->full());
351  }
352 
353  last_nonempty_->items()[last_nonempty_->position_] = item;
354  ++last_nonempty_->position_;
355  ++size_;
356  DCHECK_LE(last_nonempty_->position_, last_nonempty_->capacity_);
357 }

359 template <typename T>
360 void ZoneChunkList<T>::push_front(const T& item) {
361  if (front_ == nullptr) {
362  // Initially empty chunk list.
363  front_ = NewChunk(kInitialChunkCapacity);
364  last_nonempty_ = front_;
365  } else if (front_->full()) {
366  // First chunk at capacity, so prepend a new chunk.
367  DCHECK_NULL(front_->previous_);
368  Chunk* chunk = NewChunk(NextChunkCapacity(front_->capacity_));
369  front_->previous_ = chunk;
370  chunk->next_ = front_;
371  front_ = chunk;
372  }
373  DCHECK(!front_->full());
374 
375  T* end = front_->items() + front_->position_;
376  std::move_backward(front_->items(), end, end + 1);
377  front_->items()[0] = item;
378  ++front_->position_;
379  ++size_;
380  DCHECK_LE(front_->position_, front_->capacity_);
381 }

383 template <typename T>
384 typename ZoneChunkList<T>::SeekResult ZoneChunkList<T>::SeekIndex(
385  size_t index) const {
386  DCHECK_LT(index, size());
387  Chunk* current = front_;
388  while (index >= current->capacity_) {
389  index -= current->capacity_;
390  current = current->next_;
391  }
392  DCHECK_LT(index, current->capacity_);
393  return {current, static_cast<uint32_t>(index)};

396 template <typename T>
397 void ZoneChunkList<T>::Rewind(const size_t limit) {
398  if (limit >= size()) return;
399 
400  SeekResult seek_result = SeekIndex(limit);
401  DCHECK_NOT_NULL(seek_result.chunk_);
402 
403  // Do a partial rewind of the chunk containing the index.
404  seek_result.chunk_->position_ = seek_result.chunk_index_;
405 
406  // Set last_nonempty_ so iterators will work correctly.
407  last_nonempty_ = seek_result.chunk_;
408 
409  // Do full rewind of all subsequent chunks.
410  for (Chunk* current = seek_result.chunk_->next_; current != nullptr;
411  current = current->next_) {
412  current->position_ = 0;
413  }
414 
415  size_ = limit;
416 
417 #ifdef DEBUG
418  Verify();
419 #endif
420 }

437 template <typename T>
438 ZoneChunkList<T> ZoneChunkList<T>::SplitAt(iterator split_begin) {
439  ZoneChunkList<T> result(zone_);
440 
441  // `result` is an empty freshly-constructed list.
442  if (split_begin == end()) return result;
443 
444  // `this` is empty after the split and `result` contains everything.
445  if (split_begin == begin()) {
446  this->swap(result);
447  return result;
448  }
449 
450  // There is at least one element in both `this` and `result`.
451 
452  // Split the chunk.
453  Chunk* split_chunk = split_begin.current_;
454  DCHECK_LE(split_begin.position_, split_chunk->position_);
455  T* chunk_split_begin = split_chunk->items() + split_begin.position_;
456  T* chunk_split_end = split_chunk->items() + split_chunk->position_;
457  uint32_t new_chunk_size =
458  static_cast<uint32_t>(chunk_split_end - chunk_split_begin);
459  uint32_t new_chunk_capacity = std::max(
460  kInitialChunkCapacity, base::bits::RoundUpToPowerOfTwo32(new_chunk_size));
461  CHECK_LE(new_chunk_size, new_chunk_capacity);
462  Chunk* new_chunk = NewChunk(new_chunk_capacity);
463  std::copy(chunk_split_begin, chunk_split_end, new_chunk->items());
464  new_chunk->position_ = new_chunk_size;
465  split_chunk->position_ = split_begin.position_;
466 
467  // Split the linked list.
468  result.front_ = new_chunk;
469  result.last_nonempty_ =
470  (last_nonempty_ == split_chunk) ? new_chunk : last_nonempty_;
471  new_chunk->next_ = split_chunk->next_;
472  if (new_chunk->next_) {
473  new_chunk->next_->previous_ = new_chunk;
474  }
475 
476  last_nonempty_ = split_chunk;
477  split_chunk->next_ = nullptr;
478 
479  // Compute the new size.
480  size_t new_size = 0;
481  for (Chunk* chunk = front_; chunk != split_chunk; chunk = chunk->next_) {
482  DCHECK(!chunk->empty());
483  new_size += chunk->size();
484  }
485  new_size += split_chunk->size();
486  DCHECK_LT(new_size, size());
487  result.size_ = size() - new_size;
488  size_ = new_size;
489 
490 #ifdef DEBUG
491  Verify();
492  result.Verify();
493 #endif
494 
495  return result;
496 }
