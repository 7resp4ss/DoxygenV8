77 int CallSiteInfo::GetLineNumber(DirectHandle<CallSiteInfo> info) {
78  Isolate* isolate = Isolate::Current();
79 #if V8_ENABLE_WEBASSEMBLY
80  if (info->IsWasm() && !info->IsAsmJsWasm()) {
81  return 1;
82  }
83 #endif // V8_ENABLE_WEBASSEMBLY
84  DirectHandle<Script> script;
85  if (GetScript(isolate, info).ToHandle(&script)) {
86  int position = GetSourcePosition(info);
87  int line_number = Script::GetLineNumber(script, position) + 1;
88  if (script->HasSourceURLComment()) {
89  line_number -= script->line_offset();
90  }
91  return line_number;
92  }
93  return Message::kNoLineNumberInfo;
94 }

100 #if V8_ENABLE_WEBASSEMBLY

107  Script::PositionInfo position_info;

112  column_number -= script->column_offset();
113  }
114  return column_number;
115  }
116  return Message::kNoColumnInfo;

224 MaybeHandle<String> FormatEvalOrigin(Isolate* isolate,
225  DirectHandle<Script> script) {
226  Handle<Object> sourceURL(script->GetNameOrSourceURL(), isolate);
227  if (IsString(*sourceURL)) return Cast<String>(sourceURL);
228 
229  IncrementalStringBuilder builder(isolate);
230  builder.AppendCStringLiteral("eval at ");
231  if (script->has_eval_from_shared()) {
232  DirectHandle<SharedFunctionInfo> eval_shared(script->eval_from_shared(),
233  isolate);
234  auto eval_name = SharedFunctionInfo::DebugName(isolate, eval_shared);
235  if (eval_name->length() != 0) {
236  builder.AppendString(eval_name);
237  } else {
238  builder.AppendCStringLiteral("<anonymous>");
239  }
240  if (IsScript(eval_shared->script())) {
241  DirectHandle<Script> eval_script(Cast<Script>(eval_shared->script()),
242  isolate);
243  builder.AppendCStringLiteral(" (");
244  if (eval_script->compilation_type() == Script::CompilationType::kEval) {
245  // Eval script originated from another eval.
246  DirectHandle<String> str;
247  ASSIGN_RETURN_ON_EXCEPTION(isolate, str,
248  FormatEvalOrigin(isolate, eval_script));
249  builder.AppendString(str);
250  } else {
251  // eval script originated from "real" source.
252  DirectHandle<Object> eval_script_name(eval_script->name(), isolate);
253  if (IsString(*eval_script_name)) {
254  builder.AppendString(Cast<String>(eval_script_name));
255  Script::PositionInfo info;
256  if (Script::GetPositionInfo(eval_script,
257  Script::GetEvalPosition(isolate, script),
258  &info, Script::OffsetFlag::kNoOffset)) {
259  builder.AppendCharacter(':');
260  builder.AppendInt(info.line + 1);
261  builder.AppendCharacter(':');
262  builder.AppendInt(info.column + 1);
263  }
264  } else {
265  builder.AppendCStringLiteral("unknown source");
266  }
267  }
268  builder.AppendCharacter(')');
269  }
270  } else {
271  builder.AppendCStringLiteral("<anonymous>");
272  }
273  return indirect_handle(builder.Finish().ToHandleChecked(), isolate);
274 }

291 DirectHandle<PrimitiveHeapObject> CallSiteInfo::GetFunctionName(
292  DirectHandle<CallSiteInfo> info) {
293  Isolate* isolate = Isolate::Current();
294 #if V8_ENABLE_WEBASSEMBLY
295  if (info->IsWasm()) {
296  DirectHandle<WasmModuleObject> module_object(
297  info->GetWasmInstance()->module_object(), isolate);
298  uint32_t func_index = info->GetWasmFunctionIndex();
299  DirectHandle<String> name;
300  if (WasmModuleObject::GetFunctionNameOrNull(isolate, module_object,
301  func_index)
302  .ToHandle(&name)) {
303  return name;
304  }
305  return isolate->factory()->null_value();
306  }
307  if (info->IsBuiltin()) {
308  Builtin builtin = Builtins::FromInt(Cast<Smi>(info->function()).value());
309  return isolate->factory()->NewStringFromAsciiChecked(
310  Builtins::NameForStackTrace(isolate, builtin));
311  }
312 #endif // V8_ENABLE_WEBASSEMBLY
313  DirectHandle<JSFunction> function(Cast<JSFunction>(info->function()),
314  isolate);
315  if (function->shared()->HasBuiltinId()) {
316  Builtin builtin = function->shared()->builtin_id();
317  const char* maybe_known_name =
318  Builtins::NameForStackTrace(isolate, builtin);
319  if (maybe_known_name) {
320  // This is for cases where using the builtin's name allows us to print
321  // e.g. "String.indexOf", instead of just "indexOf" which is what we
322  // would infer below.
323  return isolate->factory()->NewStringFromAsciiChecked(maybe_known_name);
324  }
325  }
326  DirectHandle<String> name = JSFunction::GetDebugName(isolate, function);
327  if (name->length() != 0) return name;
328  if (info->IsEval()) return isolate->factory()->eval_string();
329  return isolate->factory()->null_value();
330 }

358 Tagged<PrimitiveHeapObject> InferMethodNameFromFastObject(
359  Isolate* isolate, Tagged<JSObject> receiver, Tagged<JSFunction> fun,
360  Tagged<PrimitiveHeapObject> name) {
361  ReadOnlyRoots roots(isolate);
362  Tagged<Map> map = receiver->map();
363  Tagged<DescriptorArray> descriptors = map->instance_descriptors(isolate);
364  for (auto i : map->IterateOwnDescriptors()) {
365  Tagged<PrimitiveHeapObject> key = descriptors->GetKey(i);
366  if (IsSymbol(key)) continue;
367  auto details = descriptors->GetDetails(i);
368  if (details.IsDontEnum()) continue;
369  Tagged<Object> value;
370  if (details.location() == PropertyLocation::kField) {
371  auto field_index = FieldIndex::ForPropertyIndex(
372  map, details.field_index(), details.representation());
373  if (field_index.is_double()) continue;
374  value = receiver->RawFastPropertyAt(isolate, field_index);
375  } else {
376  value = descriptors->GetStrongValue(i);
377  }
378  if (value != fun) {
379  if (!IsAccessorPair(value)) continue;
380  auto pair = Cast<AccessorPair>(value);
381  if (pair->getter() != fun && pair->setter() != fun) continue;
382  }
383  if (name != key) {
384  name = IsUndefined(name, isolate)
385  ? key
386  : Tagged<PrimitiveHeapObject>(roots.null_value());
387  }
388  }
389  return name;
390 }

392 template <typename Dictionary>
393 Tagged<PrimitiveHeapObject> InferMethodNameFromDictionary(
394  Isolate* isolate, Tagged<Dictionary> dictionary, Tagged<JSFunction> fun,
395  Tagged<PrimitiveHeapObject> name) {
396  ReadOnlyRoots roots(isolate);
397  for (auto i : dictionary->IterateEntries()) {
398  Tagged<Object> key;
399  if (!dictionary->ToKey(roots, i, &key)) continue;
400  if (IsSymbol(key)) continue;
401  auto details = dictionary->DetailsAt(i);
402  if (details.IsDontEnum()) continue;
403  auto value = dictionary->ValueAt(i);
404  if (value != fun) {
405  if (!IsAccessorPair(value)) continue;
406  auto pair = Cast<AccessorPair>(value);
407  if (pair->getter() != fun && pair->setter() != fun) continue;
408  }
409  if (name != key) {
410  name = IsUndefined(name, isolate)
411  ? Cast<PrimitiveHeapObject>(key)
412  : Tagged<PrimitiveHeapObject>(roots.null_value());
413  }
414  }
415  return name;
416 }

418 Tagged<PrimitiveHeapObject> InferMethodName(Isolate* isolate,
419  Tagged<JSReceiver> receiver,
420  Tagged<JSFunction> fun) {
421  DisallowGarbageCollection no_gc;
422  ReadOnlyRoots roots(isolate);
423  Tagged<PrimitiveHeapObject> name = roots.undefined_value();
424  for (PrototypeIterator it(isolate, receiver, kStartAtReceiver); !it.IsAtEnd();
425  it.Advance()) {
426  auto current = it.GetCurrent();
427  if (!IsJSObject(current)) break;
428  auto object = Cast<JSObject>(current);
429  if (IsAccessCheckNeeded(object)) break;
430  if (object->HasFastProperties()) {
431  name = InferMethodNameFromFastObject(isolate, object, fun, name);
432  } else if (IsJSGlobalObject(object)) {
433  name = InferMethodNameFromDictionary(
434  isolate,
435  Cast<JSGlobalObject>(object)->global_dictionary(kAcquireLoad), fun,
436  name);
437  } else if (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
438  name = InferMethodNameFromDictionary(
439  isolate, object->property_dictionary_swiss(), fun, name);
440  } else {
441  name = InferMethodNameFromDictionary(
442  isolate, object->property_dictionary(), fun, name);
443  }
444  }
445  if (IsUndefined(name, isolate)) return roots.null_value();
446  return name;
447 }

451 // static
452 DirectHandle<Object> CallSiteInfo::GetMethodName(
453  DirectHandle<CallSiteInfo> info) {
454  Isolate* isolate = Isolate::Current();
455  DirectHandle<Object> receiver_or_instance(info->receiver_or_instance(),
456  isolate);
457 #if V8_ENABLE_WEBASSEMBLY
458  if (info->IsWasm()) return isolate->factory()->null_value();
459 #endif // V8_ENABLE_WEBASSEMBLY
460  if (IsNullOrUndefined(*receiver_or_instance, isolate)) {
461  return isolate->factory()->null_value();
462  }
463 
464  DirectHandle<JSFunction> function(Cast<JSFunction>(info->function()),
465  isolate);
466  // Class members initializer function is not a method.
467  if (IsClassMembersInitializerFunction(function->shared()->kind())) {
468  return isolate->factory()->null_value();
469  }
470 
471  DirectHandle<JSReceiver> receiver =
472  Object::ToObject(isolate, receiver_or_instance).ToHandleChecked();
473  Handle<String> name(function->shared()->Name(), isolate);
474  name = String::Flatten(isolate, name);
475 
476  // ES2015 gives getters and setters name prefixes which must
477  // be stripped to find the property name.
478  if (name->HasOneBytePrefix(base::CStrVector("get ")) ||
479  name->HasOneBytePrefix(base::CStrVector("set "))) {
480  name = isolate->factory()->NewProperSubString(name, 4, name->length());
481  } else if (name->length() == 0) {
482  // The function doesn't have a meaningful "name" property, however
483  // the parser does store an inferred name "o.foo" for the common
484  // case of `o.foo = function() {...}`, so see if we can derive a
485  // property name to guess from that.
486  name = handle(function->shared()->inferred_name(), isolate);
487  for (int index = name->length(); --index >= 0;) {
488  if (name->Get(index, isolate) == '.') {
489  name = isolate->factory()->NewProperSubString(name, index + 1,
490  name->length());
491  break;
492  }
493  }
494  }
495 
496  if (name->length() != 0) {
497  PropertyKey key(isolate, Cast<Name>(name));
498  LookupIterator it(isolate, receiver, key,
499  LookupIterator::PROTOTYPE_CHAIN_SKIP_INTERCEPTOR);
500  if (it.state() == LookupIterator::DATA) {
501  if (it.GetDataValue().is_identical_to(function)) {
502  return name;
503  }
504  } else if (it.state() == LookupIterator::ACCESSOR) {
505  DirectHandle<Object> accessors = it.GetAccessors();
506  if (IsAccessorPair(*accessors)) {
507  auto pair = Cast<AccessorPair>(accessors);
508  if (pair->getter() == *function || pair->setter() == *function) {
509  return name;
510  }
511  }
512  }
513  }
514 
515  return direct_handle(InferMethodName(isolate, *receiver, *function), isolate);
516 }

586 // static
587 bool CallSiteInfo::ComputeLocation(DirectHandle<CallSiteInfo> info,
588  MessageLocation* location) {
589  Isolate* isolate = Isolate::Current();
590 #if V8_ENABLE_WEBASSEMBLY
591  if (info->IsWasm()) {
592  int pos = GetSourcePosition(info);
593  Handle<Script> script(info->GetWasmInstance()->module_object()->script(),
594  isolate);
595  *location = MessageLocation(script, pos, pos + 1);
596  return true;
597  }
598  if (info->IsBuiltin()) {
599  return false;
600  }
601 #endif // V8_ENABLE_WEBASSEMBLY
602 
603  Handle<SharedFunctionInfo> shared(info->GetSharedFunctionInfo(), isolate);
604  if (!shared->IsSubjectToDebugging()) return false;
605  Handle<Script> script(Cast<Script>(shared->script()), isolate);
606  if (IsUndefined(script->source())) return false;
607  if (info->flags() & kIsSourcePositionComputed ||
608  (shared->HasBytecodeArray() &&
609  shared->GetBytecodeArray(isolate)->HasSourcePositionTable())) {
610  int pos = GetSourcePosition(info);
611  *location = MessageLocation(script, pos, pos + 1, shared);
612  } else {
613  int code_offset = info->code_offset_or_source_position();
614  *location = MessageLocation(script, shared, code_offset);
615  }
616  return true;
617 }

619 // static
620 int CallSiteInfo::ComputeSourcePosition(DirectHandle<CallSiteInfo> info,
621  int offset) {
622  Isolate* isolate = Isolate::Current();
623 #if V8_ENABLE_WEBASSEMBLY
624 #if V8_ENABLE_DRUMBRAKE
625  if (info->IsWasmInterpretedFrame()) {
626  auto module = info->GetWasmInstance()->module();
627  uint32_t func_index = info->GetWasmFunctionIndex();
628  return wasm::GetSourcePosition(module, func_index, offset,
629  info->IsAsmJsAtNumberConversion());
630  } else {
631 #endif // V8_ENABLE_DRUMBRAKE
632  if (info->IsWasm()) {
633  auto module = info->GetWasmInstance()->trusted_data(isolate)->module();

693 void AppendFileLocation(Isolate* isolate, DirectHandle<CallSiteInfo> frame,
694  IncrementalStringBuilder* builder) {
695  DirectHandle<Object> script_name_or_source_url(
696  frame->GetScriptNameOrSourceURL(), isolate);
697  if (!IsString(*script_name_or_source_url) && frame->IsEval()) {
698  builder->AppendString(Cast<String>(CallSiteInfo::GetEvalOrigin(frame)));
699  // Expecting source position to follow.
700  builder->AppendCStringLiteral(", ");

702 
703  if (IsNonEmptyString(script_name_or_source_url)) {
704  builder->AppendString(Cast<String>(script_name_or_source_url));

706  // Source code does not originate from a file and is not native, but we

713  if (line_number != Message::kNoLineNumberInfo) {
714  builder->AppendCharacter(':');
715  builder->AppendInt(line_number);
716 
717  int column_number = CallSiteInfo::GetColumnNumber(frame);
718  if (column_number != Message::kNoColumnInfo) {
719  builder->AppendCharacter(':');
720  builder->AppendInt(column_number);
721  }
722  }

725 // Returns true iff
726 // 1. the subject ends with '.' + pattern or ' ' + pattern, or
727 // 2. subject == pattern.
728 bool StringEndsWithMethodName(Isolate* isolate, DirectHandle<String> subject,
729  DirectHandle<String> pattern) {
730  if (String::Equals(isolate, subject, pattern)) return true;
731 
732  FlatStringReader subject_reader(isolate, String::Flatten(isolate, subject));
733  FlatStringReader pattern_reader(isolate, String::Flatten(isolate, pattern));
734 
735  int pattern_index = pattern_reader.length() - 1;
736  int subject_index = subject_reader.length() - 1;
737  // Iterate over len + 1.
738  for (uint32_t i = 0; i <= pattern_reader.length(); i++) {
739  if (subject_index < 0) {
740  return false;
741  }
742 
743  const base::uc32 subject_char = subject_reader.Get(subject_index);
744  if (i == pattern_reader.length()) {
745  if (subject_char != '.' && subject_char != ' ') return false;
746  } else if (subject_char != pattern_reader.Get(pattern_index)) {
747  return false;
748  }
749 
750  pattern_index--;
751  subject_index--;
752  }
753 
754  return true;
755 }

757 void AppendMethodCall(Isolate* isolate, DirectHandle<CallSiteInfo> frame,
758  IncrementalStringBuilder* builder) {
759  DirectHandle<Object> type_name = CallSiteInfo::GetTypeName(frame);
760  DirectHandle<Object> method_name = CallSiteInfo::GetMethodName(frame);
761  DirectHandle<Object> function_name = CallSiteInfo::GetFunctionName(frame);
762 
763  if (IsNonEmptyString(function_name)) {
764  DirectHandle<String> function_string = Cast<String>(function_name);
765  if (IsNonEmptyString(type_name)) {
766  DirectHandle<String> type_string = Cast<String>(type_name);
767  if (String::IsIdentifier(isolate, function_string) &&
768  !String::Equals(isolate, function_string, type_string)) {
769  builder->AppendString(type_string);
770  builder->AppendCharacter('.');
771  }
772  }
773  builder->AppendString(function_string);
774 
775  if (IsNonEmptyString(method_name)) {
776  DirectHandle<String> method_string = Cast<String>(method_name);
777  if (!StringEndsWithMethodName(isolate, function_string, method_string)) {
778  builder->AppendCStringLiteral(" [as ");
779  builder->AppendString(method_string);
780  builder->AppendCharacter(']');
781  }
782  }
783  } else {
784  if (IsNonEmptyString(type_name)) {
785  builder->AppendString(Cast<String>(type_name));
786  builder->AppendCharacter('.');
787  }
788  if (IsNonEmptyString(method_name)) {
789  builder->AppendString(Cast<String>(method_name));
790  } else {
791  builder->AppendCStringLiteral("<anonymous>");
792  }
793  }

796 void SerializeJSStackFrame(Isolate* isolate, DirectHandle<CallSiteInfo> frame,
797  IncrementalStringBuilder* builder) {
798  DirectHandle<Object> function_name = CallSiteInfo::GetFunctionName(frame);
799  if (frame->IsAsync()) {
800  builder->AppendCStringLiteral("async ");
801  if (frame->IsPromiseAll() || frame->IsPromiseAny() ||
802  frame->IsPromiseAllSettled()) {
803  builder->AppendCStringLiteral("Promise.");
804  builder->AppendString(Cast<String>(function_name));
805  builder->AppendCStringLiteral(" (index ");
806  builder->AppendInt(CallSiteInfo::GetSourcePosition(frame));
807  builder->AppendCharacter(')');
808  return;
809  }
810  }
811  if (frame->IsMethodCall()) {
812  AppendMethodCall(isolate, frame, builder);
813  } else if (frame->IsConstructor()) {
814  builder->AppendCStringLiteral("new ");
815  if (IsNonEmptyString(function_name)) {
816  builder->AppendString(Cast<String>(function_name));
817  } else {
818  builder->AppendCStringLiteral("<anonymous>");
819  }
820  } else if (IsNonEmptyString(function_name)) {
821  builder->AppendString(Cast<String>(function_name));
822  } else {
823  AppendFileLocation(isolate, frame, builder);
824  return;
825  }
826  builder->AppendCStringLiteral(" (");
827  AppendFileLocation(isolate, frame, builder);
828  builder->AppendCharacter(')');
829 }
