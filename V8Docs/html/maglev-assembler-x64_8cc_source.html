37 template <typename T>
38 void AllocateRaw(MaglevAssembler* masm, Isolate* isolate,
39  RegisterSnapshot register_snapshot, Register object,
40  T size_in_bytes, AllocationType alloc_type,
41  AllocationAlignment alignment) {
42  // TODO(victorgomes): Call the runtime for large object allocation.
43  // TODO(victorgomes): Support double alignment.
44  DCHECK_EQ(alignment, kTaggedAligned);
45  if (v8_flags.single_generation) {
46  alloc_type = AllocationType::kOld;
47  }
48  ExternalReference top = SpaceAllocationTopAddress(isolate, alloc_type);
49  ExternalReference limit = SpaceAllocationLimitAddress(isolate, alloc_type);
50  ZoneLabelRef done(masm);
51  Register new_top = kScratchRegister;
52  // Check if there is enough space.
53  __ Move(object, __ ExternalReferenceAsOperand(top));
54  LoadNewAllocationTop(masm, new_top, object, size_in_bytes);
55  __ cmpq(new_top, __ ExternalReferenceAsOperand(limit));
56  // Otherwise call runtime.
57  __ JumpToDeferredIf(kUnsignedGreaterThanEqual, AllocateSlow<T>,
58  register_snapshot, object, AllocateBuiltin(alloc_type),
59  size_in_bytes, done);
60  // Store new top and tag object.
61  __ movq(__ ExternalReferenceAsOperand(top), new_top);
62  __ addq(object, Immediate(kHeapObjectTag));
63  __ bind(*done);
64 }

97 void MaglevAssembler::StringFromCharCode(RegisterSnapshot register_snapshot,
98  Label* char_code_fits_one_byte,
99  Register result, Register char_code,
100  Register scratch,
101  CharCodeMaskMode mask_mode) {
102  DCHECK_NE(char_code, scratch);
103  ZoneLabelRef done(this);
104  if (mask_mode == CharCodeMaskMode::kMustApplyMask) {
105  andl(char_code, Immediate(0xFFFF));
106  }
107  cmpl(char_code, Immediate(String::kMaxOneByteCharCode));
108  JumpToDeferredIf(
109  above,
110  [](MaglevAssembler* masm, RegisterSnapshot register_snapshot,
111  ZoneLabelRef done, Register result, Register char_code,
112  Register scratch) {
113  // Be sure to save {char_code}. If it aliases with {result}, use
114  // the scratch register.
115  // TODO(victorgomes): This is probably not needed any more, because
116  // we now ensure that results registers don't alias with inputs/temps.
117  // Confirm, and drop this check.
118  if (char_code == result) {
119  // This is guaranteed to be true since we've already checked
120  // char_code != scratch.
121  DCHECK_NE(scratch, result);
122  __ Move(scratch, char_code);
123  char_code = scratch;
124  }
125  DCHECK(!register_snapshot.live_tagged_registers.has(char_code));
126  register_snapshot.live_registers.set(char_code);
127  __ AllocateTwoByteString(register_snapshot, result, 1);
128  __ movw(FieldOperand(result, OFFSET_OF_DATA_START(SeqTwoByteString)),
129  char_code);
130  __ jmp(*done);
131  },
132  register_snapshot, done, result, char_code, scratch);
133  if (char_code_fits_one_byte != nullptr) {
134  bind(char_code_fits_one_byte);
135  }
136  LoadSingleCharacterString(result, char_code, scratch);
137  bind(*done);
138 }

140 void MaglevAssembler::StringCharCodeOrCodePointAt(
141  BuiltinStringPrototypeCharCodeOrCodePointAt::Mode mode,
142  RegisterSnapshot& register_snapshot, Register result, Register string,
143  Register index, Register scratch1, Register scratch2,
144  Label* result_fits_one_byte) {
145  ASM_CODE_COMMENT(this);
146 
147  ZoneLabelRef done(this);
148  Label seq_string;
149  Label cons_string;
150  Label sliced_string;
151 
152  Label* deferred_runtime_call = MakeDeferredCode(
153  [](MaglevAssembler* masm,
154  BuiltinStringPrototypeCharCodeOrCodePointAt::Mode mode,
155  RegisterSnapshot register_snapshot, ZoneLabelRef done, Register result,
156  Register string, Register index) {
157  DCHECK(!register_snapshot.live_registers.has(result));
158  DCHECK(!register_snapshot.live_registers.has(string));
159  DCHECK(!register_snapshot.live_registers.has(index));
160  {
161  SaveRegisterStateForCall save_register_state(masm, register_snapshot);
162  __ Push(string);
163  __ SmiTag(index);
164  __ Push(index);
165  __ Move(kContextRegister, masm->native_context().object());
166  // This call does not throw nor can deopt.
167  if (mode ==
168  BuiltinStringPrototypeCharCodeOrCodePointAt::kCodePointAt) {
169  __ CallRuntime(Runtime::kStringCodePointAt);
170  } else {
171  DCHECK_EQ(mode,
172  BuiltinStringPrototypeCharCodeOrCodePointAt::kCharCodeAt);
173  __ CallRuntime(Runtime::kStringCharCodeAt);
174  }
175  save_register_state.DefineSafepoint();
176  __ SmiUntag(kReturnRegister0);
177  __ Move(result, kReturnRegister0);
178  }
179  __ jmp(*done);
180  },
181  mode, register_snapshot, done, result, string, index);
182 
183  // We might need to try more than one time for ConsString, SlicedString and
184  // ThinString.
185  Label loop;
186  bind(&loop);
187 
188  if (v8_flags.debug_code) {
189  // Check if {string} is a string.
190  AssertNotSmi(string);
191  LoadMap(scratch1, string);
192  CmpInstanceTypeRange(scratch1, scratch1, FIRST_STRING_TYPE,
193  LAST_STRING_TYPE);
194  Check(below_equal, AbortReason::kUnexpectedValue);
195 
196  movl(scratch1, FieldOperand(string, offsetof(String, length_)));
197  cmpl(index, scratch1);
198  Check(below, AbortReason::kUnexpectedValue);
199  }
200 
201 #if V8_STATIC_ROOTS_BOOL
202  Register map = scratch1;
203  LoadMapForCompare(map, string);
204 #else
205  Register instance_type = scratch1;
206  // Get instance type.
207  LoadInstanceType(instance_type, string);
208 #endif
209 
210  {
211 #if V8_STATIC_ROOTS_BOOL
212  using StringTypeRange = InstanceTypeChecker::kUniqueMapRangeOfStringType;
213  // Check the string map ranges in dense increasing order, to avoid needing
214  // to subtract away the lower bound.
215  static_assert(StringTypeRange::kSeqString.first == 0);
216  CompareInt32AndJumpIf(map, StringTypeRange::kSeqString.second,
217  kUnsignedLessThanEqual, &seq_string, Label::kNear);
218 
219  static_assert(StringTypeRange::kSeqString.second + Map::kSize ==
220  StringTypeRange::kExternalString.first);
221  CompareInt32AndJumpIf(map, StringTypeRange::kExternalString.second,
222  kUnsignedLessThanEqual, deferred_runtime_call);
223  // TODO(victorgomes): Add fast path for external strings.
224 
225  static_assert(StringTypeRange::kExternalString.second + Map::kSize ==
226  StringTypeRange::kConsString.first);
227  CompareInt32AndJumpIf(map, StringTypeRange::kConsString.second,
228  kUnsignedLessThanEqual, &cons_string, Label::kNear);
229 
230  static_assert(StringTypeRange::kConsString.second + Map::kSize ==
231  StringTypeRange::kSlicedString.first);
232  CompareInt32AndJumpIf(map, StringTypeRange::kSlicedString.second,
233  kUnsignedLessThanEqual, &sliced_string, Label::kNear);
234 
235  static_assert(StringTypeRange::kSlicedString.second + Map::kSize ==
236  StringTypeRange::kThinString.first);
237  // No need to check for thin strings, they're the last string map.
238  static_assert(StringTypeRange::kThinString.second ==
239  InstanceTypeChecker::kStringMapUpperBound);
240  // Fallthrough to thin string.
241 #else
242  // TODO(victorgomes): Add fast path for external strings.
243  Register representation = kScratchRegister;
244  movl(representation, instance_type);
245  andl(representation, Immediate(kStringRepresentationMask));
246  cmpl(representation, Immediate(kSeqStringTag));
247  j(equal, &seq_string, Label::kNear);
248  cmpl(representation, Immediate(kConsStringTag));
249  j(equal, &cons_string, Label::kNear);
250  cmpl(representation, Immediate(kSlicedStringTag));
251  j(equal, &sliced_string, Label::kNear);
252  cmpl(representation, Immediate(kThinStringTag));
253  j(not_equal, deferred_runtime_call);
254  // Fallthrough to thin string.
255 #endif
256  }
257 
258  // Is a thin string.
259  {
260  LoadTaggedField(string, string, offsetof(ThinString, actual_));
261  jmp(&loop, Label::kNear);
262  }
263 
264  bind(&sliced_string);
265  {
266  Register offset = scratch1;
267  LoadAndUntagTaggedSignedField(offset, string,
268  offsetof(SlicedString, offset_));
269  LoadTaggedField(string, string, offsetof(SlicedString, parent_));
270  addl(index, offset);
271  jmp(&loop, Label::kNear);
272  }
273 
274  bind(&cons_string);
275  {
276  CompareRoot(FieldOperand(string, offsetof(ConsString, second_)),
277  RootIndex::kempty_string);
278  j(not_equal, deferred_runtime_call);
279  LoadTaggedField(string, string, offsetof(ConsString, first_));
280  jmp(&loop, Label::kNear); // Try again with first string.
281  }
282 
283  bind(&seq_string);
284  {
285  Label two_byte_string;
286 #if V8_STATIC_ROOTS_BOOL
287  if (InstanceTypeChecker::kTwoByteStringMapBit == 0) {
288  TestInt32AndJumpIfAllClear(map,
289  InstanceTypeChecker::kStringMapEncodingMask,
290  &two_byte_string, Label::kNear);
291  } else {
292  TestInt32AndJumpIfAnySet(map, InstanceTypeChecker::kStringMapEncodingMask,
293  &two_byte_string, Label::kNear);
294  }
295 #else
296  andl(instance_type, Immediate(kStringEncodingMask));
297  cmpl(instance_type, Immediate(kTwoByteStringTag));
298  j(equal, &two_byte_string, Label::kNear);
299 #endif
300  // The result of one-byte string will be the same for both modes
301  // (CharCodeAt/CodePointAt), since it cannot be the first half of a
302  // surrogate pair.
303  SeqOneByteStringCharCodeAt(result, string, index);
304  jmp(result_fits_one_byte);
305  bind(&two_byte_string);
306 
307  if (mode == BuiltinStringPrototypeCharCodeOrCodePointAt::kCharCodeAt) {
308  movzxwl(result, FieldOperand(string, index, times_2,
309  OFFSET_OF_DATA_START(SeqTwoByteString)));
310  } else {
311  DCHECK_EQ(mode,
312  BuiltinStringPrototypeCharCodeOrCodePointAt::kCodePointAt);
313  Register string_backup = string;
314  if (result == string) {
315  string_backup = scratch2;
316  movq(string_backup, string);
317  }
318  movzxwl(result, FieldOperand(string, index, times_2,
319  OFFSET_OF_DATA_START(SeqTwoByteString)));
320 
321  Register first_code_point = scratch1;
322  movl(first_code_point, result);
323  andl(first_code_point, Immediate(0xfc00));
324  cmpl(first_code_point, Immediate(0xd800));
325  j(not_equal, *done);
326 
327  Register length = scratch1;
328  StringLength(length, string_backup);
329  incl(index);
330  cmpl(index, length);
331  j(greater_equal, *done);
332 
333  Register second_code_point = scratch1;
334  movzxwl(second_code_point,
335  FieldOperand(string_backup, index, times_2,
336  OFFSET_OF_DATA_START(SeqTwoByteString)));
337  movl(scratch2, second_code_point);
338  andl(scratch2, Immediate(0xfc00));
339  cmpl(scratch2, Immediate(0xdc00));
340  j(not_equal, *done);
341 
342  int surrogate_offset = 0x10000 - (0xd800 << 10) - 0xdc00;
343  addl(second_code_point, Immediate(surrogate_offset));
344  shll(result, Immediate(10));
345  addl(result, second_code_point);
346  }
347 
348  // Fallthrough.
349  }
350 
351  bind(*done);
352 
353  if (v8_flags.debug_code) {
354  // We make sure that the user of this macro is not relying in string and
355  // index to not be clobbered.
356  if (result != string) {
357  movl(string, Immediate(0xdeadbeef));
358  }
359  if (result != index) {
360  movl(index, Immediate(0xdeadbeef));
361  }
362  }
363 }

398 void MaglevAssembler::TruncateDoubleToInt32(Register dst, DoubleRegister src) {
399  ZoneLabelRef done(this);
400 
401  Cvttsd2siq(dst, src);
402  // Check whether the Cvt overflowed.
403  cmpq(dst, Immediate(1));
404  JumpToDeferredIf(
405  overflow,
406  [](MaglevAssembler* masm, DoubleRegister src, Register dst,
407  ZoneLabelRef done) {
408  // Push the double register onto the stack as an input argument.
409  __ AllocateStackSpace(kDoubleSize);
410  __ Movsd(MemOperand(rsp, 0), src);
411  __ CallBuiltin(Builtin::kDoubleToI);
412  // DoubleToI sets the result on the stack, pop the result off the stack.
413  // Avoid using `pop` to not mix implicit and explicit rsp updates.
414  __ movl(dst, MemOperand(rsp, 0));
415  __ addq(rsp, Immediate(kDoubleSize));
416  __ jmp(*done);
417  },
418  src, dst, done);
419  bind(*done);
420  // Zero extend the converted value to complete the truncation.
421  movl(dst, dst);
422 }

424 void MaglevAssembler::TryTruncateDoubleToInt32(Register dst, DoubleRegister src,
425  Label* fail) {
426  DCHECK_NE(src, kScratchDoubleReg);
427  // Truncating conversion of the input float64 value to an int32.
428  Cvttpd2dq(kScratchDoubleReg, src);
429  // Convert that int32 value back to float64.
430  Cvtdq2pd(kScratchDoubleReg, kScratchDoubleReg);
431  // Check that the result of the float64->int32->float64 is equal to the input
432  // (i.e. that the conversion didn't truncate).
433  Ucomisd(kScratchDoubleReg, src);
434  JumpIf(parity_even, fail);
435  JumpIf(not_equal, fail);
436 
437  // Move to general purpose register.
438  Cvttsd2si(dst, src);
439 
440  // Check if {input} is -0.
441  Label check_done;
442  cmpl(dst, Immediate(0));
443  j(not_equal, &check_done);
444 
445  // In case of 0, we need to check the high bits for the IEEE -0 pattern.
446  Register high_word32_of_input = kScratchRegister;
447  Pextrd(high_word32_of_input, src, 1);
448  cmpl(high_word32_of_input, Immediate(0));
449  JumpIf(less, fail);
450 
451  bind(&check_done);
452 }

454 void MaglevAssembler::TryTruncateDoubleToUint32(Register dst,
455  DoubleRegister src,
456  Label* fail) {
457  DoubleRegister converted_back = kScratchDoubleReg;
458 
459  // Convert the input float64 value to int64.
460  Cvttsd2siq(dst, src);
461  // Truncate and zero extend to uint32.
462  movl(dst, dst);
463  // Convert that value back to float64.
464  Cvtqsi2sd(converted_back, dst);
465  // Check that the result of the float64->uint32->float64 is equal to the input
466  // (i.e. that the conversion didn't truncate.

469  JumpIf(not_equal, fail);
470 
471  // Check if {input} is -0.
472  Label check_done;

503 void MaglevAssembler::OSRPrologue(Graph* graph) {
504  DCHECK(graph->is_osr());
505  CHECK(!graph->has_recursive_calls());
506 
507  uint32_t source_frame_size =
508  graph->min_maglev_stackslots_for_unoptimized_frame_size();
509 
510  if (V8_ENABLE_SANDBOX_BOOL || v8_flags.debug_code) {
511  movq(kScratchRegister, rbp);
512  subq(kScratchRegister, rsp);
513  cmpq(kScratchRegister,
514  Immediate(source_frame_size * kSystemPointerSize +
515  StandardFrameConstants::kFixedFrameSizeFromFp));
516  SbxCheck(equal, AbortReason::kOsrUnexpectedStackSize);
517  }
518 
519  uint32_t target_frame_size =
520  graph->tagged_stack_slots() + graph->untagged_stack_slots();
521  CHECK_LE(source_frame_size, target_frame_size);
522 
523  if (source_frame_size < target_frame_size) {
524  ASM_CODE_COMMENT_STRING(this, "Growing frame for OSR");
525  Move(kScratchRegister, 0);
526  uint32_t additional_tagged =
527  source_frame_size < graph->tagged_stack_slots()
528  ? graph->tagged_stack_slots() - source_frame_size
529  : 0;
530  for (size_t i = 0; i < additional_tagged; ++i) {
531  pushq(kScratchRegister);
532  }
533  uint32_t size_so_far = source_frame_size + additional_tagged;
534  CHECK_LE(size_so_far, target_frame_size);
535  if (size_so_far < target_frame_size) {
536  subq(rsp,
537  Immediate((target_frame_size - size_so_far) * kSystemPointerSize));
538  }
539  }

542 void MaglevAssembler::Prologue(Graph* graph) {
543  DCHECK(!graph->is_osr());
544 
545  CodeEntry();
546 
547 #ifdef V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
548  AssertInSandboxedExecutionMode();
549 #endif // V8_ENABLE_SANDBOX_HARDWARE_SUPPORT
550 
551  BailoutIfDeoptimized(rbx);
552 
553  if (graph->has_recursive_calls()) {
554  BindJumpTarget(code_gen_state()->entry_label());
555  }
556 
557 #ifndef V8_ENABLE_LEAPTIERING
558  // Tiering support.
559  if (v8_flags.turbofan) {
560  using D = MaglevOptimizeCodeOrTailCallOptimizedCodeSlotDescriptor;
561  Register feedback_vector = D::GetRegisterParameter(D::kFeedbackVector);
562  DCHECK(!AreAliased(feedback_vector, kJavaScriptCallArgCountRegister,
563  kJSFunctionRegister, kContextRegister,
564  kJavaScriptCallNewTargetRegister,
565  kJavaScriptCallDispatchHandleRegister));
566  Move(feedback_vector,
567  compilation_info()->toplevel_compilation_unit()->feedback().object());
568  TailCallBuiltin(Builtin::kMaglevOptimizeCodeOrTailCallOptimizedCodeSlot,
569  CheckFeedbackVectorFlagsNeedsProcessing(feedback_vector,
570  CodeKind::MAGLEV));
571  }
572 #endif // !V8_ENABLE_LEAPTIERING
573 
574  EnterFrame(StackFrame::MAGLEV);
575  // Save arguments in frame.
576  // TODO(leszeks): Consider eliding this frame if we don't make any calls
577  // that could clobber these registers.
578  Push(kContextRegister);
579  Push(kJSFunctionRegister); // Callee's JS function.
580  Push(kJavaScriptCallArgCountRegister); // Actual argument count.
581 
582  // Initialize stack slots.
583  if (graph->tagged_stack_slots() > 0) {
584  ASM_CODE_COMMENT_STRING(this, "Initializing stack slots");
585  // TODO(leszeks): Consider filling with xmm + movdqa instead.
586  Move(rax, 0);
587 
588  // Magic value. Experimentally, an unroll size of 8 doesn't seem any
589  // worse than fully unrolled pushes.
590  const int kLoopUnrollSize = 8;
591  int tagged_slots = graph->tagged_stack_slots();
592  if (tagged_slots < 2 * kLoopUnrollSize) {
593  // If the frame is small enough, just unroll the frame fill
594  // completely.
595  for (int i = 0; i < tagged_slots; ++i) {
596  pushq(rax);
597  }
598  } else {
599  // Extract the first few slots to round to the unroll size.
