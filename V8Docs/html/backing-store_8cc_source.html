54 size_t GetReservationSize(bool has_guard_regions, size_t byte_capacity,
55  bool is_wasm_memory64) {
56 #if V8_TARGET_ARCH_64_BIT && V8_ENABLE_WEBASSEMBLY
57  DCHECK_IMPLIES(is_wasm_memory64 && has_guard_regions,
58  v8_flags.wasm_memory64_trap_handling);
59  if (has_guard_regions) {
60  if (is_wasm_memory64) {
61  DCHECK_LE(byte_capacity, wasm::kMaxMemory64Size);
62  return wasm::kMaxMemory64Size;
63  } else {
64  static_assert(kFullGuardSize32 >= size_t{4} * GB);
65  DCHECK_LE(byte_capacity, size_t{4} * GB);
66  return kFullGuardSize32;
67  }
68  }
69 #else
70  DCHECK(!has_guard_regions);
71 #endif
72 
73  return byte_capacity;
74 }

200 // Allocate a backing store using the array buffer allocator from the embedder.
201 std::unique_ptr<BackingStore> BackingStore::Allocate(
202  Isolate* isolate, size_t byte_length, SharedFlag shared,
203  InitializedFlag initialized) {
204  void* buffer_start = nullptr;
205  auto allocator = isolate->array_buffer_allocator();
206  CHECK_NOT_NULL(allocator);
207  if (byte_length > allocator->MaxAllocationSize()) return {};
208  if (byte_length != 0) {
209  auto counters = isolate->counters();
210  int mb_length = static_cast<int>(byte_length / MB);
211  if (mb_length > 0) {
212  counters->array_buffer_big_allocations()->AddSample(mb_length);
213  }
214  if (shared == SharedFlag::kShared) {
215  counters->shared_array_allocations()->AddSample(mb_length);
216  }
217  auto allocate_buffer = [allocator, initialized](size_t byte_length) {
218  if (initialized == InitializedFlag::kUninitialized) {
219  return allocator->AllocateUninitialized(byte_length);
220  }
221  return allocator->Allocate(byte_length);
222  };
223 
224  buffer_start = isolate->heap()->AllocateExternalBackingStore(
225  allocate_buffer, byte_length);
226 
227  if (buffer_start == nullptr) {
228  // Allocation failed.
229  counters->array_buffer_new_size_failures()->AddSample(mb_length);
230  return {};
231  }
232 #ifdef V8_ENABLE_SANDBOX
233  // Check to catch use of a non-sandbox-compatible ArrayBufferAllocator.
234  CHECK_WITH_MSG(isolate->isolate_group()->sandbox()->Contains(buffer_start),
235  "When the V8 Sandbox is enabled, ArrayBuffer backing stores "
236  "must be allocated inside the sandbox address space. Please "
237  "use an appropriate ArrayBuffer::Allocator to allocate "
238  "these buffers, or disable the sandbox.");
239 #endif
240  }
241 
242  PageAllocator* page_allocator =
243  isolate->isolate_group()->GetBackingStorePageAllocator();
244  auto result = new BackingStore(page_allocator,
245  buffer_start, // start
246  byte_length, // length
247  byte_length, // max length
248  byte_length, // capacity
249  shared, // shared
250  ResizableFlag::kNotResizable, // resizable
251  false, // is_wasm_memory
252  false, // is_wasm_memory64
253  false, // has_guard_regions
254  false, // custom_deleter
255  false); // empty_deleter
256 
257  TRACE_BS("BS:alloc bs=%p mem=%p (length=%zu)\n", result,
258  result->buffer_start(), byte_length);
259  result->SetAllocatorFromIsolate(isolate);
260  return std::unique_ptr<BackingStore>(result);
261 }

275 std::unique_ptr<BackingStore> BackingStore::TryAllocateAndPartiallyCommitMemory(
276  Isolate* isolate, size_t byte_length, size_t max_byte_length,
277  size_t page_size, size_t initial_pages, size_t maximum_pages,
278  WasmMemoryFlag wasm_memory, SharedFlag shared, bool has_guard_regions) {
279  // Enforce engine limitation on the maximum number of pages.
280  if (maximum_pages > std::numeric_limits<size_t>::max() / page_size) {
281  return nullptr;
282  }
283 
284  // Cannot reserve 0 pages on some OSes.
285  if (maximum_pages == 0) maximum_pages = 1;
286 
287  TRACE_BS("BSw:try %zu pages, %zu max\n", initial_pages, maximum_pages);
288 
289 #if V8_ENABLE_WEBASSEMBLY
290  bool is_wasm_memory = wasm_memory != WasmMemoryFlag::kNotWasm;
291  bool is_wasm_memory64 = wasm_memory == WasmMemoryFlag::kWasmMemory64;
292 #else
293  CHECK_EQ(WasmMemoryFlag::kNotWasm, wasm_memory);
294  constexpr bool is_wasm_memory = false;
295  constexpr bool is_wasm_memory64 = false;
296 #endif // V8_ENABLE_WEBASSEMBLY
297  DCHECK_IMPLIES(has_guard_regions, is_wasm_memory);
298 
299  // For accounting purposes, whether a GC was necessary.
300  bool did_retry = false;
301 
302  // A helper to try running a function up to 3 times, executing a GC
303  // if the first and second attempts failed.
304  auto gc_retry = [&](const std::function<bool()>& fn) {
305  for (int i = 0; i < 3; i++) {
306  if (fn()) return true;
307  // Collect garbage and retry.
308  did_retry = true;
309  if (isolate != nullptr) {
310  isolate->heap()->MemoryPressureNotification(
311  MemoryPressureLevel::kCritical, true);
312  }
313  }
314  return false;
315  };
316 
317  size_t byte_capacity = maximum_pages * page_size;
318  size_t reservation_size =
319  GetReservationSize(has_guard_regions, byte_capacity, is_wasm_memory64);
320 
321  //--------------------------------------------------------------------------
322  // Allocate pages (inaccessible by default).
323  //--------------------------------------------------------------------------
324  void* allocation_base = nullptr;
325 #ifdef V8_COMPRESS_POINTERS_IN_MULTIPLE_CAGES
326 #ifdef V8_ENABLE_SANDBOX
327  CHECK_WITH_MSG(isolate || Sandbox::current(),
328 #else
329  CHECK_WITH_MSG(isolate || IsolateGroup::current(),
330 #endif
331  "One must enter an v8::Isolate before allocating resizable "
332  "array backing stores");
333 #endif
334  IsolateGroup* group =
335  isolate ? isolate->isolate_group() : IsolateGroup::current();
336  PageAllocator* page_allocator = group ? group->GetBackingStorePageAllocator()
337  : GetArrayBufferPageAllocator();
338  auto allocate_pages = [&] {
339  allocation_base = AllocatePages(page_allocator, reservation_size, page_size,
340  PageAllocator::kNoAccess);
341  return allocation_base != nullptr;
342  };
343  if (!gc_retry(allocate_pages)) {
344  // Page allocator could not reserve enough pages.
345  if (isolate != nullptr) {
346  RecordStatus(isolate, AllocationStatus::kOtherFailure);
347  }
348  TRACE_BS("BSw:try failed to allocate pages\n");
349  return {};
350  }
351 
352  uint8_t* buffer_start = reinterpret_cast<uint8_t*>(allocation_base);
353 
354  //--------------------------------------------------------------------------
355  // Commit the initial pages (allow read/write).
356  //--------------------------------------------------------------------------
357  size_t committed_byte_length = initial_pages * page_size;
358  auto commit_memory = [&] {
359  return committed_byte_length == 0 ||
360  SetPermissions(page_allocator, buffer_start, committed_byte_length,
361  PageAllocator::kReadWrite);
362  };
363  if (!gc_retry(commit_memory)) {
364  TRACE_BS("BSw:try failed to set permissions (%p, %zu)\n", buffer_start,
365  committed_byte_length);
366  FreePages(page_allocator, allocation_base, reservation_size);
367  // SetPermissions put us over the process memory limit.
368  // We return an empty result so that the caller can throw an exception.
369  return {};
370  }
371 
372  if (isolate != nullptr) {
373  RecordStatus(isolate, did_retry ? AllocationStatus::kSuccessAfterRetry
374  : AllocationStatus::kSuccess);
375  }
376 
377  ResizableFlag resizable =
378  is_wasm_memory ? ResizableFlag::kNotResizable : ResizableFlag::kResizable;
379 
380  auto result = new BackingStore(page_allocator,
381  buffer_start, // start
382  byte_length, // length
383  max_byte_length, // max_byte_length
384  byte_capacity, // capacity
385  shared, // shared
386  resizable, // resizable
387  is_wasm_memory, // is_wasm_memory
388  is_wasm_memory64, // is_wasm_memory64
389  has_guard_regions, // has_guard_regions
390  false, // custom_deleter
391  false); // empty_deleter
392  TRACE_BS(
393  "BSw:alloc bs=%p mem=%p (length=%zu, capacity=%zu, reservation=%zu)\n",
394  result, result->buffer_start(), byte_length, byte_capacity,
395  reservation_size);
396 
397  return std::unique_ptr<BackingStore>(result);
398 }

400 #if V8_ENABLE_WEBASSEMBLY
401 // Allocate a backing store for a Wasm memory. Always use the page allocator
402 // and add guard regions.
403 std::unique_ptr<BackingStore> BackingStore::AllocateWasmMemory(
404  Isolate* isolate, size_t initial_pages, size_t maximum_pages,
405  WasmMemoryFlag wasm_memory, SharedFlag shared) {
406  // Wasm pages must be a multiple of the allocation page size.
407  DCHECK_EQ(0, wasm::kWasmPageSize % AllocatePageSize());
408  DCHECK_LE(initial_pages, maximum_pages);
409  DCHECK_LE(maximum_pages, wasm_memory == WasmMemoryFlag::kWasmMemory32
410  ? wasm::kV8MaxWasmMemory32Pages
411  : wasm::kV8MaxWasmMemory64Pages);
412 
413  DCHECK(wasm_memory == WasmMemoryFlag::kWasmMemory32 ||
414  wasm_memory == WasmMemoryFlag::kWasmMemory64);
415 
416  bool is_wasm_memory64 = wasm_memory == WasmMemoryFlag::kWasmMemory64;
417  bool has_guard_regions =
418  trap_handler::IsTrapHandlerEnabled() &&
419  (wasm_memory == WasmMemoryFlag::kWasmMemory32 ||
420  (is_wasm_memory64 && v8_flags.wasm_memory64_trap_handling));
421 
422  auto TryAllocate = [isolate, initial_pages, wasm_memory, shared,
423  has_guard_regions](size_t maximum_pages) {
424  auto result = TryAllocateAndPartiallyCommitMemory(
425  isolate, initial_pages * wasm::kWasmPageSize,
426  maximum_pages * wasm::kWasmPageSize, wasm::kWasmPageSize, initial_pages,
427  maximum_pages, wasm_memory, shared, has_guard_regions);
428  if (result && shared == SharedFlag::kShared) {
429  result->type_specific_data_.shared_wasm_memory_data =
430  new SharedWasmMemoryData();
431  }
432  return result;
433  };
434  auto backing_store = TryAllocate(maximum_pages);
435 
436  if (!backing_store &&
437  !has_guard_regions && // With guard regions we always reserved a fixed
438  // number of pages.
439  maximum_pages - initial_pages >= 4) {
440  // Retry with smaller maximum pages at each retry.
441  auto delta = (maximum_pages - initial_pages) / 4;
442  size_t sizes[] = {maximum_pages - delta, maximum_pages - 2 * delta,
443  maximum_pages - 3 * delta, initial_pages};
444 
445  for (size_t reduced_maximum_pages : sizes) {
446  backing_store = TryAllocate(reduced_maximum_pages);
447  if (backing_store) break;
448  }
449  }
450  return backing_store;
451 }

453 std::unique_ptr<BackingStore> BackingStore::CopyWasmMemory(
454  Isolate* isolate, size_t new_pages, size_t max_pages,
455  WasmMemoryFlag wasm_memory) {
456  // Note that we could allocate uninitialized to save initialization cost here,
457  // but since Wasm memories are allocated by the page allocator, the zeroing
458  // cost is already built-in.
459  auto new_backing_store = BackingStore::AllocateWasmMemory(
460  isolate, new_pages, max_pages, wasm_memory,
461  is_shared() ? SharedFlag::kShared : SharedFlag::kNotShared);
462 
463  if (!new_backing_store ||
464  new_backing_store->has_guard_regions() != has_guard_regions()) {
465  return {};
466  }
467 
468  if (byte_length_ > 0) {
469  // If the allocation was successful, then the new buffer must be at least
470  // as big as the old one.
471  DCHECK_GE(new_pages * wasm::kWasmPageSize, byte_length_);
472  memcpy(new_backing_store->buffer_start(), buffer_start_, byte_length_);
473  }
474 
475  return new_backing_store;
476 }

478 // Try to grow the size of a wasm memory in place, without realloc + copy.
479 std::optional<size_t> BackingStore::GrowWasmMemoryInPlace(Isolate* isolate,
480  size_t delta_pages,
481  size_t max_pages) {
482  // This function grows wasm memory by
483  // * changing the permissions of additional {delta_pages} pages to kReadWrite;
484  // * increment {byte_length_};
485  //
486  // As this code is executed concurrently, the following steps are executed:
487  // 1) Read the current value of {byte_length_};
488  // 2) Change the permission of all pages from {buffer_start_} to
489  // {byte_length_} + {delta_pages} * {page_size} to kReadWrite;
490  // * This operation may be executed racefully. The OS takes care of
491  // synchronization.
492  // 3) Try to update {byte_length_} with a compare_exchange;
493  // 4) Repeat 1) to 3) until the compare_exchange in 3) succeeds;
494  //
495  // The result of this function is the {byte_length_} before growing in pages.
496  // The result of this function appears like the result of an RMW-update on
497  // {byte_length_}, i.e. two concurrent calls to this function will result in
498  // different return values if {delta_pages} != 0.
499  //
500  // Invariants:
501  // * Permissions are always set incrementally, i.e. for any page {b} with
502  // kReadWrite permission, all pages between the first page {a} and page {b}
503  // also have kReadWrite permission.
504  // * {byte_length_} is always lower or equal than the amount of memory with
505  // permissions set to kReadWrite;
506  // * This is guaranteed by incrementing {byte_length_} with a
507  // compare_exchange after changing the permissions.
508  // * This invariant is the reason why we cannot use a fetch_add.
509  DCHECK(is_wasm_memory());
510  max_pages = std::min(max_pages, byte_capacity_ / wasm::kWasmPageSize);
511 
512  // Do a compare-exchange loop, because we also need to adjust page
513  // permissions. Note that multiple racing grows both try to set page
514  // permissions for the entire range (to be RW), so the operating system
515  // should deal with that raciness. We know we succeeded when we can
516  // compare/swap the old length with the new length.
517  size_t old_length = byte_length_.load(std::memory_order_relaxed);
518 
519  if (delta_pages == 0)
520  return {old_length / wasm::kWasmPageSize}; // degenerate grow.
521  if (delta_pages > max_pages) return {}; // would never work.
522 
523  size_t new_length = 0;
524  while (true) {
525  size_t current_pages = old_length / wasm::kWasmPageSize;
526 
527  // Check if we have exceed the supplied maximum.
528  if (current_pages > (max_pages - delta_pages)) return {};
529 
530  new_length = (current_pages + delta_pages) * wasm::kWasmPageSize;
531 
532  // Try to adjust the permissions on the memory.
533  if (!i::SetPermissions(GetPlatformPageAllocator(), buffer_start_,
534  new_length, PageAllocator::kReadWrite)) {
535  // This is a nondeterministic failure; mark as such in the WasmEngine (for
536  // differential fuzzing).
537  wasm::WasmEngine::set_had_nondeterminism();
538  return {};
539  }
540  if (byte_length_.compare_exchange_weak(old_length, new_length,
541  std::memory_order_acq_rel)) {
542  // Successfully updated both the length and permissions.
543  break;
544  }
545  }
546 
547  return {old_length / wasm::kWasmPageSize};
548 }

582 // Commit already reserved memory (for RAB backing stores (not shared)).
583 BackingStore::ResizeOrGrowResult BackingStore::ResizeInPlace(
584  Isolate* isolate, size_t new_byte_length) {
585  size_t page_size = AllocatePageSize();
586  size_t new_committed_pages;
587  bool round_return_value =
588  RoundUpToPageSize(new_byte_length, page_size,
589  JSArrayBuffer::kMaxByteLength, &new_committed_pages);
590  CHECK(round_return_value);
591 
592  size_t new_committed_length = new_committed_pages * page_size;
593  DCHECK_LE(new_byte_length, new_committed_length);
594  DCHECK(!is_shared());
595 
596  if (new_byte_length < byte_length_) {
597  // Zero the memory so that in case the buffer is grown later, we have
598  // zeroed the contents already. This is especially needed for the portion of
599  // the memory we're not going to decommit below (since it belongs to a
600  // committed page). In addition, we don't rely on all platforms always
601  // zeroing decommitted-then-recommitted memory, but zero the memory
602  // explicitly here.
603  memset(reinterpret_cast<uint8_t*>(buffer_start_) + new_byte_length, 0,
604  byte_length_ - new_byte_length);
605 
606  // Check if we can un-commit some pages.
607  size_t old_committed_pages;
608  round_return_value =
609  RoundUpToPageSize(byte_length_, page_size,
610  JSArrayBuffer::kMaxByteLength, &old_committed_pages);
611  CHECK(round_return_value);
612  DCHECK_LE(new_committed_pages, old_committed_pages);
613 
614  if (new_committed_pages < old_committed_pages) {
615  size_t old_committed_length = old_committed_pages * page_size;
616  if (!i::SetPermissions(
617  GetPlatformPageAllocator(),
618  reinterpret_cast<uint8_t*>(buffer_start_) + new_committed_length,
619  old_committed_length - new_committed_length,
620  PageAllocator::kNoAccess)) {
621  return kFailure;
622  }
623  }
624 
625  // Changing the byte length wouldn't strictly speaking be needed, since
626  // the JSArrayBuffer already stores the updated length. This is to keep
627  // the BackingStore and JSArrayBuffer in sync.
628  byte_length_ = new_byte_length;
629  return kSuccess;
630  }
631  if (new_byte_length == byte_length_) {
632  // i::SetPermissions with size 0 fails on some platforms, so special
633  // handling for the case byte_length_ == new_byte_length == 0 is required.
634  return kSuccess;
635  }
636 
637  // Try to adjust the permissions on the memory.
638  if (!i::SetPermissions(GetPlatformPageAllocator(), buffer_start_,
639  new_committed_length, PageAllocator::kReadWrite)) {
640  return kFailure;
641  }
642 
643  byte_length_ = new_byte_length;
644  return kSuccess;
645 }

647 // Commit already reserved memory (for GSAB backing stores (shared)).
648 BackingStore::ResizeOrGrowResult BackingStore::GrowInPlace(
649  Isolate* isolate, size_t new_byte_length) {
650  size_t page_size = AllocatePageSize();
651  size_t new_committed_pages;
652  bool round_return_value =
653  RoundUpToPageSize(new_byte_length, page_size,
654  JSArrayBuffer::kMaxByteLength, &new_committed_pages);
655  CHECK(round_return_value);
656 
657  size_t new_committed_length = new_committed_pages * page_size;
658  DCHECK_LE(new_byte_length, new_committed_length);
659  DCHECK(is_shared());
660  // See comment in GrowWasmMemoryInPlace.
661  // GrowableSharedArrayBuffer.prototype.grow can be called from several
662  // threads. If two threads try to grow() in a racy way, the spec allows the
663  // larger grow to throw also if the smaller grow succeeds first. The
664  // implementation below doesn't throw in that case - instead, it retries and
665  // succeeds. If the larger grow finishes first though, the smaller grow must
666  // throw.
667  size_t old_byte_length = byte_length_.load(std::memory_order_seq_cst);
668  while (true) {
669  if (new_byte_length < old_byte_length) {
670  // The caller checks for the new_byte_length < old_byte_length_ case. This
671  // can only happen if another thread grew the memory after that.
672  return kRace;
673  }
674  if (new_byte_length == old_byte_length) {
675  // i::SetPermissions with size 0 fails on some platforms, so special
676  // handling for the case old_byte_length == new_byte_length == 0 is
677  // required.
678  return kSuccess;
679  }
680 
681  // Try to adjust the permissions on the memory.
682  if (!i::SetPermissions(GetPlatformPageAllocator(), buffer_start_,
683  new_committed_length, PageAllocator::kReadWrite)) {
684  return kFailure;
685  }
686 
687  // compare_exchange_weak updates old_byte_length.
688  if (byte_length_.compare_exchange_weak(old_byte_length, new_byte_length,
689  std::memory_order_seq_cst)) {
690  // Successfully updated both the length and permissions.
691  break;
692  }
693  }
694  return kSuccess;
695 }

802 void GlobalBackingStoreRegistry::Purge(Isolate* isolate) {
803  // We need to keep a reference to all backing stores that are inspected
804  // in the purging loop below. Otherwise, we might get a deadlock
805  // if the temporary backing store reference created in the loop is
806  // the last reference. In that case the destructor of the backing store
807  // may try to take the &impl->mutex_ in order to unregister itself.
808  std::vector<std::shared_ptr<BackingStore>> prevent_destruction_under_lock;
809  GlobalBackingStoreRegistryImpl* impl = GetGlobalBackingStoreRegistryImpl();
810  base::MutexGuard scope_lock(&impl->mutex_);
811  // Purge all entries in the map that refer to the given isolate.
812  for (auto& entry : impl->map_) {
813  auto backing_store = entry.second.lock();
814  prevent_destruction_under_lock.emplace_back(backing_store);
815  if (!backing_store) continue; // skip entries where weak ptr is null
816  CHECK(backing_store->is_wasm_memory());
817  if (!backing_store->is_shared()) continue; // skip non-shared memory
818  SharedWasmMemoryData* shared_data =
819  backing_store->get_shared_wasm_memory_data();
820  // Remove this isolate from the isolates list.
821  std::vector<Isolate*>& isolates = shared_data->isolates_;
822  auto isolates_it = std::find(isolates.begin(), isolates.end(), isolate);
823  if (isolates_it != isolates.end()) {
824  *isolates_it = isolates.back();
825  isolates.pop_back();
826  }
827  DCHECK_EQ(isolates.end(),
828  std::find(isolates.begin(), isolates.end(), isolate));
829  }
830 }
