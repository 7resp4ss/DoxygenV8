152 inline bool AllocationSite::CanTrack(InstanceType type) {
153  if (!V8_ALLOCATION_SITE_TRACKING_BOOL) return false;
154  if (v8_flags.allocation_site_pretenuring) {
155  // TurboFan doesn't care at all about String pretenuring feedback,
156  // so don't bother even trying to track that.
157  return type == JS_ARRAY_TYPE || type == JS_OBJECT_TYPE;
158  }
159  return type == JS_ARRAY_TYPE;
160 }

248 template <AllocationSiteUpdateMode update_or_check>
249 bool AllocationSite::DigestTransitionFeedback(Isolate* isolate,
250  DirectHandle<AllocationSite> site,
251  ElementsKind to_kind) {
252  bool result = false;
253 
254  if (site->PointsToLiteral() && IsJSArray(site->boilerplate())) {
255  DirectHandle<JSArray> boilerplate(Cast<JSArray>(site->boilerplate()),
256  isolate);
257  ElementsKind kind = boilerplate->GetElementsKind();
258  // if kind is holey ensure that to_kind is as well.
259  if (IsHoleyElementsKind(kind)) {
260  to_kind = GetHoleyElementsKind(to_kind);
261  }
262  if (IsMoreGeneralElementsKindTransition(kind, to_kind)) {
263  // If the array is huge, it's not likely to be defined in a local
264  // function, so we shouldn't make new instances of it very often.
265  uint32_t length = 0;
266  CHECK(Object::ToArrayLength(boilerplate->length(), &length));
267  if (length <= kMaximumArrayBytesToPretransition) {
268  if (update_or_check == AllocationSiteUpdateMode::kCheckOnly) {
269  return true;
270  }
271  if (v8_flags.trace_track_allocation_sites) {
272  bool is_nested = site->IsNested();
273  PrintF("AllocationSite: JSArray %p boilerplate %supdated %s->%s\n",
274  reinterpret_cast<void*>(site->ptr()),
275  is_nested ? "(nested)" : " ", ElementsKindToString(kind),
276  ElementsKindToString(to_kind));
277  }
278  CHECK_NE(to_kind, DICTIONARY_ELEMENTS);
279  JSObject::TransitionElementsKind(isolate, boilerplate, to_kind);
280  DependentCode::DeoptimizeDependencyGroups(
281  isolate, *site,
282  DependentCode::kAllocationSiteTransitionChangedGroup);
283  result = true;
284  }
285  }
286  } else {
287  // The AllocationSite is for a constructed Array.
288  ElementsKind kind = site->GetElementsKind();
289  // if kind is holey ensure that to_kind is as well.
290  if (IsHoleyElementsKind(kind)) {
291  to_kind = GetHoleyElementsKind(to_kind);
292  }
293  if (IsMoreGeneralElementsKindTransition(kind, to_kind)) {
294  if (update_or_check == AllocationSiteUpdateMode::kCheckOnly) return true;
295  if (v8_flags.trace_track_allocation_sites) {
296  PrintF("AllocationSite: JSArray %p site updated %s->%s\n",
297  reinterpret_cast<void*>(site->ptr()), ElementsKindToString(kind),
298  ElementsKindToString(to_kind));
299  }
300  site->SetElementsKind(to_kind);
301  DependentCode::DeoptimizeDependencyGroups(
302  isolate, *site, DependentCode::kAllocationSiteTransitionChangedGroup);
303  result = true;
304  }
305  }
306  return result;
307 }
