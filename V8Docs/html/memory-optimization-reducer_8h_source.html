34 inline bool ValueNeedsWriteBarrier(const Graph* graph, const Operation& value,
35  Isolate* isolate) {
36  if (value.Is<Opmask::kBitcastWordPtrToSmi>()) {
37  return false;
38  } else if (const ConstantOp* constant = value.TryCast<ConstantOp>()) {
39  if (constant->kind == ConstantOp::Kind::kHeapObject) {
40  RootIndex root_index;
41  if (isolate->roots_table().IsRootHandle(constant->handle(),
42  &root_index) &&
43  RootsTable::IsImmortalImmovable(root_index)) {
44  return false;
45  }
46  } else if (constant->kind == ConstantOp::Kind::kSmi) {
47  return false;
48  }
49  } else if (const PhiOp* phi = value.TryCast<PhiOp>()) {
50  if (phi->rep == RegisterRepresentation::Tagged()) {
51  return base::any_of(phi->inputs(), [graph, isolate](OpIndex input) {
52  const Operation& input_op = graph->Get(input);
53  // If we have a Phi as the Phi's input, we give up to avoid infinite
54  // recursion.
55  if (input_op.Is<PhiOp>()) return true;
56  return ValueNeedsWriteBarrier(graph, input_op, isolate);
57  });
58  }
59  }
60  return true;
61 }

63 inline const AllocateOp* UnwrapAllocate(const Graph* graph,
64  const Operation* op) {
65  while (true) {
66  if (const AllocateOp* allocate = op->TryCast<AllocateOp>()) {
67  return allocate;
68  } else if (const TaggedBitcastOp* bitcast =
69  op->TryCast<TaggedBitcastOp>()) {
70  op = &graph->Get(bitcast->input());
71  } else if (const WordBinopOp* binop = op->TryCast<WordBinopOp>();
72  binop && binop->kind == any_of(WordBinopOp::Kind::kAdd,
73  WordBinopOp::Kind::kSub)) {
74  op = &graph->Get(binop->left());
75  } else {
76  return nullptr;
77  }
78  }

81 // The main purpose of memory optimization is folding multiple allocations
82 // into one. For this, the first allocation reserves additional space, that is
83 // consumed by subsequent allocations, which only move the allocation top
84 // pointer and are therefore guaranteed to succeed. Another nice side-effect
85 // of allocation folding is that more stores are performed on the most recent
86 // allocation, which allows us to eliminate the write barrier for the store.
87 //
88 // This analysis works by keeping track of the most recent non-folded
89 // allocation, as well as the number of bytes this allocation needs to reserve
90 // to satisfy all subsequent allocations.
91 // We can do write barrier elimination across loops if the loop does not
92 // contain any potentially allocating operations.
93 struct MemoryAnalyzer {
94  enum class AllocationFolding { kDoAllocationFolding, kDontAllocationFolding };
95 
96  PipelineData* data;
97  Zone* phase_zone;
98  const Graph& input_graph;
99  Isolate* isolate_ = data->isolate();
100  AllocationFolding allocation_folding;
101  bool is_wasm;
102  MemoryAnalyzer(PipelineData* data, Zone* phase_zone, const Graph& input_graph,
103  AllocationFolding allocation_folding, bool is_wasm)
104  : data(data),
105  phase_zone(phase_zone),
106  input_graph(input_graph),
107  allocation_folding(allocation_folding),
108  is_wasm(is_wasm) {}

110  struct BlockState {
111  const AllocateOp* last_allocation = nullptr;
112  std::optional<uint32_t> reserved_size = std::nullopt;
113 
114  bool operator!=(const BlockState& other) {
115  return last_allocation != other.last_allocation ||
116  reserved_size != other.reserved_size;
117  }
118  };
119  FixedBlockSidetable<std::optional<BlockState>> block_states{
120  input_graph.block_count(), phase_zone};
121  ZoneAbslFlatHashMap<const AllocateOp*, const AllocateOp*> folded_into{
122  phase_zone};
123  ZoneAbslFlatHashSet<V<None>> skipped_write_barriers{phase_zone};
124  ZoneAbslFlatHashMap<const AllocateOp*, uint32_t> reserved_size{phase_zone};
125  BlockIndex current_block = BlockIndex(0);
126  BlockState state;
127  TurboshaftPipelineKind pipeline_kind = data->pipeline_kind();
128 
129  bool IsPartOfLastAllocation(const Operation* op) {
130  const AllocateOp* allocation = UnwrapAllocate(&input_graph, op);
131  if (allocation == nullptr) return false;
132  if (state.last_allocation == nullptr) return false;
133  if (state.last_allocation->type != AllocationType::kYoung) return false;
134  if (state.last_allocation == allocation) return true;
135  auto it = folded_into.find(allocation);
136  if (it == folded_into.end()) return false;
137  return it->second == state.last_allocation;
138  }

140  bool SkipWriteBarrier(const StoreOp& store) {
141  const Operation& object = input_graph.Get(store.base());
142  const Operation& value = input_graph.Get(store.value());
143 
144  WriteBarrierKind write_barrier_kind = store.write_barrier;
145  if (write_barrier_kind != WriteBarrierKind::kAssertNoWriteBarrier) {
146  // If we have {kAssertNoWriteBarrier}, we cannot skip elimination
147  // checks.
148  if (ShouldSkipOptimizationStep()) return false;
149  }
150  if (IsPartOfLastAllocation(&object)) return true;
151  if (!ValueNeedsWriteBarrier(&input_graph, value, isolate_)) return true;
152  if (v8_flags.disable_write_barriers) return true;
153  if (write_barrier_kind == WriteBarrierKind::kAssertNoWriteBarrier) {
154  std::stringstream str;
155  str << "MemoryOptimizationReducer could not remove write barrier for "
156  "operation\n #"
157  << input_graph.Index(store) << ": " << store.ToString() << "\n";
158  FATAL("%s", str.str().c_str());
159  }
160  return false;
161  }
162 
163  bool IsFoldedAllocation(V<AnyOrNone> op) {
164  return folded_into.count(
165  input_graph.Get(op).template TryCast<AllocateOp>());
166  }
167 
168  std::optional<uint32_t> ReservedSize(V<AnyOrNone> alloc) {
169  if (auto it = reserved_size.find(
170  input_graph.Get(alloc).template TryCast<AllocateOp>());
171  it != reserved_size.end()) {
172  return it->second;
173  }
174  return std::nullopt;
175  }

210  V<None> REDUCE_INPUT_GRAPH(Store)(V<None> ig_index, const StoreOp& store) {
211  if (store.write_barrier != WriteBarrierKind::kAssertNoWriteBarrier) {
212  // We cannot skip this optimization if we have to eliminate a
213  // {kAssertNoWriteBarrier}.
214  if (ShouldSkipOptimizationStep()) {
215  return Next::ReduceInputGraphStore(ig_index, store);
216  }
217  }
218  if (analyzer_->skipped_write_barriers.count(ig_index)) {
219  __ Store(__ MapToNewGraph(store.base()), __ MapToNewGraph(store.index()),
220  __ MapToNewGraph(store.value()), store.kind, store.stored_rep,
221  WriteBarrierKind::kNoWriteBarrier, store.offset,
222  store.element_size_log2,
223  store.maybe_initializing_or_transitioning,
224  store.indirect_pointer_tag());
225  return V<None>::Invalid();
226  }
227  DCHECK_NE(store.write_barrier, WriteBarrierKind::kAssertNoWriteBarrier);
228  return Next::ReduceInputGraphStore(ig_index, store);
229  }

231  V<HeapObject> REDUCE(Allocate)(V<WordPtr> size, AllocationType type,
232  AllocationAlignment alignment) {
233  DCHECK_EQ(type, any_of(AllocationType::kYoung, AllocationType::kOld,
234  AllocationType::kSharedOld));
235 
236 #if V8_ENABLE_WEBASSEMBLY
237  if (type == AllocationType::kSharedOld) {
238  DCHECK_EQ(isolate_, nullptr); // Only possible in wasm.
239  DCHECK(analyzer_->is_wasm);
240  static_assert(std::is_same_v<Smi, BuiltinPtr>, "BuiltinPtr must be Smi");
241  OpIndex allocate_builtin = __ NumberConstant(
242  static_cast<int>(Builtin::kWasmAllocateInSharedHeap));
243  OpIndex allocated = __ Call(
244  allocate_builtin, {size, __ SmiConstant(Smi::FromInt(alignment))},
245  AllocateWasmSharedBuiltinDescriptor());
246  return allocated;
247  }
248 #endif
249  DCHECK_EQ(alignment, kTaggedAligned);
250  if (v8_flags.single_generation && type == AllocationType::kYoung) {
251  type = AllocationType::kOld;
252  }
253 
254  V<WordPtr> top_address;
255  if (isolate_ != nullptr) {
256  top_address = __ ExternalConstant(
257  type == AllocationType::kYoung
258  ? ExternalReference::new_space_allocation_top_address(isolate_)
259  : ExternalReference::old_space_allocation_top_address(isolate_));
260  } else {
261  // Wasm mode: producing isolate-independent code, loading the isolate
262  // address at runtime.
263 #if V8_ENABLE_WEBASSEMBLY
264  V<WasmTrustedInstanceData> instance_data = __ WasmInstanceDataParameter();
265  int top_address_offset =
266  type == AllocationType::kYoung
267  ? WasmTrustedInstanceData::kNewAllocationTopAddressOffset
268  : WasmTrustedInstanceData::kOldAllocationTopAddressOffset;
269  top_address =
270  __ Load(instance_data, LoadOp::Kind::TaggedBase().Immutable(),
271  MemoryRepresentation::UintPtr(), top_address_offset);
272 #else
273  UNREACHABLE();
274 #endif // V8_ENABLE_WEBASSEMBLY
275  }
276 
277  if (analyzer_->IsFoldedAllocation(__ current_operation_origin())) {
278  DCHECK_NE(__ GetVariable(top(type)), V<WordPtr>::Invalid());
279  V<WordPtr> obj_addr = __ GetVariable(top(type));
280  __ SetVariable(top(type), __ WordPtrAdd(__ GetVariable(top(type)), size));
281  __ StoreOffHeap(top_address, __ GetVariable(top(type)),
282  MemoryRepresentation::UintPtr());
283  return __ BitcastWordPtrToHeapObject(
284  __ WordPtrAdd(obj_addr, __ IntPtrConstant(kHeapObjectTag)));
285  }
286 
287  __ SetVariable(top(type), __ LoadOffHeap(top_address,
288  MemoryRepresentation::UintPtr()));
289 
290  V<CallTarget> allocate_builtin;
291  if (!analyzer_->is_wasm) {
292  if (type == AllocationType::kYoung) {
293  allocate_builtin =
294  __ BuiltinCode(Builtin::kAllocateInYoungGeneration, isolate_);
295  } else {
296  allocate_builtin =
297  __ BuiltinCode(Builtin::kAllocateInOldGeneration, isolate_);
298  }
299  } else {
300 #if V8_ENABLE_WEBASSEMBLY
301  // This lowering is used by Wasm, where we compile isolate-independent
302  // code. Builtin calls simply encode the target builtin ID, which will
303  // be patched to the builtin's address later.
304  if (isolate_ == nullptr) {
305  Builtin builtin;
306  if (type == AllocationType::kYoung) {
307  builtin = Builtin::kWasmAllocateInYoungGeneration;
308  } else {
309  builtin = Builtin::kWasmAllocateInOldGeneration;
310  }
311  static_assert(std::is_same_v<Smi, BuiltinPtr>,
312  "BuiltinPtr must be Smi");
313  allocate_builtin = __ NumberConstant(static_cast<int>(builtin));
314  } else {
315  if (type == AllocationType::kYoung) {
316  allocate_builtin =
317  __ BuiltinCode(Builtin::kWasmAllocateInYoungGeneration, isolate_);
318  } else {
319  allocate_builtin =
320  __ BuiltinCode(Builtin::kWasmAllocateInOldGeneration, isolate_);
321  }
322  }
323 #else
324  UNREACHABLE();
325 #endif
326  }
327 
328  Block* call_runtime = __ NewBlock();
329  Block* done = __ NewBlock();
330 
331  V<WordPtr> limit_address = GetLimitAddress(type);
332 
333  // If the allocation size is not statically known or is known to be larger
334  // than kMaxRegularHeapObjectSize, do not update {top(type)} in case of a
335  // runtime call. This is needed because we cannot allocation-fold large and
336  // normal-sized objects.
337  uint64_t constant_size{};
338  if (!__ matcher().MatchIntegralWordConstant(
339  size, WordRepresentation::WordPtr(), &constant_size) ||
340  constant_size > kMaxRegularHeapObjectSize) {
341  Variable result =
342  __ NewLoopInvariantVariable(RegisterRepresentation::Tagged());
343  if (!constant_size) {
344  // Check if we can do bump pointer allocation here.
345  V<WordPtr> top_value = __ GetVariable(top(type));
346  __ SetVariable(result,
347  __ BitcastWordPtrToHeapObject(__ WordPtrAdd(
348  top_value, __ IntPtrConstant(kHeapObjectTag))));
349  V<WordPtr> new_top = __ WordPtrAdd(top_value, size);
350  V<WordPtr> limit =
351  __ LoadOffHeap(limit_address, MemoryRepresentation::UintPtr());
352  __ GotoIfNot(LIKELY(__ UintPtrLessThan(new_top, limit)), call_runtime);
353  __ GotoIfNot(LIKELY(__ UintPtrLessThan(
354  size, __ IntPtrConstant(kMaxRegularHeapObjectSize))),
355  call_runtime);
356  __ SetVariable(top(type), new_top);
357  __ StoreOffHeap(top_address, new_top, MemoryRepresentation::UintPtr());
358  __ Goto(done);
359  }
360  if (constant_size || __ Bind(call_runtime)) {
361  __ SetVariable(
362  result, __ template Call<HeapObject>(allocate_builtin, {size},
363  AllocateBuiltinDescriptor()));
364  __ Goto(done);
365  }
366 
367  __ BindReachable(done);
368  return __ GetVariable(result);
369  }
370 
371  V<WordPtr> reservation_size;
372  if (auto c = analyzer_->ReservedSize(__ current_operation_origin())) {
373  reservation_size = __ UintPtrConstant(*c);
374  } else {
375  reservation_size = size;
376  }
377  // Check if we can do bump pointer allocation here.
378  bool reachable =
379  __ GotoIfNot(__ UintPtrLessThan(
380  size, __ IntPtrConstant(kMaxRegularHeapObjectSize)),
381  call_runtime, BranchHint::kTrue) !=
382  ConditionalGotoStatus::kGotoDestination;
383  if (reachable) {
384  V<WordPtr> limit =
385  __ LoadOffHeap(limit_address, MemoryRepresentation::UintPtr());
386  __ Branch(__ UintPtrLessThan(
387  __ WordPtrAdd(__ GetVariable(top(type)), reservation_size),
388  limit),
389  done, call_runtime, BranchHint::kTrue);
390  }
391 
392  // Call the runtime if bump pointer area exhausted.
393  if (__ Bind(call_runtime)) {
394  V<HeapObject> allocated = __ template Call<HeapObject>(
395  allocate_builtin, {reservation_size}, AllocateBuiltinDescriptor());
396  __ SetVariable(top(type),
397  __ WordPtrSub(__ BitcastHeapObjectToWordPtr(allocated),
398  __ IntPtrConstant(kHeapObjectTag)));
399  __ Goto(done);
400  }
401 
402  __ BindReachable(done);
403  // Compute the new top and write it back.
404  V<WordPtr> obj_addr = __ GetVariable(top(type));
405  __ SetVariable(top(type), __ WordPtrAdd(__ GetVariable(top(type)), size));
406  __ StoreOffHeap(top_address, __ GetVariable(top(type)),
407  MemoryRepresentation::UintPtr());
408  return __ BitcastWordPtrToHeapObject(
409  __ WordPtrAdd(obj_addr, __ IntPtrConstant(kHeapObjectTag)));
410  }

412  V<WordPtr> REDUCE(DecodeExternalPointer)(V<Word32> handle,
413  ExternalPointerTagRange tag_range) {
414 #ifdef V8_ENABLE_SANDBOX
415  // Decode loaded external pointer.
416  V<WordPtr> table;
417  if (isolate_ != nullptr) {
418  // Here we access the external pointer table through an ExternalReference.
419  // Alternatively, we could also hardcode the address of the table since it
420  // is never reallocated. However, in that case we must be able to
421  // guarantee that the generated code is never executed under a different
422  // Isolate, as that would allow access to external objects from different
423  // Isolates. It also would break if the code is serialized/deserialized at
424  // some point.
425  V<WordPtr> table_address =
426  IsSharedExternalPointerType(tag_range)
427  ? __
428  LoadOffHeap(
429  __ ExternalConstant(
430  ExternalReference::
431  shared_external_pointer_table_address_address(
432  isolate_)),
433  MemoryRepresentation::UintPtr())
434  : __ ExternalConstant(
435  ExternalReference::external_pointer_table_address(
436  isolate_));
437  table = __ LoadOffHeap(table_address,
438  Internals::kExternalPointerTableBasePointerOffset,
439  MemoryRepresentation::UintPtr());
440  } else {
441 #if V8_ENABLE_WEBASSEMBLY
442  V<WordPtr> isolate_root = __ LoadRootRegister();
443  if (IsSharedExternalPointerType(tag_range)) {
444  V<WordPtr> table_address =
445  __ Load(isolate_root, LoadOp::Kind::RawAligned(),
446  MemoryRepresentation::UintPtr(),
447  IsolateData::shared_external_pointer_table_offset());
448  table = __ Load(table_address, LoadOp::Kind::RawAligned(),
449  MemoryRepresentation::UintPtr(),
450  Internals::kExternalPointerTableBasePointerOffset);
451  } else {
452  table = __ Load(isolate_root, LoadOp::Kind::RawAligned(),
453  MemoryRepresentation::UintPtr(),
454  IsolateData::external_pointer_table_offset() +
455  Internals::kExternalPointerTableBasePointerOffset);
456  }
457 #else
458  UNREACHABLE();
459 #endif
460  }
461 
462  V<Word32> index =
463  __ Word32ShiftRightLogical(handle, kExternalPointerIndexShift);
464  V<Word64> pointer = __ LoadOffHeap(table, __ ChangeUint32ToUint64(index), 0,
465  MemoryRepresentation::Uint64());
466 
467  // We don't expect to see empty fields here. If this is ever needed,
468  // consider using an dedicated empty value entry for those tags instead
469  // (i.e. an entry with the right tag and nullptr payload).
470  DCHECK(!ExternalPointerCanBeEmpty(tag_range));
471 
472  Block* done = __ NewBlock();
473  if (tag_range.Size() == 1) {
474  // The common and simple case: we expect a specific tag.
475  V<Word64> tag_bits = __ Word64BitwiseAnd(
476  pointer, __ Word64Constant(kExternalPointerTagMask));
477  tag_bits = __ Word64ShiftRightLogical(tag_bits, kExternalPointerTagShift);
478  V<Word32> tag = __ TruncateWord64ToWord32(tag_bits);
479  V<Word32> expected_tag = __ Word32Constant(tag_range.first);
480  __ GotoIf(__ Word32Equal(tag, expected_tag), done, BranchHint::kTrue);
481  // TODO(saelo): it would be nicer to abort here with
482  // AbortReason::kExternalPointerTagMismatch. That might require adding a
483  // builtin call here though, which is not currently available.
484  __ Unreachable();
485  } else {
486  // Not currently supported. Implement once needed.
487  DCHECK_NE(tag_range, kAnyExternalPointerTagRange);
488  UNREACHABLE();
489  }
490  __ BindReachable(done);
491  return __ Word64BitwiseAnd(pointer, kExternalPointerPayloadMask);
492 #else // V8_ENABLE_SANDBOX
493  UNREACHABLE();
494 #endif // V8_ENABLE_SANDBOX
495  }
