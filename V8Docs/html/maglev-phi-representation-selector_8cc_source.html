30 BlockProcessResult MaglevPhiRepresentationSelector::PreProcessBasicBlock(
31  BasicBlock* block) {
32  PreparePhiTaggings(current_block_, block);
33  current_block_ = block;
34 
35  if (block->has_phi()) {
36  auto& phis = *block->phis();
37 
38  auto first_retry = phis.begin();
39  auto end_retry = first_retry;
40  bool any_change = false;
41 
42  for (auto it = phis.begin(); it != phis.end(); ++it) {
43  Phi* phi = *it;
44  switch (ProcessPhi(phi)) {
45  case ProcessPhiResult::kNone:
46  break;
47  case ProcessPhiResult::kChanged:
48  any_change = true;
49  break;
50  case ProcessPhiResult::kRetryOnChange:
51  if (end_retry == first_retry) {
52  first_retry = it;
53  }
54  end_retry = it;
55  ++end_retry;
56  break;
57  }
58  }
59  // Give it one more shot in case an earlier phi has a later one as input.
60  if (any_change) {
61  for (auto it = first_retry; it != end_retry; ++it) {
62  ProcessPhi(*it);
63  }
64  }
65  }
66 
67  return BlockProcessResult::kContinue;
68 }

70 bool MaglevPhiRepresentationSelector::CanHoistUntaggingTo(BasicBlock* block) {
71  if (block->successors().size() != 1) return false;
72  BasicBlock* next = block->successors()[0];
73  // To be able to hoist above resumable loops we would have to be able to
74  // convert during resumption.
75  return !next->state()->is_resumable_loop();
76 }

78 MaglevPhiRepresentationSelector::ProcessPhiResult
79 MaglevPhiRepresentationSelector::ProcessPhi(Phi* node) {
80  if (node->value_representation() != ValueRepresentation::kTagged) {
81  return ProcessPhiResult::kNone;
82  }
83 
84  if (node->is_exception_phi()) {
85  // Exception phis have no inputs (or, at least, none accessible through
86  // `node->input(...)`), so we don't know if the inputs could be untagged or
87  // not, so we just keep those Phis tagged.
88  return ProcessPhiResult::kNone;
89  }
90 
91  TRACE_UNTAGGING(
92  "Considering for untagging: " << PrintNodeLabel(graph_labeller(), node));
93 
94  // {input_mask} represents the ValueRepresentation that {node} could have,
95  // based on the ValueRepresentation of its inputs.
96  ValueRepresentationSet input_reprs;
97  HoistTypeList hoist_untagging;
98  hoist_untagging.resize(node->input_count(), HoistType::kNone);
99 
100  bool has_tagged_phi_input = false;
101  for (int i = 0; i < node->input_count(); i++) {
102  ValueNode* input = node->input(i).node();
103  if (input->Is<SmiConstant>()) {
104  // Could be any representation. We treat such inputs as Int32, since we
105  // later allow ourselves to promote Int32 to Float64 if needed (but we
106  // never downgrade Float64 to Int32, as it could cause deopt loops).
107  input_reprs.Add(ValueRepresentation::kInt32);
108  } else if (Constant* constant = input->TryCast<Constant>()) {
109  if (constant->object().IsHeapNumber()) {
110  input_reprs.Add(ValueRepresentation::kFloat64);
111  } else {
112  // Not a Constant that we can untag.
113  // TODO(leszeks): Consider treating 'undefined' as a potential
114  // HoleyFloat64.
115  input_reprs.RemoveAll();
116  break;
117  }
118  } else if (input->properties().is_conversion()) {
119  DCHECK_EQ(input->input_count(), 1);
120  // The graph builder tags all Phi inputs, so this conversion should
121  // produce a tagged value.
122  DCHECK_EQ(input->value_representation(), ValueRepresentation::kTagged);
123  // If we want to untag {node}, then we'll drop the conversion and use its
124  // input instead.
125  input_reprs.Add(
126  input->input(0).node()->properties().value_representation());
127  } else if (Phi* input_phi = input->TryCast<Phi>()) {
128  if (input_phi->value_representation() != ValueRepresentation::kTagged) {
129  input_reprs.Add(input_phi->value_representation());
130  } else {
131  // An untagged phi is an input of the current phi.
132  if (node->is_backedge_offset(i) &&
133  node->merge_state()->is_loop_with_peeled_iteration()) {
134  // This is the backedge of a loop that has a peeled iteration. We
135  // ignore it and speculatively assume that it will be the same as the
136  // 1st input.
137  DCHECK_EQ(node->input_count(), 2);
138  DCHECK_EQ(i, 1);
139  break;
140  }
141  has_tagged_phi_input = true;
142  input_reprs.RemoveAll();
143  break;
144  }
145  } else {
146  // This is the case where we don't have an existing conversion to attach
147  // the untagging to. In the general case we give up, however in the
148  // special case of the value originating from the loop entry branch, we
149  // can try to hoist untagging out of the loop.
150  if (graph_->is_osr() && v8_flags.maglev_hoist_osr_value_phi_untagging &&
151  input->Is<InitialValue>() && CanHoistUntaggingTo(*graph_->begin())) {
152  hoist_untagging[i] = HoistType::kPrologue;
153  continue;
154  }
155  if (node->is_loop_phi() && !node->is_backedge_offset(i)) {
156  BasicBlock* pred = node->merge_state()->predecessor_at(i);
157  if (CanHoistUntaggingTo(pred)) {
158  auto static_type = input->GetStaticType(graph_->broker());
159  if (NodeTypeIs(static_type, NodeType::kSmi)) {
160  input_reprs.Add(ValueRepresentation::kInt32);
161  hoist_untagging[i] = HoistType::kLoopEntryUnchecked;
162  continue;
163  }
164  if (NodeTypeIs(static_type, NodeType::kNumber)) {
165  input_reprs.Add(ValueRepresentation::kFloat64);
166  hoist_untagging[i] = HoistType::kLoopEntryUnchecked;
167  continue;
168  }
169 
170  // TODO(olivf): Unless we untag OSR values, speculatively untagging
171  // could end us in deopt loops. To enable this by default we need to
172  // add some feedback to be able to back off. Or, ideally find the
173  // respective checked conversion from within the loop to wire up the
174  // feedback collection.
175  if (v8_flags.maglev_speculative_hoist_phi_untagging) {
176  // TODO(olivf): Currently there is no hard guarantee that the phi
177  // merge state has a checkpointed jump.
178  if (pred->control_node()->Is<CheckpointedJump>()) {
179  DCHECK(!node->merge_state()->is_resumable_loop());
180  hoist_untagging[i] = HoistType::kLoopEntry;
181  continue;
182  }
183  }
184  }
185  }
186 
187  // This input is tagged, didn't require a tagging operation to be
188  // tagged and we decided not to hosit; we won't untag {node}.
189  // TODO(dmercadier): this is a bit suboptimal, because some nodes start
190  // tagged, and later become untagged (parameters for instance). Such nodes
191  // will have their untagged alternative passed to {node} without any
192  // explicit conversion, and we thus won't untag {node} even though we
193  // could have.
194  input_reprs.RemoveAll();
195  break;
196  }
197  }
198  ProcessPhiResult default_result = has_tagged_phi_input
199  ? ProcessPhiResult::kRetryOnChange
200  : ProcessPhiResult::kNone;
201 
202  UseRepresentationSet use_reprs;
203  if (node->is_loop_phi() && !node->get_same_loop_uses_repr_hints().empty()) {
204  // {node} is a loop phi that has uses inside the loop; we will tag/untag
205  // based on those uses, ignoring uses after the loop.
206  use_reprs = node->get_same_loop_uses_repr_hints();
207  TRACE_UNTAGGING(" + use_reprs : " << use_reprs << " (same loop only)");
208  } else {
209  use_reprs = node->get_uses_repr_hints();
210  TRACE_UNTAGGING(" + use_reprs : " << use_reprs << " (all uses)");
211  }
212 
213  TRACE_UNTAGGING(" + input_reprs: " << input_reprs);
214 
215  if (use_reprs.contains(UseRepresentation::kTagged) ||
216  use_reprs.contains(UseRepresentation::kUint32) || use_reprs.empty()) {
217  // We don't untag phis that are used as tagged (because we'd have to retag
218  // them later). We also ignore phis that are used as Uint32, because this is
219  // a fairly rare case and supporting it doesn't improve performance all that
220  // much but will increase code complexity.
221  // TODO(dmercadier): consider taking into account where those Tagged uses
222  // are: Tagged uses outside of a loop or for a Return could probably be
223  // ignored.
224  TRACE_UNTAGGING(" => Leaving tagged [incompatible uses]");
225  EnsurePhiInputsTagged(node);
226  return default_result;
227  }
228 
229  if (input_reprs.contains(ValueRepresentation::kTagged) ||
230  input_reprs.contains(ValueRepresentation::kIntPtr) ||
231  input_reprs.empty()) {
232  TRACE_UNTAGGING(" => Leaving tagged [tagged or intptr inputs]");
233  EnsurePhiInputsTagged(node);
234  return default_result;
235  }
236 
237  // Only allowed to have Uint32, Int32, Float64 and HoleyFloat64 inputs from
238  // here.
239  DCHECK_EQ(input_reprs -
240  ValueRepresentationSet({ValueRepresentation::kInt32,
241  ValueRepresentation::kUint32,
242  ValueRepresentation::kFloat64,
243  ValueRepresentation::kHoleyFloat64}),
244  ValueRepresentationSet());
245 
246  DCHECK_EQ(
247  use_reprs - UseRepresentationSet({UseRepresentation::kInt32,
248  UseRepresentation::kTruncatedInt32,
249  UseRepresentation::kFloat64,
250  UseRepresentation::kHoleyFloat64}),
251  UseRepresentationSet());
252 
253  // The rules for untagging are that we can only widen input representations,
254  // i.e. promote Int32 -> Float64 -> HoleyFloat64. We cannot convert from Int32
255  // to Uint32 and vise versa, but both can be converted to Float64.
256  //
257  // Inputs can always be used as more generic uses, and tighter uses always
258  // block more generic inputs. So, we can find the minimum generic use and
259  // maximum generic input, extend inputs upwards, uses downwards, and convert
260  // to the least generic use in the intersection.
261  //
262  // Of interest is the fact that we don't want to insert conversions which
263  // reduce genericity, e.g. Float64->Int32 conversions, since they could deopt
264  // and lead to deopt loops. The above logic ensures that if a Phi has Float64
265  // inputs and Int32 uses, we simply don't untag it.
266  //
267  // TODO(leszeks): The above logic could be implemented with bit magic if the
268  // representations were contiguous.
269 
270  ValueRepresentationSet possible_inputs;
271  if (input_reprs.contains(ValueRepresentation::kHoleyFloat64)) {
272  possible_inputs = {ValueRepresentation::kHoleyFloat64};
273  } else if (input_reprs.contains(ValueRepresentation::kFloat64) ||
274  input_reprs.contains(ValueRepresentation::kUint32)) {
275  possible_inputs = {ValueRepresentation::kFloat64,
276  ValueRepresentation::kHoleyFloat64};
277  } else {
278  DCHECK(input_reprs.contains_only(ValueRepresentation::kInt32));
279  possible_inputs = {ValueRepresentation::kInt32,
280  ValueRepresentation::kFloat64,
281  ValueRepresentation::kHoleyFloat64};
282  }
283 
284  ValueRepresentationSet allowed_inputs_for_uses;
285  if (use_reprs.contains(UseRepresentation::kInt32)) {
286  allowed_inputs_for_uses = {ValueRepresentation::kInt32};
287  } else if (use_reprs.contains(UseRepresentation::kFloat64)) {
288  allowed_inputs_for_uses = {ValueRepresentation::kInt32,
289  ValueRepresentation::kFloat64};
290  } else {
291  DCHECK(!use_reprs.empty() &&
292  use_reprs.is_subset_of({UseRepresentation::kHoleyFloat64,
293  UseRepresentation::kTruncatedInt32}));
294  allowed_inputs_for_uses = {ValueRepresentation::kInt32,
295  ValueRepresentation::kFloat64,
296  ValueRepresentation::kHoleyFloat64};
297  }
298 
299  // When hoisting we must ensure that we don't turn a tagged flowing into
300  // CheckedSmiUntag into a float64. This would cause us to loose the smi check
301  // which in turn can invalidate assumptions on aliasing values.
302  if (hoist_untagging.size() && node->uses_require_31_bit_value()) {
303  TRACE_UNTAGGING(" => Leaving tagged [depends on smi check]");
304  EnsurePhiInputsTagged(node);
305  return default_result;
306  }
307 
308  auto intersection = possible_inputs & allowed_inputs_for_uses;
309 
310  TRACE_UNTAGGING(" + intersection reprs: " << intersection);
311  if (intersection.contains(ValueRepresentation::kInt32) &&
312  use_reprs.contains_any(UseRepresentationSet{
313  UseRepresentation::kInt32, UseRepresentation::kTruncatedInt32})) {
314  TRACE_UNTAGGING(" => Untagging to Int32");
315  ConvertTaggedPhiTo(node, ValueRepresentation::kInt32, hoist_untagging);
316  return ProcessPhiResult::kChanged;
317  } else if (intersection.contains(ValueRepresentation::kFloat64)) {
318  TRACE_UNTAGGING(" => Untagging to kFloat64");
319  ConvertTaggedPhiTo(node, ValueRepresentation::kFloat64, hoist_untagging);
320  return ProcessPhiResult::kChanged;
321  } else if (intersection.contains(ValueRepresentation::kHoleyFloat64)) {
322  TRACE_UNTAGGING(" => Untagging to HoleyFloat64");
323  ConvertTaggedPhiTo(node, ValueRepresentation::kHoleyFloat64,
324  hoist_untagging);
325  return ProcessPhiResult::kChanged;
326  }
327 
328  DCHECK(intersection.empty());
329  // We don't untag the Phi.
330  TRACE_UNTAGGING(" => Leaving tagged [incompatible inputs/uses]");
331  EnsurePhiInputsTagged(node);
332  return default_result;
333 }

335 void MaglevPhiRepresentationSelector::EnsurePhiInputsTagged(Phi* phi) {
336  // Since we are untagging some Phis, it's possible that one of the inputs of
337  // {phi} is an untagged Phi. However, if this function is called, then we've
338  // decided that {phi} is going to stay tagged, and thus, all of its inputs
339  // should be tagged. We'll thus insert tagging operation on the untagged phi
340  // inputs of {phi}.
341 
342  const int skip_backedge = phi->is_loop_phi() ? 1 : 0;
343  for (int i = 0; i < phi->input_count() - skip_backedge; i++) {
344  ValueNode* input = phi->input(i).node();
345  if (Phi* phi_input = input->TryCast<Phi>()) {
346  phi->change_input(
347  i, EnsurePhiTagged(phi_input, phi->predecessor_at(i),
348  NewNodePosition::kEndOfBlock, nullptr, i));
349  } else {
350  // Inputs of Phis that aren't Phi should always be tagged (except for the
351  // phis untagged by this class, but {phi} isn't one of them).
352  DCHECK(input->is_tagged());
353  }
354  }

359 Opcode GetOpcodeForConversion(ValueRepresentation from, ValueRepresentation to,
360  bool truncating) {
361  DCHECK_NE(from, ValueRepresentation::kTagged);
362  DCHECK_NE(to, ValueRepresentation::kTagged);
363 
364  switch (from) {
365  case ValueRepresentation::kInt32:
366  switch (to) {
367  case ValueRepresentation::kUint32:
368  return Opcode::kCheckedInt32ToUint32;
369  case ValueRepresentation::kFloat64:
370  case ValueRepresentation::kHoleyFloat64:
371  return Opcode::kChangeInt32ToFloat64;
372 
373  case ValueRepresentation::kInt32:
374  case ValueRepresentation::kTagged:
375  case ValueRepresentation::kIntPtr:
376  UNREACHABLE();
377  }
378  case ValueRepresentation::kUint32:
379  switch (to) {
380  case ValueRepresentation::kInt32:
381  return Opcode::kCheckedUint32ToInt32;
382 
383  case ValueRepresentation::kFloat64:
384  case ValueRepresentation::kHoleyFloat64:
385  return Opcode::kChangeUint32ToFloat64;
386 
387  case ValueRepresentation::kUint32:
388  case ValueRepresentation::kTagged:
389  case ValueRepresentation::kIntPtr:
390  UNREACHABLE();
391  }
392  case ValueRepresentation::kFloat64:
393  switch (to) {
394  case ValueRepresentation::kInt32:
395  if (truncating) {
396  return Opcode::kTruncateFloat64ToInt32;
397  }
398  return Opcode::kCheckedTruncateFloat64ToInt32;
399  case ValueRepresentation::kUint32:
400  // The graph builder never inserts Tagged->Uint32 conversions, so we
401  // don't have to handle this case.
402  UNREACHABLE();
403  case ValueRepresentation::kHoleyFloat64:
404 #ifdef V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
405  // When converting to kHoleyFloat64 representation, we need to turn
406  // those NaN patterns that have a special interpretation in
407  // HoleyFloat64 (e.g. undefined and hole) into the canonical NaN so
408  // that they keep representing NaNs in the new representation.
409  return Opcode::kFloat64ToHoleyFloat64;
410 #else
411  return Opcode::kIdentity;
412 #endif // V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
413 
414  case ValueRepresentation::kFloat64:
415  case ValueRepresentation::kTagged:
416  case ValueRepresentation::kIntPtr:
417  UNREACHABLE();
418  }
419  case ValueRepresentation::kHoleyFloat64:
420  switch (to) {
421  case ValueRepresentation::kInt32:
422  // Holes are NaNs, so we can truncate them to int32 same as real NaNs.
423  if (truncating) {
424  return Opcode::kTruncateFloat64ToInt32;
425  }
426  return Opcode::kCheckedTruncateFloat64ToInt32;
427  case ValueRepresentation::kUint32:
428  // The graph builder never inserts Tagged->Uint32 conversions, so we
429  // don't have to handle this case.
430  UNREACHABLE();
431  case ValueRepresentation::kFloat64:
432  return Opcode::kHoleyFloat64ToMaybeNanFloat64;
433 
434  case ValueRepresentation::kHoleyFloat64:
435  case ValueRepresentation::kTagged:
436  case ValueRepresentation::kIntPtr:
437  UNREACHABLE();
438  }
439 
440  case ValueRepresentation::kTagged:
441  case ValueRepresentation::kIntPtr:
442  UNREACHABLE();
443  }
444  UNREACHABLE();
445 }

449 void MaglevPhiRepresentationSelector::ConvertTaggedPhiTo(
450  Phi* phi, ValueRepresentation repr, const HoistTypeList& hoist_untagging) {
451  // We currently only support Int32, Float64, and HoleyFloat64 untagged phis.
452  DCHECK(repr == ValueRepresentation::kInt32 ||
453  repr == ValueRepresentation::kFloat64 ||
454  repr == ValueRepresentation::kHoleyFloat64);
455  phi->change_representation(repr);
456  // Re-initialise register data, since we might have changed from integer
457  // registers to floating registers.
458  phi->InitializeRegisterData();
459 
460  for (int input_index = 0; input_index < phi->input_count(); input_index++) {
461  ValueNode* input = phi->input(input_index).node();
462 #define TRACE_INPUT_LABEL \
463  " @ Input " << input_index << " (" \
464  << PrintNodeLabel(graph_labeller(), input) << ")"
465 
466  if (input->Is<SmiConstant>()) {
467  switch (repr) {
468  case ValueRepresentation::kInt32:
469  TRACE_UNTAGGING(TRACE_INPUT_LABEL << ": Making Int32 instead of Smi");
470  phi->change_input(input_index,
471  graph_->GetInt32Constant(
472  input->Cast<SmiConstant>()->value().value()));
473  break;
474  case ValueRepresentation::kFloat64:
475  case ValueRepresentation::kHoleyFloat64:
476  TRACE_UNTAGGING(TRACE_INPUT_LABEL
477  << ": Making Float64 instead of Smi");
478  phi->change_input(input_index,
479  graph_->GetFloat64Constant(
480  input->Cast<SmiConstant>()->value().value()));
481  break;
482  case ValueRepresentation::kUint32:
483  UNIMPLEMENTED();
484  default:
485  UNREACHABLE();
486  }
487  } else if (Constant* constant = input->TryCast<Constant>()) {
488  TRACE_UNTAGGING(TRACE_INPUT_LABEL
489  << ": Making Float64 instead of Constant");
490  DCHECK(constant->object().IsHeapNumber());
491  DCHECK(repr == ValueRepresentation::kFloat64 ||
492  repr == ValueRepresentation::kHoleyFloat64);
493  phi->change_input(input_index,
494  graph_->GetFloat64Constant(
495  constant->object().AsHeapNumber().value()));
496  } else if (input->properties().is_conversion()) {
497  // Unwrapping the conversion.
498  DCHECK_EQ(input->value_representation(), ValueRepresentation::kTagged);
499  // Needs to insert a new conversion.
500  ValueNode* bypassed_input = input->input(0).node();
501  ValueRepresentation from_repr = bypassed_input->value_representation();
502  ValueNode* new_input;
503  if (from_repr == repr) {
504  TRACE_UNTAGGING(TRACE_INPUT_LABEL << ": Bypassing conversion");
505  new_input = bypassed_input;
506  } else {
507  Opcode conv_opcode =
508  GetOpcodeForConversion(from_repr, repr, /*truncating*/ false);
509  switch (conv_opcode) {
510  case Opcode::kChangeInt32ToFloat64: {
511  new_input =
512  GetReplacementForPhiInputConversion<ChangeInt32ToFloat64>(
513  input, phi, input_index);
514  break;
515  }
516  case Opcode::kChangeUint32ToFloat64: {
517  new_input =
518  GetReplacementForPhiInputConversion<ChangeUint32ToFloat64>(
519  input, phi, input_index);
520  break;
521  }
522 #ifdef V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
523  case Opcode::kFloat64ToHoleyFloat64: {
524  new_input =
525  GetReplacementForPhiInputConversion<Float64ToHoleyFloat64>(
526  input, phi, input_index);
527  break;
528  }
529 #endif // V8_ENABLE_EXPERIMENTAL_UNDEFINED_DOUBLE
530  case Opcode::kIdentity:
531  TRACE_UNTAGGING(TRACE_INPUT_LABEL << ": Bypassing conversion");
532  new_input = bypassed_input;
533  break;
534  default:
535  UNREACHABLE();
536  }
537  }
538  phi->change_input(input_index, new_input);
539  } else if (Phi* input_phi = input->TryCast<Phi>()) {
540  ValueRepresentation from_repr = input_phi->value_representation();
541  if (from_repr == ValueRepresentation::kTagged) {
542  // We allow speculative untagging of the backedge for loop phis from
543  // loops that have been peeled.
544  // This can lead to deopt loops (eg, if after the last iteration of a
545  // loop, a loop Phi has a specific representation that it never has in
546  // the loop), but this case should (hopefully) be rare.
547 
548  // We know that we are on the backedge input of a peeled loop, because
549  // if it wasn't the case, then Process(Phi*) would not have decided to
550  // untag this Phi, and this function would not have been called (because
551  // except for backedges of peeled loops, tagged inputs prevent phi
552  // untagging).
553  DCHECK(phi->merge_state()->is_loop_with_peeled_iteration());
554  DCHECK(phi->is_backedge_offset(input_index));
555 
556  DeoptFrame* deopt_frame = phi->merge_state()->backedge_deopt_frame();
557  switch (repr) {
558  case ValueRepresentation::kInt32: {
559  phi->change_input(
560  input_index,
561  AddNodeAtBlockEnd(
562  NodeBase::New<CheckedSmiUntag>(zone(), {input_phi}),
563  phi->predecessor_at(input_index), deopt_frame));
564  break;
565  }
566  case ValueRepresentation::kFloat64: {
567  phi->change_input(
568  input_index,
569  AddNodeAtBlockEnd(
570  NodeBase::New<CheckedNumberOrOddballToFloat64>(
571  zone(), {input_phi},
572  TaggedToFloat64ConversionType::kOnlyNumber),
573  phi->predecessor_at(input_index), deopt_frame));
574  break;
575  }
576  case ValueRepresentation::kHoleyFloat64: {
577  phi->change_input(
578  input_index,
579  AddNodeAtBlockEnd(
580  NodeBase::New<CheckedNumberOrOddballToHoleyFloat64>(
581  zone(), {input_phi}),
582  phi->predecessor_at(input_index), deopt_frame));
583  break;
584  }
585  case ValueRepresentation::kTagged:
586  case ValueRepresentation::kIntPtr:
587  case ValueRepresentation::kUint32:
588  UNREACHABLE();
589  }
590  TRACE_UNTAGGING(TRACE_INPUT_LABEL
591  << ": Eagerly untagging Phi on backedge");
592  } else if (from_repr != repr &&
593  from_repr == ValueRepresentation::kInt32) {
594  // We allow widening of Int32 inputs to Float64, which can lead to the
595  // current Phi having a Float64 representation but having some Int32
596  // inputs, which will require an Int32ToFloat64 conversion.
597  DCHECK(repr == ValueRepresentation::kFloat64 ||
598  repr == ValueRepresentation::kHoleyFloat64);
599  phi->change_input(input_index,
600  AddNodeAtBlockEnd(NodeBase::New<ChangeInt32ToFloat64>(
601  zone(), {input_phi}),
602  phi->predecessor_at(input_index)));
603  TRACE_UNTAGGING(
604  TRACE_INPUT_LABEL
605  << ": Converting phi input with a ChangeInt32ToFloat64");
606  } else {
607  // We allow Float64 to silently be used as HoleyFloat64.
608  DCHECK_IMPLIES(from_repr != repr,
609  from_repr == ValueRepresentation::kFloat64 &&
610  repr == ValueRepresentation::kHoleyFloat64);
611  TRACE_UNTAGGING(TRACE_INPUT_LABEL
612  << ": Keeping untagged Phi input as-is");
613  }
614  } else if (hoist_untagging[input_index] != HoistType::kNone) {
615  CHECK_EQ(input->value_representation(), ValueRepresentation::kTagged);
616  BasicBlock* block;
617  DeoptFrame* deopt_frame;
618  auto GetDeoptFrame = [](BasicBlock* block) {
619  return &block->control_node()
620  ->Cast<CheckpointedJump>()
621  ->eager_deopt_info()
622  ->top_frame();
623  };
624  switch (hoist_untagging[input_index]) {
625  case HoistType::kLoopEntryUnchecked:
626  block = phi->merge_state()->predecessor_at(input_index);
627  deopt_frame = nullptr;
628  break;
629  case HoistType::kLoopEntry:
630  block = phi->merge_state()->predecessor_at(input_index);
631  deopt_frame = GetDeoptFrame(block);
632  break;
633  case HoistType::kPrologue:
634  block = *graph_->begin();
635  deopt_frame = GetDeoptFrame(block);
636  break;
637  case HoistType::kNone:
638  UNREACHABLE();
639  }
640  // Ensure the hoisted value is actually live at the hoist location.
641  CHECK(input->Is<InitialValue>() ||
642  (phi->is_loop_phi() && !phi->is_backedge_offset(input_index)));
643  ValueNode* untagged;
644  switch (repr) {
645  case ValueRepresentation::kInt32:
646  if (!deopt_frame) {
647  DCHECK(NodeTypeIs(input->GetStaticType(graph_->broker()),
648  NodeType::kSmi));
649  untagged = AddNodeAtBlockEnd(
650  NodeBase::New<UnsafeSmiUntag>(zone(), {input}), block);
651 
652  } else {
653  untagged = AddNodeAtBlockEnd(
654  NodeBase::New<CheckedNumberOrOddballToFloat64>(
655  zone(), {input},
656  TaggedToFloat64ConversionType::kOnlyNumber),
657  block, deopt_frame);
658  untagged =
659  AddNodeAtBlockEnd(NodeBase::New<CheckedTruncateFloat64ToInt32>(
660  zone(), {untagged}),
661  block, deopt_frame);
662  }
663  break;
664  case ValueRepresentation::kFloat64:
665  case ValueRepresentation::kHoleyFloat64:
666  if (!deopt_frame) {
667  DCHECK(NodeTypeIs(input->GetStaticType(graph_->broker()),
668  NodeType::kNumber));
669  untagged = AddNodeAtBlockEnd(
670  NodeBase::New<UncheckedNumberOrOddballToFloat64>(
671  zone(), {input},
672  TaggedToFloat64ConversionType::kOnlyNumber),
673  block);
674  } else {
675  DCHECK(!phi->uses_require_31_bit_value());
676  untagged = AddNodeAtBlockEnd(
677  NodeBase::New<CheckedNumberOrOddballToFloat64>(
678  zone(), {input},
679  TaggedToFloat64ConversionType::kOnlyNumber),
680  block, deopt_frame);
681  if (repr != ValueRepresentation::kHoleyFloat64) {
682  untagged =
683  AddNodeAtBlockEnd(NodeBase::New<CheckedHoleyFloat64ToFloat64>(
684  zone(), {untagged}),
685  block, deopt_frame);
686  }
687  }
688  break;
689  case ValueRepresentation::kTagged:
690  case ValueRepresentation::kUint32:
691  case ValueRepresentation::kIntPtr:
692  UNREACHABLE();
693  }
694  phi->change_input(input_index, untagged);
695  } else {
696  TRACE_UNTAGGING(TRACE_INPUT_LABEL << ": Invalid input for untagged phi");
697  UNREACHABLE();
698  }
699  }

729 void MaglevPhiRepresentationSelector::UpdateUntaggingOfPhi(
730  Phi* phi, ValueNode* old_untagging) {
731  DCHECK_EQ(old_untagging->input_count(), 1);
732  DCHECK(old_untagging->input(0).node()->Is<Phi>());
733 
734  ValueRepresentation from_repr =
735  old_untagging->input(0).node()->value_representation();
736  ValueRepresentation to_repr = old_untagging->value_representation();
737 
738  // Since initially Phis are tagged, it would make not sense for
739  // {old_conversion} to convert a Phi to a Tagged value.
740  DCHECK_NE(to_repr, ValueRepresentation::kTagged);
741  // The graph builder never inserts Tagged->Uint32 conversions (and thus, we
742  // don't handle them in GetOpcodeForCheckedConversion).
743  DCHECK_NE(to_repr, ValueRepresentation::kUint32);
744 
745  if (from_repr == ValueRepresentation::kTagged) {
746  // The Phi hasn't been untagged, so we leave the conversion as it is.
747  return;
748  }
749 
750  if (from_repr == to_repr) {
751  if (from_repr == ValueRepresentation::kInt32) {
752  if (phi->uses_require_31_bit_value() &&
753  old_untagging->Is<CheckedSmiUntag>()) {
754  old_untagging->OverwriteWith<CheckedSmiSizedInt32>();
755  return;
756  }
757  }
758  old_untagging->OverwriteWith<Identity>();
759  return;
760  }
761 
762  if (old_untagging->Is<UnsafeSmiUntag>()) {
763  // UnsafeSmiTag are only inserted when the node is a known Smi. If the
764  // current phi has a Float64/Uint32 representation, then we can safely
765  // truncate it to Int32, because we know that the Float64/Uint32 fits in a
766  // Smi, and therefore in an Int32.
767  if (from_repr == ValueRepresentation::kFloat64 ||
768  from_repr == ValueRepresentation::kHoleyFloat64) {
769  old_untagging->OverwriteWith<UnsafeTruncateFloat64ToInt32>();
770  } else if (from_repr == ValueRepresentation::kUint32) {
771  old_untagging->OverwriteWith<UnsafeTruncateUint32ToInt32>();
772  } else {
773  DCHECK_EQ(from_repr, ValueRepresentation::kInt32);
774  old_untagging->OverwriteWith<Identity>();
775  }
776  return;
777  }
778 
779  // The graph builder inserts 3 kind of Tagged->Int32 conversions that can have
780  // heap number as input: CheckedTruncateNumberToInt32, which truncates its
781  // input (and deopts if it's not a HeapNumber), TruncateNumberToInt32, which
782  // truncates its input (assuming that it's indeed a HeapNumber) and
783  // CheckedSmiTag, which deopts on non-smi inputs. The first 2 cannot deopt if
784  // we have Float64 phi and will happily truncate it, but the 3rd one should
785  // deopt if it cannot be converted without loss of precision.
786  bool conversion_is_truncating_float64 =
787  old_untagging->Is<CheckedTruncateNumberOrOddballToInt32>() ||
788  old_untagging->Is<TruncateNumberOrOddballToInt32>();
789 
790  Opcode needed_conversion = GetOpcodeForConversion(
791  from_repr, to_repr, conversion_is_truncating_float64);
792 
793  if (CheckedNumberOrOddballToFloat64* number_untagging =
794  old_untagging->TryCast<CheckedNumberOrOddballToFloat64>()) {
795  if (from_repr == ValueRepresentation::kHoleyFloat64 &&
796  number_untagging->conversion_type() !=
797  TaggedToFloat64ConversionType::kNumberOrOddball) {
798  // {phi} is a HoleyFloat64 (and thus, it could be a hole), but the
799  // original untagging did not allow holes.
800  needed_conversion = Opcode::kCheckedHoleyFloat64ToFloat64;
801  }
802  }
803 
804  if (needed_conversion != old_untagging->opcode()) {
805  old_untagging->OverwriteWith(needed_conversion);
806  }
807 }

987 ValueNode* MaglevPhiRepresentationSelector::EnsurePhiTagged(
988  Phi* phi, BasicBlock* block, NewNodePosition pos,
989  const ProcessingState* state, std::optional<int> predecessor_index) {
990  DCHECK_IMPLIES(state == nullptr, pos == NewNodePosition::kEndOfBlock);
991 
992  if (phi->value_representation() == ValueRepresentation::kTagged) {
993  return phi;
994  }
995 
996  // Try to find an existing Tagged conversion for {phi} in {phi_taggings_}.
997  if (phi->has_key()) {
998  if (predecessor_index.has_value()) {
999  if (ValueNode* tagging = phi_taggings_.GetPredecessorValue(
1000  phi->key(), predecessor_index.value())) {
1001  return tagging;
1002  }
1003  } else {
1004  if (ValueNode* tagging = phi_taggings_.Get(phi->key())) {
1005  return tagging;
1006  }
1007  }
1008  }
1009 
1010  // We didn't already Tag {phi} on the current path; creating this tagging now.
1011  ValueNode* tagged = nullptr;
1012  switch (phi->value_representation()) {
1013  case ValueRepresentation::kFloat64:
1014  // It's important to use kCanonicalizeSmi for Float64ToTagged, as
1015  // otherwise, we could end up storing HeapNumbers in Smi fields.
1016  tagged = AddNode(
1017  NodeBase::New<Float64ToTagged>(
1018  zone(), {phi}, Float64ToTagged::ConversionMode::kCanonicalizeSmi),
1019  block, pos, state);
1020  break;
1021  case ValueRepresentation::kHoleyFloat64:
1022  // It's important to use kCanonicalizeSmi for HoleyFloat64ToTagged, as
1023  // otherwise, we could end up storing HeapNumbers in Smi fields.
1024  tagged =
1025  AddNode(NodeBase::New<HoleyFloat64ToTagged>(
1026  zone(), {phi},
1027  HoleyFloat64ToTagged::ConversionMode::kCanonicalizeSmi),
1028  block, pos, state);
1029  break;
1030  case ValueRepresentation::kInt32:
1031  tagged = AddNode(NodeBase::New<Int32ToNumber>(zone(), {phi}), block, pos,
1032  state);
1033  break;
1034  case ValueRepresentation::kUint32:
1035  tagged = AddNode(NodeBase::New<Uint32ToNumber>(zone(), {phi}), block, pos,
1036  state);
1037  break;
1038  case ValueRepresentation::kTagged:
1039  // Already handled at the begining of this function.
1040  case ValueRepresentation::kIntPtr:
1041  UNREACHABLE();
1042  }
1043 
1044  if (predecessor_index.has_value()) {
1045  // We inserted the new tagging node in a predecessor of the current block,
1046  // so we shouldn't update the snapshot table for the current block (and we
1047  // can't update it for the predecessor either since its snapshot is sealed).
1048  DCHECK_IMPLIES(block == current_block_,
1049  block->is_loop() && block->successors().size() == 1 &&
1050  block->successors().at(0) == block);
1051  return tagged;
1052  }
1053 
1054  if (phi->has_key()) {
1055  // The Key already existed, but wasn't set on the current path.
1056  phi_taggings_.Set(phi->key(), tagged);
1057  } else {
1058  // The Key didn't already exist, so we create it now.
1059  auto key = phi_taggings_.NewKey();
1060  phi->set_key(key);
1061  phi_taggings_.Set(key, tagged);
1062  }
1063  return tagged;
1064 }

1066 void MaglevPhiRepresentationSelector::FixLoopPhisBackedge(BasicBlock* block) {
1067  // TODO(dmercadier): it would be interesting to compute a fix point for loop
1068  // phis, or at least to go over the loop header twice.
1069  if (!block->has_phi()) return;
1070  for (Phi* phi : *block->phis()) {
1071  int last_input_idx = phi->input_count() - 1;
1072  ValueNode* backedge = phi->input(last_input_idx).node();
1073  if (phi->value_representation() == ValueRepresentation::kTagged) {
1074  // If the backedge is a Phi that was untagged, but {phi} is tagged, then
1075  // we need to retag the backedge.
1076 
1077  // Identity nodes are used to replace outdated untagging nodes after a phi
1078  // has been untagged. Here, since the backedge was initially tagged, it
1079  // couldn't have been such an untagging node, so it shouldn't be an
1080  // Identity node now.
1081  DCHECK(!backedge->Is<Identity>());
1082 
1083  if (backedge->value_representation() != ValueRepresentation::kTagged) {
1084  // Since all Phi inputs are initially tagged, the fact that the backedge
1085  // is not tagged means that it's a Phi that we recently untagged.
1086  DCHECK(backedge->Is<Phi>());
1087  phi->change_input(
1088  last_input_idx,
1089  EnsurePhiTagged(backedge->Cast<Phi>(), current_block_,
1090  NewNodePosition::kEndOfBlock, /*state*/ nullptr));
1091  }
1092  } else {
1093  // If {phi} was untagged and its backedge became Identity, then we need to
1094  // unwrap it.
1095  DCHECK_NE(phi->value_representation(), ValueRepresentation::kTagged);
1096  if (backedge->Is<Identity>()) {
1097  // {backedge} should have the same representation as {phi}, although if
1098  // {phi} has HoleyFloat64 representation, the backedge is allowed to
1099  // have Float64 representation rather than HoleyFloat64.
1100  DCHECK((backedge->input(0).node()->value_representation() ==
1101  phi->value_representation()) ||
1102  (backedge->input(0).node()->value_representation() ==
1103  ValueRepresentation::kFloat64 &&
1104  phi->value_representation() ==
1105  ValueRepresentation::kHoleyFloat64));
1106  phi->change_input(last_input_idx, backedge->input(0).node());
1107  }
1108  }
1109  }
1110 }

1150 void MaglevPhiRepresentationSelector::PreparePhiTaggings(
1151  BasicBlock* old_block, const BasicBlock* new_block) {
1152  // Sealing and saving current snapshot
1153  if (phi_taggings_.IsSealed()) {
1154  phi_taggings_.StartNewSnapshot();
1155  return;
1156  }
1157  snapshots_.emplace(old_block->id(), phi_taggings_.Seal());
1158 
1159  // Setting up new snapshot
1160  predecessors_.clear();
1161 
1162  if (!new_block->is_merge_block()) {
1163  BasicBlock* pred = new_block->predecessor();
1164  predecessors_.push_back(snapshots_.at(pred->id()));
1165  } else {
1166  int skip_backedge = new_block->is_loop();
1167  for (int i = 0; i < new_block->predecessor_count() - skip_backedge; i++) {
1168  BasicBlock* pred = new_block->predecessor_at(i);
1169  predecessors_.push_back(snapshots_.at(pred->id()));
1170  }
1171  }
1172 
1173  auto merge_taggings =
1174  [&](Key key, base::Vector<ValueNode* const> predecessors) -> ValueNode* {
1175  for (ValueNode* node : predecessors) {
1176  if (node == nullptr) {
1177  // There is a predecessor that doesn't have this Tagging, so we'll
1178  // return nullptr, and if we need it in the future, we'll have to
1179  // recreate it. An alternative would be to eagerly insert this Tagging
1180  // in all of the other predecesors, but it's possible that it's not used
1181  // anymore or not on all future path, so this could also introduce
1182  // unnecessary tagging.
1183  return static_cast<Phi*>(nullptr);
1184  }
1185  }
1186 
1187  // Only merge blocks should require Phis.
1188  DCHECK(new_block->is_merge_block());
1189 
1190  // We create a Phi to merge all of the existing taggings.
1191  int predecessor_count = new_block->predecessor_count();
1192  Phi* phi = Node::New<Phi>(zone(), predecessor_count, new_block->state(),
1193  interpreter::Register());
1194  for (int i = 0; static_cast<size_t>(i) < predecessors.size(); i++) {
1195  phi->set_input(i, predecessors[i]);
1196  }
1197  if (predecessors.size() != static_cast<size_t>(predecessor_count)) {
1198  // The backedge is omitted from {predecessors}. With set the Phi as its
1199  // own backedge.
1200  DCHECK(new_block->is_loop());
1201  phi->set_input(predecessor_count - 1, phi);
1202  }
1203  RegisterNewNode(phi);
1204  new_block->AddPhi(phi);
1205 
1206  return phi;
1207  };
1208 
1209  phi_taggings_.StartNewSnapshot(base::VectorOf(predecessors_), merge_taggings);
1210 }
