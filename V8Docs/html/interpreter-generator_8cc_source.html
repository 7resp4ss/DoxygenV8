1797  TNode<Object> lhs = LoadRegisterAtOperandIndex(0);
1798  TNode<Object> rhs = GetAccumulator();

1800

1804  TNode<Boolean> result;

1815  case Operation::kGreaterThan:
1816  case Operation::kLessThanOrEqual:
1817  case Operation::kGreaterThanOrEqual:

1824  }

1826  TNode<UintPtrT> slot_index = BytecodeOperandIdx(1);

1829  LoadFeedbackVectorOrUndefinedIfJitless(), slot_index, mode);
