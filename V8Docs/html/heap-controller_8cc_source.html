14 template <typename Trait>
15 double MemoryController<Trait>::GrowingFactor(
16  Heap* heap, size_t max_heap_size, std::optional<double> gc_speed,
17  double mutator_speed, Heap::HeapGrowingMode growing_mode) {
18  const double max_factor = MaxGrowingFactor(max_heap_size);
19  double factor = DynamicGrowingFactor(gc_speed, mutator_speed, max_factor);
20  switch (growing_mode) {
21  case Heap::HeapGrowingMode::kConservative:
22  case Heap::HeapGrowingMode::kSlow:
23  factor = std::min({factor, Trait::kConservativeGrowingFactor});
24  break;
25  case Heap::HeapGrowingMode::kMinimal:
26  factor = Trait::kMinGrowingFactor;
27  break;
28  case Heap::HeapGrowingMode::kDefault:
29  break;
30  }
31  if (v8_flags.heap_growing_percent > 0) {
32  factor = 1.0 + v8_flags.heap_growing_percent / 100.0;
33  }
34  if (V8_UNLIKELY(v8_flags.trace_gc_verbose)) {
35  Isolate::FromHeap(heap)->PrintWithTimestamp(
36  "[%s] factor %.1f based on mu=%.3f, speed_ratio=%.f "
37  "(gc=%.f, mutator=%.f)\n",
38  Trait::kName, factor, Trait::kTargetMutatorUtilization,
39  gc_speed.value_or(0) / mutator_speed, gc_speed.value_or(0),
40  mutator_speed);
41  }
42  return factor;
43 }

45 template <typename Trait>
46 double MemoryController<Trait>::MaxGrowingFactor(size_t max_heap_size) {
47  constexpr double kMinSmallFactor = 1.3;
48  constexpr double kMaxSmallFactor = 2.0;
49  constexpr double kHighFactor = 4.0;
50 
51  // If we are on a device with lots of memory, we allow a high heap
52  // growing factor.
53  if (max_heap_size >= Heap::DefaulMaxHeapSize()) {
54  return kHighFactor;
55  }
56 
57  size_t max_size = std::max({max_heap_size, Heap::DefaulMinHeapSize()});
58 
59  DCHECK_GE(max_size, Heap::DefaulMinHeapSize());
60  DCHECK_LT(max_size, Heap::DefaulMaxHeapSize());
61 
62  // On smaller devices we linearly scale the factor: C+(D-C)*(X-A)/(B-A)
63  double factor = kMinSmallFactor +
64  (kMaxSmallFactor - kMinSmallFactor) *
65  (max_size - Heap::DefaulMinHeapSize()) /
66  (Heap::DefaulMaxHeapSize() - Heap::DefaulMinHeapSize());
67  return factor;
68 }

70 // Given GC speed in bytes per ms, the allocation throughput in bytes per ms
71 // (mutator speed), this function returns the heap growing factor that will
72 // achieve the target_mutator_utilization_ if the GC speed and the mutator speed
73 // remain the same until the next GC.
74 //
75 // For a fixed time-frame T = TM + TG, the mutator utilization is the ratio
76 // TM / (TM + TG), where TM is the time spent in the mutator and TG is the
77 // time spent in the garbage collector.
78 //
79 // Let MU be target_mutator_utilization_, the desired mutator utilization for
80 // the time-frame from the end of the current GC to the end of the next GC.
81 // Based on the MU we can compute the heap growing factor F as
82 //
83 // F = R * (1 - MU) / (R * (1 - MU) - MU), where R = gc_speed / mutator_speed.
84 //
85 // This formula can be derived as follows.
86 //
87 // F = Limit / Live by definition, where the Limit is the allocation limit,
88 // and the Live is size of live objects.
89 // Letâ€™s assume that we already know the Limit. Then:
90 // TG = Limit / gc_speed
91 // TM = (TM + TG) * MU, by definition of MU.
92 // TM = TG * MU / (1 - MU)
93 // TM = Limit * MU / (gc_speed * (1 - MU))
94 // On the other hand, if the allocation throughput remains constant:
95 // Limit = Live + TM * allocation_throughput = Live + TM * mutator_speed
96 // Solving it for TM, we get
97 // TM = (Limit - Live) / mutator_speed
98 // Combining the two equation for TM:
99 // (Limit - Live) / mutator_speed = Limit * MU / (gc_speed * (1 - MU))
100 // (Limit - Live) = Limit * MU * mutator_speed / (gc_speed * (1 - MU))
101 // substitute R = gc_speed / mutator_speed
102 // (Limit - Live) = Limit * MU / (R * (1 - MU))
103 // substitute F = Limit / Live
104 // F - 1 = F * MU / (R * (1 - MU))
105 // F - F * MU / (R * (1 - MU)) = 1
106 // F * (1 - MU / (R * (1 - MU))) = 1
107 // F * (R * (1 - MU) - MU) / (R * (1 - MU)) = 1
108 // F = R * (1 - MU) / (R * (1 - MU) - MU)
109 template <typename Trait>
110 double MemoryController<Trait>::DynamicGrowingFactor(
111  std::optional<double> gc_speed, double mutator_speed, double max_factor) {
112  DCHECK_LE(Trait::kMinGrowingFactor, max_factor);
113  DCHECK_GE(Trait::kMaxGrowingFactor, max_factor);
114  if (!gc_speed || mutator_speed == 0) return max_factor;
115 
116  const double speed_ratio = *gc_speed / mutator_speed;
117 
118  const double a = speed_ratio * (1 - Trait::kTargetMutatorUtilization);
119  const double b = speed_ratio * (1 - Trait::kTargetMutatorUtilization) -
120  Trait::kTargetMutatorUtilization;
121 
122  // The factor is a / b, but we need to check for small b first.
123  double factor = (a < b * max_factor) ? a / b : max_factor;
124  DCHECK_LE(factor, max_factor);
125  factor = std::max({factor, Trait::kMinGrowingFactor});
126  return factor;
127 }

129 template <typename Trait>
130 size_t MemoryController<Trait>::MinimumAllocationLimitGrowingStep(
131  Heap::HeapGrowingMode growing_mode) {
132  const size_t kRegularAllocationLimitGrowingStep = 8;
133  const size_t kLowMemoryAllocationLimitGrowingStep = 2;
134  size_t limit = (PageMetadata::kPageSize > MB ? PageMetadata::kPageSize : MB);
135  return limit * (growing_mode == Heap::HeapGrowingMode::kConservative
136  ? kLowMemoryAllocationLimitGrowingStep
137  : kRegularAllocationLimitGrowingStep);
138 }

140 template <typename Trait>
141 size_t MemoryController<Trait>::BoundAllocationLimit(
142  Heap* heap, size_t current_size, uint64_t limit, size_t min_size,
143  size_t max_size, size_t new_space_capacity,
144  Heap::HeapGrowingMode growing_mode) {
145  CHECK_LT(0, current_size);
146  limit = std::max(limit, static_cast<uint64_t>(current_size) +
147  MinimumAllocationLimitGrowingStep(growing_mode)) +
148  new_space_capacity;
149  const uint64_t halfway_to_the_max =
150  (static_cast<uint64_t>(current_size) + max_size) / 2;
151  const uint64_t limit_or_halfway =
152  std::min<uint64_t>(limit, halfway_to_the_max);
153  const size_t result =
154  static_cast<size_t>(std::max<uint64_t>(limit_or_halfway, min_size));
155  if (V8_UNLIKELY(v8_flags.trace_gc_verbose)) {
156  Isolate::FromHeap(heap)->PrintWithTimestamp(
157  "[%s] Limit: old size: %zu KB, new limit: %zu KB\n", Trait::kName,
158  current_size / KB, result / KB);
159  }
160  return result;
161 }
