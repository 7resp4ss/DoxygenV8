73  // Snapshot blob layout:
74  // [0] number of contexts N
75  // [1] rehashability
76  // [2] checksum
77  // [3] read-only snapshot checksum
78  // [4] (64 bytes) version string
79  // [5] offset to readonly
80  // [6] offset to shared heap
81  // [7] offset to context 0
82  // [8] offset to context 1
83  // ...
84  // ... offset to context N - 1
85  // ... startup snapshot data
86  // ... read-only snapshot data
87  // ... shared heap snapshot data
88  // ... context 0 snapshot data
89  // ... context 1 snapshot data

94  kNumberOfContextsOffset + kUInt32Size;

108  static base::Vector<const uint8_t> ChecksummedContent(
109  const v8::StartupData* data) {
110  // The hashed region is everything but the header slots up-to-and-including
111  // the checksum slot itself.
112  // TODO(jgruber): We currently exclude #contexts and rehashability. This
113  // seems arbitrary and I think we could shuffle header slot order around to
114  // include them, just for consistency.
115  static_assert(kReadOnlySnapshotChecksumOffset ==
116  kChecksumOffset + kUInt32Size);
117  const uint32_t kChecksumStart = kReadOnlySnapshotChecksumOffset;
118  return base::Vector<const uint8_t>(
119  reinterpret_cast<const uint8_t*>(data->data + kChecksumStart),
120  data->raw_size - kChecksumStart);
121  }

176 bool Snapshot::Initialize(Isolate* isolate) {
177  if (!isolate->snapshot_available()) return false;
178 
179  const v8::StartupData* blob = isolate->snapshot_blob();
180  SnapshotImpl::CheckVersion(blob);
181  if (Snapshot::ShouldVerifyChecksum(blob)) {
182  CHECK(VerifyChecksum(blob));
183  }
184 
185  base::Vector<const uint8_t> startup_data =
186  SnapshotImpl::ExtractStartupData(blob);
187  base::Vector<const uint8_t> read_only_data =
188  SnapshotImpl::ExtractReadOnlyData(blob);
189  base::Vector<const uint8_t> shared_heap_data =
190  SnapshotImpl::ExtractSharedHeapData(blob);
191 
192  SnapshotData startup_snapshot_data(MaybeDecompress(isolate, startup_data));
193  SnapshotData read_only_snapshot_data(
194  MaybeDecompress(isolate, read_only_data));
195  SnapshotData shared_heap_snapshot_data(
196  MaybeDecompress(isolate, shared_heap_data));
197 
198  return isolate->InitWithSnapshot(
199  &startup_snapshot_data, &read_only_snapshot_data,
200  &shared_heap_snapshot_data, ExtractRehashability(blob));
201 }

221 void Snapshot::ClearReconstructableDataForSerialization(
222  Isolate* isolate, bool clear_recompilable_data) {
223  // Clear SFIs and JSRegExps.
224  PtrComprCageBase cage_base(isolate);
225 
226  {
227  HandleScope scope(isolate);
228  std::vector<i::Handle<i::SharedFunctionInfo>> sfis_to_clear;
229  {
230  i::HeapObjectIterator it(isolate->heap());
231  for (i::Tagged<i::HeapObject> o = it.Next(); !o.is_null();
232  o = it.Next()) {
233  if (clear_recompilable_data && IsSharedFunctionInfo(o, cage_base)) {
234  i::Tagged<i::SharedFunctionInfo> shared =
235  i::Cast<i::SharedFunctionInfo>(o);
236  if (IsScript(shared->script(cage_base), cage_base) &&
237  Cast<Script>(shared->script(cage_base))->type() ==
238  Script::Type::kExtension) {
239  continue; // Don't clear extensions, they cannot be recompiled.
240  }
241  if (shared->CanDiscardCompiled()) {
242  sfis_to_clear.emplace_back(shared, isolate);
243  }
244  } else if (IsJSRegExp(o, cage_base)) {
245  i::Tagged<i::JSRegExp> regexp = i::Cast<i::JSRegExp>(o);
246  if (regexp->has_data()) {
247  i::Tagged<i::RegExpData> data = regexp->data(isolate);
248  if (data->HasCompiledCode()) {
249  DCHECK(Is<IrRegExpData>(regexp->data(isolate)));
250  Cast<IrRegExpData>(data)->DiscardCompiledCodeForSerialization();
251  }
252  }
253  }
254  }
255  }
256 
257 #if V8_ENABLE_WEBASSEMBLY
258  // Clear the cached js-to-wasm wrappers.
259  isolate->heap()->SetJSToWasmWrappers(
260  ReadOnlyRoots(isolate).empty_weak_fixed_array());
261 #endif // V8_ENABLE_WEBASSEMBLY
262 
263  // Must happen after heap iteration since SFI::DiscardCompiled may allocate.
264  for (i::DirectHandle<i::SharedFunctionInfo> shared : sfis_to_clear) {
265  if (shared->CanDiscardCompiled()) {
266  i::SharedFunctionInfo::DiscardCompiled(isolate, shared);
267  }
268  }
269  }
270 
271  // Clear JSFunctions.
272  {
273  i::HeapObjectIterator it(isolate->heap());
274  for (i::Tagged<i::HeapObject> o = it.Next(); !o.is_null(); o = it.Next()) {
275  if (!IsJSFunction(o, cage_base)) continue;
276 
277  i::Tagged<i::JSFunction> fun = i::Cast<i::JSFunction>(o);
278  fun->CompleteInobjectSlackTrackingIfActive(isolate);
279 
280  i::Tagged<i::SharedFunctionInfo> shared = fun->shared();
281  if (IsScript(shared->script(cage_base), cage_base) &&
282  Cast<Script>(shared->script(cage_base))->type() ==
283  Script::Type::kExtension) {
284  continue; // Don't clear extensions, they cannot be recompiled.
285  }
286 
287  // Also, clear out feedback vectors and recompilable code.
288  if (fun->CanDiscardCompiled(isolate)) {
289  fun->UpdateCode(isolate, *BUILTIN_CODE(isolate, CompileLazy));
290  }
291  if (!IsUndefined(fun->raw_feedback_cell(cage_base)->value(cage_base))) {
292  fun->raw_feedback_cell(cage_base)->set_value(
293  i::ReadOnlyRoots(isolate).undefined_value());
294  }
295 #ifdef DEBUG
296  if (clear_recompilable_data) {
297  DCHECK(fun->shared()->HasBuiltinId() ||
298  fun->shared()->IsApiFunction() ||
299 #if V8_ENABLE_WEBASSEMBLY
300  fun->shared()->HasWasmExportedFunctionData(isolate) ||
301 #endif // V8_ENABLE_WEBASSEMBLY
302  fun->shared()->HasUncompiledDataWithoutPreparseData(isolate));
303  }
304 #endif // DEBUG
305  }
306  }
307 
308  // PendingOptimizeTable also contains BytecodeArray, we need to clear the
309  // recompilable code same as above.
310  ReadOnlyRoots roots(isolate);
311  isolate->heap()->SetFunctionsMarkedForManualOptimization(
312  roots.undefined_value());
313 
314 #if V8_ENABLE_WEBASSEMBLY
315  {
316  // Check if there are any asm.js / wasm functions on the heap.
317  // These cannot be serialized due to restrictions with the js-to-wasm
318  // wrapper. A tiered-up wrapper would have to be replaced with a generic
319  // wrapper which isn't supported. For asm.js there also isn't any support
320  // for the generic wrapper at all.
321  i::HeapObjectIterator it(isolate->heap(),
322  HeapObjectIterator::kFilterUnreachable);
323  for (i::Tagged<i::HeapObject> o = it.Next(); !o.is_null(); o = it.Next()) {
324  if (IsJSFunction(o)) {
325  i::Tagged<i::JSFunction> fun = i::Cast<i::JSFunction>(o);
326  if (fun->shared()->HasAsmWasmData()) {
327  FATAL("asm.js functions are not supported in snapshots");
328  }
329  if (fun->shared()->HasWasmExportedFunctionData(isolate)) {
330  FATAL(
331  "Exported WebAssembly functions are not supported in snapshots");
332  }
333  }
334  }
335  }
336 #endif // V8_ENABLE_WEBASSEMBLY
337 }

340 void Snapshot::SerializeDeserializeAndVerifyForTesting(
341  Isolate* isolate, DirectHandle<Context> default_context) {
342  StartupData serialized_data;
343  std::unique_ptr<const char[]> auto_delete_serialized_data;
344 
345  isolate->heap()->CollectAllAvailableGarbage(
346  i::GarbageCollectionReason::kSnapshotCreator);
347 
348  // Test serialization.
349  {
350  SafepointKind safepoint_kind = isolate->has_shared_space()
351  ? SafepointKind::kGlobal
352  : SafepointKind::kIsolate;
353  SafepointScope safepoint_scope(isolate, safepoint_kind);
354  DisallowGarbageCollection no_gc;
355 
356  Snapshot::SerializerFlags flags(
357  Snapshot::kAllowUnknownExternalReferencesForTesting |
358  Snapshot::kAllowActiveIsolateForTesting |
359  Snapshot::kReconstructReadOnlyAndSharedObjectCachesForTesting);
360  std::vector<Tagged<Context>> contexts{*default_context};
361  std::vector<SerializeEmbedderFieldsCallback> callbacks{
362  SerializeEmbedderFieldsCallback()};
363  serialized_data = Snapshot::Create(isolate, &contexts, callbacks,
364  safepoint_scope, no_gc, flags);
365  auto_delete_serialized_data.reset(serialized_data.data);
366  }
367 
368  // The shared heap is verified on Heap teardown, which performs a global
369  // safepoint. Both isolate and new_isolate are running in the same thread, so
370  // park isolate before running new_isolate to avoid deadlock.
371  isolate->main_thread_local_isolate()->ExecuteMainThreadWhileParked(
372  [&serialized_data]() {
373  // Test deserialization.
374  Isolate* new_isolate = Isolate::New();
375  std::unique_ptr<v8::ArrayBuffer::Allocator> array_buffer_allocator(
376  v8::ArrayBuffer::Allocator::NewDefaultAllocator());
377  {
378  // Set serializer_enabled() to not install extensions and experimental
379  // natives on the new isolate.
380  // TODO(v8:10416): This should be a separate setting on the isolate.
381  new_isolate->enable_serializer();
382  new_isolate->Enter();
383  new_isolate->set_snapshot_blob(&serialized_data);
384  new_isolate->set_array_buffer_allocator(array_buffer_allocator.get());
385  CHECK(Snapshot::Initialize(new_isolate));
386 
387  HandleScope scope(new_isolate);
388  DirectHandle<Context> new_native_context =
389  new_isolate->bootstrapper()->CreateEnvironmentForTesting();
390  CHECK(IsNativeContext(*new_native_context));
391 
392 #ifdef VERIFY_HEAP
393  if (v8_flags.verify_heap)
394  HeapVerifier::VerifyHeap(new_isolate->heap());
395 #endif // VERIFY_HEAP
396  }
397  new_isolate->Exit();
398  Isolate::Delete(new_isolate);
399  });
400 }

402 // static
403 v8::StartupData Snapshot::Create(
404  Isolate* isolate, std::vector<Tagged<Context>>* contexts,
405  const std::vector<SerializeEmbedderFieldsCallback>&
406  embedder_fields_serializers,
407  const SafepointScope& safepoint_scope,
408  const DisallowGarbageCollection& no_gc, SerializerFlags flags) {
409  TRACE_EVENT0("v8", "V8.SnapshotCreate");
410  DCHECK_EQ(contexts->size(), embedder_fields_serializers.size());
411  DCHECK_GT(contexts->size(), 0);
412  HandleScope scope(isolate);
413 
414  ReadOnlySerializer read_only_serializer(isolate, flags);
415  read_only_serializer.Serialize();
416 
417  // TODO(v8:6593): generalize rehashing, and remove this flag.
418  bool can_be_rehashed = read_only_serializer.can_be_rehashed();
419 
420  SharedHeapSerializer shared_heap_serializer(isolate, flags);
421  StartupSerializer startup_serializer(isolate, flags, &shared_heap_serializer);
422  startup_serializer.SerializeStrongReferences(no_gc);
423 
424  // Serialize each context with a new serializer.
425  const int num_contexts = static_cast<int>(contexts->size());
426  std::vector<SnapshotData*> context_snapshots;
427  context_snapshots.reserve(num_contexts);
428 
429  std::vector<int> context_allocation_sizes;
430  for (int i = 0; i < num_contexts; i++) {
431  ContextSerializer context_serializer(isolate, flags, &startup_serializer,
432  embedder_fields_serializers[i]);
433  context_serializer.Serialize(&contexts->at(i), no_gc);
434  can_be_rehashed = can_be_rehashed && context_serializer.can_be_rehashed();
435  context_snapshots.push_back(new SnapshotData(&context_serializer));
436  if (v8_flags.serialization_statistics) {
437  context_allocation_sizes.push_back(
438  context_serializer.TotalAllocationSize());
439  }
440  }
441 
442  startup_serializer.SerializeWeakReferencesAndDeferred();
443  can_be_rehashed = can_be_rehashed && startup_serializer.can_be_rehashed();
444 
445  startup_serializer.CheckNoDirtyFinalizationRegistries();
446 
447  shared_heap_serializer.FinalizeSerialization();
448  can_be_rehashed = can_be_rehashed && shared_heap_serializer.can_be_rehashed();
449 
450  if (v8_flags.serialization_statistics) {
451  DCHECK_NE(read_only_serializer.TotalAllocationSize(), 0);
452  DCHECK_NE(startup_serializer.TotalAllocationSize(), 0);
453  // The shared heap snapshot can be empty, no problem.
454  // DCHECK_NE(shared_heap_serializer.TotalAllocationSize(), 0);
455  int per_isolate_allocation_size = startup_serializer.TotalAllocationSize();
456  int per_process_allocation_size =
457  read_only_serializer.TotalAllocationSize();
458  // TODO(jgruber): At snapshot-generation time we don't know whether the
459  // shared heap snapshot will actually be shared at runtime, or if it will
460  // be deserialized into each isolate. Conservatively account to per-isolate
461  // memory here.
462  per_isolate_allocation_size += shared_heap_serializer.TotalAllocationSize();
463  // These prints must match the regexp in test/memory/Memory.json
464  PrintF("Deserialization will allocate:\n");
465  PrintF("%10d bytes per process\n", per_process_allocation_size);
466  PrintF("%10d bytes per isolate\n", per_isolate_allocation_size);
467  for (int i = 0; i < num_contexts; i++) {
468  DCHECK_NE(context_allocation_sizes[i], 0);
469  PrintF("%10d bytes per context #%d\n", context_allocation_sizes[i], i);
470  }
471  PrintF("\n");
472  }
473 
474  SnapshotData read_only_snapshot(&read_only_serializer);
475  SnapshotData shared_heap_snapshot(&shared_heap_serializer);
476  SnapshotData startup_snapshot(&startup_serializer);
477  v8::StartupData result = SnapshotImpl::CreateSnapshotBlob(
478  &startup_snapshot, &read_only_snapshot, &shared_heap_snapshot,
479  context_snapshots, can_be_rehashed);
480 
481  for (const SnapshotData* ptr : context_snapshots) delete ptr;
482 
483  CHECK(Snapshot::VerifyChecksum(&result));
484  return result;
485 }

487 v8::StartupData SnapshotImpl::CreateSnapshotBlob(
488  const SnapshotData* startup_snapshot_in,
489  const SnapshotData* read_only_snapshot_in,
490  const SnapshotData* shared_heap_snapshot_in,
491  const std::vector<SnapshotData*>& context_snapshots_in,
492  bool can_be_rehashed) {
493  TRACE_EVENT0("v8", "V8.SnapshotCompress");
494  // Have these separate from snapshot_in for compression, since we need to
495  // access the compressed data as well as the uncompressed reservations.
496  const SnapshotData* startup_snapshot;
497  const SnapshotData* read_only_snapshot;
498  const SnapshotData* shared_heap_snapshot;
499  const std::vector<SnapshotData*>* context_snapshots;
500 #ifdef V8_SNAPSHOT_COMPRESSION
501  SnapshotData startup_compressed(
502  SnapshotCompression::Compress(startup_snapshot_in));
503  SnapshotData read_only_compressed(
504  SnapshotCompression::Compress(read_only_snapshot_in));
505  SnapshotData shared_heap_compressed(
506  SnapshotCompression::Compress(shared_heap_snapshot_in));
507  startup_snapshot = &startup_compressed;
508  read_only_snapshot = &read_only_compressed;
509  shared_heap_snapshot = &shared_heap_compressed;
510  std::vector<SnapshotData> context_snapshots_compressed;
511  context_snapshots_compressed.reserve(context_snapshots_in.size());
512  std::vector<SnapshotData*> context_snapshots_compressed_ptrs;
513  for (unsigned int i = 0; i < context_snapshots_in.size(); ++i) {
514  context_snapshots_compressed.push_back(
515  SnapshotCompression::Compress(context_snapshots_in[i]));
516  context_snapshots_compressed_ptrs.push_back(
517  &context_snapshots_compressed[i]);
518  }
519  context_snapshots = &context_snapshots_compressed_ptrs;
520 #else
521  startup_snapshot = startup_snapshot_in;
522  read_only_snapshot = read_only_snapshot_in;
523  shared_heap_snapshot = shared_heap_snapshot_in;
524  context_snapshots = &context_snapshots_in;
525 #endif
526 
527  uint32_t num_contexts = static_cast<uint32_t>(context_snapshots->size());
528  uint32_t startup_snapshot_offset =
529  SnapshotImpl::StartupSnapshotOffset(num_contexts);
530  uint32_t total_length = startup_snapshot_offset;
531  total_length += static_cast<uint32_t>(startup_snapshot->RawData().length());
532  total_length += static_cast<uint32_t>(read_only_snapshot->RawData().length());
533  total_length +=
534  static_cast<uint32_t>(shared_heap_snapshot->RawData().length());
535  for (const auto context_snapshot : *context_snapshots) {
536  total_length += static_cast<uint32_t>(context_snapshot->RawData().length());
537  }
538 
539  char* data = new char[total_length];
540  // Zero out pre-payload data. Part of that is only used for padding.
541  memset(data, 0, SnapshotImpl::StartupSnapshotOffset(num_contexts));
542 
543  SnapshotImpl::SetHeaderValue(data, SnapshotImpl::kNumberOfContextsOffset,
544  num_contexts);
545  SnapshotImpl::SetHeaderValue(data, SnapshotImpl::kRehashabilityOffset,
546  can_be_rehashed ? 1 : 0);
547 
548  // Write version string into snapshot data.
549  memset(data + SnapshotImpl::kVersionStringOffset, 0,
550  SnapshotImpl::kVersionStringLength);
551  Version::GetString(
552  base::Vector<char>(data + SnapshotImpl::kVersionStringOffset,
553  SnapshotImpl::kVersionStringLength));
554 
555  // Startup snapshot (isolate-specific data).
556  uint32_t payload_offset = startup_snapshot_offset;
557  uint32_t payload_length =
558  static_cast<uint32_t>(startup_snapshot->RawData().length());
559  CopyBytes(data + payload_offset,
560  reinterpret_cast<const char*>(startup_snapshot->RawData().begin()),
561  payload_length);
562  if (v8_flags.serialization_statistics) {
563  // These prints must match the regexp in test/memory/Memory.json
564  PrintF("Snapshot blob consists of:\n");
565  PrintF("%10d bytes for startup\n", payload_length);
566  }
567  payload_offset += payload_length;
568 
569  // Read-only.
570  SnapshotImpl::SetHeaderValue(data, SnapshotImpl::kReadOnlyOffsetOffset,
571  payload_offset);
572  payload_length = read_only_snapshot->RawData().length();
573  CopyBytes(
574  data + payload_offset,
575  reinterpret_cast<const char*>(read_only_snapshot->RawData().begin()),
576  payload_length);
577  SnapshotImpl::SetHeaderValue(
578  data, SnapshotImpl::kReadOnlySnapshotChecksumOffset,
579  Checksum(base::VectorOf(
580  reinterpret_cast<const uint8_t*>(data + payload_offset),
581  payload_length)));
582  if (v8_flags.serialization_statistics) {
583  // These prints must match the regexp in test/memory/Memory.json
584  PrintF("%10d bytes for read-only\n", payload_length);
585  }
586  payload_offset += payload_length;
587 
588  // Shared heap.
589  SnapshotImpl::SetHeaderValue(data, SnapshotImpl::kSharedHeapOffsetOffset,
590  payload_offset);
591  payload_length = shared_heap_snapshot->RawData().length();
592  CopyBytes(
593  data + payload_offset,
594  reinterpret_cast<const char*>(shared_heap_snapshot->RawData().begin()),
595  payload_length);
596  if (v8_flags.serialization_statistics) {
597  // These prints must match the regexp in test/memory/Memory.json
598  PrintF("%10d bytes for shared heap\n", payload_length);
599  }
600  payload_offset += payload_length;
601 
602  // Context snapshots (context-specific data).
603  for (uint32_t i = 0; i < num_contexts; i++) {
604  SnapshotImpl::SetHeaderValue(
605  data, SnapshotImpl::ContextSnapshotOffsetOffset(i), payload_offset);
606  SnapshotData* context_snapshot = (*context_snapshots)[i];
607  payload_length = context_snapshot->RawData().length();
608  CopyBytes(
609  data + payload_offset,
610  reinterpret_cast<const char*>(context_snapshot->RawData().begin()),
611  payload_length);
612  if (v8_flags.serialization_statistics) {
613  // These prints must match the regexp in test/memory/Memory.json
614  PrintF("%10d bytes for context #%d\n", payload_length, i);
615  }
616  payload_offset += payload_length;
617  }
618  if (v8_flags.serialization_statistics) PrintF("\n");
619 
620  DCHECK_EQ(total_length, payload_offset);
621  v8::StartupData result = {data, static_cast<int>(total_length)};
622 
623  SnapshotImpl::SetHeaderValue(
624  data, SnapshotImpl::kChecksumOffset,
625  Checksum(SnapshotImpl::ChecksummedContent(&result)));
626 
627  return result;
628 }

1036 // static
1037 StartupData SnapshotCreatorImpl::CreateBlob(
1038  SnapshotCreator::FunctionCodeHandling function_code_handling,
1039  Snapshot::SerializerFlags serializer_flags) {
1040  CHECK(!created());
1041  CHECK(contexts_[kDefaultContextIndex].handle_location != nullptr);
1042 
1043  const size_t num_contexts = contexts_.size();
1044  const size_t num_additional_contexts = num_contexts - 1;
1045 
1046  // Create and store lists of embedder-provided data needed during
1047  // serialization.
1048  {
1049  HandleScope scope(isolate_);
1050 
1051  // Convert list of context-independent data to FixedArray.
1052  ConvertSerializedObjectsToFixedArray(isolate_);
1053 
1054  // Convert lists of context-dependent data to FixedArray.
1055  for (size_t i = 0; i < num_contexts; i++) {
1056  ConvertSerializedObjectsToFixedArray(isolate_, context_at(i));
1057  }
1058 
1059  // We need to store the global proxy size upfront in case we need the
1060  // bootstrapper to create a global proxy before we deserialize the context.
1061  DirectHandle<FixedArray> global_proxy_sizes =
1062  isolate_->factory()->NewFixedArray(
1063  static_cast<int>(num_additional_contexts), AllocationType::kOld);
1064  for (size_t i = kFirstAddtlContextIndex; i < num_contexts; i++) {
1065  global_proxy_sizes->set(
1066  static_cast<int>(i - kFirstAddtlContextIndex),
1067  Smi::FromInt(context_at(i)->global_proxy()->Size()));
1068  }
1069  isolate_->heap()->SetSerializedGlobalProxySizes(*global_proxy_sizes);
1070  }
1071 
1072  // We might rehash strings and re-sort descriptors. Clear the lookup cache.
1073  isolate_->descriptor_lookup_cache()->Clear();
1074 
1075  // If we don't do this then we end up with a stray root pointing at the
1076  // context even after we have disposed of the context.
1077  {
1078  // Note that we need to run a garbage collection without stack at this
1079  // point, so that all dead objects are reclaimed. This is required to avoid
1080  // conservative stack scanning and guarantee deterministic behaviour.
1081  EmbedderStackStateScope stack_scope(
1082  isolate_->heap(), EmbedderStackStateOrigin::kExplicitInvocation,
1083  StackState::kNoHeapPointers);
1084  isolate_->heap()->CollectAllAvailableGarbage(
1085  GarbageCollectionReason::kSnapshotCreator);
1086  }
1087  {
1088  HandleScope scope(isolate_);
1089  isolate_->heap()->CompactWeakArrayLists();
1090  }
1091 
1092  Snapshot::ClearReconstructableDataForSerialization(
1093  isolate_,
1094  function_code_handling == SnapshotCreator::FunctionCodeHandling::kClear);
1095 
1096  SafepointKind safepoint_kind = isolate_->has_shared_space()
1097  ? SafepointKind::kGlobal
1098  : SafepointKind::kIsolate;
1099  SafepointScope safepoint_scope(isolate_, safepoint_kind);
1100  DisallowGarbageCollection no_gc_from_here_on;
1101 
1102  // RO space is usually writable when serializing a snapshot s.t. preceding
1103  // heap initialization can also extend RO space. There are notable exceptions
1104  // though, including --stress-snapshot and serializer cctests.
1105  if (isolate_->heap()->read_only_space()->writable()) {
1106  // Promote objects from mutable heap spaces to read-only space prior to
1107  // serialization. Objects can be promoted if a) they are themselves
1108  // immutable-after-deserialization and b) all objects in the transitive
1109  // object graph also satisfy condition a).
1110  ReadOnlyPromotion::Promote(isolate_, safepoint_scope, no_gc_from_here_on);
1111  // When creating the snapshot from scratch, we are responsible for sealing
1112  // the RO heap here. Note we cannot delegate the responsibility e.g. to
1113  // Isolate::Init since it should still be possible to allocate into RO
1114  // space after the Isolate has been initialized, for example as part of
1115  // Context creation.
1116  isolate_->read_only_heap()->OnCreateHeapObjectsComplete(isolate_);
1117  }
1118 
1119  // Create a vector with all contexts and destroy associated global handles.
1120  // This is important because serialization visits active global handles as
