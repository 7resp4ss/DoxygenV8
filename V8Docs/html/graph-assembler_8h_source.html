685 template <typename... Vars>
686 void GraphAssembler::MergeState(
687  detail::GraphAssemblerLabelForVars<Vars...>* label, Vars... vars) {
688  using NodeArray = typename detail::GraphAssemblerLabelForVars<
689  Vars...>::template Array<Node*>;
690  RestoreEffectControlScope restore_effect_control_scope(this);
691 
692  const int merged_count = static_cast<int>(label->merged_count_);
693 
694  const size_t var_count = label->Count();
695  NodeArray var_array{vars...};
696 
697  const bool is_loop_exit = label->loop_nesting_level_ != loop_nesting_level_;
698  if (is_loop_exit) {
699  // This feature may only be used if it has been enabled.
700  USE(mark_loop_exits_);
701  DCHECK(mark_loop_exits_);
702  // Jumping from loops to loops not supported.
703  DCHECK(!label->IsLoop());
704  // Currently only the simple case of jumping one level is supported.
705  DCHECK_EQ(label->loop_nesting_level_, loop_nesting_level_ - 1);
706  DCHECK(!loop_headers_.empty());
707  DCHECK_NOT_NULL(*loop_headers_.back());
708 
709  // Mark this exit to enable loop peeling.
710  AddNode(graph()->NewNode(common()->LoopExit(), control(),
711  *loop_headers_.back()));
712  AddNode(graph()->NewNode(common()->LoopExitEffect(), effect(), control()));
713  for (size_t i = 0; i < var_count; i++) {
714  var_array[i] = AddNode(graph()->NewNode(
715  common()->LoopExitValue(MachineRepresentation::kTagged), var_array[i],
716  control()));
717  }
718  }
719 
720  if (label->IsLoop()) {
721  if (merged_count == 0) {
722  DCHECK(!label->IsBound());
723  label->control_ =
724  graph()->NewNode(common()->Loop(2), control(), control());
725  label->effect_ = graph()->NewNode(common()->EffectPhi(2), effect(),
726  effect(), label->control_);
727  Node* terminate = graph()->NewNode(common()->Terminate(), label->effect_,
728  label->control_);
729  NodeProperties::MergeControlToEnd(graph(), common(), terminate);
730  for (size_t i = 0; i < var_count; i++) {
731  label->bindings_[i] =
732  graph()->NewNode(common()->Phi(label->representations_[i], 2),
733  var_array[i], var_array[i], label->control_);
734  }
735  } else {
736  DCHECK(label->IsBound());
737  DCHECK_EQ(1, merged_count);
738  label->control_->ReplaceInput(1, control());
739  label->effect_->ReplaceInput(1, effect());
740  for (size_t i = 0; i < var_count; i++) {
741  label->bindings_[i]->ReplaceInput(1, var_array[i]);
742  CHECK(!NodeProperties::IsTyped(var_array[i])); // Unsupported.
743  }
744  }
745  } else {
746  DCHECK(!label->IsLoop());
747  DCHECK(!label->IsBound());
748  if (merged_count == 0) {
749  // Just set the control, effect and variables directly.
750  label->control_ = control();
751  label->effect_ = effect();
752  for (size_t i = 0; i < var_count; i++) {
753  label->bindings_[i] = var_array[i];
754  }
755  } else if (merged_count == 1) {
756  // Create merge, effect phi and a phi for each variable.
757  label->control_ =
758  graph()->NewNode(common()->Merge(2), label->control_, control());
759  label->effect_ = graph()->NewNode(common()->EffectPhi(2), label->effect_,
760  effect(), label->control_);
761  for (size_t i = 0; i < var_count; i++) {
762  label->bindings_[i] = graph()->NewNode(
763  common()->Phi(label->representations_[i], 2), label->bindings_[i],
764  var_array[i], label->control_);
765  }
766  } else {
767  // Append to the merge, effect phi and phis.
768  DCHECK_EQ(IrOpcode::kMerge, label->control_->opcode());
769  label->control_->AppendInput(graph()->zone(), control());
770  NodeProperties::ChangeOp(label->control_,
771  common()->Merge(merged_count + 1));
772 
773  DCHECK_EQ(IrOpcode::kEffectPhi, label->effect_->opcode());
774  label->effect_->ReplaceInput(merged_count, effect());
775  label->effect_->AppendInput(graph()->zone(), label->control_);
776  NodeProperties::ChangeOp(label->effect_,
777  common()->EffectPhi(merged_count + 1));
778 
779  for (size_t i = 0; i < var_count; i++) {
780  DCHECK_EQ(IrOpcode::kPhi, label->bindings_[i]->opcode());
781  label->bindings_[i]->ReplaceInput(merged_count, var_array[i]);
782  label->bindings_[i]->AppendInput(graph()->zone(), label->control_);
783  NodeProperties::ChangeOp(
784  label->bindings_[i],
785  common()->Phi(label->representations_[i], merged_count + 1));
786  if (NodeProperties::IsTyped(label->bindings_[i])) {
787  CHECK(NodeProperties::IsTyped(var_array[i]));
788  Type old_type = NodeProperties::GetType(label->bindings_[i]);
789  Type new_type = Type::Union(
790  old_type, NodeProperties::GetType(var_array[i]), graph()->zone());
791  NodeProperties::SetType(label->bindings_[i], new_type);
792  }
793  }
794  }
795  }
796  label->merged_count_++;
797 }

802  DCHECK_NULL(effect());
