33 bool AllInputsTyped(Node* node) {
34  for (int i = 0; i < node->op()->ValueInputCount(); i++) {
35  if (!NodeProperties::IsTyped(NodeProperties::GetValueInput(node, i))) {
36  return false;
37  }
38  }
39  return true;

43 Reduction WasmTyper::Reduce(Node* node) {
44  using TypeInModule = wasm::TypeInModule;
45  TypeInModule computed_type;
46  switch (node->opcode()) {
47  case IrOpcode::kTypeGuard: {
48  if (!AllInputsTyped(node)) return NoChange();
49  Type guarded_type = TypeGuardTypeOf(node->op());
50  if (!guarded_type.IsWasm()) return NoChange();
51  Type input_type =
52  NodeProperties::GetType(NodeProperties::GetValueInput(node, 0));
53  if (!input_type.IsWasm()) return NoChange();
54  TypeInModule guarded_wasm_type = guarded_type.AsWasm();
55  TypeInModule input_wasm_type = input_type.AsWasm();
56  // Note: The intersection type might be bottom. In this case, we are in a
57  // dead branch: Type this node as bottom and wait for the
58  // WasmGCOperatorReducer to remove it.
59  DCHECK_EQ(guarded_wasm_type.module, input_wasm_type.module);
60  computed_type = wasm::Intersection(
61  guarded_wasm_type.type, input_wasm_type.type, input_wasm_type.module);
62  break;
63  }
64  case IrOpcode::kWasmTypeCast:
65  case IrOpcode::kWasmTypeCastAbstract: {
66  if (!AllInputsTyped(node)) return NoChange();
67  TypeInModule object_type =
68  NodeProperties::GetType(NodeProperties::GetValueInput(node, 0))
69  .AsWasm();
70  wasm::ValueType to_type = OpParameter<WasmTypeCheckConfig>(node->op()).to;
71  computed_type =
72  wasm::Intersection(object_type.type, to_type, object_type.module);
73  break;
74  }
75  case IrOpcode::kAssertNotNull: {
76  if (!AllInputsTyped(node)) return NoChange();
77  TypeInModule object_type =
78  NodeProperties::GetType(NodeProperties::GetValueInput(node, 0))
79  .AsWasm();
80  computed_type = {object_type.type.AsNonNull(), object_type.module};
81  break;
82  }
83  case IrOpcode::kPhi: {
84  if (!AllInputsTyped(node)) {
85  bool is_loop_phi =
86  NodeProperties::GetControlInput(node)->opcode() == IrOpcode::kLoop;
87  // For a merge phi, we need all inputs to be typed.
88  if (!is_loop_phi) return NoChange();
89  // For a loop phi, we can forward the non-recursive-input type. We can
90  // recompute the type when the rest of the inputs' types are computed.
91  Node* non_recursive_input = NodeProperties::GetValueInput(node, 0);
92  if (!NodeProperties::IsTyped(non_recursive_input) ||
93  !NodeProperties::GetType(non_recursive_input).IsWasm()) {
94  return NoChange();
95  }
96  computed_type = NodeProperties::GetType(non_recursive_input).AsWasm();
97  TRACE("function: %d, loop phi node: %d, type: %s\n", function_index_,
98  node->id(), computed_type.type.name().c_str());
99  break;
100  }
101 
102  Type first_input_type =
103  NodeProperties::GetType(NodeProperties::GetValueInput(node, 0));
104  if (!first_input_type.IsWasm()) return NoChange();
105  computed_type = first_input_type.AsWasm();
106  for (int i = 1; i < node->op()->ValueInputCount(); i++) {
107  Node* input = NodeProperties::GetValueInput(node, i);
108  Type input_type = NodeProperties::GetType(input);
109  if (!input_type.IsWasm()) return NoChange();
110  TypeInModule wasm_type = input_type.AsWasm();
111  if (computed_type.type.is_bottom()) {
112  // We have not found a non-bottom branch yet.
113  computed_type = wasm_type;
114  } else if (!wasm_type.type.is_bottom()) {
115  // We do not want union of types from unreachable branches.
116  DCHECK_EQ(computed_type.module, wasm_type.module);
117  computed_type =
118  wasm::Union(computed_type.type, wasm_type.type, wasm_type.module);
119  }
120  }
121  TRACE(
122  "function: %d, phi node: %d, input#: %d, input0:%d:%s, input1:%d:%s, "
123  "type: %s\n",
124  function_index_, node->id(), node->op()->ValueInputCount(),
125  node->InputAt(0)->id(),
126  NodeProperties::GetType(node->InputAt(0))
127  .AsWasm()
128  .type.name()
129  .c_str(),
130  node->InputAt(1)->id(),
131  node->op()->ValueInputCount() > 1
132  ? NodeProperties::GetType(node->InputAt(1))
133  .AsWasm()
134  .type.name()
135  .c_str()
136  : "<control>",
137  computed_type.type.name().c_str());
138  break;
139  }
140  case IrOpcode::kWasmArrayGet: {
141  Node* object = NodeProperties::GetValueInput(node, 0);
142  // This can happen either because the object has not been typed yet, or
143  // because it is an internal VM object (e.g. the instance).
144  if (!NodeProperties::IsTyped(object)) return NoChange();
145  TypeInModule object_type = NodeProperties::GetType(object).AsWasm();
146  // {is_uninhabited} can happen in unreachable branches.
147  if (object_type.type.is_uninhabited() ||
148  object_type.type == wasm::kWasmNullRef) {
149  computed_type = {wasm::kWasmBottom, object_type.module};
150  break;
151  }
152  wasm::ModuleTypeIndex ref_index = object_type.type.ref_index();
153  DCHECK(object_type.module->has_array(ref_index));
154  const wasm::ArrayType* type_from_object =
155  object_type.module->type(ref_index).array_type;
156  computed_type = {type_from_object->element_type().Unpacked(),
157  object_type.module};
158  break;
159  }
160  case IrOpcode::kWasmStructGet: {
161  Node* object = NodeProperties::GetValueInput(node, 0);
162  // This can happen either because the object has not been typed yet.
163  if (!NodeProperties::IsTyped(object)) return NoChange();
164  TypeInModule object_type = NodeProperties::GetType(object).AsWasm();
165  // {is_uninhabited} can happen in unreachable branches.
166  if (object_type.type.is_uninhabited() ||
167  object_type.type == wasm::kWasmNullRef) {
168  computed_type = {wasm::kWasmBottom, object_type.module};
169  break;
170  }
171  WasmFieldInfo info = OpParameter<WasmFieldInfo>(node->op());
172 
173  wasm::ModuleTypeIndex ref_index = object_type.type.ref_index();
174 
175  DCHECK(object_type.module->has_struct(ref_index));
176 
177  const wasm::StructType* struct_type_from_object =
178  object_type.module->type(ref_index).struct_type;
179 
180  computed_type = {
181  struct_type_from_object->field(info.field_index).Unpacked(),
182  object_type.module};
183  break;
184  }
185  case IrOpcode::kNull: {
186  TypeInModule from_node = NodeProperties::GetType(node).AsWasm();
187  computed_type = {wasm::ToNullSentinel(from_node), from_node.module};
188  break;
189  }
190  default:
191  return NoChange();
192  }
193 
194  if (NodeProperties::IsTyped(node) && NodeProperties::GetType(node).IsWasm()) {
195  TypeInModule current_type = NodeProperties::GetType(node).AsWasm();
196  if (!(current_type.type.is_bottom() || computed_type.type.is_bottom() ||
197  wasm::IsSubtypeOf(current_type.type, computed_type.type,
198  current_type.module, computed_type.module) ||
199  wasm::IsSubtypeOf(computed_type.type, current_type.type,
200  computed_type.module, current_type.module) ||
201  // Imported strings can have more precise types.
202  (current_type.type.heap_representation() == wasm::HeapType::kExtern &&
203  computed_type.type.heap_representation() ==
204  wasm::HeapType::kString))) {
205  FATAL(
206  "Error - Incompatible types. function: %d, node: %d:%s, input0:%d, "
207  "current %s, computed %s\n",
208  function_index_, node->id(), node->op()->mnemonic(),
209  node->InputAt(0)->id(), current_type.type.name().c_str(),
210  computed_type.type.name().c_str());
211  }
212 
213  if (wasm::EquivalentTypes(current_type.type, computed_type.type,
214  current_type.module, computed_type.module)) {
215  return NoChange();
216  }
217  }
218 
219  TRACE("function: %d, node: %d:%s, from: %s, to: %s\n", function_index_,
220  node->id(), node->op()->mnemonic(),
221  NodeProperties::IsTyped(node)
222  ? NodeProperties::GetType(node).AsWasm().type.name().c_str()
223  : "<untyped>",
224  computed_type.type.name().c_str());
225 
226  NodeProperties::SetType(node, Type::Wasm(computed_type, graph_zone_));
227  return Changed(node);
228 }
