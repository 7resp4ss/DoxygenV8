14 template <typename T>
15 struct DoublyThreadedListTraits {
16  static T** prev(T t) { return t->prev(); }
17  static T* next(T t) { return t->next(); }
18  static bool non_empty(T t) { return t != nullptr; }
19 };

21 // `DoublyThreadedList` is an intrusive doubly-linked list that threads through
22 // its nodes, somewhat like `v8::base::ThreadedList`.
23 //
24 // Of interest is the fact that instead of having regular next/prev pointers,
25 // nodes have a regular "next" pointer, but their "prev" pointer contains the
26 // address of the "next" of the previous element. This way, removing an element
27 // doesn't require special treatment for the head of the list, and does not
28 // even require to know the head of the list.
29 template <class T, class DTLTraits = DoublyThreadedListTraits<T>>
30 class DoublyThreadedList {
31  public:
32  // Since C++17, it is possible to have a sentinel end-iterator that is not an
33  // iterator itself.
34  class end_iterator {};
35 
36  class iterator : public base::iterator<std::forward_iterator_tag, T> {
37  public:
38  explicit iterator(T head) : curr_(head) {}
39 
40  T operator*() { return curr_; }
41 
42  iterator& operator++() {
43  DCHECK(DTLTraits::non_empty(curr_));
44  curr_ = *DTLTraits::next(curr_);
45  return *this;
46  }
47 
48  iterator operator++(int) {
49  DCHECK(DTLTraits::non_empty(curr_));
50  iterator tmp(*this);
51  operator++();
52  return tmp;
53  }
54 
55  bool operator==(end_iterator) { return !DTLTraits::non_empty(curr_); }
56 
57  private:
58  friend DoublyThreadedList;
59  T curr_;
60  };
61 
62  // Removes `x` from the list. Iterators that are currently on `x` are
63  // invalidated. To remove while iterating, use RemoveAt.
64  static void Remove(T x) {
65  if (*DTLTraits::prev(x) == nullptr) {
66  DCHECK(empty(*DTLTraits::next(x)));
67  // {x} already removed from the list.
68  return;
69  }
70  T** prev = DTLTraits::prev(x);
71  T* next = DTLTraits::next(x);
72  **prev = *next;
73  if (DTLTraits::non_empty(*next)) *DTLTraits::prev(*next) = *prev;
74  *DTLTraits::prev(x) = nullptr;
75  *DTLTraits::next(x) = {};
76  }
77 
78  DoublyThreadedList() = default;
79 
80  // Defining move constructor so that when resizing container, the prev pointer
81  // of the next(head_) doesn't point to the old head_ but rather to the new
82  // one.
83  DoublyThreadedList(DoublyThreadedList&& other) V8_NOEXCEPT {
84  head_ = other.head_;
85  if (DTLTraits::non_empty(head_)) {
86  *DTLTraits::prev(head_) = &head_;
87  }
88  other.head_ = {};
89  }
90 
91  // Add `x` at the beginning of the list. `x` will not be visible to any
92  // existing iterator. Does not invalidate any existing iterator.
93  void PushFront(T x) {
94  DCHECK(empty(*DTLTraits::next(x)));
95  DCHECK_EQ(*DTLTraits::prev(x), nullptr);
96  *DTLTraits::next(x) = head_;
97  *DTLTraits::prev(x) = &head_;
98  if (DTLTraits::non_empty(head_)) {
99  *DTLTraits::prev(head_) = DTLTraits::next(x);
100  }
101  head_ = x;
102  }
103 
104  T Front() const {
105  DCHECK(!empty());
106  return *begin();
107  }
108 
109  void PopFront() {
110  DCHECK(!empty());
111  Remove(Front());
112  }
113 
114  bool empty() const { return !DTLTraits::non_empty(head_); }
115 
116  iterator begin() const { return iterator{head_}; }
117  end_iterator end() const { return end_iterator{}; }
118 
119  // Removes the element at `it`, and make `it` point to the next element.
120  // Iterators on the same element as `it` are invalidated. Other iterators are
121  // not affected.
122  iterator RemoveAt(iterator& it) {
123  DCHECK(DTLTraits::non_empty(it.curr_));
124  T curr = *it;
125  T next = *DTLTraits::next(curr);
126  Remove(curr);
127  return iterator{next};
128  }
129 
130  bool Contains(T needle) const {
131  const bool in_use = DTLTraits::in_use(needle);
132  DCHECK_EQ(in_use, ContainsSlow(needle));
133  return in_use;
134  }
135 
136  bool ContainsSlow(T needle) const {
137  for (T element : *this) {
138  if (element == needle) {
139  return true;
140  }
141  }
142  return false;
143  }
144 
145  private:
146  static bool empty(T x) { return !DTLTraits::non_empty(x); }
147  T head_{};
148 };
