25 CodeEntry* EntryForVMState(StateTag tag) {
26  switch (tag) {

30  case PARSER:
31  case COMPILER:

48 Symbolizer::SymbolizedSample Symbolizer::SymbolizeTickSample(
49  const TickSample& sample) {
50  ProfileStackTrace stack_trace;
51  // Conservatively reserve space for stack frames + pc + function + vm-state.
52  // There could in fact be more of them because of inlined entries.
53  stack_trace.reserve(sample.frames_count + 3);
54 
55  // The ProfileNode knows nothing about all versions of generated code for
56  // the same JS function. The position information associated with
57  // the latest version of generated code is used to find a source position
58  // for a JS function. Then, the detected source position is passed to
59  // ProfileNode to increase the tick count for this source position.
60  LineAndColumn src_pos = {};
61  bool src_pos_not_found = true;
62 
63  if (sample.pc != nullptr) {
64  if (sample.has_external_callback && sample.state == EXTERNAL) {
65  // Don't use PC when in external callback code, as it can point
66  // inside a callback's code, and we will erroneously report
67  // that a callback calls itself.
68  stack_trace.push_back(
69  {FindEntry(reinterpret_cast<Address>(sample.external_callback_entry)),
70  LineAndColumn{}});
71  } else {
72  Address attributed_pc = reinterpret_cast<Address>(sample.pc);
73  Address pc_entry_instruction_start = kNullAddress;
74  CodeEntry* pc_entry =
75  FindEntry(attributed_pc, &pc_entry_instruction_start);
76  // If there is no pc_entry, we're likely in native code. Find out if the
77  // top of the stack (the return address) was pointing inside a JS
78  // function, meaning that we have encountered a frameless invocation.
79  if (!pc_entry && !sample.has_external_callback) {
80  attributed_pc = reinterpret_cast<Address>(sample.tos);
81  pc_entry = FindEntry(attributed_pc, &pc_entry_instruction_start);
82  }
83  // If pc is in the function code before it set up stack frame or after the
84  // frame was destroyed, StackFrameIteratorForProfiler incorrectly thinks
85  // that ebp contains the return address of the current function and skips
86  // the caller's frame. Check for this case and just skip such samples.
87  if (pc_entry) {
88  int pc_offset =
89  static_cast<int>(attributed_pc - pc_entry_instruction_start);
90  // TODO(petermarshall): pc_offset can still be negative in some cases.
91  src_pos = pc_entry->GetSourcePosition(pc_offset);
92  if (src_pos.line == v8::CpuProfileNode::kNoLineNumberInfo) {
93  src_pos = pc_entry->line_and_column();
94  }
95  src_pos_not_found = false;
96  stack_trace.push_back({pc_entry, src_pos});
97 
98  if (pc_entry->builtin() == Builtin::kFunctionPrototypeApply ||
99  pc_entry->builtin() == Builtin::kFunctionPrototypeCall) {
100  // When current function is either the Function.prototype.apply or the
101  // Function.prototype.call builtin the top frame is either frame of
102  // the calling JS function or internal frame.
103  // In the latter case we know the caller for sure but in the
104  // former case we don't so we simply replace the frame with
105  // 'unresolved' entry.
106  if (!sample.has_external_callback) {
107  ProfilerStats::Instance()->AddReason(
108  ProfilerStats::Reason::kInCallOrApply);
109  stack_trace.push_back(
110  {CodeEntry::unresolved_entry(), LineAndColumn{}});
111  }
112  }
113  }
114  }
115 
116  for (unsigned i = 0; i < sample.frames_count; ++i) {
117  Address stack_pos = reinterpret_cast<Address>(sample.stack[i]);
118  Address instruction_start = kNullAddress;
119  CodeEntry* entry = FindEntry(stack_pos, &instruction_start);
120  LineAndColumn position = {};
121  if (entry) {
122  // Find out if the entry has an inlining stack associated.
123  int pc_offset = static_cast<int>(stack_pos - instruction_start);
124  // TODO(petermarshall): pc_offset can still be negative in some cases.
125  const std::vector<CodeEntryAndPosition>* inline_stack =
126  entry->GetInlineStack(pc_offset);
127  if (inline_stack) {
128  LineAndColumn most_inlined_frame_pos =
129  entry->GetSourcePosition(pc_offset);
130  for (auto inline_stack_entry : *inline_stack) {
131  stack_trace.push_back(inline_stack_entry);
132  }
133 
134  // This is a bit of a messy hack. The position for the most-inlined
135  // frame (the function at the end of the chain of function calls) has
136  // the wrong position in inline_stack. The actual position in
137  // this function is stored in the SourcePositionTable in entry. We fix
138  // up the position for the most-inlined frame here.
139  // TODO(petermarshall): Remove this and use a tree with a node per
140  // inlining_id.
141  DCHECK(!inline_stack->empty());
142  size_t index = stack_trace.size() - inline_stack->size();
143  stack_trace[index].line_and_column = most_inlined_frame_pos;
144  }
145  // Skip unresolved frames (e.g. internal frame) and get source position
146  // of the first JS caller.
147  if (src_pos_not_found) {
148  src_pos = entry->GetSourcePosition(pc_offset);
149  if (src_pos.line == v8::CpuProfileNode::kNoLineNumberInfo) {
150  src_pos = entry->line_and_column();
151  }
152  src_pos_not_found = false;
153  }
154  position = entry->GetSourcePosition(pc_offset);
155 
156  // The inline stack contains the top-level function i.e. the same
157  // function as entry. We don't want to add it twice. The one from the
158  // inline stack has the correct position for this particular inlining
159  // so we use it instead of pushing entry to stack_trace.
160  if (inline_stack) continue;
161  }
162  stack_trace.push_back({entry, position});
163  }
164  }
165 
166  if (v8_flags.prof_browser_mode) {
167  bool no_symbolized_entries = true;
168  for (auto e : stack_trace) {
169  if (e.code_entry != nullptr) {
170  no_symbolized_entries = false;
171  break;
172  }
173  }
174  // If no frames were symbolized, put the VM state entry in.
175  if (no_symbolized_entries) {
176  if (sample.pc == nullptr) {
177  ProfilerStats::Instance()->AddReason(ProfilerStats::Reason::kNullPC);
178  } else {
179  ProfilerStats::Instance()->AddReason(
180  ProfilerStats::Reason::kNoSymbolizedFrames);
181  }
182  stack_trace.push_back({EntryForVMState(sample.state), LineAndColumn{}});
183  }
184  }
185 
186  return SymbolizedSample{stack_trace, src_pos};
187 }
