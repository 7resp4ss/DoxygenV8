170 void WriteBarrier::MarkingSlow(Tagged<HeapObject> host,
171  JSDispatchHandle handle) {
172 #ifdef V8_ENABLE_LEAPTIERING
173  MarkingBarrier* marking_barrier = CurrentMarkingBarrier(host);
174 
175  // The JSDispatchTable is only marked during major GC so we can skip the
176  // barrier if we're only doing a minor GC.
177  // This is mostly an optimization, but it does help avoid scenarios where a
178  // minor GC marking barrier marks a table entry as alive but not the Code
179  // object contained in it (because it's not a young-gen object).
180  if (marking_barrier->is_minor()) return;
181 
182  // Mark both the table entry and its content.
183  JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
184  static_assert(JSDispatchTable::kWriteBarrierSetsEntryMarkBit);
185  jdt->Mark(handle);
186  marking_barrier->MarkValue(host, jdt->GetCode(handle));
187 
188  // We don't need to record a slot here because the entries in the
189  // JSDispatchTable are not compacted and because the pointers stored in the
190  // table entries are updated after compacting GC.
191  static_assert(!JSDispatchTable::kSupportsCompaction);
192 #else
193  UNREACHABLE();
194 #endif
195 }

198  Tagged<HeapObject> host = Cast<HeapObject>(Tagged<Object>(raw_host));
199  MaybeObjectSlot slot(raw_slot);
200  Address value = (*slot).ptr();
201 
202 #ifdef V8_MAP_PACKING
203  if (slot.address() == host.address()) {
204  // Clear metadata bits and fix object tag.
205  value = (value & ~Internals::kMapWordMetadataMask &
206  ~Internals::kMapWordXorMask) |

211 #if DEBUG

218  MarkingBarrier* barrier = CurrentMarkingBarrier(host);

288 // static
289 bool WriteBarrier::PageFlagsAreConsistent(Tagged<HeapObject> object) {
290  MemoryChunkMetadata* metadata = MemoryChunkMetadata::FromHeapObject(object);
291  MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
292 
293  // Slim chunk flags consistency.
294  CHECK_EQ(chunk->IsFlagSet(MemoryChunk::INCREMENTAL_MARKING),
295  chunk->IsMarking());
296 
297  if (!v8_flags.sticky_mark_bits) {
298  AllocationSpace identity = metadata->owner()->identity();
299 
300  // Generation consistency.
301  CHECK_EQ(identity == NEW_SPACE || identity == NEW_LO_SPACE,
302  chunk->InYoungGeneration());
303  }
304 
305  // Marking consistency.
306  if (metadata->IsWritable()) {
307  // RO_SPACE can be shared between heaps, so we can't use RO_SPACE objects to
308  // find a heap. The exception is when the ReadOnlySpace is writeable, during
309  // bootstrapping, so explicitly allow this case.
310  Heap* heap = Heap::FromWritableHeapObject(object);
311  if (chunk->InWritableSharedSpace()) {
312  // The marking bit is not set for chunks in shared spaces during MinorMS
313  // concurrent marking.
314  CHECK_EQ(chunk->IsMarking(),
315  heap->incremental_marking()->IsMajorMarking());
316  } else {
317  CHECK_EQ(chunk->IsMarking(), heap->incremental_marking()->IsMarking());
318  }
319  } else {
320  // Non-writable RO_SPACE must never have marking flag set.
321  CHECK(!chunk->IsMarking());
322  }
323  return true;
324 }

397 enum RangeWriteBarrierMode {
398  kDoGenerationalOrShared = 1 << 0,
399  kDoMarking = 1 << 1,
400  kDoEvacuationSlotRecording = 1 << 2,
401 };
402 
403 template <int kModeMask, typename TSlot>
404 void ForRangeImpl(Heap* heap, MemoryChunk* source_chunk,
405  Tagged<HeapObject> object, TSlot start_slot, TSlot end_slot) {
406  // At least one of generational or marking write barrier should be requested.
407  static_assert(kModeMask & (kDoGenerationalOrShared | kDoMarking));
408  // kDoEvacuationSlotRecording implies kDoMarking.
409  static_assert(!(kModeMask & kDoEvacuationSlotRecording) ||
410  (kModeMask & kDoMarking));
411 
412  MarkingBarrier* marking_barrier = nullptr;
413  static constexpr Tagged_t kPageMask =
414  ~static_cast<Tagged_t>(PageMetadata::kPageSize - 1);
415  Tagged_t cached_uninteresting_page =
416  static_cast<Tagged_t>(heap->read_only_space()->FirstPageAddress()) &
417  kPageMask;
418 
419  if (kModeMask & kDoMarking) {
420  marking_barrier = WriteBarrier::CurrentMarkingBarrier(object);
421  }
422 
423  MarkCompactCollector* collector = heap->mark_compact_collector();
424  MutablePageMetadata* source_page_metadata =
425  MutablePageMetadata::cast(source_chunk->Metadata());
426 
427  for (TSlot slot = start_slot; slot < end_slot; ++slot) {
428  // If we *only* need the generational or shared WB, we can skip objects
429  // residing on uninteresting pages.
430  Tagged_t compressed_page;
431  if (kModeMask == kDoGenerationalOrShared) {
432  Tagged_t tagged_value = *slot.location();
433  if (HAS_SMI_TAG(tagged_value)) continue;
434  compressed_page = tagged_value & kPageMask;
435  if (compressed_page == cached_uninteresting_page) {
436 #if DEBUG
437  typename TSlot::TObject value = *slot;
438  Tagged<HeapObject> value_heap_object;
439  if (value.GetHeapObject(&value_heap_object)) {
440  CHECK(!HeapLayout::InYoungGeneration(value_heap_object));
441  CHECK(!HeapLayout::InWritableSharedSpace(value_heap_object));
442  }
443 #endif // DEBUG
444  continue;
445  }
446  // Fall through to decompressing the pointer and fetching its actual
447  // page header flags.
448  }
449  typename TSlot::TObject value = *slot;
450  Tagged<HeapObject> value_heap_object;
451  if (!value.GetHeapObject(&value_heap_object)) continue;
452 
453  if (kModeMask & kDoGenerationalOrShared) {
454  if (HeapLayout::InYoungGeneration(value_heap_object)) {
455  RememberedSet<OLD_TO_NEW>::Insert<AccessMode::NON_ATOMIC>(
456  source_page_metadata, source_chunk->Offset(slot.address()));
457  } else if (HeapLayout::InWritableSharedSpace(value_heap_object)) {
458  RememberedSet<OLD_TO_SHARED>::Insert<AccessMode::ATOMIC>(
459  source_page_metadata, source_chunk->Offset(slot.address()));
460  } else if (kModeMask == kDoGenerationalOrShared) {
461  cached_uninteresting_page = compressed_page;
462  }
463  }
464 
465  if (kModeMask & kDoMarking) {
466  marking_barrier->MarkValue(object, value_heap_object);
467  if (kModeMask & kDoEvacuationSlotRecording) {
468  collector->RecordSlot(source_chunk, HeapObjectSlot(slot),
469  value_heap_object);
470  }
471  }
472  }
473 }

487 template <typename TSlot>
488 // static
489 void WriteBarrier::ForRange(Heap* heap, Tagged<HeapObject> object,
490  TSlot start_slot, TSlot end_slot) {
491  if (v8_flags.disable_write_barriers) {
492  return;
493  }
494  MemoryChunk* source_chunk = MemoryChunk::FromHeapObject(object);
495 
496  base::Flags<RangeWriteBarrierMode> mode;
497  if (!HeapLayout::InYoungGeneration(object) &&
498  !source_chunk->InWritableSharedSpace()) {
499  mode |= kDoGenerationalOrShared;
500  }
501 
502  if (heap->incremental_marking()->IsMarking()) {
503  mode |= kDoMarking;
504  if (!source_chunk->ShouldSkipEvacuationSlotRecording()) {
505  mode |= kDoEvacuationSlotRecording;
506  }
507  }
508 
509  switch (mode) {
510  // Nothing to be done.
511  case 0:
512  return;
513  // Generational only.
514  case kDoGenerationalOrShared:
515  return ForRangeImpl<kDoGenerationalOrShared>(heap, source_chunk, object,
516  start_slot, end_slot);
517  // Marking, no evacuation slot recording.
518  case kDoMarking:
519  return ForRangeImpl<kDoMarking>(heap, source_chunk, object, start_slot,
520  end_slot);
521  // Marking with evacuation slot recording.
522  case kDoMarking | kDoEvacuationSlotRecording:
523  return ForRangeImpl<kDoMarking | kDoEvacuationSlotRecording>(
524  heap, source_chunk, object, start_slot, end_slot);
525  // Generational and marking, no evacuation slot recording.
526  case kDoGenerationalOrShared | kDoMarking:
527  return ForRangeImpl<kDoGenerationalOrShared | kDoMarking>(
528  heap, source_chunk, object, start_slot, end_slot);
529  // Generational and marking with evacuation slot recording.
530  case kDoGenerationalOrShared | kDoMarking | kDoEvacuationSlotRecording:
531  return ForRangeImpl<kDoGenerationalOrShared | kDoMarking |
532  kDoEvacuationSlotRecording>(
533  heap, source_chunk, object, start_slot, end_slot);
534  default:
535  UNREACHABLE();
536  }

541 // static
542 bool WriteBarrier::VerifyDispatchHandleMarkingState(Tagged<HeapObject> host,
543  JSDispatchHandle handle,
544  WriteBarrierMode mode) {
545 #ifdef V8_ENABLE_LEAPTIERING
546  JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
547  if (mode == SKIP_WRITE_BARRIER &&
548  WriteBarrier::IsRequired(host, jdt->GetCode(handle))) {
549  return false;
550  }
551 
552  if (CurrentMarkingBarrier(host)->is_not_major()) return true;
553 
554  // Ensure we don't have a black -> white -> black edge. This could happen when
555  // skipping a write barrier while concurrently the dispatch entry is marked
556  // from another JSFunction.
557  if (ReadOnlyHeap::Contains(host) ||
558  (IsMarking(host) && mode != SKIP_WRITE_BARRIER) ||
559  !CurrentMarkingBarrier(host)->IsMarked(host)) {
560  return true;
561  }
562  if (jdt->IsMarked(handle)) {
563  return true;
564  }
565  Tagged<Code> value = jdt->GetCode(handle);
566  if (ReadOnlyHeap::Contains(value)) {
567  return true;
568  }
569  return !CurrentMarkingBarrier(host)->IsMarked(value);
570 #else
571  return true;
572 #endif // V8_ENABLE_LEAPTIERING
573 }
