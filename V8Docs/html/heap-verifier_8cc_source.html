164 template <typename TSlot>
165 void VerifyPointersVisitor::VerifyPointersImpl(TSlot start, TSlot end) {
166  for (TSlot slot = start; slot < end; ++slot) {
167  typename TSlot::TObject object = slot.load(cage_base());
168 #ifdef V8_ENABLE_DIRECT_HANDLE
169  if (object.ptr() == kTaggedNullAddress) continue;
170 #endif
171  Tagged<HeapObject> heap_object;
172  if (object.GetHeapObject(&heap_object)) {
173  VerifyHeapObjectImpl(heap_object);
174  } else {
175  CHECK(IsSmi(object) || object.IsCleared() ||
176  MapWord::IsPacked(object.ptr()));
177  }
178  }
179 }

187  CHECK_EQ(ObjectFields::kMaybePointers,

226 class VerifySharedHeapObjectVisitor : public VerifyPointersVisitor {
227  public:
228  explicit VerifySharedHeapObjectVisitor(Heap* heap)
229  : VerifyPointersVisitor(heap),
230  shared_space_(heap->shared_space()),
231  shared_trusted_space_(heap->shared_trusted_space()),
232  shared_lo_space_(heap->shared_lo_space()),
233  shared_trusted_lo_space_(heap->shared_trusted_lo_space()) {
234  CHECK_NOT_NULL(shared_space_);
235  CHECK_NOT_NULL(shared_lo_space_);
236  }
237 
238  protected:
239  void VerifyPointers(Tagged<HeapObject> host, MaybeObjectSlot start,
240  MaybeObjectSlot end) override {
241  if (!host.is_null()) {
242  Tagged<Map> map = host->map();
243  CHECK(ReadOnlyHeap::Contains(map) || shared_space_->Contains(map));
244  }
245  VerifyPointersVisitor::VerifyPointers(host, start, end);
246 
247  for (MaybeObjectSlot current = start; current < end; ++current) {
248  Tagged<HeapObject> heap_object;
249  if ((*current).GetHeapObject(&heap_object)) {
250  MemoryChunk* chunk = MemoryChunk::FromHeapObject(heap_object);
251  CHECK(chunk->InReadOnlySpace() || chunk->InWritableSharedSpace());
252  CHECK(ReadOnlyHeap::Contains(heap_object) ||
253  shared_space_->Contains(heap_object) ||
254  shared_lo_space_->Contains(heap_object) ||
255  shared_trusted_space_->Contains(heap_object) ||
256  shared_trusted_lo_space_->Contains(heap_object));
257  }
258  }
259  }
260 
261  private:
262  SharedSpace* shared_space_;
263  SharedTrustedSpace* shared_trusted_space_;
264  SharedLargeObjectSpace* shared_lo_space_;
265  SharedTrustedLargeObjectSpace* shared_trusted_lo_space_;
266 };

326 void HeapVerification::Verify() {
327  CHECK(heap()->HasBeenSetUp());
328  AllowGarbageCollection allow_gc;
329  SafepointKind safepoint_kind = isolate()->is_shared_space_isolate()
330  ? SafepointKind::kGlobal
331  : SafepointKind::kIsolate;
332  SafepointScope safepoint_scope(isolate(), safepoint_kind);
333  HandleScope scope(isolate());
334 
335  heap()->MakeHeapIterable();
336  heap()->FreeLinearAllocationAreas();
337 
338  // TODO(v8:13257): Currently we don't iterate through the stack conservatively
339  // when verifying the heap.
340  VerifyPointersVisitor visitor(heap());
341  heap()->IterateRoots(&visitor,
342  base::EnumSet<SkipRoot>{SkipRoot::kConservativeStack});
343 
344  if (!isolate()->context().is_null() &&
345  !isolate()->raw_native_context().is_null()) {
346  Tagged<Object> normalized_map_cache =
347  isolate()->raw_native_context()->normalized_map_cache();
348 
349  if (IsNormalizedMapCache(normalized_map_cache)) {
350  Cast<NormalizedMapCache>(normalized_map_cache)
351  ->NormalizedMapCacheVerify(isolate());
352  }
353  }
354 
355 #if V8_ENABLE_WEBASSEMBLY
356  // wasm_canonical_rtts holds weak references to maps or (strong) undefined.
357  Tagged<WeakFixedArray> canonical_rtts = heap()->wasm_canonical_rtts();
358  for (int i = 0, e = canonical_rtts->length(); i < e; ++i) {
359  Tagged<MaybeObject> maybe_rtt = canonical_rtts->get(i);
360  if (maybe_rtt.IsCleared()) continue;
361  CHECK(maybe_rtt.IsWeak());
362  CHECK(IsMap(maybe_rtt.GetHeapObjectAssumeWeak()));
363  }
364  wasm::WasmExportWrapperCache::Verify(heap());
365 #endif // V8_ENABLE_WEBASSEMBLY
366 
367  // The heap verifier can't deal with partially deserialized objects, so
368  // disable it if a deserializer is active.
369  // TODO(leszeks): Enable verification during deserialization, e.g. by only
370  // blocklisting objects that are in a partially deserialized state.
371  if (isolate()->has_active_deserializer()) return;
372 
373  VerifySmisVisitor smis_visitor;
374  heap()->IterateSmiRoots(&smis_visitor);
375 
376  VerifySpace(new_space());
377 
378  VerifySpace(old_space());
379  VerifySpace(shared_space());
380  VerifySpace(code_space());
381 
382  VerifySpace(lo_space());
383  VerifySpace(new_lo_space());
384  VerifySpace(shared_lo_space());
385  VerifySpace(code_lo_space());
386 
387  VerifySpace(trusted_space());
388  VerifySpace(shared_trusted_space());
389  VerifySpace(trusted_lo_space());
390  VerifySpace(shared_trusted_lo_space());
391 
392  isolate()->string_table()->VerifyIfOwnedBy(isolate());
393 
394 #if DEBUG
395  heap()->VerifyCommittedPhysicalMemory();
396 #endif // DEBUG
397 }

406 void HeapVerification::VerifyPage(const MemoryChunkMetadata* chunk_metadata) {
407  const MemoryChunk* chunk = chunk_metadata->Chunk();
408 
409  CHECK(!current_chunk_.has_value());
410  CHECK(!chunk->IsFlagSet(MemoryChunk::PAGE_NEW_OLD_PROMOTION));
411  CHECK(!chunk->IsFlagSet(MemoryChunk::FROM_PAGE));
412  CHECK(!chunk->IsFlagSet(MemoryChunk::WILL_BE_PROMOTED));
413  CHECK(!chunk->IsQuarantined());
414  if (chunk->InReadOnlySpace()) {
415  CHECK_NULL(chunk_metadata->owner());
416  } else {
417  CHECK_EQ(chunk_metadata->heap(), heap());
418  CHECK_EQ(chunk_metadata->owner()->identity(), current_space_identity());
419  }
420  current_chunk_ = chunk_metadata;
421 }

424  CHECK_EQ(chunk, *current_chunk_);
425  current_chunk_.reset();
426 }

428 void HeapVerification::VerifyObject(Tagged<HeapObject> object) {
429  CHECK_EQ(MemoryChunkMetadata::FromHeapObject(object), *current_chunk_);
430 
431  // Verify object map.
432  VerifyObjectMap(object);
433 
434  // The object itself should look OK.
435  Object::ObjectVerify(object, isolate_);
436 
437  // Verify outgoing references.
438  VerifyOutgoingPointers(object);
439 
440  // Verify remembered set.
441  if (!heap_->incremental_marking()->IsMinorMarking()) {
442  // Minor incremental marking "steals" the remembered sets from pages.
443  VerifyRememberedSetFor(object);
444  }
445 }

447 void HeapVerification::VerifyOutgoingPointers(Tagged<HeapObject> object) {
448  switch (current_space_identity()) {
449  case RO_SPACE: {
450  VerifyReadOnlyPointersVisitor visitor(heap());
451  visitor.Visit(object);
452  break;
453  }
454 
455  case SHARED_SPACE:
456  case SHARED_TRUSTED_SPACE:
457  case SHARED_LO_SPACE:
458  case SHARED_TRUSTED_LO_SPACE: {
459  VerifySharedHeapObjectVisitor visitor(heap());
460  visitor.Visit(object);
461  break;
462  }
463 
464  case NEW_SPACE:
465  case OLD_SPACE:
466  case TRUSTED_SPACE:
467  case CODE_SPACE:
468  case LO_SPACE:
469  case NEW_LO_SPACE:
470  case CODE_LO_SPACE:
471  case TRUSTED_LO_SPACE: {
472  VerifyPointersVisitor visitor(heap());
473  visitor.Visit(object);
474  break;
475  }
476  }
477 }

479 void HeapVerification::VerifyObjectMap(Tagged<HeapObject> object) {
480  // The first word should be a map, and we expect all map pointers to be
481  // in map space or read-only space.
482  MapWord map_word = object->map_word(cage_base_, kRelaxedLoad);
483  CHECK(!map_word.IsForwardingAddress());
484  Tagged<Map> map = map_word.ToMap();
485  CHECK(IsMap(map, cage_base_));
486  CHECK(ReadOnlyHeap::Contains(map) || old_space()->Contains(map) ||
487  (shared_space() && shared_space()->Contains(map)));
488 
489  if (HeapLayout::InYoungGeneration(object)) {
490  // The object should not be code or a map.
491  CHECK(!IsMap(object, cage_base_));
492  CHECK(!IsAbstractCode(object, cage_base_));
493  } else if (current_space_identity() == RO_SPACE) {
494  CHECK(!IsExternalString(object));
495  CHECK(!IsJSArrayBuffer(object));
496  }
497 }

545  RelocInfo* rinfo) override {
546  Tagged<Object> target =
547  InstructionStream::FromTargetAddress(rinfo->target_address());
548  if (ShouldHaveBeenRecorded(host, target)) {
549  CHECK(InTypedSet(SlotType::kCodeEntry, rinfo->pc()) ||
550  (rinfo->IsInConstantPool() &&
551  InTypedSet(SlotType::kConstPoolCodeEntry,
552  rinfo->constant_pool_entry_address())));
553  }

556  void VisitEmbeddedPointer(Tagged<InstructionStream> host,
557  RelocInfo* rinfo) override {
558  Tagged<Object> target = rinfo->target_object(cage_base());
559  if (ShouldHaveBeenRecorded(host, target)) {
560  CHECK(InTypedSet(SlotType::kEmbeddedObjectFull, rinfo->pc()) ||
561  InTypedSet(SlotType::kEmbeddedObjectCompressed, rinfo->pc()) ||
562  (rinfo->IsInConstantPool() &&
563  InTypedSet(SlotType::kConstPoolEmbeddedObjectCompressed,
564  rinfo->constant_pool_entry_address())) ||
565  (rinfo->IsInConstantPool() &&
566  InTypedSet(SlotType::kConstPoolEmbeddedObjectFull,
567  rinfo->constant_pool_entry_address())));
568  }
569  }

577  }

579  void VisitProtectedPointer(Tagged<TrustedObject> host,

614  bool ShouldHaveBeenRecorded(Tagged<HeapObject> host,

685  });

855  Tagged<HeapObject> object,

878 // static
879 void HeapVerifier::VerifySafeMapTransition(Heap* heap,
880  Tagged<HeapObject> object,
881  Tagged<Map> new_map) {
882  PtrComprCageBase cage_base(heap->isolate());
883 
884  if (IsJSObject(object, cage_base)) {
885  // Without double unboxing all in-object fields of a JSObject are tagged.
886  return;
887  }
888 
889  if (IsString(object, cage_base) &&
890  (new_map == ReadOnlyRoots(heap).thin_two_byte_string_map() ||
891  new_map == ReadOnlyRoots(heap).thin_one_byte_string_map())) {
892  // When transitioning a string to ThinString,
893  // Heap::NotifyObjectLayoutChange doesn't need to be invoked because only
894  // tagged fields are introduced.
895  return;
896  }
897 
898  if (v8_flags.shared_string_table && IsString(object, cage_base) &&
899  InstanceTypeChecker::IsInternalizedString(new_map->instance_type())) {
900  // In-place internalization does not change a string's fields.
901  //
902  // When sharing the string table, the setting and re-setting of maps below
903  // can race when there are parallel internalization operations, causing
904  // CHECKs to fail.
905  return;
906  }
907 
908  // Check that the set of slots before and after the transition match.
909  SlotCollectingVisitor old_visitor(heap->isolate());
910  old_visitor.Visit(object);
911  // Collect slots in object for new map.
912  SlotCollectingVisitor new_visitor(heap->isolate());
913  new_visitor.Visit(new_map, object);
914  CHECK_EQ(new_visitor.number_of_slots(), old_visitor.number_of_slots());
915  for (int i = 0; i < new_visitor.number_of_slots(); i++) {
916  CHECK_EQ(new_visitor.slot(i), old_visitor.slot(i));
917  }
918 #ifdef V8_EXTERNAL_CODE_SPACE
919  CHECK_EQ(new_visitor.number_of_code_slots(),
920  old_visitor.number_of_code_slots());
921  for (int i = 0; i < new_visitor.number_of_code_slots(); i++) {
922  CHECK_EQ(new_visitor.code_slot(i), old_visitor.code_slot(i));
923  }
924 #endif // V8_EXTERNAL_CODE_SPACE
925 }
