38 InternedV8JsScript::Type GetJsScriptType(Tagged<Script> script) {
39  if (script->compilation_type() == Script::CompilationType::kEval) {
40  return InternedV8JsScript::TYPE_EVAL;
41  }
42 
43  // TODO(carlscab): Camillo to extend the Script::Type enum. compilation_type
44  // will no longer be needed.
45 
46  switch (script->type()) {
47  case Script::Type::kNative:
48  return InternedV8JsScript::TYPE_NATIVE;
49  case Script::Type::kExtension:
50  return InternedV8JsScript::TYPE_EXTENSION;
51  case Script::Type::kNormal:
52  return InternedV8JsScript::TYPE_NORMAL;
53 #if V8_ENABLE_WEBASSEMBLY
54  case Script::Type::kWasm:
55  UNREACHABLE();
56 #endif // V8_ENABLE_WEBASSEMBLY
57  case Script::Type::kInspector:
58  return InternedV8JsScript::TYPE_INSPECTOR;
59  }
60 }

143 uint64_t CodeDataSourceIncrementalState::InternIsolate(Isolate& isolate) {
144  auto [it, was_inserted] = isolates_.emplace(isolate.id(), next_isolate_iid());
145  uint64_t iid = it->second;
146  if (!was_inserted) {
147  return iid;
148  }
149 
150  auto* isolate_proto = serialized_interned_data_->add_v8_isolate();
151  isolate_proto->set_iid(iid);
152  isolate_proto->set_isolate_id(isolate.id());
153  isolate_proto->set_pid(base::OS::GetCurrentProcessId());
154  isolate_proto->set_embedded_blob_code_start_address(
155  reinterpret_cast<uint64_t>(isolate.embedded_blob_code()));
156  isolate_proto->set_embedded_blob_code_size(isolate.embedded_blob_code_size());
157  if (auto* code_range = isolate.heap()->code_range(); code_range != nullptr) {
158  auto* v8_code_range = isolate_proto->set_code_range();
159  v8_code_range->set_base_address(code_range->base());
160  v8_code_range->set_size(code_range->size());
161  if (code_range == IsolateGroup::current()->GetCodeRange()) {
162  // FIXME(42204573): Belongs to isolate group, not process.
163  v8_code_range->set_is_process_wide(true);
164  }
165  if (auto* embedded_builtins_start = code_range->embedded_blob_code_copy();
166  embedded_builtins_start != nullptr) {
167  v8_code_range->set_embedded_blob_code_copy_start_address(
168  reinterpret_cast<uint64_t>(embedded_builtins_start));
169  }
170  }
171 
172  return iid;
173 }

175 uint64_t CodeDataSourceIncrementalState::InternJsScript(Isolate& isolate,
176  Tagged<Script> script) {
177  auto [it, was_inserted] = scripts_.emplace(
178  CodeDataSourceIncrementalState::ScriptUniqueId{isolate.id(),
179  script->id()},
180  next_script_iid());
181  uint64_t iid = it->second;
182  if (!was_inserted) {
183  return iid;
184  }
185 
186  auto* proto = serialized_interned_data_->add_v8_js_script();
187  proto->set_iid(iid);
188  proto->set_script_id(script->id());
189  proto->set_type(GetJsScriptType(script));
190  if (IsString(script->name())) {
191  PerfettoV8String(Cast<String>(script->name()))
192  .WriteToProto(*proto->set_name());
193  }
194  if (log_script_sources() && IsString(script->source())) {
195  PerfettoV8String(Cast<String>(script->source()))
196  .WriteToProto(*proto->set_source());
197  }
198 
199  return iid;
200 }

202 uint64_t CodeDataSourceIncrementalState::InternJsFunction(
203  Isolate& isolate, DirectHandle<SharedFunctionInfo> info,
204  uint64_t v8_js_script_iid, int line_num, int column_num) {
205  DirectHandle<String> function_name =
206  SharedFunctionInfo::DebugName(&isolate, info);
207  uint64_t v8_js_function_name_iid = InternJsFunctionName(*function_name);
208 
209  auto [it, was_inserted] = functions_.emplace(
210  CodeDataSourceIncrementalState::Function{
211  v8_js_script_iid, info->is_toplevel(), info->StartPosition()},
212  next_function_iid());
213  const uint64_t iid = it->second;
214  if (!was_inserted) {
215  return iid;
216  }
217 
218  auto* function_proto = serialized_interned_data_->add_v8_js_function();
219  function_proto->set_iid(iid);
220  function_proto->set_v8_js_function_name_iid(v8_js_function_name_iid);
221  function_proto->set_v8_js_script_iid(v8_js_script_iid);
222  function_proto->set_kind(GetJsFunctionKind(info->kind()));
223  int32_t start_position = info->StartPosition();
224  if (start_position >= 0) {
225  function_proto->set_byte_offset(static_cast<uint32_t>(start_position));
226  }
227  if (line_num > 0 && column_num > 0) {
228  function_proto->set_line(static_cast<uint32_t>(line_num));
229  function_proto->set_column(static_cast<uint32_t>(column_num));
230  }
231 
232  return iid;
233 }
