15 class V8_EXPORT_PRIVATE LoopFinder {
16  // This analyzer finds which loop each Block of a graph belongs to, and
17  // computes a list of all of the loops headers.
18  //
19  // A block is considered to "belong to a loop" if there is a forward-path (ie,
20  // without taking backedges) from this block to the backedge of the loop.
21  //
22  // This analysis runs in O(number of blocks), iterating each block once, and
23  // iterating blocks that are in a loop twice.
24  //
25  // Implementation:
26  // LoopFinder::Run walks the blocks of the graph backwards, and when it
27  // reaches a LoopHeader, it calls LoopFinder::VisitLoop.
28  // LoopFinder::VisitLoop iterates all of the blocks of the loop backwards,
29  // starting from the backedge, and stopping upon reaching the loop header. It
30  // marks the blocks that don't have a `parent_loops_` set as being part of the
31  // current loop (= sets their `parent_loops_` to the current loop header). If
32  // it finds a block that already has a `parent_loops_` set, it means that this
33  // loop contains an inner loop, so we skip this inner block as set the
34  // `has_inner_loops` bit.
35  //
36  // By iterating the blocks backwards in Run, we are guaranteed that inner
37  // loops are visited before their outer loops. Walking the graph forward
38  // doesn't work quite as nicely:
39  // - When seeing loop headers for the 1st time, we wouldn't have visited
40  // their inner loops yet.
41  // - If we decided to still iterate forward but to call VisitLoop when
42  // reaching their backedge rather than their header, it would work in most
43  // cases but not all, since the backedge of an outer loop can have a
44  // BlockIndex that is smaller than the one of an inner loop.
45  public:
46  enum class ConfigFlags : uint8_t { kFindCalls };
47  using Config = base::EnumSet<ConfigFlags, int8_t>;
48 
49  struct LoopInfo {
50  const Block* start = nullptr;
51  const Block* end = nullptr;
52  bool has_inner_loops = false;
53  size_t block_count = 0; // Number of blocks in this loop
54  // (excluding inner loops)
55  size_t op_count = 0; // Upper bound on the number of operations in this
56  // loop (excluding inner loops). This is computed
57  // using "end - begin" for each block, which can be
58  // more than the number of operations when some
59  // operations are large (like CallOp and
60  // FrameStateOp typically).
61  bool has_any_call = false; // True if this loop contains a Call.
62  };
63  LoopFinder(Zone* phase_zone, const Graph* input_graph, Config config)
64  : phase_zone_(phase_zone),
65  input_graph_(input_graph),
66  config_(config),
67  loop_headers_(input_graph->block_count(), nullptr, phase_zone),
68  loop_header_info_(phase_zone),
69  queue_(phase_zone) {
70  Run();
71  }
72 
73  const ZoneUnorderedMap<const Block*, LoopInfo>& LoopHeaders() const {
74  return loop_header_info_;
75  }
76  const Block* GetLoopHeader(const Block* block) const {
77  return loop_headers_[block->index()];
78  }
79  LoopInfo GetLoopInfo(const Block* block) const {
80  DCHECK(block->IsLoop());
81  auto it = loop_header_info_.find(block);
82  DCHECK_NE(it, loop_header_info_.end());
83  return it->second;
84  }

86  struct BlockCmp {
87  bool operator()(const Block* a, const Block* b) const {
88  return a->index().id() < b->index().id();
89  }
90  };
91  ZoneSet<const Block*, BlockCmp> GetLoopBody(const Block* loop_header);

94  void Run();
95  LoopInfo VisitLoop(const Block* header);
96 
97  // If Config contains kFindCalls, then CollectLoopInfo looks at the operations
98  // in {block} to see if any of them is a Call.
99  void CollectLoopInfo(const Block* block, LoopInfo* info);
100 
101  Zone* phase_zone_;
102  const Graph* input_graph_;
103  Config config_;
104 
105  // Map from block to the loop header of the closest enclosing loop. For loop
106  // headers, this map contains the enclosing loop header, rather than the
107  // identity.
108  // For instance, if a loop B1 contains a loop B2 which contains a block B3,
109  // {loop_headers_} will map:
110  // B3 -> B2
111  // B2 -> B1
112  // B1 -> nullptr (if B1 is an outermost loop)
113  FixedBlockSidetable<const Block*> loop_headers_;
114 
115  // Map from Loop headers to the LoopInfo for their loops. Only Loop blocks
116  // have entries in this map.
117  ZoneUnorderedMap<const Block*, LoopInfo> loop_header_info_;
118 
119  // {queue_} is used in `VisitLoop`, but is declared as a class variable to
120  // reuse memory.
121  ZoneVector<const Block*> queue_;
122 };
