68  enum State {
69  // The property was not found by the iterator (this is a terminal state,
70  // iteration should not continue after hitting a not-found state).
71  NOT_FOUND,
72  // This is an initial state of a lookup starting from primitive String.
73  // This case requires special handling since String is not a JSReceiver
74  // and we don't want to create a wrapper object just for this case.
75  STRING_LOOKUP_START_OBJECT,
76  // Typed arrays have special handling for "canonical numeric index string"
77  // (https://tc39.es/ecma262/#sec-canonicalnumericindexstring), where not
78  // finding such an index (either because of OOB, or because it's not a valid
79  // integer index) immediately returns undefined, without walking the
80  // prototype (https://tc39.es/ecma262/#sec-typedarray-get).
81  TYPED_ARRAY_INDEX_NOT_FOUND,
82  // The next lookup requires an access check -- we can continue iteration on
83  // a successful check, otherwise we should abort.
84  ACCESS_CHECK,
85  // Interceptors are API-level hooks for optionally handling a lookup in
86  // embedder code -- if their handling returns false, then we should continue
87  // the iteration, though we should be conscious that an interceptor may have
88  // side effects despite returning false and might invalidate the lookup
89  // iterator state.
90  INTERCEPTOR,
91  // Proxies are user-space hooks for handling lookups in JS code.
92  // https://tc39.es/ecma262/#proxy-exotic-object
93  JSPROXY,
94  // Accessors are hooks for property getters/setters -- these can be
95  // user-space accessors (AccessorPair), or API accessors (AccessorInfo).
96  ACCESSOR,
97  // Data properties are stored as data fields on an object (either properties
98  // or elements).
99  DATA,
100  // WasmGC objects are opaque in JS, and appear to have no properties.
101  WASM_OBJECT,
102 
103  // A LookupIterator in the transition state is in the middle of performing
104  // a data transition (that is, as part of a data property write, updating
105  // the receiver and its map to allow the write).
106  //
107  // This state is not expected to be observed while performing a lookup.
108  TRANSITION,
109 
110  // Set state_ to BEFORE_PROPERTY to ensure that the next lookup will be a
111  // PROPERTY lookup.
112  BEFORE_PROPERTY = INTERCEPTOR
113  };

378 // Similar to the LookupIterator, but for concurrent accesses from a background
379 // thread.
380 //
381 // Note: This is a work in progress, intended to bundle code related to
382 // concurrent lookups here. In its current state, the class is obviously not an
383 // 'iterator'. Still, keeping the name for now, with the intent to clarify
384 // names and implementation once we've gotten some experience with more
385 // involved logic.
386 // TODO(jgruber, v8:7790): Consider using a LookupIterator-style interface.
387 // TODO(jgruber, v8:7790): Consider merging back into the LookupIterator once
388 // functionality and constraints are better known.
389 class ConcurrentLookupIterator final : public AllStatic {
390  public:
391  // Tri-state to distinguish between 'not-present' and 'who-knows' failures.
392  enum Result {
393  kPresent, // The value was found.
394  kNotPresent, // No value exists.
395  kGaveUp, // The operation can't be completed.
396  };
397 
398  // Implements the own data property lookup for the specialized case of
399  // fixed_cow_array backing stores (these are only in use for array literal
400  // boilerplates). The contract is that the elements, elements kind, and array
401  // length passed to this function should all be read from the same JSArray
402  // instance; but due to concurrency it's possible that they may not be
403  // consistent among themselves (e.g. the elements kind may not match the
404  // given elements backing store). We are thus extra-careful to handle
405  // exceptional situations.
406  V8_EXPORT_PRIVATE static std::optional<Tagged<Object>> TryGetOwnCowElement(
407  Isolate* isolate, Tagged<FixedArray> array_elements,
408  ElementsKind elements_kind, int array_length, size_t index);
409 
410  // As above, the contract is that the elements and elements kind should be
411  // read from the same holder, but this function is implemented defensively to
412  // tolerate concurrency issues.
413  V8_EXPORT_PRIVATE static Result TryGetOwnConstantElement(
414  Tagged<Object>* result_out, Isolate* isolate, LocalIsolate* local_isolate,
415  Tagged<JSObject> holder, Tagged<FixedArrayBase> elements,
416  ElementsKind elements_kind, size_t index);
417 
418  // Implements the own data property lookup for the specialized case of
419  // strings.
420  V8_EXPORT_PRIVATE static Result TryGetOwnChar(Tagged<String>* result_out,
421  Isolate* isolate,
422  LocalIsolate* local_isolate,
423  Tagged<String> string,
424  size_t index);
425 
426  // This method reimplements the following sequence in a concurrent setting:
427  //
428  // LookupIterator it(holder, isolate, name, LookupIterator::OWN);
429  // it.TryLookupCachedProperty();
430  // if (it.state() == LookupIterator::DATA) it.GetPropertyCell();
431  V8_EXPORT_PRIVATE static std::optional<Tagged<PropertyCell>>
432  TryGetPropertyCell(Isolate* isolate, LocalIsolate* local_isolate,
433  DirectHandle<JSGlobalObject> holder,
434  DirectHandle<Name> name);
435 };
