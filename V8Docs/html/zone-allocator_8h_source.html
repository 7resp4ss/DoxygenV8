55 // A recycling zone allocator maintains a free list of deallocated chunks
56 // to reuse on subsequent allocations. The free list management is purposely
57 // very simple and works best for data-structures which regularly allocate and
58 // free blocks of similar sized memory (such as std::deque).
59 template <typename T>
60 class RecyclingZoneAllocator : public ZoneAllocator<T> {
61  public:
62  explicit RecyclingZoneAllocator(Zone* zone)
63  : ZoneAllocator<T>(zone), free_list_(nullptr) {}
64  template <typename U>
65  RecyclingZoneAllocator(const RecyclingZoneAllocator<U>& other) V8_NOEXCEPT
66  : ZoneAllocator<T>(other),
67  free_list_(nullptr) {}
68 
69  T* allocate(size_t n) {
70  // Only check top block in free list, since this will be equal to or larger
71  // than the other blocks in the free list.
72  if (free_list_ && free_list_->size >= n) {
73  T* return_val = reinterpret_cast<T*>(free_list_);
74  free_list_ = free_list_->next;
75  return return_val;
76  }
77  return ZoneAllocator<T>::allocate(n);
78  }
79 
80  void deallocate(T* p, size_t n) {
81  if ((sizeof(T) * n < sizeof(FreeBlock))) return;
82 
83  // Only add block to free_list if it is equal or larger than previous block
84  // so that allocation stays O(1) only having to look at the top block.
85  if (!free_list_ || free_list_->size <= n) {
86  // Store the free-list within the block being deallocated.
87  DCHECK((sizeof(T) * n >= sizeof(FreeBlock)));
88  FreeBlock* new_free_block = reinterpret_cast<FreeBlock*>(p);
89 
90  new_free_block->size = n;
91  new_free_block->next = free_list_;
92  free_list_ = new_free_block;
93  }
94  }
95 
96  private:
97  struct FreeBlock {
98  FreeBlock* next;
99  size_t size;
100  };
101 
102  FreeBlock* free_list_;
103 };
