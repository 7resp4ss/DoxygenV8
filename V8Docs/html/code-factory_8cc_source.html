141 // static
142 Callable CodeFactory::ArrayNoArgumentConstructor(
143  Isolate* isolate, ElementsKind kind,
144  AllocationSiteOverrideMode override_mode) {
145 #define CASE(kind_caps, kind_camel, mode_camel) \
146  case kind_caps: \
147  return Builtins::CallableFor( \
148  isolate, \
149  Builtin::kArrayNoArgumentConstructor_##kind_camel##_##mode_camel);
150  if (override_mode == DONT_OVERRIDE && AllocationSite::ShouldTrack(kind)) {
151  DCHECK(IsSmiElementsKind(kind));
152  switch (kind) {
153  CASE(PACKED_SMI_ELEMENTS, PackedSmi, DontOverride);
154  CASE(HOLEY_SMI_ELEMENTS, HoleySmi, DontOverride);
155  default:
156  UNREACHABLE();
157  }
158  } else {
159  DCHECK(override_mode == DISABLE_ALLOCATION_SITES ||
160  !AllocationSite::ShouldTrack(kind));
161  switch (kind) {
162  CASE(PACKED_SMI_ELEMENTS, PackedSmi, DisableAllocationSites);
163  CASE(HOLEY_SMI_ELEMENTS, HoleySmi, DisableAllocationSites);
164  CASE(PACKED_ELEMENTS, Packed, DisableAllocationSites);
165  CASE(HOLEY_ELEMENTS, Holey, DisableAllocationSites);
166  CASE(PACKED_DOUBLE_ELEMENTS, PackedDouble, DisableAllocationSites);
167  CASE(HOLEY_DOUBLE_ELEMENTS, HoleyDouble, DisableAllocationSites);
168  default:
169  UNREACHABLE();
170  }
171  }
172 #undef CASE
173 }
