69 MaybeDirectHandle<JSRelativeTimeFormat> JSRelativeTimeFormat::New(
70  Isolate* isolate, DirectHandle<Map> map, DirectHandle<Object> locales,
71  DirectHandle<Object> input_options) {
72  // 1. Let requestedLocales be ? CanonicalizeLocaleList(locales).
73  Maybe<std::vector<std::string>> maybe_requested_locales =
74  Intl::CanonicalizeLocaleList(isolate, locales);
75  MAYBE_RETURN(maybe_requested_locales, DirectHandle<JSRelativeTimeFormat>());
76  std::vector<std::string> requested_locales =
77  maybe_requested_locales.FromJust();
78 
79  // 2. Set options to ? CoerceOptionsToObject(options).
80  DirectHandle<JSReceiver> options;
81  const char* service = "Intl.RelativeTimeFormat";
82  ASSIGN_RETURN_ON_EXCEPTION(
83  isolate, options, CoerceOptionsToObject(isolate, input_options, service));
84 
85  // 4. Let opt be a new Record.
86  // 5. Let matcher be ? GetOption(options, "localeMatcher", "string", «
87  // "lookup", "best fit" », "best fit").
88  // 6. Set opt.[[localeMatcher]] to matcher.
89  Maybe<Intl::MatcherOption> maybe_locale_matcher =
90  Intl::GetLocaleMatcher(isolate, options, service);
91  MAYBE_RETURN(maybe_locale_matcher, MaybeDirectHandle<JSRelativeTimeFormat>());
92  Intl::MatcherOption matcher = maybe_locale_matcher.FromJust();
93 
94  // 7. Let _numberingSystem_ be ? GetOption(_options_, `"numberingSystem"`,
95  // `"string"`, *undefined*, *undefined*).
96  std::string numbering_system_str;
97  Maybe<bool> maybe_numberingSystem =
98  Intl::GetNumberingSystem(isolate, options, service, numbering_system_str);
99  // 8. If _numberingSystem_ is not *undefined*, then
100  // a. If _numberingSystem_ does not match the
101  // `(3*8alphanum) *("-" (3*8alphanum))` sequence, throw a *RangeError*
102  // exception.
103  MAYBE_RETURN(maybe_numberingSystem,
104  MaybeDirectHandle<JSRelativeTimeFormat>());
105 
106  // 9. Set _opt_.[[nu]] to _numberingSystem_.
107 
108  // 10. Let localeData be %RelativeTimeFormat%.[[LocaleData]].
109  // 11. Let r be
110  // ResolveLocale(%RelativeTimeFormat%.[[AvailableLocales]],
111  // requestedLocales, opt,
112  // %RelativeTimeFormat%.[[RelevantExtensionKeys]], localeData).
113  Intl::ResolvedLocale r;
114  if (!Intl::ResolveLocale(isolate, JSRelativeTimeFormat::GetAvailableLocales(),
115  requested_locales, matcher, {"nu"})
116  .To(&r)) {
117  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
118  }
119 
120  UErrorCode status = U_ZERO_ERROR;
121 
122  icu::Locale icu_locale = r.icu_locale;
123  if (maybe_numberingSystem.FromJust()) {
124  auto nu_extension_it = r.extensions.find("nu");
125  if (nu_extension_it != r.extensions.end() &&
126  nu_extension_it->second != numbering_system_str) {
127  icu_locale.setUnicodeKeywordValue("nu", nullptr, status);
128  DCHECK(U_SUCCESS(status));
129  }
130  }
131  // 12. Let locale be r.[[Locale]].
132  Maybe<std::string> maybe_locale_str = Intl::ToLanguageTag(icu_locale);
133  MAYBE_RETURN(maybe_locale_str, MaybeDirectHandle<JSRelativeTimeFormat>());
134 
135  // 13. Set relativeTimeFormat.[[Locale]] to locale.
136  DirectHandle<String> locale_str =
137  isolate->factory()->NewStringFromAsciiChecked(
138  maybe_locale_str.FromJust().c_str());
139 
140  // 14. Set relativeTimeFormat.[[NumberingSystem]] to r.[[nu]].
141  if (maybe_numberingSystem.FromJust() &&
142  Intl::IsValidNumberingSystem(numbering_system_str)) {
143  icu_locale.setUnicodeKeywordValue("nu", numbering_system_str, status);
144  DCHECK(U_SUCCESS(status));
145  }
146  // 15. Let dataLocale be r.[[DataLocale]].
147 
148  // 16. Let s be ? GetOption(options, "style", "string",
149  // «"long", "short", "narrow"», "long").
150  Maybe<Style> maybe_style = GetStringOption<Style>(
151  isolate, options, "style", service,
152  std::to_array<const std::string_view>({"long", "short", "narrow"}),
153  std::array{Style::LONG, Style::SHORT, Style::NARROW}, Style::LONG);
154  MAYBE_RETURN(maybe_style, MaybeDirectHandle<JSRelativeTimeFormat>());
155  Style style_enum = maybe_style.FromJust();
156 
157  // 17. Set relativeTimeFormat.[[Style]] to s.
158 
159  // 18. Let numeric be ? GetOption(options, "numeric", "string",
160  // «"always", "auto"», "always").
161  Maybe<Numeric> maybe_numeric = GetStringOption<Numeric>(
162  isolate, options, "numeric", service,
163  std::to_array<const std::string_view>({"always", "auto"}),
164  std::array{Numeric::ALWAYS, Numeric::AUTO}, Numeric::ALWAYS);
165  MAYBE_RETURN(maybe_numeric, MaybeDirectHandle<JSRelativeTimeFormat>());
166  Numeric numeric_enum = maybe_numeric.FromJust();
167 
168  // 19. Set relativeTimeFormat.[[Numeric]] to numeric.
169 
170  // 23. Let relativeTimeFormat.[[NumberFormat]] be
171  // ? Construct(%NumberFormat%, « nfLocale, nfOptions »).
172  icu::NumberFormat* number_format =
173  icu::NumberFormat::createInstance(icu_locale, UNUM_DECIMAL, status);
174  if (U_FAILURE(status)) {
175  // Data build filter files excluded data in "rbnf_tree" since ECMA402 does
176  // not support "algorithmic" numbering systems. Therefore we may get the
177  // U_MISSING_RESOURCE_ERROR here. Fallback to locale without the numbering
178  // system and create the object again.
179  if (status == U_MISSING_RESOURCE_ERROR) {
180  delete number_format;
181  status = U_ZERO_ERROR;
182  icu_locale.setUnicodeKeywordValue("nu", nullptr, status);
183  DCHECK(U_SUCCESS(status));
184  number_format =
185  icu::NumberFormat::createInstance(icu_locale, UNUM_DECIMAL, status);
186  }
187  if (U_FAILURE(status) || number_format == nullptr) {
188  delete number_format;
189  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
190  }
191  }
192 
193  if (number_format->getDynamicClassID() ==
194  icu::DecimalFormat::getStaticClassID()) {
195  icu::DecimalFormat* decimal_format =
196  static_cast<icu::DecimalFormat*>(number_format);
197  decimal_format->setMinimumGroupingDigits(-2);
198  }
199 
200  // Change UDISPCTX_CAPITALIZATION_NONE to other values if
201  // ECMA402 later include option to change capitalization.
202  // Ref: https://github.com/tc39/proposal-intl-relative-time/issues/11
203  std::shared_ptr<icu::RelativeDateTimeFormatter> icu_formatter =
204  std::make_shared<icu::RelativeDateTimeFormatter>(
205  icu_locale, number_format, toIcuStyle(style_enum),
206  UDISPCTX_CAPITALIZATION_NONE, status);
207  if (U_FAILURE(status) || icu_formatter == nullptr) {
208  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
209  }
210 
211  DirectHandle<String> numbering_system_string =
212  isolate->factory()->NewStringFromAsciiChecked(
213  Intl::GetNumberingSystem(icu_locale).c_str());
214 
215  DirectHandle<Managed<icu::RelativeDateTimeFormatter>> managed_formatter =
216  Managed<icu::RelativeDateTimeFormatter>::From(isolate, 0,
217  std::move(icu_formatter));
218 
219  DirectHandle<JSRelativeTimeFormat> relative_time_format_holder =
220  Cast<JSRelativeTimeFormat>(
221  isolate->factory()->NewFastOrSlowJSObjectFromMap(map));
222 
223  DisallowGarbageCollection no_gc;
224  relative_time_format_holder->set_flags(0);
225  relative_time_format_holder->set_locale(*locale_str);
226  relative_time_format_holder->set_numberingSystem(*numbering_system_string);
227  relative_time_format_holder->set_numeric(numeric_enum);
228  relative_time_format_holder->set_icu_formatter(*managed_formatter);
229 
230  // 25. Return relativeTimeFormat.
231  return relative_time_format_holder;
232 }

310 bool GetURelativeDateTimeUnit(DirectHandle<String> unit,
311  URelativeDateTimeUnit* unit_enum) {
312  std::string unit_str = unit->ToStdString();
313  if (unit_str == "second" || unit_str == "seconds") {
314  *unit_enum = UDAT_REL_UNIT_SECOND;
315  } else if (unit_str == "minute" || unit_str == "minutes") {
316  *unit_enum = UDAT_REL_UNIT_MINUTE;
317  } else if (unit_str == "hour" || unit_str == "hours") {
318  *unit_enum = UDAT_REL_UNIT_HOUR;
319  } else if (unit_str == "day" || unit_str == "days") {
320  *unit_enum = UDAT_REL_UNIT_DAY;
321  } else if (unit_str == "week" || unit_str == "weeks") {
322  *unit_enum = UDAT_REL_UNIT_WEEK;
323  } else if (unit_str == "month" || unit_str == "months") {
324  *unit_enum = UDAT_REL_UNIT_MONTH;
325  } else if (unit_str == "quarter" || unit_str == "quarters") {
326  *unit_enum = UDAT_REL_UNIT_QUARTER;
327  } else if (unit_str == "year" || unit_str == "years") {
328  *unit_enum = UDAT_REL_UNIT_YEAR;
329  } else {
330  return false;
331  }
332  return true;
333 }

335 template <typename T>
336 MaybeDirectHandle<T> FormatCommon(
337  Isolate* isolate, DirectHandle<JSRelativeTimeFormat> format,
338  Handle<Object> value_obj, Handle<Object> unit_obj, const char* func_name,
339  MaybeDirectHandle<T> (*formatToResult)(
340  Isolate*, const icu::FormattedRelativeDateTime&, DirectHandle<String>,
341  bool)) {
342  // 3. Let value be ? ToNumber(value).
343  DirectHandle<Object> value;
344  ASSIGN_RETURN_ON_EXCEPTION(isolate, value,
345  Object::ToNumber(isolate, value_obj));
346  double number = Object::NumberValue(*value);
347  // 4. Let unit be ? ToString(unit).
348  Handle<String> unit;
349  ASSIGN_RETURN_ON_EXCEPTION(isolate, unit,
350  Object::ToString(isolate, unit_obj));
351  // 4. If isFinite(value) is false, then throw a RangeError exception.
352  if (!std::isfinite(number)) {
353  THROW_NEW_ERROR(
354  isolate, NewRangeError(
355  MessageTemplate::kNotFiniteNumber,
356  isolate->factory()->NewStringFromAsciiChecked(func_name)));
357  }
358  icu::RelativeDateTimeFormatter* formatter = format->icu_formatter()->raw();
359  DCHECK_NOT_NULL(formatter);
360  URelativeDateTimeUnit unit_enum;
361  if (!GetURelativeDateTimeUnit(unit, &unit_enum)) {
362  THROW_NEW_ERROR(
363  isolate,
364  NewRangeError(MessageTemplate::kInvalidUnit,
365  isolate->factory()->NewStringFromAsciiChecked(func_name),
366  unit));
367  }
368  UErrorCode status = U_ZERO_ERROR;
369  icu::FormattedRelativeDateTime formatted =
370  (format->numeric() == JSRelativeTimeFormat::Numeric::ALWAYS)
371  ? formatter->formatNumericToValue(number, unit_enum, status)
372  : formatter->formatToValue(number, unit_enum, status);
373  if (U_FAILURE(status)) {
374  THROW_NEW_ERROR(isolate, NewTypeError(MessageTemplate::kIcuError));
375  }
376  return formatToResult(isolate, formatted, UnitAsString(isolate, unit_enum),
377  IsNaN(*value));
378 }

418 MaybeDirectHandle<JSArray> FormatToJSArray(
419  Isolate* isolate, const icu::FormattedRelativeDateTime& formatted,
420  DirectHandle<String> unit, bool is_nan) {
421  UErrorCode status = U_ZERO_ERROR;
422  icu::UnicodeString string = formatted.toString(status);
423 
424  Factory* factory = isolate->factory();
425  DirectHandle<JSArray> array = factory->NewJSArray(0);
426  icu::ConstrainedFieldPosition cfpos;
427  cfpos.constrainCategory(UFIELD_CATEGORY_NUMBER);
428  int32_t index = 0;
429 
430  int32_t previous_end = 0;
431  DirectHandle<String> substring;
432  std::vector<std::pair<int32_t, int32_t>> groups;
433  while (formatted.nextPosition(cfpos, status) && U_SUCCESS(status)) {
434  int32_t category = cfpos.getCategory();
435  int32_t field = cfpos.getField();
436  int32_t start = cfpos.getStart();
437  int32_t limit = cfpos.getLimit();
438  if (category == UFIELD_CATEGORY_NUMBER) {
439  if (field == UNUM_GROUPING_SEPARATOR_FIELD) {
440  groups.push_back(std::pair<int32_t, int32_t>(start, limit));
441  continue;
442  }
443  if (start > previous_end) {
444  Maybe<bool> maybe_added =
445  AddLiteral(isolate, array, string, index++, previous_end, start);
446  MAYBE_RETURN(maybe_added, DirectHandle<JSArray>());
447  }
448  if (field == UNUM_INTEGER_FIELD) {
449  for (auto start_limit : groups) {
450  if (start_limit.first > start) {
451  Maybe<bool> maybe_added =
452  AddUnit(isolate, array, string, index++,
453  NumberFormatSpan(field, start, start_limit.first), unit,
454  is_nan);
455  MAYBE_RETURN(maybe_added, DirectHandle<JSArray>());
456  maybe_added =
457  AddUnit(isolate, array, string, index++,
458  NumberFormatSpan(UNUM_GROUPING_SEPARATOR_FIELD,
459  start_limit.first, start_limit.second),
460  unit, is_nan);
461  MAYBE_RETURN(maybe_added, DirectHandle<JSArray>());
462  start = start_limit.second;
463  }
464  }
465  }
466  Maybe<bool> maybe_added =
467  AddUnit(isolate, array, string, index++,
468  NumberFormatSpan(field, start, limit), unit, is_nan);
469  MAYBE_RETURN(maybe_added, DirectHandle<JSArray>());
470  previous_end = limit;
471  }
472  }
473  if (U_FAILURE(status)) {
474  THROW_NEW_ERROR(isolate, NewTypeError(MessageTemplate::kIcuError));
475  }
476  if (string.length() > previous_end) {
477  Maybe<bool> maybe_added = AddLiteral(isolate, array, string, index,
478  previous_end, string.length());
479  MAYBE_RETURN(maybe_added, DirectHandle<JSArray>());
480  }
481 
482  JSObject::ValidateElements(isolate, *array);
483  return array;
484 }
