25 Reduction RedundancyElimination::Reduce(Node* node) {
26  if (node_checks_.Get(node)) return NoChange();
27  switch (node->opcode()) {
28  case IrOpcode::kCheckBigInt:
29  case IrOpcode::kCheckedBigIntToBigInt64:
30  case IrOpcode::kCheckBounds:
31  case IrOpcode::kCheckClosure:
32  case IrOpcode::kCheckEqualsInternalizedString:
33  case IrOpcode::kCheckEqualsSymbol:
34  case IrOpcode::kCheckFloat64Hole:
35  case IrOpcode::kCheckHeapObject:
36  case IrOpcode::kCheckIf:
37  case IrOpcode::kCheckInternalizedString:
38  case IrOpcode::kCheckNotTaggedHole:
39  case IrOpcode::kCheckNumber:
40  case IrOpcode::kCheckNumberFitsInt32:
41  case IrOpcode::kCheckReceiver:
42  case IrOpcode::kCheckReceiverOrNullOrUndefined:
43  case IrOpcode::kCheckSmi:
44  case IrOpcode::kCheckString:
45  case IrOpcode::kCheckStringOrStringWrapper:
46  case IrOpcode::kCheckSymbol:
47  case IrOpcode::kTypeGuard:
48  // These are not really check nodes, but behave the same in that they can be
49  // folded together if repeated with identical inputs.
50  case IrOpcode::kStringCharCodeAt:
51  case IrOpcode::kStringCodePointAt:
52  case IrOpcode::kStringFromCodePointAt:
53  case IrOpcode::kStringSubstring:
54 #define SIMPLIFIED_OP(Opcode) case IrOpcode::k##Opcode:
55  SIMPLIFIED_CHECKED_OP_LIST(SIMPLIFIED_OP)
56  SIMPLIFIED_BIGINT_BINOP_LIST(SIMPLIFIED_OP)
57 #undef SIMPLIFIED_OP
58  return ReduceCheckNode(node);
59  case IrOpcode::kSpeculativeNumberEqual:
60  case IrOpcode::kSpeculativeNumberLessThan:
61  case IrOpcode::kSpeculativeNumberLessThanOrEqual:
62  return ReduceSpeculativeNumberComparison(node);
63  case IrOpcode::kSpeculativeNumberAdd:
64  case IrOpcode::kSpeculativeNumberSubtract:
65  case IrOpcode::kSpeculativeAdditiveSafeIntegerAdd:
66  case IrOpcode::kSpeculativeAdditiveSafeIntegerSubtract:
67  case IrOpcode::kSpeculativeSmallIntegerAdd:
68  case IrOpcode::kSpeculativeSmallIntegerSubtract:
69  case IrOpcode::kSpeculativeToNumber:
70  return ReduceSpeculativeNumberOperation(node);
71  case IrOpcode::kEffectPhi:
72  return ReduceEffectPhi(node);
73  case IrOpcode::kDead:
74  break;
75  case IrOpcode::kStart:
76  return ReduceStart(node);
77  default:
78  return ReduceOtherNode(node);
79  }
80  return NoChange();
81 }

96 bool RedundancyElimination::EffectPathChecks::Equals(
97  EffectPathChecks const* that) const {
98  if (this->size_ != that->size_) return false;
99  Check* this_head = this->head_;
100  Check* that_head = that->head_;
101  while (this_head != that_head) {
102  if (this_head->node != that_head->node) return false;
103  this_head = this_head->next;
104  that_head = that_head->next;
105  }
106  return true;
107 }

109 void RedundancyElimination::EffectPathChecks::Merge(
110  EffectPathChecks const* that) {
111  // Change the current check list to a longest common tail of this check
112  // list and the other list.
113 
114  // First, we throw away the prefix of the longer list, so that
115  // we have lists of the same length.
116  Check* that_head = that->head_;
117  size_t that_size = that->size_;
118  while (that_size > size_) {
119  that_head = that_head->next;
120  that_size--;
121  }
122  while (size_ > that_size) {
123  head_ = head_->next;
124  size_--;
125  }
126 
127  // Then we go through both lists in lock-step until we find
128  // the common tail.
129  while (head_ != that_head) {
130  DCHECK_LT(0u, size_);
131  DCHECK_NOT_NULL(head_);
132  size_--;
133  head_ = head_->next;
134  that_head = that_head->next;
135  }
136 }

147 struct Subsumption {
148  enum class Kind {
149  kNone,
150  kImplicit,
151  kWithConversion,
152  };
153 
154  static Subsumption None() { return Subsumption(Kind::kNone, nullptr); }
155  static Subsumption Implicit() {
156  return Subsumption(Kind::kImplicit, nullptr);
157  }
158  static Subsumption WithConversion(const Operator* conversion_op) {
159  return Subsumption(Kind::kWithConversion, conversion_op);
160  }
161 
162  bool IsNone() const { return kind_ == Kind::kNone; }
163  bool IsImplicit() const { return kind_ == Kind::kImplicit; }
164  bool IsWithConversion() const { return kind_ == Kind::kWithConversion; }
165  const Operator* conversion_operator() const {
166  DCHECK(IsWithConversion());
167  return conversion_op_;
168  }
169 
170  private:
171  Subsumption(Kind kind, const Operator* conversion_op)
172  : kind_(kind), conversion_op_(conversion_op) {
173  DCHECK_EQ(kind_ == Kind::kWithConversion, conversion_op_ != nullptr);
174  }
175 
176  Kind kind_;
177  const Operator* conversion_op_;
178 };

180 // Does check {a} subsume check {b}?
181 Subsumption CheckSubsumes(Node const* a, Node const* b,
182  MachineOperatorBuilder* machine) {
183  Subsumption subsumption = Subsumption::Implicit();
184  if (a->op() != b->op()) {
185  if (a->opcode() == IrOpcode::kCheckInternalizedString &&
186  b->opcode() == IrOpcode::kCheckString) {
187  // CheckInternalizedString(node) implies CheckString(node)
188  } else if (a->opcode() == IrOpcode::kCheckString &&
189  b->opcode() == IrOpcode::kCheckStringOrStringWrapper) {
190  // CheckString(node) implies CheckStringOrStringWrapper(node)
191  } else if (a->opcode() == IrOpcode::kCheckInternalizedString &&
192  b->opcode() == IrOpcode::kCheckStringOrStringWrapper) {
193  // CheckInteralizedString(node) implies CheckStringOrStringWrapper(node)
194  } else if (a->opcode() == IrOpcode::kCheckSmi &&
195  b->opcode() == IrOpcode::kCheckNumber) {
196  // CheckSmi(node) implies CheckNumber(node)
197  } else if (a->opcode() == IrOpcode::kCheckSmi &&
198  b->opcode() == IrOpcode::kCheckNumberFitsInt32) {
199  // CheckSmi(node) implies CheckNumberFitsInt32(node)
200  } else if (a->opcode() == IrOpcode::kCheckNumberFitsInt32 &&
201  b->opcode() == IrOpcode::kCheckNumber) {
202  // CheckNumberFitsInt32(node) implies CheckNumber(node)
203  } else if (a->opcode() == IrOpcode::kCheckedTaggedSignedToInt32 &&
204  b->opcode() == IrOpcode::kCheckedTaggedToInt32) {
205  // CheckedTaggedSignedToInt32(node) implies CheckedTaggedToInt32(node)
206  } else if (a->opcode() == IrOpcode::kCheckedTaggedSignedToInt32 &&
207  b->opcode() == IrOpcode::kCheckedTaggedToArrayIndex) {
208  // CheckedTaggedSignedToInt32(node) implies
209  // CheckedTaggedToArrayIndex(node)
210  if (machine->Is64()) {
211  // On 64 bit architectures, ArrayIndex is 64 bit.
212  subsumption =
213  Subsumption::WithConversion(machine->ChangeInt32ToInt64());
214  }
215  } else if (a->opcode() == IrOpcode::kCheckedTaggedToInt32 &&
216  b->opcode() == IrOpcode::kCheckedTaggedToArrayIndex) {
217  // CheckedTaggedToInt32(node) implies CheckedTaggedToArrayIndex(node)
218  if (machine->Is64()) {
219  // On 64 bit architectures, ArrayIndex is 64 bit.
220  subsumption =
221  Subsumption::WithConversion(machine->ChangeInt32ToInt64());
222  }
223  } else if (a->opcode() == IrOpcode::kCheckReceiver &&
224  b->opcode() == IrOpcode::kCheckReceiverOrNullOrUndefined) {
225  // CheckReceiver(node) implies CheckReceiverOrNullOrUndefined(node)
226  } else if (a->opcode() != b->opcode()) {
227  return Subsumption::None();
228  } else {
229  switch (a->opcode()) {
230  case IrOpcode::kCheckBounds:
231  case IrOpcode::kCheckSmi:
232  case IrOpcode::kCheckString:
233  case IrOpcode::kCheckStringOrStringWrapper:
234  case IrOpcode::kCheckNumber:
235  case IrOpcode::kCheckNumberFitsInt32:
236  case IrOpcode::kCheckBigInt:
237  case IrOpcode::kCheckedBigIntToBigInt64:
238  break;
239  case IrOpcode::kTypeGuard: {
240  Type at = TypeGuardTypeOf(a->op());
241  Type bt = TypeGuardTypeOf(b->op());
242  // {a}.type has to be narrower than (or equal to) {b}.type for
243  // {b}.type to be a redundant check.
244  if (!at.Is(bt)) {
245  return Subsumption::None();
246  }
247  break;
248  }
249  case IrOpcode::kCheckedInt32ToTaggedSigned:
250  case IrOpcode::kCheckedInt64ToInt32:
251  case IrOpcode::kCheckedInt64ToAdditiveSafeInteger:
252  case IrOpcode::kCheckedInt64ToTaggedSigned:
253  case IrOpcode::kCheckedTaggedSignedToInt32:
254  case IrOpcode::kCheckedTaggedToTaggedPointer:
255  case IrOpcode::kCheckedTaggedToTaggedSigned:
256  case IrOpcode::kCheckedTaggedToArrayIndex:
257  case IrOpcode::kCheckedUint32Bounds:
258  case IrOpcode::kCheckedUint32ToInt32:
259  case IrOpcode::kCheckedUint32ToTaggedSigned:
260  case IrOpcode::kCheckedUint64Bounds:
261  case IrOpcode::kCheckedUint64ToInt32:
262  case IrOpcode::kCheckedUint64ToTaggedSigned:
263  break;
264  case IrOpcode::kCheckedFloat64ToInt32:
265  case IrOpcode::kCheckedFloat64ToAdditiveSafeInteger:
266  case IrOpcode::kCheckedFloat64ToInt64:
267  case IrOpcode::kCheckedTaggedToInt32:
268  case IrOpcode::kCheckedTaggedToAdditiveSafeInteger:
269  case IrOpcode::kCheckedTaggedToInt64: {
270  const CheckMinusZeroParameters& ap =
271  CheckMinusZeroParametersOf(a->op());
272  const CheckMinusZeroParameters& bp =
273  CheckMinusZeroParametersOf(b->op());
274  if (ap.mode() != bp.mode()) {
275  return Subsumption::None();
276  }
277  break;
278  }
279  case IrOpcode::kCheckedTaggedToFloat64:
280  case IrOpcode::kCheckedTruncateTaggedToWord32: {
281  CheckTaggedInputParameters const& ap =
282  CheckTaggedInputParametersOf(a->op());
283  CheckTaggedInputParameters const& bp =
284  CheckTaggedInputParametersOf(b->op());
285  // {a} subsumes {b} if the modes are either the same, or {a} checks
286  // for Number, in which case {b} will be subsumed no matter what.
287  if (ap.mode() != bp.mode() &&
288  ap.mode() != CheckTaggedInputMode::kNumber) {
289  return Subsumption::None();
290  }
291  break;
292  }
293  default:
294  DCHECK(!IsCheckedWithFeedback(a->op()));
295  return Subsumption::None();
296  }
297  }
298  }
299  for (int i = a->op()->ValueInputCount(); --i >= 0;) {
300  if (a->InputAt(i) != b->InputAt(i)) return Subsumption::None();
301  }
302  return subsumption;
303 }

305 bool TypeSubsumes(Node* node, Node* replacement) {
306  if (!NodeProperties::IsTyped(node) || !NodeProperties::IsTyped(replacement)) {
307  // If either node is untyped, we are running during an untyped optimization
308  // phase, and replacement is OK.
309  return true;
310  }
311  Type node_type = NodeProperties::GetType(node);
312  Type replacement_type = NodeProperties::GetType(replacement);
313  return replacement_type.Is(node_type);
314 }

318 Node* RedundancyElimination::EffectPathChecks::LookupCheck(
319  Node* node, JSGraph* jsgraph) const {
320  for (Check const* check = head_; check != nullptr; check = check->next) {
321  Subsumption subsumption =
322  CheckSubsumes(check->node, node, jsgraph->machine());
323  if (!subsumption.IsNone() && TypeSubsumes(node, check->node)) {
324  DCHECK(!check->node->IsDead());
325  Node* result = check->node;
326  if (subsumption.IsWithConversion()) {
327  result = jsgraph->graph()->NewNode(subsumption.conversion_operator(),
328  result);
329  }
330  return result;
331  }
332  }
333  return nullptr;
334 }

336 Node* RedundancyElimination::EffectPathChecks::LookupBoundsCheckFor(
337  Node* node) const {
338  for (Check const* check = head_; check != nullptr; check = check->next) {
339  if (check->node->opcode() == IrOpcode::kCheckBounds &&
340  check->node->InputAt(0) == node && TypeSubsumes(node, check->node) &&
341  !(CheckBoundsParametersOf(check->node->op()).flags() &
342  CheckBoundsFlag::kConvertStringAndMinusZero)) {
343  return check->node;

363 Reduction RedundancyElimination::ReduceCheckNode(Node* node) {
364  Node* const effect = NodeProperties::GetEffectInput(node);
365  EffectPathChecks const* checks = node_checks_.Get(effect);
366  // If we do not know anything about the predecessor, do not propagate just yet
367  // because we will have to recompute anyway once we compute the predecessor.
368  if (checks == nullptr) return NoChange();
369  // See if we have another check that dominates us.
370  if (Node* check = checks->LookupCheck(node, jsgraph_)) {
371  ReplaceWithValue(node, check);
372  return Replace(check);
373  }
374 
375  // Learn from this check.
376  return UpdateChecks(node, checks->AddCheck(zone(), node));
377 }

379 Reduction RedundancyElimination::ReduceEffectPhi(Node* node) {
380  Node* const control = NodeProperties::GetControlInput(node);
381  if (control->opcode() == IrOpcode::kLoop) {
382  // Here we rely on having only reducible loops:
383  // The loop entry edge always dominates the header, so we can just use
384  // the information from the loop entry edge.
385  return TakeChecksFromFirstEffect(node);
386  }
387  DCHECK_EQ(IrOpcode::kMerge, control->opcode());
388 
389  // Shortcut for the case when we do not know anything about some input.
390  int const input_count = node->op()->EffectInputCount();
391  for (int i = 0; i < input_count; ++i) {
392  Node* const effect = NodeProperties::GetEffectInput(node, i);
393  if (node_checks_.Get(effect) == nullptr) return NoChange();
394  }
395 
396  // Make a copy of the first input's checks and merge with the checks
397  // from other inputs.
398  EffectPathChecks* checks = EffectPathChecks::Copy(
399  zone(), node_checks_.Get(NodeProperties::GetEffectInput(node, 0)));
400  for (int i = 1; i < input_count; ++i) {
401  Node* const input = NodeProperties::GetEffectInput(node, i);
402  checks->Merge(node_checks_.Get(input));
403  }
404  return UpdateChecks(node, checks);
405 }

407 Reduction RedundancyElimination::ReduceSpeculativeNumberComparison(Node* node) {
408  NumberOperationHint const hint = NumberOperationHintOf(node->op());
409  Node* const first = NodeProperties::GetValueInput(node, 0);
410  Type const first_type = NodeProperties::GetType(first);
411  Node* const second = NodeProperties::GetValueInput(node, 1);
412  Type const second_type = NodeProperties::GetType(second);
413  Node* const effect = NodeProperties::GetEffectInput(node);
414  EffectPathChecks const* checks = node_checks_.Get(effect);
415 
416  // If we do not know anything about the predecessor, do not propagate just yet
417  // because we will have to recompute anyway once we compute the predecessor.
418  if (checks == nullptr) return NoChange();
419 
420  // Avoid the potentially expensive lookups below if the {node}
421  // has seen non-Smi inputs in the past, which is a clear signal
422  // that the comparison is probably not performed on a value that
423  // already passed an array bounds check.
424  if (hint == NumberOperationHint::kSignedSmall) {
425  // Don't bother trying to find a CheckBounds for the {first} input
426  // if it's type is already in UnsignedSmall range, since the bounds
427  // check is only going to narrow that range further, but the result
428  // is not going to make the representation selection any better.
429  if (!first_type.Is(Type::UnsignedSmall())) {
430  if (Node* check = checks->LookupBoundsCheckFor(first)) {
431  if (!first_type.Is(NodeProperties::GetType(check))) {
432  // Replace the {first} input with the {check}. This is safe,
433  // despite the fact that {check} can truncate -0 to 0, because
434  // the regular Number comparisons in JavaScript also identify
435  // 0 and -0 (unlike special comparisons as Object.is).
436  NodeProperties::ReplaceValueInput(node, check, 0);
437  return Changed(node).FollowedBy(
438  ReduceSpeculativeNumberComparison(node));
439  }
440  }
441  }
442 
443  // Don't bother trying to find a CheckBounds for the {second} input
444  // if it's type is already in UnsignedSmall range, since the bounds
445  // check is only going to narrow that range further, but the result
446  // is not going to make the representation selection any better.
447  if (!second_type.Is(Type::UnsignedSmall())) {
448  if (Node* check = checks->LookupBoundsCheckFor(second)) {
449  if (!second_type.Is(NodeProperties::GetType(check))) {
450  // Replace the {second} input with the {check}. This is safe,
451  // despite the fact that {check} can truncate -0 to 0, because
452  // the regular Number comparisons in JavaScript also identify
453  // 0 and -0 (unlike special comparisons as Object.is).
454  NodeProperties::ReplaceValueInput(node, check, 1);
455  return Changed(node).FollowedBy(
456  ReduceSpeculativeNumberComparison(node));
457  }
458  }
459  }
460  }
461 
462  return UpdateChecks(node, checks);
463 }

465 Reduction RedundancyElimination::ReduceSpeculativeNumberOperation(Node* node) {
466  DCHECK(node->opcode() == IrOpcode::kSpeculativeNumberAdd ||
467  node->opcode() == IrOpcode::kSpeculativeNumberSubtract ||
468  node->opcode() == IrOpcode::kSpeculativeAdditiveSafeIntegerAdd ||
469  node->opcode() == IrOpcode::kSpeculativeAdditiveSafeIntegerSubtract ||
470  node->opcode() == IrOpcode::kSpeculativeSmallIntegerAdd ||
471  node->opcode() == IrOpcode::kSpeculativeSmallIntegerSubtract ||
472  node->opcode() == IrOpcode::kSpeculativeToNumber);
473  DCHECK_EQ(1, node->op()->EffectInputCount());
474  DCHECK_EQ(1, node->op()->EffectOutputCount());
475 
476  Node* const first = NodeProperties::GetValueInput(node, 0);
477  Node* const effect = NodeProperties::GetEffectInput(node);
478  EffectPathChecks const* checks = node_checks_.Get(effect);
479  // If we do not know anything about the predecessor, do not propagate just yet
480  // because we will have to recompute anyway once we compute the predecessor.
481  if (checks == nullptr) return NoChange();
482 
483  // Check if there's a CheckBounds operation on {first}
484  // in the graph already, which we might be able to
485  // reuse here to improve the representation selection
486  // for the {node} later on.
487  if (Node* check = checks->LookupBoundsCheckFor(first)) {
488  // Only use the bounds {check} if its type is better
489  // than the type of the {first} node, otherwise we
490  // would end up replacing NumberConstant inputs with
491  // CheckBounds operations, which is kind of pointless.
492  if (!NodeProperties::GetType(first).Is(NodeProperties::GetType(check))) {
493  NodeProperties::ReplaceValueInput(node, check, 0);
494  }
495  }
496 
497  return UpdateChecks(node, checks);
498 }

518 Reduction RedundancyElimination::TakeChecksFromFirstEffect(Node* node) {
519  DCHECK_EQ(1, node->op()->EffectOutputCount());
520  Node* const effect = NodeProperties::GetEffectInput(node);
521  EffectPathChecks const* checks = node_checks_.Get(effect);
522  // If we do not know anything about the predecessor, do not propagate just yet
523  // because we will have to recompute anyway once we compute the predecessor.
524  if (checks == nullptr) return NoChange();
525  // We just propagate the information from the effect input (ideally,
526  // we would only revisit effect uses if there is change).
527  return UpdateChecks(node, checks);
528 }

530 Reduction RedundancyElimination::UpdateChecks(Node* node,
531  EffectPathChecks const* checks) {
532  EffectPathChecks const* original = node_checks_.Get(node);
533  // Only signal that the {node} has Changed, if the information about {checks}
534  // has changed wrt. the {original}.
535  if (checks != original) {
536  if (original == nullptr || !checks->Equals(original)) {
537  node_checks_.Set(node, checks);
538  return Changed(node);
539  }
540  }
541  return NoChange();
542 }
