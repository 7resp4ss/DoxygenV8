38 CodeKinds JSFunction::GetAvailableCodeKinds(IsolateForSandbox isolate) const {
39  CodeKinds result = GetAttachedCodeKinds(isolate);
40 
41  if ((result & CodeKindFlag::INTERPRETED_FUNCTION) == 0) {
42  // The SharedFunctionInfo could have attached bytecode.
43  if (shared()->HasBytecodeArray()) {
44  result |= CodeKindFlag::INTERPRETED_FUNCTION;
45  }
46  }
47 
48  if ((result & CodeKindFlag::BASELINE) == 0) {
49  // The SharedFunctionInfo could have attached baseline code.
50  if (shared()->HasBaselineCode()) {
51  result |= CodeKindFlag::BASELINE;
52  }
53  }
54 
55 #ifndef V8_ENABLE_LEAPTIERING
56  // Check the optimized code cache.
57  if (has_feedback_vector() && feedback_vector()->has_optimized_code() &&
58  !feedback_vector()
59  ->optimized_code(isolate)
60  ->marked_for_deoptimization()) {
61  Tagged<Code> code = feedback_vector()->optimized_code(isolate);
62  DCHECK(CodeKindIsOptimizedJSFunction(code->kind()));
63  result |= CodeKindToCodeKindFlag(code->kind());
64  }
65 #endif // !V8_ENABLE_LEAPTIERING
66 
67  DCHECK_EQ((result & ~kJSFunctionCodeKindsMask), 0);
68  return result;
69 }

143  return (masked_available_kinds & static_cast<CodeKinds>(~mask)) != 0;

165 // Returns false if no highest tier exists (i.e. the function is not compiled),
166 // otherwise returns true and sets highest_tier.
167 V8_WARN_UNUSED_RESULT bool HighestTierOf(CodeKinds kinds,
168  CodeKind* highest_tier) {
169  DCHECK_EQ((kinds & ~kJSFunctionCodeKindsMask), 0);
170  // Higher tiers > lower tiers.
171  static_assert(CodeKind::TURBOFAN_JS > CodeKind::INTERPRETED_FUNCTION);
172  if (kinds == 0) return false;
173  const int highest_tier_log2 =
174  31 - base::bits::CountLeadingZeros(static_cast<uint32_t>(kinds));
175  DCHECK(CodeKindIsJSFunction(static_cast<CodeKind>(highest_tier_log2)));
176  *highest_tier = static_cast<CodeKind>(highest_tier_log2);
177  return true;
178 }

182 std::optional<CodeKind> JSFunction::GetActiveTier(
183  IsolateForSandbox isolate) const {
184 #if V8_ENABLE_WEBASSEMBLY
185  // Asm/Wasm functions are currently not supported. For simplicity, this
186  // includes invalid asm.js functions whose code hasn't yet been updated to
187  // CompileLazy but is still the InstantiateAsmJs builtin.
188  if (shared()->HasAsmWasmData() ||
189  code(isolate)->builtin_id() == Builtin::kInstantiateAsmJs) {
190  return {};
191  }
192 #endif // V8_ENABLE_WEBASSEMBLY
193 
194  CodeKind highest_tier;
195  if (!HighestTierOf(GetAvailableCodeKinds(isolate), &highest_tier)) return {};
196 
197 #ifdef DEBUG
198  CHECK(highest_tier == CodeKind::TURBOFAN_JS ||
199  highest_tier == CodeKind::BASELINE ||
200  highest_tier == CodeKind::MAGLEV ||
201  highest_tier == CodeKind::INTERPRETED_FUNCTION);
202 
203  if (highest_tier == CodeKind::INTERPRETED_FUNCTION) {
204  CHECK(code(isolate)->is_interpreter_trampoline_builtin() ||
205  (CodeKindIsOptimizedJSFunction(code(isolate)->kind()) &&
206  code(isolate)->marked_for_deoptimization()) ||
207  (code(isolate)->builtin_id() == Builtin::kCompileLazy &&
208  shared()->HasBytecodeArray() && !shared()->HasBaselineCode()));
209  }
210 #endif // DEBUG
211 
212  return highest_tier;
213 }

231 bool JSFunction::CanDiscardCompiled(IsolateForSandbox isolate) const {
232  // Essentially, what we are asking here is, has this function been compiled
233  // from JS code? We can currently tell only indirectly, by looking at
234  // available code kinds. If any JS code kind exists, we can discard.
235  //
236  // Attached optimized code that is marked for deoptimization will not show up
237  // in the list of available code kinds, thus we must check for it manually.
238  //
239  // Note that when the function has not yet been compiled we also return
240  // false; that's fine, since nothing must be discarded in that case.
241  if (CodeKindIsOptimizedJSFunction(code(isolate)->kind())) return true;
242  CodeKinds result = GetAvailableCodeKinds(isolate);
243  return (result & kJSFunctionCodeKindsMask) != 0;
244 }

264 void JSFunction::RequestOptimization(Isolate* isolate, CodeKind target_kind,
265  ConcurrencyMode mode) {
266  if (!isolate->concurrent_recompilation_enabled() ||
267  isolate->bootstrapper()->IsActive()) {
268  mode = ConcurrencyMode::kSynchronous;
269  }
270 
271  DCHECK(CodeKindIsOptimizedJSFunction(target_kind));
272  DCHECK(!is_compiled(isolate) || ActiveTierIsIgnition(isolate) ||
273  ActiveTierIsBaseline(isolate) || ActiveTierIsMaglev(isolate));
274  DCHECK(!ActiveTierIsTurbofan(isolate));
275  DCHECK(shared()->HasBytecodeArray());
276  DCHECK(shared()->allows_lazy_compilation() ||
277  !shared()->optimization_disabled(target_kind));
278 
279  if (IsConcurrent(mode)) {
280  if (tiering_in_progress()) {
281  if (v8_flags.trace_concurrent_recompilation) {
282  PrintF(" ** Not marking ");
283  ShortPrint(*this);
284  PrintF(" -- already in optimization queue.\n");
285  }
286  return;
287  }
288  if (v8_flags.trace_concurrent_recompilation) {
289  PrintF(" ** Marking ");
290  ShortPrint(*this);
291  PrintF(" for concurrent %s recompilation.\n",
292  CodeKindToString(target_kind));
293  }
294  }
295 
296 #ifdef V8_ENABLE_LEAPTIERING
297  JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
298  switch (target_kind) {
299  case CodeKind::MAGLEV:
300  switch (mode) {
301  case ConcurrencyMode::kConcurrent:
302  jdt->SetTieringRequest(dispatch_handle(),
303  TieringBuiltin::kStartMaglevOptimizeJob,
304  isolate);
305  break;
306  case ConcurrencyMode::kSynchronous:
307  jdt->SetTieringRequest(dispatch_handle(),
308  TieringBuiltin::kOptimizeMaglevEager, isolate);
309  break;
310  }
311  break;
312  case CodeKind::TURBOFAN_JS:
313  switch (mode) {
314  case ConcurrencyMode::kConcurrent:
315  jdt->SetTieringRequest(dispatch_handle(),
316  TieringBuiltin::kStartTurbofanOptimizeJob,
317  isolate);
318  break;
319  case ConcurrencyMode::kSynchronous:
320  jdt->SetTieringRequest(dispatch_handle(),
321  TieringBuiltin::kOptimizeTurbofanEager,
322  isolate);
323  break;
324  }
325  break;
326  default:
327  UNREACHABLE();
328  }
329 #else
330  set_tiering_state(isolate, TieringStateFor(target_kind, mode));
331 #endif // V8_ENABLE_LEAPTIERING
332 }

334 void JSFunction::SetInterruptBudget(
335  Isolate* isolate, BudgetModification kind,
336  std::optional<CodeKind> override_active_tier) {
337  int32_t current = raw_feedback_cell()->interrupt_budget();

354 Maybe<bool> JSFunctionOrBoundFunctionOrWrappedFunction::CopyNameAndLength(
355  Isolate* isolate,
356  DirectHandle<JSFunctionOrBoundFunctionOrWrappedFunction> function,
357  DirectHandle<JSReceiver> target, DirectHandle<String> prefix,
358  int arg_count) {
359  // Setup the "length" property based on the "length" of the {target}.
360  // If the targets length is the default JSFunction accessor, we can keep the
361  // accessor that's installed by default on the
362  // JSBoundFunction/JSWrappedFunction. It lazily computes the value from the
363  // underlying internal length.
364  DirectHandle<AccessorInfo> function_length_accessor =
365  isolate->factory()->function_length_accessor();
366  LookupIterator length_lookup(isolate, target,
367  isolate->factory()->length_string(), target,
368  LookupIterator::OWN);
369  if (!IsJSFunction(*target) ||
370  length_lookup.state() != LookupIterator::ACCESSOR ||
371  !length_lookup.GetAccessors().is_identical_to(function_length_accessor)) {
372  DirectHandle<Object> length(Smi::zero(), isolate);
373  Maybe<PropertyAttributes> attributes =
374  JSReceiver::GetPropertyAttributes(&length_lookup);
375  if (attributes.IsNothing()) return Nothing<bool>();
376  if (attributes.FromJust() != ABSENT) {
377  DirectHandle<Object> target_length;
378  ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate, target_length,
379  Object::GetProperty(&length_lookup),
380  Nothing<bool>());
381  if (IsNumber(*target_length)) {
382  length = isolate->factory()->NewNumber(std::max(
383  0.0,
384  DoubleToInteger(Object::NumberValue(*target_length)) - arg_count));
385  }
386  }
387  LookupIterator it(isolate, function, isolate->factory()->length_string(),
388  function);
389  DCHECK_EQ(LookupIterator::ACCESSOR, it.state());
390  RETURN_ON_EXCEPTION_VALUE(isolate,
391  JSObject::DefineOwnPropertyIgnoreAttributes(
392  &it, length, it.property_attributes()),
393  Nothing<bool>());
394  }
395 
396  // Setup the "name" property based on the "name" of the {target}.
397  // If the target's name is the default JSFunction accessor, we can keep the
398  // accessor that's installed by default on the
399  // JSBoundFunction/JSWrappedFunction. It lazily computes the value from the
400  // underlying internal name.
401  DirectHandle<AccessorInfo> function_name_accessor =
402  isolate->factory()->function_name_accessor();
403  LookupIterator name_lookup(isolate, target, isolate->factory()->name_string(),
404  target);
405  if (!IsJSFunction(*target) ||
406  name_lookup.state() != LookupIterator::ACCESSOR ||
407  !name_lookup.GetAccessors().is_identical_to(function_name_accessor) ||
408  (name_lookup.IsFound() && !name_lookup.HolderIsReceiver())) {
409  DirectHandle<Object> target_name;
410  ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate, target_name,
411  Object::GetProperty(&name_lookup),
412  Nothing<bool>());
413  DirectHandle<String> name;
414  if (IsString(*target_name)) {
415  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
416  isolate, name,
417  Name::ToFunctionName(isolate, Cast<String>(target_name)),
418  Nothing<bool>());
419  if (!prefix.is_null()) {
420  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
421  isolate, name, isolate->factory()->NewConsString(prefix, name),
422  Nothing<bool>());
423  }
424  } else if (prefix.is_null()) {
425  name = isolate->factory()->empty_string();
426  } else {
427  name = prefix;
428  }
429  LookupIterator it(isolate, function, isolate->factory()->name_string());
430  DCHECK_EQ(LookupIterator::ACCESSOR, it.state());
431  RETURN_ON_EXCEPTION_VALUE(isolate,
432  JSObject::DefineOwnPropertyIgnoreAttributes(
433  &it, name, it.property_attributes()),
434  Nothing<bool>());
435  }
436 
437  return Just(true);
438 }

472 Maybe<int> JSBoundFunction::GetLength(Isolate* isolate,
473  DirectHandle<JSBoundFunction> function) {
474  int nof_bound_arguments = function->bound_arguments()->length();
475  while (IsJSBoundFunction(function->bound_target_function())) {
476  function = direct_handle(
477  Cast<JSBoundFunction>(function->bound_target_function()), isolate);
478  // Make sure we never overflow {nof_bound_arguments}, the number of
479  // arguments of a function is strictly limited by the max length of an
480  // JSAarray, Smi::kMaxValue is thus a reasonably good overestimate.
481  int length = function->bound_arguments()->length();
482  if (V8_LIKELY(Smi::kMaxValue - nof_bound_arguments > length)) {
483  nof_bound_arguments += length;
484  } else {
485  nof_bound_arguments = Smi::kMaxValue;
486  }
487  }
488  if (IsJSWrappedFunction(function->bound_target_function())) {
489  DirectHandle<JSWrappedFunction> target(
490  Cast<JSWrappedFunction>(function->bound_target_function()), isolate);
491  int target_length = 0;
492  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
493  isolate, target_length, JSWrappedFunction::GetLength(isolate, target),
494  Nothing<int>());
495  int length = std::max(0, target_length - nof_bound_arguments);
496  return Just(length);
497  }
498  // All non JSFunction targets get a direct property and don't use this
499  // accessor.
500  DirectHandle<JSFunction> target(
501  Cast<JSFunction>(function->bound_target_function()), isolate);
502  int target_length = target->length();
503 
504  int length = std::max(0, target_length - nof_bound_arguments);
505  return Just(length);
506 }

558 // static
559 MaybeDirectHandle<Object> JSWrappedFunction::Create(
560  Isolate* isolate, DirectHandle<NativeContext> creation_context,
561  DirectHandle<JSReceiver> value) {
562  // The value must be a callable according to the specification.
563  DCHECK(IsCallable(*value));
564  // The intermediate wrapped functions are not user-visible. And calling a
565  // wrapped function won't cause a side effect in the creation realm.
566  // Unwrap here to avoid nested unwrapping at the call site.
567  if (IsJSWrappedFunction(*value)) {
568  auto target_wrapped = Cast<JSWrappedFunction>(value);
569  value = direct_handle(target_wrapped->wrapped_target_function(), isolate);
570  }
571 
572  // 1. Let internalSlotsList be the internal slots listed in Table 2, plus
573  // [[Prototype]] and [[Extensible]].
574  // 2. Let wrapped be ! MakeBasicObject(internalSlotsList).
575  // 3. Set wrapped.[[Prototype]] to
576  // callerRealm.[[Intrinsics]].[[%Function.prototype%]].
577  // 4. Set wrapped.[[Call]] as described in 2.1.
578  // 5. Set wrapped.[[WrappedTargetFunction]] to Target.
579  // 6. Set wrapped.[[Realm]] to callerRealm.
580  DirectHandle<JSWrappedFunction> wrapped =
581  isolate->factory()->NewJSWrappedFunction(creation_context, value);
582 
583  // 7. Let result be CopyNameAndLength(wrapped, Target, "wrapped").
584  Maybe<bool> is_abrupt =
585  JSFunctionOrBoundFunctionOrWrappedFunction::CopyNameAndLength(
586  isolate, wrapped, value, DirectHandle<String>(), 0);
587 
588  // 8. If result is an Abrupt Completion, throw a TypeError exception.
589  if (is_abrupt.IsNothing()) {
590  DCHECK(isolate->has_exception());
591  DirectHandle<Object> exception(isolate->exception(), isolate);
592  isolate->clear_exception();
593 
594  // The TypeError thrown is created with creation Realm's TypeError
595  // constructor instead of the executing Realm's.
596  DirectHandle<JSFunction> type_error_function(
597  creation_context->type_error_function(), isolate);
598  DirectHandle<String> string =
599  Object::NoSideEffectsToString(isolate, exception);
600  THROW_NEW_ERROR_RETURN_VALUE(
601  isolate,

621 void JSFunction::EnsureClosureFeedbackCellArray(
622  Isolate* isolate, DirectHandle<JSFunction> function) {
623  DCHECK(function->shared()->is_compiled());
624  DCHECK(function->shared()->HasFeedbackMetadata());
625 #if V8_ENABLE_WEBASSEMBLY
626  if (function->shared()->HasAsmWasmData()) return;
627 #endif // V8_ENABLE_WEBASSEMBLY
628 
629  DirectHandle<SharedFunctionInfo> shared(function->shared(), isolate);
630  DCHECK(shared->HasBytecodeArray());
631  const bool has_closure_feedback_cell_array =
632  (function->has_closure_feedback_cell_array() ||
633  function->has_feedback_vector());
634 
635  if (has_closure_feedback_cell_array) {
636  return;
637  }
638 
639  // Many closure cell is used as a way to specify that there is no
640  // feedback cell for this function and a new feedback cell has to be
641  // allocated for this function. For ex: for eval functions, we have to create
642  // a feedback cell and cache it along with the code. It is safe to use
643  // many_closure_cell to indicate this because in regular cases, it should
644  // already have a feedback_vector / feedback cell array allocated.
645  const bool allocate_new_feedback_cell =
646  function->raw_feedback_cell() ==
647  *isolate->factory()->many_closures_cell();
648 
649  DirectHandle<ClosureFeedbackCellArray> feedback_cell_array =
650  ClosureFeedbackCellArray::New(isolate, shared);
651  if (allocate_new_feedback_cell) {
652  DirectHandle<FeedbackCell> feedback_cell =
653  isolate->factory()->NewOneClosureCell(feedback_cell_array);
654 #ifdef V8_ENABLE_LEAPTIERING
655  // This is a rare case where we copy the dispatch entry from a JSFunction
656  // to its FeedbackCell instead of the other way around.
657  // TODO(42204201): investigate whether this can be avoided so that we only
658  // ever copy a dispatch handle from a FeedbackCell to a JSFunction. That
659  // would probably require refactoring the way JSFunctions are built so that
660  // we always allocate a FeedbackCell up front (if needed).
661  DCHECK_NE(function->dispatch_handle(), kNullJSDispatchHandle);
662  // The feedback cell should never contain context specialized code.
663  DCHECK(!function->code(isolate)->is_context_specialized());
664  feedback_cell->set_dispatch_handle(function->dispatch_handle());
665 #endif // V8_ENABLE_LEAPTIERING
666  function->set_raw_feedback_cell(*feedback_cell, kReleaseStore);
667  } else {
668  function->raw_feedback_cell()->set_value(*feedback_cell_array,
669  kReleaseStore);
670  }
671  // Initialize the interrupt budget when initializing the feedback cell with
672  // the closure feedback cell for the first time, whether the feedback cell
673  // itself is new or not.
674  function->SetInterruptBudget(isolate, BudgetModification::kReset);

741 // static
742 void JSFunction::InitializeFeedbackCell(
743  Isolate* isolate, DirectHandle<JSFunction> function,
744  IsCompiledScope* is_compiled_scope,
745  bool reset_budget_for_feedback_allocation) {
746 #if V8_ENABLE_WEBASSEMBLY
747  // The following checks ensure that the feedback vectors are compatible with
748  // the feedback metadata. For Asm / Wasm functions we never allocate / use
749  // feedback vectors, so a mismatch between the metadata and feedback vector is
750  // harmless. The checks could fail for functions that has has_asm_wasm_broken
751  // set at runtime (for ex: failed instantiation).
752  if (function->shared()->HasAsmWasmData()) return;
753 #endif // V8_ENABLE_WEBASSEMBLY
754 
755  if (function->has_feedback_vector()) {
756  CHECK_EQ(function->feedback_vector()->length(),
757  function->feedback_vector()->metadata()->slot_count());
758  return;
759  }
760 
761  bool has_closure_feedback_cell_array =
762  function->has_closure_feedback_cell_array();
763  if (has_closure_feedback_cell_array) {
764  CHECK_EQ(
765  function->closure_feedback_cell_array()->length(),
766  function->shared()->feedback_metadata()->create_closure_slot_count());
767  }
768 
769  const bool needs_feedback_vector =
770  !v8_flags.lazy_feedback_allocation || v8_flags.always_turbofan ||
771  // We also need a feedback vector for certain log events, collecting type
772  // profile and more precise code coverage.
773  v8_flags.log_function_events ||
774  !isolate->is_best_effort_code_coverage() ||
775  function->shared()->cached_tiering_decision() !=
776  CachedTieringDecision::kPending;
777 
778  if (needs_feedback_vector) {
779  CreateAndAttachFeedbackVector(isolate, function, is_compiled_scope);
780  } else if (has_closure_feedback_cell_array) {
781  // Initialize the interrupt budget to the feedback vector allocation budget
782  // after a bytecode flush. We retain the closure feedback cell array on
783  // bytecode flush, so reset_budget_for_feedback_allocation is used to reset
784  // the budget in this case.
785  if (reset_budget_for_feedback_allocation) {
786  function->SetInterruptBudget(isolate, BudgetModification::kReset);
787  }
788  } else {
789  EnsureClosureFeedbackCellArray(isolate, function);
790  }
791 #ifdef V8_ENABLE_SPARKPLUG
792  // TODO(jgruber): Unduplicate these conditions from tiering-manager.cc.
793  if (function->shared()->cached_tiering_decision() !=
794  CachedTieringDecision::kPending &&
795  CanCompileWithBaseline(isolate, function->shared()) &&
796  function->ActiveTierIsIgnition(isolate)) {
797  if (v8_flags.baseline_batch_compilation) {
798  isolate->baseline_batch_compiler()->EnqueueFunction(function);
799  } else {
800  IsCompiledScope is_compiled_inner_scope(
801  function->shared()->is_compiled_scope(isolate));
802  Compiler::CompileBaseline(isolate, function, Compiler::CLEAR_EXCEPTION,
803  &is_compiled_inner_scope);
804  }
805  }
806 #endif // V8_ENABLE_SPARKPLUG
807 }

811 void SetInstancePrototype(Isolate* isolate, DirectHandle<JSFunction> function,
812  DirectHandle<JSReceiver> value) {
813  // Now some logic for the maps of the objects that are created by using this
814  // function as a constructor.
815  if (function->has_initial_map()) {
816  // If the function has allocated the initial map replace it with a
817  // copy containing the new prototype. Also complete any in-object
818  // slack tracking that is in progress at this point because it is
819  // still tracking the old copy.
820  function->CompleteInobjectSlackTrackingIfActive(isolate);
821 
822  DirectHandle<Map> initial_map(function->initial_map(), isolate);
823 
824  if (!isolate->bootstrapper()->IsActive() &&
825  initial_map->instance_type() == JS_OBJECT_TYPE) {
826  // Put the value in the initial map field until an initial map is needed.
827  // At that point, a new initial map is created and the prototype is put
828  // into the initial map where it belongs.
829  function->set_prototype_or_initial_map(*value, kReleaseStore);
830  if (IsJSObjectThatCanBeTrackedAsPrototype(*value)) {
831  // Optimize as prototype to detach it from its transition tree.
832  JSObject::OptimizeAsPrototype(Cast<JSObject>(value));
833  }
834  } else {
835  DirectHandle<Map> new_map =
836  Map::Copy(isolate, initial_map, "SetInstancePrototype");
837  JSFunction::SetInitialMap(isolate, function, new_map, value);
838  DCHECK_IMPLIES(!isolate->bootstrapper()->IsActive(),
839  *function != function->native_context()->array_function());
840  }
841 
842  // Deoptimize all code that embeds the previous initial map.
843  DependentCode::DeoptimizeDependencyGroups(
844  isolate, *initial_map, DependentCode::kInitialMapChangedGroup);
845  } else {
846  // Put the value in the initial map field until an initial map is
847  // needed. At that point, a new initial map is created and the
848  // prototype is put into the initial map where it belongs.
849  function->set_prototype_or_initial_map(*value, kReleaseStore);
850  if (IsJSObjectThatCanBeTrackedAsPrototype(*value)) {
851  // Optimize as prototype to detach it from its transition tree.
852  JSObject::OptimizeAsPrototype(Cast<JSObject>(value));
853  }
854  }

859 void JSFunction::SetPrototype(Isolate* isolate,
860  DirectHandle<JSFunction> function,
861  DirectHandle<Object> value) {
862  DCHECK(IsConstructor(*function) ||
863  IsGeneratorFunction(function->shared()->kind()));
864  DirectHandle<JSReceiver> construct_prototype;
865 
866  // If the value is not a JSReceiver, store the value in the map's
867  // constructor field so it can be accessed. Also, set the prototype
868  // used for constructing objects to the original object prototype.
869  // See ECMA-262 13.2.2.
870  if (!IsJSReceiver(*value)) {
871  // Copy the map so this does not affect unrelated functions.
872  // Remove map transitions because they point to maps with a
873  // different prototype.
874  DirectHandle<Map> new_map = Map::Copy(
875  isolate, direct_handle(function->map(), isolate), "SetPrototype");
876 
877  // Create a new {constructor, non-instance_prototype} tuple and store it
878  // in Map::constructor field.
879  DirectHandle<Object> constructor(new_map->GetConstructor(), isolate);
880  DirectHandle<Tuple2> non_instance_prototype_constructor_tuple =
881  isolate->factory()->NewTuple2(constructor, value, AllocationType::kOld);
882 
883  new_map->set_has_non_instance_prototype(true);
884  new_map->SetConstructor(*non_instance_prototype_constructor_tuple);
885 
886  JSObject::MigrateToMap(isolate, function, new_map);
887 
888  FunctionKind kind = function->shared()->kind();
889  DirectHandle<Context> native_context(function->native_context(), isolate);
890 
891  construct_prototype = direct_handle(
892  IsGeneratorFunction(kind)
893  ? IsAsyncFunction(kind)
894  ? native_context->initial_async_generator_prototype()
895  : native_context->initial_generator_prototype()
896  : native_context->initial_object_prototype(),
897  isolate);
898  } else {
899  construct_prototype = Cast<JSReceiver>(value);
900  function->map()->set_has_non_instance_prototype(false);
901  }
902 
903  SetInstancePrototype(isolate, function, construct_prototype);
904 }

931 void JSFunction::EnsureHasInitialMap(Isolate* isolate,
932  DirectHandle<JSFunction> function) {
933  DCHECK(function->has_prototype_slot());
934  DCHECK(IsConstructor(*function) ||
935  IsResumableFunction(function->shared()->kind()));
936  if (function->has_initial_map()) return;
937 
938  int expected_nof_properties =
939  CalculateExpectedNofProperties(isolate, function);
940 
941  // {CalculateExpectedNofProperties} can have had the side effect of creating
942  // the initial map (e.g. it could have triggered an optimized compilation
943  // whose dependency installation reentered {EnsureHasInitialMap}).
944  if (function->has_initial_map()) return;
945 
946  // Create a new map with the size and number of in-object properties suggested
947  // by the function.
948  InstanceType instance_type;
949  if (IsResumableFunction(function->shared()->kind())) {
950  instance_type = IsAsyncGeneratorFunction(function->shared()->kind())
951  ? JS_ASYNC_GENERATOR_OBJECT_TYPE
952  : JS_GENERATOR_OBJECT_TYPE;
953  } else {
954  instance_type = JS_OBJECT_TYPE;
955  }
956 
957  int instance_size;
958  int inobject_properties;
959  CalculateInstanceSizeHelper(instance_type, false, 0, expected_nof_properties,
960  &instance_size, &inobject_properties);
961 
962  DirectHandle<NativeContext> creation_context(function->native_context(),
963  isolate);
964  DirectHandle<Map> map = isolate->factory()->NewContextfulMap(
965  creation_context, instance_type, instance_size,
966  TERMINAL_FAST_ELEMENTS_KIND, inobject_properties);
967 
968  // Fetch or allocate prototype.
969  DirectHandle<JSPrototype> prototype;
970  if (function->has_instance_prototype()) {
971  prototype = direct_handle(function->instance_prototype(), isolate);
972  map->set_prototype(*prototype);
973  } else {
974  prototype = isolate->factory()->NewFunctionPrototype(function);
975  Map::SetPrototype(isolate, map, prototype);
976  }
977  DCHECK(map->has_fast_object_elements());
978 
979  // Finally link initial map and constructor function.
980  // This is a CHECK since the prototype could be Null according to the type
981  // system.
982  // TODO(leszeks): Figure out if this CHECK is needed.
983  CHECK(IsJSReceiver(*prototype));
984  JSFunction::SetInitialMap(isolate, function, map, prototype);
985  map->StartInobjectSlackTracking();

1128 bool FastInitializeDerivedMap(Isolate* isolate,
1129  DirectHandle<JSFunction> new_target,
1130  DirectHandle<JSFunction> constructor,
1131  DirectHandle<Map> constructor_initial_map) {
1132  // Use the default intrinsic prototype instead.
1133  if (!new_target->has_prototype_slot()) return false;
1134  // Check that |function|'s initial map still in sync with the |constructor|,
1135  // otherwise we must create a new initial map for |function|.
1136  if (new_target->has_initial_map() &&
1137  new_target->initial_map()->GetConstructor() == *constructor) {
1138  DCHECK(IsJSReceiver(new_target->instance_prototype()));
1139  return true;
1140  }
1141  InstanceType instance_type = constructor_initial_map->instance_type();
1142  DCHECK(CanSubclassHaveInobjectProperties(instance_type));
1143  // Create a new map with the size and number of in-object properties
1144  // suggested by |function|.
1145 
1146  // Link initial map and constructor function if the new.target is actually a
1147  // subclass constructor.
1148  if (!IsDerivedConstructor(new_target->shared()->kind())) return false;
1149 
1150  int instance_size;
1151  int in_object_properties;
1152  int embedder_fields =
1153  JSObject::GetEmbedderFieldCount(*constructor_initial_map);
1154  // Constructor expects certain number of in-object properties to be in the
1155  // object. However, CalculateExpectedNofProperties() may return smaller value
1156  // if 1) the constructor is not in the prototype chain of new_target, or
1157  // 2) the prototype chain is modified during iteration, or 3) compilation
1158  // failure occur during prototype chain iteration.
1159  // So we take the maximum of two values.
1160  int expected_nof_properties = std::max(
1161  static_cast<int>(constructor->shared()->expected_nof_properties()),
1162  JSFunction::CalculateExpectedNofProperties(isolate, new_target));
1163  JSFunction::CalculateInstanceSizeHelper(
1164  instance_type, constructor_initial_map->has_prototype_slot(),
1165  embedder_fields, expected_nof_properties, &instance_size,
1166  &in_object_properties);
1167

1500 int JSFunction::CalculateExpectedNofProperties(
1501  Isolate* isolate, DirectHandle<JSFunction> function) {
1502  int expected_nof_properties = 0;
1503  for (PrototypeIterator iter(isolate, function, kStartAtReceiver);
1504  !iter.IsAtEnd(); iter.Advance()) {
1505  DirectHandle<JSReceiver> current =
1506  PrototypeIterator::GetCurrent<JSReceiver>(iter);
1507  if (!IsJSFunction(*current)) break;
1508  DirectHandle<JSFunction> func = Cast<JSFunction>(current);
1509  // The super constructor should be compiled for the number of expected
1510  // properties to be available.
1511  DirectHandle<SharedFunctionInfo> shared(func->shared(), isolate);
1512  IsCompiledScope is_compiled_scope(shared->is_compiled_scope(isolate));
1513  if (is_compiled_scope.is_compiled() ||
1514  Compiler::Compile(isolate, func, Compiler::CLEAR_EXCEPTION,
1515  &is_compiled_scope)) {
1516  DCHECK(shared->is_compiled());
1517  int count = shared->expected_nof_properties();
1518  // Check that the estimate is sensible.
1519  if (expected_nof_properties <= JSObject::kMaxInObjectProperties - count) {
1520  expected_nof_properties += count;
1521  } else {
1522  return JSObject::kMaxInObjectProperties;
1523  }
1524  } else {
1525  // In case there was a compilation error proceed iterating in case there
1526  // will be a builtin function in the prototype chain that requires
1527  // certain number of in-object properties.
1528  continue;
1529  }
1530  }
1531  // Inobject slack tracking will reclaim redundant inobject space
1532  // later, so we can afford to adjust the estimate generously,
1533  // meaning we over-allocate by at least 8 slots in the beginning.
1534  if (expected_nof_properties > 0) {
1535  expected_nof_properties += 8;
1536  if (expected_nof_properties > JSObject::kMaxInObjectProperties) {
1537  expected_nof_properties = JSObject::kMaxInObjectProperties;
1538  }
1539  }
1540  return expected_nof_properties;
1541 }
