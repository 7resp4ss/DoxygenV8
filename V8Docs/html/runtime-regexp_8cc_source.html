52 // Looks up the capture of the given name. Returns the (1-based) numbered
53 // capture index or -1 on failure.
54 // The lookup starts at index |index_in_out|. On success |index_in_out| is set
55 // to the index after the entry was found (i.e. the start index to continue the
56 // search in the presence of duplicate group names).
57 template <typename Matcher, typename = std::enable_if<std::is_invocable_r_v<
58  bool, Matcher, Tagged<String>>>>
59 int LookupNamedCapture(Matcher name_matches,
60  Tagged<FixedArray> capture_name_map, int* index_in_out) {
61  DCHECK_GE(*index_in_out, 0);
62  // TODO(jgruber): Sort capture_name_map and do binary search via
63  // internalized strings.
64 
65  int maybe_capture_index = -1;
66  const int named_capture_count = capture_name_map->length() >> 1;
67  DCHECK_LE(*index_in_out, named_capture_count);
68  for (int j = *index_in_out; j < named_capture_count; j++) {
69  // The format of {capture_name_map} is documented at
70  // JSRegExp::kIrregexpCaptureNameMapIndex.
71  const int name_ix = j * 2;
72  const int index_ix = j * 2 + 1;
73 
74  Tagged<String> capture_name = Cast<String>(capture_name_map->get(name_ix));
75  if (!name_matches(capture_name)) continue;
76 
77  maybe_capture_index = Smi::ToInt(capture_name_map->get(index_ix));
78  *index_in_out = j + 1;
79  break;
80  }
81 
82  return maybe_capture_index;
83 }

116  struct ReplacementPart {
117  static inline ReplacementPart SubjectMatch() {
118  return ReplacementPart(SUBJECT_CAPTURE, 0);
119  }
120  static inline ReplacementPart SubjectCapture(int capture_index) {
121  return ReplacementPart(SUBJECT_CAPTURE, capture_index);
122  }
123  static inline ReplacementPart SubjectPrefix() {
124  return ReplacementPart(SUBJECT_PREFIX, 0);
125  }
126  static inline ReplacementPart SubjectSuffix(int subject_length) {
127  return ReplacementPart(SUBJECT_SUFFIX, subject_length);
128  }
129  static inline ReplacementPart ReplacementString() {
130  return ReplacementPart(REPLACEMENT_STRING, 0);
131  }
132  static inline ReplacementPart EmptyReplacement() {
133  return ReplacementPart(EMPTY_REPLACEMENT, 0);
134  }
135  static inline ReplacementPart ReplacementSubString(int from, int to) {
136  DCHECK_LE(0, from);
137  DCHECK_GT(to, from);
138  return ReplacementPart(-from, to);
139  }
140 
141  // If tag <= 0 then it is the negation of a start index of a substring of
142  // the replacement pattern, otherwise it's a value from PartType.
143  ReplacementPart(int tag, int data) : tag(tag), data(data) {
144  // Must be non-positive or a PartType value.
145  DCHECK(tag < NUMBER_OF_PART_TYPES);
146  }
147  // Either a value of PartType or a non-positive number that is
148  // the negation of an index into the replacement string.
149  int tag;
150  // The data value's interpretation depends on the value of tag:
151  // tag == SUBJECT_PREFIX ||
152  // tag == SUBJECT_SUFFIX: data is unused.
153  // tag == SUBJECT_CAPTURE: data is the number of the capture.
154  // tag == REPLACEMENT_SUBSTRING ||
155  // tag == REPLACEMENT_STRING: data is index into array of substrings
156  // of the replacement string.
157  // tag == EMPTY_REPLACEMENT: data is unused.
158  // tag <= 0: Temporary representation of the substring of the replacement
159  // string ranging over -tag .. data.
160  // Is replaced by REPLACEMENT_{SUB,}STRING when we create the
161  // substring objects.
162  int data;
163  };

165  template <typename Char>
166  bool ParseReplacementPattern(base::Vector<Char> characters,
167  Tagged<FixedArray> capture_name_map,
168  int capture_count, int subject_length) {
169  // Equivalent to String::GetSubstitution, except that this method converts
170  // the replacement string into an internal representation that avoids
171  // repeated parsing when used repeatedly.
172  int length = characters.length();
173  int last = 0;
174  for (int i = 0; i < length; i++) {
175  Char c = characters[i];
176  if (c == '$') {
177  int next_index = i + 1;
178  if (next_index == length) { // No next character!
179  break;
180  }
181  Char c2 = characters[next_index];
182  switch (c2) {
183  case '$':
184  if (i > last) {
185  // There is a substring before. Include the first "$".
186  parts_.emplace_back(
187  ReplacementPart::ReplacementSubString(last, next_index));
188  last = next_index + 1; // Continue after the second "$".
189  } else {
190  // Let the next substring start with the second "$".
191  last = next_index;
192  }
193  i = next_index;
194  break;
195  case '`':
196  if (i > last) {
197  parts_.emplace_back(
198  ReplacementPart::ReplacementSubString(last, i));
199  }
200  parts_.emplace_back(ReplacementPart::SubjectPrefix());
201  i = next_index;
202  last = i + 1;
203  break;
204  case '\'':
205  if (i > last) {
206  parts_.emplace_back(
207  ReplacementPart::ReplacementSubString(last, i));
208  }
209  parts_.emplace_back(ReplacementPart::SubjectSuffix(subject_length));
210  i = next_index;
211  last = i + 1;
212  break;
213  case '&':
214  if (i > last) {
215  parts_.emplace_back(
216  ReplacementPart::ReplacementSubString(last, i));
217  }
218  parts_.emplace_back(ReplacementPart::SubjectMatch());
219  i = next_index;
220  last = i + 1;
221  break;
222  case '0':
223  case '1':
224  case '2':
225  case '3':
226  case '4':
227  case '5':
228  case '6':
229  case '7':
230  case '8':
231  case '9': {
232  int capture_ref = c2 - '0';
233  if (capture_ref > capture_count) {
234  i = next_index;
235  continue;
236  }
237  int second_digit_index = next_index + 1;
238  if (second_digit_index < length) {
239  // Peek ahead to see if we have two digits.
240  Char c3 = characters[second_digit_index];
241  if ('0' <= c3 && c3 <= '9') { // Double digits.
242  int double_digit_ref = capture_ref * 10 + c3 - '0';
243  if (double_digit_ref <= capture_count) {
244  next_index = second_digit_index;
245  capture_ref = double_digit_ref;
246  }
247  }
248  }
249  if (capture_ref > 0) {
250  if (i > last) {
251  parts_.emplace_back(
252  ReplacementPart::ReplacementSubString(last, i));
253  }
254  DCHECK(capture_ref <= capture_count);
255  parts_.emplace_back(ReplacementPart::SubjectCapture(capture_ref));
256  last = next_index + 1;
257  }
258  i = next_index;
259  break;
260  }
261  case '<': {
262  if (capture_name_map.is_null()) {
263  i = next_index;
264  break;
265  }
266 
267  // Scan until the next '>', and let the enclosed substring be the
268  // groupName.
269 
270  const int name_start_index = next_index + 1;
271  int closing_bracket_index = -1;
272  for (int j = name_start_index; j < length; j++) {
273  if (characters[j] == '>') {
274  closing_bracket_index = j;
275  break;
276  }
277  }
278 
279  // If no closing bracket is found, '$<' is treated as a string
280  // literal.
281  if (closing_bracket_index == -1) {
282  i = next_index;
283  break;
284  }
285 
286  if (i > last) {
287  parts_.emplace_back(
288  ReplacementPart::ReplacementSubString(last, i));
289  }
290 
291  base::Vector<Char> requested_name =
292  characters.SubVector(name_start_index, closing_bracket_index);
293 
294  // If capture is undefined or does not exist, replace the text
295  // through the following '>' with the empty string.
296  // Otherwise, replace the text through the following '>' with
297  // ? ToString(capture).
298  // For duplicated capture group names we don't know which of them
299  // matches at this point in time, so we create a separate
300  // replacement for each possible match. When applying the
301  // replacement unmatched groups will be skipped.
302 
303  int capture_index = 0;
304  int capture_name_map_index = 0;
305  while (capture_index != -1) {
306  capture_index = LookupNamedCapture(
307  [=](Tagged<String> capture_name) {
308  return capture_name->IsEqualTo(requested_name);
309  },
310  capture_name_map, &capture_name_map_index);
311  DCHECK(capture_index == -1 ||
312  (1 <= capture_index && capture_index <= capture_count));
313 
314  parts_.emplace_back(
315  capture_index == -1
316  ? ReplacementPart::EmptyReplacement()
317  : ReplacementPart::SubjectCapture(capture_index));
318  }
319 
320  last = closing_bracket_index + 1;
321  i = closing_bracket_index;
322  break;
323  }
324  default:
325  i = next_index;
326  break;
327  }
328  }
329  }
330  if (length > last) {
331  if (last == 0) {
332  // Replacement is simple. Do not use Apply to do the replacement.
333  return true;
334  } else {
335  parts_.emplace_back(
336  ReplacementPart::ReplacementSubString(last, length));
337  }
338  }
339  return false;
340  }
341 
342  base::SmallVector<ReplacementPart, kStaticVectorSlots> parts_;
343  DirectHandleSmallVector<String, kStaticVectorSlots> replacement_substrings_;
344 };

346 bool CompiledReplacement::Compile(Isolate* isolate,
347  DirectHandle<JSRegExp> regexp,
348  DirectHandle<RegExpData> regexp_data,
349  DirectHandle<String> replacement,
350  int capture_count, int subject_length) {
351  {
352  DisallowGarbageCollection no_gc;
353  String::FlatContent content = replacement->GetFlatContent(no_gc);
354  DCHECK(content.IsFlat());
355 
356  Tagged<FixedArray> capture_name_map;
357  if (capture_count > 0) {
358  // capture_count > 0 implies IrRegExpData. Since capture_count is in
359  // trusted space, this is not a SBXCHECK.
360  DCHECK(Is<IrRegExpData>(*regexp_data));
361  Tagged<IrRegExpData> re_data = Cast<IrRegExpData>(*regexp_data);
362 
363  Tagged<Object> maybe_capture_name_map = re_data->capture_name_map();
364  if (IsFixedArray(maybe_capture_name_map)) {
365  capture_name_map = Cast<FixedArray>(maybe_capture_name_map);
366  }
367  }
368 
369  bool simple;
370  if (content.IsOneByte()) {
371  simple =
372  ParseReplacementPattern(content.ToOneByteVector(), capture_name_map,
373  capture_count, subject_length);
374  } else {
375  DCHECK(content.IsTwoByte());
376  simple = ParseReplacementPattern(content.ToUC16Vector(), capture_name_map,
377  capture_count, subject_length);
378  }
379  if (simple) return true;
380  }
381 
382  // Find substrings of replacement string and create them as String objects.
383  replacement_substrings_.reserve(parts());
384  int substring_index = 0;
385  for (ReplacementPart& part : parts_) {
386  int tag = part.tag;
387  if (tag <= 0) { // A replacement string slice.
388  int from = -tag;
389  int to = part.data;
390  replacement_substrings_.emplace_back(
391  isolate->factory()->NewSubString(replacement, from, to));
392  part.tag = REPLACEMENT_SUBSTRING;
393  part.data = substring_index;
394  substring_index++;
395  } else if (tag == REPLACEMENT_STRING) {
396  replacement_substrings_.emplace_back(replacement);
397  part.data = substring_index;
398  substring_index++;
399  }
400  }
401  return false;
402 }

404 void CompiledReplacement::Apply(ReplacementStringBuilder* builder,
405  int match_from, int match_to, int32_t* match) {
406  DCHECK_LT(0, parts_.size());
407  for (ReplacementPart& part : parts_) {
408  switch (part.tag) {
409  case SUBJECT_PREFIX:
410  if (match_from > 0) builder->AddSubjectSlice(0, match_from);
411  break;
412  case SUBJECT_SUFFIX: {
413  int subject_length = part.data;
414  if (match_to < subject_length) {
415  builder->AddSubjectSlice(match_to, subject_length);
416  }
417  break;
418  }
419  case SUBJECT_CAPTURE: {
420  int capture = part.data;
421  int from = match[capture * 2];
422  int to = match[capture * 2 + 1];
423  if (from >= 0 && to > from) {
424  builder->AddSubjectSlice(from, to);
425  }
426  break;
427  }
428  case REPLACEMENT_SUBSTRING:
429  case REPLACEMENT_STRING:
430  builder->AddString(replacement_substrings_[part.data]);
431  break;
432  case EMPTY_REPLACEMENT:
433  break;
434  default:
435  UNREACHABLE();
436  }
437  }

440 void FindOneByteStringIndices(base::Vector<const uint8_t> subject,
441  uint8_t pattern, std::vector<int>* indices,
442  unsigned int limit) {
443  DCHECK_LT(0, limit);
444  // Collect indices of pattern in subject using memchr.
445  // Stop after finding at most limit values.
446  const uint8_t* subject_start = subject.begin();
447  const uint8_t* subject_end = subject_start + subject.length();
448  const uint8_t* pos = subject_start;
449  while (limit > 0) {
450  pos = reinterpret_cast<const uint8_t*>(
451  memchr(pos, pattern, subject_end - pos));
452  if (pos == nullptr) return;
453  indices->push_back(static_cast<int>(pos - subject_start));
454  pos++;
455  limit--;
456  }
457 }

474 template <typename SubjectChar, typename PatternChar>
475 void FindStringIndices(Isolate* isolate,
476  base::Vector<const SubjectChar> subject,
477  base::Vector<const PatternChar> pattern,
478  std::vector<int>* indices, unsigned int limit) {
479  DCHECK_LT(0, limit);
480  // Collect indices of pattern in subject.
481  // Stop after finding at most limit values.
482  int pattern_length = pattern.length();
483  int index = 0;
484  StringSearch<PatternChar, SubjectChar> search(isolate, pattern);
485  while (limit > 0) {

487  if (index < 0) return;
488  indices->push_back(index);
489  index += pattern_length;
490  limit--;
491  }

571 template <typename ResultSeqString>
572 V8_WARN_UNUSED_RESULT static Tagged<Object>
573 StringReplaceGlobalAtomRegExpWithString(
574  Isolate* isolate, DirectHandle<String> subject,
575  DirectHandle<JSRegExp> pattern_regexp, DirectHandle<String> replacement,
576  DirectHandle<RegExpMatchInfo> last_match_info,
577  DirectHandle<AtomRegExpData> regexp_data) {
578  DCHECK(subject->IsFlat());
579  DCHECK(replacement->IsFlat());
580 
581  std::vector<int>* indices = GetRewoundRegexpIndicesList(isolate);
582 
583  Tagged<String> pattern = regexp_data->pattern();
584  int subject_len = subject->length();
585  int pattern_len = pattern->length();
586  int replacement_len = replacement->length();
587 
588  FindStringIndicesDispatch(isolate, *subject, pattern, indices, 0xFFFFFFFF);
589 
590  if (indices->empty()) return *subject;
591 
592  // Detect integer overflow.
593  int64_t result_len_64 = (static_cast<int64_t>(replacement_len) -
594  static_cast<int64_t>(pattern_len)) *
595  static_cast<int64_t>(indices->size()) +
596  static_cast<int64_t>(subject_len);
597  int result_len;
598  if (result_len_64 > static_cast<int64_t>(String::kMaxLength)) {
599  static_assert(String::kMaxLength < kMaxInt);
600  result_len = kMaxInt; // Provoke exception.
601  } else {
602  result_len = static_cast<int>(result_len_64);
603  }
604  if (result_len == 0) {
605  return ReadOnlyRoots(isolate).empty_string();
606  }
607 
608  int subject_pos = 0;
609  int result_pos = 0;
610 
611  MaybeDirectHandle<SeqString> maybe_res;
612  if (ResultSeqString::kHasOneByteEncoding) {
613  maybe_res = isolate->factory()->NewRawOneByteString(result_len);
614  } else {
615  maybe_res = isolate->factory()->NewRawTwoByteString(result_len);
616  }
617  DirectHandle<SeqString> untyped_res;
618  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, untyped_res, maybe_res);
619  DirectHandle<ResultSeqString> result = Cast<ResultSeqString>(untyped_res);
620 
621  DisallowGarbageCollection no_gc;
622  for (int index : *indices) {
623  // Copy non-matched subject content.
624  if (subject_pos < index) {
625  String::WriteToFlat(*subject, result->GetChars(no_gc) + result_pos,
626  subject_pos, index - subject_pos);
627  result_pos += index - subject_pos;
628  }
629 
630  // Replace match.
631  if (replacement_len > 0) {
632  String::WriteToFlat(*replacement, result->GetChars(no_gc) + result_pos, 0,
633  replacement_len);
634  result_pos += replacement_len;
635  }
636 
637  subject_pos = index + pattern_len;
638  }
639  // Add remaining subject content at the end.
640  if (subject_pos < subject_len) {
641  String::WriteToFlat(*subject, result->GetChars(no_gc) + result_pos,
642  subject_pos, subject_len - subject_pos);
643  }
644 
645  int32_t match_indices[] = {indices->back(), indices->back() + pattern_len};
646  RegExp::SetLastMatchInfo(isolate, last_match_info, subject, 0, match_indices);
647 
648  TruncateRegexpIndicesList(isolate);
649 
650  return *result;
651 }

657  DirectHandle<RegExpMatchInfo> last_match_info) {
658  DCHECK(subject->IsFlat());
659  DCHECK(replacement->IsFlat());
660 
661  int capture_count = regexp_data->capture_count();
662  int subject_length = subject->length();
663 
664  // Ensure the RegExp is compiled so we can access the capture-name map.
665  if (!RegExp::EnsureFullyCompiled(isolate, regexp_data, subject)) {
666  return ReadOnlyRoots(isolate).exception();
667  }
668 
669  CompiledReplacement compiled_replacement(isolate);
670  const bool simple_replace = compiled_replacement.Compile(
671  isolate, regexp, regexp_data, replacement, capture_count, subject_length);
672 
673  // Shortcut for simple non-regexp global replacements.
674  if (regexp_data->type_tag() == RegExpData::Type::ATOM && simple_replace) {
675  if (subject->IsOneByteRepresentation() &&
676  replacement->IsOneByteRepresentation()) {
677  return StringReplaceGlobalAtomRegExpWithString<SeqOneByteString>(
678  isolate, subject, regexp, replacement, last_match_info,
679  Cast<AtomRegExpData>(regexp_data));
680  } else {
681  return StringReplaceGlobalAtomRegExpWithString<SeqTwoByteString>(
682  isolate, subject, regexp, replacement, last_match_info,
683  Cast<AtomRegExpData>(regexp_data));
684  }
685  }
686 
687  RegExpGlobalExecRunner runner(direct_handle(*regexp_data, isolate), subject,
688  isolate);
689  if (runner.HasException()) return ReadOnlyRoots(isolate).exception();
690 
691  int32_t* current_match = runner.FetchNext();
692  if (current_match == nullptr) {
693  if (runner.HasException()) return ReadOnlyRoots(isolate).exception();
694  return *subject;
695  }
696 
697  // Guessing the number of parts that the final result string is built
698  // from. Global regexps can match any number of times, so we guess
699  // conservatively.
700  int expected_parts = (compiled_replacement.parts() + 1) * 4 + 1;
701  // TODO(v8:12843): improve the situation where the expected_parts exceeds
702  // the maximum size of the backing store.
703  ReplacementStringBuilder builder(isolate->heap(), subject, expected_parts);
704 
705  int prev = 0;
706 
707  do {
708  int start = current_match[0];
709  int end = current_match[1];
710 
711  if (prev < start) {
712  builder.AddSubjectSlice(prev, start);
713  }
714 
715  if (simple_replace) {
716  builder.AddString(replacement);
717  } else {
718  compiled_replacement.Apply(&builder, start, end, current_match);
719  }
720  prev = end;
721 
722  current_match = runner.FetchNext();
723  } while (current_match != nullptr);
724 
725  if (runner.HasException()) return ReadOnlyRoots(isolate).exception();
726 
727  if (prev < subject_length) {
728  builder.AddSubjectSlice(prev, subject_length);
729  }
730 
731  RegExp::SetLastMatchInfo(isolate, last_match_info, subject, capture_count,
732  runner.LastSuccessfulMatch());
733 
734  RETURN_RESULT_OR_FAILURE(isolate, builder.ToString());
735 }

740  Isolate* isolate, DirectHandle<String> subject,
741  DirectHandle<JSRegExp> regexp, DirectHandle<RegExpData> regexp_data,
742  DirectHandle<RegExpMatchInfo> last_match_info) {
743  DCHECK(subject->IsFlat());
744 
745  // Shortcut for simple non-regexp global replacements.
746  if (regexp_data->type_tag() == RegExpData::Type::ATOM) {
747  DirectHandle<String> empty_string = isolate->factory()->empty_string();
748  if (subject->IsOneByteRepresentation()) {
749  return StringReplaceGlobalAtomRegExpWithString<SeqOneByteString>(
750  isolate, subject, regexp, empty_string, last_match_info,
751  Cast<AtomRegExpData>(regexp_data));
752  } else {
753  return StringReplaceGlobalAtomRegExpWithString<SeqTwoByteString>(
754  isolate, subject, regexp, empty_string, last_match_info,
755  Cast<AtomRegExpData>(regexp_data));
756  }
757  }
758 
759  RegExpGlobalExecRunner runner(direct_handle(*regexp_data, isolate), subject,
760  isolate);
761  if (runner.HasException()) return ReadOnlyRoots(isolate).exception();
762 
763  int32_t* current_match = runner.FetchNext();
764  if (current_match == nullptr) {
765  if (runner.HasException()) return ReadOnlyRoots(isolate).exception();
766  return *subject;
767  }
768 
769  int start = current_match[0];
770  int end = current_match[1];
771  int capture_count = regexp_data->capture_count();
772  int subject_length = subject->length();
773 
774  int new_length = subject_length - (end - start);
775  if (new_length == 0) return ReadOnlyRoots(isolate).empty_string();
776 
777  DirectHandle<ResultSeqString> answer;
778  if (ResultSeqString::kHasOneByteEncoding) {
779  answer = Cast<ResultSeqString>(
780  isolate->factory()->NewRawOneByteString(new_length).ToHandleChecked());
781  } else {
782  answer = Cast<ResultSeqString>(
783  isolate->factory()->NewRawTwoByteString(new_length).ToHandleChecked());
784  }
785 
786  int prev = 0;
787  int position = 0;
788 
789  DisallowGarbageCollection no_gc;
790  do {
791  start = current_match[0];
792  end = current_match[1];
793  if (prev < start) {
794  // Add substring subject[prev;start] to answer string.
795  String::WriteToFlat(*subject, answer->GetChars(no_gc) + position, prev,
796  start - prev);
797  position += start - prev;
798  }
799  prev = end;
800 
801  current_match = runner.FetchNext();
802  } while (current_match != nullptr);
803 
804  if (runner.HasException()) return ReadOnlyRoots(isolate).exception();
805 
806  RegExp::SetLastMatchInfo(isolate, last_match_info, subject, capture_count,
807  runner.LastSuccessfulMatch());
808 
809  if (prev < subject_length) {
810  // Add substring subject[prev;length] to answer string.
811  String::WriteToFlat(*subject, answer->GetChars(no_gc) + position, prev,
812  subject_length - prev);
813  position += subject_length - prev;
814  }
815 
816  if (position == 0) return ReadOnlyRoots(isolate).empty_string();
817 
818  // Shorten string and fill
819  int string_size = ResultSeqString::SizeFor(position);
820  int allocated_string_size = ResultSeqString::SizeFor(new_length);
821  int delta = allocated_string_size - string_size;
822 
823  answer->set_length(position);
824  if (delta == 0) return *answer;
825 
826  Address end_of_string = answer->address() + string_size;
827  Heap* heap = isolate->heap();
828 
829  // The trimming is performed on a newly allocated object, which is on a
830  // freshly allocated page or on an already swept page. Hence, the sweeper
831  // thread can not get confused with the filler creation. No synchronization
832  // needed.
833  // TODO(hpayer): We should shrink the large object page if the size
834  // of the object changed significantly.
835  if (!heap->IsLargeObject(*answer)) {
836  heap->CreateFillerObjectAt(end_of_string, delta);
837  }
838  return *answer;
839 }

1257  Isolate* isolate, DirectHandle<String> subject,
1258  DirectHandle<JSRegExp> regexp, DirectHandle<RegExpData> regexp_data,
1259  DirectHandle<RegExpMatchInfo> last_match_array) {
1260  DCHECK(RegExpUtils::IsUnmodifiedRegExp(isolate, regexp));
1261  DCHECK_NE(has_capture, regexp_data->capture_count() == 0);
1262  DCHECK_IMPLIES(has_capture, Is<IrRegExpData>(*regexp_data));
1263  DCHECK(subject->IsFlat());

1265  // Force tier up to native code for global replaces. The global replace is
1266  // implemented differently for native code and bytecode execution, where the
1267  // native code expects an array to store all the matches, and the bytecode
1268  // matches one at a time, so it's easier to tier-up to native code from the
1269  // start.
1270  if (v8_flags.regexp_tier_up &&
1271  regexp_data->type_tag() == RegExpData::Type::IRREGEXP) {
1272  Cast<IrRegExpData>(regexp_data)->MarkTierUpForNextExec();
1273  if (v8_flags.trace_regexp_tier_up) {
1274  PrintF("Forcing tier-up of JSRegExp object %p in SearchRegExpMultiple\n",
1275  reinterpret_cast<void*>(regexp->ptr()));
1276  }
1277  }
1278 
1279  int capture_count = regexp_data->capture_count();
1280  int subject_length = subject->length();
1281 
1282  static const int kMinLengthToCache = 0x1000;
1283 
1284  if (subject_length > kMinLengthToCache) {
1285  Tagged<FixedArray> last_match_cache;
1286  Tagged<Object> cached_answer = RegExpResultsCache::Lookup(
1287  isolate->heap(), *subject, regexp_data->wrapper(), &last_match_cache,
1288  RegExpResultsCache::REGEXP_MULTIPLE_INDICES);
1289  if (IsFixedArray(cached_answer)) {
1290  int capture_registers = JSRegExp::RegistersForCaptureCount(capture_count);
1291  std::unique_ptr<int32_t[]> last_match(new int32_t[capture_registers]);
1292  int32_t* raw_last_match = last_match.get();
1293  for (int i = 0; i < capture_registers; i++) {
1294  raw_last_match[i] = Smi::ToInt(last_match_cache->get(i));
1295  }
1296  DirectHandle<FixedArray> cached_fixed_array(
1297  Cast<FixedArray>(cached_answer), isolate);
1298  // The cache FixedArray is a COW-array and we need to return a copy.
1299  DirectHandle<FixedArray> copied_fixed_array =
1300  isolate->factory()->CopyFixedArrayWithMap(
1301  cached_fixed_array, isolate->factory()->fixed_array_map());
1302  RegExp::SetLastMatchInfo(isolate, last_match_array, subject,
1303  capture_count, raw_last_match);
1304  return *copied_fixed_array;
1305  }
1306  }

1309  isolate);
1310  if (runner.HasException()) return ReadOnlyRoots(isolate).exception();
1311

1313

1315  int match_start = -1;
1316  int match_end = 0;
1317  bool first = true;
1318 
1319  // Two smis before and after the match, for very long strings.
1320  static const int kMaxBuilderEntriesPerRegExpMatch = 5;
1321 
1322  while (true) {
1323  int32_t* current_match = runner.FetchNext();

1325  match_start = current_match[0];
1326  builder.EnsureCapacity(isolate, kMaxBuilderEntriesPerRegExpMatch);
1327  if (match_end < match_start) {
1328  ReplacementStringBuilder::AddSubjectSlice(&builder, match_end,
1329  match_start);
1330  }
1331  match_end = current_match[1];
1332  {
1333  // Avoid accumulating new handles inside loop.
1334  HandleScope temp_scope(isolate);
1335  DirectHandle<String> match;
1336  if (!first) {
1337  match = isolate->factory()->NewProperSubString(subject, match_start,
1338  match_end);
1339  } else {
1340  match =
1341  isolate->factory()->NewSubString(subject, match_start, match_end);
1342  first = false;
1343  }
1344 
1345  if (has_capture) {
1346  // Arguments array to replace function is match, captures, index and
1347  // subject, i.e., 3 + capture count in total. If the RegExp contains
1348  // named captures, they are also passed as the last argument.
1349 
1350  // has_capture can only be true for IrRegExp.
1351  Tagged<IrRegExpData> re_data = Cast<IrRegExpData>(*regexp_data);
1352  DirectHandle<Object> maybe_capture_map(re_data->capture_name_map(),
1353  isolate);
1354  const bool has_named_captures = IsFixedArray(*maybe_capture_map);
1355

1357  has_named_captures ? 4 + capture_count : 3 + capture_count;

1360  isolate->factory()->NewFixedArray(argc);
1361  int cursor = 0;
1362 
1363  elements->set(cursor++, *match);
1364  for (int i = 1; i <= capture_count; i++) {
1365  int start = current_match[i * 2];
1366  if (start >= 0) {
1367  int end = current_match[i * 2 + 1];
1368  DCHECK(start <= end);
1369  DirectHandle<String> substring =
1370  isolate->factory()->NewSubString(subject, start, end);
1371  elements->set(cursor++, *substring);
1372  } else {
1373  DCHECK_GT(0, current_match[i * 2 + 1]);
1374  elements->set(cursor++, ReadOnlyRoots(isolate).undefined_value());
1375  }
1376  }
1377 
1378  elements->set(cursor++, Smi::FromInt(match_start));
1379  elements->set(cursor++, *subject);
1380 
1381  if (has_named_captures) {
1382  DirectHandle<FixedArray> capture_map =
1383  Cast<FixedArray>(maybe_capture_map);
1384  DirectHandle<JSObject> groups = ConstructNamedCaptureGroupsObject(
1385  isolate, capture_map, [=](int ix) { return elements->get(ix); });
1386  elements->set(cursor++, *groups);
1387  }
1388 
1389  DCHECK_EQ(cursor, argc);
1390  builder.Add(*isolate->factory()->NewJSArrayWithElements(elements));
1391  } else {
1392  builder.Add(*match);
1393  }
1394  }
1395  }
1396 
1397  if (runner.HasException()) return ReadOnlyRoots(isolate).exception();
1398 
1399  if (match_start >= 0) {
1400  // Finished matching, with at least one match.
1401  if (match_end < subject_length) {
1402  ReplacementStringBuilder::AddSubjectSlice(&builder, match_end,
1403  subject_length);
1404  }
1405 
1406  RegExp::SetLastMatchInfo(isolate, last_match_array, subject, capture_count,
1407  runner.LastSuccessfulMatch());
1408 
1409  if (subject_length > kMinLengthToCache) {
1410  // Store the last successful match into the array for caching.
1411  int capture_registers = JSRegExp::RegistersForCaptureCount(capture_count);
1412  DirectHandle<FixedArray> last_match_cache =
1413  isolate->factory()->NewFixedArray(capture_registers);
1414  int32_t* last_match = runner.LastSuccessfulMatch();
1415  for (int i = 0; i < capture_registers; i++) {
1416  last_match_cache->set(i, Smi::FromInt(last_match[i]));
1417  }
1418  DirectHandle<FixedArray> result_fixed_array =
1419  FixedArray::RightTrimOrEmpty(isolate, builder.array(),
1420  builder.length());
1421  // Cache the result and copy the FixedArray into a COW array.
1422  DirectHandle<FixedArray> copied_fixed_array =
1423  isolate->factory()->CopyFixedArrayWithMap(
1424  result_fixed_array, isolate->factory()->fixed_array_map());
1425  RegExpResultsCache::Enter(
1426  isolate, subject,
1427  direct_handle(regexp->data(isolate)->wrapper(), isolate),
1428  copied_fixed_array, last_match_cache,
1429  RegExpResultsCache::REGEXP_MULTIPLE_INDICES);
1430  }
1431  return *builder.array();
1432  } else {
1433  return ReadOnlyRoots(isolate).null_value(); // No matches at all.
1434  }

2164 template <typename SChar, typename PChar>
2165 inline void RegExpMatchGlobalAtom_OneCharPattern(
2166  Isolate* isolate, base::Vector<const SChar> subject, const PChar pattern,
2167  int start_index, int* number_of_matches, int* last_match_index,
2168  const DisallowGarbageCollection& no_gc) {
2169  static_assert(std::is_unsigned_v<SChar>);
2170  static_assert(std::is_unsigned_v<PChar>);
2171  // We can utilize SIMD to check multiple characters at once.
2172  // Since the pattern is a single char, we create a mask setting each lane in
2173  // the vector to the pattern char.
2174  // Since reductions from a vector to a general purpose register (i.e.
2175  // ReduceSum in this algorithm) are expensive, we keep a count for each lane
2176  // in a vector until the count could potentially overflow and only reduce to
2177  // a general purpose register then. I.e. if SChar is uint8_t, we have a
2178  // 16xuint8_t vector to count matches, which we reduce to an int every 255
2179  // blocks.
2180  namespace hw = hwy::HWY_NAMESPACE;
2181  hw::ScalableTag<SChar> tag;
2182  // We need a wider tag to avoid overflows on lanes when summing up submatches.
2183  using WidenedTag = hw::RepartitionToWide<decltype(tag)>;
2184  WidenedTag sum_tag;
2185  static constexpr size_t stride = hw::Lanes(tag);
2186  // Subtle: the valid variants are {SChar,PChar} in:
2187  // {uint8_t,uint8_t}, {uc16,uc16}, {uc16,uint8_t}. In the latter case,
2188  // we cast the uint8_t pattern to uc16 for the comparison.
2189  const auto mask = hw::Set(tag, static_cast<const SChar>(pattern));
2190 
2191  int matches = 0;
2192  auto submatches = hw::Zero(tag);
2193  const SChar* last_match_block = nullptr;
2194  hw::Mask<decltype(tag)> last_match_vec;
2195 
2196  const SChar* block = subject.data() + start_index;
2197  const SChar* end = subject.data() + subject.length();
2198 
2199  // ReduceSum is expensive, so we gather matches into a vector. max_count is
2200  // the maximum number of matches we can count in the vector before it
2201  // overflows.
2202  int max_count = std::numeric_limits<SChar>::max();
2203  while (block + stride * max_count <= end) {
2204  for (int i = 0; i < max_count; i++, block += stride) {
2205  const auto input = hw::LoadU(tag, block);
2206  const auto match = input == mask;
2207  // Lanes with matches have all bits set, so we subtract to increase the
2208  // count by 1.
2209  submatches = hw::Sub(submatches, hw::VecFromMask(tag, match));
2210  if (!hw::AllFalse(tag, match)) {
2211  last_match_block = block;
2212  last_match_vec = match;
2213  }
2214  }
2215  // SumsOf2 promotes the sum of 2 consecutive lanes into a wider lane.
2216  auto promoted_submatches = hw::SumsOf2(submatches);
2217  // Wider lane sums can be reduces without overflows.
2218  matches += hw::ReduceSum(sum_tag, promoted_submatches);
2219  submatches = hw::Zero(tag);
2220  }
2221 
2222  // For blocks shorter than stride * max_count, lanes in submatches can't
2223  // overflow.
2224  DCHECK_LT(end - block, stride * max_count);
2225  for (; block + stride <= end; block += stride) {
2226  const auto input = hw::LoadU(tag, block);
2227  const auto match = input == mask;
2228  submatches = hw::Sub(submatches, hw::VecFromMask(tag, match));
2229  if (!hw::AllFalse(tag, match)) {
2230  last_match_block = block;
2231  last_match_vec = match;
2232  }
2233  }
2234  auto promoted_submatches = hw::SumsOf2(submatches);
2235  matches += hw::ReduceSum(sum_tag, promoted_submatches);
2236 
2237  // Handle remaining chars.
2238  // last_match_block already contains the last match position, so use a special
2239  // vector with lane 0 set to extract the last_match_index later.
2240  const auto scalar_last_match_vec = hw::FirstN(tag, 1);
2241  for (SChar c = *block; block < end; c = *(++block)) {
2242  if (c != static_cast<const SChar>(pattern)) continue;
2243  matches++;
2244  last_match_block = block;
2245  last_match_vec = scalar_last_match_vec;
2246  }
2247 
2248  // Store results.
2249  *number_of_matches += matches;
2250  if (last_match_block != nullptr) {
2251  DCHECK(!hw::AllFalse(tag, last_match_vec));
2252  *last_match_index = static_cast<int>(
2253  last_match_block + hw::FindKnownLastTrue(tag, last_match_vec) -
2254  subject.data());
2255  }
2256 }
