104 static void Convert8Digits(uint32_t n, char* buffer) {
105  static constexpr uint32_t kUint32Mask = kMaxUInt32;
106  // 281474978 = ceil(2^48 / 1'000'000) + 1
107  uint64_t prod = n * 281474978LL;
108  prod >>= 16;
109  prod += 1;
110  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer);
111  prod = (prod & kUint32Mask) * 100;
112  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer + 2);
113  prod = (prod & kUint32Mask) * 100;
114  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer + 4);
115  prod = (prod & kUint32Mask) * 100;
116  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer + 6);
117 }

119 // Returns count of digits written.
120 uint8_t ConvertUpTo9Digits(uint32_t n, char* buffer) {
121  static constexpr uint32_t kUint32Mask = kMaxUInt32;
122 
123  if (n >= 100'000'000) {
124  // 9 digits.
125  // 1441151882 = ceil(2^57 / 100'000'000) + 1
126  uint64_t prod = n * 1441151882LL;
127  prod >>= 25;
128 
129  const uint8_t head_digit = static_cast<uint8_t>(prod >> 32);
130  DCHECK_LT(head_digit, 10);
131  *buffer = '0' + head_digit;
132 
133  // Print remaining 8 digits.
134  prod = (prod & kUint32Mask) * 100;
135  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer + 1);
136  prod = (prod & kUint32Mask) * 100;
137  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer + 3);
138  prod = (prod & kUint32Mask) * 100;
139  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer + 5);
140  prod = (prod & kUint32Mask) * 100;
141  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer + 7);
142 
143  return 9;
144  }
145  if (n >= 1'000'000) {
146  // 7 or 8 digits.
147  // 281474978 = ceil(2^48 / 1'000'000) + 1
148  uint64_t prod = n * 281474978LL;
149  prod >>= 16;
150 
151  const uint8_t head_digits = static_cast<uint8_t>(prod >> 32);
152  const uint8_t head_digit_count = ConvertHeadDigits(head_digits, buffer);
153  buffer += head_digit_count;
154 
155  // Print remaining 6 digits.
156  prod = (prod & kUint32Mask) * 100;
157  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer);
158  prod = (prod & kUint32Mask) * 100;
159  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer + 2);
160  prod = (prod & kUint32Mask) * 100;
161  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer + 4);
162 
163  return 6 + head_digit_count;
164  }
165  if (n >= 10'000) {
166  // 5 or 6 digits.
167  // 429497 = ceil(2^32 / 10'000)
168  uint64_t prod = n * 429497LL;
169 
170  const uint8_t head_digits = static_cast<uint8_t>(prod >> 32);
171  const uint8_t head_digit_count = ConvertHeadDigits(head_digits, buffer);
172  buffer += head_digit_count;
173 
174  // Print remaining 4 digits.
175  prod = (prod & kUint32Mask) * 100;
176  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer);
177  prod = (prod & kUint32Mask) * 100;
178  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer + 2);
179 
180  return 4 + head_digit_count;
181  }
182  if (n >= 100) {
183  // 3 or 4 digits.
184  // 42949673 = ceil(2^32 / 10'000)
185  uint64_t prod = n * 42949673LL;
186 
187  const uint8_t head_digits = static_cast<uint8_t>(prod >> 32);
188  const uint8_t head_digit_count = ConvertHeadDigits(head_digits, buffer);
189  buffer += head_digit_count;
190 
191  // Print remaining 2 digits.
192  prod = (prod & kUint32Mask) * 100;
193  Convert2Digits(static_cast<uint8_t>(prod >> 32), buffer);
194 
195  return 2 + head_digit_count;
196  }
197  // 1 or 2 digits.
198  return ConvertHeadDigits(n, buffer);
199 }

201 // Returns count of digits written.
202 uint8_t SignificandToChars(uint64_t n, char* buffer) {
203  DCHECK_LT(n, 99999999999999999); // Only supports up to 17 digits
204 
205  if (n >= 100'000'000) {
206  // If we have at least 9 digits, split into 2 blocks. The second one always
207  // has exactly 8 digits.
208  uint32_t first_block = static_cast<uint32_t>(n / 100'000'000);
209  uint32_t second_block = static_cast<uint32_t>(n % 100'000'000);
210 
211  uint8_t first_block_digits = ConvertUpTo9Digits(first_block, buffer);
212  Convert8Digits(second_block, buffer + first_block_digits);
213  return first_block_digits + 8;
214  } else {
215  return ConvertUpTo9Digits(static_cast<uint32_t>(n), buffer);
216  }
217 }

219 // Helper class for building result strings in a character buffer. The
220 // purpose of the class is to use safe operations that checks the
221 // buffer bounds on all operations in debug mode.
222 // This simple base class does not allow formatted output.
223 class SimpleStringBuilder final {
224  public:
225  // Create a string builder with a buffer of the given size. The
226  // buffer is allocated through NewArray<char> and must be
227  // deallocated by the caller of Finalize().
228  explicit SimpleStringBuilder(size_t size) {
229  buffer_ = base::Vector<char>::New(size);
230  cursor_ = buffer_.begin();
231  }
232 
233  SimpleStringBuilder(char* buffer, size_t size)
234  : buffer_(buffer, size), cursor_(buffer) {}
235 
236  ~SimpleStringBuilder() {
237  if (V8_UNLIKELY(!is_finalized())) Finalize();
238  }
239 
240  // Get the current position in the builder.
241  size_t position() const {
242  DCHECK(!is_finalized());
243  return cursor_ - buffer_.begin();
244  }
245 
246  // Add a single character to the builder. It is not allowed to add
247  // 0-characters; use the Finalize() method to terminate the string
248  // instead.
249  V8_INLINE void AddCharacter(char c) {
250  DCHECK_NE(c, '\0');
251  DCHECK(!is_finalized());
252  DCHECK_LT(position(), buffer_.size());
253  *cursor_++ = c;
254  }
255 
256  // Add an entire string to the builder. 'len' must be equal to strlen().
257  V8_INLINE void AddString(const char* s, size_t len) {
258  DCHECK_EQ(len, strlen(s));
259  AddSubstring(s, len);
260  }
261 
262  // Add a string literal to the builder.
263  template <size_t N>
264  V8_INLINE void AddStringLiteral(const char (&s)[N]) {
265  AddSubstring(s, N - 1);
266  }
267 
268  // Add the first 'n' characters of the given 0-terminated string 's' to the
269  // builder. The input string must have enough characters.
270  V8_INLINE void AddSubstring(const char* s, size_t n) {
271  DCHECK(!is_finalized());
272  DCHECK_LE(position() + n, buffer_.size());
273  DCHECK_LE(n, strlen(s));
274  MemCopy(cursor_, s, n * kCharSize);
275  cursor_ += n;
276  }
277 
278  // Add character padding to the builder. If count is non-positive,
279  // nothing is added to the builder.
280  V8_INLINE void AddPadding(char c, int count) {
281  DCHECK(!is_finalized());
282  DCHECK_LE(position() + std::max(0, count), buffer_.size());
283  cursor_ = std::fill_n(cursor_, count, c);
284  }
285 
286  // Add the decimal representation of the value. The value is expected to be
287  // a positive integer with at most 3 digits.
288  V8_INLINE void AddExponent(int value) {
289  DCHECK(!is_finalized());
290  DCHECK_GE(value, 0);
291  DCHECK_LE(value, 999);
292  if (value >= 100) {
293  // d1 = value / 10; d2 = value % 10;
294  // 6554 = ceil(2^16 / 10)
295  uint32_t d1 = (static_cast<uint32_t>(value) * 6554) >> 16;
296  uint32_t d2 = static_cast<uint32_t>(value) - 10 * d1;
297  DCHECK_LT(position() + 2, buffer_.length());
298  Convert2Digits(d1, cursor_);
299  cursor_ += 2;
300  AddCharacter('0' + d2);
301  } else if (value >= 10) {
302  DCHECK_LT(position() + 2, buffer_.length());
303  Convert2Digits(value, cursor_);
304  cursor_ += 2;
305  } else {
306  DCHECK_LT(value, 10);
307  AddCharacter('0' + value);
308  }
309  }
310 
311  // Finalize the string by, checking that there is no null-character in the
312  // content. Returns a pointer one past the last character.
313  char* Finalize() {
314  DCHECK(!is_finalized());
315  DCHECK_LE(position(), buffer_.size());
316 #ifdef DEBUG
317  // Make sure nobody managed to add a 0-character to the
318  // buffer while building the string.
319  for (const char* buf = buffer_.begin(); buf != cursor_; buf++) {
320  DCHECK_NE(*buf, '\0');
321  }
322 #endif
323  char* ret = cursor_;
324  cursor_ = nullptr;
325  DCHECK(is_finalized());
326  return ret;
327  }
328 
329  private:
330  base::Vector<char> buffer_;
331  char* cursor_;
332 
333  bool is_finalized() const { return cursor_ == nullptr; }
334 
335  DISALLOW_IMPLICIT_CONSTRUCTORS(SimpleStringBuilder);
336 };

366 // Returns true if a nonspace character has been found and false if the
367 // end was been reached before finding a nonspace character.
368 template <class Char>
369 inline bool AdvanceToNonspace(const Char** current, const Char* end) {
370  while (*current != end) {
371  if (!IsWhiteSpaceOrLineTerminator(**current)) return true;
372  ++*current;
373  }
374  return false;

377 // Parsing integers with radix 2, 4, 8, 16, 32. Assumes current != end.
378 template <int radix_log_2, class Char>
379 double InternalStringToIntDouble(const Char* start, const Char* end,
380  bool negative, bool allow_trailing_junk) {
381  const Char* current = start;
382  DCHECK_NE(current, end);
383 
384  // Skip leading 0s.
385  while (*current == '0') {
386  ++current;
387  if (current == end) return SignedZero(negative);
388  }
389 
390  int64_t number = 0;
391  int exponent = 0;
392  constexpr int radix = (1 << radix_log_2);
393 
394  constexpr int lim_0 = '0' + (radix < 10 ? radix : 10);
395  constexpr int lim_a = 'a' + (radix - 10);
396  constexpr int lim_A = 'A' + (radix - 10);
397 
398  do {
399  int digit;
400  if (*current >= '0' && *current < lim_0) {
401  digit = static_cast<char>(*current) - '0';
402  } else if (*current >= 'a' && *current < lim_a) {
403  digit = static_cast<char>(*current) - 'a' + 10;
404  } else if (*current >= 'A' && *current < lim_A) {
405  digit = static_cast<char>(*current) - 'A' + 10;
406  } else {
407  // We've not found any digits, this must be junk.
408  if (current == start) return JunkStringValue();
409  if (allow_trailing_junk || !AdvanceToNonspace(&current, end)) break;
410  return JunkStringValue();
411  }
412 
413  number = number * radix + digit;
414  int overflow = static_cast<int>(number >> 53);
415  if (overflow != 0) {
416  // Overflow occurred. Need to determine which direction to round the
417  // result.
418  int overflow_bits_count = 1;
419  while (overflow > 1) {
420  overflow_bits_count++;
421  overflow >>= 1;
422  }
423 
424  int dropped_bits_mask = ((1 << overflow_bits_count) - 1);
425  int dropped_bits = static_cast<int>(number) & dropped_bits_mask;
426  number >>= overflow_bits_count;
427  exponent = overflow_bits_count;
428 
429  bool zero_tail = true;
430  while (true) {
431  ++current;
432  if (current == end || !isDigit(*current, radix)) break;
433  zero_tail = zero_tail && *current == '0';
434  exponent += radix_log_2;
435  }
436 
437  if (!allow_trailing_junk && AdvanceToNonspace(&current, end)) {
438  return JunkStringValue();
439  }
440 
441  int middle_value = (1 << (overflow_bits_count - 1));
442  if (dropped_bits > middle_value) {
443  number++; // Rounding up.
444  } else if (dropped_bits == middle_value) {
445  // Rounding to even to consistency with decimals: half-way case rounds
446  // up if significant part is odd and down otherwise.
447  if ((number & 1) != 0 || !zero_tail) {
448  number++; // Rounding up.
449  }
450  }
451 
452  // Rounding up may cause overflow.
453  if ((number & (static_cast<int64_t>(1) << 53)) != 0) {
454  exponent++;
455  number >>= 1;
456  }
457  break;
458  }
459  ++current;
460  } while (current != end);
461 
462  DCHECK(number < (int64_t{1} << 53));
463  DCHECK(static_cast<int64_t>(static_cast<double>(number)) == number);
464 
465  if (exponent == 0) {
466  if (negative) {
467  if (number == 0) return -0.0;
468  number = -number;
469  }
470  return static_cast<double>(number);
471  }
472 
473  DCHECK_NE(number, 0);
474  return std::ldexp(static_cast<double>(negative ? -number : number), exponent);
475 }

592 template <class Char>
593 void StringToIntHelper::DetectRadixInternal(const Char* current,
594  size_t length) {
595  const Char* start = current;
596  length_ = length;
597  const Char* end = start + length;
598 
599  if (!AdvanceToNonspace(&current, end)) {
600  return set_state(State::kEmpty);
601  }
602 
603  if (*current == '+') {
604  // Ignore leading sign; skip following spaces.
605  ++current;
606  if (current == end) {
607  return set_state(State::kJunk);
608  }
609  sign_ = Sign::kPositive;
610  } else if (*current == '-') {
611  ++current;
612  if (current == end) {
613  return set_state(State::kJunk);
614  }
615  sign_ = Sign::kNegative;
616  }
617 
618  if (radix_ == 0) {
619  // Radix detection.
620  radix_ = 10;
621  if (*current == '0') {
622  ++current;
623  if (current == end) return set_state(State::kZero);
624  if (*current == 'x' || *current == 'X') {
625  radix_ = 16;
626  ++current;
627  if (current == end) return set_state(State::kJunk);
628  } else if (allow_binary_and_octal_prefixes_ &&
629  (*current == 'o' || *current == 'O')) {
630  radix_ = 8;
631  ++current;
632  if (current == end) return set_state(State::kJunk);
633  } else if (allow_binary_and_octal_prefixes_ &&
634  (*current == 'b' || *current == 'B')) {
635  radix_ = 2;
636  ++current;
637  if (current == end) return set_state(State::kJunk);
638  } else {
639  leading_zero_ = true;
640  }
641  }
642  } else if (radix_ == 16) {
643  if (*current == '0') {
644  // Allow "0x" prefix.
645  ++current;
646  if (current == end) return set_state(State::kZero);
647  if (*current == 'x' || *current == 'X') {
648  ++current;
649  if (current == end) return set_state(State::kJunk);
650  } else {
651  leading_zero_ = true;
652  }
653  }
654  }
655  // Skip leading zeros.
656  while (*current == '0') {
657  leading_zero_ = true;
658  ++current;
659  if (current == end) return set_state(State::kZero);
660  }
661  // Detect leading zeros with junk after them, if allowed.
662  if (leading_zero_ && allow_trailing_junk_ && !isDigit(*current, radix_)) {
663  return set_state(State::kZero);
664  }
665 
666  if (!leading_zero_ && !isDigit(*current, radix_)) {
667  return set_state(State::kJunk);
668  }
669 
670  DCHECK(radix_ >= 2 && radix_ <= 36);
671  cursor_ = current - start;
672 }

752  template <class Char>
753  void HandleBaseTenCase(const Char* current, const Char* end) {
754  // Parsing with strtod.
755  // Doubles are less than 1.8e308.
756  constexpr size_t kMaxSignificantDigits = 309;

776  result_ = Strtod(buffer_vector, 0);
777  set_state(State::kDone);
778  }

783 template <class Char>
784 void NumberParseIntHelper::HandleGenericCase(const Char* current,
785  const Char* end) {
786  // The following code causes accumulating rounding error for numbers greater
787  // than ~2^56. It's explicitly allowed in the spec: "if R is not 2, 4, 8, 10,
788  // 16, or 32, then mathInt may be an implementation-dependent approximation to
789  // the mathematical integer value" (15.1.2.2).
790 
791  int lim_0 = '0' + (radix() < 10 ? radix() : 10);
792  int lim_a = 'a' + (radix() - 10);
793  int lim_A = 'A' + (radix() - 10);
794 
795  // NOTE: The code for computing the value may seem a bit complex at
796  // first glance. It is structured to use 32-bit multiply-and-add
797  // loops as long as possible to avoid losing precision.
798 
799  bool done = false;
800  do {
801  // Parse the longest part of the string starting at {current}
802  // possible while keeping the multiplier, and thus the part
803  // itself, within 32 bits.
804  uint32_t part = 0, multiplier = 1;
805  while (true) {
806  uint32_t d;
807  if (*current >= '0' && *current < lim_0) {
808  d = *current - '0';
809  } else if (*current >= 'a' && *current < lim_a) {
810  d = *current - 'a' + 10;
811  } else if (*current >= 'A' && *current < lim_A) {
812  d = *current - 'A' + 10;
813  } else {
814  done = true;
815  break;
816  }
817 
818  // Update the value of the part as long as the multiplier fits
819  // in 32 bits. When we can't guarantee that the next iteration
820  // will not overflow the multiplier, we stop parsing the part
821  // by leaving the loop.
822  const uint32_t kMaximumMultiplier = kMaxUInt32 / 36;
823  uint32_t m = multiplier * static_cast<uint32_t>(radix());
824  if (m > kMaximumMultiplier) break;
825  part = part * radix() + d;
826  multiplier = m;
827  DCHECK(multiplier > part);
828 
829  ++current;
830  if (current == end) {
831  done = true;
832  break;
833  }
834  }
835  result_ = result_ * multiplier + part;
836  } while (!done);
837 
838  if (!allow_trailing_junk() && AdvanceToNonspace(&current, end)) {
839  return set_state(State::kJunk);
840  }
841  return set_state(State::kDone);
842 }

844 // Converts a string to a double value.
845 template <class Char>
846 double InternalStringToDouble(const Char* current, const Char* end,
847  ConversionFlag flag, double empty_string_val) {
848  // To make sure that iterator dereferencing is valid the following
849  // convention is used:
850  // 1. Each '++current' statement is followed by check for equality to 'end'.
851  // 2. If AdvanceToNonspace returned false then current == end.
852  // 3. If 'current' becomes be equal to 'end' the function returns or goes to
853  // 'parsing_done'.
854  // 4. 'current' is not dereferenced after the 'parsing_done' label.
855  // 5. Code before 'parsing_done' may rely on 'current != end'.
856  if (!AdvanceToNonspace(&current, end)) {
857  return empty_string_val;
858  }
859 
860  // The non-decimal prefix has to be the first thing after any whitespace,
861  // so check for this first.
862  if (flag == ALLOW_NON_DECIMAL_PREFIX) {
863  // Copy the current iterator, so that on a failure to find the prefix, we
864  // rewind to the start.
865  const Char* prefixed = current;
866  if (*prefixed == '0') {
867  ++prefixed;
868  if (prefixed == end) return 0;
869 
870  if (*prefixed == 'x' || *prefixed == 'X') {
871  ++prefixed;
872  if (prefixed == end) return JunkStringValue(); // "0x".
873  return InternalStringToIntDouble<4>(prefixed, end, false, false);
874  } else if (*prefixed == 'o' || *prefixed == 'O') {
875  ++prefixed;
876  if (prefixed == end) return JunkStringValue(); // "0o".
877  return InternalStringToIntDouble<3>(prefixed, end, false, false);
878  } else if (*prefixed == 'b' || *prefixed == 'B') {
879  ++prefixed;
880  if (prefixed == end) return JunkStringValue(); // "0b".
881  return InternalStringToIntDouble<1>(prefixed, end, false, false);
882  }
883  }
884  }
885 
886  // From here we are parsing a StrDecimalLiteral, as per
887  // https://tc39.es/ecma262/#sec-tonumber-applied-to-the-string-type
888  const bool allow_trailing_junk = flag == ALLOW_TRAILING_JUNK;
889 
890  double value;
891  // fast_float takes a char/char16_t instead of a uint8_t/uint16_t. Cast the
892  // pointers to match.
893  using UC = std::conditional_t<std::is_same_v<Char, uint8_t>, char, char16_t>;
894  static_assert(sizeof(UC) == sizeof(Char));
895  const UC* current_uc = reinterpret_cast<const UC*>(current);
896  const UC* end_uc = reinterpret_cast<const UC*>(end);
897  auto ret =
898  fast_float::from_chars(current_uc, end_uc, value,
899  static_cast<fast_float::chars_format>(
900  fast_float::chars_format::general |
901  fast_float::chars_format::no_infnan |
902  fast_float::chars_format::allow_leading_plus));
903  if (ret.ptr == end_uc) return value;
904  if (ret.ptr > current_uc) {
905  current = reinterpret_cast<const Char*>(ret.ptr);
906  if (!allow_trailing_junk && AdvanceToNonspace(&current, end)) {
907  return JunkStringValue();
908  }
909  return value;
910  }
911 
912  // Failed to parse any number -- handle Infinity before giving up.
913  DCHECK_EQ(ret.ptr, current_uc);
914  DCHECK_NE(current, end);
915  static constexpr char kInfinityString[] = "Infinity";
916  switch (*current) {
917  case '+':
918  // Ignore leading plus sign.
919  ++current;
920  if (current == end) return JunkStringValue();
921  if (*current != kInfinityString[0]) return JunkStringValue();
922  [[fallthrough]];
923  case kInfinityString[0]:
924  if (!SubStringEquals(&current, end, kInfinityString)) {
925  return JunkStringValue();
926  }
927  if (!allow_trailing_junk && AdvanceToNonspace(&current, end)) {
928  return JunkStringValue();
929  }
930  return V8_INFINITY;
931 
932  case '-':
933  ++current;
934  if (current == end) return JunkStringValue();
935  if (*current != kInfinityString[0]) return JunkStringValue();
936  if (!SubStringEquals(&current, end, kInfinityString)) {
937  return JunkStringValue();
938  }
939  if (!allow_trailing_junk && AdvanceToNonspace(&current, end)) {
940  return JunkStringValue();
941  }
942  return -V8_INFINITY;
943 
944  default:
945  return JunkStringValue();
946  }
947 }

1421  base::Vector<char> buffer) {

1434  size_t fraction_cursor = integer_cursor;

1437  if (negative) value = -value;
1438

1440  double integer = std::floor(value);
1441  double fraction = value - integer;
1442  // We only compute fractional digits up to the input double's precision.
1443  double delta = 0.5 * (base::Double(value).NextDouble() - value);
1444  bool delta_is_positive = true;

1446  // value. Be careful around denormal flushing when doing so.

1450  // make progress. Skip it instead.
1451  delta_is_positive = false;
1452  } else {
1453  static_assert(base::Double(0.0).NextDouble() > 0);
1454  delta = base::Double(0.0).NextDouble();

1456  }
1457  if (delta_is_positive && fraction >= delta) {
1458  // Insert decimal point.
1459  buffer[fraction_cursor++] = '.';
1460  do {
1461  // Shift up by one digit.
1462  fraction *= radix;
1463  delta *= radix;
1464  // Write digit.
1465  int digit = static_cast<int>(fraction);
1466  buffer[fraction_cursor++] = chars[digit];
1467  // Calculate remainder.
1468  fraction -= digit;
1469  // Round to even.
1470  if (fraction > 0.5 || (fraction == 0.5 && (digit & 1))) {
1471  if (fraction + delta > 1) {

1473  while (true) {

1478  integer += 1;
1479  break;

1484  if (digit + 1 < radix) {
1485  buffer[fraction_cursor++] = chars[digit + 1];
1486  break;
1487  }
1488  }
1489  break;
1490  }
1491  }
1492  } while (fraction >= delta);

1496  while (base::Double(integer / radix).Exponent() > 0) {

1498  buffer[--integer_cursor] = '0';

1502  buffer[--integer_cursor] = chars[static_cast<int>(remainder)];
1503  integer = (integer - remainder) / radix;

1508  DCHECK_LE(integer_cursor, 1u << 31); // Didn't underflow.
1509  DCHECK_GT(fraction_cursor, integer_cursor);
