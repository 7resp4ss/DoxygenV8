85 #ifdef V8_ENABLE_CHECKS
86 template <typename T, typename V = Value>
87 void TypeCheckLocal(V* value) {
88  // If `T` does not provide a `Cast` method we cannot check anything.
89  if constexpr (requires { T::Cast(value); }) {
90  // TODO(419454582): Remove all these exceptions.
91  if (std::is_same_v<Array, T> && value->IsArgumentsObject()) return;
92  if (std::is_same_v<ArrayBuffer, T> && value->IsSharedArrayBuffer()) return;
93  if (std::is_same_v<Object, T> && value->IsNull()) return;
94  if (std::is_same_v<Object, T> && value->IsString()) return;
95  if (std::is_same_v<Object, T> && value->IsUndefined()) return;
96  if (std::is_same_v<Uint32, T> && value->IsInt32()) return;
97  if (std::is_same_v<Object, T> && value->IsNumber()) return;
98  // Execute the actual check (part of the cast).
99  T::Cast(value);
100  }
101 }
102 #endif

105 /**
106  * A stack-allocated class that governs a number of local handles.
107  * After a handle scope has been created, all local handles will be
108  * allocated within that handle scope until either the handle scope is
109  * deleted or another handle scope is created. If there is already a
110  * handle scope and a new one is created, all allocations will take
111  * place in the new handle scope until it is deleted. After that,
112  * new handles will again be allocated in the original handle scope.
113  *
114  * After the handle scope of a local handle has been deleted the
115  * garbage collector will no longer track the object stored in the
116  * handle and may deallocate it. The behavior of accessing a handle
117  * for which the handle scope has been deleted is undefined.
118  */
119 class V8_EXPORT V8_NODISCARD HandleScope {
120  public:
121  explicit HandleScope(Isolate* isolate);
122 
123  ~HandleScope();
124 
125  /**
126  * Counts the number of allocated handles.
127  */
128  static int NumberOfHandles(Isolate* isolate);
129 
130  V8_INLINE Isolate* GetIsolate() const {
131  return reinterpret_cast<Isolate*>(i_isolate_);
132  }
133 
134  HandleScope(const HandleScope&) = delete;
135  void operator=(const HandleScope&) = delete;
136 
137  static internal::Address* CreateHandleForCurrentIsolate(
138  internal::Address value);
139 
140  protected:
141  V8_INLINE HandleScope() = default;
142 
143  void Initialize(Isolate* isolate);
144 
145  static internal::Address* CreateHandle(internal::Isolate* i_isolate,
146  internal::Address value);
147 
148  private:
149  // Declaring operator new and delete as deleted is not spec compliant.
150  // Therefore declare them private instead to disable dynamic alloc
151  void* operator new(size_t size);
152  void* operator new[](size_t size);
153  void operator delete(void*, size_t);
154  void operator delete[](void*, size_t);

156  internal::Isolate* i_isolate_;

159 #ifdef V8_ENABLE_CHECKS

261 /**
262  * An object reference managed by the v8 garbage collector.
263  *
264  * All objects returned from v8 have to be tracked by the garbage collector so
265  * that it knows that the objects are still alive. Also, because the garbage
266  * collector may move objects, it is unsafe to point directly to an object.
267  * Instead, all objects are stored in handles which are known by the garbage
268  * collector and updated whenever an object moves. Handles should always be
269  * passed by value (except in cases like out-parameters) and they should never
270  * be allocated on the heap.
271  *
272  * There are two types of handles: local and persistent handles.
273  *
274  * Local handles are light-weight and transient and typically used in local
275  * operations. They are managed by HandleScopes. That means that a HandleScope
276  * must exist on the stack when they are created and that they are only valid
277  * inside of the HandleScope active during their creation. For passing a local
278  * handle to an outer HandleScope, an EscapableHandleScope and its Escape()
279  * method must be used.
280  *
281  * Persistent handles can be used when storing objects across several
282  * independent operations and have to be explicitly deallocated when they're no
283  * longer used.
284  *
285  * It is safe to extract the object stored in the handle by dereferencing the
286  * handle (for instance, to extract the Object* from a Local<Object>); the value
287  * will still be governed by a handle behind the scenes and the same rules apply
288  * to these values as to their handles.
289  */
290 template <class T>
291 class V8_TRIVIAL_ABI Local : public LocalBase<T>,
292 #ifdef V8_ENABLE_LOCAL_OFF_STACK_CHECK
293  public api_internal::StackAllocated<true>
294 #else
295  public api_internal::StackAllocated<false>
296 #endif
297 {
298  public:
299  /**
300  * Default constructor: Returns an empty handle.
301  */
302  V8_INLINE Local() = default;
303 
304  /**
305  * Constructor for handling automatic up casting.
306  * Ex. Local<Object> can be passed when Local<Value> is expected but not
307  * the other way round.
308  */
309  template <class S>
310  requires std::is_base_of_v<T, S>
311  V8_INLINE Local(Local<S> that) : LocalBase<T>(that) {}
312 
313  V8_INLINE T* operator->() const { return this->template value<T>(); }
314 
315  V8_INLINE T* operator*() const { return this->operator->(); }
316 
317  /**
318  * Checks whether two handles are equal or different.
319  * They are equal iff they are both empty or they are both non-empty and the
320  * objects to which they refer are physically equal.
321  *
322  * If both handles refer to JS objects, this is the same as strict
323  * non-equality. For primitives, such as numbers or strings, a `true` return
324  * value does not indicate that the values aren't equal in the JavaScript
325  * sense. Use `Value::StrictEquals()` to check primitives for equality.
326  */
327 
328  template <class S>
329  V8_INLINE bool operator==(const Local<S>& that) const {
330  return internal::HandleHelper::EqualHandles(*this, that);
331  }
332 
333  template <class S>
334  V8_INLINE bool operator==(const PersistentBase<S>& that) const {
335  return internal::HandleHelper::EqualHandles(*this, that);
336  }
337 
338  template <class S>
339  V8_INLINE bool operator!=(const Local<S>& that) const {
340  return !operator==(that);
341  }
342 
343  template <class S>
344  V8_INLINE bool operator!=(const Persistent<S>& that) const {
345  return !operator==(that);
346  }
347 
348  /**
349  * Cast a handle to a subclass, e.g. Local<Value> to Local<Object>.
350  * This is only valid if the handle actually refers to a value of the
351  * target type or if the handle is empty.
352  */
353  template <class S>
354  V8_INLINE static Local<T> Cast(Local<S> that) {
355 #ifdef V8_ENABLE_CHECKS
356  // If we're going to perform the type check then we have to check
357  // that the handle isn't empty before doing the checked cast.
358  if (that.IsEmpty()) return Local<T>();
359  T::Cast(that.template value<S>());
360 #endif
361  return Local<T>(LocalBase<T>(that));
362  }
363 
364  /**
365  * Calling this is equivalent to Local<S>::Cast().
366  * In particular, this is only valid if the handle actually refers to a value
367  * of the target type or if the handle is empty.
368  */
369  template <class S>
370  V8_INLINE Local<S> As() const {
371  return Local<S>::Cast(*this);
372  }
373

377  * the original handle is destroyed/disposed.
378  */

654 /**
655  * A MaybeLocal<> is a wrapper around Local<> that enforces a check whether
656  * the Local<> is empty before it can be used.
657  *
658  * If an API method returns a MaybeLocal<>, the API method can potentially fail
659  * either because an exception is thrown, or because an exception is pending,
660  * e.g. because a previous API call threw an exception that hasn't been caught
661  * yet, or because a TerminateExecution exception was thrown. In that case, an
662  * empty MaybeLocal is returned.
663  */
664 template <class T>
665 class MaybeLocal {
666  public:
667  /**
668  * Default constructor: Returns an empty handle.
669  */
670  V8_INLINE MaybeLocal() = default;
671  /**
672  * Implicitly construct MaybeLocal from Local.
673  */
674  template <class S>
675  requires std::is_base_of_v<T, S>
