39 class IterationCount {
40  enum class Kind { kExact, kApprox, kUnknown };
41 
42  public:
43  // Loops with an exact number of iteration could be unrolled.
44  static IterationCount Exact(size_t count) {
45  return IterationCount(Kind::kExact, count);
46  }
47  // We can remove stack checks from loops with a small number of iterations.
48  static IterationCount Approx(size_t count) {
49  return IterationCount(Kind::kApprox, count);
50  }
51  static IterationCount Unknown() { return IterationCount(Kind::kUnknown); }
52 
53  IterationCount() : kind_(Kind::kUnknown) {}
54  explicit IterationCount(Kind kind) : kind_(kind) {
55  DCHECK_NE(kind, Kind::kExact);
56  }
57  IterationCount(Kind kind, size_t count) : kind_(kind), count_(count) {
58  DCHECK_EQ(kind, any_of(Kind::kExact, Kind::kApprox));
59  }
60 
61  size_t exact_count() const {
62  DCHECK_EQ(kind_, Kind::kExact);
63  return count_;
64  }
65  size_t approx_count() const {
66  DCHECK_EQ(kind_, Kind::kApprox);
67  return count_;
68  }
69 
70  bool IsExact() const { return kind_ == Kind::kExact; }
71  bool IsApprox() const { return kind_ == Kind::kApprox; }
72  bool IsUnknown() const { return kind_ == Kind::kUnknown; }
73 
74  bool IsSmallerThan(size_t max) {
75  return (IsExact() || IsApprox()) && count_ < max;
76  }
77 
78  private:
79  Kind kind_;
80  size_t count_;
81 };

168 class V8_EXPORT_PRIVATE LoopUnrollingAnalyzer {
169  // LoopUnrollingAnalyzer analyzes the loops of the graph, and in particular
170  // tries to figure out if some inner loops have a fixed (and known) number of
171  // iterations. In particular, it tries to pattern match loops like
172  //
173  // for (let i = 0; i < 4; i++) { ... }
174  //
175  // where `i++` could alternatively be pretty much any WordBinopOp or
176  // OverflowCheckedBinopOp, and `i < 4` could be any ComparisonOp.
177  // Such loops, if small enough, could be fully unrolled.
178  //
179  // Loops that don't have statically-known bounds could still be partially
180  // unrolled if they are small enough.
181  public:
182  LoopUnrollingAnalyzer(Zone* phase_zone, Graph* input_graph, bool is_wasm)
183  : input_graph_(input_graph),
184  matcher_(*input_graph),
185  loop_finder_(phase_zone, input_graph,
186  {LoopFinder::ConfigFlags::kFindCalls}),
187  loop_iteration_count_(phase_zone),
188  canonical_loop_matcher_(matcher_),
189  is_wasm_(is_wasm),
190  stack_checks_to_remove_(input_graph->stack_checks_to_remove()) {
191  DetectUnrollableLoops();
192  }
193 
194  bool ShouldFullyUnrollLoop(const Block* loop_header) const {
195  DCHECK(loop_header->IsLoop());
196 
197  LoopFinder::LoopInfo header_info = loop_finder_.GetLoopInfo(loop_header);
198  if (header_info.has_inner_loops) return false;
199  if (header_info.op_count > kMaxLoopSizeForFullUnrolling) return false;
200 
201  auto iter_count = GetIterationCount(loop_header);
202  return iter_count.IsExact() &&
203  iter_count.exact_count() < kMaxLoopIterationsForFullUnrolling;
204  }
205 
206  bool ShouldPartiallyUnrollLoop(const Block* loop_header) const {
207  DCHECK(loop_header->IsLoop());
208  LoopFinder::LoopInfo info = loop_finder_.GetLoopInfo(loop_header);
209  return !info.has_inner_loops && !info.has_any_call &&
210  info.op_count < kMaxLoopSizeForPartialUnrolling;
211  }
212 
213  // The returned unroll count is the total number of copies of the loop body
214  // in the resulting graph, i.e., an unroll count of N means N-1 copies of the
215  // body which were partially unrolled, and 1 for the original/remaining body.
216  size_t GetPartialUnrollCount(const Block* loop_header) const {
217  // Don't unroll if the function is already huge.
218  // Otherwise we have run into pathological runtimes or large memory usage,
219  // e.g., in register allocation in the past, see https://crbug.com/383661627
220  // for an example / reproducer.
221  // Even though we return an unroll count of one (i.e., don't unroll at all
222  // really), running this phase can speed up subsequent optimizations,
223  // probably because it produces loops in a "compact"/good block order for
224  // analyses, namely <loop header>, <loop body>, <loop exit>, <rest of code>.
225  // In principle, we should fix complexity problems in analyses, make sure
226  // loops are already produced in this order, and not rely on the "unrolling"
227  // here for the order alone, but this is a longer standing issue.
228  if (input_graph_->op_id_count() > kMaxFunctionSizeForPartialUnrolling) {
229  return 1;
230  }
231  if (is_wasm_) {
232  LoopFinder::LoopInfo info = loop_finder_.GetLoopInfo(loop_header);
233  return std::min(
234  LoopUnrollingAnalyzer::kMaxPartialUnrollingCount,
235  LoopUnrollingAnalyzer::kWasmMaxUnrolledLoopSize / info.op_count);
236  }
237  return LoopUnrollingAnalyzer::kMaxPartialUnrollingCount;
238  }
239 
240  bool ShouldRemoveLoop(const Block* loop_header) const {
241  auto iter_count = GetIterationCount(loop_header);
242  return iter_count.IsExact() && iter_count.exact_count() == 0;
243  }
244 
245  IterationCount GetIterationCount(const Block* loop_header) const {
246  DCHECK(loop_header->IsLoop());
247  auto it = loop_iteration_count_.find(loop_header);
248  if (it == loop_iteration_count_.end()) return IterationCount::Unknown();
249  return it->second;
250  }
251 
252  ZoneSet<const Block*, LoopFinder::BlockCmp> GetLoopBody(
253  const Block* loop_header) {
254  return loop_finder_.GetLoopBody(loop_header);
255  }
256 
257  const Block* GetLoopHeader(const Block* block) {
258  return loop_finder_.GetLoopHeader(block);
259  }
260 
261  bool CanUnrollAtLeastOneLoop() const { return can_unroll_at_least_one_loop_; }
262 
263  // TODO(dmercadier): consider tweaking these value for a better size-speed
264  // trade-off. In particular, having the number of iterations to unroll be a
265  // function of the loop's size and a MaxLoopSize could make sense.
266  static constexpr size_t kMaxLoopSizeForFullUnrolling = 150;
267  // This function size limit is quite arbitrary. It is large enough that we
268  // probably never hit it in JavaScript and it is lower than the operation
269  // count we have seen in some huge Wasm functions in the past, e.g., function
270  // #21937 of https://crbug.com/383661627 (1.7M operations, 2.7MB wire bytes).
271  static constexpr size_t kMaxFunctionSizeForPartialUnrolling = 1'000'000;
272  static constexpr size_t kJSMaxLoopSizeForPartialUnrolling = 50;
273  static constexpr size_t kWasmMaxLoopSizeForPartialUnrolling = 80;
274  static constexpr size_t kWasmMaxUnrolledLoopSize = 240;
275  static constexpr size_t kMaxLoopIterationsForFullUnrolling = 4;
276  static constexpr size_t kMaxPartialUnrollingCount = 4;
277  static constexpr size_t kMaxIterForStackCheckRemoval = 5000;
278 
279  private:
280  void DetectUnrollableLoops();
281  IterationCount GetLoopIterationCount(const LoopFinder::LoopInfo& info) const;
282 
283  Graph* input_graph_;
284  OperationMatcher matcher_;
285  LoopFinder loop_finder_;
286  // {loop_iteration_count_} maps loop headers to number of iterations. It
287  // doesn't contain entries for loops for which we don't know the number of
288  // iterations.
289  ZoneUnorderedMap<const Block*, IterationCount> loop_iteration_count_;
290  const StaticCanonicalForLoopMatcher canonical_loop_matcher_;
291  const bool is_wasm_;
292  const size_t kMaxLoopSizeForPartialUnrolling =
293  is_wasm_ ? kWasmMaxLoopSizeForPartialUnrolling
294  : kJSMaxLoopSizeForPartialUnrolling;
295  bool can_unroll_at_least_one_loop_ = false;
296 
297  ZoneAbslFlatHashSet<uint32_t>& stack_checks_to_remove_;
298 };

357 template <class Next>
358 class LoopUnrollingReducer : public Next {
359  public:
360  TURBOSHAFT_REDUCER_BOILERPLATE(LoopUnrolling)
361 
362 #if defined(__clang__)
363  // LoopUnrolling and LoopPeeling shouldn't be performed in the same phase, see
364  // the comment in pipeline.cc where LoopUnrolling is triggered.
365  static_assert(!reducer_list_contains<ReducerList, LoopPeelingReducer>::value);
366 
367  // TODO(dmercadier): Add static_assert that this is ran as part of a
368  // CopyingPhase.
369 #endif
370 
371  V<None> REDUCE_INPUT_GRAPH(Goto)(V<None> ig_idx, const GotoOp& gto) {
372  // Note that the "ShouldSkipOptimizationStep" are placed in the parts of
373  // this Reduce method triggering the unrolling rather than at the begining.
374  // This is because the backedge skipping is not an optimization but a
375  // mandatory lowering when unrolling is being performed.
376  LABEL_BLOCK(no_change) { return Next::ReduceInputGraphGoto(ig_idx, gto); }
377 
378  const Block* dst = gto.destination;
379  if (unrolling_ == UnrollingStatus::kNotUnrolling && dst->IsLoop() &&
380  !gto.is_backedge) {
381  // We trigger unrolling when reaching the GotoOp that jumps to the loop
382  // header (note that loop headers only have 2 predecessor, including the
383  // backedge), and that isn't the backedge.
384  if (ShouldSkipOptimizationStep()) goto no_change;
385  if (analyzer_.ShouldRemoveLoop(dst)) {
386  RemoveLoop(dst);
387  return {};
388  } else if (analyzer_.ShouldFullyUnrollLoop(dst)) {
389  FullyUnrollLoop(dst);
390  return {};
391  } else if (analyzer_.ShouldPartiallyUnrollLoop(dst)) {
392  PartiallyUnrollLoop(dst);
393  return {};
394  }
395  } else if ((unrolling_ == UnrollingStatus::kUnrolling) &&
396  dst == current_loop_header_) {
397  // Skipping the backedge of the loop: FullyUnrollLoop and
398  // PartiallyUnrollLoop will emit a Goto to the next unrolled iteration.
399  return {};
400  }
401  goto no_change;
402  }
403 
404  V<None> REDUCE_INPUT_GRAPH(Branch)(V<None> ig_idx, const BranchOp& branch) {
405  LABEL_BLOCK(no_change) {
406  return Next::ReduceInputGraphBranch(ig_idx, branch);
407  }
408 
409  if (unrolling_ == UnrollingStatus::kRemoveLoop) {
410  // We know that the branch of the final inlined header of a fully unrolled
411  // loop never actually goes to the loop, so we can replace it by a Goto
412  // (so that the non-unrolled loop doesn't get emitted). We still need to
413  // figure out if we should Goto to the true or false side of the BranchOp.
414  const Block* header = __ current_block()->OriginForBlockEnd();
415  bool is_true_in_loop = analyzer_.GetLoopHeader(branch.if_true) == header;
416  bool is_false_in_loop =
417  analyzer_.GetLoopHeader(branch.if_false) == header;
418 
419  if (is_true_in_loop && !is_false_in_loop) {
420  __ Goto(__ MapToNewGraph(branch.if_false));
421  return OpIndex::Invalid();
422  } else if (is_false_in_loop && !is_true_in_loop) {
423  __ Goto(__ MapToNewGraph(branch.if_true));
424  return OpIndex::Invalid();
425  } else {
426  // Both the true and false destinations of this block are in the loop,
427  // which means that the exit of the loop is later down the graph. We
428  // thus still emit the branch, which will lead to the loop being emitted
429  // (unless some other reducers in the stack manage to get rid of the
430  // loop).
431  DCHECK(is_true_in_loop && is_false_in_loop);
432  }
433  }
434  goto no_change;
435  }
436 
437  V<None> REDUCE_INPUT_GRAPH(JSStackCheck)(V<None> ig_idx,
438  const JSStackCheckOp& check) {
439  if (ShouldSkipOptimizationStep() || !skip_next_stack_check_) {
440  return Next::ReduceInputGraphJSStackCheck(ig_idx, check);
441  }
442  return V<None>::Invalid();
443  }
444 
445 #if V8_ENABLE_WEBASSEMBLY
446  V<None> REDUCE_INPUT_GRAPH(WasmStackCheck)(V<None> ig_idx,
447  const WasmStackCheckOp& check) {
448  if (ShouldSkipOptimizationStep() || !skip_next_stack_check_) {
449  return Next::ReduceInputGraphWasmStackCheck(ig_idx, check);
450  }
451  return V<None>::Invalid();
452  }
453 #endif
454 
455  private:
456  enum class UnrollingStatus {
457  // Not currently unrolling a loop.
458  kNotUnrolling,
459  // Currently unrolling a loop.
460  kUnrolling,
461  // We use kRemoveLoop in 2 cases:
462  // - When unrolling is finished and we are currently emitting the header
463  // one last time, and should change its final branch into a Goto.
464  // - We decided to remove a loop and will just emit its header.
465  // Both cases are fairly similar: we are currently emitting a loop header,
466  // and would like to not emit the loop body that follows.
467  kRemoveLoop,
468  };
469  void RemoveLoop(const Block* header);
470  void FullyUnrollLoop(const Block* header);
471  void PartiallyUnrollLoop(const Block* header);
472  void FixLoopPhis(const Block* input_graph_loop, Block* output_graph_loop,
473  const Block* backedge_block);
474  bool IsRunningBuiltinPipeline() {
475  return __ data() -> pipeline_kind() == TurboshaftPipelineKind::kCSA;
476  }
477  bool StopUnrollingIfUnreachable(
478  std::optional<Block*> output_graph_header = std::nullopt) {
479  if (__ generating_unreachable_operations()) {
480  // By unrolling the loop, we realized that it was actually exiting early
481  // (probably because a Branch inside the loop was using a loop Phi in a
482  // condition, and unrolling showed that this loop Phi became true or
483  // false), and that lasts iterations were unreachable. We thus don't both
484  // unrolling the next iterations of the loop.
485  unrolling_ = UnrollingStatus::kNotUnrolling;
486  if (output_graph_header.has_value()) {
487  // The loop that we're unrolling has a header (which means that we're
488  // only partially unrolling), which needs to be turned into a Merge (and
489  // its PendingLoopPhis into regular Phis).
490  __ FinalizeLoop(*output_graph_header);
491  }
492  return true;
493  }
494  return false;
495  }
496 
497  // The analysis should be ran ahead of time so that the LoopUnrollingPhase
498  // doesn't trigger the CopyingPhase if there are no loops to unroll.
499  LoopUnrollingAnalyzer& analyzer_ =
500  *__ input_graph().loop_unrolling_analyzer();
501  // {unrolling_} is true if a loop is currently being unrolled.
502  UnrollingStatus unrolling_ = UnrollingStatus::kNotUnrolling;
503  bool skip_next_stack_check_ = false;
504 
505  const Block* current_loop_header_ = nullptr;
506  JSHeapBroker* broker_ = __ data() -> broker();
507 };

509 template <class Next>
510 void LoopUnrollingReducer<Next>::PartiallyUnrollLoop(const Block* header) {
511  TRACE("LoopUnrolling: partially unrolling loop at " << header->index().id());
512  DCHECK_EQ(unrolling_, UnrollingStatus::kNotUnrolling);
513  DCHECK(!skip_next_stack_check_);
514  unrolling_ = UnrollingStatus::kUnrolling;
515 
516  auto loop_body = analyzer_.GetLoopBody(header);
517  current_loop_header_ = header;
518 
519  size_t unroll_count = analyzer_.GetPartialUnrollCount(header);
520  DCHECK_GT(unroll_count, 0);
521  TRACE("> UnrollCount: " << unroll_count);
522 
523  ScopedModification<bool> set_true(__ turn_loop_without_backedge_into_merge(),
524  false);
525 
526  // We remove the stack check of all iterations but the last one.
527  // Emitting the 1st iteration of the loop (with a proper loop header). We
528  // remove the stack check of all iterations except the last one.
529  ScopedModification<bool> skip_stack_checks(&skip_next_stack_check_, true);
530  TRACE("> Emitting first iteraton (with header)");
531  Block* output_graph_header =
532  __ CloneSubGraph(loop_body, /* keep_loop_kinds */ true);
533  if (StopUnrollingIfUnreachable(output_graph_header)) {
534  TRACE("> Next iteration is unreachable, stopping unrolling");
535  return;
536  }
537 
538  // Emitting the subsequent folded iterations. We set `unrolling_` to
539  // kUnrolling so that stack checks are skipped.
540  unrolling_ = UnrollingStatus::kUnrolling;
541  for (size_t i = 0; i < unroll_count - 1; i++) {
542  // We remove the stack check of all iterations but the last one.
543  TRACE("> Emitting iteration " << i);
544  bool is_last_iteration = i == unroll_count - 2;
545  ScopedModification<bool> inner_skip_stack_checks(&skip_next_stack_check_,
546  !is_last_iteration);
547 
548  __ CloneSubGraph(loop_body, /* keep_loop_kinds */ false);
549  if (StopUnrollingIfUnreachable(output_graph_header)) {
550  TRACE("> Next iteration is unreachable, stopping unrolling");
551  return;
552  }
553  }
554 
555  // ReduceInputGraphGoto ignores backedge Gotos while kUnrolling is true, which
556  // means that we are still missing the loop's backedge, which we thus emit
557  // now.
558  DCHECK(output_graph_header->IsLoop());
559  Block* backedge_block = __ current_block();
560  __ Goto(output_graph_header);
561  // We use a custom `FixLoopPhis` because the mapping from old->new is a bit
562  // "messed up" by having emitted multiple times the same block. See the
563  // comments in `FixLoopPhis` for more details.
564  TRACE("> Patching loop phis");
565  FixLoopPhis(header, output_graph_header, backedge_block);
566 
567  unrolling_ = UnrollingStatus::kNotUnrolling;
568  TRACE("> Finished partially unrolling loop " << header->index().id());
569 }

571 template <class Next>
572 void LoopUnrollingReducer<Next>::FixLoopPhis(const Block* input_graph_loop,
573  Block* output_graph_loop,
574  const Block* backedge_block) {
575  // FixLoopPhis for partially unrolled loops is a bit tricky: the mapping from
576  // input Loop Phis to output Loop Phis is in the Variable Snapshot of the
577  // header (`output_graph_loop`), but the mapping from the 2nd input of the
578  // input graph loop phis to the 2nd input of the output graph loop phis is in
579  // the snapshot of the backedge (`backedge_block`).
580  // VariableReducer::ReduceGotoOp (which was called right before this function
581  // because we emitted the backedge Goto) already set the current snapshot to
582  // be at the loop header. So, we start by computing the mapping input loop
583  // phis -> output loop phis (using the loop header's snapshot). Then, we
584  // restore the backedge snapshot to compute the mapping input graph 2nd phi
585  // input to output graph 2nd phi input.
586  DCHECK(input_graph_loop->IsLoop());
587  DCHECK(output_graph_loop->IsLoop());
588 
589  // The mapping InputGraphPhi -> OutputGraphPendingPhi should be retrieved from
590  // `output_graph_loop`'s snapshot (the current mapping is for the latest
591  // folded loop iteration, not for the loop header).
592  __ SealAndSaveVariableSnapshot();
593  __ RestoreTemporaryVariableSnapshotAfter(output_graph_loop);
594  base::SmallVector<std::pair<const PhiOp*, const OpIndex>, 16> phis;
595  for (const Operation& op : __ input_graph().operations(
596  input_graph_loop->begin(), input_graph_loop->end())) {
597  if (auto* input_phi = op.TryCast<PhiOp>()) {
598  OpIndex phi_index =
599  __ template MapToNewGraph<true>(__ input_graph().Index(*input_phi));
600  if (!phi_index.valid() || !output_graph_loop->Contains(phi_index)) {
601  // Unused phis are skipped, so they are not be mapped to anything in
602  // the new graph. If the phi is reduced to an operation from a
603  // different block, then there is no loop phi in the current loop
604  // header to take care of.
605  continue;
606  }
607  phis.push_back({input_phi, phi_index});
608  }
609  }
610 
611  // The mapping for the InputGraphPhi 2nd input should however be retrieved
612  // from the last block of the loop.
613  __ CloseTemporaryVariableSnapshot();
614  __ RestoreTemporaryVariableSnapshotAfter(backedge_block);
615 
616  for (auto [input_phi, output_phi_index] : phis) {
617  __ FixLoopPhi(*input_phi, output_phi_index, output_graph_loop);
618  }
619 
620  __ CloseTemporaryVariableSnapshot();
621 }

623 template <class Next>
624 void LoopUnrollingReducer<Next>::RemoveLoop(const Block* header) {
625  TRACE("LoopUnrolling: removing loop at " << header->index().id());
626  DCHECK_EQ(unrolling_, UnrollingStatus::kNotUnrolling);
627  DCHECK(!skip_next_stack_check_);
628  // When removing a loop, we still need to emit the header (since it has to
629  // always be executed before the 1st iteration anyways), but by setting
630  // {unrolling_} to `kRemoveLoop`, the final Branch of the loop will become a
631  // Goto to outside the loop.
632  unrolling_ = UnrollingStatus::kRemoveLoop;
633  __ CloneAndInlineBlock(header);
634  unrolling_ = UnrollingStatus::kNotUnrolling;
635 }

637 template <class Next>
638 void LoopUnrollingReducer<Next>::FullyUnrollLoop(const Block* header) {
639  TRACE("LoopUnrolling: fully unrolling loop at " << header->index().id());
640  DCHECK_EQ(unrolling_, UnrollingStatus::kNotUnrolling);
641  DCHECK(!skip_next_stack_check_);
642  ScopedModification<bool> skip_stack_checks(&skip_next_stack_check_, true);
643 
644  size_t iter_count = analyzer_.GetIterationCount(header).exact_count();
645  TRACE("> iter_count: " << iter_count);
646 
647  auto loop_body = analyzer_.GetLoopBody(header);
648  current_loop_header_ = header;
649 
650  unrolling_ = UnrollingStatus::kUnrolling;
651  for (size_t i = 0; i < iter_count; i++) {
652  TRACE("> Emitting iteration " << i);
653  __ CloneSubGraph(loop_body, /* keep_loop_kinds */ false);
654  if (StopUnrollingIfUnreachable()) {
655  TRACE("> Next iteration is unreachable, stopping unrolling");
656  return;
657  }
658  }
659 
660  // The loop actually finishes on the header rather than its last block. We
661  // thus inline the header, and we'll replace its final BranchOp by a GotoOp to
662  // outside of the loop.
663  TRACE("> Emitting the final header");
664  unrolling_ = UnrollingStatus::kRemoveLoop;
665  __ CloneAndInlineBlock(header);
666 
667  unrolling_ = UnrollingStatus::kNotUnrolling;
668  TRACE("> Finished fully unrolling loop " << header->index().id());
669 }
