46 void ConstantExpressionInterface::UnOp(FullDecoder* decoder, WasmOpcode opcode,
47  const Value& input, Value* result) {
48  if (!generate_value()) return;
49  switch (opcode) {
50  case kExprExternConvertAny: {
51  result->runtime_value =
52  WasmValue(WasmToJSObject(isolate_, input.runtime_value.to_ref()),
53  CanonicalValueType::RefMaybeNull(kWasmExternRef,
54  input.type.nullability()));
55  break;
56  }
57  case kExprAnyConvertExtern: {
58  const char* error_message = nullptr;
59  result->runtime_value =
60  WasmValue(JSToWasmObject(isolate_, input.runtime_value.to_ref(),
61  kWasmAnyRef, &error_message)
62  .ToHandleChecked(),
63  CanonicalValueType::RefMaybeNull(kWasmAnyRef,
64  input.type.nullability()));
65  break;
66  }
67  default:
68  UNREACHABLE();
69  }
70 }

72 void ConstantExpressionInterface::BinOp(FullDecoder* decoder, WasmOpcode opcode,
73  const Value& lhs, const Value& rhs,
74  Value* result) {
75  if (!generate_value()) return;
76  switch (opcode) {
77  case kExprI32Add:
78  result->runtime_value = WasmValue(base::AddWithWraparound(
79  lhs.runtime_value.to_i32(), rhs.runtime_value.to_i32()));
80  break;
81  case kExprI32Sub:
82  result->runtime_value = WasmValue(base::SubWithWraparound(
83  lhs.runtime_value.to_i32(), rhs.runtime_value.to_i32()));
84  break;
85  case kExprI32Mul:
86  result->runtime_value = WasmValue(base::MulWithWraparound(
87  lhs.runtime_value.to_i32(), rhs.runtime_value.to_i32()));
88  break;
89  case kExprI64Add:
90  result->runtime_value = WasmValue(base::AddWithWraparound(
91  lhs.runtime_value.to_i64(), rhs.runtime_value.to_i64()));
92  break;
93  case kExprI64Sub:
94  result->runtime_value = WasmValue(base::SubWithWraparound(
95  lhs.runtime_value.to_i64(), rhs.runtime_value.to_i64()));
96  break;
97  case kExprI64Mul:
98  result->runtime_value = WasmValue(base::MulWithWraparound(
99  lhs.runtime_value.to_i64(), rhs.runtime_value.to_i64()));
100  break;
101  default:
102  UNREACHABLE();
103  }
104 }

179 void ConstantExpressionInterface::StructNew(FullDecoder* decoder,
180  const StructIndexImmediate& imm,
181  const Value& descriptor,
182  const Value args[], Value* result) {
183  // Declaratively adding a prototype is only permitted if the initializer
184  // ends with struct.new[_default]. Since no control flow is allowed in
185  // initializers, an 'end' instruction indicates the end of the initializer.
186  int offset_to_next_instr = 2 /* prefix, opcode */ + imm.length;
187  ends_with_struct_new_ = decoder->lookahead(offset_to_next_instr, kExprEnd);
188 
189  if (!generate_value()) return;
190  DirectHandle<WasmTrustedInstanceData> data =
191  GetTrustedInstanceDataForTypeIndex(imm.index);
192  const TypeDefinition& type = module_->type(imm.index);
193  const StructType* struct_type = type.struct_type;
194  DCHECK_EQ(struct_type, imm.struct_type);
195 
196  DirectHandle<Map> rtt = GetRtt(data, imm.index, type, descriptor);
197  if (rtt.is_null()) return; // Trap (descriptor was null).
198 
199  DirectHandle<WasmStruct> obj;
200  WriteBarrierMode mode = UPDATE_WRITE_BARRIER;
201  if (type.is_descriptor()) {
202  DirectHandle<Object> first_field =
203  struct_type->first_field_can_be_prototype()
204  ? args[0].runtime_value.to_ref()
205  : direct_handle(Smi::zero(), isolate_);
206  obj = WasmStruct::AllocateDescriptorUninitialized(isolate_, data, imm.index,
207  rtt, first_field);
208  } else {
209  obj = isolate_->factory()->NewWasmStructUninitialized(struct_type, rtt);
210  mode = SKIP_WRITE_BARRIER; // Object is in new space.
211  }
212  DisallowGarbageCollection no_gc; // Must initialize fields first.
213 
214  for (uint32_t i = 0; i < struct_type->field_count(); i++) {
215  int offset = struct_type->field_offset(i);
216  if (struct_type->field(i).is_numeric()) {
217  uint8_t* address =
218  reinterpret_cast<uint8_t*>(obj->RawFieldAddress(offset));
219  args[i].runtime_value.Packed(struct_type->field(i)).CopyTo(address);
220  } else {
221  obj->SetTaggedFieldValue(offset, *args[i].runtime_value.to_ref(), mode);
222  }
223  }
224  result->runtime_value = WasmValue(
225  obj,
226  decoder->module_->canonical_type(
227  ValueType::Ref(imm.heap_type()).AsExactIfEnabled(decoder->enabled_)));
228 }

252 WasmValue DefaultValueForType(ValueType type, Isolate* isolate,
253  const WasmModule* module) {
254  switch (type.kind()) {
255  case kI32:
256  case kI8:
257  case kI16:
258  return WasmValue(0);
259  case kI64:
260  return WasmValue(int64_t{0});
261  case kF16:
262  case kF32:
263  return WasmValue(0.0f);
264  case kF64:
265  return WasmValue(0.0);
266  case kS128:
267  return WasmValue(Simd128());
268  case kRefNull:
269  return WasmValue(type.use_wasm_null()
270  ? Cast<Object>(isolate->factory()->wasm_null())
271  : Cast<Object>(isolate->factory()->null_value()),
272  module->canonical_type(type));
273  case kVoid:
274  case kRef:
275  case kTop:
276  case kBottom:
277  UNREACHABLE();
278  }
279 }

282 void ConstantExpressionInterface::StructNewDefault(
283  FullDecoder* decoder, const StructIndexImmediate& imm,
284  const Value& descriptor, Value* result) {
285  // Declaratively adding a prototype is only permitted if the initializer
286  // ends with struct.new[_default]. Since no control flow is allowed in
287  // initializers, an 'end' instruction indicates the end of the initializer.
288  int offset_to_next_instr = 2 /* prefix, opcode */ + imm.length;
289  ends_with_struct_new_ = decoder->lookahead(offset_to_next_instr, kExprEnd);
290 
291  if (!generate_value()) return;
292  DirectHandle<WasmTrustedInstanceData> data =
293  GetTrustedInstanceDataForTypeIndex(imm.index);
294  const TypeDefinition& type = module_->type(imm.index);
295  const StructType* struct_type = type.struct_type;
296  DCHECK_EQ(struct_type, imm.struct_type);
297 
298  DirectHandle<Map> rtt = GetRtt(data, imm.index, type, descriptor);
299  if (rtt.is_null()) return; // Trap (descriptor was null).
300 
301  DirectHandle<WasmStruct> obj;
302  if (type.is_descriptor()) {
303  DirectHandle<Object> first_field(Smi::zero(), isolate_);
304  obj = WasmStruct::AllocateDescriptorUninitialized(isolate_, data, imm.index,
305  rtt, first_field);
306  } else {
307  obj = isolate_->factory()->NewWasmStructUninitialized(struct_type, rtt);
308  }
309  DisallowGarbageCollection no_gc; // Must initialize fields first.
310 
311  for (uint32_t i = 0; i < struct_type->field_count(); i++) {
312  int offset = struct_type->field_offset(i);
313  ValueType ftype = struct_type->field(i);
314  if (ftype.is_numeric()) {
315  uint8_t* address =
316  reinterpret_cast<uint8_t*>(obj->RawFieldAddress(offset));
317  DefaultValueForType(ftype, isolate_, module_)
318  .Packed(ftype)
319  .CopyTo(address);
320  } else {
321  TaggedField<Object, WasmStruct::kHeaderSize>::store(
322  *obj, offset,
323  *DefaultValueForType(ftype, isolate_, module_).to_ref());
324  }
325  }
326 
327  result->runtime_value = WasmValue(
328  obj,
329  decoder->module_->canonical_type(
330  ValueType::Ref(imm.heap_type()).AsExactIfEnabled(decoder->enabled_)));
331 }

391 void ConstantExpressionInterface::ArrayNewSegment(
392  FullDecoder* decoder, const ArrayIndexImmediate& array_imm,
393  const IndexImmediate& segment_imm, const Value& offset_value,
394  const Value& length_value, Value* result) {
395  if (!generate_value()) return;
396 
397  DirectHandle<WasmTrustedInstanceData> data =
398  GetTrustedInstanceDataForTypeIndex(array_imm.index);
399 
400  DirectHandle<Map> rtt{
401  Cast<Map>(data->managed_object_maps()->get(array_imm.index.index)),
402  isolate_};
403  DCHECK_EQ(rtt->wasm_type_info()->type_index(),
404  decoder->module_->canonical_type_id(array_imm.index));
405 
406  uint32_t length = length_value.runtime_value.to_u32();
407  uint32_t offset = offset_value.runtime_value.to_u32();
408  if (length >
409  static_cast<uint32_t>(WasmArray::MaxLength(array_imm.array_type))) {
410  error_ = MessageTemplate::kWasmTrapArrayTooLarge;
411  return;
412  }
413  CanonicalValueType element_type = rtt->wasm_type_info()->element_type();
414  CanonicalValueType result_type =
415  rtt->wasm_type_info()->type().AsExactIfEnabled(decoder->enabled_);
416  if (element_type.is_numeric()) {
417  const WasmDataSegment& data_segment =
418  module_->data_segments[segment_imm.index];
419  uint32_t length_in_bytes =
420  length * array_imm.array_type->element_type().value_kind_size();
421 
422  if (!base::IsInBounds<uint32_t>(offset, length_in_bytes,
423  data_segment.source.length())) {
424  error_ = MessageTemplate::kWasmTrapDataSegmentOutOfBounds;
425  return;
426  }
427 
428  Address source =
429  data->data_segment_starts()->get(segment_imm.index) + offset;
430  DirectHandle<WasmArray> array_value =
431  isolate_->factory()->NewWasmArrayFromMemory(length, rtt, element_type,
432  source);
433  result->runtime_value = WasmValue(array_value, result_type);
434  } else {
435  const wasm::WasmElemSegment* elem_segment =
436  &decoder->module_->elem_segments[segment_imm.index];
437  // A constant expression should not observe if a passive segment is dropped.
438  // However, it should consider active and declarative segments as empty.
439  if (!base::IsInBounds<size_t>(
440  offset, length,
441  elem_segment->status == WasmElemSegment::kStatusPassive
442  ? elem_segment->element_count
443  : 0)) {
444  error_ = MessageTemplate::kWasmTrapElementSegmentOutOfBounds;
445  return;
446  }
447 
448  DirectHandle<Object> array_object =
449  isolate_->factory()->NewWasmArrayFromElementSegment(
450  trusted_instance_data_, shared_trusted_instance_data_,
451  segment_imm.index, offset, length, rtt, element_type);
452  if (IsSmi(*array_object)) {
453  // A smi result stands for an error code.
454  error_ = static_cast<MessageTemplate>(Cast<Smi>(*array_object).value());
455  } else {
456  result->runtime_value = WasmValue(array_object, result_type);
457  }
458  }

461 void ConstantExpressionInterface::RefI31(FullDecoder* decoder,
462  const Value& input, Value* result) {
463  if (!generate_value()) return;
464  Address raw = input.runtime_value.to_i32();
465  // We have to craft the Smi manually because we accept out-of-bounds inputs.
466  // For 32-bit Smi builds, set the topmost bit to sign-extend the second bit.
467  // This way, interpretation in JS (if this value escapes there) will be the
468  // same as i31.get_s.
469  static_assert((SmiValuesAre31Bits() ^ SmiValuesAre32Bits()) == 1);
470  intptr_t shifted;
471  if constexpr (SmiValuesAre31Bits()) {
472  shifted = raw << (kSmiTagSize + kSmiShiftSize);
473  } else {
474  shifted =
475  static_cast<intptr_t>(raw << (kSmiTagSize + kSmiShiftSize + 1)) >> 1;
476  }
477  result->runtime_value =
478  WasmValue(direct_handle(Tagged<Smi>(shifted), isolate_), kWasmRefI31);
479 }
