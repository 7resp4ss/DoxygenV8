29 DirectHandle<AccessorInfo> Accessors::MakeAccessor(
30  Isolate* isolate, DirectHandle<Name> name,
31  AccessorNameGetterCallback getter,
32  AccessorNameBooleanSetterCallback setter) {
33  Factory* factory = isolate->factory();
34  name = factory->InternalizeName(name);
35  DirectHandle<AccessorInfo> info = factory->NewAccessorInfo();
36  {
37  DisallowGarbageCollection no_gc;
38  Tagged<AccessorInfo> raw = *info;
39  raw->set_is_sloppy(false);
40  raw->set_replace_on_access(false);
41  raw->set_getter_side_effect_type(SideEffectType::kHasSideEffect);
42  raw->set_setter_side_effect_type(SideEffectType::kHasSideEffect);
43  raw->set_name(*name);
44  raw->set_getter(isolate, reinterpret_cast<Address>(getter));
45  if (setter == nullptr) setter = &ReconfigureToDataProperty;
46  raw->set_setter(isolate, reinterpret_cast<Address>(setter));
47  }
48  return info;
49 }

62 // Returns true for properties that are accessors to object fields.
63 // If true, *object_offset contains offset of object field.
64 bool Accessors::IsJSObjectFieldAccessor(Isolate* isolate, DirectHandle<Map> map,
65  DirectHandle<Name> name,
66  FieldIndex* index) {
67  if (map->is_dictionary_map()) {
68  // There are not descriptors in a dictionary mode map.
69  return false;
70  }
71 
72  switch (map->instance_type()) {
73  case JS_ARRAY_TYPE:
74  return CheckForName(isolate, name, isolate->factory()->length_string(),
75  JSArray::kLengthOffset, FieldIndex::kTagged, index);
76  default:
77  if (map->instance_type() < FIRST_NONSTRING_TYPE) {
78  return CheckForName(isolate, name, isolate->factory()->length_string(),
79  offsetof(String, length_), FieldIndex::kWord32,
80  index);
81  }
82 
83  return false;
84  }

87 V8_WARN_UNUSED_RESULT MaybeDirectHandle<Object>
88 Accessors::ReplaceAccessorWithDataProperty(Isolate* isolate,
89  DirectHandle<JSAny> receiver,
90  DirectHandle<JSObject> holder,
91  DirectHandle<Name> name,
92  DirectHandle<Object> value) {
93  LookupIterator it(isolate, receiver, PropertyKey(isolate, name), holder,
94  LookupIterator::OWN_SKIP_INTERCEPTOR);
95  // Skip any access checks we might hit. This accessor should never hit in a
96  // situation where the caller does not have access.
97  while (it.state() == LookupIterator::ACCESS_CHECK) {
98  CHECK(it.HasAccess());
99  it.Next();
100  }
101  DCHECK(holder.is_identical_to(it.GetHolder<JSObject>()));
102  CHECK_EQ(LookupIterator::ACCESSOR, it.state());
103  it.ReconfigureDataProperty(value, it.property_attributes());
104  return value;
105 }

169 void Accessors::ArrayLengthSetter(
170  v8::Local<v8::Name> name, v8::Local<v8::Value> val,
171  const v8::PropertyCallbackInfo<v8::Boolean>& info) {
172  i::Isolate* isolate = reinterpret_cast<i::Isolate*>(info.GetIsolate());
173  RCS_SCOPE(isolate, RuntimeCallCounterId::kArrayLengthSetter);
174  HandleScope scope(isolate);
175 
176  DCHECK(Object::SameValue(*Utils::OpenDirectHandle(*name),
177  ReadOnlyRoots(isolate).length_string()));
178 
179  DirectHandle<JSReceiver> object = Utils::OpenDirectHandle(*info.Holder());
180  DirectHandle<JSArray> array = Cast<JSArray>(object);
181  DirectHandle<Object> length_obj = Utils::OpenDirectHandle(*val);
182 
183  bool was_readonly = JSArray::HasReadOnlyLength(array);
184 
185  uint32_t length = 0;
186  if (!JSArray::AnythingToArrayLength(isolate, length_obj, &length)) {
187  return;
188  }
189 
190  if (!was_readonly && V8_UNLIKELY(JSArray::HasReadOnlyLength(array))) {
191  // AnythingToArrayLength() may have called setter re-entrantly and modified
192  // its property descriptor. Don't perform this check if "length" was
193  // previously readonly, as this may have been called during
194  // DefineOwnPropertyIgnoreAttributes().
195  if (length == Object::NumberValue(array->length())) {
196  info.GetReturnValue().Set(true);
197  } else if (info.ShouldThrowOnError()) {
198  Factory* factory = isolate->factory();
199  isolate->Throw(
200  *factory->NewTypeError(MessageTemplate::kStrictReadOnlyProperty,
201  Utils::OpenDirectHandle(*name),
202  i::Object::TypeOf(isolate, object), object));
203  } else {
204  info.GetReturnValue().Set(false);
205  }
206  return;
207  }
208 
209  if (JSArray::SetLength(isolate, array, length).IsNothing()) {
210  // TODO(victorgomes): AccessorNameBooleanSetterCallback does not handle
211  // exceptions.
212  FATAL("Fatal JavaScript invalid array length %u", length);
213  UNREACHABLE();
214  }
215 
216  uint32_t actual_new_len = 0;
217  CHECK(Object::ToArrayLength(array->length(), &actual_new_len));
218  // Fail if there were non-deletable elements.
219  if (actual_new_len != length) {
220  if (info.ShouldThrowOnError()) {
221  Factory* factory = isolate->factory();
222  isolate->Throw(*factory->NewTypeError(
223  MessageTemplate::kStrictDeleteProperty,
224  factory->NewNumberFromUint(actual_new_len - 1), array));
225  } else {
226  info.GetReturnValue().Set(false);
227  }
228  } else {
229  info.GetReturnValue().Set(true);
230  }
231 }

307 static DirectHandle<Object> GetFunctionPrototype(
308  Isolate* isolate, DirectHandle<JSFunction> function) {
309  if (!function->has_prototype()) {
310  // We lazily allocate .prototype for functions, which confuses debug
311  // evaluate which assumes we can write to temporary objects we allocated
312  // during evaluation. We err on the side of caution here and prevent the
313  // newly allocated prototype from going into the temporary objects set,
314  // which means writes to it will be considered a side effect.
315  DisableTemporaryObjectTracking no_temp_tracking(isolate->debug());
316  DirectHandle<JSObject> proto =
317  isolate->factory()->NewFunctionPrototype(function);
318  JSFunction::SetPrototype(isolate, function, proto);
319  }
320  return DirectHandle<Object>(function->prototype(), isolate);
321 }

398 Handle<JSObject> ArgumentsFromDeoptInfo(JavaScriptFrame* frame,
399  int inlined_frame_index) {
400  Isolate* isolate = frame->isolate();
401  Factory* factory = isolate->factory();
402 
403  TranslatedState translated_values(frame);
404  translated_values.Prepare(frame->fp());
405 
406  int argument_count = 0;
407  TranslatedFrame* translated_frame =
408  translated_values.GetArgumentsInfoFromJSFrameIndex(inlined_frame_index,
409  &argument_count);
410  TranslatedFrame::iterator iter = translated_frame->begin();
411 
412  // Materialize the function.
413  bool should_deoptimize = iter->IsMaterializedObject();
414  DirectHandle<JSFunction> function = Cast<JSFunction>(iter->GetValue());
415  iter++;
416 
417  // Skip the receiver.
418  iter++;
419  argument_count--;
420 
421  Handle<JSObject> arguments =
422  factory->NewArgumentsObject(function, argument_count);
423  DirectHandle<FixedArray> array = factory->NewFixedArray(argument_count);
424  for (int i = 0; i < argument_count; ++i) {
425  // If we materialize any object, we should deoptimize the frame because we
426  // might alias an object that was eliminated by escape analysis.
427  should_deoptimize = should_deoptimize || iter->IsMaterializedObject();
428  DirectHandle<Object> value = iter->GetValue();
429  array->set(i, *value);
430  iter++;
431  }
432  arguments->set_elements(*array);
433 
434  if (should_deoptimize) {
435  translated_values.StoreMaterializedValuesAndDeopt(frame);
436  }
437 
438  // Return the freshly allocated arguments object.
439  return arguments;
440 }

442 int FindFunctionInFrame(JavaScriptFrame* frame,
443  DirectHandle<JSFunction> function) {
444  FrameSummaries summaries = frame->Summarize();
445  for (int i = summaries.size(); i != 0; i--) {
446  if (*summaries.frames[i - 1].AsJavaScript().function() == *function) {
447  return static_cast<int>(i) - 1;
448  }
449  }
450  return -1;
451 }

453 Handle<JSObject> GetFrameArguments(Isolate* isolate,
454  JavaScriptStackFrameIterator* it,
455  int function_index) {
456  JavaScriptFrame* frame = it->frame();
457 
458  if (function_index > 0) {
459  // The function in question was inlined. Inlined functions have the
460  // correct number of arguments and no allocated arguments object, so
461  // we can construct a fresh one by interpreting the function's
462  // deoptimization input data.
463  return ArgumentsFromDeoptInfo(frame, function_index);
464  }
465 
466  // Construct an arguments object mirror for the right frame and the underlying
467  // function.
468  const int length = frame->GetActualArgumentCount();
469  DirectHandle<JSFunction> function(frame->function(), isolate);
470  Handle<JSObject> arguments =
471  isolate->factory()->NewArgumentsObject(function, length);
472  DirectHandle<FixedArray> array = isolate->factory()->NewFixedArray(length);
473 
474  // Copy the parameters to the arguments object.
475  DCHECK(array->length() == length);
476  for (int i = 0; i < length; i++) {
477  Tagged<Object> value = frame->GetParameter(i);
478  if (IsTheHole(value, isolate)) {
479  // Generators currently use holes as dummy arguments when resuming. We
480  // must not leak those.
481  DCHECK(IsResumableFunction(function->shared()->kind()));
482  value = ReadOnlyRoots(isolate).undefined_value();
483  }
484  array->set(i, value);
485  }
486  arguments->set_elements(*array);
487 
488  // For optimized functions, the frame arguments may be outdated, so we should
489  // update them with the deopt info, while keeping the length and extra
490  // arguments from the actual frame.
491  if (CodeKindCanDeoptimize(frame->LookupCode()->kind()) && length > 0) {
492  DirectHandle<JSObject> arguments_from_deopt_info =
493  ArgumentsFromDeoptInfo(frame, function_index);
494  DirectHandle<FixedArray> elements_from_deopt_info(
495  Cast<FixedArray>(arguments_from_deopt_info->elements()), isolate);
496  int common_length = std::min(length, elements_from_deopt_info->length());
497  for (int i = 0; i < common_length; i++) {
498  array->set(i, elements_from_deopt_info->get(i));
499  }
500  }
501 
502  // Return the freshly allocated arguments object.
503  return arguments;
504 }

508 Handle<JSObject> Accessors::FunctionGetArguments(JavaScriptFrame* frame,
509  int inlined_jsframe_index) {
510  Isolate* isolate = frame->isolate();
511  Address requested_frame_fp = frame->fp();
512  // Forward a frame iterator to the requested frame. This is needed because we
513  // potentially need for advance it to the inlined arguments frame later.
514  for (JavaScriptStackFrameIterator it(isolate); !it.done(); it.Advance()) {
515  if (it.frame()->fp() != requested_frame_fp) continue;
516  return GetFrameArguments(isolate, &it, inlined_jsframe_index);
517  }
518  UNREACHABLE(); // Requested frame not found.
519 }

521 // static
522 DirectHandle<Object> Accessors::GetLegacyFunctionArguments(
523  Isolate* isolate, DirectHandle<JSFunction> function) {
524  DirectHandle<Object> result = isolate->factory()->null_value();
525  if (!function->shared()->native()) {
526  // Find the top invocation of the function by traversing frames.
527  for (JavaScriptStackFrameIterator it(isolate); !it.done(); it.Advance()) {
528  JavaScriptFrame* frame = it.frame();
529  int function_index = FindFunctionInFrame(frame, function);

531  result = GetFrameArguments(isolate, &it, function_index);
532  break;
533  }
534  }
535  }
536  return result;

569 class FrameFunctionIterator {
570  public:
571  explicit FrameFunctionIterator(Isolate* isolate)
572  : isolate_(isolate), frame_iterator_(isolate), inlined_frame_index_(-1) {
573  GetFrames();
574  }
575 
576  // Iterate through functions until the first occurrence of 'function'.
577  // Returns true if one is found, and false if the iterator ends before.
578  bool Find(DirectHandle<JSFunction> function) {
579  do {
580  if (!next().ToHandle(&function_)) return false;
581  } while (!function_.is_identical_to(function));
582  return true;
583  }
584 
585  // Iterate through functions, at least one step, until the first candidate
586  // is found that is not toplevel and either user-provided JavaScript or
587  // "native" (not defined in user-provided scripts, but directly exposed).
588  // Returns true if one is found, and false if the iterator ends before.
589  bool FindNextNonTopLevelNativeOrUserJavaScript() {
590  do {
591  if (!next().ToHandle(&function_)) return false;
592  } while (function_->shared()->is_toplevel() ||
593  (!function_->shared()->native() &&
594  !function_->shared()->IsUserJavaScript()));
595  return true;
596  }
597 
598  // In case of inlined frames the function could have been materialized from
599  // deoptimization information. If that is the case we need to make sure that
600  // subsequent call will see the same function, since we are about to hand out
601  // the value to JavaScript. Make sure to store the materialized value and
602  // trigger a deoptimization of the underlying frame.
603  DirectHandle<JSFunction> MaterializeFunction() {
604  if (inlined_frame_index_ == 0) return function_;
605 
606  JavaScriptFrame* frame = frame_iterator_.frame();
607  TranslatedState translated_values(frame);
608  translated_values.Prepare(frame->fp());
609 
610  TranslatedFrame* translated_frame =
611  translated_values.GetFrameFromJSFrameIndex(inlined_frame_index_);
612  TranslatedFrame::iterator iter = translated_frame->begin();
613 
614  // First value is the function.
615  bool should_deoptimize = iter->IsMaterializedObject();
616  DirectHandle<Object> value = iter->GetValue();
617  if (should_deoptimize) {
618  translated_values.StoreMaterializedValuesAndDeopt(frame);
619  }
620 
621  return Cast<JSFunction>(value);
622  }
623 
624  private:
625  MaybeDirectHandle<JSFunction> next() {
626  while (true) {
627  if (inlined_frame_index_ <= 0) {
628  if (!frame_iterator_.done()) {
629  frame_iterator_.Advance();
630  summaries_.frames.clear();
631  inlined_frame_index_ = -1;
632  GetFrames();
633  }
634  if (inlined_frame_index_ == -1) return {};
635  }
636 
637  --inlined_frame_index_;
638  DirectHandle<JSFunction> next_function =
639  summaries_.frames[inlined_frame_index_].AsJavaScript().function();
640  // Skip functions from other origins.
641  if (!AllowAccessToFunction(isolate_->context(), *next_function)) continue;
642  return next_function;
643  }
644  }
645  void GetFrames() {
646  DCHECK_EQ(-1, inlined_frame_index_);
647  if (frame_iterator_.done()) return;
648  JavaScriptFrame* frame = frame_iterator_.frame();
649  summaries_ = frame->Summarize();
650  inlined_frame_index_ = static_cast<int>(summaries_.size());
651  DCHECK_LT(0, inlined_frame_index_);
652  }
653  Isolate* isolate_;
654  DirectHandle<JSFunction> function_;
655  JavaScriptStackFrameIterator frame_iterator_;
656  FrameSummaries summaries_;
657  int inlined_frame_index_;
658 };

661 MaybeDirectHandle<JSFunction> FindCaller(Isolate* isolate,
662  DirectHandle<JSFunction> function) {
663  FrameFunctionIterator it(isolate);
664  if (function->shared()->native()) {
665  return {};
666  }
667  // Find the function from the frames. Return null in case no frame
668  // corresponding to the given function was found.
669  if (!it.Find(function)) {
670  return {};
671  }
672  // Find previously called non-toplevel function that is also a user-land
673  // JavaScript function (or the entry point into native JavaScript builtins
674  // in case such a builtin was the caller).
675  if (!it.FindNextNonTopLevelNativeOrUserJavaScript()) {
676  return {};
677  }
678 
679  // Materialize the function that the iterator is currently sitting on. Note
680  // that this might trigger deoptimization in case the function was actually
681  // materialized. Identity of the function must be preserved because we are
682  // going to return it to JavaScript after this point.
683  DirectHandle<JSFunction> caller = it.MaterializeFunction();
684 
685  // Censor if the caller is not a sloppy mode function.
686  // Change from ES5, which used to throw, see:
687  // https://bugs.ecmascript.org/show_bug.cgi?id=310
688  if (is_strict(caller->shared()->language_mode())) {
689  return {};
690  }
691  // Don't return caller from another security context.
692  if (!AllowAccessToFunction(isolate->context(), *caller)) {
693  return {};
694  }
695  return caller;
696 }

702  DirectHandle<Object> result;
