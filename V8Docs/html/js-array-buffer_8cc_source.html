17 // ES#sec-canonicalnumericindexstring
18 // Returns true if the lookup_key represents a valid index string.
19 bool CanonicalNumericIndexString(Isolate* isolate,
20  const PropertyKey& lookup_key,
21  bool* is_minus_zero) {
22  // 1. Assert: Type(argument) is String.
23  DCHECK(lookup_key.is_element() || IsString(*lookup_key.name()));
24  *is_minus_zero = false;
25  if (lookup_key.is_element()) return true;
26 
27  DirectHandle<String> key = Cast<String>(lookup_key.name());
28 
29  // 3. Let n be ! ToNumber(argument).
30  DirectHandle<Object> result = String::ToNumber(isolate, key);
31  if (IsMinusZero(*result)) {
32  // 2. If argument is "-0", return -0ùîΩ.
33  // We are not performing SaveValue check for -0 because it'll be rejected
34  // anyway.
35  *is_minus_zero = true;
36  } else {
37  // 4. If SameValue(! ToString(n), argument) is false, return undefined.
38  DirectHandle<String> str =
39  Object::ToString(isolate, result).ToHandleChecked();
40  // Avoid treating strings like "2E1" and "20" as the same key.
41  if (!Object::SameValue(*str, *key)) return false;
42  }
43  return true;
44 }

47 void JSArrayBuffer::Setup(SharedFlag shared, ResizableFlag resizable,
48  std::shared_ptr<BackingStore> backing_store,
49  Isolate* isolate) {
50  auto finish_setup = [shared, isolate]() {
51  // Count usage may lead to a blink allocation, through the callback, which
52  // may trigger a GC. It is important to delay this, until the array buffer
53  // is properly initialized.
54  if (shared == SharedFlag::kShared) {
55  isolate->CountUsage(
56  v8::Isolate::UseCounterFeature::kSharedArrayBufferConstructed);
57  }
58  };
59  clear_padding();
60  init_extension();
61  set_detach_key(ReadOnlyRoots(isolate).undefined_value());
62  set_bit_field(0);
63  set_is_shared(shared == SharedFlag::kShared);
64  set_is_resizable_by_js(resizable == ResizableFlag::kResizable);
65  set_is_detachable(shared != SharedFlag::kShared);
66  SetupLazilyInitializedCppHeapPointerField(
67  JSAPIObjectWithEmbedderSlots::kCppHeapWrappableOffset);
68  for (int i = 0; i < v8::ArrayBuffer::kEmbedderFieldCount; i++) {
69  SetEmbedderField(i, Smi::zero());
70  }
71  if (!backing_store) {
72  set_backing_store(isolate, EmptyBackingStoreBuffer());
73  set_byte_length(0);
74  set_max_byte_length(0);
75  finish_setup();
76  return;
77  }
78  // Rest of the code here deals with attaching the BackingStore.
79  DCHECK_EQ(is_shared(), backing_store->is_shared());
80  DCHECK((is_resizable_by_js() == backing_store->is_resizable_by_js()) ||
81  (backing_store->is_wasm_memory() && is_shared()));
82  DCHECK_IMPLIES(
83  !backing_store->is_wasm_memory() && !backing_store->is_resizable_by_js(),
84  backing_store->byte_length() == backing_store->max_byte_length());
85 
86  void* backing_store_buffer = backing_store->buffer_start();
87  // Wasm memory always needs a backing store; this is guaranteed by reserving
88  // at least one page for the BackingStore (so {IsEmpty()} is always false).
89  DCHECK_IMPLIES(backing_store->is_wasm_memory(), !backing_store->IsEmpty());
90  // Non-empty backing stores must start at a non-null pointer.
91  DCHECK_IMPLIES(backing_store_buffer == EmptyBackingStoreBuffer(),
92  backing_store->IsEmpty());
93  // Empty backing stores can be backed by an empty buffer pointer or by an
94  // externally provided pointer: Either is acceptable. However, the pointer
95  // must always point into the sandbox, so nullptr is not acceptable if the
96  // sandbox is enabled.
97  DCHECK_IMPLIES(V8_ENABLE_SANDBOX_BOOL, backing_store_buffer != nullptr);
98  set_backing_store(isolate, backing_store_buffer);
99 
100  // GSABs need to read their byte_length from the BackingStore. Maintain the
101  // invariant that their byte_length field is always 0.
102  auto byte_len =
103  (is_shared() && is_resizable_by_js()) ? 0 : backing_store->byte_length();
104  CHECK_LE(backing_store->byte_length(), kMaxByteLength);
105  set_byte_length(byte_len);
106 
107  // For Wasm memories, it is possible for the backing store maximum to be
108  // different from the JSArrayBuffer maximum. The maximum pages allowed on a
109  // Wasm memory are tracked on the Wasm memory object, and not the
110  // JSArrayBuffer associated with it.
111  auto max_byte_len = is_resizable_by_js() ? backing_store->max_byte_length()
112  : backing_store->byte_length();
113  set_max_byte_length(max_byte_len);
114 
115  if (backing_store->is_wasm_memory()) {
116  set_is_detachable(false);
117  }
118 
119  CreateExtension(isolate, std::move(backing_store));
120  finish_setup();
121 }

123 Maybe<bool> JSArrayBuffer::Detach(DirectHandle<JSArrayBuffer> buffer,
124  bool force_for_wasm_memory,
125  DirectHandle<Object> maybe_key) {
126  Isolate* const isolate = Isolate::Current();
127 
128  DirectHandle<Object> detach_key(buffer->detach_key(), isolate);
129 
130  bool key_mismatch = false;
131 
132  if (!IsUndefined(*detach_key, isolate)) {
133  key_mismatch =
134  maybe_key.is_null() || !Object::StrictEquals(*maybe_key, *detach_key);
135  } else {
136  // Detach key is undefined; allow not passing maybe_key but disallow passing
137  // something else than undefined.
138  key_mismatch =
139  !maybe_key.is_null() && !Object::StrictEquals(*maybe_key, *detach_key);
140  }
141  if (key_mismatch) {
142  THROW_NEW_ERROR_RETURN_VALUE(
143  isolate,
144  NewTypeError(MessageTemplate::kArrayBufferDetachKeyDoesntMatch),
145  Nothing<bool>());
146  }
147 
148  if (buffer->was_detached()) return Just(true);
149 
150  if (force_for_wasm_memory) {
151  // Skip the is_detachable() check.
152  } else if (!buffer->is_detachable()) {
153  // Not detachable, do nothing.
154  return Just(true);
155  }
156 
157  buffer->DetachInternal(force_for_wasm_memory, isolate);
158  return Just(true);
159 }

161 void JSArrayBuffer::DetachInternal(bool force_for_wasm_memory,
162  Isolate* isolate) {
163  ArrayBufferExtension* extension = this->extension();
164 
165  if (extension) {
166  DisallowGarbageCollection disallow_gc;
167  isolate->heap()->DetachArrayBufferExtension(extension);
168  std::shared_ptr<BackingStore> backing_store = RemoveExtension();
169  CHECK_IMPLIES(force_for_wasm_memory, backing_store->is_wasm_memory());
170  }
171 
172  if (Protectors::IsArrayBufferDetachingIntact(isolate)) {
173  Protectors::InvalidateArrayBufferDetaching(isolate);
174  }
175 
176  DCHECK(!is_shared());
177  set_backing_store(isolate, EmptyBackingStoreBuffer());
178  set_byte_length(0);
179  set_was_detached(true);
180 }

248 ArrayBufferExtension* JSArrayBuffer::CreateExtension(
249  Isolate* isolate, std::shared_ptr<BackingStore> backing_store) {
250  // `Heap::InYoungGeneration` during full GC with sticky markbits is generally
251  // inaccurate. However, a full GC will sweep both lists and promote all to
252  // old, so it doesn't matter which list initially holds the extension in this
253  // case.
254  const auto age =
255  HeapLayout::InYoungGeneration(UncheckedCast<JSArrayBuffer>(*this))
256  ? ArrayBufferExtension::Age::kYoung
257  : ArrayBufferExtension::Age::kOld;
258  ArrayBufferExtension* extension =
259  new ArrayBufferExtension(std::move(backing_store), age);
260  set_extension(extension);
261  isolate->heap()->AppendArrayBufferExtension(extension);
262  return extension;
263 }

297 Handle<JSArrayBuffer> JSTypedArray::GetBuffer(Isolate* isolate) {
298  DirectHandle<JSTypedArray> self(*this, isolate);
299  DCHECK(IsTypedArrayOrRabGsabTypedArrayElementsKind(self->GetElementsKind()));
300  Handle<JSArrayBuffer> array_buffer(Cast<JSArrayBuffer>(self->buffer()),
301  isolate);
302  if (!is_on_heap()) {
303  // Already is off heap, so return the existing buffer.
304  return array_buffer;
305  }
306  DCHECK(!array_buffer->is_resizable_by_js());
307 
308  // The existing array buffer should be empty.
309  DCHECK(array_buffer->IsEmpty());
310 
311  // Allocate a new backing store and attach it to the existing array buffer.
312  size_t byte_length = self->byte_length();
313  auto backing_store =
314  BackingStore::Allocate(isolate, byte_length, SharedFlag::kNotShared,
315  InitializedFlag::kUninitialized);
316 
317  if (!backing_store) {
318  isolate->heap()->FatalProcessOutOfMemory("JSTypedArray::GetBuffer");
319  }
320 
321  // Copy the elements into the backing store of the array buffer.
322  if (byte_length > 0) {
323  memcpy(backing_store->buffer_start(), self->DataPtr(), byte_length);
324  }
325 
326  // Attach the backing store to the array buffer.
327  array_buffer->Setup(SharedFlag::kNotShared, ResizableFlag::kNotResizable,
328  std::move(backing_store), isolate);
329 
330  // Clear the elements of the typed array.
331  self->set_elements(ReadOnlyRoots(isolate).empty_byte_array());
332  self->SetOffHeapDataPtr(isolate, array_buffer->backing_store(), 0);
333  DCHECK(!self->is_on_heap());
334 
335  return array_buffer;
336 }

339 // static
340 Maybe<bool> JSTypedArray::DefineOwnProperty(Isolate* isolate,
341  DirectHandle<JSTypedArray> o,
342  DirectHandle<Object> key,
343  PropertyDescriptor* desc,
344  Maybe<ShouldThrow> should_throw) {
345  DCHECK(IsName(*key) || IsNumber(*key));
346  // 1. If Type(P) is String, then
347  PropertyKey lookup_key(isolate, key);
348  if (lookup_key.is_element() || IsSmi(*key) || IsString(*key)) {
349  // 1a. Let numericIndex be ! CanonicalNumericIndexString(P)
350  // 1b. If numericIndex is not undefined, then
351  bool is_minus_zero = false;
352  if (IsSmi(*key) || // Smi keys are definitely canonical
353  CanonicalNumericIndexString(isolate, lookup_key, &is_minus_zero)) {
354  // 1b i. If IsValidIntegerIndex(O, numericIndex) is false, return false.
355 
356  // IsValidIntegerIndex:
357  size_t index = lookup_key.index();
358  bool out_of_bounds = false;
359  size_t length = o->GetLengthOrOutOfBounds(out_of_bounds);
360  if (o->WasDetached() || out_of_bounds || index >= length) {
361  RETURN_FAILURE(isolate, GetShouldThrow(isolate, should_throw),
362  NewTypeError(MessageTemplate::kInvalidTypedArrayIndex));
363  }
364  if (!lookup_key.is_element() || is_minus_zero) {
365  RETURN_FAILURE(isolate, GetShouldThrow(isolate, should_throw),
366  NewTypeError(MessageTemplate::kInvalidTypedArrayIndex));
367  }
368 
369  // 1b ii. If Desc has a [[Configurable]] field and if
370  // Desc.[[Configurable]] is false, return false.
371  // 1b iii. If Desc has an [[Enumerable]] field and if Desc.[[Enumerable]]
372  // is false, return false.
373  // 1b iv. If IsAccessorDescriptor(Desc) is true, return false.
374  // 1b v. If Desc has a [[Writable]] field and if Desc.[[Writable]] is
375  // false, return false.
376 
377  if (PropertyDescriptor::IsAccessorDescriptor(desc)) {
378  RETURN_FAILURE(isolate, GetShouldThrow(isolate, should_throw),
379  NewTypeError(MessageTemplate::kRedefineDisallowed, key));
380  }
381 
382  if ((desc->has_configurable() && !desc->configurable()) ||
383  (desc->has_enumerable() && !desc->enumerable()) ||
384  (desc->has_writable() && !desc->writable())) {
385  RETURN_FAILURE(isolate, GetShouldThrow(isolate, should_throw),
386  NewTypeError(MessageTemplate::kRedefineDisallowed, key));
387  }
388 
389  // 1b vi. If Desc has a [[Value]] field, perform
390  // ? IntegerIndexedElementSet(O, numericIndex, Desc.[[Value]]).
391  if (desc->has_value()) {
392  if (!desc->has_configurable()) desc->set_configurable(true);
393  if (!desc->has_enumerable()) desc->set_enumerable(true);
394  if (!desc->has_writable()) desc->set_writable(true);
395  DirectHandle<Object> value = desc->value();
396  LookupIterator it(isolate, o, index, LookupIterator::OWN);
397  RETURN_ON_EXCEPTION_VALUE(
398  isolate,
399  DefineOwnPropertyIgnoreAttributes(&it, value, desc->ToAttributes()),
400  Nothing<bool>());
401  }
402  // 1b vii. Return true.
403  return Just(true);
404  }
405  }
406  // 4. Return ! OrdinaryDefineOwnProperty(O, P, Desc).
407  return OrdinaryDefineOwnProperty(isolate, o, lookup_key, desc, should_throw);
408 }

458 size_t JSTypedArray::GetVariableByteLengthOrOutOfBounds(
459  bool& out_of_bounds) const {
460  DCHECK(!WasDetached());
461  size_t own_byte_offset = byte_offset();
462  if (is_length_tracking()) {
463  size_t own_element_size = element_size();
464  if (is_backed_by_rab()) {
465  size_t buffer_byte_length = buffer()->byte_length();
466  if (own_byte_offset > buffer_byte_length) {
467  out_of_bounds = true;
468  return 0;
469  }
470  // Round down to the nearest multiple of element size.
471  return RoundDown(buffer_byte_length - own_byte_offset, own_element_size);
472  }
473  // GSAB-backed TypedArrays can't be out of bounds.
474  size_t buffer_byte_length =
475  buffer()->GetBackingStore()->byte_length(std::memory_order_seq_cst);
476  SBXCHECK(own_byte_offset <= buffer_byte_length);
477  // Round down to the nearest multiple of element size.
478  return RoundDown(buffer_byte_length - own_byte_offset, own_element_size);
479  }
480  DCHECK(is_backed_by_rab());
481  size_t own_byte_length = byte_length();
482  size_t buffer_byte_length = buffer()->byte_length();
483  if (own_byte_length > buffer_byte_length ||
484  own_byte_offset > buffer_byte_length - own_byte_length) {
485  out_of_bounds = true;
486  return 0;
487  }
488  return own_byte_length;
489 }
