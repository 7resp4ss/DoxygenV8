98 // Takes the scope of a parsed script, a function and a break location
99 // inside the function. The result is the innermost lexical scope around
100 // the break point, which serves as the starting point of the ScopeIterator.
101 // And the scope of the function that was passed in (called closure scope).
102 //
103 // The start scope is guaranteed to be either the closure scope itself,
104 // or a child of the closure scope.
105 class ScopeChainRetriever {
106  public:
107  ScopeChainRetriever(DeclarationScope* scope,
108  DirectHandle<JSFunction> function, int position)
109  : scope_(scope),
110  break_scope_start_(function->shared()->StartPosition()),
111  break_scope_end_(function->shared()->EndPosition()),
112  break_scope_type_(function->shared()->scope_info()->scope_type()),
113  position_(position) {
114  DCHECK_NOT_NULL(scope);
115  RetrieveScopes();
116  }
117 
118  DeclarationScope* ClosureScope() { return closure_scope_; }
119  Scope* StartScope() { return start_scope_; }
120 
121  private:
122  DeclarationScope* scope_;
123  const int break_scope_start_;
124  const int break_scope_end_;
125  const ScopeType break_scope_type_;
126  const int position_;
127 
128  DeclarationScope* closure_scope_ = nullptr;
129  Scope* start_scope_ = nullptr;
130 
131  void RetrieveScopes() {
132  // 1. Find the closure scope with a DFS.
133  RetrieveClosureScope(scope_);
134  DCHECK_NOT_NULL(closure_scope_);
135 
136  // 2. Starting from the closure scope search inwards. Given that V8's scope
137  // tree doesn't guarantee that siblings don't overlap, we look at all
138  // scopes and pick the one with the tightest bounds around `position_`.
139  start_scope_ = closure_scope_;
140  RetrieveStartScope(closure_scope_);
141  DCHECK_NOT_NULL(start_scope_);
142  }
143 
144  bool RetrieveClosureScope(Scope* scope) {
145  // The closure scope is the scope that matches exactly the function we
146  // paused in.
147  // Note that comparing the position alone is not enough and we also need to
148  // match the scope type. E.g. class member initializer have the exact same
149  // scope positions as their class scope.
150  if (break_scope_type_ == scope->scope_type() &&
151  break_scope_start_ == scope->start_position() &&
152  break_scope_end_ == scope->end_position()) {
153  closure_scope_ = scope->AsDeclarationScope();
154  return true;
155  }
156 
157  for (Scope* inner_scope = scope->inner_scope(); inner_scope != nullptr;
158  inner_scope = inner_scope->sibling()) {
159  if (RetrieveClosureScope(inner_scope)) return true;
160  }
161  return false;
162  }
163 
164  void RetrieveStartScope(Scope* scope) {
165  const int start = scope->start_position();
166  const int end = scope->end_position();
167 
168  // Update start_scope_ if scope contains `position_` and scope is a tighter
169  // fit than the currently set start_scope_.
170  // Generators have the same source position so we also check for equality.
171  if (ContainsPosition(scope) && start >= start_scope_->start_position() &&
172  end <= start_scope_->end_position()) {
173  start_scope_ = scope;
174  }
175 
176  for (Scope* inner_scope = scope->inner_scope(); inner_scope != nullptr;
177  inner_scope = inner_scope->sibling()) {
178  RetrieveStartScope(inner_scope);
179  }
180  }
181 
182  bool ContainsPosition(Scope* scope) {
183  const int start = scope->start_position();
184  const int end = scope->end_position();
185  // In case the closure_scope_ hasn't been found yet, we are less strict
186  // about recursing downwards. This might be the case for nested arrow
187  // functions that have the same end position.
188  const bool position_fits_end =
189  closure_scope_ ? position_ < end : position_ <= end;
190  // While we're evaluating a class, the calling function will have a class
191  // context on the stack with a range that starts at Token::kClass, and the
192  // source position will also point to Token::kClass. To identify the
193  // matching scope we include start in the accepted range for class scopes.
194  //
195  // Similarly "with" scopes can already have bytecodes where the source
196  // position points to the closing parenthesis with the "with" context
197  // already pushed.
198  const bool position_fits_start =
199  scope->is_class_scope() || scope->is_with_scope() ? start <= position_
200  : start < position_;
201  return position_fits_start && position_fits_end;
202  }
203 };

205 // Walks a ScopeInfo outwards until it finds a EVAL scope.
206 MaybeDirectHandle<ScopeInfo> FindEvalScope(Isolate* isolate,
207  Tagged<ScopeInfo> start_scope) {
208  Tagged<ScopeInfo> scope = start_scope;
209  while (scope->scope_type() != ScopeType::EVAL_SCOPE &&
210  scope->HasOuterScopeInfo()) {
211  scope = scope->OuterScopeInfo();
212  }
213 
214  return scope->scope_type() == ScopeType::EVAL_SCOPE
215  ? MaybeHandle<ScopeInfo>(scope, isolate)
216  : kNullMaybeHandle;
217 }

221 void ScopeIterator::TryParseAndRetrieveScopes(ReparseStrategy strategy) {
222  // Catch the case when the debugger stops in an internal function.
223  DirectHandle<SharedFunctionInfo> shared_info(function_->shared(), isolate_);
224  DirectHandle<ScopeInfo> scope_info(shared_info->scope_info(), isolate_);
225  if (IsUndefined(shared_info->script(), isolate_)) {
226  current_scope_ = closure_scope_ = nullptr;
227  context_ = handle(function_->context(), isolate_);
228  function_ = Handle<JSFunction>();
229  return;
230  }
231 
232  bool ignore_nested_scopes = false;
233  if (shared_info->HasBreakInfo(isolate_) && frame_inspector_ != nullptr) {
234  // The source position at return is always the end of the function,
235  // which is not consistent with the current scope chain. Therefore all
236  // nested with, catch and block contexts are skipped, and we can only
237  // inspect the function scope.
238  // This can only happen if we set a break point inside right before the
239  // return, which requires a debug info to be available.
240  Handle<DebugInfo> debug_info(shared_info->GetDebugInfo(isolate_), isolate_);
241 
242  // Find the break point where execution has stopped.
243  BreakLocation location = BreakLocation::FromFrame(debug_info, GetFrame());
244 
245  ignore_nested_scopes = location.IsReturn();
246  }
247 
248  if (strategy == ReparseStrategy::kScriptIfNeeded) {
249  Tagged<Object> maybe_block_list =
250  isolate_->LocalsBlockListCacheGet(scope_info);
251  calculate_blocklists_ = IsTheHole(maybe_block_list);
252  strategy = calculate_blocklists_ ? ReparseStrategy::kScriptIfNeeded
253  : ReparseStrategy::kFunctionLiteral;
254  }
255 
256  // Reparse the code and analyze the scopes.
257  // Depending on the chosen strategy, the whole script or just
258  // the closure is re-parsed for function scopes.
259  DirectHandle<Script> script(Cast<Script>(shared_info->script()), isolate_);
260 
261  // Pick between flags for a single function compilation, or an eager
262  // compilation of the whole script.
263  UnoptimizedCompileFlags flags =
264  (scope_info->scope_type() == FUNCTION_SCOPE &&
265  strategy == ReparseStrategy::kFunctionLiteral)
266  ? UnoptimizedCompileFlags::ForFunctionCompile(isolate_, *shared_info)
267  : UnoptimizedCompileFlags::ForScriptCompile(isolate_, *script)
268  .set_is_eager(true);
269  flags.set_is_reparse(true);
270 
271  MaybeDirectHandle<ScopeInfo> maybe_outer_scope;
272  if (flags.is_toplevel() &&
273  script->compilation_type() == Script::CompilationType::kEval) {
274  // Re-parsing a full eval script requires us to correctly set the outer
275  // language mode and potentially an outer scope info.
276  //
277  // We walk the runtime scope chain and look for an EVAL scope. If we don't
278  // find one, we assume sloppy mode and no outer scope info.
279 
280  DCHECK(flags.is_eval());
281 
282  DirectHandle<ScopeInfo> eval_scope;
283  if (FindEvalScope(isolate_, *scope_info).ToHandle(&eval_scope)) {
284  flags.set_outer_language_mode(eval_scope->language_mode());
285  if (eval_scope->HasOuterScopeInfo()) {
286  maybe_outer_scope =
287  direct_handle(eval_scope->OuterScopeInfo(), isolate_);
288  }
289  } else {
290  DCHECK_EQ(flags.outer_language_mode(), LanguageMode::kSloppy);
291  DCHECK(maybe_outer_scope.is_null());
292  }
293  } else if (scope_info->scope_type() == EVAL_SCOPE || script->is_wrapped()) {
294  flags.set_is_eval(true);
295  if (!IsNativeContext(*context_)) {
296  maybe_outer_scope = direct_handle(context_->scope_info(), isolate_);
297  }
298  // Language mode may be inherited from the eval caller.
299  // Retrieve it from shared function info.
300  flags.set_outer_language_mode(shared_info->language_mode());
301  } else if (scope_info->scope_type() == MODULE_SCOPE) {
302  DCHECK(script->origin_options().IsModule());
303  DCHECK(flags.is_module());
304  } else {
305  DCHECK(scope_info->is_script_scope() ||
306  scope_info->scope_type() == FUNCTION_SCOPE);
307  }
308 
309  UnoptimizedCompileState compile_state;
310 
311  reusable_compile_state_ =
312  std::make_unique<ReusableUnoptimizedCompileState>(isolate_);
313  info_ = std::make_unique<ParseInfo>(isolate_, flags, &compile_state,
314  reusable_compile_state_.get());
315 
316  const bool parse_result =
317  flags.is_toplevel()
318  ? parsing::ParseProgram(info_.get(), script, maybe_outer_scope,
319  isolate_, parsing::ReportStatisticsMode::kNo)
320  : parsing::ParseFunction(info_.get(), shared_info, isolate_,
321  parsing::ReportStatisticsMode::kNo);
322 
323  if (parse_result) {
324  DeclarationScope* literal_scope = info_->literal()->scope();
325 
326  ScopeChainRetriever scope_chain_retriever(literal_scope, function_,
327  GetSourcePosition());
328  start_scope_ = scope_chain_retriever.StartScope();
329  current_scope_ = start_scope_;
330 
331  // In case of a FUNCTION_SCOPE, the ScopeIterator expects
332  // {closure_scope_} to be set to the scope of the function.
333  closure_scope_ = scope_info->scope_type() == FUNCTION_SCOPE
334  ? scope_chain_retriever.ClosureScope()
335  : literal_scope;
336 
337  if (ignore_nested_scopes) {
338  current_scope_ = closure_scope_;
339  start_scope_ = current_scope_;
340  // ignore_nested_scopes is only used for the return-position breakpoint,
341  // so we can safely assume that the closure context for the current
342  // function exists if it needs one.
343  if (closure_scope_->NeedsContext()) {
344  context_ = handle(context_->closure_context(), isolate_);
345  }
346  }
347 
348  MaybeCollectAndStoreLocalBlocklists();
349  UnwrapEvaluationContext();
350  } else {
351  // A failed reparse indicates that the preparser has diverged from the
352  // parser, that the preparse data given to the initial parse was faulty, or
353  // a stack overflow.
354  // TODO(leszeks): This error is pretty unexpected, so we could report the
355  // error in debug mode. Better to not fail in release though, in case it's
356  // just a stack overflow.
357 
358  // Silently fail by presenting an empty context chain.
359  context_ = Handle<Context>();
360  }
361 }

363 void ScopeIterator::UnwrapEvaluationContext() {
364  if (!context_->IsDebugEvaluateContext()) return;
365  Tagged<Context> current = *context_;
366  do {
367  Tagged<Object> wrapped = current->GetNoCell(Context::WRAPPED_CONTEXT_INDEX);
368  if (IsContext(wrapped)) {

371  DCHECK(!current->previous().is_null());

373  }
374  } while (current->IsDebugEvaluateContext());
375  context_ = handle(current, isolate_);

433 bool ScopeIterator::HasContext() const {
434  // In rare cases we pause in a scope that doesn't have its context pushed yet.
435  // E.g. when pausing in for-of loop headers (see https://crbug.com/399002824).
436  //
437  // We can detect this by comparing the scope ID of the parsed scope and the
438  // runtime scope.
439  if (current_scope_ && NeedsContext() &&
440  current_scope_->UniqueIdInScript() !=
441  context_->scope_info()->UniqueIdInScript()) {
442  return false;
443  }
444 
445  return !InInnerScope() || NeedsContext();
446 }

448 bool ScopeIterator::NeedsContext() const {
449  const bool needs_context = current_scope_->NeedsContext();
450 
451  // We try very hard to ensure that a function's context is already
452  // available when we pause right at the beginning of that function.
453  // This can be tricky when we pause via stack check or via
454  // `BreakOnNextFunctionCall`, which happens normally in the middle of frame
455  // construction and we have to "step into" the function first.
456  //
457  // We check this by ensuring that the current context is not the closure
458  // context should the function need one. In that case the function has already
459  // pushed the context and we are good.
460  CHECK_IMPLIES(needs_context && current_scope_ == closure_scope_ &&
461  current_scope_->is_function_scope() && !function_.is_null(),
462  function_->context() != *context_);
463 
464  return needs_context;
465 }

510 void ScopeIterator::Next() {
511  DCHECK(!Done());
512 
513  ScopeType scope_type = Type();
514 
515  if (scope_type == ScopeTypeGlobal) {
516  // The global scope is always the last in the chain.
517  DCHECK(IsNativeContext(*context_));
518  context_ = Handle<Context>();
519  DCHECK(Done());
520  return;
521  }
522 
523  bool leaving_closure = current_scope_ == closure_scope_;
524 
525  if (scope_type == ScopeTypeScript) {
526  DCHECK_IMPLIES(InInnerScope() && !leaving_closure,
527  current_scope_->is_script_scope());
528  seen_script_scope_ = true;
529  if (context_->IsScriptContext()) {
530  context_ = handle(context_->previous(), isolate_);
531  }
532  } else if (!InInnerScope()) {
533  AdvanceContext();
534  } else {
535  DCHECK_NOT_NULL(current_scope_);
536  AdvanceScope();
537 
538  if (leaving_closure) {
539  DCHECK(current_scope_ != closure_scope_);
540  // If the current_scope_ doesn't need a context, we advance the scopes
541  // and collect the blocklist along the way until we find the scope
542  // that should match `context_`.
543  // But only do this if we have complete scope information.
544  while (!NeedsContext() && AdvanceOneScope()) {
545  }
546  }
547  }
548 
549  MaybeCollectAndStoreLocalBlocklists();
550  UnwrapEvaluationContext();
551 
552  DCHECK_IMPLIES(current_scope_ && NeedsAndHasContext(),
553  current_scope_->UniqueIdInScript() ==
554  context_->scope_info()->UniqueIdInScript());
555 
556  if (leaving_closure) function_ = Handle<JSFunction>();
557 }

559 // Return the type of the current scope.
560 ScopeIterator::ScopeType ScopeIterator::Type() const {
561  DCHECK(!Done());
562  if (InInnerScope()) {
563  switch (current_scope_->scope_type()) {
564  case FUNCTION_SCOPE:
565  DCHECK_IMPLIES(NeedsAndHasContext(),
566  context_->IsFunctionContext() ||
567  context_->IsDebugEvaluateContext());
568  return ScopeTypeLocal;
569  case MODULE_SCOPE:
570  DCHECK_IMPLIES(NeedsAndHasContext(), context_->IsModuleContext());
571  return ScopeTypeModule;
572  case SCRIPT_SCOPE:
573  case REPL_MODE_SCOPE:
574  DCHECK_IMPLIES(NeedsAndHasContext(), context_->IsScriptContext() ||
575  IsNativeContext(*context_));
576  return ScopeTypeScript;
577  case WITH_SCOPE:
578  DCHECK_IMPLIES(NeedsAndHasContext(), context_->IsWithContext());
579  return ScopeTypeWith;
580  case CATCH_SCOPE:
581  DCHECK(context_->IsCatchContext());
582  return ScopeTypeCatch;
583  case BLOCK_SCOPE:
584  case CLASS_SCOPE:
585  DCHECK_IMPLIES(NeedsAndHasContext(), context_->IsBlockContext());
586  return ScopeTypeBlock;
587  case EVAL_SCOPE:
588  DCHECK_IMPLIES(NeedsAndHasContext(), context_->IsEvalContext());
589  return ScopeTypeEval;
590  case SHADOW_REALM_SCOPE:
591  DCHECK_IMPLIES(NeedsAndHasContext(), IsNativeContext(*context_));
592  // TODO(v8:11989): New ScopeType for ShadowRealms?
593  return ScopeTypeScript;
594  }
595  UNREACHABLE();
596  }
597  if (IsNativeContext(*context_)) {
598  DCHECK(IsJSGlobalObject(context_->global_object()));
599  // If we are at the native context and have not yet seen script scope,
600  // fake it.
601  return seen_script_scope_ ? ScopeTypeGlobal : ScopeTypeScript;
602  }
603  if (context_->IsFunctionContext() || context_->IsEvalContext() ||
604  context_->IsDebugEvaluateContext()) {
605  return ScopeTypeClosure;
606  }
607  if (context_->IsCatchContext()) {
608  return ScopeTypeCatch;
609  }
610  if (context_->IsBlockContext()) {
611  return ScopeTypeBlock;
612  }
613  if (context_->IsModuleContext()) {
614  return ScopeTypeModule;
615  }
616  if (context_->IsScriptContext()) {
617  return ScopeTypeScript;
618  }
619  DCHECK(context_->IsWithContext());
620  return ScopeTypeWith;
621 }

623 Handle<JSObject> ScopeIterator::ScopeObject(Mode mode) {
624  DCHECK(!Done());
625 
626  ScopeType type = Type();
627  if (type == ScopeTypeGlobal) {
628  DCHECK_EQ(Mode::ALL, mode);
629  return handle(context_->global_proxy(), isolate_);
630  }
631  if (type == ScopeTypeWith) {
632  DCHECK_EQ(Mode::ALL, mode);
633  return WithContextExtension();
634  }
635 
636  Handle<JSObject> scope = isolate_->factory()->NewSlowJSObjectWithNullProto();
637  auto visitor = [=, this](DirectHandle<String> name,
638  DirectHandle<Object> value, ScopeType scope_type) {
639  if (IsOptimizedOut(*value, isolate_)) {
640  JSObject::SetAccessor(
641  scope, name, isolate_->factory()->value_unavailable_accessor(), NONE)
642  .Check();
643  } else if (IsTheHole(*value, isolate_)) {
644  const bool is_overriden_repl_let =
645  scope_type == ScopeTypeScript &&
646  JSReceiver::HasOwnProperty(isolate_, scope, name).FromMaybe(true);

649  // REPL mode in a script context. Don't install the unavailable accessor
650  // in that case.

661  .Check();
662  }
663  return false;
664  };

877 bool ScopeIterator::VisitLocals(const Visitor& visitor, Mode mode,
878  ScopeType scope_type) const {
879  if (mode == Mode::STACK && current_scope_->is_declaration_scope() &&
880  current_scope_->AsDeclarationScope()->has_this_declaration()) {
881  // TODO(bmeurer): We should refactor the general variable lookup
882  // around "this", since the current way is rather hacky when the
883  // receiver is context-allocated.
884  auto this_var = current_scope_->AsDeclarationScope()->receiver();
885  Handle<Object> receiver =
886  this_var->location() == VariableLocation::CONTEXT
887  ? handle(context_->GetNoCell(this_var->index()), isolate_)
888  : frame_inspector_ == nullptr ? handle(generator_->receiver(), isolate_)
889  : frame_inspector_->GetReceiver();
890  if (visitor(isolate_->factory()->this_string(), receiver, scope_type))
891  return true;
892  }
893 
894  if (current_scope_->is_function_scope()) {
895  Variable* function_var =
896  current_scope_->AsDeclarationScope()->function_var();
897  if (function_var != nullptr) {
898  Handle<JSFunction> function = frame_inspector_ == nullptr
899  ? function_
900  : frame_inspector_->GetFunction();
901  Handle<String> name = function_var->name();
902  if (visitor(name, function, scope_type)) return true;
903  }
904  }
905 
906  for (Variable* var : *current_scope_->locals()) {
907  if (ScopeInfo::VariableIsSynthetic(*var->name())) {
908  // We want to materialize "new.target" for debug-evaluate.
909  if (mode != Mode::STACK ||
910  !var->name()->Equals(*isolate_->factory()->dot_new_target_string())) {
911  continue;
912  }
913  }
914 
915  int index = var->index();
916  Handle<Object> value;
917  switch (var->location()) {
918  case VariableLocation::LOOKUP:
919  UNREACHABLE();
920 
921  case VariableLocation::REPL_GLOBAL:
922  // REPL declared variables are ignored for now.
923  case VariableLocation::UNALLOCATED:
924  continue;
925 
926  case VariableLocation::PARAMETER: {
927  if (frame_inspector_ == nullptr) {
928  // Get the variable from the suspended generator.
929  DCHECK(!generator_.is_null());
930  Tagged<FixedArray> parameters_and_registers =
931  generator_->parameters_and_registers();
932  DCHECK_LT(index, parameters_and_registers->length());
933  value = handle(parameters_and_registers->get(index), isolate_);
934  } else if (var->IsReceiver()) {
935  value = frame_inspector_->GetReceiver();
936  } else {
937  value = frame_inspector_->GetParameter(index);
938  }
939  break;
940  }
941 
942  case VariableLocation::LOCAL:
943  if (frame_inspector_ == nullptr) {
944  // Get the variable from the suspended generator.
945  DCHECK(!generator_.is_null());
946  Tagged<FixedArray> parameters_and_registers =
947  generator_->parameters_and_registers();
948  int parameter_count =
949  function_->shared()->scope_info()->ParameterCount();
950  index += parameter_count;
951  DCHECK_LT(index, parameters_and_registers->length());
952  value = handle(parameters_and_registers->get(index), isolate_);
953  } else {
954  value = frame_inspector_->GetExpression(index);
955  if (IsOptimizedOut(*value, isolate_)) {
956  // We'll rematerialize this later.
957  if (current_scope_->is_declaration_scope() &&
958  current_scope_->AsDeclarationScope()->arguments() == var) {
959  continue;
960  }
961  } else if (IsLexicalVariableMode(var->mode()) &&
962  IsUndefined(*value, isolate_) &&
963  GetSourcePosition() != kNoSourcePosition &&
964  GetSourcePosition() <= var->initializer_position()) {
965  // Variables that are `undefined` could also mean an elided hole
966  // write. We explicitly check the static scope information if we
967  // are currently stopped before the variable is actually initialized
968  // which means we are in the middle of that var's TDZ.
969  value = isolate_->factory()->the_hole_value();
970  }
971  }
972  break;
973 
974  case VariableLocation::CONTEXT:
975  if (mode == Mode::STACK) continue;
976  if (!HasContext()) {
977  // If the context was not yet pushed we report the variable as
978  // unavailable.
979  value = isolate_->factory()->the_hole_value();
980  break;
981  }
982  DCHECK(var->IsContextSlot());
983  DCHECK_EQ(context_->scope_info()->ContextSlotIndex(*var->name()),
984  index);
985  value =
986  indirect_handle(Context::Get(context_, index, isolate_), isolate_);
987  break;
988 
989  case VariableLocation::MODULE: {
990  if (mode == Mode::STACK) continue;
991  // if (var->IsExport()) continue;
992  DirectHandle<SourceTextModule> module(context_->module(), isolate_);
993  value = SourceTextModule::LoadVariable(isolate_, module, var->index());
994  break;
995  }
996  }
997 
998  if (visitor(var->name(), value, scope_type)) return true;
999  }
1000  return false;
1001 }

1017 void ScopeIterator::VisitLocalScope(const Visitor& visitor, Mode mode,
1018  ScopeType scope_type) const {
1019  if (InInnerScope()) {
1020  if (VisitLocals(visitor, mode, scope_type)) return;
1021  if (mode == Mode::STACK && Type() == ScopeTypeLocal) {
1022  // Hide |this| in arrow functions that may be embedded in other functions
1023  // but don't force |this| to be context-allocated. Otherwise we'd find the

1025  if (!closure_scope_->has_this_declaration() &&
1026  !closure_scope_->HasThisReference()) {
1027  if (visitor(isolate_->factory()->this_string(),
1028  isolate_->factory()->undefined_value(), scope_type))
1029  return;
1030  }
1031  // Add |arguments| to the function scope even if it wasn't used.
1032  // Currently we don't yet support materializing the arguments object of
1033  // suspended generators. We'd need to read the arguments out from the
1034  // suspended generator rather than from an activation as
1035  // FunctionGetArguments does.
1036  if (frame_inspector_ != nullptr && !closure_scope_->is_arrow_scope() &&
1037  (closure_scope_->arguments() == nullptr ||
1038  IsOptimizedOut(*frame_inspector_->GetExpression(
1039  closure_scope_->arguments()->index()),
1040  isolate_))) {
1041  JavaScriptFrame* frame = GetFrame();
1042  Handle<JSObject> arguments = Accessors::FunctionGetArguments(
1043  frame, frame_inspector_->inlined_frame_index());
1044  if (visitor(isolate_->factory()->arguments_string(), arguments,
1045  scope_type))
1046  return;
1047  }
1048  }
1049  } else {
1050  DCHECK_EQ(Mode::ALL, mode);
1051  DirectHandle<ScopeInfo> scope_info(context_->scope_info(), isolate_);
1052  if (VisitContextLocals(visitor, scope_info, context_, scope_type)) return;
1053  }
1054 
1055  if (mode == Mode::ALL && HasContext()) {
1056  DCHECK(!context_->IsScriptContext());
1057  DCHECK(!IsNativeContext(*context_));
1058  DCHECK(!context_->IsWithContext());
1059  if (!context_->scope_info()->SloppyEvalCanExtendVars()) return;
1060  if (context_->extension_object().is_null()) return;
1061  DirectHandle<JSObject> extension(context_->extension_object(), isolate_);
1062  DirectHandle<FixedArray> keys =
1063  KeyAccumulator::GetKeys(isolate_, extension,
1064  KeyCollectionMode::kOwnOnly, ENUMERABLE_STRINGS)
1065  .ToHandleChecked();
1066 
1067  for (int i = 0; i < keys->length(); i++) {
1068  // Names of variables introduced by eval are strings.
1069  DCHECK(IsString(keys->get(i)));
1070  Handle<String> key(Cast<String>(keys->get(i)), isolate_);
1071  Handle<Object> value =
1072  JSReceiver::GetDataProperty(isolate_, extension, key);
1073  if (visitor(key, value, scope_type)) return;
1074  }
1075  }

1080  // TODO(verwaest): Walk parameters backwards, not forwards.
1081  // TODO(verwaest): Use VariableMap rather than locals() list for lookup.
1082  for (Variable* var : *current_scope_->locals()) {

1090  *variable_name == ReadOnlyRoots(isolate_).arguments_string());

1092

1094  // Assignments to REPL declared variables are ignored for now.
1095  return false;

1106  } else {

1108  if (!frame->is_unoptimized()) return false;
1109 
1110  frame->SetParameterValue(index, *new_value);
1111  }
1112  return true;
1113  }
1114 
1115  case VariableLocation::LOCAL:
1116  if (frame_inspector_ == nullptr) {
1117  // Set the variable in the suspended generator.
1118  DCHECK(!generator_.is_null());

1120  function_->shared()->scope_info()->ParameterCount();

1126  } else {

1129  if (!frame->is_unoptimized()) return false;

1223 // Given the scope and context of a paused function, this class calculates
1224 // all the necessary block lists on the scope chain and stores them in the
1225 // global LocalsBlockListCache ephemeron table.
1226 //
1227 // Doc: bit.ly/chrome-devtools-debug-evaluate-design.
1228 //
1229 // The algorithm works in a single walk of the scope chain from the
1230 // paused function scope outwards to the script scope.
1231 //
1232 // When we step from scope "a" to its outer scope "b", we do:
1233 //
1234 // 1. Add all stack-allocated variables from "b" to the blocklists.
1235 // 2. Does "b" need a context? If yes:
1236 // - Store all current blocklists in the global table
1237 // - Start a new blocklist for scope "b"
1238 // 3. Is "b" a function scope without a context? If yes:
1239 // - Start a new blocklist for scope "b"
1240 //
1241 class LocalBlocklistsCollector {
1242  public:
1243  LocalBlocklistsCollector(Isolate* isolate, Handle<Script> script,
1244  Handle<Context> context,
1245  DeclarationScope* closure_scope);
1246  void CollectAndStore();
1247 
1248  private:
1249  void InitializeWithClosureScope();
1250  void AdvanceToNextNonHiddenScope();
1251  void CollectCurrentLocalsIntoBlocklists();
1252  DirectHandle<ScopeInfo> FindScopeInfoForScope(Scope* scope) const;
1253  void StoreFunctionBlocklists(DirectHandle<ScopeInfo> outer_scope_info);
1254 
1255  Isolate* isolate_;
1256  Handle<Script> script_;
1257  Handle<Context> context_;
1258  Scope* scope_;
1259  DeclarationScope* closure_scope_;
1260 
1261  Handle<StringSet> context_blocklist_;
1262  std::map<Scope*, IndirectHandle<StringSet>> function_blocklists_;
1263 };

1280 void LocalBlocklistsCollector::AdvanceToNextNonHiddenScope() {
1281  DCHECK(scope_ && scope_->outer_scope());
1282  do {
1283  scope_ = scope_->outer_scope();

1334 void LocalBlocklistsCollector::CollectAndStore() {
1335  InitializeWithClosureScope();
1336 
1337  while (scope_->outer_scope() && !IsNativeContext(*context_)) {
1338  AdvanceToNextNonHiddenScope();
1339  // 1. Add all stack-allocated variables of `scope_` to the various lists.
1340  CollectCurrentLocalsIntoBlocklists();
1341 
1342  // 2. If the current scope requires a context then all the blocklists "stop"
1343  // here and we store them. Next, advance the current context so
1344  // `context_` and `scope_` match again.
1345  if (scope_->NeedsContext()) {
1346  if (!context_blocklist_.is_null()) {
1347  // Only store the block list and advance the context if the
1348  // context_blocklist is set. This handles the case when we start on
1349  // a closure scope that doesn't require a context. In that case
1350  // `context_` is already the right context for `scope_` so we don't
1351  // need to advance `context_`.
1352  isolate_->LocalsBlockListCacheSet(
1353  direct_handle(context_->scope_info(), isolate_),
1354  direct_handle(context_->previous()->scope_info(), isolate_),
1355  context_blocklist_);
1356  context_ = handle(context_->previous(), isolate_);
1357  }
1358 
1359  StoreFunctionBlocklists(direct_handle(context_->scope_info(), isolate_));
1360 
1361  context_blocklist_ = StringSet::New(isolate_);
1362  function_blocklists_.clear();
1363  } else if (scope_->is_function_scope()) {
1364  // 3. If `scope` is a function scope with an SFI, start recording
1365  // locals for its ScopeInfo.
1366  CHECK(!scope_->NeedsContext());
1367  function_blocklists_.emplace(scope_, StringSet::New(isolate_));
1368  }
1369  }
1370 
1371  // In case we don't have any outer scopes we still need to record the empty
1372  // block list for the paused function to prevent future re-parses.
1373  StoreFunctionBlocklists(direct_handle(context_->scope_info(), isolate_));
1374 }
