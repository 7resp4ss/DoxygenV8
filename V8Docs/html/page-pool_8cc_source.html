31 template <typename PoolEntry>
32 std::optional<PoolEntry> PagePool::PoolImpl<PoolEntry>::Get(Isolate* isolate) {
33  DCHECK_NOT_NULL(isolate);
34  base::MutexGuard guard(&mutex_);
35 
36  // Try to get a page from the page pool for the given isolate first.
37  auto it = local_pools_.find(isolate);
38 
39  if (it == local_pools_.end()) {
40  // Pages in this pool will be flushed soon. Take them first.
41  if (!shared_pool_.empty()) {
42  auto& shared_entries = shared_pool_.back().second;
43  if (!shared_entries.empty()) {
44  PoolEntry shared_entry = std::move(shared_entries.back());
45  shared_entries.pop_back();
46  if (shared_entries.empty()) {
47  shared_pool_.pop_back();
48  }
49  return {std::move(shared_entry)};
50  }
51  }
52 
53  // Otherwise steal from some other isolate's local pool.
54  it = local_pools_.begin();
55  }
56 
57  if (it != local_pools_.end()) {
58  DCHECK(!it->second.empty());
59  std::vector<PoolEntry>& entries = it->second;
60  if (!entries.empty()) {
61  PoolEntry local_entry = std::move(entries.back());
62  entries.pop_back();
63  if (entries.empty()) {
64  local_pools_.erase(it);
65  }
66  return {std::move(local_entry)};
67  }
68  }
69 
70  return std::nullopt;
71 }

155 template <typename PoolEntry>
156 size_t PagePool::PoolImpl<PoolEntry>::ReleaseUpTo(InternalTime release_time) {
157  std::vector<std::vector<PoolEntry>> entries_to_free;
158  size_t freed = 0;
159  {
160  base::MutexGuard guard(&mutex_);
161  std::erase_if(shared_pool_,
162  [&entries_to_free, &freed, release_time](auto& entry) {
163  if (entry.first <= release_time) {
164  freed += entry.second.size();
165  entries_to_free.push_back(std::move(entry.second));
166  return true;
167  }
168  return false;
169  });
170  }
171  // Entries will be freed automatically here.
172  return freed;
173 }

175 bool PagePool::LargePagePoolImpl::Add(std::vector<LargePageMetadata*>& pages,
176  InternalTime time) {
177  bool added_to_pool = false;
178  base::MutexGuard guard(&mutex_);
179  DCHECK_EQ(total_size_, ComputeTotalSize());
180 
181  const size_t max_total_size = v8_flags.max_large_page_pool_size * MB;
182 
183  std::erase_if(pages, [this, &added_to_pool, time,
184  max_total_size](LargePageMetadata* page) {
185  if (total_size_ + page->size() > max_total_size) {
186  return false;
187  }
188 
189  total_size_ += page->size();
190  pages_.emplace_back(time,
191  LargePageMemory(page, [](LargePageMetadata* metadata) {
192  MemoryAllocator::DeleteMemoryChunk(metadata);
193  }));
194  added_to_pool = true;
195  return true;
196  });
197 
198  DCHECK_EQ(total_size_, ComputeTotalSize());
199  return added_to_pool;
200 }

202 LargePageMetadata* PagePool::LargePagePoolImpl::Remove(size_t chunk_size) {
203  base::MutexGuard guard(&mutex_);
204  auto selected = pages_.end();
205  DCHECK_EQ(total_size_, ComputeTotalSize());
206 
207  // Best-fit: Select smallest large page large with a size of at least
208  // |chunk_size|. In case the page is larger than necessary, the next full GC
209  // will trim down its size.
210  for (auto it = pages_.begin(); it != pages_.end(); it++) {
211  const size_t page_size = it->second->size();
212  if (page_size < chunk_size) continue;
213  if (selected == pages_.end() || page_size < selected->second->size()) {
214  selected = it;
215  }
216  }
217 
218  if (selected == pages_.end()) {
219  return nullptr;
220  }
221 
222  LargePageMetadata* result = selected->second.release();
223  total_size_ -= result->size();
224  pages_.erase(selected);
225  DCHECK_EQ(total_size_, ComputeTotalSize());
226  return result;
227 }

296 void PagePool::ReleaseOnTearDown(Isolate* isolate) {
297  const InternalTime time = next_time_.fetch_add(1, std::memory_order_relaxed);
298 
299  const bool shared_page_pool_populate =
300  page_pool_.MoveLocalToShared(isolate, time);
301  const bool shared_zone_pool_populated =
302  zone_pool_.MoveLocalToShared(isolate, time);
303 
304  // Always post task when there are pages in the shared pool.
305  if (shared_page_pool_populate || shared_zone_pool_populated) {
306  auto schedule_task = [this, isolate, time](Isolate* target_isolate) {
307  DCHECK_NE(isolate, target_isolate);
308  USE(isolate);
309  static constexpr base::TimeDelta kReleaseTaskDelayInSeconds =
310  base::TimeDelta::FromSeconds(8);
311  target_isolate->task_runner()->PostDelayedTask(
312  std::make_unique<ReleasePooledChunksTask>(target_isolate, this, time),
313  kReleaseTaskDelayInSeconds.InSecondsF());
314  };
315 
316  if (!isolate->isolate_group()->FindAnotherIsolateLocked(isolate,
317  schedule_task)) {
318  // No other isolate could be found. Release pooled pages right away.
319  page_pool_.ReleaseShared();
320  zone_pool_.ReleaseShared();
321  }
322  }
323 
324  large_pool_.ReleaseAll();
325 }

360 void PagePool::Add(Isolate* isolate, MutablePageMetadata* chunk) {
361  DCHECK_NOT_NULL(isolate);
362  // This method is called only on the main thread and only during the
363  // atomic pause so a lock is not needed.
364  DCHECK_NOT_NULL(chunk);
365  DCHECK_EQ(chunk->size(), PageMetadata::kPageSize);
366  DCHECK(!chunk->Chunk()->IsLargePage());
367  DCHECK(!chunk->Chunk()->IsTrusted());
368  DCHECK_NE(chunk->Chunk()->executable(), EXECUTABLE);
369  // Ensure that ReleaseAllAllocatedMemory() was called on the page.
370  DCHECK(!chunk->ContainsAnySlots());
371  page_pool_.PutLocal(isolate,
372  PageMemory(chunk, [](MutablePageMetadata* metadata) {
373  MemoryAllocator::DeleteMemoryChunk(metadata);
374  }));
375 }
