46 // Helper function to loop through Table 8 Duration Record Fields
47 // This function implement
48 // "For each row of Table 8, except the header row, in table order, do"
49 // loop. It is designed to be used to implement the common part of
50 // ToPartialDuration, ToTemporalDurationRecord
51 Maybe<bool> IterateDurationRecordFieldsTable(
52  Isolate* isolate, DirectHandle<JSReceiver> temporal_duration_like,
53  Maybe<bool> (*RowFunction)(Isolate*,
54  DirectHandle<JSReceiver> temporal_duration_like,
55  DirectHandle<String>, double*),
56  DurationRecord* record) {
57  Factory* factory = isolate->factory();
58  std::array<std::pair<DirectHandle<String>, double*>, 10> table8 = {
59  {{factory->days_string(), &record->time_duration.days},
60  {factory->hours_string(), &record->time_duration.hours},
61  {factory->microseconds_string(), &record->time_duration.microseconds},
62  {factory->milliseconds_string(), &record->time_duration.milliseconds},
63  {factory->minutes_string(), &record->time_duration.minutes},
64  {factory->months_string(), &record->months},
65  {factory->nanoseconds_string(), &record->time_duration.nanoseconds},
66  {factory->seconds_string(), &record->time_duration.seconds},
67  {factory->weeks_string(), &record->weeks},
68  {factory->years_string(), &record->years}}};
69 
70  // x. Let any be false.
71  bool any = false;
72  // x+1. For each row of Table 8, except the header row, in table order, do
73  for (const auto& row : table8) {
74  bool result;
75  // row.first is prop: the Property Name value of the current row
76  // row.second is the address of result's field whose name is the Field Name
77  // value of the current row
78  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
79  isolate, result,
80  RowFunction(isolate, temporal_duration_like, row.first, row.second),
81  Nothing<bool>());
82  any |= result;
83  }
84  return Just(any);
85 }

89 // #sec-temporal-isvalidduration
90 bool IsValidDuration(Isolate* isolate, const DurationRecord& dur) {
91  // 1. Let sign be ! DurationSign(years, months, weeks, days, hours, minutes,
92  // seconds, milliseconds, microseconds, nanoseconds).
93  int32_t sign = DurationRecord::Sign(dur);
94  // 2. For each value v of ¬´ years, months, weeks, days, hours, minutes,
95  // seconds, milliseconds, microseconds, nanoseconds ¬ª, do a. If v is not
96  // finite, return false. b. If v < 0 and sign > 0, return false. c. If v > 0
97  // and sign < 0, return false.
98  // 3. Return true.
99  const TimeDurationRecord& time = dur.time_duration;
100 
101  if (!(std::isfinite(dur.years) && std::isfinite(dur.months) &&
102  std::isfinite(dur.weeks) && std::isfinite(time.days) &&
103  std::isfinite(time.hours) && std::isfinite(time.minutes) &&
104  std::isfinite(time.seconds) && std::isfinite(time.milliseconds) &&
105  std::isfinite(time.microseconds) && std::isfinite(time.nanoseconds))) {
106  return false;
107  }
108  if ((sign > 0 && (dur.years < 0 || dur.months < 0 || dur.weeks < 0 ||
109  time.days < 0 || time.hours < 0 || time.minutes < 0 ||
110  time.seconds < 0 || time.milliseconds < 0 ||
111  time.microseconds < 0 || time.nanoseconds < 0)) ||
112  (sign < 0 && (dur.years > 0 || dur.months > 0 || dur.weeks > 0 ||
113  time.days > 0 || time.hours > 0 || time.minutes > 0 ||
114  time.seconds > 0 || time.milliseconds > 0 ||
115  time.microseconds > 0 || time.nanoseconds > 0))) {
116  return false;
117  }
118  static const double kPower32Of2 = static_cast<double>(int64_t(1) << 32);
119  static const int64_t kPower53Of2 = int64_t(1) << 53;
120  // 3. If abs(years) ‚â• 2**32, return false.
121  if (std::abs(dur.years) >= kPower32Of2) {
122  return false;
123  }
124  // 4. If abs(months) ‚â• 2**32, return false.
125  if (std::abs(dur.months) >= kPower32Of2) {
126  return false;
127  }
128  // 5. If abs(weeks) ‚â• 2**32, return false.
129  if (std::abs(dur.weeks) >= kPower32Of2) {
130  return false;
131  }
132  // 6. Let normalizedSeconds be days √ó 86,400 + hours √ó 3600 + minutes √ó 60 +
133  // seconds + ‚Ñù(ùîΩ(milliseconds)) √ó 10**-3 + ‚Ñù(ùîΩ(microseconds)) √ó 10**-6 +
134  // ‚Ñù(ùîΩ(nanoseconds)) √ó 10**-9.
135  // 7. NOTE: The above step cannot be implemented directly using floating-point
136  // arithmetic. Multiplying by 10**-3, 10**-6, and 10**-9 respectively may be
137  // imprecise when milliseconds, microseconds, or nanoseconds is an unsafe
138  // integer. This multiplication can be implemented in C++ with an
139  // implementation of std::remquo() with sufficient bits in the quotient.
140  // String manipulation will also give an exact result, since the
141  // multiplication is by a power of 10.
142  // 8. If abs(normalizedSeconds) ‚â• 2**53, return false.
143 
144  int64_t allowed = kPower53Of2;
145  double in_seconds = std::abs(time.days * 86400.0 + time.hours * 3600.0 +
146  time.minutes * 60.0 + time.seconds);
147 
148  if (in_seconds >= allowed) {
149  return false;
150  }
151  allowed -= in_seconds;
152 
153  // Check the part > 1 seconds.
154  in_seconds = std::floor(std::abs(time.milliseconds / 1e3)) +
155  std::floor(std::abs(time.microseconds / 1e6)) +
156  std::floor(std::abs(time.nanoseconds / 1e9));
157  if (in_seconds >= allowed) {
158  return false;
159  }
160  allowed -= in_seconds;
161 
162  // Sum of the three remainings will surely < 3
163  if (allowed > 3) {
164  return true;
165  }
166 
167  allowed *= 1000000000; // convert to ns
168  int64_t remainders = std::abs(fmod(time.milliseconds, 1e3)) * 1000000 +
169  std::abs(fmod(time.microseconds, 1e6)) * 1000 +
170  std::abs(fmod(time.nanoseconds, 1e9));
171  if (remainders >= allowed) {
172  return false;
173  }
174  return true;
175 }

177 // #sec-temporal-durationsign
178 int32_t DurationRecord::Sign(const DurationRecord& dur) {
179  // 1. For each value v of ¬´ years, months, weeks, days, hours, minutes,
180  // seconds, milliseconds, microseconds, nanoseconds ¬ª, do a. If v < 0, return
181  // ‚àí1. b. If v > 0, return 1.
182  // 2. Return 0.
183  if (dur.years < 0) return -1;
184  if (dur.years > 0) return 1;
185  if (dur.months < 0) return -1;
186  if (dur.months > 0) return 1;
187  if (dur.weeks < 0) return -1;
188  if (dur.weeks > 0) return 1;
189  const TimeDurationRecord& time = dur.time_duration;
190  if (time.days < 0) return -1;
191  if (time.days > 0) return 1;
192  if (time.hours < 0) return -1;
193  if (time.hours > 0) return 1;
194  if (time.minutes < 0) return -1;
195  if (time.minutes > 0) return 1;
196  if (time.seconds < 0) return -1;
197  if (time.seconds > 0) return 1;
198  if (time.milliseconds < 0) return -1;
199  if (time.milliseconds > 0) return 1;
200  if (time.microseconds < 0) return -1;
201  if (time.microseconds > 0) return 1;
202  if (time.nanoseconds < 0) return -1;
203  if (time.nanoseconds > 0) return 1;
204  return 0;
205 }

255 Maybe<DurationRecord> ToPartialDuration(
256  Isolate* isolate, DirectHandle<Object> temporal_duration_like_obj,
257  const DurationRecord& input) {
258  // 1. If Type(temporalDurationLike) is not Object, then
259  if (!IsJSReceiver(*temporal_duration_like_obj)) {
260  // a. Throw a TypeError exception.
261  THROW_NEW_ERROR_RETURN_VALUE(isolate, NEW_TEMPORAL_INVALID_ARG_TYPE_ERROR(),
262  Nothing<DurationRecord>());
263  }
264  DirectHandle<JSReceiver> temporal_duration_like =
265  Cast<JSReceiver>(temporal_duration_like_obj);
266 
267  // 2. Let result be a new partial Duration Record with each field set to
268  // undefined.
269  DurationRecord result = input;
270 
271  // 3. Let any be false.
272  bool any = false;
273 
274  // Table 8: Duration Record Fields
275  // #table-temporal-duration-record-fields
276  // 4. For each row of Table 8, except the header row, in table order, do
277  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
278  isolate, any,
279  IterateDurationRecordFieldsTable(
280  isolate, temporal_duration_like,
281  [](Isolate* isolate, DirectHandle<JSReceiver> temporal_duration_like,
282  DirectHandle<String> prop, double* field) -> Maybe<bool> {
283  bool not_undefined = false;
284  // a. Let prop be the Property value of the current row.
285  DirectHandle<Object> val;
286  // b. Let val be ? Get(temporalDurationLike, prop).
287  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
288  isolate, val,
289  JSReceiver::GetProperty(isolate, temporal_duration_like, prop),
290  Nothing<bool>());
291  // c. If val is not undefined, then
292  if (!IsUndefined(*val)) {
293  // i. Set any to true.
294  not_undefined = true;
295  // ii. Let val be ùîΩ(? ToIntegerWithoutRounding(val)).
296  // iii. Set result's field whose name is the Field Name value of
297  // the current row to val.
298  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
299  isolate, *field, ToIntegerWithoutRounding(isolate, val),
300  Nothing<bool>());
301  }
302  return Just(not_undefined);
303  },
304  &result),
305  Nothing<DurationRecord>());
306 
307  // 5. If any is false, then
308  if (!any) {
309  // a. Throw a TypeError exception.
310  THROW_NEW_ERROR_RETURN_VALUE(isolate, NEW_TEMPORAL_INVALID_ARG_TYPE_ERROR(),
311  Nothing<DurationRecord>());
312  }
313  // 6. Return result.
314  return Just(result);
315 }
