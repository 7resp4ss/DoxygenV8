46  // still pretend to have the default number of worker threads to not

51  void PostTaskOnWorkerThreadImpl(TaskPriority priority,
52  std::unique_ptr<Task> task,
53  const SourceLocation& location) override {
54  // We post worker tasks on the foreground task runner of the
55  // {kProcessGlobalPredictablePlatformWorkerTaskQueue} isolate. The task
56  // queue of the {kProcessGlobalPredictablePlatformWorkerTaskQueue} isolate
57  // is then executed on the main thread to achieve predictable behavior.
58  //
59  // In this context here it is okay to call {GetForegroundTaskRunner} from a
60  // background thread. The reason is that code is executed sequentially with
61  // the PredictablePlatform, and that the {DefaultPlatform} does not access
62  // the isolate but only uses it as the key in a HashMap.
63  platform_
64  ->GetForegroundTaskRunner(
65  kProcessGlobalPredictablePlatformWorkerTaskQueue, priority)
66  ->PostTask(std::move(task));
67  }
68 
69  void PostDelayedTaskOnWorkerThreadImpl(
70  TaskPriority priority, std::unique_ptr<Task> task,
71  double delay_in_seconds, const SourceLocation& location) override {
72  // Never run delayed tasks.
73  }
74 
75  bool IdleTasksEnabled(Isolate* isolate) override { return false; }
76 
77  std::unique_ptr<JobHandle> CreateJobImpl(
78  TaskPriority priority, std::unique_ptr<JobTask> job_task,
79  const SourceLocation& location) override {
80  // Do not call {platform_->PostJob} here, as this would create a job that
81  // posts tasks directly to the underlying default platform.
82  return platform::NewDefaultJobHandle(this, priority, std::move(job_task),
83  NumberOfWorkerThreads());
84  }
85 
86  double MonotonicallyIncreasingTime() override {
87  // In predictable mode, there should be no (observable) concurrency, but we
88  // still run some tests that explicitly specify '--predictable' in the
89  // '--isolates' variant, where several threads run the same test in
90  // different isolates. To avoid TSan issues in that scenario we use atomic
91  // increments here.
92  uint64_t synthetic_time =
93  synthetic_time_.fetch_add(1, std::memory_order_relaxed);
94  return 1e-5 * synthetic_time;
95  }
96 
97  double CurrentClockTimeMillis() override {
98  return MonotonicallyIncreasingTime() * base::Time::kMillisecondsPerSecond;
99  }

103  }

106  return platform_->GetThreadIsolatedAllocator();
107  }
108 
109  Platform* platform() const { return platform_.get(); }
110 
111  private:
112  std::atomic<uint64_t> synthetic_time_{0};
113  std::unique_ptr<Platform> platform_;
114 };

150  std::shared_ptr<TaskRunner> GetForegroundTaskRunner(
151  v8::Isolate* isolate, TaskPriority priority) override {
152  std::shared_ptr<TaskRunner> runner =
153  platform_->GetForegroundTaskRunner(isolate, priority);
154 
155  base::MutexGuard lock_guard(&mutex_);
156  // Check if we can re-materialize the weak ptr in our map.
157  std::weak_ptr<DelayedTaskRunner>& weak_delayed_runner =
158  delayed_task_runners_[runner.get()];
159  std::shared_ptr<DelayedTaskRunner> delayed_runner =
160  weak_delayed_runner.lock();
161 
162  if (!delayed_runner) {
163  // Create a new {DelayedTaskRunner} and keep a weak reference in our map.
164  delayed_runner = std::make_shared<DelayedTaskRunner>(runner, this);
165  weak_delayed_runner = delayed_runner;
166  }
167 
168  return std::move(delayed_runner);
169  }

298  class DelayedJob final : public JobTask {
299  public:
300  DelayedJob(std::unique_ptr<JobTask> job_task, int32_t delay_ms)
301  : job_task_(std::move(job_task)), delay_ms_(delay_ms) {}
302 
303  void Run(JobDelegate* delegate) override {
304  // If this job is being executed via worker tasks (as e.g. the
305  // {DefaultJobHandle} implementation does it), the worker task would
306  // already include a delay. In order to not depend on that, we add our own
307  // delay here anyway.
308  base::OS::Sleep(base::TimeDelta::FromMicroseconds(delay_ms_));
309  job_task_->Run(delegate);
310  }
311 
312  size_t GetMaxConcurrency(size_t worker_count) const override {
313  return job_task_->GetMaxConcurrency(worker_count);
314  }
315 
316  private:
317  std::unique_ptr<JobTask> job_task_;
318  int32_t delay_ms_;
319  };
320 
321  std::unique_ptr<Platform> platform_;
322 
323  // The Mutex protects the RNG, which is used by foreground and background
324  // threads, and the {delayed_task_runners_} map might be accessed concurrently
325  // by the shared_ptr destructor.
326  base::Mutex mutex_;
327  base::RandomNumberGenerator rng_;
328  std::unordered_map<TaskRunner*, std::weak_ptr<DelayedTaskRunner>>
329  delayed_task_runners_;
330 
331  int32_t GetRandomDelayInMilliseconds() {
332  base::MutexGuard lock_guard(&mutex_);
333  double delay_fraction = rng_.NextDouble();
334  // Sleep up to 100ms (100000us). Square {delay_fraction} to shift
335  // distribution towards shorter sleeps.
336  return 1e5 * (delay_fraction * delay_fraction);
337  }
338 
339  std::unique_ptr<Task> MakeDelayedTask(std::unique_ptr<Task> task) {
340  return std::make_unique<DelayedTask>(std::move(task),
341  GetRandomDelayInMilliseconds());
342  }
