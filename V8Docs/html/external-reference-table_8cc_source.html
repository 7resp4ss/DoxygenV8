104  DCHECK_EQ(is_initialized_, kInitializedIsolateIndependent);

195 void ExternalReferenceTable::AddBuiltins(
196  int* index, MemorySpan<Address> shared_external_references) {
197  CHECK_EQ(kSpecialReferenceCount + kExternalReferenceCountIsolateIndependent,
198  *index);
199 
200  static const Address c_builtins[] = {
201 #define DEF_ENTRY(Name, ...) FUNCTION_ADDR(&Builtin_##Name),
202  BUILTIN_LIST_C(DEF_ENTRY)
203 #undef DEF_ENTRY
204  };
205  for (Address addr : c_builtins) {
206  AddIsolateIndependent(ExternalReference::Create(addr).address(), index,
207  shared_external_references);
208  }
209 
210  CHECK_EQ(kSpecialReferenceCount + kExternalReferenceCountIsolateIndependent +
211  kBuiltinsReferenceCount,
212  *index);
213 }

216 void ExternalReferenceTable::AddRuntimeFunctions(

218  CHECK_EQ(kSpecialReferenceCount + kExternalReferenceCountIsolateIndependent +
219  kBuiltinsReferenceCount,

222  static constexpr Runtime::FunctionId runtime_functions[] = {

229  AddIsolateIndependent(ExternalReference::Create(fId).address(), index,

233  CHECK_EQ(kSpecialReferenceCount + kExternalReferenceCountIsolateIndependent +
234  kBuiltinsReferenceCount + kRuntimeReferenceCount,
235  *index);

238 void ExternalReferenceTable::CopyIsolateIndependentReferences(
239  int* index, MemorySpan<Address> shared_external_references) {
240  CHECK_EQ(0, *index);
241 
242  DCHECK_GE(shared_external_references.size(), kSizeIsolateIndependent);
243  std::copy(shared_external_references.data(),
244  shared_external_references.data() + kSizeIsolateIndependent,
245  ref_addr_);

263 void ExternalReferenceTable::AddAccessors(
264  int* index, MemorySpan<Address> shared_external_references) {
265  CHECK_EQ(kSpecialReferenceCount + kExternalReferenceCountIsolateIndependent +
266  kBuiltinsReferenceCount + kRuntimeReferenceCount,
267  *index);
268 
269 #define ACCESSOR_INFO_DECLARATION(_, __, AccessorName, ...) \
270  FUNCTION_ADDR(&Accessors::AccessorName##Getter),
271 #define ACCESSOR_GETTER_DECLARATION(name) FUNCTION_ADDR(&Accessors::name),
272 #define ACCESSOR_SETTER_DECLARATION(name) FUNCTION_ADDR(&Accessors::name),
273 #define ACCESSOR_CALLBACK_DECLARATION(_, AccessorName, ...) \
274  FUNCTION_ADDR(&Accessors::AccessorName),
275 
276  static const Address accessors[] = {
277  // Getters:
278  ACCESSOR_INFO_LIST_GENERATOR(ACCESSOR_INFO_DECLARATION, /* not used */)
279  // More getters:
280  ACCESSOR_GETTER_LIST(ACCESSOR_GETTER_DECLARATION)
281  // Setters:
282  ACCESSOR_SETTER_LIST(ACCESSOR_SETTER_DECLARATION)
283  // Callbacks:
284  ACCESSOR_CALLBACK_LIST_GENERATOR(ACCESSOR_CALLBACK_DECLARATION,
285  /* not used */)};
286 #undef ACCESSOR_INFO_DECLARATION
287 #undef ACCESSOR_GETTER_DECLARATION
288 #undef ACCESSOR_SETTER_DECLARATION
289 #undef ACCESSOR_CALLBACK_DECLARATION
290 
291  for (Address addr : accessors) {
292  AddIsolateIndependent(addr, index, shared_external_references);
293  }
294 
295  CHECK_EQ(kSpecialReferenceCount + kExternalReferenceCountIsolateIndependent +
296  kBuiltinsReferenceCount + kRuntimeReferenceCount +
297  kAccessorReferenceCount,
298  *index);
299 }

301 void ExternalReferenceTable::AddStubCache(Isolate* isolate, int* index) {
302  CHECK_EQ(kSizeIsolateIndependent + kExternalReferenceCountIsolateDependent +
303  kIsolateAddressReferenceCount,
304  *index);
305 
306  // Stub cache tables
307  std::array<StubCache*, 3> stub_caches{isolate->load_stub_cache(),
308  isolate->store_stub_cache(),
309  isolate->define_own_stub_cache()};
310 
311  for (StubCache* stub_cache : stub_caches) {
312  Add(stub_cache->key_reference(StubCache::kPrimary).address(), index);
313  Add(stub_cache->value_reference(StubCache::kPrimary).address(), index);
314  Add(stub_cache->map_reference(StubCache::kPrimary).address(), index);
315  Add(stub_cache->key_reference(StubCache::kSecondary).address(), index);
316  Add(stub_cache->value_reference(StubCache::kSecondary).address(), index);
317  Add(stub_cache->map_reference(StubCache::kSecondary).address(), index);
318  }
319 
320  CHECK_EQ(kSizeIsolateIndependent + kExternalReferenceCountIsolateDependent +
321  kIsolateAddressReferenceCount + kStubCacheReferenceCount,
322  *index);
323 }

325 Address ExternalReferenceTable::GetStatsCounterAddress(StatsCounter* counter) {
326  if (!counter->Enabled()) {
327  return reinterpret_cast<Address>(&dummy_stats_counter_);
328  }
329  std::atomic<int>* address = counter->GetInternalPointer();
