35  void CheckInstanceType(V<Object> input, V<FrameState> frame_state,
36  const FeedbackSource& feedback,
37  InstanceType first_instance_type,
38  InstanceType last_instance_type, bool check_smi) {
39  if (check_smi) {
40  __ DeoptimizeIf(__ IsSmi(input), frame_state,
41  DeoptimizeReason::kWrongInstanceType, feedback);
42  }
43 
44  V<i::Map> map = __ LoadMapField(input);
45 
46  if (first_instance_type == last_instance_type) {
47 #if V8_STATIC_ROOTS_BOOL
48  if (InstanceTypeChecker::UniqueMapOfInstanceType(first_instance_type)) {
49  std::optional<RootIndex> expected_index =
50  InstanceTypeChecker::UniqueMapOfInstanceType(first_instance_type);
51  CHECK(expected_index.has_value());
52  Handle<HeapObject> expected_map =
53  Cast<HeapObject>(isolate_->root_handle(expected_index.value()));
54  __ DeoptimizeIfNot(__ TaggedEqual(map, __ HeapConstant(expected_map)),
55  frame_state, DeoptimizeReason::kWrongInstanceType,
56  feedback);
57  return;
58  }
59 #endif // V8_STATIC_ROOTS_BOOL
60  V<Word32> instance_type = __ LoadInstanceTypeField(map);
61  __ DeoptimizeIfNot(__ Word32Equal(instance_type, first_instance_type),
62  frame_state, DeoptimizeReason::kWrongInstanceType,
63  feedback);
64  } else {
65  __ DeoptimizeIfNot(CheckInstanceTypeIsInRange(map, first_instance_type,
66  last_instance_type),
67  frame_state, DeoptimizeReason::kWrongInstanceType,
68  feedback);
69  }
70  }

72  V<InternalizedString> CheckedInternalizedString(
73  V<Object> object, V<FrameState> frame_state, bool check_smi,
74  const FeedbackSource& feedback) {
75  if (check_smi) {
76  __ DeoptimizeIf(__ IsSmi(object), frame_state, DeoptimizeReason::kSmi,
77  feedback);
78  }
79 
80  Label<InternalizedString> done(this);
81  V<Map> map = __ LoadMapField(object);
82  V<Word32> instance_type = __ LoadInstanceTypeField(map);
83 
84  // Go to the slow path if this is a non-string, or a non-internalised
85  // string.
86  static_assert((kStringTag | kInternalizedTag) == 0);
87  IF (UNLIKELY(__ Word32BitwiseAnd(
88  instance_type, kIsNotStringMask | kIsNotInternalizedMask))) {
89  // Deopt if this isn't a string.
90  __ DeoptimizeIf(__ Word32BitwiseAnd(instance_type, kIsNotStringMask),
91  frame_state, DeoptimizeReason::kWrongMap, feedback);
92  // Deopt if this isn't a thin string.
93  static_assert(base::bits::CountPopulation(kThinStringTagBit) == 1);
94  __ DeoptimizeIfNot(__ Word32BitwiseAnd(instance_type, kThinStringTagBit),
95  frame_state, DeoptimizeReason::kWrongMap, feedback);
96  // Load internalized string from thin string.
97  V<InternalizedString> intern_string =
98  __ template LoadField<InternalizedString>(
99  object, AccessBuilder::ForThinStringActual());
100  GOTO(done, intern_string);
101  } ELSE {
102  GOTO(done, V<InternalizedString>::Cast(object));
103  }
104 
105  BIND(done, result);
106  return result;
107  }

124  V<Object> CheckConstructResult(V<Object> construct_result,
125  V<Object> implicit_receiver) {
126  // If the result is an object (in the ECMA sense), we should get rid
127  // of the receiver and use the result; see ECMA-262 version 5.1
128  // section 13.2.2-7 on page 74.
129  Label<Object> done(this);
130 
131  GOTO_IF(
132  __ RootEqual(construct_result, RootIndex::kUndefinedValue, isolate_),
133  done, implicit_receiver);
134 
135  // If the result is a smi, it is *not* an object in the ECMA sense.
136  GOTO_IF(__ IsSmi(construct_result), done, implicit_receiver);
137 
138  // Check if the type of the result is not an object in the ECMA sense.
139  GOTO_IF(JSAnyIsNotPrimitive(V<HeapObject>::Cast(construct_result)), done,
140  construct_result);
141 
142  // Throw away the result of the constructor invocation and use the
143  // implicit receiver as the result.
144  GOTO(done, implicit_receiver);
145 
146  BIND(done, result);
147  return result;
148  }

150  void CheckDerivedConstructResult(V<Object> construct_result,
151  V<FrameState> frame_state,
152  V<NativeContext> native_context,
153  LazyDeoptOnThrow lazy_deopt_on_throw) {
154  // The result of a derived construct should be an object (in the ECMA
155  // sense).
156  Label<> do_throw(this);
157  Label<> end(this);
158 
159  // If the result is a smi, it is *not* an object in the ECMA sense.
160  GOTO_IF(UNLIKELY(__ IsSmi(construct_result)), do_throw);
161 
162  // Check if the type of the result is not an object in the ECMA sense.
163  GOTO_IF(LIKELY(JSAnyIsNotPrimitive(V<HeapObject>::Cast(construct_result))),
164  end);
165  GOTO(do_throw);
166 
167  BIND(do_throw);
168  {
169  __ CallRuntime_ThrowConstructorReturnedNonObject(
170  isolate_, frame_state, native_context, lazy_deopt_on_throw);
171  // ThrowConstructorReturnedNonObject should not return.
172  __ Unreachable();
173  }
174 
175  BIND(end);
176  }

178  V<Smi> UpdateJSArrayLength(V<Word32> length_raw, V<JSArray> object,
179  V<Word32> index) {
180  Label<Smi> done(this);
181  IF (__ Uint32LessThan(index, length_raw)) {
182  GOTO(done, __ TagSmi(length_raw));
183  } ELSE {
184  V<Word32> new_length_raw =
185  __ Word32Add(index, 1); // This cannot overflow.
186  V<Smi> new_length_tagged = __ TagSmi(new_length_raw);
187  __ Store(object, new_length_tagged, StoreOp::Kind::TaggedBase(),
188  MemoryRepresentation::TaggedSigned(),
189  WriteBarrierKind::kNoWriteBarrier, JSArray::kLengthOffset);
190  GOTO(done, new_length_tagged);
191  }
192 
193  BIND(done, length_tagged);
194  return length_tagged;
195  }

210  void TransitionElementsKind(
211  V<Object> object, V<Map> map,
212  const ZoneVector<compiler::MapRef>& transition_sources,
213  const MapRef transition_target, Label<Map>& end) {
214  // Turboshaft's TransitionElementsKind operation loads the map everytime, so
215  // we don't call it to have a single map load (in practice,
216  // LateLoadElimination should probably eliminate the subsequent map loads,
217  // but let's not risk it).
218  V<Map> target_map = __ HeapConstant(transition_target.object());
219 
220  for (const compiler::MapRef transition_source : transition_sources) {
221  bool is_simple = IsSimpleMapChangeTransition(
222  transition_source.elements_kind(), transition_target.elements_kind());
223  IF (__ TaggedEqual(map, __ HeapConstant(transition_source.object()))) {
224  if (is_simple) {
225  __ StoreField(object, AccessBuilder::ForMap(), target_map);
226  } else {
227  __ CallRuntime_TransitionElementsKind(
228  isolate_, __ NoContextConstant(), V<HeapObject>::Cast(object),
229  target_map);
230  }
231  GOTO(end, target_map);
232  }
233  }

236  V<Word32> JSAnyIsNotPrimitive(V<HeapObject> heap_object) {
237  V<Map> map = __ LoadMapField(heap_object);
238  if constexpr (V8_STATIC_ROOTS_BOOL) {
239  // All primitive object's maps are allocated at the start of the read only
240  // heap. Thus JS_RECEIVER's must have maps with larger (compressed)
241  // addresses.
242  return __ Uint32LessThanOrEqual(
243  InstanceTypeChecker::kNonJsReceiverMapLimit,
244  __ TruncateWordPtrToWord32(__ BitcastTaggedToWordPtr(map)));
245  } else {
246  static_assert(LAST_JS_RECEIVER_TYPE == LAST_TYPE);
247  return __ Uint32LessThanOrEqual(FIRST_JS_RECEIVER_TYPE,
248  __ LoadInstanceTypeField(map));
249  }
250  }

252  V<Boolean> HasInPrototypeChain(V<Object> object, HeapObjectRef prototype,
253  V<FrameState> frame_state,
254  V<NativeContext> native_context,
255  LazyDeoptOnThrow lazy_deopt_on_throw) {
256  Label<Boolean> done(this);
257 
258  V<Boolean> true_bool = __ HeapConstant(factory_->true_value());
259  V<Boolean> false_bool = __ HeapConstant(factory_->false_value());
260  V<HeapObject> target_proto = __ HeapConstant(prototype.object());
261 
262  GOTO_IF(__ IsSmi(object), done, false_bool);
263 
264  LoopLabel<Map> loop(this);
265  GOTO(loop, __ LoadMapField(object));
266 
267  BIND_LOOP(loop, map) {
268  Label<> object_is_direct(this);
269 
270  IF (UNLIKELY(CheckInstanceTypeIsInRange(map, FIRST_TYPE,
271  LAST_SPECIAL_RECEIVER_TYPE))) {
272  Label<> call_runtime(this);
273  V<Word32> instance_type = __ LoadInstanceTypeField(map);
274 
275  GOTO_IF(__ Word32Equal(instance_type, JS_PROXY_TYPE), call_runtime);
276 
277  V<Word32> bitfield =
278  __ template LoadField<Word32>(map, AccessBuilder::ForMapBitField());
279  int mask = Map::Bits1::HasNamedInterceptorBit::kMask |
280  Map::Bits1::IsAccessCheckNeededBit::kMask;
281  GOTO_IF_NOT(__ Word32BitwiseAnd(bitfield, mask), object_is_direct);
282  GOTO(call_runtime);
283 
284  BIND(call_runtime);
285  GOTO(done, __ CallRuntime_HasInPrototypeChain(
286  isolate_, frame_state, native_context,
287  lazy_deopt_on_throw, object, target_proto));
288  }
289  GOTO(object_is_direct);
290 
291  BIND(object_is_direct);
292  V<HeapObject> proto = __ template LoadField<HeapObject>(
293  map, AccessBuilder::ForMapPrototype());
294  GOTO_IF(__ RootEqual(proto, RootIndex::kNullValue, isolate_), done,
295  false_bool);
296  GOTO_IF(__ TaggedEqual(proto, target_proto), done, true_bool);
297 
298  GOTO(loop, __ LoadMapField(proto));
299  }
300 
301  BIND(done, result);
302  return result;
303  }

305  V<Map> MigrateMapIfNeeded(V<HeapObject> object, V<Map> map,
306  V<FrameState> frame_state,
307  const FeedbackSource& feedback) {
308  ScopedVar<Map> result(this, map);
309 
310  V<Word32> bitfield3 =
311  __ template LoadField<Word32>(map, AccessBuilder::ForMapBitField3());
312  IF (UNLIKELY(__ Word32BitwiseAnd(bitfield3,
313  Map::Bits3::IsDeprecatedBit::kMask))) {
314  V<Object> object_or_smi = __ CallRuntime_TryMigrateInstance(
315  isolate_, __ NoContextConstant(), object);
316  __ DeoptimizeIf(__ ObjectIsSmi(object_or_smi), frame_state,
317  DeoptimizeReason::kInstanceMigrationFailed, feedback);
318  // Reload the map since TryMigrateInstance might have changed it.
319  result = __ LoadMapField(V<HeapObject>::Cast(object_or_smi));
320  }
321 
322  return result;

325  V<PropertyArray> ExtendPropertiesBackingStore(
326  V<PropertyArray> old_property_array, V<JSObject> object, int old_length,
327  V<FrameState> frame_state, const FeedbackSource& feedback) {
328  // Allocate new PropertyArray.
329  int new_length = old_length + JSObject::kFieldsAdded;
330  Uninitialized<PropertyArray> new_property_array =
331  __ template Allocate<PropertyArray>(
332  __ IntPtrConstant(PropertyArray::SizeFor(new_length)),
333  AllocationType::kYoung, kTaggedAligned);
334  __ InitializeField(new_property_array, AccessBuilder::ForMap(),
335  __ HeapConstant(factory_->property_array_map()));
336 
337  // Copy existing properties over.
338  for (int i = 0; i < old_length; i++) {
339  V<Object> old_value = __ template LoadField<Object>(
340  old_property_array, AccessBuilder::ForPropertyArraySlot(i));
341  __ InitializeField(new_property_array,
342  AccessBuilder::ForPropertyArraySlot(i), old_value);
343  }
344 
345  // Initialize new properties to undefined.
346  V<Undefined> undefined = __ HeapConstant(factory_->undefined_value());
347  for (int i = 0; i < JSObject::kFieldsAdded; ++i) {
348  __ InitializeField(new_property_array,
349  AccessBuilder::ForPropertyArraySlot(old_length + i),
350  undefined);
351  }
352 
353  // Read the hash.
354  ScopedVar<Word32> hash(this);
355  if (old_length == 0) {
356  // The object might still have a hash, stored in properties_or_hash. If
357  // properties_or_hash is a SMI, then it's the hash. It can also be an
358  // empty PropertyArray.
359  V<Object> hash_obj = __ template LoadField<Object>(
360  object, AccessBuilder::ForJSObjectPropertiesOrHash());
361  IF (__ IsSmi(hash_obj)) {
362  hash = __ Word32ShiftLeft(__ UntagSmi(V<Smi>::Cast(hash_obj)),
363  PropertyArray::HashField::kShift);
364  } ELSE {
365  hash = __ Word32Constant(PropertyArray::kNoHashSentinel);
366  }
367  } else {
368  V<Smi> hash_smi = __ template LoadField<Smi>(
369  old_property_array, AccessBuilder::ForPropertyArrayLengthAndHash());
370  hash = __ Word32BitwiseAnd(__ UntagSmi(hash_smi),
371  PropertyArray::HashField::kMask);
372  }
373 
374  // Add the new length and write the length-and-hash field.
375  static_assert(PropertyArray::LengthField::kShift == 0);
376  V<Word32> length_and_hash = __ Word32BitwiseOr(hash, new_length);
377  __ InitializeField(new_property_array,
378  AccessBuilder::ForPropertyArrayLengthAndHash(),
379  __ TagSmi(length_and_hash));
380 
381  V<PropertyArray> initialized_new_property_array =
382  __ FinishInitialization(std::move(new_property_array));
383 
384  // Replace the old property array in {object}.
385  __ StoreField(object, AccessBuilder::ForJSObjectPropertiesOrHash(),
386  initialized_new_property_array);
387 
388  return initialized_new_property_array;
389  }

391  void GeneratorStore(V<Context> context, V<JSGeneratorObject> generator,
392  base::SmallVector<OpIndex, 32> parameters_and_registers,
393  int suspend_id, int bytecode_offset) {
394  V<FixedArray> array = __ template LoadTaggedField<FixedArray>(
395  generator, JSGeneratorObject::kParametersAndRegistersOffset);
396  for (int i = 0; static_cast<size_t>(i) < parameters_and_registers.size();
397  i++) {
398  __ Store(array, parameters_and_registers[i], StoreOp::Kind::TaggedBase(),
399  MemoryRepresentation::AnyTagged(),
400  WriteBarrierKind::kFullWriteBarrier,
401  FixedArray::OffsetOfElementAt(i));
402  }
403  __ Store(generator, __ SmiConstant(Smi::FromInt(suspend_id)),
404  StoreOp::Kind::TaggedBase(), MemoryRepresentation::TaggedSigned(),
405  WriteBarrierKind::kNoWriteBarrier,
406  JSGeneratorObject::kContinuationOffset);
407  __ Store(generator, __ SmiConstant(Smi::FromInt(bytecode_offset)),
408  StoreOp::Kind::TaggedBase(), MemoryRepresentation::TaggedSigned(),
409  WriteBarrierKind::kNoWriteBarrier,
410  JSGeneratorObject::kInputOrDebugPosOffset);
411 
412  __ Store(generator, context, StoreOp::Kind::TaggedBase(),
413  MemoryRepresentation::AnyTagged(),
414  WriteBarrierKind::kFullWriteBarrier,
415  JSGeneratorObject::kContextOffset);
416  }
