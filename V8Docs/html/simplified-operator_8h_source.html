101 // An access descriptor for loads/stores of fixed structures like field
102 // accesses of heap objects. Accesses from either tagged or untagged base
103 // pointers are supported; untagging is done automatically during lowering.
104 struct FieldAccess {
105  BaseTaggedness base_is_tagged; // specifies if the base pointer is tagged.
106  int offset; // offset of the field, without tag.
107  MaybeHandle<Name> name; // debugging only.
108  OptionalMapRef map; // map of the field value (if known).
109  Type type; // type of the field.
110  MachineType machine_type; // machine type of the field.
111  WriteBarrierKind write_barrier_kind; // write barrier hint.
112  const char* creator_mnemonic; // store the name of factory/creator method
113  ConstFieldInfo const_field_info;// the constness of this access, and the
114  // field owner map, if the access is const
115  bool is_store_in_literal; // originates from a kStoreInLiteral access
116  ExternalPointerTag external_pointer_tag = kExternalPointerNullTag;
117  bool maybe_initializing_or_transitioning_store; // store is potentially
118  // initializing a newly
119  // allocated object or part
120  // of a map transition.
121  bool is_bounded_size_access = false; // Whether this field is stored as a
122  // bounded size field. In that case,
123  // the size is shifted to the left to
124  // guarantee that the value is at most
125  // kMaxSafeBufferSizeForSandbox after
126  // decoding.
127  bool is_immutable = false; // Whether this field is known to be immutable for
128  // the purpose of loads.
129  IndirectPointerTag indirect_pointer_tag = kIndirectPointerNullTag;
130 
131  FieldAccess()
132  : base_is_tagged(kTaggedBase),
133  offset(0),
134  type(Type::None()),
135  machine_type(MachineType::None()),
136  write_barrier_kind(kFullWriteBarrier),
137  creator_mnemonic(nullptr),
138  const_field_info(ConstFieldInfo::None()),
139  is_store_in_literal(false),
140  maybe_initializing_or_transitioning_store(false) {}
141 
142  FieldAccess(BaseTaggedness base_is_tagged, int offset, MaybeHandle<Name> name,
143  OptionalMapRef map, Type type, MachineType machine_type,
144  WriteBarrierKind write_barrier_kind,
145  const char* creator_mnemonic = nullptr,
146  ConstFieldInfo const_field_info = ConstFieldInfo::None(),
147  bool is_store_in_literal = false,
148  ExternalPointerTag external_pointer_tag = kExternalPointerNullTag,
149  bool maybe_initializing_or_transitioning_store = false,
150  bool is_immutable = false,
151  IndirectPointerTag indirect_pointer_tag = kIndirectPointerNullTag)
152  : base_is_tagged(base_is_tagged),
153  offset(offset),
154  name(name),
155  map(map),
156  type(type),
157  machine_type(machine_type),
158  write_barrier_kind(write_barrier_kind),
159  const_field_info(const_field_info),
160  is_store_in_literal(is_store_in_literal),
161  external_pointer_tag(external_pointer_tag),
162  maybe_initializing_or_transitioning_store(
163  maybe_initializing_or_transitioning_store),
164  is_immutable(is_immutable),
165  indirect_pointer_tag(indirect_pointer_tag) {
166  DCHECK_GE(offset, 0);
167  DCHECK_IMPLIES(
168  machine_type.IsMapWord(),
169  offset == HeapObject::kMapOffset && base_is_tagged != kUntaggedBase);
170  DCHECK_IMPLIES(machine_type.IsMapWord(),
171  (write_barrier_kind == kMapWriteBarrier ||
172  write_barrier_kind == kNoWriteBarrier ||
173  write_barrier_kind == kAssertNoWriteBarrier));
174  #if !defined(OFFICIAL_BUILD)
175  this->creator_mnemonic = creator_mnemonic;
176  #else
177  this->creator_mnemonic = nullptr;
178  #endif
179  }
180 
181  int tag() const { return base_is_tagged == kTaggedBase ? kHeapObjectTag : 0; }
182 };

798 // Interface for building simplified operators, which represent the

801 // All operators are typed but many are representation independent.

804 // - Tagged: word-sized integer that is either
805 // - a signed small integer (31 or 32 bits plus a tag)
806 // - a tagged pointer to a HeapNumber object that has a float64 field
807 // - Int32: an untagged signed 32-bit integer
808 // - Uint32: an untagged unsigned 32-bit integer
