7 // This file implements the out of bounds trap handler for
8 // WebAssembly. Trap handlers are notoriously difficult to get
9 // right, and getting it wrong can lead to security
10 // vulnerabilities. In order to minimize this risk, here are some
11 // rules to follow.
12 //
13 // 1. Do not introduce any new external dependencies. This file needs
14 // to be self contained so it is easy to audit everything that a
15 // trap handler might do.
16 //
17 // 2. Any changes must be reviewed by someone from the crash reporting
18 // or security team. See OWNERS for suggested reviewers.
19 //
20 // For more information, see:
21 // https://docs.google.com/document/d/17y4kxuHFrVxAiuCP_FFtFA2HP5sNPsCD10KEx17Hz6M
22 //
23 // This file contains most of the code that actually runs in a trap handler
24 // context. Some additional code is used both inside and outside the trap
25 // handler. This code can be found in handler-shared.cc.

36 // This function contains the platform independent portions of fault
37 // classification.
38 bool IsFaultAddressCovered(uintptr_t fault_addr) {
39  // TODO(eholk): broad code range check
40 
41  // Taking locks in the trap handler is risky because a fault in the trap
42  // handler itself could lead to a deadlock when attempting to acquire the
43  // lock again. We guard against this case with g_thread_in_wasm_code. The
44  // lock may only be taken when not executing Wasm code (an assert in
45  // MetadataLock's constructor ensures this). The trap handler will bail
46  // out before trying to take the lock if g_thread_in_wasm_code is not set.
47  MetadataLock lock_holder;
48 
49  for (size_t i = 0; i < gNumCodeObjects; ++i) {
50  const CodeProtectionInfo* data = gCodeObjects[i].code_info;
51  if (data == nullptr) {
52  continue;
53  }
54  const uintptr_t base = data->base;
55 
56  if (fault_addr >= base && fault_addr < base + data->size) {
57  // Hurray, we found the code object. Check for protected addresses.
58  const uint32_t offset = static_cast<uint32_t>(fault_addr - base);
59  // The offset must fit in 32 bit, see comment on
60  // ProtectedInstructionData::instr_offset.
61  TH_DCHECK(base + offset == fault_addr);
62 
63 #ifdef V8_ENABLE_DRUMBRAKE
64  // Ignore the protected instruction offsets if we are running in the Wasm
65  // interpreter.
66  if (data->num_protected_instructions == 0) {
67  gRecoveredTrapCount.store(
68  gRecoveredTrapCount.load(std::memory_order_relaxed) + 1,
69  std::memory_order_relaxed);
70  return true;
71  }
72 #endif // V8_ENABLE_DRUMBRAKE
73 
74  for (unsigned j = 0; j < data->num_protected_instructions; ++j) {
75  if (data->instructions[j].instr_offset == offset) {
76  // Hurray again, we found the actual instruction.
77  gRecoveredTrapCount.store(
78  gRecoveredTrapCount.load(std::memory_order_relaxed) + 1,
79  std::memory_order_relaxed);
80 
81  return true;
82  }
83  }
84  }
85  }
86  return false;
87 }

89 bool IsAccessedMemoryCovered(uintptr_t addr) {
90  SandboxRecordsLock lock_holder;
91 
92  // Check if the access is inside a V8 sandbox (if it is enabled) as all Wasm
93  // Memory objects must be located inside some sandbox.
94  if (gSandboxRecordsHead == nullptr) {
95  return true;
96  }
97 
98  for (SandboxRecord* current = gSandboxRecordsHead; current != nullptr;
99  current = current->next) {
100  if (addr >= current->base && addr < (current->base + current->size)) {
101  return true;
102  }
103  }
104 
105  return false;
106 }
