33 int64_t CapRelativeIndex(double relative, int64_t minimum, int64_t maximum) {
34  DCHECK(!std::isnan(relative));
35  return static_cast<int64_t>(
36  relative < 0 ? std::max<double>(relative + maximum, minimum)
37  : std::min<double>(relative, maximum));
38 }

42 BUILTIN(TypedArrayPrototypeCopyWithin) {
43  HandleScope scope(isolate);
44 
45  DirectHandle<JSTypedArray> array;
46  const char* method_name = "%TypedArray%.prototype.copyWithin";
47  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
48  isolate, array,
49  JSTypedArray::Validate(isolate, args.receiver(), method_name));
50 
51  int64_t len = array->GetLength();
52  int64_t to = 0;
53  int64_t from = 0;
54  int64_t final = len;
55 
56  if (V8_LIKELY(args.length() > 1)) {
57  double num;
58  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
59  isolate, num, Object::IntegerValue(isolate, args.at<Object>(1)));
60  to = CapRelativeIndex(num, 0, len);
61 
62  if (args.length() > 2) {
63  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
64  isolate, num, Object::IntegerValue(isolate, args.at<Object>(2)));
65  from = CapRelativeIndex(num, 0, len);
66 
67  DirectHandle<Object> end = args.atOrUndefined(isolate, 3);
68  if (!IsUndefined(*end, isolate)) {
69  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
70  isolate, num, Object::IntegerValue(isolate, end));
71  final = CapRelativeIndex(num, 0, len);
72  }
73  }
74  }
75 
76  int64_t count = std::min<int64_t>(final - from, len - to);
77  if (count <= 0) return *array;
78 
79  // TypedArray buffer may have been transferred/detached during parameter
80  // processing above.
81  if (V8_UNLIKELY(array->WasDetached())) {
82  THROW_NEW_ERROR_RETURN_FAILURE(
83  isolate, NewTypeError(MessageTemplate::kDetachedOperation,
84  isolate->factory()->NewStringFromAsciiChecked(
85  method_name)));
86  }
87 
88  if (V8_UNLIKELY(array->is_backed_by_rab())) {
89  bool out_of_bounds = false;
90  int64_t new_len = array->GetLengthOrOutOfBounds(out_of_bounds);
91  if (out_of_bounds) {
92  const MessageTemplate message = MessageTemplate::kDetachedOperation;
93  DirectHandle<String> operation =
94  isolate->factory()->NewStringFromAsciiChecked(method_name);
95  THROW_NEW_ERROR_RETURN_FAILURE(isolate, NewTypeError(message, operation));
96  }
97  if (new_len < len) {
98  // We don't need to account for growing, since we only copy an already
99  // determined number of elements and growing won't change it. If to >
100  // new_len or from > new_len, the count below will be < 0, so we don't
101  // need to check them separately.
102  if (final > new_len) {
103  final = new_len;
104  }
105  count = std::min<int64_t>(final - from, new_len - to);
106  if (count <= 0) {
107  return *array;
108  }
109  }
110  }
111 
112  // Ensure processed indexes are within array bounds
113  DCHECK_GE(from, 0);
114  DCHECK_LT(from, len);
115  DCHECK_GE(to, 0);
116  DCHECK_LT(to, len);
117  DCHECK_GE(len - count, 0);
118 
119  size_t element_size = array->element_size();
120  to = to * element_size;
121  from = from * element_size;
122  count = count * element_size;
123 
124  uint8_t* data = static_cast<uint8_t*>(array->DataPtr());
125  if (array->buffer()->is_shared()) {
126  base::Relaxed_Memmove(reinterpret_cast<base::Atomic8*>(data + to),
127  reinterpret_cast<base::Atomic8*>(data + from), count);
128  } else {
129  std::memmove(data + to, data + from, count);
130  }
131 
132  return *array;
133 }

136 BUILTIN(TypedArrayPrototypeFill) {
137  HandleScope scope(isolate);
138 
139  // 1. Let O be the this value.
140  // 2. Let taRecord be ? ValidateTypedArray(O, seq-cst).
141  DirectHandle<JSTypedArray> array;
142  const char* method_name = "%TypedArray%.prototype.fill";
143  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
144  isolate, array,
145  JSTypedArray::Validate(isolate, args.receiver(), method_name));
146  ElementsKind kind = array->GetElementsKind();
147 
148  // 3. Let len be TypedArrayLength(taRecord).
149  int64_t len = array->GetLength();
150 
151  DirectHandle<Object> obj_value = args.atOrUndefined(isolate, 1);
152  if (IsBigIntTypedArrayElementsKind(kind)) {
153  // 4. If O.[[ContentType]] is bigint, set value to ? ToBigInt(value).
154  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, obj_value,
155  BigInt::FromObject(isolate, obj_value));
156  } else {
157  // 5. Otherwise, set value to ? ToNumber(value).
158  ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, obj_value,
159  Object::ToNumber(isolate, obj_value));
160  }
161 
162  int64_t start = 0;
163  int64_t end = len;
164 
165  if (args.length() > 2) {
166  // 6. Let relativeStart be ? ToIntegerOrInfinity(start).
167  DirectHandle<Object> num = args.atOrUndefined(isolate, 2);
168  double double_num;
169  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
170  isolate, double_num, Object::IntegerValue(isolate, num));
171 
172  // 7. If relativeStart = -∞, let startIndex be 0.
173  // 8. Else if relativeStart < 0, let startIndex be max(len + relativeStart,
174  // 0).
175  // 9. Else, let startIndex be min(relativeStart, len).
176  start = CapRelativeIndex(double_num, 0, len);
177 
178  // 10. If end is undefined, let relativeEnd be len; else let relativeEnd be
179  // ? ToIntegerOrInfinity(end).
180  num = args.atOrUndefined(isolate, 3);
181  if (!IsUndefined(*num, isolate)) {
182  // 11. If relativeEnd = -∞, let endIndex be 0.
183  // 12. Else if relativeEnd < 0, let endIndex be max(len + relativeEnd, 0).
184  // 13. Else, let endIndex be min(relativeEnd, len).
185  MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
186  isolate, double_num, Object::IntegerValue(isolate, num));
187  end = CapRelativeIndex(double_num, 0, len);
188  }
189  }
190 
191  // 14. Set taRecord to MakeTypedArrayWithBufferWitnessRecord(O, seq-cst).
192  // 15. If IsTypedArrayOutOfBounds(taRecord) is true, throw a TypeError
193  // exception.
194  if (V8_UNLIKELY(array->WasDetached())) {
195  THROW_NEW_ERROR_RETURN_FAILURE(
196  isolate, NewTypeError(MessageTemplate::kDetachedOperation,
197  isolate->factory()->NewStringFromAsciiChecked(
198  method_name)));
199  }
200 
201  if (V8_UNLIKELY(array->IsVariableLength())) {
202  if (array->IsOutOfBounds()) {
203  const MessageTemplate message = MessageTemplate::kDetachedOperation;
204  DirectHandle<String> operation =
205  isolate->factory()->NewStringFromAsciiChecked(method_name);
206  THROW_NEW_ERROR_RETURN_FAILURE(isolate, NewTypeError(message, operation));
207  }
208  // 16. Set len to TypedArrayLength(taRecord).
209  // 17. Set endIndex to min(endIndex, len).
210  end = std::min(end, static_cast<int64_t>(array->GetLength()));
211  }
212 
213  int64_t count = end - start;
214  if (count <= 0) return *array;
215 
216  // Ensure processed indexes are within array bounds
217  DCHECK_GE(start, 0);
218  DCHECK_LT(start, len);
219  DCHECK_GE(end, 0);
220  DCHECK_LE(end, len);
221  DCHECK_LE(count, len);
222 
223  // 19. Repeat, while k < endIndex,
224  // a. Let Pk be ! ToString(𝔽(k)).
225  // b. Perform ! Set(O, Pk, value, true).
226  // c. Set k to k + 1.
227  // 20. Return O.
228  RETURN_RESULT_OR_FAILURE(isolate, ElementsAccessor::ForKind(kind)->Fill(
229  isolate, array, obj_value, start, end));
230 }

364 template <typename T>
365 Maybe<T> MapOptionToEnum(
366  Isolate* isolate, DirectHandle<String> option_string,
367  const std::vector<std::tuple<const char*, size_t, T>>& allowed_options) {
368  option_string = String::Flatten(isolate, option_string);
369 
370  {
371  DisallowGarbageCollection no_gc;
372  String::FlatContent option_content = option_string->GetFlatContent(no_gc);
373 
374  if (option_content.IsOneByte()) {
375  const unsigned char* option_string_to_compare =
376  option_content.ToOneByteVector().data();
377  size_t length = option_content.ToOneByteVector().size();
378 
379  for (auto& [str_val, str_size, enum_val] : allowed_options) {
380  if (str_size == length &&
381  CompareCharsEqual(option_string_to_compare, str_val, str_size)) {
382  return Just<T>(enum_val);
383  }
384  }
385  } else {
386  const base::uc16* option_string_to_compare =
387  option_content.ToUC16Vector().data();
388  size_t length = option_content.ToUC16Vector().size();
389 
390  for (auto& [str_val, str_size, enum_val] : allowed_options) {
391  if (str_size == length &&
392  CompareCharsEqual(option_string_to_compare, str_val, str_size)) {
393  return Just<T>(enum_val);
394  }
395  }
396  }
397  }
398 
399  isolate->Throw(*isolate->factory()->NewTypeError(
400  MessageTemplate::kInvalidOption, option_string));
401  return Nothing<T>();
402 }

404 Maybe<std::pair<simdutf::base64_options, simdutf::last_chunk_handling_options>>
405 HandleOptionsBag(Isolate* isolate, DirectHandle<Object> options) {
406  simdutf::base64_options alphabet;
407  simdutf::last_chunk_handling_options last_chunk_handling;
408 
409  // 1. Let alphabet be ? Get(opts, "alphabet").
410  DirectHandle<Object> opt_alphabet;
411  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
412  isolate, opt_alphabet,
413  JSObject::ReadFromOptionsBag(
414  options, isolate->factory()->alphabet_string(), isolate),
415  (Nothing<std::pair<simdutf::base64_options,
416  simdutf::last_chunk_handling_options>>()));
417 
418  // 2. If alphabet is undefined, set alphabet to "base64".
419  if (IsUndefined(*opt_alphabet)) {
420  alphabet = simdutf::base64_default;
421  } else if (!IsString(*opt_alphabet)) {
422  isolate->Throw(*isolate->factory()->NewTypeError(
423  MessageTemplate::kInvalidOption, opt_alphabet));
424  return Nothing<std::pair<simdutf::base64_options,
425  simdutf::last_chunk_handling_options>>();
426  } else {
427  // 3. If alphabet is neither "base64" nor "base64url", throw a TypeError
428  // exception.
429  DirectHandle<String> alphabet_string = Cast<String>(opt_alphabet);
430  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
431  isolate, alphabet,
432  MapOptionToEnum(isolate, alphabet_string, SimdutfBase64OptionsVector()),
433  (Nothing<std::pair<simdutf::base64_options,
434  simdutf::last_chunk_handling_options>>()));
435  }
436 
437  // 4. Let lastChunkHandling be ? Get(opts, "lastChunkHandling").
438  DirectHandle<Object> opt_last_chunk_handling;
439  ASSIGN_RETURN_ON_EXCEPTION_VALUE(
440  isolate, opt_last_chunk_handling,
441  JSObject::ReadFromOptionsBag(
442  options, isolate->factory()->last_chunk_handling_string(), isolate),
443  (Nothing<std::pair<simdutf::base64_options,
444  simdutf::last_chunk_handling_options>>()));
445 
446  // 5. If lastChunkHandling is undefined, set lastChunkHandling to "loose".
447  if (IsUndefined(*opt_last_chunk_handling)) {
448  last_chunk_handling = simdutf::last_chunk_handling_options::loose;
449  } else if (!IsString(*opt_last_chunk_handling)) {
450  isolate->Throw(*isolate->factory()->NewTypeError(
451  MessageTemplate::kInvalidOption, opt_last_chunk_handling));
452  return Nothing<std::pair<simdutf::base64_options,
453  simdutf::last_chunk_handling_options>>();
454  } else {
455  // 6. If lastChunkHandling is not one of "loose", "strict", or
456  // "stop-before-partial", throw a TypeError exception.
457  DirectHandle<String> last_chunk_handling_string =
458  Cast<String>(opt_last_chunk_handling);
459 
460  MAYBE_ASSIGN_RETURN_ON_EXCEPTION_VALUE(
461  isolate, last_chunk_handling,
462  MapOptionToEnum(isolate, last_chunk_handling_string,
463  SimdutfLastChunkHandlingOptionsVector()),
464  (Nothing<std::pair<simdutf::base64_options,
465  simdutf::last_chunk_handling_options>>()));
466  }
467  return Just(std::make_pair(alphabet, last_chunk_handling));
468 }
