184 // | | 0
185 // | | 1
186 // | | 2
187 // | | ...
188 // | | 31
189 // VSX |
190 // | | 32
191 // | | 33
192 // | VMX | 34
193 // | | ...
194 // | | 63
195 //
196 // VSX registers (0 to 63) can be used by VSX vector instructions, which are
197 // mainly focused on Floating Point arithmetic. They do have few Integer
198 // Instructions such as logical operations, merge and select. The main Simd
199 // integer instructions such as add/sub/mul/ extract_lane/replace_lane,
200 // comparisons etc. are only available with VMX instructions and can only access
201 // the VMX set of vector registers (which is a subset of VSX registers). So to
202 // assure access to all Simd instructions in V8 and avoid moving data between
203 // registers, we are only using the upper 32 registers (VMX set) for Simd
204 // operations and only use the lower set for scalar (non simd) floating point
205 // operations which makes our Simd register set separate from Floating Point
206 // ones.
207 enum Simd128RegisterCode {
208 #define REGISTER_CODE(R) kSimd128Code_##R,
209  SIMD128_REGISTERS(REGISTER_CODE)
210 #undef REGISTER_CODE
211  kSimd128AfterLast
212 };
