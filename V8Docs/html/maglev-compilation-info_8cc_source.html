49 static bool SpecializeToFunctionContext(
50  Isolate* isolate, BytecodeOffset osr_offset,
51  DirectHandle<JSFunction> function,
52  std::optional<bool> specialize_to_function_context_override) {
53  if (osr_offset != BytecodeOffset::None()) return false;
54  if (!v8_flags.maglev_function_context_specialization) return false;
55  if (specialize_to_function_context_override.has_value()) {
56  return specialize_to_function_context_override.value();
57  }
58  if (function->shared()->function_context_independent_compiled()) {
59  return false;
60  }
61  return function->raw_feedback_cell()->map() ==
62  ReadOnlyRoots(isolate).one_closure_cell_map();
63 }

67 MaglevCompilationInfo::MaglevCompilationInfo(
68  Isolate* isolate, IndirectHandle<JSFunction> function,
69  BytecodeOffset osr_offset, std::optional<compiler::JSHeapBroker*> js_broker,
70  std::optional<bool> specialize_to_function_context,
71  bool for_turboshaft_frontend)
72  : zone_(isolate->allocator(), kMaglevZoneName),
73  broker_(js_broker.has_value()
74  ? js_broker.value()
75  : new compiler::JSHeapBroker(isolate, zone(),
76  v8_flags.trace_heap_broker,
77  CodeKind::MAGLEV)),
78  toplevel_function_(function),
79  osr_offset_(osr_offset),
80  owns_broker_(!js_broker.has_value()),
81  is_turbolev_(for_turboshaft_frontend)
82 #define V(Name) , Name##_(v8_flags.Name)
83  MAGLEV_COMPILATION_FLAG_LIST(V)
84 #undef V
85  ,
86  specialize_to_function_context_(SpecializeToFunctionContext(
87  isolate, osr_offset, function, specialize_to_function_context)) {
88  if (owns_broker_) {
89  canonical_handles_ = std::make_unique<CanonicalHandlesMap>(
90  isolate->heap(), ZoneAllocationPolicy(&zone_));
91  compiler::CurrentHeapBrokerScope current_broker(broker_);
92 
93  MaglevCompilationHandleScope compilation(isolate, this);
94 
95  compiler::CompilationDependencies* deps =
96  zone()->New<compiler::CompilationDependencies>(broker(), zone());
97  USE(deps); // The deps register themselves in the heap broker.
98 
99  broker()->AttachCompilationInfo(this);
100 
101  // Heap broker initialization may already use IsPendingAllocation.
102  isolate->heap()->PublishMainThreadPendingAllocations();
103  broker()->InitializeAndStartSerializing(
104  direct_handle(function->native_context(), isolate));
105  broker()->StopSerializing();
106 
107  // Serialization may have allocated.
108  isolate->heap()->PublishMainThreadPendingAllocations();
109 
110  toplevel_compilation_unit_ =
111  MaglevCompilationUnit::New(zone(), this, function);
112  } else {
113  toplevel_compilation_unit_ =
114  MaglevCompilationUnit::New(zone(), this, function);
115  }
116 
117  collect_source_positions_ = isolate->NeedsDetailedOptimizedCodeLineInfo();
118 }

139 template <typename T>
140 IndirectHandle<T> CanonicalHandle(CanonicalHandlesMap* canonical_handles,
141  Tagged<T> object, Isolate* isolate) {
142  DCHECK_NOT_NULL(canonical_handles);
143  DCHECK(PersistentHandlesScope::IsActive(isolate));
144  auto find_result = canonical_handles->FindOrInsert(object);
145  if (!find_result.already_exists) {
146  *find_result.entry = IndirectHandle<T>(object, isolate).location();
147  }
148  return IndirectHandle<T>(*find_result.entry);
149 }
