70 class YoungGenerationMarkingVerifier : public MarkingVerifierBase {
71  public:
72  explicit YoungGenerationMarkingVerifier(Heap* heap)
73  : MarkingVerifierBase(heap),
74  marking_state_(heap->non_atomic_marking_state()) {}
75 
76  const MarkingBitmap* bitmap(const MutablePageMetadata* chunk) override {
77  return chunk->marking_bitmap();
78  }
79 
80  bool IsMarked(Tagged<HeapObject> object) override {
81  return marking_state_->IsMarked(object);
82  }
83 
84  void Run() override {
85  // VerifyRoots will visit also visit the conservative stack and consider
86  // objects reachable from it, including old objects. This is fine since this
87  // verifier will only check that young objects are marked.
88  VerifyRoots();
89  if (v8_flags.sticky_mark_bits) {
90  VerifyMarking(heap_->sticky_space());
91  } else {
92  VerifyMarking(heap_->new_space());
93  }
94  }
95 
96  GarbageCollector collector() const override {
97  return GarbageCollector::MINOR_MARK_SWEEPER;
98  }
99 
100  protected:
101  void VerifyMap(Tagged<Map> map) override { VerifyHeapObjectImpl(map); }
102 
103  void VerifyPointers(ObjectSlot start, ObjectSlot end) override {
104  VerifyPointersImpl(start, end);
105  }
106 
107  void VerifyPointers(MaybeObjectSlot start, MaybeObjectSlot end) override {
108  VerifyPointersImpl(start, end);
109  }
110  void VerifyCodePointer(InstructionStreamSlot slot) override {
111  // Code slots never appear in new space because
112  // Code objects, the only object that can contain code pointers, are
113  // always allocated in the old space.
114  UNREACHABLE();
115  }
116 
117  void VisitCodeTarget(Tagged<InstructionStream> host,
118  RelocInfo* rinfo) override {
119  Tagged<InstructionStream> target =
120  InstructionStream::FromTargetAddress(rinfo->target_address());
121  VerifyHeapObjectImpl(target);
122  }
123  void VisitEmbeddedPointer(Tagged<InstructionStream> host,
124  RelocInfo* rinfo) override {
125  VerifyHeapObjectImpl(rinfo->target_object(cage_base()));
126  }
127  void VerifyRootPointers(FullObjectSlot start, FullObjectSlot end) override {
128  VerifyPointersImpl(start, end);
129  }
130 
131  private:
132  V8_INLINE void VerifyHeapObjectImpl(Tagged<HeapObject> heap_object) {
133  CHECK_IMPLIES(HeapLayout::InYoungGeneration(heap_object),
134  IsMarked(heap_object));
135  }
136 
137  template <typename TSlot>
138  V8_INLINE void VerifyPointersImpl(TSlot start, TSlot end) {
139  PtrComprCageBase cage_base =
140  GetPtrComprCageBaseFromOnHeapAddress(start.address());
141  for (TSlot slot = start; slot < end; ++slot) {
142  typename TSlot::TObject object = slot.load(cage_base);
143 #ifdef V8_ENABLE_DIRECT_HANDLE
144  if (object.ptr() == kTaggedNullAddress) continue;
145 #endif
146  Tagged<HeapObject> heap_object;
147  // Minor MS treats weak references as strong.
148  if (object.GetHeapObject(&heap_object)) {
149  VerifyHeapObjectImpl(heap_object);
150  }
151  }
152  }
153 
154  NonAtomicMarkingState* const marking_state_;
155 };

165 int EstimateMaxNumberOfRemeberedSets(Heap* heap) {
166  // old space, lo space, trusted space and trusted lo space can have a maximum
167  // of two remembered sets (OLD_TO_NEW and OLD_TO_NEW_BACKGROUND).
168  // Code space and code lo space can have typed OLD_TO_NEW in addition.
169  return 2 * (heap->old_space()->CountTotalPages() +
170  heap->lo_space()->PageCount() +
171  heap->trusted_space()->CountTotalPages() +
172  heap->trusted_lo_space()->PageCount()) +
173  3 * (heap->code_space()->CountTotalPages() +
174  heap->code_lo_space()->PageCount());
175 }

179 std::vector<YoungGenerationRememberedSetsMarkingWorklist::MarkingItem>
180 YoungGenerationRememberedSetsMarkingWorklist::CollectItems(Heap* heap) {
181  std::vector<MarkingItem> items;
182  int max_remembered_set_count = EstimateMaxNumberOfRemeberedSets(heap);
183  items.reserve(max_remembered_set_count);
184  OldGenerationMemoryChunkIterator::ForAll(
185  heap, [&items](MutablePageMetadata* chunk) {
186  SlotSet* slot_set = chunk->ExtractSlotSet<OLD_TO_NEW>();
187  SlotSet* background_slot_set =
188  chunk->ExtractSlotSet<OLD_TO_NEW_BACKGROUND>();
189  if (slot_set || background_slot_set) {
190  items.emplace_back(chunk, MarkingItem::SlotsType::kRegularSlots,
191  slot_set, background_slot_set);
192  }
193  if (TypedSlotSet* typed_slot_set =
194  chunk->ExtractTypedSlotSet<OLD_TO_NEW>()) {
195  DCHECK(IsAnyCodeSpace(chunk->owner_identity()));
196  items.emplace_back(chunk, MarkingItem::SlotsType::kTypedSlots,
197  typed_slot_set);
198  }
199  });
200  DCHECK_LE(items.size(), max_remembered_set_count);
201  return items;
202 }

204 void YoungGenerationRememberedSetsMarkingWorklist::MarkingItem::
205  MergeAndDeleteRememberedSets() {
206  DCHECK(IsAcquired());
207  if (slots_type_ == SlotsType::kRegularSlots) {
208  if (slot_set_)
209  RememberedSet<OLD_TO_NEW>::MergeAndDelete(chunk_, std::move(*slot_set_));
210  if (background_slot_set_)
211  RememberedSet<OLD_TO_NEW_BACKGROUND>::MergeAndDelete(
212  chunk_, std::move(*background_slot_set_));
213  } else {
214  DCHECK_EQ(slots_type_, SlotsType::kTypedSlots);
215  DCHECK_NULL(background_slot_set_);
216  if (typed_slot_set_)
217  RememberedSet<OLD_TO_NEW>::MergeAndDeleteTyped(
218  chunk_, std::move(*typed_slot_set_));
219  }
220 }

222 void YoungGenerationRememberedSetsMarkingWorklist::MarkingItem::
223  DeleteRememberedSets() {
224  DCHECK(IsAcquired());
225  if (slots_type_ == SlotsType::kRegularSlots) {
226  if (slot_set_) SlotSet::Delete(slot_set_);
227  if (background_slot_set_) SlotSet::Delete(background_slot_set_);
228  } else {
229  DCHECK_EQ(slots_type_, SlotsType::kTypedSlots);
230  DCHECK_NULL(background_slot_set_);
231  if (typed_slot_set_)
232  RememberedSet<OLD_TO_NEW>::DeleteTyped(std::move(*typed_slot_set_));
233  }

344 void MinorMarkSweepCollector::StartMarking(bool force_use_background_threads) {
345 #if defined(VERIFY_HEAP) && !V8_ENABLE_STICKY_MARK_BITS_BOOL
346  if (v8_flags.verify_heap) {
347  for (PageMetadata* page : *heap_->new_space()) {
348  CHECK(page->marking_bitmap()->IsClean());
349  }
350  }
351 #endif // defined(VERIFY_HEAP) && !V8_ENABLE_STICKY_MARK_BITS_BOOL
352 
353  // The state for background thread is saved here and maintained for the whole
354  // GC cycle. Both CppHeap and regular V8 heap will refer to this flag.
355  CHECK(!use_background_threads_in_cycle_.has_value());
356  // Once we decided to start concurrent marking we always need to use
357  // background threads, this is because Minor MS doesn't perform incremental
358  // marking. ShouldUseBackgroundThreads() on worker isolates can be updated
359  // concurrently from the main thread outside a task, so we shouldn't invoke it
360  // here again as it could return a different result.
361  use_background_threads_in_cycle_ =
362  force_use_background_threads || heap_->ShouldUseBackgroundThreads();
363 
364  auto* cpp_heap = CppHeap::From(heap_->cpp_heap_);
365  // CppHeap's marker must be initialized before the V8 marker to allow
366  // exchanging of worklists.
367  if (cpp_heap && cpp_heap->generational_gc_supported()) {
368  TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_MARK_EMBEDDER_PROLOGUE);
369  cpp_heap->InitializeMarking(CppHeap::CollectionType::kMinor);
370  }
371  DCHECK_NULL(ephemeron_table_list_);
372  ephemeron_table_list_ = std::make_unique<EphemeronRememberedSet::TableList>();
373  DCHECK_NULL(marking_worklists_);
374  marking_worklists_ = std::make_unique<MarkingWorklists>();
375  DCHECK_NULL(main_marking_visitor_);
376  DCHECK_NULL(pretenuring_feedback_);
377  if (v8_flags.sticky_mark_bits) {
378  DCHECK(ExternalPointerRememberedSetsEmpty(heap_->sticky_space()));
379  } else {
380  DCHECK(ExternalPointerRememberedSetsEmpty(
381  heap_->paged_new_space()->paged_space()));
382  }
383  pretenuring_feedback_ =
384  std::make_unique<PretenuringHandler::PretenuringFeedbackMap>(
385  PretenuringHandler::kInitialFeedbackCapacity);
386  main_marking_visitor_ = std::make_unique<YoungGenerationMainMarkingVisitor>(
387  heap_, pretenuring_feedback_.get());
388  DCHECK_NULL(remembered_sets_marking_handler_);
389  remembered_sets_marking_handler_ =
390  std::make_unique<YoungGenerationRememberedSetsMarkingWorklist>(heap_);
391  if (cpp_heap && cpp_heap->generational_gc_supported()) {
392  TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_MARK_EMBEDDER_PROLOGUE);
393  // StartTracing immediately starts marking which requires V8 worklists to
394  // be set up.
395  cpp_heap->StartMarking();
396  }
397 }

414 void MinorMarkSweepCollector::CollectGarbage() {
415  DCHECK(!heap_->mark_compact_collector()->in_use());
416  DCHECK(heap_->use_new_space());
417  DCHECK(!heap_->array_buffer_sweeper()->sweeping_in_progress());
418  DCHECK(!sweeper()->AreMinorSweeperTasksRunning());
419  if (v8_flags.sticky_mark_bits) {
420  DCHECK(sweeper()->IsSweepingDoneForSpace(OLD_SPACE));
421  } else {
422  DCHECK(sweeper()->IsSweepingDoneForSpace(NEW_SPACE));
423  }
424 
425  heap_->new_lo_space()->ResetPendingObject();
426 
427  is_in_atomic_pause_.store(true, std::memory_order_relaxed);
428 
429  MarkLiveObjects();
430  if (auto* cpp_heap = CppHeap::From(heap_->cpp_heap_)) {
431  cpp_heap->ProcessCrossThreadWeakness();
432  }
433  ClearNonLiveReferences();
434 #ifdef VERIFY_HEAP
435  if (v8_flags.verify_heap) {
436  TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_MARK_VERIFY);
437  YoungGenerationMarkingVerifier verifier(heap_);
438  verifier.Run();
439  }
440 #endif // VERIFY_HEAP
441 
442  if (auto* cpp_heap = CppHeap::From(heap_->cpp_heap_)) {
443  cpp_heap->FinishMarkingAndProcessWeakness();
444  }
445 
446  Sweep();
447  Finish();
448 
449  auto* isolate = heap_->isolate();
450  isolate->global_handles()->UpdateListOfYoungNodes();
451  isolate->traced_handles()->UpdateListOfYoungNodes();
452 
453  isolate->stack_guard()->ClearGC();
454  gc_finalization_requested_.store(false, std::memory_order_relaxed);
455  is_in_atomic_pause_.store(false, std::memory_order_relaxed);
456 }

460 class YoungStringForwardingTableCleaner final
461  : public StringForwardingTableCleanerBase {
462  public:
463  explicit YoungStringForwardingTableCleaner(Heap* heap)
464  : StringForwardingTableCleanerBase(heap) {}

466  // For Minor MS we don't mark forward objects, because they are always
467  // in old generation (and thus considered live).
468  // We only need to delete non-live young objects.
469  void ProcessYoungObjects() {
470  DCHECK(v8_flags.always_use_string_forwarding_table);
471  StringForwardingTable* forwarding_table =
472  isolate_->string_forwarding_table();
473  forwarding_table->IterateElements(
474  [&](StringForwardingTable::Record* record) {
475  ClearNonLiveYoungObjects(record);
476  });
477  }
478 
479  private:
480  void ClearNonLiveYoungObjects(StringForwardingTable::Record* record) {
481  Tagged<Object> original = record->OriginalStringObject(isolate_);
482  if (!IsHeapObject(original)) {
483  DCHECK_EQ(original, StringForwardingTable::deleted_element());
484  return;
485  }
486  Tagged<String> original_string = Cast<String>(original);
487  if (!HeapLayout::InYoungGeneration(original_string)) return;
488  if (!marking_state_->IsMarked(original_string)) {
489  DisposeExternalResource(record);
490  record->set_original_string(StringForwardingTable::deleted_element());
491  }
492  }
493 };

506 void MinorMarkSweepCollector::ClearNonLiveReferences() {
507  TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_CLEAR);
508 
509  if (V8_UNLIKELY(v8_flags.always_use_string_forwarding_table)) {
510  TRACE_GC(heap_->tracer(),
511  GCTracer::Scope::MINOR_MS_CLEAR_STRING_FORWARDING_TABLE);
512  // Clear non-live objects in the string fowarding table.
513  YoungStringForwardingTableCleaner forwarding_table_cleaner(heap_);
514  forwarding_table_cleaner.ProcessYoungObjects();
515  }
516 
517  Heap::ExternalStringTable& external_string_table =
518  heap_->external_string_table_;
519  if (external_string_table.HasYoung()) {
520  TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_CLEAR_STRING_TABLE);
521  // Internalized strings are always stored in old space, so there is no
522  // need to clean them here.
523  ExternalStringTableCleanerVisitor<
524  ExternalStringTableCleaningMode::kYoungOnly>
525  external_visitor(heap_);
526  external_string_table.IterateYoung(&external_visitor);
527  external_string_table.CleanUpYoung();
528  }
529 
530  Isolate* isolate = heap_->isolate();
531  if (isolate->global_handles()->HasYoung() ||
532  isolate->traced_handles()->HasYoung()) {
533  TRACE_GC(heap_->tracer(),
534  GCTracer::Scope::MINOR_MS_CLEAR_WEAK_GLOBAL_HANDLES);
535  isolate->global_handles()->ProcessWeakYoungObjects(
536  nullptr, &IsUnmarkedObjectInYoungGeneration);
537  if (auto* cpp_heap = CppHeap::From(heap_->cpp_heap_);
538  cpp_heap && cpp_heap->generational_gc_supported()) {
539  isolate->traced_handles()->ResetYoungDeadNodes(
540  &IsUnmarkedObjectInYoungGeneration);
541  } else {
542  isolate->traced_handles()->ProcessWeakYoungObjects(
543  nullptr, &IsUnmarkedObjectInYoungGeneration);
544  }
545  }
546 
547  // Clear ephemeron entries from EphemeronHashTables in the young generation
548  // whenever the entry has a dead young generation key.
549  //
550  // Worklist is collected during marking.
551  {
552  DCHECK_NOT_NULL(ephemeron_table_list_);
553  EphemeronRememberedSet::TableList::Local local_ephemeron_table_list(
554  *ephemeron_table_list_);
555  Tagged<EphemeronHashTable> table;
556  while (local_ephemeron_table_list.Pop(&table)) {
557  for (InternalIndex i : table->IterateEntries()) {
558  // Keys in EphemeronHashTables must be heap objects.
559  HeapObjectSlot key_slot(
560  table->RawFieldOfElementAt(EphemeronHashTable::EntryToIndex(i)));
561  Tagged<HeapObject> key = key_slot.ToHeapObject();
562  if (HeapLayout::InYoungGeneration(key) &&
563  non_atomic_marking_state_->IsUnmarked(key)) {
564  table->RemoveEntry(i);
565  }
566  }
567  }
568  }
569  ephemeron_table_list_.reset();
570 
571  // Clear ephemeron entries from EphemeronHashTables in the old generation
572  // whenever the entry has a dead young generation key.
573  //
574  // Does not need to be iterated as roots but is maintained in the GC to avoid
575  // treating keys as strong. The set is populated from the write barrier and
576  // the sweeper during promoted pages iteration.
577  auto* table_map = heap_->ephemeron_remembered_set()->tables();

846  free_bytes_of_class[free_bytes_index] += free_bytes;

848  free_bytes_index++;
849  }

873 bool ShouldMovePage(PageMetadata* p, intptr_t live_bytes,
874  intptr_t wasted_bytes) {
875  DCHECK(v8_flags.page_promotion);
876  DCHECK(!v8_flags.sticky_mark_bits);
877  Heap* heap = p->heap();
878  DCHECK(!p->Chunk()->NeverEvacuate());
879  const bool should_move_page =
880  ((live_bytes + wasted_bytes) > NewSpacePageEvacuationThreshold() ||
881  (p->AllocatedLabSize() == 0)) &&
882  (heap->new_space()->IsPromotionCandidate(p)) &&
883  heap->CanExpandOldGeneration(live_bytes);
884  if (v8_flags.trace_page_promotions) {
885  PrintIsolate(
886  heap->isolate(),
887  "[Page Promotion] %p: collector=mms, should move: %d"
888  ", live bytes = %zu, wasted bytes = %zu, promotion threshold = %zu"
889  ", allocated labs size = %zu\n",
890  p, should_move_page, live_bytes, wasted_bytes,
891  NewSpacePageEvacuationThreshold(), p->AllocatedLabSize());
892  }
893  if (!should_move_page &&
894  (p->AgeInNewSpace() == v8_flags.minor_ms_max_page_age)) {
895  // Don't allocate on old pages so that recently allocated objects on the
896  // page get a chance to die young. The page will be force promoted on the
897  // next GC because `AllocatedLabSize` will be 0.
898  p->Chunk()->SetFlagNonExecutable(MemoryChunk::NEVER_ALLOCATE_ON_PAGE);
899  }
900  return should_move_page;
901 }
