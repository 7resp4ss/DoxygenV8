49 bool Scavenger::MigrateObject(Tagged<Map> map, Tagged<HeapObject> source,
50  Tagged<HeapObject> target, int size,
51  PromotionHeapChoice promotion_heap_choice) {
52  // This CAS can be relaxed because we do not access the object body if the
53  // object was already copied by another thread. We only access the page header
54  // of such objects and this is safe because of the memory fence after page
55  // header initialization.
56  if (!source->relaxed_compare_and_swap_map_word_forwarded(
57  MapWord::FromMap(map), target)) {
58  // Other task migrated the object.
59  return false;
60  }
61 
62  // Copy the content of source to target. Note that we do this on purpose
63  // *after* the CAS. This avoids copying of the object in the (unlikely)
64  // failure case. It also helps us to ensure that we do not rely on non-relaxed
65  // memory ordering for the CAS above.
66  target->set_map_word(map, kRelaxedStore);
67  heap()->CopyBlock(target.address() + kTaggedSize,
68  source.address() + kTaggedSize, size - kTaggedSize);
69 
70  if (V8_UNLIKELY(is_logging_)) {
71  heap()->OnMoveEvent(source, target, size);
72  }
73 
74  PretenuringHandler::UpdateAllocationSite(heap_, map, source, size,
75  &local_pretenuring_feedback_);
76 
77  return true;
78 }

80 template <typename THeapObjectSlot>
81 CopyAndForwardResult Scavenger::SemiSpaceCopyObject(
82  Tagged<Map> map, THeapObjectSlot slot, Tagged<HeapObject> object,
83  int object_size, ObjectFields object_fields) {
84  static_assert(std::is_same_v<THeapObjectSlot, FullHeapObjectSlot> ||
85  std::is_same_v<THeapObjectSlot, HeapObjectSlot>,
86  "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
87  DCHECK(heap()->AllowedToBeMigrated(map, object, NEW_SPACE));
88  AllocationAlignment alignment = HeapObject::RequiredAlignment(NEW_SPACE, map);
89  AllocationResult allocation =
90  allocator_.Allocate(NEW_SPACE, object_size, alignment);
91 
92  Tagged<HeapObject> target;
93  if (allocation.To(&target)) {
94  DCHECK(heap()->marking_state()->IsUnmarked(target));
95  const bool self_success =
96  MigrateObject(map, object, target, object_size, kPromoteIntoLocalHeap);
97  if (!self_success) {
98  allocator_.FreeLast(NEW_SPACE, target, object_size);
99  MapWord map_word = object->map_word(kRelaxedLoad);
100  UpdateHeapObjectReferenceSlot(slot, map_word.ToForwardingAddress(object));
101  SynchronizePageAccess(*slot);
102  DCHECK(!Heap::InFromPage(*slot));
103  return Heap::InToPage(*slot)
104  ? CopyAndForwardResult::SUCCESS_YOUNG_GENERATION
105  : CopyAndForwardResult::SUCCESS_OLD_GENERATION;
106  }
107  UpdateHeapObjectReferenceSlot(slot, target);
108  if (object_fields == ObjectFields::kMaybePointers) {
109  local_copied_list_.Push(target);
110  }
111  copied_size_ += object_size;
112  return CopyAndForwardResult::SUCCESS_YOUNG_GENERATION;
113  }
114  return CopyAndForwardResult::FAILURE;
115 }

117 template <typename THeapObjectSlot,
118  Scavenger::PromotionHeapChoice promotion_heap_choice>
119 CopyAndForwardResult Scavenger::PromoteObject(Tagged<Map> map,
120  THeapObjectSlot slot,
121  Tagged<HeapObject> object,
122  int object_size,
123  ObjectFields object_fields) {
124  static_assert(std::is_same_v<THeapObjectSlot, FullHeapObjectSlot> ||
125  std::is_same_v<THeapObjectSlot, HeapObjectSlot>,
126  "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
127  DCHECK_GE(object_size, Heap::kMinObjectSizeInTaggedWords * kTaggedSize);
128  AllocationSpace target_space =
129  promotion_heap_choice == kPromoteIntoLocalHeap ? OLD_SPACE : SHARED_SPACE;
130  AllocationAlignment alignment =
131  HeapObject::RequiredAlignment(target_space, map);
132  AllocationResult allocation =
133  allocator_.Allocate(target_space, object_size, alignment);
134 
135  Tagged<HeapObject> target;
136  if (allocation.To(&target)) {
137  DCHECK(heap()->non_atomic_marking_state()->IsUnmarked(target));
138  const bool self_success =
139  MigrateObject(map, object, target, object_size, promotion_heap_choice);
140  if (!self_success) {
141  allocator_.FreeLast(target_space, target, object_size);
142 
143  MapWord map_word = object->map_word(kRelaxedLoad);
144  UpdateHeapObjectReferenceSlot(slot, map_word.ToForwardingAddress(object));
145  SynchronizePageAccess(*slot);
146  DCHECK(!Heap::InFromPage(*slot));
147  return Heap::InToPage(*slot)
148  ? CopyAndForwardResult::SUCCESS_YOUNG_GENERATION
149  : CopyAndForwardResult::SUCCESS_OLD_GENERATION;
150  }
151  UpdateHeapObjectReferenceSlot(slot, target);
152 
153  if (object_fields == ObjectFields::kMaybePointers) {
154  local_promoted_list_.Push({target, map, object_size});
155  }
156  promoted_size_ += object_size;
157  return CopyAndForwardResult::SUCCESS_OLD_GENERATION;
158  }
159  return CopyAndForwardResult::FAILURE;
160 }

169 bool Scavenger::HandleLargeObject(Tagged<Map> map, Tagged<HeapObject> object,
170  int object_size, ObjectFields object_fields) {
171  // Quick check first: A large object is the first (and only) object on a data
172  // page (code pages have a different offset but are never young). This check
173  // avoids the page flag access.
174  if (MemoryChunk::AddressToOffset(object.address()) !=
175  MemoryChunkLayout::ObjectStartOffsetInDataPage()) {
176  return false;
177  }
178 
179  // Most objects are already filtered out. Use page flags to filter out regular
180  // objects at the beginning of a page.
181  if (!MemoryChunk::FromHeapObject(object)->InNewLargeObjectSpace()) {
182  return false;
183  }
184 
185  DCHECK_EQ(NEW_LO_SPACE,
186  MutablePageMetadata::FromHeapObject(object)->owner_identity());
187  if (object->relaxed_compare_and_swap_map_word_forwarded(MapWord::FromMap(map),
188  object)) {
189  local_surviving_new_large_objects_.insert({object, map});
190  promoted_size_ += object_size;
191  if (object_fields == ObjectFields::kMaybePointers) {
192  local_promoted_list_.Push({object, map, object_size});
193  }
194  }
195  return true;
196 }

198 template <typename THeapObjectSlot,
199  Scavenger::PromotionHeapChoice promotion_heap_choice>
200 SlotCallbackResult Scavenger::EvacuateObjectDefault(
201  Tagged<Map> map, THeapObjectSlot slot, Tagged<HeapObject> object,
202  int object_size, ObjectFields object_fields) {
203  static_assert(std::is_same_v<THeapObjectSlot, FullHeapObjectSlot> ||
204  std::is_same_v<THeapObjectSlot, HeapObjectSlot>,
205  "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
206  SLOW_DCHECK(object->SizeFromMap(map) == object_size);
207  CopyAndForwardResult result;
208 
209  if (HandleLargeObject(map, object, object_size, object_fields)) {
210  return REMOVE_SLOT;
211  }
212 
213  SLOW_DCHECK(static_cast<size_t>(object_size) <=
214  MemoryChunkLayout::AllocatableMemoryInDataPage());
215 
216  if (!heap()->semi_space_new_space()->ShouldBePromoted(object.address())) {
217  // A semi-space copy may fail due to fragmentation. In that case, we
218  // try to promote the object.
219  result = SemiSpaceCopyObject(map, slot, object, object_size, object_fields);
220  if (result != CopyAndForwardResult::FAILURE) {
221  return RememberedSetEntryNeeded(result);
222  }
223  }
224 
225  // We may want to promote this object if the object was already semi-space
226  // copied in a previous young generation GC or if the semi-space copy above
227  // failed.
228  result = PromoteObject<THeapObjectSlot, promotion_heap_choice>(
229  map, slot, object, object_size, object_fields);
230  if (result != CopyAndForwardResult::FAILURE) {
231  return RememberedSetEntryNeeded(result);
232  }
233 
234  // If promotion failed, we try to copy the object to the other semi-space.
235  result = SemiSpaceCopyObject(map, slot, object, object_size, object_fields);
236  if (result != CopyAndForwardResult::FAILURE) {
237  return RememberedSetEntryNeeded(result);
238  }
239 
240  heap()->FatalProcessOutOfMemory("Scavenger: semi-space copy");
241  UNREACHABLE();
242 }

244 template <typename THeapObjectSlot>
245 SlotCallbackResult Scavenger::EvacuateThinString(Tagged<Map> map,
246  THeapObjectSlot slot,
247  Tagged<ThinString> object,
248  int object_size) {
249  static_assert(std::is_same_v<THeapObjectSlot, FullHeapObjectSlot> ||
250  std::is_same_v<THeapObjectSlot, HeapObjectSlot>,
251  "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
252  if (shortcut_strings_) {
253  // The ThinString should die after Scavenge, so avoid writing the proper
254  // forwarding pointer and instead just signal the actual object as forwarded
255  // reference.
256  Tagged<String> actual = object->actual();
257  // ThinStrings always refer to internalized strings, which are always in old
258  // space.
259  DCHECK(!HeapLayout::InYoungGeneration(actual));
260  UpdateHeapObjectReferenceSlot(slot, actual);
261  return REMOVE_SLOT;
262  }
263 
264  DCHECK_EQ(ObjectFields::kMaybePointers,
265  Map::ObjectFieldsFrom(map->visitor_id()));
266  return EvacuateObjectDefault(map, slot, object, object_size,
267  ObjectFields::kMaybePointers);

270 template <typename THeapObjectSlot>
271 SlotCallbackResult Scavenger::EvacuateShortcutCandidate(
272  Tagged<Map> map, THeapObjectSlot slot, Tagged<ConsString> object,
273  int object_size) {
274  static_assert(std::is_same_v<THeapObjectSlot, FullHeapObjectSlot> ||
275  std::is_same_v<THeapObjectSlot, HeapObjectSlot>,
276  "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
277  DCHECK(IsShortcutCandidate(map->instance_type()));
278 
279  if (shortcut_strings_ &&
280  object->unchecked_second() == ReadOnlyRoots(heap()).empty_string()) {
281  Tagged<HeapObject> first = Cast<HeapObject>(object->unchecked_first());
282 
283  UpdateHeapObjectReferenceSlot(slot, first);
284 
285  if (!HeapLayout::InYoungGeneration(first)) {
286  object->set_map_word_forwarded(first, kRelaxedStore);
287  return REMOVE_SLOT;
288  }
289 
290  MapWord first_word = first->map_word(kRelaxedLoad);
291  if (first_word.IsForwardingAddress()) {
292  Tagged<HeapObject> target = first_word.ToForwardingAddress(first);
293 
294  UpdateHeapObjectReferenceSlot(slot, target);
295  SynchronizePageAccess(target);
296  object->set_map_word_forwarded(target, kRelaxedStore);
297  return HeapLayout::InYoungGeneration(target) ? KEEP_SLOT : REMOVE_SLOT;
298  }
299  Tagged<Map> first_map = first_word.ToMap();
300  SlotCallbackResult result = EvacuateObjectDefault(
301  first_map, slot, first, first->SizeFromMap(first_map),
302  Map::ObjectFieldsFrom(first_map->visitor_id()));
303  object->set_map_word_forwarded(slot.ToHeapObject(), kRelaxedStore);
304  return result;
305  }
306  DCHECK_EQ(ObjectFields::kMaybePointers,
307  Map::ObjectFieldsFrom(map->visitor_id()));
308  return EvacuateObjectDefault(map, slot, object, object_size,
309  ObjectFields::kMaybePointers);
310 }

325 template <typename THeapObjectSlot>
326 SlotCallbackResult Scavenger::EvacuateObject(THeapObjectSlot slot,
327  Tagged<Map> map,
328  Tagged<HeapObject> source) {
329  static_assert(std::is_same_v<THeapObjectSlot, FullHeapObjectSlot> ||
330  std::is_same_v<THeapObjectSlot, HeapObjectSlot>,
331  "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
332  SLOW_DCHECK(Heap::InFromPage(source));
333  SLOW_DCHECK(!MapWord::FromMap(map).IsForwardingAddress());
334  int size = source->SizeFromMap(map);
335  // Cannot use ::cast() below because that would add checks in debug mode
336  // that require re-reading the map.
337  VisitorId visitor_id = map->visitor_id();
338  switch (visitor_id) {
339  case kVisitThinString:
340  // At the moment we don't allow weak pointers to thin strings.
341  DCHECK(!(*slot).IsWeak());
342  return EvacuateThinString(map, slot, UncheckedCast<ThinString>(source),
343  size);
344  case kVisitShortcutCandidate:
345  DCHECK(!(*slot).IsWeak());
346  // At the moment we don't allow weak pointers to cons strings.
347  return EvacuateShortcutCandidate(map, slot,
348  UncheckedCast<ConsString>(source), size);
349  case kVisitSeqOneByteString:
350  case kVisitSeqTwoByteString:
351  DCHECK(String::IsInPlaceInternalizable(map->instance_type()));
352  static_assert(Map::ObjectFieldsFrom(kVisitSeqOneByteString) ==
353  Map::ObjectFieldsFrom(kVisitSeqTwoByteString));
354  return EvacuateInPlaceInternalizableString(
355  map, slot, UncheckedCast<String>(source), size,
356  Map::ObjectFieldsFrom(kVisitSeqOneByteString));
357  default:
358  return EvacuateObjectDefault(map, slot, source, size,
359  Map::ObjectFieldsFrom(visitor_id));
360  }
361 }

363 template <typename THeapObjectSlot>
364 SlotCallbackResult Scavenger::ScavengeObject(THeapObjectSlot p,
365  Tagged<HeapObject> object) {
366  static_assert(std::is_same_v<THeapObjectSlot, FullHeapObjectSlot> ||
367  std::is_same_v<THeapObjectSlot, HeapObjectSlot>,
368  "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
369  DCHECK(Heap::InFromPage(object));
370 
371  // Check whether object was already successfully forwarded by the CAS in
372  // MigrateObject. No memory ordering required because we only access the page
373  // header of a relocated object. Page header initialization uses a memory
374  // fence.
375  MapWord first_word = object->map_word(kRelaxedLoad);
376 
377  // If the first word is a forwarding address, the object has already been
378  // copied.
379  if (first_word.IsForwardingAddress()) {
380  Tagged<HeapObject> dest = first_word.ToForwardingAddress(object);
381  DCHECK_IMPLIES(HeapLayout::IsSelfForwarded(object, first_word),
382  dest == object);
383  if (dest == object) {
384  MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
385  return chunk->IsFlagSet(MemoryChunk::WILL_BE_PROMOTED) ||
386  chunk->IsLargePage()
387  ? REMOVE_SLOT
388  : KEEP_SLOT;
389  }
390 
391  UpdateHeapObjectReferenceSlot(p, dest);
392  SynchronizePageAccess(dest);
393  // A forwarded object in new space is either in the second (to) semi space,
394  // a large object, or a pinned object on a quarantined page.
395  // Pinned objects have a self forwarding map word. However, since forwarding
396  // addresses are set with relaxed atomics and before the object is actually
397  // copied, it is unfortunately not safe to access `dest` to check whether it
398  // is pinned or not.
399  DCHECK_IMPLIES(HeapLayout::InYoungGeneration(dest),
400  Heap::InToPage(dest) || Heap::IsLargeObject(dest) ||
401  MemoryChunk::FromHeapObject(dest)->IsQuarantined());
402 
403  // This load forces us to have memory ordering for the map load above. We
404  // need to have the page header properly initialized.
405  MemoryChunk* chunk = MemoryChunk::FromHeapObject(dest);
406  return !chunk->InYoungGeneration() || chunk->IsLargePage() ? REMOVE_SLOT
407  : KEEP_SLOT;
408  }
409 
410  Tagged<Map> map = first_word.ToMap();
411  // AllocationMementos are unrooted and shouldn't survive a scavenge
412  DCHECK_NE(ReadOnlyRoots(heap()).allocation_memento_map(), map);
413  // Call the slow part of scavenge object.
414  return EvacuateObject(p, map, object);
415 }

417 template <typename TSlot>
418 SlotCallbackResult Scavenger::CheckAndScavengeObject(Heap* heap, TSlot slot) {
419  static_assert(
420  std::is_same_v<TSlot, FullMaybeObjectSlot> ||
421  std::is_same_v<TSlot, MaybeObjectSlot>,
422  "Only FullMaybeObjectSlot and MaybeObjectSlot are expected here");
423  using THeapObjectSlot = typename TSlot::THeapObjectSlot;
424  Tagged<MaybeObject> object = *slot;
425  if (Heap::InFromPage(object)) {
426  Tagged<HeapObject> heap_object = object.GetHeapObject();
427 
428  SlotCallbackResult result =
429  ScavengeObject(THeapObjectSlot(slot), heap_object);
430  DCHECK_IMPLIES(result == REMOVE_SLOT,
431  !HeapLayout::InYoungGeneration((*slot).GetHeapObject()) ||
432  MemoryChunk::FromHeapObject((*slot).GetHeapObject())
433  ->IsLargePage() ||
434  MemoryChunk::FromHeapObject((*slot).GetHeapObject())
435  ->IsFlagSet(MemoryChunk::WILL_BE_PROMOTED));
436  return result;
437  } else if (Heap::InToPage(object)) {
438  // Already updated slot. This can happen when processing of the work list
439  // is interleaved with processing roots.
440  return KEEP_SLOT;
441  }
442  // Slots can point to "to" space if the slot has been recorded multiple
443  // times in the remembered set. We remove the redundant slot now.
444  return REMOVE_SLOT;
445 }

513  for (TSlot slot = start; slot < end; ++slot) {

553 void ScavengeVisitor::VisitExternalPointer(Tagged<HeapObject> host,
554  ExternalPointerSlot slot) {
555 #ifdef V8_COMPRESS_POINTERS
556  DCHECK(!slot.tag_range().IsEmpty());
557  DCHECK(!IsSharedExternalPointerType(slot.tag_range()));
558  DCHECK(HeapLayout::InYoungGeneration(host));
559 
560  // TODO(chromium:337580006): Remove when pointer compression always uses
561  // EPT.
