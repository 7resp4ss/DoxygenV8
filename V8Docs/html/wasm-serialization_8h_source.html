30  // The data header consists of uint32_t-sized entries (see {WriteVersion}):
31  // [0] magic number
32  // [1] version hash
33  // [2] supported CPU features
34  // [3] flag hash
35  // [4] enabled features (via flags and OT)
36  // ... number of functions
37  // ... serialized functions
38  static constexpr size_t kMagicNumberOffset = 0;
39  static constexpr size_t kVersionHashOffset = kMagicNumberOffset + kUInt32Size;
40  static constexpr size_t kSupportedCPUFeaturesOffset =
41  kVersionHashOffset + kUInt32Size;
42  static constexpr size_t kFlagHashOffset =
43  kSupportedCPUFeaturesOffset + kUInt32Size;
44  static constexpr size_t kHeaderSize = 5 * kUInt32Size;
45 
46  private:

51  std::vector<WellKnownImport> import_statuses_;
52 };
