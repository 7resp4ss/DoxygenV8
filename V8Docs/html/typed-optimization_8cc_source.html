176 Reduction TypedOptimization::ReduceMaybeGrowFastElements(Node* node) {
177  Node* const elements = NodeProperties::GetValueInput(node, 1);
178  Node* const index = NodeProperties::GetValueInput(node, 2);
179  Node* const length = NodeProperties::GetValueInput(node, 3);
180  Node* const effect = NodeProperties::GetEffectInput(node);
181  Node* const control = NodeProperties::GetControlInput(node);
182 
183  Type const index_type = NodeProperties::GetType(index);
184  Type const length_type = NodeProperties::GetType(length);
185  CHECK(index_type.Is(Type::Unsigned31()));
186  CHECK(length_type.Is(Type::Unsigned31()));
187 
188  if (!index_type.IsNone() && !length_type.IsNone() &&
189  index_type.Max() < length_type.Min()) {
190  if (v8_flags.turbo_typer_hardening) {
191  Node* check_bounds = graph()->NewNode(
192  simplified()->CheckBounds(FeedbackSource{},
193  CheckBoundsFlag::kAbortOnOutOfBounds),
194  index, length, effect, control);
195  ReplaceWithValue(node, elements, check_bounds);
196  return Replace(check_bounds);
197  } else {
198  RelaxEffectsAndControls(node);
199  return Replace(elements);
200  }
201  }
202 
203  return NoChange();
204 }

234 // The CheckMaps(o, ...map...) can be eliminated if map is stable,
235 // o has type Constant(object) and map == object->map, and either
236 // (1) map cannot transition further, or
237 // (2) we can add a code dependency on the stability of map
238 // (to guard the Constant type information).
239 bool CheckMapsHelper(OptionalMapRef object_map, ZoneRefSet<Map> maps,
240  CompilationDependencies* dependencies) {
241  if (object_map.has_value()) {
242  for (MapRef map : maps) {
243  if (map.equals(*object_map)) {
244  if (object_map->CanTransition()) {
245  dependencies->DependOnStableMap(*object_map);
246  }
247  return true;
248  }
249  }
250  }
251  return false;
252 }

344 Reduction TypedOptimization::ReduceLoadField(Node* node) {
345  Node* const object = NodeProperties::GetValueInput(node, 0);
346  Type const object_type = NodeProperties::GetType(object);
347  FieldAccess const& access = FieldAccessOf(node->op());
348  if (access.base_is_tagged == kTaggedBase &&
349  access.offset == HeapObject::kMapOffset) {
350  // We can replace LoadField[Map](o) with map if is stable, and
351  // o has type Constant(object) and map == object->map, and either
352  // (1) map cannot transition further, or
353  // (2) deoptimization is enabled and we can add a code dependency on the
354  // stability of map (to guard the Constant type information).
355  OptionalMapRef object_map =
356  GetStableMapFromObjectType(broker(), object_type);
357  if (object_map.has_value()) {
358  dependencies()->DependOnStableMap(*object_map);
359  Node* const value = jsgraph()->ConstantNoHole(*object_map, broker());
360  ReplaceWithValue(node, value);
361  return Replace(value);
362  }

367 Reduction TypedOptimization::ReduceNumberFloor(Node* node) {
368  Node* const input = NodeProperties::GetValueInput(node, 0);
369  Type const input_type = NodeProperties::GetType(input);
370  if (input_type.Is(type_cache_->kIntegerOrMinusZeroOrNaN)) {
371  return Replace(input);
372  }
373  if (input_type.Is(Type::PlainNumber()) &&
374  (input->opcode() == IrOpcode::kNumberDivide ||
375  input->opcode() == IrOpcode::kSpeculativeNumberDivide)) {
376  Node* const lhs = NodeProperties::GetValueInput(input, 0);
377  Type const lhs_type = NodeProperties::GetType(lhs);
378  Node* const rhs = NodeProperties::GetValueInput(input, 1);
379  Type const rhs_type = NodeProperties::GetType(rhs);
380  if (lhs_type.IsNone() || rhs_type.IsNone()) return NoChange();
381  if (lhs_type.Is(Type::Unsigned32()) && rhs_type.Is(Type::Unsigned32())) {
382  // We can replace
383  //
384  // NumberFloor(NumberDivide(lhs: unsigned32,
385  // rhs: unsigned32)): plain-number
386  //
387  // with
388  //
389  // Unsigned32Divide(lhs, rhs)
390  //
391  // and have the new node typed to [0...lhs.Max],
392  // as the truncated result must be lower than {lhs}'s maximum
393  // value (note that {rhs} cannot be less than 1 due to the
394  // plain-number type constraint on the {node}).
395  node = graph()->NewNode(simplified()->Unsigned32Divide(), lhs, rhs);
396  return Replace(node);
397  }
398  }
399  return NoChange();
400 }

429 Reduction TypedOptimization::ReducePhi(Node* node) {
430  // Try to narrow the type of the Phi {node}, which might be more precise now
431  // after lowering based on types, i.e. a SpeculativeNumberAdd has a more
432  // precise type than the JSAdd that was in the graph when the Typer was run.
433  DCHECK_EQ(IrOpcode::kPhi, node->opcode());
434  // Prevent new types from being propagated through loop-related Phis for now.
435  // This is to avoid slow convergence of type narrowing when we learn very
436  // precise information about loop variables.
437  if (NodeProperties::GetControlInput(node, 0)->opcode() == IrOpcode::kLoop) {
438  return NoChange();
439  }
440  int arity = node->op()->ValueInputCount();
441  Type type = NodeProperties::GetType(node->InputAt(0));
442  for (int i = 1; i < arity; ++i) {
443  type = Type::Union(type, NodeProperties::GetType(node->InputAt(i)),
444  graph()->zone());
445  }
446  Type const node_type = NodeProperties::GetType(node);
447  if (!node_type.Is(type)) {
448  type = Type::Intersect(node_type, type, graph()->zone());
449  NodeProperties::SetType(node, type);
450  return Changed(node);
451  }
452  return NoChange();
453 }

455 Reduction TypedOptimization::ReduceReferenceEqual(Node* node) {
456  DCHECK_EQ(IrOpcode::kReferenceEqual, node->opcode());
457  Node* const lhs = NodeProperties::GetValueInput(node, 0);
458  Node* const rhs = NodeProperties::GetValueInput(node, 1);
459  Type const lhs_type = NodeProperties::GetType(lhs);
460  Type const rhs_type = NodeProperties::GetType(rhs);
461  if (!lhs_type.Maybe(rhs_type)) {
462  Node* replacement = jsgraph()->FalseConstant();
463  // Make sure we do not widen the type.
464  if (NodeProperties::GetType(replacement)
465  .Is(NodeProperties::GetType(node))) {
466  return Replace(jsgraph()->FalseConstant());
467  }
468  }
469  if (rhs_type.Is(Type::Boolean()) && rhs_type.IsHeapConstant() &&
470  lhs_type.Is(Type::Boolean())) {
471  std::optional<bool> maybe_result =
472  rhs_type.AsHeapConstant()->Ref().TryGetBooleanValue(broker());
473  if (maybe_result.has_value()) {
474  if (maybe_result.value()) {
475  return Replace(node->InputAt(0));
476  } else {
477  node->TrimInputCount(1);
478  NodeProperties::ChangeOp(node, simplified()->BooleanNot());
479  return Changed(node);
480  }
481  }
482  }
483  return NoChange();
484 }

501 Reduction TypedOptimization::
502  TryReduceStringComparisonOfStringFromSingleCharCodeToConstant(
503  Node* comparison, StringRef string, bool inverted) {
504  switch (comparison->opcode()) {
505  case IrOpcode::kStringEqual:
506  case IrOpcode::kStringOrOddballStrictEqual:
507  if (string.length() != 1) {
508  // String.fromCharCode(x) always has length 1.
509  return Replace(jsgraph()->BooleanConstant(false));
510  }
511  break;
512  case IrOpcode::kStringLessThan:
513  [[fallthrough]];
514  case IrOpcode::kStringLessThanOrEqual:
515  if (string.length() == 0) {
516  // String.fromCharCode(x) <= "" is always false,
517  // "" < String.fromCharCode(x) is always true.
518  return Replace(jsgraph()->BooleanConstant(inverted));
519  }
520  break;
521  default:
522  UNREACHABLE();
523  }
524  return NoChange();
525 }

527 // Try to reduces a string comparison of the form
528 // String.fromCharCode(x) {comparison} {constant} if inverted is false,
529 // and {constant} {comparison} String.fromCharCode(x) if inverted is true.
530 Reduction
531 TypedOptimization::TryReduceStringComparisonOfStringFromSingleCharCode(
532  Node* comparison, Node* from_char_code, Type constant_type, bool inverted) {
533  DCHECK_EQ(IrOpcode::kStringFromSingleCharCode, from_char_code->opcode());
534 
535  if (!constant_type.IsHeapConstant()) return NoChange();
536  ObjectRef constant = constant_type.AsHeapConstant()->Ref();
537 
538  if (!constant.IsString()) return NoChange();
539  StringRef string = constant.AsString();
540 
541  // Check if comparison can be resolved statically.
542  Reduction red = TryReduceStringComparisonOfStringFromSingleCharCodeToConstant(
543  comparison, string, inverted);
544  if (red.Changed()) return red;
545 
546  const Operator* comparison_op = NumberComparisonFor(comparison->op());
547  Node* from_char_code_repl = NodeProperties::GetValueInput(from_char_code, 0);
548  Type from_char_code_repl_type = NodeProperties::GetType(from_char_code_repl);
549  if (!from_char_code_repl_type.Is(type_cache_->kUint16)) {
550  // Convert to signed int32 to satisfy type of {NumberBitwiseAnd}.
551  from_char_code_repl =
552  graph()->NewNode(simplified()->NumberToInt32(), from_char_code_repl);
553  from_char_code_repl = graph()->NewNode(
554  simplified()->NumberBitwiseAnd(), from_char_code_repl,
555  jsgraph()->ConstantNoHole(std::numeric_limits<uint16_t>::max()));
556  }
557  if (!string.GetFirstChar(broker()).has_value()) return NoChange();
558  Node* constant_repl =
559  jsgraph()->ConstantNoHole(string.GetFirstChar(broker()).value());
560 
561  Node* number_comparison = nullptr;
562  if (inverted) {
563  // "x..." <= String.fromCharCode(z) is true if x < z.
564  if (string.length() > 1 &&
565  comparison->opcode() == IrOpcode::kStringLessThanOrEqual) {
566  comparison_op = simplified()->NumberLessThan();
567  }
568  number_comparison =
569  graph()->NewNode(comparison_op, constant_repl, from_char_code_repl);
570  } else {
571  // String.fromCharCode(z) < "x..." is true if z <= x.
572  if (string.length() > 1 &&
573  comparison->opcode() == IrOpcode::kStringLessThan) {
574  comparison_op = simplified()->NumberLessThanOrEqual();
575  }
576  number_comparison =
577  graph()->NewNode(comparison_op, from_char_code_repl, constant_repl);
578  }
579  ReplaceWithValue(comparison, number_comparison);
580  return Replace(number_comparison);
581 }

583 Reduction TypedOptimization::ReduceStringComparison(Node* node) {
584  DCHECK(IrOpcode::kStringEqual == node->opcode() ||
585  IrOpcode::kStringLessThan == node->opcode() ||
586  IrOpcode::kStringLessThanOrEqual == node->opcode() ||
587  IrOpcode::kStringOrOddballStrictEqual == node->opcode());
588  Node* const lhs = NodeProperties::GetValueInput(node, 0);
589  Node* const rhs = NodeProperties::GetValueInput(node, 1);
590  Type lhs_type = NodeProperties::GetType(lhs);
591  Type rhs_type = NodeProperties::GetType(rhs);
592  if (lhs->opcode() == IrOpcode::kStringFromSingleCharCode) {
593  if (rhs->opcode() == IrOpcode::kStringFromSingleCharCode) {
594  Node* left = NodeProperties::GetValueInput(lhs, 0);
595  Node* right = NodeProperties::GetValueInput(rhs, 0);
596  Type left_type = NodeProperties::GetType(left);
597  Type right_type = NodeProperties::GetType(right);
598  if (!left_type.Is(type_cache_->kUint16)) {
599  // Convert to signed int32 to satisfy type of {NumberBitwiseAnd}.
600  left = graph()->NewNode(simplified()->NumberToInt32(), left);
601  left = graph()->NewNode(
602  simplified()->NumberBitwiseAnd(), left,

706  DCHECK_EQ(IrOpcode::kSelect, node->opcode());
707  Node* const condition = NodeProperties::GetValueInput(node, 0);
708  Type const condition_type = NodeProperties::GetType(condition);
709  Node* const vtrue = NodeProperties::GetValueInput(node, 1);
710  Type const vtrue_type = NodeProperties::GetType(vtrue);
711  Node* const vfalse = NodeProperties::GetValueInput(node, 2);
712  Type const vfalse_type = NodeProperties::GetType(vfalse);
713  if (condition_type.Is(true_type_)) {
714  // Select(condition:true, vtrue, vfalse) => vtrue
715  return Replace(vtrue);
716  }
717  if (condition_type.Is(false_type_)) {
718  // Select(condition:false, vtrue, vfalse) => vfalse
719  return Replace(vfalse);
720  }
721  if (vtrue_type.Is(true_type_) && vfalse_type.Is(false_type_)) {
722  // Select(condition, vtrue:true, vfalse:false) => condition
723  return Replace(condition);
724  }
725  if (vtrue_type.Is(false_type_) && vfalse_type.Is(true_type_)) {
726  // Select(condition, vtrue:false, vfalse:true) => BooleanNot(condition)
727  node->TrimInputCount(1);
728  NodeProperties::ChangeOp(node, simplified()->BooleanNot());

732  // now after lowering based on types.

734  Type const node_type = NodeProperties::GetType(node);
735  if (!node_type.Is(type)) {
736  type = Type::Intersect(node_type, type, graph()->zone());
737  NodeProperties::SetType(node, type);
738  return Changed(node);
739  }
740  return NoChange();
