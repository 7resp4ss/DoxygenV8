25 // An TaggedImpl is a base class for Object (which is either a Smi or a strong
26 // reference to a HeapObject) and Tagged<MaybeObject> (which is either a Smi, a
27 // strong reference to a HeapObject, a weak reference to a HeapObject, or a
28 // cleared weak reference. This class provides storage and one canonical
29 // implementation of various predicates that check Smi and heap object tags'
30 // values and also take into account whether the tagged value is expected to be
31 // weak reference to a HeapObject or cleared weak reference.
32 template <HeapObjectReferenceType kRefType, typename StorageType>
33 class TaggedImpl {
34  public:
35  static_assert(std::is_same_v<StorageType, Address> ||
36  std::is_same_v<StorageType, Tagged_t>,
37  "StorageType must be either Address or Tagged_t");
38 
39  // True for those TaggedImpl instantiations that represent uncompressed
40  // tagged values and false for TaggedImpl instantiations that represent
41  // compressed tagged values.
42  static const bool kIsFull = sizeof(StorageType) == kSystemPointerSize;
43 
44  static const bool kCanBeWeak = kRefType == HeapObjectReferenceType::WEAK;
45 
46  V8_INLINE constexpr TaggedImpl() : ptr_{} {}
47  V8_INLINE explicit constexpr TaggedImpl(StorageType ptr) : ptr_(ptr) {}

52  // Don't use this operator for comparing with stale or invalid pointers
53  // because CheckObjectComparisonAllowed() might crash when trying to access
54  // the object's page header. Use SafeEquals() instead.
55  template <HeapObjectReferenceType kOtherRefType, typename U>
56  constexpr bool operator==(TaggedImpl<kOtherRefType, U> other) const {
57  static_assert(std::is_same_v<U, Address> || std::is_same_v<U, Tagged_t>,
58  "U must be either Address or Tagged_t");
59 #if defined(V8_EXTERNAL_CODE_SPACE) || defined(V8_ENABLE_SANDBOX)
60  // When comparing two full pointer values ensure that it's allowed.
61  if (std::is_same_v<StorageType, Address> && std::is_same_v<U, Address>) {
62  SLOW_DCHECK(CheckObjectComparisonAllowed(ptr_, other.ptr()));
63  }
64 #endif // defined(V8_EXTERNAL_CODE_SPACE) || defined(V8_ENABLE_SANDBOX)
65  return static_cast<Tagged_t>(ptr_) == static_cast<Tagged_t>(other.ptr());
66  }
67 
68  // Don't use this operator for comparing with stale or invalid pointers
69  // because CheckObjectComparisonAllowed() might crash when trying to access
70  // the object's page header. Use SafeEquals() instead.
71  template <HeapObjectReferenceType kOtherRefType, typename U>
72  constexpr bool operator!=(TaggedImpl<kOtherRefType, U> other) const {
73  static_assert(std::is_same_v<U, Address> || std::is_same_v<U, Tagged_t>,
74  "U must be either Address or Tagged_t");
75 #if defined(V8_EXTERNAL_CODE_SPACE) || defined(V8_ENABLE_SANDBOX)
76  // When comparing two full pointer values ensure that it's allowed.
77  if (std::is_same_v<StorageType, Address> && std::is_same_v<U, Address>) {
78  SLOW_DCHECK(CheckObjectComparisonAllowed(ptr_, other.ptr()));
79  }
80 #endif // defined(V8_EXTERNAL_CODE_SPACE) || defined(V8_ENABLE_SANDBOX)
81  return static_cast<Tagged_t>(ptr_) != static_cast<Tagged_t>(other.ptr());
82  }
83 
84  // A variant of operator== which allows comparing objects in different
85  // pointer compression cages. In particular, this should be used when
86  // comparing objects in trusted- or code space with objects in the main
87  // pointer compression cage.
88  template <HeapObjectReferenceType kOtherRefType>
89  constexpr bool SafeEquals(
90  TaggedImpl<kOtherRefType, StorageType> other) const {
91  static_assert(std::is_same_v<StorageType, Address>,
92  "Safe comparison is allowed only for full tagged values");
93  if (V8_EXTERNAL_CODE_SPACE_BOOL || V8_ENABLE_SANDBOX_BOOL) {
94  return ptr_ == other.ptr();
95  }
96  return this->operator==(other);
97  }

99  // For using in std::set and std::map.
100  constexpr bool operator<(TaggedImpl other) const {
101 #if defined(V8_EXTERNAL_CODE_SPACE) || defined(V8_ENABLE_SANDBOX)
102  // When comparing two full pointer values ensure that it's allowed.
103  if (std::is_same_v<StorageType, Address>) {
104  SLOW_DCHECK(CheckObjectComparisonAllowed(ptr_, other.ptr()));
105  }
106 #endif // defined(V8_EXTERNAL_CODE_SPACE) || defined(V8_ENABLE_SANDBOX)
107  return static_cast<Tagged_t>(ptr_) < static_cast<Tagged_t>(other.ptr());
108  }

136  // Returns true if this tagged value is a strong pointer to a HeapObject.
137  constexpr inline bool IsStrong() const {
138  DCHECK(kCanBeWeak || (!IsSmi() == HAS_STRONG_HEAP_OBJECT_TAG(ptr_)));
139  return kCanBeWeak ? HAS_STRONG_HEAP_OBJECT_TAG(ptr_) : !IsSmi();
140  }

142  // Returns true if this tagged value is a strong pointer to a HeapObject, or a
143  // Smi.
144  constexpr inline bool IsStrongOrSmi() const {
145  return !kCanBeWeak || !HAS_WEAK_HEAP_OBJECT_TAG(ptr_);
146  }
147 
148  // Returns true if this tagged value is a weak pointer to a HeapObject.
149  constexpr inline bool IsWeak() const {
150  return IsWeakOrCleared() && !IsCleared();

153  // Returns true if this tagged value is a weak pointer to a HeapObject or
154  // cleared weak reference.
155  constexpr inline bool IsWeakOrCleared() const {
156  return kCanBeWeak && HAS_WEAK_HEAP_OBJECT_TAG(ptr_);
