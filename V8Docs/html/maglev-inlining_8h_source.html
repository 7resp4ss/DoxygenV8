26  void Run(bool is_tracing_maglev_graphs_enabled) {
27  if (graph_->inlineable_calls().empty()) return;
28 
29  while (true) {
30  if (graph_->total_inlined_bytecode_size() >
31  v8_flags.max_maglev_inlined_bytecode_size_cumulative) {
32  // No more inlining.
33  break;
34  }
35  MaglevCallSiteInfo* call_site = ChooseNextCallSite();
36  if (!call_site) break;
37 
38  MaybeReduceResult result = BuildInlineFunction(call_site);
39  if (result.IsFail()) continue;
40  // If --trace-maglev-inlining-verbose, we print the graph after each
41  // inlining step/call.
42  if (is_tracing_maglev_graphs_enabled && v8_flags.print_maglev_graphs &&
43  v8_flags.trace_maglev_inlining_verbose) {
44  std::cout << "\nAfter inlining "
45  << call_site->generic_call_node->shared_function_info()
46  << std::endl;
47  PrintGraph(std::cout, graph_);
48  }
49  }
50 
51  // Otherwise we print just once at the end.
52  if (is_tracing_maglev_graphs_enabled && v8_flags.print_maglev_graphs &&
53  !v8_flags.trace_maglev_inlining_verbose) {
54  std::cout << "\nAfter inlining" << std::endl;
55  PrintGraph(std::cout, graph_);
56  }
57  }
58 
59  private:

62  compiler::JSHeapBroker* broker() const { return graph_->broker(); }
63  Zone* zone() const { return graph_->zone(); }

65  MaglevCallSiteInfo* ChooseNextCallSite() {
66  auto it =
67  v8_flags.maglev_inlining_following_eager_order
68  ? std::ranges::find_if(graph_->inlineable_calls(),
69  [](auto* site) { return site != nullptr; })
70  : std::ranges::max_element(
71  graph_->inlineable_calls(),
72  [](const MaglevCallSiteInfo* info1,
73  const MaglevCallSiteInfo* info2) {
74  if (info1 == nullptr || info2 == nullptr) {
75  return info2 != nullptr;
76  }
77  return info1->score < info2->score;
78  });
79  if (it == graph_->inlineable_calls().end()) return nullptr;
80  MaglevCallSiteInfo* call_site = *it;
81  *it = nullptr; // Erase call site.
82  return call_site;
83  }

85  MaybeReduceResult BuildInlineFunction(MaglevCallSiteInfo* call_site) {
86  CallKnownJSFunction* call_node = call_site->generic_call_node;
87  BasicBlock* call_block = call_node->owner();
88  MaglevCallerDetails* caller_details = &call_site->caller_details;
89  DeoptFrame* caller_deopt_frame = caller_details->deopt_frame;
90  const MaglevCompilationUnit* caller_unit =
91  &caller_deopt_frame->GetCompilationUnit();
92  compiler::SharedFunctionInfoRef shared = call_node->shared_function_info();
93 
94  if (!call_block || call_block->is_dead()) {
95  // The block containing the call is unreachable, and it was previously
96  // removed. Do not try to inline the call.
97  return ReduceResult::Fail();
98  }
99 
100  if (v8_flags.trace_maglev_inlining) {
101  std::cout << " non-eager inlining " << shared << std::endl;
102  }
103 
104  // Check if the catch block might become unreachable, ie, the call is the
105  // only instance of a throwable node in this block to the same catch block.
106  ExceptionHandlerInfo* call_exception_handler_info =
107  call_node->exception_handler_info();
108  bool catch_block_might_be_unreachable = false;
109  if (call_exception_handler_info->HasExceptionHandler() &&
110  !call_exception_handler_info->ShouldLazyDeopt()) {
111  BasicBlock* catch_block = call_exception_handler_info->catch_block();
112  catch_block_might_be_unreachable = true;
113  for (ExceptionHandlerInfo* info : call_block->exception_handlers()) {
114  if (info != call_exception_handler_info &&
115  info->HasExceptionHandler() && !info->ShouldLazyDeopt() &&
116  info->catch_block() == catch_block) {
117  catch_block_might_be_unreachable = false;
118  break;
119  }
120  }
121  }
122 
123  // Remove exception handler info from call block.
124  ExceptionHandlerInfo::List rem_handlers_in_call_block;
125  call_block->exception_handlers().TruncateAt(&rem_handlers_in_call_block,
126  call_exception_handler_info);
127  rem_handlers_in_call_block.DropHead();
128 
129  // Truncate the basic block and remove the generic call node.
130  ZoneVector<Node*> rem_nodes_in_call_block =
131  call_block->Split(call_node, zone());
132 
133  // Create a new compilation unit.
134  MaglevCompilationUnit* inner_unit = MaglevCompilationUnit::NewInner(
135  zone(), caller_unit, shared, call_site->feedback_cell);
136 
137  compiler::BytecodeArrayRef bytecode = shared.GetBytecodeArray(broker());
138  graph_->add_inlined_bytecode_size(bytecode.length());
139 
140  // Create a new graph builder for the inlined function.
141  LocalIsolate* local_isolate = broker()->local_isolate_or_isolate();
142  MaglevGraphBuilder inner_graph_builder(local_isolate, inner_unit, graph_,
143  &call_site->caller_details);
144 
145  // Update caller deopt frame with inlined arguments.
146  caller_details->deopt_frame =
147  inner_graph_builder.AddInlinedArgumentsToDeoptFrame(
148  caller_deopt_frame, inner_unit, call_node->closure().node(),
149  call_site->caller_details.arguments);
150 
151  // We truncate the graph to build the function in-place, preserving the
152  // invariant that all jumps move forward (except JumpLoop).
153  std::vector<BasicBlock*> saved_bb = TruncateGraphAt(call_block);
154  ControlNode* control_node = call_block->reset_control_node();
155 
156  // Set the inner graph builder to build in the truncated call block.
157  inner_graph_builder.set_current_block(call_block);
158 
159  ReduceResult result = inner_graph_builder.BuildInlineFunction(
160  caller_deopt_frame->GetSourcePosition(), call_node->context().node(),
161  call_node->closure().node(), call_node->new_target().node());
162 
163  if (result.IsDoneWithAbort()) {
164  // Since the rest of the block is dead, these nodes don't belong
165  // to any basic block anymore.
166  for (auto node : rem_nodes_in_call_block) {
167  node->set_owner(nullptr);
168  }
169  // Restore the rest of the graph.
170  for (auto bb : saved_bb) {
171  graph_->Add(bb);
172  }
173  RemovePredecessorFollowing(control_node, call_block);
174  // TODO(victorgomes): We probably don't need to iterate all the graph to
175  // remove unreachable blocks, but only the successors of control_node in
176  // saved_bbs.
177  RemoveUnreachableBlocks();
178  return result;
179  }
180 
181  DCHECK(result.IsDoneWithValue());
182  ValueNode* returned_value =
183  EnsureTagged(inner_graph_builder, result.value());
184 
185  // Resume execution using the final block of the inner builder.
186 
187  // Add remaining nodes to the final block and use the control flow of the
188  // old call block.
189  BasicBlock* final_block = inner_graph_builder.FinishInlinedBlockForCaller(
190  control_node, rem_nodes_in_call_block);
191  DCHECK_NOT_NULL(final_block);
192  final_block->exception_handlers().Append(
193  std::move(rem_handlers_in_call_block));
194 
195  // Update the predecessor of the successors of the {final_block}, that were
196  // previously pointing to {call_block}.
197  final_block->ForEachSuccessor(
198  [call_block, final_block](BasicBlock* successor) {
199  UpdatePredecessorsOf(successor, call_block, final_block);
200  });
201 
202  // Restore the rest of the graph.
203  for (auto bb : saved_bb) {
204  graph_->Add(bb);
205  }
206 
207  if (auto alloc = returned_value->TryCast<InlinedAllocation>()) {
208  // TODO(victorgomes): Support eliding VOs.
209  alloc->ForceEscaping();
210 #ifdef DEBUG
211  alloc->set_is_returned_value_from_inline_call();
212 #endif // DEBUG
213  }
214  call_node->OverwriteWithIdentityTo(returned_value);
215 
216  // Remove unreachable catch block if no throwable nodes were added during
217  // inlining.
218  // TODO(victorgomes): Improve this: track if we didnt indeed add a throwable
219  // node.
220  if (catch_block_might_be_unreachable) {
221  RemoveUnreachableBlocks();
222  }
223 
224  return ReduceResult::Done();
225  }
226 
227  // Truncates the graph at the given basic block `block`. All blocks
228  // following `block` (exclusive) are removed from the graph and returned.
229  // `block` itself is removed from the graph and not returned.
230  std::vector<BasicBlock*> TruncateGraphAt(BasicBlock* block) {
231  // TODO(victorgomes): Consider using a linked list of basic blocks in Maglev
232  // instead of a vector.
233  auto it =
234  std::find(graph_->blocks().begin(), graph_->blocks().end(), block);
235  CHECK_NE(it, graph_->blocks().end());
236  size_t index = std::distance(graph_->blocks().begin(), it);
237  std::vector<BasicBlock*> saved_bb(graph_->blocks().begin() + index + 1,
238  graph_->blocks().end());
239  graph_->blocks().resize(index);
240  return saved_bb;
241  }

243  template <class Node, typename... Args>
244  ValueNode* AddNodeAtBlockEnd(MaglevGraphBuilder& builder,
245  std::initializer_list<ValueNode*> inputs,
246  Args&&... args) {
247  ValueNode* node =
248  NodeBase::New<Node>(zone(), inputs, std::forward<Args>(args)...);
249  DCHECK(!node->properties().can_eager_deopt());
250  DCHECK(!node->properties().can_lazy_deopt());
251  builder.node_buffer().push_back(node);
252  RegisterNode(builder, node);
253  return node;
254  }

262  ValueNode* EnsureTagged(MaglevGraphBuilder& builder, ValueNode* node) {
263  // TODO(victorgomes): Use KNA to create better conversion nodes?
264  switch (node->value_representation()) {
265  case ValueRepresentation::kInt32:
266  return AddNodeAtBlockEnd<Int32ToNumber>(builder, {node});
267  case ValueRepresentation::kUint32:
268  return AddNodeAtBlockEnd<Uint32ToNumber>(builder, {node});
269  case ValueRepresentation::kFloat64:
270  return AddNodeAtBlockEnd<Float64ToTagged>(
271  builder, {node}, Float64ToTagged::ConversionMode::kForceHeapNumber);
272  case ValueRepresentation::kHoleyFloat64:
273  return AddNodeAtBlockEnd<HoleyFloat64ToTagged>(
274  builder, {node},
275  HoleyFloat64ToTagged::ConversionMode::kForceHeapNumber);
276  case ValueRepresentation::kIntPtr:
277  return AddNodeAtBlockEnd<IntPtrToNumber>(builder, {node});
278  case ValueRepresentation::kTagged:
279  return node;
280  }
281  }

283  static void UpdatePredecessorsOf(BasicBlock* block, BasicBlock* prev_pred,
284  BasicBlock* new_pred) {
285  if (!block->has_state()) {
286  DCHECK_EQ(block->predecessor(), prev_pred);
287  block->set_predecessor(new_pred);
288  return;
289  }
290  for (int i = 0; i < block->predecessor_count(); i++) {
291  if (block->predecessor_at(i) == prev_pred) {
292  block->state()->set_predecessor_at(i, new_pred);
293  break;
294  }
295  }
296  }
297 
298  void RemovePredecessorFollowing(ControlNode* control,
299  BasicBlock* call_block) {
300  BasicBlock::ForEachSuccessorFollowing(control, [&](BasicBlock* succ) {
301  if (!succ->has_state()) {
302  succ->set_predecessor(nullptr);
303  return;
304  }
305  if (succ->is_loop() && succ->backedge_predecessor() == call_block) {
306  succ->state()->TurnLoopIntoRegularBlock();
307  return;
308  }
309  for (int i = succ->predecessor_count() - 1; i >= 0; i--) {
310  if (succ->predecessor_at(i) == call_block) {
311  succ->state()->RemovePredecessorAt(i);
312  }
313  }
314  });
315  }
316 
317  void RemoveUnreachableBlocks() {
318  graph_->set_may_have_unreachable_blocks();
319  graph_->RemoveUnreachableBlocks();
320  }
321 };
