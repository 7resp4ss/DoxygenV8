28  void Run(bool is_tracing_maglev_graphs_enabled) {
29  if (graph_->inlineable_calls().empty()) return;
30 
31  while (true) {
32  if (graph_->total_inlined_bytecode_size() >
33  v8_flags.max_maglev_inlined_bytecode_size_cumulative) {
34  // No more inlining.
35  break;
36  }
37  MaglevCallSiteInfo* call_site = ChooseNextCallSite();
38  if (!call_site) break;
39 
40  MaybeReduceResult result = BuildInlineFunction(call_site);
41  if (result.IsFail()) continue;
42  // If --trace-maglev-inlining-verbose, we print the graph after each
43  // inlining step/call.
44  if (is_tracing_maglev_graphs_enabled && v8_flags.print_maglev_graphs &&
45  v8_flags.trace_maglev_inlining_verbose) {
46  std::cout << "\nAfter inlining "
47  << call_site->generic_call_node->shared_function_info()
48  << std::endl;
49  PrintGraph(std::cout, compilation_info_, graph_);
50  }
51  }
52 
53  // Otherwise we print just once at the end.
54  if (is_tracing_maglev_graphs_enabled && v8_flags.print_maglev_graphs &&
55  !v8_flags.trace_maglev_inlining_verbose) {
56  std::cout << "\nAfter inlining" << std::endl;
57  PrintGraph(std::cout, compilation_info_, graph_);
58  }
59  }

68  MaglevCallSiteInfo* ChooseNextCallSite() {
69  auto it =
70  v8_flags.maglev_inlining_following_eager_order
71  ? std::ranges::find_if(graph_->inlineable_calls(),
72  [](auto* site) { return site != nullptr; })
73  : std::ranges::max_element(
74  graph_->inlineable_calls(),
75  [](const MaglevCallSiteInfo* info1,
76  const MaglevCallSiteInfo* info2) {
77  if (info1 == nullptr || info2 == nullptr) {
78  return info2 != nullptr;
79  }
80  return info1->score < info2->score;
81  });
82  if (it == graph_->inlineable_calls().end()) return nullptr;
83  MaglevCallSiteInfo* call_site = *it;
84  *it = nullptr; // Erase call site.
85  return call_site;
86  }

88  MaybeReduceResult BuildInlineFunction(MaglevCallSiteInfo* call_site) {
89  CallKnownJSFunction* call_node = call_site->generic_call_node;
90  BasicBlock* call_block = call_node->owner();
91  MaglevCallerDetails* caller_details = &call_site->caller_details;
92  DeoptFrame* caller_deopt_frame = caller_details->deopt_frame;
93  const MaglevCompilationUnit* caller_unit =
94  &caller_deopt_frame->GetCompilationUnit();
95  compiler::SharedFunctionInfoRef shared = call_node->shared_function_info();
96 
97  if (!call_block || call_block->is_dead()) {
98  // The block containing the call is unreachable, and it was previously
99  // removed. Do not try to inline the call.
100  return ReduceResult::Fail();
101  }
102 
103  if (v8_flags.trace_maglev_inlining) {
104  std::cout << " non-eager inlining " << shared << std::endl;
105  }
106 
107  // Check if the catch block might become unreachable, ie, the call is the
108  // only instance of a throwable node in this block to the same catch block.
109  ExceptionHandlerInfo* call_exception_handler_info =
110  call_node->exception_handler_info();
111  bool catch_block_might_be_unreachable = false;
112  if (call_exception_handler_info->HasExceptionHandler() &&
113  !call_exception_handler_info->ShouldLazyDeopt()) {
114  BasicBlock* catch_block = call_exception_handler_info->catch_block();
115  catch_block_might_be_unreachable = true;
116  for (ExceptionHandlerInfo* info : call_block->exception_handlers()) {
117  if (info != call_exception_handler_info &&
118  info->HasExceptionHandler() && !info->ShouldLazyDeopt() &&
119  info->catch_block() == catch_block) {
120  catch_block_might_be_unreachable = false;
121  break;
122  }
123  }
124  }
125 
126  // Remove exception handler info from call block.
127  ExceptionHandlerInfo::List rem_handlers_in_call_block;
128  call_block->exception_handlers().TruncateAt(&rem_handlers_in_call_block,
129  call_exception_handler_info);
130  rem_handlers_in_call_block.DropHead();
131 
132  // Truncate the basic block and remove the generic call node.
133  ZoneVector<Node*> rem_nodes_in_call_block =
134  call_block->Split(call_node, zone());
135 
136  // Create a new compilation unit.
137  MaglevCompilationUnit* inner_unit = MaglevCompilationUnit::NewInner(
138  zone(), caller_unit, shared, call_site->feedback_cell);
139 
140  compiler::BytecodeArrayRef bytecode = shared.GetBytecodeArray(broker());
141  graph_->add_inlined_bytecode_size(bytecode.length());
142 
143  // Create a new graph builder for the inlined function.
144  LocalIsolate* local_isolate = broker()->local_isolate_or_isolate();
145  MaglevGraphBuilder inner_graph_builder(local_isolate, inner_unit, graph_,
146  &call_site->caller_details);
147 
148  // Update caller deopt frame with inlined arguments.
149  caller_details->deopt_frame =
150  inner_graph_builder.AddInlinedArgumentsToDeoptFrame(
151  caller_deopt_frame, inner_unit, call_node->closure().node(),
152  call_site->caller_details.arguments);
153 
154  // We truncate the graph to build the function in-place, preserving the
155  // invariant that all jumps move forward (except JumpLoop).
156  std::vector<BasicBlock*> saved_bb = TruncateGraphAt(call_block);
157  ControlNode* control_node = call_block->reset_control_node();
158 
159  // Set the inner graph builder to build in the truncated call block.
160  inner_graph_builder.set_current_block(call_block);
161 
162  ReduceResult result = inner_graph_builder.BuildInlineFunction(
163  caller_deopt_frame->GetSourcePosition(), call_node->context().node(),
164  call_node->closure().node(), call_node->new_target().node());
165 
166  if (result.IsDoneWithAbort()) {
167  // Since the rest of the block is dead, these nodes don't belong
168  // to any basic block anymore.
169  for (auto node : rem_nodes_in_call_block) {
170  node->set_owner(nullptr);
171  }
172  // Restore the rest of the graph.
173  for (auto bb : saved_bb) {
174  graph_->Add(bb);
175  }
176  RemovePredecessorFollowing(control_node, call_block);
177  // TODO(victorgomes): We probably don't need to iterate all the graph to
178  // remove unreachable blocks, but only the successors of control_node in
179  // saved_bbs.
180  RemoveUnreachableBlocks();
181  return result;
182  }
183 
184  DCHECK(result.IsDoneWithValue());
185  ValueNode* returned_value =
186  EnsureTagged(inner_graph_builder, result.value());
187 
188  // Resume execution using the final block of the inner builder.
189 
190  // Add remaining nodes to the final block and use the control flow of the
191  // old call block.
192  BasicBlock* final_block = inner_graph_builder.FinishInlinedBlockForCaller(
193  control_node, rem_nodes_in_call_block);
194  DCHECK_NOT_NULL(final_block);
195  final_block->exception_handlers().Append(
196  std::move(rem_handlers_in_call_block));
197 
198  // Update the predecessor of the successors of the {final_block}, that were
199  // previously pointing to {call_block}.
200  final_block->ForEachSuccessor(
201  [call_block, final_block](BasicBlock* successor) {
202  UpdatePredecessorsOf(successor, call_block, final_block);
203  });
204 
205  // Restore the rest of the graph.
206  for (auto bb : saved_bb) {
207  graph_->Add(bb);
208  }
209 
210  if (auto alloc = returned_value->TryCast<InlinedAllocation>()) {
211  // TODO(victorgomes): Support eliding VOs.
212  alloc->ForceEscaping();
213 #ifdef DEBUG
214  alloc->set_is_returned_value_from_inline_call();
215 #endif // DEBUG
216  }
217  call_node->OverwriteWithIdentityTo(returned_value);
218 
219  // Remove unreachable catch block if no throwable nodes were added during
220  // inlining.
221  // TODO(victorgomes): Improve this: track if we didnt indeed add a throwable
222  // node.
223  if (catch_block_might_be_unreachable) {
224  RemoveUnreachableBlocks();
225  }
226 
227  return ReduceResult::Done();
228  }
229 
230  // Truncates the graph at the given basic block `block`. All blocks
231  // following `block` (exclusive) are removed from the graph and returned.
232  // `block` itself is removed from the graph and not returned.
233  std::vector<BasicBlock*> TruncateGraphAt(BasicBlock* block) {
234  // TODO(victorgomes): Consider using a linked list of basic blocks in Maglev
235  // instead of a vector.
236  auto it =
237  std::find(graph_->blocks().begin(), graph_->blocks().end(), block);
238  CHECK_NE(it, graph_->blocks().end());
239  size_t index = std::distance(graph_->blocks().begin(), it);
240  std::vector<BasicBlock*> saved_bb(graph_->blocks().begin() + index + 1,
241  graph_->blocks().end());
242  graph_->blocks().resize(index);
243  return saved_bb;
244  }

246  template <class Node, typename... Args>
247  ValueNode* AddNodeAtBlockEnd(MaglevGraphBuilder& builder,
248  std::initializer_list<ValueNode*> inputs,
249  Args&&... args) {
250  ValueNode* node =
251  NodeBase::New<Node>(zone(), inputs, std::forward<Args>(args)...);
252  DCHECK(!node->properties().can_eager_deopt());
253  DCHECK(!node->properties().can_lazy_deopt());
254  builder.node_buffer().push_back(node);
255  RegisterNode(builder, node);
256  return node;
257  }

265  ValueNode* EnsureTagged(MaglevGraphBuilder& builder, ValueNode* node) {
266  // TODO(victorgomes): Use KNA to create better conversion nodes?
267  switch (node->value_representation()) {
268  case ValueRepresentation::kInt32:
269  return AddNodeAtBlockEnd<Int32ToNumber>(builder, {node});
270  case ValueRepresentation::kUint32:
271  return AddNodeAtBlockEnd<Uint32ToNumber>(builder, {node});
272  case ValueRepresentation::kFloat64:
273  return AddNodeAtBlockEnd<Float64ToTagged>(
274  builder, {node}, Float64ToTagged::ConversionMode::kForceHeapNumber);
275  case ValueRepresentation::kHoleyFloat64:
276  return AddNodeAtBlockEnd<HoleyFloat64ToTagged>(
277  builder, {node},
278  HoleyFloat64ToTagged::ConversionMode::kForceHeapNumber);
279  case ValueRepresentation::kIntPtr:
280  return AddNodeAtBlockEnd<IntPtrToNumber>(builder, {node});
281  case ValueRepresentation::kTagged:
282  return node;
283  }
284  }

286  static void UpdatePredecessorsOf(BasicBlock* block, BasicBlock* prev_pred,
287  BasicBlock* new_pred) {
288  if (!block->has_state()) {
289  DCHECK_EQ(block->predecessor(), prev_pred);
290  block->set_predecessor(new_pred);
291  return;
292  }
293  for (int i = 0; i < block->predecessor_count(); i++) {
294  if (block->predecessor_at(i) == prev_pred) {
295  block->state()->set_predecessor_at(i, new_pred);
296  break;
297  }
298  }
299  }
300 
301  void RemovePredecessorFollowing(ControlNode* control,
302  BasicBlock* call_block) {
303  BasicBlock::ForEachSuccessorFollowing(control, [&](BasicBlock* succ) {
304  if (!succ->has_state()) {
305  succ->set_predecessor(nullptr);
306  return;
307  }
308  if (succ->is_loop() && succ->backedge_predecessor() == call_block) {
309  succ->state()->TurnLoopIntoRegularBlock();
310  return;
311  }
312  for (int i = succ->predecessor_count() - 1; i >= 0; i--) {
313  if (succ->predecessor_at(i) == call_block) {
314  succ->state()->RemovePredecessorAt(i);
315  }
316  }
317  });
318  }

320  void RemoveUnreachableBlocks() {
321  graph_->set_may_have_unreachable_blocks();
322  graph_->RemoveUnreachableBlocks();
323  }
