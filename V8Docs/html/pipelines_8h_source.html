67  template <TurboshaftPhase Phase, typename... Args>
68  V8_WARN_UNUSED_RESULT auto Run(Args&&... args) {
69  // Setup run scope.
70  PhaseScope phase_scope(data_->pipeline_statistics(), Phase::phase_name());
71  ZoneWithName<Phase::kPhaseName> temp_zone(data_->zone_stats(),
72  Phase::phase_name());
73  NodeOriginTable::PhaseScope origin_scope(data_->node_origins(),
74  Phase::phase_name());
75 #ifdef V8_RUNTIME_CALL_STATS
76  RuntimeCallTimerScope runtime_call_timer_scope(data_->runtime_call_stats(),
77  Phase::kRuntimeCallCounterId,
78  Phase::kCounterMode);
79 #endif
80 
81  Phase phase;
82  using result_t =
83  decltype(phase.Run(data_, temp_zone, std::forward<Args>(args)...));
84  if constexpr (std::is_same_v<result_t, void>) {
85  phase.Run(data_, temp_zone, std::forward<Args>(args)...);
86  if constexpr (produces_printable_graph<Phase>::value) {
87  PrintGraph(temp_zone, Phase::phase_name());
88  }
89  return !data_->info()->was_cancelled();
90  } else {
91  static_assert(std::is_same_v<result_t, std::optional<BailoutReason>>);
92  auto result = phase.Run(data_, temp_zone, std::forward<Args>(args)...);
93  if constexpr (produces_printable_graph<Phase>::value) {
94  PrintGraph(temp_zone, Phase::phase_name());
95  }
96  return data_->info()->was_cancelled() ? BailoutReason::kCancelled
97  : result;
98  }
99  UNREACHABLE();
100  }

232  V8_WARN_UNUSED_RESULT bool PrepareForInstructionSelection(
233  const ProfileDataFromFile* profile = nullptr) {
234  if (V8_UNLIKELY(data()->pipeline_kind() == TurboshaftPipelineKind::kCSA ||
235  data()->pipeline_kind() ==
236  TurboshaftPipelineKind::kTSABuiltin)) {
237  if (profile) {
238  RUN_MAYBE_ABORT(ProfileApplicationPhase, profile);
239  }
240 
241  if (v8_flags.reorder_builtins &&
242  Builtins::IsBuiltinId(info()->builtin())) {
243  UnparkedScopeIfNeeded unparked_scope(data()->broker());
244  BasicBlockCallGraphProfiler::StoreCallGraph(info(), data()->graph());
245  }
246 
247  if (v8_flags.turbo_profiling) {
248  UnparkedScopeIfNeeded unparked_scope(data()->broker());
249 
250  // Basic block profiling disables concurrent compilation, so handle
251  // deref is fine.
252  AllowHandleDereference allow_handle_dereference;
253  const size_t block_count = data()->graph().block_count();
254  BasicBlockProfilerData* profiler_data =
255  BasicBlockProfiler::Get()->NewData(block_count);
256 
257  // Set the function name.
258  profiler_data->SetFunctionName(info()->GetDebugName());
259  // Capture the schedule string before instrumentation.
260  if (v8_flags.turbo_profiling_verbose) {
261  std::ostringstream os;
262  os << data()->graph();
263  profiler_data->SetSchedule(os);
264  }
265 
266  info()->set_profiler_data(profiler_data);
267 
268  RUN_MAYBE_ABORT(BlockInstrumentationPhase);
269  } else {
270  // We run an empty copying phase to make sure that we have the same
271  // control flow as when taking the profile.
272  ZoneWithName<kTempZoneName> temp_zone(data()->zone_stats(),
273  kTempZoneName);
274  CopyingPhase<>::Run(data(), temp_zone);
275  }
276  }
277 
278  // DecompressionOptimization has to run as the last phase because it
279  // constructs an (slightly) invalid graph that mixes Tagged and Compressed
280  // representations.
281  RUN_MAYBE_ABORT(DecompressionOptimizationPhase);
282 
283  return Run<SpecialRPOSchedulingPhase>();
284  }

335  V8_WARN_UNUSED_RESULT bool AllocateRegisters(
336  CallDescriptor* call_descriptor) {
337  BeginPhaseKind("V8.TFRegisterAllocation");
338 
339  bool run_verifier = v8_flags.turbo_verify_allocation;
340 
341  // Allocate registers.
342  const RegisterConfiguration* config = RegisterConfiguration::Default();
343  std::unique_ptr<const RegisterConfiguration> restricted_config;
344  if (call_descriptor->HasRestrictedAllocatableRegisters()) {
345  RegList registers = call_descriptor->AllocatableRegisters();
346  DCHECK_LT(0, registers.Count());
347  restricted_config.reset(
348  RegisterConfiguration::RestrictGeneralRegisters(registers));
349  config = restricted_config.get();
350  }
351  if (!AllocateRegisters(config, call_descriptor, run_verifier)) return false;
352 
353  // Verify the instruction sequence has the same hash in two stages.
354  VerifyGeneratedCodeIsIdempotent();
355 
356  RUN_MAYBE_ABORT(FrameElisionPhase);
357 
358  // TODO(mtrofin): move this off to the register allocator.
359  bool generate_frame_at_start =
360  data_->sequence()->instruction_blocks().front()->must_construct_frame();
361  // Optimimize jumps.
362  if (v8_flags.turbo_jt) {
363  RUN_MAYBE_ABORT(JumpThreadingPhase, generate_frame_at_start);
364  }
365 
366  EndPhaseKind();
367 
368  return !info()->was_cancelled();
369  }

376  void VerifyGeneratedCodeIsIdempotent() {
377  JumpOptimizationInfo* jump_opt = data()->jump_optimization_info();
378  if (jump_opt == nullptr) return;
379 
380  InstructionSequence* code = data()->sequence();
381  int instruction_blocks = code->InstructionBlockCount();
382  int virtual_registers = code->VirtualRegisterCount();
383  size_t hash_code =
384  base::hash_combine(instruction_blocks, virtual_registers);
385  for (Instruction* instr : *code) {
386  hash_code = base::hash_combine(hash_code, instr->opcode(),
387  instr->InputCount(), instr->OutputCount());
388  }
389  for (int i = 0; i < virtual_registers; i++) {
390  hash_code = base::hash_combine(hash_code, code->GetRepresentation(i));
391  }
392  if (jump_opt->is_collecting()) {
393  jump_opt->hash_code = hash_code;
394  } else {
395  CHECK_EQ(hash_code, jump_opt->hash_code);
396  }
397  }

425  MaybeHandle<Code> GenerateCode(CallDescriptor* call_descriptor) {
426  Linkage linkage(call_descriptor);
427  if (!PrepareForInstructionSelection()) return {};

443  MaybeIndirectHandle<Code> FinalizeCode(bool retire_broker = true) {
444  BeginPhaseKind("V8.TFFinalizeCode");
445  if (data_->broker() && retire_broker) {
446  data_->broker()->Retire();
447  }
448  RUN_MAYBE_ABORT(FinalizeCodePhase);
449 
450  MaybeIndirectHandle<Code> maybe_code = data_->code();
451  IndirectHandle<Code> code;
452  if (!maybe_code.ToHandle(&code)) {
453  return maybe_code;
454  }
455 
456  data_->info()->SetCode(code);
457  PrintCode(data_->isolate(), code, data_->info());
458 
459  // Functions with many inline candidates are sensitive to correct call
460  // frequency feedback and should therefore not be tiered up early.
461  if (v8_flags.profile_guided_optimization &&
462  info()->could_not_inline_all_candidates() &&
463  info()->shared_info()->cached_tiering_decision() !=
464  CachedTieringDecision::kDelayMaglev) {
465  info()->shared_info()->set_cached_tiering_decision(
466  CachedTieringDecision::kNormal);
467  }
468 
469  if (info()->trace_turbo_json()) {
470  TurboJsonFile json_of(info(), std::ios_base::app);
471 
472  json_of << "{\"name\":\"disassembly\",\"type\":\"disassembly\""
473  << BlockStartsAsJSON{&data_->code_generator()->block_starts()}
474  << "\"data\":\"";
475 #ifdef ENABLE_DISASSEMBLER
476  std::stringstream disassembly_stream;
477  code->Disassemble(nullptr, disassembly_stream, data_->isolate());
478  std::string disassembly_string(disassembly_stream.str());
479  for (const auto& c : disassembly_string) {
480  json_of << AsEscapedUC16ForJSON(c);
481  }
482 #endif // ENABLE_DISASSEMBLER
483  json_of << "\"}\n],\n";
484  json_of << "\"nodePositions\":";
485  // TODO(nicohartmann): We should try to always provide source positions.
486  json_of << (data_->source_position_output().empty()
487  ? "{}"
488  : data_->source_position_output())
489  << ",\n";
490  JsonPrintAllSourceWithPositions(json_of, data_->info(), data_->isolate());
491  if (info()->has_bytecode_array()) {
492  json_of << ",\n";
493  JsonPrintAllBytecodeSources(json_of, info(), data_->isolate());
494  }
495  json_of << "\n}";
496  }
497  if (info()->trace_turbo_json() || info()->trace_turbo_graph()) {
498  CodeTracer::StreamScope tracing_scope(data_->GetCodeTracer());
499  tracing_scope.stream()
500  << "---------------------------------------------------\n"
501  << "Finished compiling method " << info()->GetDebugName().get()
502  << " using TurboFan" << std::endl;
503  }
504  EndPhaseKind();
505  return code;
506  }
