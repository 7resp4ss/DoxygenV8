21 String16 calculateHash(v8::Isolate* isolate, v8::Local<v8::String> source) {
22  uint32_t length = source->Length();
23  std::unique_ptr<UChar[]> buffer(new UChar[length]);
24  source->WriteV2(isolate, 0, length,
25  reinterpret_cast<uint16_t*>(buffer.get()));
26 
27  const uint8_t* data = nullptr;
28  size_t sizeInBytes = sizeof(UChar) * length;
29  data = reinterpret_cast<const uint8_t*>(buffer.get());
30 
31  uint8_t hash[kSizeOfSha256Digest];
32  v8::internal::SHA256_hash(data, sizeInBytes, hash);
33 
34  String16Builder formatted_hash;
35  for (size_t i = 0; i < kSizeOfSha256Digest; i++)
36  formatted_hash.appendUnsignedAsHex(static_cast<uint8_t>(hash[i]));
37 
38  return formatted_hash.toString();
39 }

56 String16 V8DebuggerScript::source(size_t pos, size_t len) const {
57  v8::HandleScope scope(m_isolate);
58  v8::Local<v8::String> v8Source;
59  if (!m_scriptSource.Get(m_isolate)->JavaScriptCode().ToLocal(&v8Source)) {
60  return String16();
61  }
62  if (pos >= static_cast<size_t>(v8Source->Length())) return String16();
63  size_t substringLength =
64  std::min(len, static_cast<size_t>(v8Source->Length()) - pos);
65  std::unique_ptr<UChar[]> buffer(new UChar[substringLength]);
66  v8Source->WriteV2(m_isolate, static_cast<uint32_t>(pos),
67  static_cast<uint32_t>(substringLength),
68  reinterpret_cast<uint16_t*>(buffer.get()));
69  return String16(buffer.get(), substringLength);
70 }

118 void V8DebuggerScript::setSource(const String16& newSource, bool preview,
119  bool allowTopFrameLiveEditing,
120  v8::debug::LiveEditResult* result) {
121  v8::EscapableHandleScope scope(m_isolate);
122  v8::Local<v8::String> v8Source = toV8String(m_isolate, newSource);
123  if (!m_script.Get(m_isolate)->SetScriptSource(
124  v8Source, preview, allowTopFrameLiveEditing, result)) {
125  result->message = scope.Escape(result->message);
126  return;
127  }
128  // NOP if preview or unchanged source (diffs.empty() in PatchScript)
129  if (preview || result->script.IsEmpty()) return;
130 
131  m_hash = String16();
132  Initialize(scope.Escape(result->script));
133 }

135 bool V8DebuggerScript::getPossibleBreakpoints(
136  const v8::debug::Location& start, const v8::debug::Location& end,
137  bool restrictToFunction, std::vector<v8::debug::BreakLocation>* locations) {
138  v8::HandleScope scope(m_isolate);
139  v8::Local<v8::debug::Script> script = m_script.Get(m_isolate);
140  std::vector<v8::debug::BreakLocation> allLocations;
141  if (!script->GetPossibleBreakpoints(start, end, restrictToFunction,
142  &allLocations)) {
143  return false;
144  }
145  if (allLocations.empty()) return true;
146  v8::debug::BreakLocation current = allLocations[0];
147  for (size_t i = 1; i < allLocations.size(); ++i) {
148  if (allLocations[i].GetLineNumber() == current.GetLineNumber() &&
149  allLocations[i].GetColumnNumber() == current.GetColumnNumber()) {
150  if (allLocations[i].type() != v8::debug::kCommonBreakLocation) {
151  DCHECK(allLocations[i].type() == v8::debug::kCallBreakLocation ||
152  allLocations[i].type() == v8::debug::kReturnBreakLocation);
153  // debugger can returns more then one break location at the same
154  // source location, e.g. foo() - in this case there are two break
155  // locations before foo: for statement and for function call, we can
156  // merge them for inspector and report only one with call type.
157  current = allLocations[i];
158  }
159  } else {
160  // we assume that returned break locations are sorted.
161  DCHECK(allLocations[i].GetLineNumber() > current.GetLineNumber() ||
162  (allLocations[i].GetColumnNumber() >= current.GetColumnNumber() &&
163  allLocations[i].GetLineNumber() == current.GetLineNumber()));
164  locations->push_back(current);
165  current = allLocations[i];
166  }
167  }
168  locations->push_back(current);
169  return true;
170 }

202 const String16& V8DebuggerScript::hash() const {
203  if (!m_hash.isEmpty()) return m_hash;
204  v8::HandleScope scope(m_isolate);
205  v8::Local<v8::String> v8Source;
206  if (!m_scriptSource.Get(m_isolate)->JavaScriptCode().ToLocal(&v8Source)) {
207  v8Source = v8::String::Empty(m_isolate);
208  }
209  m_hash = calculateHash(m_isolate, v8Source);
210  DCHECK(!m_hash.isEmpty());
211  return m_hash;
212 }

214 String16 V8DebuggerScript::buildId() const {
215  if (!m_buildId.isEmpty()) return m_buildId;
216  v8::Local<v8::debug::Script> script = this->script();
217 #if V8_ENABLE_WEBASSEMBLY
218  if (m_language == Language::WebAssembly) {
219  auto maybe_build_id =
220  v8::debug::WasmScript::Cast(*script)->GetModuleBuildId();
221  if (maybe_build_id.IsJust()) {
222  v8::MemorySpan<const uint8_t> buildId = maybe_build_id.FromJust();
223  String16Builder buildIdFormatter;
224  for (size_t i = 0; i < buildId.size(); i++) {
225  buildIdFormatter.appendUnsignedAsHex(
226  static_cast<uint8_t>(buildId[i]));
227  }
228  m_buildId = buildIdFormatter.toString();
229  }
230  }
231 #endif // V8_ENABLE_WEBASSEMBLY
232  if (m_language == Language::JavaScript) {
233  v8::Local<v8::String> debugId;
234  if (script->DebugId().ToLocal(&debugId)) {
235  m_buildId = toProtocolString(m_isolate, debugId);
236  }
237  }
238  return m_buildId;
239 }

273 void V8DebuggerScript::Initialize(v8::Local<v8::debug::Script> script) {
274  v8::Local<v8::String> tmp;
275  m_hasSourceURLComment =
276  script->SourceURL().ToLocal(&tmp) && tmp->Length() > 0;
277  if (script->SourceMappingURL().ToLocal(&tmp))
278  m_sourceMappingURL = toProtocolString(m_isolate, tmp);
279  m_startLine = script->StartLine();
280  m_startColumn = script->StartColumn();
281  m_endLine = script->EndLine();
282  m_endColumn = script->EndColumn();
283 
284  USE(script->ContextId().To(&m_executionContextId));
285  m_language = V8DebuggerScript::Language::JavaScript;
286 #if V8_ENABLE_WEBASSEMBLY
287  if (script->IsWasm()) {
288  m_language = V8DebuggerScript::Language::WebAssembly;
289  }
290 #endif // V8_ENABLE_WEBASSEMBLY
291 
292  m_isModule = script->IsModule();
293 
294  bool hasHash = script->GetSha256Hash().ToLocal(&tmp) && tmp->Length() > 0;
295  if (hasHash) {
296  m_hash = toProtocolString(m_isolate, tmp);
297  }
298 
299  m_script.Reset(m_isolate, script);
300  m_script.AnnotateStrongRetainer(kGlobalDebuggerScriptHandleLabel);
301  m_scriptSource.Reset(m_isolate, script->Source());
302  m_scriptSource.AnnotateStrongRetainer(kGlobalDebuggerScriptHandleLabel);
303 }
