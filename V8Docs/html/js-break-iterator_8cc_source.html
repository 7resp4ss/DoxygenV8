20 MaybeDirectHandle<JSV8BreakIterator> JSV8BreakIterator::New(
21  Isolate* isolate, DirectHandle<Map> map, DirectHandle<Object> locales,
22  DirectHandle<Object> options_obj, const char* service) {
23  Factory* factory = isolate->factory();
24 
25  // 1. Let requestedLocales be ? CanonicalizeLocaleList(locales).
26  Maybe<std::vector<std::string>> maybe_requested_locales =
27  Intl::CanonicalizeLocaleList(isolate, locales);
28  MAYBE_RETURN(maybe_requested_locales, MaybeDirectHandle<JSV8BreakIterator>());
29  std::vector<std::string> requested_locales =
30  maybe_requested_locales.FromJust();
31 
32  DirectHandle<JSReceiver> options;
33  if (IsUndefined(*options_obj, isolate)) {
34  options = factory->NewJSObjectWithNullProto();
35  } else {
36  ASSIGN_RETURN_ON_EXCEPTION(isolate, options,
37  Object::ToObject(isolate, options_obj, service));
38  }
39 
40  // Extract locale string
41  Maybe<Intl::MatcherOption> maybe_locale_matcher =
42  Intl::GetLocaleMatcher(isolate, options, service);
43  MAYBE_RETURN(maybe_locale_matcher, MaybeDirectHandle<JSV8BreakIterator>());
44  Intl::MatcherOption matcher = maybe_locale_matcher.FromJust();
45 
46  Intl::ResolvedLocale r;
47  if (!Intl::ResolveLocale(isolate, JSV8BreakIterator::GetAvailableLocales(),
48  requested_locales, matcher, {})
49  .To(&r)) {
50  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
51  }
52 
53  // Extract type from options
54  enum class Type { CHARACTER, WORD, SENTENCE, LINE };
55  Maybe<Type> maybe_type = GetStringOption<Type>(
56  isolate, options, "type", service,
57  std::to_array<const std::string_view>(
58  {"word", "character", "sentence", "line"}),
59  std::array{Type::WORD, Type::CHARACTER, Type::SENTENCE, Type::LINE},
60  Type::WORD);
61  MAYBE_RETURN(maybe_type, MaybeDirectHandle<JSV8BreakIterator>());
62  Type type_enum = maybe_type.FromJust();
63 
64  icu::Locale icu_locale = r.icu_locale;
65  DCHECK(!icu_locale.isBogus());
66 
67  // Construct break_iterator using icu_locale and type
68  UErrorCode status = U_ZERO_ERROR;
69  std::unique_ptr<icu::BreakIterator> break_iterator = nullptr;
70  switch (type_enum) {
71  case Type::CHARACTER:
72  break_iterator.reset(
73  icu::BreakIterator::createCharacterInstance(icu_locale, status));
74  break;
75  case Type::SENTENCE:
76  break_iterator.reset(
77  icu::BreakIterator::createSentenceInstance(icu_locale, status));
78  break;
79  case Type::LINE:
80  isolate->CountUsage(
81  v8::Isolate::UseCounterFeature::kBreakIteratorTypeLine);
82  break_iterator.reset(
83  icu::BreakIterator::createLineInstance(icu_locale, status));
84  break;
85  default:
86  isolate->CountUsage(
87  v8::Isolate::UseCounterFeature::kBreakIteratorTypeWord);
88  break_iterator.reset(
89  icu::BreakIterator::createWordInstance(icu_locale, status));
90  break;
91  }
92 
93  // Error handling for break_iterator
94  if (U_FAILURE(status) || break_iterator == nullptr) {
95  THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
96  }
97  isolate->CountUsage(v8::Isolate::UseCounterFeature::kBreakIterator);
98 
99  // Construct managed objects from pointers
100  DirectHandle<Managed<icu::BreakIterator>> managed_break_iterator =
101  Managed<icu::BreakIterator>::From(isolate, 0, std::move(break_iterator));
102  DirectHandle<Managed<icu::UnicodeString>> managed_unicode_string =
103  Managed<icu::UnicodeString>::From(isolate, 0, nullptr);
104 
105  DirectHandle<String> locale_str =
106  isolate->factory()->NewStringFromAsciiChecked(r.locale.c_str());
107 
108  // Now all properties are ready, so we can allocate the result object.
109  DirectHandle<JSV8BreakIterator> break_iterator_holder =
110  Cast<JSV8BreakIterator>(
111  isolate->factory()->NewFastOrSlowJSObjectFromMap(map));
112  DisallowGarbageCollection no_gc;
113  break_iterator_holder->set_locale(*locale_str);
114  break_iterator_holder->set_break_iterator(*managed_break_iterator);
115  break_iterator_holder->set_unicode_string(*managed_unicode_string);
116 
117  // Return break_iterator_holder
118  return break_iterator_holder;
119 }

121 DirectHandle<JSObject> JSV8BreakIterator::ResolvedOptions(
122  Isolate* isolate, DirectHandle<JSV8BreakIterator> break_iterator) {
123  Factory* factory = isolate->factory();
124  const auto as_string = [isolate](icu::BreakIterator* break_iterator) {
125  // Since the developer calling the Intl.v8BreakIterator already know the
126  // type, we usually do not need to know the type unless the
127  // resolvedOptions() is called, we use the following trick to figure out the
128  // type instead of storing it with the JSV8BreakIterator object to save
129  // memory. This routine is not fast but should be seldom used only.
130 
131  // We need to clone a copy of break iterator because we need to setText to
132  // it.
133  std::unique_ptr<icu::BreakIterator> cloned_break_iterator(
134  break_iterator->clone());
135  // Use a magic string "He is." to call next().
136  // character type: will return 1 for "H"
137  // word type: will return 2 for "He"
138  // line type: will return 3 for "He "
139  // sentence type: will return 6 for "He is."
140  icu::UnicodeString data("He is.");
141  cloned_break_iterator->setText(data);
142  switch (cloned_break_iterator->next()) {
143  case 1: // After "H"
144  return isolate->factory()->character_string();
145  case 2: // After "He"
146  return isolate->factory()->word_string();
147  case 3: // After "He "
148  return isolate->factory()->line_string();
149  case 6: // After "He is."
150  return isolate->factory()->sentence_string();
151  default:
152  UNREACHABLE();
153  }
154  };
155 
156  DirectHandle<JSObject> result =
157  factory->NewJSObject(isolate->object_function());
158  DirectHandle<String> locale(break_iterator->locale(), isolate);
159 
160  JSObject::AddProperty(isolate, result, factory->locale_string(), locale,
161  NONE);
162  JSObject::AddProperty(isolate, result, factory->type_string(),
163  as_string(break_iterator->break_iterator()->raw()),
164  NONE);
165  return result;
166 }

197 Tagged<String> JSV8BreakIterator::BreakType(
198  Isolate* isolate, DirectHandle<JSV8BreakIterator> break_iterator) {
199  int32_t status = break_iterator->break_iterator()->raw()->getRuleStatus();
200  // Keep return values in sync with JavaScript BreakType enum.
201  if (status >= UBRK_WORD_NONE && status < UBRK_WORD_NONE_LIMIT) {
202  return ReadOnlyRoots(isolate).none_string();
203  }
204  if (status >= UBRK_WORD_NUMBER && status < UBRK_WORD_NUMBER_LIMIT) {
205  return ReadOnlyRoots(isolate).number_string();
206  }
207  if (status >= UBRK_WORD_LETTER && status < UBRK_WORD_LETTER_LIMIT) {
208  return ReadOnlyRoots(isolate).letter_string();
209  }
210  if (status >= UBRK_WORD_KANA && status < UBRK_WORD_KANA_LIMIT) {
211  return ReadOnlyRoots(isolate).kana_string();
212  }
213  if (status >= UBRK_WORD_IDEO && status < UBRK_WORD_IDEO_LIMIT) {
214  return ReadOnlyRoots(isolate).ideo_string();
215  }
216  return ReadOnlyRoots(isolate).unknown_string();
217 }
