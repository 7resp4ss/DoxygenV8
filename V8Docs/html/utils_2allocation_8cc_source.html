123 void* AllocWithRetry(size_t size, MallocFn malloc_fn) {
124  void* result = nullptr;
125  for (int i = 0; i < kAllocationTries; ++i) {
126  result = malloc_fn(size);
127  if (V8_LIKELY(result != nullptr)) break;
128  OnCriticalMemoryPressure();
129  }
130  return result;
131 }

133 base::AllocationResult<void*> AllocAtLeastWithRetry(size_t size) {
134  base::AllocationResult<char*> result = {nullptr, 0u};
135  for (int i = 0; i < kAllocationTries; ++i) {
136  result = base::AllocateAtLeast<char>(size);
137  if (V8_LIKELY(result.ptr != nullptr)) break;
138  OnCriticalMemoryPressure();
139  }
140  return {result.ptr, result.count};

165 void* AllocatePages(v8::PageAllocator* page_allocator, size_t size,
166  size_t alignment, PageAllocator::Permission access,
167  PageAllocator::AllocationHint hint) {
168  DCHECK_NOT_NULL(page_allocator);
169  DCHECK(IsAligned(reinterpret_cast<Address>(hint.Address()), alignment));
170  DCHECK(IsAligned(size, page_allocator->AllocatePageSize()));
171  if (!hint.Address() && v8_flags.randomize_all_allocations) {
172  hint = hint.WithAddress(
173  AlignedAddress(page_allocator->GetRandomMmapAddr(), alignment));
174  }
175  void* result = nullptr;
176  for (int i = 0; i < kAllocationTries; ++i) {
177  result = page_allocator->AllocatePages(size, alignment, access, hint);
178  if (V8_LIKELY(result != nullptr)) break;
179  OnCriticalMemoryPressure();
180  }
181  return result;
182 }

213 VirtualMemory::VirtualMemory(v8::PageAllocator* page_allocator, size_t size,
214  PageAllocator::AllocationHint hint,
215  size_t alignment,
216  PageAllocator::Permission permissions)
217  : page_allocator_(page_allocator) {
218  DCHECK_NOT_NULL(page_allocator);
219  DCHECK(IsAligned(size, page_allocator_->CommitPageSize()));
220  const size_t page_size = page_allocator_->AllocatePageSize();
221  alignment = RoundUp(alignment, page_size);
222  Address address = reinterpret_cast<Address>(AllocatePages(
223  page_allocator_, RoundUp(size, page_size), alignment, permissions, hint));
224  if (address != kNullAddress) {
225  DCHECK(IsAligned(address, alignment));
226  region_ = base::AddressRegion(address, size);
227  }

277 size_t VirtualMemory::Release(Address free_start) {
278  DCHECK(IsReserved());
279  DCHECK(IsAligned(free_start, page_allocator_->CommitPageSize()));
280  // Notice: Order is important here. The VirtualMemory object might live
281  // inside the allocated region.
282 
283  const size_t old_size = region_.size();
284  const size_t free_size = old_size - (free_start - region_.begin());
285  CHECK(InVM(free_start, free_size));
286  region_.set_size(old_size - free_size);
287  ReleasePages(page_allocator_, reinterpret_cast<void*>(region_.begin()),
288  old_size, region_.size());
289  return free_size;
290 }

292 void VirtualMemory::Free() {
293  DCHECK(IsReserved());
294  // Notice: Order is important here. The VirtualMemory object might live
295  // inside the allocated region.
296  v8::PageAllocator* page_allocator = page_allocator_;
297  base::AddressRegion region = region_;
298  Reset();
299  // FreePages expects size to be aligned to allocation granularity however
300  // ReleasePages may leave size at only commit granularity. Align it here.
301  FreePages(page_allocator, reinterpret_cast<void*>(region.begin()),
302  RoundUp(region.size(), page_allocator->AllocatePageSize()));
303 }

324 bool VirtualMemoryCage::InitReservation(
325  const ReservationParams& params, base::AddressRegion existing_reservation) {
326  DCHECK(!reservation_.IsReserved());
327 
328  const size_t allocate_page_size = params.page_allocator->AllocatePageSize();
329  CHECK(IsAligned(params.reservation_size, allocate_page_size));
330  CHECK(params.base_alignment == ReservationParams::kAnyBaseAlignment ||
331  IsAligned(params.base_alignment, allocate_page_size));
332 
333  if (!existing_reservation.is_empty()) {
334  CHECK_EQ(existing_reservation.size(), params.reservation_size);
335  CHECK(params.base_alignment == ReservationParams::kAnyBaseAlignment ||
336  IsAligned(existing_reservation.begin(), params.base_alignment));
337  reservation_ =
338  VirtualMemory(params.page_allocator, existing_reservation.begin(),
339  existing_reservation.size());
340  base_ = reservation_.address();
341  } else {
342  Address hint = params.requested_start_hint;
343  // Require the hint to be properly aligned because here it's not clear
344  // anymore whether it should be rounded up or down.
345  CHECK(IsAligned(hint, params.base_alignment));
346  VirtualMemory reservation(params.page_allocator, params.reservation_size,
347  v8::PageAllocator::AllocationHint().WithAddress(
348  reinterpret_cast<void*>(hint)),
349  params.base_alignment, params.permissions);
350  // The virtual memory reservation fails only due to OOM.
351  if (!reservation.IsReserved()) return false;
352 
353  reservation_ = std::move(reservation);
354  base_ = reservation_.address();
355  CHECK_EQ(reservation_.size(), params.reservation_size);
356  }
357  CHECK_NE(base_, kNullAddress);
358  CHECK(IsAligned(base_, params.base_alignment));
359 
360  const Address allocatable_base = RoundUp(base_, params.page_size);
361  const size_t allocatable_size = RoundDown(
362  params.reservation_size - (allocatable_base - base_), params.page_size);
363  size_ = allocatable_base + allocatable_size - base_;
364 
365  page_allocator_ = std::make_unique<base::BoundedPageAllocator>(
366  params.page_allocator, allocatable_base, allocatable_size,
367  params.page_size, params.page_initialization_mode,
368  params.page_freeing_mode);
369  return true;
370 }
