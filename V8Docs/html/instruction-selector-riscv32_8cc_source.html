21 bool RiscvOperandGenerator::CanBeImmediate(int64_t value,
22  InstructionCode opcode) {
23  switch (ArchOpcodeField::decode(opcode)) {
24  case kRiscvShl32:
25  case kRiscvSar32:
26  case kRiscvShr32:
27  return is_uint5(value);
28  case kRiscvAdd32:
29  case kRiscvAnd32:
30  case kRiscvAnd:
31  case kRiscvOr32:
32  case kRiscvOr:
33  case kRiscvTst32:
34  case kRiscvXor:
35  return is_int12(value);
36  case kRiscvLb:
37  case kRiscvLbu:
38  case kRiscvSb:
39  case kRiscvLh:
40  case kRiscvLhu:
41  case kRiscvSh:
42  case kRiscvLw:
43  case kRiscvSw:
44  case kRiscvLoadFloat:
45  case kRiscvStoreFloat:
46  case kRiscvLoadDouble:
47  case kRiscvStoreDouble:
48  return is_int32(value);
49  default:
50  return is_int12(value);
51  }
52 }

54 void EmitLoad(InstructionSelector* selector, OpIndex node,
55  InstructionCode opcode, OpIndex output = OpIndex()) {
56  RiscvOperandGenerator g(selector);
57  const Operation& op = selector->Get(node);
58  const LoadOp& load = op.Cast<LoadOp>();
59  // The LoadStoreSimplificationReducer transforms all loads into
60  // *(base + index).
61  OpIndex base = load.base();
62  OptionalOpIndex index = load.index();
63  DCHECK_EQ(load.offset, 0);
64  DCHECK_EQ(load.element_size_log2, 0);
65 
66  InstructionOperand inputs[3];
67  size_t input_count = 0;
68  InstructionOperand output_op;
69 
70  // If output is valid, use that as the output register. This is used when we
71  // merge a conversion into the load.
72  output_op = g.DefineAsRegister(output.valid() ? output : node);
73 
74  const Operation& base_op = selector->Get(base);
75  if (base_op.Is<Opmask::kExternalConstant>() && g.IsIntegerConstant(index)) {
76  const ConstantOp& constant_base = base_op.Cast<ConstantOp>();
77  if (selector->CanAddressRelativeToRootsRegister(
78  constant_base.external_reference())) {
79  ptrdiff_t const delta =
80  *g.GetOptionalIntegerConstant(index.value()) +
81  MacroAssemblerBase::RootRegisterOffsetForExternalReference(
82  selector->isolate(), constant_base.external_reference());
83  input_count = 1;
84  // Check that the delta is a 32-bit integer due to the limitations of
85  // immediate operands.
86  if (is_int32(delta)) {
87  inputs[0] = g.UseImmediate(static_cast<int32_t>(delta));
88  opcode |= AddressingModeField::encode(kMode_Root);
89  selector->Emit(opcode, 1, &output_op, input_count, inputs);
90  return;
91  }
92  }
93  }
94 
95  if (base_op.Is<LoadRootRegisterOp>() && g.IsIntegerConstant(index)) {
96  input_count = 1;
97  inputs[0] = g.UseImmediate64(*g.GetOptionalIntegerConstant(index.value()));
98  opcode |= AddressingModeField::encode(kMode_Root);
99  selector->Emit(opcode, 1, &output_op, input_count, inputs);
100  return;
101  }
102 
103  if (load.index().has_value() && g.CanBeImmediate(index.value(), opcode)) {
104  selector->Emit(opcode | AddressingModeField::encode(kMode_MRI),
105  g.DefineAsRegister(output.valid() ? output : node),
106  g.UseRegister(base), g.UseImmediate(index.value()));
107  } else {
108  if (index.has_value()) {
109  InstructionOperand addr_reg = g.TempRegister();
110  selector->Emit(kRiscvAdd32 | AddressingModeField::encode(kMode_None),
111  addr_reg, g.UseRegister(index.value()),
112  g.UseRegister(base));
113  // Emit desired load opcode, using temp addr_reg.
114  selector->Emit(opcode | AddressingModeField::encode(kMode_MRI),
115  g.DefineAsRegister(output.valid() ? output : node),
116  addr_reg, g.TempImmediate(0));
117  } else {
118  selector->Emit(opcode | AddressingModeField::encode(kMode_MRI),
119  g.DefineAsRegister(output.valid() ? output : node),
120  g.UseRegister(base), g.TempImmediate(0));
121  }
122  }
123 }

125 void EmitS128Load(InstructionSelector* selector, OpIndex node,
126  InstructionCode opcode, VSew sew, Vlmul lmul) {
127  RiscvOperandGenerator g(selector);

140  selector->Emit(kRiscvAdd32 | AddressingModeField::encode(kMode_None),

144  g.DefineAsRegister(node), addr_reg, g.TempImmediate(0),
145  g.UseImmediate(sew), g.UseImmediate(lmul));

149 void InstructionSelector::VisitStoreLane(OpIndex node) {
150  const Simd128LaneMemoryOp& store = Get(node).Cast<Simd128LaneMemoryOp>();
151  InstructionCode opcode = kRiscvS128StoreLane;
152  opcode |= LaneSizeField::encode(store.lane_size() * kBitsPerByte);
153  if (store.kind.with_trap_handler) {
154  opcode |= AccessModeField::encode(kMemoryAccessProtectedMemOutOfBounds);
155  }
156 
157  RiscvOperandGenerator g(this);
158  const Operation& op = this->Get(node);
159  DCHECK_EQ(op.input_count, 3);
160  OpIndex base = op.input(0);
161  OpIndex index = op.input(1);
162  InstructionOperand addr_reg = g.TempRegister();
163  Emit(kRiscvAdd32, addr_reg, g.UseRegister(base), g.UseRegister(index));
164  InstructionOperand inputs[4] = {
165  g.UseRegister(op.input(2)),
166  g.UseImmediate(store.lane),
167  addr_reg,
168  g.TempImmediate(0),
169  };
170  opcode |= AddressingModeField::encode(kMode_MRI);
171  Emit(opcode, 0, nullptr, 4, inputs);
172 }

194 void InstructionSelector::VisitLoad(OpIndex node) {
195  auto load = this->load_view(node);
196  LoadRepresentation load_rep = load.loaded_rep();
197  InstructionCode opcode = kArchNop;
198  switch (load_rep.representation()) {
199  case MachineRepresentation::kFloat32:
200  opcode = kRiscvLoadFloat;
201  break;
202  case MachineRepresentation::kFloat64:
203  opcode = kRiscvLoadDouble;
204  break;
205  case MachineRepresentation::kBit: // Fall through.
206  case MachineRepresentation::kWord8:
207  opcode = load_rep.IsUnsigned() ? kRiscvLbu : kRiscvLb;
208  break;
209  case MachineRepresentation::kWord16:
210  opcode = load_rep.IsUnsigned() ? kRiscvLhu : kRiscvLh;
211  break;
212  case MachineRepresentation::kTaggedSigned: // Fall through.
213  case MachineRepresentation::kTaggedPointer: // Fall through.
214  case MachineRepresentation::kTagged: // Fall through.
215  case MachineRepresentation::kWord32:
216  opcode = kRiscvLw;
217  break;
218  case MachineRepresentation::kSimd128:
219  opcode = kRiscvRvvLd;
220  break;
221  case MachineRepresentation::kCompressedPointer:
222  case MachineRepresentation::kCompressed:
223  case MachineRepresentation::kSandboxedPointer:
224  case MachineRepresentation::kMapWord: // Fall through.
225  case MachineRepresentation::kWord64:
226  case MachineRepresentation::kNone:
227  case MachineRepresentation::kSimd256: // Fall through.
228  case MachineRepresentation::kProtectedPointer: // Fall through.
229  case MachineRepresentation::kIndirectPointer:
230  case MachineRepresentation::kFloat16:
231  case MachineRepresentation::kFloat16RawBits:
232  UNREACHABLE();
233  }
234 
235  EmitLoad(this, node, opcode);
236 }

240 void InstructionSelector::VisitStore(OpIndex node) {
241  RiscvOperandGenerator g(this);
242  StoreView store_view = this->store_view(node);
243  OpIndex base = store_view.base();
244  OptionalOpIndex index = store_view.index();
245  OpIndex value = store_view.value();
246 
247  WriteBarrierKind write_barrier_kind =
248  store_view.stored_rep().write_barrier_kind();
249  MachineRepresentation rep = store_view.stored_rep().representation();
250 
251  // TODO(riscv): I guess this could be done in a better way.
252  if (write_barrier_kind != kNoWriteBarrier && index.has_value() &&
253  V8_LIKELY(!v8_flags.disable_write_barriers)) {
254  DCHECK(CanBeTaggedPointer(rep));
255  InstructionOperand inputs[4];
256  size_t input_count = 0;
257  inputs[input_count++] = g.UseUniqueRegister(base);
258  inputs[input_count++] = g.UseUniqueRegister(store_view.index().value());
259  inputs[input_count++] = g.UseUniqueRegister(value);
260  RecordWriteMode record_write_mode =
261  WriteBarrierKindToRecordWriteMode(write_barrier_kind);
262  InstructionOperand temps[] = {g.TempRegister(), g.TempRegister()};
263  size_t const temp_count = arraysize(temps);
264  InstructionCode code;
265  if (rep == MachineRepresentation::kIndirectPointer) {
266  DCHECK_EQ(write_barrier_kind, kIndirectPointerWriteBarrier);
267  // In this case we need to add the IndirectPointerTag as additional input.
268  code = kArchStoreIndirectWithWriteBarrier;
269  IndirectPointerTag tag = store_view.indirect_pointer_tag();
270  inputs[input_count++] = g.UseImmediate64(static_cast<int64_t>(tag));
271  } else {
272  code = kArchStoreWithWriteBarrier;
273  }
274  code |= RecordWriteModeField::encode(record_write_mode);
275  if (store_view.is_store_trap_on_null()) {
276  code |= AccessModeField::encode(kMemoryAccessProtectedNullDereference);
277  }
278  Emit(code, 0, nullptr, input_count, inputs, temp_count, temps);
279  } else {
280  InstructionCode code;
281  switch (rep) {
282  case MachineRepresentation::kFloat32:
283  code = kRiscvStoreFloat;
284  break;
285  case MachineRepresentation::kFloat64:
286  code = kRiscvStoreDouble;
287  break;
288  case MachineRepresentation::kBit: // Fall through.
289  case MachineRepresentation::kWord8:
290  code = kRiscvSb;
291  break;
292  case MachineRepresentation::kWord16:
293  code = kRiscvSh;
294  break;
295  case MachineRepresentation::kTaggedSigned: // Fall through.
296  case MachineRepresentation::kTaggedPointer: // Fall through.
297  case MachineRepresentation::kTagged:
298  case MachineRepresentation::kWord32:
299  code = kRiscvSw;
300  break;
301  case MachineRepresentation::kSimd128:
302  code = kRiscvRvvSt;
303  break;
304  case MachineRepresentation::kCompressedPointer: // Fall through.
305  case MachineRepresentation::kCompressed:
306  case MachineRepresentation::kSandboxedPointer:
307  case MachineRepresentation::kMapWord: // Fall through.
308  case MachineRepresentation::kNone:
309  case MachineRepresentation::kWord64:
310  case MachineRepresentation::kSimd256: // Fall through.
311  case MachineRepresentation::kProtectedPointer: // Fall through.
312  case MachineRepresentation::kIndirectPointer:
313  case MachineRepresentation::kFloat16:
314  case MachineRepresentation::kFloat16RawBits:
315  UNREACHABLE();
316  }
317 
318  if (Is<LoadRootRegisterOp>(base)) {
319  Emit(code | AddressingModeField::encode(kMode_Root), g.NoOutput(),
320  g.UseRegisterOrImmediateZero(value),
321  index.has_value() ? g.UseImmediate(store_view.index().value())
322  : g.UseImmediate(0));
323  return;
324  }
325 
326  if (index.has_value() &&
327  g.CanBeImmediate(store_view.index().value(), code)) {
328  Emit(code | AddressingModeField::encode(kMode_MRI), g.NoOutput(),
329  g.UseRegisterOrImmediateZero(value), g.UseRegister(base),
330  index.has_value() ? g.UseImmediate(store_view.index().value())
331  : g.UseImmediate(0));
332  } else {
333  if (index.has_value()) {
334  InstructionOperand addr_reg = g.TempRegister();
335  Emit(kRiscvAdd32 | AddressingModeField::encode(kMode_None), addr_reg,
336  g.UseRegister(store_view.index().value()), g.UseRegister(base));
337  // Emit desired store opcode, using temp addr_reg.
338  Emit(code | AddressingModeField::encode(kMode_MRI), g.NoOutput(),
339  g.UseRegisterOrImmediateZero(value), addr_reg, g.TempImmediate(0));
340  } else {
341  Emit(code | AddressingModeField::encode(kMode_MRI), g.NoOutput(),
342  g.UseRegisterOrImmediateZero(value), g.UseRegister(base),
343  g.UseImmediate(0));
344  }
345  }
346  }
347 }

862 // Shared routine for word comparisons against zero.
863 void InstructionSelector::VisitWordCompareZero(OpIndex user, OpIndex value,
864  FlagsContinuation* cont) {
865  // Try to combine with comparisons against 0 by simply inverting the branch.
866  while (const ComparisonOp* equal =
867  this->TryCast<Opmask::kWord32Equal>(value)) {
868  if (!CanCover(user, value)) break;
869  if (!MatchIntegralZero(equal->right())) break;
870 
871  user = value;
872  value = equal->left();
873  cont->Negate();
874  }
875 
876  const Operation& value_op = Get(value);
877  if (CanCover(user, value)) {
878  if (const ComparisonOp* comparison = value_op.TryCast<ComparisonOp>()) {
879  switch (comparison->rep.MapTaggedToWord().value()) {
880  case RegisterRepresentation::Word32():
881  cont->OverwriteAndNegateIfEqual(
882  GetComparisonFlagCondition(*comparison));
883  return VisitWordCompare(this, value, cont);
884  case RegisterRepresentation::Float32():
885  switch (comparison->kind) {
886  case ComparisonOp::Kind::kEqual:
887  cont->OverwriteAndNegateIfEqual(kEqual);
888  return VisitFloat32Compare(this, value, cont);
889  case ComparisonOp::Kind::kSignedLessThan:
890  cont->OverwriteAndNegateIfEqual(kFloatLessThan);
891  return VisitFloat32Compare(this, value, cont);
892  case ComparisonOp::Kind::kSignedLessThanOrEqual:
893  cont->OverwriteAndNegateIfEqual(kFloatLessThanOrEqual);
894  return VisitFloat32Compare(this, value, cont);
895  default:
896  UNREACHABLE();
897  }
898  case RegisterRepresentation::Float64():
899  switch (comparison->kind) {
900  case ComparisonOp::Kind::kEqual:
901  cont->OverwriteAndNegateIfEqual(kEqual);
902  return VisitFloat64Compare(this, value, cont);
903  case ComparisonOp::Kind::kSignedLessThan:
904  cont->OverwriteAndNegateIfEqual(kFloatLessThan);
905  return VisitFloat64Compare(this, value, cont);
906  case ComparisonOp::Kind::kSignedLessThanOrEqual:
907  cont->OverwriteAndNegateIfEqual(kFloatLessThanOrEqual);
908  return VisitFloat64Compare(this, value, cont);
909  default:
910  UNREACHABLE();
911  }
912  default:
913  break;
914  }
915  } else if (const ProjectionOp* projection =
916  value_op.TryCast<ProjectionOp>()) {
917  // Check if this is the overflow output projection of an
918  // <Operation>WithOverflow node.
919  if (projection->index == 1u) {
920  // We cannot combine the <Operation>WithOverflow with this branch
921  // unless the 0th projection (the use of the actual value of the
922  // <Operation> is either nullptr, which means there's no use of the
923  // actual value, or was already defined, which means it is scheduled
924  // *AFTER* this branch).
925  OpIndex node = projection->input();
926  if (const OverflowCheckedBinopOp* binop =
927  TryCast<OverflowCheckedBinopOp>(node);
928  binop && CanDoBranchIfOverflowFusion(node)) {
929  const bool is64 = binop->rep == WordRepresentation::Word64();
930  if (is64) {
931  UNREACHABLE();
932  } else {
933  switch (binop->kind) {
934  case OverflowCheckedBinopOp::Kind::kSignedAdd:
935  cont->OverwriteAndNegateIfEqual(kOverflow);
936  return VisitBinop<Int32BinopMatcher>(this, node,
937  kRiscvAddOvfWord, cont);
938  case OverflowCheckedBinopOp::Kind::kSignedSub:
939  cont->OverwriteAndNegateIfEqual(kOverflow);
940  return VisitBinop<Int32BinopMatcher>(this, node,
941  kRiscvSubOvfWord, cont);
942  case OverflowCheckedBinopOp::Kind::kSignedMul:
943  cont->OverwriteAndNegateIfEqual(kOverflow);
944  return VisitBinop<Int32BinopMatcher>(this, node, kRiscvMulOvf32,
945  cont);
946  }
947  }
948  }
949  }
950  }
951  }
952 
953  // Continuation could not be combined with a compare, emit compare against
954  // 0.
955  EmitWordCompareZero(this, value, cont);
956 }
