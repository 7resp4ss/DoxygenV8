172 // Holds all active DebugInfo objects. This is a composite data structure
173 // consisting of
174 //
175 // - an unsorted list-like structure for fast iteration and
176 // deletion-during-iteration, and
177 // - a map-like structure for fast SharedFunctionInfo-DebugInfo lookups.
178 //
179 // DebugInfos are held strongly through global handles.
180 //
181 // TODO(jgruber): Now that we use an unordered_map as the map-like structure,
182 // which supports deletion-during-iteration, the list-like part of this data
183 // structure could be removed.
184 class DebugInfoCollection final {
185  using HandleLocation = Address*;
186  using SFIUniqueId = uint32_t; // The type of SFI::unique_id.
187 
188  public:
189  explicit DebugInfoCollection(Isolate* isolate) : isolate_(isolate) {}
190 
191  void Insert(Tagged<SharedFunctionInfo> sfi, Tagged<DebugInfo> debug_info);
192 
193  bool Contains(Tagged<SharedFunctionInfo> sfi) const;
194  std::optional<Tagged<DebugInfo>> Find(Tagged<SharedFunctionInfo> sfi) const;
195 
196  void DeleteSlow(Tagged<SharedFunctionInfo> sfi);
197 
198  size_t Size() const { return list_.size(); }
199 
200  class Iterator final {
201  public:
202  explicit Iterator(DebugInfoCollection* collection)
203  : collection_(collection) {}
204 
205  bool HasNext() const {
206  return index_ < static_cast<int>(collection_->list_.size());
207  }
208 
209  Tagged<DebugInfo> Next() const {
210  DCHECK_GE(index_, 0);
211  if (!HasNext()) return {};
212  return collection_->EntryAsDebugInfo(index_);
213  }
214 
215  void Advance() {
216  DCHECK(HasNext());
217  index_++;
218  }
219 
220  void DeleteNext() {
221  DCHECK_GE(index_, 0);
222  DCHECK(HasNext());
223  collection_->DeleteIndex(index_);
224  index_--; // `Advance` must be called next.
225  }
226 
227  private:
228  using HandleLocation = DebugInfoCollection::HandleLocation;
229  DebugInfoCollection* const collection_;
230  int index_ = 0; // `int` because deletion may rewind to -1.
231  };
232 
233  private:
234  V8_EXPORT_PRIVATE Tagged<DebugInfo> EntryAsDebugInfo(size_t index) const;
235  void DeleteIndex(size_t index);
236 
237  Isolate* const isolate_;
238  std::vector<HandleLocation> list_;
239  std::unordered_map<SFIUniqueId, HandleLocation> map_;
240 };
