16 Tagged<JSFunction> SearchLiteralsMap(Tagged<CompilationCacheTable> cache,
17  InternalIndex cache_entry,
18  Tagged<Context> native_context) {
19  DisallowGarbageCollection no_gc;
20  DCHECK(IsNativeContext(native_context));
21  Tagged<Object> obj = cache->EvalJSFunctionsValueAt(cache_entry);
22 
23  // Check that there's no confusion between FixedArray and WeakFixedArray (the
24  // object used to be a FixedArray here).
25  DCHECK(!IsFixedArray(obj));
26  if (IsWeakFixedArray(obj)) {
27  Tagged<WeakFixedArray> literals_map = Cast<WeakFixedArray>(obj);
28  int length = literals_map->length();
29  for (int i = 0; i < length; i++) {
30  DCHECK(literals_map->get(i).IsWeakOrCleared());
31  Tagged<JSFunction> js_function;
32  if (literals_map->get(i).GetHeapObjectIfWeak(&js_function) &&
33  js_function->GetCreationContext().value() == native_context) {
34  return js_function;
35  }
36  }
37  }
38  return {};
39 }

41 void AddToJSFunctionMap(DirectHandle<CompilationCacheTable> cache,
42  InternalIndex cache_entry,
43  DirectHandle<JSFunction> js_function) {
44  Isolate* isolate = Isolate::Current();
45  DirectHandle<WeakFixedArray> new_js_functions;
46  int entry = -1;
47 
48  Tagged<Object> obj = cache->EvalJSFunctionsValueAt(cache_entry);
49 
50  // Check that there's no confusion between FixedArray and WeakFixedArray (the
51  // object used to be a FixedArray here).
52  DCHECK(!IsFixedArray(obj));
53  if (!IsWeakFixedArray(obj) || Cast<WeakFixedArray>(obj)->length() == 0) {
54  new_js_functions =
55  isolate->factory()->NewWeakFixedArray(1, AllocationType::kOld);
56  entry = 0;
57  } else {
58  DirectHandle<WeakFixedArray> old_js_functions(Cast<WeakFixedArray>(obj),
59  isolate);
60  // Can we reuse an entry?
61  DCHECK_LT(entry, 0);
62  int length = old_js_functions->length();
63  for (int i = 0; i < length; i++) {
64  if (old_js_functions->get(i).IsCleared()) {
65  new_js_functions = old_js_functions;
66  entry = i;
67  break;
68  }
69  }
70 
71  if (entry < 0) {
72  // Copy old optimized code map and append one new entry.
73  new_js_functions =
74  isolate->factory()->CopyWeakFixedArrayAndGrow(old_js_functions, 1);
75  entry = old_js_functions->length();
76  }
77  }
78 
79  new_js_functions->set(entry, MakeWeak(*js_function));
80 
81 #ifdef DEBUG
82  for (int i = 0; i < new_js_functions->length(); i++) {
83  Tagged<MaybeObject> object = new_js_functions->get(i);
84  DCHECK_IMPLIES(!object.IsCleared(),
85  IsJSFunction(object.GetHeapObjectAssumeWeak()));
86  }
87 #endif
88  Tagged<Object> old_js_functions = cache->EvalJSFunctionsValueAt(cache_entry);
89  if (old_js_functions != *new_js_functions) {
90  cache->SetEvalJSFunctionsValueAt(cache_entry, *new_js_functions);
91  }
92 }

94 // EvalCacheKeys are used as keys in the eval cache.
95 class EvalCacheKey : public HashTableKey {
96  public:
97  // This tuple unambiguously identifies calls to eval() or
98  // CreateDynamicFunction() (such as through the Function() constructor).
99  // * source is the string passed into eval(). For dynamic functions, this is
100  // the effective source for the function, some of which is implicitly
101  // generated.
102  // * shared is the shared function info for the function containing the call
103  // to eval(). for dynamic functions, shared is the native context closure.
104  // * When positive, position is the position in the source where eval is
105  // called. When negative, position is the negation of the position in the
106  // dynamic function's effective source where the ')' ends the parameters.
107  EvalCacheKey(DirectHandle<String> source,
108  DirectHandle<SharedFunctionInfo> shared,
109  LanguageMode language_mode, int position)
110  : HashTableKey(CompilationCacheShape::EvalHash(*source, *shared,
111  language_mode, position)),
112  source_(source),
113  shared_(shared),
114  language_mode_(language_mode),
115  position_(position) {}
116 
117  bool IsMatch(Tagged<Object> other) override {
118  DisallowGarbageCollection no_gc;
119  Tagged<FixedArray> other_array = Cast<FixedArray>(other);
120  DCHECK(IsSharedFunctionInfo(other_array->get(0)));
121  if (*shared_ != other_array->get(0)) return false;
122  int language_unchecked = Smi::ToInt(other_array->get(2));
123  DCHECK(is_valid_language_mode(language_unchecked));
124  LanguageMode language_mode = static_cast<LanguageMode>(language_unchecked);
125  if (language_mode != language_mode_) return false;
126  int position = Smi::ToInt(other_array->get(3));
127  if (position != position_) return false;
128  Tagged<String> source = Cast<String>(other_array->get(1));
129  return source->Equals(*source_);
130  }
131 
132  DirectHandle<Object> AsHandle(Isolate* isolate) {
133  DirectHandle<FixedArray> array = isolate->factory()->NewFixedArray(4);
134  array->set(0, *shared_);
135  array->set(1, *source_);
136  array->set(2, Smi::FromEnum(language_mode_));
137  array->set(3, Smi::FromInt(position_));
138  array->set_map(isolate, ReadOnlyRoots(isolate).fixed_cow_array_map());
139  return array;
140  }
141 
142  private:
143  DirectHandle<String> source_;
144  DirectHandle<SharedFunctionInfo> shared_;
145  LanguageMode language_mode_;
146  int position_;
147 };

188 Tagged<Smi> ScriptHash(Tagged<String> source,
189  MaybeDirectHandle<Object> maybe_name, int line_offset,
190  int column_offset,
191  v8::ScriptOriginOptions origin_options,
192  Isolate* isolate) {
193  DisallowGarbageCollection no_gc;
194  size_t hash = base::hash_combine(source->EnsureHash());
195  if (DirectHandle<Object> name;
196  maybe_name.ToHandle(&name) && IsString(*name, isolate)) {
197  hash =
198  base::hash_combine(hash, Cast<String>(*name)->EnsureHash(), line_offset,
199  column_offset, origin_options.Flags());
200  }
201  // The upper bits of the hash are discarded so that the value fits in a Smi.
202  return Smi::From31BitPattern(static_cast<int>(hash & (~(1u << 31))));
203 }

207 // We only reuse a cached function for some script source code if the
208 // script originates from the same place. This is to avoid issues
209 // when reporting errors, etc.
210 bool ScriptCacheKey::MatchesScript(Tagged<Script> script) {
211  DisallowGarbageCollection no_gc;
212 
213  // If the script name isn't set, the boilerplate script should have
214  // an undefined name to have the same origin.
215  Handle<Object> name;
216  if (!name_.ToHandle(&name)) {
217  return IsUndefined(script->name(), isolate_);
218  }
219  // Do the fast bailout checks first.
220  if (line_offset_ != script->line_offset()) return false;
221  if (column_offset_ != script->column_offset()) return false;
222  // Check that both names are strings. If not, no match.
223  if (!IsString(*name, isolate_) || !IsString(script->name(), isolate_))
224  return false;
225  // Are the origin_options same?
226  if (origin_options_.Flags() != script->origin_options().Flags()) {
227  return false;
228  }
229  // Compare the two name strings for equality.
230  if (!Cast<String>(*name)->Equals(Cast<String>(script->name()))) {
231  return false;
232  }
233 
234  Handle<FixedArray> wrapped_arguments_handle;
235  if (wrapped_arguments_.ToHandle(&wrapped_arguments_handle)) {
236  if (!script->is_wrapped()) {
237  return false;
238  }
239  Tagged<FixedArray> wrapped_arguments = *wrapped_arguments_handle;
240  Tagged<FixedArray> other_wrapped_arguments = script->wrapped_arguments();
241  int length = wrapped_arguments->length();
242  if (length != other_wrapped_arguments->length()) {
243  return false;
244  }
245  for (int i = 0; i < length; i++) {
246  Tagged<Object> arg = wrapped_arguments->get(i);
247  Tagged<Object> other_arg = other_wrapped_arguments->get(i);
248  DCHECK(IsString(arg));
249  DCHECK(IsString(other_arg));
250  if (!Cast<String>(arg)->Equals(Cast<String>(other_arg))) {
251  return false;
252  }
253  }
254  } else if (script->is_wrapped()) {
255  return false;
256  }
257 
258  // Don't compare host options if the script was deserialized because we didn't
259  // serialize host options (see CodeSerializer::SerializeObjectImpl())
260  if (script->deserialized() &&
261  script->host_defined_options() ==
262  ReadOnlyRoots(isolate_).empty_fixed_array()) {
263  return true;
264  }
265  // TODO(cbruni, chromium:1244145): Remove once migrated to the context
266  Handle<Object> maybe_host_defined_options;
267  if (!host_defined_options_.ToHandle(&maybe_host_defined_options)) {
268  maybe_host_defined_options = isolate_->factory()->empty_fixed_array();
269  }
270  Tagged<FixedArray> host_defined_options =
271  Cast<FixedArray>(*maybe_host_defined_options);
272  Tagged<FixedArray> script_options =
273  Cast<FixedArray>(script->host_defined_options());
274  int length = host_defined_options->length();
275  if (length != script_options->length()) return false;
276 
277  for (int i = 0; i < length; i++) {
278  // host-defined options is a v8::PrimitiveArray.
279  DCHECK(IsPrimitive(host_defined_options->get(i)));
280  DCHECK(IsPrimitive(script_options->get(i)));
281  if (!Object::StrictEquals(host_defined_options->get(i),
282  script_options->get(i))) {
283  return false;
284  }
285  }
286  return true;
287 }

329 bool ScriptCacheKey::IsMatch(Tagged<Object> other) {
330  DisallowGarbageCollection no_gc;
331  DCHECK(IsWeakFixedArray(other));
332  Tagged<WeakFixedArray> other_array = Cast<WeakFixedArray>(other);
333  DCHECK_EQ(other_array->length(), kEnd);
334 
335  // A hash check can quickly reject many non-matches, even though this step
336  // isn't strictly necessary.
337  uint32_t other_hash =
338  static_cast<uint32_t>(other_array->get(kHash).ToSmi().value());
339  if (other_hash != Hash()) return false;
340 
341  Tagged<HeapObject> other_script_object;
342  if (!other_array->get(kWeakScript)
343  .GetHeapObjectIfWeak(&other_script_object)) {
344  return false;
345  }
346  Tagged<Script> other_script = Cast<Script>(other_script_object);
347  Tagged<String> other_source = Cast<String>(other_script->source());
348 
349  return other_source->Equals(*source_) && MatchesScript(other_script);
350 }

393 CompilationCacheScriptLookupResult CompilationCacheTable::LookupScript(
394  DirectHandle<CompilationCacheTable> table, Handle<String> src,
395  const ScriptDetails& script_details, Isolate* isolate) {
396  src = String::Flatten(isolate, src);
397  ScriptCacheKey key(src, &script_details, isolate);
398  InternalIndex entry = table->FindEntry(isolate, &key);
399  if (entry.is_not_found()) return {};
400 
401  DisallowGarbageCollection no_gc;
402  Tagged<Object> key_in_table = table->KeyAt(entry);
403  Tagged<Script> script = Cast<Script>(Cast<WeakFixedArray>(key_in_table)
404  ->get(ScriptCacheKey::kWeakScript)
405  .GetHeapObjectAssumeWeak());
406 
407  Tagged<Object> obj = table->PrimaryValueAt(entry);
408  Tagged<SharedFunctionInfo> toplevel_sfi;
409  if (!IsUndefined(obj, isolate)) {
410  toplevel_sfi = Cast<SharedFunctionInfo>(obj);
411  DCHECK_EQ(toplevel_sfi->script(), script);
412  }
413 
414  return CompilationCacheScriptLookupResult::FromRawObjects(
415  std::make_pair(script, toplevel_sfi), isolate);
416 }

451 Handle<CompilationCacheTable> CompilationCacheTable::EnsureScriptTableCapacity(
452  Isolate* isolate, Handle<CompilationCacheTable> cache) {
453  if (cache->HasSufficientCapacityToAdd(1)) return cache;
454 
455  // Before resizing, delete are any entries whose keys contain cleared weak
456  // pointers.
457  {
458  DisallowGarbageCollection no_gc;
459  for (InternalIndex entry : cache->IterateEntries()) {
460  Tagged<Object> key;
461  if (!cache->ToKey(isolate, entry, &key)) continue;
462  if (Cast<WeakFixedArray>(key)
463  ->get(ScriptCacheKey::kWeakScript)
464  .IsCleared()) {
465  DCHECK(IsUndefined(cache->PrimaryValueAt(entry)));
466  cache->RemoveEntry(entry);
467  }
468  }
469  }
470 
471  return EnsureCapacity(isolate, cache);
472 }

474 DirectHandle<CompilationCacheTable> CompilationCacheTable::PutScript(
475  Handle<CompilationCacheTable> cache, Handle<String> src,
476  MaybeHandle<FixedArray> maybe_wrapped_arguments,
477  DirectHandle<SharedFunctionInfo> value, Isolate* isolate) {
478  src = String::Flatten(isolate, src);
479  DirectHandle<Script> script(Cast<Script>(value->script()), isolate);
480  MaybeHandle<Object> script_name;
481  if (IsString(script->name(), isolate)) {
482  script_name = handle(script->name(), isolate);
483  }
484  Handle<FixedArray> host_defined_options(script->host_defined_options(),
485  isolate);
486  ScriptCacheKey key(src, script_name, script->line_offset(),
487  script->column_offset(), script->origin_options(),
488  host_defined_options, maybe_wrapped_arguments, isolate);
489  DirectHandle<Object> k = key.AsHandle(isolate, value);
490 
491  // Check whether there is already a matching entry. If so, we must overwrite
492  // it. This allows an entry whose value is undefined to upgrade to contain a
493  // SharedFunctionInfo.
494  InternalIndex entry = cache->FindEntry(isolate, &key);
495  bool found_existing = entry.is_found();
496  if (!found_existing) {
497  cache = EnsureScriptTableCapacity(isolate, cache);
498  entry = cache->FindInsertionEntry(isolate, key.Hash());
499  }
500  // We might be tempted to DCHECK here that the Script in the existing entry
501  // matches the Script in the new key. However, replacing an existing Script
502  // can still happen in some edge cases that aren't common enough to be worth
503  // fixing. Consider the following unlikely sequence of events:
504  // 1. BackgroundMergeTask::SetUpOnMainThread finds a script S1 in the cache.
505  // 2. DevTools is attached and clears the cache.
506  // 3. DevTools is detached; the cache is re-enabled.
507  // 4. A new instance of the script, S2, is compiled and placed into the cache.
508  // 5. The merge from step 1 finishes on the main thread, still using S1, and
509  // places S1 into the cache, replacing S2.
510  cache->SetKeyAt(entry, *k);
511  cache->SetPrimaryValueAt(entry, *value);
512  if (!found_existing) {
513  cache->ElementAdded();
514  }
515  return cache;
516 }

536 DirectHandle<CompilationCacheTable> CompilationCacheTable::PutEval(
537  DirectHandle<CompilationCacheTable> cache, DirectHandle<String> src,
538  DirectHandle<SharedFunctionInfo> outer_info,
539  DirectHandle<JSFunction> js_function, int position) {
540  Isolate* isolate = Isolate::Current();
541  src = String::Flatten(isolate, src);
542  EvalCacheKey key(src, outer_info, js_function->shared()->language_mode(),
543  position);
544 
545  // Create a dummy entry to mark that this key has already been inserted once.
546  cache = EnsureCapacity(isolate, cache);
547  InternalIndex entry = cache->FindInsertionEntry(isolate, key.Hash());
548  DirectHandle<Object> k = key.AsHandle(isolate);
549  cache->SetKeyAt(entry, *k);
550  cache->SetPrimaryValueAt(entry, js_function->shared());
551  cache->SetEvalJSFunctionsValueAt(entry,
552  ReadOnlyRoots(isolate).the_hole_value());
553  // AddToFeedbackCellsMap may allocate a new sub-array to live in the
554  // entry, but it won't change the cache array. Therefore EntryToIndex
555  // and entry remains correct.
556  AddToJSFunctionMap(cache, entry, js_function);
557  cache->ElementAdded();
558  return cache;

585 void CompilationCacheTable::RemoveEntry(InternalIndex entry) {
586  int entry_index = EntryToIndex(entry);
587  Tagged<Object> the_hole_value = GetReadOnlyRoots().the_hole_value();
588  for (int i = 0; i < kEntrySize; i++) {
589  this->set(entry_index + i, the_hole_value, SKIP_WRITE_BARRIER);
590  }
591  ElementRemoved();
592 
593  // This table does not shrink upon deletion. The script cache depends on that
594  // fact, because EnsureScriptTableCapacity calls RemoveEntry at a time when
595  // shrinking the table would be counterproductive.
596 }
