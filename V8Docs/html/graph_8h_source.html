32 // `OperationBuffer` is a growable, Zone-allocated buffer to store Turboshaft
33 // operations. It is part of a `Graph`.
34 // The buffer can be seen as an array of 8-byte `OperationStorageSlot` values.
35 // The structure is append-only, that is, we only add operations at the end.
36 // There are rare cases (i.e., loop phis) where we overwrite an existing
37 // operation, but only if we can guarantee that the new operation is not bigger
38 // than the operation we overwrite.
39 class OperationBuffer {
40  public:
41  // A `ReplaceScope` is to overwrite an existing operation.
42  // It moves the end-pointer temporarily so that the next emitted operation
43  // overwrites an old one.
44  class ReplaceScope {
45  public:
46  ReplaceScope(OperationBuffer* buffer, OpIndex replaced)
47  : buffer_(buffer),
48  replaced_(replaced),
49  old_end_(buffer->end_),
50  old_slot_count_(buffer->SlotCount(replaced)) {
51  buffer_->end_ = buffer_->Get(replaced);
52  }
53  ~ReplaceScope() {
54  DCHECK_LE(buffer_->SlotCount(replaced_), old_slot_count_);
55  buffer_->end_ = old_end_;
56  // Preserve the original operation size in case it has become smaller.
57  buffer_->operation_sizes_[replaced_.id()] = old_slot_count_;
58  buffer_->operation_sizes_[OpIndex(replaced_.offset() +
59  static_cast<uint32_t>(old_slot_count_) *
60  sizeof(OperationStorageSlot))
61  .id() -
62  1] = old_slot_count_;
63  }
64 
65  ReplaceScope(const ReplaceScope&) = delete;
66  ReplaceScope& operator=(const ReplaceScope&) = delete;
67 
68  private:
69  OperationBuffer* buffer_;
70  OpIndex replaced_;
71  OperationStorageSlot* old_end_;
72  uint16_t old_slot_count_;
73  };
74 
75  explicit OperationBuffer(Zone* zone, size_t initial_capacity) : zone_(zone) {
76  DCHECK_NE(initial_capacity, 0);
77  begin_ = end_ =
78  zone_->AllocateArray<OperationStorageSlot>(initial_capacity);
79  operation_sizes_ =
80  zone_->AllocateArray<uint16_t>((initial_capacity + 1) / kSlotsPerId);
81  end_cap_ = begin_ + initial_capacity;
82  }
83 
84  OperationStorageSlot* Allocate(size_t slot_count) {
85  if (V8_UNLIKELY(static_cast<size_t>(end_cap_ - end_) < slot_count)) {
86  Grow(capacity() + slot_count);
87  DCHECK(slot_count <= static_cast<size_t>(end_cap_ - end_));
88  }
89  OperationStorageSlot* result = end_;
90  end_ += slot_count;
91  OpIndex idx = Index(result);
92  // Store the size in both for the first and last id corresponding to the new
93  // operation. This enables iteration in both directions. The two id's are
94  // the same if the operation is small.
95  operation_sizes_[idx.id()] = slot_count;
96  operation_sizes_[OpIndex(idx.offset() + static_cast<uint32_t>(slot_count) *
97  sizeof(OperationStorageSlot))
98  .id() -
99  1] = slot_count;
100  return result;
101  }
102 
103  void RemoveLast() {
104  size_t slot_count = operation_sizes_[EndIndex().id() - 1];
105  end_ -= slot_count;
106  DCHECK_GE(end_, begin_);
107  }
108 
109  OpIndex Index(const Operation& op) const {
110  return Index(reinterpret_cast<const OperationStorageSlot*>(&op));
111  }
112  OpIndex Index(const OperationStorageSlot* ptr) const {
113  DCHECK(begin_ <= ptr && ptr <= end_);
114  return OpIndex(static_cast<uint32_t>(reinterpret_cast<Address>(ptr) -
115  reinterpret_cast<Address>(begin_)));
116  }
117 
118  OperationStorageSlot* Get(OpIndex idx) {
119  DCHECK_LT(idx.offset() / sizeof(OperationStorageSlot), size());
120  return reinterpret_cast<OperationStorageSlot*>(
121  reinterpret_cast<Address>(begin_) + idx.offset());
122  }
123  uint16_t SlotCount(OpIndex idx) {
124  DCHECK_LT(idx.offset() / sizeof(OperationStorageSlot), size());
125  return operation_sizes_[idx.id()];
126  }
127 
128  const OperationStorageSlot* Get(OpIndex idx) const {
129  DCHECK_LT(idx.offset(), capacity() * sizeof(OperationStorageSlot));
130  return reinterpret_cast<const OperationStorageSlot*>(
131  reinterpret_cast<Address>(begin_) + idx.offset());
132  }
133 
134  OpIndex Next(OpIndex idx) const {
135  DCHECK_GT(operation_sizes_[idx.id()], 0);
136  OpIndex result = OpIndex(idx.offset() + operation_sizes_[idx.id()] *
137  sizeof(OperationStorageSlot));
138  DCHECK_LT(0, result.offset());
139  DCHECK_LE(result.offset(), capacity() * sizeof(OperationStorageSlot));
140  return result;
141  }
142  OpIndex Previous(OpIndex idx) const {
143  DCHECK_GT(idx.id(), 0);
144  DCHECK_GT(operation_sizes_[idx.id() - 1], 0);
145  OpIndex result = OpIndex(idx.offset() - operation_sizes_[idx.id() - 1] *
146  sizeof(OperationStorageSlot));
147  DCHECK_LE(0, result.offset());
148  DCHECK_LT(result.offset(), capacity() * sizeof(OperationStorageSlot));
149  return result;
150  }
151 
152  // Offset of the first operation.
153  OpIndex BeginIndex() const { return OpIndex(0); }
154  // One-past-the-end offset.
155  OpIndex EndIndex() const { return Index(end_); }
156 
157  uint32_t size() const { return static_cast<uint32_t>(end_ - begin_); }
158  uint32_t capacity() const { return static_cast<uint32_t>(end_cap_ - begin_); }
159 
160  void Grow(size_t min_capacity) {
161  size_t size = this->size();
162  size_t capacity = this->capacity();
163  size_t new_capacity = 2 * capacity;
164  while (new_capacity < min_capacity) new_capacity *= 2;
165  CHECK_LT(new_capacity, std::numeric_limits<uint32_t>::max() /
166  sizeof(OperationStorageSlot));
167 
168  OperationStorageSlot* new_buffer =
169  zone_->AllocateArray<OperationStorageSlot>(new_capacity);
170  memcpy(new_buffer, begin_, size * sizeof(OperationStorageSlot));
171 
172  uint16_t* new_operation_sizes =
173  zone_->AllocateArray<uint16_t>(new_capacity / kSlotsPerId);
174  memcpy(new_operation_sizes, operation_sizes_,
175  size / kSlotsPerId * sizeof(uint16_t));
176 
177  begin_ = new_buffer;
178  end_ = new_buffer + size;
179  end_cap_ = new_buffer + new_capacity;
180  operation_sizes_ = new_operation_sizes;
181  }
182 
183  void Reset() { end_ = begin_; }
184 
185  private:
186  Zone* zone_;
187  OperationStorageSlot* begin_;
188  OperationStorageSlot* end_;
189  OperationStorageSlot* end_cap_;
190  uint16_t* operation_sizes_;
191 };

202  // parents rather than parents to children.
203  public:

208  }

216  for (Derived* child = last_child_; child != nullptr;
217  child = child->neighboring_child_) {

1160  // to that storage to `bound_blocks_`. Initialization of the blocks is defered
1161  // to when they are actually constructed in `NewBlocks`.
1162  V8_NOINLINE V8_PRESERVE_MOST void AllocateNewBlocks() {
1163  constexpr size_t kMinCapacity = 32;
1164  size_t next_capacity = std::max(kMinCapacity, all_blocks_.size() * 2);

1171  DCHECK_EQ(new_all_blocks.size(), all_blocks_.size() + new_block_count);
1172  std::copy(all_blocks_.begin(), all_blocks_.end(), new_all_blocks.begin());
1173  Block** insert_begin = new_all_blocks.begin() + all_blocks_.size();
1174  DCHECK_EQ(insert_begin + new_block_count, new_all_blocks.end());
1175  for (size_t i = 0; i < new_block_count; ++i) {
1176  insert_begin[i] = &block_storage[i];

1185  DCHECK_LE(bound_blocks_.size(), all_blocks_.size());

1187  }

1291 inline uint32_t Block::ComputeDominator() {
1292  if (V8_UNLIKELY(LastPredecessor() == nullptr)) {
1293  // If the block has no predecessors, then it's the start block. We create a
1294  // jmp_ edge to itself, so that the SetDominator algorithm does not need a
1295  // special case for when the start block is reached.
1296  SetAsDominatorRoot();
1297  } else {
1298  // If the block has one or more predecessors, the dominator is the lowest
1299  // common ancestor (LCA) of all of the predecessors.
1300 
1301  // Note that for BranchTarget, there is a single predecessor. This doesn't
1302  // change the logic: the loop won't be entered, and the first (and only)
1303  // predecessor is set as the dominator.
1304  // Similarly, since we compute dominators on the fly, when we reach a
1305  // kLoopHeader, we haven't visited its body yet, and it should only have one
1306  // predecessor (the backedge is not here yet), which is its dominator.
1307  DCHECK_IMPLIES(kind_ == Block::Kind::kLoopHeader, PredecessorCount() == 1);
1308 
1309  Block* dominator = LastPredecessor();
1310  for (Block* pred = dominator->NeighboringPredecessor(); pred != nullptr;
1311  pred = pred->NeighboringPredecessor()) {
1312  dominator = dominator->GetCommonDominator(pred);
1313  }
1314  SetDominator(dominator);
1315  }
1316  DCHECK_NE(jmp_, nullptr);
1317  DCHECK_IMPLIES(nxt_ == nullptr, LastPredecessor() == nullptr);
1318  DCHECK_IMPLIES(len_ == 0, LastPredecessor() == nullptr);
1319  return Depth();
1320 }

1330 template <class Derived>
1331 inline void RandomAccessStackDominatorNode<Derived>::SetDominator(
1332  Derived* dominator) {
1333  DCHECK_NOT_NULL(dominator);
1334  DCHECK_NULL(static_cast<Block*>(this)->neighboring_child_);
1335  DCHECK_NULL(static_cast<Block*>(this)->last_child_);
1336  // Determining the jmp pointer
1337  Derived* t = dominator->jmp_;
1338  if (dominator->len_ - t->len_ == t->len_ - t->jmp_len_) {
1339  t = t->jmp_;
1340  } else {
1341  t = dominator;
1342  }
1343  // Initializing fields
1344  nxt_ = dominator;
1345  jmp_ = t;
1346  len_ = dominator->len_ + 1;
1347  jmp_len_ = jmp_->len_;
1348  dominator->AddChild(static_cast<Derived*>(this));
1349 }

1351 template <class Derived>
1352 inline Derived* RandomAccessStackDominatorNode<Derived>::GetCommonDominator(
1353  RandomAccessStackDominatorNode<Derived>* other) const {
1354  const RandomAccessStackDominatorNode* a = this;
1355  const RandomAccessStackDominatorNode* b = other;
1356  if (b->len_ > a->len_) {
1357  // Swapping |a| and |b| so that |a| always has a greater length.
1358  std::swap(a, b);
1359  }
1360  DCHECK_GE(a->len_, 0);
1361  DCHECK_GE(b->len_, 0);
1362 
1363  // Going up the dominators of |a| in order to reach the level of |b|.
1364  while (a->len_ != b->len_) {
1365  DCHECK_GE(a->len_, 0);
1366  if (a->jmp_len_ >= b->len_) {
1367  a = a->jmp_;
1368  } else {
1369  a = a->nxt_;
1370  }
1371  }
1372 
1373  // Going up the dominators of |a| and |b| simultaneously until |a| == |b|
1374  while (a != b) {
1375  DCHECK_EQ(a->len_, b->len_);
1376  DCHECK_GE(a->len_, 0);
1377  if (a->jmp_ == b->jmp_) {
1378  // We found a common dominator, but we actually want to find the smallest
1379  // one, so we go down in the current subtree.
1380  a = a->nxt_;
1381  b = b->nxt_;
1382  } else {
1383  a = a->jmp_;
1384  b = b->jmp_;
1385  }
1386  }
1387 
1388  return static_cast<Derived*>(
1389  const_cast<RandomAccessStackDominatorNode<Derived>*>(a));
1390 }
