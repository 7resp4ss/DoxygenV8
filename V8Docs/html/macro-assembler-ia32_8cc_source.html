91 void MacroAssembler::LoadRoot(Register destination, RootIndex index) {
92  ASM_CODE_COMMENT(this);
93  if (root_array_available()) {
94  mov(destination, RootAsOperand(index));
95  return;
96  }
97 
98  if (RootsTable::IsImmortalImmovable(index)) {
99  Handle<Object> object = isolate()->root_handle(index);
100  if (IsSmi(*object)) {
101  mov(destination, Immediate(Cast<Smi>(*object)));
102  return;
103  } else {
104  DCHECK(IsHeapObject(*object));
105  mov(destination, Cast<HeapObject>(object));
106  return;
107  }
108  }
109 
110  ExternalReference isolate_root = ExternalReference::isolate_root(isolate());
111  lea(destination,
112  Operand(isolate_root.address(), RelocInfo::EXTERNAL_REFERENCE));
113  mov(destination, Operand(destination, RootRegisterOffsetForRootIndex(index)));
114 }

183 void MacroAssembler::PushArray(Register array, Register size, Register scratch,
184  PushArrayOrder order) {
185  ASM_CODE_COMMENT(this);
186  DCHECK(!AreAliased(array, size, scratch));
187  Register counter = scratch;
188  Label loop, entry;
189  if (order == PushArrayOrder::kReverse) {
190  mov(counter, 0);
191  jmp(&entry);
192  bind(&loop);
193  Push(Operand(array, counter, times_system_pointer_size, 0));
194  inc(counter);
195  bind(&entry);
196  cmp(counter, size);
197  j(less, &loop, Label::kNear);
198  } else {
199  mov(counter, size);
200  jmp(&entry);
201  bind(&loop);
202  Push(Operand(array, counter, times_system_pointer_size, 0));
203  bind(&entry);
204  dec(counter);
205  j(greater_equal, &loop, Label::kNear);
206  }
207 }

209 Operand MacroAssembler::ExternalReferenceAsOperand(ExternalReference reference,
210  Register scratch) {
211  if (root_array_available()) {
212  if (reference.IsIsolateFieldId()) {
213  return Operand(kRootRegister, reference.offset_from_root_register());
214  }
215  if (options().enable_root_relative_access) {
216  intptr_t delta =
217  RootRegisterOffsetForExternalReference(isolate(), reference);
218  return Operand(kRootRegister, delta);
219  }
220  if (options().isolate_independent_code) {
221  if (IsAddressableThroughRootRegister(isolate(), reference)) {
222  // Some external references can be efficiently loaded as an offset from
223  // kRootRegister.
224  intptr_t offset =
225  RootRegisterOffsetForExternalReference(isolate(), reference);
226  return Operand(kRootRegister, offset);
227  } else {
228  // Otherwise, do a memory load from the external reference table.
229  mov(scratch, Operand(kRootRegister,
230  RootRegisterOffsetForExternalReferenceTableEntry(
231  isolate(), reference)));
232  return Operand(scratch, 0);
233  }
234  }
235  }
236  Move(scratch, Immediate(reference));
237  return Operand(scratch, 0);
238 }

327 int MacroAssembler::PushCallerSaved(SaveFPRegsMode fp_mode,
328  Register exclusion) {
329  ASM_CODE_COMMENT(this);
330  // We don't allow a GC in a write barrier slow path so there is no need to
331  // store the registers in any particular way, but we do have to store and
332  // restore them.
333  int bytes = 0;
334  RegList saved_regs = kCallerSaved - exclusion;
335  for (Register reg : saved_regs) {
336  push(reg);
337  bytes += kSystemPointerSize;
338  }
339 
340  if (fp_mode == SaveFPRegsMode::kSave) {
341  // Save all allocatable XMM registers.
342  int i = kAllocatableDoubleRegisters.Count();
343  const int delta = kStackSavedSavedFPSize * i;
344  AllocateStackSpace(delta);
345  for (XMMRegister reg : kAllocatableDoubleRegisters) {
346 #if V8_ENABLE_WEBASSEMBLY
347  Movdqu(Operand(esp, --i * kStackSavedSavedFPSize), reg);
348 #else
349  Movsd(Operand(esp, --i * kStackSavedSavedFPSize), reg);
350 #endif // V8_ENABLE_WEBASSEMBLY
351  }
352  bytes += delta;
353  }
354 
355  return bytes;
356 }

358 int MacroAssembler::PopCallerSaved(SaveFPRegsMode fp_mode, Register exclusion) {
359  ASM_CODE_COMMENT(this);
360  int bytes = 0;
361  if (fp_mode == SaveFPRegsMode::kSave) {
362  // Restore all allocatable XMM registers.
363  int i = kAllocatableDoubleRegisters.Count();
364  const int delta = kStackSavedSavedFPSize * i;
365  for (XMMRegister reg : kAllocatableDoubleRegisters) {
366 #if V8_ENABLE_WEBASSEMBLY
367  Movdqu(reg, Operand(esp, --i * kStackSavedSavedFPSize));
368 #else
369  Movsd(reg, Operand(esp, --i * kStackSavedSavedFPSize));
370 #endif // V8_ENABLE_WEBASSEMBLY
371  }
372  add(esp, Immediate(delta));
373  bytes += delta;
374  }
375 
376  RegList saved_regs = kCallerSaved - exclusion;
377  for (Register reg : base::Reversed(saved_regs)) {
378  pop(reg);
379  bytes += kSystemPointerSize;
380  }
381 
382  return bytes;
383 }

385 void MacroAssembler::RecordWriteField(Register object, int offset,
386  Register value, Register slot_address,
387  SaveFPRegsMode save_fp,
388  SmiCheck smi_check) {
389  ASM_CODE_COMMENT(this);
390  // First, check if a write barrier is even needed. The tests below
391  // catch stores of Smis.
392  Label done;
393 
394  // Skip barrier if writing a smi.
395  if (smi_check == SmiCheck::kInline) {
396  JumpIfSmi(value, &done);
397  }
398 
399  // Although the object register is tagged, the offset is relative to the start
400  // of the object, so so offset must be a multiple of kTaggedSize.
401  DCHECK(IsAligned(offset, kTaggedSize));
402 
403  lea(slot_address, FieldOperand(object, offset));
404  if (v8_flags.slow_debug_code) {
405  Label ok;
406  test_b(slot_address, Immediate(kTaggedSize - 1));
407  j(zero, &ok, Label::kNear);
408  int3();
409  bind(&ok);
410  }
411 
412  RecordWrite(object, slot_address, value, save_fp, SmiCheck::kOmit);
413 
414  bind(&done);
415 
416  // Clobber clobbered input registers when running with the debug-code flag
417  // turned on to provoke errors.
418  if (v8_flags.slow_debug_code) {
419  mov(value, Immediate(base::bit_cast<int32_t>(kZapValue)));
420  mov(slot_address, Immediate(base::bit_cast<int32_t>(kZapValue)));
421  }
422 }

504 void MacroAssembler::RecordWrite(Register object, Register slot_address,
505  Register value, SaveFPRegsMode fp_mode,
506  SmiCheck smi_check) {
507  ASM_CODE_COMMENT(this);
508  DCHECK(!AreAliased(object, value, slot_address));
509  AssertNotSmi(object);
510 
511  if (v8_flags.disable_write_barriers) {
512  return;
513  }
514 
515  if (v8_flags.slow_debug_code) {
516  ASM_CODE_COMMENT_STRING(this, "Verify slot_address");
517  Label ok;
518  cmp(value, Operand(slot_address, 0));
519  j(equal, &ok, Label::kNear);
520  int3();
521  bind(&ok);
522  }
523 
524  // First, check if a write barrier is even needed. The tests below
525  // catch stores of Smis and stores into young gen.
526  Label done;
527 
528  if (smi_check == SmiCheck::kInline) {
529  // Skip barrier if writing a smi.
530  JumpIfSmi(value, &done, Label::kNear);
531  }
532 
533  CheckPageFlag(value,
534  value, // Used as scratch.
535  MemoryChunk::kPointersToHereAreInterestingMask, zero, &done,
536  Label::kNear);
537  CheckPageFlag(object,
538  value, // Used as scratch.
539  MemoryChunk::kPointersFromHereAreInterestingMask, zero, &done,
540  Label::kNear);
541  RecordComment("CheckPageFlag]");
542 
543  CallRecordWriteStub(object, slot_address, fp_mode);
544 
545  bind(&done);
546 
547  // Clobber clobbered registers when running with the debug-code flag
548  // turned on to provoke errors.
549  if (v8_flags.slow_debug_code) {
550  ASM_CODE_COMMENT_STRING(this, "Clobber slot_address and value");
551  mov(slot_address, Immediate(base::bit_cast<int32_t>(kZapValue)));
552  mov(value, Immediate(base::bit_cast<int32_t>(kZapValue)));
553  }

567  Label done;
568  Register src_reg = src.is_reg_only() ? src.reg() : tmp;
569  if (src_reg == tmp) mov(tmp, src);
570  cvtsi2ss(dst, src_reg);
571  test(src_reg, src_reg);
572  j(positive, &done, Label::kNear);
573 
574  // Compute {src/2 | (src&1)} (retain the LSB to avoid rounding errors).
575  if (src_reg != tmp) mov(tmp, src_reg);

577  // The LSB is shifted into CF. If it is set, set the LSB in {tmp}.
578  Label msb_not_set;
579  j(not_carry, &msb_not_set, Label::kNear);
580  or_(tmp, Immediate(1));
581  bind(&msb_not_set);
582  cvtsi2ss(dst, tmp);
583  addss(dst, dst);
584  bind(&done);

748 void TailCallOptimizedCodeSlot(MacroAssembler* masm,
749  Register optimized_code_entry) {
750  // ----------- S t a t e -------------
751  // -- eax : actual argument count
752  // -- edx : new target (preserved for callee if needed, and caller)
753  // -- edi : target function (preserved for callee if needed, and caller)
754  // -----------------------------------
755  ASM_CODE_COMMENT(masm);
756  DCHECK(!AreAliased(edx, edi, optimized_code_entry));
757 
758  Register closure = edi;
759  __ Push(eax);
760  __ Push(edx);
761 
762  Label heal_optimized_code_slot;
763 
764  // If the optimized code is cleared, go to runtime to update the optimization
765  // marker field.
766  __ LoadWeakValue(optimized_code_entry, &heal_optimized_code_slot);
767 
768  // The entry references a CodeWrapper object. Unwrap it now.
769  __ mov(optimized_code_entry,
770  FieldOperand(optimized_code_entry, CodeWrapper::kCodeOffset));
771 
772  // Check if the optimized code is marked for deopt. If it is, bailout to a
773  // given label.
774  __ TestCodeIsMarkedForDeoptimization(optimized_code_entry);
775  __ j(not_zero, &heal_optimized_code_slot);
776 
777  // Optimized code is good, get it into the closure and link the closure
778  // into the optimized functions list, then tail call the optimized code.
779  __ Push(optimized_code_entry);
780  __ ReplaceClosureCodeWithOptimizedCode(optimized_code_entry, closure, edx,
781  ecx);
782  static_assert(kJavaScriptCallCodeStartRegister == ecx, "ABI mismatch");
783  __ Pop(optimized_code_entry);
784  __ LoadCodeInstructionStart(ecx, optimized_code_entry);
785  __ Pop(edx);
786  __ Pop(eax);
787  __ jmp(ecx);
788 
789  // Optimized code slot contains deoptimized code or code is cleared and
790  // optimized code marker isn't updated. Evict the code, update the marker
791  // and re-enter the closure's code.
792  __ bind(&heal_optimized_code_slot);
793  __ Pop(edx);
794  __ Pop(eax);
795  __ GenerateTailCallToReturnedCode(Runtime::kHealOptimizedCodeSlot);

1193 void MacroAssembler::EnterExitFrame(int extra_slots,
1194  StackFrame::Type frame_type,
1195  Register c_function) {
1196  ASM_CODE_COMMENT(this);
1197  DCHECK(frame_type == StackFrame::EXIT ||
1198  frame_type == StackFrame::BUILTIN_EXIT ||
1199  frame_type == StackFrame::API_ACCESSOR_EXIT ||
1200  frame_type == StackFrame::API_CALLBACK_EXIT);
1201 
1202  // Set up the frame structure on the stack.
1203  DCHECK_EQ(+2 * kSystemPointerSize, ExitFrameConstants::kCallerSPDisplacement);
1204  DCHECK_EQ(+1 * kSystemPointerSize, ExitFrameConstants::kCallerPCOffset);
1205  DCHECK_EQ(0 * kSystemPointerSize, ExitFrameConstants::kCallerFPOffset);
1206  push(ebp);
1207  mov(ebp, esp);
1208 
1209  push(Immediate(StackFrame::TypeToMarker(frame_type)));
1210  DCHECK_EQ(-2 * kSystemPointerSize, ExitFrameConstants::kSPOffset);
1211  push(Immediate(0)); // Saved entry sp, patched below.
1212 
1213  // Save the frame pointer and the context in top.
1214  DCHECK(!AreAliased(ebp, kContextRegister, c_function));
1215  using ER = ExternalReference;
1216  ER r0 = ER::Create(IsolateAddressId::kCEntryFPAddress, isolate());
1217  mov(ExternalReferenceAsOperand(r0, no_reg), ebp);
1218  ER r1 = ER::Create(IsolateAddressId::kContextAddress, isolate());
1219  mov(ExternalReferenceAsOperand(r1, no_reg), kContextRegister);
1220  static_assert(edx == kRuntimeCallFunctionRegister);
1221  ER r2 = ER::Create(IsolateAddressId::kCFunctionAddress, isolate());
1222  mov(ExternalReferenceAsOperand(r2, no_reg), c_function);
1223 
1224  AllocateStackSpace(extra_slots * kSystemPointerSize);
1225 
1226  // Get the required frame alignment for the OS.
1227  const int kFrameAlignment = base::OS::ActivationFrameAlignment();
1228  if (kFrameAlignment > 0) {
1229  DCHECK(base::bits::IsPowerOfTwo(kFrameAlignment));
1230  and_(esp, -kFrameAlignment);
1231  }

1234  mov(Operand(ebp, ExitFrameConstants::kSPOffset), esp);

1350 void MacroAssembler::StackOverflowCheck(Register num_args, Register scratch,
1351  Label* stack_overflow,
1352  bool include_receiver) {
1353  ASM_CODE_COMMENT(this);

1359  ExternalReference::address_of_real_jslimit(isolate());

1373  // by jumping to a builtin that throws the exception.

1377 void MacroAssembler::InvokePrologue(Register expected_parameter_count,
1378  Register actual_parameter_count,
1379  InvokeType type) {
1380  if (expected_parameter_count == actual_parameter_count) return;
1381  ASM_CODE_COMMENT(this);
1382  DCHECK_EQ(actual_parameter_count, eax);
1383  DCHECK_EQ(expected_parameter_count, ecx);
1384  Label regular_invoke;
1385 
1386  // If overapplication or if the actual argument count is equal to the
1387  // formal parameter count, no need to push extra undefined values.
1388  sub(expected_parameter_count, actual_parameter_count);
1389  j(less_equal, &regular_invoke, Label::kFar);
1390 
1391  // We need to preserve edx, edi, esi and ebx.
1392  movd(xmm0, edx);
1393  movd(xmm1, edi);
1394  movd(xmm2, esi);
1395  movd(xmm3, ebx);
1396 
1397  Label stack_overflow;
1398  StackOverflowCheck(expected_parameter_count, edx, &stack_overflow);
1399 
1400  Register scratch = esi;
1401 
1402  // Underapplication. Move the arguments already in the stack, including the
1403  // receiver and the return address.
1404  {
1405  Label copy, check;
1406  Register src = edx, dest = esp, num = edi, current = ebx;
1407  mov(src, esp);
1408  lea(scratch,
1409  Operand(expected_parameter_count, times_system_pointer_size, 0));
1410  AllocateStackSpace(scratch);
1411  // Extra words are the receiver (if not already included in argc) and the
1412  // return address (if a jump).
1413  int extra_words = type == InvokeType::kCall ? 0 : 1;
1414  lea(num, Operand(eax, extra_words)); // Number of words to copy.
1415  Move(current, 0);
1416  // Fall-through to the loop body because there are non-zero words to copy.
1417  bind(&copy);
1418  mov(scratch, Operand(src, current, times_system_pointer_size, 0));
1419  mov(Operand(dest, current, times_system_pointer_size, 0), scratch);
1420  inc(current);
1421  bind(&check);
1422  cmp(current, num);
1423  j(less, &copy);
1424  lea(edx, Operand(esp, num, times_system_pointer_size, 0));
1425  }
1426 
1427  // Fill remaining expected arguments with undefined values.
1428  movd(ebx, xmm3); // Restore root.
1429  LoadRoot(scratch, RootIndex::kUndefinedValue);
1430  {
1431  Label loop;
1432  bind(&loop);
1433  dec(expected_parameter_count);
1434  mov(Operand(edx, expected_parameter_count, times_system_pointer_size, 0),
1435  scratch);
1436  j(greater, &loop, Label::kNear);
1437  }
1438 
1439  // Restore remaining registers.
1440  movd(esi, xmm2);
1441  movd(edi, xmm1);
1442  movd(edx, xmm0);
1443 
1444  jmp(&regular_invoke);
1445 
1446  bind(&stack_overflow);
1447  {
1448  FrameScope frame(
1449  this, has_frame() ? StackFrame::NO_FRAME_TYPE : StackFrame::INTERNAL);
1450  CallRuntime(Runtime::kThrowStackOverflow);
1451  int3(); // This should be unreachable.
1452  }
1453 
1454  bind(&regular_invoke);
1455 }

1506  Register expected_parameter_count,
1507  Register actual_parameter_count,
1508  InvokeType type) {
1509  ASM_CODE_COMMENT(this);
1510  // You can't call a function without a valid frame.
1511  DCHECK_IMPLIES(type == InvokeType::kCall, has_frame());
1512  DCHECK_EQ(function, edi);
1513  DCHECK_IMPLIES(new_target.is_valid(), new_target == edx);
1514  DCHECK(expected_parameter_count == ecx || expected_parameter_count == eax);
1515  DCHECK_EQ(actual_parameter_count, eax);
1516

1548  jmp(&done, Label::kNear);

1551  bind(&debug_hook);
1552  CallDebugOnFunctionCall(function, new_target, expected_parameter_count,
1553  actual_parameter_count);

1555 
1556  bind(&done);

1967 int MacroAssembler::CallCFunction(Register function, int num_arguments,
1968  SetIsolateDataSlots set_isolate_data_slots,
1969  Label* return_location) {
1970  ASM_CODE_COMMENT(this);
1971  DCHECK_LE(num_arguments, kMaxCParameters);
1972  DCHECK(has_frame());
1973  // Check stack alignment.
1974  if (v8_flags.debug_code) {
1975  CheckStackAlignment();
1976  }
1977 
1978  Label get_pc;
1979 
1980  if (set_isolate_data_slots == SetIsolateDataSlots::kYes) {
1981  // Save the frame pointer and PC so that the stack layout remains iterable,
1982  // even without an ExitFrame which normally exists between JS and C frames.
1983  // Find two caller-saved scratch registers.
1984  Register pc_scratch = eax;
1985  Register scratch = ecx;
1986  if (function == eax) pc_scratch = edx;
1987  if (function == ecx) scratch = edx;
1988  LoadLabelAddress(pc_scratch, &get_pc);
1989 
1990  // The root array is always available in production code. Only in one unit
1991  // test it is not available. The following code is not needed in the unit
1992  // test though, so we don't provide code here for the case where the root
1993  // array is not available.
1994  CHECK(root_array_available());
1995  mov(ExternalReferenceAsOperand(IsolateFieldId::kFastCCallCallerPC),
1996  pc_scratch);
1997  mov(ExternalReferenceAsOperand(IsolateFieldId::kFastCCallCallerFP), ebp);
1998  }
1999 
2000  call(function);
2001  int call_pc_offset = pc_offset();
2002  bind(&get_pc);
2003  if (return_location) bind(return_location);
2004  // Restoring the stack pointer has to happen right after the call. The
2005  // deoptimizer may overwrite everything after restoring the SP.
2006  int before_offset = pc_offset();
2007  if (base::OS::ActivationFrameAlignment() != 0) {
2008  mov(esp, Operand(esp, num_arguments * kSystemPointerSize));
2009  } else {
2010  add(esp, Immediate(num_arguments * kSystemPointerSize));
2011  }
2012  Nop(kMaxSizeOfMoveAfterFastCall - (pc_offset() - before_offset));
2013  // We assume that with the nop padding, the move instruction uses
2014  // kMaxSizeOfMoveAfterFastCall bytes. When we patch in the deopt trampoline,
2015  // we patch it in after the move instruction, so that the stack has been
2016  // restored correctly.
2017  CHECK_EQ(kMaxSizeOfMoveAfterFastCall, pc_offset() - before_offset);
2018 
2019  if (set_isolate_data_slots == SetIsolateDataSlots::kYes) {
2020  // We don't unset the PC; the FP is the source of truth.
2021  mov(ExternalReferenceAsOperand(IsolateFieldId::kFastCCallCallerFP),
2022  Immediate(0));
2023  }
2024 
2025  return call_pc_offset;
2026 }
