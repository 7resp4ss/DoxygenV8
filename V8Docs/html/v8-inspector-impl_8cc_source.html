165 V8InspectorSessionImpl* V8InspectorImpl::connectImpl(
166  int contextGroupId, V8Inspector::Channel* channel, StringView state,
167  ClientTrustLevel client_trust_level, SessionPauseState pause_state) {
168  int sessionId = ++m_lastSessionId;
169  std::shared_ptr<V8DebuggerBarrier> debuggerBarrier;
170  if (pause_state == kWaitingForDebugger) {
171  auto it = m_debuggerBarriers.find(contextGroupId);
172  if (it != m_debuggerBarriers.end()) {
173  // Note this will be empty in case a pre-existent barrier is already
174  // released. This is by design, as a released throttle is no longer
175  // efficient.
176  debuggerBarrier = it->second.lock();
177  } else {
178  debuggerBarrier =
179  std::make_shared<V8DebuggerBarrier>(m_client, contextGroupId);
180  m_debuggerBarriers.insert(it, {contextGroupId, debuggerBarrier});
181  }
182  }
183  V8InspectorSessionImpl* session = V8InspectorSessionImpl::create(
184  this, contextGroupId, sessionId, channel, state, client_trust_level,
185  std::move(debuggerBarrier));
186  m_sessions[contextGroupId][sessionId] = session;
187  return session;
188 }

233 void V8InspectorImpl::contextCreated(const V8ContextInfo& info) {
234  int contextId = ++m_lastContextId;
235  auto* context = new InspectedContext(this, info, contextId);
236  m_contextIdToGroupIdMap[contextId] = info.contextGroupId;
237 
238  DCHECK(m_uniqueIdToContextId.find(context->uniqueId().pair()) ==
239  m_uniqueIdToContextId.end());
240  m_uniqueIdToContextId.insert(
241  std::make_pair(context->uniqueId().pair(), contextId));
242 
243  auto contextIt = m_contexts.find(info.contextGroupId);
244  if (contextIt == m_contexts.end())
245  contextIt = m_contexts
246  .insert(std::make_pair(
247  info.contextGroupId,
248  std::unique_ptr<ContextByIdMap>(new ContextByIdMap())))
249  .first;
250  const auto& contextById = contextIt->second;
251 
252  DCHECK(contextById->find(contextId) == contextById->cend());
253  (*contextById)[contextId].reset(context);
254  forEachSession(
255  info.contextGroupId, [&context](V8InspectorSessionImpl* session) {
256  session->runtimeAgent()->addBindings(context);
257  session->runtimeAgent()->reportExecutionContextCreated(context);
258  });
259 }

267 void V8InspectorImpl::contextCollected(int groupId, int contextId) {
268  m_contextIdToGroupIdMap.erase(contextId);
269 
270  auto storageIt = m_consoleStorageMap.find(groupId);
271  if (storageIt != m_consoleStorageMap.end())
272  storageIt->second->contextDestroyed(contextId);
273 
274  InspectedContext* inspectedContext = getContext(groupId, contextId);
275  if (!inspectedContext) return;
276 
277  forEachSession(groupId, [&inspectedContext](V8InspectorSessionImpl* session) {
278  session->runtimeAgent()->reportExecutionContextDestroyed(inspectedContext);
279  });
280  discardInspectedContext(groupId, contextId);
281 }

283 void V8InspectorImpl::resetContextGroup(int contextGroupId) {
284  m_consoleStorageMap.erase(contextGroupId);
285  m_muteExceptionsMap.erase(contextGroupId);
286  auto contextsIt = m_contexts.find(contextGroupId);
287  // Context might have been removed already by discardContextScript()
288  if (contextsIt != m_contexts.end()) {
289  for (const auto& map_entry : *contextsIt->second)
290  m_uniqueIdToContextId.erase(map_entry.second->uniqueId().pair());
291  m_contexts.erase(contextsIt);
292  }
293  forEachSession(contextGroupId,
294  [](V8InspectorSessionImpl* session) { session->reset(); });
295 }

419 void V8InspectorImpl::forEachContext(
420  int contextGroupId,
421  const std::function<void(InspectedContext*)>& callback) {
422  auto it = m_contexts.find(contextGroupId);
423  if (it == m_contexts.end()) return;
424  std::vector<int> ids;
425  ids.reserve(it->second->size());
426  for (auto& contextIt : *(it->second)) ids.push_back(contextIt.first);
427 
428  // Retrieve by ids each time since |callback| may destroy some contexts.
429  for (auto& contextId : ids) {
430  it = m_contexts.find(contextGroupId);
431  if (it == m_contexts.end()) continue;
432  auto contextIt = it->second->find(contextId);
433  if (contextIt != it->second->end()) callback(contextIt->second.get());
434  }
435 }

439  const std::function<void(V8InspectorSessionImpl*)>& callback) {
440  auto it = m_sessions.find(contextGroupId);
441  if (it == m_sessions.end()) return;
442  std::vector<int> ids;
443  ids.reserve(it->second.size());
444  for (auto& sessionIt : it->second) ids.push_back(sessionIt.first);

446  // Retrieve by ids each time since |callback| may destroy some contexts.

449  if (it == m_sessions.end()) continue;
450  auto sessionIt = it->second.find(sessionId);
451  if (sessionIt != it->second.end()) callback(sessionIt->second);
452  }

482 class V8InspectorImpl::EvaluateScope::TerminateTask : public v8::Task {
483  public:
484  TerminateTask(v8::Isolate* isolate, std::shared_ptr<CancelToken> token)
485  : m_isolate(isolate), m_token(std::move(token)) {}
486 
487  void Run() override {
488  // CancelToken contains m_canceled bool which may be changed from main
489  // thread, so lock mutex first.

492  m_isolate->TerminateExecution();

497  std::shared_ptr<CancelToken> m_token;

511 bool V8InspectorImpl::associateExceptionData(v8::Local<v8::Context>,
512  v8::Local<v8::Value> exception,
513  v8::Local<v8::Name> key,
514  v8::Local<v8::Value> value) {
515  if (!exception->IsObject()) {
516  return false;
517  }
518  v8::Local<v8::Context> context;
519  if (!exceptionMetaDataContext().ToLocal(&context)) return false;
520  v8::TryCatch tryCatch(m_isolate);
521  v8::Context::Scope contextScope(context);
522  v8::HandleScope handles(m_isolate);
523  if (m_exceptionMetaData.IsEmpty())
524  m_exceptionMetaData.Reset(m_isolate,
525  v8::debug::EphemeronTable::New(m_isolate));
526 
527  v8::Local<v8::debug::EphemeronTable> map = m_exceptionMetaData.Get(m_isolate);
528  v8::MaybeLocal<v8::Value> entry = map->Get(m_isolate, exception);
529  v8::Local<v8::Object> object;
530  if (entry.IsEmpty() || !entry.ToLocalChecked()->IsObject()) {
531  object =
532  v8::Object::New(m_isolate, v8::Null(m_isolate), nullptr, nullptr, 0);
533  m_exceptionMetaData.Reset(m_isolate,
534  map->Set(m_isolate, exception, object));
535  } else {
536  object = entry.ToLocalChecked().As<v8::Object>();
537  }
538  CHECK(object->IsObject());
539  v8::Maybe<bool> result = object->CreateDataProperty(context, key, value);
540  return result.FromMaybe(false);
541 }

543 v8::MaybeLocal<v8::Object> V8InspectorImpl::getAssociatedExceptionData(
544  v8::Local<v8::Value> exception) {
545  if (!exception->IsObject()) {
546  return v8::MaybeLocal<v8::Object>();
547  }
548  v8::EscapableHandleScope scope(m_isolate);
549  v8::Local<v8::Context> context;
550  if (m_exceptionMetaData.IsEmpty() ||
551  !exceptionMetaDataContext().ToLocal(&context)) {
552  return v8::MaybeLocal<v8::Object>();
553  }
554  v8::Local<v8::debug::EphemeronTable> map = m_exceptionMetaData.Get(m_isolate);
555  auto entry = map->Get(m_isolate, exception);
556  v8::Local<v8::Value> object;
557  if (!entry.ToLocal(&object) || !object->IsObject())
558  return v8::MaybeLocal<v8::Object>();
559  return scope.Escape(object.As<v8::Object>());
560 }

562 std::unique_ptr<protocol::DictionaryValue>
563 V8InspectorImpl::getAssociatedExceptionDataForProtocol(
564  v8::Local<v8::Value> exception) {
565  v8::MaybeLocal<v8::Object> maybeData = getAssociatedExceptionData(exception);
566  v8::Local<v8::Object> data;
567  if (!maybeData.ToLocal(&data)) return nullptr;
568 
569  v8::Local<v8::Context> context;
570  if (!exceptionMetaDataContext().ToLocal(&context)) return nullptr;
