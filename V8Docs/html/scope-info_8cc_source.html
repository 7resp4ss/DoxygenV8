25 bool NameToIndexHashTableEquals(Tagged<NameToIndexHashTable> a,
26  Tagged<NameToIndexHashTable> b) {
27  if (a->Capacity() != b->Capacity()) return false;
28  if (a->NumberOfElements() != b->NumberOfElements()) return false;
29 
30  InternalIndex max(a->Capacity());
31  InternalIndex entry(0);
32 
33  while (entry < max) {
34  Tagged<Object> key_a = a->KeyAt(entry);
35  Tagged<Object> key_b = b->KeyAt(entry);
36  if (key_a != key_b) return false;
37 
38  Tagged<Object> value_a = a->ValueAt(entry);
39  Tagged<Object> value_b = b->ValueAt(entry);
40  if (value_a != value_b) return false;
41  }
42  return true;
43 }

46 // TODO(crbug.com/401059828): make it DEBUG only, once investigation is over.
47 bool ScopeInfo::Equals(Tagged<ScopeInfo> other, bool is_live_edit_compare,
48  int* out_last_checked_field) const {
49  if (length() != other->length()) return false;
50  if (Flags() != other->Flags()) return false;
51  for (int index = 0; index < length(); ++index) {
52  if (out_last_checked_field) *out_last_checked_field = index;
53  if (index == kFlags) continue;
54  if (is_live_edit_compare && index >= kPositionInfoStart &&
55  index <= kPositionInfoEnd) {
56  continue;
57  }
58  Tagged<Object> entry = get(index);
59  Tagged<Object> other_entry = other->get(index);
60  if (IsSmi(entry)) {
61  if (entry != other_entry) return false;
62  } else {
63  if (Cast<HeapObject>(entry)->map()->instance_type() !=
64  Cast<HeapObject>(other_entry)->map()->instance_type()) {
65  return false;
66  }
67  if (IsString(entry)) {
68  if (!Cast<String>(entry)->Equals(Cast<String>(other_entry))) {
69  return false;
70  }
71  } else if (IsScopeInfo(entry)) {
72  if (!is_live_edit_compare && !Cast<ScopeInfo>(entry)->Equals(
73  Cast<ScopeInfo>(other_entry), false)) {
74  return false;
75  }
76  } else if (IsSourceTextModuleInfo(entry)) {
77  if (!is_live_edit_compare &&
78  !Cast<SourceTextModuleInfo>(entry)->Equals(
79  Cast<SourceTextModuleInfo>(other_entry))) {
80  return false;
81  }
82  } else if (IsOddball(entry)) {
83  if (Cast<Oddball>(entry)->kind() !=
84  Cast<Oddball>(other_entry)->kind()) {
85  return false;
86  }
87  } else if (IsDependentCode(entry)) {
88  DCHECK(IsDependentCode(other_entry));
89  // Ignore the dependent code field since all the code have to be
90  // deoptimized anyway in case of a live-edit.
91 
92  } else if (IsNameToIndexHashTable(entry)) {
93  if (!NameToIndexHashTableEquals(
94  Cast<NameToIndexHashTable>(entry),
95  Cast<NameToIndexHashTable>(other_entry))) {
96  return false;
97  }
98  } else {
99  UNREACHABLE();
100  }
101  }
102  }
103  return true;
104 }

107 template <typename IsolateT>
108 Handle<ScopeInfo> ScopeInfo::Create(IsolateT* isolate, Zone* zone, Scope* scope,
109  MaybeDirectHandle<ScopeInfo> outer_scope) {
110  // Collect variables.
111  int context_local_count = 0;
112  int module_vars_count = 0;
113  // Stack allocated block scope variables are allocated in the parent
114  // declaration scope, but are recorded in the block scope's scope info. First
115  // slot index indicates at which offset a particular scope starts in the
116  // parent declaration scope.
117  for (Variable* var : *scope->locals()) {
118  switch (var->location()) {
119  case VariableLocation::CONTEXT:
120  case VariableLocation::REPL_GLOBAL:
121  context_local_count++;
122  break;
123  case VariableLocation::MODULE:
124  module_vars_count++;
125  break;
126  default:
127  break;
128  }
129  }
130  // Determine use and location of the "this" binding if it is present.
131  VariableAllocationInfo receiver_info;
132  if (scope->is_declaration_scope() &&
133  scope->AsDeclarationScope()->has_this_declaration()) {
134  Variable* var = scope->AsDeclarationScope()->receiver();
135  if (!var->is_used()) {
136  receiver_info = VariableAllocationInfo::UNUSED;
137  } else if (var->IsContextSlot()) {
138  receiver_info = VariableAllocationInfo::CONTEXT;
139  } else {
140  DCHECK(var->IsParameter());
141  receiver_info = VariableAllocationInfo::STACK;
142  }
143  } else {
144  receiver_info = VariableAllocationInfo::NONE;
145  }
146 
147  DCHECK(module_vars_count == 0 || scope->is_module_scope());
148 
149  // Make sure we allocate the correct amount.
150  DCHECK_EQ(scope->ContextLocalCount(), context_local_count);
151 
152  // If the number of locals is small, we inline directly
153  // in the scope info object.
154  bool has_inlined_local_names =
155  context_local_count < kScopeInfoMaxInlinedLocalNamesSize;
156 
157  const bool has_new_target =
158  scope->is_declaration_scope() &&
159  scope->AsDeclarationScope()->new_target_var() != nullptr;
160  // TODO(cbruni): Don't always waste a field for the inferred name.
161  const bool has_inferred_function_name = scope->is_function_scope();
162 
163  // Determine use and location of the function variable if it is present.
164  VariableAllocationInfo function_name_info;
165  if (scope->is_function_scope()) {
166  if (scope->AsDeclarationScope()->function_var() != nullptr) {
167  Variable* var = scope->AsDeclarationScope()->function_var();
168  if (!var->is_used()) {
169  function_name_info = VariableAllocationInfo::UNUSED;
170  } else if (var->IsContextSlot()) {
171  function_name_info = VariableAllocationInfo::CONTEXT;
172  } else {
173  DCHECK(var->IsStackLocal());
174  function_name_info = VariableAllocationInfo::STACK;
175  }
176  } else {
177  // Always reserve space for the debug name in the scope info.
178  function_name_info = VariableAllocationInfo::UNUSED;
179  }
180  } else if (scope->is_module_scope() || scope->is_script_scope() ||
181  scope->is_eval_scope()) {
182  // Always reserve space for the debug name in the scope info.
183  function_name_info = VariableAllocationInfo::UNUSED;
184  } else {
185  function_name_info = VariableAllocationInfo::NONE;
186  }
187 
188  const bool has_brand =
189  scope->is_class_scope()
190  ? scope->AsClassScope()->brand() != nullptr
191  : scope->IsConstructorScope() &&
192  scope->AsDeclarationScope()->class_scope_has_private_brand();
193  const bool should_save_class_variable =
194  scope->is_class_scope()
195  ? scope->AsClassScope()->should_save_class_variable()
196  : false;
197  const bool has_function_name =
198  function_name_info != VariableAllocationInfo::NONE;
199  const int parameter_count =
200  scope->is_declaration_scope()
201  ? scope->AsDeclarationScope()->num_parameters()
202  : 0;
203  const bool has_outer_scope_info = !outer_scope.is_null();
204 
205  DirectHandle<SourceTextModuleInfo> module_info;
206  if (scope->is_module_scope()) {
207  module_info = SourceTextModuleInfo::New(isolate, zone,
208  scope->AsModuleScope()->module());
209  }
210 
211  // Make sure the Fields enum agrees with Torque-generated offsets.
212  static_assert(OffsetOfElementAt(kFlags) == kFlagsOffset);
213  static_assert(OffsetOfElementAt(kParameterCount) == kParameterCountOffset);
214  static_assert(OffsetOfElementAt(kContextLocalCount) ==
215  kContextLocalCountOffset);
216 
217  FunctionKind function_kind = FunctionKind::kNormalFunction;
218  bool sloppy_eval_can_extend_vars = false;
219  if (scope->is_declaration_scope()) {
220  function_kind = scope->AsDeclarationScope()->function_kind();
221  sloppy_eval_can_extend_vars =
222  scope->AsDeclarationScope()->sloppy_eval_can_extend_vars();
223  }
224  DCHECK_IMPLIES(sloppy_eval_can_extend_vars, scope->HasContextExtensionSlot());
225 
226  const int local_names_container_size =
227  has_inlined_local_names ? context_local_count : 1;
228 
229  const int has_dependent_code = sloppy_eval_can_extend_vars;
230  const int length =
231  kVariablePartIndex + local_names_container_size + context_local_count +
232  (should_save_class_variable ? 1 : 0) +
233  (has_function_name ? kFunctionNameEntries : 0) +
234  (has_inferred_function_name ? 1 : 0) + (has_outer_scope_info ? 1 : 0) +
235  (scope->is_module_scope()
236  ? 2 + kModuleVariableEntryLength * module_vars_count
237  : 0) +
238  (has_dependent_code ? 1 : 0);
239 
240  // Create hash table if local names are not inlined.
241  Handle<NameToIndexHashTable> local_names_hashtable;
242  if (!has_inlined_local_names) {
243  local_names_hashtable = NameToIndexHashTable::New(
244  isolate, context_local_count, AllocationType::kOld);
245  }
246 
247  Handle<ScopeInfo> scope_info_handle =
248  isolate->factory()->NewScopeInfo(length);
249  int index = kVariablePartIndex;
250  {
251  DisallowGarbageCollection no_gc;
252  Tagged<ScopeInfo> scope_info = *scope_info_handle;
253  WriteBarrierMode mode = scope_info->GetWriteBarrierMode(no_gc);
254 
255  bool has_simple_parameters = false;
256  bool is_asm_module = false;
257  if (scope->is_function_scope()) {
258  DeclarationScope* function_scope = scope->AsDeclarationScope();
259  has_simple_parameters = function_scope->has_simple_parameters();
260 #if V8_ENABLE_WEBASSEMBLY
261  is_asm_module = function_scope->is_asm_module();
262 #endif // V8_ENABLE_WEBASSEMBLY
263  }
264 
265  // Encode the flags.
266  uint32_t flags =
267  ScopeTypeBits::encode(scope->scope_type()) |
268  SloppyEvalCanExtendVarsBit::encode(sloppy_eval_can_extend_vars) |
269  LanguageModeBit::encode(scope->language_mode()) |
270  DeclarationScopeBit::encode(scope->is_declaration_scope()) |
271  ReceiverVariableBits::encode(receiver_info) |
272  ClassScopeHasPrivateBrandBit::encode(has_brand) |
273  HasSavedClassVariableBit::encode(should_save_class_variable) |
274  HasNewTargetBit::encode(has_new_target) |
275  FunctionVariableBits::encode(function_name_info) |
276  HasInferredFunctionNameBit::encode(has_inferred_function_name) |
277  IsAsmModuleBit::encode(is_asm_module) |
278  HasSimpleParametersBit::encode(has_simple_parameters) |
279  FunctionKindBits::encode(function_kind) |
280  HasOuterScopeInfoBit::encode(has_outer_scope_info) |
281  IsDebugEvaluateScopeBit::encode(false) |
282  ForceContextAllocationBit::encode(
283  scope->ForceContextForLanguageMode()) |
284  PrivateNameLookupSkipsOuterClassBit::encode(
285  scope->private_name_lookup_skips_outer_class()) |
286  HasContextExtensionSlotBit::encode(scope->HasContextExtensionSlot()) |
287  IsHiddenBit::encode(scope->is_hidden()) |
288  IsWrappedFunctionBit::encode(scope->is_wrapped_function()) |
289  HasContextCellsBit::encode(scope->has_context_cells());
290  scope_info->set_flags(flags, kRelaxedStore);
291 
292  scope_info->set_parameter_count(parameter_count);
293  scope_info->set_context_local_count(context_local_count);
294 
295  scope_info->set_position_info_start(scope->start_position());
296  scope_info->set_position_info_end(scope->end_position());
297 
298  if (scope->is_module_scope()) {
299  scope_info->set_module_variable_count(module_vars_count);
300  ++index;
301  }
302  if (!has_inlined_local_names) {
303  scope_info->set_context_local_names_hashtable(*local_names_hashtable);
304  }
305 
306  // Add context locals' names and info, module variables' names and info.
307  // Context locals are added using their index.
308  int context_local_base = index;
309  int context_local_info_base =
310  context_local_base + local_names_container_size;
311  int module_var_entry = scope_info->ModuleVariablesIndex();
312 
313  for (Variable* var : *scope->locals()) {
314  switch (var->location()) {
315  case VariableLocation::CONTEXT:
316  case VariableLocation::REPL_GLOBAL: {
317  // Due to duplicate parameters, context locals aren't guaranteed to
318  // come in order.
319  int local_index = var->index() - scope->ContextHeaderLength();
320  DCHECK_LE(0, local_index);
321  DCHECK_LT(local_index, context_local_count);
322  uint32_t info =
323  VariableModeBits::encode(var->mode()) |
324  InitFlagBit::encode(var->initialization_flag()) |
325  MaybeAssignedFlagBit::encode(var->maybe_assigned()) |
326  ParameterNumberBits::encode(ParameterNumberBits::kMax) |
327  IsStaticFlagBit::encode(var->is_static_flag());
328  if (has_inlined_local_names) {
329  scope_info->set(context_local_base + local_index, *var->name(),
330  mode);
331  } else {
332  Handle<NameToIndexHashTable> new_table = NameToIndexHashTable::Add(
333  isolate, local_names_hashtable, var->name(), local_index);
334  DCHECK_EQ(*new_table, *local_names_hashtable);
335  USE(new_table);
336  }
337  scope_info->set(context_local_info_base + local_index,
338  Smi::FromInt(info));
339  break;
340  }
341  case VariableLocation::MODULE: {
342  scope_info->set(
343  module_var_entry +
344  TorqueGeneratedModuleVariableOffsets::kNameOffset /
345  kTaggedSize,
346  *var->name(), mode);
347  scope_info->set(
348  module_var_entry +
349  TorqueGeneratedModuleVariableOffsets::kIndexOffset /
350  kTaggedSize,
351  Smi::FromInt(var->index()));
352  uint32_t properties =
353  VariableModeBits::encode(var->mode()) |
354  InitFlagBit::encode(var->initialization_flag()) |
355  MaybeAssignedFlagBit::encode(var->maybe_assigned()) |
356  ParameterNumberBits::encode(ParameterNumberBits::kMax) |
357  IsStaticFlagBit::encode(var->is_static_flag());
358  scope_info->set(
359  module_var_entry +
360  TorqueGeneratedModuleVariableOffsets::kPropertiesOffset /
361  kTaggedSize,
362  Smi::FromInt(properties));
363  module_var_entry += kModuleVariableEntryLength;
364  break;
365  }
366  default:
367  break;
368  }
369  }
370 
371  if (scope->is_declaration_scope()) {
372  // Mark contexts slots with the parameter number they represent. We walk
373  // the list of parameters. That can include duplicate entries if a
374  // parameter name is repeated. By walking upwards, we'll automatically
375  // mark the context slot with the highest parameter number that uses this
376  // variable. That will be the parameter number that is represented by the
377  // context slot. All lower parameters will only be available on the stack
378  // through the arguments object.
379  for (int i = 0; i < parameter_count; i++) {
380  Variable* parameter = scope->AsDeclarationScope()->parameter(i);
381  if (parameter->location() != VariableLocation::CONTEXT) continue;
382  int param_index = parameter->index() - scope->ContextHeaderLength();
383  int info_index = context_local_info_base + param_index;
384  int info = Smi::ToInt(scope_info->get(info_index));
385  info = ParameterNumberBits::update(info, i);
386  scope_info->set(info_index, Smi::FromInt(info));
387  }
388  }
389 
390  // Advance past local names and local names info.
391  index += local_names_container_size + context_local_count;
392 
393  DCHECK_EQ(index, scope_info->SavedClassVariableInfoIndex());
394  // If the scope is a class scope and has used static private methods,
395  // save context slot index if locals are inlined, otherwise save the name.
396  if (should_save_class_variable) {
397  Variable* class_variable = scope->AsClassScope()->class_variable();
398  DCHECK_EQ(class_variable->location(), VariableLocation::CONTEXT);
399  if (has_inlined_local_names) {
400  scope_info->set(index++, Smi::FromInt(class_variable->index()));
401  } else {
402  scope_info->set(index++, *class_variable->name());
403  }
404  }
405 
406  // If present, add the function variable name and its index.
407  DCHECK_EQ(index, scope_info->FunctionVariableInfoIndex());
408  if (has_function_name) {
409  Variable* var = scope->AsDeclarationScope()->function_var();
410  int var_index = -1;
411  Tagged<Object> name = Smi::zero();
412  if (var != nullptr) {
413  var_index = var->index();
414  name = *var->name();
415  }
416  scope_info->set(index++, name, mode);
417  scope_info->set(index++, Smi::FromInt(var_index));
418  DCHECK(function_name_info != VariableAllocationInfo::CONTEXT ||
419  var_index == scope_info->ContextLength() - 1);
420  }
421 
422  DCHECK_EQ(index, scope_info->InferredFunctionNameIndex());
423  if (has_inferred_function_name) {
424  // The inferred function name is taken from the SFI.
425  index++;
426  }
427 
428  // If present, add the outer scope info.
429  DCHECK_EQ(index, scope_info->OuterScopeInfoIndex());
430  if (has_outer_scope_info) {
431  scope_info->set(index++, *outer_scope.ToHandleChecked(), mode);
432  }
433 
434  // Module-specific information (only for module scopes).
435  if (scope->is_module_scope()) {
436  DCHECK_EQ(index, scope_info->ModuleInfoIndex());
437  scope_info->set(index++, *module_info);
438  DCHECK_EQ(index, scope_info->ModuleVariablesIndex());
439  // The variable entries themselves have already been written above.
440  index += kModuleVariableEntryLength * module_vars_count;
441  }
442 
443  DCHECK_EQ(index, scope_info->DependentCodeIndex());
444  if (has_dependent_code) {
445  ReadOnlyRoots roots(isolate);
446  scope_info->set(index++, DependentCode::empty_dependent_code(roots));
447  }
448  }
449 
450  DCHECK_EQ(index, scope_info_handle->length());
451  DCHECK_EQ(length, scope_info_handle->length());
452  DCHECK_EQ(parameter_count, scope_info_handle->ParameterCount());
453  DCHECK_EQ(scope->num_heap_slots(), scope_info_handle->ContextLength());
454 
455  return scope_info_handle;
456 }

549 // static
550 DirectHandle<ScopeInfo> ScopeInfo::CreateForBootstrapping(
551  Isolate* isolate, BootstrappingType type) {
552  const int parameter_count = 0;
553  const bool is_empty_function = type == BootstrappingType::kFunction;
554  const bool is_native_context = (type == BootstrappingType::kNative) ||
555  (type == BootstrappingType::kShadowRealm);
556  const bool is_script = type == BootstrappingType::kScript;
557  const bool is_shadow_realm = type == BootstrappingType::kShadowRealm;
558  const int context_local_count =
559  is_empty_function || is_native_context ? 0 : 1;
560  const bool has_inferred_function_name = is_empty_function;
561  // NOTE: Local names are always inlined here, since context_local_count < 2.
562  DCHECK_LT(context_local_count, kScopeInfoMaxInlinedLocalNamesSize);
563  const int length = kVariablePartIndex + 2 * context_local_count +
564  (is_empty_function ? kFunctionNameEntries : 0) +
565  (has_inferred_function_name ? 1 : 0);
566 
567  Factory* factory = isolate->factory();
568  DirectHandle<ScopeInfo> scope_info =
569  factory->NewScopeInfo(length, AllocationType::kReadOnly);
570  DisallowGarbageCollection _nogc;
571  // Encode the flags.
572  DCHECK_IMPLIES(is_shadow_realm || is_script, !is_empty_function);
573  uint32_t flags =
574  ScopeTypeBits::encode(
575  is_empty_function
576  ? FUNCTION_SCOPE
577  : (is_shadow_realm ? SHADOW_REALM_SCOPE : SCRIPT_SCOPE)) |
578  SloppyEvalCanExtendVarsBit::encode(false) |
579  LanguageModeBit::encode(LanguageMode::kSloppy) |
580  DeclarationScopeBit::encode(true) |
581  ReceiverVariableBits::encode(is_script ? VariableAllocationInfo::CONTEXT
582  : VariableAllocationInfo::UNUSED) |
583  ClassScopeHasPrivateBrandBit::encode(false) |
584  HasSavedClassVariableBit::encode(false) | HasNewTargetBit::encode(false) |
585  FunctionVariableBits::encode(is_empty_function
586  ? VariableAllocationInfo::UNUSED
587  : VariableAllocationInfo::NONE) |
588  HasInferredFunctionNameBit::encode(has_inferred_function_name) |
589  IsAsmModuleBit::encode(false) | HasSimpleParametersBit::encode(true) |
590  FunctionKindBits::encode(FunctionKind::kNormalFunction) |
591  HasOuterScopeInfoBit::encode(false) |
592  IsDebugEvaluateScopeBit::encode(false) |
593  ForceContextAllocationBit::encode(false) |
594  PrivateNameLookupSkipsOuterClassBit::encode(false) |
595  HasContextExtensionSlotBit::encode(is_native_context) |
596  IsHiddenBit::encode(false) | IsWrappedFunctionBit::encode(false) |
597  HasContextCellsBit::encode(false);
598  Tagged<ScopeInfo> raw_scope_info = *scope_info;
599  raw_scope_info->set_flags(flags, kRelaxedStore);
600  raw_scope_info->set_parameter_count(parameter_count);
601  raw_scope_info->set_context_local_count(context_local_count);
602  raw_scope_info->set_position_info_start(0);
603  raw_scope_info->set_position_info_end(0);
604 
605  int index = kVariablePartIndex;
606 
607  // Here we add info for context-allocated "this".
608  DCHECK_EQ(index, raw_scope_info->ContextLocalNamesIndex());
609  ReadOnlyRoots roots(isolate);
610  if (context_local_count) {
611  raw_scope_info->set(index++, roots.this_string());
612  }
613  DCHECK_EQ(index, raw_scope_info->ContextLocalInfosIndex());
614  if (context_local_count > 0) {
615  const uint32_t value =
616  VariableModeBits::encode(VariableMode::kConst) |
617  InitFlagBit::encode(kCreatedInitialized) |
618  MaybeAssignedFlagBit::encode(kNotAssigned) |
619  ParameterNumberBits::encode(ParameterNumberBits::kMax) |
620  IsStaticFlagBit::encode(IsStaticFlag::kNotStatic);
621  raw_scope_info->set(index++, Smi::FromInt(value));
622  }
623 
624  DCHECK_EQ(index, raw_scope_info->FunctionVariableInfoIndex());
625  if (is_empty_function) {
626  raw_scope_info->set(index++, roots.empty_string());
627  raw_scope_info->set(index++, Smi::zero());
628  }
629  DCHECK_EQ(index, raw_scope_info->InferredFunctionNameIndex());
630  if (has_inferred_function_name) {
631  raw_scope_info->set(index++, roots.empty_string());
632  }
633  DCHECK_EQ(index, raw_scope_info->OuterScopeInfoIndex());
634  DCHECK_EQ(index, raw_scope_info->DependentCodeIndex());
635  DCHECK_EQ(index, raw_scope_info->length());
636  DCHECK_EQ(length, raw_scope_info->length());
637  DCHECK_EQ(raw_scope_info->ParameterCount(), parameter_count);
638  if (is_empty_function || is_native_context) {
639  DCHECK_EQ(raw_scope_info->ContextLength(), 0);
640  } else {
641  DCHECK_EQ(raw_scope_info->ContextLength(),
642  raw_scope_info->ContextHeaderLength() + 1);
643  }
644 
645  return scope_info;
646 }
