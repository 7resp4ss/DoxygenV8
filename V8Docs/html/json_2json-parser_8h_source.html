166  V8_WARN_UNUSED_RESULT static MaybeHandle<Object> Parse(
167  Isolate* isolate, Handle<String> source, Handle<Object> reviver) {
168  HighAllocationThroughputScope high_throughput_scope(
169  V8::GetCurrentPlatform());
170  Handle<Object> result;
171  MaybeHandle<Object> val_node;
172  {
173  JsonParser parser(isolate, source);
174  ASSIGN_RETURN_ON_EXCEPTION(isolate, result, parser.ParseJson(reviver));
175  val_node = parser.parsed_val_node_;
176  }
177  if (IsCallable(*reviver)) {
178  return JsonParseInternalizer::Internalize(isolate, result, reviver,
179  source, val_node);
180  }
181  return result;
182  }

265  template <size_t N>
266  void ScanLiteral(const char (&s)[N]) {
267  DCHECK(!is_at_end());
268  // There's at least 1 character, we always consume a character and compare
269  // the next character. The first character was compared before we jumped
270  // to ScanLiteral.
271  static_assert(N > 2);
272  size_t remaining = remaining_chars();
273  if (V8_LIKELY(remaining >= N - 1 &&
274  CompareCharsEqual(s + 1, cursor_ + 1, N - 2))) {
275  cursor_ += N - 1;
276  return;
277  }
278 
279  cursor_++;
280  for (size_t i = 0; i < std::min(N - 2, remaining - 1); i++) {
281  if (*(s + 1 + i) != *cursor_) {
282  ReportUnexpectedCharacter(*cursor_);
283  return;
284  }
285  cursor_++;
286  }
287 
288  DCHECK(is_at_end());
289  ReportUnexpectedToken(JsonToken::EOS);
290  }

389  void UpdatePointers() {
390  DisallowGarbageCollection no_gc;
391  const Char* chars = Cast<SeqString>(source_)->GetChars(no_gc);
392  if (chars_ != chars) {
393  size_t position = cursor_ - chars_;
394  size_t length = end_ - chars_;
395  chars_ = chars;
396  cursor_ = chars_ + position;
397  end_ = chars_ + length;
398  }
399  }
