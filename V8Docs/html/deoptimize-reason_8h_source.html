158 constexpr bool IsDeoptimizationWithoutCodeInvalidation(
159  DeoptimizeReason reason) {
160  // Maglev OSRs into Turbofan by first deoptimizing in order to restore the
161  // unoptimized frame layout. Since no actual assumptions in the Maglev code
162  // object are violated, it (and any associated cached optimized code) should
163  // not be invalidated s.t. we may reenter it in the future.
164  return reason == DeoptimizeReason::kPrepareForOnStackReplacement ||
165  reason == DeoptimizeReason::kOSREarlyExit;
166 }
