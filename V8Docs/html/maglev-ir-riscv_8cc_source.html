236 void CheckFloat64SameValue::GenerateCode(MaglevAssembler* masm,
237  const ProcessingState& state) {
238  Label* fail = __ GetDeoptLabel(this, deoptimize_reason());
239  MaglevAssembler::TemporaryRegisterScope temps(masm);
240  DoubleRegister target = ToDoubleRegister(target_input());
241  if (value().is_nan()) {
242  __ JumpIfNotNan(target, fail);
243  } else {
244  DoubleRegister double_scratch = temps.AcquireScratchDouble();
245  Register scratch = temps.AcquireScratch();
246  __ Move(double_scratch, value().get_scalar());
247  __ CompareF64(scratch, EQ, double_scratch, target);
248  __ BranchFalseF(scratch, fail);
249  if (value().get_scalar() == 0) { // +0.0 or -0.0.
250  __ MacroAssembler::Move(scratch, target);
251  __ And(scratch, scratch, Operand(1ULL << 63));
252  if (value().get_bits() == 0) {
253  __ BranchTrueF(scratch, fail);
254  } else {
255  __ BranchFalseF(scratch, fail);
256  }
257  }
258  }

320 void Int32MultiplyWithOverflow::GenerateCode(MaglevAssembler* masm,
321  const ProcessingState& state) {
322  Register left = ToRegister(left_input());
323  Register right = ToRegister(right_input());
324  Register out = ToRegister(result());
325 
326  // TODO(leszeks): peephole optimise multiplication by a constant.
327 
328  MaglevAssembler::TemporaryRegisterScope temps(masm);
329  bool out_alias_input = out == left || out == right;
330  Register res = out;
331  if (out_alias_input) {
332  res = temps.Acquire();
333  }
334 
335  Register scratch = temps.Acquire();
336  __ MulOverflow32(res, left, Operand(right), scratch, false);
337 
338  static_assert(Int32MultiplyWithOverflow::kProperties.can_eager_deopt());
339  // if res != (res[0:31] sign extended to 64 bits), then the multiplication
340  // result is too large for 32 bits.
341  Label* fail = __ GetDeoptLabel(this, DeoptimizeReason::kOverflow);
342  __ RecordComment("-- Jump to eager deopt");
343  __ MacroAssembler::Branch(fail, ne, scratch, Operand(zero_reg));
344 
345  // If the result is zero, check if either lhs or rhs is negative.
346  Label end;
347  __ MacroAssembler::Branch(&end, ne, res, Operand(zero_reg),
348  Label::Distance::kNear);
349  {
350  Register maybeNegative = scratch;
351  __ Or(maybeNegative, left, Operand(right));
352  // TODO(Vladimir Kempik): consider usage of bexti instruction if Zbs
353  // extension is available
354  __ And(maybeNegative, maybeNegative, Operand(0x80000000)); // 1 << 31
355  // If one of them is negative, we must have a -0 result, which is non-int32,
356  // so deopt.
357  // TODO(leszeks): Consider splitting these deopts to have distinct deopt
358  // reasons. Otherwise, the reason has to match the above.
359  __ RecordComment("-- Jump to eager deopt if the result is negative zero");
360  Label* deopt_label = __ GetDeoptLabel(this, DeoptimizeReason::kOverflow);
361  __ MacroAssembler::Branch(deopt_label, ne, maybeNegative,
362  Operand(zero_reg));
363  }
364 
365  __ bind(&end);
366  if (out_alias_input) {
367  __ Move(out, res);
368  }
369 
370  // The output register shouldn't be a register input into the eager deopt
371  // info.
372  DCHECK_REGLIST_EMPTY(RegList{out} &
373  GetGeneralRegistersUsedAsInputs(eager_deopt_info()));
374 }

382 void Int32DivideWithOverflow::GenerateCode(MaglevAssembler* masm,
383  const ProcessingState& state) {
384  Register left = ToRegister(left_input());
385  Register right = ToRegister(right_input());
386  Register out = ToRegister(result());
387 
388  // TODO(leszeks): peephole optimise division by a constant.
389 
390  static_assert(Int32DivideWithOverflow::kProperties.can_eager_deopt());
391  ZoneLabelRef done(masm);
392  Label* deferred_overflow_checks = __ MakeDeferredCode(
393  [](MaglevAssembler* masm, ZoneLabelRef done, Register left,
394  Register right, Int32DivideWithOverflow* node) {
395  // {right} is negative or zero.
396 
397  // TODO(leszeks): Using kNotInt32 here, but in same places
398  // kDivisionByZerokMinusZero/kMinusZero/kOverflow would be better. Right
399  // now all eager deopts in a node have to be the same -- we should allow
400  // a node to emit multiple eager deopts with different reasons.
401  Label* deopt = __ GetDeoptLabel(node, DeoptimizeReason::kNotInt32);
402 
403  // Check if {right} is zero.
404  __ RecordComment("-- Jump to eager deopt if right is zero");
405  __ MacroAssembler::Branch(deopt, eq, right, Operand(zero_reg));
406 
407  // Check if {left} is zero, as that would produce minus zero.
408  __ RecordComment("-- Jump to eager deopt if left is zero");
409  __ MacroAssembler::Branch(deopt, eq, left, Operand(zero_reg));
410 
411  // Check if {left} is kMinInt and {right} is -1, in which case we'd have
412  // to return -kMinInt, which is not representable as Int32.
413  __ MacroAssembler::Branch(*done, ne, left, Operand(kMinInt));
414  __ MacroAssembler::Branch(*done, ne, right, Operand(-1));
415  __ JumpToDeopt(deopt);
416  },
417  done, left, right, this);
418 
419  // Check if {right} is positive and not zero.
420  __ MacroAssembler::Branch(deferred_overflow_checks, less_equal, right,
421  Operand(zero_reg));
422  __ bind(*done);
423 
424  // Perform the actual integer division.
425  MaglevAssembler::TemporaryRegisterScope temps(masm);
426  bool out_alias_input = out == left || out == right;
427  Register res = out;
428  if (out_alias_input) {
429  res = temps.Acquire();
430  }
431  __ Div32(res, left, right);
432 
433  // Check that the remainder is zero.
434  Register temp = temps.Acquire();
435  __ remw(temp, left, right);
436  Label* deopt = __ GetDeoptLabel(this, DeoptimizeReason::kNotInt32);
437  __ RecordComment("-- Jump to eager deopt if remainder is zero");
438  __ MacroAssembler::Branch(deopt, ne, temp, Operand(zero_reg));
439 
440  // The output register shouldn't be a register input into the eager deopt
441  // info.
442  DCHECK_REGLIST_EMPTY(RegList{res} &
443  GetGeneralRegistersUsedAsInputs(eager_deopt_info()));
444  __ Move(out, res);
445 }

453 void Int32ModulusWithOverflow::GenerateCode(MaglevAssembler* masm,
454  const ProcessingState& state) {
455  // If AreAliased(lhs, rhs):
456  // deopt if lhs < 0 // Minus zero.
457  // 0
458  //
459  // Using same algorithm as in EffectControlLinearizer:
460  // if rhs <= 0 then
461  // rhs = -rhs
462  // deopt if rhs == 0
463  // if lhs < 0 then
464  // let lhs_abs = -lhs in
465  // let res = lhs_abs % rhs in
466  // deopt if res == 0
467  // -res
468  // else
469  // let msk = rhs - 1 in
470  // if rhs & msk == 0 then
471  // lhs & msk
472  // else
473  // lhs % rhs
474 
475  Register lhs = ToRegister(left_input());
476  Register rhs = ToRegister(right_input());
477  Register out = ToRegister(result());
478 
479  static_assert(Int32ModulusWithOverflow::kProperties.can_eager_deopt());
480  static constexpr DeoptimizeReason deopt_reason =
481  DeoptimizeReason::kDivisionByZero;
482 
483  // For the modulus algorithm described above, lhs and rhs must not alias
484  // each other.
485  if (lhs == rhs) {
486  // TODO(victorgomes): This ideally should be kMinusZero, but Maglev only
487  // allows one deopt reason per IR.
488  Label* deopt = __ GetDeoptLabel(this, DeoptimizeReason::kDivisionByZero);
489  __ RecordComment("-- Jump to eager deopt");
490  __ MacroAssembler::Branch(deopt, less, lhs, Operand(zero_reg));
491  __ Move(out, zero_reg);
492  return;
493  }
494 
495  DCHECK(!AreAliased(lhs, rhs));
496 
497  ZoneLabelRef done(masm);
498  ZoneLabelRef rhs_checked(masm);
499 
500  Label* deferred_rhs_check = __ MakeDeferredCode(
501  [](MaglevAssembler* masm, ZoneLabelRef rhs_checked, Register rhs,
502  Int32ModulusWithOverflow* node) {
503  __ negw(rhs, rhs);
504  __ MacroAssembler::Branch(*rhs_checked, ne, rhs, Operand(zero_reg));
505  __ EmitEagerDeopt(node, deopt_reason);
506  },
507  rhs_checked, rhs, this);
508  __ MacroAssembler::Branch(deferred_rhs_check, less_equal, rhs,
509  Operand(zero_reg));
510  __ bind(*rhs_checked);
511 
512  Label* deferred_lhs_check = __ MakeDeferredCode(
513  [](MaglevAssembler* masm, ZoneLabelRef done, Register lhs, Register rhs,
514  Register out, Int32ModulusWithOverflow* node) {
515  MaglevAssembler::TemporaryRegisterScope temps(masm);
516  Register lhs_abs = temps.AcquireScratch();
517  __ negw(lhs_abs, lhs);
518  Register res = lhs_abs;
519  __ remw(res, lhs_abs, rhs);
520  __ negw(out, res);
521  __ MacroAssembler::Branch(*done, ne, res, Operand(zero_reg));
522  // TODO(victorgomes): This ideally should be kMinusZero, but Maglev
523  // only allows one deopt reason per IR.
524  __ EmitEagerDeopt(node, deopt_reason);
525  },
526  done, lhs, rhs, out, this);
527  __ MacroAssembler::Branch(deferred_lhs_check, less, lhs, Operand(zero_reg));
528 
529  Label rhs_not_power_of_2;
530  MaglevAssembler::TemporaryRegisterScope temps(masm);
531  Register scratch = temps.AcquireScratch();
532  Register msk = temps.AcquireScratch();
533  __ Sub32(msk, rhs, Operand(1));
534  __ And(scratch, rhs, msk);
535  __ MacroAssembler::Branch(&rhs_not_power_of_2, not_equal, scratch,
536  Operand(zero_reg), Label::kNear);
537  // {rhs} is power of 2.
538  __ And(out, lhs, msk);
539  __ MacroAssembler::Branch(*done, Label::kNear);
540 
541  __ bind(&rhs_not_power_of_2);
542  __ remw(out, lhs, rhs);
543 
544  __ bind(*done);
545 }

701 void Float64Round::GenerateCode(MaglevAssembler* masm,
702  const ProcessingState& state) {
703  DoubleRegister in = ToDoubleRegister(input());
704  DoubleRegister out = ToDoubleRegister(result());
705  MaglevAssembler::TemporaryRegisterScope temps(masm);
706  DoubleRegister fscratch1 = temps.AcquireScratchDouble();
707 
708  if (kind_ == Kind::kNearest) {
709  // RISC-V Rounding Mode RNE means "Round to Nearest, ties to Even", while JS
710  // expects it to round towards +Infinity (see ECMA-262, 20.2.2.28).
711  // The best seems to be to add 0.5 then round with RDN mode.
712 
713  DoubleRegister half_one = temps.AcquireDouble(); // available in this mode
714  __ LoadFPRImmediate(half_one, 0.5);
715  DoubleRegister tmp = half_one;
716  __ fadd_d(tmp, in, half_one);
717  __ Floor_d_d(out, tmp, fscratch1);
718  __ fsgnj_d(out, out, in);
719  } else if (kind_ == Kind::kCeil) {
720  __ Ceil_d_d(out, in, fscratch1);
721  } else if (kind_ == Kind::kFloor) {
722  __ Floor_d_d(out, in, fscratch1);
723  } else {
724  UNREACHABLE();
725  }

826 void HandleInterruptsAndTiering(MaglevAssembler* masm, ZoneLabelRef done,
827  Node* node, ReduceInterruptBudgetType type,
828  Register scratch0) {
829  // For loops, first check for interrupts. Don't do this for returns, as we
830  // can't lazy deopt to the end of a return.
831  if (type == ReduceInterruptBudgetType::kLoop) {
832  Label next;
833  // Here, we only care about interrupts since we've already guarded against
834  // real stack overflows on function entry.
835  {
836  Register stack_limit = scratch0;
837  __ LoadStackLimit(stack_limit, StackLimitKind::kInterruptStackLimit);
838  __ MacroAssembler::Branch(&next, ugt, sp, Operand(stack_limit),
839  Label::Distance::kNear);
840  }
841 
842  // An interrupt has been requested and we must call into runtime to handle
843  // it; since we already pay the call cost, combine with the TieringManager
844  // call.
845  {
846  SaveRegisterStateForCall save_register_state(masm,
847  node->register_snapshot());
848  Register function = scratch0;
849  __ LoadWord(function,
850  MemOperand(fp, StandardFrameConstants::kFunctionOffset));
851  __ Push(function);
852  // Move into kContextRegister after the load into scratch0, just in case
853  // scratch0 happens to be kContextRegister.
854  __ Move(kContextRegister, masm->native_context().object());
855  __ CallRuntime(Runtime::kBytecodeBudgetInterruptWithStackCheck_Maglev, 1);
856  save_register_state.DefineSafepointWithLazyDeopt(node->lazy_deopt_info());
857  }
858  __ MacroAssembler::Branch(*done); // All done, continue.
859  __ bind(&next);
860  }
861 
862  // No pending interrupts. Call into the TieringManager if needed.
863  {
864  SaveRegisterStateForCall save_register_state(masm,
865  node->register_snapshot());
866  Register function = scratch0;
867  __ LoadWord(function,
868  MemOperand(fp, StandardFrameConstants::kFunctionOffset));
869  __ Push(function);
870  // Move into kContextRegister after the load into scratch0, just in case
871  // scratch0 happens to be kContextRegister.
872  __ Move(kContextRegister, masm->native_context().object());
873  // Note: must not cause a lazy deopt!
874  __ CallRuntime(Runtime::kBytecodeBudgetInterrupt_Maglev, 1);
875  save_register_state.DefineSafepoint();
876  }
877  __ MacroAssembler::Branch(*done);
878 }

880 void GenerateReduceInterruptBudget(MaglevAssembler* masm, Node* node,
881  Register feedback_cell,
882  ReduceInterruptBudgetType type, int amount) {
883  MaglevAssembler::TemporaryRegisterScope temps(masm);
884  Register scratch = temps.Acquire();
885  Register budget = scratch;
886

888  FieldMemOperand(feedback_cell, FeedbackCell::kInterruptBudgetOffset));

890  __ Sw(budget,

899  done, node, type, scratch);

903 }

936 void Return::GenerateCode(MaglevAssembler* masm, const ProcessingState& state) {
937  DCHECK_EQ(ToRegister(value_input()), kReturnRegister0);
938  // Read the formal number of parameters from the top level compilation unit
939  // (i.e. the outermost, non inlined function).
940  int formal_params_size =
941  masm->compilation_info()->toplevel_compilation_unit()->parameter_count();
942 
943  // We're not going to continue execution, so we can use an arbitrary register
944  // here instead of relying on temporaries from the register allocator.
945  // We cannot use scratch registers, since they're used in LeaveFrame and
946  // DropArguments.
947  Register actual_params_size = a5;
948 
949  // Compute the size of the actual parameters + receiver (in bytes).
950  // TODO(leszeks): Consider making this an input into Return to reuse the
951  // incoming argc's register (if it's still valid).
952  __ LoadWord(actual_params_size,
953  MemOperand(fp, StandardFrameConstants::kArgCOffset));
954 
955  // Leave the frame.
956  __ LeaveFrame(StackFrame::MAGLEV);
957 
958  // If actual is bigger than formal, then we should use it to free up the stack
959  // arguments.
960  Label corrected_args_count;
961  __ MacroAssembler::Branch(&corrected_args_count, gt, actual_params_size,
962  Operand(formal_params_size),
963  Label::Distance::kNear);
964  __ Move(actual_params_size, formal_params_size);
965 
966  __ bind(&corrected_args_count);
967  // Drop receiver + arguments according to dynamic arguments size.
968  __ DropArguments(actual_params_size);
969  __ Ret();
970 }
