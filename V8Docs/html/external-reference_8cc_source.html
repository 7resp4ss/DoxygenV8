426 // Helper function to verify that all types in a list of types are scalar.
427 // This includes primitive types (int, Address) and pointer types. We also
428 // allow void.
429 template <typename T>
430 constexpr bool AllScalar() {
431  return std::is_scalar_v<T> || std::is_void_v<T>;
432 }
433 
434 template <typename T1, typename T2, typename... Rest>
435 constexpr bool AllScalar() {
436  return AllScalar<T1>() && AllScalar<T2, Rest...>();
437 }
438 
439 // Checks a function pointer's type for compatibility with the
440 // ExternalReference calling mechanism. Specifically, all arguments
441 // as well as the result type must pass the AllScalar check above,
442 // because we expect each item to fit into one register or stack slot.
443 template <typename T>
444 struct IsValidExternalReferenceType;
445 
446 template <typename Result, typename... Args>
447 struct IsValidExternalReferenceType<Result (*)(Args...)> {
448  static const bool value = AllScalar<Result, Args...>();
449 };
450 
451 template <typename Result, typename Class, typename... Args>
452 struct IsValidExternalReferenceType<Result (Class::*)(Args...)> {
453  static const bool value = AllScalar<Result, Args...>();
454 };
