105 // CpuFeatures keeps track of which features are supported by the target CPU.
106 // Supported features must be enabled by a CpuFeatureScope before use.
107 // Example:
108 // if (assembler->IsSupported(SSE3)) {
109 // CpuFeatureScope fscope(assembler, SSE3);
110 // // Generate code containing SSE3 instructions.
111 // } else {
112 // // Generate alternative code.
113 // }
114 class V8_EXPORT_PRIVATE CpuFeatures : public AllStatic {
115  public:
116  CpuFeatures(const CpuFeatures&) = delete;
117  CpuFeatures& operator=(const CpuFeatures&) = delete;
118 
119  static void Probe(bool cross_compile) {
120  static_assert(NUMBER_OF_CPU_FEATURES <= kBitsPerInt);
121  if (initialized_) return;
122  initialized_ = true;
123  ProbeImpl(cross_compile);
124  }
125 
126  static unsigned SupportedFeatures() {
127  Probe(false);
128  return supported_;
129  }
130 
131  static bool IsSupported(CpuFeature f) {
132  return (supported_ & (1u << f)) != 0;
133  }
134 
135  static void SetSupported(CpuFeature f) { supported_ |= 1u << f; }
136  static void SetUnsupported(CpuFeature f) { supported_ &= ~(1u << f); }
137 
138  static bool SupportsWasmSimd128();
139 
140  static inline bool SupportsOptimizer();
141 
142  static inline unsigned icache_line_size() {
143  DCHECK_NE(icache_line_size_, 0);
144  return icache_line_size_;
145  }
146 
147  static inline unsigned dcache_line_size() {
148  DCHECK_NE(dcache_line_size_, 0);
149  return dcache_line_size_;

151 
152  static void PrintTarget();
153  static void PrintFeatures();
154 
155  private:

157  friend class ExternalReference;
158  // Flush instruction cache.
159  static void FlushICache(void* start, size_t size);
160 
161  // Platform-dependent implementation.
162  static void ProbeImpl(bool cross_compile);
163 
164  static unsigned supported_;
165  static unsigned icache_line_size_;
166  static unsigned dcache_line_size_;
167  static bool initialized_;
168  // This variable is only used for certain archs to query SupportWasmSimd128()
169  // at runtime in builtins using an extern ref. Other callers should use
170  // CpuFeatures::SupportWasmSimd128().
171  static bool supports_wasm_simd_128_;
172  static bool supports_cetss_;
