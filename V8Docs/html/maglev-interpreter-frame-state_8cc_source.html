26 void KnownNodeAspects::Merge(const KnownNodeAspects& other, Zone* zone) {
27  bool any_merged_map_is_unstable = false;
28  DestructivelyIntersect(node_infos, other.node_infos,
29  [&](NodeInfo& lhs, const NodeInfo& rhs) {
30  lhs.MergeWith(rhs, zone, any_merged_map_is_unstable);
31  return !lhs.no_info_available();
32  });
33 
34  if (effect_epoch_ != other.effect_epoch_) {
35  effect_epoch_ = std::max(effect_epoch_, other.effect_epoch_) + 1;
36  }
37  DestructivelyIntersect(
38  available_expressions, other.available_expressions,
39  [&](const AvailableExpression& lhs, const AvailableExpression& rhs) {
40  DCHECK_IMPLIES(lhs.node == rhs.node,
41  lhs.effect_epoch == rhs.effect_epoch);
42  DCHECK_NE(lhs.effect_epoch, kEffectEpochOverflow);
43  DCHECK_EQ(Node::needs_epoch_check(lhs.node->opcode()),
44  lhs.effect_epoch != kEffectEpochForPureInstructions);
45 
46  return lhs.node == rhs.node && lhs.effect_epoch >= effect_epoch_;
47  });
48 
49  this->any_map_for_any_node_is_unstable = any_merged_map_is_unstable;
50 
51  auto merge_loaded_properties =
52  [](ZoneMap<ValueNode*, ValueNode*>& lhs,
53  const ZoneMap<ValueNode*, ValueNode*>& rhs) {
54  // Loaded properties are maps of maps, so just do the destructive
55  // intersection recursively.
56  DestructivelyIntersect(lhs, rhs);
57  return !lhs.empty();
58  };
59  DestructivelyIntersect(loaded_constant_properties,
60  other.loaded_constant_properties,
61  merge_loaded_properties);
62  DestructivelyIntersect(loaded_properties, other.loaded_properties,
63  merge_loaded_properties);
64  DestructivelyIntersect(loaded_context_constants,
65  other.loaded_context_constants);
66  may_have_aliasing_contexts_ = ContextSlotLoadsAliasMerge(
67  may_have_aliasing_contexts_, other.may_have_aliasing_contexts());
68  DestructivelyIntersect(loaded_context_slots, other.loaded_context_slots);
69 }

73 template <typename Key>
74 bool NextInIgnoreList(typename ZoneSet<Key>::const_iterator& ignore,
75  typename ZoneSet<Key>::const_iterator& ignore_end,
76  const Key& cur) {
77  while (ignore != ignore_end && *ignore < cur) {
78  ++ignore;
79  }
80  return ignore != ignore_end && *ignore == cur;

85 bool CheckAllPhiInputsAreContextType(compiler::JSHeapBroker* broker,
86  LocalIsolate* local_isolate,
87  const KnownNodeAspects& node_aspects,
88  Phi* phi) {
89  for (Input& input : *phi) {
90  if (!input.node()) continue;
91  if (phi->is_loop_phi() && &phi->backedge_input() == &input) continue;
92  if (NodeTypeIs(node_aspects.GetType(broker, input.node()),
93  NodeType::kContext)) {
94  continue;
95  }
96  // OSR values can leak a context with InitialValue that we are unable to
97  // type.
98  if (input.node()->Is<InitialValue>() ||
99  input.node()->Is<GeneratorRestoreRegister>()) {
100  continue;
101  }
102  if (phi->Cast<Phi>()->owner() == interpreter::Register::current_context()) {
103  // We are unable to type loop phis holding the context from resumable
104  // loops, since they start with an empty KNA.
105  // Instead here, we just check if the phi makes sense,
106  return true;
107  }
108  if (auto next_phi = input.node()->TryCast<Phi>()) {
109  CheckAllPhiInputsAreContextType(broker, local_isolate, node_aspects,
110  next_phi);
111  continue;
112  }
113  return false;
114  }
115  return true;
116 }

122 void KnownNodeAspects::UpdateMayHaveAliasingContexts(
123  compiler::JSHeapBroker* broker, LocalIsolate* local_isolate,
124  ValueNode* context) {
125  while (true) {
126  if (auto load_prev_ctxt =
127  context->TryCast<LoadTaggedFieldForContextSlotNoCells>()) {
128  DCHECK_EQ(load_prev_ctxt->offset(),
129  Context::OffsetOfElementAt(Context::PREVIOUS_INDEX));
130  // Recurse until we find the root.
131  context = load_prev_ctxt->input(0).node();
132  continue;
133  }
134  break;
135  }
136 
137  switch (context->opcode()) {
138  case Opcode::kInitialValue:
139  may_have_aliasing_contexts_ = ContextSlotLoadsAliasMerge(
140  may_have_aliasing_contexts_,
141  ContextSlotLoadsAlias::kOnlyLoadsRelativeToCurrentContext);
142  break;
143  case Opcode::kConstant:
144  may_have_aliasing_contexts_ = ContextSlotLoadsAliasMerge(
145  may_have_aliasing_contexts_,
146  ContextSlotLoadsAlias::kOnlyLoadsRelativeToConstant);
147  DCHECK(NodeTypeIs(GetType(broker, context), NodeType::kContext));
148  break;
149  case Opcode::kCreateFunctionContext:
150  case Opcode::kInlinedAllocation:
151  // These can be precisely tracked.
152  DCHECK(NodeTypeIs(GetType(broker, context), NodeType::kContext));
153  break;
154  case Opcode::kLoadTaggedField: {
155  LoadTaggedField* load = context->Cast<LoadTaggedField>();
156  // Currently, the only way to leak a context via load tagged field is with
157  // JSFunction or JSGeneratorObject.
158  USE(load);
159  DCHECK(load->offset() == JSFunction::kContextOffset ||
160  load->offset() == JSGeneratorObject::kContextOffset);
161  may_have_aliasing_contexts_ = ContextSlotLoadsAlias::kYes;
162  DCHECK(NodeTypeIs(GetType(broker, context), NodeType::kContext));
163  break;
164  }
165  case Opcode::kCallRuntime:
166  DCHECK(NodeTypeIs(GetType(broker, context), NodeType::kContext));
167  may_have_aliasing_contexts_ = ContextSlotLoadsAlias::kYes;
168  break;
169  case Opcode::kGeneratorRestoreRegister:
170  may_have_aliasing_contexts_ = ContextSlotLoadsAlias::kYes;
171  break;
172  case Opcode::kPhi:
173  SLOW_DCHECK(CheckAllPhiInputsAreContextType(broker, local_isolate, *this,
174  context->Cast<Phi>()));
175  may_have_aliasing_contexts_ = ContextSlotLoadsAlias::kYes;
176  break;
177  default:
178  // This DCHECK only shows which kind of nodes can be a context, even if
179  // this DCHECK fails, it is always safe to set aliasing context mode to
180  // kYes.
181  DCHECK(false);
182  may_have_aliasing_contexts_ = ContextSlotLoadsAlias::kYes;
183  break;
184  }
185 }

206 KnownNodeAspects::KnownNodeAspects(const KnownNodeAspects& other,
207  bool optimistic_initial_state,
208  LoopEffects* loop_effects, Zone* zone)
209  : any_map_for_any_node_is_unstable(false),
210  loaded_constant_properties(other.loaded_constant_properties),
211  loaded_properties(zone),
212  loaded_context_constants(other.loaded_context_constants),
213  loaded_context_slots(zone),
214  available_expressions(zone),
215  may_have_aliasing_contexts_(
216  KnownNodeAspects::ContextSlotLoadsAlias::kNone),
217  effect_epoch_(other.effect_epoch_),
218  node_infos(zone) {
219  if (!other.any_map_for_any_node_is_unstable) {
220  node_infos = other.node_infos;
221 #ifdef DEBUG
222  for (const auto& it : node_infos) {
223  DCHECK(!it.second.any_map_is_unstable());
224  }
225 #endif
226  } else if (optimistic_initial_state &&
227  !loop_effects->unstable_aspects_cleared) {
228  node_infos = other.node_infos;
229  any_map_for_any_node_is_unstable = other.any_map_for_any_node_is_unstable;
230  } else {
231  for (const auto& it : other.node_infos) {
232  node_infos.emplace(it.first,
233  NodeInfo::ClearUnstableMapsOnCopy{it.second});
234  }
235  }
236  if (optimistic_initial_state && !loop_effects->unstable_aspects_cleared) {
237  // IMPORTANT: Whatever we clone here needs to be checked for consistency
238  // in when we try to terminate the loop in `IsCompatibleWithLoopHeader`.
239  if (loop_effects->objects_written.empty() &&
240  loop_effects->keys_cleared.empty()) {
241  loaded_properties = other.loaded_properties;
242  } else {
243  auto cleared_key = loop_effects->keys_cleared.begin();
244  auto cleared_keys_end = loop_effects->keys_cleared.end();
245  auto cleared_obj = loop_effects->objects_written.begin();
246  auto cleared_objs_end = loop_effects->objects_written.end();
247  for (auto loaded_key : other.loaded_properties) {
248  if (NextInIgnoreList(cleared_key, cleared_keys_end, loaded_key.first)) {
249  continue;
250  }
251  auto& props_for_key =
252  loaded_properties.try_emplace(loaded_key.first, zone).first->second;
253  for (auto loaded_obj : loaded_key.second) {
254  if (!NextInIgnoreList(cleared_obj, cleared_objs_end,
255  loaded_obj.first)) {
256  props_for_key.emplace(loaded_obj);
257  }
258  }
259  }
260  }
261  if (loop_effects->context_slot_written.empty()) {
262  loaded_context_slots = other.loaded_context_slots;
263  } else {
264  auto slot_written = loop_effects->context_slot_written.begin();
265  auto slot_written_end = loop_effects->context_slot_written.end();
266  for (auto loaded : other.loaded_context_slots) {
267  if (!NextInIgnoreList(slot_written, slot_written_end, loaded.first)) {
268  loaded_context_slots.emplace(loaded);
269  }
270  }
271  }
272  if (!loaded_context_slots.empty()) {
273  if (loop_effects->may_have_aliasing_contexts) {
274  may_have_aliasing_contexts_ = ContextSlotLoadsAlias::kYes;
275  } else {
276  DCHECK_EQ(may_have_aliasing_contexts_, ContextSlotLoadsAlias::kNone);
277  may_have_aliasing_contexts_ = other.may_have_aliasing_contexts();
278  }
279  }
280  }
281 
282  // To account for the back-jump we must not allow effects to be reshuffled
283  // across loop headers.
284  // TODO(olivf): Only do this if the loop contains write effects.
285  increment_effect_epoch();
286  for (const auto& e : other.available_expressions) {
287  if (e.second.effect_epoch >= effect_epoch()) {
288  available_expressions.emplace(e);
289  }
290  }
291 }

295 // Takes two ordered maps and ensures that every element in `as` is
296 // * also present in `bs` and
297 // * `Compare(a, b)` holds for each value.
298 template <typename As, typename Bs, typename CompareFunction,
299  typename IsEmptyFunction = std::nullptr_t>
300 bool AspectIncludes(const As& as, const Bs& bs, const CompareFunction& Compare,
301  const IsEmptyFunction IsEmpty = nullptr) {
302  typename As::const_iterator a = as.begin();
303  typename Bs::const_iterator b = bs.begin();
304  while (a != as.end()) {
305  if constexpr (!std::is_same_v<IsEmptyFunction, std::nullptr_t>) {
306  if (IsEmpty(a->second)) {
307  ++a;
308  continue;
309  }
310  }
311  if (b == bs.end()) return false;
312  while (b->first < a->first) {
313  ++b;
314  if (b == bs.end()) return false;
315  }
316  if (!(a->first == b->first)) return false;
317  if (!Compare(a->second, b->second)) {
318  return false;
319  }
320  ++a;
321  ++b;
322  }
323  return true;
324 }

326 // Same as above but allows `as` to contain empty collections as values, which
327 // do not need to be present in `bs`.
328 template <typename As, typename Bs, typename Function>
329 bool MaybeEmptyAspectIncludes(const As& as, const Bs& bs,
330  const Function& Compare) {
331  return AspectIncludes<As, Bs, Function>(as, bs, Compare,
332  [](auto x) { return x.empty(); });
333 }

337  const Function& Compare) {
338  return AspectIncludes<As, Bs, Function>(as, bs, Compare,
339  [](auto x) { return x == nullptr; });

369 bool KnownNodeAspects::IsCompatibleWithLoopHeader(
370  const KnownNodeAspects& loop_header) const {
371  // Needs to be in sync with `CloneForLoopHeader(zone, true)`.
372 
373  // Analysis state can change with loads.
374  if (!loop_header.loaded_context_slots.empty() &&
375  loop_header.may_have_aliasing_contexts() != ContextSlotLoadsAlias::kYes &&
376  loop_header.may_have_aliasing_contexts() !=
377  may_have_aliasing_contexts() &&
378  may_have_aliasing_contexts() != ContextSlotLoadsAlias::kNone) {
379  if (V8_UNLIKELY(v8_flags.trace_maglev_loop_speeling)) {
380  std::cout << "KNA after loop has incompatible "
381  "loop_header.may_have_aliasing_contexts\n";
382  }
383  return false;
384  }
385 
386  bool had_effects = effect_epoch() != loop_header.effect_epoch();
387 
388  if (!had_effects) {
389  if (!AspectIncludes(loop_header.node_infos, node_infos, NodeInfoTypeIs,
390  NodeInfoIsEmpty)) {
391  if (V8_UNLIKELY(v8_flags.trace_maglev_loop_speeling)) {
392  std::cout << "KNA after effectless loop has incompatible node_infos\n";
393  }
394  return false;
395  }
396  // In debug builds we do a full comparison to ensure that without an effect
397  // epoch change all unstable properties still hold.
398 #ifndef DEBUG
399  return true;
400 #endif
401  }
402 
403  if (!AspectIncludes(loop_header.node_infos, node_infos, NodeInfoIncludes,
404  NodeInfoIsEmpty)) {
405  if (V8_UNLIKELY(v8_flags.trace_maglev_loop_speeling)) {
406  std::cout << "KNA after loop has incompatible node_infos\n";
407  }
408  DCHECK(had_effects);
409  return false;
410  }
411 
412  if (!MaybeEmptyAspectIncludes(
413  loop_header.loaded_properties, loaded_properties,
414  [](auto a, auto b) { return AspectIncludes(a, b, SameValue); })) {
415  if (V8_UNLIKELY(v8_flags.trace_maglev_loop_speeling)) {
416  std::cout << "KNA after loop has incompatible loaded_properties\n";
417  }
418  DCHECK(had_effects);
419  return false;
420  }
421 
422  if (!MaybeNullAspectIncludes(loop_header.loaded_context_slots,
423  loaded_context_slots, SameValue)) {
424  if (V8_UNLIKELY(v8_flags.trace_maglev_loop_speeling)) {
425  std::cout << "KNA after loop has incompatible loaded_context_slots\n";
426  }
427  DCHECK(had_effects);
428  return false;
429  }
430 
431  return true;
432 }

435 MergePointInterpreterFrameState* MergePointInterpreterFrameState::New(
436  const MaglevCompilationUnit& info, const InterpreterFrameState& state,
437  int merge_offset, int predecessor_count, BasicBlock* predecessor,
438  const compiler::BytecodeLivenessState* liveness) {
439  MergePointInterpreterFrameState* merge_state =
440  info.zone()->New<MergePointInterpreterFrameState>(
441  info, merge_offset, predecessor_count, 1,
442  info.zone()->AllocateArray<BasicBlock*>(predecessor_count),
443  BasicBlockType::kDefault, liveness);
444  int i = 0;
445  merge_state->frame_state_.ForEachValue(
446  info, [&](ValueNode*& entry, interpreter::Register reg) {
447  entry = state.get(reg);
448  // Initialise the alternatives list and cache the alternative
449  // representations of the node.
450  Alternatives::List* per_predecessor_alternatives =
451  new (&merge_state->per_predecessor_alternatives_[i])
452  Alternatives::List();
453  per_predecessor_alternatives->Add(info.zone()->New<Alternatives>(
454  state.known_node_aspects()->TryGetInfoFor(entry)));
455  i++;
456  });
457  merge_state->predecessors_[0] = predecessor;
458  merge_state->known_node_aspects_ =
459  state.known_node_aspects()->Clone(info.zone());
460  state.virtual_objects().Snapshot();
461  merge_state->set_virtual_objects(state.virtual_objects());
462  return merge_state;
463 }

471 MergePointInterpreterFrameState* MergePointInterpreterFrameState::NewForLoop(
472  const InterpreterFrameState& start_state, Graph* graph,
473  const MaglevCompilationUnit& info, int merge_offset, int predecessor_count,
474  const compiler::BytecodeLivenessState* liveness,
475  const compiler::LoopInfo* loop_info, bool has_been_peeled) {
476  MergePointInterpreterFrameState* state =
477  info.zone()->New<MergePointInterpreterFrameState>(
478  info, merge_offset, predecessor_count, 0,
479  info.zone()->AllocateArray<BasicBlock*>(predecessor_count),
480  BasicBlockType::kLoopHeader, liveness);
481  state->bitfield_ =
482  kIsLoopWithPeeledIterationBit::update(state->bitfield_, has_been_peeled);
483  state->loop_metadata_ = LoopMetadata{loop_info, nullptr};
484  if (loop_info->resumable()) {
485  state->known_node_aspects_ =
486  info.zone()->New<KnownNodeAspects>(info.zone());
487  state->set_is_resumable_loop(graph);
488  }
489  auto& assignments = loop_info->assignments();
490  auto& frame_state = state->frame_state_;
491  int i = 0;
492  frame_state.ForEachParameter(
493  info, [&](ValueNode*& entry, interpreter::Register reg) {
494  entry = nullptr;
495  if (assignments.ContainsParameter(reg.ToParameterIndex())) {
496  entry = state->NewLoopPhi(info.zone(), reg);
497  } else if (state->is_resumable_loop()) {
498  // Copy initial values out of the start state.
499  entry = start_state.get(reg);
500  // Initialise the alternatives list for this value.
501  new (&state->per_predecessor_alternatives_[i]) Alternatives::List();
502  DCHECK(entry->Is<InitialValue>());
503  }
504  ++i;
505  });
506  frame_state.context(info) = nullptr;
507  if (state->is_resumable_loop()) {
508  // While contexts are always the same at specific locations, resumable loops
509  // do have different nodes to set the context across resume points. Create a
510  // phi for them.
511  frame_state.context(info) = state->NewLoopPhi(
512  info.zone(), interpreter::Register::current_context());
513  }
514  frame_state.ForEachLocal(
515  info, [&](ValueNode*& entry, interpreter::Register reg) {
516  entry = nullptr;
517  if (assignments.ContainsLocal(reg.index())) {
518  entry = state->NewLoopPhi(info.zone(), reg);
519  }
520  });
521  DCHECK(!frame_state.liveness()->AccumulatorIsLive());
522  return state;
523 }

671 void MergePointInterpreterFrameState::MergeVirtualObjects(
672  MaglevGraphBuilder* builder, MaglevCompilationUnit& compilation_unit,
673  const VirtualObjectList unmerged_vos,
674  const KnownNodeAspects& unmerged_aspects) {
675  if (frame_state_.virtual_objects().is_empty()) return;
676  if (unmerged_vos.is_empty()) return;
677 
678  frame_state_.virtual_objects().Snapshot();
679 
680  PrintVirtualObjects(builder->graph_labeller(), unmerged_vos,
681  "VOs before merge:");
682 
683  SmallZoneMap<InlinedAllocation*, VirtualObject*, 10> unmerged_map(
684  builder->zone());
685  SmallZoneMap<InlinedAllocation*, VirtualObject*, 10> merged_map(
686  builder->zone());
687 
688  // We iterate both list in reversed order of ids collecting the umerged
689  // objects into the map, until we find a common virtual object.
690  VirtualObjectList::WalkUntilCommon(
691  frame_state_.virtual_objects(), unmerged_vos,
692  [&](VirtualObject* vo, VirtualObjectList vos) {
693  // If we have a version in the map, it should be the most up-to-date,
694  // since the list is in reverse order.
695  auto& map = unmerged_vos == vos ? unmerged_map : merged_map;
696  map.emplace(vo->allocation(), vo);
697  });
698 
699  // Walk the merged map (values from the merged state) and merge values.
700  for (auto [_, merged] : merged_map) {
701  VirtualObject* unmerged = nullptr;
702  auto it = unmerged_map.find(merged->allocation());
703  if (it != unmerged_map.end()) {
704  unmerged = it->second;
705  unmerged_map.erase(it);
706  } else {
707  unmerged = unmerged_vos.FindAllocatedWith(merged->allocation());
708  }
709  if (unmerged != nullptr) {
710  MergeVirtualObject(builder, unmerged_vos, unmerged_aspects, merged,
711  unmerged);
712  }
713  }
714 
715  // Walk the unmerged map (values from the interpreter frame state) and merge
716  // values. If the value was already merged, we would have removed from the
717  // unmerged_map.
718  for (auto [_, unmerged] : unmerged_map) {
719  VirtualObject* merged = nullptr;
720  auto it = merged_map.find(unmerged->allocation());
721  if (it != merged_map.end()) {
722  merged = it->second;
723  } else {
724  merged = frame_state_.virtual_objects().FindAllocatedWith(
725  unmerged->allocation());
726  }
727  if (merged != nullptr) {
728  MergeVirtualObject(builder, unmerged_vos, unmerged_aspects, merged,
729  unmerged);
730  }
731  }
732 
733  PrintVirtualObjects(builder->graph_labeller(), unmerged_vos,
734  "VOs after merge:");
735 }

813  builder->zone()->New<DeoptFrame>(builder->GetLatestCheckpointedFrame());

831  //

833  // Phi by checking if the LoopInfo of the loop Phi "Contains" the current

835  // was inlined (because the LoopInfo contains the first and last bytecode

837  //
838  // - LoopInfo is obtained from the {header_to_info_} member of
839  // BytecodeAnalysis, but the BytecodeAnalysis is a member of the

841  // created for inlining is destructed. LoopInfo would then become a stale
842  // pointer.
843  ClearLoopInfo();

846 bool MergePointInterpreterFrameState::TryMergeLoop(
847  MaglevGraphBuilder* builder, InterpreterFrameState& loop_end_state,
848  const std::function<BasicBlock*()>& FinishBlock) {
849  // This should be the last predecessor we try to merge.
850  DCHECK_EQ(predecessors_so_far_, predecessor_count_ - 1);
851  DCHECK(is_unmerged_loop());
852 
853  backedge_deopt_frame_ =
854  builder->zone()->New<DeoptFrame>(builder->GetLatestCheckpointedFrame());
855 
856  auto& compilation_unit = *builder->compilation_unit();
857 
858  DCHECK_NOT_NULL(known_node_aspects_);
859  DCHECK(v8_flags.maglev_optimistic_peeled_loops);
860 
861  // TODO(olivf): This could be done faster by consulting loop_effects_
862  if (!loop_end_state.known_node_aspects()->IsCompatibleWithLoopHeader(
863  *known_node_aspects_)) {
864  if (v8_flags.trace_maglev_graph_building) {
865  std::cout << "Merging failed, peeling loop instead... " << std::endl;
866  }
867  ClearLoopInfo();
868  return false;
869  }
870 
871  bool phis_can_merge = true;
872  frame_state_.ForEachValue(compilation_unit, [&](ValueNode* value,
873  interpreter::Register reg) {
874  if (!value->Is<Phi>()) return;
875  Phi* phi = value->Cast<Phi>();
876  if (!phi->is_loop_phi()) return;
877  if (phi->merge_state() != this) return;
878  NodeType old_type = known_node_aspects_->GetType(builder->broker(), phi);
879  if (old_type != NodeType::kUnknown) {
880  NodeType new_type = loop_end_state.known_node_aspects()->GetType(
881  builder->broker(), loop_end_state.get(reg));
882  if (!NodeTypeIs(new_type, old_type)) {
883  if (v8_flags.trace_maglev_loop_speeling) {
884  std::cout << "Cannot merge " << new_type << " into " << old_type
885  << " for r" << reg.index() << "\n";
886  }
887  phis_can_merge = false;
888  }
889  }
890  });
891  if (!phis_can_merge) {
892  ClearLoopInfo();
893  return false;
894  }
895 
896  BasicBlock* loop_end_block = FinishBlock();
897  int input = predecessor_count_ - 1;
898  loop_end_block->set_predecessor_id(input);
899  predecessors_[input] = loop_end_block;
900 
901  if (v8_flags.trace_maglev_graph_building) {
902  std::cout << "Next peeling not needed due to compatible state" << std::endl;
903  }
904 
905  frame_state_.ForEachValue(
906  compilation_unit, [&](ValueNode* value, interpreter::Register reg) {
907  PrintBeforeMerge(builder->graph_labeller(), value,
908  loop_end_state.get(reg), reg, known_node_aspects_);
909  MergeLoopValue(builder, reg, *loop_end_state.known_node_aspects(),
910  value, loop_end_state.get(reg));
911  PrintAfterMerge(builder->graph_labeller(), value, known_node_aspects_);
912  });
913  predecessors_so_far_++;
914  DCHECK_EQ(predecessors_so_far_, predecessor_count_);
915  ClearLoopInfo();
916  return true;
917 }
